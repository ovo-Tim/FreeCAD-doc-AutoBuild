291 bool Enumeration::operator==(const Enumeration& other) const
292 {
293  if (_index != other._index || enumArray.size() != other.enumArray.size()) {
294  return false;
295  }
296  for (size_t i = 0; i < enumArray.size(); ++i) {
297  if (enumArray[i]->data() == other.enumArray[i]->data()) {
298  continue;
299  }
300  if (!enumArray[i]->data() || !other.enumArray[i]->data()) {
301  return false;

303  if (!enumArray[i]->isEqual(other.enumArray[i]->data())) {
304  return false;
305  }
306  }
