53 void pagesize(string& page_template, int dims[4], int block[4])
54 {
55  dims[0] = 10; // default to A4_Landscape with 10mm margins
56  dims[1] = 10;
57  dims[2] = 287;
58  dims[3] = 200;
59 
60  block[0] = block[1] = 0; // default to no title block
61  block[2] = block[3] = 0;
62 
63  int t0, t1, t2, t3 = 0;
64 
65  // code below copied from FeaturePage.cpp
66  Base::FileInfo fi(page_template);
67  if (!fi.isReadable()) {
68  fi.setFile(App::Application::getResourceDir() + "Mod/Drawing/Templates/" + fi.fileName());
69  if (!fi.isReadable()) { // if so then really shouldn't have been able to get this far, but
70  // just in case...
71  return;
72  }
73  }
74 
75  // open Template file
76  string line;
77  ifstream file(fi.filePath().c_str());
78 
79  try {
80  while (getline(file, line)) {
81  if (line.find("<!-- Working space") != string::npos) {
82  (void)sscanf(line.c_str(),
83  "%*s %*s %*s %d %d %d %d",
84  &dims[0],
85  &dims[1],
86  &dims[2],
87  &dims[3]); // eg " <!-- Working space 10 10 410 287 -->"
88  getline(file, line);
89 
90  if (line.find("<!-- Title block") != string::npos) {
91  (void)sscanf(line.c_str(),
92  "%*s %*s %*s %d %d %d %d",
93  &t0,
94  &t1,
95  &t2,
96  &t3); // eg " <!-- Working space 10 10 410 287 -->"
97  }
98 
99  break;
100  }
101 
102  if (line.find("metadata") != string::npos) { // give up if we meet a metadata tag
103  break;
104  }
105  }
106  }
107  catch (Standard_Failure&) {
108  }
109 
110 
111  if (t3 != 0) {
112  block[2] = t2 - t0; // block width
113  block[3] = t3 - t1; // block height
114 
115  if (t0 <= dims[0]) { // title block on left
116  block[0] = -1;
117  }
118  else if (t2 >= dims[2]) { // title block on right
119  block[0] = 1;
120  }
121 
122  if (t1 <= dims[1]) { // title block at top
123  block[1] = 1;
124  }
125  else if (t3 >= dims[3]) { // title block at bottom
126  block[1] = -1;
127  }
128  }
129 
130  dims[2] -= dims[0]; // width
131  dims[3] -= dims[1]; // height
132 }

233 void orthoview::set_projection(const gp_Ax2& cs)
234 {
235  gp_Ax2 actual_cs;
236  gp_Dir actual_X;
237 
238  // coord system & directions for desired projection
239  X_dir = cs.XDirection();
240  Y_dir = cs.YDirection();
241  Z_dir = cs.Direction();
242 
243  // coord system of created view - same code as used in projection algos
244  // actual_cs = gp_Ax2(gp_Pnt(0,0,0), gp_Dir(Z_dir.X(),Z_dir.Y(),Z_dir.Z()));
245 
246  // but as the file gets saved the projection direction gets rounded.
247  // this can lead to choosing a different normal x-direction when the file
248  // gets reloaded see issue #1909
249  // we anticipate the actual_cs after reloading by rounding the Z_dir now
250  const double x = round(Z_dir.X() * 1e12) / 1e12;
251  const double y = round(Z_dir.Y() * 1e12) / 1e12;
252  const double z = round(Z_dir.Z() * 1e12) / 1e12;
253  actual_cs = gp_Ax2(gp_Pnt(0, 0, 0), gp_Dir(x, y, z));
254 
255  actual_X = actual_cs.XDirection();
256 
257  // angle between desired projection and actual projection
258  float rotation = X_dir.Angle(actual_X);
259 
260  if (rotation != 0 && abs(M_PI - rotation) > 0.05) {
261  if (!Z_dir.IsEqual(actual_X.Crossed(X_dir), 0.05)) {
262  rotation = -rotation;
263  }
264  }
265 
266  calcCentre();
267 
268  // this_view->Direction.setValue(Z_dir.X(), Z_dir.Y(), Z_dir.Z());
269  this_view->Direction.setValue(x, y, z);
270  this_view->Rotation.setValue(180 * rotation / M_PI);
271 }

391 void OrthoViews::calc_layout_size() // calculate the real world size of given view layout, assuming
392  // no space
393 {
394  // note that views in relative positions x = -4, -2, 0 , 2 etc etc
395  // have width = orientated part width
396  // while those in relative positions x = -3, -1, 1 etc
397  // have width = orientated part depth
398 
399  // similarly in y positions, height = part height or depth
400 
401  layout_width = (1 + floor(max_r_x / 2.0) + floor(-min_r_x / 2.0)) * width;
402  layout_width += (ceil(max_r_x / 2.0) + ceil(-min_r_x / 2.0)) * depth;
403  layout_height = (1 + floor(max_r_y / 2.0) + floor(-min_r_y / 2.0)) * height;
404  layout_height += (ceil(max_r_y / 2.0) + ceil(-min_r_y / 2.0)) * depth;
405 }

407 void OrthoViews::choose_page() // chooses which bit of page space to use depending upon layout &
408  // titleblock
409 {
410  int h = abs(*horiz); // how many views in direction of title block (horiz points to min_r_x or
411  // max_r_x)
412  int v = abs(*vert);
413  float layout_corner_width = (1 + floor(h / 2.0)) * width
414  + ceil(h / 2.0) * depth; // from (0, 0) view inclusively, how wide and tall is the layout
415  // in the direction of the title block
416  float layout_corner_height = (1 + floor(v / 2.0)) * height + ceil(v / 2.0) * depth;
417  float rel_space_x = layout_corner_width / layout_width
418  - 1.0 * block[2] / large[2]; // relative to respective sizes, how much space between (0, 0)
419  // and title block,
420  float rel_space_y = layout_corner_height / layout_height
421  - 1.0 * block[3] / large[3]; // can be -ve if block extends into /
422  // beyond (0, 0) view
423  float view_x, view_y, v_x_r, v_y_r;
424  bool interferes = false;
425  float a, b;
426 
427  for (int i = min_r_x; i <= max_r_x; i++) {
428  for (int j = min_r_y; j <= max_r_y; j++) {
429  if (index(i, j) != -1) // is there a view in this position?
430  {
431  a = i * block[0]
432  * 0.5; // reflect i and j so as +ve is in direction of title block ##
433  b = j * block[1] * 0.5;
434  view_x = ceil(a + 0.5) * width
435  + ceil(a) * depth; // extreme coords of view in direction of block, measured
436  // from far corner of (0, 0) view,
437  view_y = ceil(b + 0.5) * height
438  + ceil(b) * depth; // can be -ve if view is on opposite
439  // side of (0, 0) from title block
440  v_x_r = view_x / layout_width; // make relative
441  v_y_r = view_y / layout_height;
442  if (v_x_r > rel_space_x
443  && v_y_r > rel_space_y) { // ## so that can use > in this condition regardless
444  // of position of block
445  interferes = true;
446  }
447  }
448  }
449  }
450 
451  if (!interferes) {
452  page_dims = large;
453  }
454  else {
455  if (min(small_h[2] / layout_width, small_h[3] / layout_height)
456  > min(small_v[2] / layout_width, small_v[3] / layout_height)) {
457  page_dims = small_h;
458  }
459  else {
460  page_dims = small_v;
461  }
462  }
463 }

465 void OrthoViews::calc_scale() // compute scale required to meet minimum space requirements
466 {
467  float scale_x, scale_y, working_scale;
468 
469  scale_x = (page_dims[2] - num_gaps_x * min_space) / layout_width;
470  scale_y = (page_dims[3] - num_gaps_y * min_space) / layout_height;
471 
472  working_scale = min(scale_x, scale_y);
473 
474  // which gives the largest scale for which the min_space requirements can be met, but we want a
475  // 'sensible' scale, rather than 0.28457239... eg if working_scale = 0.115, then we want to use
476  // 0.1, similarly 7.65 -> 5, and 76.5 -> 50
477 
478  float exponent = floor(log10(working_scale)); // if working_scale = a * 10^b, what is b?
479  working_scale *= pow(10, -exponent); // now find what 'a' is.
480 
481  float valid_scales[2][8] = {
482  {1, 1.25, 2, 2.5, 3.75, 5, 7.5, 10}, // equate to 1:10, 1:8, 1:5, 1:4, 3:8, 1:2, 3:4, 1:1
483  {1, 1.5, 2, 3, 4, 5, 8, 10}}; // equate to 1:1, 3:2, 2:1, 3:1, 4:1, 5:1, 8:1, 10:1
484 
485  int i = 7;
486  while (valid_scales[(exponent >= 0)][i]
487  > working_scale) { // choose closest value smaller than 'a' from list.
488  i -= 1; // choosing top list if exponent -ve, bottom list for +ve exponent
489  }
490 
491  scale = valid_scales[(exponent >= 0)][i]
492  * pow(10, exponent); // now have the appropriate scale, reapply the *10^b
493 }

495 void OrthoViews::calc_offsets() // calcs SVG coords for centre of upper left view
496 {
497  // space_x is the empty clear white space between views
498  // gap_x is the centre - centre distance between views
499 
500  float space_x = (page_dims[2] - scale * layout_width) / num_gaps_x;
501  float space_y = (page_dims[3] - scale * layout_height) / num_gaps_y;
502 
503  gap_x = space_x + scale * (width + depth) * 0.5;
504  gap_y = space_y + scale * (height + depth) * 0.5;
505 
506  if (min_r_x % 2 == 0) {
507  offset_x = page_dims[0] + space_x + 0.5 * scale * width;
508  }
509  else {
510  offset_x = page_dims[0] + space_x + 0.5 * scale * depth;
511  }
512 
513  if (max_r_y % 2 == 0) {
514  offset_y = page_dims[1] + space_y + 0.5 * scale * height;
515  }
516  else {
517  offset_y = page_dims[1] + space_y + 0.5 * scale * depth;
518  }
519 }
