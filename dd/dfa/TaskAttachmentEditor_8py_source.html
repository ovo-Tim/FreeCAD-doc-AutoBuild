59 def linkSubList_convertToOldStyle(references):
60  ("input: [(obj1, (sub1, sub2)), (obj2, (sub1, sub2))]\n"
61  "output: [(obj1, sub1), (obj1, sub2), (obj2, sub1), (obj2, sub2)]")
62  result = []
63  for tup in references:
64  if type(tup[1]) is tuple or type(tup[1]) is list:
65  for subname in tup[1]:
66  result.append((tup[0], subname))
67  if len(tup[1]) == 0:
68  result.append((tup[0], ''))
69  elif isinstance(tup[1],str):
70  # old style references, no conversion required
71  result.append(tup)
72  return result

120 def PlacementsFuzzyCompare(plm1, plm2):
121  pos_eq = (plm1.Base - plm2.Base).Length < 1e-7 # 1e-7 is OCC's Precision::Confusion
122 
123  q1 = plm1.Rotation.Q
124  q2 = plm2.Rotation.Q
125  # rotations are equal if q1 == q2 or q1 == -q2.
126  # Invert one of Q's if their scalar product is negative, before comparison.
127  if q1[0]*q2[0] + q1[1]*q2[1] + q1[2]*q2[2] + q1[3]*q2[3] < 0:
128  q2 = [-v for v in q2]
129  rot_eq = ( abs(q1[0]-q2[0]) +
130  abs(q1[1]-q2[1]) +
131  abs(q1[2]-q2[2]) +
132  abs(q1[3]-q2[3]) ) < 1e-12 # 1e-12 is OCC's Precision::Angular (in radians)
133  return pos_eq and rot_eq
