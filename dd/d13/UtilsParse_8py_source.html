375 def drill_translate(
376  values: Values,
377  gcode: Gcode,
378  command: str,
379  params: PathParameters,
380  motion_location: PathParameters,
381  drill_retract_mode: str,
382 ) -> None:
383  """Translate drill cycles.
384 
385  Currently only cycles in XY are provided (G17).
386  XZ (G18) and YZ (G19) are not dealt with.
387  In other words only Z drilling can be translated.
388  """
389  cmd: str
390  comment: str
391  drill_x: float
392  drill_y: float
393  drill_z: float
394  motion_z: float
395  nl: str = "\n"
396  retract_z: float
397  F_feedrate: str
398  G0_retract_z: str
399 
400  if values["MOTION_MODE"] == "G91":
401  # force absolute coordinates during cycles
402  gcode.append(f"{linenumber(values)}G90{nl}")
403 
404  drill_x = Units.Quantity(params["X"], Units.Length)
405  drill_y = Units.Quantity(params["Y"], Units.Length)
406  drill_z = Units.Quantity(params["Z"], Units.Length)
407  retract_z = Units.Quantity(params["R"], Units.Length)
408  if retract_z < drill_z: # R less than Z is error
409  comment = create_comment(values, "Drill cycle error: R less than Z")
410  gcode.append(f"{linenumber(values)}{comment}{nl}")
411  return
412  motion_z = Units.Quantity(motion_location["Z"], Units.Length)
413  if values["MOTION_MODE"] == "G91": # relative movements
414  drill_x += Units.Quantity(motion_location["X"], Units.Length)
415  drill_y += Units.Quantity(motion_location["Y"], Units.Length)
416  drill_z += motion_z
417  retract_z += motion_z
418  if drill_retract_mode == "G98" and motion_z >= retract_z:
419  retract_z = motion_z
420 
421  cmd = format_command_line(values, ["G0", f"Z{format_for_axis(values, retract_z)}"])
422  G0_retract_z = f"{cmd}{nl}"
423  cmd = format_for_feed(values, Units.Quantity(params["F"], Units.Velocity))
424  F_feedrate = f'{values["COMMAND_SPACE"]}F{cmd}{nl}'
425 
426  # preliminary movement(s)
427  if motion_z < retract_z:
428  gcode.append(f"{linenumber(values)}{G0_retract_z}")
429  cmd = format_command_line(
430  values,
431  [
432  "G0",
433  f"X{format_for_axis(values, drill_x)}",
434  f"Y{format_for_axis(values, drill_y)}",
435  ],
436  )
437  gcode.append(f"{linenumber(values)}{cmd}{nl}")
438  if motion_z > retract_z:
439  # NIST GCODE 3.5.16.1 Preliminary and In-Between Motion says G0 to retract_z
440  # Here use G1 since retract height may be below surface !
441  cmd = format_command_line(values, ["G1", f"Z{format_for_axis(values, retract_z)}"])
442  gcode.append(f"{linenumber(values)}{cmd}{F_feedrate}")
443 
444  # drill moves
445  if command in ("G81", "G82"):
446  output_G81_G82_drill_moves(
447  values, gcode, command, params, drill_z, F_feedrate, G0_retract_z
448  )
449  elif command in ("G73", "G83"):
450  output_G73_G83_drill_moves(
451  values, gcode, command, params, drill_z, retract_z, F_feedrate, G0_retract_z
452  )

540 def output_G73_G83_drill_moves(
541  values: Values,
542  gcode: Gcode,
543  command: str,
544  params: PathParameters,
545  drill_z: float,
546  retract_z: float,
547  F_feedrate: str,
548  G0_retract_z: str,
549 ) -> None:
550  """Output the movement G code for G73 and G83."""
551  a_bit: float
552  chip_breaker_height: float
553  clearance_depth: float
554  cmd: str
555  drill_step: float
556  last_stop_z: float
557  next_stop_z: float
558  nl: str = "\n"
559 
560  last_stop_z = retract_z
561  drill_step = Units.Quantity(params["Q"], Units.Length)
562  # NIST 3.5.16.4 G83 Cycle: "current hole bottom, backed off a bit."
563  a_bit = drill_step * 0.05
564  if drill_step != 0:
565  while True:
566  if last_stop_z != retract_z:
567  # rapid move to just short of last drilling depth
568  clearance_depth = last_stop_z + a_bit
569  cmd = format_command_line(
570  values,
571  ["G0", f"Z{format_for_axis(values, clearance_depth)}"],
572  )
573  gcode.append(f"{linenumber(values)}{cmd}{nl}")
574  next_stop_z = last_stop_z - drill_step
575  if next_stop_z > drill_z:
576  cmd = format_command_line(
577  values, ["G1", f"Z{format_for_axis(values, next_stop_z)}"]
578  )
579  gcode.append(f"{linenumber(values)}{cmd}{F_feedrate}")
580  if command == "G73":
581  # Rapid up "a small amount".
582  chip_breaker_height = next_stop_z + values["CHIPBREAKING_AMOUNT"]
583  cmd = format_command_line(
584  values,
585  [
586  "G0",
587  f"Z{format_for_axis(values, chip_breaker_height)}",
588  ],
589  )
590  gcode.append(f"{linenumber(values)}{cmd}{nl}")
591  elif command == "G83":
592  # Rapid up to the retract height
593  gcode.append(f"{linenumber(values)}{G0_retract_z}")
594  last_stop_z = next_stop_z
595  else:
596  cmd = format_command_line(values, ["G1", f"Z{format_for_axis(values, drill_z)}"])
597  gcode.append(f"{linenumber(values)}{cmd}{F_feedrate}")
598  gcode.append(f"{linenumber(values)}{G0_retract_z}")
599  break
