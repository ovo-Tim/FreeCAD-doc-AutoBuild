158 void SoFCColorGradient::setViewportSize(const SbVec2s& size)
159 {
160  float fMinX {};
161  float fMinY {};
162  float fMaxX {};
163  float fMaxY {};
164  float boxWidth = getBounds(size, fMinX, fMinY, fMaxX, fMaxY);
165 
166  // search for the labels
167  int num = 0;
168  for (int i = 0; i < labels->getNumChildren(); i++) {
169  if (labels->getChild(i)->getTypeId() == SoTransform::getClassTypeId()) {
170  num++;
171  }
172  }
173 
174  if (num > 2) {
175  bool first = true;
176  float fStep = (fMaxY - fMinY) / ((float)num - 2);
177 
178  for (int j = 0; j < labels->getNumChildren(); j++) {
179  if (labels->getChild(j)->getTypeId() == SoTransform::getClassTypeId()) {
180  auto transform = static_cast<SoTransform*>(labels->getChild(j)); // NOLINT
181  if (first) {
182  first = false;
183  transform->translation.setValue(fMaxX + spaceX - boxWidth,
184  fMaxY - spaceY + fStep,
185  0.0F);
186  }
187  else {
188  transform->translation.setValue(0, -fStep, 0.0F);
189  }
190  }
191  }
192  }
193 
194  // gradient bar is shifted to the left by width of the labels to assure that labels are fully visible
195  _bbox.setBounds(fMinX - boxWidth, fMinY, fMaxX - boxWidth, fMaxY);
196  modifyPoints(_bbox);
197 }

201  bool isScientific(float fMin, float fMax, int prec, int numColors)
202  {
203  const float base10 = 10.0F;
204  float eps = std::pow(base10, static_cast<float>(-prec));
205  float value_min = std::min<float>(fabs(fMin), fabs(fMax));
206  float value_max = std::max<float>(fabs(fMin), fabs(fMax));
207 
208  if (value_min < eps && value_min > 0.0F) {
209  return true;
210  }
211 
212  if ((value_max - value_min) < eps * static_cast<float>(numColors -1)) {
213  return true;
214  }
215 
216  return value_max > upperLimit;
217  }

261 std::vector<float> SoFCColorGradient::getMarkerValues(float fMin, float fMax, int count) const
262 {
263  std::vector<float> labels;
264 
265  // the middle of the bar is zero
266  if (isZeroBased(fMin, fMax)) {
267  if (count % 2 == 0) {
268  count++;
269  }
270  int half = count / 2;
271  for (int j = 0; j < half + 1; j++) {
272  float w = (float)j / ((float)half);
273  float fValue = (1.0F - w) * fMax;
274  labels.push_back(fValue);
275  }
276  for (int k = half + 1; k < count; k++) {
277  float w = (float)(k - half + 1) / ((float)(count - half));
278  float fValue = w * fMin;
279  labels.push_back(fValue);
280  }
281  }
282  else { // either not zero based or 0 is not in between [fMin,fMax]
283  for (int j = 0; j < count; j++) {
284  float w = (float)j / ((float)count - 1.0F);
285  float fValue = (1.0F - w) * fMax + w * fMin;
286  labels.push_back(fValue);
287  }
288  }
289 
290  return labels;
291 }

293 void SoFCColorGradient::modifyPoints(const SbBox2f& box)
294 {
295  float fMinX = box.getMin()[0];
296  float fMinY = box.getMin()[1];
297  float fMaxX = box.getMax()[0];
298  float fMaxY = box.getMax()[1];
299 
300  // set the vertices spanning the faces for the color gradient
301  int intFields = coords->point.getNum() / 2;
302  for (int i = 0; i < intFields; i++) {
303  float w = static_cast<float>(i) / static_cast<float>(intFields - 1);
304  float fPosY = (1.0F - w) * fMaxY + w * fMinY;
305  coords->point.set1Value(2 * i, fMinX, fPosY, 0.0F);
306  coords->point.set1Value(2 * i + 1, fMaxX, fPosY, 0.0F);
307  }
308 }

322 SoIndexedFaceSet* SoFCColorGradient::createFaceSet(int numFaces) const
323 {
324  // NOLINTBEGIN
325  // for numFaces colors we need 2*(numFaces-1) faces and therefore an array with
326  // 8*(numFaces-1) face indices
327  auto faceset = new SoIndexedFaceSet;
328  faceset->coordIndex.setNum(8 * (numFaces - 1));
329  for (int j = 0; j < numFaces - 1; j++) {
330  faceset->coordIndex.set1Value(8 * j, 2 * j);
331  faceset->coordIndex.set1Value(8 * j + 1, 2 * j + 3);
332  faceset->coordIndex.set1Value(8 * j + 2, 2 * j + 1);
333  faceset->coordIndex.set1Value(8 * j + 3, SO_END_FACE_INDEX);
334  faceset->coordIndex.set1Value(8 * j + 4, 2 * j);
335  faceset->coordIndex.set1Value(8 * j + 5, 2 * j + 2);
336  faceset->coordIndex.set1Value(8 * j + 6, 2 * j + 3);
337  faceset->coordIndex.set1Value(8 * j + 7, SO_END_FACE_INDEX);
338  }
339  // NOLINTEND
340 
341  return faceset;
342 }

386 {

389  setCoordSize(2 * numColors);

391

393  auto ttype = createTransparencyType();

395  auto matBinding = createMaterialBinding();
396 
397  // first clear the children

399  coinRemoveAllChildren(this);

402  addChild(ttype);

405  addChild(mat);
406  addChild(matBinding);
407  addChild(faceset);
