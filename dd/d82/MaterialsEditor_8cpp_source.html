190 void MaterialsEditor::saveFavorites()
191 {
192  auto param = App::GetApplication().GetParameterGroupByPath(
193  "User parameter:BaseApp/Preferences/Mod/Material/Favorites");
194 
195  // Clear out the existing favorites
196  int count = param->GetInt("Favorites", 0);
197  for (int i = 0; static_cast<long>(i) < count; i++) {
198  QString key = QString::fromLatin1("FAV%1").arg(i);
199  param->RemoveASCII(key.toStdString().c_str());
200  }
201 
202  // Add the current values
203  param->SetInt("Favorites", _favorites.size());
204  int j = 0;
205  for (auto& favorite : _favorites) {
206  QString key = QString::fromLatin1("FAV%1").arg(j);
207  param->SetASCII(key.toStdString().c_str(), favorite.toStdString());
208 
209  j++;
210  }

268 void MaterialsEditor::saveRecents()
269 {
270  auto param = App::GetApplication().GetParameterGroupByPath(
271  "User parameter:BaseApp/Preferences/Mod/Material/Recent");
272 
273  // Clear out the existing favorites
274  int count = param->GetInt("Recent", 0);
275  for (int i = 0; static_cast<long>(i) < count; i++) {
276  QString key = QString::fromLatin1("MRU%1").arg(i);

291  j++;

294  }

298 void MaterialsEditor::addRecent(const QString& uuid)
299 {
300  // Ensure it is a material. New, unsaved materials will not be
301  try {
302  auto material = _materialManager.getMaterial(uuid);
303  Q_UNUSED(material)
304  }
305  catch (const Materials::MaterialNotFound&) {
306  return;
307  }
308 
309  // Ensure no duplicates
310  if (isRecent(uuid)) {
311  _recents.remove(uuid);
312  }
313 
314  _recents.push_front(uuid);
315  while (_recents.size() > static_cast<std::size_t>(_recentMax)) {
316  _recents.pop_back();
317  }
318 
319  saveRecents();
320 }

665 void MaterialsEditor::addMaterials(
666  QStandardItem& parent,
667  const std::shared_ptr<std::map<QString, std::shared_ptr<Materials::MaterialTreeNode>>>
668  modelTree,
669  const QIcon& folderIcon,
670  const QIcon& icon,
671  const Base::Reference<ParameterGrp>& param)
672 {
673  auto childParam = param->GetGroup(parent.text().toStdString().c_str());
674  auto tree = ui->treeMaterials;
675  for (auto& mat : *modelTree) {
676  std::shared_ptr<Materials::MaterialTreeNode> nodePtr = mat.second;
677  if (nodePtr->getType() == Materials::MaterialTreeNode::DataNode) {
678  auto material = nodePtr->getData();
679  QString uuid = material->getUUID();
680 
681  QIcon matIcon = icon;
682  if (material->isOldFormat()) {
683  matIcon = _warningIcon;
684  }
685  auto card = new QStandardItem(matIcon, mat.first);
686  card->setFlags(Qt::ItemIsSelectable | Qt::ItemIsEnabled | Qt::ItemIsDragEnabled
687  | Qt::ItemIsDropEnabled);
688  card->setData(QVariant(uuid), Qt::UserRole);
689  if (material->isOldFormat()) {
690  card->setToolTip(tr("This card uses the old format and must be saved before use"));
691  }
692 
693  addExpanded(tree, &parent, card);
694  }
695  else {
696  auto node = new QStandardItem(folderIcon, mat.first);
697  addExpanded(tree, &parent, node, childParam);
698  node->setFlags(Qt::ItemIsEnabled | Qt::ItemIsDragEnabled | Qt::ItemIsDropEnabled);
699  auto treeMap = nodePtr->getFolder();
700  addMaterials(*node, treeMap, folderIcon, icon, childParam);
701  }
702  }
703 }

1016 QString MaterialsEditor::getColorHash(const QString& colorString, int colorRange)
1017 {
1018  /*
1019  returns a '#000000' string from a '(0.1,0.2,0.3)' string. Optionally the string
1020  has a fourth value for alpha (transparency)
1021  */
1022  std::stringstream stream(colorString.toStdString());
1023 
1024  char c;
1025  stream >> c; // read "("
1026  double red;
1027  stream >> red;
1028  stream >> c; // ","
1029  double green;
1030  stream >> green;
1031  stream >> c; // ","
1032  double blue;
1033  stream >> blue;
1034  stream >> c; // ","
1035  double alpha = 1.0;
1036  if (c == ',') {
1037  stream >> alpha;
1038  }
1039 
1040  QColor color(static_cast<int>(red * colorRange),
1041  static_cast<int>(green * colorRange),
1042  static_cast<int>(blue * colorRange),
1043  static_cast<int>(alpha * colorRange));
1044  return color.name();
1045 }
