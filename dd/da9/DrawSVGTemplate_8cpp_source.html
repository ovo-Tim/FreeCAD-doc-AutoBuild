77 void DrawSVGTemplate::onChanged(const App::Property* prop)
78 {
79  if (prop == &Template && !isRestoring()) {
80  //if we are restoring an existing file we just want the properties set as they were saved,
81  //but if we are not restoring, we need to replace the embedded file and extract the new
82  //EditableTexts.
83  //We could try to find matching field names are preserve the values from
84  //the old template, but there is no guarantee that the same fields will be present.
85  replaceFileIncluded(Template.getValue());
86  EditableTexts.setValues(getEditableTextsFromTemplate());
87  QDomDocument templateDocument;
88  if (getTemplateDocument(PageResult.getValue(), templateDocument)) {
89  extractTemplateAttributes(templateDocument);
90  }
91  }
92 
93  TechDraw::DrawTemplate::onChanged(prop);
94 }

123 //parse the Svg code, inserting current EditableTexts values, and return the result as a QString.
124 //While parsing, note the Orientation, Width and Height values in the Svg code.
125 QString DrawSVGTemplate::processTemplate()
126 {
127  if (isRestoring()) {
128  //until everything is fully restored, the embedded file is not available, so we
129  //can't do anything
130  return QString();
131  }
132 
133  QDomDocument templateDocument;
134  if (!getTemplateDocument(PageResult.getValue(), templateDocument)) {
135  return QString();
136  }
137 
138  XMLQuery query(templateDocument);
139  std::map<std::string, std::string> substitutions = EditableTexts.getValues();
140  // auto captureTextValues = m_initialTextValues;
141 
142  // XPath query to select all <tspan> nodes whose <text> parent
143  // has "freecad:editable" attribute
144  query.processItems(QString::fromUtf8(
145  "declare default element namespace \"" SVG_NS_URI "\"; "
146  "declare namespace freecad=\"" FREECAD_SVG_NS_URI "\"; "
147  "//text[@" FREECAD_ATTR_EDITABLE "]/tspan"),
148  // [this, &substitutions, &templateDocument, &captureTextValues](QDomElement& tspan) -> bool {
149  [&substitutions, &templateDocument](QDomElement& tspan) -> bool {
150  // Replace the editable text spans with new nodes holding actual values
151 
152  QString editableName = tspan.parentNode().toElement().attribute(QString::fromUtf8(FREECAD_ATTR_EDITABLE));
153  std::map<std::string, std::string>::iterator item =
154  substitutions.find(editableName.toStdString());
155  if (item != substitutions.end()) {
156  // we have an editable text
157  QDomElement parent = tspan.parentNode().toElement();
158  QString editableValue = QString::fromUtf8(item->second.c_str());
159 
160  // Keep all spaces in the text node
161  tspan.setAttribute(QString::fromUtf8("xml:space"), QString::fromUtf8("preserve"));
162 
163  // Remove all child nodes and append text node with editable replacement as the only descendant
164  while (!tspan.lastChild().isNull()) {
165  tspan.removeChild(tspan.lastChild());
166  }
167  tspan.appendChild(templateDocument.createTextNode(editableValue));
168  }
169  return true;
170  });
171 
172  extractTemplateAttributes(templateDocument);
173  //all Qt holds on files should be released on exit #4085
174  return templateDocument.toString();
175 }

177 // find the width, height and orientation of the template and update the properties
178 void DrawSVGTemplate::extractTemplateAttributes(QDomDocument& templateDocument)
179 {
180  QDomElement docElement = templateDocument.documentElement();
181  Base::Quantity quantity;
182 
183  // Obtain the width
184  QString str = docElement.attribute(QString::fromLatin1("width"));
185  quantity = Base::Quantity::parse(str.toStdString());
186  quantity.setUnit(Base::Unit::Length);
187 
188  Width.setValue(quantity.getValue());
189 
190  str = docElement.attribute(QString::fromLatin1("height"));
191  quantity = Base::Quantity::parse(str.toStdString());
192  quantity.setUnit(Base::Unit::Length);
193 
194  Height.setValue(quantity.getValue());
195 
196  bool isLandscape = getWidth() / getHeight() >= 1.;
197 
198  Orientation.setValue(isLandscape ? 1 : 0);
199 }

201 // load the included template file as a QDomDocument
202 bool DrawSVGTemplate::getTemplateDocument(std::string sourceFile, QDomDocument& templateDocument) const
203 {
204  if (sourceFile.empty()) {
205  return false;
206  }
207  QFile templateFile(QString::fromStdString(sourceFile));
208  if (!templateFile.open(QIODevice::ReadOnly)) {
209  Base::Console().Error("DrawSVGTemplate::processTemplate can't read embedded template %s!\n", PageResult.getValue());
210  return false;
211  }
212 
213  if (!templateDocument.setContent(&templateFile)) {
214  Base::Console().Error("DrawSVGTemplate::processTemplate - failed to parse file: %s\n",
215  PageResult.getValue());
216  return false;
217  }
218  // no errors templateDocument is loaded
219  return true;
220 }

248 std::map<std::string, std::string> DrawSVGTemplate::getEditableTextsFromTemplate()
249 {
250  std::map<std::string, std::string> editables;
251 
252  QDomDocument templateDocument;
253  if (!getTemplateDocument(PageResult.getValue(), templateDocument)) {
254  return editables;
255  }
256 
257  XMLQuery query(templateDocument);
258 
259  // XPath query to select all <tspan> nodes whose <text> parent
260  // has "freecad:editable" attribute
261  query.processItems(QString::fromUtf8(
262  "declare default element namespace \"" SVG_NS_URI "\"; "
263  "declare namespace freecad=\"" FREECAD_SVG_NS_URI "\"; "
264  "//text[@" FREECAD_ATTR_EDITABLE "]/tspan"),
265  [this, &editables](QDomElement& tspan) -> bool {
266  QDomElement parent = tspan.parentNode().toElement();
267  QString editableName = parent.attribute(QString::fromUtf8(FREECAD_ATTR_EDITABLE));
268  QString editableValue;
269  if (parent.hasAttribute(QString::fromUtf8(FREECAD_ATTR_AUTOFILL))) {
270  QString autofillName = parent.attribute(QString::fromUtf8(FREECAD_ATTR_AUTOFILL));
271  QString autofillValue = getAutofillValue(autofillName);
272  if (!autofillValue.isEmpty()) {
273  editableValue = autofillValue;
274  }
275  }
276 
277  // If the autofill value is not specified or unsupported, use the default text value
278  if (editableValue.isEmpty()) {
279  editableValue = tspan.firstChild().nodeValue();
280  }
281 
282  editables[std::string(editableName.toUtf8().constData())] =
283  std::string(editableValue.toUtf8().constData());
284  return true;
285  });
286 
287  return editables;
288 }

290 QString DrawSVGTemplate::getAutofillByEditableName(QString nameToMatch)
291 {
292  QString result;
293  QString nameCapture{nameToMatch};
294 
295  QDomDocument templateDocument;
296  if (!getTemplateDocument(PageResult.getValue(), templateDocument)) {
297  return {};
298  }
299 
300  XMLQuery query(templateDocument);
301 
302  // XPath query to select all <tspan> nodes whose <text> parent
303  // has "freecad:editable" attribute
304  query.processItems(QString::fromUtf8(
305  "declare default element namespace \"" SVG_NS_URI "\"; "
306  "declare namespace freecad=\"" FREECAD_SVG_NS_URI "\"; "
307  "//text[@" FREECAD_ATTR_EDITABLE "]/tspan"),
308  [this, &nameCapture, &result](QDomElement& tspan) -> bool {
309  QDomElement parent = tspan.parentNode().toElement();
310  QString editableName = parent.attribute(QString::fromUtf8(FREECAD_ATTR_EDITABLE));
311  if (editableName == nameCapture &&
312  parent.hasAttribute(QString::fromUtf8(FREECAD_ATTR_AUTOFILL))) {
313  QString autofillName = parent.attribute(QString::fromUtf8(FREECAD_ATTR_AUTOFILL));
314  QString autofillValue = getAutofillValue(autofillName);
315  if (!autofillValue.isEmpty()) {
316  result = autofillValue;
317  }
318  }
319  return true;
320  });
321  return result;
322 }
