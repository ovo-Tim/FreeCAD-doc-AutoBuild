432  y = last.y
433  z = last.z
434  if p.get("X"):
435  x = round(p["X"], 2)
436  gcode += " X" + str(x)
437  if p.get("Y"):

439  gcode += " Y" + str(y)

441  z = round(p["Z"], 2)
442  gcode += " Z" + str(z)
443  last.x = x
444  last.y = y
445  last.z = z
446  gcode_list.append(gcode)
447  elif includeLines and name in ["G1", "G01"]:
448  gcode = name

452  if p.get("X"):
453  x = round(p["X"], 2)
