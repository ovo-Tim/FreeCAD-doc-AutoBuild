165 PyObject* MaterialTreeWidgetPy::setFilter(PyObject* args)
166 {
167  PyObject* obj;
168  if (!PyArg_ParseTuple(args, "O", &obj)) {
169  return nullptr;
170  }
171  if (PyObject_TypeCheck(obj, &(Materials::MaterialFilterPy::Type))) {
172  auto filter = static_cast<Materials::MaterialFilterPy*>(obj)->getMaterialFilterPtr();
173  auto filterPtr = std::make_shared<Materials::MaterialFilter>(*filter);
174  getMaterialTreeWidgetPtr()->setFilter(filterPtr);
175  }
176  else if (PyList_Check(obj)) {
177  // The argument is a list of filters
178  Py_ssize_t n = PyList_Size(obj);
179  if (n < 0) {
180  Py_Return;
181  }
182  PyObject* item;
183  auto filterList = std::make_shared<std::list<std::shared_ptr<Materials::MaterialFilter>>>();
184  for (int i = 0; i < n; i++) {
185  item = PyList_GetItem(obj, i);
186  if (PyObject_TypeCheck(item, &(Materials::MaterialFilterPy::Type))) {
187  auto filter =
188  static_cast<Materials::MaterialFilterPy*>(item)->getMaterialFilterPtr();
189  auto filterPtr = std::make_shared<Materials::MaterialFilter>(*filter);
190  filterList->push_back(filterPtr);
191  }
192  else {
193  PyErr_Format(PyExc_TypeError,
194  "List entry must be of type 'MaterialFilter' not '%s'",
195  obj->ob_type->tp_name);
196  return nullptr;
197  }
198  }
199  getMaterialTreeWidgetPtr()->setFilter(filterList);
200  }
201  else {
202  PyErr_Format(PyExc_TypeError,
203  "Type must be 'MaterialFilter' or list of 'MaterialFilter' not '%s'",
204  obj->ob_type->tp_name);
205  return nullptr;
206  }
207 
208  Py_Return;
209 }
