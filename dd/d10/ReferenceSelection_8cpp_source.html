60 bool ReferenceSelection::allow(App::Document* pDoc, App::DocumentObject* pObj, const char* sSubName)
61 {
62  PartDesign::Body *body = getBody();
63  App::OriginGroupExtension* originGroup = getOriginGroupExtension(body);
64 
65  // Don't allow selection in other document
66  if (support && pDoc != support->getDocument()) {
67  return false;
68  }
69 
70  // Enable selection from origin of current part/
71  if (pObj->isDerivedFrom<App::DatumElement>()) {
72  return allowOrigin(body, originGroup, pObj);
73  }
74 
75  if (pObj->isDerivedFrom<Part::Datum>()) {
76  return allowDatum(body, pObj);
77  }
78 
79  // The flag was used to be set. So, this block will never be treated and really doesn't make sense anyway
80 #if 0
81  if (!type.testFlag(AllowSelection::OTHERBODY)) {
82  if (support == NULL)
83  return false;
84  if (pObj != support)
85  return false;
86  }
87 #endif
88  // Handle selection of geometry elements
89  if (!sSubName || sSubName[0] == '\0')
90  return type.testFlag(AllowSelection::WHOLE);
91 
92  // resolve links if needed
93  if (!pObj->isDerivedFrom<Part::Feature>()) {
94  pObj = Part::Feature::getShapeOwner(pObj, sSubName);
95  }
96 
97  if (pObj && pObj->isDerivedFrom<Part::Feature>()) {
98  return allowPartFeature(pObj, sSubName);
99  }
100 
101  return false;
102 }

138 bool ReferenceSelection::allowOrigin(PartDesign::Body *body, App::OriginGroupExtension* originGroup, App::DocumentObject* pObj) const
139 {
140  bool fits = false;
141  if (type.testFlag(AllowSelection::FACE) && pObj->isDerivedFrom<App::Plane>()) {
142  fits = true;
143  }
144  else if (type.testFlag(AllowSelection::EDGE) && pObj->isDerivedFrom<App::Line>()) {
145  fits = true;
146  }
147 
148  if (fits) { // check that it actually belongs to the chosen body or part
149  try { // here are some throwers
150  if (body) {
151  if (body->hasObject(pObj, true) ) {
152  return true;
153  }
154  }
155  else if (originGroup ) {
156  if (originGroup->hasObject(pObj, true)) {
157  return true;
158  }
159  }
160  }
161  catch (const Base::Exception&) {
162  }
163  }
164  return false; // The Plane/Axis doesn't fits our needs

167 bool ReferenceSelection::allowDatum(PartDesign::Body *body, App::DocumentObject* pObj) const
168 {
169  if (!body) { // Allow selecting Part::Datum features from the active Body
170  return false;
171  }
172  else if (!type.testFlag(AllowSelection::OTHERBODY) && !body->hasObject(pObj)) {
173  return false;
174  }
175 
176  if (type.testFlag(AllowSelection::FACE) && (pObj->isDerivedFrom<PartDesign::Plane>()))
177  return true;
178  if (type.testFlag(AllowSelection::EDGE) && (pObj->isDerivedFrom<PartDesign::Line>()))
179  return true;
180  if (type.testFlag(AllowSelection::POINT) && (pObj->isDerivedFrom<PartDesign::Point>()))
181  return true;
182 
183  return false;
184 }

186 bool ReferenceSelection::allowPartFeature(App::DocumentObject* pObj, const char* sSubName) const
187 {
188  std::string subName(sSubName);
189  if (type.testFlag(AllowSelection::POINT) && subName.compare(0, 6, "Vertex") == 0) {
190  return true;
191  }
192 
193  if (type.testFlag(AllowSelection::EDGE) && subName.compare(0, 4, "Edge") == 0) {
194  if (isEdge(pObj, sSubName))
195  return true;
196  }
197 
198  if (type.testFlag(AllowSelection::CIRCLE) && subName.compare(0, 4, "Edge") == 0) {
199  if (isCircle(pObj, sSubName))
200  return true;
201  }
202 
203  if (type.testFlag(AllowSelection::FACE) && subName.compare(0, 4, "Face") == 0) {
204  if (isFace(pObj, sSubName))
205  return true;
206  }
207 
208  return false;
209 }

216  if (!edgeShape.IsNull()) {

218  BRepAdaptor_Curve adapt(edgeShape);

223  return true;
224  }
225  }

284 bool getReferencedSelection(const App::DocumentObject* thisObj, const Gui::SelectionChanges& msg,
285  App::DocumentObject*& selObj, std::vector<std::string>& selSub)
286 {
287  selObj = nullptr;
288  if (!thisObj)
289  return false;
290 
291  if (strcmp(thisObj->getDocument()->getName(), msg.pDocName) != 0)
292  return false;
293 
294  selObj = thisObj->getDocument()->getObject(msg.pObjectName);
295  if (selObj == thisObj)
296  return false;
297 
298  std::string subname = msg.pSubName;
299 
300  //check if the selection is an external reference and ask the user what to do
301  //of course only if thisObj is in a body, as otherwise the old workflow would not
302  //be supported
303  PartDesign::Body* body = PartDesignGui::getBodyFor(thisObj, false);
304  bool originfeature = selObj->isDerivedFrom(App::DatumElement::getClassTypeId());
305  if (!originfeature && body) {
306  PartDesign::Body* selBody = PartDesignGui::getBodyFor(selObj, false);
307  if (!selBody || body != selBody) {
308  QDialog dia(Gui::getMainWindow());
309  Ui_DlgReference dlg;
310  dlg.setupUi(&dia);
311  dia.setModal(true);
312  int result = dia.exec();
313  if (result == QDialog::DialogCode::Rejected) {
314  selObj = nullptr;
315  return false;
316  }
317 
318  if (!dlg.radioXRef->isChecked()) {
319  App::Document* document = thisObj->getDocument();
320  document->openTransaction("Make copy");
321  auto copy = PartDesignGui::TaskFeaturePick::makeCopy(selObj, subname, dlg.radioIndependent->isChecked());
322  body->addObject(copy);
323 
324  selObj = copy;
325  subname.erase(std::remove_if(subname.begin(), subname.end(), &isdigit), subname.end());
326  subname.append("1");
327  }
328  }
329  }
330 
331  // Remove subname for planes and datum features
332  if (PartDesign::Feature::isDatum(selObj)) {
333  subname = "";
334  }
335 
336  selSub = std::vector<std::string>(1,subname);
337 
338  return true;
339 }
