282 def parse(pathobj):
283  global PRECISION
284  global MODAL
285  global OUTPUT_DOUBLES
286  global UNIT_FORMAT
287  global UNIT_SPEED_FORMAT
288 
289  out = ""
290  lastcommand = None
291  precision_string = "." + str(PRECISION) + "f"
292  currLocation = {} # keep track for no doubles
293 
294  # the order of parameters
295  # linuxcnc doesn't want K properties on XY plane Arcs need work.
296  params = [
297  "X",
298  "Y",
299  "Z",
300  "A",
301  "B",
302  "C",
303  "I",
304  "J",
305  "F",
306  "S",
307  "T",
308  "Q",
309  "R",
310  "L",
311  "H",
312  "D",
313  "P",
314  ]
315  firstmove = Path.Command("G0", {"X": -1, "Y": -1, "Z": -1, "F": 0.0})
316  currLocation.update(firstmove.Parameters) # set First location Parameters
317 
318  if hasattr(pathobj, "Group"): # We have a compound or project.
319  # if OUTPUT_COMMENTS:
320  # out += linenumber() + "(compound: " + pathobj.Label + ")\n"
321  for p in pathobj.Group:
322  out += parse(p)
323  return out
324  else: # parsing simple path
325 
326  # groups might contain non-path things like stock.
327  if not hasattr(pathobj, "Path"):
328  return out
329 
330  # if OUTPUT_COMMENTS:
331  # out += linenumber() + "(" + pathobj.Label + ")\n"
332 
333  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
334 
335  outstring = []
336  command = c.Name
337  outstring.append(command)
338 
339  # if modal: suppress the command if it is the same as the last one
340  if MODAL is True:
341  if command == lastcommand:
342  outstring.pop(0)
343 
344  if c.Name[0] == "(" and not OUTPUT_COMMENTS: # command is a comment
345  continue
346 
347  # Now add the remaining parameters in order
348  for param in params:
349  if param in c.Parameters:
350  if param == "F" and (
351  currLocation[param] != c.Parameters[param] or OUTPUT_DOUBLES
352  ):
353  if c.Name not in [
354  "G0",
355  "G00",
356  ]: # linuxcnc doesn't use rapid speeds
357  speed = Units.Quantity(c.Parameters["F"], FreeCAD.Units.Velocity)
358  if speed.getValueAs(UNIT_SPEED_FORMAT) > 0.0:
359  outstring.append(
360  param
361  + format(
362  float(speed.getValueAs(UNIT_SPEED_FORMAT)),
363  precision_string,
364  )
365  )
366  else:
367  continue
368  elif param == "T":
369  outstring.append(param + str(int(c.Parameters["T"])))
370  elif param == "H":
371  outstring.append(param + str(int(c.Parameters["H"])))
372  elif param == "D":
373  outstring.append(param + str(int(c.Parameters["D"])))
374  elif param == "S":
375  outstring.append(param + str(int(c.Parameters["S"])))
376  else:
377  if (
378  (not OUTPUT_DOUBLES)
379  and (param in currLocation)
380  and (currLocation[param] == c.Parameters[param])
381  ):
382  continue
383  else:
384  pos = Units.Quantity(c.Parameters[param], FreeCAD.Units.Length)
385  outstring.append(
386  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
387  )
388 
389  # store the latest command
390  lastcommand = command
391  currLocation.update(c.Parameters)
392 
393  # Check for Tool Change:
394  if command == "M6":
395  # if OUTPUT_COMMENTS:
396  # out += linenumber() + "(begin toolchange)\n"
397  for line in TOOL_CHANGE.splitlines(True):
398  out += linenumber() + line
399 
400  if command == "message":
401  if OUTPUT_COMMENTS is False:
402  out = []
403  else:
404  outstring.pop(0) # remove the command
405 
406  # prepend a line number and append a newline
407  if len(outstring) >= 1:
408  if OUTPUT_LINE_NUMBERS:
409  outstring.insert(0, (linenumber()))
410 
411  # append the line to the final output
412  for w in outstring:
413  out += w + COMMAND_SPACE
414  out = out.strip() + "\n"
415 
416  return out
