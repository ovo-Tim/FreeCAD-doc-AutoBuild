105 void BomObject::saveCustomColumnData()
106 {
107  // This function saves data that is not automatically generated.
108  dataElements.clear();
109  std::tuple<App::CellAddress, App::CellAddress> res = getUsedRange();
110  int maxRow = std::get<1>(res).row();
111  int nameColIndex = getColumnIndex("Name");
112 
113  for (int row = 1; row <= maxRow; ++row) {
114 
115  size_t col = 0;
116  // Here we do not iterate columnName : columnsNames.getValues() because they may have
117  // changed
118  for (size_t i = 0; i < columnsNames.getValues().size(); ++i) {
119  std::string columnName = getText(0, i);
120  if (columnName != "Index" && columnName != "Name" && columnName != "Quantity"
121  && columnName != "File Name") {
122  // Base::Console().Warning("row col %d %d\n", row, col);
123  // save custom data if any.
124  std::string text = getText(row, col);
125  if (text != "") {
126  std::string objName = getText(row, nameColIndex);
127  BomDataElement el(objName, columnName, text);
128  dataElements.push_back(el);
129  }
130  }
131 
132  ++col;
133  }
134  }
135 }

137 void BomObject::generateBOM()
138 {
139  saveCustomColumnData();
140  clearAll();
141  obj_list.clear();
142  size_t row = 0;
143  size_t col = 0;
144 
145  // Populate headers
146  for (auto& columnName : columnsNames.getValues()) {
147  setCell(App::CellAddress(row, col), columnName.c_str());
148  ++col;
149  }
150  ++row;
151 
152  auto* assembly = getAssembly();
153  if (assembly) {
154  addObjectChildrenToBom(assembly->getOutList(), row, "");
155  }
156  else {
157  addObjectChildrenToBom(getDocument()->getRootObjectsIgnoreLinks(), row, "");
158  }

161 void BomObject::addObjectChildrenToBom(std::vector<App::DocumentObject*> objs,
162  size_t& row,
163  std::string index)
164 {
165  int quantityColIndex = getColumnIndex("Quantity");
166  bool hasQuantityCol = hasQuantityColumn();
167 
168  size_t siblingsInitialRow = row;
169 
170  if (index != "") {
171  index = index + ".";
172  }
173 
174  size_t sub_i = 1;
175 
176  for (auto* child : objs) {
177  if (!child) {
178  continue;
179  }
180  if (child->isDerivedFrom<App::Link>()) {
181  child = static_cast<App::Link*>(child)->getLinkedObject();
182  if (!child) {
183  continue;
184  }
185  }
186 
187  if (!child->isDerivedFrom<AssemblyObject>() && !child->isDerivedFrom<App::Part>()
188  && !(child->isDerivedFrom<Part::Feature>() && !onlyParts.getValue())) {
189  continue;
190  }
191 
192  if (hasQuantityCol && row != siblingsInitialRow) {
193  // Check if the object is not already in (case of links). And if so just increment.
194  // Note: an object can be used in several parts. In which case we do no want to blindly
195  // increment.
196  bool found = false;
197  for (size_t i = siblingsInitialRow; i <= row; ++i) {
198  size_t idInList = i - 1; // -1 for the header
199  if (idInList < obj_list.size() && child == obj_list[idInList]) {
200 
201  int qty = std::stoi(getText(i, quantityColIndex)) + 1;
202  setCell(App::CellAddress(i, quantityColIndex), std::to_string(qty).c_str());
203  found = true;
204  break;
205  }
206  }
207  if (found) {
208  continue;
209  }
210  }
211 
212  std::string sub_index = index + std::to_string(sub_i);
213  ++sub_i;
214 
215  addObjectToBom(child, row, sub_index);
216  ++row;
217 
218  if ((child->isDerivedFrom<AssemblyObject>() && detailSubAssemblies.getValue())
219  || (child->isDerivedFrom<App::Part>() && detailParts.getValue())) {
220  addObjectChildrenToBom(child->getOutList(), row, sub_index);
221  }
222  }
223 }

225 void BomObject::addObjectToBom(App::DocumentObject* obj, size_t row, std::string index)
226 {
227  obj_list.push_back(obj);
228  size_t col = 0;
229  for (auto& columnName : columnsNames.getValues()) {
230  if (columnName == "Index") {
231  setCell(App::CellAddress(row, col), (std::string("'") + index).c_str());
232  }
233  else if (columnName == "Name") {
234  setCell(App::CellAddress(row, col), obj->Label.getValue());
235  }
236  else if (columnName == "File Name") {
237  setCell(App::CellAddress(row, col), obj->getDocument()->getFileName());
238  }
239  else if (columnName == "Quantity") {
240  setCell(App::CellAddress(row, col), std::to_string(1).c_str());
241  }
242  else {
243  // load custom data if any.
244  for (auto& el : dataElements) {
245  if (el.objName == obj->Label.getValue() && el.columnName == columnName) {
246  setCell(App::CellAddress(row, col), el.value.c_str());
247  break;
248  }
249  }
250  }
251  ++col;
252  }
253 }

275 std::string Assembly::BomObject::getText(size_t row, size_t col)
276 {
277  Spreadsheet::Cell* cell = getCell(App::CellAddress(row, col));
278  std::string cellName;
279  if (cell) {
280  cell->getStringContent(cellName);
281 
282  // getStringContent is adding a ' before the string for whatever reason.
283  if (!cellName.empty() && cellName.front() == '\'') {
284  cellName.erase(0, 1); // Remove the first character if it's a '
285  }
286  }
287 
288  return cellName;
