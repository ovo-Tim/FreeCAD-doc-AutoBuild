75 PaperAttributes PagePrinter::getPaperAttributes(TechDraw::DrawPage* dPage)
76 {
77  PaperAttributes result;
78  if (!dPage) {
79  return result;
80  }
81  double width = A4Widthmm;
82  double height = A4Heightmm;
83  auto pageTemplate(dynamic_cast<TechDraw::DrawTemplate*>(dPage->Template.getValue()));
84  if (pageTemplate) {
85  width = pageTemplate->Width.getValue();
86  height = pageTemplate->Height.getValue();
87  }
88  // result.m_pagewidth = width;
89  // result.m_pageheight = height;
90 
91  //Qt's page size determination assumes Portrait orientation. To get the right paper size
92  //we need to ask in the proper form.
93  QPageSize::PageSizeId paperSizeID =
94  QPageSize::id(QSizeF(std::min(width, height), std::max(width, height)),
95  QPageSize::Millimeter, QPageSize::FuzzyOrientationMatch);
96  auto paperSize = paperSizeID;
97 
98  auto orientation = (QPageLayout::Orientation)dPage->getOrientation();
99  if (paperSize == QPageSize::Ledger) {
100  // Ledger size paper orientation is reversed inside Qt
101  orientation = (QPageLayout::Orientation)(1 - orientation);
102  }
103 
104  return {orientation, paperSize, width, height};
105 }

130 void PagePrinter::printAll(QPrinter* printer, App::Document* doc)
131 {
132  QPageLayout pageLayout = printer->pageLayout();
133  std::vector<App::DocumentObject*> docObjs =
134  doc->getObjectsOfType(TechDraw::DrawPage::getClassTypeId());
135  auto firstPage = docObjs.front();
136 
137  auto dPage = static_cast<TechDraw::DrawPage*>(firstPage);
138  double width = A4Heightmm; // default to A4 Landscape 297 x 210
139  double height = A4Widthmm;
140  makePageLayout(dPage, pageLayout, width, height);
141  printer->setPageLayout(pageLayout);
142  QPainter painter(printer);
143 
144  auto ourDoc = Gui::Application::Instance->getDocument(doc);
145  auto docModifiedState = ourDoc->isModified();
146 
147  bool firstTime = true;
148  for (auto& obj : docObjs) {
149  Gui::ViewProvider* vp = Gui::Application::Instance->getViewProvider(obj);
150  if (!vp) {
151  continue; // can't print this one
152  }
153  auto* vpp = dynamic_cast<TechDrawGui::ViewProviderPage*>(vp);
154  if (!vpp) {
155  continue; // can't print this one
156  }
157 
158  auto dPage = static_cast<TechDraw::DrawPage*>(obj);
159  double width = A4Heightmm; // default to A4 Landscape 297 x 210
160  double height = A4Widthmm;
161  makePageLayout(dPage, pageLayout, width, height);
162  printer->setPageLayout(pageLayout);
163 
164  if (!firstTime) {
165  printer->newPage();
166  }
167  firstTime = false;
168  QRectF sourceRect(0.0, Rez::guiX(-height), Rez::guiX(width), Rez::guiX(height));
169  QRect targetRect = printer->pageLayout().fullRectPixels(printer->resolution());
170  renderPage(vpp, painter, sourceRect, targetRect);
171  dPage->redrawCommand();
172  }
173 
174  ourDoc->setModified(docModifiedState);
175 }

178 void PagePrinter::printAllPdf(QPrinter* printer, App::Document* doc)
179 {
180  double dpmm = printer->resolution() / mmPerInch;
181 
182  QString outputFile = printer->outputFileName();
183  QString documentName = QString::fromUtf8(doc->getName());
184  QPdfWriter pdfWriter(outputFile);
185 
186  // set the printed PDF Version to comply with PDF/A-1b, more details under:
187  // https://www.kdab.com/creating-pdfa-documents-qt/
188  pdfWriter.setPdfVersion(QPagedPaintDevice::PdfVersion_A1b);
189 
190  pdfWriter.setTitle(documentName);
191  pdfWriter.setResolution(printer->resolution());
192  QPageLayout pageLayout = printer->pageLayout();
193  // we want to set the layout for the first page before we make the painter(&pdfWriter) or the layout for the first page will
194  // not be correct.
195  std::vector<App::DocumentObject*> docObjs =
196  doc->getObjectsOfType(TechDraw::DrawPage::getClassTypeId());
197  auto firstPage = docObjs.front();
198 
199  auto dPage = static_cast<TechDraw::DrawPage*>(firstPage);
200  double width = A4Heightmm;//default to A4 Landscape 297 x 210
201  double height = A4Widthmm;
202  makePageLayout(dPage, pageLayout, width, height);
203 
204  pdfWriter.setPageLayout(pageLayout);
205  // to get several pages into the same pdf, we must use the same painter for each page and not have any
206  // start() or end() until all the pages are printed.
207  QPainter painter(&pdfWriter);
208 
209  auto ourDoc = Gui::Application::Instance->getDocument(doc);
210  auto docModifiedState = ourDoc->isModified();
211 
212  bool firstTime = true;
213  for (auto& obj : docObjs) {
214  Gui::ViewProvider* vp = Gui::Application::Instance->getViewProvider(obj);
215  if (!vp) {
216  continue;// can't print this one
217  }
218  auto vpp = dynamic_cast<TechDrawGui::ViewProviderPage*>(vp);
219  if (!vpp) {
220  continue;// can't print this one
221  }
222 
223  auto ourScene = vpp->getQGSPage();
224  ourScene->setExportingPdf(true);
225 
226  auto dPage = static_cast<TechDraw::DrawPage*>(obj);
227  double width{0};
228  double height{0};
229  makePageLayout(dPage, pageLayout, width, height);
230  pdfWriter.setPageLayout(pageLayout);
231  if (!firstTime) {
232  pdfWriter.newPage();
233  }
234  firstTime = false;
235 
236  QRectF sourceRect(0.0, Rez::guiX(-height), Rez::guiX(width), Rez::guiX(height));
237  QRect targetRect(0, 0, width * dpmm, height * dpmm);
238  renderPage(vpp, painter, sourceRect, targetRect);
239  dPage->redrawCommand();
240 
241  ourScene->setExportingPdf(true);
242  }
243 
244  ourDoc->setModified(docModifiedState);
245 }

249 void PagePrinter::printBannerPage(QPrinter* printer, QPainter& painter, QPageLayout& pageLayout,
250  App::Document* doc, std::vector<App::DocumentObject*>& docObjs)
251 {
252  QFont savePainterFont = painter.font();
253  QFont painterFont;
254  painterFont.setFamily(Preferences::labelFontQString());
255  int fontSizeMM = Preferences::labelFontSizeMM();
256  double dpmm = printer->resolution() / mmPerInch;
257  int fontSizePx = fontSizeMM * dpmm;
258  painterFont.setPixelSize(fontSizePx);
259  painter.setFont(painterFont);
260 
261  //print a header
262  QString docLine = QObject::tr("Document Name:") + QLatin1String(" ") + QString::fromUtf8(doc->getName());
263  int leftMargin = pageLayout.margins().left() * dpmm + 5 * dpmm; //layout margin + 5mm
264  int verticalPos = pageLayout.margins().top() * dpmm + 20 * dpmm;//layout margin + 20mm
265  int verticalSpacing = 2; //double space
266  painter.drawText(leftMargin, verticalPos, docLine);
267 
268  //leave some blank space between document name and page entries
269  verticalPos += 2 * verticalSpacing * fontSizePx;
270  for (auto& obj : docObjs) {
271  //print a line for each page
272  QString pageLine = QString::fromUtf8(obj->getNameInDocument()) + QString::fromUtf8(" / ")
273  + QString::fromUtf8(obj->Label.getValue());
274  painter.drawText(leftMargin, verticalPos, pageLine);
275  verticalPos += verticalSpacing * fontSizePx;
276  }
277  painter.setFont(savePainterFont);//restore the original font
278 }

281 void PagePrinter::renderPage(ViewProviderPage* vpp, QPainter& painter, QRectF& sourceRect,
282  QRect& targetRect)
283 {
284  //turn off view frames for print
285  bool saveState = vpp->getFrameState();
286  vpp->setFrameState(false);
287  vpp->setTemplateMarkers(false);
288 
289  //scene might be drawn in light text. we need to redraw in normal text.
290  bool saveLightOnDark = Preferences::lightOnDark();
291  if (Preferences::lightOnDark()) {
292  Preferences::lightOnDark(false);
293  vpp->getQGSPage()->redrawAllViews();
294  }
295 
296  vpp->getQGSPage()->refreshViews();
297  vpp->getQGSPage()->render(&painter, targetRect, sourceRect);
298 
299  // Reset
300  vpp->setFrameState(saveState);
301  vpp->setTemplateMarkers(saveState);
302  Preferences::lightOnDark(saveLightOnDark);
303 
304  vpp->getQGSPage()->refreshViews();
305 }

339 void PagePrinter::printPdf(ViewProviderPage* vpPage, const std::string& file)
340 {
341  if (file.empty()) {
342  Base::Console().Warning("PagePrinter - no file specified\n");
343  return;
344  }
345 
346  auto filespec = Base::Tools::escapeEncodeFilename(file);
347  filespec = DU::cleanFilespecBackslash(filespec);
348 
349  // set up the pdfwriter
350  QString outputFile = QString::fromStdString(filespec);
351  QPdfWriter pdfWriter(outputFile);
352  pdfWriter.setPdfVersion(QPagedPaintDevice::PdfVersion_A1b);
353  QPageLayout pageLayout = pdfWriter.pageLayout();
354  auto marginsdb = pageLayout.margins(QPageLayout::Millimeter);
355  QString documentName = QString::fromUtf8(vpPage->getDrawPage()->getNameInDocument());
356  pdfWriter.setTitle(documentName);
357  // default pdfWriter dpi is 1200.
358 
359  // set up the page layout
360  auto dPage = vpPage->getDrawPage();
361  double width = A4Heightmm; // default to A4 Landscape 297 x 210
362  double height = A4Widthmm;
363  makePageLayout(dPage, pageLayout, width, height);
364  pdfWriter.setPageLayout(pageLayout);
365  marginsdb = pageLayout.margins(QPageLayout::Millimeter);
366 
367  // first page does not respect page layout unless painter is created after
368  // pdfWriter layout is established.
369  QPainter painter(&pdfWriter);
370 
371  auto ourScene = vpPage->getQGSPage();
372  ourScene->setExportingPdf(true);
373  auto ourDoc = Gui::Application::Instance->getDocument(dPage->getDocument());
374  auto docModifiedState = ourDoc->isModified();
375 
376  // render the page
377  QRectF sourceRect(0.0, Rez::guiX(-height), Rez::guiX(width), Rez::guiX(height));
378  double dpmm = pdfWriter.resolution() / mmPerInch;
379  int twide = int(std::round(width * dpmm));
380  int thigh = int(std::round(height * dpmm));
381  QRect targetRect(0, 0, twide, thigh);
382  renderPage(vpPage, painter, sourceRect, targetRect);
383 
384  ourScene->setExportingPdf(false);
385  ourDoc->setModified(docModifiedState);
386  dPage->redrawCommand();
387 }
