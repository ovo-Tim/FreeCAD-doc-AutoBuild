46 QList<int> getPenStyles()
47 {
48  QList<int> styles;
49  styles << 0b1111111111111111 // solid
50  << 0b1110111011101110 // dashed 3:1
51  << 0b1111110011111100 // dashed 6:2
52  << 0b0000111100001111 // dashed 4:4
53  << 0b1010101010101010 // point 1:1
54  << 0b1110010011100100 // dash point
55  << 0b1111111100111100; // dash long-dash
56  return styles;
57 }

59 const QVector<qreal> binaryPatternToDashPattern(int binaryPattern)
60 {
61  QVector<qreal> dashPattern;
62  int count = 0;
63  bool isDash = (binaryPattern & 0x8000) != 0; // Check the highest bit
64 
65  for (int i = 0; i < 16; ++i) {
66  bool currentBit = (binaryPattern & (0x8000 >> i)) != 0;
67  if (currentBit == isDash) {
68  ++count; // Counting dashes or spaces
69  }
70  else {
71  // Adjust count to be odd for dashes and even for spaces (see qt doc)
72  count = (count % 2 == (isDash ? 0 : 1)) ? count + 1 : count;
73  dashPattern << count;
74  count = 1; // Reset count for next dash/space
75  isDash = !isDash;
76  }
77  }
78  count = (count % 2 == (isDash ? 0 : 1)) ? count + 1 : count;
79  dashPattern << count; // Add the last count
80 
81  if ((dashPattern.size() % 2) == 1) {
82  // prevent this error : qWarning("QPen::setDashPattern: Pattern not of even length");
83  dashPattern << 1;
84  }
85 
86  return dashPattern;
87 }

104 void SketcherSettings::saveSettings()
105 {
106  // Sketch editing
107  ui->checkBoxAdvancedSolverTaskBox->onSave();
108  ui->checkBoxRecalculateInitialSolutionWhileDragging->onSave();
109  ui->checkBoxEnableEscape->onSave();
110  ui->checkBoxDisableShading->onSave();
111  ui->checkBoxNotifyConstraintSubstitutions->onSave();
112  ui->checkBoxAutoRemoveRedundants->onSave();
113  ui->checkBoxUnifiedCoincident->onSave();
114  ui->checkBoxHorVerAuto->onSave();
115  ui->checkBoxAddExtGeo->onSave();
116 
117  enum
118  {
119  DimensionSingleTool,
120  DimensionSeparateTools,
121  DimensionBoth
122  };
123 
124  // Dimensioning constraints mode
125  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
126  "User parameter:BaseApp/Preferences/Mod/Sketcher/dimensioning");
127  bool singleTool = true;
128  bool SeparatedTools = false;
129  int index = ui->dimensioningMode->currentIndex();
130  switch (index) {
131  case DimensionSeparateTools:
132  singleTool = false;
133  SeparatedTools = true;
134  break;
135  case DimensionBoth:
136  singleTool = true;
137  SeparatedTools = true;
138  break;
139  }
140  hGrp->SetBool("SingleDimensioningTool", singleTool);
141  hGrp->SetBool("SeparatedDimensioningTools", SeparatedTools);
142 
143  ui->radiusDiameterMode->setEnabled(index != 1);
144 
145  enum
146  {
147  DimensionAutoRadiusDiam,
148  DimensionDiameter,
149  DimensionRadius
150  };
151 
152  bool Diameter = true;
153  bool Radius = true;
154  index = ui->radiusDiameterMode->currentIndex();
155  switch (index) {
156  case DimensionDiameter:
157  Diameter = true;
158  Radius = false;
159  break;
160  case DimensionRadius:
161  Diameter = false;
162  Radius = true;
163  break;
164  }
165  hGrp->SetBool("DimensioningDiameter", Diameter);
166  hGrp->SetBool("DimensioningRadius", Radius);
167 
168  hGrp = App::GetApplication().GetParameterGroupByPath(
169  "User parameter:BaseApp/Preferences/Mod/Sketcher/Tools");
170 
171  index = ui->ovpVisibility->currentIndex();
172  hGrp->SetInt("OnViewParameterVisibility", index);
173 
174  checkForRestart();
175 }

177 void SketcherSettings::loadSettings()
178 {

181  ui->checkBoxRecalculateInitialSolutionWhileDragging->onRestore();

184  ui->checkBoxNotifyConstraintSubstitutions->onRestore();
185  ui->checkBoxAutoRemoveRedundants->onRestore();
186  ui->checkBoxUnifiedCoincident->onRestore();
187  setProperty("checkBoxUnifiedCoincident", ui->checkBoxUnifiedCoincident->isChecked());
188  ui->checkBoxHorVerAuto->onRestore();
189  setProperty("checkBoxHorVerAuto", ui->checkBoxHorVerAuto->isChecked());
190  ui->checkBoxAddExtGeo->onRestore();
191 
192  // Dimensioning constraints mode
193  ui->dimensioningMode->clear();
194  ui->dimensioningMode->addItem(tr("Single tool"));
195  ui->dimensioningMode->addItem(tr("Separated tools"));
196  ui->dimensioningMode->addItem(tr("Both"));
197 
198  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
199  "User parameter:BaseApp/Preferences/Mod/Sketcher/dimensioning");
200  bool singleTool = hGrp->GetBool("SingleDimensioningTool", true);
201  bool SeparatedTools = hGrp->GetBool("SeparatedDimensioningTools", false);
202  int index = SeparatedTools ? (singleTool ? 2 : 1) : 0;
203  ui->dimensioningMode->setCurrentIndex(index);
204  setProperty("dimensioningMode", index);
205  connect(ui->dimensioningMode,
206  QOverload<int>::of(&QComboBox::currentIndexChanged),
207  this,
208  &SketcherSettings::dimensioningModeChanged);
209 
210  ui->radiusDiameterMode->setEnabled(index != 1);
211 
212  // Dimensioning constraints mode
213  ui->radiusDiameterMode->clear();
214  ui->radiusDiameterMode->addItem(tr("Auto"));
215  ui->radiusDiameterMode->addItem(tr("Diameter"));
216  ui->radiusDiameterMode->addItem(tr("Radius"));
217 
218  bool Diameter = hGrp->GetBool("DimensioningDiameter", true);
219  bool Radius = hGrp->GetBool("DimensioningRadius", true);
220  index = Diameter ? (Radius ? 0 : 1) : 2;
221  ui->radiusDiameterMode->setCurrentIndex(index);
222 
223  hGrp = App::GetApplication().GetParameterGroupByPath(
224  "User parameter:BaseApp/Preferences/Mod/Sketcher/Tools");
225  ui->ovpVisibility->clear();
226  ui->ovpVisibility->addItem(tr("None"));
227  ui->ovpVisibility->addItem(tr("Dimensions only"));
228  ui->ovpVisibility->addItem(tr("Position and dimensions"));
229 
230  index = hGrp->GetInt("OnViewParameterVisibility", 1);
231  ui->ovpVisibility->setCurrentIndex(index);
232 }

469 void SketcherSettingsDisplay::onBtnTVApplyClicked(bool)
470 {
471  QString errMsg;
472  try {
473  Gui::Command::doCommand(Gui::Command::Gui,
474  "for name,doc in App.listDocuments().items():\n"
475  " for sketch in doc.findObjects('Sketcher::SketchObject'):\n"
476  " sketch.ViewObject.HideDependent = %s\n"
477  " sketch.ViewObject.ShowLinks = %s\n"
478  " sketch.ViewObject.ShowSupport = %s\n"
479  " sketch.ViewObject.RestoreCamera = %s\n"
480  " sketch.ViewObject.ForceOrtho = %s\n"
481  " sketch.ViewObject.SectionView = %s\n",
482  this->ui->checkBoxTVHideDependent->isChecked() ? "True" : "False",
483  this->ui->checkBoxTVShowLinks->isChecked() ? "True" : "False",
484  this->ui->checkBoxTVShowSupport->isChecked() ? "True" : "False",
485  this->ui->checkBoxTVRestoreCamera->isChecked() ? "True" : "False",
486  this->ui->checkBoxTVForceOrtho->isChecked() ? "True" : "False",
487  this->ui->checkBoxTVSectionView->isChecked() ? "True" : "False");
488  }
489  catch (Base::PyException& e) {
490  Base::Console().DeveloperError("SketcherSettings", "error in onBtnTVApplyClicked:\n");
491  e.ReportException();
492  errMsg = QString::fromLatin1(e.what());
493  }
494  catch (...) {
495  errMsg = tr("Unexpected C++ exception");
496  }
497  if (errMsg.length() > 0) {
498  QMessageBox::warning(this, tr("Sketcher"), errMsg);
499  }
