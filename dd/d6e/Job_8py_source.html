637  def getCycleTime(self):
638  seconds = 0
639 
640  if len(self.obj.Operations.Group):
641  for op in self.obj.Operations.Group:
642 
643  # Skip inactive operations
644  if PathUtil.opProperty(op, "Active") is False:
645  continue
646 
647  # Skip operations that don't have a cycletime attribute
648  if PathUtil.opProperty(op, "CycleTime") is None:
649  continue
650 
651  formattedCycleTime = PathUtil.opProperty(op, "CycleTime")
652  opCycleTime = 0
653  try:
654  # Convert the formatted time from HH:MM:SS to just seconds
655  opCycleTime = sum(
656  x * int(t)
657  for x, t in zip([1, 60, 3600], reversed(formattedCycleTime.split(":")))
658  )
659  except Exception:
660  continue
661 
662  if opCycleTime > 0:
663  seconds = seconds + opCycleTime
664 
665  cycleTimeString = time.strftime("%H:%M:%S", time.gmtime(seconds))
666  self.obj.CycleTime = cycleTimeString
