182  if solverSection not in allSections:

189  section["Name"] = name

313  def _writeAttribute(self, key, data):
314  if isinstance(data, Section):

350  def _writeScalarAttr(self, key, data):
351  attrType = self._getAttrTypeScalar(data)
352  if attrType is None:
353  raise ValueError("Unsupported data type: %s" % type(data))
354  self._stream.write(_INDENT)
355  self._stream.write(key)
356  self._stream.write(_WHITESPACE)
357  self._stream.write("=")
358  self._stream.write(_WHITESPACE)
359  # check if we have a variable string
360  if attrType is _TYPE_STRING:
361  if data.startswith("Variable"):
362  attrType = _TYPE_VARIABLE
363  if attrType is not _TYPE_VARIABLE:
364  self._stream.write(attrType)
365  self._stream.write(_WHITESPACE)
366  output = self._preprocess(data, type(data))
367  # in case of a variable the output must be without the quatoation marks
368  if attrType is _TYPE_VARIABLE:
369  output = output.lstrip('"')
370  # we cannot use rstrip because there are two subsequent " at the end
371  output = output[:-1]
372  self._stream.write(output)

374  def _writeArrAttr(self, key, data):

377  self._stream.write(key)
378  self._stream.write("(%d)" % len(data))
379  self._stream.write(_WHITESPACE)

381  self._stream.write(_WHITESPACE)
