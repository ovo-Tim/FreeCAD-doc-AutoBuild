255  def get_dimension(self):
256  # Dimension
257  # known_element_dimensions = ["From Shape", "1D", "2D", "3D"]
258  # if not given, Gmsh uses the highest available.
259  # A use case for not "From Shape" would be a surface (2D) mesh of a solid

262  if shty == "Solid" or shty == "CompSolid":
263  # print("Found: " + shty)
264  self.dimension = "3"
265  elif shty == "Face" or shty == "Shell":
266  # print("Found: " + shty)
267  self.dimension = "2"
268  elif shty == "Edge" or shty == "Wire":
269  # print("Found: " + shty)
270  self.dimension = "1"
271  elif shty == "Vertex":
272  # print("Found: " + shty)
273  Console.PrintError("You can not mesh a Vertex.\n")
274  self.dimension = "0"
275  elif shty == "Compound":
276  # print(" Found a " + shty)
277  Console.PrintLog(
278  " Found a Compound. Since it could contain"
279  "any kind of shape dimension 3 is used.\n"
280  )
281  self.dimension = "3" # dimension 3 works for 2D and 1d shapes as well
282  else:
283  self.dimension = "0"
284  Console.PrintError(
285  "Could not retrieve Dimension from shape type. Please choose dimension.\n"
286  )
287  elif self.dimension == "3D":
288  self.dimension = "3"
289  elif self.dimension == "2D":
290  self.dimension = "2"
291  elif self.dimension == "1D":
292  self.dimension = "1"

339  def get_gmsh_command(self):
340  from platform import system
341 
342  gmsh_std_location = FreeCAD.ParamGet(
343  "User parameter:BaseApp/Preferences/Mod/Fem/Gmsh"
344  ).GetBool("UseStandardGmshLocation")
345  if gmsh_std_location:
346  if system() == "Windows":
347  gmsh_path = FreeCAD.getHomePath() + "bin/gmsh.exe"
348  FreeCAD.ParamGet("User parameter:BaseApp/Preferences/Mod/Fem/Gmsh").SetString(
349  "gmshBinaryPath", gmsh_path
350  )
351  self.gmsh_bin = gmsh_path
352  elif system() == "Linux":
353  p1 = subprocess.Popen(["which", "gmsh"], stdout=subprocess.PIPE)
354  if p1.wait() == 0:
355  output = p1.stdout.read()
356  output = output.decode("utf-8")
357  gmsh_path = output.split("\n")[0]
358  elif p1.wait() == 1:
359  error_message = (
360  "Gmsh binary gmsh not found in standard system binary path. "
361  "Please install Gmsh or set path to binary "
362  "in FEM preferences tab Gmsh.\n"
363  )
364  Console.PrintError(error_message)
365  raise GmshError(error_message)
366  self.gmsh_bin = gmsh_path
367  elif system() == "Darwin":
368  # https://forum.freecad.org/viewtopic.php?f=13&t=73041&p=642026#p642022
369  gmsh_path = "/Applications/Gmsh.app/Contents/MacOS/gmsh"
370  FreeCAD.ParamGet("User parameter:BaseApp/Preferences/Mod/Fem/Gmsh").SetString(
371  "gmshBinaryPath", gmsh_path
372  )
373  self.gmsh_bin = gmsh_path
374  else:
375  error_message = (
376  "No standard location implemented for your operating system. "
377  "Set GMHS binary path in FEM preferences.\n"
378  )
379  Console.PrintError(error_message)
380  raise GmshError(error_message)
381  else:
382  if not self.gmsh_bin:
383  self.gmsh_bin = FreeCAD.ParamGet(
384  "User parameter:BaseApp/Preferences/Mod/Fem/Gmsh"
385  ).GetString("gmshBinaryPath", "")
386  if not self.gmsh_bin: # in prefs not set, we will try to use something reasonable
387  if system() == "Linux":
388  self.gmsh_bin = "gmsh"
389  elif system() == "Windows":
390  self.gmsh_bin = FreeCAD.getHomePath() + "bin/gmsh.exe"
391  else:
392  self.gmsh_bin = "gmsh"
393  Console.PrintMessage(" " + self.gmsh_bin + "\n")
