484  return PointIdToPosId[layerindex][pointindex];

491  }
492 
495  MultiFieldId getIndexLayer(int geoid, Sketcher::PointPos pos)
496  {
497  auto indexit = GeoElementId2SetId.find(Sketcher::GeoElementId(geoid, pos));
498 
499  if (indexit != GeoElementId2SetId.end()) {
500  return indexit->second;
501  }
502 
503  return MultiFieldId::Invalid;
504  }
505 
508  MultiFieldId getIndexLayer(int vertexId)
509  {
510 
511  for (size_t l = 0; l < PointIdToVertexId.size(); l++) {
512  auto indexit =
513  std::find(PointIdToVertexId[l].begin(), PointIdToVertexId[l].end(), vertexId);
514 
515  if (indexit != PointIdToVertexId[l].end()) {
516  return MultiFieldId(std::distance(PointIdToVertexId[l].begin(), indexit), l);
517  }
518  }
519 
520  return MultiFieldId::Invalid;
521  }
522 
523  //* These map a MF index (second index) within a coin layer (first index) for points or curves
524  // to a GeoId */
525  std::vector<std::vector<std::vector<int>>>
526  CurvIdToGeoId; // conversion of SoLineSet index to GeoId
527  std::vector<std::vector<int>> PointIdToGeoId; // conversion of SoCoordinate3 index to GeoId
528  std::vector<std::vector<Sketcher::PointPos>> PointIdToPosId; // SoCoordinate3 index to PosId
529 
530  //* This maps an MF index (second index) of a point within a coin layer (first index) to a

532  std::vector<std::vector<int>> PointIdToVertexId;
533 
536  std::map<Sketcher::GeoElementId, MultiFieldId> GeoElementId2SetId;
