60 std::vector<App::DocumentObject*> ViewProviderOriginGroupExtension::constructChildren (
61  const std::vector<App::DocumentObject*> &children ) const
62 {
63  auto* obj = getExtendedViewProvider()->getObject();
64  auto* group = obj ? obj->getExtensionByType<App::OriginGroupExtension>() : nullptr;
65  if(!group)
66  return children;
67 
68  App::DocumentObject *originObj = group->Origin.getValue();
69 
70  // Origin must be first
71  if (originObj) {
72  std::vector<App::DocumentObject*> rv;
73  rv.push_back (originObj);
74  std::copy (children.begin(), children.end(), std::back_inserter (rv));
75  return rv;
76  } else { // Generally shouldn't happen but must be handled in case origin is lost
77  return children;
78  }
79 }
