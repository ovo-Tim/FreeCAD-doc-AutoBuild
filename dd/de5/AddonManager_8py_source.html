108 def get_icon(repo: Addon, update: bool = False) -> QtGui.QIcon:
109  """Returns an icon for an Addon. Uses a cached icon if possible, unless update is True,
110  in which case the icon is regenerated."""
111 
112  if not update and repo.icon and not repo.icon.isNull() and repo.icon.isValid():
113  return repo.icon
114 
115  path = ":/icons/" + repo.name.replace(" ", "_")
116  default_icon = QtGui.QIcon(":/icons/document-package.svg")
117  if repo.repo_type == Addon.Kind.WORKBENCH:
118  path += "_workbench_icon.svg"
119  default_icon = QtGui.QIcon(":/icons/document-package.svg")
120  elif repo.repo_type == Addon.Kind.MACRO:
121  if repo.macro and repo.macro.icon:
122  if os.path.isabs(repo.macro.icon):
123  path = repo.macro.icon
124  default_icon = QtGui.QIcon(":/icons/document-python.svg")
125  else:
126  path = os.path.join(os.path.dirname(repo.macro.src_filename), repo.macro.icon)
127  default_icon = QtGui.QIcon(":/icons/document-python.svg")
128  elif repo.macro and repo.macro.xpm:
129  cache_path = FreeCAD.getUserCachePath()
130  am_path = os.path.join(cache_path, "AddonManager", "MacroIcons")
131  os.makedirs(am_path, exist_ok=True)
132  path = os.path.join(am_path, repo.name + "_icon.xpm")
133  if not os.path.exists(path):
134  with open(path, "w") as f:
135  f.write(repo.macro.xpm)
136  default_icon = QtGui.QIcon(repo.macro.xpm)
137  else:
138  path += "_macro_icon.svg"
139  default_icon = QtGui.QIcon(":/icons/document-python.svg")
140  elif repo.repo_type == Addon.Kind.PACKAGE:
141  # The cache might not have been downloaded yet, check to see if it's there...
142  if os.path.isfile(repo.get_cached_icon_filename()):
143  path = repo.get_cached_icon_filename()
144  elif repo.contains_workbench():
145  path += "_workbench_icon.svg"
146  default_icon = QtGui.QIcon(":/icons/document-package.svg")
147  elif repo.contains_macro():
148  path += "_macro_icon.svg"
149  default_icon = QtGui.QIcon(":/icons/document-python.svg")
150  else:
151  default_icon = QtGui.QIcon(":/icons/document-package.svg")
152 
153  if QtCore.QFile.exists(path):
154  addon_icon = QtGui.QIcon(path)
155  else:
156  addon_icon = default_icon
157  repo.icon = addon_icon
158 
159  return addon_icon

335  def cleanup_workers(self) -> None:
336  """Ensure that no workers are running by explicitly asking them to stop and waiting for
337  them until they do"""
338  for worker in self.workers:
339  if hasattr(self, worker):
340  thread = getattr(self, worker)
341  if thread:
342  if not thread.isFinished():
343  thread.blockSignals(True)
344  thread.requestInterruption()
345  for worker in self.workers:
346  if hasattr(self, worker):
347  thread = getattr(self, worker)
348  if thread:
349  if not thread.isFinished():
350  finished = thread.wait(500)
351  if not finished:
352  FreeCAD.Console.PrintWarning(
353  translate(
354  "AddonsInstaller",
355  "Worker process {} is taking a long time to stop...",
356  ).format(worker)
357  + "\n"
358  )

439  def startup(self) -> None:
440  """Downloads the available packages listings and populates the table
441 
442  This proceeds in four stages: first, the main GitHub repository is queried for a list of
443  possible addons. Each addon is specified as a git submodule with name and branch
444  information. The actual specific commit ID of the submodule (as listed on GitHub) is
445  ignored. Any extra repositories specified by the user are appended to this list.
446 
447  Second, the list of macros is downloaded from the FreeCAD/FreeCAD-macros repository and
448  the wiki.
449 
450  Third, each of these items is queried for a package.xml metadata file. If that file exists
451  it is downloaded, cached, and any icons that it references are also downloaded and cached.
452 
453  Finally, for workbenches that are not contained within a package (e.g. they provide no
454  metadata), an additional git query is made to see if an update is available. Macros are
455  checked for file changes.
456 
457  Each of these stages is launched in a separate thread to ensure that the UI remains
458  responsive, and the operation can be cancelled.
459 
460  Each stage is also subject to caching, so may return immediately, if no cache update has
461  been requested."""
462 
463  # Each function in this list is expected to launch a thread and connect its completion
464  # signal to self.do_next_startup_phase, or to shortcut to calling
465  # self.do_next_startup_phase if it is not launching a worker
466  self.startup_sequence = [
467  self.populate_packages_table,
468  self.activate_table_widgets,
469  self.populate_macros,
470  self.update_metadata_cache,
471  self.check_updates,
472  self.check_python_updates,
473  self.fetch_addon_stats,
474  self.fetch_addon_score,
475  self.select_addon,
476  ]
477  pref = FreeCAD.ParamGet("User parameter:BaseApp/Preferences/Addons")
478  if pref.GetBool("DownloadMacros", True):
479  self.startup_sequence.append(self.load_macro_metadata)
480  self.number_of_progress_regions = len(self.startup_sequence)
481  self.current_progress_region = 0
482  self.do_next_startup_phase()
483
