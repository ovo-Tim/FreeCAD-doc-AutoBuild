337 def external_pin_diameter_and_distance(z, m, a, x):
338  """Calculates the ideal pin diameter for over pins measurement and its distance
339  for extrnal spur gears.
340 
341  z is the number of teeth
342  m is the module, in millimeter
343  a is the pressure angle, in radians
344  x is the profile shift coefficient
345 
346  returns the tuple of ideal pin diameter and its center distance from the gear's center
347  """
348  # Equations taken from http://qtcgears.com/tools/catalogs/PDF_Q420/Tech.pdf
349  # Table 10-13 (1-4) and Table 10-14 (4a)
350 
351  # 1. Half Tooth Space Angle at Base Circle
352  nu = pi / (2 * z) - inv(a) - 2 * x * tan(a) / z
353 
354  # 2. The Pressure Angle at the Point Pin is Tangent to Tooth Surface
355  ap = acos(z * m * cos(a) / (z * m + 2 * x * m))
356 
357  # 3. The Pressure Angle at Pin Center
358  phi = tan(ap) + nu
359 
360  # 4. Ideal Pin Diameter
361  dp = z * m * cos(a) * (inv(phi) + nu)
362 
363  # 4a. Over Pins Measurement, even number of teeth
364  # As we return the distance from the gear's center, we need dm to pass thought this center
365  # and that's only the case for a dm for an even number of teeth. However, this center distance
366  # is also valid for an odd number of teeth, as we don't measure pin-to-pin but pin-to-center.
367  dm = z * m * cos(a) / cos(phi) + dp
368 
369  # Eq. 10-12 on page T46
370  rc = (dm - dp) / 2
371  return (dp, rc)

374 def internal_pin_diameter_and_distance(z, m, a, x):
375  """Calculates the ideal pin diameter for over pins measurement and its distance
376  for intrnal spur gears.
377 
378  z is the number of teeth
379  m is the module, in millimeter
380  a is the pressure angle, in radians
381  x is the profile shift coefficient
382 
383  returns the tuple of ideal pin diameter and its center distance from the gear's center
384  """
385  # Equations taken from http://qtcgears.com/tools/catalogs/PDF_Q420/Tech.pdf
386  # Table 10-17 (1-4) and Table 10-18 (4a)
387 
388  # 1. Half Tooth Space Angle at Base Circle
389  nu = pi / (2 * z) + inv(a) + 2 * x * tan(a) / z
390 
391  # 2. The Pressure Angle at the Point Pin is Tangent to Tooth Surface
392  ap = acos(z * m * cos(a) / (z * m + 2 * x * m))
393 
394  # 3. The Pressure Angle at Pin Center
395  phi = tan(ap) - nu
396 
397  # 4. Ideal Pin Diameter
398  dp = z * m * cos(a) * (nu - inv(phi))
399 
400  # 4a. Over Pins Measurement, even number of teeth
401  # As we return the distance from the gear's center, we need dm to pass thought this center
402  # and that's only the case for a dm for an even number of teeth. However, this center distance
403  # is also valid for an odd number of teeth, as we don't measure pin-to-pin but pin-to-center.
404  dm = z * m * cos(a) / cos(phi) - dp
405 
406  rc = (dm + dp) / 2
407  return (dp, rc)
