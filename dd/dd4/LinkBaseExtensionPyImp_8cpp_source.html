46 static bool getProperty(PropTmpMap& props,
47  const LinkBaseExtension::PropInfoMap& infoMap,
48  const PropMap& propMap,
49  PyObject* key,
50  PyObject* value)
51 {
52  std::ostringstream str;
53 
54  if (!PyUnicode_Check(key)) {
55  PyErr_SetString(PyExc_TypeError, "key must be a unicode string");
56  return false;
57  }
58  const char* keyStr = PyUnicode_AsUTF8(key);
59  auto it = infoMap.find(keyStr);
60  if (it == infoMap.end()) {
61  str << "unknown key '" << keyStr << "'";
62  PyErr_SetString(PyExc_KeyError, str.str().c_str());
63  return false;
64  }
65 
66  const char* valStr = nullptr;
67  if (key == value) {
68  valStr = keyStr;
69  }
70  else if (value != Py_None) {
71  if (!PyUnicode_Check(value)) {
72  PyErr_SetString(PyExc_TypeError, "value must be unicode string");
73  return false;
74  }
75  valStr = PyUnicode_AsUTF8(value);
76  }
77 
78  App::Property* prop = nullptr;
79  auto& info = it->second;
80  if (valStr) {
81  auto pIt = propMap.find(valStr);
82  if (pIt == propMap.end()) {
83  str << "cannot find property '" << valStr << "'";
84  PyErr_SetString(PyExc_ValueError, str.str().c_str());
85  return false;
86  }
87  prop = pIt->second;
88  if (!prop->isDerivedFrom(info.type)) {
89  str << "expect property '" << keyStr << "(" << valStr << ") to be derived from '"
90  << info.type.getName() << "', instead of '" << prop->getTypeId().getName() << "'";
91  PyErr_SetString(PyExc_TypeError, str.str().c_str());
92  }
93  }
94  props[keyStr] = std::make_pair(info.index, prop);
95  return true;
96 }

98 PyObject* LinkBaseExtensionPy::configLinkProperty(PyObject* args, PyObject* keywds)
99 {
100  auto ext = getLinkBaseExtensionPtr();
101  const auto& info = ext->getPropertyInfoMap();
102 
103  PropMap propMap;
104  ext->getExtendedContainer()->getPropertyMap(propMap);
105 
106  PropTmpMap props;
107 
108  if (args && PyTuple_Check(args)) {
109  for (Py_ssize_t pos = 0; pos < PyTuple_GET_SIZE(args); ++pos) {
110  auto key = PyTuple_GET_ITEM(args, pos);
111  if (!getProperty(props, info, propMap, key, key)) {
112  return nullptr;
113  }
114  }
115  }
116  if (keywds && PyDict_Check(keywds)) {
117  PyObject *key, *value;
118  Py_ssize_t pos = 0;
119  while (PyDict_Next(keywds, &pos, &key, &value)) {
120  if (!getProperty(props, info, propMap, key, value)) {
121  return nullptr;
122  }
123  }
124  }
125  for (auto& v : props) {
126  ext->setProperty(v.second.first, v.second.second);
127  }
128  Py_Return;
129 }

203  Py::TupleN ret(Py::String(info.type.getName()), Py::String(info.doc));

215 void parseLink(LinkBaseExtension* ext, int index, PyObject* value)
216 {
217  App::DocumentObject* obj = nullptr;
218  PropertyStringList subs;
219  PropertyString sub;
220  if (value != Py_None) {
221  if (PyObject_TypeCheck(value, &DocumentObjectPy::Type)) {
222  obj = static_cast<DocumentObjectPy*>(value)->getDocumentObjectPtr();
223  }
224  else if (!PySequence_Check(value)) {
225  throw Base::TypeError("Expects type of DocumentObject or sequence");
226  }
227  else {
228  Py::Sequence seq(value);
229  if (seq[0].ptr() != Py_None) {
230  if (!PyObject_TypeCheck(seq[0].ptr(), &DocumentObjectPy::Type)) {
231  throw Base::TypeError(
232  "Expects the first argument to be DocumentObject in sequence");
233  }
234  obj = static_cast<DocumentObjectPy*>(seq[0].ptr())->getDocumentObjectPtr();
235  if (seq.size() > 1) {
236  sub.setPyObject(seq[1].ptr());
237  if (seq.size() > 2) {
238  subs.setPyObject(seq[2].ptr());
239  }
240  }
241  }
242  }
243  }
244  ext->setLink(index, obj, sub.getValue(), subs.getValue());
245 }

247 PyObject* LinkBaseExtensionPy::setLink(PyObject* _args)
248 {
249  Py::Sequence args(_args);
250  PY_TRY
251  {
252  auto ext = getLinkBaseExtensionPtr();
253  PyObject* pcObj = args.size() ? args[0].ptr() : Py_None;
254  if (pcObj == Py_None) {
255  ext->setLink(-1, nullptr);
256  }
257  else if (PyDict_Check(pcObj)) {
258  PyObject *key, *value;
259  Py_ssize_t pos = 0;
260  while (PyDict_Next(pcObj, &pos, &key, &value)) {
261  parseLink(ext, Py::Int(key), value);
262  }
263  }
264  else if (PySequence_Check(pcObj)) {
265  ext->setLink(-1, nullptr);
266  Py::Sequence seq(pcObj);
267  for (Py_ssize_t i = 0; i < seq.size(); ++i) {
268  parseLink(ext, i, seq[i].ptr());
269  }
270  }
271  else {
272  parseLink(ext, -1, _args);
273  }
274 
275  Py_Return;
276  }
277  PY_CATCH
278 }
