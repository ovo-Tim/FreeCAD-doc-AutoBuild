36 DlgSettingsFemGeneralImp::DlgSettingsFemGeneralImp(QWidget* parent)
37  : PreferencePage(parent)
38  , ui(new Ui_DlgSettingsFemGeneralImp)
39 {
40  ui->setupUi(this);
41 
42  // fill solvers combo with available solvers
43  ui->cmb_def_solver->clear();
44  std::vector<std::string> Solvers = {"None"};
45 
46  if (!Fem::Tools::checkIfBinaryExists("Ccx", "ccx", "ccx").empty()) {
47  Solvers.emplace_back("CalculiX");
48  }
49  if (!Fem::Tools::checkIfBinaryExists("Elmer", "elmer", "ElmerSolver").empty()) {
50  Solvers.emplace_back("Elmer");
51  }
52  // also check the multi-CPU Elmer build
53  else if (!Fem::Tools::checkIfBinaryExists("Elmer", "elmer", "ElmerSolver_mpi").empty()) {
54  Solvers.emplace_back("Elmer");
55  }
56  if (!Fem::Tools::checkIfBinaryExists("Mystran", "mystran", "mystran").empty()) {
57  Solvers.emplace_back("Mystran");
58  }
59  if (!Fem::Tools::checkIfBinaryExists("Z88", "z88", "z88r").empty()) {
60  Solvers.emplace_back("Z88");
61  }
62 
63  QStringList solversList;
64  for (auto item : Solvers) {
65  solversList << QLatin1String(item.c_str());
66  }
67  ui->cmb_def_solver->addItems(solversList);
68 
69  // if the "DefaultSolver" parameter is not yet set and there is only
70  // one available solver, set this solver
71  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
72  "User parameter:BaseApp/Preferences/Mod/Fem/General");
73  auto DefaultSolver = hGrp->GetInt("DefaultSolver", 0);
74  if (!DefaultSolver && ui->cmb_def_solver->count() == 2) {
75  ui->cmb_def_solver->setCurrentIndex(1);
76  }
77 }
