89  int ret;
90  // An arc created with Part.Arc will be converted into a Part.ArcOfCircle
91  if (geo->is<Part::GeomTrimmedCurve>()) {
92  Handle(Geom_TrimmedCurve) trim = Handle(Geom_TrimmedCurve)::DownCast(geo->handle());
93  Handle(Geom_Circle) circle = Handle(Geom_Circle)::DownCast(trim->BasisCurve());
94  Handle(Geom_Ellipse) ellipse = Handle(Geom_Ellipse)::DownCast(trim->BasisCurve());
95  if (!circle.IsNull()) {

97  Part::GeomArcOfCircle aoc;
98  aoc.setHandle(trim);
99  ret = this->getSketchObjectPtr()->addGeometry(&aoc, isConstruction);
100  }
101  else if (!ellipse.IsNull()) {
102  // create the definition struct for that geom
103  Part::GeomArcOfEllipse aoe;
104  aoe.setHandle(trim);
105  ret = this->getSketchObjectPtr()->addGeometry(&aoe, isConstruction);
106  }
107  else {
108  std::stringstream str;
109  str << "Unsupported geometry type: " << geo->getTypeId().getName();
110  PyErr_SetString(PyExc_TypeError, str.str().c_str());
111  return nullptr;
112  }
113  }
114  else if (geo->is<Part::GeomPoint>() || geo->is<Part::GeomCircle>()
115  || geo->is<Part::GeomEllipse>() || geo->is<Part::GeomArcOfCircle>()
116  || geo->is<Part::GeomArcOfEllipse>() || geo->is<Part::GeomArcOfHyperbola>()

124  PyErr_SetString(PyExc_TypeError, str.str().c_str());

142  Handle(Geom_Circle) circle = Handle(Geom_Circle)::DownCast(trim->BasisCurve());

359 PyObject* SketchObjectPy::addConstraint(PyObject* args)
360 {
361  PyObject* pcObj;
362  if (!PyArg_ParseTuple(args, "O", &pcObj)) {
363  return nullptr;
364  }
365 
366  if (PyObject_TypeCheck(pcObj, &(Sketcher::ConstraintPy::Type))) {
367  Sketcher::Constraint* constr =
368  static_cast<Sketcher::ConstraintPy*>(pcObj)->getConstraintPtr();
369  if (!this->getSketchObjectPtr()->evaluateConstraint(constr)) {
370  PyErr_SetString(PyExc_IndexError, "Constraint has invalid indexes");
371  return nullptr;
372  }
373  int ret = this->getSketchObjectPtr()->addConstraint(constr);
374  // this solve is necessary because:
375  // 1. The addition of constraint is part of a command addition
376  // 2. This solve happens before the command is committed
377  // 3. A constraint, may effect a geometry change (think of coincident,
378  // a line's point moves to meet the other line's point
379  // 4. The transaction is committed before any other solve, for example
380  // the one of execute() triggered by a recompute (UpdateActive) is generated.
381  // 5. Upon "undo", the constraint is removed (it was before the command was committed)
382  // however, the geometry changed after the command was committed, so the point that
383  // moved do not go back to the position where it was.
384  //
385  // N.B.: However, the solve itself may be inhibited in cases where groups of
386  // geometry/constraints
387  // are added together, because in that case undoing will also make the geometry
388  // disappear.
389  this->getSketchObjectPtr()->solve();
390  // if the geometry moved during the solve, then the initial solution is invalid
391  // at this point, so a point movement may not work in cases where redundant constraints
392  // exist. this forces recalculation of the initial solution (not a full solve)
393  if (this->getSketchObjectPtr()->noRecomputes) {
394  this->getSketchObjectPtr()->setUpSketch();
395  this->getSketchObjectPtr()->Constraints.touch(); // update solver information
396  }
397  return Py::new_reference_to(Py::Long(ret));
398  }
399  else if (PyObject_TypeCheck(pcObj, &(PyList_Type))
400  || PyObject_TypeCheck(pcObj, &(PyTuple_Type))) {
401  std::vector<Constraint*> values;
402  Py::Sequence list(pcObj);
403  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
404  if (PyObject_TypeCheck((*it).ptr(), &(ConstraintPy::Type))) {
405  Constraint* con = static_cast<ConstraintPy*>((*it).ptr())->getConstraintPtr();
406  values.push_back(con);
407  }
408  }
409 
410  for (std::vector<Constraint*>::iterator it = values.begin(); it != values.end(); ++it) {
411  if (!this->getSketchObjectPtr()->evaluateConstraint(*it)) {
412  PyErr_SetString(
413  PyExc_IndexError,
414  QT_TRANSLATE_NOOP(
415  "Notifications",
416  "The constraint has invalid index information and is malformed."));
417  return nullptr;
418  }
419  }
420  int ret = getSketchObjectPtr()->addConstraints(values) + 1;
421  std::size_t numCon = values.size();
422  Py::Tuple tuple(numCon);
423  for (std::size_t i = 0; i < numCon; ++i) {
424  int conId = ret - int(numCon - i);
425  tuple.setItem(i, Py::Long(conId));
426  }
427  return Py::new_reference_to(tuple);
428  }
429 
430  std::string error = std::string("type must be 'Constraint' or list of 'Constraint', not ");
431  error += pcObj->ob_type->tp_name;
432  throw Py::TypeError(error);
433 }
