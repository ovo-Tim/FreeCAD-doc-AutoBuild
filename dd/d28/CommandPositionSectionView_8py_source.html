114  def getTrianglePoint(self,p1,dir,p2):
115  '''
116  Get third point of a perpendicular triangle
117  p1, p2 ...vertexes of hypothenusis, dir ...direction of one kathete, p3 ...3rd vertex
118  '''
119  a = -dir.y
120  b = dir.x
121  c1 = p1.x * a + p1.y * b
122  c2 = -p2.x * b + p2.y * a
123  ab = a * a + b * b
124  x = (c1 * a - c2 * b) / ab
125  y = (c2 * a + c1 * b) / ab
126  return App.Vector(x,y,0.0)
