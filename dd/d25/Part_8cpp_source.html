65 static App::Part* _getPartOfObject(const DocumentObject* obj,
66  std::set<const DocumentObject*>* objset)
67 {
68  // as a Part is a geofeaturegroup it must directly link to all
69  // objects it contains, even if they are in additional groups etc.
70  // But we still must call 'hasObject()' to exclude link brought in by
71  // expressions.
72  for (auto inObj : obj->getInList()) {
73  if (objset && !objset->insert(inObj).second) {
74  continue;
75  }
76  auto group = inObj->getExtensionByType<GeoFeatureGroupExtension>(true);
77  if (group && group->hasObject(obj)) {
78  if (inObj->isDerivedFrom(App::Part::getClassTypeId())) {
79  return static_cast<App::Part*>(inObj);
80  }
81  else if (objset) {
82  return _getPartOfObject(inObj, objset);
83  }
84  // Only one parent geofeature group per object, so break
85  break;
86  }
87  }
88 
89  return nullptr;
90 }

92 App::Part* Part::getPartOfObject(const DocumentObject* obj, bool recursive)
93 {
94  if (!recursive) {
95  return _getPartOfObject(obj, nullptr);
96  }
97  std::set<const DocumentObject*> objset;
98  objset.insert(obj);
99  return _getPartOfObject(obj, &objset);
100 }
