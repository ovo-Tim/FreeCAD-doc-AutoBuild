406 def get_extended_wire(wire, offset_start, offset_end):
407  """Return a wire trimmed (negative offset) or extended (positive offset) at its first vertex, last vertex or both ends.
408 
409  get_extended_wire(wire, -100.0, 0.0) -> returns a copy of the wire with its first 100 mm removed
410  get_extended_wire(wire, 0.0, 100.0) -> returns a copy of the wire extended by 100 mm after it's last vertex
411  """

413  App.Console.PrintError("debug: get_extended_wire error, wire's length insufficient for trimming.\n")
414  return wire

416  offset_start = -offset_start

418  for edge in wire.OrderedEdges:

420  offset_start -= edge.Length

422  if edge.Orientation == "Forward":
