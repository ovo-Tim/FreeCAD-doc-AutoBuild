245  void createShape(bool onlyeditoutline) override
246  {
247  Sketcher::SketchObject* Obj = sketchgui->getSketchObject();
248 
249  ShapeGeometry.clear();
250 
251  int numberOfCopiesToMake = numberOfCopies;
252  if (numberOfCopies == 0) {
253  numberOfCopiesToMake = 1;
254  deleteOriginal = 1;
255  }
256  else {
257  deleteOriginal = 0;
258  }
259 
260  for (int k = 0; k < secondNumberOfCopies; k++) {
261  for (int i = 0; i <= numberOfCopiesToMake; i++) {
262  if ((k == 0 && i == 0)) {
263  continue;
264  }
265  Base::Vector3d vec = firstTranslationVector * i + secondTranslationVector * k;
266 
267  for (auto& geoId : listOfGeoIds) {
268  const Part::Geometry* pGeo = Obj->getGeometry(geoId);
269  auto geoUniquePtr = std::unique_ptr<Part::Geometry>(pGeo->copy());
270  Part::Geometry* geo = geoUniquePtr.get();
271 
272  if (isCircle(*geo)) {
273  Part::GeomCircle* circle = static_cast<Part::GeomCircle*>(geo); // NOLINT
274  circle->setCenter(circle->getCenter() + vec);
275  }
276  else if (isArcOfCircle(*geo)) {
277  Part::GeomArcOfCircle* arc =
278  static_cast<Part::GeomArcOfCircle*>(geo); // NOLINT
279  arc->setCenter(arc->getCenter() + vec);
280  }
281  else if (isEllipse(*geo)) {
282  Part::GeomEllipse* ellipse =
283  static_cast<Part::GeomEllipse*>(geo); // NOLINT
284  ellipse->setCenter(ellipse->getCenter() + vec);
285  }
286  else if (isArcOfEllipse(*geo)) {
287  Part::GeomArcOfEllipse* aoe =
288  static_cast<Part::GeomArcOfEllipse*>(geo); // NOLINT
289  aoe->setCenter(aoe->getCenter() + vec);
290  }
291  else if (isArcOfHyperbola(*geo)) {
292  Part::GeomArcOfHyperbola* aoh =
293  static_cast<Part::GeomArcOfHyperbola*>(geo); // NOLINT
294  aoh->setCenter(aoh->getCenter() + vec);
295  }
296  else if (isArcOfParabola(*geo)) {
297  Part::GeomArcOfParabola* aop =
298  static_cast<Part::GeomArcOfParabola*>(geo); // NOLINT
299  aop->setCenter(aop->getCenter() + vec);
300  }
301  else if (isLineSegment(*geo)) {
302  auto* line = static_cast<Part::GeomLineSegment*>(geo); // NOLINT
303  line->setPoints(line->getStartPoint() + vec, line->getEndPoint() + vec);
304  }
305  else if (isBSplineCurve(*geo)) {
306  auto* bSpline = static_cast<Part::GeomBSplineCurve*>(geo); // NOLINT
307  std::vector<Base::Vector3d> poles = bSpline->getPoles();
308  for (size_t p = 0; p < poles.size(); p++) {
309  poles[p] = poles[p] + vec;
310  }
311  bSpline->setPoles(poles);
312  }
313  else if (isPoint(*geo)) {
314  auto* point = static_cast<Part::GeomPoint*>(geo); // NOLINT
315  point->setPoint(point->getPoint() + vec);
316  }
317 
318  ShapeGeometry.emplace_back(std::move(geoUniquePtr));
319  }
320  }
321  }
322 
323  if (onlyeditoutline) {
324  // Add the lines to show angle
325  if (firstTranslationVector.Length() > Precision::Confusion()) {
326  addLineToShapeGeometry(toVector3d(referencePoint),
327  toVector3d(firstTranslationPoint),
328  true);
329  }
330 
331  if (secondTranslationVector.Length() > Precision::Confusion()) {
332  addLineToShapeGeometry(toVector3d(referencePoint),
333  toVector3d(secondTranslationPoint),
334  true);
335  }
336  }
337  else {
338  int firstCurveCreated = getHighestCurveIndex() + 1;
339  int size = static_cast<int>(listOfGeoIds.size());
340 
341  const std::vector<Sketcher::Constraint*>& vals = Obj->Constraints.getValues();
342  // avoid applying equal several times if cloning distanceX and distanceY of the
343  // same part.
344  std::vector<int> geoIdsWhoAlreadyHasEqual = {};
345 
346  for (auto& cstr : vals) {
347  int firstIndex = indexOfGeoId(listOfGeoIds, cstr->First);
348  int secondIndex = indexOfGeoId(listOfGeoIds, cstr->Second);
349  int thirdIndex = indexOfGeoId(listOfGeoIds, cstr->Third);
350 
351  for (int k = 0; k < secondNumberOfCopies; k++) {
352  for (int i = 0; i <= numberOfCopiesToMake; i++) {
353  if (k == 0 && i == 0) {
354  continue;
355  }
356 
357  int firstIndexi = firstCurveCreated + firstIndex + size * (i - 1)
358  + size * (numberOfCopiesToMake + 1) * k;
359  int secondIndexi = firstCurveCreated + secondIndex + size * (i - 1)
360  + size * (numberOfCopiesToMake + 1) * k;
361  int thirdIndexi = firstCurveCreated + thirdIndex + size * (i - 1)
362  + size * (numberOfCopiesToMake + 1) * k;
363 
364  auto newConstr = std::unique_ptr<Constraint>(cstr->copy());
365  newConstr->First = firstIndexi;
366 
367  if ((cstr->Type == Symmetric || cstr->Type == Tangent
368  || cstr->Type == Perpendicular || cstr->Type == Angle)
369  && firstIndex >= 0 && secondIndex >= 0 && thirdIndex >= 0) {
370  newConstr->Second = secondIndexi;
371  newConstr->Third = thirdIndexi;
372  }
373  else if ((cstr->Type == Coincident || cstr->Type == Tangent
374  || cstr->Type == Symmetric || cstr->Type == Perpendicular
375  || cstr->Type == Parallel || cstr->Type == Equal
376  || cstr->Type == Angle || cstr->Type == PointOnObject
377  || cstr->Type == Horizontal || cstr->Type == Vertical
378  || cstr->Type == InternalAlignment)
379  && firstIndex >= 0 && secondIndex >= 0
380  && thirdIndex == GeoEnum::GeoUndef) {
381  newConstr->Second = secondIndexi;
382  }
383  else if ((cstr->Type == Radius || cstr->Type == Diameter
384  || cstr->Type == Weight)
385  && firstIndex >= 0) {
386  if (deleteOriginal || !cloneConstraints) {
387  newConstr->setValue(cstr->getValue());
388  }
389  else {
390  newConstr->Type = Equal;
391  newConstr->First = cstr->First;
392  newConstr->Second = firstIndexi;
393  }
394  }
395  else if ((cstr->Type == Distance || cstr->Type == DistanceX
396  || cstr->Type == DistanceY)
397  && firstIndex >= 0 && secondIndex >= 0) {
398  if (!deleteOriginal && cloneConstraints
399  && cstr->First == cstr->Second) { // only line distances
400  if (indexOfGeoId(geoIdsWhoAlreadyHasEqual, secondIndexi) != -1) {
401  continue;
402  }
403  newConstr->Type = Equal;
404  newConstr->First = cstr->First;
405  newConstr->Second = secondIndexi;
406  geoIdsWhoAlreadyHasEqual.push_back(secondIndexi);
407  }
408  else {
409  newConstr->Second = secondIndexi;
410  }
411  }
412  else if ((cstr->Type == Block || cstr->Type == Horizontal
413  || cstr->Type == Vertical)
414  && firstIndex >= 0) {
415  newConstr->First = firstIndexi;
416  }
417  else {
418  continue;
419  }
420 
421  ShapeConstraints.push_back(std::move(newConstr));
422  }
423  }
424  }
425  }
426  }

556 template<>
557 void DSHTranslateControllerBase::doEnforceControlParameters(Base::Vector2d& onSketchPos)
558 {
559  switch (handler->state()) {
560  case SelectMode::SeekFirst: {
561  if (onViewParameters[OnViewParameter::First]->isSet) {
562  onSketchPos.x = onViewParameters[OnViewParameter::First]->getValue();
563  }
564 
565  if (onViewParameters[OnViewParameter::Second]->isSet) {
566  onSketchPos.y = onViewParameters[OnViewParameter::Second]->getValue();
567  }
568  } break;
569  case SelectMode::SeekSecond: {
570  Base::Vector2d dir = onSketchPos - handler->referencePoint;
571  if (dir.Length() < Precision::Confusion()) {
572  dir.x = 1.0; // if direction null, default to (1,0)
573  }
574  double length = dir.Length();
575 
576  if (onViewParameters[OnViewParameter::Third]->isSet) {
577  length = onViewParameters[OnViewParameter::Third]->getValue();
578  if (length < Precision::Confusion()) {
579  unsetOnViewParameter(onViewParameters[OnViewParameter::Third].get());
580  return;
581  }
582 
583  onSketchPos = handler->referencePoint + length * dir.Normalize();
584  }
585 
586  if (onViewParameters[OnViewParameter::Fourth]->isSet) {
587  double angle = onViewParameters[OnViewParameter::Fourth]->getValue() * M_PI / 180;
588  onSketchPos.x = handler->referencePoint.x + cos(angle) * length;
589  onSketchPos.y = handler->referencePoint.y + sin(angle) * length;
590  }
591  } break;
592  case SelectMode::SeekThird: {
593  Base::Vector2d dir = onSketchPos - handler->referencePoint;
594  if (dir.Length() < Precision::Confusion()) {
595  dir.x = 1.0; // if direction null, default to (1,0)
596  }
597  double length = dir.Length();
598 
599  if (onViewParameters[OnViewParameter::Fifth]->isSet) {
600  length = onViewParameters[OnViewParameter::Fifth]->getValue();
601  if (length < Precision::Confusion()) {
602  unsetOnViewParameter(onViewParameters[OnViewParameter::Fifth].get());
603  return;
604  }
605 
606  onSketchPos = handler->referencePoint + length * dir.Normalize();
607  }
608 
609  if (onViewParameters[OnViewParameter::Sixth]->isSet) {
610  double angle = onViewParameters[OnViewParameter::Sixth]->getValue() * M_PI / 180;
611  onSketchPos.x = handler->referencePoint.x + cos(angle) * length;
612  onSketchPos.y = handler->referencePoint.y + sin(angle) * length;
613  }
614  } break;
615  default:
616  break;
617  }

620 template<>
621 void DSHTranslateController::adaptParameters(Base::Vector2d onSketchPos)
622 {
623  switch (handler->state()) {
624  case SelectMode::SeekFirst: {
625  if (!onViewParameters[OnViewParameter::First]->isSet) {
626  setOnViewParameterValue(OnViewParameter::First, onSketchPos.x);
627  }
628 
629  if (!onViewParameters[OnViewParameter::Second]->isSet) {
630  setOnViewParameterValue(OnViewParameter::Second, onSketchPos.y);
631  }
632 
633  bool sameSign = onSketchPos.x * onSketchPos.y > 0.;
634  onViewParameters[OnViewParameter::First]->setLabelAutoDistanceReverse(!sameSign);
635  onViewParameters[OnViewParameter::Second]->setLabelAutoDistanceReverse(sameSign);
636  onViewParameters[OnViewParameter::First]->setPoints(Base::Vector3d(),
637  toVector3d(onSketchPos));
638  onViewParameters[OnViewParameter::Second]->setPoints(Base::Vector3d(),
639  toVector3d(onSketchPos));
640  } break;
641  case SelectMode::SeekSecond: {
642  if (!onViewParameters[OnViewParameter::Third]->isSet) {
643  double length = (onSketchPos - handler->referencePoint).Length();
644  setOnViewParameterValue(OnViewParameter::Third, length);
645  }
646 
647  Base::Vector2d vec2d = Base::Vector2d(handler->firstTranslationVector.x,
648  handler->firstTranslationVector.y);
649  double angle = vec2d.Angle();
650  double range = angle * 180 / M_PI;
651 
652  if (!onViewParameters[OnViewParameter::Fourth]->isSet) {
653  setOnViewParameterValue(OnViewParameter::Fourth, range, Base::Unit::Angle);
654  }
655 
656  Base::Vector3d start = toVector3d(handler->referencePoint);
657  Base::Vector3d end = toVector3d(onSketchPos);
658 
659  onViewParameters[OnViewParameter::Third]->setPoints(start, end);
660  onViewParameters[OnViewParameter::Fourth]->setPoints(start, Base::Vector3d());
661  onViewParameters[OnViewParameter::Fourth]->setLabelRange(angle);
662  } break;
663  case SelectMode::SeekThird: {
664  if (!onViewParameters[OnViewParameter::Fifth]->isSet) {
665  double length = (onSketchPos - handler->referencePoint).Length();
666  setOnViewParameterValue(OnViewParameter::Fifth, length);
