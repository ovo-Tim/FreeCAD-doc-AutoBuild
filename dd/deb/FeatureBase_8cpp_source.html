59 App::DocumentObjectExecReturn* FeatureBase::execute()
60 {
61 
62  if (!BaseFeature.getValue()) {
63  return new App::DocumentObjectExecReturn(
64  QT_TRANSLATE_NOOP("Exception", "BaseFeature link is not set"));
65  }
66 
67  if (!BaseFeature.getValue()->isDerivedFrom(Part::Feature::getClassTypeId())) {
68  return new App::DocumentObjectExecReturn(
69  QT_TRANSLATE_NOOP("Exception", "BaseFeature must be a Part::Feature"));
70  }
71 
72  auto shape = Part::Feature::getTopoShape(BaseFeature.getValue());
73  if (!shape.countSubShapes(TopAbs_SOLID)) {
74  shape = shape.makeElementSolid();
75  }
76  if (shape.isNull()) {
77  return new App::DocumentObjectExecReturn(
78  QT_TRANSLATE_NOOP("Exception", "BaseFeature has an empty shape"));
79  }
80 
81  Shape.setValue(shape);
82 
83  return StdReturn;
84 }
