192 // Export an instance of the base class (to avoid warning C4275, see also
193 // C++ Language Reference/General Rules and Limitations on MSDN for more details.)
194 //
195 // For compiler gcc4.1 we need to define the template class outside namespace 'Gui'
196 // otherwise we get the compiler error:
197 // 'explicit instantiation of 'class Base::Subject<const Gui::SelectionChanges&>'
198 // in namespace 'Gui' (which does not enclose namespace 'Base')
199 //
200 // It seems that this construct is not longer needed for gcc4.4 and even leads to
201 // errors under Mac OS X. Thus, we check for version between 4.1 and 4.4.
202 // It seems that for Mac OS X this can be completely ignored

744 template<typename T>
745 inline std::vector<T*> SelectionSingleton::getObjectsOfType(const char* pDocName, ResolveMode resolve) const
746 {
747  std::vector<T*> type;
748  std::vector<App::DocumentObject*> obj = this->getObjectsOfType(T::getClassTypeId(), pDocName, resolve);
749  type.reserve(obj.size());
750  for (auto it = obj.begin(); it != obj.end(); ++it)
751  type.push_back(static_cast<T*>(*it));
