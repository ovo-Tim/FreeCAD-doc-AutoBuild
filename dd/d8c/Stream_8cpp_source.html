474 std::streambuf::int_type IODeviceIStreambuf::underflow()
475 {
476 #ifndef _MSC_VER
477  using std::memcpy;
478 #endif
479 
480  // is read position before end of buffer?
481  if (gptr() < egptr()) {
482  return *gptr();
483  }
484 
485  /* process size of putback area
486  * - use number of characters read
487  * - but at most size of putback area
488  */
489  int numPutback {};
490  numPutback = gptr() - eback();
491  if (numPutback > pbSize) {
492  numPutback = pbSize;
493  }
494 
495  /* copy up to pbSize characters previously read into
496  * the putback area
497  */
498  memcpy(buffer + (pbSize - numPutback), gptr() - numPutback, numPutback);
499 
500  // read at most bufSize new characters
501  int num {};
502  num = device->read(buffer + pbSize, bufSize);
503  if (num <= 0) {
504  // ERROR or EOF
505  return EOF;
506  }
507 
508  // reset buffer pointers
509  setg(buffer + (pbSize - numPutback), // beginning of putback area
510  buffer + pbSize, // read position
511  buffer + pbSize + num); // end of buffer
512 
513  // return next character
514  return *gptr();
515 }

578 PyStreambuf::int_type PyStreambuf::underflow()
579 {
580  if (gptr() < egptr()) {
581  return traits_type::to_int_type(*gptr());
582  }
583 
584  char* base = &buffer.front();
585  char* start = base;
586 
587  if (eback() == base) { // true when this isn't the first fill
588  std::memmove(base, egptr() - put_back, put_back);
589  start += put_back;
590  }
591 
592  std::size_t n {};
593  Py::Tuple arg(1);
594  long len = static_cast<long>(buffer.size() - (start - base));
595  arg.setItem(0, Py::Long(len));
596  Py::Callable meth(Py::Object(inp).getAttr("read"));
597 
598  try {
599  std::string c;
600  Py::Object res(meth.apply(arg));
601  if (res.isBytes()) {
602  c = static_cast<std::string>(Py::Bytes(res));
603  }
604  else if (res.isString()) {
605  c = static_cast<std::string>(Py::String(res));
606  }
607  else {
608  // wrong type
609  return traits_type::eof();
610  }
611 
612  n = c.size();
613  if (n == 0) {
614  return traits_type::eof();
615  }
616 
617  // Check: bugprone-not-null-terminated-result
618  std::memcpy(start, c.data(), c.size());
619  }
620  catch (Py::Exception& e) {
621  e.clear();
622  return traits_type::eof();
623  }
624 
625  setg(base, start, start + n);
626  return traits_type::to_int_type(*gptr());
627 }

671 bool PyStreambuf::writeStr(const char* str, std::streamsize num)
672 {
673  try {
674  Py::Tuple arg(1);
675  Py::Callable meth(Py::Object(inp).getAttr("write"));
676 
677  if (type == StringIO) {
678  arg.setItem(0, Py::String(str, num));
679  meth.apply(arg);
680  return true;
681  }
682  if (type == BytesIO) {
683  arg.setItem(0, Py::Bytes(str, num));
684  meth.apply(arg);
685  return true;
686  }
687 
688  // try out what works
689  try {
690  arg.setItem(0, Py::String(str, num));
691  meth.apply(arg);
692  type = StringIO;
693  return true;
694  }
695  catch (Py::Exception& e) {
696  if (PyErr_ExceptionMatches(PyExc_TypeError)) {
697  e.clear();
698  arg.setItem(0, Py::Bytes(str, num));
699  meth.apply(arg);
700  type = BytesIO;
701  return true;
702  }
703 
704  throw; // re-throw
705  }
706  }
707  catch (Py::Exception& e) {
708  e.clear();
709  }
710 
711  return false;
712 }

726 PyStreambuf::pos_type PyStreambuf::seekoff(PyStreambuf::off_type offset,
727  PyStreambuf::seekdir dir,
728  PyStreambuf::openmode /*mode*/)
729 {
730  int whence = 0;
731  switch (dir) {
732  case std::ios_base::beg:
733  whence = 0;
734  break;
735  case std::ios_base::cur:
736  whence = 1;
737  break;
738  case std::ios_base::end:
739  whence = 2;
740  break;
741  default:
742  return {off_type(-1)};
743  }
744 
745  try {
746  Py::Tuple arg(2);
747  arg.setItem(0, Py::Long(static_cast<long>(offset)));
748  arg.setItem(1, Py::Long(whence));
749  Py::Callable seek(Py::Object(inp).getAttr("seek"));
750  seek.apply(arg);
751 
752  // get current position
753  Py::Tuple arg2;
754  Py::Callable tell(Py::Object(inp).getAttr("tell"));
755  Py::Long pos(tell.apply(arg2));
756  long cur_pos = static_cast<long>(pos);
757  return static_cast<pos_type>(cur_pos);
758  }
759  catch (Py::Exception& e) {
760  e.clear();
761  return {off_type(-1)};
762  }
763 }
