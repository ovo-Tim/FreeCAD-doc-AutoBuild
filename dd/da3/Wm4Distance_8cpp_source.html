92 template <class Real, class TVector>
93 Real Distance<Real,TVector>::GetDerivative (Real fT,
94  const TVector& rkVelocity0, const TVector& rkVelocity1)
95 {
96  // Use a finite difference approximation: f'(t) = (f(t+h)-f(t-h))/(2*h)
97  Real fFp = Get(fT+m_fDifferenceStep,rkVelocity0,rkVelocity1);
98  Real fFm = Get(fT-m_fDifferenceStep,rkVelocity0,rkVelocity1);
99  Real fDFApprox = m_fInvTwoDifferenceStep*(fFp-fFm);
100  return fDFApprox;
101 }

103 template <class Real, class TVector>
104 Real Distance<Real,TVector>::GetDerivativeSquared (Real fT,
105  const TVector& rkVelocity0, const TVector& rkVelocity1)
106 {
107  // A derived class should override this only if there is a faster method
108  // to compute the derivative of the squared distance for the specific
109  // class.
110  Real fDistance = Get(fT,rkVelocity0,rkVelocity1);
111  Real fDerivative = GetDerivative(fT,rkVelocity0,rkVelocity1);
112  return ((Real)2.0)*fDistance*fDerivative;
113 }
114 //----------------------------------------------------------------------------
115 template <class Real, class TVector>
116 Real Distance<Real,TVector>::Get (Real fTMin, Real fTMax,
117  const TVector& rkVelocity0, const TVector& rkVelocity1)
118 {
119  // The assumption is that distance f(t) is a convex function. If
120  // f'(tmin) >= 0, then the minimum occurs at tmin. If f'(tmax) <= 0,
121  // then the minimum occurs at tmax. Otherwise, f'(0) < 0 and
122  // f'(tmax) > 0 and the minimum occurs at some t in (tmin,tmax).
123 
124  Real fT0 = fTMin;
125  Real fF0 = Get(fT0,rkVelocity0,rkVelocity1);
126  if (fF0 <= ZeroThreshold)
127  {
128  // The distance is effectively zero. The objects are initially in
129  // contact.
130  m_fContactTime = fT0;
131  return (Real)0.0;
132  }
133  Real fDF0 = GetDerivative(fT0,rkVelocity0,rkVelocity1);
134  if (fDF0 >= (Real)0.0)
135  {
136  // The distance is increasing on [0,tmax].
137  m_fContactTime = fT0;
138  return fF0;
139  }
140 
141  Real fT1 = fTMax;
142  Real fF1 = Get(fT1,rkVelocity0,rkVelocity1);
143  if (fF1 <= ZeroThreshold)
144  {
145  // The distance is effectively zero.
146  m_fContactTime = fT1;
147  return (Real)0.0;
148  }
149  Real fDF1 = GetDerivative(fT1,rkVelocity0,rkVelocity1);
150  if (fDF1 <= (Real)0.0)
151  {
152  // The distance is decreasing on [0,tmax].
153  m_fContactTime = fT1;
154  return fF1;
155  }
156 
157  // Start the process with Newton's method for computing a time when the
158  // distance is zero. During this process we will switch to a numerical
159  // minimizer if we decide that the distance cannot be zero.
160  int i;
161  for (i = 0; i < MaximumIterations; i++)
162  {
163  // compute the next Newton's iterate
164  Real fT = fT0 - fF0/fDF0;
165  if (fT >= fTMax)
166  {
167  // The convexity of the graph guarantees that when this condition
168  // happens, the distance is always positive. Switch to a
169  // numerical minimizer.
170  break;
171  }
172 
173  Real fF = Get(fT,rkVelocity0,rkVelocity1);
174  if (fF <= ZeroThreshold)
175  {
176  // The distance is effectively zero.
177  m_fContactTime = fT;
178  return (Real)0.0;
179  }
180 
181  Real fDF = GetDerivative(fT,rkVelocity0,rkVelocity1);
182  if (fDF >= (Real)0.0)
183  {
184  // The convexity of the graph guarantees that when this condition
185  // happens, the distance is always positive. Switch to a
186  // numerical minimizer.
187  break;
188  }
189 
190  fT0 = fT;
191  fF0 = fF;
192  fDF0 = fDF;
193  }
194 
195  if (i == MaximumIterations)
196  {
197  // Failed to converge within desired number of iterations. To
198  // reach here, the derivative values were always negative, so report
199  // the distance at the last time.
200  m_fContactTime = fT0;
201  return fF0;
202  }
203 
204  // The distance is always positive. Use bisection to find the root of
205  // the derivative function.
206  Real fTm = fT0;
207  for (i = 0; i < MaximumIterations; i++)
208  {
209  fTm = ((Real)0.5)*(fT0 + fT1);
210  Real fDFm = GetDerivative(fTm,rkVelocity0,rkVelocity1);
211  Real fProduct = fDFm*fDF0;
212  if (fProduct < -ZeroThreshold)
213  {
214  fT1 = fTm;
215  fDF1 = fDFm;
216  }
217  else if (fProduct > ZeroThreshold)
218  {
219  fT0 = fTm;
220  fDF0 = fDFm;
221  }
222  else
223  {
224  break;
225  }
226  }
227 
228  // This is the time at which the minimum occurs and is not the contact
229  // time. Store it anyway for debugging purposes.
230  m_fContactTime = fTm;
231  Real fFm = Get(fTm,rkVelocity0,rkVelocity1);
232  return fFm;
233 }
