168 void paintGL()
169 {
170  const SbViewportRegion vp = view->getSoRenderManager()->getViewportRegion();
171  SbVec2s size = vp.getViewportSizePixels();
172 
173  glDisable(GL_LIGHTING);
174  glViewport(0, 0, size[0], size[1]);
175  glMatrixMode(GL_PROJECTION);
176  glLoadIdentity();
177  glMatrixMode(GL_MODELVIEW);
178  glLoadIdentity();
179  glDisable(GL_DEPTH_TEST);
180 
181  glClear(GL_COLOR_BUFFER_BIT);
182  glEnable(GL_TEXTURE_2D);
183  glBindTexture(GL_TEXTURE_2D, fbObject->texture());
184  glColor3f(1.0, 1.0, 1.0);
185  GLfloat s = size[0] / GLfloat(fbObject->size().width());
186  GLfloat t = size[1] / GLfloat(fbObject->size().height());
187 
188  glBegin(GL_QUADS);
189  glTexCoord2f(0.0, 0.0);
190  glVertex2f(-1.0, -1.0);
191  glTexCoord2f(s, 0.0);
192  glVertex2f(1.0, -1.0);
193  glTexCoord2f(s, t);
194  glVertex2f(1.0, 1.0);
195  glTexCoord2f(0.0, t);
196  glVertex2f(-1.0, 1.0);
197  glEnd();
198 
199  if (rubberBandIsShown) {
200  glMatrixMode(GL_PROJECTION);
201  glOrtho(0, size[0], size[1], 0, 0, 100);
202  glMatrixMode(GL_MODELVIEW);
203  glDisable(GL_TEXTURE_2D);
204  glEnable(GL_BLEND);
205  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
206  glLineWidth(4.0);
207  glColor4f(1.0f, 1.0f, 1.0f, 0.2f);
208  glRecti(rubberBandCorner1.x(), rubberBandCorner1.y(),
209  rubberBandCorner2.x(), rubberBandCorner2.y());
210  glColor4f(1.0, 1.0, 0.0, 0.5);
211  glLineStipple(3, 0xAAAA);
212  glEnable(GL_LINE_STIPPLE);
213 
214  glBegin(GL_LINE_LOOP);
215  glVertex2i(rubberBandCorner1.x(), rubberBandCorner1.y());
216  glVertex2i(rubberBandCorner2.x(), rubberBandCorner1.y());
217  glVertex2i(rubberBandCorner2.x(), rubberBandCorner2.y());
218  glVertex2i(rubberBandCorner1.x(), rubberBandCorner2.y());
219  glEnd();
220 
221  glLineWidth(1.0);
222  glDisable(GL_LINE_STIPPLE);
223  glDisable(GL_BLEND);
224  }
225 
226  glEnable(GL_LIGHTING);
227  glEnable(GL_DEPTH_TEST);
228 }

352 void DrawingPlane::draw ()
353 {return;
354  if (1/*mustRedraw*/) {
355  SbVec2s view = _pcView3D->getSoRenderManager()->getSize();
356  static_cast<QtGLWidget*>(_pcView3D->getGLWidget())->makeCurrent();
357  glMatrixMode(GL_PROJECTION);
358  glPushMatrix();
359  glLoadIdentity();
360  glOrtho(0, view[0], 0, view[1], -1, 1);
361 
362  // Store GL state
363  glPushAttrib(GL_ALL_ATTRIB_BITS);
364  GLfloat depthrange[2];
365  glGetFloatv(GL_DEPTH_RANGE, depthrange);
366  GLdouble projectionmatrix[16];
367  glGetDoublev(GL_PROJECTION_MATRIX, projectionmatrix);
368 
369  glDepthFunc(GL_ALWAYS);
370  glDepthMask(GL_TRUE);
371  glDepthRange(0,0);
372  glEnable(GL_DEPTH_TEST);
373  glDisable(GL_LIGHTING);
374  glEnable(GL_COLOR_MATERIAL);
375  glDisable(GL_BLEND);
376 
377  glLineWidth(1.0f);
378  glColor4f(1.0, 1.0, 1.0, 0.0);
379  glViewport(0, 0, view[0], view[1]);
380 
381  glEnable(GL_COLOR_LOGIC_OP);
382  glLogicOp(GL_XOR);
383  glDrawBuffer(GL_FRONT);
384 
385  glFlush();
386  glLogicOp(GL_COPY);
387  glDisable(GL_COLOR_LOGIC_OP);
388 
389  // Reset original state
390  glDepthRange(depthrange[0], depthrange[1]);
391  glMatrixMode(GL_PROJECTION);
392  glLoadMatrixd(projectionmatrix);
393 
394  glPopAttrib();
395  glPopMatrix();
396  }
