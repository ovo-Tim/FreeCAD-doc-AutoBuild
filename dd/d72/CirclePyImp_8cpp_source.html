42 // returns a string which represents the object e.g. when printed in python
43 std::string CirclePy::representation() const
44 {
45  Handle(Geom_Circle) circle = Handle(Geom_Circle)::DownCast(getGeomCirclePtr()->handle());
46  gp_Ax1 axis = circle->Axis();
47  gp_Dir dir = axis.Direction();
48  gp_Pnt loc = axis.Location();
49  Standard_Real fRad = circle->Radius();
50 
51  std::stringstream str;
52  str << "Circle (";

54  str << "Position : (" << loc.X() << ", "<< loc.Y() << ", "<< loc.Z() << "), ";
55  str << "Direction : (" << dir.X() << ", "<< dir.Y() << ", "<< dir.Z() << ")";
56  str << ")";
57 
58  return str.str();
59 }
