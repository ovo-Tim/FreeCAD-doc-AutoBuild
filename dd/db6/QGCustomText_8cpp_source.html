186 QRectF QGCustomText::tightBoundingRect() const
187 {
188  QFontMetrics qfm(font());
189  QRectF result = QGraphicsTextItem::boundingRect();
190  QRectF tight = qfm.tightBoundingRect(toPlainText());
191  qreal x_adj = (result.width() - tight.width())/4.0;
192  qreal y_adj = (result.height() - tight.height())/4.0;
193 
194  // Adjust the bounding box 50% towards the Qt tightBoundingRect(),
195  // except chomp some extra empty space above the font (1.75*y_adj)
196  // wf: this does not work with all fonts. it depends on where the glyph is located within
197  // the em square. see https://github.com/FreeCAD/FreeCAD/issues/11452
198  // TODO: how to know where the glyph is going to be placed?
199  result.adjust(x_adj, 1.75*y_adj, -x_adj, -y_adj);
200 
201  return result;
202 }
