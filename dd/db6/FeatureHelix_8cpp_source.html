127 App::DocumentObjectExecReturn* Helix::execute()
128 {
129 
130  if (onlyHasToRefine()){
131  TopoShape result = refineShapeIfActive(rawShape, RefineErrorPolicy::Warn);
132  Shape.setValue(result);
133  return App::DocumentObject::StdReturn;
134  }
135 
136  // Validate and normalize parameters
137  HelixMode mode = static_cast<HelixMode>(Mode.getValue());
138  if (mode == HelixMode::pitch_height_angle) {
139  if (Pitch.getValue() < Precision::Confusion())
140  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Pitch too small!"));
141  if (Height.getValue() < Precision::Confusion())
142  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: height too small!"));
143  Turns.setValue(Height.getValue() / Pitch.getValue());
144  Growth.setValue(Pitch.getValue() * tan(Base::toRadians(Angle.getValue())));
145  }
146  else if (mode == HelixMode::pitch_turns_angle) {
147  if (Pitch.getValue() < Precision::Confusion())
148  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: pitch too small!"));
149  if (Turns.getValue() < Precision::Confusion())
150  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: turns too small!"));
151  Height.setValue(Turns.getValue() * Pitch.getValue());
152  Growth.setValue(Pitch.getValue() * tan(Base::toRadians(Angle.getValue())));
153  }
154  else if (mode == HelixMode::height_turns_angle) {
155  if (Height.getValue() < Precision::Confusion())
156  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: height too small!"));
157  if (Turns.getValue() < Precision::Confusion())
158  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: turns too small!"));
159  Pitch.setValue(Height.getValue() / Turns.getValue());
160  Growth.setValue(Pitch.getValue() * tan(Base::toRadians(Angle.getValue())));
161  }
162  else if (mode == HelixMode::height_turns_growth) {
163  if (Turns.getValue() < Precision::Confusion())
164  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: turns too small!"));
165  if ((Height.getValue() < Precision::Confusion())
166  && (abs(Growth.getValue()) < Precision::Confusion())
167  && Turns.getValue() > 1.0)
168  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: either height or growth must not be zero!"));
169  Pitch.setValue(Height.getValue() / Turns.getValue());
170  if (Height.getValue() > 0) {
171  Angle.setValue(Base::toDegrees(atan(
172  Turns.getValue() * Growth.getValue() / Height.getValue())));
173  }
174  else {
175  // On purpose, we're doing nothing here; the else-branch is just for this comment.
176  // - we don't print a warning, as for a flat spiral a zero-height is perfectly fine
177  // - we don't void the angle (somehow) so that it keeps its value. This allows in
178  // interactive usage to just go back to another mode and everything keeps working
179  }
180  }
181  else {
182  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: unsupported mode"));
183  }
184 
185  TopoDS_Shape sketchshape; // Fixme: Should this be TopoShape here and below?
186  try {
187  sketchshape = getVerifiedFace();
188  }
189  catch (const Base::Exception& e) {
190  return new App::DocumentObjectExecReturn(e.what());
191  }
192 
193  if (sketchshape.IsNull())
194  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: No valid sketch or face"));
195  else {
196  //TODO: currently we only allow planar faces. the reason for this is that with other faces in front, we could
197  //not use the current simulate approach and build the start and end face from the wires. As the shell
198  //begins always at the spine and not the profile, the sketchshape cannot be used directly as front face.
199  //We would need a method to translate the front shape to match the shell starting position somehow...
200  TopoDS_Face face = TopoDS::Face(sketchshape);
201  BRepAdaptor_Surface adapt(face);
202  if (adapt.GetType() != GeomAbs_Plane)
203  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Face must be planar"));
204  }
205 
206  // if the Base property has a valid shape, fuse the AddShape into it
207  TopoShape base;
208  try {
209  base = getBaseTopoShape();
210  }
211  catch (const Base::Exception&) {
212  // fall back to support (for legacy features)
213  base = TopoShape();
214  }
215 
216  // update Axis from ReferenceAxis
217  try {
218  updateAxis();
219  }
220  catch (const Base::Exception& e) {
221  return new App::DocumentObjectExecReturn(e.what());
222  }
223 
224  try {
225  this->positionByPrevious();
226  TopLoc_Location invObjLoc = this->getLocation().Inverted();
227 
228  base.move(invObjLoc);
229 
230  TopoDS_Shape result;
231 
232  // generate the helix path
233  TopoDS_Shape path;
234  if (Angle.getValue()==0.){
235  // breaking the path at each turn prevents an OCC issue
236  path = generateHelixPath();
237  } else {
238  // don't break the path or the generated solid is invalid
239  path = generateHelixPath(1000.);
240  }
241 
242  TopoDS_Shape face = sketchshape;
243  face.Move(invObjLoc);
244 
245  Bnd_Box bounds;
246  BRepBndLib::Add(path, bounds);
247  double size=sqrt(bounds.SquareExtent());
248  ShapeFix_ShapeTolerance fix;
249  fix.LimitTolerance(path, Precision::Confusion() * 1e-6 * size ); // needed to produce valid Pipe for very big parts
250  // We introduce final part tolerance with the second call to LimitTolerance below, however
251  // OCCT has a bug where the side-walls of the Pipe disappear with very large (km range) pieces
252  // increasing a tiny bit of extra tolerance to the path fixes this. This will in any case
253  // be less than the tolerance lower limit below, but sufficient to avoid the bug
254 
255  BRepOffsetAPI_MakePipe mkPS(TopoDS::Wire(path), face, GeomFill_Trihedron::GeomFill_IsFrenet, Standard_False);
256  result = mkPS.Shape();
257 
258  BRepClass3d_SolidClassifier SC(result);
259  SC.PerformInfinitePoint(Precision::Confusion());
260  if (SC.State() == TopAbs_IN) {
261  result.Reverse();
262  }
263 
264  fix.LimitTolerance(result, Precision::Confusion() * size * Tolerance.getValue() ); // significant precision reduction due to helical approximation - needed to allow fusion to succeed
265 
266  AddSubShape.setValue(result);
267 
268  if (base.isNull()) {
269 
270  if (getAddSubType() == FeatureAddSub::Subtractive){
271  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: There is nothing to subtract"));
272  }
273 
274  if (!isSingleSolidRuleSatisfied(result)) {
275  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Result has multiple solids"));
276  }
277 
278  // store shape before refinement
279  this->rawShape = result;
280  Shape.setValue(getSolid(result));
281  return App::DocumentObject::StdReturn;
282  }
283 
284  if (getAddSubType() == FeatureAddSub::Additive) {
285 
286  FCBRepAlgoAPI_Fuse mkFuse(base.getShape(), result);
287  if (!mkFuse.IsDone()){
288  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Adding the helix failed"));
289  }
290  // we have to get the solids (fuse sometimes creates compounds)
291  TopoShape boolOp = this->getSolid(mkFuse.Shape());
292 
293  // lets check if the result is a solid
294  if (boolOp.isNull()){
295  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Result is not a solid"));
296  }
297 
298  if (!isSingleSolidRuleSatisfied(boolOp.getShape())) {
299  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Result has multiple solids"));
300  }
301 
302  // store shape before refinement
303  this->rawShape = boolOp;
304  boolOp = refineShapeIfActive(boolOp, RefineErrorPolicy::Warn);
305  Shape.setValue(getSolid(boolOp));
306  }
307  else if (getAddSubType() == FeatureAddSub::Subtractive) {
308 
309  TopoShape boolOp;
310 
311  if (Outside.getValue()) { // are we subtracting the inside or the outside of the profile.
312  FCBRepAlgoAPI_Common mkCom(result, base.getShape());
313  if (!mkCom.IsDone())
314  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Intersecting the helix failed"));
315  boolOp = this->getSolid(mkCom.Shape());
316 
317  }
318  else {
319  FCBRepAlgoAPI_Cut mkCut(base.getShape(), result);
320  if (!mkCut.IsDone())
321  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Subtracting the helix failed"));
322  boolOp = this->getSolid(mkCut.Shape());
323  }
324 
325  // lets check if the result is a solid
326  if (boolOp.isNull())
327  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Result is not a solid"));
328 
329  if (!isSingleSolidRuleSatisfied(boolOp.getShape())) {
330  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Result has multiple solids"));
331  }
332 
333  // store shape before refinement
334  this->rawShape = boolOp;
335  boolOp = refineShapeIfActive(boolOp, RefineErrorPolicy::Warn);
336  Shape.setValue(getSolid(boolOp));
337  }
338 
339  return App::DocumentObject::StdReturn;
340  }
341  catch (Standard_Failure& e) {
342 
343  if (std::string(e.GetMessageString()) == "TopoDS::Face")
344  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Error: Could not create face from sketch"));
345  else
346  return new App::DocumentObjectExecReturn(e.GetMessageString());
347  }
348  catch (Base::Exception& e) {
349  return new App::DocumentObjectExecReturn(e.what());
350  }
351 }

365 TopoDS_Shape Helix::generateHelixPath(double breakAtTurn)
366 {
367  double turns = Turns.getValue();
368  double height = Height.getValue();
369  bool leftHanded = LeftHanded.getValue();
370  bool reversed = Reversed.getValue();
371  double angle = Angle.getValue();
372  double growth = Growth.getValue();
373 
374  if (fabs(angle) < Precision::Confusion())
375  angle = 0.0;
376 
377  // get revolve axis
378  Base::Vector3d baseVector = Base.getValue();
379  gp_Pnt pnt(baseVector.x, baseVector.y, baseVector.z);
380  Base::Vector3d axisVector = Axis.getValue();
381  gp_Dir dir(axisVector.x, axisVector.y, axisVector.z);
382 
383  Base::Vector3d normal = getProfileNormal();
384  Base::Vector3d start = axisVector.Cross(normal); // pointing towards the desired helix start point.
385 
386  // if our axis is (nearly) aligned with the profile's normal, we're only interested in the "twist"
387  // of the helix. The actual starting point, and thus the radius, isn't important as long as it's
388  // somewhere in the profile's plane: an arbitrary vector perpendicular to the normal.
389  if (start.IsNull()) {
390  auto hopefullyNotParallel = Base::Vector3d(1.0, 2.0, 3.0);
391  start = normal.Cross(hopefullyNotParallel);
392  if (start.IsNull()) {
393  // bad luck
394  hopefullyNotParallel = Base::Vector3d(3.0, 2.0, 1.0);
395  start = normal.Cross(hopefullyNotParallel);
396  }
397  }
398 
399  gp_Dir dir_start(start.x, start.y, start.z);
400 
401  // Find out in what quadrant relative to the axis the profile is located, and the exact position.
402  Base::Vector3d profileCenter = getProfileCenterPoint();
403 
404  // The factor of 100 below ensures that profile size is small compared to the curvature of the helix.
405  // This improves the issue reported in https://forum.freecad.org/viewtopic.php?f=10&t=65048
406  double axisOffset = 100.0 * (profileCenter * start - baseVector * start);
407  double radius = std::fabs(axisOffset);
408  bool turned = axisOffset < 0;
409  // since the factor does not only change the radius but also the path position, we must shift its offset back
410  // using the square of the factor
411  double startOffset = 10000.0 * std::fabs((angle <= 0. ? 1. : 0.) * (profileCenter * axisVector) - baseVector * axisVector);
412 
413  if (radius < Precision::Confusion()) {
414  // in this case ensure that axis is not in the sketch plane
415  if (fabs(axisVector * normal) < Precision::Confusion())
416  throw Base::ValueError("Error: Result is self intersecting");
417  radius = 1000.0; //fallback to radius 1000
418  }
419 
420  bool growthMode = std::string(Mode.getValueAsString()).find("growth") != std::string::npos;
421  double radiusTop;
422  if (growthMode)
423  radiusTop = radius + turns * growth;
424  else
425  radiusTop = radius + height * tan(Base::toRadians(angle));
426 
427  //build the helix path
428  TopoDS_Shape path = TopoShape().makeSpiralHelix(radius, radiusTop, height, turns, breakAtTurn, leftHanded);
429 
430  /*
431  * The helix wire is created with the axis coinciding with z-axis and the start point at (radius, 0, 0)
432  * We want to move it so that the axis becomes aligned with "dir" and "pnt", we also want (radius,0,0) to
433  * map to the sketch plane.
434  */
435 
436  gp_Pnt origo(0.0, 0.0, 0.0);
437  gp_Dir dir_axis1(0.0, 0.0, 1.0); // pointing along the helix axis, as created.
438  gp_Dir dir_axis2(1.0, 0.0, 0.0); // pointing towards the helix start point, as created.
439  gp_Trsf mov;
440 
441  if (abs(startOffset) > 0) { // translate the helix so that the starting point aligns with the profile
442  mov.SetTranslation(startOffset * gp_Vec(dir_axis1));
443  TopLoc_Location loc(mov);
444  path.Move(loc);
445  }
446 
447  // because of the radius factor we used above, we must reverse after the
448  // startOffset movement (that brings the path back to the desired position)
449  if (reversed) {
450  mov.SetRotation(gp_Ax1(origo, dir_axis2), M_PI);
451  TopLoc_Location loc(mov);
452  path.Move(loc);
453  }
454 
455  if (turned) { // turn the helix so that the starting point aligns with the profile
456  mov.SetRotation(gp_Ax1(origo, dir_axis1), M_PI);
457  TopLoc_Location loc(mov);
458  path.Move(loc);
459  }
460 
461  gp_Ax3 sourceCS(origo, dir_axis1, dir_axis2);
462  gp_Ax3 targetCS(pnt, dir, dir_start);
463 
464  mov.SetTransformation(sourceCS, targetCS);
465  TopLoc_Location loc(mov);
466  path.Move(loc.Inverted());
467 
468  TopLoc_Location invObjLoc = this->getLocation().Inverted();
469  path.Move(invObjLoc);
470 
471  return path;
472 }

474 // this function calculates self intersection safe pitch based on the profile bounding box.
475 double Helix::safePitch()
476 {
477  Base::Vector3d axisVec = Axis.getValue();
478  Base::Vector3d startVec = axisVec.Cross(getProfileNormal()); // pointing towards the helix start point
479  HelixMode mode = static_cast<HelixMode>(Mode.getValue());
480  double growthValue = Growth.getValue();
481  double turnsValue = Turns.getValue();
482 
483  // handle case if axis is orthogonal to profile
484  // since startVec.IsNull() fails sometimes, for e.g. (0.0, 0.0, -0.0)
485  // we take the precision into account
486  if (startVec.Length() < Precision::Confusion()) {
487  // when not in growth mode any pitch > 0 is safe
488  if (mode != HelixMode::height_turns_growth) {
489  return Precision::Confusion();
490  }
491  // if growth is not zero, there will in many cases be intersections
492  // when the turn is >= 1, thus return an 'infinite' pitch
493  // Note: The resulting helix body is in this case often garbage since
494  // the OCC algorithm to create the helix fails.
495  // Nevertheless, the result is a valid body so it should be valuable for users
496  // to get this correct warning anyway.
497  else {
498  if (abs(turnsValue) >= 1.0 && abs(growthValue) > 0.0)
499  return Precision::Infinite();
500  }
501  }
502 
503  double angle = Angle.getValue() / 180.0 * M_PI;
504  gp_Dir direction(axisVec.x, axisVec.y, axisVec.z);
505  gp_Dir directionStart(startVec.x, startVec.y, startVec.z);
506  TopoDS_Shape sketchshape = getVerifiedFace();
507  Bnd_Box boundingBox;
508  BRepBndLib::Add(sketchshape, boundingBox);
509 
510  // get boundary and dimensions of boundingBox
511  double Xmin, Ymin, Zmin, Xmax, Ymax, Zmax;
512  boundingBox.Get(Xmin, Ymin, Zmin, Xmax, Ymax, Zmax);
513  double X = Xmax - Xmin, Y = Ymax - Ymin, Z = Zmax - Zmin;
514  gp_Vec boundingBoxVec(X, Y, Z);
515 
516  // Below is an approximation because since we take the bounding box it is
517  // impossible to calculate it precisely. For example a circle has as bounding
518  // box a square and thus results in a larger pitch than really necessary
519 
520  // minimal safe pitch if the angle or growth is 0
521  double pitch0 = boundingBoxVec * direction;
522 
523  if (mode == HelixMode::height_turns_growth) {
524  // if the distance perpendicular to axisVec
525  // between two turns is larger than the bounding box size in this direction
526  // the minimal necessary pitch is zero
527  if (abs(growthValue) > abs(boundingBoxVec * directionStart)) {
528  return 0.0;
529  }
530  else {
531  // if less than one turn, every pitch is safe
532  // Note: at the moment helices with a growth end with a plane
533  // whose normal is the final direction of the helix path.
534  // In case this might be changed in future, also 1.0 turn would be safe.
535  if (turnsValue < 1.0)
536  return 0.0;
537  else
538  return pitch0;
539  }
