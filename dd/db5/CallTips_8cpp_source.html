125 void CallTipsList::keyboardSearch(const QString& wordPrefix)
126 {
127  // first search for the item that matches perfectly
128  for (int i=0; i<count(); ++i) {
129  QString text = item(i)->text();
130  if (text.startsWith(wordPrefix)) {
131  setCurrentRow(i);
132  return;
133  }
134  }
135 
136  // now do a case insensitive comparison
137  for (int i=0; i<count(); ++i) {
138  QString text = item(i)->text();
139  if (text.startsWith(wordPrefix, Qt::CaseInsensitive)) {
140  setCurrentRow(i);
141  return;
142  }
143  }
144 
145  if (currentItem())
146  currentItem()->setSelected(false);
147 }

149 void CallTipsList::validateCursor()
150 {
151  QTextCursor cursor = textEdit->textCursor();
152  int currentPos = cursor.position();
153  if (currentPos < this->cursorPos) {
154  hide();
155  }
156  else {
157  cursor.setPosition(this->cursorPos);
158  cursor.movePosition(QTextCursor::EndOfWord, QTextCursor::KeepAnchor);
159  QString word = cursor.selectedText();
160  if (!word.isEmpty()) {
161  // the following text might be an operator, brackets, ...
162  const QChar underscore = QLatin1Char('_');
163  const QChar ch = word.at(0);
164  if (!ch.isLetterOrNumber() && ch != underscore)
165  word.clear();
166  }
167  if (currentPos > this->cursorPos+word.length()) {
168  hide();
169  }
170  else if (!word.isEmpty()){
171  // If the word is empty we should not allow one to do a search,
172  // otherwise we may select the next item which is not okay in this
173  // context. This might happen if e.g. Shift is pressed.
174  keyboardSearch(word);
175  }
176  }

179 QString CallTipsList::extractContext(const QString& line) const
180 {
181  int len = line.size();
182  int index = len-1;
183  for (int i=0; i<len; i++) {
184  int pos = len-1-i;
185  const char ch = line.at(pos).toLatin1();
186  if ((ch >= 48 && ch <= 57) || // Numbers
187  (ch >= 65 && ch <= 90) || // Uppercase letters
188  (ch >= 97 && ch <= 122) || // Lowercase letters
189  (ch == '.') || (ch == '_') || // dot or underscore
190  (ch == ' ') || (ch == '\t')) // whitespace (between dot and text)
191  index = pos;
192  else
193  break;
194  }
195 
196  return line.mid(index);
197 }

199 QMap<QString, CallTip> CallTipsList::extractTips(const QString& context) const
200 {
201  Base::PyGILStateLocker lock;
202  QMap<QString, CallTip> tips;
203  if (context.isEmpty())
204  return tips;
205 
206  try {
207  PyErr_Clear();
208  Py::Module module("__main__");
209  if (module.ptr() == nullptr) {
210  return tips;
211  }
212  Py::Dict dict = module.getDict();
213 
214  // this is used to filter out input of the form "1."
215  QStringList items = context.split(QLatin1Char('.'));
216  QString modname = items.front();
217  items.pop_front();
218  if (!dict.hasKey(std::string(modname.toLatin1())))
219  return tips; // unknown object
220  // Don't use hasattr & getattr because if a property is bound to a method this will be executed twice.
221  PyObject* code = Py_CompileString(static_cast<const char*>(context.toLatin1()), "<CallTipsList>", Py_eval_input);
222  if (!code) {
223  PyErr_Clear();
224  return tips;
225  }
226 
227  PyObject* eval = nullptr;
228  if (PyCode_Check(code)) {
229  eval = PyEval_EvalCode(code, dict.ptr(), dict.ptr());
230  }
231  Py_DECREF(code);
232  if (!eval) {
233  PyErr_Clear();
234  return tips;
235  }
236  Py::Object obj(eval, true);
237 
238  // Checks whether the type is a subclass of PyObjectBase because to get the doc string
239  // of a member we must get it by its type instead of its instance otherwise we get the
240  // wrong string, namely that of the type of the member.
241  // Note: 3rd party libraries may use their own type object classes so that we cannot
242  // reliably use Py::Type. To be on the safe side we should use Py::Object to assign
243  // the used type object to.
244  //Py::Object type = obj.type();
245  Py::Object type(PyObject_Type(obj.ptr()), true);
246  Py::Object inst = obj; // the object instance
247  PyObject* typeobj = Base::getTypeAsObject(&Base::PyObjectBase::Type);
248  PyObject* typedoc = Base::getTypeAsObject(&App::DocumentObjectPy::Type);
249  PyObject* basetype = Base::getTypeAsObject(&PyBaseObject_Type);
250 
251  if (PyObject_IsSubclass(type.ptr(), typedoc) == 1) {
252  // From the template Python object we don't query its type object because there we keep
253  // a list of additional methods that we won't see otherwise. But to get the correct doc
254  // strings we query the type's dict in the class itself.
255  // To see if we have a template Python object we check for the existence of '__fc_template__'
256  // See also: FeaturePythonPyT
257  if (!obj.hasAttr("__fc_template__")) {
258  obj = type;
259  }
260  }
261  else if (PyObject_IsSubclass(type.ptr(), typeobj) == 1) {
262  obj = type;
263  }
264  else if (PyObject_IsInstance(obj.ptr(), basetype) == 1) {
265  // New style class which can be a module, type, list, tuple, int, float, ...
266  // Make sure it's not a type object
267  PyObject* typetype = Base::getTypeAsObject(&PyType_Type);
268  if (PyObject_IsInstance(obj.ptr(), typetype) != 1) {
269  // For wrapped objects with PySide2 use the object, not its type
270  // as otherwise attributes added at runtime won't be listed (e.g. MainWindowPy)
271  QString typestr(QLatin1String(Py_TYPE(obj.ptr())->tp_name));
272 
273  // this should be now a user-defined Python class
274  // http://stackoverflow.com/questions/12233103/in-python-at-runtime-determine-if-an-object-is-a-class-old-and-new-type-instan
275  if (!typestr.startsWith(QLatin1String("PySide")) && Py_TYPE(obj.ptr())->tp_flags & Py_TPFLAGS_HEAPTYPE) {
276  obj = type;
277  }
278  }
279  }
280 
281  // If we have an instance of PyObjectBase then determine whether it's valid or not
282  if (PyObject_IsInstance(inst.ptr(), typeobj) == 1) {
283  auto baseobj = static_cast<Base::PyObjectBase*>(inst.ptr());
284  validObject = baseobj->isValid();
285  }
286  else {
287  // PyObject_IsInstance might set an exception
288  PyErr_Clear();
289  }
290 
291  Py::List list(obj.dir());
292 
293  // If we derive from PropertyContainerPy we can search for the properties in the
294  // C++ twin class.
295  PyObject* proptypeobj = Base::getTypeAsObject(&App::PropertyContainerPy::Type);
296  if (PyObject_IsSubclass(type.ptr(), proptypeobj) == 1) {
297  // These are the attributes of the instance itself which are NOT accessible by
298  // its type object
299  extractTipsFromProperties(inst, tips);
300  }
301 
302  // If we derive from App::DocumentPy we have direct access to the objects by their internal
303  // names. So, we add these names to the list, too.
304  PyObject* appdoctypeobj = Base::getTypeAsObject(&App::DocumentPy::Type);
305  if (PyObject_IsSubclass(type.ptr(), appdoctypeobj) == 1) {
306  auto docpy = static_cast<App::DocumentPy*>(inst.ptr());
307  auto document = docpy->getDocumentPtr();
308  // Make sure that the C++ object is alive
309  if (document) {
310  std::vector<App::DocumentObject*> objects = document->getObjects();
311  Py::List list;
312  for (const auto & object : objects)
313  list.append(Py::String(object->getNameInDocument()));
314  extractTipsFromObject(inst, list, tips);
315  }
316  }
317 
318  // If we derive from Gui::DocumentPy we have direct access to the objects by their internal
319  // names. So, we add these names to the list, too.
320  PyObject* guidoctypeobj = Base::getTypeAsObject(&Gui::DocumentPy::Type);
321  if (PyObject_IsSubclass(type.ptr(), guidoctypeobj) == 1) {
322  auto docpy = static_cast<Gui::DocumentPy*>(inst.ptr());
323  if (docpy->getDocumentPtr()) {
324  App::Document* document = docpy->getDocumentPtr()->getDocument();
325  // Make sure that the C++ object is alive
326  if (document) {
327  std::vector<App::DocumentObject*> objects = document->getObjects();
328  Py::List list;
329  for (const auto & object : objects)
330  list.append(Py::String(object->getNameInDocument()));
331  extractTipsFromObject(inst, list, tips);
332  }
333  }
334  }
335 
336  // These are the attributes from the type object
337  extractTipsFromObject(obj, list, tips);
338  }
339  catch (Py::Exception& e) {
340  // Just clear the Python exception
341  e.clear();
342  }
343 
344  return tips;
345 }

347 bool shibokenMayCrash(void)
348 {
349  // Shiboken 6.4.8 to 6.7.3 crash if we try to read their object
350  // attributes without a current stack frame.
351  // FreeCAD issue: https://github.com/FreeCAD/FreeCAD/issues/14101
352  // Qt issue: https://bugreports.qt.io/browse/PYSIDE-2796
353 
354  Py::Module shiboken("shiboken6");
355  Py::Tuple version(shiboken.getAttr("__version_info__"));
356  int major = Py::Long(version.getItem(0));
357  int minor = Py::Long(version.getItem(1));
358  int patch = Py::Long(version.getItem(2));
359  bool brokenVersion = (major == 6 && minor >= 4 && minor < 8);
360  bool fixedVersion = (major == 6 && minor == 7 && patch > 2);
361  return brokenVersion && !fixedVersion;
362 }

364 Py::Object CallTipsList::getAttrWorkaround(Py::Object& obj, Py::String& name) const
365 {
366  QString typestr(QLatin1String(Py_TYPE(obj.ptr())->tp_name));
367  bool hasWorkingGetAttr =
368  !(typestr == QLatin1String("Shiboken.ObjectType") && shibokenMayCrash());
369 
370  if (hasWorkingGetAttr) {
371  return obj.getAttr(name.as_string());
372  }
373 
374  Py::Dict globals;
375  Py::Dict locals;
376  locals.setItem("obj", obj);
377  locals.setItem("attr", name);
378 
379  Py::Object bouncer(Py_CompileString("getattr(obj, attr)", "<CallTipsList>", Py_eval_input));
380  Py::Object attr(PyEval_EvalCode(bouncer.ptr(), globals.ptr(), locals.ptr()));
381 
382  return attr;
383 }

385 void CallTipsList::extractTipsFromObject(Py::Object& obj, Py::List& list, QMap<QString, CallTip>& tips) const
386 {
387  for (Py::List::iterator it = list.begin(); it != list.end(); ++it) {
388  try {
389  Py::String attrname(*it);
390  std::string name(attrname.as_string());
391 
392  Py::Object attr = getAttrWorkaround(obj, attrname);
393  if (!attr.ptr()) {
394  Base::Console().Log("Python attribute '%s' returns null!\n", name.c_str());
395  continue;
396  }
397 
398  CallTip tip;
399  QString str = QString::fromLatin1(name.c_str());
400  tip.name = str;
401 
402  if (attr.isCallable()) {
403  PyObject* basetype = Base::getTypeAsObject(&PyBaseObject_Type);
404  if (PyObject_IsSubclass(attr.ptr(), basetype) == 1) {
405  tip.type = CallTip::Class;
406  }
407  else {
408  PyErr_Clear(); // PyObject_IsSubclass might set an exception
409  tip.type = CallTip::Method;
410  }
411  }
412  else if (PyModule_Check(attr.ptr())) {
413  tip.type = CallTip::Module;
414  }
415  else {
416  tip.type = CallTip::Member;
417  }
418 
419  if (str == QLatin1String("__doc__") && attr.isString()) {
420  Py::Object help = attr;
421  if (help.isString()) {
422  Py::String doc(help);
423  QString longdoc = QString::fromUtf8(doc.as_string().c_str());
424  int pos = longdoc.indexOf(QLatin1Char('\n'));
425  pos = qMin(pos, 70);
426  if (pos < 0)
427  pos = qMin(longdoc.length(), 70);
428  tip.description = stripWhiteSpace(longdoc);
429  tip.parameter = longdoc.left(pos);
430  }
431  }
432  else if (attr.hasAttr("__doc__")) {
433  Py::Object help = attr.getAttr("__doc__");
434  if (help.isString()) {
435  Py::String doc(help);
436  QString longdoc = QString::fromUtf8(doc.as_string().c_str());
437  int pos = longdoc.indexOf(QLatin1Char('\n'));
438  pos = qMin(pos, 70);
439  if (pos < 0)
440  pos = qMin(longdoc.length(), 70);
441  tip.description = stripWhiteSpace(longdoc);
442  tip.parameter = longdoc.left(pos);
443  }
444  }
445 
446  // Do not override existing items
447  QMap<QString, CallTip>::iterator pos = tips.find(str);
448  if (pos == tips.end())
449  tips[str] = tip;
450  }
451  catch (Py::Exception& e) {
452  // Just clear the Python exception
453  e.clear();
454  }
455  }

459 {

462  // Make sure that the C++ object is alive
463  if (!container)

466  container->getPropertyMap(Map);
467

497 void CallTipsList::showTips(const QString& line)
498 {
499  // search only once
500  static QPixmap type_module_icon = BitmapFactory().pixmap("ClassBrowser/type_module.svg");
501  static QPixmap type_class_icon = BitmapFactory().pixmap("ClassBrowser/type_class.svg");
502  static QPixmap method_icon = BitmapFactory().pixmap("ClassBrowser/method.svg");
503  static QPixmap member_icon = BitmapFactory().pixmap("ClassBrowser/member.svg");
504  static QPixmap property_icon = BitmapFactory().pixmap("ClassBrowser/property.svg");
505 
506  // object is in error state
507  static QPixmap forbidden_icon(Gui::BitmapFactory().pixmapFromSvg("forbidden", property_icon.size() / 4));
508  static QPixmap forbidden_type_module_icon = BitmapFactory().merge(type_module_icon,forbidden_icon,BitmapFactoryInst::BottomLeft);
509  static QPixmap forbidden_type_class_icon = BitmapFactory().merge(type_class_icon,forbidden_icon,BitmapFactoryInst::BottomLeft);
510  static QPixmap forbidden_method_icon = BitmapFactory().merge(method_icon,forbidden_icon,BitmapFactoryInst::BottomLeft);
511  static QPixmap forbidden_member_icon = BitmapFactory().merge(member_icon,forbidden_icon,BitmapFactoryInst::BottomLeft);
512  static QPixmap forbidden_property_icon = BitmapFactory().merge(property_icon,forbidden_icon,BitmapFactoryInst::BottomLeft);
513 
514  this->validObject = true;
515  QString context = extractContext(line);
516  context = context.simplified();
517  QMap<QString, CallTip> tips = extractTips(context);
518  clear();
519  for (QMap<QString, CallTip>::Iterator it = tips.begin(); it != tips.end(); ++it) {
520  addItem(it.key());
521  QListWidgetItem *item = this->item(this->count()-1);
522  item->setData(Qt::ToolTipRole, QVariant(it.value().description));
523  item->setData(Qt::UserRole, QVariant::fromValue( it.value() )); //< store full CallTip data
524  switch (it.value().type)
525  {
526  case CallTip::Module:
527  {
528  item->setIcon((this->validObject ? type_module_icon : forbidden_type_module_icon));
529  } break;
530  case CallTip::Class:
531  {
532  item->setIcon((this->validObject ? type_class_icon : forbidden_type_class_icon));
533  } break;
534  case CallTip::Method:
535  {
536  item->setIcon((this->validObject ? method_icon : forbidden_method_icon));
537  } break;
538  case CallTip::Member:
539  {
540  item->setIcon((this->validObject ? member_icon : forbidden_member_icon));
541  } break;
542  case CallTip::Property:
543  {
544  item->setIcon((this->validObject ? property_icon : forbidden_property_icon));
545  } break;
546  default:
547  break;
548  }
549  }
550 
551  if (count()==0)
552  return; // nothing found
553 
554  // get the minimum width and height of the box
555  int h = 0;
556  int w = 0;
557  for (int i = 0; i < count(); ++i) {
558  QRect r = visualItemRect(item(i));
559  w = qMax(w, r.width());
560  h += r.height();
561  }
562 
563  // Add an offset
564  w += 2*frameWidth();
565  h += 2*frameWidth();
566 
567  // get the start position of the word prefix
568  QTextCursor cursor = textEdit->textCursor();
569  this->cursorPos = cursor.position();
570  QRect rect = textEdit->cursorRect(cursor);
571  int posX = rect.x();
572  int posY = rect.y();
573  int boxH = h;
574 
575  // Decide whether to show downstairs or upstairs
576  if (posY > textEdit->viewport()->height()/2) {
577  h = qMin(qMin(h,posY), 250);
578  if (h < boxH)
579  w += textEdit->style()->pixelMetric(QStyle::PM_ScrollBarExtent);
580  setGeometry(posX,posY-h, w, h);
581  }
582  else {
583  h = qMin(qMin(h,textEdit->viewport()->height()-fontMetrics().height()-posY), 250);
584  if (h < boxH)
585  w += textEdit->style()->pixelMetric(QStyle::PM_ScrollBarExtent);
586  setGeometry(posX, posY+fontMetrics().height(), w, h);
587  }
588 
589  setCurrentRow(0);
590  show();
591 }

610 bool CallTipsList::eventFilter(QObject * watched, QEvent * event)
611 {
612  // This is a trick to avoid to hide the tooltip window after the defined time span
613  // of 10 seconds. We just filter out all timer events to keep the label visible.
614  if (watched->inherits("QLabel")) {
615  auto label = qobject_cast<QLabel*>(watched);
616  // Ignore the timer events to prevent from being closed
617  if (label->windowFlags() & Qt::ToolTip && event->type() == QEvent::Timer)
618  return true;
619  }
620  if (isVisible() && watched == textEdit->viewport()) {
621  if (event->type() == QEvent::MouseButtonPress)
622  hide();
623  }
624  else if (isVisible() && watched == textEdit) {
625  if (event->type() == QEvent::KeyPress) {
626  auto ke = static_cast<QKeyEvent*>(event);

665  return true;
666  }
667  }
668  else if (event->type() == QEvent::KeyRelease) {
669  auto ke = static_cast<QKeyEvent*>(event);
670  if (ke->key() == Qt::Key_Up || ke->key() == Qt::Key_Down ||
671  ke->key() == Qt::Key_PageUp || ke->key() == Qt::Key_PageDown) {

686  if (!hasFocus())

688  }

696  hide();

708  const QChar ch = sel.at(sel.size()-1);
709  if (!ch.isLetterOrNumber() && ch != underscore)
710  cursor.movePosition(QTextCursor::Left, QTextCursor::KeepAnchor);

721  cursor.insertText( QLatin1String("()") ); //< just append parenthesis to identifier even inserted.

732  cursor.movePosition( QTextCursor::Left, QTextCursor::MoveAnchor, 1 );

747 QString CallTipsList::stripWhiteSpace(const QString& str) const
748 {
749  QString stripped = str;
750  QStringList lines = str.split(QLatin1String("\n"));
751  int minspace=INT_MAX;
752  int line=0;
753  for (QStringList::iterator it = lines.begin(); it != lines.end(); ++it, ++line) {
754  if (it->size() > 0 && line > 0) {
755  int space = 0;
756  for (int i=0; i<it->size(); i++) {
757  if ((*it)[i] == QLatin1Char('\t'))
758  space++;
759  else
760  break;
761  }
762 
763  if (it->size() > space)
764  minspace = std::min<int>(minspace, space);
765  }
766  }
767 
768  // remove all leading tabs from each line
769  if (minspace > 0 && minspace < INT_MAX) {
770  int line=0;
771  QStringList strippedlines;
772  for (QStringList::iterator it = lines.begin(); it != lines.end(); ++it, ++line) {
773  if (line == 0 && !it->isEmpty()) {
774  strippedlines << *it;
775  }
776  else if (it->size() > 0 && line > 0) {
777  strippedlines << it->mid(minspace);
778  }
779  }
780 
781  stripped = strippedlines.join(QLatin1String("\n"));
782  }
783 
784  return stripped;
785 }
