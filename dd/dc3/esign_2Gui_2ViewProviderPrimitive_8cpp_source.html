58 bool ViewProviderPrimitive::setEdit(int ModNum)
59 {
60  if (ModNum == ViewProvider::Default ) {
61  // When double-clicking on the item for this fillet the
62  // object unsets and sets its edit mode without closing
63  // the task panel
64  Gui::TaskView::TaskDialog *dlg = Gui::Control().activeDialog();
65  TaskPrimitiveParameters *primitiveDlg = qobject_cast<TaskPrimitiveParameters *>(dlg);
66  if (dlg && !primitiveDlg) {
67  QMessageBox msgBox;
68  msgBox.setText(QObject::tr("A dialog is already open in the task panel"));
69  msgBox.setInformativeText(QObject::tr("Do you want to close this dialog?"));
70  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
71  msgBox.setDefaultButton(QMessageBox::Yes);
72  int ret = msgBox.exec();
73  if (ret == QMessageBox::Yes)
74  Gui::Control().closeDialog();
75  else
76  return false;
77  }
78 
79  // clear the selection (convenience)
80  Gui::Selection().clearSelection();
81 
82  // always change to PartDesign WB, remember where we come from
83  oldWb = Gui::Command::assureWorkbench("PartDesignWorkbench");
84 
85  // start the edit dialog
86  // another pad left open its task panel
87  if (primitiveDlg)
88  Gui::Control().showDialog(primitiveDlg);
89  else
90  Gui::Control().showDialog(new TaskPrimitiveParameters(this));
91 
92  setPreviewDisplayMode(true);
93 
94  return true;
95  }
96  else {
97  return ViewProviderAddSub::setEdit(ModNum);
98  }
99 }
