44 {

65 // DE BOOR ALGORITHM FROM OPENGLIDER
66 std::function<double(double)> get_basis(int degree, int i, Eigen::VectorXd knots)
67  // Return a basis_function for the given degree """
68 {
69  if (degree == 0)
70  {
71  return [degree, i, knots](double t)
72  {
73  // The basis function for degree = 0 as per eq. 7
74  (void)degree;
75  double t_this = knots[i];
76  double t_next = knots[i+1];
77  if (t == knots[0])
78  return (double)(t_next >= t && t >= t_this);
79  return (double)(t_next >= t && t > t_this);
80  };
81  }
82  else
83  {
84  return [degree, i, knots](double t)
85  {
86  // The basis function for degree > 0 as per eq. 8
87  if (t < knots[0])
88  return get_basis(degree, i, knots)(knots[0]);
89  else if (t > knots[knots.rows() - 1])
90  return get_basis(degree, i, knots)(knots[knots.rows() - 1]);
91  double out = 0.;
92  double t_this = knots[i];
93  double t_next = knots[i + 1];
94  double t_precog = knots[i + degree];
95  double t_horizon = knots[i + degree + 1];
96  if (t_this == t_horizon)
97  return 0.;
98 
99  double bottom = (t_precog - t_this);
100  out = divide(t - t_this, bottom) * get_basis(degree - 1, i, knots)(t);
101 
102  bottom = (t_horizon - t_next);
103  out += divide(t_horizon - t, bottom) * get_basis(degree - 1, i + 1, knots)(t);
104  return out;
105  };
106  }
107 }

110 std::function<double(double)> get_basis_derivative(int order, int degree, int i, Eigen::VectorXd knots)
111  // Return the derivation of the basis function
112  // order of basis function
113  // degree of basis function
114  //
115  // knots sequence
116 {
117  if (order == 1)
118  {
119  return [degree, i, knots, order](double t)
120  {
121  (void)order;
122  double out = 0;
123  if (!(knots[i + degree] - knots[i] == 0))
124  {
125  out += get_basis(degree - 1, i, knots)(t) *
126  degree / (knots[i + degree] - knots[i]);
127  }
128  if (!(knots[i + degree + 1] - knots[i + 1] == 0))
129  {
130  out -= get_basis(degree - 1, i + 1, knots)(t) *
131  degree / (knots[i + degree + 1] - knots[i + 1]);
132  }
133  return out;
134  };
135  }
136  else
137  {
138  return [degree, i, knots, order](double t)
139  {
140  double out = 0;
141  if (!(knots[i + degree] - knots[i] == 0))
142  {
143  out += get_basis_derivative(order - 1, degree - 1, i, knots)(t) *
144  degree / (knots[i + degree] - knots[i]);
145  }
146  if (!(knots[i + degree + 1] - knots[i + 1] == 0))
147  {
148  out -= get_basis_derivative(order - 1, degree - 1, i + 1, knots)(t) *
149  degree / (knots[i + degree + 1] - knots[i + 1]);
150  }
151  return out;
152  };
153  }

172 Eigen::VectorXd NurbsBase2D::getInfluenceVector(Eigen::Vector2d u)
173 {
174  Eigen::VectorXd n_u, n_v;
175  double sum_weights = 0;
176  Eigen::VectorXd infl(this->u_functions.size() * this->v_functions.size());
177  int i = 0;
178 
179  n_u.resize(this->u_functions.size());
180  n_v.resize(this->v_functions.size());
181 
182  for (unsigned int i = 0; i < this->u_functions.size(); i ++)
183  n_u[i] = this->u_functions[i](u.x());
184  for (unsigned int i = 0; i < this->v_functions.size(); i ++)
185  n_v[i] = this->v_functions[i](u.y());
186 
187  for (unsigned int u_i = 0; u_i < this->u_functions.size(); u_i++)
188  {
189  for (unsigned int v_i = 0; v_i < this->v_functions.size(); v_i++)
190  {
191  sum_weights += weights[i] * n_u[u_i] * n_v[v_i];
192  infl[i] = weights[i] * n_u[u_i] * n_v[v_i];
193  i ++;
194  }
195  }
196  return infl / sum_weights;
197 }

235 Eigen::VectorXd NurbsBase2D::getDuVector(Eigen::Vector2d u)
236 {
237  Eigen::VectorXd A1, A2;
238  double C1, C2;
239  A1.resize(this->u_functions.size() * this->v_functions.size());
240  A2.resize(this->u_functions.size() * this->v_functions.size());
241  double A3 = 0;
242  double A5 = 0;
243  int i = 0;
244  Eigen::VectorXd n_u, n_v, Dn_u;
245  n_u.resize(this->u_functions.size());
246  Dn_u.resize(this->u_functions.size());
247  n_v.resize(this->v_functions.size());
248  for (unsigned int u_i=0; u_i < this->u_functions.size(); u_i++)
249  {
250  n_u[u_i] = this->u_functions[u_i](u.x());
251  Dn_u[u_i] = this->Du_functions[u_i](u.x());
252  }
253  for (unsigned int v_i=0; v_i < this->v_functions.size(); v_i++)
254  {
255  n_v[v_i] = this->v_functions[v_i](u.y());
256  }
257 
258  for (unsigned int u_i=0; u_i < this->u_functions.size(); u_i++)
259  {
260  for (unsigned int v_i=0; v_i < this->v_functions.size(); v_i++)
261  {
262  C1 = weights[i] * Dn_u[u_i] * n_v[v_i];
263  C2 = weights[i] * n_u[u_i] * n_v[v_i];
264  A1[i] = C1;
265  A2[i] = C2;
266  A3 += C2;
267  A5 += C1;
268  i ++;
269  }
270  }
271  return (A1 * A3 - A2 * A5) / A3 / A3;
272 }

335 std::tuple<NurbsBase2D, Eigen::MatrixXd> NurbsBase2D::interpolateUBS(
336  Eigen::Matrix<double, Eigen::Dynamic, 3> poles,
337  int degree_u,
338  int degree_v,
339  int num_u_poles,
340  int num_v_poles,
341  int num_u_points,
342  int num_v_points)
343 {
344  double u_min = this->u_knots(0);
345  double u_max = this->u_knots(this->u_knots.size() - 1);
346  double v_min = this->v_knots(0);
347  double v_max = this->v_knots(this->v_knots.size() - 1);
348  Eigen::VectorXd weights, u_knots, v_knots;
349  u_knots = NurbsBase1D::getKnotSequence(u_min, u_max, degree_u, num_u_poles);
350  v_knots = NurbsBase1D::getKnotSequence(v_min, v_max, degree_v, num_v_poles);
351 
352  weights.resize((u_knots.rows() - degree_u - 1) * (v_knots.rows() - degree_v - 1));
353  weights.setOnes();
354  NurbsBase2D new_base(u_knots, v_knots, weights, degree_u, degree_v);
355  Eigen::Matrix<double, Eigen::Dynamic, 2> uv_points = this->getUVMesh(num_u_points, num_v_points);
356  Eigen::Matrix<double, Eigen::Dynamic, 3> xyz_points = this->getInfluenceMatrix(uv_points) * poles;
357  spMat A = new_base.getInfluenceMatrix(uv_points);
358  Eigen::LeastSquaresConjugateGradient<spMat > solver;
359  solver.compute(A);
360  Eigen::Matrix<double, Eigen::Dynamic, 3> new_poles = solver.solve(xyz_points);
361  return std::tuple<NurbsBase2D, Eigen::MatrixXd >(new_base, new_poles);
362 }

364 Eigen::Matrix<double, Eigen::Dynamic, 2> NurbsBase2D::getUVMesh(int num_u_points, int num_v_points)
365 {
366  double u_min = this->u_knots(0);
367  double u_max = this->u_knots(this->u_knots.size() - 1);
368  double v_min = this->v_knots(0);
369  double v_max = this->v_knots(this->v_knots.size() - 1);
370  Eigen::Matrix<double, Eigen::Dynamic, 2> uv_points;
371  uv_points.resize(num_u_points * num_v_points, 2);
372  int i = 0;
373  for (int u = 0; u < num_u_points; u++)
374  {
375  for (int v = 0; v < num_v_points; v++)
376  {
377  uv_points(i, 0) = u_min + (u_max - u_min) * u / (num_u_points - 1);
378  uv_points(i, 1) = v_min + (v_max - v_min) * v / (num_v_points - 1);
379  i++;
380  }
381 
382  }
383  return uv_points;
384 }
