142  # generic writer for constraints mesh sets and constraints property data
143  # write constraint node sets, constraint face sets, constraint element sets
144  def write_constraints_meshsets(self, f, femobjs, con_module):
145  if not femobjs:
146  return
147 
148  analysis_types = con_module.get_analysis_types()
149  if analysis_types != "all" and self.analysis_type not in analysis_types:
150  return
151 
152  def constraint_sets_loop_writing(the_file, femobjs, write_before, write_after):
153  if write_before != "":
154  the_file.write(write_before)
155  for femobj in femobjs:
156  # femobj --> dict, FreeCAD document object is femobj["Object"]
157  the_obj = femobj["Object"]
158  the_file.write(f"** {the_obj.Label}\n")
159  con_module.write_meshdata_constraint(the_file, femobj, the_obj, self)
160  if write_after != "":
161  the_file.write(write_after)
162 
163  write_before = con_module.get_before_write_meshdata_constraint()
164  write_after = con_module.get_after_write_meshdata_constraint()
165 
166  # write sets to file
167  write_name = con_module.get_sets_name()
168  f.write("\n{}\n".format(59 * "*"))
169  f.write("** {}\n".format(write_name.replace("_", " ")))
170 
171  if self.split_inpfile is True:
172  file_name_split = f"{self.mesh_name}_{write_name}.inp"
173  f.write(f"*INCLUDE,INPUT={file_name_split}\n")
174  inpfile_split = open(join(self.dir_name, file_name_split), "w")
175  constraint_sets_loop_writing(inpfile_split, femobjs, write_before, write_after)
176  inpfile_split.close()
177  else:
178  constraint_sets_loop_writing(f, femobjs, write_before, write_after)
