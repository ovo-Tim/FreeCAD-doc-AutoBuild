31  // A quadric surface is defined implicitly by
32  //
33  // 0 = a0 + a1*x[0] + a2*x[1] + a3*x[2] + a4*x[0]^2 + a5*x[0]*x[1] +
34  // a6*x[0]*x[2] + a7*x[1]^2 + a8*x[1]*x[2] + a9*x[2]^2
35  //
36  // = a0 + [a1 a2 a3]*X + X^T*[a4 a5/2 a6/2]*X
37  // [a5/2 a7 a8/2]
38  // [a6/2 a8/2 a9 ]
39  // = C + B^T*X + X^T*A*X
40  //
41  // The matrix A is symmetric.
