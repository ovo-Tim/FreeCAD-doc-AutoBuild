104 void TaskBooleanParameters::onSelectionChanged(const Gui::SelectionChanges& msg)
105 {
106  if (selectionMode == none) {
107  return;
108  }
109 
110  if (msg.Type == Gui::SelectionChanges::AddSelection) {
111  if (strcmp(msg.pDocName, BooleanView->getObject()->getDocument()->getName()) != 0) {
112  return;
113  }
114 
115  // get the selected object
116  PartDesign::Boolean* pcBoolean =
117  BooleanView->getObject<PartDesign::Boolean>();
118  std::string body(msg.pObjectName);
119  if (body.empty()) {
120  return;
121  }
122  App::DocumentObject* pcBody = pcBoolean->getDocument()->getObject(body.c_str());
123  if (!pcBody) {
124  return;
125  }
126 
127  // if the selected object is not a body then get the body it is part of
128  if (!pcBody->isDerivedFrom<PartDesign::Body>()) {
129  pcBody = PartDesign::Body::findBodyOf(pcBody);
130  if (!pcBody) {
131  return;
132  }
133  body = pcBody->getNameInDocument();
134  }
135 
136  std::vector<App::DocumentObject*> bodies = pcBoolean->Group.getValues();
137 
138  if (selectionMode == bodyAdd) {
139  if (std::find(bodies.begin(), bodies.end(), pcBody) == bodies.end()) {
140  bodies.push_back(pcBody);
141  pcBoolean->Group.setValues(std::vector<App::DocumentObject*>());
142  pcBoolean->addObjects(bodies);
143 
144  QListWidgetItem* item = new QListWidgetItem(ui->listWidgetBodies);
145  item->setText(QString::fromUtf8(pcBody->Label.getValue()));
146  item->setData(Qt::UserRole, QString::fromLatin1(pcBody->getNameInDocument()));
147 
148  pcBoolean->getDocument()->recomputeFeature(pcBoolean);
149  ui->buttonBodyAdd->setChecked(false);
150  exitSelectionMode();
151 
152  // Hide the bodies
153  if (bodies.size() == 1) {
154  // Hide base body and added body
155  Gui::ViewProviderDocumentObject* vp =
156  dynamic_cast<Gui::ViewProviderDocumentObject*>(
157  Gui::Application::Instance->getViewProvider(
158  pcBoolean->BaseFeature.getValue()));
159  if (vp) {
160  vp->hide();
161  }
162  vp = dynamic_cast<Gui::ViewProviderDocumentObject*>(
163  Gui::Application::Instance->getViewProvider(bodies.front()));
164  if (vp) {
165  vp->hide();
166  }
167  BooleanView->show();
168  }
169  else {
170  // Hide newly added body
171  Gui::ViewProviderDocumentObject* vp =
172  dynamic_cast<Gui::ViewProviderDocumentObject*>(
173  Gui::Application::Instance->getViewProvider(bodies.back()));
174  if (vp) {
175  vp->hide();
176  }
177  }
178  }
179  }
180  else if (selectionMode == bodyRemove) {
181  std::vector<App::DocumentObject*>::iterator b =
182  std::find(bodies.begin(), bodies.end(), pcBody);
183  if (b != bodies.end()) {
184  bodies.erase(b);
185  pcBoolean->setObjects(bodies);
186 
187  QString internalName = QString::fromStdString(body);
188  for (int row = 0; row < ui->listWidgetBodies->count(); row++) {
189  QListWidgetItem* item = ui->listWidgetBodies->item(row);
190  QString name = item->data(Qt::UserRole).toString();
191  if (name == internalName) {
192  ui->listWidgetBodies->takeItem(row);
193  delete item;
194  break;
195  }
196  }
197 
198  pcBoolean->getDocument()->recomputeFeature(pcBoolean);
199  ui->buttonBodyRemove->setChecked(false);
200  exitSelectionMode();
201 
202  // Make bodies visible again
203  Gui::ViewProviderDocumentObject* vp =
204  dynamic_cast<Gui::ViewProviderDocumentObject*>(
205  Gui::Application::Instance->getViewProvider(pcBody));
206  if (vp) {
207  vp->show();
208  }
209  if (bodies.empty()) {
210  Gui::ViewProviderDocumentObject* vp =
211  dynamic_cast<Gui::ViewProviderDocumentObject*>(
212  Gui::Application::Instance->getViewProvider(
213  pcBoolean->BaseFeature.getValue()));
214  if (vp) {
215  vp->show();
216  }
217  BooleanView->hide();
218  }
219  }
220  }
221  }
222 }

293 void TaskBooleanParameters::onBodyDeleted()
294 {
295  PartDesign::Boolean* pcBoolean = BooleanView->getObject<PartDesign::Boolean>();
296  std::vector<App::DocumentObject*> bodies = pcBoolean->Group.getValues();
297  int index = ui->listWidgetBodies->currentRow();
298  if (index < 0 && (size_t)index > bodies.size()) {
299  return;
300  }
301 
302  App::DocumentObject* body = bodies[index];
303  QString internalName = ui->listWidgetBodies->item(index)->data(Qt::UserRole).toString();
304  for (auto it = bodies.begin(); it != bodies.end(); ++it) {
305  if (internalName == QLatin1String((*it)->getNameInDocument())) {
306  body = *it;
307  bodies.erase(it);
308  break;
309  }
310  }
311 
312  ui->listWidgetBodies->model()->removeRow(index);
313  pcBoolean->setObjects(bodies);
314  pcBoolean->getDocument()->recomputeFeature(pcBoolean);
315 
316  // Make bodies visible again
317  Gui::ViewProviderDocumentObject* vp = dynamic_cast<Gui::ViewProviderDocumentObject*>(
318  Gui::Application::Instance->getViewProvider(body));
319  if (vp) {
320  vp->show();
321  }
322  if (bodies.empty()) {
323  Gui::ViewProviderDocumentObject* vp = dynamic_cast<Gui::ViewProviderDocumentObject*>(
324  Gui::Application::Instance->getViewProvider(pcBoolean->BaseFeature.getValue()));
325  if (vp) {
326  vp->show();
327  }
328  BooleanView->hide();
329  }
330 }

380 bool TaskDlgBooleanParameters::accept()
381 {
382  auto obj = BooleanView->getObject();
383  if (!obj || !obj->isAttachedToDocument()) {
384  return false;
385  }
386  BooleanView->Visibility.setValue(true);
387 
388  try {
389  std::vector<std::string> bodies = parameter->getBodies();
390  if (bodies.empty()) {
391  QMessageBox::warning(parameter,
392  tr("Empty body list"),
393  tr("The body list cannot be empty"));
394  return false;
395  }
396  std::stringstream str;
397  str << Gui::Command::getObjectCmd(obj) << ".setObjects( [";
398  for (const auto& body : bodies) {
399  str << "App.getDocument('" << obj->getDocument()->getName() << "').getObject('" << body
400  << "'),";
401  }
402  str << "])";
403  Gui::Command::runCommand(Gui::Command::Doc, str.str().c_str());
404  }
405  catch (const Base::Exception& e) {
406  QMessageBox::warning(parameter,
407  tr("Boolean: Accept: Input error"),
408  QCoreApplication::translate("Exception", e.what()));
409  return false;
410  }
411 
412  FCMD_OBJ_CMD(obj, "Type = " << parameter->getType());
413  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
414  Gui::Command::doCommand(Gui::Command::Gui, "Gui.activeDocument().resetEdit()");
415  Gui::Command::commitCommand();
416 
417  return true;
418 }

420 bool TaskDlgBooleanParameters::reject()
421 {
422  // Show the bodies again
423  PartDesign::Boolean* obj = BooleanView->getObject<PartDesign::Boolean>();
424  Gui::Document* doc = Gui::Application::Instance->activeDocument();
425  if (doc) {
426  if (obj->BaseFeature.getValue()) {
427  doc->setShow(obj->BaseFeature.getValue()->getNameInDocument());
428  std::vector<App::DocumentObject*> bodies = obj->Group.getValues();
429  for (auto body : bodies) {
430  doc->setShow(body->getNameInDocument());
431  }
432  }
433  }
434 
435  // roll back the done things
436  Gui::Command::abortCommand();
437  Gui::Command::doCommand(Gui::Command::Gui, "Gui.activeDocument().resetEdit()");
438 
439 
440  return true;
