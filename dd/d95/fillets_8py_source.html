382 def filletWire(aWire, r, chamfer=False):
383  """Fillet each angle of a wire with r as radius.
384 
385  If chamfer is true, a `chamfer` is made instead, and `r` is the
386  size of the chamfer.
387  """
388  edges = aWire.Edges
389  edges = Part.__sortEdges__(edges)
390  filEdges = [edges[0]]
391 
392  for i in range(len(edges) - 1):
393  result = fillet([filEdges[-1], edges[i+1]], r, chamfer)
394  if len(result) > 2:
395  filEdges[-1:] = result[0:3]
396  else:
397  filEdges[-1:] = result[0:2]
398 
399  if isReallyClosed(aWire):
400  result = fillet([filEdges[-1], filEdges[0]], r, chamfer)
401  if len(result) > 2:
402  filEdges[-1:] = result[0:2]
403  filEdges[0] = result[2]
404 
405  return Part.Wire(filEdges)
