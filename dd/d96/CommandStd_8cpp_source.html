83 void StdCmdWorkbench::activated(int i)
84 {
85  try {
86  Workbench* w = WorkbenchManager::instance()->active();
87  QList<QAction*> items = static_cast<WorkbenchGroup*>(_pcAction)->actions();
88  std::string switch_to = (const char*)items[i]->objectName().toLatin1();
89  if (w) {
90  std::string current_w = w->name();
91  if (switch_to == current_w)
92  return;
93  }
94  doCommand(Gui, "Gui.activateWorkbench(\"%s\")", switch_to.c_str());
95  }
96  catch(const Base::PyException& e) {
97  QString msg(QLatin1String(e.what()));
98  // ignore '<type 'exceptions.*Error'>' prefixes
99  QRegularExpression rx;
100  rx.setPattern(QLatin1String(R"(^\s*<type 'exceptions.\w*'>:\s*)"));
101  auto match = rx.match(msg);
102  if (match.hasMatch())
103  msg = msg.mid(match.capturedLength());
104  QMessageBox::critical(getMainWindow(), QObject::tr("Cannot load workbench"), msg);
105  }
106  catch(...) {
107  QMessageBox::critical(getMainWindow(), QObject::tr("Cannot load workbench"),
108  QObject::tr("A general error occurred while loading the workbench"));
109  }
110 }

342 void StdCmdRestartInSafeMode::activated(int iMsg)
343 {
344  Q_UNUSED(iMsg);
345 
346  QMessageBox restartBox;
347  restartBox.setIcon(QMessageBox::Warning);
348  restartBox.setWindowTitle(QObject::tr("Restart in safe mode"));
349  restartBox.setText(QObject::tr("Are you sure you want to restart FreeCAD and enter safe mode?"));
350  restartBox.setInformativeText(QObject::tr("Safe mode temporarily disables your configuration and addons."));
351  restartBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
352  restartBox.setDefaultButton(QMessageBox::No);
353 
354  if (restartBox.exec() == QMessageBox::Yes) {
355  //restart FreeCAD after a delay to give time to this dialog to close
356  const int ms = 1000;
357  QTimer::singleShot(ms, []()
358  {
359  QStringList args = QApplication::arguments();
360  args.pop_front();
361  auto const safeModeArgument = QString::fromLatin1("--safe-mode");
362  if (!args.contains(safeModeArgument)) {
363  args.append(safeModeArgument);
364  }
365  if (getMainWindow()->close()) {
366  QProcess::startDetached(QApplication::applicationFilePath(), args);
367  }
368  });
369  }
370 }
