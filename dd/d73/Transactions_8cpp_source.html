64 Transaction::~Transaction()
65 {
66  auto& index = _Objects.get<0>();
67  for (const auto& It : index) {
68  if (It.second->status == TransactionObject::New) {
69  // If an object has been removed from the document the transaction
70  // status is 'New'. The 'pcNameInDocument' member serves as criterion
71  // to check whether the object is part of the document or not.
72  // Note, it's possible that the transaction status is 'New' while the
73  // object is (again) part of the document. This usually happens when
74  // a previous removal is undone.
75  // Thus, if the object has been removed, i.e. the status is 'New' and
76  // is still not part of the document the object must be destroyed not
77  // to cause a memory leak. This usually is the case when the removal
78  // of an object is not undone or when an addition is undone.
79 
80  if (!It.first->isAttachedToDocument()) {
81  if (It.first->isDerivedFrom<DocumentObject>()) {
82  // #0003323: Crash when clearing transaction list
83  // It can happen that when clearing the transaction list several objects
84  // are destroyed with dependencies which can lead to dangling pointers.
85  // When setting the 'Destroy' flag of an object the destructors of link
86  // properties don't ry to remove backlinks, i.e. they don't try to access
87  // possible dangling pointers.
88  // An alternative solution is to call breakDependency inside
89  // Document::_removeObject. Make this change in v0.18.
90  const DocumentObject* obj = static_cast<const DocumentObject*>(It.first);
91  const_cast<DocumentObject*>(obj)->setStatus(ObjectStatus::Destroy, true);
92  }
93  delete It.first;
94  }
95  }
96  delete It.second;
97  }
98 }

170 void Transaction::apply(Document& Doc, bool forward)
171 {
172  std::string errMsg;
173  try {
174  auto& index = _Objects.get<0>();
175  for (auto& info : index) {
176  info.second->applyDel(Doc, const_cast<TransactionalObject*>(info.first));
177  }
178  for (auto& info : index) {
179  info.second->applyNew(Doc, const_cast<TransactionalObject*>(info.first));
180  }
181  for (auto& info : index) {
182  info.second->applyChn(Doc, const_cast<TransactionalObject*>(info.first), forward);
183  }
184  }
185  catch (Base::Exception& e) {
186  e.ReportException();
187  errMsg = e.what();
188  }
189  catch (std::exception& e) {
190  errMsg = e.what();
191  }
192  catch (...) {
193  errMsg = "Unknown exception";
194  }
195  if (!errMsg.empty()) {
196  FC_ERR("Exception on " << (forward ? "redo" : "undo") << " '" << Name << "':" << errMsg);
197  }

200 void Transaction::addObjectNew(TransactionalObject* Obj)
201 {
202  auto& index = _Objects.get<1>();
203  auto pos = index.find(Obj);
204  if (pos != index.end()) {
205  if (pos->second->status == TransactionObject::Del) {
206  // first remove the item from the container before deleting it
207  auto second = pos->second;
208  auto first = pos->first;
209  index.erase(pos);
210  delete second;
211  delete first;
212  }
213  else {
214  pos->second->status = TransactionObject::New;
215  pos->second->_NameInDocument = Obj->detachFromDocument();
216  // move item at the end to make sure the order of removal is kept
217  auto& seq = _Objects.get<0>();
218  seq.relocate(seq.end(), _Objects.project<0>(pos));
219  }
220  }
221  else {
222  TransactionObject* To = TransactionFactory::instance().createTransaction(Obj->getTypeId());
223  To->status = TransactionObject::New;
224  To->_NameInDocument = Obj->detachFromDocument();
225  index.emplace(Obj, To);
226  }
227 }

303 void TransactionObject::applyChn(Document& /*Doc*/, TransactionalObject* pcObj, bool /* Forward */)
304 {
305  if (status == New || status == Chn) {
306  // Property change order is not preserved, as it is recursive in nature
307  for (auto& v : _PropChangeMap) {
308  auto& data = v.second;
309  auto prop = const_cast<Property*>(data.propertyOrig);
310 
311  if (!data.property) {
312  // here means we are undoing/redoing and property add operation
313  pcObj->removeDynamicProperty(v.second.name.c_str());
314  continue;
315  }
316 
317  // getPropertyName() is specially coded to be safe even if prop has
318  // been destroies. We must prepare for the case where user removed
319  // a dynamic property but does not recordered as transaction.
320  auto name = pcObj->getPropertyName(prop);
321  if (!name || (!data.name.empty() && data.name != name)
322  || data.propertyType != prop->getTypeId()) {
323  // Here means the original property is not found, probably removed
324  if (data.name.empty()) {
325  // not a dynamic property, nothing to do
326  continue;
327  }
328 
329  // It is possible for the dynamic property to be removed and
330  // restored. But since restoring property is actually creating
331  // a new property, the property key inside redo stack will not
332  // match. So we search by name first.
333  prop = pcObj->getDynamicPropertyByName(data.name.c_str());
334  if (!prop) {
335  // Still not found, re-create the property
336  prop = pcObj->addDynamicProperty(data.propertyType.getName(),
337  data.name.c_str(),
338  data.group.c_str(),
339  data.doc.c_str(),
340  data.attr,
341  data.readonly,
342  data.hidden);
343  if (!prop) {
344  continue;
345  }
346  prop->setStatusValue(data.property->getStatus());
347  }
348  }
349 
350  // Many properties do not bother implement Copy() and accepts
351  // derived types just fine in Paste(). So we do not enforce type
352  // matching here. But instead, strengthen type checking in all
353  // Paste() implementation.
354  //
355  // if(data.propertyType != prop->getTypeId()) {
356  // FC_WARN("Cannot " << (Forward?"redo":"undo")
357  // << " change of property " << prop->getName()
358  // << " because of type change: "
359  // << data.propertyType.getName()
360  // << " -> " << prop->getTypeId().getName());
361  // continue;
362  // }
363  try {
364  prop->Paste(*data.property);
365  }
366  catch (Base::Exception& e) {
367  e.ReportException();
368  FC_ERR("exception while restoring " << prop->getFullName() << ": " << e.what());
369  }
370  catch (std::exception& e) {
371  FC_ERR("exception while restoring " << prop->getFullName() << ": " << e.what());
372  }
373  catch (...) {
374  }
375  }
376  }
377 }

392 void TransactionObject::addOrRemoveProperty(const Property* pcProp, bool add)
393 {
394  (void)add;
395  if (!pcProp || !pcProp->getContainer()) {
396  return;
397  }
398 
399  auto& data = _PropChangeMap[pcProp->getID()];
400  if (!data.name.empty()) {
401  if (!add && !data.property) {
402  // this means add and remove the same property inside a single
403  // transaction, so they cancel each other out.
404  _PropChangeMap.erase(pcProp->getID());
405  }
406  return;
407  }
408  if (data.property) {
409  delete data.property;
410  data.property = nullptr;
411  }
412  data.propertyOrig = pcProp;
413  static_cast<DynamicProperty::PropData&>(data) =
414  pcProp->getContainer()->getDynamicPropertyData(pcProp);
415  if (add) {
416  data.property = nullptr;
417  }
418  else {
419  data.property = pcProp->Copy();
420  data.propertyType = pcProp->getTypeId();
421  data.property->setStatusValue(pcProp->getStatus());
422  }

462 void TransactionDocumentObject::applyDel(Document& Doc, TransactionalObject* pcObj)
463 {
464  if (status == Del) {
465  DocumentObject* obj = static_cast<DocumentObject*>(pcObj);
466 
467 #ifndef USE_OLD_DAG
468  // Make sure the backlinks of all linked objects are updated. As the links of the removed
469  // object are never set to [] they also do not remove the backlink. But as they are
470  // not in the document anymore we need to remove them anyway to ensure a correct graph
471  auto list = obj->getOutList();
472  for (auto link : list) {
473  link->_removeBackLink(obj);
474  }
475 #endif
476 
477  // simply filling in the saved object
478  Doc._removeObject(obj);
479  }
480 }
