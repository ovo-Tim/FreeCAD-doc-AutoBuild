159 void ModelSelect::saveRecents()
160 {
161  auto param = App::GetApplication().GetParameterGroupByPath(
162  "User parameter:BaseApp/Preferences/Mod/Material/Models/Recent");
163 
164  // Clear out the existing favorites
165  int count = param->GetInt("Recent", 0);
166  for (int i = 0; static_cast<long>(i) < count; i++) {
167  QString key = QString::fromLatin1("MRU%1").arg(i);
168  param->RemoveASCII(key.toStdString().c_str());
169  }
170 
171  // Add the current values
172  int size = _recents.size();
173  if (size > _recentMax) {
174  size = _recentMax;
175  }
176  param->SetInt("Recent", size);
177  int j = 0;
178  for (auto& recent : _recents) {
179  QString key = QString::fromLatin1("MRU%1").arg(j);
180  param->SetASCII(key.toStdString().c_str(), recent.toStdString());
181 
182  j++;
183  if (j >= size) {
184  break;
185  }
186  }

189 void ModelSelect::addRecent(const QString& uuid)
190 {
191  // Ensure no duplicates
192  if (isRecent(uuid)) {
193  _recents.remove(uuid);
194  }
195 
196  _recents.push_front(uuid);
197  while (_recents.size() > static_cast<std::size_t>(_recentMax)) {
198  _recents.pop_back();
199  }
200 
201  saveRecents();
202 }

234 void ModelSelect::addModels(
235  QStandardItem& parent,
236  const std::shared_ptr<std::map<QString, std::shared_ptr<Materials::ModelTreeNode>>> modelTree,
237  const QIcon& icon)
238 {
239  auto tree = ui->treeModels;
240  for (auto& mod : *modelTree) {
241  std::shared_ptr<Materials::ModelTreeNode> nodePtr = mod.second;
242  if (nodePtr->getType() == Materials::ModelTreeNode::DataNode) {
243  auto model = nodePtr->getData();
244  QString uuid = model->getUUID();
245 
246  auto card = new QStandardItem(icon, model->getName());
247  card->setFlags(Qt::ItemIsSelectable | Qt::ItemIsEnabled | Qt::ItemIsDragEnabled
248  | Qt::ItemIsDropEnabled);
249  card->setData(QVariant(uuid), Qt::UserRole);
250 
251  addExpanded(tree, &parent, card);
252  }
253  else {
254  auto node = new QStandardItem(mod.first);
255  addExpanded(tree, &parent, node);
256  node->setFlags(Qt::ItemIsEnabled | Qt::ItemIsDragEnabled | Qt::ItemIsDropEnabled);
257  auto treeMap = nodePtr->getFolder();
258  addModels(*node, treeMap, icon);
259  }
260  }
261 }

326 void ModelSelect::fillTree()
327 {
328  auto tree = ui->treeModels;
329  auto model = static_cast<QStandardItemModel*>(tree->model());
330  model->clear();
331 
332  auto lib = new QStandardItem(tr("Favorites"));
333  lib->setFlags(Qt::ItemIsEnabled | Qt::ItemIsDragEnabled | Qt::ItemIsDropEnabled);

339  addExpanded(tree, model, lib);

346  addExpanded(tree, model, lib);
347 
348  auto modelTree = getModelManager().getModelTree(library, _filter);
349  addModels(*lib, modelTree, QIcon(library->getIconPath()));

464 void ModelSelect::onSelectModel(const QItemSelection& selected, const QItemSelection& deselected)
465 {
466  Q_UNUSED(deselected);
467 
468  auto model = dynamic_cast<QStandardItemModel*>(ui->treeModels->model());
469  QModelIndexList indexes = selected.indexes();
470  for (auto it = indexes.begin(); it != indexes.end(); it++) {
471  QStandardItem* item = model->itemFromIndex(*it);
472  if (item) {
473  try {
474  _selected = item->data(Qt::UserRole).toString();
475  updateMaterialModel(_selected);
476  ui->standardButtons->button(QDialogButtonBox::Ok)->setEnabled(true);
477  ui->buttonFavorite->setEnabled(true);
