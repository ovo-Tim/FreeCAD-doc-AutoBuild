65 void DlgStringListEditor::fillList(std::vector<std::string> texts)
66 {
67  QString qText;
68  int textCount = texts.size();
69  int i = 0;
70  for (; i < textCount; i++) {
71  qText = QString::fromStdString(texts[i]);
72  QListWidgetItem* item = new QListWidgetItem(qText);
73  item->setFlags(item->flags() | Qt::ItemIsEditable);
74  ui->lwTexts->addItem(item);
75  }
76  //add a blank line at the end to allow extending the list
77  QListWidgetItem* item = new QListWidgetItem(QString::fromUtf8(""));
78  item->setFlags(item->flags() | Qt::ItemIsEditable);

87 void DlgStringListEditor::slotAddItem()
88 {
89  QString newText = ui->leNewItem->text();
90  QListWidgetItem* item = new QListWidgetItem(newText);
91  item->setFlags(item->flags() | Qt::ItemIsEditable);
92  int row = ui->lwTexts->currentRow();
93  if (row < 0) {
94  //no location set yet, add to end of list
95  ui->lwTexts->addItem(item);
96  }
97  else {
98  //insert item at current row and push the rest down 1 position
99  ui->lwTexts->insertItem(row, item);
100  }
101  ui->leNewItem->clear();
102  //TODO: how to append to end of list?
103 }

117 std::vector<std::string> DlgStringListEditor::getTexts() const
118 {
119  std::vector<std::string> outTexts;
120  if (ui->lwTexts->count() < 1) {
121  return outTexts;
122  }
123 
124  for (int iRow = 0; iRow < ui->lwTexts->count(); iRow++) {
125  QString itemText = ui->lwTexts->item(iRow)->text();
126  outTexts.push_back(itemText.toStdString());
127  }
128  if (outTexts.back().empty()) {
129  outTexts.pop_back();
130  }
131  return outTexts;
132 }
