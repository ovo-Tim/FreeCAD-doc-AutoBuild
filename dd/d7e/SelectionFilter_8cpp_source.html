87 {
88  Base::PyGILStateLocker lock;
89  try {

91  Py::Callable method(this->gate.getAttr(std::string("allow")));
92  Py::Object pyDoc = Py::asObject(doc->getPyObject());
93  Py::Object pyObj = Py::asObject(obj->getPyObject());
94  Py::Object pySub = Py::None();

96  pySub = Py::String(sub);
97  Py::Tuple args(3);

99  args.setItem(1, pyObj);

103  }
104  }
105  catch (Py::Exception&) {
106  Base::PyException e; // extract the Python error text
107  e.ReportException();
108  }

161 bool SelectionFilter::match()
162 {
163  if (!Ast)
164  return false;
165  Result.clear();
166 
167  for (const auto& it : Ast->Objects) {
168  std::size_t min = 1;
169  std::size_t max = 1;
170 
171  if (it->Slice) {
172  min = it->Slice->Min;
173  max = it->Slice->Max;
174  }
175 
176  std::vector<Gui::SelectionObject> temp = Gui::Selection().getSelectionEx(nullptr, it->ObjectType);
177 
178  // test if subnames present
179  if (it->SubName.empty()) {
180  // if no subnames the count of the object get tested
181  if (temp.size() < min || temp.size() > max)
182  return false;
183  }
184  else {
185  // if subnames present count all subs over the selected object of type
186  std::size_t subCount = 0;
187  for (const auto & it2 : temp) {
188  const std::vector<std::string>& subNames = it2.getSubNames();
189  if (subNames.empty())
190  return false;
191  for (const auto & subName : subNames) {
192  if (subName.find(it->SubName) != 0)
193  return false;
194  }
195  subCount += subNames.size();
196  }
197 
198  if (subCount < min || subCount > max)
199  return false;
200  }
201 
202  Result.push_back(temp);
203  }
204  return true;
205 }

207 bool SelectionFilter::test(App::DocumentObject*pObj, const char*sSubName)
208 {
209  if (!Ast)
210  return false;
211 
212  for (const auto& it : Ast->Objects) {
213  if (pObj->getTypeId().isDerivedFrom(it->ObjectType)) {
214  if (!sSubName)
215  return true;
216  if (it->SubName.empty())
217  return true;
218  if (std::string(sSubName).find(it->SubName) == 0)
219  return true;
220  }
221  }
222  return false;
223 }
