77 //trap Enter press in dsb? so as not to invoke task accept processing
78 bool VectorEditWidget::eventFilter(QObject *target, QEvent *event)
79 {
80  if (target == dsbX ||
81  target == dsbY ||
82  target == dsbZ) {
83  if (event->type() == QEvent::KeyPress) {
84  QKeyEvent *keyEvent = static_cast<QKeyEvent *>(event);
85  if (keyEvent->key() == Qt::Key_Return ||
86  keyEvent->key() == Qt::Key_Enter) {
87  QDoubleSpinBox* dsb = static_cast<QDoubleSpinBox*>(target);
88  dsb->interpretText();
89  Q_EMIT dsb->valueChanged(dsb->value());
90  return true;
91  }
92  }
93  }
94  return QWidget::eventFilter(target, event);
95 }
