71 void ViewProviderFemConstraintTransform::updateData(const App::Property* prop)
72 {
73  auto obj = this->getObject<Fem::ConstraintTransform>();
74 
75  if (prop == &obj->Rotation) {
76  updateSymbol();
77  }
78  else if (prop == &obj->TransformType || prop == &obj->References) {
79  std::string transType = obj->TransformType.getValueAsString();
80  auto sw = static_cast<SoSwitch*>(getSymbolSeparator()->getChild(0));
81  auto swExtra = static_cast<SoSwitch*>(getExtraSymbolSeparator()->getChild(0));
82 
83  if (transType == "Rectangular") {
84  sw->whichChild.setValue(0);
85  swExtra->whichChild.setValue(-1);
86  }
87  else if (transType == "Cylindrical") {
88  sw->whichChild.setValue(1);
89  if (obj->References.getSize()) {
90  swExtra->whichChild.setValue(0);
91  }
92  else {
93  swExtra->whichChild.setValue(-1);
94  }
95  }
96  updateSymbol();
97  }
98  else if (prop == &obj->BasePoint || prop == &obj->Axis) {
99  updateSymbol();
100  }
101 
102  ViewProviderFemConstraint::updateData(prop);
103 }

105 void ViewProviderFemConstraintTransform::transformSymbol(const Base::Vector3d& point,
106  const Base::Vector3d& normal,
107  SbMatrix& mat) const
108 {
109  auto obj = this->getObject<const Fem::ConstraintTransform>();
110 
111  std::string transType = obj->TransformType.getValueAsString();
112  if (transType == "Rectangular") {
113  Base::Rotation rot = obj->Rotation.getValue();
114  Base::Vector3d axis;
115  double angle;
116  rot.getValue(axis, angle);
117  float s = obj->getScaleFactor();
118 
119  mat.setTransform(SbVec3f(point.x, point.y, point.z),
120  SbRotation(SbVec3f(axis.x, axis.y, axis.z), angle),
121  SbVec3f(s, s, s));
122  }
123  else if (transType == "Cylindrical") {
124  float s = obj->getScaleFactor();
125 
126  mat.setTransform(SbVec3f(point.x, point.y, point.z),
127  SbRotation(SbVec3f(0, 1, 0), SbVec3f(normal.x, normal.y, normal.z)),
128  SbVec3f(s, s, s));
129  }
130 }
