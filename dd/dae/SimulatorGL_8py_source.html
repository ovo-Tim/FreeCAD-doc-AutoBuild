117  def FindClosestEdge(self, edges, px, pz):
118  """Convert tool shape to tool profile needed by GL simulator"""
119  for edge in edges:
120  p1 = edge.FirstParameter
121  p2 = edge.LastParameter
122  rad1 = RadiusAt(edge, p1)
123  z1 = edge.valueAt(p1).z
124  if IsSame(px, rad1) and IsSame(pz, z1):
125  return edge, p1, p2
126  rad2 = RadiusAt(edge, p2)
127  z2 = edge.valueAt(p2).z
128  if IsSame(px, rad2) and IsSame(pz, z2):
129  return edge, p2, p1
130  # sometimes a flat circle is without edge, so return edge with
131  # same height and later a connecting edge will be interpolated
132  if IsSame(pz, z1):
133  return edge, p1, p2
134  if IsSame(pz, z2):
135  return edge, p2, p1
136  return None, 0.0, 0.0

138  def FindTopMostEdge(self, edges):
139  """Examine tool solid edges and find the top most one"""
140  maxz = -99999999.0
141  topedge = None
142  top_p1 = 0.0
143  top_p2 = 0.0
144  for edge in edges:
145  p1 = edge.FirstParameter
146  p2 = edge.LastParameter
147  z = edge.valueAt(p1).z
148  if z > maxz:
149  topedge = edge
150  top_p1 = p1
151  top_p2 = p2
152  maxz = z
153  z = edge.valueAt(p2).z
154  if z > maxz:
155  topedge = edge
156  top_p1 = p2
157  top_p2 = p1
158  maxz = z
159  return topedge, top_p1, top_p2

161  def GetToolProfile(self, tool, resolution):
162  """Get the edge profile of a tool solid. Basically locating the
163  side edge that OCC creates on any revolved object
164  """
165  originalPlacement = tool.Placement
166  tool.Placement = Placement(Vector(0, 0, 0), Rotation(Vector(0, 0, 1), 0), Vector(0, 0, 0))
167  shape = tool.Shape
168  tool.Placement = originalPlacement
169  sideEdgeList = []
170  for _i, edge in enumerate(shape.Edges):
171  if not edge.isClosed():
172  # v1 = edge.firstVertex()
173  # v2 = edge.lastVertex()
174  # tp = "arc" if type(edge.Curve) is Part.Circle else "line"
175  sideEdgeList.append(edge)
176 
177  # sort edges as a single 3d line on the x-z plane
178 
179  # first find the topmost edge
180  edge, p1, p2 = self.FindTopMostEdge(sideEdgeList)
181  profile = [RadiusAt(edge, p1), edge.valueAt(p1).z]
182  endrad = 0.0
183  # one by one find all connecting edges
184  while edge is not None:
185  sideEdgeList.remove(edge)
186  if isinstance(edge.Curve, Part.Circle):
187  # if edge is curved, approximate it with lines based on resolution
188  nsegments = int(edge.Length / resolution) + 1
189  step = (p2 - p1) / nsegments
190  location = p1 + step
191  while nsegments > 0:
192  endrad = RadiusAt(edge, location)
193  endz = edge.valueAt(location).z
194  profile.append(endrad)
195  profile.append(endz)
196  location += step
197  nsegments -= 1
198  else:
199  endrad = RadiusAt(edge, p2)
200  endz = edge.valueAt(p2).z
201  profile.append(endrad)
202  profile.append(endz)
203  edge, p1, p2 = self.FindClosestEdge(sideEdgeList, endrad, endz)
204  if edge is None:
205  break
206  startrad = RadiusAt(edge, p1)
207  if not IsSame(startrad, endrad):
208  profile.append(startrad)
209  startz = edge.valueAt(p1).z
210  profile.append(startz)
211 
212  return profile
