96 void Flag::drawLine (View3DInventorViewer* v, int tox, int toy)
97 {
98  if (!isVisible())
99  return;
100 
101  // Get position of line
102  QSize s = parentWidget()->size();
103  SbVec2s view(s.width(), s.height());
104  int fromx = pos().x();
105  int fromy = pos().y() + height()/2;
106  if (false) fromx += width();
107 
108  GLPainter p;
109  p.begin(v->getGLWidget());
110 
111  // the line
112  p.setLineWidth(1.0f);
113  p.drawLine(fromx, fromy, tox, toy);
114  // the point
115  p.setPointSize(3.0f);
116  p.drawPoint(tox, toy);

247 void FlagLayout::setGeometry(const QRect &rect)
248 {
249  int topHeight = 0;
250  int bottomHeight = 0;
251 
252  QLayout::setGeometry(rect);
253 
254  // left side
255  for (ItemWrapper *wrapper : list) {
256  QLayoutItem *item = wrapper->item;
257  Position position = wrapper->position;
258 
259  if (position == TopLeft) {
260  topHeight += spacing();
261  item->setGeometry(QRect(rect.x() + spacing(), topHeight,
262  item->sizeHint().width(), item->sizeHint().height()));
263 
264  topHeight += item->geometry().height();
265  } else if (position == BottomLeft) {
266  bottomHeight += item->geometry().height() + spacing();
267  item->setGeometry(QRect(rect.x() + spacing(), rect.height() - bottomHeight,
268  item->sizeHint().width(), item->sizeHint().height()));
269  }
270  }
271 
272  // right side
273  topHeight = 0;
274  bottomHeight = 0;
275  for (ItemWrapper *wrapper : list) {
276  QLayoutItem *item = wrapper->item;
277  Position position = wrapper->position;
278 
279  int rightpos = item->sizeHint().width() + spacing();
280  if (position == TopRight) {
281  topHeight += spacing();
282  item->setGeometry(QRect(rect.x() + rect.width() - rightpos, topHeight,
283  item->sizeHint().width(), item->sizeHint().height()));
284 
285  topHeight += item->geometry().height();
286  } else if (position == BottomRight) {
287  bottomHeight += item->geometry().height() + spacing();
288  item->setGeometry(QRect(rect.x() + rect.width() - rightpos, rect.height() - bottomHeight,
289  item->sizeHint().width(), item->sizeHint().height()));
290  }
291  }
292 }

313 QSize FlagLayout::calculateSize(SizeType sizeType) const
314 {
315  QSize totalSize;
316 
317  for (ItemWrapper *wrapper : list) {
318  QSize itemSize;
319 
320  if (sizeType == MinimumSize)
321  itemSize = wrapper->item->minimumSize();
322  else // (sizeType == SizeHint)
323  itemSize = wrapper->item->sizeHint();
324 
325  totalSize.rheight() += itemSize.height();
326  totalSize.rwidth() = qMax<int>(totalSize.width(),itemSize.width());
327  }
328  return totalSize;
329 }

400 void GLFlagWindow::paintGL()
401 {
402  // draw lines for the flags
403  if (_flagLayout) {
404  // it can happen that the GL widget gets replaced internally (SoQt only, not with quarter) which
405  // causes to destroy the FlagLayout instance
406  int ct = _flagLayout->count();
407  const SbViewportRegion vp = _viewer->getSoRenderManager()->getViewportRegion();
408  SbVec2s size = vp.getViewportSizePixels();
409  float aspectratio = float(size[0])/float(size[1]);
410  SbViewVolume vv = _viewer->getSoRenderManager()->getCamera()->getViewVolume(aspectratio);
411  for (int i=0; i<ct;i++) {
412  Flag* flag = qobject_cast<Flag*>(_flagLayout->itemAt(i)->widget());
413  if (flag) {
414  SbVec3f pt = flag->getOrigin();
415  vv.projectToScreen(pt, pt);
416  int tox = (int)(pt[0] * size[0]);
417  int toy = (int)((1.0f-pt[1]) * size[1]);
418  flag->drawLine(_viewer, tox, toy);
419  }
420  }
421  }
422 }
