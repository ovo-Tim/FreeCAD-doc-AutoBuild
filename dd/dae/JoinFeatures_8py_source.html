38 def shapeOfMaxVol(compound):
39  if compound.ShapeType == "Compound":
40  maxVol = 0
41  cntEq = 0
42  shMax = None
43  for sh in compound.childShapes():
44  v = sh.Volume
45  if v > maxVol + 1e-8:
46  maxVol = v
47  shMax = sh
48  cntEq = 1
49  elif abs(v - maxVol) <= 1e-8:
50  cntEq = cntEq + 1
51  if cntEq > 1:
52  raise ValueError("Equal volumes, can't figure out what to cut off!")
53  return shMax
54  else:
55  return compound
