33 def makeTube(outerRadius, innerRadius, height):
34  outer_cylinder = Part.makeCylinder(outerRadius, height)
35  shape = outer_cylinder
36  if innerRadius > 0 and innerRadius < outerRadius:
37  inner_cylinder = Part.makeCylinder(innerRadius, height)
38  shape = outer_cylinder.cut(inner_cylinder)
39  return shape
