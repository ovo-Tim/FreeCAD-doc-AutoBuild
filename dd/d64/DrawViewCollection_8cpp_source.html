84 int DrawViewCollection::addView(App::DocumentObject* docObj)
85 {
86  // Add the new view to the collection
87 
88  if (!docObj->isDerivedFrom<DrawView>()
89  && !docObj->isDerivedFrom<App::Link>()) {
90  return -1;
91  }
92 
93  auto* view = dynamic_cast<DrawView*>(docObj);
94 
95  if (!view) {
96  auto* link = dynamic_cast<App::Link*>(docObj);
97  if (!link) {
98  return -1;
99  }
100 
101  if (link) {
102  view = dynamic_cast<DrawView*>(link->getLinkedObject());
103  if (!view) {
104  return -1;
105  }
106  }
107  }
108 
109  std::vector<App::DocumentObject*> newViews(Views.getValues());
110  newViews.push_back(docObj);
111  Views.setValues(newViews);
112 
113  return Views.getSize();
114 }

150 void DrawViewCollection::rebuildViewList()
151 {
152  const std::vector<App::DocumentObject*> currViews = Views.getValues();
153  std::vector<App::DocumentObject*> newViews;
154  for (auto* child : getOutList()) {
155  if (child->isDerivedFrom<DrawView>() ||
156  (child->isDerivedFrom<App::Link>()
157  && static_cast<App::Link*>(child)->getLinkedObject()->isDerivedFrom<DrawView>())) {
158  bool found = false;
159  for (auto& v:currViews) {
160  if (v == child) {
161  found = true;
162  break;
163  }
164  }
165  if (found) {
166  newViews.push_back(child);
167  }
168  }
169  } // newViews contains only valid items, but may have duplicates
170  sort( newViews.begin(), newViews.end() );
171  newViews.erase( unique( newViews.begin(), newViews.end() ), newViews.end() );
172  Views.setValues(newViews);
173 }
