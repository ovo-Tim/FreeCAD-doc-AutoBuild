35  {
36  //Check sizes when in debug mode
37  if(q.rows()!=nj || q_dot.rows()!=nj || q_dotdot.rows()!=nj || torques.rows()!=nj || f_ext.size()!=ns)
38  return (error = -1);
39  unsigned int j=0;
40 
41  //Sweep from root to leaf
42  for(unsigned int i=0;i<ns;i++){
43  double q_,qdot_,qdotdot_;
44  if(chain.getSegment(i).getJoint().getType()!=Joint::None){
45  q_=q(j);
46  qdot_=q_dot(j);
47  qdotdot_=q_dotdot(j);
48  j++;
49  }else
50  q_=qdot_=qdotdot_=0.0;
51 
52  //Calculate segment properties: X,S,vj,cj
53  X[i]=chain.getSegment(i).pose(q_);//Remark this is the inverse of the
54  //frame for transformations from
55  //the parent to the current coord frame
56  //Transform velocity and unit velocity to segment frame
57  Twist vj=X[i].M.Inverse(chain.getSegment(i).twist(q_,qdot_));
58  S[i]=X[i].M.Inverse(chain.getSegment(i).twist(q_,1.0));
59  //We can take cj=0, see remark section 3.5, page 55 since the unit velocity vector S of our joints is always time constant
60  //calculate velocity and acceleration of the segment (in segment coordinates)
61  if(i==0){
62  v[i]=vj;
63  a[i]=X[i].Inverse(ag)+S[i]*qdotdot_+v[i]*vj;
64  }else{
65  v[i]=X[i].Inverse(v[i-1])+vj;
66  a[i]=X[i].Inverse(a[i-1])+S[i]*qdotdot_+v[i]*vj;
67  }
68  //Calculate the force for the joint
69  //Collect RigidBodyInertia and external forces
70  RigidBodyInertia Ii=chain.getSegment(i).getInertia();
71  f[i]=Ii*a[i]+v[i]*(Ii*v[i])-f_ext[i];
72  //std::cout << "a[i]=" << a[i] << "\n f[i]=" << f[i] << "\n S[i]" << S[i] << std::endl;
73  }
74  //Sweep from leaf to root
75  j=nj-1;
76  for(int i=ns-1;i>=0;i--){
77  if(chain.getSegment(i).getJoint().getType()!=Joint::None)
78  torques(j--)=dot(S[i],f[i]);

80  f[i-1]=f[i-1]+X[i]*f[i];
