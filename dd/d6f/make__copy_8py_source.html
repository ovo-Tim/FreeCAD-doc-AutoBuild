35 def make_copy(obj, force=None, reparent=False, simple_copy=False):
36  """make_copy(object, [force], [reparent], [simple_copy])
37 
38  Make an exact copy of an object and return it.
39 
40  Parameters
41  ----------
42  obj :
43  Object to copy.
44 
45  force :
46  Obsolete, not used anymore.
47 
48  reparent :
49  Group the new object in the same group of the old one.
50 
51  simple_copy :
52  Create a simple copy of the object (a new non parametric
53  Part::Feature with the same Shape as the given object).
54  """
55  if not App.ActiveDocument:
56  App.Console.PrintError("No active document. Aborting\n")
57  return
58 
59  newobj = None
60 
61  if simple_copy and hasattr(obj, 'Shape'):
62  # this was the old implementation that is actually not used by default
63  _name = utils.get_real_name(obj.Name)
64  newobj = App.ActiveDocument.addObject("Part::Feature", _name)
65  newobj.Shape = obj.Shape
66  gui_utils.format_object(newobj, obj)
67  elif not simple_copy:
68  # this is the new implementation using doc.copyObject API
69  if obj.hasExtension("App::OriginGroupExtension"):
70  # always copy with dependencies when copying App::Part and PartDesign::Body
71  newobj = App.ActiveDocument.copyObject(obj, True)
72  else:
73  newobj = App.ActiveDocument.copyObject(obj)
74 
75  if not newobj:
76  return None
77 
78  if reparent:
79  parents = obj.InList
80  if parents:
81  for par in parents:
82  if par.isDerivedFrom("App::DocumentObjectGroup") or par.isDerivedFrom("App::Part"):
83  par.addObject(newobj)
84  else:
85  # That's the case of Arch_BuildingParts or Draft_Layers for example
86  if "Group" in par.PropertiesList:
87  if obj in par.Group:
88  group = par.Group
89  group.append(newobj)
90  par.Group = group
91 
92  return newobj
