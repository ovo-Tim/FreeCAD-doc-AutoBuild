135 QuarterWidgetP_cachecontext *
136 QuarterWidgetP::findCacheContext(QuarterWidget * widget, const QtGLWidget * sharewidget)
137 {
138  if (!cachecontext_list) {
139  // FIXME: static memory leak
140  cachecontext_list = new SbList <QuarterWidgetP_cachecontext*>;
141  }
142  for (int i = 0; i < cachecontext_list->getLength(); i++) {
143  QuarterWidgetP_cachecontext * cachecontext = (*cachecontext_list)[i];
144 
145  for (int j = 0; j < cachecontext->widgetlist.getLength(); j++) {
146  if (cachecontext->widgetlist[j] == sharewidget) {
147  cachecontext->widgetlist.append(static_cast<const QtGLWidget*>(widget->viewport()));
148  return cachecontext;
149  }
150  }
151  }
152  QuarterWidgetP_cachecontext * cachecontext = new QuarterWidgetP_cachecontext;
153  cachecontext->id = SoGLCacheContextElement::getUniqueCacheContext();
154  cachecontext->widgetlist.append(static_cast<const QtGLWidget*>(widget->viewport()));
155  cachecontext_list->append(cachecontext);
156 
157  return cachecontext;
158 }

160 void
161 QuarterWidgetP::removeFromCacheContext(QuarterWidgetP_cachecontext * context, const QtGLWidget * widget)
162 {
163  context->widgetlist.removeItem((const QtGLWidget*) widget);
164 
165  if (context->widgetlist.getLength() == 0) { // last context in this share group?
166  assert(cachecontext_list);
167 
168  for (int i = 0; i < cachecontext_list->getLength(); i++) {
169  if ((*cachecontext_list)[i] == context) {
170  QtGLContext* glcontext = widget->context();
171  if (glcontext) {
172  // set the context while calling destructingContext() (might trigger OpenGL calls)
173  if (glcontext->isValid()) {
174  const_cast<QtGLWidget*> (widget)->makeCurrent();
175  }
176  // fetch the cc_glglue context instance as a workaround for a bug fixed in Coin r12818
177  (void) cc_glglue_instance(context->id);
178  }
179  cachecontext_list->removeFast(i);
180  SoContextHandler::destructingContext(context->id);
181  if (glcontext) {
182  if (glcontext->isValid()) {
183  const_cast<QtGLWidget*> (widget)->doneCurrent();
184  }
185  }
186  delete context;
187  return;
188  }
189  }
190  }

332 bool
333 QuarterWidgetP::nativeEventFilter(void * message, long * result)
334 {
335 #ifdef HAVE_SPACENAV_LIB
336  XEvent * event = (XEvent *) message;
337  if (event->type == ClientMessage) {
338  // FIXME: I don't really like this, but the original XEvent will
339  // die before reaching the destination within the Qt system. To
340  // avoid this, we'll have to make a copy. We should try to find a
341  // workaround for this. (20101020 handegar)
342 
343  // The copy will automatically be deleted when the NativeEvent dies.
344  XEvent * copy = (XEvent *) malloc(sizeof(XEvent));
345  memcpy(copy, event, sizeof(XEvent));
346  NativeEvent * ne = new NativeEvent(copy);
347 
348  qApp->postEvent(QApplication::focusWidget(), ne);
349  return true;
350  }
351 #else
352  Q_UNUSED(message);
353  Q_UNUSED(result);
354 #endif // HAVE_SPACENAV_LIB
355 
356  return false;
357 }
