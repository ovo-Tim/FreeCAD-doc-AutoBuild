71 void TemplateTextField::mouseReleaseEvent(QGraphicsSceneMouseEvent *event)
72 {
73  if ( tmplte && m_rect->rect().contains(event->pos()) ) {
74  event->accept();
75 
76  DlgTemplateField ui;
77 
78  ui.setFieldName(fieldNameStr);
79  ui.setFieldContent(tmplte->EditableTexts[fieldNameStr]);
80 
81  auto qName = QString::fromStdString(fieldNameStr);
82  auto svgTemplate = dynamic_cast<DrawSVGTemplate*>(tmplte);
83  if (svgTemplate) {
84  // preset the autofill with the current value - something might have changed since this field was created
85  m_autofillString = svgTemplate->getAutofillByEditableName(qName);
86  }
87  ui.setAutofillContent(m_autofillString.toStdString());
88 
89  if (ui.exec() == QDialog::Accepted) {
90  QString qsClean = ui.getFieldContent();
91  std::string utf8Content = qsClean.toUtf8().constData();
92  if (ui.getAutofillState()) {
93  if (svgTemplate) {
94  // unlikely, but something could have changed since we grabbed the autofill value
95  QString fieldName = QString::fromStdString(fieldNameStr);
96  QString autofillValue = svgTemplate->getAutofillByEditableName(fieldName);
97  if (!autofillValue.isEmpty()) {
98  utf8Content = autofillValue.toUtf8().constData();
99  }
100  }
101  }
102  tmplte->EditableTexts.setValue(fieldNameStr, utf8Content);
103  }
104 
105  } else {
106  QGraphicsItemGroup::mouseReleaseEvent(event);
107  }
108 }
