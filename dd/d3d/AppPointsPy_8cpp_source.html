76  Py::Object open(const Py::Tuple& args)
77  {
78  char* Name {};
79  if (!PyArg_ParseTuple(args.ptr(), "et", "utf-8", &Name)) {
80  throw Py::Exception();
81  }
82  std::string EncodedName = std::string(Name);
83  PyMem_Free(Name);
84 
85  try {
86  Base::Console().Log("Open in Points with %s", EncodedName.c_str());
87  Base::FileInfo file(EncodedName.c_str());
88 
89  // extract ending
90  if (file.extension().empty()) {
91  throw Py::RuntimeError("No file extension");
92  }
93 
94  std::unique_ptr<Reader> reader;
95  if (file.hasExtension("asc")) {
96  reader = std::make_unique<AscReader>();
97  }
98  else if (file.hasExtension("e57")) {
99  auto setting = readE57Settings();
100  reader = std::make_unique<E57Reader>(std::get<0>(setting),
101  std::get<1>(setting),
102  std::get<2>(setting));
103  }
104  else if (file.hasExtension("ply")) {
105  reader = std::make_unique<PlyReader>();
106  }
107  else if (file.hasExtension("pcd")) {
108  reader = std::make_unique<PcdReader>();
109  }
110  else {
111  throw Py::RuntimeError("Unsupported file extension");
112  }
113 
114  reader->read(EncodedName);
115 
116  App::Document* pcDoc = App::GetApplication().newDocument();
117 
118  Points::Feature* pcFeature = nullptr;
119  if (reader->hasProperties()) {
120  // Scattered or structured points?
121  if (reader->isStructured()) {
122  pcFeature = new Points::StructuredCustom();
123 
124  App::PropertyInteger* width =
125  static_cast<App::PropertyInteger*>(pcFeature->getPropertyByName("Width"));
126  if (width) {
127  width->setValue(reader->getWidth());
128  }
129  App::PropertyInteger* height =
130  static_cast<App::PropertyInteger*>(pcFeature->getPropertyByName("Height"));
131  if (height) {
132  height->setValue(reader->getHeight());
133  }
134  }
135  else {
136  pcFeature = new Points::FeatureCustom();
137  }
138 
139  pcFeature->Points.setValue(reader->getPoints());
140  // add gray values
141  if (reader->hasIntensities()) {
142  Points::PropertyGreyValueList* prop =
143  static_cast<Points::PropertyGreyValueList*>(
144  pcFeature->addDynamicProperty("Points::PropertyGreyValueList",
145  "Intensity"));
146  if (prop) {
147  prop->setValues(reader->getIntensities());
148  }
149  }
150  // add colors
151  if (reader->hasColors()) {
152  App::PropertyColorList* prop = static_cast<App::PropertyColorList*>(
153  pcFeature->addDynamicProperty("App::PropertyColorList", "Color"));
154  if (prop) {
155  prop->setValues(reader->getColors());
156  }
157  }
158  // add normals
159  if (reader->hasNormals()) {
160  Points::PropertyNormalList* prop = static_cast<Points::PropertyNormalList*>(
161  pcFeature->addDynamicProperty("Points::PropertyNormalList", "Normal"));
162  if (prop) {
163  prop->setValues(reader->getNormals());
164  }
165  }
166 
167  // delayed adding of the points feature
168  pcDoc->addObject(pcFeature, file.fileNamePure().c_str());
169  pcDoc->recomputeFeature(pcFeature);
170  pcFeature->purgeTouched();
171  }
172  else {
173  if (reader->isStructured()) {
174  Structured* structured = new Points::Structured();
175  structured->Width.setValue(reader->getWidth());
176  structured->Height.setValue(reader->getHeight());
177  pcFeature = structured;
178  }
179  else {
180  pcFeature = new Points::Feature();
181  }
182 
183  // delayed adding of the points feature
184  pcFeature->Points.setValue(reader->getPoints());
185  pcDoc->addObject(pcFeature, file.fileNamePure().c_str());
186  pcDoc->recomputeFeature(pcFeature);
187  pcFeature->purgeTouched();
188  }
189  }
190  catch (const Base::Exception& e) {
191  throw Py::RuntimeError(e.what());
192  }
193 
194  return Py::None();
195  }

312  Py::Object exporter(const Py::Tuple& args)
313  {
314  PyObject* object {};
315  char* Name {};
316 
317  if (!PyArg_ParseTuple(args.ptr(), "Oet", &object, "utf-8", &Name)) {
318  throw Py::Exception();
319  }
320 
321  std::string encodedName = std::string(Name);
322  PyMem_Free(Name);
323 
324  Base::FileInfo file(encodedName);
325 
326  // extract ending
327  if (file.extension().empty()) {
328  throw Py::RuntimeError("No file extension");
329  }
330 
331  Py::Sequence list(object);
332  Base::Type pointsId = Base::Type::fromName("Points::Feature");
333  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
334  PyObject* item = (*it).ptr();
335  if (PyObject_TypeCheck(item, &(App::DocumentObjectPy::Type))) {
336  App::DocumentObject* obj =
337  static_cast<App::DocumentObjectPy*>(item)->getDocumentObjectPtr();
338  if (obj->getTypeId().isDerivedFrom(pointsId)) {
339  // get relative placement
340  Points::Feature* fea = static_cast<Points::Feature*>(obj);
341  Base::Placement globalPlacement = fea->globalPlacement();
342 
343  const PointKernel& kernel = fea->Points.getValue();
344  std::unique_ptr<Writer> writer;
345  if (file.hasExtension("asc")) {
346  writer = std::make_unique<AscWriter>(kernel);
347  }
348  else if (file.hasExtension("ply")) {
349  writer = std::make_unique<PlyWriter>(kernel);
350  }
351  else if (file.hasExtension("pcd")) {
352  writer = std::make_unique<PcdWriter>(kernel);
353  }
354  else {
355  throw Py::RuntimeError("Unsupported file extension");
356  }
357 
358  // get additional properties if there
359  App::PropertyInteger* width =
360  dynamic_cast<App::PropertyInteger*>(fea->getPropertyByName("Width"));
361  if (width) {
362  writer->setWidth(width->getValue());
363  }
364  App::PropertyInteger* height =
365  dynamic_cast<App::PropertyInteger*>(fea->getPropertyByName("Height"));
366  if (height) {
367  writer->setHeight(height->getValue());
368  }
369  // get gray values
370  Points::PropertyGreyValueList* grey =
371  dynamic_cast<Points::PropertyGreyValueList*>(
372  fea->getPropertyByName("Intensity"));
373  if (grey) {
374  writer->setIntensities(grey->getValues());
375  }
376  // get colors
377  App::PropertyColorList* col =
378  dynamic_cast<App::PropertyColorList*>(fea->getPropertyByName("Color"));
379  if (col) {
380  writer->setColors(col->getValues());
381  }
382  // get normals
383  Points::PropertyNormalList* nor =
384  dynamic_cast<Points::PropertyNormalList*>(fea->getPropertyByName("Normal"));
385  if (nor) {
386  writer->setNormals(nor->getValues());
387  }
388 
389  writer->setPlacement(globalPlacement);
390  writer->write(encodedName);
391 
392  break;
393  }
394  else {
395  Base::Console().Message("'%s' is not a point object, export will be ignored.\n",
396  obj->Label.getValue());
397  }
398  }
399  }
400 
401  return Py::None();
402  }
