49 App::DocumentObjectExecReturn* FeatureCompound::execute()
50 {
51  const std::vector<DocumentObject*>& Paths = Group.getValues();
52  Path::Toolpath result;
53 
54  for (std::vector<DocumentObject*>::const_iterator it = Paths.begin(); it != Paths.end(); ++it) {
55  if ((*it)->isDerivedFrom<Path::Feature>()) {
56  const std::vector<Command*>& cmds =
57  static_cast<Path::Feature*>(*it)->Path.getValue().getCommands();
58  const Base::Placement pl = static_cast<Path::Feature*>(*it)->Placement.getValue();
59  for (std::vector<Command*>::const_iterator it2 = cmds.begin(); it2 != cmds.end();
60  ++it2) {
61  if (UsePlacements.getValue()) {
62  result.addCommand((*it2)->transform(pl));
63  }
64  else {
65  result.addCommand(**it2);
66  }
67  }
68  }
69  else {
70  return new App::DocumentObjectExecReturn("Not all objects in group are paths!");
71  }
72  }
73 
74  result.setCenter(Path.getValue().getCenter());
75  Path.setValue(result);
76 
77  return App::DocumentObject::StdReturn;
78 }
