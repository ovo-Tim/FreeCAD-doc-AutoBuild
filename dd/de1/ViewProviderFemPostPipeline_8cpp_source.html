84 void ViewProviderFemPostPipeline::updateFunctionSize()
85 {
86 
87  // we need to get the bounding box and set the function provider size
88  Fem::FemPostPipeline* obj = getObject<Fem::FemPostPipeline>();
89 
90  if (!obj->Functions.getValue()
91  || !obj->Functions.getValue()->isDerivedFrom(
92  Fem::FemPostFunctionProvider::getClassTypeId())) {
93  return;
94  }
95 
96  // get the function provider
97  FemGui::ViewProviderFemPostFunctionProvider* vp =
98  static_cast<FemGui::ViewProviderFemPostFunctionProvider*>(
99  Gui::Application::Instance->getViewProvider(obj->Functions.getValue()));
100 
101  if (obj->Data.getValue() && obj->Data.getValue()->IsA("vtkDataSet")) {
102  vtkBoundingBox box = obj->getBoundingBox();
103 
104  vp->SizeX.setValue(box.GetLength(0) * 1.2);
105  vp->SizeY.setValue(box.GetLength(1) * 1.2);
106  vp->SizeZ.setValue(box.GetLength(2) * 1.2);
107  }
108 }

179 void ViewProviderFemPostPipeline::transformField(char* FieldName, double FieldFactor)
180 {
181  Fem::FemPostPipeline* obj = getObject<Fem::FemPostPipeline>();
182 
183  vtkSmartPointer<vtkDataObject> data = obj->Data.getValue();
184  vtkDataSet* dset = vtkDataSet::SafeDownCast(data);
185  if (!dset) {
186  return;
187  }
188  vtkDataArray* pdata = dset->GetPointData()->GetArray(FieldName);
189  if (!pdata) {
190  return;
191  }
192 
193  auto strFieldName = std::string(FieldName);
194 
195  // for EigenModes, we need to step through all available modes
196  if (strFieldName.find("EigenMode") != std::string::npos) {
197  int modeCount;
198  std::string testFieldName;
199  // since a valid FieldName must have been passed
200  // we assume the mode number was < 10 and we can strip the last char
201  strFieldName.pop_back();
202  for (modeCount = 1; pdata; ++modeCount) {
203  testFieldName = strFieldName + std::to_string(modeCount);
204  pdata = dset->GetPointData()->GetArray(testFieldName.c_str());
205  if (pdata) {
206  scaleField(dset, pdata, FieldFactor);
207  }
208  }
209  }
210  else {
211  scaleField(dset, pdata, FieldFactor);
212  }
213 }

215 void ViewProviderFemPostPipeline::scaleField(vtkDataSet* dset,
216  vtkDataArray* pdata,
217  double FieldFactor)
218 {
219  // safe guard
220  if (!dset || !pdata) {
221  return;
222  }
223 
224  // step through all mesh points and scale them
225  for (int i = 0; i < dset->GetNumberOfPoints(); ++i) {
226  double value = 0;
227  if (pdata->GetNumberOfComponents() == 1) {
228  value = pdata->GetComponent(i, 0);
229  pdata->SetComponent(i, 0, value * FieldFactor);
230  }
231  // if field is a vector
232  else {
233  for (int j = 0; j < pdata->GetNumberOfComponents(); ++j) {
234  value = pdata->GetComponent(i, j);
235  pdata->SetComponent(i, j, value * FieldFactor);
236  }
237  }
238  }
239 }
