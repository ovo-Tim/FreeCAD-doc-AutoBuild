134 void DlgBooleanOperation::slotChangedObject(const App::DocumentObject& obj,
135  const App::Property& prop)
136 {
137  std::list<const App::DocumentObject*>::iterator it;
138  it = std::find(observe.begin(), observe.end(), &obj);

141  if (!shape.IsNull()) {
142  Gui::Document* activeGui = Gui::Application::Instance->getDocument(obj.getDocument());
143  QString label = QString::fromUtf8(obj.Label.getValue());
144  QString name = QString::fromLatin1(obj.getNameInDocument());

146  QTreeWidgetItem* child = new BooleanOperationItem();

148  child->setText(0, label);

151  Gui::ViewProvider* vp = activeGui->getViewProvider(&obj);
152  if (vp)
153  child->setIcon(0, vp->getIcon());
154 
155  QTreeWidgetItem* copy = new BooleanOperationItem();
156  copy->setCheckState(0, Qt::Unchecked);
157  copy->setText(0, label);

162 
163  TopAbs_ShapeEnum type = shape.ShapeType();

165  ui->firstShape->topLevelItem(0)->addChild(child);
166  ui->secondShape->topLevelItem(0)->addChild(copy);
167  ui->firstShape->topLevelItem(0)->setExpanded(true);
168  ui->secondShape->topLevelItem(0)->setExpanded(true);
169  }
170  else if (type == TopAbs_SHELL) {
171  ui->firstShape->topLevelItem(1)->addChild(child);
172  ui->secondShape->topLevelItem(1)->addChild(copy);
173  ui->firstShape->topLevelItem(1)->setExpanded(true);
174  ui->secondShape->topLevelItem(1)->setExpanded(true);
175  }
176  else if (type == TopAbs_COMPOUND || type == TopAbs_COMPSOLID) {
177  ui->firstShape->topLevelItem(2)->addChild(child);
178  ui->secondShape->topLevelItem(2)->addChild(copy);
179  ui->firstShape->topLevelItem(2)->setExpanded(true);
180  ui->secondShape->topLevelItem(2)->setExpanded(true);

186  ui->secondShape->topLevelItem(3)->setExpanded(true);

189  delete child; child = nullptr;

212 void DlgBooleanOperation::findShapes()
213 {
214  App::Document* activeDoc = App::GetApplication().getActiveDocument();
215  if (!activeDoc)
216  return;
217  Gui::Document* activeGui = Gui::Application::Instance->getDocument(activeDoc);
218  if (!activeGui)
219  return;
220 
221  std::vector<App::DocumentObject*> objs = activeDoc->getObjectsOfType
222  (Part::Feature::getClassTypeId());
223 
224  QTreeWidgetItem *item_left=nullptr, *item_right=nullptr;
225  for (auto obj : objs) {
226  const TopoDS_Shape& shape = static_cast<Part::Feature*>(obj)->Shape.getValue();
227  if (!shape.IsNull()) {
228  QString label = QString::fromUtf8(obj->Label.getValue());
229  QString name = QString::fromLatin1(obj->getNameInDocument());
230 
231  QTreeWidgetItem* child = new BooleanOperationItem();
232  child->setCheckState(0, Qt::Unchecked);
233  child->setText(0, label);
234  child->setToolTip(0, label);
235  child->setData(0, Qt::UserRole, name);
236  Gui::ViewProvider* vp = activeGui->getViewProvider(obj);
237  if (vp)
238  child->setIcon(0, vp->getIcon());
239 
240  QTreeWidgetItem* copy = new BooleanOperationItem();
241  copy->setCheckState(0, Qt::Unchecked);
242  copy->setText(0, label);
243  copy->setToolTip(0, label);
244  copy->setData(0, Qt::UserRole, name);
245  if (vp)
246  copy->setIcon(0, vp->getIcon());
247 
248  TopAbs_ShapeEnum type = shape.ShapeType();
249  if (type == TopAbs_SOLID) {
250  ui->firstShape->topLevelItem(0)->addChild(child);
251  ui->secondShape->topLevelItem(0)->addChild(copy);
252  }
253  else if (type == TopAbs_SHELL) {
254  ui->firstShape->topLevelItem(1)->addChild(child);
255  ui->secondShape->topLevelItem(1)->addChild(copy);
256  }
257  else if (type == TopAbs_COMPOUND || type == TopAbs_COMPSOLID) {
258  ui->firstShape->topLevelItem(2)->addChild(child);
259  ui->secondShape->topLevelItem(2)->addChild(copy);
260  }
261  else if (type == TopAbs_FACE) {
262  ui->firstShape->topLevelItem(3)->addChild(child);
263  ui->secondShape->topLevelItem(3)->addChild(copy);
264  }
265  else { // belongs to none of these groups
266  delete child; child = nullptr;
267  delete copy; copy = nullptr;
268  }
269 
270  if (!item_left || !item_right) {
271  bool selected = Gui::Selection().isSelected(obj);
272  if (!item_left && selected)
273  item_left = child;
274  else if (!item_right && selected)
275  item_right = copy;
276  }
277  }
278  }
279 
280  if (item_left) {
281  item_left->setCheckState(0, Qt::Checked);
282  ui->firstShape->setCurrentItem(item_left);
283  }
284  if (item_right) {
285  item_right->setCheckState(0, Qt::Checked);
286  ui->secondShape->setCurrentItem(item_right);
287  }
288  for (int i = 0; i < ui->firstShape->topLevelItemCount(); i++) {
289  QTreeWidgetItem* group = ui->firstShape->topLevelItem(i);
290  group->setFlags(Qt::ItemIsEnabled);
291  if (group->childCount() > 0)
292  group->setExpanded(true);
293  }
294  for (int i = 0; i < ui->secondShape->topLevelItemCount(); i++) {
295  QTreeWidgetItem* group = ui->secondShape->topLevelItem(i);
296  group->setFlags(Qt::ItemIsEnabled);
297  if (group->childCount() > 0)
298  group->setExpanded(true);
299  }

324 void DlgBooleanOperation::onSwapButtonClicked()
325 {
326  QTreeWidgetItem* lChild = ui->firstShape->currentItem();
327  bool lsel = (lChild && (lChild->checkState(0) & Qt::Checked));
328  QTreeWidgetItem* rChild = ui->secondShape->currentItem();
329  bool rsel = (rChild && (rChild->checkState(0) & Qt::Checked));

335  child->setCheckState(0, Qt::Checked);
336  ui->firstShape->setCurrentItem(child);
337  }
338  }
339  if (lsel) {
340  int top_index, child_ind;
341  if (indexOfCurrentItem(lChild, top_index, child_ind)) {
342  QTreeWidgetItem* child = ui->secondShape->topLevelItem(top_index)->child(child_ind);
343  child->setCheckState(0, Qt::Checked);
344  ui->secondShape->setCurrentItem(child);
345  }
346  }

349 void DlgBooleanOperation::accept()
350 {
351  int ltop, lchild, rtop, rchild;
352 
353  QTreeWidgetItem* litem = nullptr;
354  int numLChild = ui->firstShape->topLevelItemCount();
355  for (int i=0; i<numLChild; i++) {
356  QTreeWidgetItem* item = ui->firstShape->topLevelItem(i);
357  for (int j=0; j<item->childCount(); j++) {
358  QTreeWidgetItem* child = item->child(j);
359  if (child && child->checkState(0) & Qt::Checked) {
360  litem = child;
361  break;
362  }
363  }
364 
365  if (litem)
366  break;
367  }
368 
369  QTreeWidgetItem* ritem = nullptr;
370  int numRChild = ui->secondShape->topLevelItemCount();
371  for (int i=0; i<numRChild; i++) {
372  QTreeWidgetItem* item = ui->secondShape->topLevelItem(i);
373  for (int j=0; j<item->childCount(); j++) {
374  QTreeWidgetItem* child = item->child(j);
375  if (child && child->checkState(0) & Qt::Checked) {
376  ritem = child;
377  break;
378  }
379  }
380 
381  if (ritem)
382  break;
383  }
384 
385  if (!litem || !indexOfCurrentItem(litem,ltop,lchild)) {
386  QMessageBox::critical(this, windowTitle(),
387  tr("Select a shape on the left side, first"));
388  return;
389  }
390  if (!ritem || !indexOfCurrentItem(ritem,rtop,rchild)) {
391  QMessageBox::critical(this, windowTitle(),
392  tr("Select a shape on the right side, first"));
393  return;
394  }
395  if (ltop == rtop && lchild == rchild) {
396  QMessageBox::critical(this, windowTitle(),
397  tr("Cannot perform a boolean operation with the same shape"));
398  return;
399  }
400 
401  std::string shapeOne, shapeTwo;
402  shapeOne = (const char*)litem->data(0, Qt::UserRole).toByteArray();
403  shapeTwo = (const char*)ritem->data(0, Qt::UserRole).toByteArray();
404  App::Document* activeDoc = App::GetApplication().getActiveDocument();
405  if (!activeDoc) {
406  QMessageBox::critical(this, windowTitle(),
407  tr("No active document available"));
408  return;
409  }
410 
411  std::string method;
412  App::DocumentObject* obj1 = activeDoc->getObject(shapeOne.c_str());
413  App::DocumentObject* obj2 = activeDoc->getObject(shapeTwo.c_str());
414  if (!obj1 || !obj2) {
415  // objects don't exists (anymore)
416  QMessageBox::critical(this, windowTitle(),
417  tr("One of the selected objects doesn't exist anymore"));
418  return;
419  }
420 
421  if (ui->unionButton->isChecked()) {
422  if (!hasSolids(obj1) || !hasSolids(obj2)) {
423  QMessageBox::critical(this, windowTitle(),
424  tr("Performing union on non-solids is not possible"));
425  return;
426  }
427  method = "make_fuse";
428  }
429  else if (ui->interButton->isChecked()) {
430  if (!hasSolids(obj1) || !hasSolids(obj2)) {
431  QMessageBox::critical(this, windowTitle(),
432  tr("Performing intersection on non-solids is not possible"));
433  return;
434  }
435  method = "make_common";
436  }
437  else if (ui->diffButton->isChecked()) {
438  if (!hasSolids(obj1) || !hasSolids(obj2)) {
439  QMessageBox::critical(this, windowTitle(),
440  tr("Performing difference on non-solids is not possible"));
441  return;
442  }
443  method = "make_cut";
444  }
445  else if (ui->sectionButton->isChecked()) {
446  method = "make_section";
447  }
448 
449  try {
450  Gui::WaitCursor wc;
451  activeDoc->openTransaction("Boolean operation");
452  std::vector<std::string> names;
453  names.push_back(Base::Tools::quoted(shapeOne.c_str()));
454  names.push_back(Base::Tools::quoted(shapeTwo.c_str()));
455  Gui::Command::doCommand(Gui::Command::Doc,
456  "from BOPTools import BOPFeatures");
457  Gui::Command::doCommand(Gui::Command::Doc,
458  "bp = BOPFeatures.BOPFeatures(App.activeDocument())");
459  Gui::Command::doCommand(Gui::Command::Doc,
460  "bp.%s([%s])", method.c_str(), Base::Tools::joinList(names).c_str());
461  activeDoc->commitTransaction();
462  activeDoc->recompute();
463  }
464  catch (const Base::Exception& e) {
465  e.ReportException();
466  }
467 }
