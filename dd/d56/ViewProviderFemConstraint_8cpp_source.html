102 void ViewProviderFemConstraint::loadSymbol(const char* fileName)
103 {
104  ivFile = fileName;
105  SoInput in;
106  if (!in.openFile(ivFile)) {
107  std::stringstream str;
108  str << "Error opening symbol file " << fileName;
109  throw Base::ImportError(str.str());
110  }
111  SoSeparator* nodes = SoDB::readAll(&in);
112  if (!nodes) {
113  std::stringstream str;
114  str << "Error reading symbol file " << fileName;
115  throw Base::ImportError(str.str());
116  }
117 
118  nodes->ref();
119  pSymbol = dynamic_cast<SoSeparator*>(nodes->getChild(0));
120  pShapeSep->addChild(pMultCopy);
121  if (pSymbol) {
122  pMultCopy->addChild(pSymbol);
123  }
124  if (nodes->getNumChildren() == 2) {
125  pExtraSymbol = dynamic_cast<SoSeparator*>(nodes->getChild(1));
126  if (pExtraSymbol) {
127  pShapeSep->addChild(pExtraTrans);
128  pShapeSep->addChild(pExtraSymbol);
129  }
130  }
131  pMultCopy->matrix.setNum(0);
132  nodes->unref();
133 }

203 void ViewProviderFemConstraint::updateSymbol()
204 {
205  auto obj = this->getObject<const Fem::Constraint>();
206  const std::vector<Base::Vector3d>& points = obj->Points.getValue();
207  const std::vector<Base::Vector3d>& normals = obj->Normals.getValue();
208  if (points.size() != normals.size()) {
209  return;
210  }
211 
212  pMultCopy->matrix.setNum(points.size());
213  SbMatrix* mat = pMultCopy->matrix.startEditing();
214 
215  for (size_t i = 0; i < points.size(); ++i) {
216  transformSymbol(points[i], normals[i], mat[i]);
217  }
218 
219  pMultCopy->matrix.finishEditing();
220 
221  transformExtraSymbol();
222 }

224 void ViewProviderFemConstraint::transformSymbol(const Base::Vector3d& point,
225  const Base::Vector3d& normal,
226  SbMatrix& mat) const
227 {
228  auto obj = this->getObject<const Fem::Constraint>();
229  SbVec3f axisY(0, 1, 0);
230  float s = obj->getScaleFactor();
231  SbVec3f scale(s, s, s);
232  SbVec3f norm = rotateSymbol ? SbVec3f(normal.x, normal.y, normal.z) : axisY;
233  SbRotation rot(axisY, norm);
234  SbVec3f tra(static_cast<float>(point.x),
235  static_cast<float>(point.y),
236  static_cast<float>(point.z));
237  mat.setTransform(tra, rot, scale);
238 }
