184 def init_toolbar(workbench, toolbar, cmd_list):
185  """Initialize a toolbar.
186 
187  Parameters
188  ----------
189  workbench: Gui.Workbench
190  The workbench. The commands from cmd_list must be available.
191 
192  toolbar: string
193  The name of the toolbar.
194 
195  cmd_list: list of strings or list of strings and tuples
196  See f.e. the return value of get_draft_drawing_commands.
197  """
198  for cmd in cmd_list:
199  if isinstance(cmd, tuple):
200  if len(cmd) == 1:
201  workbench.appendToolbar(toolbar, [cmd[0]])
202  else:
203  workbench.appendToolbar(toolbar, [cmd])

206 def init_menu(workbench, menu_list, cmd_list):
207  """Initialize a menu.
208 
209  Parameters
210  ----------
211  workbench: Gui.Workbench
212  The workbench. The commands from cmd_list must be available.
213 
214  menu_list: list of strings
215  The main and optional submenu(s). The commands, and additional
216  submenus (if any), are added to the last (sub)menu in the list.
217 
218  cmd_list: list of strings or list of strings and tuples
219  See f.e. the return value of get_draft_drawing_commands.
220  """
221  for cmd in cmd_list:
222  if isinstance(cmd, tuple):
223  if len(cmd) == 2:
224  workbench.appendMenu(menu_list + cmd[0], cmd[1])
225  else:
226  workbench.appendMenu(menu_list, [cmd])
