179 bool CheckCompileResult(int shaderId, const char* shaderName, bool isVertex)
180 {
181  char log[1024];
182  int res = 0;
183  GLsizei len;
184  glGetShaderiv(shaderId, GL_COMPILE_STATUS, &res);
185  if (res != 0) {
186  return false;
187  }
188  int headerLen = snprintf(log,
189  48,
190  "Error compiling %s %s shader: ",
191  shaderName,
192  isVertex ? "vertex" : "fragment");
193  glGetShaderInfoLog(shaderId, 1020 - headerLen, &len, log + headerLen);
194  len += headerLen;
195  if (len > 1020) {
196  len = 1020;
197  }
198  log[len] = 0;
199  Base::Console().Error(log);
200  return true;
201 }

203 unsigned int
204 Shader::CompileShader(const char* name, const char* _vertShader, const char* _fragShader)
205 {
206  vertShader = _vertShader;
207  fragShader = _fragShader;
208  const GLuint vertex_shader = glCreateShader(GL_VERTEX_SHADER);
209  GLint res = 0;
210  glShaderSource(vertex_shader, 1, &vertShader, NULL);
211  glCompileShader(vertex_shader);
212  if (CheckCompileResult(vertex_shader, name, true)) {
213  glDeleteShader(vertex_shader);
214  return 0xdeadbeef;
215  }
216 
217  const GLuint fragment_shader = glCreateShader(GL_FRAGMENT_SHADER);
218  glShaderSource(fragment_shader, 1, &fragShader, NULL);
219  glCompileShader(fragment_shader);
220  if (CheckCompileResult(fragment_shader, name, false)) {
221  glDeleteShader(fragment_shader);
222  glDeleteShader(vertex_shader);
223  return 0xdeadbeef;
224  }
225 
226  shaderId = glCreateProgram();
227  glAttachShader(shaderId, vertex_shader);
228  glAttachShader(shaderId, fragment_shader);
229  glLinkProgram(shaderId);
230 
231  glGetProgramiv(shaderId, GL_LINK_STATUS, &res);
232  if (res == 0) {
233  Destroy();
234  return 0xdeadbeef;
235  }
236 
237  // get all uniform parameters positions
238  mModelPos = glGetUniformLocation(shaderId, "model");
239  mNormalRotPos = glGetUniformLocation(shaderId, "normalRot");
240  mProjectionPos = glGetUniformLocation(shaderId, "projection");
241  mViewPos = glGetUniformLocation(shaderId, "view");
242  mLightPosPos = glGetUniformLocation(shaderId, "lightPos");
243  mLightColorPos = glGetUniformLocation(shaderId, "lightColor");
244  mLightLinearPos = glGetUniformLocation(shaderId, "lightLinear");
245  mLightAmbientPos = glGetUniformLocation(shaderId, "lightAmbient");
246  mObjectColorPos = glGetUniformLocation(shaderId, "objectColor");
247  mObjectColorAlphaPos = glGetUniformLocation(shaderId, "objectColorAlpha");
248  mTexSlotPos = glGetUniformLocation(shaderId, "texSlot");
249  mInvertedNormalsPos = glGetUniformLocation(shaderId, "invertedNormals");
250  mSsaoActivePos = glGetUniformLocation(shaderId, "ssaoActive");
251  mAlbedoPos = glGetUniformLocation(shaderId, "ColorTex");
252  mPositionPos = glGetUniformLocation(shaderId, "PositionTex");
253  mNormalPos = glGetUniformLocation(shaderId, "NormalTex");
254  mSsaoPos = glGetUniformLocation(shaderId, "AoTex");
255  mRandTexPos = glGetUniformLocation(shaderId, "RandTex");
256  mSamplesPos = glGetUniformLocation(shaderId, "SampleKernel");
257  mCurSegmentPos = glGetUniformLocation(shaderId, "curSegment");
258  mScreenWidthPos = glGetUniformLocation(shaderId, "screenWidth");
259  mScreenHeightPos = glGetUniformLocation(shaderId, "screenHeight");
260 
261  Activate();
262  return shaderId;
263 }

480 const char* FragShaderSSAO = R"(
481  #version 330 core
482  layout (location = 0) out float AoData;
483 
484  in vec2 texCoord;
485 
486  uniform sampler2D RandTex;
487  uniform sampler2D PositionTex;
488  uniform sampler2D NormalTex;
489 
490  const int kernelSize = 64;
491  uniform vec3 SampleKernel[kernelSize];
492  uniform float Radius = 20.0f;
493  uniform float screenWidth = 800.0;
494  uniform float screenHeight = 600.0;
495 
496  uniform mat4 projection;
497 
498  void main()
499  {
500  // Create the random tangent space matrix
501  vec2 randScale = vec2( screenWidth / 4.0, screenHeight / 4.0 );
502  vec3 randDir = normalize( texture(RandTex, texCoord.xy * randScale).xyz );
503  vec3 n = normalize( texture(NormalTex, texCoord).xyz );
504  vec3 biTang = cross( n, randDir );
505  if( length(biTang) < 0.0001 ) // If n and randDir are parallel, n is in x-y plane
506  biTang = cross( n, vec3(0,0,1));
507  biTang = normalize(biTang);
508  vec3 tang = cross(biTang, n);
509  mat3 toCamSpace = mat3(tang, biTang, n);
510 
511  float occlusionSum = 0.0;
512  vec3 camPos = texture(PositionTex, texCoord).xyz;
513  for( int i = 0; i < kernelSize; i++ ) {
514  vec3 samplePos = camPos + Radius * (toCamSpace * SampleKernel[i]);
515 
516  // Project point
517  vec4 p = projection * vec4(samplePos,1);
518  p *= 1.0 / p.w;
519  p.xyz = p.xyz * 0.5 + 0.5;
520 
521  // Access camera space z-coordinate at that point
522  float surfaceZ = texture(PositionTex, p.xy).z;
523  float zDist = surfaceZ - camPos.z;
524 
525  // Count points that ARE occluded
526  if( zDist >= 0.0 && zDist <= Radius && surfaceZ > samplePos.z ) occlusionSum += 1.0;
527  }
528 
529  float occ = occlusionSum / kernelSize;
530  AoData = 1.0 - occ;
531  }
532 )";
