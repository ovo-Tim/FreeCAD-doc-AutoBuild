78 class MirrorPlaneSelection : public Gui::SelectionFilterGate
79 {
80 public:
81  explicit MirrorPlaneSelection()
82  : Gui::SelectionFilterGate()
83  {
84  }
93  bool allow(App::Document* /*pDoc*/, App::DocumentObject* pObj, const char* sSubName) override
94  {
95  std::string subString(sSubName);
96 
97  if (pObj->isDerivedFrom(Part::Plane::getClassTypeId()) || pObj->isDerivedFrom<App::Plane>()
98  || (strstr(pObj->getNameInDocument(), "Plane") && pObj->isDerivedFrom(Part::Datum::getClassTypeId()))) {
99  return true;
100  // reference is an app::link or a part::feature or some subobject
101  } else if (pObj->isDerivedFrom<Part::Feature>() || pObj->isDerivedFrom<App::Link>()) {
102  bool isFace = false; //will be true if user selected face subobject or if object only has 1 face
103  bool isEdge = false; //will be true if user selected edge subobject or if object only has 1 edge
104  TopoDS_Shape shape;
105  if (subString.length() > 0){
106  shape = Part::Feature::getTopoShape(pObj, subString.c_str(), true).getShape();
107  if (strstr(subString.c_str(), "Face")){
108  isFace = true; //was face subobject, e.g. Face3
109  } else {
110  if (strstr(subString.c_str(), "Edge")){
111  isEdge = true; //was edge subobject, e.g. Edge7
112  }
113  }
114  } else {
115  shape = Part::Feature::getShape(pObj); //no subobjects were selected, so this is entire shape of feature
116  }
117 
118  // if there is only 1 face or 1 edge, then we don't need to force the user to select that face or edge
119  // instead we can infer what was intended
120  int faceCount = Part::TopoShape(shape).countSubShapes(TopAbs_FACE);
121  int edgeCount = Part::TopoShape(shape).countSubShapes(TopAbs_EDGE);
122 
123  TopoDS_Face face;
124  TopoDS_Edge edge;
125 
126  if (isFace) { //user selected a face, so use shape to get the TopoDS::Face
127  face = TopoDS::Face(shape);
128  } else {
129  if (faceCount == 1) { //entire feature selected, but it only has 1 face, so get that face
130  TopoDS_Shape tdface = Part::TopoShape(shape).getSubShape(std::string("Face1").c_str());
131  face = TopoDS::Face(tdface);
132  isFace = true;
133  }
134  }
135  if (!isFace && isEdge){ //don't bother with edge if we already have a face to work with
136  edge = TopoDS::Edge(shape); //isEdge means an edge was selected
137  } else {
138  if (edgeCount == 1){ //we don't have a face yet and there were no edges in the subobject selection
139  //but since this object only has 1 edge, we use it
140  TopoDS_Shape tdedge = Part::TopoShape(shape).getSubShape(std::string("Edge1").c_str());
141  edge = TopoDS::Edge(tdedge);
142  isEdge = true;
143  }
144  }
145 
146  if (isFace && face.IsNull()) { //ensure we have a good face to work with
147  return false;
148  }
149  if (isEdge && edge.IsNull()){ //ensure we have a good edge to work with
150  return false;
151  }
152  if (!isFace && !isEdge){
153  return false;
154  }
155 
156  if (isFace) {
157  BRepAdaptor_Surface adapt(face);
158  if (adapt.GetType() != GeomAbs_Plane){
159  return false;
160  }
161  return true;
162  } else {
163  if (isEdge){
164  BRepAdaptor_Curve curve(edge);
165  if (!(curve.GetType() == GeomAbs_Circle)) {
166  return false;
167  }
168  return true;
169  }
170  }
171  } //end of if(derived from part::feature)
172  return true;
173  }//end of allow()
174

242 void Mirroring::findShapes()
243 {
244  App::Document* activeDoc = App::GetApplication().getActiveDocument();
245  if (!activeDoc)
246  return;
247  Gui::Document* activeGui = Gui::Application::Instance->getDocument(activeDoc);
248  if (!activeGui)
249  return;
250 
251  this->document = QString::fromLatin1(activeDoc->getName());
252  std::vector<App::DocumentObject*> objs = activeDoc->getObjectsOfType<App::DocumentObject>();
253 
254  for (auto obj : objs) {
255  Part::TopoShape shape = Part::Feature::getTopoShape(obj);
256  if (!shape.isNull()) {
257  QString label = QString::fromUtf8(obj->Label.getValue());
258  QString name = QString::fromLatin1(obj->getNameInDocument());
259 
260  QTreeWidgetItem* child = new QTreeWidgetItem();
261  child->setText(0, label);
262  child->setToolTip(0, label);
263  child->setData(0, Qt::UserRole, name);
264  Gui::ViewProvider* vp = activeGui->getViewProvider(obj);
265  if (vp) child->setIcon(0, vp->getIcon());
266  ui->shapes->addTopLevelItem(child);
267  }
268  }

277 bool Mirroring::accept()
278 {
279  if (ui->shapes->selectedItems().isEmpty()) {
280  QMessageBox::critical(this, windowTitle(),
281  tr("Select a shape for mirroring, first."));
282  return false;
283  }
284 
285  App::Document* activeDoc = App::GetApplication().getDocument((const char*)this->document.toLatin1());
286  if (!activeDoc) {
287  QMessageBox::critical(this, windowTitle(),
288  tr("No such document '%1'.").arg(this->document));
289  return false;
290  }
291 
292  Gui::WaitCursor wc;
293  unsigned int count = activeDoc->countObjectsOfType<Part::Mirroring>();
294  activeDoc->openTransaction("Mirroring");
295 
296  QString shape, label, selectionString;
297  QRegularExpression rx(QString::fromLatin1(R"( \(Mirror #\d+\)$)"));
298  QList<QTreeWidgetItem *> items = ui->shapes->selectedItems();
299  float normx=0, normy=0, normz=0;
300  int index = ui->comboBox->currentIndex();
301  std::string selection(""); //set MirrorPlane property to empty string unless
302  //user has selected Use selected reference in combobox
303 
304  if (index == 0){
305  normz = 1.0f;
306  } else if (index == 1){
307  normy = 1.0f;
308  } else if (index == 2){
309  normx = 1.0f;
310  } else if (index == 3){ //use selected reference
311  std::vector<Gui::SelectionObject> selobjs = Gui::Selection().getSelectionEx();
312  if (selobjs.size() == 1) {
313  selection = selobjs[0].getAsPropertyLinkSubString();
314  }
315  }
316  double basex = ui->baseX->value().getValue();
317  double basey = ui->baseY->value().getValue();
318  double basez = ui->baseZ->value().getValue();
319  for (auto item : items) {
320  shape = item->data(0, Qt::UserRole).toString();
321  std::string escapedstr = Base::Tools::escapedUnicodeFromUtf8(item->text(0).toUtf8());
322  label = QString::fromStdString(escapedstr);
323  selectionString = QString::fromStdString(selection);
324 
325  // if we already have the suffix " (Mirror #<number>)" remove it
326  int pos = label.indexOf(rx);
327  if (pos > -1)
328  label = label.left(pos);
329  label.append(QString::fromLatin1(" (Mirror #%1)").arg(++count));
330 
331  QString code = QString::fromLatin1(
332  "__doc__=FreeCAD.getDocument(\"%1\")\n"
333  "__doc__.addObject(\"Part::Mirroring\")\n"
334  "__doc__.ActiveObject.Source=__doc__.getObject(\"%2\")\n"
335  "__doc__.ActiveObject.Label=u\"%3\"\n"
336  "__doc__.ActiveObject.Normal=(%4,%5,%6)\n"
337  "__doc__.ActiveObject.Base=(%7,%8,%9)\n"
338  "__doc__.ActiveObject.MirrorPlane=(%10)\n"
339  "del __doc__")
340  .arg(this->document, shape, label)
341  .arg(normx).arg(normy).arg(normz)
342  .arg(basex).arg(basey).arg(basez)
343  .arg(selectionString);
344  Gui::Command::runCommand(Gui::Command::App, code.toLatin1());
345  QByteArray from = shape.toLatin1();
346  Gui::Command::copyVisual("ActiveObject", "ShapeAppearance", from);
347  Gui::Command::copyVisual("ActiveObject", "LineColor", from);
348  Gui::Command::copyVisual("ActiveObject", "PointColor", from);
349  }
350 
351  activeDoc->commitTransaction();
352  activeDoc->recompute();
353  Gui::Selection().rmvSelectionGate();
354  return true;
355 }
