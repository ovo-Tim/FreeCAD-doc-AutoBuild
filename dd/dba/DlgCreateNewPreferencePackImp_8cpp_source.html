64 void DlgCreateNewPreferencePackImp::setPreferencePackTemplates(const std::vector<Gui::PreferencePackManager::TemplateFile>& availableTemplates)
65 {
66  ui->treeWidget->clear();
67  _groups.clear();
68 
69  ui->treeWidget->header()->setDefaultSectionSize(250);
70 
71  _templates = availableTemplates;
72  for (const auto &t : _templates) {
73 
74  QTreeWidgetItem* group;
75  if (auto foundGroup = _groups.find(t.group); foundGroup != _groups.end()) {
76  group = foundGroup->second;
77  }
78  else {
79  group = new QTreeWidgetItem(ui->treeWidget, QStringList(QString::fromStdString(t.group)));
80  group->setCheckState(0, Qt::Checked);
81  group->setExpanded(true);
82  _groups.insert(std::make_pair(t.group, group));
83  }
84 
85  QStringList itemColumns;
86  itemColumns.push_back(QString::fromStdString(t.name));
87  auto newItem = new QTreeWidgetItem(group, itemColumns);
88  newItem->setCheckState(0, Qt::Checked);
89  if (group->checkState(0) != newItem->checkState(0))
90  group->setCheckState(0, Qt::PartiallyChecked);
91  newItem->setData(0, TemplateRole, QVariant::fromValue(t));
92  group->addChild(newItem);
93  }
94 }

102 {

124 void DlgCreateNewPreferencePackImp::onItemChanged(QTreeWidgetItem* item, int column)
125 {
126  Q_UNUSED(column);
127  const QSignalBlocker blocker(ui->treeWidget);
128  if (auto group = item->parent(); group) {
129  // Child clicked
130  bool firstItemChecked = false;
131  for (int childIndex = 0; childIndex < group->childCount(); ++childIndex) {
132  auto child = group->child(childIndex);
133  if (childIndex == 0) {
134  firstItemChecked = child->checkState(0) == Qt::Checked;
135  }
136  else {
137  bool thisItemChecked = child->checkState(0) == Qt::Checked;
138  if (firstItemChecked != thisItemChecked) {
139  group->setCheckState(0, Qt::PartiallyChecked);
140  return;
141  }
142  }
143  }
144  group->setCheckState(0, firstItemChecked ? Qt::Checked : Qt::Unchecked);
145  }
146  else {
147  // Group clicked:
148  auto groupCheckState = item->checkState(0);
149  for (int childIndex = 0; childIndex < item->childCount(); ++childIndex) {
150  auto child = item->child(childIndex);
151  child->setCheckState(0, groupCheckState);
152  }
153  }
154 }
