40 def check_member_for_solver_calculix(analysis, solver, mesh, member):
41 
42  message = ""
43 
44  # solver
45  if solver.AnalysisType not in ANALYSIS_TYPES:
46  message += f"Unknown analysis type: {solver.AnalysisType}\n"
47  if solver.AnalysisType == "frequency":
48  if not hasattr(solver, "EigenmodeHighLimit"):
49  message += "Frequency analysis: Solver has no EigenmodeHighLimit.\n"
50  elif not hasattr(solver, "EigenmodeLowLimit"):
51  message += "Frequency analysis: Solver has no EigenmodeLowLimit.\n"
52  elif not hasattr(solver, "EigenmodesCount"):
53  message += "Frequency analysis: Solver has no EigenmodesCount.\n"
54  if hasattr(solver, "MaterialNonlinearity") and solver.MaterialNonlinearity == "nonlinear":
55  if not member.mats_nonlinear:
56  message += (
57  "Solver is set to nonlinear materials, "
58  "but there is no nonlinear material in the analysis.\n"
59  )
60 
61  # mesh
62  if not mesh:
63  message += "A single mesh object must be defined in the analysis.\n"
64  if mesh:
65  if (
66  mesh.FemMesh.VolumeCount == 0
67  and mesh.FemMesh.FaceCount > 0
68  and not member.geos_shellthickness
69  ):
70  message += (
71  "FEM mesh has no volume elements, "
72  "either define shell thicknesses or "
73  "provide a FEM mesh with volume elements.\n"
74  )
75  if (
76  mesh.FemMesh.VolumeCount == 0
77  and mesh.FemMesh.FaceCount == 0
78  and mesh.FemMesh.EdgeCount > 0
79  and not member.geos_beamsection
80  and not member.geos_fluidsection
81  ):
82  message += (
83  "FEM mesh has no volume and no shell elements, "
84  "either define a beam/fluid section or provide "
85  "a FEM mesh with volume elements.\n"
86  )
87  if (
88  mesh.FemMesh.VolumeCount == 0
89  and mesh.FemMesh.FaceCount == 0
90  and mesh.FemMesh.EdgeCount == 0
91  ):
92  message += (
93  "FEM mesh has neither volume nor shell or edge elements. "
94  "Provide a FEM mesh with elements.\n"
95  )
96 
97  # material linear and nonlinear
98  if not member.mats_linear:
99  message += "No material object defined in the analysis.\n"
100  has_no_references = False
101  for m in member.mats_linear:
102  if len(m["Object"].References) == 0:
103  if has_no_references is True:
104  message += (
105  "More than one material has an empty references list "
106  "(Only one empty references list is allowed!).\n"
107  )
108  has_no_references = True
109  mat_ref_shty = ""
110  for m in member.mats_linear:
111  ref_shty = femutils.get_refshape_type(m["Object"])
112  if ref_shty == "Compound":
113  ref_shty = "Solid"
114  if not mat_ref_shty:
115  mat_ref_shty = ref_shty
116  if mat_ref_shty and ref_shty and ref_shty != mat_ref_shty:
117  # mat_ref_shty could be empty in one material
118  # only the not empty ones should have the same shape type
119  message += (
120  "Some material objects do not have the same reference shape type "
121  "(all material objects must have the same reference shape type, "
122  "at the moment).\n"
123  )
124  for m in member.mats_linear:
125  mat_map = m["Object"].Material
126  mat_obj = m["Object"]
127  if mat_obj.Category == "Solid":
128  if "YoungsModulus" in mat_map:
129  # print(Units.Quantity(mat_map["YoungsModulus"]).Value)
130  if not Units.Quantity(mat_map["YoungsModulus"]).Value:
131  message += "Value of YoungsModulus is set to 0.0.\n"
132  else:
133  message += "No YoungsModulus defined for at least one material.\n"
134  if "PoissonRatio" not in mat_map:
135  # PoissonRatio is allowed to be 0.0 (in ccx), but it should be set anyway.
136  message += "No PoissonRatio defined for at least one material.\n"
137  if solver.AnalysisType == "frequency" or member.cons_selfweight:
138  if "Density" not in mat_map:
139  message += "No Density defined for at least one material.\n"
140  if solver.AnalysisType == "thermomech":
141  if "ThermalConductivity" in mat_map:
142  if not Units.Quantity(mat_map["ThermalConductivity"]).Value:
143  message += "Value of ThermalConductivity is set to 0.0.\n"
144  else:
145  message += (
146  "Thermomechanical analysis: No ThermalConductivity defined "
147  "for at least one material.\n"
148  )
149  if "ThermalExpansionCoefficient" not in mat_map and mat_obj.Category == "Solid":
150  message += (
151  "Thermomechanical analysis: No ThermalExpansionCoefficient defined "
152  "for at least one material.\n" # allowed to be 0.0 (in ccx)
153  )
154  if "SpecificHeat" not in mat_map:
155  message += (
156  "Thermomechanical analysis: No SpecificHeat "
157  "defined for at least one material.\n" # allowed to be 0.0 (in ccx)
158  )
159  if femutils.is_of_type(mat_obj, "Fem::MaterialReinforced"):
160  # additional tests for reinforced materials,
161  # they are needed for result calculation, not for ccx analysis
162  mat_map_m = mat_obj.Material
163  if "AngleOfFriction" in mat_map_m:
164  # print(Units.Quantity(mat_map_m["AngleOfFriction"]).Value)
165  if not Units.Quantity(mat_map_m["AngleOfFriction"]).Value:
166  message += (
167  "Value of AngleOfFriction is set to 0.0 "
168  "for the matrix of a reinforced material.\n"
169  )
170  else:
171  message += (
172  "No AngleOfFriction defined for the matrix "
173  "of at least one reinforced material.\n"
174  )
175  if "CompressiveStrength" in mat_map_m:
176  # print(Units.Quantity(mat_map_m["CompressiveStrength"]).Value)
177  if not Units.Quantity(mat_map_m["CompressiveStrength"]).Value:
178  message += (
179  "Value of CompressiveStrength is set to 0.0 "
180  "for the matrix of a reinforced material.\n"
181  )
182  else:
183  message += (
184  "No CompressiveStrength defined for the matrix "
185  "of at least one reinforced material.\n"
186  )
187  mat_map_r = mat_obj.Reinforcement
188  if "YieldStrength" in mat_map_r:
189  # print(Units.Quantity(mat_map_r["YieldStrength"]).Value)
190  if not Units.Quantity(mat_map_r["YieldStrength"]).Value:
191  message += (
192  "Value of YieldStrength is set to 0.0 "
193  "for the reinforcement of a reinforced material.\n"
194  )
195  else:
196  message += (
197  "No YieldStrength defined for the reinforcement "
198  "of at least one reinforced material.\n"
199  )
200  if len(member.mats_linear) == 1:
201  mobj = member.mats_linear[0]["Object"]
202  if hasattr(mobj, "References") and mobj.References:
203  FreeCAD.Console.PrintError(
204  "Only one material object, but this one has a reference shape. "
205  "The reference shape will be ignored.\n"
206  )
207  for m in member.mats_linear:
208  has_nonlinear_material = False
209  for nlm in member.mats_nonlinear:
210  if nlm["Object"].LinearBaseMaterial == m["Object"]:
211  if has_nonlinear_material is False:
212  has_nonlinear_material = True
213  else:
214  message += (
215  "At least two nonlinear materials use the same linear base material. "
216  "Only one nonlinear material for each linear material allowed.\n"
217  )
218 
219  # which analysis needs which constraints
220  # no check in the regard of loads existence (constraint force, pressure, self weight)
221  # is done, because an analysis without loads at all is a valid analysis too
222  if solver.AnalysisType == "static":
223  if not (member.cons_fixed or member.cons_displacement or member.cons_rigidbody):
224  message += "Static analysis: No mechanical boundary conditions defined.\n"
225  if solver.AnalysisType == "thermomech":
226  if not member.cons_initialtemperature:
227  if not member.geos_fluidsection:
228  message += "Thermomechanical analysis: No initial temperature defined.\n"
229  if len(member.cons_initialtemperature) > 1:
230  message += "Thermomechanical analysis: Only one initial temperature is allowed.\n"
231 
232  # constraints
233  # fixed
234  if member.cons_fixed:
235  for c in member.cons_fixed:
236  if len(c["Object"].References) == 0:
237  message += "{} has empty references.".format(c["Object"].Name)
238  # displacement
239  if member.cons_displacement:
240  for di in member.cons_displacement:
241  if len(di["Object"].References) == 0:
242  message += "{} has empty references.".format(c["Object"].Name)
243  # plane rotation
244  if member.cons_planerotation:
245  for c in member.cons_planerotation:
246  if len(c["Object"].References) == 0:
247  message += "{} has empty references.".format(c["Object"].Name)
248  # contact
249  if member.cons_contact:
250  for c in member.cons_contact:
251  if len(c["Object"].References) == 0:
252  message += "{} has empty references.".format(c["Object"].Name)
253  # tie
254  if member.cons_tie:
255  for c in member.cons_tie:
256  items = 0
257  for reference in c["Object"].References:
258  items += len(reference[1])
259  if items != 2:
260  message += "{} doesn't reference exactly two needed faces.\n".format(
261  c["Object"].Name
262  )
263  # sectionprint
264  if member.cons_sectionprint:
265  for c in member.cons_sectionprint:
266  items = 0
267  for reference in c["Object"].References:
268  items += len(reference[1])
269  if items != 1:
270  message += "{} doesn't reference exactly one needed face.\n".format(
271  c["Object"].Name
272  )
273  # transform
274  if member.cons_transform:
275  for c in member.cons_transform:
276  if len(c["Object"].References) == 0:
277  message += "{} has empty references.".format(c["Object"].Name)
278  # pressure
279  if member.cons_pressure:
280  for c in member.cons_pressure:
281  if len(c["Object"].References) == 0:
282  message += "{} has empty references.".format(c["Object"].Name)
283  # force
284  if member.cons_force:
285  for c in member.cons_force:
286  if len(c["Object"].References) == 0:
287  message += "{} has empty references.".format(c["Object"].Name)
288  # temperature
289  if member.cons_temperature:
290  for c in member.cons_temperature:
291  if len(c["Object"].References) == 0:
292  message += "{} has empty references.".format(c["Object"].Name)
293  # heat flux
294  if member.cons_heatflux:
295  for c in member.cons_heatflux:
296  if len(c["Object"].References) == 0:
297  message += "{} has empty references.".format(c["Object"].Name)
298 
299  # geometries
300  # beam section
301  if member.geos_beamsection:
302  if member.geos_shellthickness:
303  # this needs to be checked only once either here or in shell_thicknesses
304  message += (
305  "Beam sections and shell thicknesses in one analysis "
306  "are not supported at the moment.\n"
307  )
308  if member.geos_fluidsection:
309  # this needs to be checked only once either here or in shell_thicknesses
310  message += (
311  "Beam sections and fluid sections in one analysis "
312  "are not supported at the moment.\n"
313  )
314  has_no_references = False
315  for b in member.geos_beamsection:
316  if len(b["Object"].References) == 0:
317  if has_no_references is True:
318  message += (
319  "More than one beam section has an empty references "
320  "list (Only one empty references list is allowed!).\n"
321  )
322  has_no_references = True
323  if mesh:
324  if mesh.FemMesh.FaceCount > 0 or mesh.FemMesh.VolumeCount > 0:
325  message += "Beam sections defined but FEM mesh has volume or shell elements.\n"
326  if mesh.FemMesh.EdgeCount == 0:
327  message += "Beam sections defined but FEM mesh has no edge elements.\n"
328  if not (hasattr(mesh, "Shape") or hasattr(mesh, "Part")):
329  message += (
330  "Mesh without geometry link. "
331  "The mesh needs to know its geometry for the beam rotations.\n"
332  )
333  if len(member.geos_beamrotation) > 1:
334  message += "Multiple beam rotations in one analysis are not supported at the moment.\n"
335  # beam rotations
336  if member.geos_beamrotation and not member.geos_beamsection:
337  message += "Beam rotations in the analysis but no beam sections defined.\n"
338  # shell thickness
339  if member.geos_shellthickness:
340  has_no_references = False
341  for s in member.geos_shellthickness:
342  if len(s["Object"].References) == 0:
343  if has_no_references is True:
344  message += (
345  "More than one shell thickness has an empty references "
346  "list (Only one empty references list is allowed!).\n"
347  )
348  has_no_references = True
349  if mesh:
350  if mesh.FemMesh.VolumeCount > 0:
351  message += "Shell thicknesses defined but FEM mesh has volume elements.\n"
352  if mesh.FemMesh.FaceCount == 0:
353  message += "Shell thicknesses defined but FEM mesh has no shell elements.\n"
354  # fluid section
355  if member.geos_fluidsection:
356  if not member.cons_selfweight:
357  message += "A fluid network analysis requires self weight constraint to be applied\n"
358  if solver.AnalysisType != "thermomech":
359  message += "A fluid network analysis can only be done in a thermomech analysis\n"
360  has_no_references = False
361  for f in member.geos_fluidsection:
362  if len(f["Object"].References) == 0:
363  if has_no_references is True:
364  message += (
365  "More than one fluid section has an empty references list "
366  "(Only one empty references list is allowed!).\n"
367  )
368  has_no_references = True
369  if mesh:
370  if mesh.FemMesh.FaceCount > 0 or mesh.FemMesh.VolumeCount > 0:
371  message += "Fluid sections defined but FEM mesh has volume or shell elements.\n"
372  if mesh.FemMesh.EdgeCount == 0:
373  message += "Fluid sections defined but FEM mesh has no edge elements.\n"
374 
375  return message
