237 void QGILeaderLine::onLineEditFinished(QPointF tipDisplace, std::vector<QPointF> scenePoints)
238 {
239  // Base::Console().Message("QGILL::onLineEditFinished(%s, %d)\n",
240  // TechDraw::DrawUtil::formatVector(tipDisplace).c_str(),
241  // scenePoints.size());
242 
243  m_blockDraw = true;
244  auto featLeader = getLeaderFeature();
245  if (!featLeader) {
246  return;
247  }
248  double baseScale = featLeader->getBaseScale();
249 
250  if (!(TechDraw::DrawUtil::fpCompare(tipDisplace.x(), 0.0)
251  && TechDraw::DrawUtil::fpCompare(tipDisplace.y(), 0.0))) {
252  //tip was moved. need to change AttachPoint
253  QPointF oldAttach = getAttachFromFeature();
254  QPointF newAttach = oldAttach + (tipDisplace / baseScale);
255  featLeader->setPosition(Rez::appX(newAttach.x()), Rez::appX(-newAttach.y()), true);
256  }
257 
258  std::vector<Base::Vector3d> pageDeltas;
259  for (auto& pt : scenePoints) {
260  QPointF distFromP0 = pt - scenePoints.front();
261  // convert deltas to mm
262  Base::Vector3d deltaInPageCoords = Rez::appX(DU::toVector3d(distFromP0));
263  pageDeltas.push_back(deltaInPageCoords);
264  }
265  pageDeltas.at(0) = Base::Vector3d(0.0, 0.0, 0.0);
266 
267  // to canonical form here
268  auto temp = featLeader->makeCanonicalPointsInverted(pageDeltas);
269  featLeader->WayPoints.setValues(temp);
270 
271  Q_EMIT editComplete();//tell task editing is complete
272 
273  m_blockDraw = false;
274  m_editPath->hide();
275  draw();
276 }

342 void QGILeaderLine::draw()
343 {
344  if (m_blockDraw) {
345  return;
346  }
347  if (!isVisible()) {
348  return;
349  }
350  TechDraw::DrawLeaderLine* featLeader = getLeaderFeature();
351  if (!featLeader) {
352  return;
353  }
354  auto vp = static_cast<ViewProviderLeader*>(getViewProvider(getViewObject()));
355  if (!vp) {
356  return;
357  }
358 
359  TechDraw::DrawView* parent = featLeader->getBaseView();
360 
361  if (!parent) {
362  return;
363  }
364 
365  if (m_editPath->inEdit()) {
366  return;
367  }
368 
369  //********
370  if (featLeader->isLocked()) {
371  setFlag(QGraphicsItem::ItemIsMovable, false);
372  }
373  else {
374  setFlag(QGraphicsItem::ItemIsMovable, true);
375  }
376 
377  // set the leader's Qt position from feature's X,Y and scale.
378  // the feature's x,y is unscaled, unrotated and conventional Y
379  QPointF qPoint = DU::toQPointF(getAttachPoint());
380  // ???? why does the attach point not need Rez applied?
381  setPos(qPoint);
382 
383  // line style is standing in for line number here?
384  m_lineStyle = static_cast<Qt::PenStyle>(vp->LineStyle.getValue());
385 
386  m_line->setFillStyle(Qt::NoBrush);
387  m_line->setStyle(m_lineStyle);
388  m_line->setWidth(getLineWidth());
389 
390  m_line->setPos(0, 0); //make m_line coords == leader coords
391 
392  std::vector<QPointF> qPoints = getWayPointsFromFeature();
393  if (qPoints.empty() ) {
394  Base::Console().Message("QGILL::draw - no points\n");
395  return;
396  }
397 
398  setNormalColorAll();
399  m_line->setPath(makeLeaderPath(qPoints));
400  setArrows(qPoints);
401 
402  if (isSelected()) {
403  setPrettySel();
404  }
405  else if (m_hasHover) {
406  setPrettyPre();
407  }
408  else {
409  setPrettyNormal();
410  }
411  update(boundingRect());
412 }

414 QPainterPath QGILeaderLine::makeLeaderPath(std::vector<QPointF> qPoints)
415 {
416  QPainterPath result;
417  DrawLeaderLine* featLeader = getLeaderFeature();
418  if (!featLeader) {
419  // Base::Console().Message("QGILL::makeLeaderPath - featLeader is nullptr\n");
420  return result;
421  }
422 
423  QPointF startAdjVec(0.0, 0.0);
424  double startAdjLength(0.0);
425  QPointF endAdjVec(0.0, 0.0);
426  double endAdjLength(0.0);
427  if (qPoints.size() > 1) {
428  //make path adjustment to hide leaderline ends behind arrowheads
429  if (featLeader->StartSymbol.getValue() != ArrowType::NONE) {
430  startAdjLength = QGIArrow::getOverlapAdjust(featLeader->StartSymbol.getValue(),
431  QGIArrow::getPrefArrowSize());
432  }
433  if (featLeader->EndSymbol.getValue() != ArrowType::NONE) {
434  endAdjLength = QGIArrow::getOverlapAdjust(featLeader->EndSymbol.getValue(),
435  QGIArrow::getPrefArrowSize());
436  }
437 
438  //get adjustment directions
439  startAdjVec = qPoints.at(1) - qPoints.front();
440  endAdjVec = (*(qPoints.end() - 2)) - qPoints.back();
441 
442  //get adjustment vectors
443  QVector2D startTemp(startAdjVec);
444  QVector2D endTemp(endAdjVec);
445  startTemp.normalize();
446  endTemp.normalize();
447  startAdjVec = startTemp.toPointF() * startAdjLength;
448  endAdjVec = endTemp.toPointF() * endAdjLength;
449 
450  qPoints.front() += startAdjVec;
451  qPoints.back() += endAdjVec;
452  result.moveTo(qPoints.front());
453  for (int i = 1; i < (int)qPoints.size(); i++) {
454  result.lineTo(qPoints.at(i));
455  }
456  }
457  return result;
458 }

476 std::vector<QPointF> QGILeaderLine::getWayPointsFromFeature()
477 {
478  DrawLeaderLine* featLeader = getLeaderFeature();
479  if (!featLeader) {
480  // Base::Console().Message("QGILL::getWayPointsFromFeature - featLeader is nullptr\n");
481  return {};
482  }
483 
484  // vPoints are in mm with conventional Y axis
485  auto doScale = featLeader->Scalable.getValue();
486  auto doRotate = featLeader->RotatesWithParent.getValue();
487  auto vPoints = featLeader->getScaledAndRotatedPoints(doScale, doRotate);
488  if (featLeader->AutoHorizontal.getValue()) {
489  vPoints = DrawLeaderLine::horizLastSegment(vPoints, featLeader->getBaseView()->Rotation.getValue());
490  }
491 
492  std::vector<QPointF> qPoints;
493  qPoints.reserve(vPoints.size());
494  // now convert to sceneUnits and Qt Y axis
495  for (auto& entry : vPoints) {
496  if (useOldCoords()) {
497  // use points as saved in <= v0.21 (in Qt form, Rez'd and
498  qPoints.push_back(DU::toQPointF(entry));
499  } else {
500  // use points as saved in >= v0.22
501  qPoints.push_back(DU::toQPointF(Rez::guiX(entry)));
502  }
503  }
504 
505  if (qPoints.empty()) {
506  Base::Console().Warning("QGILeaderLine::getWayPointsFromFeature - no points\n");
507  }
508  return qPoints;
509 }

511 void QGILeaderLine::setArrows(std::vector<QPointF> pathPoints)
512 {
513  Base::Vector3d stdX(1.0, 0.0, 0.0);
514  TechDraw::DrawLeaderLine* featLeader = getLeaderFeature();
515 
516  QPointF lastOffset = (pathPoints.back() - pathPoints.front());
517 
518  if (featLeader->StartSymbol.getValue() != ArrowType::NONE) {
519  m_arrow1->setStyle(featLeader->StartSymbol.getValue());
520  m_arrow1->setWidth(getLineWidth());
521  m_arrow1->setSize(QGIArrow::getPrefArrowSize());
522  m_arrow1->setDirMode(true);
523  m_arrow1->setDirection(stdX);
524  if (pathPoints.size() > 1) {
525  auto it = pathPoints.begin();
526  QPointF s = (*it);
527  QPointF e = (*(it + 1));
528  QPointF qsVec = s - e;
529  Base::Vector3d sVec(qsVec.x(), qsVec.y(), 0.0);
530  m_arrow1->setDirection(sVec);
531  m_arrow1->setPos(0.0, 0.0);
532  }
533  m_arrow1->draw();
534  m_arrow1->show();
535  }
536  else {
537  m_arrow1->hide();
538  }
539 
540  if (featLeader->EndSymbol.getValue() != ArrowType::NONE) {
541  m_arrow2->setStyle(featLeader->EndSymbol.getValue());
542  m_arrow2->setWidth(getLineWidth());
543  m_arrow2->setSize(QGIArrow::getPrefArrowSize());
544  m_arrow2->setDirMode(true);
545  m_arrow2->setDirection(-stdX);
546  if (pathPoints.size() > 1) {
547  auto itr = pathPoints.rbegin();
548  QPointF s = (*itr);
549  QPointF e = (*(itr + 1));
550  QPointF qeVec = s - e;
551  Base::Vector3d eVec(qeVec.x(), qeVec.y(), 0.0);
552  m_arrow2->setDirection(eVec);
553  m_arrow2->setPos(lastOffset);
554  }
555  m_arrow2->draw();
556  m_arrow2->show();
557  }
558  else {
559  m_arrow2->hide();
560  }

571 Base::Vector3d QGILeaderLine::getAttachPoint()
572 {
573  TechDraw::DrawLeaderLine* featLeader = getLeaderFeature();
574  if (!featLeader) {
575  return Base::Vector3d(0, 0, 0);
576  }
577  TechDraw::DrawView* parent = featLeader->getBaseView();
578 
579  if (!parent) {
580  return Base::Vector3d(0, 0, 0);
581  }
582 
583  double baseScale = featLeader->getBaseScale();
584  double xPos = Rez::guiX(featLeader->X.getValue());
585  double yPos = Rez::guiX(featLeader->Y.getValue());
586  Base::Vector3d vAttachPoint{xPos, yPos};
587  vAttachPoint = vAttachPoint * baseScale;
588  double rotationRad = parent->Rotation.getValue() * M_PI / DegreesHalfCircle;
589  if (rotationRad != 0.0) {
590  vAttachPoint.RotateZ(rotationRad);
591  }
592  vAttachPoint = DU::invertY(vAttachPoint);
593  return vAttachPoint;
594 }
