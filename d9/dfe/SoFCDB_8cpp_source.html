274 SoNode* replaceSwitches(SoNodeList* children, SoGroup* parent)
275 {
276  if (!children) {
277  return parent;
278  }
279 
280  for (int i=0; i<children->getLength(); i++) {
281  SoNode* node = (*children)[i];
282  if (node->getTypeId().isDerivedFrom(SoGroup::getClassTypeId())) {
283  if (node->getTypeId().isDerivedFrom(SoSwitch::getClassTypeId())) {
284  auto group = static_cast<SoSwitch*>(node);
285  int which = group->whichChild.getValue();
286  if (which == SO_SWITCH_NONE)
287  continue;
288  auto newParent = new SoGroup();
289  SoNodeList c;
290  if (which >= 0) {
291  c.append(group->getChild(which));
292  }
293  else {
294  // SO_SWITCH_INHERIT or SO_SWITCH_ALL
295  for (int i=0; i<group->getNumChildren(); i++)
296  c.append(group->getChild(i));
297  }
298 
299  replaceSwitches(&c, newParent);
300  parent->addChild(newParent);
301  }
302  else {
303  auto newParent = static_cast<SoGroup*>(node->getTypeId().createInstance());
304  replaceSwitches(node->getChildren(), newParent);
305  parent->addChild(newParent);
306  }
307  }
308  else {
309  parent->addChild(node);
310  }
311  }
312 
313  return parent;
314 }

352 bool Gui::SoFCDB::writeToVRML(SoNode* node, const char* filename, bool binary)
353 {
354  std::string buffer;
355  writeToVRML(node, buffer);
356 
357  Base::FileInfo fi(filename);
358  if (binary) {
359  // We want to write compressed VRML but Coin 2.4.3 doesn't do it even though
360  // SoOutput::getAvailableCompressionMethods() delivers a string list that
361  // contains 'GZIP'. setCompression() was called directly after opening the file,
362  // returned true and no error message appeared but anyway it didn't work.
363  // Strange is that reading GZIPped VRML files works.
364  // So, we do the compression on our own.
365  Base::ofstream str(fi, std::ios::out | std::ios::binary);
366  zipios::GZIPOutputStream gzip(str);
367 
368  if (gzip) {
369  gzip << buffer;
370  gzip.close();
371  return true;
372  }
373  }
374  else {
375  Base::ofstream str(fi, std::ios::out);
376 
377  if (str) {
378  str << buffer;
379  str.close();
380  return true;
381  }
382  }
383 
384  return false;
385 }

416 bool Gui::SoFCDB::writeToX3D(SoNode* node, bool exportViewpoints, std::string& buffer)
417 {
418  SoNode* noSwitches = replaceSwitchesInSceneGraph(node);
419  noSwitches->ref();
420  SoVRMLAction vrml2;
421  vrml2.setOverrideMode(true);
422  vrml2.apply(noSwitches);
423  SoToVRML2Action tovrml2;
424  tovrml2.apply(noSwitches);
425  SoVRMLGroup* vrmlRoot = tovrml2.getVRML2SceneGraph();
426 
427  vrmlRoot->setInstancePrefix(SbString("o"));
428  vrmlRoot->ref();
429 
430  // Search for SoVRMLIndexedFaceSet nodes and set creaseAngle to 0.5
431  {
432  SoSearchAction sa;
433  sa.setType(SoVRMLShape::getClassTypeId());
434  sa.setInterest(SoSearchAction::ALL);
435  sa.setSearchingAll(true);
436  sa.apply(vrmlRoot);
437  SoPathList& paths = sa.getPaths();
438  for (int i=0; i<paths.getLength(); i++) {
439  SoPath* path = paths[i];
440  auto shape = static_cast<SoVRMLShape*>(path->getTail());
441  SoNode* geom = shape->geometry.getValue();
442  if (geom && geom->getTypeId() == SoVRMLIndexedFaceSet::getClassTypeId()) {
443  SoNode* norm = static_cast<SoVRMLIndexedFaceSet*>(geom)->normal.getValue();
444  if (norm && norm->getTypeId() == SoVRMLNormal::getClassTypeId()) {
445  // if empty then nullify the normal field node
446  if (static_cast<SoVRMLNormal*>(norm)->vector.getNum() == 0)
447  static_cast<SoVRMLIndexedFaceSet*>(geom)->normal.setValue(nullptr);
448  }
449  else {
450  static_cast<SoVRMLIndexedFaceSet*>(geom)->creaseAngle.setValue(0.5f);
451  }
452  }
453  }
454  sa.reset(); // clear the internal cache
455  }
456 
457  std::stringstream out;
458  writeX3D(vrmlRoot, exportViewpoints, out);
459  buffer = out.str();
460 
461  vrmlRoot->unref(); // release the memory as soon as possible
462 
463  // restore old settings
464  vrml2.setOverrideMode(false);
465  vrml2.apply(noSwitches);
466  noSwitches->unref();
467 
468  return true;
469 }

471 void Gui::SoFCDB::writeX3DFields(SoNode* node, std::map<SoNode*, std::string>& nodeMap,
472  bool isRoot, int& numDEF, int spaces, std::ostream& out)
473 {
474  // remove the VRML prefix from the type name
475  std::string type(node->getTypeId().getName().getString());
476  type = type.substr(4);
477 
478  out << Base::blanks(spaces) << "<" << type;
479  if (node->getRefCount() > 1 && !isRoot) {
480  SbName name = node->getName();
481  std::stringstream str;
482  if (name.getLength() == 0)
483  str << "o" << numDEF++;
484  else
485  str << name.getString();
486 
487  nodeMap[node] = str.str();
488  out << " DEF=\"" << str.str() << "\"";
489  }
490 
491  const SoFieldData* fielddata = node->getFieldData();
492  if (fielddata) {
493  int numFieldNodes = 0;
494 
495  // process non-SoSFNode and non-SoMFNode fields
496  for (int i=0; i<fielddata->getNumFields(); i++) {
497  SoField* field = fielddata->getField(node, i);
498  if (!field->isDefault()) {
499  if (!field->isOfType(SoSFNode::getClassTypeId()) &&
500  !field->isOfType(SoMFNode::getClassTypeId())) {
501  SbString value;
502  field->get(value);
503  QByteArray ba(value.getString(), value.getLength());
504  ba.replace('\n', ' ');
505  if (field->isOfType(SoMField::getClassTypeId())) {
506  ba.replace('[', ' ');
507  ba.replace(']', ' ');
508  ba = ba.simplified();
509  }
510 
511  out << '\n' << Base::blanks(spaces+2) << fielddata->getFieldName(i).getString() << "=\"" << ba.data() << "\" ";
512  }
513  else {
514  numFieldNodes++;
515  }
516  }
517  }
518 
519  if (numFieldNodes > 0) {
520  out << ">\n";
521  }
522  else {
523  out << "/>\n";
524  }
525 
526  // process SoSFNode or SoMFNode fields
527  for (int i=0; i<fielddata->getNumFields(); i++) {
528  SoField* field = fielddata->getField(node, i);
529  if (!field->isDefault()) {
530  if (field->isOfType(SoSFNode::getClassTypeId())) {
531  auto sfNode = static_cast<SoSFNode*>(field);
532  writeX3DChild(sfNode->getValue(), nodeMap, numDEF, spaces+2, out);
533  }
534  else if (field->isOfType(SoMFNode::getClassTypeId())) {
535  auto mfNode = static_cast<SoMFNode*>(field);
536  for (int j=0; j<mfNode->getNum(); j++) {
537  writeX3DChild(mfNode->getNode(j), nodeMap, numDEF, spaces+2, out);
538  }
539  }
540  }
541  }
542 
543  if (numFieldNodes > 0) {
544  out << Base::blanks(spaces) << "</" << type << ">\n";
545  }
546  }
547 }

549 void Gui::SoFCDB::writeX3DChild(SoNode* node, std::map<SoNode*, std::string>& nodeMap,
550  int& numDEF, int spaces, std::ostream& out)
551 {
552  if (!node)
553  return;
554 
555  // check if the node is already used
556  auto mapIt = nodeMap.find(node);
557  if (mapIt == nodeMap.end()) {
558  writeX3DFields(node, nodeMap, false, numDEF, spaces, out);
559  }
560  else {
561  // remove the VRML prefix from the type name
562  std::string sftype(node->getTypeId().getName().getString());
563  sftype = sftype.substr(4);
564  out << Base::blanks(spaces) << "<" << sftype << " USE=\"" << mapIt->second << "\" />\n";
565  }

568 void Gui::SoFCDB::writeX3D(SoVRMLGroup* node, bool exportViewpoints, std::ostream& out)
569 {
570  out << "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n";
571  out << "<!DOCTYPE html PUBLIC \"-//W3C//DTD XHTML 1.0 Strict//EN\" \"http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd\">\n";
572  out << "<X3D profile=\"Immersive\" version=\"3.2\" xmlns:xsd=\"http://www.w3.org/2001/XMLSchema-instance\" "
573  "xsd:noNamespaceSchemaLocation=\"http://www.web3d.org/specifications/x3d-3.2.xsd\" width=\"1280px\" height=\"1024px\">\n";
574  out << " <head>\n"
575  " <meta name=\"generator\" content=\"FreeCAD\"/>\n"
576  " <meta name=\"author\" content=\"\"/>\n"
577  " <meta name=\"company\" content=\"\"/>\n"
578  " </head>\n";
579 
580  std::map<SoNode*, std::string> nodeMap;
581  out << " <Scene>\n";
582 
583  // compute a sensible view point
584  SoGetBoundingBoxAction bboxAction(SbViewportRegion(1280, 1024));
585  bboxAction.apply(node);
586  SbBox3f bbox = bboxAction.getBoundingBox();
587  SbSphere bs;
588  bs.circumscribe(bbox);
589  const SbVec3f& cnt = bs.getCenter();
590  float dist = 2.4f * bs.getRadius();
591  float dist3 = 0.577350f * dist; // sqrt(1/3) * dist
592 
593  if (exportViewpoints) {
594  auto viewpoint = [&out](const char* text, const SbVec3f& cnt,
595  const SbVec3f& pos, const SbRotation& rot) {
596  SbVec3f axis; float angle;
597  rot.getValue(axis, angle);
598  out << " <Viewpoint id=\"" << text
599  << "\" centerOfRotation=\"" << cnt[0] << " " << cnt[1] << " " << cnt[2]
600  << "\" position=\"" << pos[0] << " " << pos[1] << " " << pos[2]
601  << "\" orientation=\"" << axis[0] << " " << axis[1] << " " << axis[2] << " " << angle
602  << R"(" description="camera" fieldOfView="0.9">)"
603  << "</Viewpoint>\n";
604  };
605 
606  viewpoint("Iso", cnt, SbVec3f(cnt[0] + dist3, cnt[1] - dist3, cnt[2] + dist3), Camera::rotation(Camera::Isometric));
607  viewpoint("Front", cnt, SbVec3f(cnt[0], cnt[1] - dist, cnt[2]), Camera::rotation(Camera::Front));
608  viewpoint("Back", cnt, SbVec3f(cnt[0], cnt[1] + dist, cnt[2]), Camera::rotation(Camera::Rear));
609  viewpoint("Right", cnt, SbVec3f(cnt[0] + dist, cnt[1], cnt[2]), Camera::rotation(Camera::Right));
610  viewpoint("Left", cnt, SbVec3f(cnt[0] - dist, cnt[1], cnt[2]), Camera::rotation(Camera::Left));
611  viewpoint("Top", cnt, SbVec3f(cnt[0], cnt[1], cnt[2] + dist), Camera::rotation(Camera::Top));
612  viewpoint("Bottom", cnt, SbVec3f(cnt[0], cnt[1], cnt[2] - dist), Camera::rotation(Camera::Bottom));
613  }
614 
615  int numDEF = 0;
616  writeX3DFields(node, nodeMap, true, numDEF, 4, out);
617  out << " </Scene>\n";
618  out << "</X3D>\n";
619 }

621 bool Gui::SoFCDB::writeToX3DOM(SoNode* node, std::string& buffer)
622 {
623  std::string x3d;
624  if (!writeToX3D(node, true, x3d))
625  return false;
626 
627  // remove the first two lines from the x3d output as this duplicates
628  // the xml and doctype header
629  std::size_t pos = x3d.find('\n');
630  pos = x3d.find('\n', pos + 1);
631  x3d = x3d.erase(0, pos + 1);
632
