46 std::vector<App::DocumentObject*> ViewProviderPipe::claimChildren()const
47 {
48  std::vector<App::DocumentObject*> temp;
49 
50  PartDesign::Pipe* pcPipe = getObject<PartDesign::Pipe>();
51 
52  App::DocumentObject* sketch = pcPipe->getVerifiedSketch(true);
53  if (sketch)
54  temp.push_back(sketch);
55 
56  for(App::DocumentObject* obj : pcPipe->Sections.getValues()) {
57  if (obj && obj->isDerivedFrom(Part::Part2DObject::getClassTypeId()))
58  temp.push_back(obj);
59  }
60 
61  App::DocumentObject* spine = pcPipe->Spine.getValue();
62  if (spine && spine->isDerivedFrom(Part::Part2DObject::getClassTypeId()))
63  temp.push_back(spine);
64 
65  App::DocumentObject* auxspine = pcPipe->AuxillerySpine.getValue();
66  if (auxspine && auxspine->isDerivedFrom(Part::Part2DObject::getClassTypeId()))
67  temp.push_back(auxspine);
68 
69  return temp;
70 }

113 void ViewProviderPipe::highlightReferences(ViewProviderPipe::Reference mode, bool on)
114 {
115  PartDesign::Pipe* pcPipe = getObject<PartDesign::Pipe>();
116 
117  switch (mode) {
118  case Spine:
119  highlightReferences(dynamic_cast<Part::Feature*>(pcPipe->Spine.getValue()),
120  pcPipe->Spine.getSubValuesStartsWith("Edge"), on);
121  break;
122  case AuxiliarySpine:
123  highlightReferences(dynamic_cast<Part::Feature*>(pcPipe->AuxillerySpine.getValue()),
124  pcPipe->AuxillerySpine.getSubValuesStartsWith("Edge"), on);
125  break;
126  case Profile:
127  highlightReferences(dynamic_cast<Part::Feature*>(pcPipe->Profile.getValue()),
128  pcPipe->Profile.getSubValuesStartsWith("Edge"), on);
129  break;
130  case Section:
131  {
132  std::vector<App::DocumentObject*> sections = pcPipe->Sections.getValues();
133  for (auto it : sections) {
134  highlightReferences(dynamic_cast<Part::Feature*>(it),
135  std::vector<std::string>(), on);
136  }
137  }
138  break;
139  default:
140  break;
141  }
142 }

144 void ViewProviderPipe::highlightReferences(Part::Feature* base, const std::vector<std::string>& edges, bool on)
145 {
146  if (!base)
147  return;
148 
149  PartGui::ViewProviderPart* svp = dynamic_cast<PartGui::ViewProviderPart*>(
150  Gui::Application::Instance->getViewProvider(base));
151  if (!svp)
152  return;
153 
154  std::vector<App::Color>& edgeColors = originalLineColors[base->getID()];
155 
156  if (on) {
157  if (edgeColors.empty()) {
158  edgeColors = svp->LineColorArray.getValues();
159  std::vector<App::Color> colors = edgeColors;
160 
161  PartGui::ReferenceHighlighter highlighter(base->Shape.getValue(), svp->LineColor.getValue());
162  highlighter.getEdgeColors(edges, colors);
163  svp->LineColorArray.setValues(colors);
164  }
165  } else {
166  if (!edgeColors.empty()) {
167  svp->LineColorArray.setValues(edgeColors);
168  edgeColors.clear();
169  }
170  }
171 }
