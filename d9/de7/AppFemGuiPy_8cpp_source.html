116  Py::Object open(const Py::Tuple& args)
117  {
118  char* Name;
119  const char* DocName;
120  if (!PyArg_ParseTuple(args.ptr(), "et|s", "utf-8", &Name, &DocName)) {
121  throw Py::Exception();
122  }
123 
124  std::string EncodedName = std::string(Name);
125  PyMem_Free(Name);
126 
127  QString fileName = QString::fromUtf8(EncodedName.c_str());
128  QFileInfo fi;
129  fi.setFile(fileName);
130  QString ext = fi.completeSuffix().toLower();
131  QList<Gui::EditorView*> views = Gui::getMainWindow()->findChildren<Gui::EditorView*>();
132  for (auto view : views) {
133  if (view->fileName() == fileName) {
134  view->setFocus();
135  return Py::None();
136  }
137  }
138 
139  if ((ext == QLatin1String("inp")) || (ext == QLatin1String("sif"))
140  || (ext == QLatin1String("txt"))) {
141  Gui::TextEditor* editor = new Gui::TextEditor();
142  editor->setWindowIcon(Gui::BitmapFactory().pixmap(":/icons/fem-solver-inp-editor.svg"));
143  Gui::EditorView* edit = new Gui::EditorView(editor, Gui::getMainWindow());
144  if (ext == QLatin1String("inp")) {
145  editor->setSyntaxHighlighter(new FemGui::AbaqusHighlighter(editor));
146  }
147  edit->setDisplayName(Gui::EditorView::FileName);
148  edit->open(fileName);
149  edit->resize(400, 300);
150  Gui::getMainWindow()->addWindow(edit);
151 
152  QFont font = editor->font();
153  font.setFamily(QString::fromLatin1("Arial"));
154  editor->setFont(font);
155  }
156 
157  return Py::None();
158  }
