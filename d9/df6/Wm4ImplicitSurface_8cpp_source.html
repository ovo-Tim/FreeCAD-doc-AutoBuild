72 template <class Real>
73 bool ImplicitSurface<Real>::ComputePrincipalCurvatureInfo (
74  const Vector3<Real>& rkP, Real& rfCurv0, Real& rfCurv1,
75  Vector3<Real>& rkDir0, Vector3<Real>& rkDir1)
76 {
77  // Principal curvatures and directions for implicitly defined surfaces
78  // F(x,y,z) = 0.
79  //
80  // DF = (Fx,Fy,Fz), L = Length(DF)
81  //
82  // D^2 F = +- -+
83  // | Fxx Fxy Fxz |
84  // | Fxy Fyy Fyz |
85  // | Fxz Fyz Fzz |
86  // +- -+
87  //
88  // adj(D^2 F) = +- -+
89  // | Fyy*Fzz-Fyz*Fyz Fyz*Fxz-Fxy*Fzz Fxy*Fyz-Fxz*Fyy |
90  // | Fyz*Fxz-Fxy*Fzz Fxx*Fzz-Fxz*Fxz Fxy*Fxz-Fxx*Fyz |
91  // | Fxy*Fyz-Fxz*Fyy Fxy*Fxz-Fxx*Fyz Fxx*Fyy-Fxy*Fxy |
92  // +- -+
93  //
94  // Gaussian curvature = [DF^t adj(D^2 F) DF]/L^4
95  //
96  // Mean curvature = 0.5*[trace(D^2 F)/L - (DF^t D^2 F DF)/L^3]
97 
98  // first derivatives
99  Real fFX = FX(rkP);
100  Real fFY = FY(rkP);
101  Real fFZ = FZ(rkP);
102  Real fL = Math<Real>::Sqrt(fFX*fFX + fFY*fFY + fFZ*fFZ);
103  if (fL <= Math<Real>::ZERO_TOLERANCE)
104  {
105  return false;
106  }
107 
108  Real fFXFX = fFX*fFX;
109  Real fFXFY = fFX*fFY;
110  Real fFXFZ = fFX*fFZ;
111  Real fFYFY = fFY*fFY;
112  Real fFYFZ = fFY*fFZ;
113  Real fFZFZ = fFZ*fFZ;
114 
115  Real fInvL = ((Real)1.0)/fL;
116  Real fInvL2 = fInvL*fInvL;
117  Real fInvL3 = fInvL*fInvL2;
118  Real fInvL4 = fInvL2*fInvL2;
119 
120  // second derivatives
121  Real fFXX = FXX(rkP);
122  Real fFXY = FXY(rkP);
123  Real fFXZ = FXZ(rkP);
124  Real fFYY = FYY(rkP);
125  Real fFYZ = FYZ(rkP);
126  Real fFZZ = FZZ(rkP);
127 
128  // mean curvature
129  Real fMCurv = ((Real)0.5)*fInvL3*(fFXX*(fFYFY+fFZFZ) + fFYY*(fFXFX+fFZFZ)
130  + fFZZ*(fFXFX+fFYFY)
131  - ((Real)2.0)*(fFXY*fFXFY+fFXZ*fFXFZ+fFYZ*fFYFZ));
132 
133  // Gaussian curvature
134  Real fGCurv = fInvL4*(fFXFX*(fFYY*fFZZ-fFYZ*fFYZ)
135  + fFYFY*(fFXX*fFZZ-fFXZ*fFXZ) + fFZFZ*(fFXX*fFYY-fFXY*fFXY)
136  + ((Real)2.0)*(fFXFY*(fFXZ*fFYZ-fFXY*fFZZ)
137  + fFXFZ*(fFXY*fFYZ-fFXZ*fFYY)
138  + fFYFZ*(fFXY*fFXZ-fFXX*fFYZ)));
139 
140  // solve for principal curvatures
141  Real fDiscr = Math<Real>::Sqrt(Math<Real>::FAbs(fMCurv*fMCurv-fGCurv));
142  rfCurv0 = fMCurv - fDiscr;
143  rfCurv1 = fMCurv + fDiscr;
144 
145  Real fM00 = ((-(Real)1.0 + fFXFX*fInvL2)*fFXX)*fInvL + (fFXFY*fFXY)*fInvL3
146  + (fFXFZ*fFXZ)*fInvL3;
147  Real fM01 = ((-(Real)1.0 + fFXFX*fInvL2)*fFXY)*fInvL + (fFXFY*fFYY)*fInvL3
148  + (fFXFZ*fFYZ)*fInvL3;
149  Real fM02 = ((-(Real)1.0 + fFXFX*fInvL2)*fFXZ)*fInvL + (fFXFY*fFYZ)*fInvL3
150  + (fFXFZ*fFZZ)*fInvL3;
151  Real fM10 = (fFXFY*fFXX)*fInvL3 + ((-(Real)1.0 + fFYFY*fInvL2)*fFXY)*fInvL
152  + (fFYFZ*fFXZ)*fInvL3;
153  Real fM11 = (fFXFY*fFXY)*fInvL3 + ((-(Real)1.0 + fFYFY*fInvL2)*fFYY)*fInvL
154  + (fFYFZ*fFYZ)*fInvL3;
155  Real fM12 = (fFXFY*fFXZ)*fInvL3 + ((-(Real)1.0 + fFYFY*fInvL2)*fFYZ)*fInvL
156  + (fFYFZ*fFZZ)*fInvL3;
157  Real fM20 = (fFXFZ*fFXX)*fInvL3 + (fFYFZ*fFXY)*fInvL3 + ((-(Real)1.0
158  + fFZFZ*fInvL2)*fFXZ)*fInvL;
159  Real fM21 = (fFXFZ*fFXY)*fInvL3 + (fFYFZ*fFYY)*fInvL3 + ((-(Real)1.0
160  + fFZFZ*fInvL2)*fFYZ)*fInvL;
161  Real fM22 = (fFXFZ*fFXZ)*fInvL3 + (fFYFZ*fFYZ)*fInvL3 + ((-(Real)1.0
162  + fFZFZ*fInvL2)*fFZZ)*fInvL;
163 
164  // solve for principal directions
165  Real fTmp1 = fM00 + rfCurv0;
166  Real fTmp2 = fM11 + rfCurv0;
167  Real fTmp3 = fM22 + rfCurv0;
168 
169  Vector3<Real> akU[3];
170  Real afLength[3];
171 
172  akU[0].X() = fM01*fM12-fM02*fTmp2;
173  akU[0].Y() = fM02*fM10-fM12*fTmp1;
174  akU[0].Z() = fTmp1*fTmp2-fM01*fM10;
175  afLength[0] = akU[0].Length();
176 
177  akU[1].X() = fM01*fTmp3-fM02*fM21;
178  akU[1].Y() = fM02*fM20-fTmp1*fTmp3;
179  akU[1].Z() = fTmp1*fM21-fM01*fM20;
180  afLength[1] = akU[1].Length();
181 
182  akU[2].X() = fTmp2*fTmp3-fM12*fM21;
183  akU[2].Y() = fM12*fM20-fM10*fTmp3;
184  akU[2].Z() = fM10*fM21-fM20*fTmp2;
185  afLength[2] = akU[2].Length();
186 
187  int iMaxIndex = 0;
188  Real fMax = afLength[0];
189  if (afLength[1] > fMax)
190  {
191  iMaxIndex = 1;
192  fMax = afLength[1];
193  }
194  if (afLength[2] > fMax)
195  {
196  iMaxIndex = 2;
197  }
198 
199  Real fInvLength = ((Real)1.0)/afLength[iMaxIndex];
200  akU[iMaxIndex] *= fInvLength;
201 
202  rkDir1 = akU[iMaxIndex];
203  rkDir0 = rkDir1.UnitCross(Vector3<Real>(fFX,fFY,fFZ));
204 
205  return true;
206 }
