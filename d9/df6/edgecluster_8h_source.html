37 struct Edgesort_gp_Pnt_Less
38 {
39  bool operator()(const gp_Pnt & _Left, const gp_Pnt & _Right) const
40  {
41  Standard_Real x1,y1,z1,x2,y2,z2;
42  _Left.Coord(x1,y1,z1);
43  _Right.Coord(x2,y2,z2);
44  if ( fabs(x1- x2) > 0.2 )
45  return x1 < x2;
46  else if ( fabs(y1 -y2) > 0.2 )
47  return y1 < y2;
48  else if ( fabs(z1 -z2) > 0.2 )
49  return z1 < z2;
50  return false;
51  }
52 };
