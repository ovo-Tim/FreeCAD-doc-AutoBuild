39 // returns a string which represents the object e.g. when printed in python
40 std::string WaypointPy::representation() const
41 {
42  double A, B, C;
43  std::stringstream str;
44  getWaypointPtr()->EndPos.getRotation().getYawPitchRoll(A, B, C);
45  str.precision(5);
46  str << "Waypoint [";
47  if (getWaypointPtr()->Type == Waypoint::PTP) {
48  str << "PTP ";
49  }
50  else if (getWaypointPtr()->Type == Waypoint::LINE) {
51  str << "LIN ";
52  }
53  else if (getWaypointPtr()->Type == Waypoint::CIRC) {
54  str << "CIRC ";
55  }
56  else if (getWaypointPtr()->Type == Waypoint::WAIT) {
57  str << "WAIT ";
58  }
59  else if (getWaypointPtr()->Type == Waypoint::UNDEF) {
60  str << "UNDEF ";
61  }
62  str << getWaypointPtr()->Name;
63  str << " (";
64  str << getWaypointPtr()->EndPos.getPosition().x << ","
65  << getWaypointPtr()->EndPos.getPosition().y << ","
66  << getWaypointPtr()->EndPos.getPosition().z;
67  str << ";" << A << "," << B << "," << C << ")";
68  str << "v=" << getWaypointPtr()->Velocity << " ";
69  if (getWaypointPtr()->Cont) {
70  str << "Cont ";
71  }
72  if (getWaypointPtr()->Tool != 0) {
73  str << "Tool" << getWaypointPtr()->Tool << " ";
74  }
75  if (getWaypointPtr()->Base != 0) {
76  str << "Tool" << getWaypointPtr()->Base << " ";
77  }
78  str << "]";
79 
80  return str.str();
81 }
