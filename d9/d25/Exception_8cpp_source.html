90 void Exception::ReportException() const
91 {
92  if (!_isReported) {
93  const char* msg {};
94  if (_sErrMsg.empty()) {
95  msg = typeid(*this).name();
96  }
97  else {
98  msg = _sErrMsg.c_str();
99  }
100 #ifdef FC_DEBUG
101  if (!_function.empty()) {
102  _FC_ERR(_file.c_str(), _line, _function << " -- " << msg);
103  }
104  else
105 #endif
106  {
107  _FC_ERR(_file.c_str(), _line, msg);
108  }
109  _isReported = true;
110  }
111 }

113 PyObject* Exception::getPyObject()
114 {
115  Py::Dict edict;
116  edict.setItem("sclassname", Py::String(typeid(*this).name()));
117  edict.setItem("sErrMsg", Py::String(this->getMessage()));
118  edict.setItem("sfile", Py::String(this->getFile()));
119  edict.setItem("iline", Py::Long(this->getLine()));
120  edict.setItem("sfunction", Py::String(this->getFunction()));
121  edict.setItem("swhat", Py::String(this->what()));
122  edict.setItem("btranslatable", Py::Boolean(this->getTranslatable()));
123  edict.setItem("breported", Py::Boolean(this->_isReported));
124  return Py::new_reference_to(edict);
125 }

127 void Exception::setPyObject(PyObject* pydict)
128 {
129  try {
130  if (pydict && Py::_Dict_Check(pydict)) {
131  Py::Dict edict(pydict);
132  if (edict.hasKey("sfile")) {
133  _file = static_cast<std::string>(Py::String(edict.getItem("sfile")));
134  }
135 
136  if (edict.hasKey("sfunction")) {
137  _function = static_cast<std::string>(Py::String(edict.getItem("sfunction")));
138  }
139 
140  if (edict.hasKey("sErrMsg")) {
141  _sErrMsg = static_cast<std::string>(Py::String(edict.getItem("sErrMsg")));
142  }
143 
144  if (edict.hasKey("iline")) {
145  _line = static_cast<long>(Py::Long(edict.getItem("iline")));
146  }
147  if (edict.hasKey("btranslatable")) {
148  _isTranslatable = static_cast<bool>(Py::Boolean(edict.getItem("btranslatable")));
149  }
150  if (edict.hasKey("breported")) {
151  _isReported = static_cast<bool>(Py::Boolean(edict.getItem("breported")));
152  }
153  }
154  }
155  catch (Py::Exception& e) {
156  e.clear(); // should never happen
157  }
158 }
