507  "Grounding a part permanently locks its position in the assembly, preventing any movement or rotation. You need at least one grounded part before starting to assemble.",
508  )

519  def Activated(self):
520  assembly = UtilsAssembly.activeAssembly()
521  if not assembly:
522  return
523 
524  joint_group = UtilsAssembly.getJointGroup(assembly)
525 
526  selection = Gui.Selection.getSelectionEx("*", 0)
527  if not selection:
528  return
529 
530  App.setActiveTransaction("Toggle grounded")
531  for sel in selection:
532  # If you select 2 solids (bodies for example) within an assembly.
533  # There'll be a single sel but 2 SubElementNames.
534  for sub in sel.SubElementNames:
535  ref = [sel.Object, [sub, sub]]
536  moving_part = UtilsAssembly.getMovingPart(assembly, ref)
537 
538  # Only objects within the assembly.
539  if moving_part is None:
540  continue
541 
542  # Check if part is grounded and if so delete the joint.
543  ungrounded = False
544  for joint in joint_group.Group:
545  if hasattr(joint, "ObjectToGround") and joint.ObjectToGround == moving_part:
546  commands = (
547  "doc = App.ActiveDocument\n"
548  f'doc.removeObject("{joint.Name}")\n'
549  "doc.recompute()\n"
550  )
551  Gui.doCommand(commands)
552  ungrounded = True
553  break
554  if ungrounded:
555  continue
556 
557  # Create groundedJoint.
558  createGroundedJoint(moving_part)
559  App.closeActiveTransaction()
