132  Py::Object exporter(const Py::Tuple& args)
133  {
134  PyObject* object;
135  char* Name;
136  if (!PyArg_ParseTuple(args.ptr(), "Oet", &object, "utf-8", &Name)) {
137  throw Py::Exception();
138  }
139 
140  std::string EncodedName = std::string(Name);
141  PyMem_Free(Name);
142 
143  Py::Sequence list(object);
144  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
145  PyObject* item = (*it).ptr();
146  if (PyObject_TypeCheck(item, &(App::DocumentObjectPy::Type))) {
147  App::DocumentObject* obj =
148  static_cast<App::DocumentObjectPy*>(item)->getDocumentObjectPtr();
149  if (obj->isDerivedFrom<Drawing::FeaturePage>()) {
150  Base::FileInfo fi_out(EncodedName.c_str());
151  Base::ofstream str_out(fi_out, std::ios::out | std::ios::binary);
152  if (!str_out) {
153  std::stringstream str;
154  str << "Cannot open file '" << EncodedName << "' for writing";
155  throw Py::Exception(PyExc_IOError, str.str().c_str());
156  }
157  if (fi_out.hasExtension("svg")) {
158  std::string fn =
159  static_cast<Drawing::FeaturePage*>(obj)->PageResult.getValue();
160  Base::FileInfo fi_in(fn);
161  Base::ifstream str_in(fi_in, std::ios::in | std::ios::binary);
162  if (!str_in) {
163  std::stringstream str;
164  str << "Cannot open file '" << fn << "' for reading";
165  throw Py::Exception(PyExc_IOError, str.str().c_str());
166  }
167 
168  str_in >> str_out.rdbuf();
169  str_in.close();
170  str_out.close();
171  break;
172  }
173  else if (fi_out.hasExtension("dxf")) {
174  const std::vector<App::DocumentObject*>& views =
175  static_cast<Drawing::FeaturePage*>(obj)->Group.getValues();
176  for (std::vector<App::DocumentObject*>::const_iterator it = views.begin();
177  it != views.end();
178  ++it) {
179  if ((*it)->isDerivedFrom<Drawing::FeatureViewPart>()) {
180  Drawing::FeatureViewPart* view =
181  static_cast<Drawing::FeatureViewPart*>(*it);
182  App::DocumentObject* link = view->Source.getValue();
183  if (!link) {
184  throw Py::ValueError("No object linked");
185  }
186  if (!link->isDerivedFrom<Part::Feature>()) {
187  throw Py::TypeError("Linked object is not a Part object");
188  }
189  TopoDS_Shape shape =
190  static_cast<Part::Feature*>(link)->Shape.getShape().getShape();
191  if (!shape.IsNull()) {
192  Base::Vector3d dir = view->Direction.getValue();
193  bool hidden = view->ShowHiddenLines.getValue();
194  bool smooth = view->ShowSmoothLines.getValue();
195  Drawing::ProjectionAlgos::ExtractionType type =
196  Drawing::ProjectionAlgos::Plain;
197  if (hidden) {
198  type = (Drawing::ProjectionAlgos::ExtractionType)(
199  type | Drawing::ProjectionAlgos::WithHidden);
200  }
201  if (smooth) {
202  type = (Drawing::ProjectionAlgos::ExtractionType)(
203  type | Drawing::ProjectionAlgos::WithSmooth);
204  }
205  float scale = view->Scale.getValue();
206  float tol = view->Tolerance.getValue();
207 
208  Drawing::ProjectionAlgos project(shape, dir);
209  str_out << project.getDXF(type, scale, tol);
210  break; // TODO: How to add several shapes?
211  }
212  }
213  }
214  str_out.close();
215  break;
216  }
217  else {
218  throw Py::TypeError("Export of page object as this file format is not "
219  "supported by Drawing module");
220  }
221  }
222  else {
223  throw Py::TypeError(
224  "Export of this object type is not supported by Drawing module");
225  }
226  }
227  }
228 
229  return Py::None();
230  }
