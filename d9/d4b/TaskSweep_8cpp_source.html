70  class EdgeSelection : public Gui::SelectionFilterGate
71  {
72  public:
73  EdgeSelection()
74  : Gui::SelectionFilterGate(nullPointer())
75  {
76  }
77  bool allow(App::Document* /*pDoc*/, App::DocumentObject*pObj, const char*sSubName) override
78  {
79  if (!sSubName || sSubName[0] == '\0') {
80  // If selecting again the same edge the passed sub-element is empty. If the whole
81  // shape is an edge or wire we can use it completely.
82  Part::TopoShape topoShape = Part::Feature::getTopoShape(pObj);
83  if (topoShape.isNull()) {
84  return false;
85  }
86  const TopoDS_Shape shape = topoShape.getShape();
87  if (!shape.IsNull()) {
88  // a single edge
89  if (shape.ShapeType() == TopAbs_EDGE) {
90  return true;
91  }
92  // a single wire
93  if (shape.ShapeType() == TopAbs_WIRE) {
94  return true;
95  }
96  // a compound of only edges or wires
97  if (shape.ShapeType() == TopAbs_COMPOUND) {
98  TopoDS_Iterator it(shape);
99  for (; it.More(); it.Next()) {
100  if (it.Value().IsNull())
101  return false;
102  if ((it.Value().ShapeType() != TopAbs_EDGE) &&
103  (it.Value().ShapeType() != TopAbs_WIRE))
104  return false;
105  }
106 
107  return true;
108  }
109  }
110  }
111  else {
112  std::string element(sSubName);
113  return element.substr(0,4) == "Edge";
114  }
115 
116  return false;
117  }
118  };
119 };

153 void SweepWidget::findShapes()
154 {
155  App::Document* activeDoc = App::GetApplication().getActiveDocument();
156  Gui::Document* activeGui = Gui::Application::Instance->getDocument(activeDoc);
157  if (!activeGui)
158  return;
159  d->document = activeDoc->getName();
160 
161  std::vector<App::DocumentObject*> objs = activeDoc->getObjectsOfType<App::DocumentObject>();
162 
163  for (auto obj : objs) {
164  Part::TopoShape topoShape = Part::Feature::getTopoShape(obj);
165  if (topoShape.isNull()) {
166  continue;
167  }
168  TopoDS_Shape shape = topoShape.getShape();
169  if (shape.IsNull()) continue;
170 
171  // also allow compounds with a single face, wire or vertex or
172  // if there are only edges building one wire
173  if (shape.ShapeType() == TopAbs_COMPOUND) {
174  Handle(TopTools_HSequenceOfShape) hEdges = new TopTools_HSequenceOfShape();
175  Handle(TopTools_HSequenceOfShape) hWires = new TopTools_HSequenceOfShape();
176 
177  TopoDS_Iterator it(shape);
178  int numChilds=0;
179  TopoDS_Shape child;
180  for (; it.More(); it.Next(), numChilds++) {
181  if (!it.Value().IsNull()) {
182  child = it.Value();
183  if (child.ShapeType() == TopAbs_EDGE) {
184  hEdges->Append(child);
185  }
186  }
187  }
188 
189  // a single child
190  if (numChilds == 1) {
191  shape = child;
192  }
193  // or all children are edges
194  else if (hEdges->Length() == numChilds) {
195  ShapeAnalysis_FreeBounds::ConnectEdgesToWires(hEdges,
196  Precision::Confusion(), Standard_False, hWires);
197  if (hWires->Length() == 1)
198  shape = hWires->Value(1);
199  }
200  }
201 
202  if (!shape.Infinite() &&
203  (shape.ShapeType() == TopAbs_FACE ||
204  shape.ShapeType() == TopAbs_WIRE ||
205  shape.ShapeType() == TopAbs_EDGE ||
206  shape.ShapeType() == TopAbs_VERTEX)) {
207  QString label = QString::fromUtf8(obj->Label.getValue());
208  QString name = QString::fromLatin1(obj->getNameInDocument());
209 
210  QTreeWidgetItem* child = new QTreeWidgetItem();
211  child->setText(0, label);
212  child->setToolTip(0, label);
213  child->setData(0, Qt::UserRole, name);
214  Gui::ViewProvider* vp = activeGui->getViewProvider(obj);
215  if (vp) child->setIcon(0, vp->getIcon());
216  d->ui.selector->availableTreeWidget()->addTopLevelItem(child);
217  }
218  }

221 bool SweepWidget::isPathValid(const Gui::SelectionObject& sel) const
222 {
223  const App::DocumentObject* path = sel.getObject();
224  const std::vector<std::string>& sub = sel.getSubNames();
225 
226  TopoDS_Shape pathShape;
227  const Part::TopoShape& shape = Part::Feature::getTopoShape(path);
228  if (shape.isNull()){
229  return false;
230  }
231  if (!sub.empty()) {
232  try {
233  BRepBuilderAPI_MakeWire mkWire;
234  for (const auto & it : sub) {
235  TopoDS_Shape subshape = shape.getSubShape(it.c_str());
236  mkWire.Add(TopoDS::Edge(subshape));
237  }
238  pathShape = mkWire.Wire();
239  }
240  catch (...) {
241  return false;
242  }
243  }
244  else if (shape.getShape().ShapeType() == TopAbs_EDGE) {
245  pathShape = shape.getShape();
246  }
247  else if (shape.getShape().ShapeType() == TopAbs_WIRE) {
248  BRepBuilderAPI_MakeWire mkWire(TopoDS::Wire(shape.getShape()));
249  pathShape = mkWire.Wire();
250  }
251  else if (shape.getShape().ShapeType() == TopAbs_COMPOUND) {
252  try {
253  TopoDS_Iterator it(shape.getShape());
254  for (; it.More(); it.Next()) {
255  if ((it.Value().ShapeType() != TopAbs_EDGE) &&
256  (it.Value().ShapeType() != TopAbs_WIRE)) {
257  return false;
258  }
259  }
260  Handle(TopTools_HSequenceOfShape) hEdges = new TopTools_HSequenceOfShape();
261  Handle(TopTools_HSequenceOfShape) hWires = new TopTools_HSequenceOfShape();
262  for (TopExp_Explorer xp(shape.getShape(), TopAbs_EDGE); xp.More(); xp.Next())
263  hEdges->Append(xp.Current());
264 
265  ShapeAnalysis_FreeBounds::ConnectEdgesToWires(hEdges, Precision::Confusion(), Standard_True, hWires);
266  int len = hWires->Length();
267  if (len != 1)
268  return false;
269  pathShape = hWires->Value(1);
270  }
271  catch (...) {
272  return false;
273  }
274  }
275 
276  return (!pathShape.IsNull());
277 }

279 bool SweepWidget::accept()
280 {
281  if (d->ui.buttonPath->isChecked())
282  return false;
283  const App::DocumentObject* docobj = nullptr;
284  std::string selection;
285  const std::vector<Gui::SelectionObject> selobjs = Gui::Selection().getSelectionEx();
286  std::vector<Part::TopoShape> subShapes;
287  Part::TopoShape topoShape = Part::TopoShape();
288  std::string spineObject, spineLabel;
289 
290  bool ok = true;
291  if (selobjs.size() == 1) {
292  selection = selobjs[0].getAsPropertyLinkSubString();
293  const std::vector<std::string>& subnames = selobjs[0].getSubNames();
294  docobj = selobjs[0].getObject();
295  spineObject = selobjs[0].getFeatName();
296  spineLabel = docobj->Label.getValue();
297  topoShape = Part::Feature::getTopoShape(docobj);
298  if (!topoShape.isNull()) {
299  for (std::vector<std::string>::const_iterator it = subnames.begin(); it != subnames.end(); ++it) {
300  subShapes.push_back(Part::Feature::getTopoShape(docobj, subnames[0].c_str(), true /*need element*/));
301  }
302  for (const auto & it : subShapes) {
303  TopoDS_Shape dsShape = it.getShape();
304  if (dsShape.IsNull() || dsShape.ShapeType() != TopAbs_EDGE) { //only edge selection allowed
305  ok = false;
306  }
307  }
308  } else { //could be not a part::feature or app:link to non-part::feature or app::part without a visible part::feature
309  ok = false;
310  }
311 
312  } else { //not just one object selected
313  ok = false;
314  }
315 
316  QString list, solid, frenet;
317  if (d->ui.checkSolid->isChecked())
318  solid = QString::fromLatin1("True");
319  else
320  solid = QString::fromLatin1("False");
321 
322  if (d->ui.checkFrenet->isChecked())
323  frenet = QString::fromLatin1("True");
324  else
325  frenet = QString::fromLatin1("False");
326 
327  QTextStream str(&list);
328 
329  int count = d->ui.selector->selectedTreeWidget()->topLevelItemCount();
330  if (count < 1) {
331  QMessageBox::critical(this, tr("Too few elements"), tr("At least one edge or wire is required."));
332  return false;
333  }
334  if (!ok) {
335  QMessageBox::critical(this, tr("Invalid selection"), tr("Select one or more edges from a single object."));
336  return false;
337  }
338  for (int i=0; i<count; i++) {
339  QTreeWidgetItem* child = d->ui.selector->selectedTreeWidget()->topLevelItem(i);
340  QString name = child->data(0, Qt::UserRole).toString();
341  if (name == QLatin1String(spineObject.c_str())) {
342  QMessageBox::critical(this, tr("Wrong selection"), tr("'%1' cannot be used as profile and path.")
343  .arg(QString::fromUtf8(spineLabel.c_str())));
344  return false;
345  }
346  str << "App.getDocument('" << d->document.c_str() << "')." << name << ", ";
347  }
348 
349  try {
350  Gui::WaitCursor wc;
351  QString cmd;
352  cmd = QString::fromLatin1(
353  "App.getDocument('%5').addObject('Part::Sweep','Sweep')\n"
354  "App.getDocument('%5').ActiveObject.Sections=[%1]\n"
355  "App.getDocument('%5').ActiveObject.Spine=%2\n"
356  "App.getDocument('%5').ActiveObject.Solid=%3\n"
357  "App.getDocument('%5').ActiveObject.Frenet=%4\n"
358  )
359  .arg(list,
360  QLatin1String(selection.c_str()),
361  solid,
362  frenet,
363  QString::fromLatin1(d->document.c_str()));
364 
365  Gui::Document* doc = Gui::Application::Instance->getDocument(d->document.c_str());
366  if (!doc)
367  throw Base::RuntimeError("Document doesn't exist anymore");
368  doc->openCommand(QT_TRANSLATE_NOOP("Command", "Sweep"));
369  Gui::Command::runCommand(Gui::Command::App, cmd.toLatin1());
370  doc->getDocument()->recompute();
371  App::DocumentObject* obj = doc->getDocument()->getActiveObject();
372  if (obj && !obj->isValid()) {
373  std::string msg = obj->getStatusString();
374  doc->abortCommand();
375  throw Base::RuntimeError(msg);
376  }
377  doc->commitCommand();
378  }
379  catch (const Base::Exception& e) {
380  QMessageBox::warning(this, tr("Input error"), QCoreApplication::translate("Exception", e.what()));
381  return false;
382  }
383 
384  return true;
385 }

421  else {
422  QList<QWidget*> c = this->findChildren<QWidget*>();
423  for (auto it : c)
424  it->setEnabled(true);
425  d->ui.buttonPath->setText(d->buttonText);
426  d->ui.labelPath->clear();
427  Gui::Selection().rmvSelectionGate();
428 
429  Gui::SelectionFilter edgeFilter ("SELECT Part::Feature SUBELEMENT Edge COUNT 1..");
430  Gui::SelectionFilter partFilter ("SELECT Part::Feature COUNT 1");
431  bool matchEdge = edgeFilter.match();
432  bool matchPart = partFilter.match();
433  if (matchEdge) {
434  // check if path is valid
435  const std::vector<Gui::SelectionObject>& result = edgeFilter.Result[0];
436  if (!isPathValid(result.front())) {
437  QMessageBox::critical(this, tr("Sweep path"), tr("The selected sweep path is invalid."));
438  Gui::Selection().clearSelection();
439  }
440  }
441  else if (matchPart) {
442  // check if path is valid
443  const std::vector<Gui::SelectionObject>& result = partFilter.Result[0];
444  if (!isPathValid(result.front())) {
445  QMessageBox::critical(this, tr("Sweep path"), tr("The selected sweep path is invalid."));
446  Gui::Selection().clearSelection();
447  }
448  }
449  }
