125 PreferenceUiForm::PreferenceUiForm(const QString& fn, QWidget* parent)
126  : PreferencePage(parent)
127  , _form(nullptr)
128 {
129  auto loader = UiLoader::newInstance();
130  loader->setWorkingDirectory(QFileInfo(fn).absolutePath());
131  QFile file(fn);
132  if (file.open(QFile::ReadOnly)) {
133  _form = loader->load(&file, this);
134  }
135  file.close();
136  if (_form) {
137  this->setWindowTitle(_form->windowTitle());
138  auto layout = new QVBoxLayout;
139  layout->addWidget(_form);
140  setLayout(layout);
141  }
142  else {
143  Base::Console().Error("Failed to load UI file from '%s'\n",
144  (const char*)fn.toUtf8());
145  }
146 }

218 void PreferencePage::resetSettingsToDefaults()
219 {
220  auto prefs = this->findChildren<QObject*>();
221 
222  for (const auto& pref : prefs) {
223  if (!pref->property("prefPath").isNull() && !pref->property("prefEntry").isNull()) {
224  std::string path = pref->property("prefPath").toString().toStdString();
225  std::string entry = pref->property("prefEntry").toString().toStdString();
226 
227  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
228  std::string("User parameter:BaseApp/Preferences/" + path).c_str());
229 
230  for (const auto& pn : hGrp->GetParameterNames(entry.c_str())) {
231  hGrp->RemoveAttribute(pn.first, pn.second.c_str());
232  }
233  }
234  }
235 }

246 bool CustomizeActionPage::event(QEvent* e)
247 {
248  bool ok = QWidget::event(e);
249 
250  if (e->type() == QEvent::ParentChange || e->type() == QEvent::ParentAboutToChange) {
251  QWidget* topLevel = this->parentWidget();
252  while (topLevel && !topLevel->inherits("QDialog"))
253  topLevel = topLevel->parentWidget();
254  if (topLevel) {
255  int index = topLevel->metaObject()->indexOfSignal( QMetaObject::normalizedSignature("addMacroAction(const QByteArray&)") );
256  if (index >= 0) {
257  if (e->type() == QEvent::ParentChange) {
258  connect(topLevel, SIGNAL(addMacroAction( const QByteArray& )),
259  this, SLOT(onAddMacroAction( const QByteArray& )));
260  connect(topLevel, SIGNAL(removeMacroAction( const QByteArray& )),
261  this, SLOT(onRemoveMacroAction( const QByteArray& )));
262  connect(topLevel, SIGNAL(modifyMacroAction( const QByteArray& )),
263  this, SLOT(onModifyMacroAction( const QByteArray& )));
264  }
265  else {
266  disconnect(topLevel, SIGNAL(addMacroAction( const QByteArray& )),
267  this, SLOT(onAddMacroAction( const QByteArray& )));
268  disconnect(topLevel, SIGNAL(removeMacroAction( const QByteArray& )),
269  this, SLOT(onRemoveMacroAction( const QByteArray& )));
270  disconnect(topLevel, SIGNAL(modifyMacroAction( const QByteArray& )),
271  this, SLOT(onModifyMacroAction( const QByteArray& )));
272  }
273  }
274  }
275  }
276 
277  return ok;
278 }
