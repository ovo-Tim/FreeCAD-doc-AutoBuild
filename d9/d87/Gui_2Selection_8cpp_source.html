323 std::vector<SelectionObject> SelectionSingleton::getObjectList(const char* pDocName, Base::Type typeId,
324  std::list<_SelObj> &objList,
325  ResolveMode resolve, bool single) const
326 {
327  std::vector<SelectionObject> temp;
328  if (single)
329  temp.reserve(1);
330  std::map<App::DocumentObject*,size_t> SortMap;
331 
332  // check the type
333  if (typeId == Base::Type::badType())
334  return temp;
335 
336  App::Document *pcDoc = nullptr;
337  if (!pDocName || strcmp(pDocName,"*") != 0) {
338  pcDoc = getDocument(pDocName);
339  if (!pcDoc)
340  return temp;
341  }
342 
343  for (auto &sel : objList) {
344  if(!sel.pDoc)
345  continue;
346  const char *subelement = nullptr;
347  auto obj = getObjectOfType(sel,typeId,resolve,&subelement);
348  if (!obj || (pcDoc && sel.pObject->getDocument() != pcDoc))
349  continue;
350  auto it = SortMap.find(obj);
351  if(it!=SortMap.end()) {
352  // only add sub-element
353  if (subelement && *subelement) {
354  if (resolve != ResolveMode::NoResolve && !temp[it->second]._SubNameSet.insert(subelement).second)
355  continue;
356  temp[it->second].SubNames.emplace_back(subelement);
357  temp[it->second].SelPoses.emplace_back(sel.x,sel.y,sel.z);
358  }
359  }
360  else {
361  if (single && !temp.empty()) {
362  temp.clear();
363  break;
364  }
365  // create a new entry
366  temp.emplace_back(obj);
367  if (subelement && *subelement) {
368  temp.back().SubNames.emplace_back(subelement);
369  temp.back().SelPoses.emplace_back(sel.x,sel.y,sel.z);
370  if (resolve != ResolveMode::NoResolve)
371  temp.back()._SubNameSet.insert(subelement);
372  }
373  SortMap.insert(std::make_pair(obj,temp.size()-1));
374  }
375  }
376 
377  return temp;
378 }

394 void SelectionSingleton::notify(SelectionChanges &&Chng)
395 {
396  if(Notifying) {
397  NotificationQueue.push_back(std::move(Chng));
398  return;
399  }
400  Base::FlagToggler<bool> flag(Notifying);
401  NotificationQueue.push_back(std::move(Chng));
402  while(!NotificationQueue.empty()) {
403  const auto &msg = NotificationQueue.front();
404  bool notify;
405  switch(msg.Type) {
406  case SelectionChanges::AddSelection:
407  notify = isSelected(msg.pDocName, msg.pObjectName, msg.pSubName, ResolveMode::NoResolve);
408  break;
409  case SelectionChanges::RmvSelection:
410  notify = !isSelected(msg.pDocName, msg.pObjectName, msg.pSubName, ResolveMode::NoResolve);
411  break;
412  case SelectionChanges::SetPreselect:
413  notify = CurrentPreselection.Type==SelectionChanges::SetPreselect
414  && CurrentPreselection.Object == msg.Object;
415  break;
416  case SelectionChanges::RmvPreselect:
417  notify = CurrentPreselection.Type==SelectionChanges::ClrSelection;
418  break;
419  default:
420  notify = true;
421  }
422  if(notify) {
423  Notify(msg);
424  try {
425  signalSelectionChanged(msg);
426  }
427  catch (const boost::exception&) {
428  // reported by code analyzers
429  Base::Console().Warning("notify: Unexpected boost exception\n");
430  }
431  }
432  NotificationQueue.pop_front();
433  }

597 int SelectionSingleton::setPreselect(const char* pDocName, const char* pObjectName, const char* pSubName,
598  float x, float y, float z, SelectionChanges::MsgSource signal)
599 {
600  if (!pDocName || !pObjectName) {
601  rmvPreselect();
602  return 0;
603  }
604  if (!pSubName) pSubName = "";
605 
606  if (DocName==pDocName && FeatName==pObjectName && SubName==pSubName) {
607  return -1;
608  }
609 
610  rmvPreselect();
611 
612  if (ActiveGate && signal != SelectionChanges::MsgSource::Internal) {
613  App::Document* pDoc = getDocument(pDocName);
614  if (!pDoc || !pObjectName)
615  return 0;
616  App::ElementNamePair elementName;
617  auto pObject = pDoc->getObject(pObjectName);
618  if(!pObject)
619  return 0;
620 
621  const char *subelement = pSubName;
622  if (gateResolve != ResolveMode::NoResolve) {
623  auto &newElementName = elementName.newName;
624  auto &oldElementName = elementName.oldName;
625  pObject = App::GeoFeature::resolveElement(pObject,pSubName,elementName);
626  if (!pObject)
627  return 0;
628  if (gateResolve > ResolveMode::OldStyleElement)
629  subelement = !newElementName.empty() ? newElementName.c_str() : oldElementName.c_str();
630  else
631  subelement = oldElementName.c_str();
632  }
633  if (!ActiveGate->allow(pObject->getDocument(), pObject, subelement)) {
634  QString msg;
635  if (ActiveGate->notAllowedReason.length() > 0){
636  msg = QObject::tr(ActiveGate->notAllowedReason.c_str());
637  } else {
638  msg = QCoreApplication::translate("SelectionFilter","Not allowed:");
639  }
640  msg.append(QString::fromLatin1(" %1.%2.%3 ")
641  .arg(QString::fromLatin1(pDocName),
642  QString::fromLatin1(pObjectName),
643  QString::fromLatin1(pSubName)));
644 
645  if (getMainWindow()) {
646  getMainWindow()->showMessage(msg);
647  Gui::MDIView* mdi = Gui::Application::Instance->activeDocument()->getActiveView();
648  mdi->setOverrideCursor(QCursor(Qt::ForbiddenCursor));
649  }
650  return 0;
651  }
652  Gui::MDIView* mdi = Gui::Application::Instance->activeDocument()->getActiveView();
653  mdi->restoreOverrideCursor();
654  }
655 
656  DocName = pDocName;
657  FeatName= pObjectName;
658  SubName = pSubName;
659  hx = x;
660  hy = y;
661  hz = z;
662 
663  // set up the change object
664  SelectionChanges Chng(signal == SelectionChanges::MsgSource::Internal
665  ? SelectionChanges::SetPreselectSignal
666  : SelectionChanges::SetPreselect,
667  DocName,FeatName,SubName,std::string(),x,y,z,signal);
668 
669  if (Chng.Type==SelectionChanges::SetPreselect) {
670  CurrentPreselection = Chng;
671  FC_TRACE("preselect "<<DocName<<'#'<<FeatName<<'.'<<SubName);
672  }
673  else {
674  FC_TRACE("preselect signal "<<DocName<<'#'<<FeatName<<'.'<<SubName);
675  }
676 
677  notify(Chng);
678 
679  if (signal == SelectionChanges::MsgSource::Internal && !DocName.empty()) {
680  FC_TRACE("preselect "<<DocName<<'#'<<FeatName<<'.'<<SubName);
681  Chng.Type = SelectionChanges::SetPreselect;
682  CurrentPreselection = Chng;
683  notify(std::move(Chng));
684  }
685 
686  // It is possible the preselect is removed during notification
687  return DocName.empty()?0:1;
688 }

1483 int SelectionSingleton::checkSelection(const char *pDocName, const char *pObjectName, const char *pSubName,
1484  ResolveMode resolve, _SelObj &sel, const std::list<_SelObj> *selList) const
1485 {
1486  sel.pDoc = getDocument(pDocName);
1487  if(!sel.pDoc) {
1488  if(!selList)
1489  FC_ERR("Cannot find document");
1490  return -1;
1491  }
1492  pDocName = sel.pDoc->getName();
1493  sel.DocName = pDocName;
1494 
1495  if(pObjectName)
1496  sel.pObject = sel.pDoc->getObject(pObjectName);
1497  else
1498  sel.pObject = nullptr;
1499  if (!sel.pObject) {
1500  if(!selList)
1501  FC_ERR("Object not found");
1502  return -1;
1503  }
1504  if (sel.pObject->testStatus(App::ObjectStatus::Remove))
1505  return -1;
1506  if (pSubName)
1507  sel.SubName = pSubName;
1508  if (resolve == ResolveMode::NoResolve)
1509  TreeWidget::checkTopParent(sel.pObject,sel.SubName);
1510  pSubName = !sel.SubName.empty()?sel.SubName.c_str():nullptr;
1511  sel.FeatName = sel.pObject->getNameInDocument();
1512  sel.TypeName = sel.pObject->getTypeId().getName();
1513  const char *element = nullptr;
1514  sel.pResolvedObject = App::GeoFeature::resolveElement(sel.pObject,
1515  pSubName,sel.elementName,false,App::GeoFeature::Normal,nullptr,&element);
1516  if(!sel.pResolvedObject) {
1517  if(!selList)
1518  FC_ERR("Sub-object " << sel.DocName << '#' << sel.FeatName << '.' << sel.SubName << " not found");
1519  return -1;
1520  }
1521  if(sel.pResolvedObject->testStatus(App::ObjectStatus::Remove))
1522  return -1;
1523  std::string subname;
1524  std::string prefix;
1525  if(pSubName && element) {
1526  prefix = std::string(pSubName, element-pSubName);
1527  if(!sel.elementName.newName.empty()) {
1528  // make sure the selected sub name is a new style if available
1529  subname = prefix + sel.elementName.newName;
1530  pSubName = subname.c_str();
1531  sel.SubName = subname;
1532  }
1533  }
1534  if(!selList)
1535  selList = &_SelList;
1536 
1537  if(!pSubName)
1538  pSubName = "";
1539 
1540  for (auto &s : *selList) {
1541  if (s.DocName==pDocName && s.FeatName==sel.FeatName) {
1542  if(s.SubName==pSubName)
1543  return 1;
1544  if (resolve > ResolveMode::OldStyleElement && boost::starts_with(s.SubName,prefix))
1545  return 1;
1546  }
1547  }
1548  if (resolve == ResolveMode::OldStyleElement) {
1549  for(auto &s : *selList) {
1550  if(s.pResolvedObject != sel.pResolvedObject)
1551  continue;
1552  if(!pSubName[0])
1553  return 1;
1554  if (!s.elementName.newName.empty()) {
1555  if (s.elementName.newName == sel.elementName.newName)
1556  return 1;
1557  }
1558  else if(s.SubName == sel.elementName.oldName)
1559  return 1;
1560  }
1561  }
1562  return 0;
1563 }

1584 void SelectionSingleton::slotDeletedObject(const App::DocumentObject& Obj)
