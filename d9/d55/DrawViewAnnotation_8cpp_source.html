100 void DrawViewAnnotation::handleChangedPropertyType(Base::XMLReader &reader, const char *TypeName, App::Property *prop)
101 // transforms properties that had been changed
102 {
103  // also check for changed properties of the base class
104  DrawView::handleChangedPropertyType(reader, TypeName, prop);
105 
106  // property LineSpace had the App::PropertyInteger and was changed to App::PropertyPercent
107  // the above change does not allow lines to be spaced wider than single space, so the change has been
108  // reverted, but we have to handle existing files that have a PropertyPercent
109  if (prop == &LineSpace && strcmp(TypeName, "App::PropertyPercent") == 0) {
110  App::PropertyPercent LineSpaceProperty;
111  // restore the PropertyPercent to be able to set its value
112  LineSpaceProperty.Restore(reader);
113  LineSpace.setValue(LineSpaceProperty.getValue());
114  }
115  // property MaxWidth had the App::PropertyFloat and was changed to App::PropertyLength
116  // the above change did not consider that -1 is a magic value that indicates the line
117  // should not be broken.
118  else if (prop == &MaxWidth && strcmp(TypeName, "App::PropertyLength") == 0) {
119  App::PropertyFloat MaxWidthProperty;
120  MaxWidthProperty.Restore(reader);
121  MaxWidth.setValue(MaxWidthProperty.getValue());
122  }
123 }

125 QRectF DrawViewAnnotation::getRect() const
126 {
127  double tSize = TextSize.getValue();
128  int lines = Text.getValues().size();
129  int chars = 1;
130  for (auto& l:Text.getValues()) {
131  if ((int)l.size() > chars) {
132  chars = (int)l.size();
133  }
134  }
135  int w = chars * std::max(1, (int)tSize);
136  int h = lines * std::max(1, (int)tSize);
137  return { 0, 0, getScale() * w, getScale() * h};
138 }
