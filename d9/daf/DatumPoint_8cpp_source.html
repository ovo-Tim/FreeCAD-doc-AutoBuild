68 void Point::makeShape()
69 {
70  // Create a shape, which will be used by Sketcher, attachables, and whatever. Them main function is to avoid a dependency of
71  // Sketcher on the PartDesign module
72  BRepBuilderAPI_MakeVertex builder(gp_Pnt(0,0,0));
73  if (!builder.IsDone())
74  return;
75  Part::TopoShape tshape(builder.Shape());
76  tshape.setPlacement(this->Placement.getValue());
