25 #move a simple view and its dependents (dimensions, balloons, etc) from fromPage to toPage
26 def simpleViewMove(view, fromPage, toPage, copy):
27  deps = [x for x in view.InList if x.isDerivedFrom("TechDraw::DrawView")]
28  if not copy:
29  for d in deps:
30  rc = fromPage.removeView(d)
31  rc = fromPage.removeView(view)
32  #redraw fromPage without view
33  fromPage.requestPaint()
34 
35  toPage.addView(view)
36  for d in deps:
37  toPage.addView(d)
38 
39  App.ActiveDocument.recompute() #update views & dependents if necessary
40  return

48 #move a projection group and all its dependents (items, dimensions, balloons, etc) from fromPage to toPage
49 def projGroupMove(view, fromPage, toPage, copy):
50  items = [x for x in view.Views if x.isDerivedFrom("TechDraw::DrawProjGroupItem")]
51  if not copy:
