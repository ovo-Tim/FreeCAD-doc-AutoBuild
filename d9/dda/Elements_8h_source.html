963 inline bool MeshGeomFacet::ContainedByOrIntersectBoundingBox(const Base::BoundBox3f& rclBB) const
964 {
965  // Test, if all corner points of the facet are on one of the 6 sides of the BB
966  if (!(GetBoundBox() && rclBB)) {
967  return false;
968  }
969 
970  // Test, whether Facet-BB is completely within BB
971  if (rclBB.IsInBox(GetBoundBox())) {
972  return true;
973  }
974 
975  // Test, whether one of the corner points is in BB
976  for (auto pnt : _aclPoints) {
977  if (rclBB.IsInBox(pnt)) {
978  return true;
979  }
980  }
981 
982  // "real" test for cutting
983  return (IntersectBoundingBox(rclBB));
984 }
