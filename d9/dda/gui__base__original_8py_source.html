60 
61  This class connects with the `draftToolBar` and `Snapper` classes
62  that are installed in the `FreeCADGui` namespace in order to set up some
63  properties of the running tools such as the task panel, the snapping
64  functions, and the grid trackers.
65 
66  It also connects with the `WorkingPlane` class.

79  def Activated(self, name="None", is_subtool=False):
80  """Execute when the command is called.
81 
82  If an active Gui Command exists, it will call the `finish` method
83  of it to terminate it.
84 
85  If no active Gui Command exists, it will proceed with configuration
86  of the tool. The child class that subclasses this class
87  then should continue with its own Activated method.
88 
89  Parameters
90  ----------
91  name: str, optional
92  It defaults to `'None'`.
93  It is the `featureName` of the object, to know what is being run.
94 
95  is_subtool: bool, optional
96  It defaults to `False`.
97  This is set to `True` when we want to modify an object
98  by using the mechanism of a `subtool`, introduced
99  through the `Draft_SubelementHighlight` command.
100  That is, first we run `Draft_SubelementHighlight`
101  then we can use `Draft_Move` while setting `is_subtool` to `True`.
102  """

148  def finish(self, cont=False):
149  """Finish the current command.
150 
151  These are general cleaning tasks that are performed
152  when terminating all commands.
153 
154  These include setting the node list to empty,
155  setting to `None` the active command,
156  turning off the graphical interface (task panel),
157  finishing the plane tracker, restoring the working plane,
158  turning off the snapper.
159 
160  If a callback is installed in the 3D view, the callback is removed,
161  and set to `None`.
162 
163  If the commit list is non-empty it will commit the instructions on
164  the list with `draftutils.todo.ToDo.delayCommit`,
165  and the list will be set to empty.
166  """
167  self.node = []
168  App.activeDraftCommand = None
169  if self.ui:
170  self.ui.offUi()
171  self.ui.sourceCmd = None
172  if hasattr(Gui, "Snapper"):
173  Gui.Snapper.off()
174  if self.planetrack:
175  self.planetrack.finalize()
176  self.wp._restore()
177  if self.commitList:
178  last_cmd = self.commitList[-1][1][-1]
179  if last_cmd.find("recompute") >= 0:
180  self.commitList[-1] = (self.commitList[-1][0], self.commitList[-1][1][:-1])
181  todo.ToDo.delayCommit(self.commitList)
182  todo.ToDo.delayAfter(Gui.doCommand, last_cmd)
183  else:
184  todo.ToDo.delayCommit(self.commitList)
185  self.commitList = []

204  def getStrings(self, addrot=None):
205  """Return useful strings that will be used to build commands.
206 
207  Returns
208  -------
209  str, str, str, str
210  A tuple of four strings that represent useful information.
211 
212  * the current working plane rotation quaternion as a string
213  * the support object if available as a string
214  * the list of nodes inside the `node` attribute as a string
215  * the string `'True'` or `'False'` depending on the fill mode
216  of the current tool
217  """
218  # Current plane rotation as a string
219  qr = self.wp.get_placement().Rotation.Q
220  qr = "({0}, {1}, {2}, {3})".format(qr[0], qr[1], qr[2], qr[3])
221 
222  # Support object
223  if self.support and params.get_param("useSupport"):
224  sup = 'FreeCAD.ActiveDocument.getObject'
225  sup += '("{}")'.format(self.support.Name)
226  else:
227  sup = 'None'
228 
229  # Contents of self.node
230  points = '['
231  for n in self.node:
232  if len(points) > 1:
233  points += ', '
234  points += DraftVecUtils.toString(n)
235  points += ']'
236 
237  # Fill mode
238  if self.ui:
239  fil = str(bool(self.ui.fillmode))
240  else:
241  fil = "True"
242 
243  return qr, sup, points, fil
