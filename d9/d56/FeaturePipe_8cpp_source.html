113  auto getSectionShape = [](App::DocumentObject* feature,
114  const std::vector<std::string>& subs) -> TopoDS_Shape {
115  if (!feature || !feature->isDerivedFrom(Part::Feature::getClassTypeId()))
116  throw Base::TypeError("Pipe: Invalid profile/section");
117 
118  auto subName = subs.empty() ? "" : subs.front();
119 
120  // only take the entire shape when we have a sketch selected, but
121  // not a point of the sketch
122  if (feature->isDerivedFrom(Part::Part2DObject::getClassTypeId())
123  && subName.compare(0, 6, "Vertex") != 0)
124  return static_cast<Part::Part2DObject*>(feature)->Shape.getValue();
125  else {
126  if (subName.empty())
127  throw Base::ValueError("Pipe: No valid subelement linked in Part::Feature");
128  return static_cast<Part::Feature*>(feature)->Shape.getShape().getSubShape(
129  subName.c_str());

133  auto addWiresToWireSections =
134  [](TopoDS_Shape& section, std::vector<std::vector<TopoDS_Shape>>& wiresections) -> size_t {
135  TopExp_Explorer ex;
136  size_t i = 0;
137  bool initialWireSectionsEmpty = wiresections.empty();
138  for (ex.Init(section, TopAbs_WIRE); ex.More(); ex.Next(), ++i) {
139  // if profile was just a point then this is where we can first set our list
140  if (i >= wiresections.size()) {
141  if (initialWireSectionsEmpty)
142  wiresections.emplace_back(1, ex.Current());
143  else
144  throw Base::ValueError(
145  "Pipe: Sections need to have the same amount of inner wires (except "
146  "profile and last section, which can be points)");
147  }
148  else
149  wiresections[i].push_back(TopoDS::Wire(ex.Current()));
150  }
151  return i;
152  };
153 
154  // TODO: currently we can only allow planar faces, so add that check.
155  // The reason for this is that with other faces in front, we could not use the
156  // current simulate approach and build the start and end face from the wires.
157  // As the shell begins always at the spine and not the profile, the sketchshape
158  // cannot be used directly as front face. We would need a method to translate
159  // the front shape to match the shell starting position somehow...
160  std::vector<TopoDS_Wire> wires;
161  TopoDS_Shape profilePoint;
162 
163  // if the Base property has a valid shape, fuse the pipe into it
164  TopoShape base;
165  try {
166  base = getBaseTopoShape();
167  } catch (const Base::Exception&) {
168  base = TopoShape();
169  }
170 
171  try {
172  // setup the location
173  this->positionByPrevious();
174  TopLoc_Location invObjLoc = this->getLocation().Inverted();
175  if (!base.isNull())
176  base.move(invObjLoc);
177 
178  // setup the profile section
179  TopoDS_Shape profileShape = getSectionShape(Profile.getValue(),
180  Profile.getSubValues());
181  if (profileShape.IsNull())
182  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Pipe: Could not obtain profile shape"));
183 
184  // build the paths
185  App::DocumentObject* spine = Spine.getValue();
186  if (!(spine && spine->isDerivedFrom<Part::Feature>()))
187  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "No spine linked"));
188 
189  std::vector<std::string> subedge = Spine.getSubValues();
190  TopoDS_Shape path;
191  const Part::TopoShape& shape = static_cast<Part::Feature*>(spine)->Shape.getValue();
192  buildPipePath(shape, subedge, path);
193  path.Move(invObjLoc);
194 
195  // auxiliary
196  TopoDS_Shape auxpath;
197  if (Mode.getValue() == 3) {
198  App::DocumentObject* auxspine = AuxillerySpine.getValue();
199  if (!(auxspine && auxspine->isDerivedFrom<Part::Feature>()))
200  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "No auxiliary spine linked."));
201  std::vector<std::string> auxsubedge = AuxillerySpine.getSubValues();
202 
203  const Part::TopoShape& auxshape =
204  static_cast<Part::Feature*>(auxspine)->Shape.getValue();
205  buildPipePath(auxshape, auxsubedge, auxpath);
206  auxpath.Move(invObjLoc);
207  }
208 
209  // build up multisections
210  auto multisections = Sections.getSubListValues();
211  std::vector<std::vector<TopoDS_Shape>> wiresections;
212 
213  size_t numWires = addWiresToWireSections(profileShape, wiresections);
214  if (numWires == 0) {
215  // profileShape had no wires so only other valid option is single point section
216  TopExp_Explorer ex;
217  size_t i = 0;
218  for (ex.Init(profileShape, TopAbs_VERTEX); ex.More(); ex.Next(), ++i)
219  profilePoint = ex.Current();
220  if (i > 1)
221  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception",
222  "Pipe: Only one isolated point is needed if using a sketch with isolated "
223  "points for section"));
224  }
225 
226  if (!profilePoint.IsNull() && (Transformation.getValue() != 1 || multisections.empty()))
227  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception",
228  "Pipe: At least one section is needed when using a single point for profile"));
229 
230  // maybe we need a scaling law
231  Handle(Law_Function) scalinglaw;
232 
233  bool isLastSectionVertex = false;
234 
235  // see if we shall use multiple sections
236  if (Transformation.getValue() == 1) {
237  // TODO: we need to order the sections to prevent occ from crashing,
238  // as makepipeshell connects the sections in the order of adding
239  for (auto& subSet : multisections) {
240  if (!subSet.first->isDerivedFrom(Part::Feature::getClassTypeId()))
241  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception",
242  "Pipe: All sections need to be part features"));
243 
244  // if the section is an object's face then take just the face
245  TopoDS_Shape shape = getSectionShape(subSet.first, subSet.second);
246  if (shape.IsNull())
247  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception",
248  "Pipe: Could not obtain section shape"));
249 
250  size_t nWiresAdded = addWiresToWireSections(shape, wiresections);
251  if (nWiresAdded == 0) {
252  TopExp_Explorer ex;
253  size_t i = 0;
254  for (ex.Init(shape, TopAbs_VERTEX); ex.More(); ex.Next(), ++i) {
255  if (isLastSectionVertex)
256  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception",
257  "Pipe: Only the profile and last section can be vertices"));
258  isLastSectionVertex = true;
259  for (auto& wires : wiresections)
260  wires.push_back(ex.Current());
261  }
262  }
263 
264  if (!isLastSectionVertex && nWiresAdded < wiresections.size())
265  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception",
266  "Multisections need to have the same amount of inner wires as the base "
267  "section"));
268  }
269  }
270  /*//build the law functions instead
271  else if (Transformation.getValue() == 2) {
272  if (ScalingData.getValues().size()<1)
273  return new App::DocumentObjectExecReturn("No valid data given for linear scaling mode");
274 
275  Handle(Law_Linear) lin = new Law_Linear();
276  lin->Set(0, 1, 1, ScalingData[0].x);
277 
278  scalinglaw = lin;
279  }
280  else if (Transformation.getValue() == 3) {
281  if (ScalingData.getValues().size()<1)
282  return new App::DocumentObjectExecReturn("No valid data given for S-shape scaling mode");
283 
284  Handle(Law_S) s = new Law_S();
285  s->Set(0, 1, ScalingData[0].y, 1, ScalingData[0].x, ScalingData[0].z);
286 
287  scalinglaw = s;
288  }*/
289 
290  // Verify that path is not a null shape
291  if (path.IsNull())
292  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP(
293  "Exception", "Path must not be a null shape"));
294 
295  // build all shells
296  std::vector<TopoDS_Shape> shells;
297 
298  TopoDS_Shape copyProfilePoint(profilePoint);
299  if (!profilePoint.IsNull())
300  copyProfilePoint.Move(invObjLoc);
301 
302  std::vector<TopoDS_Wire> frontwires, backwires;
303  for (auto& wires : wiresections) {
304  BRepOffsetAPI_MakePipeShell mkPS(TopoDS::Wire(path));
305  setupAlgorithm(mkPS, auxpath);
306 
307  if (!scalinglaw) {
308  if (!profilePoint.IsNull())
309  mkPS.Add(copyProfilePoint);
310 
311  for (auto& wire : wires) {
312  wire.Move(invObjLoc);
313  mkPS.Add(wire);
314  }
315  }
316  else {
317  if (!profilePoint.IsNull())
318  mkPS.SetLaw(copyProfilePoint, scalinglaw);
319 
320  for (auto& wire : wires) {
321  wire.Move(invObjLoc);
322  mkPS.SetLaw(wire, scalinglaw);
323  }
324  }
325 
326  if (!mkPS.IsReady())
327  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Pipe could not be built"));
328 
329  shells.push_back(mkPS.Shape());
330 
331  if (!mkPS.Shape().Closed()) {
332  // shell is not closed - use simulate to get the end wires
333  TopTools_ListOfShape sim;
334  mkPS.Simulate(2, sim);
335 
336  // Note that while we call them front and back, these sections
337  // appear to correspond to the front or back of the path. When one
338  // or both ends of the pipe are points, one or both of these wires
339  // (and eventually faces) will be null.
340  frontwires.push_back(TopoDS::Wire(sim.First()));
341  backwires.push_back(TopoDS::Wire(sim.Last()));
342  }
343  }
344 
345  BRepBuilderAPI_MakeSolid mkSolid;
346 
347  if (!frontwires.empty() || !backwires.empty()) {
348  BRepBuilderAPI_Sewing sewer;
349  sewer.SetTolerance(Precision::Confusion());
350 
351  // build the end faces, sew the shell and build the final solid
352  if (!frontwires.empty()) {
353  TopoDS_Shape front = Part::FaceMakerCheese::makeFace(frontwires);
354  sewer.Add(front);
355  }
356  if (!backwires.empty()) {
357  TopoDS_Shape back = Part::FaceMakerCheese::makeFace(backwires);
358  sewer.Add(back);
359  }
360  for (TopoDS_Shape& s : shells)
361  sewer.Add(s);
362 
363  sewer.Perform();
364  mkSolid.Add(TopoDS::Shell(sewer.SewedShape())); } else {
365  // shells are already closed - add them directly
366  for (TopoDS_Shape& s : shells) {
367  mkSolid.Add(TopoDS::Shell(s));
368  }
369  }
370 
371  if (!mkSolid.IsDone())
372  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Result is not a solid"));
373 
374  TopoDS_Shape result = mkSolid.Shape();
375  BRepClass3d_SolidClassifier SC(result);
376  SC.PerformInfinitePoint(Precision::Confusion());
377  if (SC.State() == TopAbs_IN) {
378  result.Reverse();
379  }
380 
381  //result.Move(invObjLoc);
382  AddSubShape.setValue(result); // Converts result to a TopoShape, but no tag.
383 
384  if (base.isNull()) {
385  if (getAddSubType() == FeatureAddSub::Subtractive)
386  return new App::DocumentObjectExecReturn(
387  QT_TRANSLATE_NOOP("Exception", "Pipe: There is nothing to subtract from"));
388 
389  // store shape before refinement
390  this->rawShape = result;
391  auto ts_result = refineShapeIfActive(result);
392  Shape.setValue(getSolid(ts_result));
393  return App::DocumentObject::StdReturn;
394  }
395 
396  if (getAddSubType() == FeatureAddSub::Additive) {
397 
398  FCBRepAlgoAPI_Fuse mkFuse(base.getShape(), result);
399  if (!mkFuse.IsDone())
400  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Adding the pipe failed"));
401  // we have to get the solids (fuse sometimes creates compounds)
402  TopoShape boolOp = this->getSolid(mkFuse.Shape());
403  // lets check if the result is a solid
404  if (boolOp.isNull())
405  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Resulting shape is not a solid"));
406 
407  int solidCount = countSolids(boolOp.getShape());
408  if (solidCount > 1) {
409  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception",
410  "Result has multiple solids: that is not currently supported."));
411  }
412 
413  // store shape before refinement
414  this->rawShape = boolOp;
415  boolOp = refineShapeIfActive(boolOp);
416  Shape.setValue(getSolid(boolOp));
417  }
418  else if (getAddSubType() == FeatureAddSub::Subtractive) {
419 
420  FCBRepAlgoAPI_Cut mkCut(base.getShape(), result);
421  if (!mkCut.IsDone())
422  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Subtracting the pipe failed"));
423  // we have to get the solids (fuse sometimes creates compounds)
424  TopoShape boolOp = this->getSolid(mkCut.Shape());
425  // lets check if the result is a solid
426  if (boolOp.isNull())
427  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Resulting shape is not a solid"));
428 
429  int solidCount = countSolids(boolOp.getShape());
430  if (solidCount > 1) {
431  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception",
432  "Result has multiple solids: that is not currently supported."));
433  }
434 
435  // store shape before refinement
436  this->rawShape = boolOp;
437  boolOp = refineShapeIfActive(boolOp);
438  Shape.setValue(getSolid(boolOp));
439  }
440 
441  return App::DocumentObject::StdReturn;
442  }
443  catch (Standard_Failure& e) {
444 
445  return new App::DocumentObjectExecReturn(e.GetMessageString());
446  }
447  catch (...) {
448  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "A fatal error occurred when making the pipe"));
449  }

454  mkPipeShell.SetTolerance(Precision::Confusion());

475  mkPipeShell.SetMode(true);
476  break;
477  case 3:
478  auxiliary = true;
479  break;
480  case 4:
481  mkPipeShell.SetMode(gp_Dir(bVec.x, bVec.y, bVec.z));
482  break;
483  }
484 
485  if (auxiliary) {
486  mkPipeShell.SetMode(TopoDS::Wire(auxshape), AuxilleryCurvelinear.getValue());
487  // mkPipeShell.SetMode(TopoDS::Wire(auxshape), AuxilleryCurvelinear.getValue(),

544 void Pipe::buildPipePath(const Part::TopoShape& shape, const std::vector<std::string>& subedge,
545  TopoDS_Shape& path)
546 {
547  if (!shape.getShape().IsNull()) {
548  try {
549  if (!subedge.empty()) {
550  //if (SpineTangent.getValue())
551  //getContinuousEdges(shape, subedge);
552 
553  BRepBuilderAPI_MakeWire mkWire;
554  for (const auto & it : subedge) {
555  TopoDS_Shape subshape = shape.getSubShape(it.c_str());
556  mkWire.Add(TopoDS::Edge(subshape));
557  }
558  path = mkWire.Wire();
559  }
560  else if (shape.getShape().ShapeType() == TopAbs_EDGE) {
561  path = shape.getShape();
562  }
563  else if (shape.getShape().ShapeType() == TopAbs_WIRE) {
564  BRepBuilderAPI_MakeWire mkWire(TopoDS::Wire(shape.getShape()));
565  path = mkWire.Wire();
566  }
567  else if (shape.getShape().ShapeType() == TopAbs_COMPOUND) {
568  TopoDS_Iterator it(shape.getShape());
569  for (; it.More(); it.Next()) {
570  if (it.Value().IsNull())
571  throw Base::ValueError(QT_TRANSLATE_NOOP("Exception", "Invalid element in spine."));
572  if ((it.Value().ShapeType() != TopAbs_EDGE) &&
573  (it.Value().ShapeType() != TopAbs_WIRE)) {
574  throw Base::TypeError(QT_TRANSLATE_NOOP("Exception", "Element in spine is neither an edge nor a wire."));
575  }
576  }
577 
578  Handle(TopTools_HSequenceOfShape) hEdges = new TopTools_HSequenceOfShape();
579  Handle(TopTools_HSequenceOfShape) hWires = new TopTools_HSequenceOfShape();
580  for (TopExp_Explorer xp(shape.getShape(), TopAbs_EDGE); xp.More(); xp.Next())
581  hEdges->Append(xp.Current());
582 
583  ShapeAnalysis_FreeBounds::ConnectEdgesToWires(
584  hEdges, Precision::Confusion(), Standard_True, hWires);
585  int len = hWires->Length();
586  if (len != 1)
587  throw Base::ValueError(QT_TRANSLATE_NOOP("Exception", "Spine is not connected."));
588  path = hWires->Value(1);
589  }
590  else {
591  throw Base::TypeError(QT_TRANSLATE_NOOP("Exception", "Spine is neither an edge nor a wire."));
592  }
