71 void ViewProviderDatum::attach(App::DocumentObject* pcObject)
72 {
73  ViewProviderGeometryObject::attach(pcObject);
74 
75 
76  // Create an external separator
77  auto sep = new SoSeparator();
78 
79  // Add material from the base class
80  sep->addChild(pcShapeMaterial);
81 
82  // Bind same material to all part
83  auto matBinding = new SoMaterialBinding;
84  matBinding->value = SoMaterialBinding::OVERALL;
85  sep->addChild(matBinding);
86 
87  // Setup font size
88  auto font = new SoFont();
89  static const float size = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/View")->GetFloat("DatumFontSize", 15.0);
90  font->size.setValue(size);
91  sep->addChild(font);
92 
93  // Create the selection node
94  auto highlight = new SoFCSelection();
95  highlight->applySettings();
96  if (!Selectable.getValue()) {
97  highlight->selectionMode = Gui::SoFCSelection::SEL_OFF;
98  }
99  highlight->objectName = getObject()->getNameInDocument();
100  highlight->documentName = getObject()->getDocument()->getName();
101  highlight->style = SoFCSelection::EMISSIVE_DIFFUSE;
102 
103  // Visible features
104  auto visible = new SoSeparator();
105  // Style for normal (visible) lines
106  auto style = new SoDrawStyle();
107  style->lineWidth = lineThickness;
108  visible->addChild(style);
109 
110  // Visible lines
111  visible->addChild(pRoot);
112 
113  // Hidden features
114  auto hidden = new SoAnnotation();
115 
116  // Style for hidden lines
117  style = new SoDrawStyle();
118  style->lineWidth = lineThickness;
119  style->linePattern.setValue(0xFF00); // (dash-skip)
120  hidden->addChild(style);
121 
122  // Hidden lines
123  hidden->addChild(pRoot);
124 
125  visible->addChild(hidden);
126 
127  sep->addChild(visible);
128 
129  soScale->setPart("shape", sep);
130  resetTemporarySize();
131 
132  highlight->addChild(soScale);
133 
134  addDisplayMaskMode(highlight, "Base");
135 }
