42 def downgrade(objects, delete=False, force=None):
43  """Downgrade the given objects.
44 
45  This is a counterpart to `upgrade`.
46 
47  Parameters
48  ----------
49  objects: Part::Feature or list
50  A single object to downgrade or a list
51  containing various such objects.
52 
53  delete: bool, optional
54  It defaults to `False`.
55  If it is `True`, the old objects are deleted, and only the resulting
56  object is kept.
57 
58  force: str, optional
59  It defaults to `None`.
60  Its value can be used to force a certain method of downgrading.
61  It can be any of: `'explode'`, `'shapify'`, `'subtr'`, `'splitFaces'`,
62  `'cut2'`, `'getWire'`, `'splitWires'`, or `'splitCompounds'`.
63 
64  Returns
65  -------
66  tuple
67  A tuple containing two lists, a list of new objects
68  and a list of objects to be deleted.
69 
70  None
71  If there is a problem it will return `None`.
72 
73  See Also
74  --------
75  upgrade
76  """
77  _name = "downgrade"
78 
79  if not isinstance(objects, list):
80  objects = [objects]
81 
82  delete_list = []
83  add_list = []
84  doc = App.ActiveDocument

86  # actions definitions
87  def explode(obj):
88  """Explode a Draft block."""
89  pl = obj.Placement
90  for o in obj.Components:
91  o.Placement = pl.multiply(o.Placement)
92  if App.GuiUp:
93  o.ViewObject.Visibility = True
94  delete_list.append(obj)
95  return True
96 
97  def cut2(objects):
98  """Cut first object from the last one."""
99  newobj = cut.cut(objects[0], objects[1])
100  if newobj:
101  add_list.append(newobj)
102  return newobj
103  return None

105  def splitCompounds(objects):
106  """Split solids contained in compound objects into new objects."""
107  result = False
108  for o in objects:
109  if o.Shape.Solids:
110  for s in o.Shape.Solids:
111  newobj = doc.addObject("Part::Feature", "Solid")
112  newobj.Shape = s
113  add_list.append(newobj)
114  result = True
115  delete_list.append(o)
116  return result

118  def splitFaces(objects):
119  """Split faces contained in objects into new objects."""
120  result = False
121  preserveFaceColor = params.get_param("preserveFaceColor")
122  preserveFaceNames = params.get_param("preserveFaceNames")
123  for o in objects:
124  if App.GuiUp and preserveFaceColor and o.ViewObject:
125  voDColors = o.ViewObject.DiffuseColor
126  else:
127  voDColors = None
128  oLabel = o.Label if hasattr(o, 'Label') else ""
129  if o.Shape.Faces:
130  for ind, f in enumerate(o.Shape.Faces):
131  newobj = doc.addObject("Part::Feature", "Face")
132  newobj.Shape = f
133  if preserveFaceNames:
134  newobj.Label = "{} {}".format(oLabel, newobj.Label)
135  if App.GuiUp and preserveFaceColor and voDColors:
136  # At this point, some single-color objects might have
137  # just a single value in voDColors for all faces,
138  # so we handle that
139  if ind < len(voDColors):
140  tcolor = voDColors[ind]
141  else:
142  tcolor = voDColors[0]
143  # does is not applied visually on its own
144  # just in case
145  newobj.ViewObject.DiffuseColor[0] = tcolor
146  # this gets applied, works by itself too
147  newobj.ViewObject.ShapeColor = tcolor
148  add_list.append(newobj)
149  result = True
150  delete_list.append(o)
151  return result

153  def subtr(objects):
154  """Subtract objects from the first one."""
155  faces = []
156  for o in objects:
157  if o.Shape.Faces:
158  faces.extend(o.Shape.Faces)
159  delete_list.append(o)
160  u = faces.pop(0)
161  for f in faces:
162  u = u.cut(f)
163  if not u.isNull():
164  newobj = doc.addObject("Part::Feature", "Subtraction")
165  newobj.Shape = u
166  add_list.append(newobj)
167  return newobj
168  return None

170  def getWire(obj):

172  result = False

194  def delete_object(obj):
195  if obj.FullName == "?": # Already deleted.
196  return
197  # special case: obj is a body or belongs to a body:
198  if obj.TypeId == "PartDesign::Body":
199  obj.removeObjectsFromDocument()
200  if hasattr(obj, "_Body") and obj._Body is not None:
201  obj = obj._Body
202  obj.removeObjectsFromDocument()
203  else:
204  for parent in obj.InList:
205  if parent.TypeId == "PartDesign::Body" \
206  and obj in parent.Group:
207  obj = parent
208  obj.removeObjectsFromDocument()
209  break
210  doc.removeObject(obj.Name)
211 
212  # analyzing objects
213  faces = []
214  edges = []
215  onlyedges = True
216  parts = []
217  solids = []
218  result = None
219 
220  for o in objects:
221  if hasattr(o, 'Shape'):
222  for s in o.Shape.Solids:
223  solids.append(s)
224  for f in o.Shape.Faces:
225  faces.append(f)
226  for e in o.Shape.Edges:
227  edges.append(e)
228  if o.Shape.ShapeType != "Edge":
229  onlyedges = False
230  parts.append(o)

233  if force:
234  if force in ("explode", "shapify", "subtr", "splitFaces",
235  "cut2", "getWire", "splitWires"):
236  # TODO: Using eval to evaluate a string is not ideal
237  # and potentially a security risk.

239  # Best case, a series of if-then statements.
240  shapify = utils.shapify

244  result = None
245  else:

249  result = explode(objects[0])

252

254  elif len(objects) == 1 \

257  result = splitCompounds(objects)
258  # print(result)
259  if result:
260  _msg(translate("draft","Found 1 multi-solids compound: exploding it"))
261 
262  # special case, we have one parametric object: we "de-parametrize" it
263  elif len(objects) == 1 \
264  and hasattr(objects[0], "Shape") \
265  and hasattr(objects[0], "Base") \
266  and not objects[0].isDerivedFrom("PartDesign::Feature"):
267  result = utils.shapify(objects[0])
268  if result:
269  _msg(translate("draft","Found 1 parametric object: breaking its dependencies"))
270  add_list.append(result)
271  # delete_list.append(objects[0])
272 
273  # we have only 2 objects: cut 2nd from 1st
274  elif len(objects) == 2:
275  result = cut2(objects)
276  if result:
277  _msg(translate("draft","Found 2 objects: subtracting them"))
278 
279  elif len(faces) > 1:
280  # one object with several faces: split it
281  if len(objects) == 1:
282  result = splitFaces(objects)
283  if result:
284  _msg(translate("draft","Found several faces: splitting them"))
285  # several objects: remove all the faces from the first one
286  else:
287  result = subtr(objects)
288  if result:
289  _msg(translate("draft","Found several objects: subtracting them from the first one"))
290 
291  # only one face: we extract its wires
292  elif len(faces) > 0:
293  result = getWire(objects[0])
294  if result:
295  _msg(translate("draft","Found 1 face: extracting its wires"))
296 
297  # no faces: split wire into single edges
298  elif not onlyedges:
299  result = splitWires(objects)
300  if result:
301  _msg(translate("draft","Found only wires: extracting their edges"))
302 
303  # no result has been obtained
304  if not result:
305  _msg(translate("draft","No more downgrade possible"))
306 
307  if delete:
308  for o in delete_list:
309  delete_object(o)
310  delete_list = []
311 
312  gui_utils.select(add_list)
313  return add_list, delete_list
