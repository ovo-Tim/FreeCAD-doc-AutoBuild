26 """Provides the object code for the dimension objects.
27 
28 This includes the `LinearDimension` and `AgularDimension`.
29 The first one measures a distance between two points or vertices
30 in an object; it includes radial dimensions of circular arcs.
31 The second one creates an arc between two straight lines to measure
32 the angle between both.
33 
34 To Do
35 -----
36 1. Currently the angular dimension does not use linked geometrical
37 elements, meaning that it cannot update its `Angle` by picking lines or edges
38 from objects. If fact, `LinkedGeometry` is hidden to prevent the user
39 from picking any object.
40 
41 At the moment the user must manually modify `FirstAngle` and `LastAngle`
42 to obtain a new `Angle`, but since the values are manually entered
43 the result is not parametrically tied to any actual object in the document.
44 
45 We introduced a function `measure_two_obj_angles` to calculate
46 the corresponding parameters from a pair of objects and their edges.
47 Currently this function is deactivated because we don't consider it
48 to be ready; it is there for testing purposes only.
49 This needs to be improved because at the moment it only gives
50 one possible angle. We should be able to get the four angles
51 of a two-line intersection. Maybe a new property is required
52 to indicate the quadrant to choose and display.
53 
54 2. In general, the `LinkedGeometry` property must be changed in type,
55 as it does not need to be an `App::PropertyLinkSubList`.
56 A `LinkSubList` is to select multiple subelements (vertices, edges)
57 from multiple objects (two lines). However, since we typically measure
58 a single object, for example, a single line or circle, the subelements
59 that we can choose must belong to this object only.
60 
61 Therefore, just like in the case of the `PathArray` class the best property
62 that could be used is `App::PropertyLinkSub`.
63 Then in the property editor we will be unable to select more than one object
64 thus preventing errors of the subelements not matching the measured object.
65 
66 3. Currently the `LinearDimension` class is able to measure the distance
67 between two arbitrary vertices in two distinct objects.
68 For this case `App::PropertyLinkSubList` is in fact the right property
69 to use, however, neither the `make_dimension` functions
70 nor the Gui Command are set up to use this type of information.
71 This has to be done manually by picking the two objects and the two vertices
72 in the property editor. That is, this functionality is not entirely intuitive,
73 so it is somewhat 'hidden' from the user.
74 
75 So, the make function and the Gui Command should be expanded to consider
76 this case.
77 
78 Another possibility would be to use one property (LinkSub) for single-object
79 measurements (linear, radial), and a second property (LinkSubList)
80 for two-object measurements (linear, angular). This would require adjustments
81 to the `execute` method to handle both cases properly. It may be necessary
82 to have another property to control which type to use.
83 
84 4. The `Support` property is not used at all, so it should be removed.
85 It is just set at creation time by the `make_dimension` function
86 but it actually isn't used in the `execute` code.
87 
88 5. In fact the `DimensionBase` class is not the best base class
89 than can be used as parent for all dimensions because it defines `Normal`,
90 `Support`, and `LinkedGeometry`, which aren't used in all cases.
91 In some of the derived classes, these properties are hidden.
92 
93 So, together with what is explained in point 3, we probably need to use
94 a more generic base class, while at the same time improve the way
95 the link properties are used.
96 """

323  def execute(self, obj):
324  """Execute when the object is created or recomputed.
325 
326  Set start point and end point according to the linked geometry
327  and the number of subelements.
328 
329  If it has one subelement, we assume a straight edge or a circular edge.
330  If it has two subelements, we assume a straight edge (two vertices).
331  """
332  if obj.LinkedGeometry:
333  if len(obj.LinkedGeometry) == 1:
334  linked_obj = obj.LinkedGeometry[0][0]
335  sub_list = obj.LinkedGeometry[0][1]
336 
337  if len(sub_list) == 1:
338  # If it has one subelement, we assume an edge
339  # that can be a straight line, or a circular edge
340  subelement = sub_list[0]
341  (obj.Start,
342  obj.End) = measure_one_obj_edge(linked_obj,
343  subelement,
344  obj.Dimline,
345  obj.Diameter)
346  elif len(sub_list) == 2:
347  # If it has two subelements, we assume a straight edge
348  # that is measured by two vertices
349  (obj.Start,
350  obj.End) = measure_one_obj_vertices(linked_obj,
351  sub_list)
352 
353  elif len(obj.LinkedGeometry) == 2:
354  # If the list has two objects, it measures the distance
355  # between the two vertices in those two objects
356  (obj.Start,
357  obj.End) = measure_two_objects(obj.LinkedGeometry[0],
358  obj.LinkedGeometry[1])
359 
360  # Update the distance property by comparing the floats
361  # with the precision
362  net_length = (obj.Start.sub(obj.End)).Length
363  rounded_1 = round(obj.Distance.Value, utils.precision())
364  rounded_2 = round(net_length, utils.precision())
365 
366  if rounded_1 != rounded_2:
367  obj.Distance = net_length
368 
369  # The lines and text are created in the viewprovider, so we should
370  # update it whenever the object is recomputed
371  if App.GuiUp and obj.ViewObject:
372  obj.ViewObject.update()

375 def measure_one_obj_edge(obj, subelement, dim_point, diameter=False):
376  """Measure one object with one subelement, a straight or circular edge.
377 
378  Parameters
379  ----------
380  obj: Part::Feature
381  The object that is measured.
382 
383  subelement: str
384  The subelement that is measured, for example, `'Edge1'`.
385 
386  dim_line: Base::Vector3
387  A point through which the dimension goes through.
388  """
389  start = App.Vector()
390  end = App.Vector()
391 
392  if "Edge" in subelement:
393  n = int(subelement[4:]) - 1
394  edge = obj.Shape.Edges[n]
395 
396  if DraftGeomUtils.geomType(edge) == "Line":
397  start = edge.Vertexes[0].Point
398  end = edge.Vertexes[-1].Point
399  elif DraftGeomUtils.geomType(edge) == "Circle":
400  center = edge.Curve.Center
401  radius = edge.Curve.Radius
402  axis = edge.Curve.Axis
403  dim_line = dim_point.sub(center)
404 
405  # The ray is projected to the plane on which the circle lies,
406  # but if the projection is not successful, try in the other planes
407  ray = dim_line.projectToPlane(App.Vector(0, 0, 0), axis)
408 
409  if ray.Length == 0:
410  ray = axis.cross(App.Vector(1, 0, 0))
411  if ray.Length == 0:
412  ray = axis.cross(App.Vector(0, 1, 0))
413 
414  # The ray is made as large as the arc's radius
415  # and optionally the diameter
416  ray = DraftVecUtils.scaleTo(ray, radius)
417 
418  if diameter:
419  # The start and end points lie on the arc
420  start = center.add(ray.negative())
421  end = center.add(ray)
422  else:
423  # The start is th center and the end lies on the arc
424  start = center
425  end = center.add(ray)
426 
427  return start, end

618 def measure_two_obj_angles(link_sub_1, link_sub_2):
619  """Measure two edges from two different objects to measure the angle.
620 
621  This function is a prototype because it does not determine all possible
622  starting and ending angles that could be used to draw the dimension line,
623  which is a circular arc.
624 
625  Parameters
626  ----------
627  link_sub_1: tuple
628  A tuple containing one object and a list of subelement strings,
629  which may be empty. Only the first subelement is considered, which
630  must be an edge.
631  ::
632  link_sub_1 = (obj1, ['EdgeN', ...])
633 
634  link_sub_2: tuple
635  Same.
636  """
637  start = 0
638  end = 0
639 
640  obj1 = link_sub_1[0]
641  lsub1 = link_sub_1[1]
642 
643  obj2 = link_sub_2[0]
644  lsub2 = link_sub_2[1]
645 
646  # The subelement list may be empty so we test it first
647  # and pick only the first item
648  if lsub1 and lsub2:
649  subelement1 = lsub1[0]
650  subelement2 = lsub2[0]
651 
652  if "Edge" in subelement1 and "Edge" in subelement2:
653  n1 = int(subelement1[4:]) - 1
654  n2 = int(subelement2[4:]) - 1
655  start = obj1.Shape.Edges[n1].Curve.Direction
656  end = obj2.Shape.Edges[n2].Curve.Direction
657 
658  # We get the angle from the U axis of the working plane
659  wp = WorkingPlane.get_working_plane(update=False)
660  start_r = DraftVecUtils.angle(start, wp.u)
661  end_r = DraftVecUtils.angle(end, wp.u)
662  start = math.degrees(start_r)
663  end = math.degrees(end_r)
664 
665  # We make the angle positive because when tested, some errors
666  # were produced in the code that calculates the 'Angle'.
667  # This code is actually inside the viewprovider.
668  if start < 0:
669  start = abs(start)
670  if end < 0:
671  end = abs(end)
672 
673  return start, end
