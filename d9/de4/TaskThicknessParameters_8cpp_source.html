64 void TaskThicknessParameters::initControls()
65 {
66  auto thickness = getObject<PartDesign::Thickness>();
67  double a = thickness->Value.getValue();
68 
69  ui->Value->setMinimum(0.0);
70  ui->Value->setValue(a);
71  ui->Value->selectAll();
72  QMetaObject::invokeMethod(ui->Value, "setFocus", Qt::QueuedConnection);
73 
74  // Bind input fields to properties
75  ui->Value->bind(thickness->Value);
76 
77  bool r = thickness->Reversed.getValue();
78  ui->checkReverse->setChecked(r);
79 
80  bool i = thickness->Intersection.getValue();
81  ui->checkIntersection->setChecked(i);
82 
83  std::vector<std::string> strings = thickness->Base.getSubValues();
84  for (const auto& string : strings) {
85  ui->listWidgetReferences->addItem(QString::fromStdString(string));
86  }
87 
88  setupConnections();
89 
90  int mode = static_cast<int>(thickness->Mode.getValue());
91  ui->modeComboBox->setCurrentIndex(mode);
92 
93  int join = static_cast<int>(thickness->Join.getValue());
94  ui->joinComboBox->setCurrentIndex(join);
95 
96  if (strings.empty()) {
97  setSelectionMode(refSel);
98  }
99  else {
100  hideOnError();
101  }
102 }
