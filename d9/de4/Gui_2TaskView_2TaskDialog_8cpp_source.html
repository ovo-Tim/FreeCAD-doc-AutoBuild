97 bool TaskDialog::canClose() const
98 {
99  QMessageBox msgBox;
100  msgBox.setText(tr("A dialog is already open in the task panel"));
101  msgBox.setInformativeText(QObject::tr("Do you want to close this dialog?"));
102  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
103  msgBox.setDefaultButton(QMessageBox::Yes);
104  int ret = msgBox.exec();
105  return (ret == QMessageBox::Yes);
106 }

108 void TaskDialog::associateToObject3dView(App::DocumentObject* obj)
109 {
110  if (!obj) {
111  return;
112  }
113 
114  Gui::Document* guiDoc = Gui::Application::Instance->getDocument(obj->getDocument());
115  auto* vp = Gui::Application::Instance->getViewProvider(obj);
116  auto* vpdo = static_cast<Gui::ViewProviderDocumentObject*>(vp);
117  auto* view = guiDoc->openEditingView3D(vpdo);
118 
119  if (!view) {
120  return;
121  }
122 
123  setAssociatedView(view);
124  setAutoCloseOnClosedView(true);
125 }
