97 /*
98  * When building up a mesh then usually the class MeshBuilder is used. This
99  * class uses internally a std::set<MeshPoint> which uses the '<' operator of
100  * MeshPoint to sort the points. Thus to be consistent (and avoid using the
101  * '==' operator of MeshPoint) we use the same operator when comparing the
102  * points in the function object.
103  */
104 struct Vertex_EqualTo
105 {
106  bool operator()(const VertexIterator& x, const VertexIterator& y) const
107  {
108  if ((*x) < (*y)) {
109  return false;
110  }
111  if ((*y) < (*x)) {
112  return false;
113  }
114  return true;
115  }
116 };

121  {

175 bool MeshFixDuplicatePoints::Fixup()
176 {
177  // Note: We must neither use map or set to get duplicated indices because
178  // the sort algorithms deliver different results compared to std::sort of
179  // a vector.
180  const MeshPointArray& rPoints = _rclMesh.GetPoints();
181  std::vector<VertexIterator> vertices;
182  vertices.reserve(rPoints.size());
183  for (auto it = rPoints.begin(); it != rPoints.end(); ++it) {
184  vertices.push_back(it);
185  }
186 
187  // get the indices of adjacent vertices which have the same coordinates
188  std::sort(vertices.begin(), vertices.end(), Vertex_Less());
189 
190  Vertex_EqualTo pred;
191  std::vector<VertexIterator>::iterator next = vertices.begin();
192  std::map<PointIndex, PointIndex> mapPointIndex;
193  std::vector<PointIndex> pointIndices;
194  while (next < vertices.end()) {
195  next = std::adjacent_find(next, vertices.end(), pred);
196  if (next < vertices.end()) {
197  auto first = next;
198  PointIndex first_index = *first - rPoints.begin();
199  ++next;
200  while (next < vertices.end() && pred(*first, *next)) {
201  PointIndex next_index = *next - rPoints.begin();
202  mapPointIndex[next_index] = first_index;
203  pointIndices.push_back(next_index);
204  ++next;
205  }
206  }
207  }
208 
209  // now set all facets to the correct index
210  MeshFacetArray& rFacets = _rclMesh._aclFacetArray;
211  for (auto& it : rFacets) {
212  for (PointIndex& point : it._aulPoints) {
213  auto pt = mapPointIndex.find(point);
214  if (pt != mapPointIndex.end()) {
215  point = pt->second;
216  }
217  }
218  }
219 
220  // remove invalid indices
221  _rclMesh.DeletePoints(pointIndices);
222  _rclMesh.RebuildNeighbours();
223 
224  return true;
225 }

280 struct MeshFacet_Less
281 {

283  {
284  PointIndex tmp {};
285  PointIndex x0 = x->_aulPoints[0];
286  PointIndex x1 = x->_aulPoints[1];
287  PointIndex x2 = x->_aulPoints[2];
288  PointIndex y0 = y->_aulPoints[0];
289  PointIndex y1 = y->_aulPoints[1];
290  PointIndex y2 = y->_aulPoints[2];
291 
292  if (x0 > x1) {
293  tmp = x0;
294  x0 = x1;
295  x1 = tmp;
296  }
297  if (x0 > x2) {
298  tmp = x0;
299  x0 = x2;
300  x2 = tmp;
301  }
302  if (x1 > x2) {
303  tmp = x1;
304  x1 = x2;
305  x2 = tmp;
306  }
307  if (y0 > y1) {
308  tmp = y0;
309  y0 = y1;
310  y1 = tmp;
311  }
312  if (y0 > y2) {
313  tmp = y0;
314  y0 = y2;
315  y2 = tmp;
316  }
317  if (y1 > y2) {
318  tmp = y1;
319  y1 = y2;
320  y2 = tmp;
321  }
322 
323  if (x0 < y0) {
324  return true;
325  }
326  if (x0 > y0) {
327  return false;
328  }
329  if (x1 < y1) {
330  return true;
331  }
332  if (x1 > y1) {
333  return false;
334  }
335  if (x2 < y2) {
336  return true;
337  }
338 
339  return false;
340  }

536 bool MeshRemoveNeedles::Fixup()
537 {
538  using FaceEdge = std::pair<unsigned long, int>; // (face, edge) pair
539  using FaceEdgePriority = std::pair<float, FaceEdge>;
540 
541  MeshTopoAlgorithm topAlg(_rclMesh);
542  MeshRefPointToFacets vf_it(_rclMesh);
543  const MeshFacetArray& rclFAry = _rclMesh.GetFacets();
544  const MeshPointArray& rclPAry = _rclMesh.GetPoints();
545  rclFAry.ResetInvalid();
546  rclPAry.ResetInvalid();
547  rclPAry.ResetFlag(MeshPoint::VISIT);
548  std::size_t facetCount = rclFAry.size();
549 
550  std::priority_queue<FaceEdgePriority, std::vector<FaceEdgePriority>, std::greater<>> todo;
551  for (std::size_t index = 0; index < facetCount; index++) {
552  const MeshFacet& facet = rclFAry[index];
553  MeshGeomFacet tria(_rclMesh.GetFacet(facet));
554  float perimeter = tria.Perimeter();
555  float fMinLen = perimeter * fMinEdgeLength;
556  for (int i = 0; i < 3; i++) {
557  const Base::Vector3f& p1 = rclPAry[facet._aulPoints[i]];
558  const Base::Vector3f& p2 = rclPAry[facet._aulPoints[(i + 1) % 3]];
559 
560  float distance = Base::Distance(p1, p2);
561  if (distance < fMinLen) {
562  unsigned long facetIndex = static_cast<unsigned long>(index);
563  todo.push(std::make_pair(distance, std::make_pair(facetIndex, i)));
564  }
565  }
566  }
567 
568  bool removedEdge = false;
569  while (!todo.empty()) {
570  FaceEdge faceedge = todo.top().second;
571  todo.pop();
572 
573  // check if one of the face pairs was already processed
574  if (!rclFAry[faceedge.first].IsValid()) {
575  continue;
576  }
577 
578  // the facet points may have changed, so check the current distance again
579  const MeshFacet& facet = rclFAry[faceedge.first];
580  MeshGeomFacet tria(_rclMesh.GetFacet(facet));
581  float perimeter = tria.Perimeter();
582  float fMinLen = perimeter * fMinEdgeLength;
583  const Base::Vector3f& p1 = rclPAry[facet._aulPoints[faceedge.second]];
584  const Base::Vector3f& p2 = rclPAry[facet._aulPoints[(faceedge.second + 1) % 3]];
585  float distance = Base::Distance(p1, p2);
586  if (distance >= fMinLen) {
587  continue;
588  }
589 
590  // collect the collapse-edge information
591  EdgeCollapse ce;
592  ce._fromPoint = rclFAry[faceedge.first]._aulPoints[faceedge.second];
593  ce._toPoint = rclFAry[faceedge.first]._aulPoints[(faceedge.second + 1) % 3];
594 
595  ce._removeFacets.push_back(faceedge.first);
596  FacetIndex neighbour = rclFAry[faceedge.first]._aulNeighbours[faceedge.second];
597  if (neighbour != FACET_INDEX_MAX) {
598  ce._removeFacets.push_back(neighbour);
599  }
600 
601  std::set<FacetIndex> vf = vf_it[ce._fromPoint];
602  vf.erase(faceedge.first);
603  if (neighbour != FACET_INDEX_MAX) {
604  vf.erase(neighbour);
605  }
606  ce._changeFacets.insert(ce._changeFacets.begin(), vf.begin(), vf.end());
607 
608  // get adjacent points
609  std::set<PointIndex> vv;
610  vv = vf_it.NeighbourPoints(ce._fromPoint);
611  ce._adjacentFrom.insert(ce._adjacentFrom.begin(), vv.begin(), vv.end());
612  vv = vf_it.NeighbourPoints(ce._toPoint);
613  ce._adjacentTo.insert(ce._adjacentTo.begin(), vv.begin(), vv.end());
614 
615  if (topAlg.IsCollapseEdgeLegal(ce)) {
616  topAlg.CollapseEdge(ce);
617  for (auto it : ce._removeFacets) {
618  vf_it.RemoveFacet(it);
619  }
620  for (auto it : ce._changeFacets) {
621  vf_it.RemoveNeighbour(ce._fromPoint, it);
622  vf_it.AddNeighbour(ce._toPoint, it);
623  }
624  removedEdge = true;
625  }
626  }
627 
628  if (removedEdge) {
629  topAlg.Cleanup();
630  _rclMesh.RebuildNeighbours();
631  }
632 
633  return true;
634 }

671  FaceVertex facevertex = todo.top().second;
672  todo.pop();
673 
674  // the facet points may have changed, so check the current distance again
675  const MeshFacet& facet = rclFAry[facevertex.first];

677  const Base::Vector3f& p2 = rclPAry[facet._aulPoints[(facevertex.second + 1) % 3]];

685  // an earlier edge-swap has an impact
686  float fCosAngle = dir1.Dot(dir2);

689  }

692  // the maximum angle must have a clear distance to the other corner points
693  // as factor we choose a default value of 25% of the corresponding edge length
694  Base::Vector3f p4 = p1.Perpendicular(p2, p3 - p2);
695  float distP2P3 = Base::Distance(p2, p3);
696  float distP2P4 = Base::Distance(p2, p4);
697  float distP3P4 = Base::Distance(p3, p4);

746 {

748  float fCosMaxAngle = cos(fMaxAngle);
749 
750  Base::Vector3f u, v;
751  MeshTopoAlgorithm cTopAlg(_rclMesh);
752 
753  MeshFacetIterator it(_rclMesh);
754  for (it.Init(); it.More(); it.Next()) {
755  // possibly deformed but not degenerated
756  if (!it->IsDegenerated(fEpsilon)) {
757  // store the angles to avoid to compute twice
758  float fCosAngles[3] = {0, 0, 0};
759  bool done = false;
760 
761  for (int i = 0; i < 3; i++) {
762  u = it->_aclPoints[(i + 1) % 3] - it->_aclPoints[i];
763  v = it->_aclPoints[(i + 2) % 3] - it->_aclPoints[i];
764  u.Normalize();
765  v.Normalize();
766 
767  float fCosAngle = u * v;
768  fCosAngles[i] = fCosAngle;
769  }
770 
771  // first check for angle > 120 deg: in this case we swap with the opposite edge
772  for (int i = 0; i < 3; i++) {
773  float fCosAngle = fCosAngles[i];
774  if (fCosAngle < fCosMaxAngle) {
775  const MeshFacet& face = it.GetReference();
776  FacetIndex uNeighbour = face._aulNeighbours[(i + 1) % 3];
777  if (uNeighbour != FACET_INDEX_MAX
778  && cTopAlg.ShouldSwapEdge(it.Position(), uNeighbour, fMaxSwapAngle)) {
779  cTopAlg.SwapEdge(it.Position(), uNeighbour);
780  done = true;
781  }
782  break;
783  }
784  }
785 
786  // we have swapped already
787  if (done) {
788  continue;
789  }
790 
791  // now check for angle < 30 deg: in this case we swap with one of the edges the corner
792  // is part of
793  for (int j = 0; j < 3; j++) {
794  float fCosAngle = fCosAngles[j];
795  if (fCosAngle > fCosMinAngle) {
796  const MeshFacet& face = it.GetReference();
797 
798  FacetIndex uNeighbour = face._aulNeighbours[j];
799  if (uNeighbour != FACET_INDEX_MAX
800  && cTopAlg.ShouldSwapEdge(it.Position(), uNeighbour, fMaxSwapAngle)) {
801  cTopAlg.SwapEdge(it.Position(), uNeighbour);
802  break;
803  }
804 
805  uNeighbour = face._aulNeighbours[(j + 2) % 3];
806  if (uNeighbour != FACET_INDEX_MAX
807  && cTopAlg.ShouldSwapEdge(it.Position(), uNeighbour, fMaxSwapAngle)) {
808  cTopAlg.SwapEdge(it.Position(), uNeighbour);
809  break;
810  }
811  }
812  }
813  }
814  }
