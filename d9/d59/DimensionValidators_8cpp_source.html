44 TechDraw::DrawViewPart* TechDraw::getReferencesFromSelection(ReferenceVector& references2d,
45  ReferenceVector& references3d)
46 {
47  TechDraw::DrawViewPart* dvp(nullptr);
48  TechDraw::DrawViewDimension* dim(nullptr);
49  constexpr bool allowOnlySingle{false};
50  std::vector<Gui::SelectionObject> selectionAll =
51  Gui::Selection().getSelectionEx("*",
52  App::DocumentObject::getClassTypeId(),
53  Gui::ResolveMode::NoResolve,
54  allowOnlySingle);
55 
56  for (auto& selItem : selectionAll) {
57  if (selItem.getObject()->isDerivedFrom(TechDraw::DrawViewDimension::getClassTypeId())) {
58  //we are probably repairing a dimension, but we will check later
59  dim = static_cast<TechDraw::DrawViewDimension*>(selItem.getObject()); //NOLINT cppcoreguidelines-pro-type-static-cast-downcast
60  } else if (selItem.getObject()->isDerivedFrom(TechDraw::DrawViewPart::getClassTypeId())) {
61  //this could be a 2d geometry selection or just a DrawViewPart for context in
62  //a 3d selection
63  dvp = static_cast<TechDraw::DrawViewPart*>(selItem.getObject()); //NOLINT cppcoreguidelines-pro-type-static-cast-downcast
64  if (selItem.getSubNames().empty()) {
65  //there are no subNames, so we think this is a 3d case,
66  //and we only need to select the view. We set the reference
67  //subName to a null string to avoid later misunderstandings.
68  ReferenceEntry ref(dvp, std::string());
69  references2d.push_back(ref);
70  continue;
71  }
72  for (auto& sub : selItem.getSubNames()) {
73  // plain ordinary 2d view + geometry reference
74 
75  ReferenceEntry ref(dvp, ShapeFinder::getLastTerm(sub));
76  references2d.push_back(ref);
77  }
78  } else if (!selItem.getObject()->isDerivedFrom(TechDraw::DrawView::getClassTypeId())) {
79  App::DocumentObject* obj3d = selItem.getObject();
80  // this is a regular 3d reference in form obj + long subelement
81  for (auto& sub3d : selItem.getSubNames()) {
82  ReferenceEntry ref(obj3d, sub3d);
83  references3d.push_back(ref);
84  }
85  }
86  }
87  if (dim) {
88  if (!dvp) {
89  ReferenceEntry ref(dim->getViewPart(), std::string());
90  references2d.push_back(ref);
91  return dim->getViewPart();
92  }
93  }
94  return dvp;
95 }

98 DimensionGeometryType TechDraw::validateDimSelection(
99  const ReferenceVector& references, //[(dvp*, std::string),...,(dvp*, std::string)]
100  const StringVector& acceptableGeometry,//"Edge", "Vertex", etc
101  const std::vector<int>& minimumCounts, //how many of each geometry are needed for a good dimension
102  const std::vector<DimensionGeometryType>& acceptableDimensionGeometrys)//isVertical, isHorizontal, ...
103 {
104  StringVector subNames;
105  TechDraw::DrawViewPart* dvpSave(nullptr);
106  for (auto& ref : references) {
107  auto dvp = dynamic_cast<TechDraw::DrawViewPart*>(ref.getObject());
108  if (dvp) {
109  dvpSave = dvp;
110  if (!ref.getSubName().empty()) {
111  subNames.push_back(ref.getSubName());
112  }
113  }
114  }
115  if (!dvpSave) {
116  //must have 1 DVP in selection
117  return isInvalid;
118  }
119 
120  if (subNames.empty()) {
121  //no geometry referenced. can not make a dim from this mess. We are being called to validate
122  //a selection for a 3d reference
123  return isViewReference;
124  }
125 
126  if (subNames.front().empty()) {
127  //can this still happen?
128  return isViewReference;
129  }
130 
131  //check for invalid geometry descriptors in the subNames
132  std::unordered_set<std::string> acceptableGeometrySet(acceptableGeometry.begin(),
133  acceptableGeometry.end());
134  if (!TechDraw::validateSubnameList(subNames, acceptableGeometrySet)) {
135  //can not make a dimension from this
136  return isInvalid;
137  }
138 
139  //check for wrong number of geometry
140  GeomCountVector foundCounts;
141  GeomCountMap minimumCountMap = loadRequiredCounts(acceptableGeometry, minimumCounts);
142  if (!checkGeometryOccurrences(subNames, minimumCountMap)) {
143  //too many or too few geometry descriptors.
144  return isInvalid;
145  }
146 
147  //we have a (potentially valid collection of 2d geometry
148  ReferenceVector valid2dReferences;
149  for (auto& sub : subNames) {
150  ReferenceEntry validEntry(dvpSave, sub);
151  valid2dReferences.push_back(validEntry);
152  }
153 
154  DimensionGeometryType foundGeometry = getGeometryConfiguration(valid2dReferences);
155  if (acceptableDimensionGeometrys.empty()) {
156  //if the list is empty, we are accepting anything
157  return foundGeometry;
158  }
159  for (auto& acceptable : acceptableDimensionGeometrys) {
160  if (foundGeometry == acceptable) {
161  return foundGeometry;
162  }
163  }
164 
165  return isInvalid;
166 }

224 //count how many of each "Edge", "Vertex, etc and compare totals to required minimum
225 bool TechDraw::checkGeometryOccurrences(StringVector subNames, GeomCountMap keyedMinimumCounts)
226 {
227  //how many of each geometry descriptor are input
228  GeomCountMap foundCounts;
229  for (auto& sub : subNames) {
230  std::string geometryType = DrawUtil::getGeomTypeFromName(ShapeFinder::getLastTerm(sub));
231  std::map<std::string, int>::iterator it0(foundCounts.find(geometryType));
232  if (it0 == foundCounts.end()) {
233  //first occurrence of this geometryType
234  foundCounts[geometryType] = 1;
235  } else {
236  //already have this geometryType
237  it0->second++;
238  }
239  }
240 
241  //hybrid dims (vertex-edge) can skip this check
242  if (foundCounts.size() > 1) {
243  //this is a hybrid dimension
244  return true;
245  }
246 
247  //check found geometry counts against required counts
248  for (auto& foundItem : foundCounts) {
249  std::string currentKey = foundItem.first;
250  int foundCount = foundItem.second;
251  auto itAccept = keyedMinimumCounts.find(currentKey);
252  if (itAccept == keyedMinimumCounts.end()) {
253  //not supposed to happen by this point
254  throw Base::IndexError("Dimension validation counts and geometry do not match");
255  }
256  if (foundCount < keyedMinimumCounts[currentKey]) {
257  //not enough of this type of geom to make a good dimension - ex 1 Vertex
258  return false;
259  }
260  }
261  //we have no complaints about the input
262  return true;
263 }

296  ReferenceVector valid3dReferences)
297 {

300  ReferenceVector subElementRefs;

304  } else {
305  subElementRefs.push_back(ref);
306  }
307  }
308  if (subElementRefs.empty()) {
309  //only whole object references
310  return isMultiEdge;
311  }
312  if (!wholeObjectRefs.empty()) {
313  //mix of whole object and subelement refs
314  return isMultiEdge;//??? correct ???
315  }
316 
317  //only have subelement refs

319  if (config > isInvalid) {

321  }

324  return config;
325  }

362 DimensionGeometryType TechDraw::isValidSingleEdge(ReferenceEntry ref)
363 {
364  auto objFeat(dynamic_cast<TechDraw::DrawViewPart*>(ref.getObject()));
365  if (!objFeat) {
366  return isInvalid;
367  }
368 
369  //the Name starts with "Edge"
370  std::string geomName = DrawUtil::getGeomTypeFromName(ref.getSubName());
371  if (geomName != "Edge") {
372  return isInvalid;
373  }
374 
375  //the geometry exists (redundant?)
376  int GeoId(TechDraw::DrawUtil::getIndexFromName(ref.getSubName()));
377  TechDraw::BaseGeomPtr geom = objFeat->getGeomByIndex(GeoId);
378  if (!geom) {
379  return isInvalid;
380  }
381 
382  if (geom->getGeomType() == TechDraw::GENERIC) {
383  TechDraw::GenericPtr gen1 = std::static_pointer_cast<TechDraw::Generic>(geom);
384  if (gen1->points.size() < 2) {
385  return isInvalid;
386  }
387  Base::Vector3d line = gen1->points.at(1) - gen1->points.at(0);
388  if (fabs(line.y) < FLT_EPSILON) {
389  return TechDraw::isHorizontal;
390  } else if (fabs(line.x) < FLT_EPSILON) {
391  return TechDraw::isVertical;
392  } else {
393  return TechDraw::isDiagonal;
394  }
395  } else if (geom->getGeomType() == TechDraw::CIRCLE || geom->getGeomType() == TechDraw::ARCOFCIRCLE) {
396  return isCircle;
397  } else if (geom->getGeomType() == TechDraw::ELLIPSE || geom->getGeomType() == TechDraw::ARCOFELLIPSE) {
398  return isEllipse;
399  } else if (geom->getGeomType() == TechDraw::BSPLINE) {
400  TechDraw::BSplinePtr spline = std::static_pointer_cast<TechDraw::BSpline>(geom);
401  if (spline->isCircle()) {
402  return isBSplineCircle;
403  } else {
404  return isBSpline;
405  }
406  }
407  return isInvalid;
408 }

411 DimensionGeometryType TechDraw::isValidSingleEdge3d(DrawViewPart* dvp, ReferenceEntry ref)
412 {
413  (void)dvp;
414  //the Name starts with "Edge"
415  std::string geomName = DrawUtil::getGeomTypeFromName(ref.getSubName());
416  if (geomName != "Edge") {
417  return isInvalid;
418  }
419 
420  TopoDS_Shape refShape = ref.getGeometry();
421  if (refShape.IsNull() || refShape.ShapeType() != TopAbs_EDGE) {
422  throw Base::RuntimeError("Geometry for reference is not an edge.");
423  }
424 
425  TopoDS_Edge occEdge = TopoDS::Edge(refShape);
426  BRepAdaptor_Curve adapt(occEdge);
427  if (adapt.GetType() == GeomAbs_Line) {
428  Base::Vector3d point0 = DU::toVector3d(BRep_Tool::Pnt(TopExp::FirstVertex(occEdge)));
429  point0 = dvp->projectPoint(point0);
430  Base::Vector3d point1 = DU::toVector3d(BRep_Tool::Pnt(TopExp::LastVertex(occEdge)));
431  point1 = dvp->projectPoint(point1);
432  Base::Vector3d line = point1 - point0;
433  if (fabs(line.y) < FLT_EPSILON) {
434  return TechDraw::isHorizontal;
435  } else if (fabs(line.x) < FLT_EPSILON) {
436  return TechDraw::isVertical;
437  }
438  // else if (fabs(line.z) < FLT_EPSILON) {
439  // return TechDraw::isZLimited;
440  // }
441  else {
442  return TechDraw::isDiagonal;
443  }
444  } else if (adapt.GetType() == GeomAbs_Circle) {
445  return isCircle;
446  } else if (adapt.GetType() == GeomAbs_Ellipse) {
447  return isEllipse;
448  } else if (adapt.GetType() == GeomAbs_BSplineCurve) {
449  if (GeometryUtils::isCircle(occEdge)) {
450  return isBSplineCircle;
451  } else {
452  return isBSpline;
453  }
454  }
455 
456  return isInvalid;
457 }

502 DimensionGeometryType TechDraw::isValidMultiEdge(ReferenceVector refs)
503 {
504  //there has to be at least 2
505  if (refs.size() < 2) {
506  return isInvalid;
507  }
508 
509  //they all must start with "Edge"
510  const std::string matchToken{"Edge"};
511  if (!refsMatchToken(refs, matchToken)) {
512  return isInvalid;
513  }
514 
515  auto objFeat0(dynamic_cast<TechDraw::DrawViewPart*>(refs.at(0).getObject()));
516  if (!objFeat0) {
517  //probably redundant
518  throw Base::RuntimeError("Logic error in isValidMultiEdge");
519  }
520 
521  if (refs.size() > 2) {
522  //many edges, must be an extent?
523  return isMultiEdge;
524  }
525 
526  //exactly 2 edges. could be angle, could be distance
527  int GeoId0(TechDraw::DrawUtil::getIndexFromName(refs.at(0).getSubName()));
528  int GeoId1(TechDraw::DrawUtil::getIndexFromName(refs.at(1).getSubName()));
529  TechDraw::BaseGeomPtr geom0 = objFeat0->getGeomByIndex(GeoId0);
530  TechDraw::BaseGeomPtr geom1 = objFeat0->getGeomByIndex(GeoId1);
531 
532  if (geom0->getGeomType() == TechDraw::GENERIC && geom1->getGeomType() == TechDraw::GENERIC) {
533  TechDraw::GenericPtr gen0 = std::static_pointer_cast<TechDraw::Generic>(geom0);
534  TechDraw::GenericPtr gen1 = std::static_pointer_cast<TechDraw::Generic>(geom1);
535  if (gen0->points.size() > 2 || gen1->points.size() > 2) {//the edge is a polyline
536  return isInvalid; //not supported yet
537  }
538  Base::Vector3d line0 = gen0->points.at(1) - gen0->points.at(0);
539  line0.Normalize();
540  Base::Vector3d line1 = gen1->points.at(1) - gen1->points.at(0);
541  line1.Normalize();
542  double dot = fabs(line0.Dot(line1));
543  if (DU::fpCompare(dot, 1.0, EWTOLERANCE)) {
544  return isDiagonal; //distance || line
545  }
546  return isAngle; //angle or distance
547  }
548  return isDiagonal; //two edges, not both straight lines
549 }

553 DimensionGeometryType TechDraw::isValidMultiEdge3d(DrawViewPart* dvp, ReferenceVector refs)
554 {
555  (void)dvp;
556  //there has to be at least 2
557  if (refs.size() < 2) {
558  return isInvalid;
559  }
560 
561  //they all must start with "Edge"
562  const std::string matchToken{"Edge"};
563  if (!refsMatchToken(refs, matchToken)) {
564  return isInvalid;
565  }
566 
567  std::vector<TopoDS_Edge> edges;
568  for (auto& ref : refs) {
569  std::vector<TopoDS_Shape> shapesAll = ShapeExtractor::getShapesFromObject(ref.getObject());
570  if (shapesAll.empty()) {
571  //reference has no geometry
572  return isInvalid;
573  }
574  }
575  std::vector<TopoDS_Edge> edgesAll;
576  std::vector<int> typeAll;
577  for (auto& ref : refs) {
578  TopoDS_Shape geometry = ref.getGeometry();
579  if (geometry.ShapeType() != TopAbs_EDGE) {
580  return isInvalid;
581  }
582  TopoDS_Edge edge = TopoDS::Edge(geometry);
583  BRepAdaptor_Curve adapt(edge);
584  if (adapt.GetType() != GeomAbs_Line) {
585  //not a line, so this must be an extent dim?
586  return isMultiEdge;
587  }
588  edgesAll.push_back(edge);
589  }
590  if (edgesAll.size() > 2) {
591  //must be an extent dimension of lines?
592  return isMultiEdge;
593  }
594 
595  if (edgesAll.size() == 2) {
596  Base::Vector3d first0 = DU::toVector3d(BRep_Tool::Pnt(TopExp::FirstVertex(edgesAll.at(0))));
597  Base::Vector3d last0 = DU::toVector3d(BRep_Tool::Pnt(TopExp::LastVertex(edgesAll.at(1))));
598  Base::Vector3d line0 = last0 - first0;
599  Base::Vector3d first1 = DU::toVector3d(BRep_Tool::Pnt(TopExp::FirstVertex(edgesAll.at(0))));
600  Base::Vector3d last1 = DU::toVector3d(BRep_Tool::Pnt(TopExp::LastVertex(edgesAll.at(1))));
601  Base::Vector3d line1 = last1 - first1;
602  line0.Normalize();
603  line1.Normalize();
604  auto dot = fabs(line0.Dot(line1));
605  if (DU::fpCompare(dot, 1.0, EWTOLERANCE)) {
606  //lines are parallel, must be distance dim
607  return isDiagonal;
608  }
609  //lines are skew, could be angle, could be distance?
610  return isAngle;
611  }
612 
613  return isInvalid;
614 }

617 DimensionGeometryType TechDraw::isValidVertexes(ReferenceVector refs)
618 {
619  TechDraw::DrawViewPart* dvp(dynamic_cast<TechDraw::DrawViewPart*>(refs.front().getObject()));
620  if (!dvp) {
621  //probably redundant
622  throw Base::RuntimeError("Logic error in isValidMultiEdge");
623  }
624 
625  const std::string matchToken{"Vertex"};
626  if (!refsMatchToken(refs, matchToken)) {
627  return isInvalid;
628  }
629

631  //2 vertices can only make a distance dimension
632  TechDraw::VertexPtr v0 = dvp->getVertex(refs.at(0).getSubName());

634  Base::Vector3d line = v1->point() - v0->point();

636  return isHorizontal;

638  return isVertical;
639  } else {
640  return isDiagonal;
641  }
642  } else if (refs.size() == 3) {
643  //three vertices make an angle dimension
644  return isAngle3Pt;
645  }
646 
647  // did not find a valid configuration
648  return isInvalid;

686  }
