190  def onChanged(self, vobj, prop):
191  """Execute when a view property is changed."""
192  properties = vobj.PropertiesList
193  meta = vobj.Object.Document.Meta
194 
195  if prop == "AnnotationStyle" and "AnnotationStyle" in properties:
196  if not vobj.AnnotationStyle or vobj.AnnotationStyle == "":
197  # unset style
198  for visprop in utils.get_default_annotation_style().keys():
199  if visprop in properties:
200  # make property writable
201  vobj.setPropertyStatus(visprop, '-ReadOnly')
202  else:
203  # set style
204  styles = {}
205  for key, value in meta.items():
206  if key.startswith("Draft_Style_"):
207  styles[key[12:]] = json.loads(value)
208 
209  if vobj.AnnotationStyle in styles:
210  style = styles[vobj.AnnotationStyle]
211  for visprop in style.keys():
212  if visprop in properties:
213  # make property read-only
214  vobj.setPropertyStatus(visprop, "ReadOnly")
215  value = style[visprop]
216  try:
217  if vobj.getTypeIdOfProperty(visprop) == "App::PropertyColor":
218  value = value & 0xFFFFFF00
219  setattr(vobj, visprop, value)
220  except:
221  pass
