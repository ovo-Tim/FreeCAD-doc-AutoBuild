181  It appends all identified tokens to self.tokenList."""

185  keyToken = False

202  self.getNextToken(theExpr)
203  else:

208  theExpr = theExpr[1:]

239  token = self.tokenList[treeNode.lIndex]
240  treeNode.lIndex += 1
241  if token in tokenDic:

245  newToken = token
246  funcState = 0
247  # print('treeNode.state: ', treeNode.state, ' my.index: ', treeNode.lIndex-1, ' ', token, ' fState: ', funcState)
248 
249  if token == ",":

252  treeNode.state = 6
253  if treeNode.state == 3:
254  newToken = "?"
255  treeNode.state = 4
256

258  funcState = 0
259  newNode = exprNode(treeNode, 3, treeNode.lIndex)
260  self.parseIF(newNode)
261  else:
