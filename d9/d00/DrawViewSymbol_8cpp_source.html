114 //get editable fields from symbol
115 std::vector<std::string> DrawViewSymbol::getEditableFields()
116 {
117  QDomDocument symbolDocument;
118  std::vector<std::string> editables;
119 
120  bool rc = loadQDomDocument(symbolDocument);
121  if (rc) {
122  XMLQuery query(symbolDocument);
123 
124  // XPath query to select all <tspan> nodes whose <text> parent
125  // has "freecad:editable" attribute
126  query.processItems(QString::fromUtf8("declare default element namespace \"" SVG_NS_URI "\"; "
127  "declare namespace freecad=\"" FREECAD_SVG_NS_URI "\"; "
128  "//text[@" FREECAD_ATTR_EDITABLE "]/tspan"),
129  [&editables](QDomElement& tspan) -> bool {
130  QString editableValue = tspan.firstChild().nodeValue();
131  editables.emplace_back(editableValue.toStdString());
132  return true;
133  });
134  }
135  return editables;
136 }

138 //replace editable field in symbol with values from property
139 void DrawViewSymbol::updateFieldsInSymbol()
140 {
141  const std::vector<std::string>& editText = EditableTexts.getValues();
142  if (editText.empty()) {
143  return;
144  }
145 
146  QDomDocument symbolDocument;
147 
148  bool rc = loadQDomDocument(symbolDocument);
149  if (rc) {
150  XMLQuery query(symbolDocument);
151  std::size_t count = 0;
152 
153  // XPath query to select all <tspan> nodes whose <text> parent
154  // has "freecad:editable" attribute
155  query.processItems(QString::fromUtf8("declare default element namespace \"" SVG_NS_URI "\"; "
156  "declare namespace freecad=\"" FREECAD_SVG_NS_URI "\"; "
157  "//text[@" FREECAD_ATTR_EDITABLE "]/tspan"),
158  [&symbolDocument, &editText, &count](QDomElement& tspanElement) -> bool {
159 
160  if (count >= editText.size()) {
161  return false;
162  }
163  // Keep all spaces in the text node
164  tspanElement.setAttribute(QString::fromUtf8("xml:space"),
165  QString::fromUtf8("preserve"));
166 
167  // Remove all child nodes (if any)
168  while (!tspanElement.lastChild().isNull()) {
169  tspanElement.removeChild(tspanElement.lastChild());
170  }
171 
172  // Finally append text node with editable replacement as the only <tspan> descendant
173  tspanElement.appendChild(
174  symbolDocument.createTextNode(QString::fromStdString(editText[count])));
175  ++count;
176  return true;
177  });
178 
179  Symbol.setValue(symbolDocument.toString(1).toStdString());
180  }
181 }

183 //load QDomDocument
184 bool DrawViewSymbol::loadQDomDocument(QDomDocument& symbolDocument)
185 {
186  const char* symbol = Symbol.getValue();
187  QByteArray qba(symbol);
188  if (qba.isEmpty()) {
189  return false;
190  }
191 #if QT_VERSION < QT_VERSION_CHECK(6,5,0) // New setContent interface added in Qt 6.5
192  QString errorMsg;
193  int errorLine;
194  int errorCol;
195  bool nsProcess = false;
196  bool rc = symbolDocument.setContent(qba, nsProcess, &errorMsg, &errorLine, &errorCol);
197  if (!rc) {
198  //invalid SVG message
199  Base::Console().Warning("DrawViewSymbol - %s - SVG for Symbol is not valid. See log.\n",
200  getNameInDocument());
201  Base::Console().Log("DrawViewSymbol - %s - len: %d rc: %d error: %s line: %d col: %d\n",
202  getNameInDocument(), strlen(symbol), rc, qPrintable(errorMsg),
203  errorLine, errorCol);
204  }
205  return rc;
206 #else
207  QDomDocument::ParseResult rc = symbolDocument.setContent(qba); // Use the default ParseOptions
208  if (!rc) {
209  //invalid SVG message
210  Base::Console().Warning("DrawViewSymbol - %s - SVG for Symbol is not valid. See log.\n",
211  getNameInDocument());
212  Base::Console().Log("DrawViewSymbol - %s - len: %d error: %s line: %d col: %d\n",
213  getNameInDocument(), strlen(symbol), qPrintable(rc.errorMessage),
214  rc.errorLine, rc.errorColumn);
215  }
216  return static_cast<bool>(rc);
217 #endif
218 }
