50 void ToolBarAreaWidget::addWidget(QWidget* widget)
51 {
52  // if widget already exist don't do anything
53  if (_layout->indexOf(widget) >= 0) {
54  return;
55  }
56 
57  if (auto toolbar = qobject_cast<ToolBar*>(widget)) {
58  toolbar->updateCustomGripVisibility();

60 
61  _layout->addWidget(widget);
62  adjustParent();
63 
64  QString name = widget->objectName();
65 
66  if (!name.isEmpty()) {
67  Base::ConnectionBlocker block(_conn);

72 void ToolBarAreaWidget::insertWidget(int index, QWidget* widget)
73 {
74  int currentIndex = _layout->indexOf(widget);
75 
76  // we are inserting widget at the same place, this is no-op
77  if (currentIndex == index) {
78  return;
79  }
80 
81  // widget already exists in the area, we need to first remove it and then recreate
82  if (currentIndex > 0) {
83  _layout->removeWidget(widget);
84  }
85 
86  _layout->insertWidget(index, widget);
87 
88  if (auto toolbar = qobject_cast<ToolBar*>(widget)) {
89  toolbar->updateCustomGripVisibility();
90  }
91 
92  adjustParent();
93  saveState();
94 }
