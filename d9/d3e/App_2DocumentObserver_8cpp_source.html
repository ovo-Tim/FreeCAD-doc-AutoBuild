366 bool normalizeConvertIndex(const std::vector<App::DocumentObject*>& objs, const unsigned int idx)
367 {
368  if (auto ext = objs[idx - 1]->getExtensionByType<App::LinkBaseExtension>(true)) {
369  if ((ext->getElementCountValue() != 0) && !ext->getShowElementValue()) {
370  // If the parent is a collapsed link array element, then we
371  // have to keep the index no matter what, because there is
372  // no sub-object corresponding to an array element.
373  return true;
374  }
375  }
376  return false;
377 }

380 bool SubObjectT::normalize(NormalizeOptions options)
381 {
382  bool noElement = options.testFlag(NormalizeOption::NoElement);
383  bool flatten = !options.testFlag(NormalizeOption::NoFlatten);
384  bool keepSub = options.testFlag(NormalizeOption::KeepSubName);
385  bool convertIndex = options.testFlag(NormalizeOption::ConvertIndex);
386 
387  std::ostringstream ss;
388  std::vector<int> subs;
389  auto obj = getObject();
390  if (!obj) {
391  return false;
392  }
393  auto objs = obj->getSubObjectList(subname.c_str(), &subs, flatten);
394  if (objs.empty()) {
395  return false;
396  }
397  for (unsigned i = 1; i < objs.size(); ++i) {
398  // Keep digit-only subname, as it maybe an index to an array, which does
399  // not expand its elements as objects.
400  const char* end = subname.c_str() + subs[i];
401  const char* sub = end - 2;
402  for (;; --sub) {
403  if (sub < subname.c_str()) {
404  sub = subname.c_str();
405  break;
406  }
407  if (*sub == '.') {
408  ++sub;
409  break;
410  }
411  }
412  bool _keepSub {};
413  if (!std::isdigit(sub[0])) {
414  _keepSub = keepSub;
415  }
416  else if (!convertIndex) {
417  _keepSub = true;
418  }
419  else {
420  _keepSub = normalizeConvertIndex(objs, i);
421  }
422  if (_keepSub) {
423  ss << std::string(sub, end);
424  }
425  else {
426  ss << objs[i]->getNameInDocument() << ".";
427  }
428  }
429  if (objs.size() > 1 && objs.front()->getSubObject(ss.str().c_str()) != objs.back()) {
430  // something went wrong
431  return false;
432  }
433  if (!noElement) {
434  ss << getOldElementName();
435  }
436  std::string sub = ss.str();
437  if (objs.front() != obj || subname != sub) {
438  *this = objs.front();
439  subname = std::move(sub);
440  return true;
441  }
442  return false;
443 }

494 std::string SubObjectT::getOldElementName(int* index) const
495 {
496  ElementNamePair element;
497  auto obj = getObject();
498  if (!obj) {
499  return {};
500  }
501  GeoFeature::resolveElement(obj, subname.c_str(), element);
502  if (!index) {
503  return std::move(element.oldName);
504  }
505  std::size_t pos = element.oldName.find_first_of("0123456789");
506  if (pos == std::string::npos) {
507  *index = -1;
508  }
509  else {
510  *index = std::atoi(element.oldName.c_str() + pos);
511  element.oldName.resize(pos);
512  }
513  return std::move(element.oldName);

745 class DocumentObjectWeakPtrT::Private
746 {
747 public:
748  explicit Private(App::DocumentObject* obj)
749  : object(obj)
750  {
751  set(obj);
752  }
753  void deletedDocument(const App::Document& doc)
754  {
755  // When deleting document then there is no way to undo it
756  if (object && object->getDocument() == &doc) {
757  reset();
758  }
759  }
760  void createdObject(const App::DocumentObject& obj) noexcept
761  {
762  // When undoing the removal
763  if (object == &obj) {
764  indocument = true;
765  }
766  }
767  void deletedObject(const App::DocumentObject& obj) noexcept
768  {
769  if (object == &obj) {
770  indocument = false;
771  }
772  }
773  void reset()
774  {
775  connectApplicationDeletedDocument.disconnect();
776  connectDocumentCreatedObject.disconnect();
777  connectDocumentDeletedObject.disconnect();
778  object = nullptr;
779  indocument = false;
780  }
781  void set(App::DocumentObject* obj)
782  {
783  object = obj;
784  if (obj) {
785  // NOLINTBEGIN
786  indocument = true;
787  connectApplicationDeletedDocument = App::GetApplication().signalDeleteDocument.connect(
788  std::bind(&Private::deletedDocument, this, sp::_1));
789  App::Document* doc = obj->getDocument();
790  connectDocumentCreatedObject =
791  doc->signalNewObject.connect(std::bind(&Private::createdObject, this, sp::_1));
792  connectDocumentDeletedObject =
793  doc->signalDeletedObject.connect(std::bind(&Private::deletedObject, this, sp::_1));
794  // NOLINTEND
795  }
796  }
797  App::DocumentObject* get() const noexcept
798  {
799  return indocument ? object : nullptr;
800  }
801 
802  App::DocumentObject* object;
803  bool indocument {false};
804  using Connection = boost::signals2::scoped_connection;
805  Connection connectApplicationDeletedDocument;
806  Connection connectDocumentCreatedObject;
807  Connection connectDocumentDeletedObject;
808 };
