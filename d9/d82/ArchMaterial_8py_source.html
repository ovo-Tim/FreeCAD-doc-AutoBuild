100  def mergeByName(self):
101  if hasattr(self,"Object"):
102  mats = [o for o in self.Object.Group if o.isDerivedFrom("App::MaterialObject")]
103  todelete = []
104  for mat in mats:
105  orig = None
106  for om in mats:
107  if om.Label == mat.Label:
108  orig = om
109  break
110  else:
111  if mat.Label[-1].isdigit() and mat.Label[-2].isdigit() and mat.Label[-3].isdigit():
112  for om in mats:
113  if om.Label == mat.Label[:-3].strip():
114  orig = om
115  break
116  if orig:
117  for par in mat.InList:
118  for prop in par.PropertiesList:
119  if getattr(par,prop) == mat:
120  FreeCAD.Console.PrintMessage("Changed property '"+prop+"' of object "+par.Label+" from "+mat.Label+" to "+orig.Label+"\n")
121  setattr(par,prop,orig)
122  todelete.append(mat)
123  for tod in todelete:
124  if not tod.InList:
125  FreeCAD.Console.PrintMessage("Merging duplicate material "+tod.Label+"\n")
126  FreeCAD.ActiveDocument.removeObject(tod.Name)
127  elif (len(tod.InList) == 1) and (tod.InList[0].isDerivedFrom("App::DocumentObjectGroup")):
128  FreeCAD.Console.PrintMessage("Merging duplicate material "+tod.Label+"\n")
129  FreeCAD.ActiveDocument.removeObject(tod.Name)

178  def isSameColor(self,c1,c2):
179 
180  r = 4

182  if round(c1[1],r) == round(c2[1],r):
183  if round(c1[2],r) == round(c2[2],r):
