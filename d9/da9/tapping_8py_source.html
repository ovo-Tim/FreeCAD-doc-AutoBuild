41 def generate(edge, dwelltime=0.0, repeat=1, retractheight=None, righthand=True):
42  """
43  Generates Gcode for tapping a single hole.
44 
45  Takes as input an edge. It assumes the edge is trivial with just two vectors.
46  The edge must be aligned with the Z axes (Vector(0,0,1)) or it is an error.
47 
48  The first vertex of the edge will be the startpoint
49  The second vertex of the edge will be the endpoint.
50  All other vertices are ignored.
51 
52  additionally, you can pass in a dwelltime,and repeat value.
53 
54  These will result in appropriate G74 and G84 codes.
55 
56  """
57  startPoint = edge.Vertexes[0].Point
58  endPoint = edge.Vertexes[1].Point
59 
60  Path.Log.debug(startPoint)
61  Path.Log.debug(endPoint)
62 
63  Path.Log.debug(numpy.isclose(startPoint.sub(endPoint).x, 0, rtol=1e-05, atol=1e-06))
64  Path.Log.debug(numpy.isclose(startPoint.sub(endPoint).y, 0, rtol=1e-05, atol=1e-06))
65  Path.Log.debug(endPoint)
66 
67  if repeat < 1:
68  raise ValueError("repeat must be 1 or greater")
69 
70  if not type(repeat) is int:
71  raise ValueError("repeat value must be an integer")
72 
73  if not type(dwelltime) is float:
74  raise ValueError("dwelltime must be a float")
75 
76  if retractheight is not None and not type(retractheight) is float:
77  raise ValueError("retractheight must be a float")
78 
79  if not (
80  numpy.isclose(startPoint.sub(endPoint).x, 0, rtol=1e-05, atol=1e-06)
81  and (numpy.isclose(startPoint.sub(endPoint).y, 0, rtol=1e-05, atol=1e-06))
82  ):
83  raise ValueError("edge is not aligned with Z axis")
84 
85  if startPoint.z < endPoint.z:
86  raise ValueError("start point is below end point")
87 
88  cmdParams = {}
89  cmdParams["X"] = startPoint.x
90  cmdParams["Y"] = startPoint.y
91  cmdParams["Z"] = endPoint.z

93 
94  if repeat < 1:

96

98  raise ValueError("repeat value must be an integer")

102

105 
106  # Check if tool is lefthand or righthand, set appropriate G-code
