82 short Chamfer::mustExecute() const
83 {
84  bool touched = false;
85 
86  auto chamferType = ChamferType.getValue();
87 
88  switch (chamferType) {
89  case 0: // "Equal distance"
90  touched = Size.isTouched() || ChamferType.isTouched();
91  break;
92  case 1: // "Two distances"
93  touched = Size.isTouched() || ChamferType.isTouched() || Size2.isTouched();
94  break;
95  case 2: // "Distance and Angle"
96  touched = Size.isTouched() || ChamferType.isTouched() || Angle.isTouched();
97  break;
98  }
99 
100  if (Placement.isTouched() || touched)
101  return 1;
102  return DressUp::mustExecute();
103 }

105 App::DocumentObjectExecReturn *Chamfer::execute()
106 {
107  if (onlyHasToRefine()){
108  TopoShape result = refineShapeIfActive(rawShape);
109  Shape.setValue(result);
110  return App::DocumentObject::StdReturn;
111  }
112 
113  // NOTE: Normally the Base property and the BaseFeature property should point to the same object.
114  // The only difference is that the Base property also stores the edges that are to be chamfered
115  Part::TopoShape TopShape;
116  try {
117  TopShape = getBaseTopoShape();
118  }
119  catch (Base::Exception& e) {
120  return new App::DocumentObjectExecReturn(e.what());
121  }
122 
123  TopShape.setTransform(Base::Matrix4D());
124 
125  auto edges = UseAllEdges.getValue() ? TopShape.getSubTopoShapes(TopAbs_EDGE)
126  : getContinuousEdges(TopShape);
127 
128  if (edges.empty()) {
129  return new App::DocumentObjectExecReturn(
130  QT_TRANSLATE_NOOP("Exception", "No edges specified"));
131  }
132  const int chamferType = ChamferType.getValue();
133  const double size = Size.getValue();
134  double size2 = Size2.getValue();
135  const double angle = Angle.getValue();
136  const bool flipDirection = FlipDirection.getValue();
137 
138  auto res = validateParameters(chamferType, size, size2, angle);
139  if (res != App::DocumentObject::StdReturn) {
140  return res;
141  }
142 
143  this->positionByBaseFeature();
144 
145  if ( static_cast<Part::ChamferType>(chamferType) == Part::ChamferType::distanceAngle ) {
146  size2 = angle;
147  }
148  try {
149  TopoShape shape(0);
150  shape.makeElementChamfer(TopShape,
151  edges,
152  static_cast<Part::ChamferType>(chamferType),
153  size,
154  size2,
155  nullptr,
156  flipDirection ? Part::Flip::flip : Part::Flip::none);
157  if (shape.isNull()) {
158  return new App::DocumentObjectExecReturn(
159  QT_TRANSLATE_NOOP("Exception", "Failed to create chamfer"));
160  }
161 
162  TopTools_ListOfShape aLarg;
163  aLarg.Append(TopShape.getShape());
164  bool failed = false;
165  if (!BRepAlgo::IsValid(aLarg, shape.getShape(), Standard_False, Standard_False)) {
166  ShapeFix_ShapeTolerance aSFT;
167  aSFT.LimitTolerance(shape.getShape(),
168  Precision::Confusion(),
169  Precision::Confusion(),
170  TopAbs_SHAPE);
171  }
172  if (!failed) {
173 
174  // store shape before refinement
175  this->rawShape = shape;
176  shape = refineShapeIfActive(shape);
177  shape = getSolid(shape);
178  }
179  if (!isSingleSolidRuleSatisfied(shape.getShape())) {
180  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Result has multiple solids: that is not currently supported."));
181  }
182  this->Shape.setValue(shape);
183  if (failed) {
184  return new App::DocumentObjectExecReturn(
185  QT_TRANSLATE_NOOP("Exception", "Resulting shape is invalid"));
186  }
187  return App::DocumentObject::StdReturn;
188  }
189  catch (Standard_Failure& e) {
190  return new App::DocumentObjectExecReturn(e.GetMessageString());
191  }
192 }

221 void Chamfer::updateProperties()
222 {
223  auto chamferType = ChamferType.getValue();
224 
225  auto disableproperty = [](App::Property * prop, bool on) {
226  prop->setStatus(App::Property::ReadOnly, on);
227  };
228 
229  switch (chamferType) {
230  case 0: // "Equal distance"
231  disableproperty(&this->Angle, true);
232  disableproperty(&this->Size2, true);
233  break;
234  case 1: // "Two distances"
235  disableproperty(&this->Angle, true);
236  disableproperty(&this->Size2, false);
237  break;
238  case 2: // "Distance and Angle"
239  disableproperty(&this->Angle, false);
240  disableproperty(&this->Size2, true);
241  break;
242  }
243 }

245 static App::DocumentObjectExecReturn *validateParameters(int chamferType, double size, double size2, double angle)
246 {
247  // Size is common to all chamfer types.
248  if (size <= 0) {
249  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Size must be greater than zero"));
250  }
251 
252  switch (chamferType) {
253  case 0: // Equal distance
254  // Nothing to do.
255  break;
256  case 1: // Two distances
257  if (size2 <= 0) {
258  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Size2 must be greater than zero"));
259  }
260  break;
261  case 2: // Distance and angle
262  if (angle <= 0 || angle >= 180.0) {
263  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Angle must be greater than 0 and less than 180"));
264  }
265  break;
266  }
267 
268  return App::DocumentObject::StdReturn;
269 }
