161 void ViewProviderTransformed::recomputeFeature(bool recompute)
162 {
163  PartDesign::Transformed* pcTransformed = getObject<PartDesign::Transformed>();
164  if(recompute || (pcTransformed->isError() || pcTransformed->mustExecute()))
165  pcTransformed->recomputeFeature(true);
166 
167  unsigned rejected = 0;
168  TopoDS_Shape cShape = pcTransformed->rejected;
169  TopExp_Explorer xp;
170  xp.Init(cShape, TopAbs_SOLID);
171  for (; xp.More(); xp.Next()) {
172  rejected++;
173  }
174 
175  QString msg = QString::fromLatin1("%1");
176  if (rejected > 0) {
177  msg = QString::fromLatin1("<font color='orange'>%1<br/></font>\r\n%2");
178  if (rejected == 1)
179  msg = msg.arg(QObject::tr("One transformed shape does not intersect the support"));
180  else {
181  msg = msg.arg(QObject::tr("%1 transformed shapes do not intersect the support"));
182  msg = msg.arg(rejected);
183  }
184  }
185  auto error = pcTransformed->getDocument()->getErrorDescription(pcTransformed);
186  if (error) {
187  msg = msg.arg(QString::fromLatin1("<font color='red'>%1<br/></font>"));
188  msg = msg.arg(QString::fromUtf8(error));
189  } else {
190  msg = msg.arg(QString::fromLatin1("<font color='green'>%1<br/></font>"));
191  msg = msg.arg(QObject::tr("Transformation succeeded"));
192  }
193  diagMessage = msg;
194  signalDiagnosis(msg);
195 
196  // Clear all the rejected stuff
197  while (pcRejectedRoot->getNumChildren() > 7) {
198  SoSeparator* sep = static_cast<SoSeparator*>(pcRejectedRoot->getChild(7));
199  SoMultipleCopy* rejectedTrfms = static_cast<SoMultipleCopy*>(sep->getChild(2));
200  Gui::coinRemoveAllChildren(rejectedTrfms);
201  sep->removeChild(1);
202  sep->removeChild(0);
203  pcRejectedRoot ->removeChild(7);
204  }
205 
206  // Display the rejected transformations in red
207  if (rejected > 0) {
208  showRejectedShape(cShape);
209  }
210 }

212 void ViewProviderTransformed::showRejectedShape(TopoDS_Shape shape)
213 {
214  try {
215  // calculating the deflection value
216  Standard_Real xMin, yMin, zMin, xMax, yMax, zMax;
217  {
218  Bnd_Box bounds;
219  BRepBndLib::Add(shape, bounds);
220  bounds.SetGap(0.0);
221  bounds.Get(xMin, yMin, zMin, xMax, yMax, zMax);
222  }
223  Standard_Real deflection = ((xMax-xMin)+(yMax-yMin)+(zMax-zMin))/300.0 * Deviation.getValue();
224 
225  // create or use the mesh on the data structure
226  // Note: This DOES have an effect on shape
227  Standard_Real AngDeflectionRads = AngularDeflection.getValue() / 180.0 * M_PI;
228  BRepMesh_IncrementalMesh(shape, deflection, Standard_False, AngDeflectionRads, Standard_True);
229 
230  // We must reset the location here because the transformation data
231  // are set in the placement property
232  TopLoc_Location aLoc;
233  shape.Location(aLoc);
234 
235  // count triangles and nodes in the mesh
236  int nbrTriangles=0, nbrNodes=0;
237  TopExp_Explorer Ex;
238  for (Ex.Init(shape, TopAbs_FACE); Ex.More(); Ex.Next()) {
239  Handle (Poly_Triangulation) mesh = BRep_Tool::Triangulation(TopoDS::Face(Ex.Current()), aLoc);
240  // Note: we must also count empty faces
241  if (!mesh.IsNull()) {
242  nbrTriangles += mesh->NbTriangles();
243  nbrNodes += mesh->NbNodes();
244  }
245  }
246 
247  // create memory for the nodes and indexes
248  SoCoordinate3* rejectedCoords = new SoCoordinate3();
249  rejectedCoords ->point .setNum(nbrNodes);
250  SoNormal* rejectedNorms = new SoNormal();
251  rejectedNorms ->vector .setNum(nbrNodes);
252  SoIndexedFaceSet* rejectedFaceSet = new SoIndexedFaceSet();
253  rejectedFaceSet ->coordIndex .setNum(nbrTriangles*4);
254 
255  // get the raw memory for fast fill up
256  SbVec3f* verts = rejectedCoords ->point .startEditing();
257  SbVec3f* norms = rejectedNorms ->vector .startEditing();
258  int32_t* index = rejectedFaceSet ->coordIndex .startEditing();
259 
260  // preset the normal vector with null vector
261  for (int i=0; i < nbrNodes; i++)
262  norms[i]= SbVec3f(0.0,0.0,0.0);
263 
264  int FaceNodeOffset=0,FaceTriaOffset=0;
265  for (Ex.Init(shape, TopAbs_FACE); Ex.More(); Ex.Next()) {
266  const TopoDS_Face &actFace = TopoDS::Face(Ex.Current());
267 
268  // get triangulation
269  std::vector<gp_Pnt> points;
270  std::vector<Poly_Triangle> facets;
271  if (!Part::Tools::getTriangulation(actFace, points, facets))
272  continue;
273 
274  // get normal per vertex
275  std::vector<gp_Vec> vertexnormals;
276  Part::Tools::getPointNormals(points, facets, vertexnormals);
277 
278  // getting size of node and triangle array of this face
279  std::size_t nbNodesInFace = points.size();
280  std::size_t nbTriInFace = facets.size();
281 
282  for (std::size_t i = 0; i < points.size(); i++) {
283  verts[FaceNodeOffset+i] = SbVec3f(points[i].X(), points[i].Y(), points[i].Z());
284  }
285 
286  for (std::size_t i = 0; i < vertexnormals.size(); i++) {
287  norms[FaceNodeOffset+i] = SbVec3f(vertexnormals[i].X(), vertexnormals[i].Y(), vertexnormals[i].Z());
288  }
289 
290  // cycling through the poly mesh
291  for (std::size_t g=0; g < nbTriInFace; g++) {
292  // Get the triangle
293  Standard_Integer N1,N2,N3;
294  facets[g].Get(N1,N2,N3);
295 
296  // set the index vector with the 3 point indexes and the end delimiter
297  index[FaceTriaOffset*4+4*g] = FaceNodeOffset+N1;
298  index[FaceTriaOffset*4+4*g+1] = FaceNodeOffset+N2;
299  index[FaceTriaOffset*4+4*g+2] = FaceNodeOffset+N3;
300  index[FaceTriaOffset*4+4*g+3] = SO_END_FACE_INDEX;
301  }
302 
303  // counting up the per Face offsets
304  FaceNodeOffset += nbNodesInFace;
305  FaceTriaOffset += nbTriInFace;
306 
307  // normalize all normals
308  for (int i=0; i < nbrNodes; i++)
309  norms[i].normalize();
310 
311  // end the editing of the nodes
312  rejectedCoords ->point .finishEditing();
313  rejectedNorms ->vector .finishEditing();
314  rejectedFaceSet ->coordIndex .finishEditing();
315 
316  // fill in the transformation matrices
317  SoMultipleCopy* rejectedTrfms = new SoMultipleCopy();
318  rejectedTrfms->matrix.finishEditing();
319  rejectedTrfms->addChild(rejectedFaceSet);
320  SoSeparator* sep = new SoSeparator();
321  sep->addChild(rejectedCoords);
322  sep->addChild(rejectedNorms);
323  sep->addChild(rejectedTrfms);
324  pcRejectedRoot->addChild(sep);
325  }
326  }
327  catch (...) {
328  Base::Console().Error("Cannot compute Inventor representation for the rejected transformations of shape of %s.\n",
329  getObject()->getNameInDocument());
330  }
331 }
