23 template <class Real>
24 Delaunay1<Real>::Delaunay1 (int iVertexQuantity, Real* afVertex,
25  Real fEpsilon, bool bOwner, Query::Type eQueryType)
26  :
27  Delaunay<Real>(iVertexQuantity,fEpsilon,bOwner,eQueryType)
28 {
29  assert(afVertex);
30  m_afVertex = afVertex;
31 
32  std::vector<SortedVertex> kArray(m_iVertexQuantity);
33  int i;
34  for (i = 0; i < m_iVertexQuantity; i++)
35  {
36  kArray[i].Value = m_afVertex[i];
37  kArray[i].Index = i;
38  }
39  std::sort(kArray.begin(),kArray.end());
40 
41  Real fRange = kArray[m_iVertexQuantity-1].Value - kArray[0].Value;
42  if (fRange >= m_fEpsilon)
43  {
44  m_iDimension = 1;
45  m_iSimplexQuantity = m_iVertexQuantity - 1;
46  m_aiIndex = WM4_NEW int[2*m_iSimplexQuantity];
47  for (i = 0; i < m_iSimplexQuantity; i++)
48  {
49  m_aiIndex[2*i] = kArray[i].Index;
50  m_aiIndex[2*i+1] = kArray[i+1].Index;
51  }
52 
53  m_aiAdjacent = WM4_NEW int[2*m_iSimplexQuantity];
54  for (i = 0; i < m_iSimplexQuantity; i++)
55  {
56  m_aiAdjacent[2*i] = i-1;
57  m_aiAdjacent[2*i+1] = i+1;
58  }
59  m_aiAdjacent[2*m_iSimplexQuantity-1] = -1;
60  }
61 }

178 template <class Real>
179 bool Delaunay1<Real>::GetBarycentricSet (int i, const Real fP, Real afBary[2])
180  const
181 {
182  assert(m_iDimension == 1);
183  if (m_iDimension != 1)
184  {
185  return false;
186  }
187 
188  if (0 <= i && i < m_iSimplexQuantity)
189  {
190  Real fV0 = m_afVertex[m_aiIndex[2*i]];
191  Real fV1 = m_afVertex[m_aiIndex[2*i+1]];
192  Real fDenom = fV1 - fV0;
193  if (fDenom > m_fEpsilon)
194  {
195  afBary[0] = (fV1 - fP)/fDenom;
196  }
197  else
198  {
199  afBary[0] = (Real)1.0;
200  }
201 
202  afBary[1] = (Real)1.0 - afBary[0];
203  return true;
204  }
205  return false;
