84 void
85 SoShapeScale::GLRender(SoGLRenderAction * action)
86 {
87  auto* scale = static_cast<SoScale*>(this->getAnyPart(SbName("scale"), true));
88  if (!this->active.getValue()) {
89  SbVec3f v(1.0f, 1.0f, 1.0f);
90  if (scale->scaleFactor.getValue() != v){
91  scale->scaleFactor = v;
92  }
93  }
94  else {
95  SoState* state = action->getState();
96  const SbViewportRegion & vp = SoViewportRegionElement::get(state);
97  const SbViewVolume & vv = SoViewVolumeElement::get(state);
98 
99  SbVec3f center(0.0f, 0.0f, 0.0f);
100  float nsize = this->scaleFactor.getValue() / float(vp.getViewportSizePixels()[0]);
101  SoModelMatrixElement::get(state).multVecMatrix(center, center); // world coords
102  float sf = vv.getWorldToScreenScale(center, nsize);
103 
104  sf *= SoDevicePixelRatioElement::get(state);
105 
106  SbVec3f v(sf, sf, sf);
107  if (scale->scaleFactor.getValue() != v){
108  scale->scaleFactor = v;
109  }
110  }
111 
112  inherited::GLRender(action);
113 }
