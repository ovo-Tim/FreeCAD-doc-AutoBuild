50 def make_point_array(base_object, point_object, extra=None, use_link=True):
51  """Make a Draft PointArray object.
52 
53  Create copies of a `base_object` at the points defined by
54  a `point_object`.
55 
56  Parameters
57  ----------
58  base_object: Part::Feature or str
59  Any of object that has a `Part::TopoShape` that can be duplicated.
60  This means most 2D and 3D objects produced with any workbench.
61  If it is a string, it must be the `Label` of that object.
62  Since a label is not guaranteed to be unique in a document,
63  it will use the first object found with this label.
64 
65  point_object: Part::Feature, Sketcher::SketchObject, Mesh::Feature,
66  Points::FeatureCustom or str
67  The object must have vertices and/or points.
68 
69  extra: Base::Placement, Base::Vector3, or Base::Rotation, optional
70  It defaults to `None`.
71  If it is provided, it is an additional placement that is applied
72  to each copy of the array.
73  The input could be a full placement, just a vector indicating
74  the additional translation, or just a rotation.
75 
76  Returns
77  -------
78  Part::FeaturePython
79  A scripted object of type `'PointArray'`.
80  Its `Shape` is a compound of the copies of the original object.
81 
82  None
83  If there is a problem it will return `None`.
84  """
85  _name = "make_point_array"
86 
87  found, doc = utils.find_doc(App.activeDocument())
88  if not found:
89  _err(translate("draft", "No active document. Aborting."))
90  return None
91 
92  found, base_object = utils.find_object(base_object, doc)
93  if not found:
94  _err(translate("draft", "Wrong input: base_object not in document."))
95  return None
96 
97  found, point_object = utils.find_object(point_object, doc)
98  if not found:
99  _err(translate("draft", "Wrong input: point_object not in document."))
100  return None
101 
102  if not ((hasattr(point_object, "Shape") and hasattr(point_object.Shape, "Vertexes"))
103  or hasattr(point_object, "Mesh")
104  or hasattr(point_object, "Points")):
105  _err(translate("draft", "Wrong input: object has the wrong type."))
106  return None
107 
108  if not extra:
109  extra = App.Placement()
110 
111  try:
112  utils.type_check([(extra, (App.Placement,
113  App.Vector,
114  App.Rotation))],
115  name=_name)
116  except TypeError:
117  _err(translate("draft", "Wrong input: must be a placement, a vector, or a rotation."))
118  return None
119 
120  # Convert the vector or rotation to a full placement
121  if isinstance(extra, App.Vector):
122  extra = App.Placement(extra, App.Rotation())
123  elif isinstance(extra, App.Rotation):
124  extra = App.Placement(App.Vector(), extra)
125 
126  if use_link:
127  # The PointArray class must be called in this special way
128  # to make it a LinkArray
129  new_obj = doc.addObject("Part::FeaturePython", "PointArray",
130  PointArray(None), None, True)
131  else:
132  new_obj = doc.addObject("Part::FeaturePython", "PointArray")
133  PointArray(new_obj)
134 
135  new_obj.Base = base_object
136  new_obj.PointObject = point_object
137  new_obj.ExtraPlacement = extra
138 
139  if App.GuiUp:
140  if use_link:
141  ViewProviderDraftLink(new_obj.ViewObject)
142  else:
143  new_obj.Proxy.execute(new_obj) # Updates Count which is required for correct DiffuseColor.
144  ViewProviderDraftArray(new_obj.ViewObject)
145  gui_utils.format_object(new_obj, new_obj.Base)
146  new_obj.ViewObject.Proxy.resetColors(new_obj.ViewObject)
147  new_obj.Base.ViewObject.hide()
148  gui_utils.select(new_obj)
149 
150  return new_obj
