387 void TaskHoleParameters::holeCutTypeChanged(int index)
388 {
389  if (index < 0) {
390  return;
391  }
392 
393  auto hole = getObject<PartDesign::Hole>();
394 
395  // the HoleCutDepth is something different for countersinks and counterbores
396  // therefore reset it, it will be reset to sensible values by setting the new HoleCutType
397  hole->HoleCutDepth.setValue(0.0);
398  hole->HoleCutType.setValue(index);
399 
400  // when holeCutType was changed, reset HoleCutCustomValues to false because it should
401  // be a purpose decision to overwrite the normed values
402  // we will handle the case that there is no normed value later in this routine
403  ui->HoleCutCustomValues->setChecked(false);
404  hole->HoleCutCustomValues.setValue(false);
405 
406  // recompute to get the info about the HoleCutType properties
407  recomputeFeature();
408 
409  // apply the result to the widgets
410  ui->HoleCutCustomValues->setDisabled(hole->HoleCutCustomValues.isReadOnly());
411  ui->HoleCutCustomValues->setChecked(hole->HoleCutCustomValues.getValue());
412 
413  // HoleCutCustomValues is only enabled for screw definitions
414  // we must do this after recomputeFeature() because this gives us the info if
415  // the type is a countersink and thus if HoleCutCountersinkAngle can be enabled
416  std::string HoleCutTypeString = hole->HoleCutType.getValueAsString();
417  if (HoleCutTypeString == "None" || HoleCutTypeString == "Counterbore"
418  || HoleCutTypeString == "Countersink" || HoleCutTypeString == "Counterdrill") {
419  ui->HoleCutCustomValues->setEnabled(false);
420  if (HoleCutTypeString == "None") {
421  ui->HoleCutDiameter->setEnabled(false);
422  ui->HoleCutDepth->setEnabled(false);
423  ui->HoleCutCountersinkAngle->setEnabled(false);
424  }
425  if (HoleCutTypeString == "Counterbore") {
426  ui->HoleCutCountersinkAngle->setEnabled(false);
427  }
428  if (HoleCutTypeString == "Countersink") {
429  ui->HoleCutCountersinkAngle->setEnabled(true);
430  }
431  }
432  else { // screw definition
433  // we can have the case that we have no normed values
434  // in this case HoleCutCustomValues is read-only AND true
435  if (ui->HoleCutCustomValues->isChecked()) {
436  ui->HoleCutDiameter->setEnabled(true);
437  ui->HoleCutDepth->setEnabled(true);
438  if (!hole->HoleCutCountersinkAngle.isReadOnly()) {
439  ui->HoleCutCountersinkAngle->setEnabled(true);
440  }
441  }
442  else {
443  ui->HoleCutCustomValues->setEnabled(true);
444  ui->HoleCutDiameter->setEnabled(false);
445  ui->HoleCutDepth->setEnabled(false);
446  ui->HoleCutCountersinkAngle->setEnabled(false);
447  }
448  }
449 }

481 void TaskHoleParameters::holeCutDepthChanged(double value)
482 {
483  auto hole = getObject<PartDesign::Hole>();
484  if (!hole) {
485  return;
486  }
487 
488  std::string HoleCutTypeString = hole->HoleCutType.getValueAsString();
489 
490  if (ui->HoleCutCountersinkAngle->isEnabled() && HoleCutTypeString != "Counterdrill") {
491  // we have a countersink and recalculate the HoleCutDiameter
492 
493  // store current depth
494  double DepthDifference = value - hole->HoleCutDepth.getValue();
495  // new diameter is the old one + 2 * tan(angle / 2) * DepthDifference
496  double newDiameter = hole->HoleCutDiameter.getValue()
497  + 2 * tan(Base::toRadians(hole->HoleCutCountersinkAngle.getValue() / 2))
498  * DepthDifference;
499  // only apply if the result is not smaller than the hole diameter
500  if (newDiameter > hole->Diameter.getValue()) {
501  hole->HoleCutDiameter.setValue(newDiameter);
502  hole->HoleCutDepth.setValue(value);
503  }
504  }
505  else {
506  hole->HoleCutDepth.setValue(value);
507  }
508 
509  recomputeFeature();
510 }

614 void TaskHoleParameters::threadTypeChanged(int index)
615 {
616  if (index < 0) {
617  return;
618  }
619 
620  auto hole = getObject<PartDesign::Hole>();
621  if (!hole) {
622  return;
623  }
624 
625  // A typical case is that users change from an ISO profile to another one.
626  // When they had e.g. the size "M3" in one profile they expect
627  // the same size in the other profile if it exists there.
628  // Besides the size also the thread class" and hole cut type are affected.
629 
630  // at first check what type class is used
631  QByteArray TypeClass = ui->ThreadType->itemData(index).toByteArray();
632 
633  // store the current size
634  QString ThreadSizeString = ui->ThreadSize->currentText();
635  // store the current class
636  QString ThreadClassString = ui->ThreadClass->currentText();
637  // store the current type
638  QString CutTypeString = ui->HoleCutType->currentText();
639 
640  // now set the new type, this will reset the comboboxes to item 0
641  hole->ThreadType.setValue(index);
642 
643  // Threaded checkbox is meaningless if no thread profile is selected.
644  ui->Threaded->setDisabled(std::string(hole->ThreadType.getValueAsString()) == "None");
645 
646  // size and clearance
647  if (TypeClass == QByteArray("ISO")) {
648  // the size for ISO type has either the form "M3x0.35" or just "M3"
649  // so we need to check if the size contains a 'x'. If yes, check if the string
650  // up to the 'x' is exists in the new list
651  if (ThreadSizeString.indexOf(QString::fromLatin1("x")) > -1) {
652  // we have an ISO fine size
653  // cut of the part behind the 'x'
654  ThreadSizeString =
655  ThreadSizeString.left(ThreadSizeString.indexOf(QString::fromLatin1("x")));
656  }
657  // search if the string exists in the combobox
658  int threadSizeIndex = ui->ThreadSize->findText(ThreadSizeString, Qt::MatchContains);
659  if (threadSizeIndex > -1) {
660  // we can set it
661  ui->ThreadSize->setCurrentIndex(threadSizeIndex);
662  }
663  // the names of the clearance types are different in ISO and UTS
664  ui->ThreadFit->setItemText(
665  0,
666  QCoreApplication::translate("TaskHoleParameters", "Standard", nullptr));
667  ui->ThreadFit->setItemText(
668  1,
669  QCoreApplication::translate("TaskHoleParameters", "Close", nullptr));
670  ui->ThreadFit->setItemText(
671  2,
672  QCoreApplication::translate("TaskHoleParameters", "Wide", nullptr));
673  }
674  else if (TypeClass == QByteArray("UTS")) {
675  // for all UTS types the size entries are the same
676  int threadSizeIndex = ui->ThreadSize->findText(ThreadSizeString, Qt::MatchContains);
677  if (threadSizeIndex > -1) {
678  ui->ThreadSize->setCurrentIndex(threadSizeIndex);
679  }
680  // the names of the clearance types are different in ISO and UTS
681  ui->ThreadFit->setItemText(
682  0,
683  QCoreApplication::translate("TaskHoleParameters", "Normal", nullptr));
684  ui->ThreadFit->setItemText(
685  1,
686  QCoreApplication::translate("TaskHoleParameters", "Close", nullptr));
687  ui->ThreadFit->setItemText(
688  2,
689  QCoreApplication::translate("TaskHoleParameters", "Loose", nullptr));
690  }
691 
692  // Class and cut type
693  // the class and cut types are the same for both TypeClass so we don't need to distinguish
694  // between ISO and UTS
695  int threadClassIndex = ui->ThreadClass->findText(ThreadClassString, Qt::MatchContains);
696  if (threadClassIndex > -1) {
697  ui->ThreadClass->setCurrentIndex(threadClassIndex);
698  }
699  int holeCutIndex = ui->HoleCutType->findText(CutTypeString, Qt::MatchContains);
700  if (holeCutIndex > -1) {
701  ui->HoleCutType->setCurrentIndex(holeCutIndex);
702  }
703 
704  // we must set the read-only state according to the new HoleCutType
705  holeCutTypeChanged(ui->HoleCutType->currentIndex());
706 
707  recomputeFeature();
708 }
