43  // Coordinate system. The origin is valid for any dimension d. The
44  // unit-length direction vector is valid only for 0 <= i < d. The extreme
45  // index is relative to the array of input points, and is also valid only
46  // for 0 <= i < d. If d = 0, all points are effectively the same, but the
47  // use of an epsilon may lead to an extreme index that is not zero. If
48  // d = 1, all points effectively lie on a line segment. The extreme
49  // indices correspond to input points that are the end points of the
50  // segment. If d = 2, the first two extreme indices correspond to a
51  // line segment. The next extreme index corresponds to the input point
52  // that is farthest from this line segment in a direction perpendicular
53  // to the segment. If d = 3, the first three extreme indices correspond
54  // to a triangle. The next extreme index corresponds to the input point
55  // that is farthest from this triangle in a direction perpendicular to
56  // the triangle.
57  const Vector3<Real>& GetOrigin () const;
58  const Vector3<Real>& GetDirection (int i) const;
59  int GetExtremeIndex (int i) const;
60 
61  // If d = 3, the direction vectors {U0,U1,U2} form a right-handed set.
62  // The four extreme points form a tetrahedron. This function indicates
63  // if that tetrahedron is counterclockwise ordered.
64  bool GetExtremeCCW () const;
65 
66 private:
67  // Axis-aligned bounding box of input points. The maximum range is the
68  // larger of max[0]-min[0], max[1]-min[1], and max[2]-min[2].
69  Vector3<Real> m_kMin, m_kMax;
70  Real m_fMaxRange;
71 
72  // The intrinsic dimension of the input set. The parameter fEpsilon to
73  // the constructor is used to provide a tolerance when determining the
74  // dimension.
75  int m_iDimension;
76 
77  // The indices that define the maximum dimensional extents. The values
78  // m_aiExtreme[0] and m_aiExtreme[1] are the indices for the vertices
79  // which define the largest extent in one of the coordinate axis
80  // directions. If the intrinsic dimensionality is 2, then m_aiExtreme[2]
81  // is the index for the vertex which causes the largest extent in the
82  // direction perpendicular to the line through the vertices corresponding
83  // to m_aiExtreme[0] and m_aiExtreme[1]. Furthermore, if the intrinsic
84  // dimensionality is 3, then m_aiExtreme[3] is the index for the vertex
85  // which causes the largest extent in the direction perpendicular to the
86  // triangle defined by the other extreme points. The tetrahedron
87  // <V[extreme0],V[extreme1],V[extreme2],V[extreme3]> can be clockwise or
88  // counterclockwise, the condition stored in m_bExtremeCCW.
89  int m_aiExtreme[4];
