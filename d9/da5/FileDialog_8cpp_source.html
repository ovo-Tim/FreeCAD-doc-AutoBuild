156 void FileDialog::getSuffixesDescription(QStringList& suffixes, const QString* suffixDescriptions)
157 {
158  QRegularExpression rx;
159  // start the raw string with a (
160  // match a *, a . and at least one word character (a-z, A-Z, 0-9, _) with \*\.\w+
161  // end the raw string with a )
162  rx.setPattern(QLatin1String(R"(\*\.\w+)"));
163 
164  QRegularExpressionMatchIterator i = rx.globalMatch(*suffixDescriptions);
165  while (i.hasNext()) {
166  QRegularExpressionMatch match = i.next();
167  QString suffix = match.captured(0);
168  suffixes << suffix;
169  }
170 }

175 QString FileDialog::getSaveFileName (QWidget * parent, const QString & caption, const QString & dir,
176  const QString & filter, QString * selectedFilter, Options options)
177 {
178  QString dirName = dir;
179  bool hasFilename = false;
180  if (dirName.isEmpty()) {
181  dirName = getWorkingDirectory();
182  } else {
183  QFileInfo fi(dir);
184  if (fi.isRelative()) {
185  dirName = getWorkingDirectory();
186  dirName += QLatin1String("/");
187  dirName += fi.fileName();
188  }
189  if (!fi.fileName().isEmpty()) {
190  hasFilename = true;
191  }
192 
193  // get the suffix for the filter: use the selected filter if there is one,
194  // otherwise find the first valid suffix in the complete list of filters
195  const QString *filterToSearch;
196  if (selectedFilter && !selectedFilter->isEmpty()) {
197  filterToSearch = selectedFilter;
198  }
199  else {
200  filterToSearch = &filter;
201  }
202 
203  QStringList filterSuffixes;
204  getSuffixesDescription(filterSuffixes, filterToSearch);
205  QString fiSuffix = QLatin1String("*.") + fi.suffix(); // To match with filterSuffixes
206  if (fi.suffix().isEmpty() || !filterSuffixes.contains(fiSuffix)) {
207  // there is no suffix or not a suffix that matches the filter, so
208  // default to the first suffix of the filter
209  dirName += filterSuffixes[0].mid(1);
210  }
211  }
212 
213  QString windowTitle = caption;
214  if (windowTitle.isEmpty())
215  windowTitle = FileDialog::tr("Save as");
216 
217  // NOTE: We must not change the specified file name afterwards as we may return the name of an already
218  // existing file. Hence we must extract the first matching suffix from the filter list and append it
219  // before showing the file dialog.
220  QString file;
221  if (DialogOptions::dontUseNativeFileDialog()) {
222  QList<QUrl> urls = fetchSidebarUrls();
223 
224  options |= QFileDialog::DontUseNativeDialog;
225 
226  FileDialog dlg(parent);
227  dlg.setOptions(options);
228  dlg.setWindowTitle(windowTitle);
229  dlg.setSidebarUrls(urls);
230  auto iconprov = std::make_unique<FileIconProvider>();
231  dlg.setIconProvider(iconprov.get());
232  dlg.setFileMode(QFileDialog::AnyFile);
233  dlg.setAcceptMode(QFileDialog::AcceptSave);
234  dlg.setDirectory(dirName);
235  if (hasFilename)
236  dlg.selectFile(dirName);
237  dlg.setNameFilters(filter.split(QLatin1String(";;")));
238  if (selectedFilter && !selectedFilter->isEmpty())
239  dlg.selectNameFilter(*selectedFilter);
240  dlg.onSelectedFilter(dlg.selectedNameFilter());
241  dlg.setOption(QFileDialog::HideNameFilterDetails, false);
242  dlg.setOption(QFileDialog::DontConfirmOverwrite, false);
243  if (dlg.exec() == QDialog::Accepted) {
244  if (selectedFilter)
245  *selectedFilter = dlg.selectedNameFilter();
246  file = dlg.selectedFiles().constFirst();
247  }
248  }
249  else {
250  file = QFileDialog::getSaveFileName(parent, windowTitle, dirName, filter, selectedFilter, options);
251  file = QDir::fromNativeSeparators(file);
252  }
253 
254  if (!file.isEmpty()) {
255  setWorkingDirectory(file);
256  return file;
257  } else {
258  return {};
259  }
260 }

470 void FileOptionsDialog::accept()
471 {
472  // Fixes a bug of the default implementation when entering an asterisk
473  auto filename = this->findChild<QLineEdit*>();
474  QString fn = filename->text();
475  if (fn.startsWith(QLatin1String("*"))) {
476  QFileInfo fi(fn);
477  QString ext = fi.suffix();
478  ext.prepend(QLatin1String("*."));
479  QStringList filters = this->nameFilters();
480  bool ok=false;
481  // Compare the given suffix with the suffixes of all filters
482  QString filter;
483  for (const auto & it : filters) {
484  if (it.contains(ext)) {
485  filter = it;
486  ok = true;
487  break;
488  }
489  }
490 
491  // if no appropriate filter was found the add the 'All files' filter
492  if (!ok) {
493  filter = tr("All files (*.*)");
494  filters << filter;
495  setNameFilters(filters);
496  }
497 
498  // empty the line edit
499  filename->blockSignals(true);
500  filename->clear();
501  filename->blockSignals(false);
502  selectNameFilter(filter);
503 
504  return;
505  }
506  else if (!fn.isEmpty()) {
507  QFileInfo fi(fn);
508  QString ext = fi.completeSuffix();
509  QRegularExpression rx(QLatin1String(R"(\(\*.(\w+))"));
510  QString suf = selectedNameFilter();
511  auto match = rx.match(suf);
512  if (match.hasMatch())
513  suf = match.captured(1);
514  if (ext.isEmpty())
515  setDefaultSuffix(suf);
516  else if (ext.toLower() != suf.toLower()) {
517  fn = QString::fromLatin1("%1.%2").arg(fn, suf);
518  selectFile(fn);
519  // That's the built-in line edit (fixes Debian bug #811200)
520  auto fileNameEdit = this->findChild<QLineEdit*>(QString::fromLatin1("fileNameEdit"));
521  if (fileNameEdit)
522  fileNameEdit->setText(fn);
523  }
524  }
525 
526  QFileDialog::accept();
527 }

604 QIcon FileIconProvider::icon(const QFileInfo & info) const
605 {
606  auto toUrl = [](const QFileInfo & info) {
607  QFileInfo fi(info);
608  fi.makeAbsolute();
609  QString fileName = fi.absoluteFilePath();
610  if (fi.isSymLink()) {
611  fileName = fi.symLinkTarget();
612  }
613 
614  return QUrl::fromLocalFile(fileName).toString();
615  };
616 
617  auto urlToThumbnail = [](const QString& filename) {
618  QString hash = QString::fromLatin1(QCryptographicHash::hash(filename.toUtf8(), QCryptographicHash::Md5).toHex());
619  QString cache = QStandardPaths::writableLocation(QStandardPaths::GenericCacheLocation);
620  return QString::fromLatin1("%1/thumbnails/normal/%2.png").arg(cache, hash);
621  };
622 
623  auto iconFromFile = [](const QString& filename) {
624  if (QFile::exists(filename)) {
625  QIcon icon(filename);
626  if (!icon.isNull())
627  return icon;
628  }
629 
630  return QIcon(QString::fromLatin1(":/icons/freecad-doc.png"));
631  };
632 
633  if (info.suffix().toLower() == QLatin1String("fcstd")) {
634  // Check if a thumbnail is available
635  QString fileName = toUrl(info);
636  QString thumb = urlToThumbnail(fileName);
637  return iconFromFile(thumb);
638  }
639  else if (info.suffix().toLower().startsWith(QLatin1String("fcstd"))) {
640  QIcon icon(QString::fromLatin1(":/icons/freecad-doc.png"));

644  darkIcon.addPixmap(icon.pixmap(w, w, QIcon::Disabled, QIcon::On ), QIcon::Normal, QIcon::On );
645  return darkIcon;

952 SelectModule::Dict SelectModule::exportHandler(const QStringList& fileNames, const QString& filter)
953 {
954  // first check if there is a certain filter selected
955  SelectModule::Dict dict;
956  if (!filter.isEmpty()) {
957  // If an export filter is specified search directly for the module
958  std::map<std::string, std::string> filterList = App::GetApplication().getExportFilters();
959  std::map<std::string, std::string>::const_iterator it;
960  it = filterList.find((const char*)filter.toUtf8());
961  if (it != filterList.end()) {
962  QString module = QString::fromLatin1(it->second.c_str());
963  for (const auto & fileName : fileNames) {
964  dict[fileName] = module;
965  }
966  return dict;
967  }
968  }
969 
970  // the global filter (or no filter) was selected. We now try to sort filetypes that are
971  // handled by more than one module and ask to the user to select one.
972  QMap<QString, SelectModule::Dict> filetypeHandler;
973  QMap<QString, QStringList > fileExtension;
974  for (const auto & fileName : fileNames) {
975  QFileInfo fi(fileName);
976  QString ext = fi.completeSuffix().toLower();
977  std::map<std::string, std::string> filters = App::GetApplication().getExportFilters(ext.toLatin1());
978 
979  if (filters.empty()) {
980  ext = fi.suffix().toLower();
981  filters = App::GetApplication().getExportFilters(ext.toLatin1());
982  }
983 
984  fileExtension[ext].push_back(fileName);
985  for (const auto & filter : filters)
986  filetypeHandler[ext][QString::fromUtf8(filter.first.c_str())] = QString::fromLatin1(filter.second.c_str());
987  // set the default module handler
988  if (!filters.empty())
989  dict[fileName] = QString::fromLatin1(filters.begin()->second.c_str());
990  }
991 
992  for (QMap<QString, SelectModule::Dict>::const_iterator it = filetypeHandler.cbegin();
993  it != filetypeHandler.cend(); ++it) {
994  if (it.value().size() > 1) {
995  SelectModule dlg(it.key(),it.value(), getMainWindow());
996  QApplication::beep();
997  if (dlg.exec()) {
998  QString mod = dlg.getModule();
999  const QStringList& files = fileExtension[it.key()];
1000  for (const auto & file : files)
1001  dict[file] = mod;
1002  }
1003  }
1004  }
1005 
1006  return dict;
1007 }
