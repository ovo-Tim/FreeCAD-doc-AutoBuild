23 template <class Real>
24 MeshCurvature<Real>::MeshCurvature (int iVQuantity,
25  const Vector3<Real>* akVertex, int iTQuantity, const int* aiIndex)
26 {
27  m_iVQuantity = iVQuantity;
28  m_akVertex = akVertex;
29  m_iTQuantity = iTQuantity;
30  m_aiIndex = aiIndex;
31 
32  // compute normal vectors
33  m_akNormal = WM4_NEW Vector3<Real>[m_iVQuantity];
34  std::fill_n(m_akNormal,m_iVQuantity,Vector3<Real>{0,0,0});
35  int i, iV0, iV1, iV2;
36  for (i = 0; i < m_iTQuantity; i++)
37  {
38  // get vertex indices
39  iV0 = *aiIndex++;
40  iV1 = *aiIndex++;
41  iV2 = *aiIndex++;
42 
43  // compute the normal (length provides a weighted sum)
44  Vector3<Real> kEdge1 = m_akVertex[iV1] - m_akVertex[iV0];
45  Vector3<Real> kEdge2 = m_akVertex[iV2] - m_akVertex[iV0];
46  Vector3<Real> kNormal = kEdge1.Cross(kEdge2);
47 
48  m_akNormal[iV0] += kNormal;
49  m_akNormal[iV1] += kNormal;
50  m_akNormal[iV2] += kNormal;
51  }
52  for (i = 0; i < m_iVQuantity; i++)
53  {
54  m_akNormal[i].Normalize();
55  }
56 
57  // compute the matrix of normal derivatives
58  Matrix3<Real>* akDNormal = WM4_NEW Matrix3<Real>[m_iVQuantity];
59  Matrix3<Real>* akWWTrn = WM4_NEW Matrix3<Real>[m_iVQuantity];
60  Matrix3<Real>* akDWTrn = WM4_NEW Matrix3<Real>[m_iVQuantity];
61  std::fill_n(akWWTrn,m_iVQuantity,Matrix3<Real>{0,0,0,0,0,0,0,0,0});
62  std::fill_n(akDWTrn,m_iVQuantity,Matrix3<Real>{0,0,0,0,0,0,0,0,0});
63 
64  int iRow, iCol;
65  aiIndex = m_aiIndex;
66  for (i = 0; i < m_iTQuantity; i++)
67  {
68  // get vertex indices
69  int aiV[3];
70  aiV[0] = *aiIndex++;
71  aiV[1] = *aiIndex++;
72  aiV[2] = *aiIndex++;
73 
74  for (int j = 0; j < 3; j++)
75  {
76  iV0 = aiV[j];
77  iV1 = aiV[(j+1)%3];
78  iV2 = aiV[(j+2)%3];
79 
80  // Compute edge from V0 to V1, project to tangent plane of vertex,
81  // and compute difference of adjacent normals.
82  Vector3<Real> kE = m_akVertex[iV1] - m_akVertex[iV0];
83  Vector3<Real> kW = kE - (kE.Dot(m_akNormal[iV0]))*m_akNormal[iV0];
84  Vector3<Real> kD = m_akNormal[iV1] - m_akNormal[iV0];
85  for (iRow = 0; iRow < 3; iRow++)
86  {
87  for (iCol = 0; iCol < 3; iCol++)
88  {
89  akWWTrn[iV0][iRow][iCol] += kW[iRow]*kW[iCol];
90  akDWTrn[iV0][iRow][iCol] += kD[iRow]*kW[iCol];
91  }
92  }
93 
94  // Compute edge from V0 to V2, project to tangent plane of vertex,
95  // and compute difference of adjacent normals.
96  kE = m_akVertex[iV2] - m_akVertex[iV0];
97  kW = kE - (kE.Dot(m_akNormal[iV0]))*m_akNormal[iV0];
98  kD = m_akNormal[iV2] - m_akNormal[iV0];
99  for (iRow = 0; iRow < 3; iRow++)
100  {
101  for (iCol = 0; iCol < 3; iCol++)
102  {
103  akWWTrn[iV0][iRow][iCol] += kW[iRow]*kW[iCol];
104  akDWTrn[iV0][iRow][iCol] += kD[iRow]*kW[iCol];
105  }
106  }
107  }
108  }
109 
110  // Add in N*N^T to W*W^T for numerical stability. In theory 0*0^T gets
111  // added to D*W^T, but of course no update needed in the implementation.
112  // Compute the matrix of normal derivatives.
113  for (i = 0; i < m_iVQuantity; i++)
114  {
115  for (iRow = 0; iRow < 3; iRow++)
116  {
117  for (iCol = 0; iCol < 3; iCol++)
118  {
119  akWWTrn[i][iRow][iCol] = ((Real)0.5)*akWWTrn[i][iRow][iCol] +
120  m_akNormal[i][iRow]*m_akNormal[i][iCol];
121  akDWTrn[i][iRow][iCol] *= (Real)0.5;
122  }
123  }
124 
125  akDNormal[i] = akDWTrn[i]*akWWTrn[i].Inverse();
126  }
127 
128  WM4_DELETE[] akWWTrn;
129  WM4_DELETE[] akDWTrn;
130 
131  // If N is a unit-length normal at a vertex, let U and V be unit-length
132  // tangents so that {U, V, N} is an orthonormal set. Define the matrix
133  // J = [U | V], a 3-by-2 matrix whose columns are U and V. Define J^T
134  // to be the transpose of J, a 2-by-3 matrix. Let dN/dX denote the
135  // matrix of first-order derivatives of the normal vector field. The
136  // shape matrix is
137  // S = (J^T * J)^{-1} * J^T * dN/dX * J = J^T * dN/dX * J
138  // where the superscript of -1 denotes the inverse. (The formula allows
139  // for J built from non-perpendicular vectors.) The matrix S is 2-by-2.
140  // The principal curvatures are the eigenvalues of S. If k is a principal
141  // curvature and W is the 2-by-1 eigenvector corresponding to it, then
142  // S*W = k*W (by definition). The corresponding 3-by-1 tangent vector at
143  // the vertex is called the principal direction for k, and is J*W.
144  m_afMinCurvature = WM4_NEW Real[m_iVQuantity];
145  m_afMaxCurvature = WM4_NEW Real[m_iVQuantity];
146  m_akMinDirection = WM4_NEW Vector3<Real>[m_iVQuantity];
147  m_akMaxDirection = WM4_NEW Vector3<Real>[m_iVQuantity];
148  for (i = 0; i < m_iVQuantity; i++)
149  {
150  // compute U and V given N
151  Vector3<Real> kU, kV;
152  Vector3<Real>::GenerateComplementBasis(kU,kV,m_akNormal[i]);
153 
154  // Compute S = J^T * dN/dX * J. In theory S is symmetric, but
155  // because we have estimated dN/dX, we must slightly adjust our
156  // calculations to make sure S is symmetric.
157  Real fS01 = kU.Dot(akDNormal[i]*kV);
158  Real fS10 = kV.Dot(akDNormal[i]*kU);
159  Real fSAvr = ((Real)0.5)*(fS01+fS10);
160  Matrix2<Real> kS
161  (
162  kU.Dot(akDNormal[i]*kU), fSAvr,
163  fSAvr, kV.Dot(akDNormal[i]*kV)
164  );
165 
166  // compute the eigenvalues of S (min and max curvatures)
167  Real fTrace = kS[0][0] + kS[1][1];
168  Real fDet = kS[0][0]*kS[1][1] - kS[0][1]*kS[1][0];
169  Real fDiscr = fTrace*fTrace - ((Real)4.0)*fDet;
170  Real fRootDiscr = Math<Real>::Sqrt(Math<Real>::FAbs(fDiscr));
171  m_afMinCurvature[i] = ((Real)0.5)*(fTrace - fRootDiscr);
172  m_afMaxCurvature[i] = ((Real)0.5)*(fTrace + fRootDiscr);
173 
174  // compute the eigenvectors of S
175  Vector2<Real> kW0(kS[0][1],m_afMinCurvature[i]-kS[0][0]);
176  Vector2<Real> kW1(m_afMinCurvature[i]-kS[1][1],kS[1][0]);
177  if (kW0.SquaredLength() >= kW1.SquaredLength())
178  {
179  kW0.Normalize();
180  m_akMinDirection[i] = kW0.X()*kU + kW0.Y()*kV;
181  }
182  else
183  {
184  kW1.Normalize();
185  m_akMinDirection[i] = kW1.X()*kU + kW1.Y()*kV;
186  }
187 
188  kW0 = Vector2<Real>(kS[0][1],m_afMaxCurvature[i]-kS[0][0]);
189  kW1 = Vector2<Real>(m_afMaxCurvature[i]-kS[1][1],kS[1][0]);
190  if (kW0.SquaredLength() >= kW1.SquaredLength())
191  {
192  kW0.Normalize();
193  m_akMaxDirection[i] = kW0.X()*kU + kW0.Y()*kV;
194  }
195  else
196  {
197  kW1.Normalize();
198  m_akMaxDirection[i] = kW1.X()*kU + kW1.Y()*kV;
199  }
200  }
201 
202  WM4_DELETE[] akDNormal;
203 }
