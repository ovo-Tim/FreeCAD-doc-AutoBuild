67 DocumentObjectExecReturn* FemPostFilter::execute()
68 {
69  if (!m_pipelines.empty() && !m_activePipeline.empty()) {
70  FemPostFilter::FilterPipeline& pipe = m_pipelines[m_activePipeline];
71  vtkSmartPointer<vtkDataObject> data = getInputData();
72  if (!data || !data->IsA("vtkDataSet")) {
73  return StdReturn;
74  }
75 
76  if ((m_activePipeline == "DataAlongLine") || (m_activePipeline == "DataAtPoint")) {
77  pipe.filterSource->SetSourceData(getInputData());
78  pipe.filterTarget->Update();
79  Data.setValue(pipe.filterTarget->GetOutputDataObject(0));
80  }
81  else {
82  pipe.source->SetInputDataObject(data);
83  pipe.target->Update();
84  Data.setValue(pipe.target->GetOutputDataObject(0));
85  }
86  }
87 
88  return StdReturn;
89 }

91 vtkDataObject* FemPostFilter::getInputData()
92 {
93  if (Input.getValue()) {
94  if (Input.getValue()->getTypeId().isDerivedFrom(
95  Base::Type::fromName("Fem::FemPostObject"))) {
96  return Input.getValue<FemPostObject*>()->Data.getValue();
97  }
98  else {
99  throw std::runtime_error(
100  "The filter's Input object is not a 'Fem::FemPostObject' object!");
101  }
102  }
103  else {
104  // get the pipeline and use the pipelinedata
105  std::vector<App::DocumentObject*> objs =
106  getDocument()->getObjectsOfType(FemPostPipeline::getClassTypeId());
107  for (auto it : objs) {
108  if (static_cast<FemPostPipeline*>(it)->holdsPostObject(this)) {
109  return static_cast<FemPostObject*>(it)->Data.getValue();
110  }
111  }
112  }
113 
114  return nullptr;
115 }

258 void FemPostDataAlongLineFilter::GetAxisData()
259 {
260  std::vector<double> coords;
261  std::vector<double> values;
262 
263  vtkSmartPointer<vtkDataObject> data = m_probe->GetOutputDataObject(0);
264  vtkDataSet* dset = vtkDataSet::SafeDownCast(data);
265  if (!dset) {
266  return;
267  }
268  vtkDataArray* pdata = dset->GetPointData()->GetArray(PlotData.getValue());
269  // VTK cannot deliver data when the filer relies e.g. on a scalar clip filter
270  // whose value is set so that all data are clipped
271  if (!pdata) {
272  return;
273  }
274 
275  // expected "Magnitude" -> 0; "X" -> 1; "Y" -> 2, "Z" -> 3
276  vtkIdType component = PlotDataComponent.getValue();
277  // prevent selecting a component out of range
278  if (!PlotDataComponent.isValid() || component > pdata->GetNumberOfComponents()) {
279  return;
280  }
281 
282  vtkDataArray* tcoords = dset->GetPointData()->GetTCoords("Texture Coordinates");
283 
284  const Base::Vector3d& vec1 = Point1.getValue();
285  const Base::Vector3d& vec2 = Point2.getValue();
286  const Base::Vector3d diff = vec1 - vec2;
287  double Len = diff.Length();
288 
289  for (vtkIdType i = 0; i < dset->GetNumberOfPoints(); ++i) {
290  double value = 0;
291  if (pdata) {
292  if (pdata->GetNumberOfComponents() == 1) {
293  value = pdata->GetComponent(i, 0);
294  }
295  else if (pdata->GetNumberOfComponents() > 1) {
296  if (component) {
297  value = pdata->GetComponent(i, component - 1);
298  }
299  else {
300  // compute magnitude
301  for (vtkIdType j = 0; j < pdata->GetNumberOfComponents(); ++j) {
302  value += std::pow(pdata->GetComponent(i, j), 2);
303  }
304 
305  value = std::sqrt(value);
306  }
307  }
308  }
309 
310  values.push_back(value);
311  double tcoord = tcoords->GetComponent(i, 0);
312  coords.push_back(tcoord * Len);
313  }
314 
315  YAxisData.setValues(values);
316  XAxisData.setValues(coords);
317 }

656 {

693  // would not get any data
694  if ((prop == &Field || prop == &VectorMode || prop == &NumberOfContours || prop == &Data)
695  && (Field.getValue() >= 0)) {
696  double p[2];
697 
698  // get the field and its data
699  vtkSmartPointer<vtkDataObject> data = getInputData();
700  vtkDataSet* dset = vtkDataSet::SafeDownCast(data);
701  if (!dset) {
702  return;
703  }
704  vtkDataArray* pdata = dset->GetPointData()->GetArray(Field.getValueAsString());
705  if (!pdata) {
706  return;
707  }
708  if (pdata->GetNumberOfComponents() == 1) {
709  // if we have a scalar, we can directly use the array
710  m_contours->SetInputArrayToProcess(0,
711  0,
712  0,
713  vtkDataObject::FIELD_ASSOCIATION_POINTS,
714  Field.getValueAsString());
715  pdata->GetRange(p);
716  recalculateContours(p[0], p[1]);
717  }
718  else {
719  // The contour filter handles vectors by taking always its first component.
720  // There is no other solution than to make the desired vectorn component a
721  // scalar array and append this temporarily to the data. (vtkExtractVectorComponents
722  // does not work because our data is an unstructured data set.)
723  int component = -1;
724  if (VectorMode.getValue() == 1) {
725  component = 0;
726  }
727  else if (VectorMode.getValue() == 2) {
728  component = 1;
729  }
730  else if (VectorMode.getValue() == 3) {
731  component = 2;
732  }
733  // extract the component to a new array
734  vtkSmartPointer<vtkDoubleArray> componentArray = vtkSmartPointer<vtkDoubleArray>::New();
735  componentArray->SetNumberOfComponents(1);
736  vtkIdType numTuples = pdata->GetNumberOfTuples();
737  componentArray->SetNumberOfTuples(numTuples);
738 
739  if (component >= 0) {
740  for (vtkIdType tupleIdx = 0; tupleIdx < numTuples; ++tupleIdx) {
741  componentArray->SetComponent(tupleIdx,
742  0,
743  pdata->GetComponent(tupleIdx, component));
744  }
745  }
746  else {
747  for (vtkIdType tupleIdx = 0; tupleIdx < numTuples; ++tupleIdx) {
748  componentArray->SetComponent(
749  tupleIdx,
750  0,
751  std::sqrt(
752  pdata->GetComponent(tupleIdx, 0) * pdata->GetComponent(tupleIdx, 0)
753  + pdata->GetComponent(tupleIdx, 1) * pdata->GetComponent(tupleIdx, 1)
754  + pdata->GetComponent(tupleIdx, 2) * pdata->GetComponent(tupleIdx, 2)));
755  }
756  }
757  // name the array
758  contourFieldName = std::string(Field.getValueAsString()) + "_contour";
759  componentArray->SetName(contourFieldName.c_str());
760 
761  // add the array as new field and use it for the contour filter
762  dset->GetPointData()->AddArray(componentArray);
763  m_contours->SetInputArrayToProcess(0,
764  0,
765  0,
766  vtkDataObject::FIELD_ASSOCIATION_POINTS,
767  contourFieldName.c_str());
768  componentArray->GetRange(p);
769  recalculateContours(p[0], p[1]);
770  if (prop == &Data) {
771  // we must recalculate to pass the new created contours field
772  // to ViewProviderFemPostObject
773  m_blockPropertyChanges = true;
774  execute();
775  m_blockPropertyChanges = false;
776  }
777  }
778  }
779 
780  Fem::FemPostFilter::onChanged(prop);
781 }
