88 {
89  static const std::array<const char*, 5> kwlist {"shape",
90  "toolnumber",
91  "diameter",
92  "resolution",
93  nullptr};
94  PyObject* pObjToolShape;
95  int toolNumber;
96  float resolution;
97  float diameter;
98  if (!Base::Wrapped_ParseTupleAndKeywords(args,
99  kwds,
100  "Oiff",
101  kwlist,
102  &pObjToolShape,
103  &toolNumber,
104  &diameter,
105  &resolution)) {
106  return nullptr;
107  }
108  // The tool shape is defined by a list of 2d points that represents the tool revolving profile
109  Py_ssize_t num_floats = PyList_Size(pObjToolShape);
110  std::vector<float> toolProfile;
111  for (Py_ssize_t i = 0; i < num_floats; ++i) {
112  PyObject* item = PyList_GetItem(pObjToolShape, i);
113  toolProfile.push_back(static_cast<float>(PyFloat_AsDouble(item)));
114  }
115 
116  CAMSim* sim = getCAMSimPtr();
117  sim->addTool(toolProfile, toolNumber, diameter, resolution);
118 
119  Py_INCREF(Py_None);
120  return Py_None;
121 }
