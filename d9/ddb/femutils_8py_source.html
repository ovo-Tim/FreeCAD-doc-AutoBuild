48 def createObject(doc, name, proxy, viewProxy=None):
49  """Add python object to document using python type string.
50 
51  Add a document object suitable for the *proxy* and the *viewProxy* to *doc*
52  and attach it to the *proxy* and the *viewProxy*. This function can only be
53  used with python proxies that specify their C++ type via the BaseType class
54  member (e.g. Cube.BaseType). If there already exists a object with *name* a
55  suitable unique name is generated. To auto generate a name pass ``""``.
56 
57  :param doc: document object to which the object is added
58  :param name: string of the name of new object in *doc*, use
59  ``""`` to generate a name
60  :param proxy: python proxy for new object
61  :param viewProxy: view proxy for new object
62 
63  :returns: reference to new object
64  """
65  obj = doc.addObject(proxy.BaseType, name)
66  proxy(obj)
67  if FreeCAD.GuiUp and viewProxy is not None:
68  viewProxy(obj.ViewObject)
69  return obj

73 def type_of_obj(obj):
74  """Return type of *obj* honoring the special typesystem of Fem.
75 
76  Python objects of the Fem workbench define their type via a class member
77  ``<Class>.Type``. Return this type if the property exists. If not return
78  the conventional ``TypeId`` value.
79 
80  :para obj: a document object
81  """
82  if hasattr(obj, "Proxy") and hasattr(obj.Proxy, "Type"):
83  return obj.Proxy.Type
84  return obj.TypeId

87 def is_of_type(obj, ty):
88  """Compare type of *obj* with *ty* honoring Fems typesystem.
89 
90  See :py:func:`type_of_obj` for more info about the special typesystem of
91  the Fem module.
92 
93  :returns:
94  ``True`` if *obj* is of type *ty*, ``False`` otherwise. Type must match
95  exactly: Derived objects are not considered to be of type of one of their
96  super classes.
97  """
98  return type_of_obj(obj) == ty

101 def is_derived_from(obj, t):
102  """Check if *obj* is derived from *t* honoring Fems typesystem.
103 
104  Essentially just call ``obj.isDerivedFrom(t)`` and return it's value. For
105  objects using Fems typesystem (see :py:func:`type_of_obj`) return always
106  True if the Fem type is equal to *t*.
107 
108  :note:
109  Inheritance of Fem types is not checked. If *obj* uses Fems typesystem the
110  type is just checked for equality. If the type doesn't match
111  ``obj.isDerivedFrom`` is called as usual. See
112  https://forum.freecad.org/viewtopic.php?f=10&t=32625
113  """
114  if hasattr(obj, "Proxy") and hasattr(obj.Proxy, "Type") and obj.Proxy.Type == t:
115  return True
116  return obj.isDerivedFrom(t)

236 def getBoundBoxOfAllDocumentShapes(doc):
237  """Calculate bounding box containing all objects inside *doc*.
238 
239  :returns:
240  A bounding box containing all objects that have a *Shape* attribute (all
241  Part and PartDesign objects). If the document contains no such objects or
242  no objects at all return ``None``.
243  """
244  overallboundbox = None
245  # netgen mesh obj has an attribute Shape which is an Document obj, which has no BB
246  # a FemMesh without a Shape could be clipped too
247  # https://forum.freecad.org/viewtopic.php?f=18&t=52920
248  for o in doc.Objects:
249 
250  FreeCAD.Console.PrintMessage(":\n") # debug only
251  bb = None
252 
253  try:
254  FreeCAD.Console.PrintMessage(
255  f"trying: {o.Label}: getPropertyOfGeometry()\n"
256  ) # debug only
257  bb = o.getPropertyOfGeometry().BoundBox
258  FreeCAD.Console.PrintMessage(f"{bb}\n") # debug only
259  except Exception:
260  FreeCAD.Console.PrintMessage("exception \n") # debug only
261 
262  if bb is None:
263  try:
264  FreeCAD.Console.PrintMessage(f"trying: {o.Label}: FemMesh\n") # debug only
265  bb = o.FemMesh.BoundBox
266  FreeCAD.Console.PrintMessage(f"{bb}\n") # debug only
267  except Exception:
268  FreeCAD.Console.PrintMessage("exception \n") # debug only
269 
270  if bb:
271  if bb.isValid():
272  if not overallboundbox:
273  overallboundbox = bb
274  else:
275  overallboundbox.add(bb)
276  else: # debug only
277  FreeCAD.Console.PrintMessage("no bb\n") # debug only
278 
279  FreeCAD.Console.PrintMessage("overallBB:" + str(overallboundbox) + "\n") # debug only
280  return overallboundbox

312 def get_refshape_type(fem_doc_object):
313  """Return shape type the constraints references.
314 
315  Determine single shape type of references of *fem_doc_object* which must be
316  a constraint (=have a *References* property). All references must be of the
317  same type which is than returned as a string. A type can be "Vertex",
318  "Edge", "Face" or "Solid".
319 
320  :param fem_doc_object:
321  A constraint object with a *References* property.
322 
323  :returns:
324  A string representing the shape type ("Vertex", "Edge", "Face" or
325  "Solid"). If *fem_doc_object* isn't a constraint ``""`` is returned.
326 
327  :note:
328  Undefined behaviour if the type of the references of one object are
329  not all the same.
330 
331  :note:
332  Undefined behaviour if constraint contains no references (empty list).
333  """
334  from femtools.geomtools import get_element
335 
336  if hasattr(fem_doc_object, "References") and fem_doc_object.References:
337  first_ref_obj = fem_doc_object.References[0]
338  first_ref_shape = get_element(first_ref_obj[0], first_ref_obj[1][0])
339  st = first_ref_shape.ShapeType
340  FreeCAD.Console.PrintLog(
341  f"References: {st} in {fem_doc_object.Name}, {fem_doc_object.Label}\n"
342  )
343  return st
344  else:
345  FreeCAD.Console.PrintLog(
346  f"References: empty in {fem_doc_object.Name}, {fem_doc_object.Label}\n"
347  )
348  return ""
