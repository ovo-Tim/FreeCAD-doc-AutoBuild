94 Attacher::eMapMode SuggestAutoMapMode(Attacher::SuggestResult::eSuggestResult* pMsgId = nullptr,
95  QString* message = nullptr,
96  std::vector<Attacher::eMapMode>* allmodes = nullptr)
97 {
98  // convert pointers into valid references, to avoid checking for null pointers everywhere
99  Attacher::SuggestResult::eSuggestResult buf;
100  if (!pMsgId)
101  pMsgId = &buf;
102  Attacher::SuggestResult::eSuggestResult& msg = *pMsgId;
103  QString buf2;
104  if (!message)
105  message = &buf2;
106  QString& msg_str = *message;
107 
108  App::PropertyLinkSubList tmpSupport;
109  Gui::Selection().getAsPropertyLinkSubList(tmpSupport);
110 
111  Attacher::SuggestResult sugr;
112  AttachEngine3D eng;
113  eng.setUp(tmpSupport);
114  eng.suggestMapModes(sugr);
115  if (allmodes)
116  *allmodes = sugr.allApplicableModes;
117  msg = sugr.message;
118  switch (msg) {
119  case Attacher::SuggestResult::srOK:
120  break;
121  case Attacher::SuggestResult::srNoModesFit:
122  msg_str = QObject::tr("There are no modes that accept the selected set of subelements");
123  break;
124  case Attacher::SuggestResult::srLinkBroken:
125  msg_str = QObject::tr("Broken link to support subelements");
126  break;
127  case Attacher::SuggestResult::srUnexpectedError:
128  msg_str = QObject::tr("Unexpected error");
129  break;
130  case Attacher::SuggestResult::srIncompatibleGeometry:
131  if (tmpSupport.getSubValues()[0].substr(0, 4) == std::string("Face"))
132  msg_str = QObject::tr("Face is non-planar");
133  else
134  msg_str = QObject::tr("Selected shapes are of wrong form (e.g., a curved edge "
135  "where a straight one is needed)");
136  break;
137  default:
138  msg_str = QObject::tr("Unexpected error");
139  assert(0 /*no message for eSuggestResult enum item*/);
140  }
141 
142  return sugr.bestFitMode;
143 }

569 
570  // check that selection is valid for at least some mapping mode.
571  Attacher::SuggestResult::eSuggestResult msgid = Attacher::SuggestResult::srOK;
572  suggMapMode = SuggestAutoMapMode(&msgid, &msg_str, &validModes);
573  bool sketchInSelection = false;
574  std::vector<App::DocumentObject*> selectedSketches = Gui::Selection()
575  .getObjectsOfType(Part::Part2DObject::getClassTypeId());
576  App::Document* doc = App::GetApplication().getActiveDocument();
577  std::vector<App::DocumentObject*> sketches =
578  doc->getObjectsOfType(Part::Part2DObject::getClassTypeId());
579 
586  auto newEnd = std::remove_if(sketches.begin(), sketches.end(),
587  [&selectedSketches, &sketchInSelection](App::DocumentObject* obj) {
588  Part::Part2DObject* sketch = dynamic_cast<Part::Part2DObject*>(obj);
589  if (sketch && std::find(selectedSketches.begin(),
590  selectedSketches.end(), sketch) != selectedSketches.end()) {
591  sketchInSelection = true;
592  return true;
593  }
594  return false;
595  });
596  sketches.erase(newEnd, sketches.end());
597 
598  if (sketches.empty()) {
599  Gui::TranslatedUserWarning(
600  doc->Label.getStrValue(),
601  qApp->translate("Sketcher_MapSketch", "No sketch found"),
602  sketchInSelection
603  ? qApp->translate("Sketcher_MapSketch", "Cannot attach sketch to itself!")
604  : qApp->translate("Sketcher_MapSketch", "The document doesn't have a sketch"));
605 
606  return;
607  }
608  std::sort(sketches.begin(), sketches.end(), [](const auto &a, const auto &b) {
609  return QString::fromUtf8(a->Label.getValue()) < QString::fromUtf8(b->Label.getValue());
610  });
611 
612  bool ok;
613  QStringList items;
614  for (std::vector<App::DocumentObject*>::iterator it = sketches.begin();
615  it != sketches.end();
616  ++it)
617  items.push_back(QString::fromUtf8((*it)->Label.getValue()));
618  QString text = QInputDialog::getItem(
619  Gui::getMainWindow(),
620  qApp->translate("Sketcher_MapSketch", "Select sketch"),
621  sketchInSelection
622  ? qApp->translate("Sketcher_MapSketch",
623  "Select a sketch (some sketches not shown to prevent a circular dependency)")
624  : qApp->translate("Sketcher_MapSketch", "Select a sketch from the list"),
625  items,
626  0,
627  false,
628  &ok,
629  Qt::MSWindowsFixedSizeDialogHint);
630  if (!ok)
631  return;
632  int index = items.indexOf(text);
633  Part2DObject* sketch = static_cast<Part2DObject*>(sketches[index]);
634 
635  // check circular dependency
636  std::vector<Gui::SelectionObject> selobjs = Gui::Selection().getSelectionEx();
637  for (size_t i = 0; i < selobjs.size(); ++i) {
638  App::DocumentObject* part = static_cast<Part::Feature*>(selobjs[i].getObject());
