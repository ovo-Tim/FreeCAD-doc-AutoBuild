170 def _param_observer_callback_svg_pattern():
171  # imports have to happen here to avoid circular imports
172  from draftutils import utils
173  from draftviewproviders import view_base
174  utils.load_svg_patterns()
175  if App.ActiveDocument is None:
176  return
177 
178  pats = list(utils.svg_patterns())
179  pats.sort()
180  pats = ["None"] + pats
181 
182  data = []
183  for doc in App.listDocuments().values():
184  vobjs = []
185  for obj in doc.Objects:
186  if hasattr(obj, "ViewObject"):
187  vobj = obj.ViewObject
188  if hasattr(vobj, "Pattern") \
189  and hasattr(vobj, "Proxy") \
190  and isinstance(vobj.Proxy, view_base.ViewProviderDraft) \
191  and vobj.getEnumerationsOfProperty("Pattern") != pats:
192  vobjs.append(vobj)
193  if vobjs:
194  data.append([doc, vobjs])
195  if not data:
196  return
197 
198  msg = translate("draft",
199 """Do you want to update the SVG pattern options
200 of existing objects in all opened documents?""")
201  res = QtWidgets.QMessageBox.question(None, "Update SVG patterns", msg,
202  QtWidgets.QMessageBox.Yes | QtWidgets.QMessageBox.No,
203  QtWidgets.QMessageBox.No)
204  if res == QtWidgets.QMessageBox.No:
205  return
206 
207  for doc, vobjs in data:
208  doc.openTransaction("SVG pattern update")
209  for vobj in vobjs:
210  old = vobj.Pattern
211  if old in pats:
212  vobj.Pattern = pats
213  else:
214  tmp_pats = [old] + pats[1:]
215  tmp_pats.sort()
216  vobj.Pattern = ["None"] + tmp_pats
217  vobj.Pattern = old
218  doc.commitTransaction()

610 def get_param(entry, path="Mod/Draft", ret_default=False):
611  """Return a stored parameter value or its default.
612 
613  Parameters
614  ----------
615  entry: str
616  Name of the parameter.
617  path: str, optional
618  Defaults to "Mod/Draft".
619  The path where the parameter can be found.
620  This string is appended to "User parameter:BaseApp/Preferences/".
621  ret_default: bool, optional
622  Defaults to `False`.
623  If `True`, always return the default value even if a stored value is available.
624 
625  Returns
626  -------
627  bool, float, int or str (if successful) or `None`.
628  """
629  if path not in PARAM_DICT or entry not in PARAM_DICT[path]:
630  print(f"draftutils.params.get_param: Unable to find '{entry}' in '{path}'")
631  return None
