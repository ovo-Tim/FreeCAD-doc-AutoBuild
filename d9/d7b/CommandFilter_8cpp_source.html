54 void PartCmdSelectFilter::activated(int iMsg)
55 {
56  Gui::CommandManager &rcCmdMgr = Gui::Application::Instance->commandManager();
57  if (iMsg==0) {
58  rcCmdMgr.runCommandByName("Part_VertexSelection");
59  }
60  else if (iMsg==1) {
61  rcCmdMgr.runCommandByName("Part_EdgeSelection");
62  }
63  else if (iMsg==2) {
64  rcCmdMgr.runCommandByName("Part_FaceSelection");
65  }
66  else if (iMsg==3) {
67  rcCmdMgr.runCommandByName("Part_RemoveSelectionGate");
68  }
69  else {
70  return;
71  }
72 
73  // Since the default icon is reset when enabling/disabling the command we have
74  // to explicitly set the icon of the used command.
75  auto pcAction = qobject_cast<Gui::ActionGroup*>(_pcAction);
76  QList<QAction*> act = pcAction->actions();
77 
78  assert(iMsg < act.size());
79  pcAction->setIcon(act[iMsg]->icon());
80 }
