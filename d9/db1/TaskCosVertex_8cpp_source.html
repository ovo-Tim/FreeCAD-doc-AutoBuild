182 void TaskCosVertex::startTracker()
183 {
184 // Base::Console().Message("TCV::startTracker()\n");
185  if (m_trackerMode == QGTracker::TrackerMode::None) {
186  return;
187  }
188 
189  if (!m_tracker) {
190  m_tracker = new QGTracker(m_vpp->getQGSPage(), m_trackerMode);
191  QObject::connect(
192  m_tracker, &QGTracker::drawingFinished,
193  this, &TaskCosVertex::onTrackerFinished
194  );
195  }
196  else {
197  //this is too harsh. but need to avoid restarting process
198  throw Base::RuntimeError("TechDrawNewLeader - tracker already active\n");
199  }
200  setEditCursor(Qt::CrossCursor);
201  QString msg = tr("Left click to set a point");
202  Gui::getMainWindow()->statusBar()->show();
203  Gui::getMainWindow()->showMessage(msg, 3000);
204 }

206 void TaskCosVertex::onTrackerFinished(std::vector<QPointF> pts, QGIView* qgParent)
207 {
208  (void)qgParent;
209  if (pts.empty()) {
210  Base::Console().Error("TaskCosVertex - no points available\n");
211  return;
212  }
213 
214  QPointF dragEnd = pts.front(); //scene pos of mouse click
215 
216  double x = Rez::guiX(m_baseFeat->X.getValue());
217  double y = Rez::guiX(m_baseFeat->Y.getValue());
218 
219  DrawViewPart* dvp = m_baseFeat;
220  DrawProjGroupItem* dpgi = dynamic_cast<DrawProjGroupItem*>(dvp);
221  if (dpgi) {
222  DrawProjGroup* dpg = dpgi->getPGroup();
223  if (dpg) {
224  x += Rez::guiX(dpg->X.getValue());
225  y += Rez::guiX(dpg->Y.getValue());
226  }
227  }
228  //x, y are scene pos of dvp/dpgi
229 
230  QPointF basePosScene(x, -y); //base position in scene coords
231  QPointF scenePosCV = dragEnd - basePosScene;
232 
233  // Invert Y value so the math works.
234  // scenePosCV is effectively a scaled (and rotated), inverted value
235  // Base::Vector3d posToRotate = DU::invertY(DU::toVector3d(scenePosCV));
236 
237  // unscale and rotate the picked point
238  Base::Vector3d posToRotate = CosmeticVertex::makeCanonicalPointInverted(m_baseFeat, DU::toVector3d(scenePosCV));
239  // now put Y value back to display form
240  scenePosCV = DU::toQPointF(posToRotate);
241 
242  m_savePoint = Rez::appX(scenePosCV);
243  updateUi();
244 
245  m_tracker->sleep(true);
246  m_inProgressLock = false;
247  m_pbTrackerState = TrackerAction::PICK;
248  ui->pbTracker->setText(tr("Pick Points"));
249  ui->pbTracker->setEnabled(true);
250  enableTaskButtons(true);
251  setEditCursor(Qt::ArrowCursor);
252  m_vpp->getMDIViewPage()->setContextMenuPolicy(m_saveContextPolicy);
253 
254 }

299 bool TaskCosVertex::accept()
300 {
301  Gui::Document* doc = Gui::Application::Instance->getDocument(m_basePage->getDocument());
302  if (!doc)
303  return false;
304 
305  removeTracker();
306  // whatever is in the ui for x,y is treated as an unscaled, unrotated, conventional Y position.
307  // the position from the tracker is unscaled & unrotated before updating the ui
308  double x = ui->dsbX->value().getValue();
309  double y = ui->dsbY->value().getValue();
310  QPointF uiPoint(x, y);
311 
312  addCosVertex(uiPoint);
313 
314  m_baseFeat->recomputeFeature();
315  m_baseFeat->requestPaint();
316  m_vpp->getMDIViewPage()->setContextMenuPolicy(m_saveContextPolicy);
317  m_trackerMode = QGTracker::TrackerMode::None;
318  Gui::Command::doCommand(Gui::Command::Gui, "Gui.ActiveDocument.resetEdit()");
319 
320  return true;
321 }
322 
323 bool TaskCosVertex::reject()

325  Gui::Document* doc = Gui::Application::Instance->getDocument(m_basePage->getDocument());
326  if (!doc)
327  return false;
328 
329  removeTracker();
330  m_trackerMode = QGTracker::TrackerMode::None;
331  if (m_vpp->getMDIViewPage()) {
332  m_vpp->getMDIViewPage()->setContextMenuPolicy(m_saveContextPolicy);
333  }
334 
335  //make sure any dangling objects are cleaned up
336  Gui::Command::doCommand(Gui::Command::Gui, "App.activeDocument().recompute()");
337  Gui::Command::doCommand(Gui::Command::Gui, "Gui.ActiveDocument.resetEdit()");
338
