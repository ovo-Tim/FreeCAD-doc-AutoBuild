44 bool ViewProviderBase::doubleClicked()
45 {
46  // If the Placement is mutable then open the transform panel.
47  // If the Placement can't be modified then just do nothing on double-click.
48  PartDesign::FeatureBase* base = getObject<PartDesign::FeatureBase>();
49  if (!base->Placement.testStatus(App::Property::Immutable) &&
50  !base->Placement.testStatus(App::Property::ReadOnly) &&
51  !base->Placement.testStatus(App::Property::Hidden)) {
52 
53  try {
54  std::string Msg("Edit ");
55  Msg += base->Label.getValue();
56  Gui::Command::openCommand(Msg.c_str());
57  FCMD_SET_EDIT(base);
58  }
59  catch (const Base::Exception&) {
60  Gui::Command::abortCommand();
61  }
62  return true;
63  }
64 
65  return false;
66 }
