170  def _calculateEigenfrequencies(self, output):
171  # takes the EigenSolve results and performs the calculation
172  # sqrt(aResult) / 2*PI but with aResult as complex number
173 
174  # first search the output file for the results
175  OutputList = output.split("\n")
176  modeNumber = 0
177  modeCount = 0
178  real = 0
179  imaginary = 0
180  haveImaginary = False
181  FrequencyList = []
182  for line in OutputList:
183  LineList = line.split(" ")
184  if len(LineList) > 1 and LineList[0] == "EigenSolve:" and LineList[1] == "Computed":
185  # we found a result and take now the next LineList[2] lines
186  modeCount = int(LineList[2])
187  modeNumber = modeCount
188  continue
189  if modeCount > 0:
190  for LineString in reversed(LineList):
191  # the output of Elmer may vary, we only know the last float
192  # is the imaginary and second to last float the real part
193  if self._isNumber(LineString):
194  if not haveImaginary:
195  imaginary = float(LineString)
196  haveImaginary = True
197  else:
198  real = float(LineString)
199  break
200  eigenFreq = complex(real, imaginary)
201  haveImaginary = False
202  # now we can perform the calculation
203  eigenFreq = cmath.sqrt(eigenFreq) / (2 * cmath.pi)
204  # create an output line
205  FrequencyList.append(f"Mode {modeNumber - modeCount + 1}: {eigenFreq.real} Hz")
206  modeCount = modeCount - 1
207  if modeNumber > 0:
208  # push the results and append to output
209  self.pushStatus("\n\nEigenfrequency results:")
210  output = output + "\n\nEigenfrequency results:"
211  for i in range(0, modeNumber):
212  output = output + "\n" + FrequencyList[i]
213  self.pushStatus("\n" + FrequencyList[i])
214  self.pushStatus("\n")
215  return output

266  def _handleTransientResults(self):
267  # for transient results we must create a result pipeline for every time
268  # the connection between result files and and their time is in the FreeCAD.pvd file
269  # therefore first open FreeCAD.pvd
270  pvdFilePath = os.path.join(self.directory, "FreeCAD.pvd")
271  if not os.path.exists(pvdFilePath):
272  self.pushStatus("\nNo result file was created.\n")
273  self.fail()
274  return
275  pvdFile = open(pvdFilePath)
276  # read all lines
277  pvdContent = pvdFile.readlines()
278  # skip header and footer line and evaluate all lines
279  # a line has the form like this:
280  # <DataSet timestep=" 5.000E-02" group="" part="0" file="FreeCAD_t0001.vtu"/>
281  # so .split("\"") gives as 2nd the time and as 7th the filename

283  # get time

291  # for eigen analyses the resulting values are by a factor 1000 to high
