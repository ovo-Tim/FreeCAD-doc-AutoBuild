54 App::DocumentObjectExecReturn *Compound::execute()
55 {
56  try {
57  // avoid duplicates without changing the order
58  // See also ViewProviderCompound::updateData
59  std::set<DocumentObject*> tempLinks;
60 
61  std::vector<TopoShape> shapes;
62  for (auto obj : Links.getValues()) {
63  if (!tempLinks.insert(obj).second) {
64  continue;
65  }
66  auto sh = Feature::getTopoShape(obj);
67  if (!sh.isNull()) {
68  shapes.push_back(sh);
69  }
70  }
71  this->Shape.setValue(TopoShape().makeElementCompound(shapes));
72  if (Links.getSize() > 0) {
73  App::DocumentObject* link = Links.getValues()[0];
74  copyMaterial(link);
75  }
76  return Part::Feature::execute();
77  }
78  catch (Standard_Failure& e) {
79  return new App::DocumentObjectExecReturn(e.GetMessageString());
80  }
81 }
