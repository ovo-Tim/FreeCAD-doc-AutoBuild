302 GeometryObjectPtr DrawViewPart::makeGeometryForShape(TopoDS_Shape& shape)
303 {
304 // Base::Console().Message("DVP::makeGeometryForShape() - %s\n", getNameInDocument());
305 
306  // if we use the passed reference directly, the centering doesn't work. Maybe the underlying OCC TShape
307  // isn't modified? using a copy works and the referenced shape (from getSourceShape in execute())
308  // isn't used for anything anyway.
309  bool copyGeometry = true;
310  bool copyMesh = false;
311  BRepBuilderAPI_Copy copier(shape, copyGeometry, copyMesh);
312  TopoDS_Shape localShape = copier.Shape();
313 
314  gp_Pnt gCentroid = ShapeUtils::findCentroid(localShape, getProjectionCS());
315  m_saveCentroid = DU::toVector3d(gCentroid);
316  m_saveShape = centerScaleRotate(this, localShape, m_saveCentroid);
317 
318  return buildGeometryObject(localShape, getProjectionCS());

322 TopoDS_Shape DrawViewPart::centerScaleRotate(const DrawViewPart *dvp, TopoDS_Shape& inOutShape,
323  Base::Vector3d centroid)
324 {
325 // Base::Console().Message("DVP::centerScaleRotate() - %s\n", dvp->getNameInDocument());
326  gp_Ax2 viewAxis = dvp->getProjectionCS();
327 
328  //center shape on origin
329  TopoDS_Shape centeredShape = ShapeUtils::moveShape(inOutShape, centroid * -1.0);
330 
331  inOutShape = ShapeUtils::scaleShape(centeredShape, dvp->getScale());
332  if (!DrawUtil::fpCompare(dvp->Rotation.getValue(), 0.0)) {
333  inOutShape = ShapeUtils::rotateShape(inOutShape, viewAxis,

341 TechDraw::GeometryObjectPtr DrawViewPart::buildGeometryObject(TopoDS_Shape& shape,
342  const gp_Ax2& viewAxis)
343 {
344 // Base::Console().Message("DVP::buildGeometryObject() - %s\n", getNameInDocument());
345  showProgressMessage(getNameInDocument(), "is finding hidden lines");
346 
347  TechDraw::GeometryObjectPtr go(
348  std::make_shared<TechDraw::GeometryObject>(getNameInDocument(), this));
349  go->setIsoCount(IsoCount.getValue());
350  go->isPerspective(Perspective.getValue());
351  go->setFocus(Focus.getValue());
352  go->usePolygonHLR(CoarseView.getValue());
353  go->setScrubCount(ScrubCount.getValue());
354 
355  if (CoarseView.getValue()) {
356  //the polygon approximation HLR process runs quickly, so doesn't need to be in a
357  //separate thread
358  go->projectShapeWithPolygonAlgo(shape, viewAxis);
359  }
360  else {
361  //projectShape (the HLR process) runs in a separate thread since it can take a long time
362  //note that &m_hlrWatcher in the third parameter is not strictly required, but using the
363  //4 parameter signature instead of the 3 parameter signature prevents clazy warning:
364  //https://github.com/KDE/clazy/blob/1.11/docs/checks/README-connect-3arg-lambda.md
365  connectHlrWatcher = QObject::connect(&m_hlrWatcher, &QFutureWatcherBase::finished,
366  &m_hlrWatcher, [this] { this->onHlrFinished(); });
367 
368  // We create a lambda closure to hold a copy of go, shape and viewAxis.
369  // This is important because those variables might be local to the calling
370  // function and might get destructed before the parallel processing finishes.
371  auto lambda = [go, shape, viewAxis]{go->projectShape(shape, viewAxis);};
372  m_hlrFuture = QtConcurrent::run(std::move(lambda));
373  m_hlrWatcher.setFuture(m_hlrFuture);
374  waitingForHlr(true);
375  }
376  return go;
377 }

380 void DrawViewPart::onHlrFinished()
381 {
382  // Base::Console().Message("DVP::onHlrFinished() - %s\n", getNameInDocument());
383 
384  //now that the new GeometryObject is fully populated, we can replace the old one
385  if (m_tempGeometryObject) {
386  geometryObject = m_tempGeometryObject;//replace with new
387  m_tempGeometryObject = nullptr; //superfluous?
388  }
389  if (!geometryObject) {
390  throw Base::RuntimeError("DrawViewPart has lost its geometry object");
391  }
392 
393  if (!hasGeometry()) {
394  Base::Console().Error("TechDraw did not retrieve any geometry for %s/%s\n",
395  getNameInDocument(), Label.getValue());
396  }
397 
398  //the last hlr related task is to make a bbox of the results
399  bbox = geometryObject->calcBoundingBox();
400 
401  waitingForHlr(false);
402  QObject::disconnect(connectHlrWatcher);
403  showProgressMessage(getNameInDocument(), "has finished finding hidden lines");
404 
405  postHlrTasks();//application level tasks that depend on HLR/GO being complete
406 
407  //start face finding in a separate thread. We don't find faces when using the polygon
408  //HLR method.
409  if (handleFaces() && !CoarseView.getValue()) {
410  try {
411  //note that &m_faceWatcher in the third parameter is not strictly required, but using the
412  //4 parameter signature instead of the 3 parameter signature prevents clazy warning:
413  //https://github.com/KDE/clazy/blob/1.11/docs/checks/README-connect-3arg-lambda.md
414  connectFaceWatcher =
415  QObject::connect(&m_faceWatcher, &QFutureWatcherBase::finished, &m_faceWatcher,
416  [this] { this->onFacesFinished(); });
417 
418  auto lambda = [this]{this->extractFaces();};
419  m_faceFuture = QtConcurrent::run(std::move(lambda));
420  m_faceWatcher.setFuture(m_faceFuture);
421  waitingForFaces(true);
422  }
423  catch (Standard_Failure& e) {
424  waitingForFaces(false);
425  Base::Console().Error("DVP::partExec - %s - extractFaces failed - %s **\n",
426  getNameInDocument(), e.GetMessageString());
427  throw Base::RuntimeError("DVP::onHlrFinished - error extracting faces");
428  }
429  }
430 }

514 // use the revised face finder algo
515 void DrawViewPart::findFacesNew(const std::vector<BaseGeomPtr> &goEdges)
516 {
517  std::vector<TopoDS_Edge> closedEdges;
518  std::vector<TopoDS_Edge> cleanEdges = DrawProjectSplit::scrubEdges(goEdges, closedEdges);
519 
520  if (cleanEdges.empty() && closedEdges.empty()) {
521  //how does this happen? something wrong somewhere
522  // Base::Console().Message("DVP::findFacesNew - no clean or closed wires\n"); //debug
523  return;
524  }
525 
526  //use EdgeWalker to make wires from edges
527  EdgeWalker eWalker;
528  std::vector<TopoDS_Wire> sortedWires;
529  try {
530  if (!cleanEdges.empty()) {
531  sortedWires = eWalker.execute(cleanEdges, true);//include outer wire
532  }
533  }
534  catch (Base::Exception& e) {
535  throw Base::RuntimeError(e.what());
536  }
537  geometryObject->clearFaceGeom();
538 
539  std::vector<TopoDS_Wire> closedWires;
540  for (auto& edge : closedEdges) {
541  BRepBuilderAPI_MakeWire mkWire(edge);
542  TopoDS_Wire wire = mkWire.Wire();
543  closedWires.push_back(wire);
544  }
545  if (!closedWires.empty()) {
546  sortedWires.insert(sortedWires.end(), closedWires.begin(), closedWires.end());
547  //inserting the closedWires that did not go through EdgeWalker into
548  //sortedWires ruins EdgeWalker's sort by size, so we have to do it again.
549  sortedWires = eWalker.sortWiresBySize(sortedWires);
550  }
551 
552  if (sortedWires.empty()) {
553  Base::Console().Warning(
554  "DVP::findFacesNew - %s - Can't make faces from projected edges\n",
555  getNameInDocument());
556  }
557  else {
558  constexpr double minWireArea = 0.000001;//arbitrary very small face size
559  auto itWire = sortedWires.begin();
560  for (; itWire != sortedWires.end(); itWire++) {
561  if (!BRep_Tool::IsClosed(*itWire)) {
562  continue;//can not make a face from open wire
563  }
564 
565  double area = ShapeAnalysis::ContourArea(*itWire);
566  if (area <= minWireArea) {
567  continue;//can not make a face from wire with no area
568  }
569 
570  TechDraw::FacePtr face(std::make_shared<TechDraw::Face>());
571  const TopoDS_Wire& wire = (*itWire);
572  face->wires.push_back(new TechDraw::Wire(wire));
573  if (geometryObject) {
574  geometryObject->addFaceGeom(face);
575  }
576  }
577  }
578 }

580 // original face finding method. This is retained only to produce the same face geometry in older
581 // documents.
582 void DrawViewPart::findFacesOld(const std::vector<BaseGeomPtr> &goEdges)
583 {
584  //make a copy of the input edges so the loose tolerances of face finding are
585  //not applied to the real edge geometry. See TopoDS_Shape::TShape().
586  std::vector<TopoDS_Edge> copyEdges;
587  bool copyGeometry = true;
588  bool copyMesh = false;
589  for (const auto& e : goEdges) {
590  BRepBuilderAPI_Copy copier(e->getOCCEdge(), copyGeometry, copyMesh);
591  copyEdges.push_back(TopoDS::Edge(copier.Shape()));
592  }
593  std::vector<TopoDS_Edge> nonZero;
594  for (auto& e : copyEdges) {//drop any zero edges (shouldn't be any by now!!!)
595  if (!DrawUtil::isZeroEdge(e)) {
596  nonZero.push_back(e);
597  }
598  }
599 
600  //HLR algo does not provide all edge intersections for edge endpoints.
601  //need to split long edges touched by Vertex of another edge
602  std::vector<splitPoint> splits;
603  std::vector<TopoDS_Edge>::iterator itOuter = nonZero.begin();
604  int iOuter = 0;
605  for (; itOuter != nonZero.end(); ++itOuter, iOuter++) {//*** itOuter != nonZero.end() - 1
606  TopoDS_Vertex v1 = TopExp::FirstVertex((*itOuter));
607  TopoDS_Vertex v2 = TopExp::LastVertex((*itOuter));
608  Bnd_Box sOuter;
609  BRepBndLib::AddOptimal(*itOuter, sOuter);
610  sOuter.SetGap(0.1);
611  if (sOuter.IsVoid()) {
612  continue;
613  }
614  if (DrawUtil::isZeroEdge(*itOuter)) {
615  continue; //skip zero length edges. shouldn't happen ;)
616  }
617  int iInner = 0;
618  std::vector<TopoDS_Edge>::iterator itInner = nonZero.begin();//***sb itOuter + 1;
619  for (; itInner != nonZero.end(); ++itInner, iInner++) {
620  if (iInner == iOuter) {
621  continue;
622  }
623  if (DrawUtil::isZeroEdge((*itInner))) {
624  continue;//skip zero length edges. shouldn't happen ;)
625  }
626 
627  Bnd_Box sInner;
628  BRepBndLib::AddOptimal(*itInner, sInner);
629  sInner.SetGap(0.1);
630  if (sInner.IsVoid()) {
631  continue;
632  }
633  if (sOuter.IsOut(sInner)) {//bboxes of edges don't intersect, don't bother
634  continue;
635  }
636 
637  double param = -1;
638  if (DrawProjectSplit::isOnEdge((*itInner), v1, param, false)) {
639  gp_Pnt pnt1 = BRep_Tool::Pnt(v1);
640  splitPoint s1;
641  s1.i = iInner;
642  s1.v = Base::Vector3d(pnt1.X(), pnt1.Y(), pnt1.Z());
643  s1.param = param;
644  splits.push_back(s1);
645  }
646  if (DrawProjectSplit::isOnEdge((*itInner), v2, param, false)) {
647  gp_Pnt pnt2 = BRep_Tool::Pnt(v2);
648  splitPoint s2;
649  s2.i = iInner;
650  s2.v = Base::Vector3d(pnt2.X(), pnt2.Y(), pnt2.Z());
651  s2.param = param;
652  splits.push_back(s2);
653  }
654  }//inner loop
655  } //outer loop
656 
657  std::vector<splitPoint> sorted = DrawProjectSplit::sortSplits(splits, true);
658  auto last = std::unique(sorted.begin(), sorted.end(),
659  DrawProjectSplit::splitEqual);//duplicates to back
660  sorted.erase(last, sorted.end()); //remove dupl splits
661  std::vector<TopoDS_Edge> newEdges = DrawProjectSplit::splitEdges(nonZero, sorted);
662 
663  if (newEdges.empty()) {
664  return;
665  }
666 
667  newEdges = DrawProjectSplit::removeDuplicateEdges(newEdges);
668 
669  geometryObject->clearFaceGeom();
670 
671  //find all the wires in the pile of faceEdges
672  std::vector<TopoDS_Wire> sortedWires;
673  EdgeWalker eWalker;
674  sortedWires = eWalker.execute(newEdges);
675  if (sortedWires.empty()) {
676  Base::Console().Warning(
677  "DVP::findFacesOld - %s -Can't make faces from projected edges\n",
678  getNameInDocument());
679  return;
680  }
681  else {
682  std::vector<TopoDS_Wire>::iterator itWire = sortedWires.begin();
683  for (; itWire != sortedWires.end(); itWire++) {
684  //version 1: 1 wire/face - no voids in face
685  TechDraw::FacePtr f(std::make_shared<TechDraw::Face>());
686  const TopoDS_Wire& wire = (*itWire);
687  TechDraw::Wire* w = new TechDraw::Wire(wire);
688  f->wires.push_back(w);
689  if (geometryObject) {
690  geometryObject->addFaceGeom(f);
691  }
692  }
693  }

1054 {

1064  //we should use the original X to prevent unwanted mirroring.
1065  //There might be a better choice of tolerance than EWTOLERANCE
1066  gNewX = gOldX;
