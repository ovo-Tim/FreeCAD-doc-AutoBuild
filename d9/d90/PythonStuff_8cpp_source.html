332 bp::list AdaptiveExecute(AdaptivePath::Adaptive2d& ada,
333  const boost::python::list& stock_paths,
334  const boost::python::list& in_paths,
335  boost::python::object progressCallbackFn)
336 {
337  bp::list out_list;
338 
339  // convert stock paths
340  AdaptivePath::DPaths stock_dpaths;
341  for (bp::ssize_t i = 0; i < bp::len(stock_paths); i++) {
342  bp::list in_path = bp::extract<boost::python::list>(stock_paths[i]);
343  AdaptivePath::DPath dpath;
344  for (bp::ssize_t j = 0; j < bp::len(in_path); j++) {
345  bp::list in_point = bp::extract<bp::list>(in_path[j]);
346  dpath.push_back(pair<double, double>(bp::extract<double>(in_point[0]),
347  bp::extract<double>(in_point[1])));
348  }
349  stock_dpaths.push_back(dpath);
350  }
351 
352  // convert inputs
353  AdaptivePath::DPaths dpaths;
354  for (bp::ssize_t i = 0; i < bp::len(in_paths); i++) {
355  bp::list in_path = bp::extract<boost::python::list>(in_paths[i]);
356  AdaptivePath::DPath dpath;
357  for (bp::ssize_t j = 0; j < bp::len(in_path); j++) {
358  bp::list in_point = bp::extract<bp::list>(in_path[j]);
359  dpath.push_back(pair<double, double>(bp::extract<double>(in_point[0]),
360  bp::extract<double>(in_point[1])));
361  }
362  dpaths.push_back(dpath);
363  }
364  // Execute with callback
365  std::list<AdaptivePath::AdaptiveOutput> result =
366  ada.Execute(stock_dpaths, dpaths, [progressCallbackFn](AdaptivePath::TPaths tp) -> bool {
367  bp::list out_paths;
368  for (const auto& in_pair : tp) {
369  bp::list path;
370  for (const auto& in_pt : in_pair.second) {
371  path.append(bp::make_tuple(in_pt.first, in_pt.second));
372  }
373  out_paths.append(bp::make_tuple(in_pair.first, path));
374  }
375  return bp::extract<bool>(progressCallbackFn(out_paths));
376  });
377  // convert outputs back
378  BOOST_FOREACH (const auto& res, result) {
379  out_list.append(res);
380  }
381  return out_list;
382 }
