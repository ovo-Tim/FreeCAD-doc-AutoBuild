54 def cmdCreateJoinFeature(name, mode):
55  """cmdCreateJoinFeature(name, mode): generalized implementation of GUI commands."""
56  sel = FreeCADGui.Selection.getSelectionEx()
57 
58  FreeCAD.ActiveDocument.openTransaction("Create "+mode)
59  FreeCADGui.addModule("BOPTools.JoinFeatures")
60  FreeCADGui.doCommand("j = BOPTools.JoinFeatures.make{mode}(name='{name}')".format(mode=mode, name=name))
61  if mode == "Embed" or mode == "Cutout":
62  FreeCADGui.doCommand("j.Base = App.ActiveDocument."+sel[0].Object.Name)
63  FreeCADGui.doCommand("j.Tool = App.ActiveDocument."+sel[1].Object.Name)
64  elif mode == "Connect":
65  FreeCADGui.doCommand("j.Objects = {sel}".format(
66  sel= "[" + ", ".join(["App.ActiveDocument."+so.Object.Name for so in sel]) + "]"
67  ))
68  else:
69  raise ValueError("cmdCreateJoinFeature: Unexpected mode {mode}".format(mode=repr(mode)))
70 
71  try:
72  FreeCADGui.doCommand("j.Proxy.execute(j)")
73  FreeCADGui.doCommand("j.purgeTouched()")
74  except Exception as err:
75  mb = QtGui.QMessageBox()
76  mb.setIcon(mb.Icon.Warning)
77  error_text1 = translate("Part_JoinFeatures", "Computing the result failed with an error:")
78  error_text2 = translate("Part_JoinFeatures", "Click 'Continue' to create the feature anyway, or 'Abort' to cancel.")
79  mb.setText(error_text1 + "\n\n" + str(err) + "\n\n" + error_text2)
80  mb.setWindowTitle(translate("Part_JoinFeatures","Bad selection", None))
81  btnAbort = mb.addButton(QtGui.QMessageBox.StandardButton.Abort)
82  btnOK = mb.addButton(translate("Part_JoinFeatures","Continue",None),
83  QtGui.QMessageBox.ButtonRole.ActionRole)
84  mb.setDefaultButton(btnOK)
85  mb.exec_()
86 
87  if mb.clickedButton() is btnAbort:
88  FreeCAD.ActiveDocument.abortTransaction()
89  return

92  " obj.ViewObject.hide()")

113 class FeatureConnect:
114  """The PartJoinFeature object."""
115 
116  def __init__(self,obj):
117  obj.addProperty("App::PropertyLinkList","Objects","Connect","Object to be connected.")
118  obj.addProperty("App::PropertyBool","Refine","Connect",
119  "True = refine resulting shape. False = output as is.")
120  obj.Refine = getParamRefine()
121  obj.addProperty("App::PropertyLength","Tolerance","Connect",
122  "Tolerance when intersecting (fuzzy value). "
123  "In addition to tolerances of the shapes.")
124 
125  obj.Proxy = self
126  self.Type = "FeatureConnect"
127 
128  def execute(self,selfobj):
129  rst = JoinAPI.connect([obj.Shape for obj in selfobj.Objects], selfobj.Tolerance)
130  if selfobj.Refine:
131  rst = rst.removeSplitter()
132  selfobj.Shape = rst
