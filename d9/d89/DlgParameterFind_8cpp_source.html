119 bool DlgParameterFind::matches(QTreeWidgetItem* item, const Options& opt) const
120 {
121  // check the group name
122  if (opt.group) {
123  // whole word matches
124  if (opt.match) {
125  if (item->text(0).compare(opt.text, Qt::CaseInsensitive) == 0) {
126  return true;
127  }
128  }
129  else {
130  if (item->text(0).indexOf(opt.text, 0, Qt::CaseInsensitive) >= 0) {
131  return true;
132  }
133  }
134  }
135 
136  if (opt.name || opt.value) {
137  auto group = static_cast<ParameterGroupItem*>(item);
138  Base::Reference<ParameterGrp> hGrp = group->_hcGrp;
139 
140  auto boolMap = hGrp->GetBoolMap();
141  auto intMap = hGrp->GetIntMap();
142  auto uintMap = hGrp->GetUnsignedMap();
143  auto floatMap = hGrp->GetFloatMap();
144  auto asciiMap = hGrp->GetASCIIMap();
145 
146  // check the name of an entry in the group
147  if (opt.name) {
148  if (opt.match) {
149  for (const auto& it : boolMap) {
150  QString text = QString::fromUtf8(it.first.c_str());
151  if (text.compare(opt.text, Qt::CaseInsensitive) == 0) {
152  return true;
153  }
154  }
155  for (const auto& it : intMap) {
156  QString text = QString::fromUtf8(it.first.c_str());
157  if (text.compare(opt.text, Qt::CaseInsensitive) == 0) {
158  return true;
159  }
160  }
161  for (const auto& it : uintMap) {
162  QString text = QString::fromUtf8(it.first.c_str());
163  if (text.compare(opt.text, Qt::CaseInsensitive) == 0) {
164  return true;
165  }
166  }
167  for (const auto& it : floatMap) {
168  QString text = QString::fromUtf8(it.first.c_str());
169  if (text.compare(opt.text, Qt::CaseInsensitive) == 0) {
170  return true;
171  }
172  }
173  for (const auto& it : asciiMap) {
174  QString text = QString::fromUtf8(it.first.c_str());
175  if (text.compare(opt.text, Qt::CaseInsensitive) == 0) {
176  return true;
177  }
178  }
179  }
180  else {
181  for (const auto& it : boolMap) {
182  QString text = QString::fromUtf8(it.first.c_str());
183  if (text.indexOf(opt.text, 0, Qt::CaseInsensitive) >= 0) {
184  return true;
185  }
186  }
187  for (const auto& it : intMap) {
188  QString text = QString::fromUtf8(it.first.c_str());
189  if (text.indexOf(opt.text, 0, Qt::CaseInsensitive) >= 0) {
190  return true;
191  }
192  }
193  for (const auto& it : uintMap) {
194  QString text = QString::fromUtf8(it.first.c_str());
195  if (text.indexOf(opt.text, 0, Qt::CaseInsensitive) >= 0) {
196  return true;
197  }
198  }
199  for (const auto& it : floatMap) {
200  QString text = QString::fromUtf8(it.first.c_str());
201  if (text.indexOf(opt.text, 0, Qt::CaseInsensitive) >= 0) {
202  return true;
203  }
204  }
205  for (const auto& it : asciiMap) {
206  QString text = QString::fromUtf8(it.first.c_str());
207  if (text.indexOf(opt.text, 0, Qt::CaseInsensitive) >= 0) {
208  return true;
209  }
210  }
211  }
212  }
213 
214  // check the value of an entry in the group
215  if (opt.value) {
216  if (opt.match) {
217  for (const auto& it : asciiMap) {
218  QString text = QString::fromUtf8(it.second.c_str());
219  if (text.compare(opt.text, Qt::CaseInsensitive) == 0) {
220  return true;
221  }
222  }
223  }
224  else {
225  for (const auto& it : asciiMap) {
226  QString text = QString::fromUtf8(it.second.c_str());
227  if (text.indexOf(opt.text, 0, Qt::CaseInsensitive) >= 0) {
228  return true;
229  }
230  }
231  }
232  }
233  }
234 
235  return false;

238 QTreeWidgetItem* DlgParameterFind::findItem(QTreeWidgetItem* root, const Options& opt) const
239 {
240  if (!root) {
241  return nullptr;
242  }
243 
244  if (matches(root, opt)) {
245  // if the root matches then only return if it's not the current element
246  // as otherwise it would never move forward
247  if (root->treeWidget()->currentItem() != root) {
248  return root;
249  }
250  }
251 
252  for (int i = 0; i < root->childCount(); i++) {
253  QTreeWidgetItem* item = root->child(i);
254  if (matches(item, opt)) {
255  return item;
256  }
257  item = findItem(item, opt);
258  if (item) {
259  return item;
260  }
261  }
262 
263  return nullptr;
264 }

266 void DlgParameterFind::accept()
267 {
268  auto groupTree = dialog->findChild<ParameterGroup*>();
269  if (groupTree) {
270  Options opt;
271  opt.text = ui->lineEdit->text();
272  opt.group = ui->checkGroups->isChecked();
273  opt.name = ui->checkNames->isChecked();
274  opt.value = ui->checkValues->isChecked();
275  opt.match = ui->checkMatch->isChecked();
276 
277  QTreeWidgetItem* current = groupTree->currentItem();
278  QTreeWidgetItem* next = findItem(current, opt);
279  while (!next && current) {
280  // go to the parent item and try again for each sibling after the current item
281  QTreeWidgetItem* parent = current->parent();
282  if (!parent) {
283  // switch from one top-level group to the next
284  QTreeWidgetItem* root = groupTree->invisibleRootItem();
285  if (root->indexOfChild(current) >= 0) {
286  parent = root;
287  }
288  }
289  if (parent) {
290  int index = parent->indexOfChild(current);
291  for (int i = index + 1; i < parent->childCount(); i++) {
292  next = findItem(parent->child(i), opt);
293  if (next) {
294  break;
295  }
296  }
297  }
298 
299  if (!next) {
300  current = parent;
301  }
302  }
303 
304  // if search was successful then make it the current item
305  if (next) {
306  groupTree->setCurrentItem(next);
307  }
308  else {
309  QMessageBox::warning(this,
310  tr("Not found"),
311  tr("Can't find the text: %1").arg(opt.text));
312  }
313  }
314 }
