53  // measurements
54  bsplineGeoIds.clear();
55  arcGeoIds.clear();

57  // end information layer
58  Points.clear();
59  Coords.clear();
60  Index.clear();
61 
62  coinMapping.clear();
63 
64  pointCounter.clear();
65 
66  for (auto l = 0; l < geometryLayerParameters.getCoinLayerCount(); l++) {
67  Points.emplace_back();
68  Coords.emplace_back();
69  Index.emplace_back();
70 
71  coinMapping.CurvIdToGeoId.emplace_back();

73  Coords[l].emplace_back();
74  Index[l].emplace_back();

82  pointCounter.resize(geometryLayerParameters.getCoinLayerCount(), 0);
83 
84  // RootPoint
85  // TODO: RootPoint is here added in layer0. However, this layer may be hidden. The point should,
86  // when that functionality is provided, be added to the first visible layer, or may even a new
87  // empty layer.

93 
94  coinMapping.GeoElementId2SetId.emplace(std::piecewise_construct,
95  std::forward_as_tuple(Sketcher::GeoElementId::RtPnt),
96  std::forward_as_tuple(pointCounter[0]++, 0));

98  auto setTracking = [this](int geoId,
99  int coinLayer,
100  EditModeGeometryCoinConverter::PointsMode pointmode,
101  int numberCurves,
102  int sublayer) {
103  int numberPoints = 0;
104 
105  if (pointmode == PointsMode::InsertSingle) {
106  numberPoints = 1;
107 
108  coinMapping.GeoElementId2SetId.emplace(
109  std::piecewise_construct,
110  std::forward_as_tuple(geoId, Sketcher::PointPos::start),
111  std::forward_as_tuple(pointCounter[coinLayer]++, coinLayer));
112  }
113  else if (pointmode == PointsMode::InsertStartEnd) {
114  numberPoints = 2;
115 
116  coinMapping.GeoElementId2SetId.emplace(
117  std::piecewise_construct,
118  std::forward_as_tuple(geoId, Sketcher::PointPos::start),
119  std::forward_as_tuple(pointCounter[coinLayer]++, coinLayer));
120 
121  coinMapping.GeoElementId2SetId.emplace(
122  std::piecewise_construct,
123  std::forward_as_tuple(geoId, Sketcher::PointPos::end),
124  std::forward_as_tuple(pointCounter[coinLayer]++, coinLayer));
125  }
126  else if (pointmode == PointsMode::InsertMidOnly) {
127  numberPoints = 1;
128 
129  coinMapping.GeoElementId2SetId.emplace(
130  std::piecewise_construct,
131  std::forward_as_tuple(geoId, Sketcher::PointPos::mid),
132  std::forward_as_tuple(pointCounter[coinLayer]++, coinLayer));
133  }
134  else if (pointmode == PointsMode::InsertStartEndMid) {
135  numberPoints = 3;
136 
137  coinMapping.GeoElementId2SetId.emplace(
138  std::piecewise_construct,
139  std::forward_as_tuple(geoId, Sketcher::PointPos::start),
140  std::forward_as_tuple(pointCounter[coinLayer]++, coinLayer));
141 
142  coinMapping.GeoElementId2SetId.emplace(
143  std::piecewise_construct,
144  std::forward_as_tuple(geoId, Sketcher::PointPos::end),
145  std::forward_as_tuple(pointCounter[coinLayer]++, coinLayer));
146 
147  coinMapping.GeoElementId2SetId.emplace(
148  std::piecewise_construct,
149  std::forward_as_tuple(geoId, Sketcher::PointPos::mid),
150  std::forward_as_tuple(pointCounter[coinLayer]++, coinLayer));
151  }
152 
153  for (int i = 0; i < numberPoints; i++) {
154  coinMapping.PointIdToGeoId[coinLayer].push_back(geoId);
155  Sketcher::PointPos pos;
156  if (i == 0) {
157  if (pointmode == PointsMode::InsertMidOnly) {
158  pos = Sketcher::PointPos::mid;
159  }
160  else {
161  pos = Sketcher::PointPos::start;
162  }
163  }
164  else if (i == 1) {
165  pos = Sketcher::PointPos::end;
166  }
167  else {
168  pos = Sketcher::PointPos::mid;
169  }
170 
171  coinMapping.PointIdToPosId[coinLayer].push_back(pos);
172  coinMapping.PointIdToVertexId[coinLayer].push_back(vertexCounter++);
173  }
174 
175  if (numberCurves > 0) { // insert the first segment of the curve into the map
176  coinMapping.GeoElementId2SetId.emplace(
177  std::piecewise_construct,
178  std::forward_as_tuple(geoId, Sketcher::PointPos::none),
179  std::forward_as_tuple(
180  static_cast<int>(coinMapping.CurvIdToGeoId[coinLayer][sublayer].size()),
181  coinLayer,
182  sublayer));
183  }
184 
185  for (int i = 0; i < numberCurves; i++) {
186  coinMapping.CurvIdToGeoId[coinLayer][sublayer].push_back(geoId);
187  }
188  };

190  for (size_t i = 0; i < geolistfacade.geomlist.size() - 2; i++) {
191 
192  const auto GeoId = geolistfacade.getGeoIdFromGeomListIndex(i);
193  const auto geom = geolistfacade.getGeometryFacadeFromGeoId(GeoId);

197  int subLayerId = geometryLayerParameters.getSubLayerIndex(GeoId, geom);
198

200

202  convert<Part::GeomPoint,
203  EditModeGeometryCoinConverter::PointsMode::InsertSingle,
204  EditModeGeometryCoinConverter::CurveMode::NoCurve,
205  EditModeGeometryCoinConverter::AnalyseMode::BoundingBoxMagnitude>(geom,
206  GeoId,
207  subLayerId);
208  setTracking(GeoId,
209  coinLayer,
210  EditModeGeometryCoinConverter::PointsMode::InsertSingle,
211  0,
212  subLayerId);
213  }
214  else if (type == Part::GeomLineSegment::getClassTypeId()) { // add a line
215  convert<Part::GeomLineSegment,
216  EditModeGeometryCoinConverter::PointsMode::InsertStartEnd,
217  EditModeGeometryCoinConverter::CurveMode::StartEndPointsOnly,
218  EditModeGeometryCoinConverter::AnalyseMode::BoundingBoxMagnitude>(geom,
219  GeoId,
220  subLayerId);
221  setTracking(GeoId,
222  coinLayer,
223  EditModeGeometryCoinConverter::PointsMode::InsertStartEnd,
224  1,
225  subLayerId);
226  }
227  else if (type.isDerivedFrom(
228  Part::GeomConic::getClassTypeId())) { // add a closed curve conic
229  convert<Part::GeomConic,
230  EditModeGeometryCoinConverter::PointsMode::InsertMidOnly,
231  EditModeGeometryCoinConverter::CurveMode::ClosedCurve,
232  EditModeGeometryCoinConverter::AnalyseMode::BoundingBoxMagnitude>(geom,
233  GeoId,
234  subLayerId);
235  setTracking(GeoId,
236  coinLayer,
237  EditModeGeometryCoinConverter::PointsMode::InsertMidOnly,
238  1,
239  subLayerId);
240  }
241  else if (type.isDerivedFrom(
242  Part::GeomArcOfConic::getClassTypeId())) { // add an arc of conic
243  convert<Part::GeomArcOfConic,
244  EditModeGeometryCoinConverter::PointsMode::InsertStartEndMid,
245  EditModeGeometryCoinConverter::CurveMode::OpenCurve,
246  EditModeGeometryCoinConverter::AnalyseMode::BoundingBoxMagnitude>(geom,
247  GeoId,
248  subLayerId);
249  setTracking(GeoId,
250  coinLayer,
251  EditModeGeometryCoinConverter::PointsMode::InsertStartEndMid,
252  1,
253  subLayerId);
254  arcGeoIds.push_back(GeoId);
255  }
256  else if (type == Part::GeomBSplineCurve::getClassTypeId()) { // add a bspline (a bounded
257  // curve that is not a conic)
258  convert<Part::GeomBSplineCurve,
259  EditModeGeometryCoinConverter::PointsMode::InsertStartEnd,
260  EditModeGeometryCoinConverter::CurveMode::OpenCurve,

268  bsplineGeoIds.push_back(GeoId);

271 
272  // Coin Nodes Editing
273  int vOrFactor = ViewProviderSketchCoinAttorney::getViewOrientationFactor(viewProvider);
274  double linez = vOrFactor * drawingParameters.zLowLines; // NOLINT
275  double pointz = vOrFactor * drawingParameters.zLowPoints;
276 
277  for (auto l = 0; l < geometryLayerParameters.getCoinLayerCount(); l++) {
278  geometryLayerNodes.PointsCoordinate[l]->point.setNum(Points[l].size());
279  geometryLayerNodes.PointsMaterials[l]->diffuseColor.setNum(Points[l].size());
280  SbVec3f* pverts = geometryLayerNodes.PointsCoordinate[l]->point.startEditing();
281 
282  int i = 0; // setting up the point set
283  for (auto& point : Points[l]) {
284  pverts[i++].setValue(point.x, point.y, pointz);
285  }
286  geometryLayerNodes.PointsCoordinate[l]->point.finishEditing();
287 
288  for (auto t = 0; t < geometryLayerParameters.getSubLayerCount(); t++) {
289  geometryLayerNodes.CurvesCoordinate[l][t]->point.setNum(Coords[l][t].size());
290  geometryLayerNodes.CurveSet[l][t]->numVertices.setNum(Index[l][t].size());
291  geometryLayerNodes.CurvesMaterials[l][t]->diffuseColor.setNum(Index[l][t].size());
292 
293  SbVec3f* verts = geometryLayerNodes.CurvesCoordinate[l][t]->point.startEditing();
294  int32_t* index = geometryLayerNodes.CurveSet[l][t]->numVertices.startEditing();
295 
296  i = 0; // setting up the line set
297  for (auto& coord : Coords[l][t]) {
298  verts[i++].setValue(coord.x, coord.y, linez); // NOLINT
299  }
300 
301  i = 0; // setting up the indexes of the line set
302  for (auto it : Index[l][t]) {
303  index[i++] = it;
304  }
305 
306  geometryLayerNodes.CurvesCoordinate[l][t]->point.finishEditing();
307  geometryLayerNodes.CurveSet[l][t]->numVertices.finishEditing();
308  }
309  }

312 template<typename GeoType,
313  EditModeGeometryCoinConverter::PointsMode pointmode,
314  EditModeGeometryCoinConverter::CurveMode curvemode,
315  EditModeGeometryCoinConverter::AnalyseMode analysemode>
316 void EditModeGeometryCoinConverter::convert(const Sketcher::GeometryFacade* geometryfacade,
317  [[maybe_unused]] int geoid,
318  [[maybe_unused]] int subLayer)
319 {
320  auto geo = static_cast<const GeoType*>(geometryfacade->getGeometry());
321  auto layerId = getSafeGeomLayerId(geometryfacade);
322 
323  auto coinLayer = geometryLayerParameters.getSafeCoinLayer(layerId);
324 
325  auto addPoint = [&dMg = boundingBoxMaxMagnitude](auto& pushvector, Base::Vector3d point) {
326  if constexpr (analysemode == AnalyseMode::BoundingBoxMagnitude
327  || analysemode == AnalyseMode::BoundingBoxMagnitudeAndBSplineCurvature) {
328  dMg = dMg > std::abs(point.x) ? dMg : std::abs(point.x);
329  dMg = dMg > std::abs(point.y) ? dMg : std::abs(point.y);
330  pushvector.push_back(point);
331  }
332  };
333 
334  // Points
335  if constexpr (pointmode == PointsMode::InsertSingle) {
336  addPoint(Points[coinLayer], geo->getPoint());
337  }
338  else if constexpr (pointmode == PointsMode::InsertStartEnd) {
339  addPoint(Points[coinLayer], geo->getStartPoint());
340  addPoint(Points[coinLayer], geo->getEndPoint());
341  }
342  else if constexpr (pointmode == PointsMode::InsertStartEndMid) {
343  // All in this group are Trimmed Curves (see Geometry.h)
344  addPoint(Points[coinLayer], geo->getStartPoint(/*emulateCCW=*/true));
345  addPoint(Points[coinLayer], geo->getEndPoint(/*emulateCCW=*/true));
346  addPoint(Points[coinLayer], geo->getCenter());
347  }
348  else if constexpr (pointmode == PointsMode::InsertMidOnly) {
349  addPoint(Points[coinLayer], geo->getCenter());
350  }
351 
352  // Curves
353  if constexpr (curvemode == CurveMode::StartEndPointsOnly) {
354  addPoint(Coords[coinLayer][subLayer], geo->getStartPoint());
355  addPoint(Coords[coinLayer][subLayer], geo->getEndPoint());
356  Index[coinLayer][subLayer].push_back(2);
357  }
358  else if constexpr (curvemode == CurveMode::ClosedCurve) {
359  int numSegments = drawingParameters.curvedEdgeCountSegments;
360  if constexpr (std::is_same<GeoType, Part::GeomBSplineCurve>::value) {
361  numSegments *= geo->countKnots();
362  }
363 
364  double segment = (geo->getLastParameter() - geo->getFirstParameter()) / numSegments;
365 
366  for (int i = 0; i < numSegments; i++) {
367  Base::Vector3d pnt = geo->value(i * segment);
368  addPoint(Coords[coinLayer][subLayer], pnt);
369  }
370 
371  Base::Vector3d pnt = geo->value(0);
372  addPoint(Coords[coinLayer][subLayer], pnt);
373 
374  Index[coinLayer][subLayer].push_back(numSegments + 1);
375  }
376  else if constexpr (curvemode == CurveMode::OpenCurve) {
377  int numSegments = drawingParameters.curvedEdgeCountSegments;
378  if constexpr (std::is_same<GeoType, Part::GeomBSplineCurve>::value) {
379  numSegments *= (geo->countKnots() - 1); // one less segments than knots
380  }
381 
382  double segment = (geo->getLastParameter() - geo->getFirstParameter()) / numSegments;
383 
384  for (int i = 0; i < numSegments; i++) {
385  Base::Vector3d pnt = geo->value(geo->getFirstParameter() + i * segment);
386  addPoint(Coords[coinLayer][subLayer], pnt);
387  }
388 
389  Base::Vector3d pnt = geo->value(geo->getLastParameter());
390  addPoint(Coords[coinLayer][subLayer], pnt);
391 
392  Index[coinLayer][subLayer].push_back(numSegments + 1);
393 
394  if constexpr (analysemode == AnalyseMode::BoundingBoxMagnitudeAndBSplineCurvature) {
395  //***************************************************************************************************************
396  // global information gathering for geometry information layer
397 
398  std::vector<Base::Vector3d> poles = geo->getPoles();
399 
400  Base::Vector3d midp = Base::Vector3d(0, 0, 0);
401 
402  for (std::vector<Base::Vector3d>::iterator it = poles.begin(); it != poles.end();
403  ++it) {
404  midp += (*it);
405  }
406 
407  midp /= poles.size();
408 
409  double firstparam = geo->getFirstParameter();
410  double lastparam = geo->getLastParameter();
411 
412  const int ndiv = poles.size() > 4 ? poles.size() * 16 : 64;
413  double step = (lastparam - firstparam) / (ndiv - 1);
414 
415  std::vector<double> paramlist(ndiv);
416  std::vector<Base::Vector3d> pointatcurvelist(ndiv);
417  std::vector<double> curvaturelist(ndiv);
418  std::vector<Base::Vector3d> normallist(ndiv);
419 
420  double maxcurv = 0;
421  double maxdisttocenterofmass = 0;
422 
423  for (int i = 0; i < ndiv; i++) {
424  paramlist[i] = firstparam + i * step;
425  pointatcurvelist[i] = geo->pointAtParameter(paramlist[i]);
426 
427  try {
428  curvaturelist[i] = geo->curvatureAt(paramlist[i]);
429  }
430  catch (Base::CADKernelError& e) {
431  // it is "just" a visualisation matter OCC could not calculate the curvature
432  // terminating here would mean that the other shapes would not be drawn.
433  // Solution: Report the issue and set dummy curvature to 0
434  e.ReportException();
435  Base::Console().DeveloperError(
436  "EditModeGeometryCoinConverter",
437  "Curvature graph for B-spline with GeoId=%d could not be calculated.\n",
438  geoid); // TODO: Fix identification of curve.
439  curvaturelist[i] = 0;
440  }
441 
442  if (curvaturelist[i] > maxcurv) {
443  maxcurv = curvaturelist[i];
444  }
445 
446  double tempf = (pointatcurvelist[i] - midp).Length();
447 
448  if (tempf > maxdisttocenterofmass) {
449  maxdisttocenterofmass = tempf;
450  }
451  }
452 
453  double temprepscale = 0;
454  if (maxcurv > 0) {
455  temprepscale = (0.5 * maxdisttocenterofmass)
456  / maxcurv; // just a factor to make a comb reasonably visible
457  }
458 
459  if (temprepscale > combrepscale) {
460  combrepscale = temprepscale;
461  }
462  }
463  }
464 }
