138 void ImportOCAF::loadShapes(const TDF_Label& label,
139  const TopLoc_Location& loc,
140  const std::string& defaultname,
141  const std::string& assembly,
142  bool isRef,
143  std::vector<App::DocumentObject*>& lValue)
144 {
145  int hash = 0;
146 #ifdef HAVE_TBB
147  using namespace tbb;
148  task_group g;
149 #endif
150  TopoDS_Shape aShape;
151 
152  std::vector<App::DocumentObject*> localValue;
153 
154  if (aShapeTool->GetShape(label, aShape)) {
155  hash = Part::ShapeMapHasher {}(aShape);
156  }
157 
158  Handle(TDataStd_Name) name;
159  std::string part_name = defaultname;
160  if (label.FindAttribute(TDataStd_Name::GetID(), name)) {
161  TCollection_ExtendedString extstr = name->Get();
162  char* str = new char[extstr.LengthOfCString() + 1];
163  extstr.ToUTF8CString(str);
164  part_name = str;
165  delete[] str;
166  if (part_name.empty()) {
167  part_name = defaultname;
168  }
169  else {
170  bool ws = true;
171  for (char it : part_name) {
172  if (it != ' ') {
173  ws = false;
174  break;
175  }
176  }
177  if (ws) {
178  part_name = defaultname;
179  }
180  }
181  }
182 
183  TopLoc_Location part_loc = loc;
184  Handle(XCAFDoc_Location) hLoc;
185  if (label.FindAttribute(XCAFDoc_Location::GetID(), hLoc)) {
186  if (isRef) {
187  part_loc = part_loc * hLoc->Get();
188  }
189  else {
190  part_loc = hLoc->Get();
191  }
192  }
193 
194 #ifdef FC_DEBUG
195  Base::Console().Log("H:%d, N:%s, T:%d, A:%d, S:%d, C:%d, SS:%d, F:%d, R:%d, C:%d, SS:%d\n",
196  hash,
197  part_name.c_str(),
198  aShapeTool->IsTopLevel(label),
199  aShapeTool->IsAssembly(label),
200  aShapeTool->IsShape(label),
201  aShapeTool->IsCompound(label),
202  aShapeTool->IsSimpleShape(label),
203  aShapeTool->IsFree(label),
204  aShapeTool->IsReference(label),
205  aShapeTool->IsComponent(label),
206  aShapeTool->IsSubShape(label));
207 #endif
208 
209 #if defined(OCAF_KEEP_PLACEMENT)
210  std::string asm_name = part_name;
211  (void)assembly;
212 #else
213  std::string asm_name = assembly;
214  if (aShapeTool->IsAssembly(label)) {
215  asm_name = part_name;
216  }
217 #endif
218 
219  TDF_Label ref;
220  if (aShapeTool->IsReference(label) && aShapeTool->GetReferredShape(label, ref)) {
221  loadShapes(ref, part_loc, part_name, asm_name, true, lValue);
222  }
223 
224  if (isRef || myRefShapes.find(hash) == myRefShapes.end()) {
225  TopoDS_Shape aShape;
226  if (isRef && aShapeTool->GetShape(label, aShape)) {
227  myRefShapes.insert(Part::ShapeMapHasher {}(aShape));
228  }
229 
230  if (aShapeTool->IsSimpleShape(label) && (isRef || aShapeTool->IsFree(label))) {
231  if (!asm_name.empty()) {
232  part_name = asm_name;
233  }
234 
235  // TODO: The merge parameter (last one from createShape) should become an
236  // Importer/Exporter option within the FreeCAD preference menu Currently it is merging
237  // STEP Compound Shape into a single Shape Part::Feature which is an OpenCascade
238  // computed Compound
239  if (isRef) {
240  createShape(label, loc, part_name, lValue, this->merge);
241  }
242  else {
243  createShape(label, part_loc, part_name, localValue, this->merge);
244  }
245  }
246  else {
247  if (aShapeTool->IsSimpleShape(label)) {
248  // We are not creating a list of Part::Feature in that case but just
249  // a single Part::Feature which has as a Shape a Compound of the Subshapes contained
250  // within the global shape
251  // This is standard behavior of many STEP reader and avoid to register a crazy
252  // amount of Shape within the Tree as STEP file do mostly contain large assemblies
253  return;
254  }
255 
256  // This is probably an Assembly let's try to create a Compound with the name
257  for (TDF_ChildIterator it(label); it.More(); it.Next()) {
258  if (isRef) {
259  loadShapes(it.Value(), part_loc, part_name, asm_name, false, localValue);
260  }
261  else {
262  loadShapes(it.Value(), part_loc, part_name, asm_name, isRef, localValue);
263  }
264  }
265 
266  if (!localValue.empty()) {
267  if (aShapeTool->IsAssembly(label)) {
268  App::Part* pcPart = nullptr;
269  pcPart = static_cast<App::Part*>(doc->addObject("App::Part", asm_name.c_str()));
270  pcPart->Label.setValue(asm_name);
271  pcPart->addObjects(localValue);
272 
273  // STEP reader is now a hierarchical reader. Node and leaf must have
274  // there local placement updated and relative to the STEP file content
275  // standard FreeCAD placement was absolute we are now moving to relative
276 
277  tryPlacementFromLoc(pcPart, part_loc);
278  lValue.push_back(pcPart);
279  }
280  }
281  }
282  }
283 }

285 void ImportOCAF::createShape(const TDF_Label& label,
286  const TopLoc_Location& loc,
287  const std::string& name,
288  std::vector<App::DocumentObject*>& lValue,
289  bool mergeShape)
290 {
291  const TopoDS_Shape& aShape = aShapeTool->GetShape(label);
292 #ifdef HAVE_TBB
293  using namespace tbb;
294  task_group g;
295 #endif
296 
297  if (!aShape.IsNull() && aShape.ShapeType() == TopAbs_COMPOUND) {
298  TopExp_Explorer xp;
299  int ctSolids = 0, ctShells = 0, ctVertices = 0, ctEdges = 0;
300  std::vector<App::DocumentObject*> localValue;
301  App::Part* pcPart = nullptr;
302 
303  if (mergeShape) {
304 
305  // We should do that only if there is more than a single shape inside
306  // Computing Compounds takes time
307  // We must keep track of the Color. If there is more than 1 Color into
308  // a STEP Compound then the Merge can't be done and we cancel the operation
309 
310  BRep_Builder builder;
311  TopoDS_Compound comp;
312  builder.MakeCompound(comp);
313 
314  for (xp.Init(aShape, TopAbs_SOLID); xp.More(); xp.Next(), ctSolids++) {
315  const TopoDS_Shape& sh = xp.Current();
316  if (!sh.IsNull()) {
317  builder.Add(comp, sh);
318  }
319  }
320 
321  for (xp.Init(aShape, TopAbs_SHELL, TopAbs_SOLID); xp.More(); xp.Next(), ctShells++) {
322  const TopoDS_Shape& sh = xp.Current();
323  if (!sh.IsNull()) {
324  builder.Add(comp, sh);
325  }
326  }
327 
328  for (xp.Init(aShape, TopAbs_EDGE); xp.More(); xp.Next(), ctEdges++) {
329  const TopoDS_Shape& sh = xp.Current();
330  if (!sh.IsNull()) {
331  builder.Add(comp, sh);
332  }
333  }
334 
335  for (xp.Init(aShape, TopAbs_VERTEX); xp.More(); xp.Next(), ctVertices++) {
336  const TopoDS_Shape& sh = xp.Current();
337  if (!sh.IsNull()) {
338  builder.Add(comp, sh);
339  }
340  }
341 
342  // Ok we got a Compound which is computed
343  // Just need to add it to a Part::Feature and push it to lValue
344  if (!comp.IsNull() && (ctSolids || ctShells || ctEdges || ctVertices)) {
345  Part::Feature* part = static_cast<Part::Feature*>(doc->addObject("Part::Feature"));
346  // Let's allocate the relative placement of the Compound from the STEP file
347  tryPlacementFromLoc(part, loc);
348  if (!loc.IsIdentity()) {
349  part->Shape.setValue(comp.Moved(loc));
350  }
351  else {
352  part->Shape.setValue(comp);
353  }
354 
355  part->Label.setValue(name);
356  lValue.push_back(part);
357 
358  loadColors(part, aShape);
359  }
360  }
361  else {
362  for (xp.Init(aShape, TopAbs_SOLID); xp.More(); xp.Next(), ctSolids++) {
363  createShape(xp.Current(), loc, name, localValue);
364  }
365  for (xp.Init(aShape, TopAbs_SHELL, TopAbs_SOLID); xp.More(); xp.Next(), ctShells++) {
366  createShape(xp.Current(), loc, name, localValue);
367  }
368  }
369 
370  if (!localValue.empty() && !mergeShape) {
371  pcPart = static_cast<App::Part*>(doc->addObject("App::Part", name.c_str()));
372  pcPart->Label.setValue(name);
373 
374  // localValue contain the objects that must added to the local Part
375  // We must add the PartOrigin and the Part itself
376  pcPart->addObjects(localValue);
377 
378  lValue.push_back(pcPart);
379  }
380 
381  if (ctSolids > 0 || ctShells > 0) {
382  return;
383  }
384  }
385  else if (!aShape.IsNull()) {
386  createShape(aShape, loc, name, lValue);
387  }
388 }

410 void ImportOCAF::loadColors(Part::Feature* part, const TopoDS_Shape& aShape)
411 {
412  Quantity_ColorRGBA aColor;
413  App::Color color(0.8f, 0.8f, 0.8f);
414  if (aColorTool->GetColor(aShape, XCAFDoc_ColorGen, aColor)
415  || aColorTool->GetColor(aShape, XCAFDoc_ColorSurf, aColor)
416  || aColorTool->GetColor(aShape, XCAFDoc_ColorCurv, aColor)) {
417  color = convertColor(aColor);
418  std::vector<App::Color> colors;
419  colors.push_back(color);
420  applyColors(part, colors);
421  }
422 
423  TopTools_IndexedMapOfShape faces;
424  TopExp_Explorer xp(aShape, TopAbs_FACE);
425  while (xp.More()) {
426  faces.Add(xp.Current());
427  xp.Next();
428  }
429 
430  bool found_face_color = false;
431  std::vector<App::Color> faceColors;
432  faceColors.resize(faces.Extent(), color);
433  xp.Init(aShape, TopAbs_FACE);
434  while (xp.More()) {
435  if (aColorTool->GetColor(xp.Current(), XCAFDoc_ColorGen, aColor)
436  || aColorTool->GetColor(xp.Current(), XCAFDoc_ColorSurf, aColor)
437  || aColorTool->GetColor(xp.Current(), XCAFDoc_ColorCurv, aColor)) {
438  int index = faces.FindIndex(xp.Current());
439  color = convertColor(aColor);
440  faceColors[index - 1] = color;
441  found_face_color = true;
442  }
443  xp.Next();
444  }
445 
446  if (found_face_color) {
447  applyColors(part, faceColors);
448  }
449 }

475 void ImportXCAF::loadShapes()
476 {
477  // collect sequence of labels to display
478  TDF_LabelSequence shapeLabels, colorLabels;
479  aShapeTool->GetFreeShapes(shapeLabels);
480  hColors->GetColors(colorLabels);
481 
482  // set presentations and show
483  for (Standard_Integer i = 1; i <= shapeLabels.Length(); i++) {
484  // get the shapes and attributes
485  const TDF_Label& label = shapeLabels.Value(i);
486  loadShapes(label);
487  }
488  std::map<Standard_Integer, TopoDS_Shape>::iterator it;
489  // go through solids
490  for (it = mySolids.begin(); it != mySolids.end(); ++it) {
491  createShape(it->second, true, true);
492  }
493  // go through shells
494  for (it = myShells.begin(); it != myShells.end(); ++it) {
495  createShape(it->second, true, true);
496  }
497  // go through compounds
498  for (it = myCompds.begin(); it != myCompds.end(); ++it) {
499  createShape(it->second, true, true);
500  }
501  // do the rest
502  if (!myShapes.empty()) {
503  BRep_Builder builder;
504  TopoDS_Compound comp;
505  builder.MakeCompound(comp);
506  for (it = myShapes.begin(); it != myShapes.end(); ++it) {
507  builder.Add(comp, it->second);
508  }
509  createShape(comp, true, false);
510  }
511 }

513 void ImportXCAF::createShape(const TopoDS_Shape& shape, bool perface, bool setname) const
514 {
515  Part::Feature* part;
516  part = static_cast<Part::Feature*>(doc->addObject("Part::Feature", default_name.c_str()));
517  part->Label.setValue(default_name);
518  part->Shape.setValue(shape);
519  std::map<Standard_Integer, Quantity_ColorRGBA>::const_iterator jt;
520  jt = myColorMap.find(Part::ShapeMapHasher {}(shape));
521 
522  App::Color partColor(0.8f, 0.8f, 0.8f);
523 
524 
525  // set label name if defined
526  if (setname && !myNameMap.empty()) {
527  std::map<Standard_Integer, std::string>::const_iterator jt;
528  jt = myNameMap.find(Part::ShapeMapHasher {}(shape));
529  if (jt != myNameMap.end()) {
530  part->Label.setValue(jt->second);
531  }
532  }
533 
534  // check for colors per face
535  if (perface && !myColorMap.empty()) {
536  TopTools_IndexedMapOfShape faces;
537  TopExp_Explorer xp(shape, TopAbs_FACE);
538  while (xp.More()) {
539  faces.Add(xp.Current());
540  xp.Next();
541  }
542 
543  std::vector<App::Color> faceColors;
544  faceColors.resize(faces.Extent(), partColor);
545  xp.Init(shape, TopAbs_FACE);
546  while (xp.More()) {
547  jt = myColorMap.find(Part::ShapeMapHasher {}(xp.Current()));
548  if (jt != myColorMap.end()) {
549  int index = faces.FindIndex(xp.Current());
550  faceColors[index - 1] = convertColor(jt->second);
551  }
552  xp.Next();
553  }
554  }

557 void ImportXCAF::loadShapes(const TDF_Label& label)
558 {
559  TopoDS_Shape aShape;
560  if (aShapeTool->GetShape(label, aShape)) {
561  if (aShapeTool->IsTopLevel(label)) {
562  int ctSolids = 0, ctShells = 0, ctComps = 0;
563  // add the shapes
564  TopExp_Explorer xp;
565  for (xp.Init(aShape, TopAbs_SOLID); xp.More(); xp.Next(), ctSolids++) {
566  this->mySolids[Part::ShapeMapHasher {}(xp.Current())] = (xp.Current());
567  }
568  for (xp.Init(aShape, TopAbs_SHELL, TopAbs_SOLID); xp.More(); xp.Next(), ctShells++) {
569  this->myShells[Part::ShapeMapHasher {}(xp.Current())] = (xp.Current());
570  }
571  // if no solids and no shells were found then go for compounds
572  if (ctSolids == 0 && ctShells == 0) {
573  for (xp.Init(aShape, TopAbs_COMPOUND); xp.More(); xp.Next(), ctComps++) {
574  this->myCompds[Part::ShapeMapHasher {}(xp.Current())] = (xp.Current());
575  }
576  }
577  if (ctComps == 0) {
578  for (xp.Init(aShape, TopAbs_FACE, TopAbs_SHELL); xp.More(); xp.Next()) {
579  this->myShapes[Part::ShapeMapHasher {}(xp.Current())] = (xp.Current());
580  }
581  for (xp.Init(aShape, TopAbs_WIRE, TopAbs_FACE); xp.More(); xp.Next()) {
582  this->myShapes[Part::ShapeMapHasher {}(xp.Current())] = (xp.Current());
583  }
584  for (xp.Init(aShape, TopAbs_EDGE, TopAbs_WIRE); xp.More(); xp.Next()) {
585  this->myShapes[Part::ShapeMapHasher {}(xp.Current())] = (xp.Current());
586  }
587  for (xp.Init(aShape, TopAbs_VERTEX, TopAbs_EDGE); xp.More(); xp.Next()) {
588  this->myShapes[Part::ShapeMapHasher {}(xp.Current())] = (xp.Current());
589  }
590  }
591  }
592 
593  // getting color
594  Quantity_ColorRGBA col;
595  if (hColors->GetColor(label, XCAFDoc_ColorGen, col)
596  || hColors->GetColor(label, XCAFDoc_ColorSurf, col)
597  || hColors->GetColor(label, XCAFDoc_ColorCurv, col)) {
598  // add defined color
599  myColorMap[Part::ShapeMapHasher {}(aShape)] = col;
600  }
601  else {
602  // http://www.opencascade.org/org/forum/thread_17107/
603  TopoDS_Iterator it;
604  for (it.Initialize(aShape); it.More(); it.Next()) {
605  if (hColors->GetColor(it.Value(), XCAFDoc_ColorGen, col)
606  || hColors->GetColor(it.Value(), XCAFDoc_ColorSurf, col)
607  || hColors->GetColor(it.Value(), XCAFDoc_ColorCurv, col)) {
608  // add defined color
609  myColorMap[Part::ShapeMapHasher {}(it.Value())] = col;
610  }
611  }
612  }
613 
614  // getting names
615  Handle(TDataStd_Name) name;
616  if (label.FindAttribute(TDataStd_Name::GetID(), name)) {
617  TCollection_ExtendedString extstr = name->Get();
618  char* str = new char[extstr.LengthOfCString() + 1];
619  extstr.ToUTF8CString(str);
620  std::string labelName(str);
621  if (!labelName.empty()) {
622  myNameMap[Part::ShapeMapHasher {}(aShape)] = labelName;
623  }
624  delete[] str;
625  }
626 
627  if (label.HasChild()) {
628  TDF_ChildIterator it;
629  for (it.Initialize(label); it.More(); it.Next()) {
630  loadShapes(it.Value());
631  }
632  }
