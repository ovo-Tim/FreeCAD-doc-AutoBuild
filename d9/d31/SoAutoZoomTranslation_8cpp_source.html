69 float SoAutoZoomTranslation::getScaleFactor(SoAction* action) const
70 {
71  float scale = scaleFactor.getValue();
72  if(!scale)
73  return 1.0;
74  // Dividing by 5 seems to work well
75  SbViewVolume vv = SoViewVolumeElement::get(action->getState());
76  float aspectRatio = SoViewportRegionElement::get(action->getState()).getViewportAspectRatio();
77  scale *= vv.getWorldToScreenScale(SbVec3f(0.f, 0.f, 0.f), 0.1f) / (5*aspectRatio);
78  return scale;
79 }

95 void SoAutoZoomTranslation::doAction(SoAction * action)
96 {
97  float sf = this->getScaleFactor(action);
98  auto state = action->getState();
99  SbRotation r,so;
100  SbVec3f s,t;
101  SbMatrix matrix = SoModelMatrixElement::get(action->getState()); // clazy:exclude=rule-of-two-soft
102  matrix.getTransform(t,r,s,so);
103  matrix.multVecMatrix(SbVec3f(0,0,0),t);
104  // reset current model scale factor
105  matrix.setTransform(t,r,SbVec3f(sf,sf,sf));
106  SoModelMatrixElement::set(state,this,matrix);
107 }

120 void SoAutoZoomTranslation::getMatrix(SoGetMatrixAction * action)
121 {
122  float sf = this->getScaleFactor(action);
123 
124  SbMatrix &m = action->getMatrix();
125 
126  SbRotation r,so;
