72 def display_external(internal_value,
73  decimals=None, dim='Length', showUnit=True, unit=None):
74  """Return a converted value for display, according to the unit schema.
75 
76  Parameters
77  ----------
78  internal_value: float
79  A value that will be transformed depending on the other parameters.
80 
81  decimals: float, optional
82  It defaults ot `None`, in which case, the decimals are 2.
83 
84  dim: str, optional
85  It defaults to `'Length'`. It can also be `'Angle'`.
86 
87  showUnit: bool, optional
88  It defaults to `True`.
89  If it is `False` it won't show the unit.
90 
91  unit: str, optional
92  A unit string such as `'mm'`, `'cm'`, `'m'`, `'in'`, `'ft'`,
93  in which to express the returned value.
94  """
95  if dim == 'Length':
96  q = App.Units.Quantity(internal_value, App.Units.Length)
97  if not unit:
98  if decimals is None and showUnit:
99  return q.UserString
100 
101  conversion = q.getUserPreferred()[1]
102  uom = q.getUserPreferred()[2]
103  elif unit.lower() == "arch":
104  return App.Units.schemaTranslate(q,5)[0].replace("+"," ")
105  else:
106  try:
107  uom = unit
108  internal_value = q.getValueAs(unit)
109  conversion = 1
110  except Exception:
111  conversion = q.getUserPreferred()[1]
112  uom = q.getUserPreferred()[2]
113  elif dim == 'Angle':
114  q = App.Units.Quantity(internal_value, App.Units.Angle)
115  if decimals is None:
116  return q.UserString
117 
118  conversion = q.getUserPreferred()[1]
119  uom = q.getUserPreferred()[2]
120  else:
121  conversion = 1.0
122  if not decimals:
123  decimals = 2
124  uom = "??"
125 
126  if not showUnit:
127  uom = ""
128 
129  decimals = abs(decimals) # prevent negative values
130  fmt = "{0:." + str(decimals) + "f} " + uom
131  display = fmt.format(float(internal_value) / float(conversion))
132  display = display.replace(".", QtCore.QLocale().decimalPoint())
133 
134  return display
