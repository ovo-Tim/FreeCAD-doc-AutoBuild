142 def type_check(args_and_types, name="?"):
143  """Check that the arguments are instances of certain types.
144 
145  Parameters
146  ----------
147  args_and_types : list
148  A list of tuples. The first element of a tuple is tested as being
149  an instance of the second element.
150  ::
151  args_and_types = [(a, Type), (b, Type2), ...]
152 
153  Then
154  ::
155  isinstance(a, Type)
156  isinstance(b, Type2)
157 
158  A `Type` can also be a tuple of many types, in which case
159  the check is done for any of them.
160  ::
161  args_and_types = [(a, (Type3, int, float)), ...]
162 
163  isinstance(a, (Type3, int, float))
164 
165  name : str, optional
166  Defaults to `'?'`. The name of the check.
167 
168  Raises
169  ------
170  TypeError
171  If the first element in the tuple is not an instance of the second
172  element, it raises `Draft.name`.
173  """
174  for v, t in args_and_types:
175  if not isinstance(v, t):
176  w = "typecheck[{}]: '{}' is not {}".format(name, v, t)
177  _wrn(w)
178  raise TypeError("Draft." + str(name))

255  value depending on the type of parameter determined

273  if default is None:

278  elif t == "string":

280  default = ""

282  elif t == "float":
283  if default is None:
284  default = 0
285  return p.GetFloat(param, default)
286  elif t == "bool":
287  if default is None:
288  default = False

290  elif t == "unsigned":
291  if default is None:
292  default = 0

479 def is_clone(obj, objtype=None, recursive=False):
480  """Return True if the given object is a clone of a certain type.
481 
482  A clone is of type `'Clone'`, and has a reference
483  to the original object inside its `Objects` attribute,
484  which is an `'App::PropertyLinkListGlobal'`.
485 
486  The `Objects` attribute can point to another `'Clone'` object.
487  If `recursive` is `True`, the function will be called recursively
488  to further test this clone, until the type of the original object
489  can be compared to `objtype`.
490 
491  Parameters
492  ----------
493  obj : App::DocumentObject
494  The clone object that will be tested for a certain type.
495 
496  objtype : str or list of str
497  A type string such as one obtained from `get_type`.
498  Or a list of such types.
499 
500  recursive : bool, optional
501  It defaults to `False`.
502  If it is `True`, this same function will be called recursively
503  with `obj.Object[0]` as input.
504 
505  This option only works if `obj.Object[0]` is of type `'Clone'`,
506  that is, if `obj` is a clone of a clone.
507 
508  Returns
509  -------
510  bool
511  Returns `True` if `obj` is of type `'Clone'`,
512  and `obj.Object[0]` is of type `objtype`.
513 
514  If `objtype` is a list, then `obj.Objects[0]`
515  will be tested against each of the elements in the list,
516  and it will return `True` if at least one element matches the type.
517 
518  If `obj` isn't of type `'Clone'` but has the `CloneOf` attribute,
519  it will also return `True`.
520 
521  It returns `False` otherwise, for example,
522  if `obj` is not even a clone.
523  """
524  if isinstance(objtype, list):
525  return any([is_clone(obj, t, recursive) for t in objtype])
526  if getType(obj) == "Clone":
527  if len(obj.Objects) == 1:
528  if objtype:
529  if getType(obj.Objects[0]) == objtype:
530  return True
531  elif recursive and (getType(obj.Objects[0]) == "Clone"):
532  return is_clone(obj.Objects[0], objtype, recursive)
533  elif hasattr(obj, "CloneOf"):
534  if obj.CloneOf:
535  if objtype:
536  if getType(obj.CloneOf) == objtype:
537  return True
538  else:
539  return True
540  return False

599 def shapify(obj):
600  """Transform a parametric object into a static, non-parametric shape.
601 
602  Parameters
603  ----------
604  obj : App::DocumentObject
605  Any type of scripted object.
606 
607  This object will be removed, and a non-parametric object
608  with the same topological shape (`Part::TopoShape`)
609  will be created.
610 
611  Returns
612  -------
613  Part::Feature
614  The new object that takes `obj.Shape` as its own.
615 
616  Depending on the contents of the Shape, the resulting object
617  will be named `'Face'`, `'Solid'`, `'Compound'`,
618  `'Shell'`, `'Wire'`, `'Line'`, `'Circle'`,
619  or the name returned by `get_real_name(obj.Name)`.
620 
621  If there is a problem with `obj.Shape`, it will return `None`,
622  and the original object will not be modified.
623  """
624  try:
625  shape = obj.Shape
626  except Exception:
627  return None
628 
629  if len(shape.Faces) == 1:
630  name = "Face"
631  elif len(shape.Solids) == 1:
632  name = "Solid"
633  elif len(shape.Solids) > 1:
634  name = "Compound"
635  elif len(shape.Faces) > 1:
636  name = "Shell"
637  elif len(shape.Wires) == 1:
638  name = "Wire"
639  elif len(shape.Edges) == 1:
640  import DraftGeomUtils
641  if DraftGeomUtils.geomType(shape.Edges[0]) == "Line":
642  name = "Line"
643  else:
644  name = "Circle"
645  else:
646  name = getRealName(obj.Name)
647 
648  App.ActiveDocument.removeObject(obj.Name)
649  newobj = App.ActiveDocument.addObject("Part::Feature", name)
650  newobj.Shape = shape
651 
652  return newobj

913 def utf8_decode(text):
914  r"""Decode the input string and return a unicode string.
915 
916  Python 2:
917  ::
918  str -> unicode
919  unicode -> unicode
920 
921  Python 3:
922  ::
923  str -> str
924  bytes -> str
925 
926  It runs
927  ::
928  try:
929  return text.decode("utf-8")
930  except AttributeError:
931  return text
932 
933  Parameters
934  ----------
935  text : str, unicode or bytes
936  A str, unicode, or bytes object that may have unicode characters
937  like accented characters.
938 
939  In Python 2, a `bytes` object can include accented characters,
940  but in Python 3 it must only contain ASCII literal characters.
941 
942  Returns
943  -------
944  unicode or str
945  In Python 2 it will try decoding the `bytes` string
946  and return a `'utf-8'` decoded string.
947 
948  >>> "Aá".decode("utf-8")
949  >>> b"Aá".decode("utf-8")
950  u'A\xe1'
951 
952  In Python 2 the unicode string is prefixed with `u`,
953  and unicode characters are replaced by their two-digit hexadecimal
954  representation, or four digit unicode escape.
955 
956  >>> "AáBẃCñ".decode("utf-8")
957  u'A\xe1B\u1e83C\xf1'
958 
959  In Python 2 it will always return a `unicode` object.
960 
961  In Python 3 a regular string is already unicode encoded,
962  so strings have no `decode` method. In this case, `text`
963  will be returned as is.
964 
965  In Python 3, if `text` is a `bytes` object, then it will be converted
966  to `str`; in this case, the `bytes` object cannot have accents,
967  it must only contain ASCII literal characters.
968 
969  >>> b"ABC".decode("utf-8")
970  'ABC'
971 
972  In Python 3 it will always return a `str` object, with no prefix.
973  """
974  try:
975  return text.decode("utf-8")
976  except AttributeError:
977  return text
