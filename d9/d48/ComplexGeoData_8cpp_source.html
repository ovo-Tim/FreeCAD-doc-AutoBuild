264 Data::MappedElement
265 ComplexGeoData::getElementName(const char* name, ElementIDRefs* sid, bool copy) const
266 {
267  IndexedName element(name, getElementTypes());
268  if (element) {
269  return {getMappedName(element, false, sid), element};
270  }
271 
272  const char* mapped = isMappedElement(name);
273  if (mapped) {
274  name = mapped;
275  }
276 
277  MappedElement result;
278  // Strip out the trailing '.XXXX' if any
279  const char* dot = strchr(name, '.');
280  if (dot) {
281  result.name = MappedName(name, static_cast<int>(dot - name));
282  }
283  else if (copy) {
284  result.name = name;
285  }
286  else {
287  result.name = MappedName(name);
288  }
289  result.index = getIndexedName(result.name, sid);
290  return result;

374 char ComplexGeoData::elementType(const Data::IndexedName& element) const
375 {
376  if (!element) {
377  return 0;
378  }
379  for (auto& type : getElementTypes()) {
380  if (boost::equals(element.getType(), type)) {
381  return type[0];
382  }
383  }
384  return 0;
385 }
386 
387 // The elementType function can take a char *, in which case it tries a sequence of checks to
388 // see what it got.
389 // 1) Check to see if it is an indexedName, and if so directly look up the type
390 // 2) If not:
391 // a) Remove any element map prefix that is present
392 // b) See if the name contains a dot:
393 // i) If yes, create a MappedName from the part before the dot, and set the type to the
394 // part after the dot
395 // ii) If no, create a MappedName from the whole name
396 // c) Try to get the elementType based on the MappedName. Return it if found
397 // 3) Check to make sure the discovered type is in the list of types, and return its first
398 // character if so.
399 char ComplexGeoData::elementType(const char* name) const
400 {
401  if (!name) {
402  return 0;
403  }
404 
405  const char* type = nullptr;
406  IndexedName element(name, getElementTypes());
407  if (element) {
408  type = element.getType();
409  }
410  else {
411  const char* mapped = isMappedElement(name);
412  if (mapped) {
413  name = mapped;
414  }
415 
416  MappedName mappedName;
417  const char* dot = strchr(name, '.');
418  if (dot) {
419  mappedName = MappedName(name, static_cast<int>(dot - name));
420  type = dot + 1;
421  }
422  else {
423  mappedName = MappedName::fromRawData(name);
424  }
425  char res = elementType(mappedName);
426  if (res != 0) {
427  return res;
428  }
429  }
430 
431  if (type && (type[0] != 0)) {
432  for (auto& elementTypes : getElementTypes()) {
433  if (boost::starts_with(type, elementTypes)) {
434  return type[0];
435  }
436  }
437  }
438  return 0;
439 }

476 void ComplexGeoData::Restore(Base::XMLReader& reader)
477 {
478  resetElementMap();
479 
480  reader.readElement("ElementMap");
481  bool newTag = false;
482  if (reader.hasAttribute("new") && reader.getAttributeAsInteger("new") > 0) {
483  reader.readEndElement("ElementMap");
484  reader.readElement("ElementMap2");
485  newTag = true;
486  }
487 
488  const char* file = "";
489  if (reader.hasAttribute("file")) {
490  file = reader.getAttribute("file");
491  }
492  if (*file != 0) {
493  reader.addFile(file, this);
494  return;
495  }
496 
497  std::size_t count = 0;
498  if (reader.hasAttribute("count")) {
499  count = reader.getAttributeAsUnsigned("count");
500  }
501  if (count == 0) {
502  return;
503  }
504 
505  if (newTag) {
506  resetElementMap(std::make_shared<ElementMap>());
507  _elementMap =
508  _elementMap->restore(Hasher, reader.beginCharStream(Base::CharStreamFormat::Raw));
509  reader.endCharStream();
510  reader.readEndElement("ElementMap2");
511  return;
512  }
513 
514  if (reader.FileVersion > 1) {
515  restoreStream(reader.beginCharStream(Base::CharStreamFormat::Raw), count);
516  reader.endCharStream();
517  return;
518  }
519  readElements(reader, count);
520  reader.readEndElement("ElementMap");
521 }

523 void ComplexGeoData::readElements(Base::XMLReader& reader, size_t count)
524 {
525  size_t invalid_count = 0;
526  bool warned = false;
527 
528  const auto& types = getElementTypes();
529 
530  for (size_t i = 0; i < count; ++i) {
531  reader.readElement("Element");
532  ElementIDRefs sids;
533  if (reader.hasAttribute("sid")) {
534  if (!Hasher) {
535  if (!warned) {
536  warned = true;
537  FC_ERR("missing hasher"); // NOLINT
538  }
539  }
540  else {
541  const char* attr = reader.getAttribute("sid");
542  bio::stream<bio::array_source> iss(attr, std::strlen(attr));
543  long id {};
544  while ((iss >> id)) {
545  if (id == 0) {
546  continue;
547  }
548  auto sid = Hasher->getID(id);
549  if (!sid) {
550  ++invalid_count;
551  }
552  else {
553  sids.push_back(sid);
554  }
555  char sep {};
556  iss >> sep;
557  }
558  }
559  }
560  ensureElementMap()->setElementName(IndexedName(reader.getAttribute("value"), types),
561  MappedName(reader.getAttribute("key")),
562  Tag,
563  &sids);
564  }
565  if (invalid_count != 0) {
566  FC_ERR("Found " << invalid_count << " invalid string id"); // NOLINT
567  }
568 }

570 void ComplexGeoData::restoreStream(std::istream& stream, std::size_t count)
571 {
572  resetElementMap();
573 
574  size_t invalid_count = 0;
575  std::string key;
576  std::string value;
577  std::string sid;
578  bool warned = false;
579 
580  const auto& types = getElementTypes();
581  try {
582  for (size_t i = 0; i < count; ++i) {
583  ElementIDRefs sids;
584  std::size_t sCount = 0;
585  if (!(stream >> value >> key >> sCount)) {
586  // NOLINTNEXTLINE
587  FC_THROWM(Base::RuntimeError, "Failed to restore element map " << _persistenceName);
588  }
589  sids.reserve(static_cast<int>(sCount));
590  for (std::size_t j = 0; j < sCount; ++j) {
591  long id = 0;
592  if (!(stream >> id)) {
593  // NOLINTNEXTLINE
594  FC_THROWM(Base::RuntimeError,
595  "Failed to restore element map " << _persistenceName);
596  }
597  if (Hasher) {
598  auto hasherSID = Hasher->getID(id);
599  if (!hasherSID) {
600  ++invalid_count;
601  }
602  else {
603  sids.push_back(hasherSID);
604  }
605  }
606  }
607  if (sCount != 0 && !Hasher) {
608  sids.clear();
609  if (!warned) {
610  warned = true;
611  FC_ERR("missing hasher"); // NOLINT
612  }
613  }
614  _elementMap->setElementName(IndexedName(value.c_str(), types),
615  MappedName(key),
616  Tag,
617  &sids);
618  }
619  }
620  catch (Base::Exception& e) {
621  e.ReportException();
622  _restoreFailed = true;
623  _elementMap.reset();
624  }
625  if (invalid_count != 0) {
626  FC_ERR("Found " << invalid_count << " invalid string id"); // NOLINT
627  }
628 }
