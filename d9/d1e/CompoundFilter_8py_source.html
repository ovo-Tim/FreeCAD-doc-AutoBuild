91  def execute(self, obj):
92  # When operating on the object, it is to be treated as a lattice object.
93  # If False, treat as a regular shape.
94  if hasattr(obj, "isLattice"):
95  if 'On' in obj.isLattice:
96  print(obj.Name + " A generic shape is expected, but an array of placements was supplied. "
97  "It will be treated as a generic shape.\n")
98 
99  rst = [] # variable to receive the final list of shapes
100  shps = obj.Base.Shape.childShapes()
101  if obj.FilterType == 'bypass':
102  rst = shps
103  elif obj.FilterType == 'specific items':
104  rst = []
105  flags = [False] * len(shps)
106  if not obj.items:
107  raise ValueError("The 'items' property must have a number to use this filter: '{}'".format(obj.FilterType))
108  ranges = obj.items.split(';')
109  for r in ranges:
110  r_v = r.split(':')
111  if len(r_v) == 1:
112  try:
113  i = int(r_v[0])
114  except ValueError:
115  raise ValueError("Make sure the 'item' does not have spaces; "
116  "a semicolon (;) can only appear between two numbers; "
117  "filter: '{}'".format(obj.FilterType))
118  try:
119  rst.append(shps[i])
120  except IndexError:
121  raise ValueError("Item index '{}' is out of range for this filter: '{}'".format(i, obj.FilterType))
122  flags[i] = True
123  elif len(r_v) == 2 or len(r_v) == 3:
124  if len(r_v) == 2:
125  # fix issue #1: instead of checking length
126  # here and there, simply add the missing field =) (DeepSOIC)
127  r_v.append("")
128  ifrom = None if len(r_v[0].strip()) == 0 else int(r_v[0])
129  ito = None if len(r_v[1].strip()) == 0 else int(r_v[1])
130  istep = None if len(r_v[2].strip()) == 0 else int(r_v[2])
131  rst = rst + shps[ifrom:ito:istep]
132  for b in flags[ifrom:ito:istep]:
133  b = True
134  else:
135  raise ValueError("index range cannot be parsed: '{}'".format(r))
136  if obj.Invert:
137  rst = []
138  for i in range(len(shps)):
139  if not flags[i]:
140  rst.append(shps[i])
141  elif obj.FilterType == 'collision-pass':
142  if not obj.Stencil:
143  raise ValueError("A 'Stencil' object must be set to use this filter: '{}'".format(obj.FilterType))
144 
145  stencil = obj.Stencil.Shape
146  for s in shps:
147  d = s.distToShape(stencil)
148  if bool(d[0] < Part.Precision.confusion()) ^ bool(obj.Invert):
149  rst.append(s)
150  elif obj.FilterType in ('window-volume', 'window-area',
151  'window-length', 'window-distance'):
152  vals = [0.0] * len(shps)
153  for i in range(len(shps)):
154  if obj.FilterType == 'window-volume':
155  vals[i] = shps[i].Volume
156  elif obj.FilterType == 'window-area':
157  vals[i] = shps[i].Area
158  elif obj.FilterType == 'window-length':
159  vals[i] = shps[i].Length
160  elif obj.FilterType == 'window-distance':
161  if not obj.Stencil:
162  raise ValueError("A 'Stencil' object must be set to use this filter: '{}'".format(obj.FilterType))
163  vals[i] = shps[i].distToShape(obj.Stencil.Shape)[0]
164 
165  maxval = max(vals)
166  if obj.Stencil:
167  if obj.FilterType == 'window-volume':
168  maxval = obj.Stencil.Shape.Volume
169  elif obj.FilterType == 'window-area':
170  maxval = obj.Stencil.Shape.Area
171  elif obj.FilterType == 'window-length':
172  maxval = obj.Stencil.Shape.Length
173  if obj.OverrideMaxVal:
174  maxval = obj.OverrideMaxVal
175 
176  valFrom = obj.WindowFrom / 100.0 * maxval
177  valTo = obj.WindowTo / 100.0 * maxval
178 
179  for i in range(len(shps)):
180  if bool(vals[i] >= valFrom and vals[i] <= valTo) ^ obj.Invert:
181  rst.append(shps[i])
182  else:
183  raise ValueError("Filter mode not implemented: '{}'".format(obj.FilterType))
184 
185  if len(rst) == 0:
186  scale = 1.0
187  if not obj.Base.Shape.isNull():
188  scale = obj.Base.Shape.BoundBox.DiagonalLength / math.sqrt(3) / math.sqrt(len(shps))
189  if scale < Part.Precision.confusion() * 100:
190  scale = 1.0
191  print(scale)
192  obj.Shape = getNullShapeShape(scale)
193  # Feeding empty compounds to FreeCAD seems to cause rendering issues,
194  # otherwise it would have been a good idea to output nothing.
195  raise ValueError('Nothing passes through the filter')
196 
197  if len(rst) > 1:
198  obj.Shape = Part.makeCompound(rst)
199  else: # don't make compound of one shape, output it directly
200  sh = rst[0]
201  sh.transformShape(sh.Placement.toMatrix(), True) # True = make copy
202  sh.Placement = FreeCAD.Placement()
203  obj.Shape = sh
204 
205  return
