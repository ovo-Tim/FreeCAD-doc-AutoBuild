123 bool SketcherGui::tryAutoRecompute(Sketcher::SketchObject* obj, bool& autoremoveredundants)
124 {
125  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
126  "User parameter:BaseApp/Preferences/Mod/Sketcher");
127  bool autoRecompute = hGrp->GetBool("AutoRecompute", false);
128  bool autoRemoveRedundants = hGrp->GetBool("AutoRemoveRedundants", false);
129 
130  // We need to make sure the solver has right redundancy information before trying to remove the
131  // redundants. for example if a non-driving constraint has been added.
132  if (autoRemoveRedundants && autoRecompute) {
133  obj->solve();
134  }
135 
136  if (autoRemoveRedundants) {
137  obj->autoRemoveRedundants();
138  }
139 
140  if (autoRecompute) {
141  Gui::Command::updateActive();
142  }
143 
144  autoremoveredundants = autoRemoveRedundants;
145 
146  return autoRecompute;
147 }

361 bool SketcherGui::IsPointAlreadyOnCurve(int GeoIdCurve,
362  int GeoIdPoint,
363  Sketcher::PointPos PosIdPoint,
364  Sketcher::SketchObject* Obj)
365 {
366  // This func is a "smartness" behind three-element tangent-, perp.- and angle-via-point.
367  // We want to find out, if the point supplied by user is already on
368  // both of the curves. If not, necessary point-on-object constraints
369  // are to be added automatically.
370  // Simple geometric test seems to be the best, because a point can be
371  // constrained to a curve in a number of ways (e.g. it is an endpoint of an
372  // arc, or is coincident to endpoint of an arc, or it is an endpoint of an
373  // ellipse's major diameter line). Testing all those possibilities is way
374  // too much trouble, IMO(DeepSOIC).
375  // One exception: check for knots on their B-splines, at least until point on B-spline is
376  // implemented. (Ajinkya)
377  if (isBsplineKnot(Obj, GeoIdPoint)) {
378  const Part::Geometry* geoCurve = Obj->getGeometry(GeoIdCurve);
379  if (geoCurve->is<Part::GeomBSplineCurve>()) {
380  const std::vector<Constraint*>& constraints = Obj->Constraints.getValues();
381  for (const auto& constraint : constraints) {
382  if (constraint->Type == Sketcher::ConstraintType::InternalAlignment
383  && constraint->First == GeoIdPoint && constraint->Second == GeoIdCurve) {
384  return true;
385  }
386  }
387  }
388  }
389 
390  Base::Vector3d p = Obj->getPoint(GeoIdPoint, PosIdPoint);
391  return Obj->isPointOnCurve(GeoIdCurve, p.x, p.y);
392 }

438 // Set the two points on circles at minimal distance
439 // in concentric case set points on relative X axis
440 void SketcherGui::GetCirclesMinimalDistance(const Part::Geometry* geom1,
441  const Part::Geometry* geom2,
442  Base::Vector3d& point1,
443  Base::Vector3d& point2)
444 {
445  // This will throw if geom1 or geom2 are not circles or arcs
446  auto [radius1, center1] = getRadiusCenterCircleArc(geom1);
447  auto [radius2, center2] = getRadiusCenterCircleArc(geom2);
448 
449  point1 = center1;
450  point2 = center2;
451 
452  Base::Vector3d v = point2 - point1;
453  double length = v.Length();
454 
455  if (length == 0) { // concentric case
456  point1.x += radius1;
457  point2.x += radius2;
458  }
459  else {
460  v = v.Normalize();
461  if (length <= std::max(radius1, radius2)) { // inner case
462  if (radius1 > radius2) {
463  point1 += v * radius1;
464  point2 += v * radius2;
465  }
466  else {
467  point1 += -v * radius1;
468  point2 += -v * radius2;
469  }
470  }
471  else { // outer case
472  point1 += v * radius1;
473  point2 += -v * radius2;
474  }
475  }
476 }

555 void SketcherGui::removeRedundantHorizontalVertical(Sketcher::SketchObject* psketch,
556  std::vector<AutoConstraint>& sug1,
557  std::vector<AutoConstraint>& sug2)
558 {
559  if (!sug1.empty() && !sug2.empty()) {
560 
561  bool rmvhorvert = false;
562 
563  // we look for:
564  // 1. Coincident to external on both endpoints
565  // 2. Coincident in one endpoint to origin and pointonobject/tangent to an axis on the other
566  auto detectredundant =
567  [psketch](std::vector<AutoConstraint>& sug, bool& ext, bool& orig, bool& axis) {
568  ext = false;
569  orig = false;
570  axis = false;
571 
572  for (std::vector<AutoConstraint>::const_iterator it = sug.begin(); it != sug.end();
573  ++it) {
574  if ((*it).Type == Sketcher::Coincident && !ext) {
575  const std::map<int, Sketcher::PointPos> coincidents =
576  psketch->getAllCoincidentPoints((*it).GeoId, (*it).PosId);
577 
578  if (!coincidents.empty()) {
579  // the keys are ordered, so if the first is negative, it is coincident
580  // with external
581  ext = coincidents.begin()->first < 0;
582 
583  std::map<int, Sketcher::PointPos>::const_iterator geoId1iterator;
584 
585  geoId1iterator = coincidents.find(-1);
586 
587  if (geoId1iterator != coincidents.end()) {
588  if ((*geoId1iterator).second == Sketcher::PointPos::start) {
589  orig = true;
590  }
591  }
592  }
593  else { // it may be that there is no constraint at all, but there is
594  // external geometry
595  ext = (*it).GeoId < 0;
596  orig = ((*it).GeoId == -1 && (*it).PosId == Sketcher::PointPos::start);
597  }
598  }
599  else if ((*it).Type == Sketcher::PointOnObject && !axis) {
600  axis = (((*it).GeoId == -1 && (*it).PosId == Sketcher::PointPos::none)
601  || ((*it).GeoId == -2 && (*it).PosId == Sketcher::PointPos::none));
602  }
603  }
604  };
605 
606  bool firstext = false, secondext = false, firstorig = false, secondorig = false,
607  firstaxis = false, secondaxis = false;
608 
609  detectredundant(sug1, firstext, firstorig, firstaxis);
610  detectredundant(sug2, secondext, secondorig, secondaxis);
611 
612 
613  rmvhorvert =
614  ((firstext && secondext) || // coincident with external on both endpoints
615  (firstorig && secondaxis) || // coincident origin and point on object on other
616  (secondorig && firstaxis));
617 
618  if (rmvhorvert) {
619  for (std::vector<AutoConstraint>::reverse_iterator it = sug2.rbegin();
620  it != sug2.rend();
621  ++it) {
622  if ((*it).Type == Sketcher::Horizontal || (*it).Type == Sketcher::Vertical) {
623  sug2.erase(std::next(it).base());
624  it = sug2.rbegin(); // erase invalidates the iterator
625  }
626  }
627  }
628  }
629 }

632  Sketcher::GeoElementId attachment,
633  double distance,

704 // convert value to display format %0.[digits]f. Units are displayed if
705 // preference "ShowUnits" is true, or if the unit schema in effect uses
706 // multiple units (ex. Ft/In). Digits parameter is ignored for multi-unit
707 // schemata
708 // TODO:: if the user string is delivered in 1.23e45 format, this might not work
709 // correctly.
710 std::string SketcherGui::lengthToDisplayFormat(double value, int digits)
711 {
712  Base::Quantity asQuantity;
713  asQuantity.setValue(value);
714  asQuantity.setUnit(Base::Unit::Length);
715  std::string userString = asQuantity.getUserString();
716  if (Base::UnitsApi::isMultiUnitLength() || (!hideUnits() && useSystemDecimals())) {
717  // just return the user string
718  return userString;
719  }
720 
721  // find the unit of measure
722  double factor = 1.0;
723  std::string unitString;
724  std::string translate = Base::UnitsApi::schemaTranslate(asQuantity, factor, unitString);
725  std::string unitPart = " " + unitString;
726 
727  // get the numeric part of the user string
728  QRegularExpression rxNoUnits(
729  QString::fromUtf8("(.*) \\D*$")); // text before space + any non digits at end of string
730  QRegularExpressionMatch match = rxNoUnits.match(QString::fromStdString(userString));
731  if (!match.hasMatch()) {
732  // no units in userString?
733  return userString;
734  }
735  QString matched = match.captured(1); // matched is the numeric part of user string
736  auto smatched = matched.toStdString();
737  int dpPos = matched.indexOf(QLocale().decimalPoint());
738  if (dpPos < 0) {
739  // no decimal separator (ie an integer), return all the digits
740  if (!hideUnits()) {
741  smatched.append(unitPart);
742  }
743  return smatched;
744  }
745 
746  // real number
747  if (useSystemDecimals() && hideUnits()) {
748  // return just the numeric part of the user string
749  return smatched;
750  }
751 
752  // real number and not using system decimals
753  int requiredLength = dpPos + digits + 1;
754  if (requiredLength > matched.size()) {
755  // just take the whole thing
756  requiredLength = matched.size();
757  }
758  auto numericPart = matched.left(requiredLength).toStdString();
759  if (!hideUnits()) {
760  numericPart.append(unitPart);
761  }
762  return numericPart;
763 }
764 
765 // convert value to display format %0.[digits]f. Units are always displayed for
766 // angles - 123.456° or 12°34'56". Digits parameter is ignored for multi-unit
767 // schemata. Note small differences between this method and lengthToDisplyFormat
768 // TODO:: if the user string is delivered in 1.23e45 format, this might not work
769 // correctly.
770 std::string SketcherGui::angleToDisplayFormat(double value, int digits)
771 {
772  Base::Quantity asQuantity;
773  asQuantity.setValue(value);
774  asQuantity.setUnit(Base::Unit::Angle);
775  QString qUserString = QString::fromStdString(asQuantity.getUserString());
776  if (Base::UnitsApi::isMultiUnitAngle()) {
777  // just return the user string
778  // Coin SbString doesn't handle utf8 well, so we convert to ascii
779  QString schemeMinute = QString::fromUtf8("\xE2\x80\xB2"); // prime symbol
780  QString schemeSecond = QString::fromUtf8("\xE2\x80\xB3"); // double prime symbol
781  QString escapeMinute = QString::fromLatin1("\'"); // substitute ascii single quote
782  QString escapeSecond = QString::fromLatin1("\""); // substitute ascii double quote
783  QString displayString = qUserString.replace(schemeMinute, escapeMinute);
784  displayString = displayString.replace(schemeSecond, escapeSecond);
785  return displayString.toStdString();
786  }
787 
788  // we always use use U+00B0 (°) as the unit of measure for angles in
789  // single unit schema. Will need a change to support rads or grads.
790  std::string unitString = "°";
791  auto decimalSep = QLocale().decimalPoint();
792 
793  // get the numeric part of the user string
794  QRegularExpression rxNoUnits(QString::fromUtf8("(\\d*\\%1?\\d*)(\\D*)$")
795  .arg(decimalSep)); // number + non digits at end of string
796  QRegularExpressionMatch match = rxNoUnits.match(qUserString);
797  if (!match.hasMatch()) {
798  // no units in userString?
799  return qUserString.toStdString();
800  }
801  QString matched = match.captured(1); // matched is the numeric part of user string
802  int dpPos = matched.indexOf(decimalSep);
803  if (dpPos < 0 || useSystemDecimals()) {
804  // just the numeric part of the user string + degree symbol
805  auto angle = matched.toStdString();
806  angle.append(unitString);
807  return angle;
808  }
809 
810  // real number and not using system decimals
811  int requiredLength = dpPos + digits + 1;
812  if (requiredLength > matched.size()) {
813  // just take the whole thing
814  requiredLength = matched.size();
815  }
816  auto numericPart = matched.left(requiredLength).toStdString();
817  numericPart.append(unitString);
818  return numericPart;
819 }

822 bool SketcherGui::areCollinear(const Base::Vector2d& p1,
823  const Base::Vector2d& p2,
824  const Base::Vector2d& p3)
825 {
826  Base::Vector2d u = p2 - p1;
827  Base::Vector2d v = p3 - p2;
828  Base::Vector2d w = p1 - p3;
829 
830  double uu = u * u;
831  double vv = v * v;
832  double ww = w * w;
833 
834  double eps2 = Precision::SquareConfusion();
835  if (uu < eps2 || vv < eps2 || ww < eps2) {
836  return true;
837  }
838 
839  double uv = -(u * v);
840  double vw = -(v * w);
841  double uw = -(u * w);
842 
843  double w0 = (2 * sqrt(abs(uu * ww - uw * uw)) * uw / (uu * ww));
844  double w1 = (2 * sqrt(abs(uu * vv - uv * uv)) * uv / (uu * vv));
845  double w2 = (2 * sqrt(abs(vv * ww - vw * vw)) * vw / (vv * ww));
846 
847  double wx = w0 + w1 + w2;
848 
849  if (abs(wx) < Precision::Confusion()) {
850  return true;
851  }
852 
853  return false;
854 }
