401  def extract_metadata_dependencies(self, metadata: Metadata):
402  """Read dependency information from a metadata object and store it in this
403  Addon"""
404 
405  # Version check: if this piece of metadata doesn't apply to this version of
406  # FreeCAD, just skip it.
407  if not Addon.version_is_ok(metadata):
408  return
409 
410  if metadata.pythonmin:
411  self.python_min_version["major"] = metadata.pythonmin.version_as_list[0]
412  self.python_min_version["minor"] = metadata.pythonmin.version_as_list[1]
413 
414  for dep in metadata.depend:
415  if dep.dependency_type == DependencyType.internal:
416  if dep.package in INTERNAL_WORKBENCHES:
417  self.requires.add(dep.package)
418  else:
419  fci.Console.PrintWarning(
420  translate(
421  "AddonsInstaller",
422  "{}: Unrecognized internal workbench '{}'",
423  ).format(self.name, dep.package)
424  )
425  elif dep.dependency_type == DependencyType.addon:
426  self.requires.add(dep.package)
427  elif dep.dependency_type == DependencyType.python:
428  if dep.optional:
429  self.python_optional.add(dep.package)
430  else:
431  self.python_requires.add(dep.package)
432  else:
433  # Automatic resolution happens later, once we have a complete list of
434  # Addons
435  self.requires.add(dep.package)
436 
437  for dep in metadata.conflict:
438  self.blocks.add(dep.package)
439 
440  # Recurse
441  content = metadata.content
442  for _, value in content.items():
443  for item in value:
444  self.extract_metadata_dependencies(item)

592  def walk_dependency_tree(self, all_repos, deps):
593  """Compute the total dependency tree for this repo (recursive)
594  - all_repos is a dictionary of repos, keyed on the name of the repo
595  - deps is an Addon.Dependency object encapsulating all the types of dependency
596  information that may be needed.
597  """
598 
599  deps.python_requires |= self.python_requires
600  deps.python_optional |= self.python_optional
601 
602  deps.python_min_version["major"] = max(
603  deps.python_min_version["major"], self.python_min_version["major"]
604  )
605  if deps.python_min_version["major"] == 3:
606  deps.python_min_version["minor"] = max(
607  deps.python_min_version["minor"], self.python_min_version["minor"]
608  )
609  else:
610  fci.Console.PrintWarning("Unrecognized Python version information")
611 
612  for dep in self.requires:
613  if dep in all_repos:
614  if dep not in deps.required_external_addons:
615  deps.required_external_addons.append(all_repos[dep])
616  all_repos[dep].walk_dependency_tree(all_repos, deps)
617  else:
618  # See if this is an internal workbench:
619  if dep.upper().endswith("WB"):
620  real_name = dep[:-2].strip().lower()
621  elif dep.upper().endswith("WORKBENCH"):
622  real_name = dep[:-9].strip().lower()
623  else:
624  real_name = dep.strip().lower()
625 
626  if real_name in INTERNAL_WORKBENCHES:
627  deps.internal_workbenches.add(INTERNAL_WORKBENCHES[real_name])
628  else:
629  # Assume it's a Python requirement of some kind:
630  deps.python_requires.add(dep)
631 
632  for dep in self.blocks:
633  if dep in all_repos:
634  deps.blockers[dep] = all_repos[dep]

748  """Attempt to locate a line with an addWorkbench command in the workbench's
749  Python files. If it is directly instantiating a workbench, then we can use
