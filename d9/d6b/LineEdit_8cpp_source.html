43 bool LineEdit::eventFilter(QObject* object, QEvent* event)
44 {
45  Q_UNUSED(object);
46  if (event && event->type() == QEvent::KeyPress) {
47  QKeyEvent* keyEvent = static_cast<QKeyEvent*>(event);
48  if (keyEvent->key() == Qt::Key_Tab) {
49  // Special tab handling -- must be done via a QApplication event filter, otherwise the
50  // widget system will always grab the tab events
51  if (completerActive()) {
52  hideCompleter();
53  event->accept();
54  return true; // To make sure this tab press doesn't do anything else
55  }
56  else {
57  lastKeyPressed = keyEvent->key();
58  lastModifiers = keyEvent->modifiers();
59  }
60  }
61  }
62  return false; // We don't usually actually "handle" the tab event, we just keep track of it
63 }

65 bool LineEdit::event(QEvent* event)
66 {
67  if (event && event->type() == QEvent::FocusIn) {
68  qApp->installEventFilter(this);
69  }
70  else if (event && event->type() == QEvent::FocusOut) {
71  qApp->removeEventFilter(this);
72  if (lastKeyPressed) {
73  Q_EMIT finishedWithKey(lastKeyPressed, lastModifiers);
74  }
75  lastKeyPressed = 0;
76  }
77  else if (event && event->type() == QEvent::KeyPress && !completerActive()) {
78  QKeyEvent* kevent = static_cast<QKeyEvent*>(event);
79  lastKeyPressed = kevent->key();
80  lastModifiers = kevent->modifiers();
81  }
82  return Gui::ExpressionLineEdit::event(event);
83 }
