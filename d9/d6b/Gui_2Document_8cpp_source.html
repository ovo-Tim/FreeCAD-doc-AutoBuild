162  void throwIfNotInMap(App::DocumentObject* obj, App::Document* doc) const
163  {
164  if (_ViewProviderMap.find(obj) == _ViewProviderMap.end()) {
165  // We can actually support editing external object, by calling
166  // View3DInventViewer::setupEditingRoot() before exiting from
167  // ViewProvider::setEditViewer(), which transfer all child node of the view
168  // provider into an editing node inside the viewer of this document. And
169  // that's may actually be the case, as the subname referenced sub object
170  // is allowed to be in other documents.
171  //
172  // We just disabling editing external parent object here, for bug
173  // tracking purpose. Because, bringing an unrelated external object to
174  // the current view for editing will confuse user, and is certainly a
175  // bug. By right, the top parent object should always belong to the
176  // editing document, and the actually editing sub object can be
177  // external.
178  //
179  // So, you can either call setEdit() with subname set to 0, which cause
180  // the code above to auto detect selection context, and dispatch the
181  // editing call to the correct document. Or, supply subname yourself,
182  // and make sure you get the document right.
183  //
184  std::stringstream str;
185  str << "cannot edit object '"
186  << obj->getNameInDocument()
187  << "': not found in document "
188  << "'"
189  << doc->getName()
190  << "'";
191  throw Base::RuntimeError(str.str());

344  bool findParent()
345  {
346  auto result = findParentAndSubName(obj);
347  App::DocumentObject* parentObj = std::get<0>(result);
348 
349  if (parentObj) {
350  subname = std::get<1>(result);
351  obj = parentObj;
352  vp = findParentObject(parentObj, subname.c_str());
353  return true;
354  }
355 
356  return false;
357  }
358 
359 private:
360  static std::tuple<App::DocumentObject*, std::string>
361  findParentAndSubName(App::DocumentObject* obj)
362  {
363  // No subname reference is given, we try to extract one from the current
364  // selection in order to obtain the correct transformation matrix below
365  auto sels = Gui::Selection().getCompleteSelection(ResolveMode::NoResolve);
366  App::DocumentObject* parentObj = nullptr;
367  std::string _subname;
368  for (auto &sel : sels) {
369  if (!sel.pObject || !sel.pObject->isAttachedToDocument()) {
370  continue;
371  }
372  if (!parentObj) {
373  parentObj = sel.pObject;
374  }
375  else if (parentObj != sel.pObject) {
376  FC_LOG("Cannot deduce subname for editing, more than one parent?");
377  parentObj = nullptr;
378  break;
379  }
380 
381  auto sobj = parentObj->getSubObject(sel.SubName);
382  if (!sobj || (sobj != obj && sobj->getLinkedObject(true) != obj)) {
383  FC_LOG("Cannot deduce subname for editing, subname mismatch");
384  parentObj = nullptr;
385  break;
386  }
387 
388  _subname = sel.SubName;
389  }
390 
391  return std::make_tuple(parentObj, _subname);
392  }

614 {

616 
617  resetIfEditing();
618

620 
621  std::string _subname = subname ? subname : "";
622  if (_subname.empty()) {
623  ParentFinder finder(obj, vp, _subname);
624  if (finder.findParent()) {
625  _subname = finder.getSubname();
626  obj = finder.getObject();
627  vp = finder.getViewProvider();
628  if (vp->getDocument() != this) {
629  return vp->getDocument()->setEdit(vp, ModNum, _subname.c_str());
630  }
631  }
632  }
633 
634  d->throwIfNotInMap(obj, getDocument());
635 
636  Application::Instance->setEditDocument(this);
637 
638  if (!d->tryStartEditing(vp, obj, _subname.c_str(), ModNum)) {
639  return false;

645  d->setEditingViewerIfPossible(view3d, ModNum);
646  d->signalEditMode();
