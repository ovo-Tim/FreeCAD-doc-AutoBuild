46 int ChainIdSolver_Vereshchagin::CartToJnt(const JntArray &q, const JntArray &q_dot, JntArray &q_dotdot, const Jacobian& alfa, const JntArray& beta, const Wrenches& f_ext, JntArray &torques)
47 {
48  //Check sizes always
49  if (q.rows() != nj || q_dot.rows() != nj || q_dotdot.rows() != nj || torques.rows() != nj || f_ext.size() != ns)
50  return -1;
51  if (alfa.columns() != nc || beta.rows() != nc)
52  return -2;
53  //do an upward recursion for position and velocities
54  this->initial_upwards_sweep(q, q_dot, q_dotdot, f_ext);
55  //do an inward recursion for inertia, forces and constraints
56  this->downwards_sweep(alfa, torques);
57  //Solve for the constraint forces
58  this->constraint_calculation(beta);
59  //do an upward recursion to propagate the result
60  this->final_upwards_sweep(q_dotdot, torques);
61  return 0;
62 }

64 void ChainIdSolver_Vereshchagin::initial_upwards_sweep(const JntArray &q, const JntArray &qdot, const JntArray &/*qdotdot*/, const Wrenches& f_ext)
65 {
66  //if (q.rows() != nj || qdot.rows() != nj || qdotdot.rows() != nj || f_ext.size() != ns)
67  // return -1;
68 
69  unsigned int j = 0;
70  F_total = Frame::Identity();
71  for (unsigned int i = 0; i < ns; i++)
72  {
73  //Express everything in the segments reference frame (body coordinates)
74  //which is at the segments tip, i.e. where the next joint is attached.
75 
76  //Calculate segment properties: X,S,vj,cj
77  const Segment& segment = chain.getSegment(i);
78  segment_info& s = results[i + 1];
79  //The pose between the joint root and the segment tip (tip expressed in joint root coordinates)
80  s.F = segment.pose(q(j)); //X pose of each link in link coord system
81 
82  F_total = F_total * s.F; //X pose of the each link in root coord system
83  s.F_base = F_total; //X pose of the each link in root coord system for getter functions
84 
85  //The velocity due to the joint motion of the segment expressed in the segments reference frame (tip)
86  Twist vj = s.F.M.Inverse(segment.twist(q(j), qdot(j))); //XDot of each link
87  //Twist aj = s.F.M.Inverse(segment.twist(q(j), qdotdot(j))); //XDotDot of each link
88 
89  //The unit velocity due to the joint motion of the segment expressed in the segments reference frame (tip)
90  s.Z = s.F.M.Inverse(segment.twist(q(j), 1.0));
91  //Put Z in the joint root reference frame:
92  s.Z = s.F * s.Z;
93 
94  //The total velocity of the segment expressed in the segments reference frame (tip)
95  if (i != 0)
96  {
97  s.v = s.F.Inverse(results[i].v) + vj; // recursive velocity of each link in segment frame
98  //s.A=s.F.Inverse(results[i].A)+aj;
99  s.A = s.F.M.Inverse(results[i].A);
100  }
101  else
102  {
103  s.v = vj;
104  s.A = s.F.M.Inverse(acc_root);
105  }
106  //c[i] = cj + v[i]xvj (remark: cj=0, since our S is not time dependent in local coordinates)
107  //The velocity product acceleration
108  //std::cout << i << " Initial upward" << s.v << std::endl;
109  s.C = s.v*vj; //This is a cross product: cartesian space BIAS acceleration in local link coord.
110  //Put C in the joint root reference frame
111  s.C = s.F * s.C; //+F_total.M.Inverse(acc_root));
112  //The rigid body inertia of the segment, expressed in the segments reference frame (tip)
113  s.H = segment.getInertia();
114 
115  //wrench of the rigid body bias forces and the external forces on the segment (in body coordinates, tip)
116  //external forces are taken into account through s.U.
117  Wrench FextLocal = F_total.M.Inverse() * f_ext[i];
118  s.U = s.v * (s.H * s.v) - FextLocal; //f_ext[i];
119  if (segment.getJoint().getType() != Joint::None)
120  j++;
121  }

125 void ChainIdSolver_Vereshchagin::downwards_sweep(const Jacobian& alfa, const JntArray &torques)
126 {
127  int j = nj - 1;
128  for (int i = ns; i >= 0; i--)
129  {
130  //Get a handle for the segment we are working on.
131  segment_info& s = results[i];
132  //For segment N,
133  //tilde is in the segment refframe (tip, not joint root)
134  //without tilde is at the joint root (the childs tip!!!)
135  //P_tilde is the articulated body inertia
136  //R_tilde is the sum of external and coriolis/centrifugal forces
137  //M is the (unit) acceleration energy already generated at link i
138  //G is the (unit) magnitude of the constraint forces at link i
139  //E are the (unit) constraint forces due to the constraints
140  if (i == (int)ns)
141  {
142  s.P_tilde = s.H;
143  s.R_tilde = s.U;
144  s.M.setZero();
145  s.G.setZero();
146  //changeBase(alfa_N,F_total.M.Inverse(),alfa_N2);
147  for (unsigned int r = 0; r < 3; r++)
148  for (unsigned int c = 0; c < nc; c++)
149  {
150  //copy alfa constrain force matrix in E~
151  s.E_tilde(r, c) = alfa(r + 3, c);
152  s.E_tilde(r + 3, c) = alfa(r, c);
153  }
154  //Change the reference frame of alfa to the segmentN tip frame
155  //F_Total holds end effector frame, if done per segment bases then constraints could be extended to all segments
156  Rotation base_to_end = F_total.M.Inverse();
157  for (unsigned int c = 0; c < nc; c++)
158  {
159  Wrench col(Vector(s.E_tilde(3, c), s.E_tilde(4, c), s.E_tilde(5, c)),
160  Vector(s.E_tilde(0, c), s.E_tilde(1, c), s.E_tilde(2, c)));
161  col = base_to_end*col;
162  s.E_tilde.col(c) << Vector3d::Map(col.torque.data), Vector3d::Map(col.force.data);
163  }
164  }
165  else
166  {
167  //For all others:
168  //Everything should expressed in the body coordinates of segment i
169  segment_info& child = results[i + 1];
170  //Copy PZ into a vector so we can do matrix manipulations, put torques above forces
171  Vector6d vPZ;
172  vPZ << Vector3d::Map(child.PZ.torque.data), Vector3d::Map(child.PZ.force.data);
173  Matrix6d PZDPZt;
174  PZDPZt.noalias() = vPZ * vPZ.transpose();
175  PZDPZt /= child.D;
176 
177  //equation a) (see Vereshchagin89) PZDPZt=[I,H;H',M]
178  //Azamat:articulated body inertia as in Featherstone (7.19)
179  s.P_tilde = s.H + child.P - ArticulatedBodyInertia(PZDPZt.bottomRightCorner<3,3>(), PZDPZt.topRightCorner<3,3>(), PZDPZt.topLeftCorner<3,3>());
180  //equation b) (see Vereshchagin89)
181  //Azamat: bias force as in Featherstone (7.20)
182  s.R_tilde = s.U + child.R + child.PC + (child.PZ / child.D) * child.u;
183  //equation c) (see Vereshchagin89)
184  s.E_tilde = child.E;
185 
186  //Azamat: equation (c) right side term
187  s.E_tilde.noalias() -= (vPZ * child.EZ.transpose()) / child.D;
188 
189  //equation d) (see Vereshchagin89)
190  s.M = child.M;
191  //Azamat: equation (d) right side term
192  s.M.noalias() -= (child.EZ * child.EZ.transpose()) / child.D;
193 
194  //equation e) (see Vereshchagin89)
195  s.G = child.G;
196  Twist CiZDu = child.C + (child.Z / child.D) * child.u;
197  Vector6d vCiZDu;
198  vCiZDu << Vector3d::Map(CiZDu.rot.data), Vector3d::Map(CiZDu.vel.data);
199  s.G.noalias() += child.E.transpose() * vCiZDu;
200  }
201  if (i != 0)
202  {
203  //Transform all results to joint root coordinates of segment i (== body coordinates segment i-1)
204  //equation a)
205  s.P = s.F * s.P_tilde;
206  //equation b)
207  s.R = s.F * s.R_tilde;
208  //equation c), in matrix: torques above forces, so switch and switch back
209  for (unsigned int c = 0; c < nc; c++)
210  {
211  Wrench col(Vector(s.E_tilde(3, c), s.E_tilde(4, c), s.E_tilde(5, c)),
212  Vector(s.E_tilde(0, c), s.E_tilde(1, c), s.E_tilde(2, c)));
213  col = s.F*col;
214  s.E.col(c) << Vector3d::Map(col.torque.data), Vector3d::Map(col.force.data);
215  }
216 
217  //needed for next recursion
218  s.PZ = s.P * s.Z;
219  s.D = dot(s.Z, s.PZ);
220  s.PC = s.P * s.C;
221 
222  //u=(Q-Z(R+PC)=sum of external forces along the joint axes,
223  //R are the forces coming from the children,
224  //Q is taken zero (do we need to take the previous calculated torques?)
225 
226  //projection of coriolis and centrepital forces into joint subspace (0 0 Z)
227  s.totalBias = -dot(s.Z, s.R + s.PC);
228  s.u = torques(j) + s.totalBias;
229 
230  //Matrix form of Z, put rotations above translations
231  Vector6d vZ;
232  vZ << Vector3d::Map(s.Z.rot.data), Vector3d::Map(s.Z.vel.data);
233  s.EZ.noalias() = s.E.transpose() * vZ;
234 
235  if (chain.getSegment(i - 1).getJoint().getType() != Joint::None)
236  j--;
237  }
238  }

241 void ChainIdSolver_Vereshchagin::constraint_calculation(const JntArray& beta)
242 {
243  //equation f) nu = M_0_inverse*(beta_N - E0_tilde`*acc0 - G0)
244  //M_0_inverse, always nc*nc symmetric matrix
245  //std::cout<<"M0: "<<results[0].M<<std::endl;
246  //results[0].M-=MatrixXd::Identity(nc,nc);
247  //std::cout<<"augmented M0: "<<results[0].M<<std::endl;
248 
249 
250  //ToDo: Need to check ill conditions
251 
252  //M_0_inverse=results[0].M.inverse();
253  svd_eigen_HH(results[0].M, Um, Sm, Vm, tmpm);
254  //truncated svd, what would sdls, dls physically mean?
255  for (unsigned int i = 0; i < nc; i++)
256  if (Sm(i) < 1e-14)
257  Sm(i) = 0.0;
258  else
259  Sm(i) = 1 / Sm(i);
260 
261  results[0].M.noalias() = Vm * Sm.asDiagonal();
262  M_0_inverse.noalias() = results[0].M * Um.transpose();
263  //results[0].M.ldlt().solve(MatrixXd::Identity(nc,nc),&M_0_inverse);
264  //results[0].M.computeInverse(&M_0_inverse);
265  Vector6d acc;
266  acc << Vector3d::Map(acc_root.rot.data), Vector3d::Map(acc_root.vel.data);
267  nu_sum.noalias() = -(results[0].E_tilde.transpose() * acc);
268  //nu_sum.setZero();
269  nu_sum += beta.data;
270  nu_sum -= results[0].G;
271 
272  //equation f) nu = M_0_inverse*(beta_N - E0_tilde`*acc0 - G0)
273  nu.noalias() = M_0_inverse * nu_sum;

276 void ChainIdSolver_Vereshchagin::final_upwards_sweep(JntArray &q_dotdot, JntArray &torques)
277 {
278  unsigned int j = 0;
279 
280  for (unsigned int i = 1; i <= ns; i++)
281  {
282  segment_info& s = results[i];
283  //Calculation of joint and segment accelerations
284  //equation g) qdotdot[i] = D^-1*(Q - Z'(R + P(C + acc[i-1]) + E*nu))
285  // = D^-1(u - Z'(P*acc[i-1] + E*nu)
286  Twist a_g;
287  Twist a_p;
288  if (i == 1)
289  {
290  a_p = acc_root;
291  }
292  else
293  {
294  a_p = results[i - 1].acc;
295  }
296 
297  //The contribution of the constraint forces at segment i
298  Vector6d tmp = s.E*nu;
299  Wrench constraint_force = Wrench(Vector(tmp(3), tmp(4), tmp(5)),
300  Vector(tmp(0), tmp(1), tmp(2)));
301 
302  //acceleration components are also computed
303  //Contribution of the acceleration of the parent (i-1)
304  Wrench parent_force = s.P*a_p;
305  double parent_forceProjection = -dot(s.Z, parent_force);
306  double parentAccComp = parent_forceProjection / s.D;
307 
308  //The constraint force and acceleration force projected on the joint axes -> axis torque/force
309  double constraint_torque = -dot(s.Z, constraint_force);
310  //The result should be the torque at this joint
311 
312  torques(j) = constraint_torque;
313  //s.constAccComp = torques(j) / s.D;
314  s.constAccComp = constraint_torque / s.D;
315  s.nullspaceAccComp = s.u / s.D;
316  //total joint space acceleration resulting from accelerations of parent joints, constraint forces and
317  // nullspace forces.
318  q_dotdot(j) = (s.nullspaceAccComp + parentAccComp + s.constAccComp);
319  s.acc = s.F.Inverse(a_p + s.Z * q_dotdot(j) + s.C);//returns acceleration in link distal tip coordinates. For use needs to be transformed
320  if (chain.getSegment(i - 1).getJoint().getType() != Joint::None)
321  j++;
322  }
