63 struct Vertex_Less
64 {
65  explicit Vertex_Less(double tolerance)
66  : tolerance(tolerance)
67  {}
68  bool operator()(const VertexIds& x, const VertexIds& y) const
69  {
70  if (fabs(x.v.x - y.v.x) > tolerance) {
71  return x.v.x < y.v.x;
72  }
73  if (fabs(x.v.y - y.v.y) > tolerance) {
74  return x.v.y < y.v.y;
75  }
76  if (fabs(x.v.z - y.v.z) > tolerance) {
77  return x.v.z < y.v.z;
78  }
79  return false; // points are considered to be equal
80  }
81 
82 private:
83  double tolerance;
84 };

254  std::list<ConstraintIds> getMissingCoincidences(std::vector<Sketcher::Constraint*>& allcoincid,
255  double precision)
256  {
257  std::list<ConstraintIds> missingCoincidences; // Holds the list of missing coincidences
258 
259  // Sort points in geographic order
260  std::sort(vertexIds.begin(), vertexIds.end(), Vertex_Less(precision));
261 
262  auto vt = vertexIds.begin();
263  Vertex_EqualTo pred(precision);
264 
265  // Comparing existing constraints and find missing ones
266 
267  while (vt < vertexIds.end()) {
268  // Seeking for adjacent group of vertices
269  vt = std::adjacent_find(vt, vertexIds.end(), pred);
270  if (vt < vertexIds.end()) { // If one found
271  std::vector<VertexIds>::iterator vn;
272  // Holds a single group of adjacent vertices
273  std::set<VertexIds, VertexID_Less> vertexGrp;
274  // Extract the group of adjacent vertices
275  vertexGrp.insert(*vt);
276  for (vn = vt + 1; vn < vertexIds.end(); ++vn) {
277  if (pred(*vt, *vn)) {
278  vertexGrp.insert(*vn);
279  }
280  else {
281  break;
282  }
283  }
284 
285  // Holds groups of coincident vertices
286  std::vector<std::set<VertexIds, VertexID_Less>> coincVertexGrps;
287 
288  // Decompose the group of adjacent vertices into groups of coincident vertices
289  // Going through existent coincidences
290  for (auto& coincidence : allcoincid) {
291  VertexIds v1;
292  VertexIds v2;
293  v1.GeoId = coincidence->First;
294  v1.PosId = coincidence->FirstPos;
295  v2.GeoId = coincidence->Second;
296  v2.PosId = coincidence->SecondPos;
297 
298  // Look if coincident vertices are in the group of adjacent ones we are
299  // processing
300  auto nv1 = vertexGrp.extract(v1);
301  auto nv2 = vertexGrp.extract(v2);
302 
303  // Maybe if both empty, they already have been extracted by other coincidences
304  // We have to check in existing coincident groups and eventually merge
305  if (nv1.empty() && nv2.empty()) {
306  std::set<VertexIds, VertexID_Less>* tempGrp = nullptr;
307  for (auto it = coincVertexGrps.begin(); it < coincVertexGrps.end(); ++it) {
308  if ((it->find(v1) != it->end()) || (it->find(v2) != it->end())) {
309  if (!tempGrp) {
310  tempGrp = &*it;
311  }
312  else {
313  tempGrp->insert(it->begin(), it->end());
314  coincVertexGrps.erase(it);
315  break;
316  }
317  }
318  }
319  continue;
320  }
321 
322  // Look if one of the constrained vertices is already in a group of coincident
323  // vertices
324  for (std::set<VertexIds, VertexID_Less>& grp : coincVertexGrps) {
325  if ((grp.find(v1) != grp.end()) || (grp.find(v2) != grp.end())) {
326  // If yes add them to the existing group
327  if (!nv1.empty()) {
328  grp.insert(nv1.value());
329  }
330  if (!nv2.empty()) {
331  grp.insert(nv2.value());
332  }
333  continue;
334  }
335  }
336 
337  if (nv1.empty() || nv2.empty()) {
338  continue;
339  }
340 
341  // If no, create a new group of coincident vertices
342  std::set<VertexIds, VertexID_Less> newGrp;
343  newGrp.insert(nv1.value());
344  newGrp.insert(nv2.value());
345  coincVertexGrps.push_back(newGrp);
346  }
347 
348  // If there are remaining vertices in the adjacent group (not in any existing
349  // constraint) add them as being each a separate coincident group
350  for (auto& lonept : vertexGrp) {
351  std::set<VertexIds, VertexID_Less> newGrp;
352  newGrp.insert(lonept);
353  coincVertexGrps.push_back(newGrp);
354  }
355 
356  // If there is more than 1 coincident group into adjacent group, constraint(s)
357  // is(are) missing Virtually generate the missing constraint(s)
358  if (coincVertexGrps.size() > 1) {
359  std::vector<std::set<VertexIds, VertexID_Less>>::iterator vn;
360  // Starting from the 2nd coincident group, generate a constraint between
361  // this group first vertex, and previous group first vertex
362  for (vn = coincVertexGrps.begin() + 1; vn < coincVertexGrps.end(); ++vn) {
363  ConstraintIds id;
364  id.Type = Coincident; // default point on point restriction
365  id.v = (vn - 1)->begin()->v;
366  id.First = (vn - 1)->begin()->GeoId;
367  id.FirstPos = (vn - 1)->begin()->PosId;
368  id.Second = vn->begin()->GeoId;
369  id.SecondPos = vn->begin()->PosId;
370  missingCoincidences.push_back(id);
371  }
372  }
373 
374  vt = vn;
375  }
376  }
377 
378  return missingCoincidences;
379  }
380 
381 private:
382  // Holds a list of all vertices in the sketch
383  std::vector<VertexIds> vertexIds;
384 };

504 int SketchAnalysis::detectMissingPointOnPointConstraints(double precision,
505  bool includeconstruction /*=true*/)
506 {
507  PointConstraints pointConstr;
508 
509  // Build the list of sketch vertices
510  const std::vector<Part::Geometry*>& geom = sketch->getInternalGeometry();
511  for (std::size_t i = 0; i < geom.size(); i++) {
512  auto gf = GeometryFacade::getFacade(geom[i]);
513 
514  if (gf->getConstruction() && !includeconstruction) {
515  continue;
516  }
517 
518  pointConstr.addGeometry(gf->getGeometry(), int(i));
519  // TODO take into account single vertices ?
520  }
521 
522  // Build a list of all coincidences in the sketch
523 
524  std::vector<Sketcher::Constraint*> coincidences = sketch->Constraints.getValues();
525  for (auto& constraint : sketch->Constraints.getValues()) {
526  // clang-format off
527  if (constraint->Type == Sketcher::Coincident ||
528  constraint->Type == Sketcher::Tangent ||
529  constraint->Type == Sketcher::Perpendicular) {
530  coincidences.push_back(constraint);
531  }
532  // clang-format on
533  // TODO optimizing by removing constraints not applying on vertices ?
534  }
535 
536  // Holds the list of missing coincidences
537  std::list<ConstraintIds> missingCoincidences =
538  pointConstr.getMissingCoincidences(coincidences, precision);
539 
540  // Update list of missing constraints stored as member variable of sketch
541  this->vertexConstraints.clear();
542  this->vertexConstraints.reserve(missingCoincidences.size());
543 
544  for (auto& coincidence : missingCoincidences) {
545  this->vertexConstraints.push_back(coincidence);
546  }
547 
548  // Return number of missing constraints
549  return int(this->vertexConstraints.size());
550 }

552 void SketchAnalysis::analyseMissingPointOnPointCoincident(double angleprecision)
553 {
554  for (auto& vc : vertexConstraints) {
555 
556  auto geo1 = sketch->getGeometry(vc.First);
557  auto geo2 = sketch->getGeometry(vc.Second);
558 
559  // tangency point-on-point
560  const auto* curve1 = dynamic_cast<const Part::GeomCurve*>(geo1);
561  const auto* curve2 = dynamic_cast<const Part::GeomCurve*>(geo2);
562 
563  if (curve1 && curve2) {
564 
565  const auto* segm1 = dynamic_cast<const Part::GeomLineSegment*>(geo1);
566  const auto* segm2 = dynamic_cast<const Part::GeomLineSegment*>(geo2);
567 
568  if (segm1 && segm2) {
569 
570  Base::Vector3d dir1 = segm1->getEndPoint() - segm1->getStartPoint();
571  Base::Vector3d dir2 = segm2->getEndPoint() - segm2->getStartPoint();
572 
573  if ((checkVertical(dir1, angleprecision) || checkHorizontal(dir1, angleprecision))
574  && (checkVertical(dir2, angleprecision)
575  || checkHorizontal(dir2, angleprecision))) {
576  // this is a job for horizontal/vertical constraints alone
577  continue;
578  }
579  }
580 
581  try {
582  double u1 {};
583  double u2 {};
584 
585  curve1->closestParameter(vc.v, u1);
586  curve2->closestParameter(vc.v, u2);
587 
588  Base::Vector3d tgv1 = curve1->firstDerivativeAtParameter(u1).Normalize();
589  Base::Vector3d tgv2 = curve2->firstDerivativeAtParameter(u2).Normalize();
590 
591  if (fabs(tgv1 * tgv2) > fabs(cos(angleprecision))) {
592  vc.Type = Sketcher::Tangent;
593  }
594  else if (fabs(tgv1 * tgv2) < fabs(cos(M_PI / 2 - angleprecision))) {
595  vc.Type = Sketcher::Perpendicular;
596  }
597  }
598  catch (Base::Exception&) {
599  Base::Console().Warning("Point-On-Point Coincidence analysis: unable to obtain "
600  "derivative. Detection ignored.\n");
601  continue;
602  }
603  }
604  }
605 }

737 int SketchAnalysis::detectMissingEqualityConstraints(double precision)
738 {
739  EqualityConstraints equalConstr;
740 
741  const std::vector<Part::Geometry*>& geom = sketch->getInternalGeometry();
742  for (std::size_t i = 0; i < geom.size(); i++) {
743  Part::Geometry* g = geom[i];
744  equalConstr.addGeometry(g, int(i));
745  }
746 
747  std::list<ConstraintIds> equallines = equalConstr.getEqualLines(precision);
748  std::list<ConstraintIds> equalradius = equalConstr.getEqualRadius(precision);
749 
750  // Go through the available 'Coincident', 'Tangent' or 'Perpendicular' constraints
751  // and check which of them is forcing two vertexes to be coincident.
752  // If there is none but two vertexes can be considered equal a coincident constraint is missing.
753  std::vector<Sketcher::Constraint*> constraint = sketch->Constraints.getValues();
754  for (auto it : constraint) {
755  if (it->Type == Sketcher::Equal) {
756  ConstraintIds id {Base::Vector3d {},
757  it->First,
758  it->Second,
759  it->FirstPos,
760  it->SecondPos,
761  it->Type};
762 
763  auto pos = std::find_if(equallines.begin(), equallines.end(), Constraint_Equal(id));
764 
765  if (pos != equallines.end()) {
766  equallines.erase(pos);
767  }
768 
769  pos = std::find_if(equalradius.begin(), equalradius.end(), Constraint_Equal(id));
770 
771  if (pos != equalradius.end()) {
772  equalradius.erase(pos);
773  }
774  }
775  }
776 
777  this->lineequalityConstraints.clear();
778  this->lineequalityConstraints.reserve(equallines.size());
779 
780  for (const auto& it : equallines) {
781  this->lineequalityConstraints.push_back(it);
782  }
783 
784  this->radiusequalityConstraints.clear();
785  this->radiusequalityConstraints.reserve(equalradius.size());
786 
787  for (const auto& it : equalradius) {
788  this->radiusequalityConstraints.push_back(it);
789  }
790 
791  return int(this->lineequalityConstraints.size() + this->radiusequalityConstraints.size());
792 }

909 int SketchAnalysis::autoconstraint(double precision,
910  double angleprecision,
911  bool includeconstruction)
912 {
913  autoDeleteAllConstraints();
914 
915  // STAGE 1: Vertical/Horizontal Line Segments
916  int nhv = detectMissingVerticalHorizontalConstraints(angleprecision);
917 
918  // STAGE 2: Point-on-Point constraint (Coincidents, endpoint perp, endpoint tangency)
919  // Note: We do not apply the vertical/horizontal constraints before calculating the pointonpoint
920  // constraints
921  // as the solver may move the geometry in the meantime and prevent correct detection
922  int nc = detectMissingPointOnPointConstraints(precision, includeconstruction);
923 
924  if (nc > 0) { // STAGE 2a: Classify point-on-point into coincidents, endpoint perp, endpoint
925  // tangency
926  analyseMissingPointOnPointCoincident(angleprecision);
927  }
928 
929  // STAGE 3: Equality constraint detection
930  int ne = detectMissingEqualityConstraints(precision);
931 
932  Base::Console().Log("Constraints: Vertical/Horizontal: %d found. "
933  "Point-on-point: %d. Equality: %d\n",
934  nhv,
935  nc,
936  ne);
937 
938  // Applying STAGE 1, if any
939  if (nhv > 0) {
940  autoHorizontalVerticalConstraints();
941  }
942 
943  // Applying STAGE 2
944  if (nc > 0) {
945  autoPointOnPointCoincident();
946  }
947 
948  // Applying STAGE 3
949  if (ne > 0) {
950  autoMissingEquality();
951  }
952 
953  return 0;
954 }
