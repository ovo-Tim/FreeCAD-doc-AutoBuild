81 App::DocumentObjectExecReturn* Revolution::execute()
82 {
83  if (onlyHasToRefine()){
84  TopoShape result = refineShapeIfActive(rawShape);
85  Shape.setValue(result);
86  return App::DocumentObject::StdReturn;
87  }
88 
89  // Validate parameters
90  // All angles are in radians unless explicitly stated
91  double angleDeg = Angle.getValue();
92  if (angleDeg > 360.0) {
93  return new App::DocumentObjectExecReturn(
94  QT_TRANSLATE_NOOP("Exception", "Angle of revolution too large"));
95  }
96 
97  double angle = Base::toRadians<double>(angleDeg);
98  if (angle < Precision::Angular()) {
99  return new App::DocumentObjectExecReturn(
100  QT_TRANSLATE_NOOP("Exception", "Angle of revolution too small"));
101  }
102 
103  double angle2 = Base::toRadians(Angle2.getValue());
104 
105  TopoShape sketchshape = getTopoShapeVerifiedFace();
106 
107  // if the Base property has a valid shape, fuse the AddShape into it
108  TopoShape base;
109  try {
110  base = getBaseTopoShape();
111  }
112  catch (const Base::Exception&) {
113  // fall back to support (for legacy features)
114  }
115 
116  // update Axis from ReferenceAxis
117  try {
118  updateAxis();
119  }
120  catch (const Base::Exception& e) {
121  return new App::DocumentObjectExecReturn(e.what());
122  }
123 
124  try {
125  // get revolve axis
126  Base::Vector3d b = Base.getValue();
127  gp_Pnt pnt(b.x, b.y, b.z);
128  Base::Vector3d v = Axis.getValue();
129 
130  if (v.IsNull()) {
131  return new App::DocumentObjectExecReturn(
132  QT_TRANSLATE_NOOP("Exception", "Reference axis is invalid"));
133  }
134 
135  gp_Dir dir(v.x, v.y, v.z);
136 
137  if (sketchshape.isNull()) {
138  return new App::DocumentObjectExecReturn(
139  QT_TRANSLATE_NOOP("Exception", "Creating a face from sketch failed"));
140  }
141 
142  RevolMethod method = methodFromString(Type.getValueAsString());
143 
144  this->positionByPrevious();
145  TopLoc_Location invObjLoc = this->getLocation().Inverted();
146  pnt.Transform(invObjLoc.Transformation());
147  dir.Transform(invObjLoc.Transformation());
148  base.move(invObjLoc);
149  sketchshape.move(invObjLoc);
150 
151  // Check distance between sketchshape and axis - to avoid failures and crashes
152  TopExp_Explorer xp;
153  xp.Init(sketchshape.getShape(), TopAbs_FACE);
154  for (; xp.More(); xp.Next()) {
155  if (checkLineCrossesFace(gp_Lin(pnt, dir), TopoDS::Face(xp.Current()))) {
156  return new App::DocumentObjectExecReturn(
157  QT_TRANSLATE_NOOP("Exception", "Revolve axis intersects the sketch"));
158  }
159  }
160 
161  // Create a fresh support even when base exists so that it can be used for patterns
162  TopoShape result(0);
163  TopoShape supportface(0);
164  try {
165  supportface = getSupportFace();
166  }
167  catch(...) {
168  //do nothing, null shape is handle below
169  }
170 
171  supportface.move(invObjLoc);
172 
173  if (method == RevolMethod::ToFace || method == RevolMethod::ToFirst
174  || method == RevolMethod::ToLast) {
175  TopoShape upToFace;
176  if (method == RevolMethod::ToFace) {
177  getUpToFaceFromLinkSub(upToFace, UpToFace);
178  upToFace.move(invObjLoc);
179  }
180  else {
181  throw Base::RuntimeError(
182  "ProfileBased: Revolution up to first/last is not yet supported");
183  }
184 
185  if (Reversed.getValue()) {
186  dir.Reverse();
187  }
188 
189  TopExp_Explorer Ex(supportface.getShape(), TopAbs_WIRE);
190  if (!Ex.More()) {
191  supportface = TopoDS_Face();
192  }
193 
194  try {
195  result = base.makeElementRevolution(base,
196  TopoDS::Face(sketchshape.getShape()),
197  gp_Ax1(pnt, dir),
198  TopoDS::Face(supportface.getShape()),
199  TopoDS::Face(upToFace.getShape()),
200  nullptr,
201  Part::RevolMode::None,
202  Standard_True);
203  }
204  catch (Standard_Failure&) {
205  return new App::DocumentObjectExecReturn("Could not revolve the sketch!");
206  }
207  }
208  else {
209  bool midplane = Midplane.getValue();
210  bool reversed = Reversed.getValue();
211  generateRevolution(result,
212  sketchshape.getShape(),
213  gp_Ax1(pnt, dir),
214  angle,
215  angle2,
216  midplane,
217  reversed,
218  method);
219  }
220 
221  if (!result.isNull()) {
222  // store shape before refinement
223  this->rawShape = result;
224  result = refineShapeIfActive(result);
225  // set the additive shape property for later usage in e.g. pattern
226  this->AddSubShape.setValue(result);
227 
228  if (!base.isNull()) {
229  result = result.makeElementFuse(base);
230  // store shape before refinement
231  this->rawShape = result;
232  result = refineShapeIfActive(result);
233  }
234  this->Shape.setValue(getSolid(result));
235  }
236  else {
237  return new App::DocumentObjectExecReturn(
238  QT_TRANSLATE_NOOP("Exception", "Could not revolve the sketch!"));
239  }
240 
241  // eventually disable some settings that are not valid for the current method
242  updateProperties(method);
243 
244  return App::DocumentObject::StdReturn;
245  }
246  catch (Standard_Failure& e) {
247 
248  if (std::string(e.GetMessageString()) == "TopoDS::Face") {
249  return new App::DocumentObjectExecReturn(
250  QT_TRANSLATE_NOOP("Exception",
251  "Could not create face from sketch.\n"
252  "Intersecting sketch entities in a sketch are not allowed."));
253  }
254  else {
255  return new App::DocumentObjectExecReturn(e.GetMessageString());
256  }
257  }
258  catch (Base::Exception& e) {
259  return new App::DocumentObjectExecReturn(e.what());
260  }
261 }

286 Revolution::RevolMethod Revolution::methodFromString(const std::string& methodStr)
287 {
288  if (methodStr == "Angle")
289  return RevolMethod::Dimension;
290  if (methodStr == "UpToLast")
291  return RevolMethod::ToLast;
292  if (methodStr == "ThroughAll")
293  return RevolMethod::ThroughAll;
294  if (methodStr == "UpToFirst")
295  return RevolMethod::ToFirst;
296  if (methodStr == "UpToFace")
297  return RevolMethod::ToFace;
298  if (methodStr == "TwoAngles")
299  return RevolMethod::TwoDimensions;
300 
301  throw Base::ValueError("Revolution:: No such method");
302  return RevolMethod::Dimension;
303 }

305 void Revolution::generateRevolution(TopoShape& revol,
306  const TopoShape& sketchshape,
307  const gp_Ax1& axis,
308  const double angle,
309  const double angle2,
310  const bool midplane,
311  const bool reversed,
312  RevolMethod method)
313 {
314  if (method == RevolMethod::Dimension || method == RevolMethod::TwoDimensions || method == RevolMethod::ThroughAll) {
315  double angleTotal = angle;
316  double angleOffset = 0.;
317 
318  if (method == RevolMethod::TwoDimensions) {
319  // Rotate the face by `angle2`/`angle` to get "second" angle
320  angleTotal += angle2;
321  angleOffset = angle2 * -1.0;
322  }
323  else if (midplane) {
324  // Rotate the face by half the angle to get Revolution symmetric to sketch plane
325  angleOffset = -angle / 2;
326  }
327 
328  if (fabs(angleTotal) < Precision::Angular())
329  throw Base::ValueError("Cannot create a revolution with zero angle.");
330 
331  gp_Ax1 revolAx(axis);
332  if (reversed) {
333  revolAx.Reverse();
334  }
335 
336  TopoShape from = sketchshape;
337  if (method == RevolMethod::TwoDimensions || midplane) {
338  gp_Trsf mov;
339  mov.SetRotation(revolAx, angleOffset);
340  TopLoc_Location loc(mov);
341  from.move(loc);
342  }
343 
344  revol = from;
345  revol = revol.makeElementRevolve(revolAx,angleTotal);
346  revol.Tag = -getID();
347  } else {
348  std::stringstream str;
349  str << "ProfileBased: Internal error: Unknown method for generateRevolution()";
350  throw Base::RuntimeError(str.str());
351  }
352 }

375 void Revolution::updateProperties(RevolMethod method)
376 {
377  // disable settings that are not valid on the current method
378  // disable everything unless we are sure we need it
379  bool isAngleEnabled = false;
380  bool isAngle2Enabled = false;
381  bool isMidplaneEnabled = false;
382  bool isReversedEnabled = false;
383  bool isUpToFaceEnabled = false;
384  if (method == RevolMethod::Dimension) {
385  isAngleEnabled = true;
386  isMidplaneEnabled = true;
387  isReversedEnabled = !Midplane.getValue();
388  }
389  else if (method == RevolMethod::ToLast) {
390  isReversedEnabled = true;
391  }
392  else if (method == RevolMethod::ThroughAll) {
393  isMidplaneEnabled = true;
394  isReversedEnabled = !Midplane.getValue();
395  }
396  else if (method == RevolMethod::ToFirst) {
397  isReversedEnabled = true;
398  }
399  else if (method == RevolMethod::ToFace) {
400  isReversedEnabled = true;
401  isUpToFaceEnabled = true;
402  }
403  else if (method == RevolMethod::TwoDimensions) {
404  isAngleEnabled = true;
405  isAngle2Enabled = true;
406  isReversedEnabled = true;
407  }
408 
409  Angle.setReadOnly(!isAngleEnabled);
410  Angle2.setReadOnly(!isAngle2Enabled);
411  Midplane.setReadOnly(!isMidplaneEnabled);
412  Reversed.setReadOnly(!isReversedEnabled);
413  UpToFace.setReadOnly(!isUpToFaceEnabled);
414 }
