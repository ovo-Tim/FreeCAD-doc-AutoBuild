37 DrawTile::DrawTile()
38 {
39  static const char *group = "Tile";
40 
41  ADD_PROPERTY_TYPE(TileParent, (nullptr), group, (App::PropertyType)(App::Prop_None),
42  "Object to which this tile is attached");
43  ADD_PROPERTY_TYPE(TileRow, (0), group, App::Prop_None, "Row in parent object\n 0 for arrow side, -1 for other side");
44  ADD_PROPERTY_TYPE(TileColumn, (0), group, App::Prop_None, "Column in parent object");
45 
46  // there is currently only one column, this don't allow one to edit
47  TileColumn.setStatus(App::Property::ReadOnly, true);
48  // the row can only have the value 0 or -1
49  // allow its editing because this way the tiles can be flipped
50  TileRowConstraints.LowerBound = -1;
51  TileRowConstraints.UpperBound = 0;
52  TileRowConstraints.StepSize = 1;
53  TileRow.setConstraints(&TileRowConstraints);
54 }
