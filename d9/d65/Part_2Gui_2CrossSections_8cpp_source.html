211 void CrossSections::apply()
212 {
213  std::vector<App::DocumentObject*> docobjs = Gui::Selection().
214  getObjectsOfType(App::DocumentObject::getClassTypeId());
215  std::vector<App::DocumentObject*> obj;
216  for (auto it : docobjs) {
217  if (!Part::Feature::getTopoShape(it).isNull()) {
218  obj.push_back(it);
219  }
220  }
221 
222  std::vector<double> d;
223  if (ui->sectionsBox->isChecked())
224  d = getPlanes();
225  else
226  d.push_back(ui->position->value().getValue());
227  double a=0,b=0,c=0;
228  switch (plane()) {
229  case CrossSections::XY:
230  c = 1.0;
231  break;
232  case CrossSections::XZ:
233  b = 1.0;
234  break;
235  case CrossSections::YZ:
236  a = 1.0;
237  break;
238  }
239 
240 #ifdef CS_FUTURE
241  Standard::SetReentrant(Standard_True);
242  for (std::vector<App::DocumentObject*>::iterator it = obj.begin(); it != obj.end(); ++it) {
243  Part::CrossSection cs(a,b,c,static_cast<Part::Feature*>(*it)->Shape.getValue());
244  QFuture< std::list<TopoDS_Wire> > future = QtConcurrent::mapped
245  (d, std::bind(&Part::CrossSection::section, &cs, sp::_1));
246  future.waitForFinished();
247  QFuture< std::list<TopoDS_Wire> >::const_iterator ft;
248  TopoDS_Compound comp;
249  BRep_Builder builder;
250  builder.MakeCompound(comp);
251 
252  for (ft = future.begin(); ft != future.end(); ++ft) {
253  const std::list<TopoDS_Wire>& w = *ft;
254  for (std::list<TopoDS_Wire>::const_iterator wt = w.begin(); wt != w.end(); ++wt) {
255  if (!wt->IsNull())
256  builder.Add(comp, *wt);
257  }
258  }
259 
260  App::Document* doc = (*it)->getDocument();
261  std::string s = (*it)->getNameInDocument();
262  s += "_cs";
263  Part::Feature* section = static_cast<Part::Feature*>
264  (doc->addObject("Part::Feature",s.c_str()));
265  section->Shape.setValue(comp);
266  section->purgeTouched();
267  }
268 #else
269  Base::SequencerLauncher seq("Cross-sections...", obj.size() * (d.size() +1));
270  Gui::Command::runCommand(Gui::Command::App, "import Part\n");
271  Gui::Command::runCommand(Gui::Command::App, "from FreeCAD import Base\n");
272  for (auto it : obj) {
273  App::Document* doc = it->getDocument();
274  std::string s = it->getNameInDocument();
275  s += "_cs";
276  Gui::Command::runCommand(Gui::Command::App, QString::fromLatin1(
277  "wires=list()\n"
278  "shape=FreeCAD.getDocument(\"%1\").%2.Shape\n")
279  .arg(QLatin1String(doc->getName()),
280  QLatin1String(it->getNameInDocument())).toLatin1());
281 
282  for (double jt : d) {
283  Gui::Command::runCommand(Gui::Command::App, QString::fromLatin1(
284  "for i in shape.slice(Base.Vector(%1,%2,%3),%4):\n"
285  " wires.append(i)\n"
286  ).arg(a).arg(b).arg(c).arg(jt).toLatin1());
287  seq.next();
288  }
289 
290  Gui::Command::runCommand(Gui::Command::App, QString::fromLatin1(
291  "comp=Part.Compound(wires)\n"
292  "slice=FreeCAD.getDocument(\"%1\").addObject(\"Part::Feature\",\"%2\")\n"
293  "slice.Shape=comp\n"
294  "slice.purgeTouched()\n"
295  "del slice,comp,wires,shape")
296  .arg(QLatin1String(doc->getName()),
297  QLatin1String(s.c_str())).toLatin1());
298 
299  seq.next();
300  }
301 #endif
302 }

362 void CrossSections::sectionsBoxToggled(bool b)

422 void CrossSections::calcPlane(Plane type, double pos)
423 {
424  double bound[4];
425  switch (type) {
426  case XY:
427  bound[0] = bbox.MinX;
428  bound[1] = bbox.MaxX;
429  bound[2] = bbox.MinY;

431  break;
432  case XZ:
433  bound[0] = bbox.MinX;
434  bound[1] = bbox.MaxX;

439  bound[0] = bbox.MinY;
440  bound[1] = bbox.MaxY;

442  bound[3] = bbox.MaxZ;
443  break;
444  }
445 
446  std::vector<double> d;
447  d.push_back(pos);
448  makePlanes(type, d, bound);

479 std::vector<double> CrossSections::getPlanes() const
480 {
481  int count = ui->countSections->value();
482  double pos = ui->position->value().getValue();
483  double stp = ui->distance->value().getValue();
484  bool both = ui->checkBothSides->isChecked();
485 
486  std::vector<double> d;
487  if (both) {
488  double start = pos-0.5f*(count-1)*stp;
489  for (int i=0; i<count; i++) {
490  d.push_back(start+i*stp);
491  }
492  }
493  else {
494  for (int i=0; i<count; i++) {
495  d.push_back(pos+i*stp);
496  }
497  }
498  return d;
499 }

501 void CrossSections::makePlanes(Plane type, const std::vector<double>& d, double bound[4])
502 {
503  std::vector<Base::Vector3f> points;
504  for (double it : d) {
505  Base::Vector3f v[4];
506  switch (type) {
507  case XY:
508  v[0].Set(bound[0],bound[2],it);
509  v[1].Set(bound[1],bound[2],it);
510  v[2].Set(bound[1],bound[3],it);
511  v[3].Set(bound[0],bound[3],it);
512  break;
513  case XZ:
514  v[0].Set(bound[0],it,bound[2]);
515  v[1].Set(bound[1],it,bound[2]);
516  v[2].Set(bound[1],it,bound[3]);
517  v[3].Set(bound[0],it,bound[3]);
518  break;
519  case YZ:
520  v[0].Set(it,bound[0],bound[2]);
521  v[1].Set(it,bound[1],bound[2]);
522  v[2].Set(it,bound[1],bound[3]);
523  v[3].Set(it,bound[0],bound[3]);
524  break;
525  }
526 
527  points.push_back(v[0]);
528  points.push_back(v[1]);
529  points.push_back(v[2]);
530  points.push_back(v[3]);
531  points.push_back(v[0]);
532  }
533  vp->setCoords(points);
534 }
