40 def make_clone(obj, delta=None, forcedraft=False):
41  """clone(obj,[delta,forcedraft])
42 
43  Makes a clone of the given object(s).
44  The clone is an exact, linked copy of the given object. If the original
45  object changes, the final object changes too.
46 
47  Parameters
48  ----------
49  obj :
50 
51  delta : Base.Vector
52  Delta Vector to move the clone from the original position.
53 
54  forcedraft : bool
55  If forcedraft is True, the resulting object is a Draft clone
56  even if the input object is an Arch object.
57 
58  """
59 
60  prefix = params.get_param("ClonePrefix")
61 
62  cl = None
63 
64  if prefix:
65  prefix = prefix.strip() + " "
66 
67  if not isinstance(obj,list):
68  obj = [obj]
69 
70  if (len(obj) == 1) and obj[0].isDerivedFrom("Part::Part2DObject"):
71  cl = App.ActiveDocument.addObject("Part::Part2DObjectPython","Clone2D")
72  cl.Label = prefix + obj[0].Label + " (2D)"
73  elif (len(obj) == 1) and (hasattr(obj[0],"CloneOf") or (utils.get_type(obj[0]) == "BuildingPart")) and (not forcedraft):
74  # arch objects can be clones
75  try:
76  import Arch
77  except:
78  # BIM not present
79  pass
80  else:
81  if utils.get_type(obj[0]) == "BuildingPart":
82  cl = Arch.makeComponent()
83  else:
84  try: # new-style make function
85  cl = getattr(Arch, "make_" + obj[0].Proxy.Type.lower())()
86  except Exception:
87  try: # old-style make function
88  cl = getattr(Arch, "make" + obj[0].Proxy.Type)()
89  except Exception:
90  pass # not a standard Arch object... Fall back to Draft mode
91  if cl:
92  base = utils.get_clone_base(obj[0])
93  cl.Label = prefix + base.Label
94  cl.CloneOf = base
95  if utils.get_type(obj[0]) != "BuildingPart":
96  cl.Placement = obj[0].Placement
97  for prop in ("Description", "IfcType", "Material", "Subvolume", "Tag"):
98  try:
99  setattr(cl, prop, getattr(base, prop))
100  except Exception:
101  pass
102  if App.GuiUp:
103  gui_utils.format_object(cl, base)
104  gui_utils.select(cl)
105  return cl
106 
107  # fall back to Draft clone mode
108  if not cl:
109  cl = App.ActiveDocument.addObject("Part::FeaturePython","Clone")
110  cl.addExtension("Part::AttachExtensionPython")
111  cl.Label = prefix + obj[0].Label
112  Clone(cl)
113  cl.Objects = obj
114  if delta:
115  cl.Placement.move(delta)
116  elif (len(obj) == 1) and hasattr(obj[0],"Placement"):
117  cl.Placement = obj[0].Placement
118  if hasattr(cl,"LongName") and hasattr(obj[0],"LongName"):
119  cl.LongName = obj[0].LongName
120  if App.GuiUp:
121  ViewProviderClone(cl.ViewObject)
122  gui_utils.format_object(cl, obj[0])
123  gui_utils.select(cl)
