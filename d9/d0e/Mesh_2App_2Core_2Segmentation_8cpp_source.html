556 void MeshSegmentAlgorithm::FindSegments(std::vector<MeshSurfaceSegmentPtr>& segm)
557 {
558  // reset VISIT flags
559  FacetIndex startFacet {};
560  MeshCore::MeshAlgorithm cAlgo(myKernel);
561  cAlgo.ResetFacetFlag(MeshCore::MeshFacet::VISIT);
562 
563  const MeshCore::MeshFacetArray& rFAry = myKernel.GetFacets();
564  MeshCore::MeshFacetArray::_TConstIterator iCur = rFAry.begin();
565  MeshCore::MeshFacetArray::_TConstIterator iBeg = rFAry.begin();
566  MeshCore::MeshFacetArray::_TConstIterator iEnd = rFAry.end();
567 
568  // start from the first not visited facet
569  cAlgo.CountFacetFlag(MeshCore::MeshFacet::VISIT);
570  std::vector<FacetIndex> resetVisited;
571 
572  for (auto& it : segm) {
573  cAlgo.ResetFacetsFlag(resetVisited, MeshCore::MeshFacet::VISIT);
574  resetVisited.clear();
575 
576  MeshCore::MeshIsNotFlag<MeshCore::MeshFacet> flag;
577  iCur = std::find_if(iBeg, iEnd, [flag](const MeshFacet& f) {
578  return flag(f, MeshFacet::VISIT);
579  });
580  if (iCur < iEnd) {
581  startFacet = iCur - iBeg;
582  }
583  else {
584  startFacet = FACET_INDEX_MAX;
585  }
586  while (startFacet != FACET_INDEX_MAX) {
587  // collect all facets of the same geometry
588  std::vector<FacetIndex> indices;
589  it->Initialize(startFacet);
590  if (it->TestInitialFacet(startFacet)) {
591  indices.push_back(startFacet);
592  }
593  MeshSurfaceVisitor pv(*it, indices);
594  myKernel.VisitNeighbourFacets(pv, startFacet);
595 
596  // add or discard the segment
597  if (indices.size() <= 1) {
598  resetVisited.push_back(startFacet);
599  }
600  else {
601  it->AddSegment(indices);
602  }
603 
604  // search for the next start facet
605  iCur = std::find_if(iCur, iEnd, [flag](const MeshFacet& f) {
606  return flag(f, MeshFacet::VISIT);
607  });
608  if (iCur < iEnd) {
609  startFacet = iCur - iBeg;
610  }
611  else {
612  startFacet = FACET_INDEX_MAX;
613  }
614  }
615  }
