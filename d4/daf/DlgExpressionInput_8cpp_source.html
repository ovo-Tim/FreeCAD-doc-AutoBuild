161 Base::Type DlgExpressionInput::determineTypeVarSet()
162 {
163  Base::Type typePath = getTypePath();
164 
165  // The type of the path is leading. If it is one of the types below, we
166  // can create a property in the varset.
167  if (typePath == App::PropertyString::getClassTypeId() ||
168  typePath.isDerivedFrom(App::PropertyFloat::getClassTypeId()) ||
169  typePath.isDerivedFrom(App::PropertyInteger::getClassTypeId())) {
170  return typePath;
171  }
172 
173  // If we cannot determine the type by means of the path, for example when
174  // dealing with a sketcher constraint list or with the x, y, or z of a
175  // Placement, the type of the unit allows us to create a property in the
176  // varset. Since unit properties are derived from App::PropertyFloat, it
177  // allows us to create a property and set the value.
178 
179  std::string unitTypeString = impliedUnit.getTypeString();
180  if (unitTypeString.empty()) {
181  // no type was provided
182  return Base::Type::badType();
183  }
184 
185  std::string typeString = "App::Property" + unitTypeString;
186  // may return badType
187  return Base::Type::fromName(typeString.c_str());
188 }

239 void NumberRange::throwIfOutOfRange(const Base::Quantity& value) const
240 {
241  if (!defined)
242  return;
243 
244  if (value.getValue() < minimum || value.getValue() > maximum) {
245  Base::Quantity minVal(minimum, value.getUnit());
246  Base::Quantity maxVal(maximum, value.getUnit());
247  auto valStr = value.getUserString();
248  auto minStr = minVal.getUserString();
249  auto maxStr = maxVal.getUserString();
250  throw Base::ValueError(fmt::format("Value out of range ({} out of [{}, {}])", valStr, minStr, maxStr));
251  }
252 }

269 void DlgExpressionInput::checkExpression(const QString& text)
270 {
271  //now handle expression
272  std::shared_ptr<Expression> expr(ExpressionParser::parse(path.getDocumentObject(), text.toUtf8().constData()));
273 
274  if (expr) {
275  std::string error = path.getDocumentObject()->ExpressionEngine.validateExpression(path, expr);
276 
277  if (!error.empty())
278  throw Base::RuntimeError(error.c_str());
279 
280  std::unique_ptr<Expression> result(expr->eval());
281 
282  expression = expr;
283  ui->okBtn->setEnabled(true);
284  ui->msg->clear();
285 
286  //set default palette as we may have read text right now
287  ui->msg->setPalette(ui->okBtn->palette());
288 
289  auto * n = Base::freecad_dynamic_cast<NumberExpression>(result.get());
290  if (n) {
291  Base::Quantity value = n->getQuantity();
292  if (!value.isValid()) {
293  throw Base::ValueError("Not a number");
294  }
295 
296  auto msg = value.getUserString();
297  if (!impliedUnit.isEmpty()) {
298  if (!value.getUnit().isEmpty() && value.getUnit() != impliedUnit)
299  throw Base::UnitsMismatchError("Unit mismatch between result and required unit");
300 
301  value.setUnit(impliedUnit);
302 
303  }
304  else if (!value.getUnit().isEmpty()) {
305  msg += " (Warning: unit discarded)";
306 
307  QPalette p(ui->msg->palette());
308  p.setColor(QPalette::WindowText, Qt::red);
309  ui->msg->setPalette(p);
310  }
311 
312  numberRange.throwIfOutOfRange(value);
313 
314  ui->msg->setText(QString::fromStdString(msg));
315  }
316  else {
317  ui->msg->setText(QString::fromStdString(result->toString()));
318  }
319 
320  }
321 }

325 void DlgExpressionInput::textChanged(const QString &text)
326 {
327  if (text.isEmpty()) {
328  ui->okBtn->setDisabled(true);
329  ui->discardBtn->setDefault(true);
330  return;
331  }
332 
333  ui->okBtn->setDefault(true);
334 
335  try {
336  //resize the input field according to text size
337  QFontMetrics fm(ui->expression->font());
338  int width = QtTools::horizontalAdvance(fm, text) + 15;
339  if (width < minimumWidth)
340  ui->expression->setMinimumWidth(minimumWidth);
341  else
342  ui->expression->setMinimumWidth(width);
343 
344  if(this->width() < ui->expression->minimumWidth())
345  setMinimumWidth(ui->expression->minimumWidth());
346 
347  checkExpression(text);
348  if (varSetsVisible) {
349  // If varsets are visible, check whether the varset info also
350  // agrees that the button should be enabled.
351  // No need to check the expression in that function.
352  updateVarSetInfo(NO_CHECK_EXPR);
353  }
354  }
355  catch (Base::Exception & e) {
356  ui->msg->setText(QString::fromUtf8(e.what()));
357  QPalette p(ui->msg->palette());
358  p.setColor(QPalette::WindowText, Qt::red);
359  ui->msg->setPalette(p);
360  ui->okBtn->setDisabled(true);
361  }

420 static bool isNamePropOk(const QString& nameProp, App::DocumentObject* obj,
421  std::stringstream& message)
422 {
423  if (!obj) {
424  message << "Unknown object";
425  return false;
426  }
427 
428  std::string name = nameProp.toStdString();
429  if (name.empty()) {
430  message << "Please provide a name for the property.";
431  return false;
432  }
433 
434  if (name != Base::Tools::getIdentifier(name)) {
435  message << "Invalid property name (must only contain alphanumericals, underscore, "
436  << "and must not start with digit";
437  return false;
438  }
439 
440  if (ExpressionParser::isTokenAUnit(name) || ExpressionParser::isTokenAConstant(name)) {
441  message << name << " is a reserved word";
442  return false;
443  }
444 
445  auto prop = obj->getPropertyByName(name.c_str());
446  if (prop && prop->getContainer() == obj) {
447  message << name << " already exists";
448  return false;
449  }
450 
451  return true;
452 }

465 void DlgExpressionInput::acceptWithVarSet()
466 {
467  // all checks have been performed in updateVarSetInfo and textChanged that
468  // decide to enable the button
469 
470  // create a property in the VarSet
471  QTreeWidgetItem *selected = treeWidget->currentItem();
472  QString nameVarSet = getValue(selected, ROLE_VARSET_NAME);
473  QString nameGroup = ui->lineEditGroup->text();
474  QString nameProp = ui->lineEditPropNew->text();
475 
476  QString nameDoc = getValue(selected, ROLE_DOC);
477  App::Document* doc = App::GetApplication().getDocument(nameDoc.toUtf8());
478  App::DocumentObject* obj = doc->getObject(nameVarSet.toUtf8());
479 
480  std::string name = nameProp.toStdString();
481  std::string group = nameGroup.toStdString();
482  std::string type = getType();
483  auto prop = obj->addDynamicProperty(type.c_str(), name.c_str(), group.c_str());
484 
485  // Set the value of the property in the VarSet
486  //
487  // The value of the property is going to be the value that was originally
488  // meant to be the value for the property that this dialog is targeting.
489  Expression* exprSimplfied = expression->simplify();
490  auto ne = dynamic_cast<NumberExpression*>(exprSimplfied);
491  auto se = dynamic_cast<StringExpression*>(exprSimplfied);
492  if (ne) {
493  // the value is a number: directly assign it to the property instead of
494  // making it an expression in the variable set
495  Gui::Command::doCommand(Gui::Command::Doc, "App.getDocument('%s').getObject('%s').%s = %f",
496  obj->getDocument()->getName(),
497  obj->getNameInDocument(),
498  prop->getName(), ne->getValue());
499  }
500  else if (se) {
501  // the value is a string: directly assign it to the property.
502  Gui::Command::doCommand(Gui::Command::Doc, "App.getDocument('%s').getObject('%s').%s = \"%s\"",
503  obj->getDocument()->getName(),
504  obj->getNameInDocument(),
505  prop->getName(), se->getText().c_str());
506  }
507  else {
508  // the value is an epxression: make an expression binding in the variable set.
509  ObjectIdentifier objId(*prop);
510  Binding binding;
511  binding.bind(objId);
512  binding.setExpression(expression);
513  binding.apply();
514  }
515 
516  // Create a new expression that refers to the property in the variable set
517  // for the original property that is the target of this dialog.
518  expression.reset(ExpressionParser::parse(path.getDocumentObject(),
519  prop->getFullName().c_str()));
520 }
