62 void ViewProviderPlane::attach(App::DocumentObject * obj) {
63  ViewProviderDatum::attach(obj);
64 
65  std::string role = getRole();
66 
67  SoSeparator* sep = getDatumRoot();
68 
69  // Setup colors
70  // Can't use transparency because of https://github.com/FreeCAD/FreeCAD/issues/18395
71  // When this issue is fixed then we can use the below and remove the material here
72  // and faceSeparator...
73  //ShapeAppearance.setTransparency(0.8);
74  auto material = new SoMaterial();
75  material->transparency.setValue(0.95f);
76  if (!role.empty()) {
77  ShapeAppearance.setDiffuseColor(getColor(role));
78  SbColor color;
79  float alpha = 0.0f;
80  color.setPackedValue(getColor(role), alpha);
81  material->ambientColor.setValue(color);
82  material->diffuseColor.setValue(color);
83  }
84 
85  static const float size = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/View")->GetFloat("DatumPlaneSize", 40.0);
86  static const float startSize = 0.25 * size; //NOLINT
87 
88  SbVec3f verts[4];
89  if (role.empty()) {
90  verts[0] = SbVec3f(size, size, 0);
91  verts[1] = SbVec3f(size, -size, 0);
92  verts[2] = SbVec3f(-size, -size, 0);
93  verts[3] = SbVec3f(-size, size, 0);
94  }
95  else {
96  verts[0] = SbVec3f(size, size, 0);
97  verts[1] = SbVec3f(size, startSize, 0);
98  verts[2] = SbVec3f(startSize, startSize, 0);
99  verts[3] = SbVec3f(startSize, size, 0);
100  }
101 
102  auto pCoords = new SoCoordinate3();
103  pCoords->point.setNum(4);
104  pCoords->point.setValues(0, 4, verts);
105  sep->addChild(pCoords);
106 
107  auto lineSeparator = new SoSeparator();
108  auto pLines = new SoIndexedLineSet();
109  static const int32_t lines[6] = { 0, 1, 2, 3, 0, -1 };
110  pLines->coordIndex.setNum(6);
111  pLines->coordIndex.setValues(0, 6, lines);
112 
113  auto ps = new SoPickStyle();
114  ps->style.setValue(SoPickStyle::SHAPE_ON_TOP);
115  lineSeparator->addChild(ps);
116  lineSeparator->addChild(pLines);
117  sep->addChild(lineSeparator);
118 
119  // add semi transparent face
120  auto faceSeparator = new SoSeparator();
121  sep->addChild(faceSeparator);
122 
123  faceSeparator->addChild(material);
124 
125  // disable backface culling and render with two-sided lighting
126  auto shapeHints = new SoShapeHints();
127  shapeHints->vertexOrdering = SoShapeHints::COUNTERCLOCKWISE;
128  shapeHints->shapeType = SoShapeHints::UNKNOWN_SHAPE_TYPE;
129  faceSeparator->addChild(shapeHints);
130 
131  auto faceSet = new SoFaceSet();
132  auto vertexProperty = new SoVertexProperty();
133  vertexProperty->vertex.setValues(0, 4, verts);
134  faceSet->vertexProperty.setValue(vertexProperty);
135  faceSeparator->addChild(faceSet);
136 
137  auto textTranslation = new SoTranslation();
138  SbVec3f centeringVec = size * SbVec3f(0.36, 0.49, 0); // NOLINT
139  textTranslation->translation.setValue(centeringVec);
140  sep->addChild(textTranslation);
141 
142  pLabel->string.setValue(getLabelText(role).c_str());
143 
144  labelSwitch = new SoSwitch();
145  setLabelVisibility(false);
146  labelSwitch->addChild(pLabel);
147  sep->addChild(labelSwitch);
148 }

155 unsigned long ViewProviderPlane::getColor(std::string& role)
156 {
157  auto planesRoles = App::LocalCoordinateSystem::PlaneRoles;
158  if (role == planesRoles[0]) {
159  return ViewParams::instance()->getAxisZColor(); // XY-plane
160  }
161  else if (role == planesRoles[1]) {
162  return ViewParams::instance()->getAxisYColor(); // XZ-plane

166  }

195  return planesRoles[1];

198  return planesRoles[2];
