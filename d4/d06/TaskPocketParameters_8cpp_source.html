88 void TaskPocketParameters::onModeChanged(int index)
89 {
90  auto pcPocket = getObject<PartDesign::Pocket>();
91 
92  switch (static_cast<Mode>(index)) {
93  case Mode::Dimension:
94  // Why? See below for "UpToFace"
95  if (oldLength < Precision::Confusion())
96  oldLength = 5.0;
97  pcPocket->Length.setValue(oldLength);
98  ui->lengthEdit->setValue(oldLength);
99  pcPocket->Type.setValue("Length");
100  break;
101  case Mode::ThroughAll:
102  oldLength = pcPocket->Length.getValue();
103  pcPocket->Type.setValue("ThroughAll");
104  break;
105  case Mode::ToFirst:
106  oldLength = pcPocket->Length.getValue();
107  pcPocket->Type.setValue("UpToFirst");
108  break;
109  case Mode::ToFace:
110  // Note: ui->checkBoxReversed is purposely enabled because the selected face
111  // could be a circular one around the sketch
112  // Also note: Because of the code at the beginning of Pocket::execute() which is used
113  // to detect broken legacy parts, we must set the length to zero here!
114  oldLength = pcPocket->Length.getValue();
115  pcPocket->Type.setValue("UpToFace");
116  pcPocket->Length.setValue(0.0);
117  ui->lengthEdit->setValue(0.0);
118  if (ui->lineFaceName->text().isEmpty()) {
119  ui->buttonFace->setChecked(true);
120  handleLineFaceNameClick(); // sets placeholder text
121  }
122  break;
123  case Mode::TwoDimensions:
124  oldLength = pcPocket->Length.getValue();
125  pcPocket->Type.setValue("TwoLengths");
126  break;
127  case Mode::ToShape:
128  pcPocket->Type.setValue("UpToShape");
129  break;
130  }
131 
132  updateUI(index);
133  recomputeFeature();
134 }
