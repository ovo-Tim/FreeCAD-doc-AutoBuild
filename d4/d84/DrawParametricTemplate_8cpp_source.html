88 App::DocumentObjectExecReturn *DrawParametricTemplate::execute()
89 {
90  std::string temp = Template.getValue();
91  if (temp.empty()) {
92  return App::DocumentObject::StdReturn;
93  }
94 
95  Base::FileInfo tfi(temp);
96  if (!tfi.isReadable()) {
97  // if there is a old absolute template file set use a redirect
98  return App::DocumentObject::StdReturn;
99  }
100 
101  try {
102  Base::Interpreter().runFile(temp.c_str(), true);
103  }
104  catch(const Base::Exception& e) {
105  PyErr_SetString(PyExc_ImportError, e.what());
106  }
107  return App::DocumentObject::StdReturn;
108 }
