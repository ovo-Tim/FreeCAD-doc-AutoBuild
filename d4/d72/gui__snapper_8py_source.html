383  needSubElement=True,
384  noElementMap=True)

388

390  # Special snapping for polygons: add the center
391  snaps.extend(self.snapToPolygon(obj))
392 
393  elif (Draft.getType(obj) == "BuildingPart"
394  and self.isEnabled("Center")):
395  # snap to the base placement of empty BuildingParts
396  snaps.append([obj.Placement.Base, 'center',
397  self.toWP(obj.Placement.Base)])
398

401  # we are snapping to an edge
402  if shape.ShapeType == "Edge":

404  snaps.extend(self.snapToNear(edge, point))
405  snaps.extend(self.snapToEndpoints(edge))

414  snaps.extend(self.snapToAngles(edge))
415  snaps.extend(self.snapToCenter(edge))
416  elif et == "Ellipse":
417  # extra ellipse options
418  snaps.extend(self.snapToCenter(edge))
419  elif "Face" in comp:
420  # we are snapping to a face
421  if shape.ShapeType == "Face":
422  face = shape
423  snaps.extend(self.snapToNearFace(face, point))
424  snaps.extend(self.snapToPerpendicularFace(face, lastpoint))
425  snaps.extend(self.snapToCenterFace(face))

427  # we are snapping to a vertex
428  if shape.ShapeType == "Vertex":
429  snaps.extend(self.snapToEndpoints(shape))

433  # or vertices.
434  snaps.extend(self.snapToNearUnprojected(point))
435 
436  elif Draft.getType(obj) in ("LinearDimension", "AngularDimension"):

438  snaps.extend(self.snapToDim(obj))
439 
440  elif Draft.getType(obj) == "Axis":
441  for edge in obj.Shape.Edges:
442  snaps.extend(self.snapToEndpoints(edge))
443  snaps.extend(self.snapToIntersection(edge))
444 
445  elif Draft.getType(obj).startswith("Mesh::"):
446  snaps.extend(self.snapToNearUnprojected(point))
447  snaps.extend(self.snapToEndpoints(obj.Mesh))
448 
449  elif Draft.getType(obj).startswith("Points::"):
450  # for points we only snap to points
451  snaps.extend(self.snapToEndpoints(obj.Points))
452 
453  elif (Draft.getType(obj) in ("WorkingPlaneProxy", "BuildingPart")
454  and self.isEnabled("Center")):
455  # snap to the center of WPProxies or to the base
456  # placement of no empty BuildingParts
457  snaps.append([obj.Placement.Base, 'center',
458  self.toWP(obj.Placement.Base)])
459 
460  elif Draft.getType(obj) == "SectionPlane":
461  # snap to corners of section planes
462  snaps.extend(self.snapToEndpoints(obj.Shape))
463 
464  # updating last objects list
465  if obj.Name in self.lastObj:
466  self.lastObj.remove(obj.Name)
467  self.lastObj.append(obj.Name)
468  if len(self.lastObj) > 8:
469  self.lastObj = self.lastObj[-8:]
470 
471  if not snaps:
472  return None

475  shortest = 1000000000000000000
476  origin = App.Vector(self.snapInfo['x'],
477  self.snapInfo['y'],
478  self.snapInfo['z'])
479  winner = None
480  fp = point
481  for snap in snaps:
482  if (not snap) or (snap[0] is None):
483  pass
484  # print("debug: Snapper: invalid snap point: ",snaps)
485  else:
486  delta = snap[0].sub(origin)
487  if delta.Length < shortest:
488  shortest = delta.Length
489  winner = snap
490 
491  if winner:
492  # setting the cursors
493  if self.tracker and not self.selectMode:
494  self.tracker.setCoords(winner[2])
495  self.tracker.setMarker(self.mk[winner[1]])
496  self.tracker.on()

498  fp = self.cstr(lastpoint, constrain, winner[2])
499  if self.trackLine and lastpoint:
500  self.trackLine.p2(fp)
501  self.trackLine.setColor()
502  self.trackLine.on()
503  # set the cursor
504  self.setCursor(winner[1])
505 
506  # set the arch point tracking
507  if lastpoint:
508  self.setArchDims(lastpoint, fp)
509 
510  # return the final point
511  self.spoint = fp
512  self.running = False
513  return self.spoint
