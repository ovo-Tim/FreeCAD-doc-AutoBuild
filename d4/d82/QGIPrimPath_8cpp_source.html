242 void QGIPrimPath::mousePressEvent(QGraphicsSceneMouseEvent *event)
243 {
244  Qt::KeyboardModifiers originalModifiers = event->modifiers();
245  if (event->button()&Qt::LeftButton) {
246  multiselectActivated = false;
247  }
248 
249  if (event->button() == Qt::LeftButton
250  && multiselectEligible()
251  && PreferencesGui::multiSelection()) {
252 
253  auto parent = dynamic_cast<QGIView *>(parentItem());
254  if (parent) {
255  std::vector<Gui::SelectionObject> selection = Gui::Selection().getSelectionEx();
256  if (DGU::findObjectInSelection(selection, *(parent->getViewObject()))) {
257  // if our parent is already in the selection, then allow addition
258  // primitives to be selected.
259  multiselectActivated = true;
260  event->setModifiers(originalModifiers | Qt::ControlModifier);
261  }
262  }
263  }
264 
265  QGraphicsPathItem::mousePressEvent(event);
266 
267  event->setModifiers(originalModifiers);
268 }
