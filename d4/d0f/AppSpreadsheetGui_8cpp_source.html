67 private:
68  void load(App::Document* pcDoc, const std::string& Name)
69  {
70  try {
71  Base::FileInfo file(Name);
72  Spreadsheet::Sheet* pcSheet = static_cast<Spreadsheet::Sheet*>(
73  pcDoc->addObject("Spreadsheet::Sheet", file.fileNamePure().c_str()));
74 
75  pcSheet->importFromFile(Name, '\t', '"', '\\');
76  pcSheet->execute();
77  }
78  catch (const Base::Exception& e) {
79  throw Py::RuntimeError(e.what());
80  }
81  }
82  Py::Object open(const Py::Tuple& args)
83  {
84  char* Name;
85  const char* DocName = nullptr;
86  if (!PyArg_ParseTuple(args.ptr(), "et|s", "utf-8", &Name, &DocName)) {
87  throw Py::Exception();
88  }
89  std::string EncodedName = std::string(Name);
90  PyMem_Free(Name);
91 
92  App::Document* pcDoc =
93  App::GetApplication().newDocument(DocName ? DocName : QT_TR_NOOP("Unnamed"));
94  load(pcDoc, EncodedName);
95 
96  return Py::None();
97  }
98 
99  Py::Object insert(const Py::Tuple& args)
100  {
101  char* Name;
102  const char* DocName = nullptr;
103  if (!PyArg_ParseTuple(args.ptr(), "et|s", "utf-8", &Name, &DocName)) {
104  throw Py::Exception();
105  }
106  std::string EncodedName = std::string(Name);
107  PyMem_Free(Name);
108 
109  App::Document* pcDoc = App::GetApplication().getDocument(DocName);
110  if (!pcDoc) {
111  pcDoc = App::GetApplication().newDocument(DocName ? DocName : QT_TR_NOOP("Unnamed"));
112  }
113  load(pcDoc, EncodedName);
114 
115  return Py::None();
116  }
