140 class MoveArc(Instruction):
141  def anglesOfTangents(self):
142  """anglesOfTangents() ... return a tuple with the tangent angles at begin and end position"""
143  begin = self.xyBegin()
144  end = self.xyEnd()
145  center = self.xyCenter()
146  # calculate angle of the hypotenuse at begin and end
147  s0 = Path.Geom.getAngle(begin - center)
148  s1 = Path.Geom.getAngle(end - center)
149  # the tangents are perpendicular to the hypotenuse with the sign determined by the
150  # direction of the arc
151  return (
152  Path.Geom.normalizeAngle(s0 + self.arcDirection()),
153  Path.Geom.normalizeAngle(s1 + self.arcDirection()),
154  )

168  def arcAngle(self):
169  """arcAngle() ... return the angle of the arc opening"""
170  begin = self.xyBegin()
171  end = self.xyEnd()
172  center = self.xyCenter()
173  s0 = Path.Geom.getAngle(begin - center)
174  s1 = Path.Geom.getAngle(end - center)
175 
176  if self.isCW():
177  while s0 < s1:
178  s0 = s0 + 2 * math.pi
179  return s0 - s1
180 
181  # CCW
182  while s1 < s0:
183  s1 = s1 + 2 * math.pi
184  return s1 - s0
