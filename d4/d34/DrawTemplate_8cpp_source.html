104 // Return the counts related to pages, namely collated page index and total page count
105 std::pair<int, int> DrawTemplate::getPageNumbers() const
106 {
107  std::vector<DocumentObject *> pages = getDocument()->getObjectsOfType(TechDraw::DrawPage::getClassTypeId());
108  std::vector<QString> pageNames;
109  for (auto page : pages) {
110  if (page->isAttachedToDocument() &&
111  !page->testStatus(App::ObjectStatus::Remove)) {
112  pageNames.push_back(QString::fromUtf8(page->Label.getValue()));
113  }
114  }
115  QCollator collator;
116  std::sort(pageNames.begin(), pageNames.end(), collator);
117 
118  int pos = 0;
119  DrawPage *page = getParentPage();
120  if (page) {
121  auto it = std::find(pageNames.begin(), pageNames.end(), QString::fromUtf8(page->Label.getValue()));
122  if (it != pageNames.end()) {
123  pos = it - pageNames.begin() + 1;
124  }
125  }
126 
127  return std::pair<int, int>(pos, (int) pageNames.size());
128 }

131 QString DrawTemplate::getAutofillValue(const QString &id) const
132 {
133  constexpr int ISODATELENGTH {10};
134  auto doc = getDocument();
135  if (!doc) {
136  return QString();
137  }
138  // author
139  if (id.compare(QString::fromUtf8(Autofill::Author)) == 0) {
140  auto value = QString::fromUtf8(doc->CreatedBy.getValue());
141  if (!value.isEmpty()) {
142  return value;
143  }
144  }
145  // date
146  else if (id.compare(QString::fromUtf8(Autofill::Date)) == 0) {
147  auto timeLocale = std::setlocale(LC_TIME, nullptr);
148  QDateTime date = QDateTime::currentDateTime();
149  if (Preferences::enforceISODate()) {
150  auto rawDate = date.toString(Qt::ISODate);
151  return rawDate.left(ISODATELENGTH);
152  }
153  auto qTimeLocale = QString::fromUtf8(timeLocale);
154  return date.toString(QLocale(qTimeLocale).dateFormat(QLocale::ShortFormat));
155  }
156  // organization ( also organisation/owner/company )
157  else if (id.compare(QString::fromUtf8(Autofill::Organization)) == 0 ||
158  id.compare(QString::fromUtf8(Autofill::Organisation)) == 0 ||
159  id.compare(QString::fromUtf8(Autofill::Owner)) == 0 ||
160  id.compare(QString::fromUtf8(Autofill::Company)) == 0 ) {
161  auto value = QString::fromUtf8(doc->Company.getValue());
162  if (!value.isEmpty()) {
163  return value;
164  }
165  }
166  // scale
167  else if (id.compare(QString::fromUtf8(Autofill::Scale)) == 0) {
168  DrawPage *page = getParentPage();
169  if (page) {
170  std::pair<int, int> scale = DrawUtil::nearestFraction(page->Scale.getValue());
171  return QString::asprintf("%d : %d", scale.first, scale.second);
172  }
173  }
174  // sheet
175  else if (id.compare(QString::fromUtf8(Autofill::Sheet)) == 0) {
176  std::pair<int, int> pageNumbers = getPageNumbers();
177  return QString::asprintf("%d / %d", pageNumbers.first, pageNumbers.second);
178  }
179  // title
180  else if (id.compare(QString::fromUtf8(Autofill::Title)) == 0) {
181  return QString::fromUtf8(getDocument()->Label.getValue());
182  }
183  // page number
184  else if (id.compare(QString::fromUtf8(Autofill::PageNumber)) == 0) {
185  std::pair<int, int> pageNumbers = getPageNumbers();
186  return QString::number(pageNumbers.first);
187  }
188  // page total
189  else if (id.compare(QString::fromUtf8(Autofill::PageCount)) == 0) {
190  std::pair<int, int> pageNumbers = getPageNumbers();
191  return QString::number(pageNumbers.second);
192  }
193 
194  return QString();
195 }
