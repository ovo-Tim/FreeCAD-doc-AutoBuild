223  std::pair<Base::Vector3d, Base::Vector3d> ends;

236  m_hShift, m_vShift, m_rotate);
237  } else if (m_type == CLTYPE::EDGE) {
238  ends = calcEndPoints2Lines(partFeat,
239  m_edges,
240  m_mode,
241  m_extendBy,
242  m_hShift, m_vShift, m_rotate, m_flip2Line);
243  } else if (m_type == CLTYPE::VERTEX) {
244  ends = calcEndPoints2Points(partFeat,
245  m_verts,
246  m_mode,
247  m_extendBy,
248  m_hShift, m_vShift, m_rotate, m_flip2Line);
249  }
250  }
251

255  }
256 
257  Base::Vector3d p1 = ends.first;
258  Base::Vector3d p2 = ends.second;

268  TopoDS_Shape s = ShapeUtils::scaleShape(e, scale);
269  TopoDS_Edge newEdge = TopoDS::Edge(s);
270  TechDraw::BaseGeomPtr newGeom = TechDraw::BaseGeom::baseFactory(newEdge);
271  newGeom->setClassOfEdge(ecHARD);
272  newGeom->setHlrVisible( true);
273  newGeom->setCosmetic(true);
274  newGeom->source(CENTERLINE);

395 std::pair<Base::Vector3d, Base::Vector3d> CenterLine::rotatePointsAroundMid(const Base::Vector3d& p1,
396  const Base::Vector3d& p2,
397  const Base::Vector3d& mid,
398  const double angleDeg)
399 {
400  std::pair<Base::Vector3d, Base::Vector3d> result;
401  double angleRad = angleDeg * M_PI / 180.0;
402 
403  result.first.x = ((p1.x - mid.x) * cos(angleRad)) - ((p1.y - mid.y) * sin(angleRad)) + mid.x;
404  result.first.y = ((p1.x - mid.x) * sin(angleRad)) + ((p1.y - mid.y) * cos(angleRad)) + mid.y;
405  result.first.z = 0.0;
406 
407  result.second.x = ((p2.x - mid.x) * cos(angleRad)) - ((p2.y - mid.y) * sin(angleRad)) + mid.x;
408  result.second.y = ((p2.x - mid.x) * sin(angleRad)) + ((p2.y - mid.y) * cos(angleRad)) + mid.y;
409  result.second.z = 0.0;
410 
411  return result;
412 }

415 //end points for centerline with no geometry reference
416 std::pair<Base::Vector3d, Base::Vector3d> CenterLine::calcEndPointsNoRef(const Base::Vector3d& start,
417  const Base::Vector3d& end,
418  const double scale,
419  const double ext,
420  const double hShift,
421  const double vShift,
422  const double rotate, const double viewAngleDeg)
423 {
424 // Base::Console().Message("CL::calcEndPointsNoRef()\n");
425  Base::Vector3d p1 = start;
426  Base::Vector3d p2 = end;
427  Base::Vector3d mid = (p1 + p2) / 2.0;
428 
429  //extend
430  Base::Vector3d clDir = p2 - p1;
431  clDir.Normalize();
432  p1 = p1 - (clDir * ext);
433  p2 = p2 + (clDir * ext);
434 
435  //rotate
436  if (!DrawUtil::fpCompare(rotate, 0.0)) {
437  //rotate p1, p2 about mid point
438  std::tie(p1, p2) = rotatePointsAroundMid(p1, p2, mid, rotate);
439  }
440 
441  //shift
442  if (!DrawUtil::fpCompare(hShift, 0.0)) {
443  double hss = hShift * scale;
444  p1.x = p1.x + hss;
445  p2.x = p2.x + hss;
446  }
447  if (!DrawUtil::fpCompare(vShift, 0.0)) {
448  double vss = vShift * scale;
449  p1.y = p1.y + vss;
450  p2.y = p2.y + vss;
451  }
452 
453  // rotate the endpoints so that when the View's Rotation is applied, the
454  // centerline is aligned correctly
455  std::pair<Base::Vector3d, Base::Vector3d> result;
456  result.first = p1 / scale;
457  result.second = p2 / scale;
458  Base::Vector3d midpoint = (result.first + result.second) / 2.0;
459  result = rotatePointsAroundMid(result.first, result.second, midpoint, viewAngleDeg * -1.0);
460 
461  return result;
462 }

464 //end points for face centerline
465 std::pair<Base::Vector3d, Base::Vector3d> CenterLine::calcEndPoints(const DrawViewPart* partFeat,
466  const std::vector<std::string>& faceNames,
467  const int mode,
468  const double ext,
469  const double hShift,
470  const double vShift,
471  const double rotate)
472 {
473 // Base::Console().Message("CL::calcEndPoints()\n");
474  if (faceNames.empty()) {
475  Base::Console().Warning("CL::calcEndPoints - no faces!\n");
476  return std::pair<Base::Vector3d, Base::Vector3d>();
477  }
478 
479  Bnd_Box faceBox;
480  faceBox.SetGap(0.0);
481 
482  double scale = partFeat->getScale();
483 
484  std::vector<TopoDS_Edge> faceEdgesAll;
485  for (auto& fn: faceNames) {
486  if (TechDraw::DrawUtil::getGeomTypeFromName(fn) != "Face") {
487  continue;
488  }
489  int idx = TechDraw::DrawUtil::getIndexFromName(fn);
490  std::vector<TechDraw::BaseGeomPtr> faceGeoms =
491  partFeat->getFaceEdgesByIndex(idx);
492  if (!faceGeoms.empty()) {
493  for (auto& fe: faceGeoms) {
494  if (!fe->getCosmetic()) {
495  faceEdgesAll.push_back(fe->getOCCEdge());
496  }
497  }
498  }
499  }
500  TopoDS_Shape faceEdgeCompound = DU::vectorToCompound(faceEdgesAll);
501 
502  if (partFeat->Rotation.getValue() != 0.0) {
503  // unrotate the input shape to align with the cardinal axes so we can use bbox to
504  // get size measurements
505  faceEdgeCompound = ShapeUtils::rotateShape(faceEdgeCompound,
506  partFeat->getProjectionCS(),
507  partFeat->Rotation.getValue() * -1.0);
508  }
509  // get the center of the unrotated, scaled face
510  Base::Vector3d faceCenter = ShapeUtils::findCentroidVec(faceEdgeCompound, partFeat->getProjectionCS());
511  // we need to move the edges to the origin here to get the right limits from the bounding box
512  faceEdgeCompound = ShapeUtils::moveShape(faceEdgeCompound, faceCenter * -1.0);
513 
514  // get the bounding box of the centered and unrotated face
515  BRepBndLib::AddOptimal(faceEdgeCompound, faceBox);
516 
517  if (faceBox.IsVoid()) {
518  Base::Console().Error("CL::calcEndPoints - faceBox is void!\n");
519  throw Base::IndexError("CenterLine wrong number of faces.");
520  }
521 
522  double Xmin, Ymin, Zmin, Xmax, Ymax, Zmax;
523  faceBox.Get(Xmin, Ymin, Zmin, Xmax, Ymax, Zmax);
524 
525  double Xspan = fabs(Xmax - Xmin);
526  Xspan = (Xspan / 2.0);
527  double Xmid = 0.0;
528  Xmax = Xmid + Xspan;
529  Xmin = Xmid - Xspan;
530  double Yspan = fabs(Ymax - Ymin);
531  Yspan = (Yspan / 2.0);
532  double Ymid = 0.0;
533  Ymax = Ymid + Yspan;
534  Ymin = Ymid - Yspan;
535 
536  Base::Vector3d p1, p2;
537  if (mode == CenterLine::VERTICAL) { //vertical
538  p1 = Base::Vector3d(Xmid, Ymax, 0.0);
539  p2 = Base::Vector3d(Xmid, Ymin, 0.0);
540  } else if (mode == CenterLine::HORIZONTAL) { //horizontal
541  p1 = Base::Vector3d(Xmin, Ymid, 0.0);
542  p2 = Base::Vector3d(Xmax, Ymid, 0.0);
543  } else { //vert == CenterLine::ALIGNED //aligned, but aligned doesn't make sense for face(s) bbox
544  Base::Console().Message("CL::calcEndPoints - aligned is not applicable to Face center lines\n");
545  p1 = Base::Vector3d(Xmid, Ymax, 0.0);
546  p2 = Base::Vector3d(Xmid, Ymin, 0.0);
547  }
548 
549  // now move the extents back to the face center. this should give us the scaled,
550  // unrotated ends of the cl (but in 3d coordinates, which will be handled at the time
551  // the cl is added to the view)
552  p1 += faceCenter;
553  p2 += faceCenter;
554 
555  Base::Vector3d mid = (p1 + p2) / 2.0;
556 
557  //extend
558  Base::Vector3d clDir = p2 - p1;
559  clDir.Normalize();
560  p1 = p1 - (clDir * ext);
561  p2 = p2 + (clDir * ext);
562 
563  //rotate
564  if (!DrawUtil::fpCompare(rotate, 0.0)) {
565  //rotate p1, p2 about mid point
566  std::tie(p1, p2) = rotatePointsAroundMid(p1, p2, mid, rotate);
567  }
568 
569  //shift
570  if (!DrawUtil::fpCompare(hShift, 0.0)) {
571  double hss = hShift * scale;
572  p1.x = p1.x + hss;
573  p2.x = p2.x + hss;
574  }
575  if (!DrawUtil::fpCompare(vShift, 0.0)) {
576  double vss = vShift * scale;
577  p1.y = p1.y + vss;
578  p2.y = p2.y + vss;
579  }
580 
581  std::pair<Base::Vector3d, Base::Vector3d> result;
582  result.first = p1 / scale;
583  result.second = p2 / scale;
584  return result;
585 }

587 std::pair<Base::Vector3d, Base::Vector3d> CenterLine::calcEndPoints2Lines(const DrawViewPart* partFeat,
588  const std::vector<std::string>& edgeNames,
589  const int mode,
590  const double ext,
591  const double hShift,
592  const double vShift,
593  const double rotate,
594  const bool flip)
595 
596 {
597  Q_UNUSED(flip)
598 
599 // Base::Console().Message("CL::calc2Lines() - mode: %d flip: %d edgeNames: %d\n", mode, flip, edgeNames.size());
600  std::pair<Base::Vector3d, Base::Vector3d> result;
601  if (edgeNames.empty()) {
602  Base::Console().Warning("CL::calcEndPoints2Lines - no edges!\n");
603  return result;
604  }
605 
606  double scale = partFeat->getScale();
607 
608  std::vector<TechDraw::BaseGeomPtr> edges;
609  for (auto& en: edgeNames) {
610  if (TechDraw::DrawUtil::getGeomTypeFromName(en) != "Edge") {
611  continue;
612  }
613  int idx = TechDraw::DrawUtil::getIndexFromName(en);
614  TechDraw::BaseGeomPtr bg = partFeat->getGeomByIndex(idx);
615  if (bg) {
616  edges.push_back(bg);
617  } else {
618  Base::Console().Message("CL::calcEndPoints2Lines - no geom for index: %d\n", idx);
619  }
620  }
621  if (edges.size() != 2) {
622  Base::Console().Message("CL::calcEndPoints2Lines - wrong number of edges: %d!\n", edges.size());
623  throw Base::IndexError("CenterLine wrong number of edges.");
624  }
625 
626  // these points are centered, rotated, scaled and inverted.
627  // invert the points so the math works correctly
628  Base::Vector3d l1p1 = DU::invertY(edges.front()->getStartPoint());
629  Base::Vector3d l1p2 = DU::invertY(edges.front()->getEndPoint());
630  Base::Vector3d l2p1 = DU::invertY(edges.back()->getStartPoint());
631  Base::Vector3d l2p2 = DU::invertY(edges.back()->getEndPoint());
632 
633  // The centerline is drawn using the midpoints of the two lines that connect l1p1-l2p1 and l1p2-l2p2.
634  // However, we don't know which point should be l1p1 to get a geometrically correct result, see
635  // https://wiki.freecad.org/File:TD-CenterLineFlip.png for an illustration of the problem.
636  // Thus we test this by a circulation test, see this post for a brief explanation:
637  // https://forum.freecad.org/viewtopic.php?p=505733#p505615
638  if (DrawUtil::circulation(l1p1, l1p2, l2p1) != DrawUtil::circulation(l1p2, l2p2, l2p1)) {
639  Base::Vector3d temp; // reverse line 1
640  temp = l1p1;
641  l1p1 = l1p2;
642  l1p2 = temp;
643  }
644 
645  Base::Vector3d p1 = (l1p1 + l2p1) / 2.0;
646  Base::Vector3d p2 = (l1p2 + l2p2) / 2.0;
647  Base::Vector3d mid = (p1 + p2) / 2.0;
648 
649  // if the proposed end points prevent the creation of a vertical or horizontal centerline, we need
650  // to prevent the "orientation" code below from creating identical endpoints. This would create a
651  // zero length edge and cause problems later.
652  bool inhibitVertical = false;
653  bool inhibitHorizontal = false;
654  if (DU::fpCompare(p1.y, p2.y, EWTOLERANCE)) {
655  // proposed end points are aligned vertically, so we can't draw a vertical line to connect them
656  inhibitVertical = true;
657  }
658  if (DU::fpCompare(p1.x, p2.x, EWTOLERANCE)) {
659  // proposed end points are aligned horizontally, so we can't draw a horizontal line to connect them
660  inhibitHorizontal = true;
661  }
662 
663  //orientation
664  if (partFeat->Rotation.getValue() == 0.0) {
665  // if the view is rotated, then horizontal and vertical lose their meaning
666  if (mode == 0 && !inhibitVertical) { //Vertical
667  p1.x = mid.x;
668  p2.x = mid.x;
669  } else if (mode == 1 && !inhibitHorizontal) { //Horizontal
670  p1.y = mid.y;
671  p2.y = mid.y;
672  } else if (mode == 2) { //Aligned
673  // no op
674  }
675  }
676 
677  //extend
678  Base::Vector3d clDir = p2 - p1;
679  clDir.Normalize();
680  p1 = p1 - (clDir * ext);
681  p2 = p2 + (clDir * ext);
682 
683  //rotate
684  if (!DrawUtil::fpCompare(rotate, 0.0)) {
685  //rotate p1, p2 about mid
686  std::tie(p1, p2) = rotatePointsAroundMid(p1, p2, mid, rotate);
687  }
688 
689  //shift
690  if (!DrawUtil::fpCompare(hShift, 0.0)) {
691  double hss = hShift * scale;
692  p1.x = p1.x + hss;
693  p2.x = p2.x + hss;
694  }
695  if (!DrawUtil::fpCompare(vShift, 0.0)) {
696  double vss = vShift * scale;
697  p1.y = p1.y + vss;
698  p2.y = p2.y + vss;
699  }
700 
701  // the cl will be scaled when drawn, so unscale now.
702  result.first = p1 / scale;
703  result.second = p2 / scale;
704  return result;
705 }

707 std::pair<Base::Vector3d, Base::Vector3d> CenterLine::calcEndPoints2Points(const DrawViewPart* partFeat,
708  const std::vector<std::string>& vertNames,
709  const int mode,
710  const double ext,
711  const double hShift,
712  const double vShift,
713  const double rotate,
714  const bool flip)
715 
716 {
717 // Base::Console().Message("CL::calc2Points() - mode: %d\n", mode);
718  if (vertNames.empty()) {
719  Base::Console().Warning("CL::calcEndPoints2Points - no points!\n");
720  return std::pair<Base::Vector3d, Base::Vector3d>();
721  }
722 
723  double scale = partFeat->getScale();
724 
725  std::vector<TechDraw::VertexPtr> points;
726  for (auto& vn: vertNames) {
727  if (TechDraw::DrawUtil::getGeomTypeFromName(vn) != "Vertex") {
728  continue;
729  }

731  TechDraw::VertexPtr v = partFeat->getProjVertexByIndex(idx);
732  if (v) {
733  points.push_back(v);
734  }
735  }
736  if (points.size() != 2) {
737  throw Base::IndexError("CenterLine wrong number of points.");
738  }

741  Base::Vector3d v2 = DU::invertY(points.back()->point());

743  Base::Vector3d dir = v2 - v1;
744 
745  // if the proposed end points prevent the creation of a vertical or horizontal centerline, we need
746  // to prevent the "orientation" code below from creating identical endpoints. This would create a
747  // zero length edge and cause problems later.
748

750  bool inhibitHorizontal = false;

752  // proposed end points are aligned horizontally, can not draw horizontal CL
753  inhibitHorizontal = true;
754  }
755  if (DU::fpCompare(v1.x, v2.x, EWTOLERANCE)) {
756  // proposed end points are aligned vertically, can not draw vertical CL
757  inhibitVertical = true;
758  }
759 
760  //orientation
761  if (partFeat->Rotation.getValue() == 0.0) {
762  // if the view is rotated, then horizontal and vertical lose their meaning
763  if (mode == CenterLine::VERTICAL && !inhibitVertical) {
764  //Vertical
765  v1.x = mid.x;
766  v2.x = mid.x;
767  } else if (mode == CenterLine::HORIZONTAL && !inhibitHorizontal) {
768  //Horizontal
769  v1.y = mid.y;
770  v2.y = mid.y;
771  } else if (mode == CenterLine::ALIGNED) { //Aligned
772  // no op
773  }
774  }
775 
776  double length = dir.Length();
777  dir.Normalize();
778  Base::Vector3d clDir(dir.y, -dir.x, dir.z);
779 
780  Base::Vector3d p1 = mid + clDir * (length / 2.0);
781  Base::Vector3d p2 = mid - clDir * (length / 2.0);
782 
783  if (flip) { //is flip relevant to 2 point???
784  Base::Vector3d temp = p1;
785  p1 = p2;
786  p2 = temp;
787  }
788 
789  //extend
790  p1 = p1 + (clDir * ext);
791  p2 = p2 - (clDir * ext);
792 
793  //rotate
794  if (!DrawUtil::fpCompare(rotate, 0.0)) {
795  //rotate p1, p2 about mid
796  std::tie(p1, p2) = rotatePointsAroundMid(p1, p2, mid, rotate);
797  }
798 
799  //shift
800  if (!DrawUtil::fpCompare(hShift, 0.0)) {
801  double hss = hShift * scale;
802  p1.x = p1.x + hss;
803  p2.x = p2.x + hss;
804  }
805  if (!DrawUtil::fpCompare(vShift, 0.0)) {
806  double vss = vShift * scale;
807  p1.y = p1.y + vss;
808  p2.y = p2.y + vss;
809  }
810 
811  std::pair<Base::Vector3d, Base::Vector3d> result;
812  result.first = p1 / scale;
813  result.second = p2 / scale;
814 
815  return result;
816 }
