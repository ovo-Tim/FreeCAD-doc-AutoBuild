42 // A helper function that traverses all child elements recursively and check for
43 // elements of name "text" with an attribute "freecad:editable"
44 // If the query string contains "tspan" the first sub-element is used or the
45 // found text element otherwise.
46 static bool processElements(const QDomElement& element, const QString& queryStr,
47  const std::function<bool(QDomElement&)>& process)
48 {
49  bool find_tspan = queryStr.contains(QLatin1String("tspan"));
50  QDomNodeList editable = element.elementsByTagName(QString(QLatin1String("text")));
51  if (editable.count() > 0) {
52  for(int i = 0; i < editable.count(); i++) {
53  QDomNode node = editable.item(i);
54  QDomElement element = node.toElement();
55  if (element.hasAttribute(QString(QLatin1String(FREECAD_ATTR_EDITABLE)))) {
56  if (find_tspan) {
57  element = element.firstChildElement();
58  }
59 
60  if (!process(element)) {
61  return false;
62  }
63  }
64  }
65  }
66  else {
67  QDomElement child;
68  child = element.firstChildElement();
69  while (!child.isNull()) {
70  if (!processElements(child, queryStr, process)) {
71  return false;
72  }
73  child = child.nextSiblingElement();
74  }
75  }
76 
77  return true;
78 }
