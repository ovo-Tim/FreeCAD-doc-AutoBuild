83  void updateDataAndDrawToPosition(Base::Vector2d onSketchPos) override
84  {
85  switch (state()) {
86  case SelectMode::SeekFirst: {
87  toolWidgetManager.drawPositionAtCursor(onSketchPos);
88 
89  startPoint = onSketchPos;
90 
91  seekAndRenderAutoConstraint(sugConstraints[0],
92  onSketchPos,
93  Base::Vector2d(0.f, 0.f));
94  } break;
95  case SelectMode::SeekSecond: {
96  toolWidgetManager.drawDirectionAtCursor(onSketchPos, startPoint);
97 
98  secondPoint = onSketchPos;
99  angle = (secondPoint - startPoint).Angle();
100  checkHorizontalVertical();
101  length = (secondPoint - startPoint).Length();
102  const double scale = 0.2;
103  radius = length * scale; // radius chosen at 1/5 of length
104 
105  CreateAndDrawShapeGeometry();
106 
107  seekAndRenderAutoConstraint(sugConstraints[1],
108  onSketchPos,
109  secondPoint - startPoint,
110  AutoConstraint::VERTEX_NO_TANGENCY);
111  } break;
112  case SelectMode::SeekThird: {
113  /*To follow the cursor, r should adapt depending on the position of the cursor. If
114  cursor is 'between' the center points, then its distance to that line and not
115  distance to the second center. A is "between" B and C if angle ?ABC and angle ?ACB
116  are both less than or equal to ninety degrees. An angle ?ABC is greater than ninety
117  degrees iff AB^2 + BC^2 < AC^2.*/
118 
119  double L1 = (onSketchPos - startPoint)
120  .Length(); // distance between first center and onSketchPos
121  double L2 = (onSketchPos - secondPoint)
122  .Length(); // distance between second center and onSketchPos
123 
124  if ((L1 * L1 + length * length > L2 * L2)
125  && (L2 * L2 + length * length > L1 * L1)) {
126  // distance of onSketchPos to the line StartPos-SecondPos
127  radius = (abs((secondPoint.y - startPoint.y) * onSketchPos.x
128  - (secondPoint.x - startPoint.x) * onSketchPos.y
129  + secondPoint.x * startPoint.y - secondPoint.y * startPoint.x))
130  / length;
131  }
132  else {
133  radius = std::min(L1, L2);
134  }
135 
136  toolWidgetManager.drawDoubleAtCursor(onSketchPos, radius);
137 
138  CreateAndDrawShapeGeometry();
139  } break;
140  default:
141  break;
142  }
143  }

242  void createShape(bool onlyeditoutline) override
243  {
244  ShapeGeometry.clear();
245 
246  if (length < Precision::Confusion() || radius < Precision::Confusion()) {
247  return;
248  }
249 
250  Part::GeomArcOfCircle* arc1 = addArcToShapeGeometry(toVector3d(startPoint),
251  M_PI / 2 + angle,
252  1.5 * M_PI + angle,
253  radius,
254  isConstructionMode());
255 
256  Part::GeomArcOfCircle* arc2 = addArcToShapeGeometry(toVector3d(secondPoint),
257  1.5 * M_PI + angle,
258  M_PI / 2 + angle,
259  radius,
260  isConstructionMode());
261 
262  Base::Vector3d p11 = arc1->getStartPoint();
263  Base::Vector3d p12 = arc1->getEndPoint();
264  Base::Vector3d p21 = arc2->getStartPoint();
265  Base::Vector3d p22 = arc2->getEndPoint();
266 
267  addLineToShapeGeometry(p11, p22, isConstructionMode());
268 
269  addLineToShapeGeometry(p12, p21, isConstructionMode());
270 
271  if (!onlyeditoutline) {
272  addToShapeConstraints(Sketcher::Tangent,
273  firstCurve,
274  Sketcher::PointPos::start,
275  firstCurve + 2,
276  Sketcher::PointPos::start);
277  addToShapeConstraints(Sketcher::Tangent,
278  firstCurve,
279  Sketcher::PointPos::end,
280  firstCurve + 3,
281  Sketcher::PointPos::start);
282  addToShapeConstraints(Sketcher::Tangent,
283  firstCurve + 1,
284  Sketcher::PointPos::end,
285  firstCurve + 2,
286  Sketcher::PointPos::end);
287  addToShapeConstraints(Sketcher::Tangent,
288  firstCurve + 1,
289  Sketcher::PointPos::start,
290  firstCurve + 3,
291  Sketcher::PointPos::end);
292  addToShapeConstraints(Sketcher::Equal,
293  firstCurve,
294  Sketcher::PointPos::none,
295  firstCurve + 1);
296 
297  // Prevent duplicate with Autocontraint
298  AutoConstraint lastCons = {Sketcher::None,
299  Sketcher::GeoEnum::GeoUndef,
300  Sketcher::PointPos::none};
301  if (!sugConstraints[1].empty()) {
302  lastCons = sugConstraints[1].back();
303  }
304 
305  if (isHorizontal || isVertical) {
306  addToShapeConstraints(isHorizontal ? Sketcher::Horizontal : Sketcher::Vertical,
307  firstCurve + 3);
308 
309 
310  if (lastCons.Type == Sketcher::Horizontal || lastCons.Type == Sketcher::Vertical) {
311  sugConstraints[1].pop_back();
312  }
313  }
314  else {
315  // If horizontal/vertical Autoconstraint suggested, applied it on first line
316  // (rather than last arc)
317  if (lastCons.Type == Sketcher::Horizontal || lastCons.Type == Sketcher::Vertical) {
318  sugConstraints[1].back().GeoId = firstCurve + 2;
319  }
320  }
321  }
322  }

324  void checkHorizontalVertical()
325  {
326  isHorizontal = false;
327  isVertical = false;
328 
329  if (fmod(fabs(angle), M_PI) < Precision::Confusion()) {
330  isHorizontal = true;
331  }
332  else if (fmod(fabs(angle + M_PI / 2), M_PI) < Precision::Confusion()) {
333  isVertical = true;
334  }
335  }

380 template<>
381 void DSHSlotControllerBase::doEnforceControlParameters(Base::Vector2d& onSketchPos)
382 {
383  switch (handler->state()) {
384  case SelectMode::SeekFirst: {
385  if (onViewParameters[OnViewParameter::First]->isSet) {
386  onSketchPos.x = onViewParameters[OnViewParameter::First]->getValue();
387  }
388 
389  if (onViewParameters[OnViewParameter::Second]->isSet) {
390  onSketchPos.y = onViewParameters[OnViewParameter::Second]->getValue();
391  }
392  } break;
393  case SelectMode::SeekSecond: {
394  Base::Vector2d dir = onSketchPos - handler->startPoint;
395  if (dir.Length() < Precision::Confusion()) {
396  dir.x = 1.0; // if direction null, default to (1,0)
397  }
398  double length = dir.Length();
399 
400  if (onViewParameters[OnViewParameter::Third]->isSet) {
401  length = onViewParameters[OnViewParameter::Third]->getValue();
402  if (length < Precision::Confusion()) {
403  unsetOnViewParameter(onViewParameters[OnViewParameter::Third].get());
404  return;
405  }
406 
407  onSketchPos = handler->startPoint + length * dir.Normalize();
408  }
409 
410  if (onViewParameters[OnViewParameter::Fourth]->isSet) {
411  double angle =
412  Base::toRadians(onViewParameters[OnViewParameter::Fourth]->getValue());
413  onSketchPos.x = handler->startPoint.x + cos(angle) * length;
414  onSketchPos.y = handler->startPoint.y + sin(angle) * length;
415  }
416  } break;
417  case SelectMode::SeekThird: {
418  if (onViewParameters[OnViewParameter::Fifth]->isSet) {
419  double radius = onViewParameters[OnViewParameter::Fifth]->getValue();
420 
421  if (radius < Precision::Confusion()) {
422  unsetOnViewParameter(onViewParameters[OnViewParameter::Fifth].get());
423  return;
424  }
425 
426  onSketchPos.x = handler->secondPoint.x + cos(handler->angle) * radius;
427  onSketchPos.y = handler->secondPoint.y + sin(handler->angle) * radius;
428  }
429  } break;
430  default:
431  break;
432  }
433 }

435 template<>
436 void DSHSlotController::adaptParameters(Base::Vector2d onSketchPos)
437 {
438  switch (handler->state()) {
439  case SelectMode::SeekFirst: {
440  if (!onViewParameters[OnViewParameter::First]->isSet) {
441  setOnViewParameterValue(OnViewParameter::First, onSketchPos.x);
442  }
443 
444  if (!onViewParameters[OnViewParameter::Second]->isSet) {
445  setOnViewParameterValue(OnViewParameter::Second, onSketchPos.y);
446  }
447 
448  bool sameSign = onSketchPos.x * onSketchPos.y > 0.;
449  onViewParameters[OnViewParameter::First]->setLabelAutoDistanceReverse(!sameSign);
450  onViewParameters[OnViewParameter::Second]->setLabelAutoDistanceReverse(sameSign);
451  onViewParameters[OnViewParameter::First]->setPoints(Base::Vector3d(),
452  toVector3d(onSketchPos));
453  onViewParameters[OnViewParameter::Second]->setPoints(Base::Vector3d(),
454  toVector3d(onSketchPos));
455  } break;
456  case SelectMode::SeekSecond: {
457  Base::Vector3d start = toVector3d(handler->startPoint);
458  Base::Vector3d end = toVector3d(handler->secondPoint);
459  Base::Vector3d vec = end - start;
460 
461  if (!onViewParameters[OnViewParameter::Third]->isSet) {
462  setOnViewParameterValue(OnViewParameter::Third, vec.Length());
463  }
464 
465  double range = (handler->secondPoint - handler->startPoint).Angle();
466  if (!onViewParameters[OnViewParameter::Fourth]->isSet) {
467  setOnViewParameterValue(OnViewParameter::Fourth,
468  Base::toDegrees(range),
469  Base::Unit::Angle);
470  }
471 
472  onViewParameters[OnViewParameter::Third]->setPoints(start, end);
473  onViewParameters[OnViewParameter::Fourth]->setPoints(start, Base::Vector3d());
474  onViewParameters[OnViewParameter::Fourth]->setLabelRange(range);
475  } break;
476  case SelectMode::SeekThird: {
477  if (!onViewParameters[OnViewParameter::Fifth]->isSet) {
478  setOnViewParameterValue(OnViewParameter::Fifth, handler->radius);
479  }
480 
481  Base::Vector3d labelSecondPoint = Base::Vector3d();
482  labelSecondPoint.x = handler->secondPoint.x + cos(handler->angle) * handler->radius;
483  labelSecondPoint.y = handler->secondPoint.y + sin(handler->angle) * handler->radius;
484 
485  onViewParameters[OnViewParameter::Fifth]->setPoints(toVector3d(handler->secondPoint),
486  labelSecondPoint);
487 
488  } break;
489  default:
490  break;
491  }

523 template<>
524 void DSHSlotController::addConstraints()
525 {
526  App::DocumentObject* obj = handler->sketchgui->getObject();
527 
528  int firstCurve = handler->getHighestCurveIndex() - 3;
529 
530  auto x0 = onViewParameters[OnViewParameter::First]->getValue();
531  auto y0 = onViewParameters[OnViewParameter::Second]->getValue();
532 
533  auto x0set = onViewParameters[OnViewParameter::First]->isSet;
534  auto y0set = onViewParameters[OnViewParameter::Second]->isSet;
535  auto lengthSet = onViewParameters[OnViewParameter::Third]->isSet;
536  auto angleSet = onViewParameters[OnViewParameter::Fourth]->isSet;
537  auto radiusSet = onViewParameters[OnViewParameter::Fifth]->isSet;
538 
539  using namespace Sketcher;
540 
541  auto constraintToOrigin = [&]() {
542  ConstraintToAttachment(GeoElementId(firstCurve, PointPos::mid),
543  GeoElementId::RtPnt,
544  x0,
545  obj);
546  };
547 
548  auto constraintx0 = [&]() {
549  ConstraintToAttachment(GeoElementId(firstCurve, PointPos::mid),
550  GeoElementId::VAxis,
551  x0,
552  obj);
553  };
554 
555  auto constrainty0 = [&]() {
556  ConstraintToAttachment(GeoElementId(firstCurve, PointPos::mid),
557  GeoElementId::HAxis,
558  y0,
559  obj);
560  };
561 
562  auto constraintLength = [&]() {
563  Gui::cmdAppObjectArgs(obj,
564  "addConstraint(Sketcher.Constraint('Distance',%d,%d,%d,%d,%f)) ",
565  firstCurve,
566  3,
567  firstCurve + 1,
568  3,
569  handler->length);
570  };
571 
572  auto constraintAngle = [&]() {
573  if (!handler->isHorizontal && !handler->isVertical) {
574  Gui::cmdAppObjectArgs(obj,
575  "addConstraint(Sketcher.Constraint('Angle',%d,%d,%f)) ",
576  Sketcher::GeoEnum::HAxis,
577  firstCurve + 2,
578  handler->angle);
579 
580  // Prevent duplicate with Autocontraint
581  AutoConstraint lastCons = {Sketcher::None,
582  Sketcher::GeoEnum::GeoUndef,
583  Sketcher::PointPos::none};
584  if (!handler->sugConstraints[1].empty()) {
585  lastCons = handler->sugConstraints[1].back();
586  if (lastCons.Type == Sketcher::Horizontal || lastCons.Type == Sketcher::Vertical) {
587  handler->AutoConstraints.pop_back();
588  }
589  }
590  }
591  };
592 
593  if (handler->AutoConstraints.empty()) { // No valid diagnosis. Every constraint can be added.
594 
595  if (x0set && y0set && x0 == 0. && y0 == 0.) {
596  constraintToOrigin();
597  }
598  else {
599  if (x0set) {
600  constraintx0();
601  }
602 
603  if (y0set) {
604  constrainty0();
605  }
606  }
607 
608  if (lengthSet) {
609  constraintLength();
610  }
611 
612  if (angleSet) {
613  constraintAngle();
614  }
615  }
616  else { // Valid diagnosis. Must check which constraints may be added.
617  auto startpointinfo = handler->getPointInfo(GeoElementId(firstCurve, PointPos::mid));
618 
619  if (x0set && startpointinfo.isXDoF()) {
620  constraintx0();
621 
622  handler->diagnoseWithAutoConstraints(); // ensure we have recalculated parameters after
623  // each constraint addition
624 
625  startpointinfo = handler->getPointInfo(
626  GeoElementId(firstCurve, PointPos::start)); // get updated point position
627  }
628 
629  if (y0set && startpointinfo.isYDoF()) {
630  constrainty0();
631 
632  handler->diagnoseWithAutoConstraints(); // ensure we have recalculated parameters after
633  // each constraint addition
634 
635  startpointinfo = handler->getPointInfo(
636  GeoElementId(firstCurve, PointPos::start)); // get updated point position
637  }
638 
639  auto endpointinfo = handler->getPointInfo(GeoElementId(firstCurve + 1, PointPos::mid));
640 
641  int DoFs = startpointinfo.getDoFs();
642  DoFs += endpointinfo.getDoFs();
643 
644  if (lengthSet && DoFs > 0) {
645  constraintLength();
646  DoFs--;
647  }
648 
649  if (angleSet && DoFs > 0) {
650  constraintAngle();
651  }
652  }
653 
654  // No auto constraint in seekThird.
655  if (radiusSet) {
656  Gui::cmdAppObjectArgs(obj,
657  "addConstraint(Sketcher.Constraint('Radius',%d,%f)) ",
658  firstCurve,
659  handler->radius);
660  }
661 }
