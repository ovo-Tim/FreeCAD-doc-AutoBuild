95 static inline void drawBorder(QPainter* painter,
96  const QStyleOptionViewItem& option,
97  unsigned flags,
98  QColor color,
99  Qt::PenStyle style)
100 {
101  if (!flags) {
102  return;
103  }
104  QPen pen(color);
105  pen.setWidth(2);
106  pen.setStyle(style);
107  painter->setPen(pen);
108 
109  QRect rect = option.rect.adjusted(1, 1, 0, 0);
110  if (flags == Sheet::BorderAll) {
111  painter->drawRect(rect.adjusted(0, 0, -1, -1));
112  return;
113  }
114  if (flags & Sheet::BorderLeft) {
115  painter->drawLine(rect.topLeft(), rect.bottomLeft());
116  }
117  if (flags & Sheet::BorderTop) {
118  painter->drawLine(rect.topLeft(), rect.topRight());
119  }
120  if (flags & Sheet::BorderRight) {
121  painter->drawLine(rect.topRight(), rect.bottomRight());
122  }
123  if (flags & Sheet::BorderBottom) {
124  painter->drawLine(rect.bottomLeft(), rect.bottomRight());
125  }
126 }
