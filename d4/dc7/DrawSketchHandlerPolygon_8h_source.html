231  void createShape(bool onlyeditoutline) override
232  {
233  Q_UNUSED(onlyeditoutline)
234 
235  ShapeGeometry.clear();
236 
237  Base::Vector2d prevCorner = firstCorner;
238  Base::Vector2d dV = firstCorner - centerPoint;
239  radius = dV.Length();
240 
241  if (radius < Precision::Confusion()) {
242  return;
243  }
244 
245  double angleOfSeparation = 2.0 * M_PI / static_cast<double>(numberOfCorners); // NOLINT
246  double cos_v = cos(angleOfSeparation);
247  double sin_v = sin(angleOfSeparation);
248 
249  double rx = dV.x;
250  double ry = dV.y;
251 
252  for (int i = 1; i <= static_cast<int>(numberOfCorners); i++) {
253  const double old_rx = rx;
254  rx = cos_v * rx - sin_v * ry;
255  ry = cos_v * ry + sin_v * old_rx;
256  Base::Vector2d newCorner = Base::Vector2d(centerPoint.x + rx, centerPoint.y + ry);
257  addLineToShapeGeometry(toVector3d(prevCorner),
258  toVector3d(newCorner),
259  isConstructionMode());
260  prevCorner = newCorner;
261  }
262  }

333 template<>
334 void DSHPolygonControllerBase::doEnforceControlParameters(Base::Vector2d& onSketchPos)
335 {
336  switch (handler->state()) {
337  case SelectMode::SeekFirst: {
338  if (onViewParameters[OnViewParameter::First]->isSet) {
339  onSketchPos.x = onViewParameters[OnViewParameter::First]->getValue();
340  }
341 
342  if (onViewParameters[OnViewParameter::Second]->isSet) {
343  onSketchPos.y = onViewParameters[OnViewParameter::Second]->getValue();
344  }
345  } break;
346  case SelectMode::SeekSecond: {
347  Base::Vector2d dir = onSketchPos - handler->centerPoint;
348  if (dir.Length() < Precision::Confusion()) {
349  dir.x = 1.0; // if direction null, default to (1,0)
350  }
351  double length = dir.Length();
352 
353  if (onViewParameters[OnViewParameter::Third]->isSet) {
354  length = onViewParameters[OnViewParameter::Third]->getValue();
355  if (length < Precision::Confusion()) {
356  unsetOnViewParameter(onViewParameters[OnViewParameter::Third].get());
357  return;
358  }
359 
360  onSketchPos = handler->centerPoint + length * dir.Normalize();
361  }
362 
363  if (onViewParameters[OnViewParameter::Fourth]->isSet) {
364  double angle =
365  Base::toRadians(onViewParameters[OnViewParameter::Fourth]->getValue());
366  onSketchPos.x = handler->centerPoint.x + cos(angle) * length;
367  onSketchPos.y = handler->centerPoint.y + sin(angle) * length;
368  }
369  } break;
370  default:
371  break;
372  }
373 }

376 void DSHPolygonController::adaptParameters(Base::Vector2d onSketchPos)
377 {
378  switch (handler->state()) {
379  case SelectMode::SeekFirst: {
380  if (!onViewParameters[OnViewParameter::First]->isSet) {
381  setOnViewParameterValue(OnViewParameter::First, onSketchPos.x);
382  }
383 
384  if (!onViewParameters[OnViewParameter::Second]->isSet) {
385  setOnViewParameterValue(OnViewParameter::Second, onSketchPos.y);
386  }
387 
388  bool sameSign = onSketchPos.x * onSketchPos.y > 0.;
389  onViewParameters[OnViewParameter::First]->setLabelAutoDistanceReverse(!sameSign);
390  onViewParameters[OnViewParameter::Second]->setLabelAutoDistanceReverse(sameSign);
391  onViewParameters[OnViewParameter::First]->setPoints(Base::Vector3d(),
392  toVector3d(onSketchPos));
393  onViewParameters[OnViewParameter::Second]->setPoints(Base::Vector3d(),
394  toVector3d(onSketchPos));
395  } break;
396  case SelectMode::SeekSecond: {
397  Base::Vector3d start = toVector3d(handler->centerPoint);
398  Base::Vector3d end = toVector3d(handler->firstCorner);
399  Base::Vector3d vec = end - start;
400 
401  if (!onViewParameters[OnViewParameter::Third]->isSet) {
402  setOnViewParameterValue(OnViewParameter::Third, vec.Length());
403  }
404 
405  double range = (handler->firstCorner - handler->centerPoint).Angle();
406  if (!onViewParameters[OnViewParameter::Fourth]->isSet) {
407  setOnViewParameterValue(OnViewParameter::Fourth,
408  Base::toDegrees(range),
409  Base::Unit::Angle);
410  }
411 
412  onViewParameters[OnViewParameter::Third]->setPoints(start, end);
413  onViewParameters[OnViewParameter::Fourth]->setPoints(start, Base::Vector3d());
414  onViewParameters[OnViewParameter::Fourth]->setLabelRange(range);
415 
416  } break;
417  default:
418  break;
419  }

445 template<>
446 void DSHPolygonController::addConstraints()
447 {
448  int lastCurve = handler->getHighestCurveIndex();
449 
450  auto x0 = onViewParameters[OnViewParameter::First]->getValue();
451  auto y0 = onViewParameters[OnViewParameter::Second]->getValue();
452  auto radius = onViewParameters[OnViewParameter::Third]->getValue();
453 
454  auto x0set = onViewParameters[OnViewParameter::First]->isSet;
455  auto y0set = onViewParameters[OnViewParameter::Second]->isSet;
456  auto radiusSet = onViewParameters[OnViewParameter::Third]->isSet;
457 
458  using namespace Sketcher;
459 
460  auto constraintx0 = [&]() {
461  ConstraintToAttachment(GeoElementId(lastCurve, PointPos::mid),
462  GeoElementId::VAxis,
463  x0,
464  handler->sketchgui->getObject());
465  };
466 
467  auto constrainty0 = [&]() {
468  ConstraintToAttachment(GeoElementId(lastCurve, PointPos::mid),
469  GeoElementId::HAxis,
470  y0,
471  handler->sketchgui->getObject());
472  };
473 
474  auto constraintradius = [&]() {
475  Gui::cmdAppObjectArgs(handler->sketchgui->getObject(),
476  "addConstraint(Sketcher.Constraint('Radius',%d,%f)) ",
477  lastCurve,
478  radius);
479  };
480 
481  // NOTE: if AutoConstraints is empty, we can add constraints directly without any diagnose. No
482  // diagnose was run.
483  if (handler->AutoConstraints.empty()) {
484  if (x0set) {
485  constraintx0();
486  }
487 
488  if (y0set) {
489  constrainty0();
490  }
491 
492  if (radiusSet) {
493  constraintradius();
494  }
495  }
496  else { // There is a valid diagnose.
497  auto startpointinfo = handler->getPointInfo(GeoElementId(lastCurve, PointPos::mid));
498 
499  // if Autoconstraints is empty we do not have a diagnosed system and the parameter will
500  // always be set
501  if (x0set && startpointinfo.isXDoF()) {
502  constraintx0();
503 
504  handler->diagnoseWithAutoConstraints(); // ensure we have recalculated parameters after
505  // each constraint addition
506 
507  startpointinfo = handler->getPointInfo(
508  GeoElementId(lastCurve, PointPos::mid)); // get updated point position
509  }
510 
511  // if Autoconstraints is empty we do not have a diagnosed system and the parameter will
512  // always be set
513  if (y0set && startpointinfo.isYDoF()) {
514  constrainty0();
515 
516  handler->diagnoseWithAutoConstraints(); // ensure we have recalculated parameters after
517  // each constraint addition
518  }
519 
520  auto edgeinfo = handler->getEdgeInfo(lastCurve);
521  auto circle = static_cast<SolverGeometryExtension::Circle&>(edgeinfo);
522 
523  // if Autoconstraints is empty we do not have a diagnosed system and the parameter will
524  // always be set
525  if (radiusSet && circle.isRadiusDoF()) {
526  constraintradius();
527  }
528  }
529 }
