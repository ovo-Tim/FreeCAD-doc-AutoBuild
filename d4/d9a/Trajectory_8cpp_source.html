149 void Trajectory::generateTrajectory()
150 {
151  if (vpcWaypoints.empty()) {
152  return;
153  }
154 
155  // delete the old and create a new one
156  if (pcTrajectory) {
157  delete (pcTrajectory);
158  }
159  pcTrajectory = new KDL::Trajectory_Composite();
160 
161  // pointer to the pieces while iterating
162  std::unique_ptr<KDL::Trajectory_Segment> pcTrak;
163  std::unique_ptr<KDL::VelocityProfile> pcVelPrf;
164  std::unique_ptr<KDL::Path_RoundedComposite> pcRoundComp;
165  KDL::Frame Last;
166 
167  try {
168  // handle the first waypoint special
169  bool first = true;
170 
171  for (std::vector<Waypoint*>::const_iterator it = vpcWaypoints.begin();
172  it != vpcWaypoints.end();
173  ++it) {
174  if (first) {
175  Last = toFrame((*it)->EndPos);
176  first = false;
177  }
178  else {
179  // destinct the type of movement
180  switch ((*it)->Type) {
181  case Waypoint::LINE:
182  case Waypoint::PTP: {
183  KDL::Frame Next = toFrame((*it)->EndPos);
184  // continues the movement until no continuous waypoint or the end
185  bool Cont = (*it)->Cont && !(it == --vpcWaypoints.end());
186  // start of a continue block
187  if (Cont && !pcRoundComp) {
188  pcRoundComp = std::make_unique<KDL::Path_RoundedComposite>(
189  3,
190  3,
191  new KDL::RotationalInterpolation_SingleAxis());
192  // the velocity of the first waypoint is used
193  pcVelPrf =
194  std::make_unique<KDL::VelocityProfile_Trap>((*it)->Velocity,
195  (*it)->Acceleration);
196  pcRoundComp->Add(Last);
197  pcRoundComp->Add(Next);
198 
199  // continue a continues block
200  }
201  else if (Cont && pcRoundComp) {
202  pcRoundComp->Add(Next);
203  // end a continuous block
204  }
205  else if (!Cont && pcRoundComp) {
206  // add the last one
207  pcRoundComp->Add(Next);
208  pcRoundComp->Finish();
209  pcVelPrf->SetProfile(0, pcRoundComp->PathLength());
210  pcTrak =
211  std::make_unique<KDL::Trajectory_Segment>(pcRoundComp.release(),
212  pcVelPrf.release());
213 
214  // normal block
215  }
216  else if (!Cont && !pcRoundComp) {
217  KDL::Path* pcPath;
218  pcPath =
219  new KDL::Path_Line(Last,
220  Next,
221  new KDL::RotationalInterpolation_SingleAxis(),
222  1.0,
223  true);
224 
225  pcVelPrf =
226  std::make_unique<KDL::VelocityProfile_Trap>((*it)->Velocity,
227  (*it)->Acceleration);
228  pcVelPrf->SetProfile(0, pcPath->PathLength());
229  pcTrak = std::make_unique<KDL::Trajectory_Segment>(pcPath,
230  pcVelPrf.release());
231  }
232  Last = Next;
233  break;
234  }
235  case Waypoint::WAIT:
236  break;
237  default:
238  break;
239  }
240 
241  // add the segment if no continuous block is running
242  if (!pcRoundComp && pcTrak) {
243  pcTrajectory->Add(pcTrak.release());
244  }
245  }
246  }
247  }
248  catch (KDL::Error& e) {
249  throw Base::RuntimeError(e.Description());
250  }
251 }

253 std::string Trajectory::getUniqueWaypointName(const char* Name) const
254 {
255  if (!Name || *Name == '\0') {
256  return {};
257  }
258 
259  // check for first character whether it's a digit
260  std::string CleanName = Name;
261  if (!CleanName.empty() && CleanName[0] >= 48 && CleanName[0] <= 57) {
262  CleanName[0] = '_';
263  }
264  // strip illegal chars
265  for (char& it : CleanName) {
266  if (!((it >= 48 && it <= 57) || // number
267  (it >= 65 && it <= 90) || // uppercase letter
268  (it >= 97 && it <= 122))) { // lowercase letter
269  it = '_'; // it's neither number nor letter
270  }
271  }
272 
273  // name in use?
274  std::vector<Robot::Waypoint*>::const_iterator it;
275  for (it = vpcWaypoints.begin(); it != vpcWaypoints.end(); ++it) {
276  if ((*it)->Name == CleanName) {
277  break;
278  }
279  }
280 
281  if (it == vpcWaypoints.end()) {
282  // if not, name is OK
283  return CleanName;
284  }
285  else {
286  // find highest suffix
287  int nSuff = 0;
288  for (it = vpcWaypoints.begin(); it != vpcWaypoints.end(); ++it) {
289  const std::string& ObjName = (*it)->Name;
290  if (ObjName.substr(0, CleanName.length()) == CleanName) { // same prefix
291  std::string clSuffix(ObjName.substr(CleanName.length()));
292  if (!clSuffix.empty()) {
293  std::string::size_type nPos = clSuffix.find_first_not_of("0123456789");
294  if (nPos == std::string::npos) {
295  nSuff = std::max<int>(nSuff, std::atol(clSuffix.c_str()));
296  }
297  }
298  }
299  }
300 
301  std::stringstream str;
302  str << CleanName << (nSuff + 1);
303  return str.str();
304  }
305 }
