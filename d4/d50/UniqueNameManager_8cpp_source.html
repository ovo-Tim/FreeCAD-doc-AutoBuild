32 void Base::UniqueNameManager::PiecewiseSparseIntegerSet::add(unsigned int value)
33 {
34  SpanType newSpan(value, 1);
35  // Look for the smallest entry not less than newSpan.
36  // Bear in mind that overlapping spans are neither less than nor greater than ech other.
37  auto above = spans.lower_bound(newSpan);
38  if (above != spans.end() && above->first <= value) {
39  // A span was found that includes value so there is nothing to do as it is already in the
40  // set.
41  return;
42  }
43 
44  // Set below to the next span below 'after', if any, otherwise to spans.end().
45  // Logically, we want spans.begin()-1 so 'below' is always the entry before 'after',
46  // but that is not an allowed reference so spans.end() is used
47  std::set<SpanType, Comparer>::iterator below;
48  if (above == spans.begin()) {
49  // No spans are less than newSpan
50  // (possibly spans is empty, in which case above == spans.end() too)
51  below = spans.end();
52  }
53  else {
54  // At least one span is less than newSpan,
55  // and 'above' is the next span above that
56  // (or above == spans.end() if all spans are below newSpan)
57  below = above;
58  --below;
59  }
60 
61  // Determine whether the span above (if any) and/or the span below (if any)
62  // are adjacent to newSpan and if so, merge them appropriately and remove the
63  // original span(s) that was/were merged, updating newSpan to be the new merged
64  // span.
65  if (above != spans.end() && below != spans.end()
66  && above->first - below->first + 1 == below->second) {
67  // below and above have a gap of exactly one between them, and this must be value
68  // so we coalesce the two spans (and the gap) into one.
69  newSpan = SpanType(below->first, below->second + above->second + 1);
70  spans.erase(above);
71  above = spans.erase(below);
72  }
73  else if (below != spans.end() && value - below->first == below->second) {
74  // value is adjacent to the end of below, so just expand below by one
75  newSpan = SpanType(below->first, below->second + 1);
76  above = spans.erase(below);
77  }
78  else if (above != spans.end() && above->first - value == 1) {
79  // value is adjacent to the start of above, so just expand above down by one
80  newSpan = SpanType(above->first - 1, above->second + 1);
81  above = spans.erase(above);
82  }
83  // else value is not adjacent to any existing span, so just make anew span for it
84  spans.insert(above, newSpan);
85 }
86 void Base::UniqueNameManager::PiecewiseSparseIntegerSet::remove(unsigned int value)
87 {
88  SpanType newSpan(value, 1);
89  auto at = spans.lower_bound(newSpan);
90  if (at == spans.end() || at->first > value) {
91  // The found span does not include value so there is nothing to do, as it is already not in
92  // the set.
93  return;
94  }
95  if (at->second == 1) {
96  // value is the only in this span, just remove the span
97  spans.erase(at);
98  }
99  else if (at->first == value) {
100  // value is the first in this span, trim the lower end
101  SpanType replacement(at->first + 1, at->second - 1);
102  spans.insert(spans.erase(at), replacement);
103  }
104  else if (value - at->first == at->second - 1) {
105  // value is the last in this span, trim the upper end
106  SpanType replacement(at->first, at->second - 1);
107  spans.insert(spans.erase(at), replacement);
108  }
109  else {
110  // value is in the moddle of the span, so we must split it.
111  SpanType firstReplacement(at->first, value - at->first);
112  SpanType secondReplacement(value + 1, at->second - ((value + 1) - at->first));
113  // Because erase returns the iterator after the erased element, and insert returns the
114  // iterator for the inserted item, we want to insert secondReplacement first.
115  spans.insert(spans.insert(spans.erase(at), secondReplacement), firstReplacement);
116  }
117 }

124 std::tuple<std::string, std::string, unsigned int, unsigned int>
125 Base::UniqueNameManager::decomposeName(const std::string& name) const
126 {
127  auto suffixStart = getNameSuffixStartPosition(name);
128  auto digitsStart = std::find_if_not(suffixStart, name.crend(), [](char c) {
129  return std::isdigit(c);
130  });
131  unsigned int digitCount = digitsStart - suffixStart;
132  return std::tuple<std::string, std::string, unsigned int, unsigned int> {
133  name.substr(0, name.crend() - digitsStart),
134  name.substr(name.crend() - suffixStart),
135  digitCount,
136  digitCount == 0 ? 0U : std::stoul(name.substr(name.crend() - digitsStart, digitCount))};
137 }
138 bool Base::UniqueNameManager::sameBaseName(const std::string& first,
139  const std::string& second) const
140 {
141  auto firstSuffixStart = getNameSuffixStartPosition(first);
142  auto secondSuffixStart = getNameSuffixStartPosition(second);
143  if (firstSuffixStart - first.crbegin() != secondSuffixStart - second.crbegin()) {
144  // The suffixes are different lengths
145  return false;
146  }
147  auto firstDigitsStart = std::find_if_not(firstSuffixStart, first.crend(), [](char c) {
148  return std::isdigit(c);
149  });
150  auto secondDigitsStart = std::find_if_not(secondSuffixStart, second.crend(), [](char c) {
151  return std::isdigit(c);
152  });
153  return std::equal(firstDigitsStart, first.crend(), secondDigitsStart, second.crend());
154 }

156 void Base::UniqueNameManager::addExactName(const std::string& name)
157 {
158  auto [baseName, nameSuffix, digitCount, digitsValue] = decomposeName(name);
159  baseName += nameSuffix;
160  auto baseNameEntry = uniqueSeeds.find(baseName);
161  if (baseNameEntry == uniqueSeeds.end()) {
162  // First use of baseName
163  baseNameEntry =
164  uniqueSeeds.emplace(baseName, std::vector<PiecewiseSparseIntegerSet>()).first;
165  }
166  if (digitCount >= baseNameEntry->second.size()) {
167  // First use of this digitCount
168  baseNameEntry->second.resize(digitCount + 1);
169  }
170  PiecewiseSparseIntegerSet& baseNameAndDigitCountEntry = baseNameEntry->second[digitCount];
171 
172  if (baseNameAndDigitCountEntry.contains(digitsValue)) {
173  // We already have at least one instance of the name.
174  // Increment the duplicateCounts entry for that name,
175  // making one if none is present with an initial count of 1
176  // representing the singleton element we already have.
177  duplicateCounts.try_emplace(name, 1U).first->second++;
178  return;
179  }
180  baseNameAndDigitCountEntry.add(digitsValue);
181 }
182 std::string Base::UniqueNameManager::makeUniqueName(const std::string& modelName,
183  std::size_t minDigits) const
184 {
185  auto [namePrefix, nameSuffix, digitCount, digitsValue] = decomposeName(modelName);
186  std::string baseName = namePrefix + nameSuffix;
187  auto baseNameEntry = uniqueSeeds.find(baseName);
188  if (baseNameEntry == uniqueSeeds.end()) {
189  // First use of baseName, just return it with no unique digits
190  return baseName;
191  }
192  // We don't care about the digit count or value of the suggested name,
193  // we always use at least the most digits ever used before.
194  digitCount = baseNameEntry->second.size() - 1;
195  if (digitCount < minDigits) {
196  // Caller is asking for more digits than we have in any registered name.
197  // We start the longer digit string at 000...0001 even though we might have shorter strings
198  // with larger numeric values.
199  digitCount = minDigits;
200  digitsValue = 1;
201  }
202  else {
203  digitsValue = baseNameEntry->second[digitCount].next();
204  }
205  std::string digits = std::to_string(digitsValue);
206  if (digitCount > digits.size()) {
207  namePrefix += std::string(digitCount - digits.size(), '0');
208  }
209  return namePrefix + digits + nameSuffix;
210 }

212 void Base::UniqueNameManager::removeExactName(const std::string& name)
213 {
214  auto duplicateCountFound = duplicateCounts.find(name);
215  if (duplicateCountFound != duplicateCounts.end()) {
216  // The name has duplicates. Decrement the duplicate count.
217  if (--duplicateCountFound->second <= 1) {
218  // After removal, there are no duplicates, only a single instance.
219  // Remove the duplicateCounts entry.
220  duplicateCounts.erase(duplicateCountFound);
221  }
222  return;
223  }
224  auto [baseName, nameSuffix, digitCount, digitsValue] = decomposeName(name);
225  baseName += nameSuffix;
226  auto baseNameEntry = uniqueSeeds.find(baseName);
227  if (baseNameEntry == uniqueSeeds.end()) {
228  // name must not be registered, so nothing to do.
229  return;
230  }
231  auto& digitValueSets = baseNameEntry->second;
232  if (digitCount >= digitValueSets.size()) {
233  // First use of this digitCount, name must not be registered, so nothing to do.
234  return;
235  }
236  digitValueSets[digitCount].remove(digitsValue);
237  // an element of digitValueSets may now be newly empty and so may other elements below it
238  // Prune off all such trailing empty entries.
239  auto lastNonemptyEntry =
240  std::find_if(digitValueSets.crbegin(), digitValueSets.crend(), [](auto& it) {
241  return !it.empty();
242  });
243  if (lastNonemptyEntry == digitValueSets.crend()) {
244  // All entries are empty, so the entire baseName can be forgotten.
245  uniqueSeeds.erase(baseName);
246  }
247  else {
248  digitValueSets.resize(digitValueSets.crend() - lastNonemptyEntry);
249  }
250 }

252 bool Base::UniqueNameManager::containsName(const std::string& name) const
253 {
254  if (duplicateCounts.find(name) != duplicateCounts.end()) {
255  // There are at least two instances of the name
256  return true;
257  }
258  auto [baseName, nameSuffix, digitCount, digitsValue] = decomposeName(name);
259  baseName += nameSuffix;
260  auto baseNameEntry = uniqueSeeds.find(baseName);
261  if (baseNameEntry == uniqueSeeds.end()) {
262  // base name is not registered
263  return false;
264  }
265  if (digitCount >= baseNameEntry->second.size()) {
266  // First use of this digitCount, name must not be registered, so not in collection
267  return false;
268  }
269  return baseNameEntry->second[digitCount].contains(digitsValue);
