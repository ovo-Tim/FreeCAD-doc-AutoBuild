51 template <class Real>
52 Real DistLine3Segment3<Real>::GetSquared ()
53 {
54  Vector3<Real> kDiff = m_rkLine.Origin - m_rkSegment.Origin;
55  Real fA01 = -m_rkLine.Direction.Dot(m_rkSegment.Direction);
56  Real fB0 = kDiff.Dot(m_rkLine.Direction);
57  Real fC = kDiff.SquaredLength();
58  Real fDet = Math<Real>::FAbs((Real)1.0 - fA01*fA01);
59  Real fB1, fS0, fS1, fSqrDist, fExtDet;
60 
61  if (fDet >= Math<Real>::ZERO_TOLERANCE)
62  {
63  // The line and segment are not parallel.
64  fB1 = -kDiff.Dot(m_rkSegment.Direction);
65  fS1 = fA01*fB0-fB1;
66  fExtDet = m_rkSegment.Extent*fDet;
67 
68  if (fS1 >= -fExtDet)
69  {
70  if (fS1 <= fExtDet)
71  {
72  // Two interior points are closest, one on the line and one
73  // on the segment.
74  Real fInvDet = ((Real)1.0)/fDet;
75  fS0 = (fA01*fB1-fB0)*fInvDet;
76  fS1 *= fInvDet;
77  fSqrDist = fS0*(fS0+fA01*fS1+((Real)2.0)*fB0) +
78  fS1*(fA01*fS0+fS1+((Real)2.0)*fB1)+fC;
79  }
80  else
81  {
82  // The end point e1 of the segment and an interior point of
83  // the line are closest.
84  fS1 = m_rkSegment.Extent;
85  fS0 = -(fA01*fS1+fB0);
86  fSqrDist = -fS0*fS0+fS1*(fS1+((Real)2.0)*fB1)+fC;
87  }
88  }
89  else
90  {
91  // The end point e0 of the segment and an interior point of the
92  // line are closest.
93  fS1 = -m_rkSegment.Extent;
94  fS0 = -(fA01*fS1+fB0);
95  fSqrDist = -fS0*fS0+fS1*(fS1+((Real)2.0)*fB1)+fC;
96  }
97  }
98  else
99  {
100  // The line and segment are parallel. Choose the closest pair so that
101  // one point is at segment origin.
102  fS1 = (Real)0.0;
103  fS0 = -fB0;
104  fSqrDist = fB0*fS0+fC;
105  }
106 
107  m_kClosestPoint0 = m_rkLine.Origin + fS0*m_rkLine.Direction;
108  m_kClosestPoint1 = m_rkSegment.Origin + fS1*m_rkSegment.Direction;
109  m_fLineParameter = fS0;
110  m_fSegmentParameter = fS1;
111  return Math<Real>::FAbs(fSqrDist);
112 }
