31 def get_geometry_tree(element, prefix=""):
32  """Returns a list of elements representing an object's representation"""
33 
34  result = [prefix + str(element)]
35  prefix += TAB * " "
36  if getattr(element, "Representation", None):
37  reps = element.Representation
38  result.append(prefix + str(reps))
39  prefix += prefix
40  for rep in reps.Representations:
41  result.extend(get_geometry_tree(rep, prefix))
42  elif getattr(element, "Items", None):
43  for it in element.Items:
44  result.extend(get_geometry_tree(it, prefix))
45  elif element.is_a("IfcPolyline"):
46  for p in element.Points:
47  result.append(prefix + str(p))
48  elif element.is_a("IfcExtrudedAreaSolid"):
49  result.append(prefix + str(element.ExtrudedDirection))
50  result.extend(get_geometry_tree(element.SweptArea, prefix))
51  elif element.is_a("IfcArbitraryClosedProfileDef"):
52  result.extend(get_geometry_tree(element.OuterCurve, prefix))
53  elif element.is_a("IfcArbitraryProfileDefWithVoids"):
54  result.extend(get_geometry_tree(element.OuterCurve, prefix))
55  for inn in element.InnerCurves:
56  result.extend(get_geometry_tree(inn, prefix))
57  elif element.is_a("IfcMappedItem"):
58  result.extend(get_geometry_tree(element.MappingSource[1], prefix))
59  elif element.is_a("IfcBooleanClippingResult"):
60  result.extend(get_geometry_tree(element.FirstOperand, prefix))
61  result.extend(get_geometry_tree(element.SecondOperand, prefix))
62  elif element.is_a("IfcBooleanResult"):
63  result.extend(get_geometry_tree(element.FirstOperand, prefix))
64  result.extend(get_geometry_tree(element.SecondOperand, prefix))
65  elif element.is_a("IfcHalfSpaceSolid"):
66  result.extend(get_geometry_tree(element.BaseSurface, prefix))
67  return result
