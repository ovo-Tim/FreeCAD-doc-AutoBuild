39  self.svg.append('A %s,%s 0,0,%s %s,%s' % (r, r, str(sweep), p[0], p[1]))

42  def curve(self, *points):
43  """Add a Bezier curve from self.pos to points[-1]
44  every other points are the control points of the Bezier curve (which
45  will thus be of degree len(points) )
46  """
47  assert len(points) == 3
48  points = [rotate(p, self.theta) for p in points]
49  self.svg.append('C %s,%s %s,%s %s,%s' % tuple(itertools.chain(*points)))
50  self.pos = points[-1]

52  def close(self):
