267  def _get_installation_details_string(self) -> str:
268  version = self.installed_version
269  date = ""
270  installed_version_string = ""
271  if self.installed_timestamp:
272  date = QtCore.QLocale().toString(
273  QtCore.QDateTime.fromSecsSinceEpoch(int(round(self.installed_timestamp, 0))),
274  QtCore.QLocale.ShortFormat,
275  )
276  if version and date:
277  installed_version_string += (
278  translate("AddonsInstaller", "Version {version} installed on {date}").format(
279  version=version, date=date
280  )
281  + ". "
282  )
283  elif version:
284  installed_version_string += (
285  translate("AddonsInstaller", "Version {version} installed") + "."
286  ).format(version=version)
287  elif date:
288  installed_version_string += (
289  translate("AddonsInstaller", "Installed on {date}") + "."
290  ).format(date=date)
291  else:
292  installed_version_string += translate("AddonsInstaller", "Installed") + "."
293  return installed_version_string

295  def _get_update_status_string(self) -> str:
296  if self.update_info.check_in_progress:
297  return translate("AddonsInstaller", "Update check in progress") + "."
298  elif self.update_info.unchecked:
299  return ""
300  if self.update_info.detached_head:
301  return (
302  translate(
303  "AddonsInstaller", "Git tag '{}' checked out, no updates possible"
304  ).format(self.update_info.tag)
305  + "."
306  )
307  if self.update_info.update_available:
308  if self.installed_branch and self.update_info.branch:
309  if self.installed_branch != self.update_info.branch:
310  return (
311  translate(
312  "AddonsInstaller", "Currently on branch {}, name changed to {}"
313  ).format(self.installed_branch, self.update_info.branch)
314  + "."
315  )
316  if self.update_info.version:
317  return (
318  translate(
319  "AddonsInstaller",
320  "Currently on branch {}, update available to version {}",
321  ).format(self.installed_branch, str(self.update_info.version).strip())
322  + "."
323  )
324  return translate("AddonsInstaller", "Update available") + "."
325  if self.update_info.version:
326  return (
327  translate("AddonsInstaller", "Update available to version {}").format(
328  str(self.update_info.version).strip()
329  )
330  + "."
331  )
332  return translate("AddonsInstaller", "Update available") + "."
333  return translate("AddonsInstaller", "This is the latest version available") + "."
