145  def modify(self, detail, mild_restore=None):
146  """modify(detail, mild_restore = True): modifies scene detail through this TV.
147  The value is provided as an instance of SceneDetail implementation.
148  The procedure takes care to account for the stack - that is, if in a TV applied
149  later than this one this detail was changed too, the value saved therein is altered,
150  rather than applied to the scene.
151 
152  mild_restore: if True, when restoring later, checks if the value was changed
153  by user after last call to modify(), and doesn't restore if it was changed.
154 
155  Example: tv.modify(VProperty(obj, "Visibility", True))"""
156 
157  self._change()

160  detail.mild_restore = mild_restore
161 
162  # save current
163  self.save(detail, detail.mild_restore)
164 
165  # apply
166  tv1, curr = self._value_after(detail)
167  if tv1 is not None:
