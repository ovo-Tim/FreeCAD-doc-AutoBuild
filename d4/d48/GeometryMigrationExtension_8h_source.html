33 // This is a light-weight c++ only geometry extension to enable migration of information that was stored within
34 // the Part WB and should be migrated to another WB
35 //
36 // It is designed so that a single extension can migrate different types of data.
37 //
38 // To migrate data:
39 // 1. Add an enum bit to indicate the type of migration type
40 // 2. Add the data members to store the information and accessors
41 class PartExport GeometryMigrationExtension : public Part::GeometryExtension
42 {
43  TYPESYSTEM_HEADER_WITH_OVERRIDE();
44 public:
45 
46  // Indicates the type of migration to be performed, it is stored as a bitset, so several
47  // migrations may take place in a single extension.
48  // It is intended to support also LinkStage3 migration with a single framework (Id, Ref, ...)
49  enum MigrationType {
50  None = 0,
51  Construction = 1,
52  GeometryId = 2,
53  ExternalReference = 3,
54  NumMigrationType // Must be the last
55  };
