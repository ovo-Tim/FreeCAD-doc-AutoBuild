195  def buildShape(self, obj, pl, pls):
196  """Build the shape of the link object."""
197  if self.use_link:
198  if not getattr(obj, 'ExpandArray', False) or obj.Count != len(pls):
199  obj.setPropertyStatus('PlacementList', '-Immutable')
200  obj.PlacementList = pls
201  obj.setPropertyStatus('PlacementList', 'Immutable')
202  obj.Count = len(pls)
203  if getattr(obj, 'ExpandArray', False) \
204  and getattr(obj, 'AlwaysSyncPlacement', False):
205  for pla,child in zip(pls,obj.ElementList):
206  child.Placement = pla
207  elif obj.Count != len(pls):
208  obj.Count = len(pls)
209 
210  if obj.Base:
211  shape = getattr(obj.Base, 'Shape', None)
212  if not isinstance(shape, Part.Shape):
213  obj.Shape = Part.Shape()
214  elif shape.isNull():
215  _err_msg = ("'{}' cannot build shape "
216  "from '{}'\n".format(obj.Label, obj.Base.Label))
217  raise RuntimeError(_err_msg)
218  else:
219  # Resetting the Placement of the copied shape does not work for
220  # Part_Vertex and Draft_Point objects, we need to transform:
221  place = shape.Placement
222  shape = shape.copy()
223  shape.transformShape(place.Matrix.inverse())
224  base = []
225  for i, pla in enumerate(pls):
226  vis = getattr(obj, 'VisibilityList', [])
227  if len(vis) > i and not vis[i]:
228  continue
229 
230  base.append(shape.transformed(pla.toMatrix()))
231 
232  if getattr(obj, 'Fuse', False) and len(base) > 1:
233  obj.Shape = base[0].multiFuse(base[1:]).removeSplitter()
234  else:
235  obj.Shape = Part.makeCompound(base)
236 
237  if not DraftGeomUtils.isNull(pl):
238  obj.Placement = pl
239 
240  if self.use_link:
241  return False # return False to call LinkExtension::execute()
