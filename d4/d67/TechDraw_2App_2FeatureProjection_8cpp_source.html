65 App::DocumentObjectExecReturn *FeatureProjection::execute()
66 {
67  App::DocumentObject* link = Source.getValue();
68  if (!link)
69  return new App::DocumentObjectExecReturn("No object linked");
70  if (!link->isDerivedFrom<Part::Feature>())
71  return new App::DocumentObjectExecReturn("Linked object is not a Part object");
72  const TopoDS_Shape& shape = static_cast<Part::Feature*>(link)->Shape.getShape().getShape();
73  if (shape.IsNull())
74  return new App::DocumentObjectExecReturn("Linked shape object is empty");
75 
76  try {
77  const Base::Vector3d& dir = Direction.getValue();
78  TechDraw::ProjectionAlgos alg(shape, dir);
79 
80  TopoDS_Compound comp;
81  BRep_Builder builder;
82  builder.MakeCompound(comp);
83  if (!alg.V.IsNull() && VCompound.getValue())
84  builder.Add(comp, alg.V);
85  if (!alg.V1.IsNull() && Rg1LineVCompound.getValue())
86  builder.Add(comp, alg.V1);
87  if (!alg.VN.IsNull() && RgNLineVCompound.getValue())
88  builder.Add(comp, alg.VN);
89  if (!alg.VO.IsNull() && OutLineVCompound.getValue())
90  builder.Add(comp, alg.VO);
91  if (!alg.VI.IsNull() && IsoLineVCompound.getValue())
92  builder.Add(comp, alg.VI);
93  if (!alg.H.IsNull() && HCompound.getValue())
94  builder.Add(comp, alg.H);
95  if (!alg.H1.IsNull() && Rg1LineHCompound.getValue())
96  builder.Add(comp, alg.H1);
97  if (!alg.HN.IsNull() && RgNLineHCompound.getValue())
98  builder.Add(comp, alg.HN);
99  if (!alg.HO.IsNull() && OutLineHCompound.getValue())
100  builder.Add(comp, alg.HO);
101  if (!alg.HI.IsNull() && IsoLineHCompound.getValue())
102  builder.Add(comp, alg.HI);
103 
104  Shape.setValue(comp);
105  return App::DocumentObject::StdReturn;
106  }
107  catch (Standard_Failure& e) {
108  return new App::DocumentObjectExecReturn(e.GetMessageString());
109  }
110 }
