123 void TaskDraftParameters::onSelectionChanged(const Gui::SelectionChanges& msg)
124 {
125  // executed when the user selected something in the CAD object
126  // adds/deletes the selection accordingly
127 
128  if (msg.Type == Gui::SelectionChanges::AddSelection) {
129  if (selectionMode == refSel) {
130  referenceSelected(msg, ui->listWidgetReferences);
131  }
132  else if (selectionMode == plane) {
133  auto pcDraft = getObject<PartDesign::Draft>();
134  std::vector<std::string> planes;
135  App::DocumentObject* selObj {};
136  getReferencedSelection(pcDraft, msg, selObj, planes);
137  if (!selObj) {
138  return;
139  }
140  setupTransaction();
141  pcDraft->NeutralPlane.setValue(selObj, planes);
142  ui->linePlane->setText(getRefStr(selObj, planes));
143 
144  pcDraft->getDocument()->recomputeFeature(pcDraft);
145  // highlight existing references for possible further selections
146  getDressUpView()->highlightReferences(true);
147  // hide the draft if there was a computation error
148  hideOnError();
149  }
150  else if (selectionMode == line) {
151  auto pcDraft = getObject<PartDesign::Draft>();
152  std::vector<std::string> edges;
153  App::DocumentObject* selObj;
154  getReferencedSelection(pcDraft, msg, selObj, edges);
155  if (!selObj) {
156  return;
157  }
158  setupTransaction();
159  pcDraft->PullDirection.setValue(selObj, edges);
160  ui->lineLine->setText(getRefStr(selObj, edges));
161 
162  pcDraft->getDocument()->recomputeFeature(pcDraft);
163  // highlight existing references for possible further selections
164  getDressUpView()->highlightReferences(true);
165  // hide the draft if there was a computation error
166  hideOnError();
167  }
168  }
169 }
