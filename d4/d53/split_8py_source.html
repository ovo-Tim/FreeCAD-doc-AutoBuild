57 def split_open_wire(wire, newPoint, edgeIndex):
58  wire1Points = []
59  wire2Points = []
60  for index, point in enumerate(wire.Points):
61  if index == edgeIndex:
62  wire1Points.append(wire.Placement.inverse().multVec(newPoint))
63  wire2Points.append(newPoint)
64  wire2Points.append(wire.Placement.multVec(point))
65  elif index < edgeIndex:
66  wire1Points.append(point)
67  elif index > edgeIndex:
68  wire2Points.append(wire.Placement.multVec(point))
69  wire.Points = wire1Points
