156  {

159  if (vpd && vpd->getObject()) {
160  map[vpd->getObject()] = vpd;

164  {
165  auto vpd =
166  Base::freecad_dynamic_cast<ViewProviderDocumentObject>(const_cast<ViewProvider*>(&vp));
167  if (vpd && vpd->getObject()) {
168  map.erase(vpd->getObject());
169  }
170  }

173  for (auto obj : doc.getObjects()) {
174  map.erase(obj);
175  }
176  }

178  {
179  auto it = map.find(obj);
180  if (it == map.end()) {
181  return nullptr;
182  }

184  }

227 static PyObject* FreeCADGui_subgraphFromObject(PyObject* /*self*/, PyObject* args)
228 {
229  PyObject* o;
230  if (!PyArg_ParseTuple(args, "O!", &(App::DocumentObjectPy::Type), &o)) {
231  return nullptr;
232  }
233  App::DocumentObject* obj = static_cast<App::DocumentObjectPy*>(o)->getDocumentObjectPtr();
234  std::string vp = obj->getViewProviderName();
235  SoNode* node = nullptr;
236  try {
237  auto base =
238  static_cast<Base::BaseClass*>(Base::Type::createInstanceByName(vp.c_str(), true));
239  if (base && base->isDerivedFrom<Gui::ViewProviderDocumentObject>()) {
240  std::unique_ptr<Gui::ViewProviderDocumentObject> vp(
241  static_cast<Gui::ViewProviderDocumentObject*>(base));
242  std::map<std::string, App::Property*> Map;
243  obj->getPropertyMap(Map);
244  vp->attach(obj);
245 
246  // this is needed to initialize Python-based view providers
247  App::Property* pyproxy = vp->getPropertyByName("Proxy");
248  if (pyproxy && pyproxy->is<App::PropertyPythonObject>()) {
249  static_cast<App::PropertyPythonObject*>(pyproxy)->setValue(Py::Long(1));
250  }
251 
252  for (const auto& it : Map) {
253  vp->updateData(it.second);
254  }
255 
256  std::vector<std::string> modes = vp->getDisplayModes();
257  if (!modes.empty()) {
258  vp->setDisplayMode(modes.front().c_str());
259  }
260  node = vp->getRoot()->copy();
261  node->ref();
262  std::string prefix = "So";
263  std::string type = node->getTypeId().getName().getString();
264  // doesn't start with the prefix 'So'
265  if (type.rfind("So", 0) != 0) {
266  type = prefix + type;
267  }
268  else if (type == "SoFCSelectionRoot") {
269  type = "SoSeparator";
270  }
271 
272  type += " *";
273  PyObject* proxy = nullptr;
274  proxy = Base::Interpreter().createSWIGPointerObj("pivy.coin",
275  type.c_str(),
276  static_cast<void*>(node),
277  1);
278  return Py::new_reference_to(Py::Object(proxy, true));
279  }
280  }
281  catch (const Base::Exception& e) {
282  if (node) {
283  node->unref();
284  }
285  PyErr_SetString(PyExc_RuntimeError, e.what());
286  return nullptr;
287  }
288 
289  Py_INCREF(Py_None);
290  return Py_None;
291 }

293 static PyObject* FreeCADGui_exportSubgraph(PyObject* /*self*/, PyObject* args)
294 {
295  const char* format = "VRML";
296  PyObject* proxy;
297  PyObject* output;
298  if (!PyArg_ParseTuple(args, "OO|s", &proxy, &output, &format)) {
299  return nullptr;
300  }
301 
302  void* ptr = nullptr;
303  try {
304  Base::Interpreter().convertSWIGPointerObj("pivy.coin", "SoNode *", proxy, &ptr, 0);
305  auto node = static_cast<SoNode*>(ptr);
306  if (node) {
307  std::string formatStr(format);
308  std::string buffer;
309 
310  if (formatStr == "VRML") {
311  SoFCDB::writeToVRML(node, buffer);
312  }
313  else if (formatStr == "IV") {
314  buffer = SoFCDB::writeNodesToString(node);
315  }
316  else {
317  throw Base::ValueError("Unsupported format");
318  }
319 
320  Base::PyStreambuf buf(output);
321  std::ostream str(nullptr);
322  str.rdbuf(&buf);
323  str << buffer;
324  }
325 
326  Py_INCREF(Py_None);
327  return Py_None;
328  }
329  catch (const Base::Exception& e) {
330  PyErr_SetString(PyExc_RuntimeError, e.what());
331  return nullptr;
332  }

616 void Application::open(const char* FileName, const char* Module)
617 {

619  wc.setIgnoreEvents(WaitCursor::NoEvents);
620  Base::FileInfo File(FileName);
621  string te = File.extension();
622  string unicodepath = Base::Tools::escapedUnicodeFromUtf8(File.filePath().c_str());
623  unicodepath = Base::Tools::escapeEncodeFilename(unicodepath);
624 
625  // if the active document is empty and not modified, close it
626  // in case of an automatically created empty document at startup
627  App::Document* act = App::GetApplication().getActiveDocument();
628  Gui::Document* gui = this->getDocument(act);
629  if (act && act->countObjects() == 0 && gui && !gui->isModified()) {
630  Command::doCommand(Command::App, "App.closeDocument('%s')", act->getName());
631  qApp->processEvents(); // an update is needed otherwise the new view isn't shown
632  }
633 
634  if (Module) {
635  try {
636  if (File.hasExtension("FCStd")) {
637  bool handled = false;
638  std::string filepath = File.filePath();
639  for (auto& v : d->documents) {
640  auto doc = v.second->getDocument();
641  std::string fi = Base::FileInfo(doc->FileName.getValue()).filePath();
642  if (filepath == fi) {
643  handled = true;
644  Command::doCommand(Command::App, "FreeCADGui.reload('%s')", doc->getName());
645  break;
646  }
647  }
648 
649  if (!handled) {
650  Command::doCommand(Command::App,
651  "FreeCAD.openDocument('%s')",
652  unicodepath.c_str());
653  Gui::Application::checkForRecomputes();
654  }
655  }
656  else {
657  std::string code = fmt::format("from freecad import module_io\n"
658  "module_io.OpenInsertObject(\"{}\", \"{}\", \"{}\")\n",
659  Module, unicodepath, "open");
660  Gui::Command::runCommand(Gui::Command::App, code.c_str());
661 
662  // ViewFit
663  if (sendHasMsgToActiveView("ViewFit")) {
664  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
665  "User parameter:BaseApp/Preferences/View");
666  if (hGrp->GetBool("AutoFitToView", true)) {
667  Command::doCommand(Command::Gui, "Gui.SendMsgToActiveView(\"ViewFit\")");
668  }
669  }
670  }
671 
672  // the original file name is required
673  QString filename = QString::fromUtf8(File.filePath().c_str());

1526 {

1528  WaitCursor wc;
1529  Workbench* oldWb = WorkbenchManager::instance()->active();
1530  if (oldWb && oldWb->name() == name) {
1531  return false; // already active
1532  }
1533 
1534  Base::PyGILStateLocker lock;
1535  // we check for the currently active workbench and call its 'Deactivated'
1536  // method, if available
1537  PyObject* pcOldWorkbench = nullptr;
1538  if (oldWb) {
1539  pcOldWorkbench = PyDict_GetItemString(_pcWorkbenchDictionary, oldWb->name().c_str());
1540  }
1541 
1542  // get the python workbench object from the dictionary
1543  PyObject* pcWorkbench = nullptr;
1544  pcWorkbench = PyDict_GetItemString(_pcWorkbenchDictionary, name);
1545  // test if the workbench exists
1546  if (!pcWorkbench) {
1547  return false;
1548  }
1549 
1550  try {
1551  std::string type;
1552  Py::Object handler(pcWorkbench);
1553  if (!handler.hasAttr(std::string("__Workbench__"))) {
1554  // call its GetClassName method if possible
1555  Py::Callable method(handler.getAttr(std::string("GetClassName")));
1556  Py::Tuple args;
1557  Py::String result(method.apply(args));
1558  type = result.as_std_string("ascii");
1559  if (Base::Type::fromName(type.c_str())
1560  .isDerivedFrom(Gui::PythonBaseWorkbench::getClassTypeId())) {
1561  Workbench* wb = WorkbenchManager::instance()->createWorkbench(name, type);
1562  if (!wb) {
1563  throw Py::RuntimeError("Failed to instantiate workbench of type " + type);
1564  }
1565  handler.setAttr(std::string("__Workbench__"), Py::Object(wb->getPyObject(), true));
1566  }
1567 
1568  // import the matching module first
1569  Py::Callable activate(handler.getAttr(std::string("Initialize")));
1570  activate.apply(args);
1571 
1572  // Dependent on the implementation of a workbench handler the type
1573  // can be defined after the call of Initialize()
1574  if (type.empty()) {
1575  Py::String result(method.apply(args));
1576  type = result.as_std_string("ascii");
1577  }
1578  }
1579 
1580  // does the Python workbench handler have changed the workbench?
1581  Workbench* curWb = WorkbenchManager::instance()->active();
1582  if (curWb && curWb->name() == name) {
1583  ok = true; // already active
1584  }
1585  // now try to create and activate the matching workbench object
1586  else if (WorkbenchManager::instance()->activate(name, type)) {
1587  getMainWindow()->activateWorkbench(QString::fromLatin1(name));
1588  this->signalActivateWorkbench(name);
1589  ok = true;
1590  }
1591 
1592  // if we still not have this member then it must be built-in C++ workbench
1593  // which could be created after loading the appropriate module
1594  if (!handler.hasAttr(std::string("__Workbench__"))) {
1595  Workbench* wb = WorkbenchManager::instance()->getWorkbench(name);
1596  if (wb) {
1597  handler.setAttr(std::string("__Workbench__"), Py::Object(wb->getPyObject(), true));
1598  }
1599  }
1600 
1601  // If the method Deactivate is available we call it
1602  if (pcOldWorkbench) {
1603  Py::Object handler(pcOldWorkbench);
1604  if (handler.hasAttr(std::string("Deactivated"))) {
1605  Py::Object method(handler.getAttr(std::string("Deactivated")));
1606  if (method.isCallable()) {
1607  Py::Tuple args;
1608  Py::Callable activate(method);
1609  activate.apply(args);
1610  }
1611  }
1612  }
1613 
1614  if (oldWb) {
1615  oldWb->deactivated();
1616  }
1617 
1618  // If the method Activate is available we call it
1619  if (handler.hasAttr(std::string("Activated"))) {
1620  Py::Object method(handler.getAttr(std::string("Activated")));
1621  if (method.isCallable()) {
1622  Py::Tuple args;
1623  Py::Callable activate(method);
1624  activate.apply(args);
1625  }
1626  }
1627 
1628  // now get the newly activated workbench
1629  Workbench* newWb = WorkbenchManager::instance()->active();
1630  if (newWb) {
1631  if (!Instance->d->startingUp) {
1632  std::string nameWb = newWb->name();
1633  App::GetApplication()
1634  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")
1635  ->SetASCII("LastModule", nameWb.c_str());
1636  }
1637  newWb->activated();
1638  }
1639  }
1640  catch (Py::Exception&) {
1641  Base::PyException e; // extract the Python error text
1642  QString msg = QString::fromUtf8(e.what());
1643  QRegularExpression rx;
1644  // ignore '<type 'exceptions.ImportError'>' prefixes
1645  rx.setPattern(QLatin1String("^\\s*<type 'exceptions.ImportError'>:\\s*"));
1646  auto match = rx.match(msg);
1647  while (match.hasMatch()) {
1648  msg = msg.mid(match.capturedLength());
1649  match = rx.match(msg);
1650  }
1651 
1652  Base::Console().Error("%s\n", (const char*)msg.toUtf8());
1653  if (!d->startingUp) {
1654  Base::Console().Error("%s\n", e.getStackTrace().c_str());
1655  }
1656  else {
1657  Base::Console().Log("%s\n", e.getStackTrace().c_str());
1658  }
1659 
1660  if (!d->startingUp) {
1661  wc.restoreCursor();
1662  QMessageBox::critical(getMainWindow(),
1663  QObject::tr("Workbench failure"),
1664  QObject::tr("%1").arg(msg));
1665  wc.setWaitCursor();
1666  }
1667  }
1668 
1669  return ok;
1670 }
