284 bool MaterialTreeWidget::findInTree(const QStandardItem& node,
285  QModelIndex* index,
286  const QString& uuid)
287 {
288  auto vv = node.data(Qt::UserRole);
289  if (vv.isValid() && vv == uuid) {
290  *index = node.index();
291  return true;
292  }
293 
294  if (node.hasChildren()) {
295  for (int i = 0; i < node.rowCount(); i++) {
296  auto child = node.child(i);
297  if (findInTree(*child, index, uuid)) {
298  return true;
299  }
300  }
301  }
302 
303  return false;
304 }

306 QModelIndex MaterialTreeWidget::findInTree(const QString& uuid)
307 {
308  auto model = dynamic_cast<QStandardItemModel*>(m_materialTree->model());
309  auto root = model->invisibleRootItem();
310 
311  QModelIndex index;
312  // Find the original item, not the reference in favourites or recents
313  for (int i = 0; i < root->rowCount(); i++) {
314  auto child = root->child(i);
315  if (child->text() != tr("Favorites") && child->text() != tr("Recent")) {
316  if (findInTree(*child, &index, uuid)) {
317  return index;
318  }
319  }
320  }
321 
322  return {};

453 void MaterialTreeWidget::saveRecents()
454 {
455  auto param = App::GetApplication().GetParameterGroupByPath(
456  "User parameter:BaseApp/Preferences/Mod/Material/Recent");
457 
458  // Clear out the existing favorites
459  int count = param->GetInt("Recent", 0);
460  for (int i = 0; static_cast<long>(i) < count; i++) {
461  QString key = QString::fromLatin1("MRU%1").arg(i);
462  param->RemoveASCII(key.toStdString().c_str());
463  }
464 
465  // Add the current values
466  int size = _recents.size();
467  if (size > _recentMax) {
468  size = _recentMax;
469  }
470  param->SetInt("Recent", size);
471  int j = 0;
472  for (auto& recent : _recents) {
473  QString key = QString::fromLatin1("MRU%1").arg(j);
474  param->SetASCII(key.toStdString().c_str(), recent.toStdString());
475 
476  j++;
477  if (j >= size) {
478  break;
479  }
480  }
481 }

483 void MaterialTreeWidget::addRecent(const QString& uuid)
484 {
485  if (uuid.isEmpty()) {
486  return;
487  }
488  // Ensure it is a material. New, unsaved materials will not be
489  try {
490  auto material = _materialManager.getMaterial(uuid);
491  Q_UNUSED(material)
492  }
493  catch (const Materials::MaterialNotFound&) {
494  return;
495  }
496 
497  // Ensure no duplicates
498  if (isRecent(uuid)) {
499  _recents.remove(uuid);
500  }
501 
502  _recents.push_front(uuid);
503  while (_recents.size() > static_cast<std::size_t>(_recentMax)) {
504  _recents.pop_back();
505  }
506 
507  saveRecents();
508 }

541

543

564  showLibraries = true;
565  }
566 
567  if (showLibraries) {
568  auto lib = new QStandardItem(library->getName());
569  lib->setFlags(Qt::ItemIsEnabled);
570  addExpanded(model, lib, param);
571 
572  QIcon icon(library->getIconPath());
573  QIcon folderIcon(QString::fromStdString(":/icons/folder.svg"));
574

576  }
