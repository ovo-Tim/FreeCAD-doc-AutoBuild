98  double length() const
99  {
100  return sqrt(x * x + y * y);
101  }
102  // returns length and writes length deriv into the dlength argument.
103  double length(double& dlength) const;
104 
105  // unlike other vectors in FreeCAD, this normalization creates a new vector instead of
106  // modifying existing one.
107  DeriVector2 getNormalized() const; // returns zero vector if the original is zero.
108  // calculates scalar product of two vectors and returns the result. The derivative
109  // of the result is written into argument dprd.
110  double scalarProd(const DeriVector2& v2, double* dprd = nullptr) const;
111  // calculates the norm of the cross product of the two vectors.
112  // DeriVector2 are considered as 3d vectors with null z. The derivative
113  // of the result is written into argument dprd.
114  double crossProdNorm(const DeriVector2& v2, double& dprd) const;
115  DeriVector2 sum(const DeriVector2& v2) const
116  { // adds two vectors and returns result
117  return DeriVector2(x + v2.x, y + v2.y, dx + v2.dx, dy + v2.dy);
118  }
119  DeriVector2 subtr(const DeriVector2& v2) const
120  { // subtracts two vectors and returns result
121  return DeriVector2(x - v2.x, y - v2.y, dx - v2.dx, dy - v2.dy);
122  }
123  DeriVector2 mult(double val) const
124  {
125  return DeriVector2(x * val, y * val, dx * val, dy * val);
126  } // multiplies the vector by a number. Derivatives are scaled.
127  DeriVector2 multD(double val, double dval) const
128  { // multiply vector by a variable with a derivative.
129  return DeriVector2(x * val, y * val, dx * val + x * dval, dy * val + y * dval);
130  }
131  // divide vector by a variable with a derivative
132  DeriVector2 divD(double val, double dval) const;
133  DeriVector2 rotate90ccw() const
134  {
135  return DeriVector2(-y, x, -dy, dx);
136  }
137  DeriVector2 rotate90cw() const
138  {
139  return DeriVector2(y, -x, dy, -dx);
140  }
141  DeriVector2 linCombi(double m1, const DeriVector2& v2, double m2) const
142  { // linear combination of two vectors
143  return DeriVector2(x * m1 + v2.x * m2,
144  y * m1 + v2.y * m2,
145  dx * m1 + v2.dx * m2,
146  dy * m1 + v2.dy * m2);
147  }

155 class SketcherExport Curve
156 {
157 public:
158  virtual ~Curve()
159  {}
160  // returns normal vector. The vector should point to the left when one
161  // walks along the curve from start to end. Ellipses and circles are
162  // assumed to be walked counterclockwise, so the vector should point
163  // into the shape.
164  // derivparam is a pointer to a curve parameter (or point coordinate) to
165  // compute the derivative for. The derivative is returned through dx,dy
166  // fields of DeriVector2.
167  virtual DeriVector2 CalculateNormal(const Point& p,
168  const double* derivparam = nullptr) const = 0;
169 
170  // returns normal vector at parameter instead of at the given point.
171  virtual DeriVector2 CalculateNormal(const double* param,
172  const double* derivparam = nullptr) const
173  {
174  DeriVector2 pointDV = Value(*param, 0.0);
175  Point p(&pointDV.x, &pointDV.y);
176  return CalculateNormal(p, derivparam);
177  }
178 
186  virtual DeriVector2 Value(double u, double du, const double* derivparam = nullptr) const;
187 
188  // adds curve's parameters to pvec (used by constraints)
189  virtual int PushOwnParams(VEC_pD& pvec) = 0;
190  // recunstruct curve's parameters reading them from pvec starting from index cnt.
191  // cnt will be incremented by the same value as returned by PushOwnParams()
192  virtual void ReconstructOnNewPvec(VEC_pD& pvec, int& cnt) = 0;
193  // DeepSOIC: I haven't found a way to simply copy a curve object provided pointer to a curve
194  // object.
195  virtual Curve* Copy() = 0;
196 };
