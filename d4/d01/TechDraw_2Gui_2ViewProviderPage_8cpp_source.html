148 void ViewProviderPage::updateData(const App::Property* prop)
149 {
150  auto page = getDrawPage();

153  return;

185 bool ViewProviderPage::onDelete(const std::vector<std::string>&)
186 {
187  // warn the user if the Page is not empty
188  // but don't do this if there is just the template
189 
190  // check if there are items in the group
191  auto objs = claimChildren();
192 
193  // check if there is just a template
194  // if there are several objects, the template is never the last one
195  // the ExportName of a template always begins with "Template"
196  bool isTemplate = false;
197  for (auto objsIterator : objs) {
198  if (objsIterator->getExportName().substr(0, 8).compare(std::string("Template")) == 0)
199  isTemplate = true;
200  else
201  isTemplate = false;
202  }
203 
204  if (!objs.empty() && !isTemplate) {
205  // generate dialog
206  QString bodyMessage;
207  QTextStream bodyMessageStream(&bodyMessage);
208  bodyMessageStream << qApp->translate(
209  "Std_Delete",
210  "The page is not empty, therefore the\nfollowing referencing objects might be lost:");
211  bodyMessageStream << '\n';
212  for (auto ObjIterator : objs)
213  bodyMessageStream << '\n' << QString::fromUtf8(ObjIterator->Label.getValue());
214  bodyMessageStream << "\n\n" << QObject::tr("Are you sure you want to continue?");
215  // show and evaluate the dialog
216  int DialogResult = QMessageBox::warning(
217  Gui::getMainWindow(), qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
218  QMessageBox::Yes, QMessageBox::No);
219  if (DialogResult == QMessageBox::Yes) {
220  removeMDIView();
221  return true;
222  }
223  else
224  return false;
225  }
226  else {
227  removeMDIView();
228  return true;
229  }
230 }

388 std::vector<App::DocumentObject*> ViewProviderPage::claimChildren(void) const
389 {
390  std::vector<App::DocumentObject*> temp;
391 
392  App::DocumentObject* templateFeat = nullptr;
393  templateFeat = getDrawPage()->Template.getValue();
394 
395  if (templateFeat) {
396  temp.push_back(templateFeat);
397  }
398 
399  // Collect any child views
400  // for Page, valid children are any View except: DrawViewDimension
401  // DrawViewBalloon
402  // any FeatureView in a DrawViewClip
403  // DrawHatch
404  // DrawGeomHatch
405  // ?? leaders?
406 
407  try {
408  for (auto* obj : getDrawPage()->Views.getValues()) {
409  auto* featView = dynamic_cast<TechDraw::DrawView*>(obj);
410 
411  // If the child view appoints a parent, skip it
412  if (featView && featView->claimParent()) {
413  continue;
414  }
415 
416  // Don't collect if dimension, balloon, hatch or member of ClipGroup as these should be grouped elsewhere
417  if (obj->isDerivedFrom<TechDraw::DrawViewDimension>()
418  || obj->isDerivedFrom<TechDraw::DrawHatch>()
419  || obj->isDerivedFrom<TechDraw::DrawGeomHatch>()
420  || obj->isDerivedFrom<TechDraw::DrawViewBalloon>()
421  || (featView && featView->isInClip())) {
422  continue;
423  }
424  else {
425  temp.push_back(obj);
426  }
427  }
428  return temp;
429  }
430  catch (...) {
431  return std::vector<App::DocumentObject*>();
432  }
433 }
