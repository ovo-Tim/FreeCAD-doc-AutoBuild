146 void QGIRichAnno::setTextItem()
147 {
148 // Base::Console().Message("QGIRA::setTextItem() - %s - exportingSvg: %d\n", getViewName(), getExportingSvg());
149  TechDraw::DrawRichAnno* annoFeat = getFeature();
150 
151  // convert the text size
152  QString inHtml = QString::fromUtf8(annoFeat->AnnoText.getValue());
153  QString outHtml = convertTextSizes(inHtml);
154 
155  //position the text
156  prepareGeometryChange();
157  // control auto line break
158  if (annoFeat->MaxWidth.getValue() > 0.0) {
159  // we have set a maximum width, so convert it to scene units
160  m_text->setTextWidth(Rez::guiX(annoFeat->MaxWidth.getValue()));
161  } else {
162  // we don't want to break lines
163  m_text->setTextWidth(annoFeat->MaxWidth.getValue());
164  }
165  m_text->setHtml(outHtml);
166  if (getExportingSvg()) {
167  // lines are correctly spaced on screen or in pdf, but svg needs this
168  setLineSpacing(100);
169  }
170 
171  if (!getExportingSvg()) {
172  // screen or pdf rendering
173  m_text->centerAt(0.0, 0.0);
174  }
175 
176  // align the frame rectangle to the text
177  constexpr double frameMargin{10.0};
178  QRectF outRect = m_text->boundingRect().adjusted(-frameMargin, -frameMargin, frameMargin, frameMargin);
179  m_rect->setPen(rectPen());
180  m_rect->setBrush(Qt::NoBrush);
181  if (!getExportingSvg()) {
182  m_rect->setRect(outRect);
183  m_rect->setPos(m_text->pos().x() - frameMargin, m_text->pos().y() - frameMargin);
184  }
185 
186  if (annoFeat->ShowFrame.getValue()) {
187  m_rect->show();
188  } else {
189  m_rect->hide();
190  }
191 }

193 // attempt to space the lines correctly after the font sizes are changed to match
194 // the Svg rendering of the QGraphicsTextItem.
195 void QGIRichAnno::setLineSpacing(int lineSpacing)
196 {
197  // left to itself, Qt appears to space the lines according to this formula
198  // DeltaY(in mm) = (1 + 2*pointSize + margin) which vastly under spaces the
199  // lines
200  m_text->document()->setUseDesignMetrics(true);
201 
202  QTextBlock block = m_text->document()->begin();
203  for (; block.isValid(); block = block.next()) {
204  QTextCursor tc = QTextCursor(block);
205  QTextBlockFormat fmt = tc.blockFormat();
206  QTextCharFormat cFmt = tc.charFormat();
207  // this is already converted to pixels in setTextItem
208  //
209  double cssPixelSize = cFmt.font().pointSizeF();
210  // css Pixels are treated as if they were points in the conversion to Svg
211  double textHeightSU = Rez::guiX(cssPixelSize * 25.4 / 72.0); // actual height of text in scene units
212  double pointSize = cssPixelSize * 72.0 / 96.0;
213  double deltaYSU = 1.0 + 2.0 * pointSize; // how far Qt will space lines (based on samples)
214  double linegap = 0.4 * cssPixelSize; // 20% gaps above and below
215  // margins will be included in Qt's calculation of spacing
216 
217  double margin = linegap * pointSize / 10.0;
218  QTextBlockFormat spacerFmt = QTextBlockFormat();
219  if (block.previous().isValid()) {
220  // there is a block before this one, so add a top margin
221  spacerFmt.setTopMargin(margin);
222  }
223  if (block.next().isValid()) {
224  // there is another block after this, so add a bottom margin
225  spacerFmt.setBottomMargin(margin);
226  }
227  double requiredSpacing = (textHeightSU / (deltaYSU - 1.0)) * lineSpacing;
228  spacerFmt.setLineHeight(requiredSpacing, QTextBlockFormat::ProportionalHeight);
229  tc.mergeBlockFormat(spacerFmt);
230  }
231 }

235 QString QGIRichAnno::convertTextSizes(const QString& inHtml) const
236 {
237  constexpr double mmPerPoint{0.353}; // 25.4 mm/in / 72 points/inch
238  constexpr double cssPxPerPoint{1.333333}; // CSS says 12 pt text is 16 px high
239  double sceneUnitsPerPoint = Rez::getRezFactor() * mmPerPoint; // scene units per point: 3.53
240 
241  QRegularExpression rxFontSize(QString::fromUtf8("font-size:([0-9]*)pt;"));
242  QRegularExpressionMatch match;
243  QStringList findList;
244  QStringList replList;
245 
246  // find each occurrence of "font-size:..." and calculate the equivalent size in scene units
247  // or CSS pixels
248  int pos = 0;
249  while ((pos = inHtml.indexOf(rxFontSize, pos, &match)) != -1) {
250  QString found = match.captured(0);
251  findList << found;
252  QString qsOldSize = match.captured(1);
253 
254  QString repl = found;
255  double newSize = qsOldSize.toDouble(); // in points
256  // The font size in the QGraphicsTextItem html is interpreted differently
257  // in QSvgRenderer rendering compared to painting the screen or pdf
258  if (getExportingSvg()) {
259  // scale point size to CSS pixels
260  newSize = newSize * cssPxPerPoint;
261  } else {
262  // scale point size to scene units
263  newSize = newSize * sceneUnitsPerPoint;
264  }
265  QString qsNewSize = QString::number(newSize, 'f', 2);
266  repl.replace(qsOldSize, qsNewSize);
267  replList << repl;
268  pos += match.capturedLength();
269  }
270  QString outHtml = inHtml;
271  int iRepl = 0;
272  //TODO: check list for duplicates?
273  for ( ; iRepl < findList.size(); iRepl++) {
274  outHtml = outHtml.replace(findList[iRepl], replList[iRepl]);
275  }
276 
277  return outHtml;
278 }
