47 def mirror(objlist, p1, p2):
48  """Create a mirror object from the provided list and line.
49 
50  It creates a `Part::Mirroring` object from the given `objlist` using
51  a plane that is defined by the two given points `p1` and `p2`,
52  and the Draft working plane normal.
53 
54  Parameters
55  ----------
56  objlist: single object or a list of objects
57  A single object or a list of objects.
58 
59  p1: Base::Vector3
60  Point 1 of the mirror plane. It is also used as the `Placement.Base`
61  of the resulting object.
62 
63  p2: Base::Vector3
64  Point 2 of the mirror plane.
65 
66  Returns
67  -------
68  None
69  If the operation fails.
70 
71  list
72  List of `Part::Mirroring` objects, or a single one
73  depending on the input `objlist`.
74 
75  To Do
76  -----
77  Implement a mirror tool specific to the workbench that does not
78  just use `Part::Mirroring`. It should create a derived object,
79  that is, it should work similar to `Draft.offset`.
80  """
81 
82  if not objlist:
83  _err(translate("draft","No object given"))
84  return
85 
86  if p1 == p2:
87  _err(translate("draft","The two points are coincident"))
88  return
89 
90  if not isinstance(objlist, list):
91  objlist = [objlist]
92 
93  norm = WorkingPlane.get_working_plane(update=False).axis
94  pnorm = p2.sub(p1).cross(norm).normalize()
95 
96  result = []
97 
98  for obj in objlist:
99  mir = App.ActiveDocument.addObject("Part::Mirroring", "Mirror")
100  mir.Label = obj.Label + " (" + translate("draft", "mirrored") + ")"
101  mir.Source = obj
102  mir.Base = p1
103  mir.Normal = pnorm
104  gui_utils.format_object(mir, obj)
105  result.append(mir)
106 
107  if len(result) == 1:
108  result = result[0]
109  gui_utils.select(result)
110 
111  return result
