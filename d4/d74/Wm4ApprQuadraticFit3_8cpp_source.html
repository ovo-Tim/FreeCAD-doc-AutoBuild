163 template <class Real>
164 Real QuadraticSphereFit3 (int iQuantity, const Vector3<Real>* akPoint,
165  Vector3<Real>& rkCenter, Real& rfRadius)
166 {
167  Eigen<Real> kES(5);
168  int iRow, iCol;
169  for (iRow = 0; iRow < 5; iRow++)
170  {
171  for (iCol = 0; iCol < 5; iCol++)
172  {
173  kES(iRow,iCol) = (Real)0.0;
174  }
175  }
176 
177  for (int i = 0; i < iQuantity; i++)
178  {
179  Real fX = akPoint[i].X();
180  Real fY = akPoint[i].Y();
181  Real fZ = akPoint[i].Z();
182  Real fX2 = fX*fX;
183  Real fY2 = fY*fY;
184  Real fZ2 = fZ*fZ;
185  Real fXY = fX*fY;
186  Real fXZ = fX*fZ;
187  Real fYZ = fY*fZ;
188  Real fR2 = fX2+fY2+fZ2;
189  Real fXR2 = fX*fR2;
190  Real fYR2 = fY*fR2;
191  Real fZR2 = fZ*fR2;
192  Real fR4 = fR2*fR2;
193 
194  kES(0,1) += fX;
195  kES(0,2) += fY;
196  kES(0,3) += fZ;
197  kES(0,4) += fR2;
198  kES(1,1) += fX2;
199  kES(1,2) += fXY;
200  kES(1,3) += fXZ;
201  kES(1,4) += fXR2;
202  kES(2,2) += fY2;
203  kES(2,3) += fYZ;
204  kES(2,4) += fYR2;
205  kES(3,3) += fZ2;
206  kES(3,4) += fZR2;
207  kES(4,4) += fR4;
208  }
209 
210  kES(0,0) = (Real)iQuantity;
211 
212  for (iRow = 0; iRow < 5; iRow++)
213  {
214  for (iCol = 0; iCol < iRow; iCol++)
215  {
216  kES(iRow,iCol) = kES(iCol,iRow);
217  }
218  }
219 
220  Real fInvQuantity = ((Real)1.0)/(Real)iQuantity;
221  for (iRow = 0; iRow < 5; iRow++)
222  {
223  for (iCol = 0; iCol < 5; iCol++)
224  {
225  kES(iRow,iCol) *= fInvQuantity;
226  }
227  }
228 
229  kES.IncrSortEigenStuffN();
230 
231  GVector<Real> kEVector = kES.GetEigenvector(0);
232  Real fInv = ((Real)1.0)/kEVector[4]; // beware zero divide
233  Real afCoeff[4];
234  for (iRow = 0; iRow < 4; iRow++)
235  {
236  afCoeff[iRow] = fInv*kEVector[iRow];
237  }
238 
239  rkCenter.X() = -((Real)0.5)*afCoeff[1];
240  rkCenter.Y() = -((Real)0.5)*afCoeff[2];
241  rkCenter.Z() = -((Real)0.5)*afCoeff[3];
242  rfRadius = Math<Real>::Sqrt(Math<Real>::FAbs(rkCenter.X()*rkCenter.X() +
243  rkCenter.Y()*rkCenter.Y() + rkCenter.Z()*rkCenter.Z() - afCoeff[0]));
244 
245  // For exact fit, numeric round-off errors may make the minimum
246  // eigenvalue just slightly negative. Return absolute value since
247  // application may rely on the return value being nonnegative.
248  return Math<Real>::FAbs(kES.GetEigenvalue(0));
249 }
