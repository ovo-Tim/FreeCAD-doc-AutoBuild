147 int TopoShapePy::PyInit(PyObject* args, PyObject* keywds)
148 {
149  static const std::array<const char*, 5> kwlist{ "shape",
150  "op",
151  "tag",
152  "hasher",
153  nullptr };
154  long tag = 0;
155  PyObject* pyHasher = nullptr;
156  const char* op = nullptr;
157  PyObject* pcObj = nullptr;
158  if (!Base::Wrapped_ParseTupleAndKeywords(args,
159  keywds,
160  "|OsiO!",
161  kwlist,
162  &pcObj,
163  &op,
164  &tag,
165  &App::StringHasherPy::Type,
166  &pyHasher)) {
167  return -1;
168  }
169  auto& self = *getTopoShapePtr();
170  self.Tag = tag;
171  if (pyHasher) {
172  self.Hasher = static_cast<App::StringHasherPy*>(pyHasher)->getStringHasherPtr();
173  }
174  auto shapes = getPyShapes(pcObj);
175  PY_TRY
176  {
177  if (shapes.size() == 1 && !op) {
178  auto s = shapes.front();
179  if (self.Tag) {
180  if ((s.Tag && self.Tag != s.Tag)
181  || (self.Hasher && s.getElementMapSize() && self.Hasher != s.Hasher)) {
182  s.reTagElementMap(self.Tag, self.Hasher);
183  }
184  else {
185  s.Tag = self.Tag;
186  s.Hasher = self.Hasher;
187  }
188  }
189  self = s;
190  }
191  else if (shapes.size()) {
192  if (!op) {
193  op = Part::OpCodes::Fuse;
194  }
195  self.makeElementBoolean(op, shapes);
196  }
197  }
198  _PY_CATCH_OCC(return (-1))
199  return 0;
200 }
