40 class HashableShape_Deep(object):
41  """Similar to HashableShape, except that the things the shape is composed of are compared.
42 
43 Example:
44  >>> wire2 = Part.Wire(wire1.childShapes())
45  >>> wire2.isSame(wire1)
46  False # <--- the wire2 is a new wire, although made of edges of wire1
47  >>> HashableShape_Deep(wire2) == HashableShape_Deep(wire1)
48  True # <--- made of same set of elements
49  """
50 
51  def __init__(self, shape):
52  self.Shape = shape
53  self.hash = 0
54  for el in shape.childShapes():
55  self.hash = self.hash ^ el.hashCode()
56 
57  def __eq__(self, other):
58  # avoiding extensive comparison for now. Just doing a few extra tests should reduce the already-low chances of false-positives
59  if self.hash == other.hash:
60  if len(self.Shape.childShapes()) == len(other.Shape.childShapes()):
61  if self.Shape.ShapeType == other.Shape.ShapeType:
62  return True
63  return False
64 
65  def __hash__(self):
66  return self.hash

68 def compoundLeaves(shape_or_compound):
69  """compoundLeaves(shape_or_compound): extracts all non-compound shapes from a nested compound.
70  Note: shape_or_compound may be a non-compound; then, it is the only thing in the
71  returned list."""
72 
73  if shape_or_compound.ShapeType == "Compound":
74  leaves = []
75  for child in shape_or_compound.childShapes():
76  leaves.extend( compoundLeaves(child) )
77  return leaves
78  else:
79  return [shape_or_compound]

81 def upgradeToAggregateIfNeeded(list_of_shapes, types = None):
82  """upgradeToAggregateIfNeeded(list_of_shapes, types = None): upgrades non-aggregate type
83  shapes to aggregate-type shapes if the list has a mix of aggregate and non-aggregate
84  type shapes. Returns the new list. Recursively traverses into compounds.
85 
86  aggregate shape types are Wire, Shell, CompSolid
87  non-aggregate shape types are Vertex, Edge, Face, Solid
88  Compounds are something special: they are recursively traversed to upgrade the
89  contained shapes.
90 
91  Examples:
92  list_of_shapes contains only faces -> nothing happens
93  list_of_shapes contains faces and shells -> faces are converted to shells
94 
95  'types' argument is needed for recursive traversal. Do not supply."""
96 
97  import Part
98  if types is None:
99  types = set()
100  for shape in list_of_shapes:
101  types.add(shape.ShapeType)
102  subshapes = compoundLeaves(shape)
103  for subshape in subshapes:
104  types.add(subshape.ShapeType)
105  if "Wire" in types:
106  list_of_shapes = [(Part.Wire([shape]) if shape.ShapeType == "Edge" else shape) for shape in list_of_shapes]
107  if "Shell" in types:
108  list_of_shapes = [(Part.Shell([shape]) if shape.ShapeType == "Face" else shape) for shape in list_of_shapes]
109  if "CompSolid" in types:
110  list_of_shapes = [(Part.CompSolid([shape]) if shape.ShapeType == "Solid" else shape) for shape in list_of_shapes]
111  if "Compound" in types:
112  list_of_shapes = [(Part.Compound(upgradeToAggregateIfNeeded(shape.childShapes(), types)) if shape.ShapeType == "Compound" else shape) for shape in list_of_shapes]
113  return list_of_shapes
