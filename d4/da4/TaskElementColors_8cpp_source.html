143  void addItem(int i, const char* sub, bool push = false)
144  {
145  auto itE = elements.find(sub);
146  if (i < 0 && itE != elements.end()) {
147  if (push && !ViewProvider::hasHiddenMarker(sub)) {
148  items.push_back(itE->second);
149  }
150  return;
151  }
152 
153  const char* marker = ViewProvider::hasHiddenMarker(sub);
154  if (marker) {
155  auto icon = BitmapFactory().pixmap("Invisible");
156  auto item =
157  new QListWidgetItem(icon,
158  QString::fromLatin1(std::string(sub, marker - sub).c_str()),
159  ui->elementList);
160  item->setData(Qt::UserRole, QColor());
161  item->setData(Qt::UserRole + 1, QString::fromLatin1(sub));
162  elements.emplace(sub, item);
163  return;
164  }
165 
166  for (auto& v : vp->getElementColors(sub)) {
167  auto it = elements.find(v.first);
168  if (it != elements.end()) {
169  if (push) {
170  items.push_back(it->second);
171  }
172  continue;
173  }
174  auto color = v.second;
175  QColor c;
176  c.setRgbF(color.r, color.g, color.b, 1.0 - color.a);
177  px.fill(c);
178  auto item = new QListWidgetItem(
179  QIcon(px),
180  QString::fromLatin1(Data::oldElementName(v.first.c_str()).c_str()),
181  ui->elementList);
182  item->setData(Qt::UserRole, c);
183  item->setData(Qt::UserRole + 1, QString::fromLatin1(v.first.c_str()));
184  if (push) {
185  items.push_back(item);
186  }
187  elements.emplace(v.first, item);
188  }
189  }

273  void onSelectionChanged(const SelectionChanges& msg)
274  {
275  // no object selected in the combobox or no sub-element was selected
276  if (busy) {
277  return;
278  }
279  busy = true;
280  switch (msg.Type) {
281  case SelectionChanges::ClrSelection:
282  ui->elementList->clearSelection();
283  break;
284  case SelectionChanges::AddSelection:
285  case SelectionChanges::RmvSelection:
286  if (msg.pDocName && msg.pObjectName && msg.pSubName && msg.pSubName[0]) {
287  if (editDoc == msg.pDocName && editObj == msg.pObjectName
288  && boost::starts_with(msg.pSubName, editSub)) {
289  const auto items = ui->elementList->findItems(
290  QString::fromLatin1(msg.pSubName - editSub.size()),
291  Qt::MatchExactly);
292  for (auto item : items) {
293  item->setSelected(msg.Type == SelectionChanges::AddSelection);
294  }
295  }
296  }
297  default:
298  break;
299  }
300  busy = false;
301  }
302 
303  void onSelectionChanged()
304  {
305  if (busy) {
306  return;
307  }
308  busy = true;
309  std::map<std::string, int> sels;
310  for (auto& sel : Selection().getSelectionEx(editDoc.c_str(),
311  App::DocumentObject::getClassTypeId(),
312  ResolveMode::NoResolve)) {
313  if (sel.getFeatName() != editObj) {
314  continue;
315  }
316  for (auto& sub : sel.getSubNames()) {
317  if (boost::starts_with(sub, editSub)) {
318  sels[sub.c_str() + editSub.size()] = 1;
319  }
320  }
321  break;
322  }
323  const auto items = ui->elementList->selectedItems();
324  for (auto item : items) {
325  std::string name(qPrintable(item->data(Qt::UserRole + 1).value<QString>()));
326  if (ViewProvider::hasHiddenMarker(name.c_str())) {
327  continue;
328  }
329  auto& v = sels[name];
330  if (!v) {
331  Selection().addSelection(editDoc.c_str(),
332  editObj.c_str(),
333  (editSub + name).c_str());
334  }
335  v = 2;
336  }
337  for (auto& v : sels) {
338  if (v.second != 2) {
339  Selection().rmvSelection(editDoc.c_str(),
340  editObj.c_str(),
341  (editSub + v.first).c_str());
342  }
343  }
344  busy = false;
345  }
