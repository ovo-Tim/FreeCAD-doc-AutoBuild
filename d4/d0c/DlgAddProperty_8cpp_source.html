97 void DlgAddProperty::accept()
98 {
99  std::string name = ui->edtName->text().toUtf8().constData();
100  std::string group = ui->edtGroup->text().toUtf8().constData();
101  if(name.empty() || group.empty()
102  || name != Base::Tools::getIdentifier(name)
103  || group != Base::Tools::getIdentifier(group))
104  {
105  QMessageBox::critical(getMainWindow(),
106  QObject::tr("Invalid name"),
107  QObject::tr("The property name or group name must only contain alpha numericals,\n"
108  "underscore, and must not start with a digit."));
109  return;
110  }
111 
112  if(ui->chkAppend->isChecked())
113  name = group + "_" + name;
114 
115  if (App::ExpressionParser::isTokenAUnit(name) || App::ExpressionParser::isTokenAConstant(name)) {
116  QMessageBox::critical(getMainWindow(),
117  QObject::tr("Invalid name"),
118  QObject::tr("The property name is a reserved word."));
119  return;
120  }
121 
122  for(auto c : containers) {
123  auto prop = c->getPropertyByName(name.c_str());
124  if(prop && prop->getContainer() == c) {
125  QMessageBox::critical(getMainWindow(),
126  QObject::tr("Invalid name"),
127  QObject::tr("The property '%1' already exists in '%2'").arg(
128  QString::fromLatin1(name.c_str()),
129  QString::fromLatin1(containerName(c).c_str())));
130  return;
131  }
132  }
133 
134  std::string type = ui->comboType->currentText().toLatin1().constData();
135 
136  for(auto it=containers.begin();it!=containers.end();++it) {
137  try {
138  (*it)->addDynamicProperty(type.c_str(),name.c_str(),
139  group.c_str(),ui->edtDoc->toPlainText().toUtf8().constData());
140  } catch(Base::Exception &e) {
141  e.ReportException();
142  for(auto it2=containers.begin();it2!=it;++it2) {
143  try {
144  (*it2)->removeDynamicProperty(name.c_str());
145  } catch(Base::Exception &e) {
146  e.ReportException();
147  }
148  }
149  QMessageBox::critical(getMainWindow(),
150  QObject::tr("Add property"),
151  QObject::tr("Failed to add property to '%1': %2").arg(
152  QString::fromLatin1(containerName(*it).c_str()),
153  QString::fromUtf8(e.what())));
154  return;
155  }
156  }
157  auto hGrp = App::GetApplication().GetParameterGroupByPath(
158  "User parameter:BaseApp/Preferences/PropertyView");
159  hGrp->SetASCII("NewPropertyType",type.c_str());
160  hGrp->SetASCII("NewPropertyGroup",group.c_str());
161  hGrp->SetBool("NewPropertyAppend",ui->chkAppend->isChecked());
162  QDialog::accept();
163 }
