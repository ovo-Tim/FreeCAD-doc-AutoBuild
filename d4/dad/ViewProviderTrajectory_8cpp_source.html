126 void ViewProviderTrajectory::updateData(const App::Property* prop)
127 {
128  Robot::TrajectoryObject* pcTracObj = static_cast<Robot::TrajectoryObject*>(pcObject);
129  if (prop == &pcTracObj->Trajectory) {
130  const Trajectory& trak = pcTracObj->Trajectory.getValue();
131 
132  pcCoords->point.deleteValues(0);
133  pcCoords->point.setNum(trak.getSize());
134 
135  for (unsigned int i = 0; i < trak.getSize(); ++i) {
136  Base::Vector3d pos = trak.getWaypoint(i).EndPos.getPosition();
137  pcCoords->point.set1Value(i, pos.x, pos.y, pos.z);
138  }
139  pcLines->numVertices.set1Value(0, trak.getSize());
140  }
141 }
