257 PyObject* createWeakRef(PyObjectBase* ptr)
258 {
259  static bool init = false;
260  if (!init) {
261  init = true;
262  if (PyType_Ready(&PyBaseProxyType) < 0) {
263  return nullptr;
264  }
265  }
266 
267  PyObject* proxy = ptr->baseProxy;
268  if (!proxy) {
269  proxy = PyType_GenericAlloc(&PyBaseProxyType, 0);
270  ptr->baseProxy = proxy;
271  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
272  reinterpret_cast<PyBaseProxy*>(proxy)->baseobject = ptr;
273  }
274 
275  PyObject* ref = PyWeakref_NewRef(proxy, nullptr);
276  return ref;
277 }

299 PyObject* PyObjectBase::__getattro(PyObject * obj, PyObject *attro)
300 {
301  const char *attr{};
302  attr = PyUnicode_AsUTF8(attro);
303 
304  // For the __class__ attribute get it directly as with
305  // ExtensionContainerPy::getCustomAttributes we may get
306  // the wrong type object (#0003311)
307  if (streq(attr, "__class__")) {
308  PyObject* res = PyObject_GenericGetAttr(obj, attro);
309  if (res) {
310  return res;
311  }
312  }
313 
314  // This should be the entry in Type
315  PyObjectBase* pyObj = static_cast<PyObjectBase*>(obj);
316  if (!pyObj->isValid()){
317  PyErr_Format(PyExc_ReferenceError, "Cannot access attribute '%s' of deleted object", attr);
318  return nullptr;
319  }
320 
321 #ifdef ATTR_TRACKING
322  // If an attribute references this as parent then reset it (bug #0002902)
323  PyObject* cur = pyObj->getTrackedAttribute(attr);
324  if (cur) {
325  if (PyObject_TypeCheck(cur, &(PyObjectBase::Type))) {
326  PyObjectBase* base = static_cast<PyObjectBase*>(cur);
327  base->resetAttribute();
328  pyObj->untrackAttribute(attr);
329  }
330  }
331 #endif
332 
333  PyObject* value = pyObj->_getattr(attr);
334 #ifdef ATTR_TRACKING
335  if (value && PyObject_TypeCheck(value, &(PyObjectBase::Type))) {
336  if (!static_cast<PyObjectBase*>(value)->isConst() &&
337  !static_cast<PyObjectBase*>(value)->isNotTracking()) {
338  static_cast<PyObjectBase*>(value)->setAttributeOf(attr, pyObj);
339  pyObj->trackAttribute(attr, value);
340  }
341  }
342  else
343 #endif
344  if (value && PyCFunction_Check(value)) {
345  // ExtensionContainerPy::initialization() transfers the methods of an
346  // extension object by creating PyCFunction objects.
347  // At this point no 'self' object is passed but is handled and determined
348  // in ExtensionContainerPy::getCustomAttributes().
349  // So, if we come through this section then it's an indication that
350  // something is wrong with the Python types. For example, a C++ class
351  // that adds an extension uses the same Python type as a wrapper than
352  // another C++ class without this extension.
353  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
354  PyCFunctionObject* cfunc = reinterpret_cast<PyCFunctionObject*>(value);
355  if (!cfunc->m_self) {
356  Py_DECREF(cfunc);
357  value = nullptr;
358  PyErr_Format(PyExc_AttributeError, "<no object bound to built-in method %s>", attr);
359  }
360  }
361 
362  return value;
363 }

365 int PyObjectBase::__setattro(PyObject *obj, PyObject *attro, PyObject *value)
366 {
367  const char *attr{};
368  attr = PyUnicode_AsUTF8(attro);
369 
370  //Hint: In general we don't allow one to delete attributes (i.e. value=0). However, if we want to allow
371  //we must check then in _setattr() of all subclasses whether value is 0.
372  if (!value) {
373  PyErr_Format(PyExc_AttributeError, "Cannot delete attribute: '%s'", attr);
374  return -1;
375  }
376  if (!static_cast<PyObjectBase*>(obj)->isValid()){
377  PyErr_Format(PyExc_ReferenceError, "Cannot access attribute '%s' of deleted object", attr);
378  return -1;
379  }
380 
381 #ifdef ATTR_TRACKING
382  // If an attribute references this as parent then reset it
383  // before setting the new attribute
384  PyObject* cur = static_cast<PyObjectBase*>(obj)->getTrackedAttribute(attr);
385  if (cur) {
386  if (PyObject_TypeCheck(cur, &(PyObjectBase::Type))) {
387  PyObjectBase* base = static_cast<PyObjectBase*>(cur);
388  base->resetAttribute();
389  static_cast<PyObjectBase*>(obj)->untrackAttribute(attr);
390  }
391  }
392 #endif
393 
394  int ret = static_cast<PyObjectBase*>(obj)->_setattr(attr, value);
395 #ifdef ATTR_TRACKING
396  if (ret == 0) {
397  static_cast<PyObjectBase*>(obj)->startNotify();
398  }
399 #endif
400  return ret;
401 }

406 PyObject *PyObjectBase::_getattr(const char *attr)
407 {
408  if (streq(attr, "__class__")) {
409  // Note: We must return the type object here,
410  // so that our own types feel as really Python objects
411  Py_INCREF(Py_TYPE(this));
412  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
413  return reinterpret_cast<PyObject *>(Py_TYPE(this));
414  }
415  if (streq(attr, "__members__")) {
416  // Use __dict__ instead as __members__ is deprecated
417  return nullptr;
418  }
419  if (streq(attr,"__dict__")) {
420  // Return the default dict
421  PyTypeObject *tp = Py_TYPE(this);
422  Py_XINCREF(tp->tp_dict);
423  return tp->tp_dict;
424  }
425  if (streq(attr,"softspace")) {
426  // Internal Python stuff
427  return nullptr;
428  }
429  // As fallback solution use Python's default method to get generic attributes
430  PyObject *w{}, *res{};
431  w = PyUnicode_InternFromString(attr);
432  if (w) {
433  res = PyObject_GenericGetAttr(this, w);
434  Py_XDECREF(w);
435  return res;
436  }
437  // Throw an exception for unknown attributes
438  PyTypeObject *tp = Py_TYPE(this);
439  PyErr_Format(PyExc_AttributeError, "%.50s instance has no attribute '%.400s'", tp->tp_name, attr);
440  return nullptr;
441 }

443 int PyObjectBase::_setattr(const char *attr, PyObject *value)
444 {
445  if (streq(attr,"softspace")) {
446  return -1; // filter out softspace
447  }
448  PyObject *w{};
449  // As fallback solution use Python's default method to get generic attributes
450  w = PyUnicode_InternFromString(attr); // new reference
451  if (w) {
452  // call methods from tp_getset if defined
453  int res = PyObject_GenericSetAttr(this, w, value);
454  Py_DECREF(w);
455  return res;
456  }
457  // Throw an exception for unknown attributes
458  PyTypeObject *tp = Py_TYPE(this);
459  PyErr_Format(PyExc_AttributeError, "%.50s instance has no attribute '%.400s'", tp->tp_name, attr);
460  return -1;
461 }

478 void PyObjectBase::resetAttribute()
479 {
480  if (attrDict) {
481  // This is the attribute name to the parent structure
482  // which we search for in the dict
483  PyObject* key1 = PyBytes_FromString("__attribute_of_parent__");
484  PyObject* key2 = PyBytes_FromString("__instance_of_parent__");
485  PyObject* attr = PyDict_GetItem(attrDict, key1);
486  PyObject* inst = PyDict_GetItem(attrDict, key2);
487  if (attr) {
488  PyDict_DelItem(attrDict, key1);
489  }
490  if (inst) {
491  PyDict_DelItem(attrDict, key2);
492  }
493  Py_DECREF(key1);
494  Py_DECREF(key2);
495  }
496 }

514 void PyObjectBase::startNotify()
515 {
516  if (!shouldNotify()) {
517  return;
518  }
519 
520  if (attrDict) {
521  // This is the attribute name to the parent structure
522  // which we search for in the dict
523  PyObject* key1 = PyBytes_FromString("__attribute_of_parent__");
524  PyObject* key2 = PyBytes_FromString("__instance_of_parent__");
525  PyObject* attr = PyDict_GetItem(attrDict, key1);
526  PyObject* parent = PyDict_GetItem(attrDict, key2);
527  if (attr && parent) {
528  // Inside __setattr of the parent structure the 'attr'
529  // is being removed from the dict and thus its reference
530  // counter will be decremented. To avoid to be deleted we
531  // must tmp. increment it and afterwards decrement it again.
532  Py_INCREF(parent);
533  Py_INCREF(attr);
534  Py_INCREF(this);
535 
536  __setattro(parent, attr, this);
537 
538  Py_DECREF(parent); // might be destroyed now
539  Py_DECREF(attr); // might be destroyed now
540  Py_DECREF(this); // might be destroyed now
541 
542  if (PyErr_Occurred()) {
543  PyErr_Clear();
544  }
545  }
546  Py_DECREF(key1);
547  Py_DECREF(key2);
548  }
549 }
