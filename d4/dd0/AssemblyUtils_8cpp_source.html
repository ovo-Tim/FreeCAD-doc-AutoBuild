64 void swapJCS(const App::DocumentObject* joint)
65 {
66  if (!joint) {
67  return;
68  }
69 
70  auto pPlc1 = joint->getPropertyByName<App::PropertyPlacement>("Placement1");
71  auto pPlc2 = joint->getPropertyByName<App::PropertyPlacement>("Placement2");
72  if (pPlc1 && pPlc2) {
73  const auto temp = pPlc1->getValue();
74  pPlc1->setValue(pPlc2->getValue());
75  pPlc2->setValue(temp);
76  }
77  auto pRef1 = joint->getPropertyByName<App::PropertyXLinkSub>("Reference1");
78  auto pRef2 = joint->getPropertyByName<App::PropertyXLinkSub>("Reference2");
79  if (pRef1 && pRef2) {
80  auto temp = pRef1->getValue();
81  auto subs1 = pRef1->getSubValues();
82  auto subs2 = pRef2->getSubValues();
83  pRef1->setValue(pRef2->getValue());
84  pRef1->setSubValues(std::move(subs2));
85  pRef2->setValue(temp);
86  pRef2->setSubValues(std::move(subs1));
87  }
88 }

91  const std::string& elName,
92  const GeomAbs_CurveType type)
93 {

95  if (!base) {
96  return false;

98 
99  const auto& TopShape = base->Shape.getShape();
100 
101  // Check for valid face types
102  const auto edge = TopoDS::Edge(TopShape.getSubShape(elName.c_str()));

104 
105  return sf.GetType() == type;

126 double getFaceRadius(const App::DocumentObject* obj, const std::string& elt)
127 {
128  auto* base = dynamic_cast<const PartApp::Feature*>(obj);
129  if (!base) {
130  return 0.0;
131  }
132 
133  const PartApp::TopoShape& TopShape = base->Shape.getShape();
134 
135  // Check for valid face types
136  TopoDS_Face face = TopoDS::Face(TopShape.getSubShape(elt.c_str()));

138 
139  const auto type = sf.GetType();

161 DistanceType getDistanceType(App::DocumentObject* joint)
162 {
163  if (!joint) {
164  return DistanceType::Other;
165  }
166 
167  const auto type1 = getElementTypeFromProp(joint, "Reference1");
168  const auto type2 = getElementTypeFromProp(joint, "Reference2");
169  auto elt1 = getElementFromProp(joint, "Reference1");
170  auto elt2 = getElementFromProp(joint, "Reference2");
171  auto* obj1 = getLinkedObjFromRef(joint, "Reference1");
172  auto* obj2 = getLinkedObjFromRef(joint, "Reference2");
173 
174  if (type1 == "Vertex" && type2 == "Vertex") {
175  return DistanceType::PointPoint;
176  }
177  else if (type1 == "Edge" && type2 == "Edge") {
178  if (isEdgeType(obj1, elt1, GeomAbs_Line) || isEdgeType(obj2, elt2, GeomAbs_Line)) {
179  if (!isEdgeType(obj1, elt1, GeomAbs_Line)) {
180  swapJCS(joint); // make sure that line is first if not 2 lines.
181  std::swap(elt1, elt2);
182  std::swap(obj1, obj2);
183  }
184 
185  if (isEdgeType(obj2, elt2, GeomAbs_Line)) {
186  return DistanceType::LineLine;
187  }
188  else if (isEdgeType(obj2, elt2, GeomAbs_Circle)) {
189  return DistanceType::LineCircle;
190  }
191  // TODO : other cases Ellipse, parabola, hyperbola...
192  }
193 
194  else if (isEdgeType(obj1, elt1, GeomAbs_Circle) || isEdgeType(obj2, elt2, GeomAbs_Circle)) {
195  if (!isEdgeType(obj1, elt1, GeomAbs_Circle)) {
196  swapJCS(joint); // make sure that circle is first if not 2 lines.
197  std::swap(elt1, elt2);
198  std::swap(obj1, obj2);
199  }
200 
201  if (isEdgeType(obj2, elt2, GeomAbs_Circle)) {
202  return DistanceType::CircleCircle;
203  }
204  // TODO : other cases Ellipse, parabola, hyperbola...
205  }
206  }
207  else if (type1 == "Face" && type2 == "Face") {
208  if (isFaceType(obj1, elt1, GeomAbs_Plane) || isFaceType(obj2, elt2, GeomAbs_Plane)) {
209  if (!isFaceType(obj1, elt1, GeomAbs_Plane)) {
210  swapJCS(joint); // make sure plane is first if its not 2 planes.
211  std::swap(elt1, elt2);
212  std::swap(obj1, obj2);
213  }
214 
215  if (isFaceType(obj2, elt2, GeomAbs_Plane)) {
216  return DistanceType::PlanePlane;
217  }
218  else if (isFaceType(obj2, elt2, GeomAbs_Cylinder)) {
219  return DistanceType::PlaneCylinder;
220  }
221  else if (isFaceType(obj2, elt2, GeomAbs_Sphere)) {
222  return DistanceType::PlaneSphere;
223  }
224  else if (isFaceType(obj2, elt2, GeomAbs_Cone)) {
225  return DistanceType::PlaneCone;
226  }
227  else if (isFaceType(obj2, elt2, GeomAbs_Torus)) {
228  return DistanceType::PlaneTorus;
229  }
230  }
231 
232  else if (isFaceType(obj1, elt1, GeomAbs_Cylinder)
233  || isFaceType(obj2, elt2, GeomAbs_Cylinder)) {
234  if (!isFaceType(obj1, elt1, GeomAbs_Cylinder)) {
235  swapJCS(joint); // make sure cylinder is first if its not 2 cylinders.
236  std::swap(elt1, elt2);
237  std::swap(obj1, obj2);
238  }
239 
240  if (isFaceType(obj2, elt2, GeomAbs_Cylinder)) {
241  return DistanceType::CylinderCylinder;
242  }
243  else if (isFaceType(obj2, elt2, GeomAbs_Sphere)) {
244  return DistanceType::CylinderSphere;
245  }
246  else if (isFaceType(obj2, elt2, GeomAbs_Cone)) {
247  return DistanceType::CylinderCone;
248  }
249  else if (isFaceType(obj2, elt2, GeomAbs_Torus)) {
250  return DistanceType::CylinderTorus;
251  }
252  }
253 
254  else if (isFaceType(obj1, elt1, GeomAbs_Cone) || isFaceType(obj2, elt2, GeomAbs_Cone)) {
255  if (!isFaceType(obj1, elt1, GeomAbs_Cone)) {
256  swapJCS(joint); // make sure cone is first if its not 2 cones.
257  std::swap(elt1, elt2);
258  std::swap(obj1, obj2);
259  }
260 
261  if (isFaceType(obj2, elt2, GeomAbs_Cone)) {
262  return DistanceType::ConeCone;
263  }
264  else if (isFaceType(obj2, elt2, GeomAbs_Torus)) {
265  return DistanceType::ConeTorus;
266  }
267  else if (isFaceType(obj2, elt2, GeomAbs_Sphere)) {
268  return DistanceType::ConeSphere;
269  }
270  }
271 
272  else if (isFaceType(obj1, elt1, GeomAbs_Torus) || isFaceType(obj2, elt2, GeomAbs_Torus)) {
273  if (!isFaceType(obj1, elt1, GeomAbs_Torus)) {
274  swapJCS(joint); // make sure torus is first if its not 2 torus.
275  std::swap(elt1, elt2);
276  std::swap(obj1, obj2);
277  }
278 
279  if (isFaceType(obj2, elt2, GeomAbs_Torus)) {
280  return DistanceType::TorusTorus;
281  }
282  else if (isFaceType(obj2, elt2, GeomAbs_Sphere)) {
283  return DistanceType::TorusSphere;
284  }
285  }
286 
287  else if (isFaceType(obj1, elt1, GeomAbs_Sphere) || isFaceType(obj2, elt2, GeomAbs_Sphere)) {
288  if (!isFaceType(obj1, elt1, GeomAbs_Sphere)) {
289  swapJCS(joint); // make sure sphere is first if its not 2 spheres.
290  std::swap(elt1, elt2);
291  std::swap(obj1, obj2);
292  }
293 
294  if (isFaceType(obj2, elt2, GeomAbs_Sphere)) {
295  return DistanceType::SphereSphere;
296  }
297  }
298  }
299  else if ((type1 == "Vertex" && type2 == "Face") || (type1 == "Face" && type2 == "Vertex")) {
300  if (type1 == "Vertex") { // Make sure face is the first.
301  swapJCS(joint);
302  std::swap(elt1, elt2);
303  std::swap(obj1, obj2);
304  }
305  if (isFaceType(obj1, elt1, GeomAbs_Plane)) {
306  return DistanceType::PointPlane;
307  }
308  else if (isFaceType(obj1, elt1, GeomAbs_Cylinder)) {
309  return DistanceType::PointCylinder;
310  }
311  else if (isFaceType(obj1, elt1, GeomAbs_Sphere)) {
312  return DistanceType::PointSphere;
313  }
314  else if (isFaceType(obj1, elt1, GeomAbs_Cone)) {
315  return DistanceType::PointCone;
316  }
317  else if (isFaceType(obj1, elt1, GeomAbs_Torus)) {
318  return DistanceType::PointTorus;
319  }
320  }
321  else if ((type1 == "Edge" && type2 == "Face") || (type1 == "Face" && type2 == "Edge")) {
322  if (type1 == "Edge") { // Make sure face is the first.
323  swapJCS(joint);
324  std::swap(elt1, elt2);
325  std::swap(obj1, obj2);
326  }
327  if (isEdgeType(obj2, elt2, GeomAbs_Line)) {
328  if (isFaceType(obj1, elt1, GeomAbs_Plane)) {
329  return DistanceType::LinePlane;
330  }
331  else if (isFaceType(obj1, elt1, GeomAbs_Cylinder)) {
332  return DistanceType::LineCylinder;
333  }
334  else if (isFaceType(obj1, elt1, GeomAbs_Sphere)) {
335  return DistanceType::LineSphere;
336  }
337  else if (isFaceType(obj1, elt1, GeomAbs_Cone)) {
338  return DistanceType::LineCone;
339  }
340  else if (isFaceType(obj1, elt1, GeomAbs_Torus)) {
341  return DistanceType::LineTorus;
342  }
343  }
344  else {
345  // For other curves we consider them as planes for now. Can be refined later.
346  if (isFaceType(obj1, elt1, GeomAbs_Plane)) {
347  return DistanceType::CurvePlane;
348  }
349  else if (isFaceType(obj1, elt1, GeomAbs_Cylinder)) {
350  return DistanceType::CurveCylinder;
351  }
352  else if (isFaceType(obj1, elt1, GeomAbs_Sphere)) {
353  return DistanceType::CurveSphere;
354  }
355  else if (isFaceType(obj1, elt1, GeomAbs_Cone)) {
356  return DistanceType::CurveCone;
357  }
358  else if (isFaceType(obj1, elt1, GeomAbs_Torus)) {
359  return DistanceType::CurveTorus;
360  }
361  }
362  }
363  else if ((type1 == "Vertex" && type2 == "Edge") || (type1 == "Edge" && type2 == "Vertex")) {
364  if (type1 == "Vertex") { // Make sure edge is the first.
365  swapJCS(joint);
366  std::swap(elt1, elt2);
367  std::swap(obj1, obj2);
368  }
369  if (isEdgeType(obj1, elt1, GeomAbs_Line)) { // Point on line joint.
370  return DistanceType::PointLine;
371  }
372  else {
373  // For other curves we do a point in plane-of-the-curve.
374  // Maybe it would be best tangent / distance to the conic? For arcs and
375  // circles we could use ASMTRevSphJoint. But is it better than pointInPlane?
376  return DistanceType::PointCurve;
377  }
378  }
379  return DistanceType::Other;
380 }

525 App::DocumentObject* getObjFromRef(const App::DocumentObject* obj, const std::string& sub)
526 {
527  if (!obj) {
528  return nullptr;
529  }
530 
531  const auto* doc = obj->getDocument();
532  const auto names = Base::Tools::splitSubName(sub);
533 
534  // Lambda function to check if the typeId is a BodySubObject
535  const auto isBodySubObject = [](App::DocumentObject* obj) -> bool {
536  // PartDesign::Point + Line + Plane + CoordinateSystem
537  // getViewProviderName instead of isDerivedFrom to avoid dependency on sketcher
538  const auto isDerivedFromVpSketch =
539  strcmp(obj->getViewProviderName(), "SketcherGui::ViewProviderSketch") == 0;
540  return isDerivedFromVpSketch || obj->isDerivedFrom<PartApp::Datum>()
541  || obj->isDerivedFrom<App::DatumElement>()
542  || obj->isDerivedFrom<App::LocalCoordinateSystem>();
543  };
544 
545  // Helper function to handle PartDesign::Body objects
546  const auto handlePartDesignBody =
547  [&](App::DocumentObject* obj,
548  std::vector<std::string>::const_iterator it) -> App::DocumentObject* {
549  const auto nextIt = std::next(it);
550  if (nextIt != names.end()) {
551  for (auto* obji : obj->getOutList()) {
552  if (*nextIt == obji->getNameInDocument() && isBodySubObject(obji)) {
553  return obji;
554  }
555  }
556  }
557  return obj;
558  };
559 
560 
561  for (auto it = names.begin(); it != names.end(); ++it) {
562  App::DocumentObject* obj = doc->getObject(it->c_str());
563  if (!obj) {
564  return nullptr;
565  }
566 
567  if (obj->isDerivedFrom<App::DocumentObjectGroup>()) {
568  continue;
569  }
570 
571  // The last but one name should be the selected
572  if (std::next(it) == std::prev(names.end())) {
573  return obj;
574  }
575 
576  if (obj->isDerivedFrom<App::Part>() || obj->isLinkGroup()) {
577  continue;
578  }
579  else if (obj->isDerivedFrom<PartDesign::Body>()) {
580  return handlePartDesignBody(obj, it);
581  }
582  else if (obj->isDerivedFrom<PartApp::Feature>()) {
583  // Primitive, fastener, gear, etc.
584  return obj;
585  }
586  else if (obj->isLink()) {
587  App::DocumentObject* linked_obj = obj->getLinkedObject();
588  if (linked_obj->isDerivedFrom<PartDesign::Body>()) {
589  auto* retObj = handlePartDesignBody(linked_obj, it);
590  return retObj == linked_obj ? obj : retObj;
591  }
592  else if (linked_obj->isDerivedFrom<PartApp::Feature>()) {
593  return obj;
594  }
595  else {
596  doc = linked_obj->getDocument();
597  continue;
598  }
599  }
600  }
601 
602  return nullptr;
603 }

646 App::DocumentObject* getMovingPartFromRef(const AssemblyObject* assemblyObject,
647  App::DocumentObject* obj,
648  const std::string& sub)
649 {
650  if (!obj) {
651  return nullptr;
652  }
653 
654  auto* doc = obj->getDocument();
655 
656  auto names = Base::Tools::splitSubName(sub);
657  names.insert(names.begin(), obj->getNameInDocument());
658 
659  bool assemblyPassed = false;
660 
661  for (const auto& objName : names) {
662  obj = doc->getObject(objName.c_str());
663  if (!obj) {
664  continue;
665  }
666 
667  if (obj->isLink()) { // update the document if necessary for next object
668  doc = obj->getLinkedObject()->getDocument();
669  }
670 
671  if (obj == assemblyObject) {
672  // We make sure we pass the assembly for cases like part.assembly.part.body
673  assemblyPassed = true;
674  continue;
675  }
676  if (!assemblyPassed) {
677  continue;
678  }
