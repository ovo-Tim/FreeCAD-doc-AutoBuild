63 std::string App::quote(const std::string& input, bool toPython)
64 {
65  std::stringstream output;
66 
67  std::string::const_iterator cur = input.begin();
68  std::string::const_iterator end = input.end();
69 
70  output << (toPython ? "'" : "<<");
71  while (cur != end) {
72  switch (*cur) {
73  case '\t':
74  output << "\\t";
75  break;
76  case '\n':
77  output << "\\n";
78  break;
79  case '\r':
80  output << "\\r";
81  break;
82  case '\\':
83  output << "\\\\";
84  break;
85  case '\'':
86  output << "\\'";
87  break;
88  case '"':
89  output << "\\\"";
90  break;
91  case '>':
92  output << (toPython ? ">" : "\\>");
93  break;
94  default:
95  output << *cur;
96  }
97  ++cur;
98  }
99  output << (toPython ? "'" : ">>");
100 
101  return output.str();
102 }

316 {

319  if (silent) {

323  }

335  if (silent) {
336  return false;
337  }
338  FC_THROWM(Base::ValueError, "Invalid property path: name mismatch");
339  }
340  return true;

469  const App::DocumentObject* parent,
470  App::DocumentObject* oldObj,
471  App::DocumentObject* newObj) const
472 {
473  ResolveResults result(*this);
474 
475  if (!result.resolvedDocumentObject) {
476  return false;
477  }
478 
479  auto r = PropertyLinkBase::tryReplaceLink(owner,
480  result.resolvedDocumentObject,
481  parent,
482  oldObj,
483  newObj,
484  subObjectName.getString().c_str());
485 
486  if (!r.first) {
487  return false;
488  }
489 
490  res = *this;
491  if (r.first != result.resolvedDocumentObject) {
492  if (r.first->getDocument() != owner->getDocument()) {
493  auto doc = r.first->getDocument();
494  bool useLabel = res.documentName.isRealString();
495  const char* name = useLabel ? doc->Label.getValue() : doc->getName();
496  res.setDocumentName(String(name, useLabel), true);
497  }
498  if (documentObjectName.isRealString()) {
499  res.documentObjectName = String(r.first->Label.getValue(), true);
500  }
501  else {
502  res.documentObjectName = String(r.first->getNameInDocument(), false, true);
503  }
504  }
505  res.subObjectName = String(r.second, true);
506  res._cache.clear();
507  res.shadowSub.newName.clear();
508  res.shadowSub.oldName.clear();
509  return true;

522 bool ObjectIdentifier::updateLabelReference(App::DocumentObject* obj,
523  const std::string& ref,
524  const char* newLabel)
525 {
526  if (!owner) {
527  return false;
528  }
529 
530  ResolveResults result(*this);
531 
532  if (!subObjectName.getString().empty() && result.resolvedDocumentObject) {
533  std::string sub = PropertyLinkBase::updateLabelReference(result.resolvedDocumentObject,
534  subObjectName.getString().c_str(),
535  obj,
536  ref,
537  newLabel);
538  if (!sub.empty()) {
539  subObjectName = String(sub, true);
540  _cache.clear();
541  return true;
542  }
543  }
544 
545  if (result.resolvedDocument != obj->getDocument()) {
546  return false;
547  }
548 
549  if (!documentObjectName.getString().empty()) {
550  if (documentObjectName.isForceIdentifier()) {
551  return false;
552  }
553 
554  if (!documentObjectName.isRealString()
555  && documentObjectName.getString() == obj->getNameInDocument()) {
556  return false;
557  }
558 
559  if (documentObjectName.getString() != obj->Label.getValue()) {
560  return false;
561  }
562 
563  documentObjectName = ObjectIdentifier::String(newLabel, true);
564 
565  _cache.clear();
566  return true;
567  }
568 
569  if (result.resolvedDocumentObject == obj && result.propertyIndex == 1
570  && result.resolvedDocumentObjectName.isRealString()
571  && result.resolvedDocumentObjectName.getString() == obj->Label.getValue()) {
572  components[0].name = ObjectIdentifier::String(newLabel, true);
573  _cache.clear();
574  return true;
575  }
576 
577  // If object identifier uses the label then resolving the document object will fail.
578  // So, it must be checked if using the new label will succeed
579  if (components.size() > 1 && components[0].getName() == obj->Label.getValue()) {
580  ObjectIdentifier id(*this);
581  id.components[0].name.str = newLabel;
582 
583  ResolveResults result(id);
584 
585  if (result.propertyIndex == 1 && result.resolvedDocumentObject == obj) {
586  components[0].name = id.components[0].name;
587  _cache.clear();
588  return true;
589  }
590  }
591 
592  return false;
593 }

684 Py::Object ObjectIdentifier::Component::get(const Py::Object& pyobj) const
685 {
686  Py::Object res;
687  if (isSimple()) {
688  if (!pyobj.hasAttr(getName())) {
689  FC_THROWM(Base::AttributeError, "No attribute named '" << getName() << "'");
690  }
691  res = pyobj.getAttr(getName());
692  }
693  else if (isArray()) {
694  if (pyobj.isMapping()) {
695  res = Py::Mapping(pyobj).getItem(Py::Int(begin));
696  }
697  else {
698  res = Py::Sequence(pyobj).getItem(begin);
699  }
700  }
701  else if (isMap()) {
702  res = Py::Mapping(pyobj).getItem(getName());
703  }
704  else {
705  assert(isRange());

707  end != INT_MAX ? Py::Int(end).ptr() : nullptr,
708  step != 1 ? Py::Int(step).ptr() : nullptr),
709  true);
710  PyObject* r = PyObject_GetItem(pyobj.ptr(), slice.ptr());
711  if (!r) {
712  Base::PyException::ThrowException();
713  }
714  res = Py::asObject(r);
715  }
716  if (!res.ptr()) {
717  Base::PyException::ThrowException();
718  }
719  if (PyModule_Check(res.ptr()) && !ExpressionParser::isModuleImported(res.ptr())) {
720  FC_THROWM(Base::RuntimeError, "Module '" << getName() << "' access denied.");
721  }
722  return res;

976  flags.set(ResolveByLabel);

981  flags.set(ResolveByLabel);

1721  switch (ptype) {
1722  case PseudoApp:
1723  GET_MODULE(FreeCAD);

1729  GET_MODULE(Part);
1730  break;
1731  case PseudoCadquery:
1732  GET_MODULE(freecad.fc_cadquery);
1733  break;
1734  case PseudoRegex:
1735  GET_MODULE(re);
1736  break;
1737  case PseudoBuiltins:
1738  GET_MODULE(builtins);
1739  break;
1740  case PseudoMath:
1741  GET_MODULE(math);
1742  break;
1743  case PseudoCollections:
1744  GET_MODULE(collections);
1745  break;
1746  case PseudoShape: {
1747  GET_MODULE(Part);
1748  Py::Callable func(pyobj.getAttr("getShape"));
1749  Py::Tuple tuple(1);
1750  tuple.setItem(0, Py::Object(result.resolvedDocumentObject->getPyObject(), true));
1751  if (result.subObjectName.getString().empty()) {
1752  pyobj = func.apply(tuple);
