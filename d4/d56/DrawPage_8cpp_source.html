97 void DrawPage::onChanged(const App::Property* prop)
98 {
99  if (prop == &KeepUpdated && KeepUpdated.getValue()) {
100  if (!isRestoring() && !isUnsetting()) {
101  //would be nice if this message was displayed immediately instead of after the recomputeFeature
102  Base::Console().Message("Rebuilding Views for: %s/%s\n", getNameInDocument(),
103  Label.getValue());
104  updateAllViews();
105  purgeTouched();
106  }
107  }
108  else if (prop == &Template) {
109  if (!isRestoring() && !isUnsetting()) {
110  //nothing to page to do??
111  }
112  }
113  else if (prop == &Scale) {
114  // touch all views in the Page as they may be dependent on this scale
115  // WF: not sure this loop is required. Views figure out their scale as required. but maybe
116  // this is needed just to mark the Views to recompute??
117  if (!isRestoring()) {
118  for (auto* obj : getViews()) {
119  auto* view = dynamic_cast<DrawView*>(obj);
120  if (view && view->ScaleType.isValue("Page")) {
121  if (std::abs(view->Scale.getValue() - Scale.getValue()) > FLT_EPSILON) {
122  view->Scale.setValue(Scale.getValue());
123  }
124  }
125  }
126  }
127  }
128  else if (prop == &ProjectionType) {
129  // touch all ortho views in the Page as they may be dependent on Projection Type //(is this true?)
130  for (auto* obj : getViews()) {
131  auto* view = dynamic_cast<DrawProjGroup*>(obj);
132  if (view && view->ProjectionType.isValue("Default")) {
133  view->ProjectionType.touch();
134  }
135  }
136 
137  // TODO: Also update Template graphic.
138  }
139  App::DocumentObject::onChanged(prop);
140 }

235 int DrawPage::addView(App::DocumentObject* docObj, bool setPosition)
236 {
237  if (!docObj->isDerivedFrom<DrawView>()
238  && !docObj->isDerivedFrom<App::Link>()) {
239  return -1;
240  }
241 
242  auto* view = dynamic_cast<DrawView*>(docObj);
243 
244  if (!view) {
245  auto* link = dynamic_cast<App::Link*>(docObj);
246  if (!link) {
247  return -1;
248  }
249 
250  view = dynamic_cast<DrawView*>(link->getLinkedObject());
251  if (!view) {
252  return -1;
253  }
254  }
255 
256  //position all new views without owners in center of Page (exceptDVDimension)
257  if (!view->claimParent()
258  && !docObj->isDerivedFrom<DrawViewDimension>()
259  && !docObj->isDerivedFrom<DrawViewBalloon>()
260  && setPosition) {
261  view->X.setValue(getPageWidth() / 2.0);
262  view->Y.setValue(getPageHeight() / 2.0);
263  }
264 
265  //add view to list
266  std::vector<App::DocumentObject*> newViews(Views.getValues());
267  newViews.push_back(docObj);
268  Views.setValues(newViews);
269 
270 
271  //check if View fits on Page
272  if (!view->checkFit(this)) {
273  Base::Console().Warning("%s is larger than page. Will be scaled.\n",
274  view->getNameInDocument());
275  view->ScaleType.setValue("Automatic");
276  }
277 
278  view->checkScale();
279 
280  return Views.getSize();
281 }

284 int DrawPage::removeView(App::DocumentObject* docObj)
285 {
286  if (!docObj->isDerivedFrom<DrawView>() && !docObj->isDerivedFrom<App::Link>()) {
287  return -1;
288  }
289 
290  App::Document* doc = docObj->getDocument();
291  if (!doc) {
292  return -1;
293  }
294 
295  if (!docObj->isAttachedToDocument()) {
296  return -1;
297  }
298  std::vector<App::DocumentObject*> newViews;
299  for (auto* view : Views.getValues()) {
300  App::Document* viewDoc = view->getDocument();
301  if (!viewDoc) {
302  continue;
303  }
304 
305  std::string viewName = docObj->getNameInDocument();
306  if (viewName.compare(view->getNameInDocument()) != 0) {
307  newViews.push_back(view);
308  }
309  }
310  Views.setValues(newViews);
311  return Views.getSize();
312 }

334 void DrawPage::updateAllViews()
335 {
336  // Base::Console().Message("DP::updateAllViews()\n");
337  //unordered list of views within page
338  std::vector<App::DocumentObject*> featViews = getAllViews();
339 
340  //first, make sure all the Parts have been executed so GeometryObjects exist
341  for (auto& v : featViews) {
342  auto* part = dynamic_cast<DrawViewPart*>(v);
343  if (part) {
344  //view, section, detail, dpgi
345  part->recomputeFeature();
346  }
347  }
348  //second, do the rest of the views that may depend on a part view
349  //TODO: check if we have 2 layers of dependency (ex. leader > weld > tile?)
350  for (auto& v : featViews) {
351  auto* part = dynamic_cast<DrawViewPart*>(v);
352  if (part) {
353  continue;
354  }
355 
356  auto* view = dynamic_cast<DrawView*>(v);
357  if (view) {
358  view->overrideKeepUpdated(true);
359  view->recomputeFeature();
360  }
361  }
362 }

364 std::vector<App::DocumentObject*> DrawPage::getViews() const
365 {
366  std::vector<App::DocumentObject*> views = Views.getValues();
367  std::vector<App::DocumentObject*> allViews;
368  for (auto& v : views) {
369  bool addChildren = false;
370 
371  if (v->isDerivedFrom<App::Link>()) {
372  // In the case of links, child object of the view need to be added since
373  // they are not in the page Views property.
374  v = static_cast<App::Link*>(v)->getLinkedObject();
375  addChildren = true;
376  }
377 
378  if (!v->isDerivedFrom<DrawView>()) {
379  continue;
380  }
381 
382  allViews.push_back(v);
383 
384  if (addChildren) {
385  for (auto* dep : v->getInList()) {
386  if (dep && dep->isDerivedFrom<TechDraw::DrawView>()) {
387  allViews.push_back(dep);
388  }
389  }
390  }
391  }
392  return allViews;
393 }

395 std::vector<App::DocumentObject*> DrawPage::getAllViews() const
396 {
397  std::vector<App::DocumentObject*> views = Views.getValues();
398  std::vector<App::DocumentObject*> allViews;
399  for (auto& v : views) {
400  if (v->isDerivedFrom<App::Link>()) {
401  v = static_cast<App::Link*>(v)->getLinkedObject();
402  }
403 
404  if (!v->isDerivedFrom<DrawView>()) {
405  continue;
406  }
407 
408  allViews.push_back(v);
409  if (v->isDerivedFrom<DrawProjGroup>()) {

415  }

420 void DrawPage::unsetupObject()
421 {
422  nowUnsetting = true;
423 
424  // Remove the Page's views & template from document
425  App::Document* doc = getDocument();
426  std::string docName = doc->getName();
427  std::string pageName = getNameInDocument();
428 
429  try {
430  for (auto& v : Views.getValues()) {
431  //NOTE: the order of objects in Page.Views does not reflect the object hierarchy
432  // this means that a ProjGroup could be deleted before its child ProjGroupItems.
433  // this causes problems when removing objects from document
434  if (v->isAttachedToDocument()) {
435  std::string viewName = v->getNameInDocument();
436  Base::Interpreter().runStringArg("App.getDocument(\"%s\").removeObject(\"%s\")",
437  docName.c_str(), viewName.c_str());
438  }
439  }
440  std::vector<App::DocumentObject*> emptyViews;//probably superfluous
441  Views.setValues(emptyViews);
442  }
443  catch (...) {
444  Base::Console().Warning("DP::unsetupObject - %s - error while deleting children\n",
445  getNameInDocument());
446  }
447 
448  App::DocumentObject* tmp = Template.getValue();
449  if (tmp) {
450  std::string templateName = Template.getValue()->getNameInDocument();
451  Base::Interpreter().runStringArg("App.getDocument(\"%s\").removeObject(\"%s\")",
452  docName.c_str(), templateName.c_str());
453  }
454  Template.setValue(nullptr);
455 }
