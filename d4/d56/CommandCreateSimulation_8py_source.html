674  def positionDialog(self):

679  screen_geometry = (
680  screen.availableGeometry()
681  if screen
682  else QApplication.primaryScreen().availableGeometry()
683  )
684 
685  # Calculate the position of the dialog to ensure it stays within the screen
686  dialog_position = self.initialPos
687 
688  # Adjust position to keep the dialog within the screen bounds
689  if dialog_position.x() + self.dialog.width() > screen_geometry.right():
690  dialog_position.setX(screen_geometry.right() - self.dialog.width())
691  if dialog_position.y() + self.dialog.height() > screen_geometry.bottom():
692  dialog_position.setY(screen_geometry.bottom() - self.dialog.height())
693 
694  # Ensure the dialog does not go above or to the left of the screen
695  if dialog_position.x() < screen_geometry.left():
696  dialog_position.setX(screen_geometry.left())
697  if dialog_position.y() < screen_geometry.top():
698  dialog_position.setY(screen_geometry.top())
