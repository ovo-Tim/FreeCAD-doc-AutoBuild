57 def get_init_values(path, count=6):
58  """Set values needed to create the array."""
59  norm = App.Vector(0, 0, 1)
60 
61  # Currently this works with a sketch that has a single edge.
62  # Here we need a more general function to extract all edges from a shape,
63  # so that the array uses all of them.
64  edge = path.Shape.Edges[0]
65  edge_length = edge.Length
66 
67  n = max((count - 1), 1)
68  step = edge_length / n
69  inc = 360 / n
70 
71  return norm, edge, step, inc

74 def get_n_params(edge, number, step, norm):
75  """Get the parameters needed in each iteration."""
76  parameter = edge.getParameterByLength(number * step)
77 
78  v0 = edge.valueAt(parameter)
79  tan = edge.tangentAt(parameter).normalize()
80  binorm = tan.cross(norm).normalize()
81  rot = App.Rotation(binorm, tan, norm)
82 
83  return v0, tan, rot

86 def get_twisted_placements(path, count=15, rot_factor=0.25):
87  """Get the placements of the twisted array elements."""
88  count = max(count, 1)
89 
90  (norm, edge,
91  step, inc) = get_init_values(path, count)
92 
93  increment = 0
94  places = []
95  params = []
96 
97  for number in range(count):
98  v0, tan, rot = get_n_params(edge, number, step, norm)
99 
100  angle = increment * rot_factor
101  place = App.Placement(v0, tan, angle)
102  place.Rotation = place.Rotation * rot
103  places.append(place)
104 
105  params.append((v0, tan, angle, rot))
106 
107  increment += inc
108 
109  return places, params

174 def make_walkway(path, width=100, thickness=10):
175  """Construct the walkway of the twisted bridge array."""
176  spine = path.Shape.Edges[0]
177 
178  half_size = width/2
179  offset_height = thickness
180 
181  norm1 = App.Vector(0, 0, 1)
182  v1 = spine.valueAt(1)
183  tan1 = spine.tangentAt(1).normalize()
184  binorm1 = tan1.cross(norm1)
185 
186  place = App.Placement()
187  place.Rotation = App.Rotation(binorm1, norm1, tan1)
188  place.move(v1 - App.Vector(0, 0, 3 * offset_height))
189 
190  plane = Part.makePlane(width, thickness,
191  App.Vector(-half_size, -2 * offset_height, 0))
192  face = Part.Face(plane)
193  face.Placement = place.multiply(face.Placement)
194 
195  wire = Part.Wire(spine)
196  sweep = wire.makePipe(face)
197 
198  return sweep
