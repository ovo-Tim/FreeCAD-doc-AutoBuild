935  def _calculateOffsetValue(self, isHole, isVoid=False):
936  """_calculateOffsetValue(self.obj, isHole, isVoid) ... internal function.
937  Calculate the offset for the Path.Area() function."""
938  self.JOB = PathUtils.findParentJob(self.obj)
939  # We need to offset by at least our linear tessellation deflection
940  # (default GeometryTolerance / 4) to avoid false retracts at the
941  # boundaries.
942  tolrnc = max(self.JOB.GeometryTolerance.Value / 10.0, self.obj.LinearDeflection.Value)
943 
944  if isVoid is False:
945  if isHole is True:
946  offset = -1 * self.obj.InternalFeaturesAdjustment.Value
947  offset += self.radius + tolrnc
948  else:
949  offset = -1 * self.obj.BoundaryAdjustment.Value
950  if self.obj.BoundaryEnforcement is True:
951  offset += self.radius + tolrnc
952  else:
953  offset -= self.radius + tolrnc
954  offset = 0.0 - offset
955  else:
956  offset = -1 * self.obj.BoundaryAdjustment.Value
957  offset += self.radius + tolrnc
958 
959  return offset

962  """Wrapper around area and wire based region unification
963  implementations."""
964  Path.Log.debug("findUnifiedRegions()")
965  # Allow merging of faces within the LinearDeflection tolerance.
966  tolerance = self.obj.LinearDeflection.Value
967  # Default: normal to Z=1 (XY plane), at Z=0
968  try:
969  # Use Area based implementation

972  shapes,

975  0.0 - tolerance / 10,
976  removeHoles=True, # Outline has holes filled in
977  tolerance=tolerance,
978  plane=self.wpc,
979  )
980  projectionShape = PathUtils.getOffsetArea(
981  shapes,
982  # Make the projection very slightly larger
983  tolerance / 10,
984  removeHoles=False, # Projection has holes preserved
985  tolerance=tolerance,
986  plane=self.wpc,
987  )
