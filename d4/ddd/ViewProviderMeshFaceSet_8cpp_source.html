100 void ViewProviderMeshFaceSet::updateData(const App::Property* prop)
101 {
102  ViewProviderMesh::updateData(prop);
103  if (const auto* meshProp = dynamic_cast<const Mesh::PropertyMeshKernel*>(prop)) {
104  const Mesh::MeshObject* mesh = meshProp->getValuePtr();
105 
106  bool direct = MeshRenderer::shouldRenderDirectly(mesh->countFacets() > this->triangleCount);
107  if (direct) {
108  this->pcMeshNode->mesh.setValue(mesh);
109  // Needs to update internal bounding box caches
110  this->pcMeshShape->touch();
111  pcMeshCoord->point.setNum(0);
112  pcMeshFaces->coordIndex.setNum(0);
113  }
114  else {
115  ViewProviderMeshBuilder builder;
116  builder.createMesh(prop, pcMeshCoord, pcMeshFaces);
117  pcMeshFaces->invalidate();
118  }
119 
120  if (direct != directRendering) {
121  directRendering = direct;
122  Gui::coinRemoveAllChildren(pcShapeGroup);
123 
124  if (directRendering) {
125  pcShapeGroup->addChild(pcMeshNode);
126  pcShapeGroup->addChild(pcMeshShape);
127  }
128  else {
129  pcShapeGroup->addChild(pcMeshCoord);
130  pcShapeGroup->addChild(pcMeshFaces);
131  }
132  }
133 
134  showOpenEdges(OpenEdges.getValue());
135  std::vector<Mesh::FacetIndex> selection;
136  mesh->getFacetsFromSelection(selection);
137  if (selection.empty()) {
138  unhighlightSelection();
139  }
140  else {
141  highlightSelection();
142  }
143  }

146 void ViewProviderMeshFaceSet::showOpenEdges(bool show)
147 {
148  if (pcOpenEdge) {
149  // remove the node and destroy the data
150  pcRoot->removeChild(pcOpenEdge);
151  pcOpenEdge = nullptr;
152  }
153 
154  if (show) {
155  pcOpenEdge = new SoSeparator();
156  pcOpenEdge->addChild(pcLineStyle);
157  pcOpenEdge->addChild(pOpenColor);
158 
159  if (directRendering) {
160  pcOpenEdge->addChild(pcMeshNode);
161  pcOpenEdge->addChild(new SoFCMeshObjectBoundary);
162  }
163  else {
164  pcOpenEdge->addChild(pcMeshCoord);
165  auto lines = new SoIndexedLineSet;
166  pcOpenEdge->addChild(lines);
167 
168  // Build up the lines with indices to the list of vertices 'pcMeshCoord'
169  int index = 0;
170  const Mesh::MeshObject& mesh = getMeshObject();
171  const MeshCore::MeshKernel& kernel = mesh.getKernel();
172  const MeshCore::MeshFacetArray& rFaces = kernel.GetFacets();
173  for (const auto& rFace : rFaces) {
174  for (int i = 0; i < 3; i++) {
175  // NOLINTBEGIN
176  if (rFace._aulNeighbours[i] == MeshCore::FACET_INDEX_MAX) {
177  lines->coordIndex.set1Value(index++, rFace._aulPoints[i]);
178  lines->coordIndex.set1Value(index++, rFace._aulPoints[(i + 1) % 3]);
179  lines->coordIndex.set1Value(index++, SO_END_LINE_INDEX);
180  }
181  // NOLINTEND
182  }
183  }
184  }
185 
186  // add to the highlight node
187  pcRoot->addChild(pcOpenEdge);
188  }
189 }
