96 QPen LineGenerator::getLinePen(size_t lineNumber, double nominalLineWidth)
97 {
98 // Base::Console().Message("LG::getLinePen(%d, %.3f)\n",
99 // lineNumber, nominalLineWidth);
100  QPen linePen;
101  linePen.setWidthF(nominalLineWidth);
102 
103  // Note: if the cap style is Round or Square, the lengths of the lines, or
104  // dots/dashes within the line, will be wrong by 1 pen width. To get the
105  // exact line lengths or dash pattern, you must use Flat caps. Flat caps
106  // look terrible at the corners.
107  linePen.setCapStyle((Qt::PenCapStyle)Preferences::LineCapStyle());
108 
109  double proportionalAdjust{1.0};
110  if (!isCurrentProportional()) {
111  // ANSI.Y14.2M.1992 lines are specified in actual element lengths, but Qt will draw
112  // them as proportional to the line width.
113  proportionalAdjust = nominalLineWidth;
114  }
115 
116  // valid line numbers are [1, number of line definitions]
117  // line 1 is always (?) continuous
118  // 0 substitutes for LineFormat::InvalidLine here
119  if (lineNumber < 2 ||
120  lineNumber > m_lineDefs.size()) {
121  // plain boring solid line (or possibly an invalid line number)
122  linePen.setStyle(Qt::SolidLine);
123  return linePen;
124  }
125 
126  int lineIndex = lineNumber - 1;
127  std::vector<std::string> elements = m_lineDefs.at(lineIndex);
128 
129  // there are some lines with numbers >1 that are also continuous, and
130  // a dash pattern is not applicable.
131  std::string naToken{"n/a"};
132  if (elements.empty() || elements.front() == naToken) {
133  // plain boring solid line (or possibly an invalid line number)
134  linePen.setStyle(Qt::SolidLine);
135  return linePen;
136  }
137 
138  // there is at least one line style (ASME #11 Other) that is "invisible"
139  std::string noLineToken{"noline"};
140  if (elements.front() == noLineToken) {
141  linePen.setStyle(Qt::NoPen);
142  return linePen;
143  }
144 
145  // interesting line styles
146  linePen.setStyle(Qt::CustomDashLine);
147  std::vector<double> dashPattern;
148  bool firstElement(true);
149  for (auto& entry : elements) {
150  if (firstElement &&
151  (entry == "Gap" || entry == "Space") ) {
152  // some dash patterns MAY begin with a gap/space, but Qt dash patterns are always
153  // "mark, space, mark, space", so we handle this by offsetting the pattern
154  // and skipping the first element.
155  linePen.setDashOffset(static_cast< double >(m_elementDefs[entry]) / proportionalAdjust);
156  firstElement = false;
157  continue;
158  }
159  firstElement = false;
160  dashPattern.push_back(static_cast< double >(m_elementDefs[entry]) / proportionalAdjust);
161  }
162 
163 #if QT_VERSION < QT_VERSION_CHECK(5, 14, 0)
164  QVector<double> qDashPattern = QVector<double>::fromStdVector(dashPattern);
165 #else
166  QVector<double> qDashPattern(dashPattern.begin(), dashPattern.end());
167 #endif
168 
169  linePen.setDashPattern(qDashPattern);
170  linePen.setWidthF(nominalLineWidth);
171  return linePen;
172 }

176 int LineGenerator::fromQtStyle(Qt::PenStyle style)
177 {
178 // Base::Console().Message("DLG::fromQtStyle(%d)\n", style);
179 
180  // the 4 standard Qt::PenStyles and ISO128 equivalents
181  int dashed = 2;
182  int dotted = 7;
183  int dashDot = 10;
184  int dashDotDot = 12;
185  if (Preferences::lineStandard() == ANSI) {
186  dashed = 2;
187  dotted = 2; // no dotted line in Ansi Y14.2?
188  dashDot = 2;
189  dashDotDot = 2;
190  }
191  if (Preferences::lineStandard() == ASME) {
192  dashed = 2;
193  dotted = 16;
194  dashDot = 17;
195  dashDotDot = 14;
196  }
197 
198  switch (style) {
199  case Qt::NoPen:
200  case Qt::SolidLine:
201  return 1;
202  case Qt::DashLine:
203  return dashed;
204  case Qt::DotLine:
205  return dotted;
206  case Qt::DashDotLine:
207  return dashDot;
208  case Qt::DashDotDotLine:
209  return dashDotDot;
210  case Qt::CustomDashLine:
211  // not sure what to do here. we would have to match the custom pattern
212  // to the patterns of the ISO lines and set the dash pattern accordingly.
213  return 2;
214  default:
215  return 0;
216  }
217 }

222 std::map<std::string, int> LineGenerator::loadElements()
223 {
224 // Base::Console().Message("DLG::loadElements()\n");
225  std::map<std::string, int> result;
226  // open file, for each record, parse element name and length, then insert into
227  // the output map.
228 
229  std::string parmFile = Preferences::currentElementDefFile();
230  std::string record;
231  Base::FileInfo fi(parmFile);
232  Base::ifstream inFile(fi, std::ifstream::in);
233  if(!inFile.is_open()) {
234  Base::Console().Message( "Cannot open line element def file: %s\n", parmFile.c_str());
235  return result;
236  }
237  std::string line;
238  while ( std::getline(inFile, line) ){
239  if ( line.empty() || line.at(0) == '#' ) {
240  // this is a comment or a blank line, ignore it
241  continue;
242  }
243  std::vector<std::string> tokens = DU::tokenize(line, ",");
244  // should be 2 tokens: elementName, elementLength
245  result[tokens.front()] = std::stoi(tokens.back(), nullptr);
246  }
247  inFile.close();
248  return result;
249 }

253 std::vector< std::vector<std::string> > LineGenerator::getLineDefinitions()
254 {
255 // Base::Console().Message("DLG::loadLineDefinitions()\n");
256  std::vector< std::vector<std::string> > lineDefs;
257  std::string record;
258  Base::FileInfo fi(Preferences::currentLineDefFile());
259  Base::ifstream inFile(fi, std::ifstream::in);
260  if(!inFile.is_open()) {
261  Base::Console().Message( "Cannot open line def file: %s\n", fi.filePath().c_str());
262  return lineDefs;
263  }
264 
265  std::string line;
266  while ( std::getline(inFile, line) ) {
267  if (line.empty() ||
268  line.at(0) == '#') {
269  // this is a comment or a blank line, ignore it
270  continue;
271  }
272  // strip out any null tokens that may be caused by trailing ',' in the input
273  std::vector<std::string> validTokens;
274  for (auto& token : DU::tokenize(line, ",")) {
275  if (!token.empty()) {
276  validTokens.emplace_back(token);
277  }
278  }
279  std::vector<std::string> lineDefRow;
280  lineDefRow.insert(lineDefRow.end(), validTokens.begin()+2, validTokens.end());
281  lineDefs.push_back(lineDefRow);
282  }
283 
284  inFile.close();
285  return lineDefs;
286 }

289 std::vector<std::string> LineGenerator::getAvailableLineStandards()
290 {
291  std::vector<std::string> result;
292  std::string lineDefToken{"LineDef"};
293  Base::FileInfo fi(Preferences::lineDefinitionLocation());
294  auto fiAll = fi.getDirectoryContent();
295  for (auto& entry : fiAll) {
296  if (!entry.isFile()) {
297  continue;
298  }
299  auto fileName = entry.fileNamePure();
300  size_t position = fileName.find(lineDefToken);
301  if (position != std::string::npos) {
302  result.emplace_back(fileName.substr(0, position - 1));
303  }
304  }
305  std::sort(result.begin(),result.end());
306  return result;
307 }

353 std::string LineGenerator::getLineStandardsBody()
354 {
355  int activeStandard = Preferences::lineStandard();
356  std::vector<std::string> choices = getAvailableLineStandards();
357  if (activeStandard < 0 ||
358  (size_t) activeStandard >= choices.size()) {
359  // there is a condition where the LineStandard parameter exists, but is -1 (the
360  // qt value for no current index in a combobox). This is likely caused by an old
361  // development version writing an unvalidated value. In this case, the existing but
362  // invalid value will be returned. This is a temporary fix and can be removed for
363  // production.
364  // Preferences::lineStandard() will print a message about this every time it is called
365  // (lots of messages!).
366  activeStandard = 0;
367  }
368  return getBodyFromString(choices.at(activeStandard));
369 }

385 {
386  std::vector<std::string> choices = getAvailableLineStandards();
387  if (standardIndex > choices.size()) {
388  // we don't have a standard for the specified index.
389  return true;
390  }
391  std::string bodyName = getBodyFromString(choices.at(standardIndex));
392  if (bodyName == "ANSI") {
