50 TopoDS_Shape SketchExportHelper::projectShape(const TopoDS_Shape& inShape,
51  const gp_Ax2& projectionCS)
52 {
53  Handle(HLRBRep_Algo) brep_hlr = new HLRBRep_Algo();
54  brep_hlr->Add(inShape);
55  HLRAlgo_Projector projector(projectionCS);
56  brep_hlr->Projector(projector);
57  brep_hlr->Update();
58  brep_hlr->Hide();
59  HLRBRep_HLRToShape hlrToShape(brep_hlr);
60  BRep_Builder builder;
61  TopoDS_Compound comp;
62  builder.MakeCompound(comp);
63  if (!hlrToShape.VCompound().IsNull()) {
64  builder.Add(comp, hlrToShape.VCompound());
65  }
66  if (!hlrToShape.OutLineVCompound().IsNull()) {
67  builder.Add(comp, hlrToShape.OutLineVCompound());
68  }
69  return comp;
70 }

89 TopoDS_Shape SketchExportHelper::getFlatSketchXY(App::DocumentObject* obj)
90 {
91  // since we can't reference Sketcher module here, we will cast obj to
92  // a Part::Feature instead
93  auto sketch = dynamic_cast<Part::Feature*>(obj);
94  if (!sketch || !isSketch(obj)) {
95  return {};
96  }
97 
98  auto plm = sketch->Placement.getValue();
99  Base::Rotation rot = plm.getRotation();
100 
101  // get the sketch normal
102  Base::Vector3d stdZ {0.0, 0.0, 1.0};
103  Base::Vector3d sketchNormal;
104  rot.multVec(stdZ, sketchNormal);
105  Base::Vector3d stdX {1.0, 0.0, 0.0};
106  Base::Vector3d sketchX;
107  rot.multVec(stdX, sketchX);
108 
109  // get the sketch origin
110  Base::Vector3d position = plm.getPosition();
111  gp_Ax2 projectionCS(gp_Pnt(position.x, position.y, position.z),
112  gp_Dir(sketchNormal.x, sketchNormal.y, sketchNormal.z),
113  gp_Dir(sketchX.x, sketchX.y, sketchX.z));
114  const TopoDS_Shape& shape = sketch->Shape.getValue();
115  return projectShape(shape, projectionCS);
116 }
