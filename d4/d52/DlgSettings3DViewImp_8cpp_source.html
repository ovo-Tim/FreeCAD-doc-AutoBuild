101 void DlgSettings3DViewImp::addAntiAliasing()
102 {
103  ui->comboAliasing->clear();
104 
105  // Do the samples checks only once
106  static std::vector<std::pair<QString, AntiAliasing>> modes;
107  static bool formatCheck = true;
108  if (formatCheck) {
109  formatCheck = false;
110 
111  Multisample check;
112  modes = check.supported();
113  }
114 
115  for (const auto& it : modes) {
116  ui->comboAliasing->addItem(it.first, int(it.second));
117  }
118 }
