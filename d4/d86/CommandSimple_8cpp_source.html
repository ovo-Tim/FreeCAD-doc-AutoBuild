141 void CmdPartPointsFromMesh::activated(int iMsg)
142 {
143  Q_UNUSED(iMsg);
144 
145  auto getDefaultDistance = [](Part::Feature* geometry) {
146  auto bbox = geometry->Shape.getBoundingBox();
147  int steps{20};
148  return bbox.CalcDiagonalLength() / steps;
149  };
150 
151  Base::Type geoid = Base::Type::fromName("App::GeoFeature");
152  std::vector<App::DocumentObject*> geoms;
153  geoms = Gui::Selection().getObjectsOfType(geoid);
154 
155  double distance{1.0};
156  auto found = std::find_if(geoms.begin(), geoms.end(), [](App::DocumentObject* obj) {
157  return Base::freecad_dynamic_cast<Part::Feature>(obj);
158  });
159 
160  if (found != geoms.end()) {
161 
162  double defaultDistance = getDefaultDistance(Base::freecad_dynamic_cast<Part::Feature>(*found));
163 
164  double STD_OCC_TOLERANCE = 1e-6;
165 
166  int decimals = Base::UnitsApi::getDecimals();
167  double tolerance_from_decimals = pow(10., -decimals);
168 
169  double minimal_tolerance = tolerance_from_decimals < STD_OCC_TOLERANCE ? STD_OCC_TOLERANCE : tolerance_from_decimals;
170 
171  bool ok;
172  distance = QInputDialog::getDouble(Gui::getMainWindow(), QObject::tr("Distance in parameter space"),
173  QObject::tr("Enter distance:"), defaultDistance, minimal_tolerance, 10.0 * defaultDistance, decimals, &ok, Qt::MSWindowsFixedSizeDialogHint);
174  if (!ok) {
175  return;
176  }
177  }
178 
179  Gui::WaitCursor wc;
180  openCommand(QT_TRANSLATE_NOOP("Command", "Points from geometry"));
181 
182  Base::PyGILStateLocker lock;
183  try {
184  PyObject* module = PyImport_ImportModule("BasicShapes.Utils");
185  if (!module) {
186  throw Py::Exception();
187  }
188  Py::Module utils(module, true);
189 
190  for (auto it : geoms) {
191  Py::Tuple args(2);
192  args.setItem(0, Py::asObject(it->getPyObject()));
193  args.setItem(1, Py::Float(distance));
194  utils.callMemberFunction("showCompoundFromPoints", args);
195  }
196  }
197  catch (Py::Exception&) {
198  Base::PyException e;
199  e.ReportException();
200  }
201 
202  commitCommand();
203 }

227 static void _copyShape(const char *cmdName, bool resolve,bool needElement=false, bool refine=false) {
228  Gui::WaitCursor wc;
229  Gui::Command::openCommand(cmdName);
230  for(auto &sel : Gui::Selection().getSelectionEx("*", App::DocumentObject::getClassTypeId(),
231  resolve ? Gui::ResolveMode::OldStyleElement : Gui::ResolveMode::NoResolve)) {
232  std::map<std::string,App::DocumentObject*> subMap;
233  auto obj = sel.getObject();
234  if (!obj)
235  continue;
236  if (resolve || !sel.hasSubNames()) {
237  subMap.emplace("",obj);
238  }
239  else {
240  for(const auto &sub : sel.getSubNames()) {
241  const char *element = nullptr;
242  auto sobj = obj->resolve(sub.c_str(),nullptr,nullptr,&element);
243  if(!sobj) continue;
244  if(!needElement && element)
245  subMap.emplace(sub.substr(0,element-sub.c_str()),sobj);
246  else
247  subMap.emplace(sub,sobj);
248  }
249  if(subMap.empty())
250  continue;
251  }
252  auto parentName = Gui::Command::getObjectCmd(obj);
253  for(auto &v : subMap) {
254  Gui::Command::doCommand(Gui::Command::Doc,
255  "__shape = Part.getShape(%s,'%s',needSubElement=%s,refine=%s)%s\n"
256  "App.ActiveDocument.addObject('Part::Feature','%s').Shape=__shape\n"
257  "App.ActiveDocument.ActiveObject.Label=%s.Label\n",
258  parentName.c_str(), v.first.c_str(),
259  needElement ? "True" : "False",
260  refine ? "True" : "False",
261  needElement ? ".copy()" : "",
262  v.second->getNameInDocument(),
263  Gui::Command::getObjectCmd(v.second).c_str());
264  auto newObj = App::GetApplication().getActiveDocument()->getActiveObject();
265  Gui::Command::copyVisual(newObj, "ShapeAppearance", v.second);
266  Gui::Command::copyVisual(newObj, "LineColor", v.second);
267  Gui::Command::copyVisual(newObj, "PointColor", v.second);
268  }
269  }
270  Gui::Command::commitCommand();
271  Gui::Command::updateActive();
272 }
