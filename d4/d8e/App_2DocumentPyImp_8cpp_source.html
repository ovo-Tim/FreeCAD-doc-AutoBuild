313 PyObject* DocumentPy::addObject(PyObject* args, PyObject* kwd)
314 {
315  char *sType, *sName = nullptr, *sViewType = nullptr;
316  PyObject* obj = nullptr;
317  PyObject* view = nullptr;
318  PyObject* attach = Py_False;
319  static const std::array<const char*, 7>
320  kwlist {"type", "name", "objProxy", "viewProxy", "attach", "viewType", nullptr};
321  if (!Base::Wrapped_ParseTupleAndKeywords(args,
322  kwd,
323  "s|sOOO!s",
324  kwlist,
325  &sType,
326  &sName,
327  &obj,
328  &view,
329  &PyBool_Type,
330  &attach,
331  &sViewType)) {
332  return nullptr;
333  }
334 
335  DocumentObject* pcFtr = nullptr;
336 
337  if (!obj || !Base::asBoolean(attach)) {
338  pcFtr = getDocumentPtr()->addObject(sType, sName, true, sViewType);
339  }
340  else {
341  Base::Type type =
342  Base::Type::getTypeIfDerivedFrom(sType, DocumentObject::getClassTypeId(), true);
343  if (type.isBad()) {
344  std::stringstream str;
345  str << "'" << sType << "' is not a document object type";
346  throw Base::TypeError(str.str());
347  }
348  pcFtr = static_cast<DocumentObject*>(type.createInstance());
349  }
350  // the type instance could be a null pointer
351  if (!pcFtr) {
352  std::stringstream str;
353  str << "No document object found of type '" << sType << "'" << std::ends;
354  throw Py::TypeError(str.str());
355  }
356  // Allows one to hide the handling with Proxy in client python code
357  if (obj) {
358  try {
359  // the python binding class to the document object
360  Py::Object pyftr = Py::asObject(pcFtr->getPyObject());
361  // 'pyobj' is the python class with the implementation for DocumentObject
362  Py::Object pyobj(obj);
363  if (pyobj.hasAttr("__object__")) {
364  pyobj.setAttr("__object__", pyftr);
365  }
366  pyftr.setAttr("Proxy", pyobj);
367 
368  if (Base::asBoolean(attach)) {
369  getDocumentPtr()->addObject(pcFtr, sName);
370 
371  try {
372  Py::Callable method(pyobj.getAttr("attach"));
373  if (!method.isNone()) {
374  Py::TupleN arg(pyftr);
375  method.apply(arg);
376  }
377  }
378  catch (Py::Exception&) {
379  Base::PyException e;
380  e.ReportException();
381  }
382  }
383 
384  // if a document class is set we also need a view provider defined which must be
385  // something different to None
386  Py::Object pyvp;
387  if (view) {
388  pyvp = Py::Object(view);
389  }
390  if (pyvp.isNone()) {
391  pyvp = Py::Int(1);
392  }
393  // 'pyvp' is the python class with the implementation for ViewProvider
394  if (pyvp.hasAttr("__vobject__")) {
395  pyvp.setAttr("__vobject__", pyftr.getAttr("ViewObject"));
396  }
397 
398  Py::Object pyprx(pyftr.getAttr("ViewObject"));
399  pyprx.setAttr("Proxy", pyvp);
400  return Py::new_reference_to(pyftr);
401  }
402  catch (Py::Exception& e) {
403  e.clear();
404  }
405  }
406  return pcFtr->getPyObject();
407 }
