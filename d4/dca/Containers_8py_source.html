45  def getStaticChildren(self):

66  def getDynamicChildren(self):
67  """Returns dynamic children, i.e. the stuff that can be removed from the container."""
68  self.self_check()
69  container = self.Object
70 
71  if container.isDerivedFrom("App::Document"):
72  # find all objects not contained by any Part or Body
73  result = set(container.Objects)
74  for obj in container.Objects:
75  if isAContainer(obj):
76  children = set(Container(obj).getAllChildren())
77  result = result - children
78  return list(result)
79  elif container.hasExtension("App::GroupExtension"):
80  result = container.Group
81  if container.hasExtension("App::GeoFeatureGroupExtension"):
82  # geofeaturegroup's group contains all objects within the CS, we don't want that
83  result = [obj for obj in result if obj.getParentGroup() is not container]
84  return result
85  elif container.isDerivedFrom("App::Origin"):
86  return []
87  elif container.hasChildElement():
88  result = []

157 def _getMetacontainerChildren(container, isrightcontainer_func):
158  """Gathers up children of metacontainer - a container structure formed by containers of specific type.
159  For example, coordinate systems form a kind of container structure.

161  container: instance of Container class

164  result = []

184 def isAContainer(obj, links_too=False):
185  """isAContainer(obj, links_too): returns True if obj is an object container, such as
186  Group, Part, Body. The important characteristic of an object being a
187  container is that it can be activated to receive new objects. Documents
188  are considered containers, too.
189  If links_too, App::Link objects are considered containers, too. Then, container tree
190  isn't necessarily a tree."""
191 
192  if obj.isDerivedFrom("App::Document"):
193  return True
194  if obj.hasExtension("App::GroupExtension"):
195  return True
196  if obj.isDerivedFrom("App::Origin"):
197  return True
198  if obj.hasChildElement():
199  return True if links_too else False
200  return False

224 def ContainerChain(feat):
225  """ContainerChain(feat): container path to feat (not including feat itself).
226  Last container directly contains the feature.
227  Example of output: [<document>,<SuperPart>,<Part>,<Body>]"""
228 
229  if feat.isDerivedFrom("App::Document"):
230  return []
231 
232  list_traversing_now = [feat]
233  set_of_deps = set()
234  list_of_deps = []
235 
236  while len(list_traversing_now) > 0:
237  list_to_be_traversed_next = []
238  for feat in list_traversing_now:
239  for dep in feat.InList:
240  if isAContainer(dep) and Container(dep).hasObject(feat):
241  if not (dep in set_of_deps):
242  set_of_deps.add(dep)
243  list_of_deps.append(dep)
244  list_to_be_traversed_next.append(dep)
245  if len(list_to_be_traversed_next) > 1:
246  raise ContainerTreeError("Container tree is not a tree")
247  list_traversing_now = list_to_be_traversed_next
248 
249  return [feat.Document] + list_of_deps[::-1]
