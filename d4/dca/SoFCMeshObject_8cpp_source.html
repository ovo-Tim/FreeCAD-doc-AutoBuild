107 class SoInputStreambuf: public std::streambuf
108 {
109 public:
110  explicit SoInputStreambuf(SoInput* o)
111  : inp(o)
112  {
113  setg(buffer + pbSize, buffer + pbSize, buffer + pbSize);
114  }
115 
116 protected:
117  int underflow() override
118  {
119  if (gptr() < egptr()) {
120  return *gptr();
121  }
122 
123  int numPutback {};
124  numPutback = gptr() - eback();
125  if (numPutback > pbSize) {
126  numPutback = pbSize;
127  }
128 
129  memcpy(buffer + (pbSize - numPutback), gptr() - numPutback, numPutback);
130 
131  int num = 0;
132  for (int i = 0; i < bufSize; i++) {
133  char c {};
134  SbBool ok = inp->get(c);
135  if (ok) {
136  num++;
137  buffer[pbSize + i] = c;
138  if (c == '\n') {
139  break;
140  }
141  }
142  else if (num == 0) {
143  return EOF;
144  }
145  }
146 
147  setg(buffer + (pbSize - numPutback), buffer + pbSize, buffer + pbSize + num);
148 
149  return *gptr();
150  }
151 
152 private:
153  static const int pbSize = 4;
154  static const int bufSize = 1024;
155  char buffer[bufSize + pbSize] {};
156  SoInput* inp;
157 };

193 // successfully.
194 SbBool SoSFMeshObject::readValue(SoInput* in)
195 {
196  if (!in->isBinary()) {
197  SoInputStream str(in);
198  MeshCore::MeshKernel kernel;
199  MeshCore::MeshInput(kernel).LoadMeshNode(str);
200  value = new Mesh::MeshObject(kernel);
201 
202  // We need to trigger the notification chain here, as this function
203  // can be used on a node in a scene graph in any state -- not only
204  // during initial scene graph import.
205  this->valueChanged();
206 
207  return true;
208  }
209 
210  int32_t countPt {};
211  in->read(countPt);
212  std::vector<float> verts(countPt);
213  in->readBinaryArray(verts.data(), countPt);
214 
215  MeshCore::MeshPointArray rPoints;
216  rPoints.reserve(countPt / 3);
217  for (auto it = verts.begin(); it != verts.end();) {
218  Base::Vector3f p;
219  p.x = *it;
220  ++it;
221  p.y = *it;
222  ++it;
223  p.z = *it;
224  ++it;
225  rPoints.push_back(p);
226  }
227 
228  int32_t countFt {};
229  in->read(countFt);
230  std::vector<int32_t> faces(countFt);
231  in->readBinaryArray(faces.data(), countFt);
232 
233  MeshCore::MeshFacetArray rFacets;
234  rFacets.reserve(countFt / 3);
235  for (auto it = faces.begin(); it != faces.end();) {
236  MeshCore::MeshFacet f;
237  f._aulPoints[0] = *it;
238  ++it;
239  f._aulPoints[1] = *it;
240  ++it;
241  f._aulPoints[2] = *it;
242  ++it;
243  rFacets.push_back(f);
244  }
245 
246  MeshCore::MeshKernel kernel;
247  kernel.Adopt(rPoints, rFacets, true);
248  value = new Mesh::MeshObject(kernel);
249 
250  // We need to trigger the notification chain here, as this function
251  // can be used on a node in a scene graph in any state -- not only
252  // during initial scene graph import.
253  this->valueChanged();
254 
255  return true;
256 }

446  const SbVec3s& len = lenGrid.getValue();

449  float minX {}, minY {}, minZ {};
450  min.getValue(minX, minY, minZ);
451  float maxX {}, maxY {}, maxZ {};
452  max.getValue(maxX, maxY, maxZ);
453  float dx = (maxX - minX) / (float)u;

629 void SoFCMeshObjectShape::GLRender(SoGLRenderAction* action)
630 {
631  if (shouldGLRender(action)) {
632  SoState* state = action->getState();
633 
634  // Here we must save the model and projection matrices because
635  // we need them later for picking
636  glGetFloatv(GL_MODELVIEW_MATRIX, this->modelview);
637  glGetFloatv(GL_PROJECTION_MATRIX, this->projection);
638 
639  SbBool mode = Gui::SoFCInteractiveElement::get(state);
640  const Mesh::MeshObject* mesh = SoFCMeshObjectElement::get(state);
641  if (!mesh || mesh->countPoints() == 0) {
642  return;
643  }
644 
645  Binding mbind = this->findMaterialBinding(state);
646 
647  SoMaterialBundle mb(action);
648  // SoTextureCoordinateBundle tb(action, true, false);
649 
650  SbBool needNormals = !mb.isColorOnly() /* || tb.isFunction()*/;
651  mb.sendFirst(); // make sure we have the correct material
652 
653  SbBool ccw = true;
654  if (SoShapeHintsElement::getVertexOrdering(state) == SoShapeHintsElement::CLOCKWISE) {
655  ccw = false;
656  }
657 
658  if (!mode || mesh->countFacets() <= this->renderTriangleLimit) {
659  if (mbind != OVERALL) {
660  drawFaces(mesh, &mb, mbind, needNormals, ccw);
661  }
662  else {
663 #ifdef RENDER_GLARRAYS
664  if (updateGLArray) {
665  updateGLArray = false;
666  generateGLArrays(state);
667  }
668  renderFacesGLArray(action);
669 #else
670  drawFaces(mesh, 0, mbind, needNormals, ccw);
671 #endif
672  }
673  }
674  else {
675 #if 0 && defined(RENDER_GLARRAYS)
676  renderCoordsGLArray(action);
677 #else
678  drawPoints(mesh, needNormals, ccw);

722 void SoFCMeshObjectShape::drawFaces(const Mesh::MeshObject* mesh,
723  SoMaterialBundle* mb,
724  Binding bind,
725  SbBool needNormals,
726  SbBool ccw) const
727 {
728  const MeshCore::MeshPointArray& rPoints = mesh->getKernel().GetPoints();
729  const MeshCore::MeshFacetArray& rFacets = mesh->getKernel().GetFacets();
730  bool perVertex = (mb && bind == PER_VERTEX_INDEXED);
731  bool perFace = (mb && bind == PER_FACE_INDEXED);
732 
733  if (needNormals) {
734  glBegin(GL_TRIANGLES);
735  if (ccw) {
736  // counterclockwise ordering
737  for (auto it = rFacets.begin(); it != rFacets.end(); ++it) {
738  const MeshCore::MeshPoint& v0 = rPoints[it->_aulPoints[0]];
