82 bool ViewProvider::setEdit(int ModNum)
83 {
84  if (ModNum == ViewProvider::Transform) {
85  if (forwardToLink()) {
86  return true;
87  }
88 
89  // this is feature so we need to forward the transform to the body
90  forwardedViewProvider = getBodyViewProvider();
91  return forwardedViewProvider->startEditing(ModNum);
92  }
93  else if (ModNum == ViewProvider::Default) {
94  // When double-clicking on the item for this feature the
95  // object unsets and sets its edit mode without closing
96  // the task panel
97  Gui::TaskView::TaskDialog *dlg = Gui::Control().activeDialog();
98  TaskDlgFeatureParameters *featureDlg = qobject_cast<TaskDlgFeatureParameters *>(dlg);
99  // NOTE: if the dialog is not partDesigan dialog the featureDlg will be NULL
100  if (featureDlg && featureDlg->getViewObject() != this) {
101  featureDlg = nullptr; // another feature left open its task panel
102  }
103  if (dlg && !featureDlg) {
104  QMessageBox msgBox;
105  msgBox.setText(QObject::tr("A dialog is already open in the task panel"));
106  msgBox.setInformativeText(QObject::tr("Do you want to close this dialog?"));
107  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
108  msgBox.setDefaultButton(QMessageBox::Yes);
109  int ret = msgBox.exec();
110  if (ret == QMessageBox::Yes) {
111  Gui::Control().reject();
112  } else {
113  return false;
114  }
115  }
116 
117  // clear the selection (convenience)
118  Gui::Selection().clearSelection();
119 
120  // always change to PartDesign WB, remember where we come from
121  oldWb = Gui::Command::assureWorkbench("PartDesignWorkbench");
122 
123  // start the edit dialog if
124  if (!featureDlg) {
125  featureDlg = this->getEditDialog();
126  if (!featureDlg) { // Shouldn't generally happen
127  throw Base::RuntimeError ("Failed to create new edit dialog.");
128  }
129  }
130 
131  Gui::Control().showDialog(featureDlg);
132  return true;
133  } else {
134  return PartGui::ViewProviderPart::setEdit(ModNum);
135  }
136 }

182 void ViewProvider::onChanged(const App::Property* prop) {
183 
184  //if the object is inside of a body we make sure it is the only visible one on activation
185  if(prop == &Visibility && Visibility.getValue()) {
186 
187  Part::BodyBase* body = Part::BodyBase::findBodyOf(getObject());
188  if(body) {
189 
190  //hide all features in the body other than this object
191  for(App::DocumentObject* obj : body->Group.getValues()) {
192 
193  if(obj->isDerivedFrom(PartDesign::Feature::getClassTypeId()) && obj != getObject()) {
194  auto vpd = Base::freecad_dynamic_cast<Gui::ViewProviderDocumentObject>(
195  Gui::Application::Instance->getViewProvider(obj));
196  if(vpd && vpd->Visibility.getValue())
197  vpd->Visibility.setValue(false);
198  }
199  }
200  }
201  }
202 
203  PartGui::ViewProviderPartExt::onChanged(prop);
204 }

240 bool ViewProvider::onDelete(const std::vector<std::string> &)
241 {
242  PartDesign::Feature* feature = getObject<PartDesign::Feature>();
243 
244  App::DocumentObject* previousfeat = feature->BaseFeature.getValue();
245 
246  // Visibility - we want:
247  // 1. If the visible object is not the one being deleted, we leave that one visible.
248  // 2. If the visible object is the one being deleted, we make the previous object visible.
249  if (isShow() && previousfeat && Gui::Application::Instance->getViewProvider(previousfeat)) {
250  Gui::Application::Instance->getViewProvider(previousfeat)->show();
251  }
252 
253  // find surrounding features in the tree
254  Part::BodyBase* body = PartDesign::Body::findBodyOf(getObject());
255 
256  if (body) {
257  // Deletion from the tree of a feature is handled by Document.removeObject, which has no clue
258  // about what a body is. Therefore, Bodies, although an "activable" container, know nothing
259  // about what happens at Document level with the features they contain.
260  //
261  // The Deletion command StdCmdDelete::activated, however does notify the viewprovider corresponding
262  // to the feature (not body) of the imminent deletion (before actually doing it).
263  //
264  // Consequently, the only way of notifying a body of the imminent deletion of one of its features
265  // so as to do the clean up required (moving basefeature references, tip management) is from the
266  // viewprovider, so we call it here.
267  //
268  // fixes (#3084)
269 
270  FCMD_OBJ_CMD(body,"removeObject(" << Gui::Command::getObjectCmd(feature) << ')');
271  }
272 
273  return true;
274 }

276 void ViewProvider::setBodyMode(bool bodymode) {
277 
278  std::vector<App::Property*> props;
279  getPropertyList(props);
280 
281  auto vp = getBodyViewProvider();
282  if(!vp)
283  return;
284 
285  for(App::Property* prop : props) {
286 
287  //we keep visibility and selectibility per object
288  if(prop == &Visibility ||
289  prop == &Selectable)
290  continue;
291 
292  //we hide only properties which are available in the body, not special ones
293  if(!vp->getPropertyByName(prop->getName()))
294  continue;
295 
296  prop->setStatus(App::Property::Hidden, bodymode);
297  }
298 }

321 ViewProviderBody* ViewProvider::getBodyViewProvider() {
322
