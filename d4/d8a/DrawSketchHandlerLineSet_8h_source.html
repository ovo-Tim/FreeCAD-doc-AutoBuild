99  void registerPressedKey(bool pressed, int key) override
100  {
101  if (Mode == STATUS_SEEK_Second && key == SoKeyboardEvent::M && pressed
102  && previousCurve != -1) {
103  // loop through the following modes:
104  // SEGMENT_MODE_Line, TRANSITION_MODE_Free / TRANSITION_MODE_Tangent
105  // SEGMENT_MODE_Line, TRANSITION_MODE_Perpendicular_L
106  // SEGMENT_MODE_Line, TRANSITION_MODE_Tangent / TRANSITION_MODE_Free
107  // SEGMENT_MODE_Arc, TRANSITION_MODE_Tangent
108  // SEGMENT_MODE_Arc, TRANSITION_MODE_Perpendicular_L
109  // SEGMENT_MODE_Arc, TRANSITION_MODE_Perpendicular_R
110 
111  SnapMode = SNAP_MODE_Free;
112 
113  Base::Vector2d onSketchPos;
114  if (SegmentMode == SEGMENT_MODE_Line) {
115  onSketchPos = EditCurve[EditCurve.size() - 1];
116  }
117  else {
118  onSketchPos = EditCurve[29];
119  }
120 
121  const Part::Geometry* geom = sketchgui->getSketchObject()->getGeometry(previousCurve);
122 
123  if (SegmentMode == SEGMENT_MODE_Line) {
124  switch (TransitionMode) {
125  case TRANSITION_MODE_Free:
126  if (geom->is<Part::GeomArcOfCircle>()) { // 3rd mode
127  SegmentMode = SEGMENT_MODE_Arc;
128  TransitionMode = TRANSITION_MODE_Tangent;
129  }
130  else { // 1st mode
131  TransitionMode = TRANSITION_MODE_Perpendicular_L;
132  }
133  break;
134  case TRANSITION_MODE_Perpendicular_L: // 2nd mode
135  if (geom->is<Part::GeomArcOfCircle>()) {
136  TransitionMode = TRANSITION_MODE_Free;
137  }
138  else {
139  TransitionMode = TRANSITION_MODE_Tangent;
140  }
141  break;
142  case TRANSITION_MODE_Tangent:
143  if (geom->is<Part::GeomArcOfCircle>()) { // 1st mode
144  TransitionMode = TRANSITION_MODE_Perpendicular_L;
145  }
146  else { // 3rd mode
147  SegmentMode = SEGMENT_MODE_Arc;
148  TransitionMode = TRANSITION_MODE_Tangent;
149  }
150  break;
151  default: // unexpected mode
152  TransitionMode = TRANSITION_MODE_Free;
153  break;
154  }
155  }
156  else {
157  switch (TransitionMode) {
158  case TRANSITION_MODE_Tangent: // 4th mode
159  TransitionMode = TRANSITION_MODE_Perpendicular_L;
160  break;
161  case TRANSITION_MODE_Perpendicular_L: // 5th mode
162  TransitionMode = TRANSITION_MODE_Perpendicular_R;
163  break;
164  default: // 6th mode (Perpendicular_R) + unexpected mode
165  SegmentMode = SEGMENT_MODE_Line;
166  if (geom->is<Part::GeomArcOfCircle>()) {
167  TransitionMode = TRANSITION_MODE_Tangent;
168  }
169  else {
170  TransitionMode = TRANSITION_MODE_Free;
171  }
172  break;
173  }
174  }
175 
176  if (SegmentMode == SEGMENT_MODE_Line) {
177  EditCurve.resize(TransitionMode == TRANSITION_MODE_Free ? 2 : 3);
178  }
179  else {
180  EditCurve.resize(32);
181  }
182  mouseMove(onSketchPos); // trigger an update of EditCurve
183  }
184  else {
185  DrawSketchHandler::registerPressedKey(pressed, key);
186  }
187  }

189  void mouseMove(Base::Vector2d onSketchPos) override
190  {
191  suppressTransition = false;
192  if (Mode == STATUS_SEEK_First) {
193  setPositionText(onSketchPos);
194  seekAndRenderAutoConstraint(sugConstr1, onSketchPos, Base::Vector2d(0.f, 0.f));
195  }
196  else if (Mode == STATUS_SEEK_Second) {
197  if (SegmentMode == SEGMENT_MODE_Line) {
198  EditCurve[EditCurve.size() - 1] = onSketchPos;
199  if (TransitionMode == TRANSITION_MODE_Tangent) {
200  Base::Vector2d Tangent(dirVec.x, dirVec.y);
201  EditCurve[1].ProjectToLine(EditCurve[2] - EditCurve[0], Tangent);
202  if (EditCurve[1] * Tangent < 0) {
203  EditCurve[1] = EditCurve[2];
204  suppressTransition = true;
205  }
206  else {
207  EditCurve[1] = EditCurve[0] + EditCurve[1];
208  }
209  }
210  else if (TransitionMode == TRANSITION_MODE_Perpendicular_L
211  || TransitionMode == TRANSITION_MODE_Perpendicular_R) {
212  Base::Vector2d Perpendicular(-dirVec.y, dirVec.x);
213  EditCurve[1].ProjectToLine(EditCurve[2] - EditCurve[0], Perpendicular);
214  EditCurve[1] = EditCurve[0] + EditCurve[1];
215  }
216 
217  drawEdit(EditCurve);
218 
219  float length = (EditCurve[1] - EditCurve[0]).Length();
220  float angle = (EditCurve[1] - EditCurve[0]).GetAngle(Base::Vector2d(1.f, 0.f));
221 
222  if (showCursorCoords()) {
223  SbString text;
224  std::string lengthString = lengthToDisplayFormat(length, 1);
225  std::string angleString = angleToDisplayFormat(angle * 180.0 / M_PI, 1);
226  text.sprintf(" (%s, %s)", lengthString.c_str(), angleString.c_str());
227  setPositionText(EditCurve[1], text);
228  }
229 
230  if (TransitionMode == TRANSITION_MODE_Free) {
231  seekAndRenderAutoConstraint(sugConstr2,
232  onSketchPos,
233  onSketchPos - EditCurve[0]);
234  }
235  }
236  else if (SegmentMode == SEGMENT_MODE_Arc) {
237 
238  if (QApplication::keyboardModifiers() == Qt::ControlModifier) {
239  SnapMode = SNAP_MODE_45Degree;
240  }
241  else {
242  SnapMode = SNAP_MODE_Free;
243  }
244 
245  Base::Vector2d Tangent;
246  if (TransitionMode == TRANSITION_MODE_Tangent) {
247  Tangent = Base::Vector2d(dirVec.x, dirVec.y);
248  }
249  else if (TransitionMode == TRANSITION_MODE_Perpendicular_L) {
250  Tangent = Base::Vector2d(-dirVec.y, dirVec.x);
251  }
252  else if (TransitionMode == TRANSITION_MODE_Perpendicular_R) {
253  Tangent = Base::Vector2d(dirVec.y, -dirVec.x);
254  }
255 
256  double theta = Tangent.GetAngle(onSketchPos - EditCurve[0]);
257 
258  arcRadius = (onSketchPos - EditCurve[0]).Length() / (2.0 * sin(theta));
259 
260  // At this point we need a unit normal vector pointing towards
261  // the center of the arc we are drawing. Derivation of the formula
262  // used here can be found at
263  // http://people.richland.edu/james/lecture/m116/matrices/area.html
264  double x1 = EditCurve[0].x;
265  double y1 = EditCurve[0].y;
266  double x2 = x1 + Tangent.x;
267  double y2 = y1 + Tangent.y;
268  double x3 = onSketchPos.x;
269  double y3 = onSketchPos.y;
270  if ((x2 * y3 - x3 * y2) - (x1 * y3 - x3 * y1) + (x1 * y2 - x2 * y1) > 0) {
271  arcRadius *= -1;
272  }
273  if (boost::math::isnan(arcRadius) || boost::math::isinf(arcRadius)) {
274  arcRadius = 0.f;
275  }
276 
277  CenterPoint =
278  EditCurve[0] + Base::Vector2d(arcRadius * Tangent.y, -arcRadius * Tangent.x);
279 
280  double rx = EditCurve[0].x - CenterPoint.x;
281  double ry = EditCurve[0].y - CenterPoint.y;
282 
283  startAngle = atan2(ry, rx);
284 
285  double rxe = onSketchPos.x - CenterPoint.x;
286  double rye = onSketchPos.y - CenterPoint.y;
287  double arcAngle = atan2(-rxe * ry + rye * rx, rxe * rx + rye * ry);
288  if (boost::math::isnan(arcAngle) || boost::math::isinf(arcAngle)) {
289  arcAngle = 0.f;
290  }
291  if (arcRadius >= 0 && arcAngle > 0) {
292  arcAngle -= 2 * M_PI;
293  }
294  if (arcRadius < 0 && arcAngle < 0) {
295  arcAngle += 2 * M_PI;
296  }
297 
298  if (SnapMode == SNAP_MODE_45Degree) {
299  arcAngle = round(arcAngle / (M_PI / 4)) * M_PI / 4;
300  }
301 
302  endAngle = startAngle + arcAngle;
303 
304  for (int i = 1; i <= 29; i++) {
305  double angle = i * arcAngle / 29.0;
306  double dx = rx * cos(angle) - ry * sin(angle);
307  double dy = rx * sin(angle) + ry * cos(angle);
308  EditCurve[i] = Base::Vector2d(CenterPoint.x + dx, CenterPoint.y + dy);
309  }
310 
311  EditCurve[30] = CenterPoint;
312  EditCurve[31] = EditCurve[0];
313 
314  drawEdit(EditCurve);
315 
316  if (showCursorCoords()) {
317  SbString text;
318  std::string radiusString = lengthToDisplayFormat(std::abs(arcRadius), 1);
319  std::string angleString = angleToDisplayFormat(arcAngle * 180.0 / M_PI, 1);
320  text.sprintf(" (R%s, %s)", radiusString.c_str(), angleString.c_str());
321  setPositionText(onSketchPos, text);
322  }
323 
324  seekAndRenderAutoConstraint(sugConstr2, onSketchPos, Base::Vector2d(0.f, 0.f));
325  }
326  }
327  }

329  bool pressButton(Base::Vector2d onSketchPos) override
330  {
331  if (Mode == STATUS_SEEK_First) {
332 
333  EditCurve[0] = onSketchPos; // this may be overwritten if previousCurve is found
334 
335  virtualsugConstr1 = sugConstr1; // store original autoconstraints.
336 
337  // here we check if there is a preselected point and
338  // we set up a transition from the neighbouring segment.
339  // (peviousCurve, previousPosId, dirVec, TransitionMode)
340  for (unsigned int i = 0; i < sugConstr1.size(); i++) {
341  if (sugConstr1[i].Type == Sketcher::Coincident) {
342  const Part::Geometry* geom =
343  sketchgui->getSketchObject()->getGeometry(sugConstr1[i].GeoId);
344  if ((geom->is<Part::GeomLineSegment>() || geom->is<Part::GeomArcOfCircle>())
345  && (sugConstr1[i].PosId == Sketcher::PointPos::start
346  || sugConstr1[i].PosId == Sketcher::PointPos::end)) {
347  previousCurve = sugConstr1[i].GeoId;
348  previousPosId = sugConstr1[i].PosId;
349  updateTransitionData(previousCurve,
350  previousPosId); // -> dirVec, EditCurve[0]
351  if (geom->is<Part::GeomArcOfCircle>()) {
352  TransitionMode = TRANSITION_MODE_Tangent;
353  SnapMode = SNAP_MODE_Free;
354  }
355  sugConstr1.erase(sugConstr1.begin()
356  + i); // actually we should clear the vector completely
357  break;
358  }
359  }
360  }
361 
362  // remember our first point (even if we are doing a transition from a previous curve)
363  firstCurve = getHighestCurveIndex() + 1;
364  firstPosId = Sketcher::PointPos::start;
365 
366  if (SegmentMode == SEGMENT_MODE_Line) {
367  EditCurve.resize(TransitionMode == TRANSITION_MODE_Free ? 2 : 3);
368  }
369  else if (SegmentMode == SEGMENT_MODE_Arc) {
370  EditCurve.resize(32);
371  }
372  Mode = STATUS_SEEK_Second;
373  }
374  else if (Mode == STATUS_SEEK_Second) {
375  // exit on clicking exactly at the same position (e.g. double click)
376  if (onSketchPos == EditCurve[0]) {
377  unsetCursor();
378  resetPositionText();
379  EditCurve.clear();
380  drawEdit(EditCurve);
381 
382  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
383  "User parameter:BaseApp/Preferences/Mod/Sketcher");
384  bool continuousMode = hGrp->GetBool("ContinuousCreationMode", true);
385 
386  if (continuousMode) {
387  // This code enables the continuous creation mode.
388  Mode = STATUS_SEEK_First;
389  SegmentMode = SEGMENT_MODE_Line;
390  TransitionMode = TRANSITION_MODE_Free;
391  SnapMode = SNAP_MODE_Free;
392  suppressTransition = false;
393  firstCurve = -1;
394  previousCurve = -1;
395  firstPosId = Sketcher::PointPos::none;
396  previousPosId = Sketcher::PointPos::none;
397  EditCurve.clear();
398  drawEdit(EditCurve);
399  EditCurve.resize(2);
400  applyCursor();
401  /* this is ok not to call to purgeHandler
402  * in continuous creation mode because the
403  * handler is destroyed by the quit() method on pressing the
404  * right button of the mouse */
405  return true;
406  }
407  else {
408  sketchgui->purgeHandler(); // no code after this line, Handler get deleted in
409  // ViewProvider
410  return true;
411  }
412  }
413 
414  Mode = STATUS_Do;
415 
416  if (getPreselectPoint() != -1 && firstPosId != Sketcher::PointPos::none) {
417  int GeoId;
418  Sketcher::PointPos PosId;
419  sketchgui->getSketchObject()->getGeoVertexIndex(getPreselectPoint(), GeoId, PosId);
420  if (sketchgui->getSketchObject()->arePointsCoincident(GeoId,
421  PosId,
422  firstCurve,
423  firstPosId)) {
424  Mode = STATUS_Close;
425  }
426  }
427  else if (getPreselectCross() == 0 && firstPosId != Sketcher::PointPos::none) {
428  // close line started at root point
429  if (sketchgui->getSketchObject()->arePointsCoincident(-1,
430  Sketcher::PointPos::start,
431  firstCurve,
432  firstPosId)) {
433  Mode = STATUS_Close;
434  }
435  }
436  }
437  return true;
438  }

440  bool releaseButton(Base::Vector2d onSketchPos) override
441  {
442  if (Mode == STATUS_Do || Mode == STATUS_Close) {
443  bool addedGeometry = true;
444  if (SegmentMode == SEGMENT_MODE_Line) {
445  // issue the geometry
446  try {
447  // open the transaction
448  Gui::Command::openCommand(
449  QT_TRANSLATE_NOOP("Command", "Add line to sketch polyline"));
450  Gui::cmdAppObjectArgs(
451  sketchgui->getObject(),
452  "addGeometry(Part.LineSegment(App.Vector(%f,%f,0),App.Vector(%f,%f,0)),%s)",
453  EditCurve[0].x,

472  return true;

495  Gui::Command::abortCommand();

504  Sketcher::PointPos lastStartPosId =
505  (SegmentMode == SEGMENT_MODE_Arc && startAngle > endAngle)
506  ? Sketcher::PointPos::end
507  : Sketcher::PointPos::start;
508  Sketcher::PointPos lastEndPosId =
509  (SegmentMode == SEGMENT_MODE_Arc && startAngle > endAngle)
510  ? Sketcher::PointPos::start
511  : Sketcher::PointPos::end;
512  // in case of a tangency constraint, the coincident constraint is redundant
513  std::string constrType = "Coincident";
514  if (!suppressTransition && previousCurve != -1) {
515  if (TransitionMode == TRANSITION_MODE_Tangent) {
516  constrType = "Tangent";
517  }
518  else if (TransitionMode == TRANSITION_MODE_Perpendicular_L

520  constrType = "Perpendicular";
521  }
522  }
523  Gui::cmdAppObjectArgs(sketchgui->getObject(),
524  "addConstraint(Sketcher.Constraint('%s',%i,%i,%i,%i)) ",

528  lastCurve,
529  static_cast<int>(lastStartPosId));

531  if (SnapMode == SNAP_MODE_45Degree && Mode != STATUS_Close) {
532  // -360, -315, -270, -225, -180, -135, -90, -45, 0, 45, 90, 135, 180, 225,
533  // 270, 315, 360
534  // N/A, a, perp, a, par, a,perp, a,N/A, a,perp, a, par, a,perp,
535  // a, N/A
536 
537  // #3974: if in radians, the printf %f defaults to six decimals, which leads to
538  // loss of precision
539  double arcAngle =
540  abs(round((endAngle - startAngle) / (M_PI / 4)) * 45); // in degrees
541 
542  Gui::cmdAppObjectArgs(sketchgui->getObject(),
543  "addConstraint(Sketcher.Constraint('Angle',%i,App.Units."
544  "Quantity('%f deg'))) ",
545  lastCurve,
546  arcAngle);
547  }
548  if (Mode == STATUS_Close) {
549  // close the loop by constrain to the first curve point
550  Gui::cmdAppObjectArgs(
551  sketchgui->getObject(),
552  "addConstraint(Sketcher.Constraint('Coincident',%i,%i,%i,%i)) ",
553  lastCurve,
554  static_cast<int>(lastEndPosId),
555  firstCurve,
556  static_cast<int>(firstPosId));
557  firstsegment = true;
558  }
559  Gui::Command::commitCommand();
560 
561  tryAutoRecomputeIfNotSolve(sketchgui->getObject<Sketcher::SketchObject>());
562  }

565  "User parameter:BaseApp/Preferences/Mod/Sketcher");

567  sketchgui->AvoidRedundant.getValue() && sketchgui->Autoconstraints.getValue();
568 
569  if (Mode == STATUS_Close) {
570

572  if (SegmentMode == SEGMENT_MODE_Line) { // avoid redundant constraints.

612  bool continuousMode = hGrp->GetBool("ContinuousCreationMode", true);
613

615  // This code enables the continuous creation mode.
616  Mode = STATUS_SEEK_First;
617  SegmentMode = SEGMENT_MODE_Line;
618  TransitionMode = TRANSITION_MODE_Free;
619  SnapMode = SNAP_MODE_Free;
