57 void ExtrusionHelper::makeDraft(const TopoDS_Shape& shape,
58  const gp_Dir& direction,
59  const double LengthFwd,
60  const double LengthRev,
61  const double AngleFwd,
62  const double AngleRev,
63  bool isSolid,
64  std::list<TopoDS_Shape>& drafts,
65  bool isPartDesign)
66 {
67  std::vector<std::vector<TopoDS_Shape>> wiresections;
68 
69  auto addWiresToWireSections =
70  [&shape](std::vector<std::vector<TopoDS_Shape>>& wiresections) -> size_t {
71  TopExp_Explorer ex;
72  size_t i = 0;
73  for (ex.Init(shape, TopAbs_WIRE); ex.More(); ex.Next(), ++i) {
74  wiresections.emplace_back();
75  wiresections[i].push_back(TopoDS::Wire(ex.Current()));
76  }
77  return i;
78  };
79 
80  double distanceFwd = tan(AngleFwd) * LengthFwd;
81  double distanceRev = tan(AngleRev) * LengthRev;
82  gp_Vec vecFwd = gp_Vec(direction) * LengthFwd;
83  gp_Vec vecRev = gp_Vec(direction.Reversed()) * LengthRev;
84 
85  bool bFwd = fabs(LengthFwd) > Precision::Confusion();
86  bool bRev = fabs(LengthRev) > Precision::Confusion();
87  // only if there is a 2nd direction and the negated angle is equal to the first one
88  // we can omit the source shape as loft section
89  bool bMid = !bFwd || !bRev || -1.0 * AngleFwd != AngleRev;
90 
91  if (shape.IsNull())
92  Standard_Failure::Raise("Not a valid shape");
93 
94  // store all wires of the shape into an array
95  size_t numWires = addWiresToWireSections(wiresections);
96  if (numWires == 0)
97  Standard_Failure::Raise("Extrusion: Input must not only consist if a vertex");
98 
99  // to store the sections for the loft
100  std::list<TopoDS_Wire> list_of_sections;
101 
102  // we need for all found wires an offset copy of them
103  // we store them in an array
104  TopoDS_Wire offsetWire;
105  std::vector<std::vector<TopoDS_Shape>> extrusionSections(wiresections.size(), std::vector<TopoDS_Shape>());
106  size_t rows = 0;
107  int numEdges = 0;
108 
109  // We need to find out what are outer wires and what are inner ones
110  // methods like checking the center of mass etc. don't help us here.
111  // As solution we build a prism with every wire, then subtract every prism from each other.
112  // If the moment of inertia changes by a subtraction, we have an inner wire prism.
113  //
114  // first build the prisms
115  std::vector<TopoDS_Shape> resultPrisms;
116  TopoDS_Shape singlePrism;
117  for (auto& wireVector : wiresections) {
118  for (auto& singleWire : wireVector) {
119  BRepBuilderAPI_MakeFace mkFace(TopoDS::Wire(singleWire));
120  auto tempFace = mkFace.Shape();
121  BRepPrimAPI_MakePrism mkPrism(tempFace, vecFwd);
122  if (!mkPrism.IsDone())
123  Standard_Failure::Raise("Extrusion: Generating prism failed");
124  singlePrism = mkPrism.Shape();
125  resultPrisms.push_back(singlePrism);
126  }
127  }
128  // create an array with false to store later which wires are inner ones
129  std::vector<bool> isInnerWire(resultPrisms.size(), false);
130  std::vector<bool> checklist(resultPrisms.size(), true);
131  // finally check reecursively for inner wires
132  checkInnerWires(isInnerWire, direction, checklist, false, resultPrisms);
133 
134  // count the number of inner wires
135  int numInnerWires = 0;
136  for (auto isInner : isInnerWire) {
137  if (isInner)
138  ++numInnerWires;
139  }
140 
141  // at first create offset wires for the reversed part of extrusion
142  // it is important that these wires are the first loft section
143  if (bRev) {
144  // create an offset for all source wires
145  rows = 0;
146  for (auto& wireVector : wiresections) {
147  for (auto& singleWire : wireVector) {
148  // count number of edges
149  numEdges = 0;
150  TopExp_Explorer xp(singleWire, TopAbs_EDGE);
151  while (xp.More()) {
152  numEdges++;
153  xp.Next();
154  }
155  // create an offset copy of the wire
156  if (!isInnerWire[rows]) {
157  // this is an outer wire
158  createTaperedPrismOffset(TopoDS::Wire(singleWire), vecRev, distanceRev, true, offsetWire);
159  }
160  else {
161  // there is an OCC bug with single-edge wires (circles), see inside createTaperedPrismOffset
162  if (numEdges > 1 || !isPartDesign)
163  // inner wires must get the negated offset
164  createTaperedPrismOffset(TopoDS::Wire(singleWire), vecRev, -distanceRev, true, offsetWire);
165  else
166  // circles in PartDesign must not get the negated offset
167  createTaperedPrismOffset(TopoDS::Wire(singleWire), vecRev, distanceRev, true, offsetWire);
168  }
169  if (offsetWire.IsNull())
170  return;
171  extrusionSections[rows].push_back(offsetWire);
172  }
173  ++rows;
174  }
175  }
176 
177  // add the source wire as middle section
178  // it is important to add them after the reversed part
179  if (bMid) {
180  // transfer all source wires as they are to the array from which we build the shells
181  rows = 0;
182  for (auto& wireVector : wiresections) {
183  for (auto& singleWire : wireVector) {
184  extrusionSections[rows].push_back(singleWire);
185  }
186  rows++;
187  }
188  }
189 
190  // finally add the forward extrusion offset wires
191  // these wires must be the last loft section
192  if (bFwd) {
193  rows = 0;
194  for (auto& wireVector : wiresections) {
195  for (auto& singleWire : wireVector) {
196  // count number of edges
197  numEdges = 0;
198  TopExp_Explorer xp(singleWire, TopAbs_EDGE);
199  while (xp.More()) {
200  numEdges++;
201  xp.Next();
202  }
203  // create an offset copy of the wire
204  if (!isInnerWire[rows]) {
205  // this is an outer wire
206  createTaperedPrismOffset(TopoDS::Wire(singleWire), vecFwd, distanceFwd, false, offsetWire);
207  }
208  else {
209  // there is an OCC bug with single-edge wires (circles), see inside createTaperedPrismOffset
210  if (numEdges > 1 || !isPartDesign)
211  // inner wires must get the negated offset
212  createTaperedPrismOffset(TopoDS::Wire(singleWire), vecFwd, -distanceFwd, false, offsetWire);
213  else
214  // circles in PartDesign must not get the negated offset
215  createTaperedPrismOffset(TopoDS::Wire(singleWire), vecFwd, distanceFwd, false, offsetWire);
216  }
217  if (offsetWire.IsNull())
218  return;
219  extrusionSections[rows].push_back(offsetWire);
220  }
221  ++rows;
222  }
223  }
224 
225  try {
226  // build all shells
227  std::vector<TopoDS_Shape> shells;
228 
229  for (auto& wires : extrusionSections) {
230  BRepOffsetAPI_ThruSections mkTS(isSolid, /*ruled=*/Standard_True, Precision::Confusion());
231 
232  for (auto& singleWire : wires) {
233  if (singleWire.ShapeType() == TopAbs_VERTEX)
234  mkTS.AddVertex(TopoDS::Vertex(singleWire));
235  else
236  mkTS.AddWire(TopoDS::Wire(singleWire));
237  }
238  mkTS.Build();
239  if (!mkTS.IsDone())
240  Standard_Failure::Raise("Extrusion: Loft could not be built");
241 
242  shells.push_back(mkTS.Shape());
243  }
244 
245  if (isSolid) {
246  // we only need to cut if we have inner wires
247  if (numInnerWires > 0) {
248  // we take every outer wire prism and cut subsequently all inner wires prisms from it
249  // every resulting shape is the final drafted extrusion shape
250  GProp_GProps tempProperties;
251  Standard_Real momentOfInertiaInitial;
252  Standard_Real momentOfInertiaFinal;
253  std::vector<bool>::iterator isInnerWireIterator = isInnerWire.begin();
254  std::vector<bool>::iterator isInnerWireIteratorLoop;
255  for (auto itOuter = shells.begin(); itOuter != shells.end(); ++itOuter) {
256  if (*isInnerWireIterator) {
257  ++isInnerWireIterator;
258  continue;
259  }
260  isInnerWireIteratorLoop = isInnerWire.begin();
261  for (auto itInner = shells.begin(); itInner != shells.end(); ++itInner) {
262  if (itOuter == itInner || !*isInnerWireIteratorLoop) {
263  ++isInnerWireIteratorLoop;
264  continue;
265  }
266  // get MomentOfInertia of first shape
267  BRepGProp::VolumeProperties(*itOuter, tempProperties);
268  momentOfInertiaInitial = tempProperties.MomentOfInertia(gp_Ax1(gp_Pnt(), direction));
269  FCBRepAlgoAPI_Cut mkCut(*itOuter, *itInner);
270  if (!mkCut.IsDone())
271  Standard_Failure::Raise("Extrusion: Final cut out failed");
272  BRepGProp::VolumeProperties(mkCut.Shape(), tempProperties);
273  momentOfInertiaFinal = tempProperties.MomentOfInertia(gp_Ax1(gp_Pnt(), direction));
274  // if the whole shape was cut away the resulting shape is not Null but its MomentOfInertia is 0.0
275  // therefore we have a valid cut if the MomentOfInertia is not zero and changed
276  if ((momentOfInertiaInitial != momentOfInertiaFinal)
277  && (momentOfInertiaFinal > Precision::Confusion())) {
278  // immediately update the outer shape since more inner wire prism might cut it
279  *itOuter = mkCut.Shape();
280  }
281  ++isInnerWireIteratorLoop;
282  }
283  drafts.push_back(*itOuter);
284  ++isInnerWireIterator;
285  }
286  }
287  else
288  // we already have the results
289  for (const auto & shell : shells)
290  drafts.push_back(shell);
291  }
292  else { // no solid
293  BRepBuilderAPI_Sewing sewer;
294  sewer.SetTolerance(Precision::Confusion());
295  for (TopoDS_Shape& s : shells)
296  sewer.Add(s);
297  sewer.Perform();
298  drafts.push_back(sewer.SewedShape());
299  }
300  }
301  catch (Standard_Failure& e) {
302  throw Base::RuntimeError(e.GetMessageString());
303  }
304  catch (const Base::Exception& e) {
305  throw Base::RuntimeError(e.what());
306  }
307  catch (...) {
308  throw Base::CADKernelError("Extrusion: A fatal error occurred when making the loft");
309  }
310 }

312 void ExtrusionHelper::checkInnerWires(std::vector<bool>& isInnerWire, const gp_Dir direction,
313  std::vector<bool>&checklist, bool forInner, std::vector<TopoDS_Shape> prisms)
314 {
315  // store the number of wires to be checked
316  size_t numCheckWiresInitial = 0;
317  for (auto checks : checklist) {
318  if (checks)
319  ++numCheckWiresInitial;
320  }
321  GProp_GProps tempProperties;
322  Standard_Real momentOfInertiaInitial;
323  Standard_Real momentOfInertiaFinal;
324  size_t numCheckWires = 0;
325  std::vector<bool>::iterator isInnerWireIterator = isInnerWire.begin();
326  std::vector<bool>::iterator toCheckIterator = checklist.begin();
327  // create an array with false used later to store what can be cancelled from the checklist
328  std::vector<bool> toDisable(checklist.size(), false);
329  int outer = -1;
330  // we cut every prism to be checked from the other to be checked ones
331  // if nothing happens, a prism can be cancelled from the checklist
332  for (auto itOuter = prisms.begin(); itOuter != prisms.end(); ++itOuter) {
333  ++outer;
334  if (!*toCheckIterator) {
335  ++isInnerWireIterator;
336  ++toCheckIterator;
337  continue;
338  }
339  auto toCheckIteratorInner = checklist.begin();
340  bool saveIsInnerWireIterator = *isInnerWireIterator;
341  for (auto itInner = prisms.begin(); itInner != prisms.end(); ++itInner) {
342  if (itOuter == itInner || !*toCheckIteratorInner) {
343  ++toCheckIteratorInner;
344  continue;
345  }
346  // get MomentOfInertia of first shape
347  BRepGProp::VolumeProperties(*itInner, tempProperties);
348  momentOfInertiaInitial = tempProperties.MomentOfInertia(gp_Ax1(gp_Pnt(), direction));
349  FCBRepAlgoAPI_Cut mkCut(*itInner, *itOuter);
350  if (!mkCut.IsDone())
351  Standard_Failure::Raise("Extrusion: Cut out failed");
352  BRepGProp::VolumeProperties(mkCut.Shape(), tempProperties);
353  momentOfInertiaFinal = tempProperties.MomentOfInertia(gp_Ax1(gp_Pnt(), direction));
354  // if the whole shape was cut away the resulting shape is not Null but its MomentOfInertia is 0.0
355  // therefore we have an inner wire if the MomentOfInertia is not zero and changed
356  if ((momentOfInertiaInitial != momentOfInertiaFinal)
357  && (momentOfInertiaFinal > Precision::Confusion())) {
358  *isInnerWireIterator = !forInner;
359  ++numCheckWires;
360  *toCheckIterator = true;
361  break;
362  }
363  ++toCheckIteratorInner;
364  }
365  if (saveIsInnerWireIterator == *isInnerWireIterator)
366  // nothing was changed and we can remove it from the list to be checked
367  // but we cannot do this before the for loop was fully run
368  toDisable[outer] = true;
369  ++isInnerWireIterator;
370  ++toCheckIterator;
371  }
372 
373  // cancel prisms from the checklist whose wire state did not change
374  size_t i = 0;
375  for (auto disable : toDisable) {
376  if (disable)
377  checklist[i] = false;
378  ++i;
379  }
380 
381  // if all wires are inner ones, we take the first one as outer and issue a warning
382  if (numCheckWires == isInnerWire.size()) {
383  isInnerWire[0] = false;
384  checklist[0] = false;
385  --numCheckWires;
386  Base::Console().Warning("Extrusion: could not determine what structure is the outer one.\n\
387  The first input one will now be taken as outer one.\n");
388  }
389 
390  // There can be cases with several wires all intersecting each other.
391  // Then it is impossible to find out what wire is an inner one
392  // and we can only treat all wires in the checklist as outer ones.
393  if (numCheckWiresInitial == numCheckWires) {
394  i = 0;
395  for (auto checks : checklist) {
396  if (checks) {
397  isInnerWire[i] = false;
398  checklist[i] = false;
399  --numCheckWires;
400  }
401  ++i;
402  }
403  Base::Console().Warning("Extrusion: too many self-intersection structures!\n\
404  Impossible to determine what structure is an inner one.\n\
405  All undeterminable structures will therefore be taken as outer ones.\n");
406  }
407 
408  // recursively call the function until all wires are checked
409  if (numCheckWires > 1)
410  checkInnerWires(isInnerWire, direction, checklist, !forInner, prisms);
411 }

413 void ExtrusionHelper::createTaperedPrismOffset(TopoDS_Wire sourceWire,
414  const gp_Vec& translation,
415  double offset,
416  bool isSecond,
417  TopoDS_Wire& result) {
418 
419  // if the wire consists of a single edge which has applied a placement
420  // then this placement must be reset because otherwise
421  // BRepOffsetAPI_MakeOffset shows weird behaviour by applying the placement, see
422  // https://dev.opencascade.org/content/brepoffsetapimakeoffset-wire-and-face-odd-occt-740
423  // therefore we use here the workaround of BRepOffsetAPI_MakeOffsetFix and not BRepOffsetAPI_MakeOffset
424 
425  gp_Trsf tempTransform;
426  tempTransform.SetTranslation(translation);
427  TopLoc_Location loc(tempTransform);
428  TopoDS_Wire movedSourceWire = TopoDS::Wire(sourceWire.Moved(loc));
429 
430  TopoDS_Shape offsetShape;
431  if (fabs(offset) > Precision::Confusion()) {
432  TopLoc_Location edgeLocation;
433  // create the offset shape
434  BRepOffsetAPI_MakeOffsetFix mkOffset;
435  mkOffset.Init(GeomAbs_Arc);
436  mkOffset.Init(GeomAbs_Intersection);
437  mkOffset.AddWire(movedSourceWire);
438  try {
439  mkOffset.Perform(offset);
440  offsetShape = mkOffset.Shape();
441  }
442  catch (const Base::Exception& e) {
443  throw Base::RuntimeError(e.what());
444  }
445  if (!mkOffset.IsDone()) {
446  Standard_Failure::Raise("Extrusion: Offset could not be created");
447  }
448  }
449  else {
450  offsetShape = movedSourceWire;
451  }
452  if (offsetShape.IsNull()) {
453  if (isSecond)
454  Base::Console().Error("Extrusion: end face of tapered against extrusion is empty\n" \
455  "This means most probably that the against taper angle is too large or small.\n");
456  else
457  Base::Console().Error("Extrusion: end face of tapered along extrusion is empty\n" \
458  "This means most probably that the along taper angle is too large or small.\n");
459  Standard_Failure::Raise("Extrusion: end face of tapered extrusion is empty");
460  }
461  // assure we return a wire and no edge
462  TopAbs_ShapeEnum type = offsetShape.ShapeType();
463  if (type == TopAbs_WIRE) {
464  result = TopoDS::Wire(offsetShape);
465  }
466  else if (type == TopAbs_EDGE) {
467  BRepBuilderAPI_MakeWire mkWire2(TopoDS::Edge(offsetShape));
468  result = mkWire2.Wire();
469  }
470  else {
471  // this happens usually if type == TopAbs_COMPOUND and means the angle is too small
472  // since this is a common mistake users will quickly do, issue a warning dialog
473  // FIXME: Standard_Failure::Raise or App::DocumentObjectExecReturn don't output the message to the user
474  result = TopoDS_Wire();
475  if (isSecond)
476  Base::Console().Error("Extrusion: type of against extrusion end face is not supported.\n" \
477  "This means most probably that the against taper angle is too large or small.\n");
478  else
479  Base::Console().Error("Extrusion: type of along extrusion is not supported.\n" \
480  "This means most probably that the along taper angle is too large or small.\n");
481  }
482 
483 }

485 void ExtrusionHelper::makeElementDraft(const ExtrusionParameters& params,
486  const TopoShape& _shape,
487  std::vector<TopoShape>& drafts,
488  App::StringHasherRef hasher)
489 {
490  double distanceFwd = tan(params.taperAngleFwd) * params.lengthFwd;
491  double distanceRev = tan(params.taperAngleRev) * params.lengthRev;
492 
493  gp_Vec vecFwd = gp_Vec(params.dir) * params.lengthFwd;
494  gp_Vec vecRev = gp_Vec(params.dir.Reversed()) * params.lengthRev;
495 
496  bool bFwd = fabs(params.lengthFwd) > Precision::Confusion();
497  bool bRev = fabs(params.lengthRev) > Precision::Confusion();
498  bool bMid = !bFwd || !bRev
499  || params.lengthFwd * params.lengthRev > 0.0; // include the source shape as loft section?
500 
501  TopoShape shape = _shape;
502  TopoShape sourceWire;
503  if (shape.isNull()) {
504  Standard_Failure::Raise("Not a valid shape");
505  }
506 
507  if (params.solid && !shape.hasSubShape(TopAbs_FACE)) {
508  shape = shape.makeElementFace(nullptr, params.faceMakerClass.c_str());
509  }
510 
511  if (shape.shapeType() == TopAbs_FACE) {
512  std::vector<TopoShape> wires;
513  TopoShape outerWire = shape.splitWires(&wires, TopoShape::ReorientForward);
514  if (outerWire.isNull()) {
515  Standard_Failure::Raise("Missing outer wire");
516  }
517  if (wires.empty()) {
518  shape = outerWire;
519  }
520  else {
521  unsigned pos = drafts.size();
522  makeElementDraft(params, outerWire, drafts, hasher);
523  if (drafts.size() != pos + 1) {
524  Standard_Failure::Raise("Failed to make drafted extrusion");
525  }
526  std::vector<TopoShape> inner;
527  TopoShape innerWires(0);
528  innerWires.makeElementCompound(
529  wires,
530  "",
531  TopoShape::SingleShapeCompoundCreationPolicy::returnShape);
532  makeElementDraft(params, innerWires, inner, hasher);
533  if (inner.empty()) {
534  Standard_Failure::Raise("Failed to make drafted extrusion with inner hole");
535  }
536  inner.insert(inner.begin(), drafts.back());
537  drafts.back().makeElementCut(inner);
538  return;
539  }
540  }
541 
542  if (shape.shapeType() == TopAbs_WIRE) {
543  ShapeFix_Wire aFix;
544  aFix.Load(TopoDS::Wire(shape.getShape()));
545  aFix.FixReorder();
546  aFix.FixConnected();
547  aFix.FixClosed();
548  sourceWire.setShape(aFix.Wire());
549  sourceWire.Tag = shape.Tag;
550  sourceWire.mapSubElement(shape);
551  }
552  else if (shape.shapeType() == TopAbs_COMPOUND) {
553  for (auto& s : shape.getSubTopoShapes()) {
554  makeElementDraft(params, s, drafts, hasher);
555  }
556  }
557  else {
558  Standard_Failure::Raise("Only a wire or a face is supported");
559  }
560 
561  if (!sourceWire.isNull()) {
562  std::vector<TopoShape> list_of_sections;
563  if (bRev) {
564  TopoDS_Wire offsetWire;
565  createTaperedPrismOffset(TopoDS::Wire(sourceWire.getShape()),
566  vecRev,
567  distanceRev,
568  false,
569  offsetWire);
570  list_of_sections.push_back(TopoShape(offsetWire, sourceWire.Tag));
571  }
572 
573  // next. Add source wire as middle section. Order is important.
574  if (bMid) {
575  list_of_sections.push_back(sourceWire);
576  }
577 
578  // finally. Forward extrusion offset wire.
579  if (bFwd) {
580  TopoDS_Wire offsetWire;
581  createTaperedPrismOffset(TopoDS::Wire(sourceWire.getShape()),
582  vecFwd,
583  distanceFwd,
584  false,
585  offsetWire);
586  list_of_sections.push_back(TopoShape(offsetWire, sourceWire.Tag));
587  }
588 
589  try {
590 #if defined(__GNUC__) && defined(FC_OS_LINUX)
591  Base::SignalException se;
592 #endif
593 
594  // make loft
595  BRepOffsetAPI_ThruSections mkGenerator(params.solid ? Standard_True : Standard_False,
596  /*ruled=*/Standard_True);
597  for (auto& s : list_of_sections) {
598  mkGenerator.AddWire(TopoDS::Wire(s.getShape()));
599  }
600 
601  mkGenerator.Build();
602  drafts.push_back(TopoShape(0, hasher).makeElementShape(mkGenerator, list_of_sections));
603  }
604  catch (Standard_Failure&) {
605  throw;
606  }
607  catch (...) {
608  throw Base::CADKernelError("Unknown exception from BRepOffsetAPI_ThruSections");
609  }
610  }
