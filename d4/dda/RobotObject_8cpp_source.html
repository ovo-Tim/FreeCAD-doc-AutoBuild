125 void RobotObject::onChanged(const Property* prop)
126 {
127 
128  if (prop == &RobotKinematicFile) {
129  // load the new kinematic
130  robot.readKinematic(RobotKinematicFile.getValue());
131  }
132 
133  if (prop == &Axis1 && !block) {
134  robot.setAxis(0, Axis1.getValue());
135  block = true;
136  Tcp.setValue(robot.getTcp());
137  block = false;
138  }
139  if (prop == &Axis2 && !block) {
140  robot.setAxis(1, Axis2.getValue());
141  block = true;
142  Tcp.setValue(robot.getTcp());
143  block = false;
144  }
145  if (prop == &Axis3 && !block) {
146  robot.setAxis(2, Axis3.getValue());
147  block = true;
148  Tcp.setValue(robot.getTcp());
149  block = false;
150  }
151  if (prop == &Axis4 && !block) {
152  robot.setAxis(3, Axis4.getValue());
153  block = true;
154  Tcp.setValue(robot.getTcp());
155  block = false;
156  }
157  if (prop == &Axis5 && !block) {
158  robot.setAxis(4, Axis5.getValue());
159  block = true;
160  Tcp.setValue(robot.getTcp());
161  block = false;
162  }
163  if (prop == &Axis6 && !block) {
164  robot.setAxis(5, Axis6.getValue());
165  block = true;
166  Tcp.setValue(robot.getTcp());
167  block = false;
168  }
169  if (prop == &Tcp && !block) {
170  robot.setTo(Tcp.getValue());
171  block = true;
172  Axis1.setValue((float)robot.getAxis(0));
173  Axis2.setValue((float)robot.getAxis(1));
174  Axis3.setValue((float)robot.getAxis(2));
175  Axis4.setValue((float)robot.getAxis(3));
176  Axis5.setValue((float)robot.getAxis(4));
177  Axis6.setValue((float)robot.getAxis(5));
178  block = false;
179  }
180  App::GeoFeature::onChanged(prop);
181 }
