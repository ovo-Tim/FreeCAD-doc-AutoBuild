58 std::string DimensionFormatter::formatValue(const qreal value,
59  const QString& qFormatSpec,
60  const int partial,
61  const bool isDim) const
62 {
63 // Base::Console().Message("DF::formatValue() - %s isRestoring: %d\n",
64 // m_dimension->getNameInDocument(), m_dimension->isRestoring());
65  bool angularMeasure = m_dimension->Type.isValue("Angle") || m_dimension->Type.isValue("Angle3Pt");
66  bool areaMeasure = m_dimension->Type.isValue("Area");
67  QLocale loc;
68 
69  Base::Quantity asQuantity;
70  asQuantity.setValue(value);
71  if (angularMeasure) {
72  asQuantity.setUnit(Base::Unit::Angle);
73  }
74  else if (areaMeasure) {
75  asQuantity.setUnit(Base::Unit::Area);
76  }
77  else {
78  asQuantity.setUnit(Base::Unit::Length);
79  }
80 
81  // this handles mm to inch/km/parsec etc and decimal positions but
82  // won't give more than Global_Decimals precision
83  QString qUserString = QString::fromStdString(asQuantity.getUserString());
84 
85  //get formatSpec prefix/suffix/specifier
86  QStringList qsl = getPrefixSuffixSpec(qFormatSpec);
87  QString formatPrefix = qsl[0]; //FormatSpec prefix
88  QString formatSuffix = qsl[1]; //FormatSpec suffix
89  QString formatSpecifier = qsl[2]; //FormatSpec specifier
90 
91  QString qMultiValueStr;
92  QString qBasicUnit = QString::fromStdString(Base::UnitsApi::getBasicLengthUnit());
93 
94  QString formattedValue;
95  if (isMultiValueSchema() && partial == 0) {
96  //handle multi value schemes (yd/ft/in, dms, etc). don't even try to use Alt Decimals or hide units
97  qMultiValueStr = formatPrefix + qUserString + formatSuffix;
98  return qMultiValueStr.toStdString();
99  } else {
100  //not multivalue schema
101  if (formatSpecifier.isEmpty()) {
102  Base::Console().Warning("Warning - no numeric format in Format Spec %s - %s\n",
103  qPrintable(qFormatSpec), m_dimension->getNameInDocument());
104  return qFormatSpec.toStdString();
105  }
106 
107  // for older TD drawings the formatSpecifier "%g" was used, but the number of decimals was
108  // neverheless limited. To keep old drawings, we limit the number of decimals too
109  // if the TD preferences option to use the global decimal number is set
110  // the formatSpecifier can have a prefix and/or suffix
111  if (m_dimension->useDecimals() && formatSpecifier.contains(QString::fromLatin1("%g"), Qt::CaseInsensitive)) {
112  int globalPrecision = Base::UnitsApi::getDecimals();
113  // change formatSpecifier to e.g. "%.2f"
114  QString newSpecifier = QString::fromStdString("%." + std::to_string(globalPrecision) + "f");
115  formatSpecifier.replace(QString::fromLatin1("%g"), newSpecifier, Qt::CaseInsensitive);
116  }
117 
118  // since we are not using a multiValueSchema, we know that angles are in '°' and for
119  // lengths we can get the unit of measure from UnitsApi::getBasicLengthUnit.
120 
121  // TODO: check the weird schemas (MKS, Imperial1)that report different UoM
122  // for different values
123 
124  // get value in the base unit with default decimals
125  // for the conversion we use the same method as in DlgUnitsCalculator::valueChanged
126  // get the conversion factor for the unit
127  // the result is now just val / convertValue because val is always in the base unit
128  // don't do this for angular values since they are not in the BaseLengthUnit
129  double userVal;
130  if (angularMeasure) {
131  userVal = asQuantity.getValue();
132  qBasicUnit = QString::fromUtf8("°");
133  }
134  else {
135  double convertValue = Base::Quantity::parse("1" + qBasicUnit.toStdString()).getValue();
136  userVal = asQuantity.getValue() / convertValue;
137  if (areaMeasure) {
138  userVal = userVal / convertValue; // divide again as area is length²
139  qBasicUnit = qBasicUnit + QString::fromUtf8("²");
140  }
141  }
142 
143  if (isTooSmall(userVal, formatSpecifier)) {
144  Base::Console().Warning("Dimension %s value %.6f is too small for format specifier: %s\n",
145  m_dimension->getNameInDocument(), userVal, qPrintable(formatSpecifier));
146  }
147 
148  formattedValue = formatValueToSpec(userVal, formatSpecifier);
149 
150  // replace decimal sign if necessary
151  QChar dp = QChar::fromLatin1('.');
152  if (loc.decimalPoint() != dp) {
153  formattedValue.replace(dp, loc.decimalPoint());
154  }
155  }
156 
157  //formattedValue is now in formatSpec format with local decimal separator
158  std::string formattedValueString = formattedValue.toStdString();
159  if (partial == 0) { //prefix + unit subsystem string + suffix
160  return formatPrefix.toStdString() +
161  qUserString.toStdString() +
162  formatSuffix.toStdString();
163  }
164  else if (partial == 1) { // prefix number[unit] suffix
165  if (angularMeasure) {
166  //always insert unit after value
167  return formatPrefix.toStdString() + formattedValueString + "°" +
168  formatSuffix.toStdString();
169  }
170  else if (m_dimension->showUnits() || areaMeasure){
171  if (isDim && m_dimension->haveTolerance()) {
172  //unit will be included in tolerance so don't repeat it here
173  return formatPrefix.toStdString() +
174  formattedValueString +
175  formatSuffix.toStdString();
176  }
177  else {
178  //no tolerance, so we need to include unit
179  return formatPrefix.toStdString() +
180  formattedValueString + " " +
181  qBasicUnit.toStdString() +
182  formatSuffix.toStdString();
183  }
184  }
185  else {
186  //showUnits is false
187  return formatPrefix.toStdString() +
188  formattedValueString +
189  formatSuffix.toStdString();
190  }
191  }
192  else if (partial == 2) { // just the unit
193  if (angularMeasure) {
194  return qBasicUnit.toStdString();
195  }
196  else if (m_dimension->showUnits() || areaMeasure) {
197  return qBasicUnit.toStdString();
198  }
199  else {
200  return "";
201  }
202  }
203 
204  return formattedValueString;
205 }

254 //partial = 2 unit only
255 std::string DimensionFormatter::getFormattedDimensionValue(const int partial) const
256 {
257  QString qFormatSpec = QString::fromUtf8(m_dimension->FormatSpec.getStrValue().data());
258 
259  if ( (m_dimension->Arbitrary.getValue() && !m_dimension->EqualTolerance.getValue())
260  || (m_dimension->Arbitrary.getValue() && m_dimension->TheoreticalExact.getValue()) ) {
261  return m_dimension->FormatSpec.getStrValue();
262  }
263 
264  if (m_dimension->Arbitrary.getValue()) {
265  return m_dimension->FormatSpec.getStrValue();
266  }
267 
268  // if there is an equal over-/undertolerance (so only 1 tolerance to show with +/-) and
269  // not theoretically exact (which has no tolerance), and
270  // tolerance has been specified, ie
271  // (OverTolerance != 0.0 (so a tolerance has been specified) or
272  // ArbitraryTolerances are specified)
273  // concatenate the tolerance to dimension
274  if (m_dimension->EqualTolerance.getValue() &&
275  !m_dimension->TheoreticalExact.getValue() &&
276  (!DrawUtil::fpCompare(m_dimension->OverTolerance.getValue(), 0.0) ||
277  m_dimension->ArbitraryTolerances.getValue())) {
278  QString labelText = QString::fromUtf8(formatValue(m_dimension->getDimValue(),
279  qFormatSpec,
280  1,
281  true).c_str()); //just the number pref/spec[unit]/suf
282  QString unitText = QString::fromUtf8(formatValue(m_dimension->getDimValue(),
283  qFormatSpec,
284  2,
285  false).c_str()); //just the unit
286  QString tolerance = QString::fromStdString(getFormattedToleranceValue(1).c_str());
287 
288  // tolerance might start with a plus sign that we don't want, so cut it off
289  // note plus sign is not at pos = 0!
290  QRegularExpression plus(QString::fromUtf8("^\\s*\\+"));
291  tolerance.remove(plus);
292 
293  return (labelText +
294  QString::fromUtf8(" \xC2\xB1 ") + // +/- symbol
295  tolerance).toStdString();
296 
297  if (partial == 2) {
298  return unitText.toStdString();
299  }
300 
301  return "";
302  }
303 
304  //tolerance not specified, so just format dimension value?
305  std::string formattedValue = formatValue(m_dimension->getDimValue(), qFormatSpec, partial, true);
306 
307  return formattedValue;
308 }

310 // format the value using the formatSpec. Also, handle the non-standard format-
311 // specifier '%w', which has the following rules: works as %f, but no trailing zeros
312 QString DimensionFormatter::formatValueToSpec(const double value, const QString& formatSpecifier) const
313 {
314  QString formattedValue;
315  if (formatSpecifier.contains(QRegularExpression(QStringLiteral("%.*[wW]")))) {
316  QString fs = formatSpecifier;
317  fs.replace(QRegularExpression(QStringLiteral("%(.*)w")), QStringLiteral("%\\1f"));
318  fs.replace(QRegularExpression(QStringLiteral("%(.*)W")), QStringLiteral("%\\1F"));
319  formattedValue = QString::asprintf(fs.toStdString().c_str(), value);
320  // First, try to cut trailing zeros, if AFTER decimal dot there are nonzero numbers
321  // Second, try to cut also decimal dot and zeros, if there are just zeros after it
322  formattedValue.replace(QRegularExpression(QStringLiteral("([0-9][0-9]*\\.[0-9]*[1-9])00*$")), QStringLiteral("\\1"));
323  formattedValue.replace(QRegularExpression(QStringLiteral("([0-9][0-9]*)\\.0*$")), QStringLiteral("\\1"));
324  } else {
325  if (isNumericFormat(formatSpecifier)) {
326  formattedValue = QString::asprintf(formatSpecifier.toStdString().c_str(), value);
327  }
328  }
329 
330  return formattedValue;
331 }

345 QStringList DimensionFormatter::getPrefixSuffixSpec(const QString& fSpec) const
346 {
347  QStringList result;
348  //find the %x.y tag in FormatSpec
349  QRegularExpression rxFormat(QStringLiteral("%[+-]?[0-9]*\\.*[0-9]*[aefgwAEFGW]")); //printf double format spec
350  QRegularExpressionMatch rxMatch;
351  int pos = fSpec.indexOf(rxFormat, 0, &rxMatch);
352  if (pos != -1) {
353  QString match = rxMatch.captured(0); //entire capture of rx
354  QString formatPrefix = fSpec.left(pos);
355  result.append(formatPrefix);
356  QString formatSuffix = fSpec.right(fSpec.size() - pos - match.size());
357  result.append(formatSuffix);
358  result.append(match);
359  } else { //printf format not found!
360  Base::Console().Warning("Warning - no numeric format in formatSpec %s - %s\n",
361  qPrintable(fSpec), m_dimension->getNameInDocument());
362  result.append(QString());
363  result.append(QString());
364  result.append(fSpec);
365  }
366  return result;
367 }

407 //true if value is too small to display using formatSpec
408 bool DimensionFormatter::isTooSmall(const double value, const QString& formatSpec) const
409 {
410  if (TechDraw::DrawUtil::fpCompare(value, 0.0)) {
411  //zero values always fit, so it isn't too small
412  return false;
413  }
414 
415  QRegularExpression rxFormat(QStringLiteral("%[+-]?[0-9]*\\.*([0-9]*)[aefgwAEFGW]")); //printf double format spec
416  QRegularExpressionMatch rxMatch = rxFormat.match(formatSpec);
417  if (rxMatch.hasMatch()) {
418  QString decimalGroup = rxMatch.captured(1);
419  int factor = decimalGroup.toInt();
420  double minValue = pow(10.0, -factor);
421  if (std::fabs(value) < minValue) {
422  return true;
423  }
424  } else {
425  Base::Console().Warning("Failed to parse dimension format spec\n");
426  }
427  return false;
428 }
