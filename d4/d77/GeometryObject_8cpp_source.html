143 void GeometryObject::projectShape(const TopoDS_Shape& inShape, const gp_Ax2& viewAxis)
144 {
145  clear();
146 
147  Handle(HLRBRep_Algo) brep_hlr;
148  try {
149  brep_hlr = new HLRBRep_Algo();
150  // brep_hlr->Debug(true);
151  brep_hlr->Add(inShape, m_isoCount);
152  if (m_isPersp) {
153  double fLength = std::max(Precision::Confusion(), m_focus);
154  HLRAlgo_Projector projector(viewAxis, fLength);
155  brep_hlr->Projector(projector);
156  }
157  else {
158  HLRAlgo_Projector projector(viewAxis);
159  brep_hlr->Projector(projector);
160  }
161  brep_hlr->Update();
162  brep_hlr->Hide();
163  }
164  catch (const Standard_Failure& e) {
165  Base::Console().Error("GO::projectShape - OCC error - %s - while projecting shape\n",
166  e.GetMessageString());
167  throw Base::RuntimeError("GeometryObject::projectShape - OCC error");
168  }
169  catch (...) {
170  throw Base::RuntimeError("GeometryObject::projectShape - unknown error");
171  }
172 
173  try {
174  HLRBRep_HLRToShape hlrToShape(brep_hlr);
175 
176  if (!hlrToShape.VCompound().IsNull()) {
177  visHard = hlrToShape.VCompound();
178  BRepLib::BuildCurves3d(visHard);
179  visHard =ShapeUtils::invertGeometry(visHard);
180  // BRepTools::Write(visHard, "GOvisHard.brep"); //debug
181  }
182 
183  if (!hlrToShape.Rg1LineVCompound().IsNull()) {
184  visSmooth = hlrToShape.Rg1LineVCompound();
185  BRepLib::BuildCurves3d(visSmooth);
186  visSmooth =ShapeUtils::invertGeometry(visSmooth);
187  }
188 
189  if (!hlrToShape.RgNLineVCompound().IsNull()) {
190  visSeam = hlrToShape.RgNLineVCompound();
191  BRepLib::BuildCurves3d(visSeam);
192  visSeam =ShapeUtils::invertGeometry(visSeam);
193  }
194 
195  if (!hlrToShape.OutLineVCompound().IsNull()) {
196  // BRepTools::Write(hlrToShape.OutLineVCompound(), "GOOutLineVCompound.brep"); //debug
197  visOutline = hlrToShape.OutLineVCompound();
198  BRepLib::BuildCurves3d(visOutline);
199  visOutline =ShapeUtils::invertGeometry(visOutline);
200  }
201 
202  if (!hlrToShape.IsoLineVCompound().IsNull()) {
203  visIso = hlrToShape.IsoLineVCompound();
204  BRepLib::BuildCurves3d(visIso);
205  visIso =ShapeUtils::invertGeometry(visIso);
206  }
207 
208  if (!hlrToShape.HCompound().IsNull()) {
209  hidHard = hlrToShape.HCompound();
210  BRepLib::BuildCurves3d(hidHard);
211  hidHard =ShapeUtils::invertGeometry(hidHard);
212  }
213 
214  if (!hlrToShape.Rg1LineHCompound().IsNull()) {
215  hidSmooth = hlrToShape.Rg1LineHCompound();
216  BRepLib::BuildCurves3d(hidSmooth);
217  hidSmooth =ShapeUtils::invertGeometry(hidSmooth);
218  }
219 
220  if (!hlrToShape.RgNLineHCompound().IsNull()) {
221  hidSeam = hlrToShape.RgNLineHCompound();
222  BRepLib::BuildCurves3d(hidSeam);
223  hidSeam =ShapeUtils::invertGeometry(hidSeam);
224  }
225 
226  if (!hlrToShape.OutLineHCompound().IsNull()) {
227  hidOutline = hlrToShape.OutLineHCompound();
228  BRepLib::BuildCurves3d(hidOutline);
229  hidOutline =ShapeUtils::invertGeometry(hidOutline);
230  }
231 
232  if (!hlrToShape.IsoLineHCompound().IsNull()) {
233  hidIso = hlrToShape.IsoLineHCompound();
234  BRepLib::BuildCurves3d(hidIso);
235  hidIso =ShapeUtils::invertGeometry(hidIso);
236  }
237  }
238  catch (const Standard_Failure&) {
239  throw Base::RuntimeError(
240  "GeometryObject::projectShape - OCC error occurred while extracting edges");
241  }
242  catch (...) {
243  throw Base::RuntimeError(
244  "GeometryObject::projectShape - unknown error occurred while extracting edges");
245  }
246 
247  makeTDGeometry();
248 }

250 //convert the hlr output into TD Geometry
251 void GeometryObject::makeTDGeometry()
252 {
253 // Base::Console().Message("GO::makeTDGeometry()\n");
254  extractGeometry(TechDraw::ecHARD, //always show the hard&outline visible lines
255  true);
256  extractGeometry(TechDraw::ecOUTLINE,
257  true);
258 
259  const DrawViewPart* dvp = static_cast<const DrawViewPart*>(m_parent);
260  if (!dvp) {
261  return;//some routines do not have a dvp (ex shape outline)
262  }
263 
264  if (dvp->SmoothVisible.getValue()) {
265  extractGeometry(TechDraw::ecSMOOTH, true);
266  }
267  if (dvp->SeamVisible.getValue()) {
268  extractGeometry(TechDraw::ecSEAM, true);
269  }
270  if ((dvp->IsoVisible.getValue()) && (dvp->IsoCount.getValue() > 0)) {
271  extractGeometry(TechDraw::ecUVISO, true);
272  }
273  if (dvp->HardHidden.getValue()) {
274  extractGeometry(TechDraw::ecHARD, false);
275  extractGeometry(TechDraw::ecOUTLINE, false);
276  }
277  if (dvp->SmoothHidden.getValue()) {
278  extractGeometry(TechDraw::ecSMOOTH, false);
279  }
280  if (dvp->SeamHidden.getValue()) {
281  extractGeometry(TechDraw::ecSEAM, false);
282  }
283  if (dvp->IsoHidden.getValue() && (dvp->IsoCount.getValue() > 0)) {
284  extractGeometry(TechDraw::ecUVISO, false);
285  }

290 void GeometryObject::projectShapeWithPolygonAlgo(const TopoDS_Shape& input, const gp_Ax2& viewAxis)
291 {
292 // Base::Console().Message("GO::projectShapeWithPolygonAlgo()\n");
293  // Clear previous Geometry
294  clear();
295 
296  //work around for Mantis issue #3332
297  //if 3332 gets fixed in OCC, this will produce shifted views and will need
298  //to be reverted.
299  TopoDS_Shape inCopy;
300  if (!m_isPersp) {
301  gp_Pnt gCenter = ShapeUtils::findCentroid(input, viewAxis);
302  Base::Vector3d motion(-gCenter.X(), -gCenter.Y(), -gCenter.Z());
303  inCopy = ShapeUtils::moveShape(input, motion);
304  }
305  else {
306  BRepBuilderAPI_Copy BuilderCopy(input);
307  inCopy = BuilderCopy.Shape();
308  }
309 
310  Handle(HLRBRep_PolyAlgo) brep_hlrPoly;
311 
312  try {
313  // HLRBRep_PolyAlgo will fail if the whole input shape has not been meshed.
314  // meshing the faces is not sufficient.
315  BRepMesh_IncrementalMesh(inCopy, 0.10);
316 
317  brep_hlrPoly = new HLRBRep_PolyAlgo();
318  brep_hlrPoly->Load(inCopy);
319 
320  if (m_isPersp) {
321  double fLength = std::max(Precision::Confusion(), m_focus);
322  HLRAlgo_Projector projector(viewAxis, fLength);
323  brep_hlrPoly->Projector(projector);
324  }
325  else {// non perspective
326  HLRAlgo_Projector projector(viewAxis);
327  brep_hlrPoly->Projector(projector);
328  }
329  brep_hlrPoly->Update();
330  }
331  catch (const Standard_Failure& e) {
332  Base::Console().Error(
333  "GO::projectShapeWithPolygonAlgo - OCC error - %s - while projecting shape\n",
334  e.GetMessageString());
335  throw Base::RuntimeError("GeometryObject::projectShapeWithPolygonAlgo - OCC error");
336  }
337  catch (...) {
338  throw Base::RuntimeError("GeometryObject::projectShapeWithPolygonAlgo - unknown error");
339  }
340 
341  try {
342  HLRBRep_PolyHLRToShape polyhlrToShape;
343  polyhlrToShape.Update(brep_hlrPoly);
344 
345  visHard = polyhlrToShape.VCompound();
346  BRepLib::BuildCurves3d(visHard);
347  visHard =ShapeUtils::invertGeometry(visHard);
348  // BRepTools::Write(visHard, "GOvisHardi.brep"); //debug
349 
350  visSmooth = polyhlrToShape.Rg1LineVCompound();
351  BRepLib::BuildCurves3d(visSmooth);
352  visSmooth =ShapeUtils::invertGeometry(visSmooth);
353 
354  visSeam = polyhlrToShape.RgNLineVCompound();
355  BRepLib::BuildCurves3d(visSeam);
356  visSeam =ShapeUtils::invertGeometry(visSeam);
357 
358  visOutline = polyhlrToShape.OutLineVCompound();
359  BRepLib::BuildCurves3d(visOutline);
360  visOutline =ShapeUtils::invertGeometry(visOutline);
361 
362  hidHard = polyhlrToShape.HCompound();
363  BRepLib::BuildCurves3d(hidHard);
364  hidHard =ShapeUtils::invertGeometry(hidHard);
365  // BRepTools::Write(hidHard, "GOhidHardi.brep"); //debug
366 
367  hidSmooth = polyhlrToShape.Rg1LineHCompound();
368  BRepLib::BuildCurves3d(hidSmooth);
369  hidSmooth =ShapeUtils::invertGeometry(hidSmooth);
370 
371  hidSeam = polyhlrToShape.RgNLineHCompound();
372  BRepLib::BuildCurves3d(hidSeam);
373  hidSeam =ShapeUtils::invertGeometry(hidSeam);
374 
375  hidOutline = polyhlrToShape.OutLineHCompound();
376  BRepLib::BuildCurves3d(hidOutline);
377  hidOutline =ShapeUtils::invertGeometry(hidOutline);
378  }
379  catch (const Standard_Failure& e) {
380  Base::Console().Error(
381  "GO::projectShapeWithPolygonAlgo - OCC error - %s - while extracting edges\n",
382  e.GetMessageString());
383  throw Base::RuntimeError("GeometryObject::projectShapeWithPolygonAlgo - OCC error occurred "
384  "while extracting edges");
385  }
386  catch (...) {
387  throw Base::RuntimeError("GeometryObject::projectShapeWithPolygonAlgo - unknown error "
388  "occurred while extracting edges");
389  }
390 
391  makeTDGeometry();
392 }

512 void GeometryObject::addGeomFromCompound(TopoDS_Shape edgeCompound, edgeClass category,
513  bool hlrVisible)
514 {
515 // Base::Console().Message("GO::addGeomFromCompound(%d, %d)\n", category, hlrVisible);
516  if (edgeCompound.IsNull()) {
517  return; // There is no OpenCascade Geometry to be calculated
518  }
519 
520  // remove overlapping edges
521  TopoDS_Shape cleanShape;
522  if (m_scrubCount > 0) {
523  std::vector<TopoDS_Edge> edgeVector = DU::shapeToVector(edgeCompound);
524  for (int iPass = 0; iPass < m_scrubCount; iPass++) {
525  edgeVector = DrawProjectSplit::removeOverlapEdges(edgeVector);
526  }
527  bool invertResult = false;
528  cleanShape = DU::vectorToCompound(edgeVector, invertResult);
529 
530  } else {
531  cleanShape = edgeCompound;
532  }
533 
534  BaseGeomPtr base;
535  TopExp_Explorer edges(cleanShape, TopAbs_EDGE);
536  int i = 1;
537  for (; edges.More(); edges.Next(), i++) {
538  const TopoDS_Edge& edge = TopoDS::Edge(edges.Current());
539  if (edge.IsNull()) {
540  continue;
541  }
542  if (DU::isZeroEdge(edge)) {
543  continue;
544  }
545  if (DU::isCrazy(edge)) {
546  continue;
547  }
548 
549  base = BaseGeom::baseFactory(edge);
550  if (!base) {
551  continue;
552  // throw Base::ValueError("GeometryObject::addGeomFromCompound - baseFactory failed");
553  }
554 
555  base->source(0);//object geometry
556  base->sourceIndex(i - 1);
557  base->setClassOfEdge(category);
558  base->setHlrVisible(hlrVisible);
559  edgeGeom.push_back(base);
560 
561  //add vertices of new edge if not already in list
562  if (hlrVisible) {
563  BaseGeomPtr lastAdded = edgeGeom.back();
564  bool v1Add = true, v2Add = true;
565  bool c1Add = true;
566  TechDraw::VertexPtr v1 = std::make_shared<TechDraw::Vertex>(lastAdded->getStartPoint());
567  TechDraw::VertexPtr v2 = std::make_shared<TechDraw::Vertex>(lastAdded->getEndPoint());
568  TechDraw::CirclePtr circle = std::dynamic_pointer_cast<TechDraw::Circle>(lastAdded);
569  TechDraw::VertexPtr c1;
570  if (circle) {
571  c1 = std::make_shared<TechDraw::Vertex>(circle->center);
572  c1->isCenter(true);
573  c1->setHlrVisible(true);
574  }
575 
576  std::vector<VertexPtr>::iterator itVertex = vertexGeom.begin();
577  for (; itVertex != vertexGeom.end(); itVertex++) {
578  if ((*itVertex)->isEqual(*v1, Precision::Confusion())) {
579  v1Add = false;
580  }
581  if ((*itVertex)->isEqual(*v2, Precision::Confusion())) {
582  v2Add = false;
583  }
584  if (circle) {
585  if ((*itVertex)->isEqual(*c1, Precision::Confusion())) {
586  c1Add = false;
587  }
588  }
589  }
590  if (v1Add) {
591  vertexGeom.push_back(v1);
592  v1->setHlrVisible( true);
593  }
594  else {
595  // delete v1;
596  }
597  if (v2Add) {
598  vertexGeom.push_back(v2);
599  v2->setHlrVisible( true);
600  }
601  else {
602  // delete v2;
603  }
604 
605  if (circle) {
606  if (c1Add) {
607  vertexGeom.push_back(c1);
608  c1->setHlrVisible( true);
609  }
610  else {
611  // delete c1;
612  }
613  }
614  }
615  }//end TopExp

760 bool GeometryObject::isWithinArc(double theta, double first, double last, bool cw) const
761 {
762  if (fabs(last - first) >= 2 * M_PI) {
763  return true;
764  }
765 
766  // Put params within [0, 2*pi) - not totally sure this is necessary
767  theta = fmod(theta, 2 * M_PI);
768  if (theta < 0) {
769  theta += 2 * M_PI;
770  }
771 
772  first = fmod(first, 2 * M_PI);
773  if (first < 0) {
774  first += 2 * M_PI;
775  }
776 
777  last = fmod(last, 2 * M_PI);
778  if (last < 0) {
779  last += 2 * M_PI;
780  }
781 
782  if (cw) {
783  if (first > last) {
784  return theta <= first && theta >= last;
785  }
786  else {
787  return theta <= first || theta >= last;
788  }
789  }
790  else {
791  if (first > last) {
792  return theta >= first || theta <= last;
793  }
794  else {
795  return theta >= first && theta <= last;
796  }
797  }
798 }
