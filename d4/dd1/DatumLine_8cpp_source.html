44 Line::Line()
45 {
46  // These properties are only relevant for the visual appearance.
47  // Since they are getting changed from within its view provider
48  // their type is set to "Output" to avoid that they are marked as
49  // touched all the time.
50  ADD_PROPERTY_TYPE(ResizeMode,(static_cast<long>(0)), "Size", App::Prop_Output, "Automatic or manual resizing");
51  ResizeMode.setEnums(ResizeModeEnums);
52  ADD_PROPERTY_TYPE(Length,(20), "Size", App::Prop_Output, "Length of the line");
53  Length.setReadOnly(true);
54 
55  this->setAttacher(new AttachEngineLine);
56  // Create a shape, which will be used by the Sketcher. Them main function is to avoid a dependency of
57  // Sketcher on the PartDesign module
58  BRepBuilderAPI_MakeEdge builder(gp_Lin(gp_Pnt(0,0,0), gp_Dir(0,0,1)));
59  if (!builder.IsDone())
60  return;
61  TopoDS_Shape myShape = builder.Shape();
62  myShape.Infinite(Standard_True);
63  Shape.setValue(myShape);
64 
65  AttachmentSupport.touch();
66 }
