80  void updateDataAndDrawToPosition(Base::Vector2d onSketchPos) override
81  {
82  switch (state()) {
83  case SelectMode::SeekFirst: {
84  toolWidgetManager.drawPositionAtCursor(onSketchPos);
85  if (constructionMethod() == ConstructionMethod::Center) {
86  centerPoint = onSketchPos;
87 
88  seekAndRenderAutoConstraint(sugConstraints[0], onSketchPos, Base::Vector2d());
89  }
90  else {
91  firstPoint = onSketchPos;
92 
93  seekAndRenderAutoConstraint(sugConstraints[0],
94  onSketchPos,
95  Base::Vector2d(),
96  AutoConstraint::CURVE);
97  }
98  } break;
99  case SelectMode::SeekSecond: {
100  if (constructionMethod() == ConstructionMethod::ThreeRim) {
101  centerPoint = (onSketchPos - firstPoint) / 2 + firstPoint;
102  secondPoint = onSketchPos;
103  }
104 
105  radius = (onSketchPos - centerPoint).Length();
106 
107  CreateAndDrawShapeGeometry();
108 
109  if (constructionMethod() == ConstructionMethod::Center) {
110  toolWidgetManager.drawDoubleAtCursor(onSketchPos, radius);
111  }
112  else {
113  toolWidgetManager.drawPositionAtCursor(onSketchPos);
114  }
115 
116  seekAndRenderAutoConstraint(sugConstraints[1],
117  onSketchPos,
118  constructionMethod() == ConstructionMethod::Center
119  ? onSketchPos - centerPoint
120  : Base::Vector2d(),
121  AutoConstraint::CURVE);
122  } break;
123  case SelectMode::SeekThird: {
124  try {
125  if (areCollinear(firstPoint, secondPoint, onSketchPos)) {
126  // If points are collinear then we can't calculate the center.
127  return;
128  }
129 
130  centerPoint =
131  Part::Geom2dCircle::getCircleCenter(firstPoint, secondPoint, onSketchPos);
132 
133  radius = (onSketchPos - centerPoint).Length();
134 
135  toolWidgetManager.drawPositionAtCursor(onSketchPos);
136 
137  CreateAndDrawShapeGeometry();
138 
139  seekAndRenderAutoConstraint(sugConstraints[2],
140  onSketchPos,
141  Base::Vector2d(0.f, 0.f),
142  AutoConstraint::CURVE);
143  }
144  catch (Base::ValueError& e) {
145  e.ReportException();
146  }
147  } break;
148  default:
149  break;
150  }

417 template<>
418 void DSHCircleControllerBase::doEnforceControlParameters(Base::Vector2d& onSketchPos)
419 {
420  switch (handler->state()) {
421  case SelectMode::SeekFirst: {
422  if (onViewParameters[OnViewParameter::First]->isSet) {
423  onSketchPos.x = onViewParameters[OnViewParameter::First]->getValue();
424  }
425 
426  if (onViewParameters[OnViewParameter::Second]->isSet) {
427  onSketchPos.y = onViewParameters[OnViewParameter::Second]->getValue();
428  }
429  } break;
430  case SelectMode::SeekSecond: {
431  if (handler->constructionMethod()
432  == DrawSketchHandlerCircle::ConstructionMethod::Center) {
433  if (onViewParameters[OnViewParameter::Third]->isSet) {
434  double radius = (handler->isDiameter ? 0.5 : 1)
435  * onViewParameters[OnViewParameter::Third]->getValue();
436  if (radius < Precision::Confusion()) {
437  unsetOnViewParameter(onViewParameters[OnViewParameter::Third].get());
438  return;
439  }
440 
441  auto dir = (onSketchPos - handler->centerPoint);
442 
443  if (dir.Length() < Precision::Confusion()) {
444  dir.x = 1.0; // if direction null, default to (1,0)
445  }
446 
447  onSketchPos = handler->centerPoint + radius * dir.Normalize();
448  }
449  }
450  else {
451  if (onViewParameters[OnViewParameter::Third]->isSet) {
452  onSketchPos.x = onViewParameters[OnViewParameter::Third]->getValue();
453  }
454 
455  if (onViewParameters[OnViewParameter::Fourth]->isSet) {
456  onSketchPos.y = onViewParameters[OnViewParameter::Fourth]->getValue();
457  }
458 
459  if (onViewParameters[OnViewParameter::Third]->isSet
460  && onViewParameters[OnViewParameter::Fourth]->isSet
461  && (onSketchPos - handler->firstPoint).Length() < Precision::Confusion()) {
462  unsetOnViewParameter(onViewParameters[OnViewParameter::Third].get());
463  unsetOnViewParameter(onViewParameters[OnViewParameter::Fourth].get());
464  }
465  }
466  } break;
467  case SelectMode::SeekThird: { // 3 rims only
468  if (onViewParameters[OnViewParameter::Fifth]->isSet) {
469  onSketchPos.x = onViewParameters[OnViewParameter::Fifth]->getValue();
470  }
471 
472  if (onViewParameters[OnViewParameter::Sixth]->isSet) {
473  onSketchPos.y = onViewParameters[OnViewParameter::Sixth]->getValue();
474  }
475  if (onViewParameters[OnViewParameter::Fifth]->isSet
476  && onViewParameters[OnViewParameter::Sixth]->isSet
477  && areCollinear(handler->firstPoint, handler->secondPoint, onSketchPos)) {
478  unsetOnViewParameter(onViewParameters[OnViewParameter::Fifth].get());
479  unsetOnViewParameter(onViewParameters[OnViewParameter::Sixth].get());
480  }
481  } break;
482  default:
483  break;
484  }

489 {
490  switch (handler->state()) {
491  case SelectMode::SeekFirst: {
492  if (!onViewParameters[OnViewParameter::First]->isSet) {
493  setOnViewParameterValue(OnViewParameter::First, onSketchPos.x);
494  }
495 
496  if (!onViewParameters[OnViewParameter::Second]->isSet) {
497  setOnViewParameterValue(OnViewParameter::Second, onSketchPos.y);
498  }
499 
500  bool sameSign = onSketchPos.x * onSketchPos.y > 0.;
501  onViewParameters[OnViewParameter::First]->setLabelAutoDistanceReverse(!sameSign);
502  onViewParameters[OnViewParameter::Second]->setLabelAutoDistanceReverse(sameSign);
503  onViewParameters[OnViewParameter::First]->setPoints(Base::Vector3d(),
504  toVector3d(onSketchPos));
505  onViewParameters[OnViewParameter::Second]->setPoints(Base::Vector3d(),
506  toVector3d(onSketchPos));
507  } break;
508  case SelectMode::SeekSecond: {
509  if (handler->constructionMethod()
510  == DrawSketchHandlerCircle::ConstructionMethod::Center) {
511  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
512  "User parameter:BaseApp/Preferences/Mod/Sketcher/dimensioning");
513  bool dimDiameter = hGrp->GetBool("DimensioningDiameter", true);
514  bool dimRadius = hGrp->GetBool("DimensioningRadius", true);
515  bool useRadius = dimRadius && !dimDiameter;
516 
517  if (!onViewParameters[OnViewParameter::Third]->isSet) {
518  double val = handler->radius * (useRadius ? 1 : 2);
519  setOnViewParameterValue(OnViewParameter::Third, val);
520  }
521 
522  Base::Vector3d start = toVector3d(handler->centerPoint);
523  Base::Vector3d end = toVector3d(onSketchPos);

525  start = toVector3d(handler->centerPoint - (onSketchPos - handler->centerPoint));
526  }
527 
528  onViewParameters[OnViewParameter::Third]->setPoints(start, end);
529  }
530  else {
531  if (!onViewParameters[OnViewParameter::Third]->isSet) {
532  setOnViewParameterValue(OnViewParameter::Third, onSketchPos.x);
533  }
534 
535  if (!onViewParameters[OnViewParameter::Fourth]->isSet) {

538

541  onViewParameters[OnViewParameter::Fourth]->setLabelAutoDistanceReverse(sameSign);
542  onViewParameters[OnViewParameter::Third]->setPoints(Base::Vector3d(),

608 template<>
609 void DSHCircleController::addConstraints()
610 {
611  if (handler->constructionMethod() == DrawSketchHandlerCircle::ConstructionMethod::Center) {
612  int firstCurve = handler->getHighestCurveIndex();
613 
614  auto x0 = onViewParameters[OnViewParameter::First]->getValue();
615  auto y0 = onViewParameters[OnViewParameter::Second]->getValue();
616 
617  auto x0set = onViewParameters[OnViewParameter::First]->isSet;
618  auto y0set = onViewParameters[OnViewParameter::Second]->isSet;
619  auto radiusSet = onViewParameters[OnViewParameter::Third]->isSet;
620 
621  using namespace Sketcher;
622 
623  auto constraintx0 = [&]() {
624  ConstraintToAttachment(GeoElementId(firstCurve, PointPos::mid),
625  GeoElementId::VAxis,
626  x0,
627  handler->sketchgui->getObject());
628  };
629 
630  auto constrainty0 = [&]() {
631  ConstraintToAttachment(GeoElementId(firstCurve, PointPos::mid),
632  GeoElementId::HAxis,
633  y0,
634  handler->sketchgui->getObject());
635  };
636 
637  auto constraintradius = [&]() {
638  if (handler->isDiameter) {
639  Gui::cmdAppObjectArgs(handler->sketchgui->getObject(),
640  "addConstraint(Sketcher.Constraint('Diameter',%d,%f)) ",
641  firstCurve,
642  handler->radius * 2);
643  }
644  else {
645  Gui::cmdAppObjectArgs(handler->sketchgui->getObject(),
646  "addConstraint(Sketcher.Constraint('Radius',%d,%f)) ",
647  firstCurve,
648  handler->radius);
649  }
650  };
651 
652  // NOTE: if AutoConstraints is empty, we can add constraints directly without any diagnose.
653  // No diagnose was run.
654  if (handler->AutoConstraints.empty()) {
655  if (x0set) {
656  constraintx0();
657  }
658 
659  if (y0set) {
660  constrainty0();
661  }
662 
663  if (radiusSet) {
664  constraintradius();
665  }
666  }
667  else { // There is a valid diagnose.
668  auto startpointinfo = handler->getPointInfo(GeoElementId(firstCurve, PointPos::mid));
669 
670  // if Autoconstraints is empty we do not have a diagnosed system and the parameter will
671  // always be set
672  if (x0set && startpointinfo.isXDoF()) {
673  constraintx0();
674 
675  handler->diagnoseWithAutoConstraints(); // ensure we have recalculated parameters
676  // after each constraint addition
677 
678  startpointinfo = handler->getPointInfo(
679  GeoElementId(firstCurve, PointPos::mid)); // get updated point position
680  }
681 
682  // if Autoconstraints is empty we do not have a diagnosed system and the parameter will
683  // always be set
684  if (y0set && startpointinfo.isYDoF()) {
685  constrainty0();
686 
687  handler->diagnoseWithAutoConstraints(); // ensure we have recalculated parameters
688  // after each constraint addition
689  }
690 
691  auto edgeinfo = handler->getEdgeInfo(firstCurve);
