99 void QGIHighlight::makeReference()
100 {
101  prepareGeometryChange();
102  int fontSize = QGIView::exactFontSize(m_refFont.family().toStdString(),
103  m_refSize);
104  m_refFont .setPixelSize(fontSize);
105  m_reference->setFont(m_refFont);
106  m_reference->setPlainText(m_refText);
107 
108  double vertOffset = 0.0;
109  if (m_referenceAngle >= 0.0 &&
110  m_referenceAngle <= 180.0) {
111  //above the X axis
112  //referenceText is positioned by top-left, need to adjust upwards by text height
113  vertOffset = m_reference->boundingRect().height();
114  } else {
115  //below X axis. need to adjust upwards a bit because there is blank space above text
116  vertOffset = m_reference->tightBoundingAdjust().y();
117  }
118 
119  double horizOffset = 0.0;
120  if (m_referenceAngle > 90.0 &&
121  m_referenceAngle < 270.0) {
122  //to left of Y axis
123  horizOffset = -m_reference->boundingRect().width();
124  }
125  QRectF r(m_start, m_end);
126  double radius = r.width() / 2.0;
127  QPointF center = r.center();
128  double angleRad = m_referenceAngle * M_PI / 180.0;
129  double posX = center.x() + cos(angleRad) * radius + horizOffset;
130  double posY = center.y() - sin(angleRad) * radius - vertOffset;
131  m_reference->setPos(posX, posY);
132 
133  double highRot = rotation();
134  if (!TechDraw::DrawUtil::fpCompare(highRot, 0.0)) {
135  QRectF refBR = m_reference->boundingRect();
136  QPointF refCenter = refBR.center();
137  m_reference->setTransformOriginPoint(refCenter);
138  m_reference->setRotation(-highRot);
139  }
140 }
