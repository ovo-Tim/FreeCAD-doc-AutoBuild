141 void PropertyConstraintList::setValue(const Constraint* lValue)
142 {
143  if (lValue) {
144  aboutToSetValue();
145  Constraint* newVal = lValue->clone();
146  std::set<App::ObjectIdentifier> removed;
147  std::map<App::ObjectIdentifier, App::ObjectIdentifier> renamed;
148  int start = 0;
149 
150  /* Determine if it is a rename or not * */
151  if (!_lValueList.empty() && lValue->tag == _lValueList[0]->tag) {
152  renamed[makePath(0, _lValueList[0])] = makePath(0, lValue);
153  start = 1;
154  }
155 
156  /* Signal rename changes */
157  if (!renamed.empty()) {
158  signalConstraintsRenamed(renamed);
159  }
160 
161  /* Collect info about removals */
162  for (unsigned int i = start; i < _lValueList.size(); i++) {
163  valueMap.erase(_lValueList[i]->tag);
164  removed.insert(makePath(i, _lValueList[i]));

169  signalConstraintsRemoved(removed);
170  }
171

174  delete _lValueList[i];

176 
177  /* Set new data */
178  _lValueList.resize(1);

202 void PropertyConstraintList::applyValues(std::vector<Constraint*>&& lValue)
203 {
204  std::set<Constraint*> oldVals(_lValueList.begin(), _lValueList.end());
205  std::map<App::ObjectIdentifier, App::ObjectIdentifier> renamed;
206  std::set<App::ObjectIdentifier> removed;
207  boost::unordered_map<boost::uuids::uuid, std::size_t> newValueMap;
208 
209  /* Check for renames */
210  for (unsigned int i = 0; i < lValue.size(); i++) {
211  boost::unordered_map<boost::uuids::uuid, std::size_t>::const_iterator j =
212  valueMap.find(lValue[i]->tag);
213 
214  if (j != valueMap.end()) {
215  if (i != j->second || _lValueList[j->second]->Name != lValue[i]->Name) {
216  App::ObjectIdentifier old_oid(makePath(j->second, _lValueList[j->second]));
217  App::ObjectIdentifier new_oid(makePath(i, lValue[i]));
218  renamed[old_oid] = new_oid;
219  }
220  valueMap.erase(j);
221  }
222 
223  newValueMap[lValue[i]->tag] = i;
224 
225  // safety insurance in case new new values contain some pointers of the old values
226  oldVals.erase(lValue[i]);
227  }
228 
229  /* Collect info about removed elements */
230  for (auto& v : valueMap) {
231  removed.insert(makePath(v.second, _lValueList[v.second]));
232  }
233 
234  /* Update value map with new tags from new array */
235  valueMap = std::move(newValueMap);
236 
237  /* Signal removes first, in case renamed values below have the same names as some of the removed
238  * ones. */
239  if (!removed.empty() && !restoreFromTransaction) {
240  signalConstraintsRemoved(removed);
241  }
242 
243  /* Signal renames */
244  if (!renamed.empty() && !restoreFromTransaction) {
245  signalConstraintsRenamed(renamed);
246  }
247 
248  _lValueList = std::move(lValue);
249 
250  /* Clean-up; remove old values */
251  for (auto& v : oldVals) {
252  delete v;
253  }
254 }

424 bool PropertyConstraintList::checkConstraintIndices(int geomax, int geomin)
425 {
426  int mininternalgeoid = std::numeric_limits<int>::max();
427  int maxinternalgeoid = GeoEnum::GeoUndef;
428 
429  auto cmin = [](int previousmin, int cindex) {
430  if (cindex == GeoEnum::GeoUndef) {
431  return previousmin;
432  }
433 
434  return (cindex < previousmin) ? cindex : previousmin;
435  };
436 
437  auto cmax = [](int previousmax, int cindex) {
438  return (cindex > previousmax) ? cindex : previousmax;
439  };
440 
441  for (const auto& v : _lValueList) {
442 
443  mininternalgeoid = cmin(mininternalgeoid, v->First);
444  mininternalgeoid = cmin(mininternalgeoid, v->Second);
445  mininternalgeoid = cmin(mininternalgeoid, v->Third);
446 
447  maxinternalgeoid = cmax(maxinternalgeoid, v->First);
448  maxinternalgeoid = cmax(maxinternalgeoid, v->Second);
449  maxinternalgeoid = cmax(maxinternalgeoid, v->Third);
450  }
451 
452  if (maxinternalgeoid > geomax || mininternalgeoid < geomin) {
453  invalidIndices = true;
454  }
455  else {
456  invalidIndices = false;
457  }
458 
459  return invalidIndices;
460 }

518 void PropertyConstraintList::setPathValue(const ObjectIdentifier& path, const boost::any& value)
519 {
520  if (path.numSubComponents() != 2 || path.getPropertyComponent(0).getName() != getName()) {
521  FC_THROWM(Base::ValueError, "invalid constraint path " << path.toString());
522  }
523 
524  const ObjectIdentifier::Component& c1 = path.getPropertyComponent(1);
525  double dvalue;
526 
527  if (value.type() == typeid(double)) {
528  dvalue = boost::any_cast<double>(value);
529  }
530  else if (value.type() == typeid(float)) {
531  dvalue = App::any_cast<float>(value);
532  }
533  else if (value.type() == typeid(long)) {
534  dvalue = App::any_cast<long>(value);
535  }
536  else if (value.type() == typeid(int)) {
537  dvalue = App::any_cast<int>(value);
538  }
539  else if (value.type() == typeid(Quantity)) {
540  dvalue = (App::any_cast<const Quantity&>(value)).getValue();
541  }
542  else {
543  throw std::bad_cast();
544  }
545 
546  if (c1.isArray()) {
547  size_t index = c1.getIndex(_lValueList.size());
548  switch (_lValueList[index]->Type) {
549  case Angle:
550  dvalue = Base::toRadians<double>(dvalue);
551  break;
552  default:
553  break;
554  }
555  aboutToSetValue();
556  _lValueList[index]->setValue(dvalue);
557  hasSetValue();
558  return;
559  }
560  else if (c1.isSimple()) {
561  for (std::vector<Constraint*>::const_iterator it = _lValueList.begin();
562  it != _lValueList.end();
563  ++it) {
564  int index = it - _lValueList.begin();
565 
566  if ((*it)->Name == c1.getName()) {
567  switch (_lValueList[index]->Type) {
568  case Angle:
569  dvalue = Base::toRadians<double>(dvalue);
570  break;
571  default:
572  break;
573  }
574  aboutToSetValue();
575  _lValueList[index]->setValue(dvalue);
576  hasSetValue();
577  return;
578  }
579  }
580  }
581  FC_THROWM(Base::ValueError, "invalid constraint path " << path.toString());
582 }
