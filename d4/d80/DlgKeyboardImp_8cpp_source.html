182 void DlgCustomKeyboardImp::populateCommandList(QTreeWidget* commandTreeWidget,
183  QTreeWidgetItem* separatorItem,
184  QComboBox* combo)
185 {
186  QByteArray current;
187  if (auto item = commandTreeWidget->currentItem()) {
188  current = item->data(1, Qt::UserRole).toByteArray();
189  }
190 
191  if (separatorItem) {
192  commandTreeWidget->takeTopLevelItem(commandTreeWidget->indexOfTopLevelItem(separatorItem));
193  }
194  commandTreeWidget->clear();
195  if (separatorItem) {
196  commandTreeWidget->addTopLevelItem(separatorItem);
197  }
198 
199  CommandManager& cCmdMgr = Application::Instance->commandManager();
200  auto group = combo->itemData(combo->currentIndex(), Qt::UserRole).toByteArray();
201  auto cmds =
202  group == "All" ? cCmdMgr.getAllCommands() : cCmdMgr.getGroupCommands(group.constData());
203  QTreeWidgetItem* currentItem = nullptr;
204  for (const Command* cmd : cmds) {
205  QTreeWidgetItem* item = new QTreeWidgetItem(commandTreeWidget);
206  item->setText(1, Action::commandMenuText(cmd));
207  item->setToolTip(1, Action::commandToolTip(cmd));
208  item->setData(1, Qt::UserRole, QByteArray(cmd->getName()));
209  item->setSizeHint(0, QSize(32, 32));
210  if (auto pixmap = cmd->getPixmap()) {
211  item->setIcon(0, BitmapFactory().iconFromTheme(pixmap));
212  }
213  item->setText(2, cmd->getShortcut());
214  if (auto accel = cmd->getAccel()) {
215  item->setText(3, QKeySequence(QString::fromLatin1(accel)).toString());
216  }
217 
218  if (current == cmd->getName()) {
219  currentItem = item;
220  }
221  }
222  if (currentItem) {
223  commandTreeWidget->setCurrentItem(currentItem);
224  }
225  commandTreeWidget->resizeColumnToContents(2);
226  commandTreeWidget->resizeColumnToContents(3);

270 void DlgCustomKeyboardImp::initPriorityList(QTreeWidget* priorityList,
271  QAbstractButton* buttonUp,
272  QAbstractButton* buttonDown)
273 {
274  QStringList labels;
275  labels << tr("Name") << tr("Title");
276  priorityList->setHeaderLabels(labels);
277  priorityList->header()->hide();
278  priorityList->header()->setSectionResizeMode(0, QHeaderView::ResizeToContents);
279  priorityList->header()->setSectionResizeMode(1, QHeaderView::ResizeToContents);
280 
281  auto updatePriorityList = [priorityList](bool up) {
282  auto item = priorityList->currentItem();
283  if (!item) {
284  return;
285  }
286 
287  int index = priorityList->indexOfTopLevelItem(item);
288  if (index < 0) {
289  return;
290  }
291  if ((index == 0 && up) || (index == priorityList->topLevelItemCount() - 1 && !up)) {
292  return;
293  }
294 
295  std::vector<QByteArray> actions;
296  for (int i = 0; i < priorityList->topLevelItemCount(); ++i) {
297  auto item = priorityList->topLevelItem(i);
298  actions.push_back(item->data(0, Qt::UserRole).toByteArray());
299  }
300 
301  auto it = actions.begin() + index;
302  auto itNext = up ? it - 1 : it + 1;
303  std::swap(*it, *itNext);
304  ShortcutManager::instance()->setPriorities(actions);
305  };
306 
307  QObject::connect(buttonUp, &QAbstractButton::clicked, [=]() {
308  updatePriorityList(true);
309  });
310  QObject::connect(buttonDown, &QAbstractButton::clicked, [=]() {
311  updatePriorityList(false);
312  });
313  QObject::connect(priorityList, &QTreeWidget::currentItemChanged, [=](QTreeWidgetItem* item) {
314  buttonUp->setEnabled(item != nullptr);
315  buttonDown->setEnabled(item != nullptr);
316  });
317 }

360 void DlgCustomKeyboardImp::populatePriorityList(QTreeWidget* priorityList,
361  Gui::AccelLineEdit* editor,
362  Gui::AccelLineEdit* curShortcut)
363 {
364  QByteArray current;
365  if (auto currentItem = priorityList->currentItem()) {
366  current = currentItem->data(0, Qt::UserRole).toByteArray();
367  }
368 
369  priorityList->clear();
370  QString sc;
371  if (!editor->isNone() && editor->text().size()) {
372  sc = editor->text();
373  }
374  else if (curShortcut && !curShortcut->isNone()) {
375  sc = curShortcut->text();
376  }
377 
378  auto actionList = ShortcutManager::instance()->getActionsByShortcut(sc);
379  QTreeWidgetItem* currentItem = nullptr;
380  for (const auto& info : actionList) {
381  if (!info.second) {
382  continue;
383  }
384  QTreeWidgetItem* item = new QTreeWidgetItem(priorityList);
385  item->setText(0, QString::fromUtf8(info.first));
386  item->setText(1, Action::cleanTitle(info.second->text()));
387  item->setToolTip(0, info.second->toolTip());
388  item->setIcon(0, info.second->icon());
389  item->setData(0, Qt::UserRole, info.first);
390  if (current == info.first) {
391  currentItem = item;
392  }
393  }
394  priorityList->resizeColumnToContents(0);
395  priorityList->resizeColumnToContents(1);
396  if (currentItem) {
397  priorityList->setCurrentItem(currentItem);
398  }

401 void DlgCustomKeyboardImp::populateCommandGroups(QComboBox* combo)
402 {
403  CommandManager& cCmdMgr = Application::Instance->commandManager();
404  std::map<std::string, Command*> sCommands = cCmdMgr.getCommands();
405 
406  GroupMap groupMap;
407  groupMap.push_back(std::make_pair(QLatin1String("File"), QString()));
408  groupMap.push_back(std::make_pair(QLatin1String("Edit"), QString()));
409  groupMap.push_back(std::make_pair(QLatin1String("View"), QString()));
410  groupMap.push_back(std::make_pair(QLatin1String("Standard-View"), QString()));
411  groupMap.push_back(std::make_pair(QLatin1String("Tools"), QString()));
412  groupMap.push_back(std::make_pair(QLatin1String("Window"), QString()));
413  groupMap.push_back(std::make_pair(QLatin1String("Help"), QString()));
414  groupMap.push_back(
415  std::make_pair(QLatin1String("Macros"), qApp->translate("Gui::MacroCommand", "Macros")));
416 
417  for (const auto& sCommand : sCommands) {
418  QLatin1String group(sCommand.second->getGroupName());
419  QString text = sCommand.second->translatedGroupName();
420  GroupMap::iterator jt;
421  jt = std::find_if(groupMap.begin(), groupMap.end(), GroupMap_find(group));
422  if (jt != groupMap.end()) {
423  if (jt->second.isEmpty()) {
424  jt->second = text;
425  }
426  }
427  else {
428  groupMap.push_back(std::make_pair(group, text));
429  }
430  }
431  groupMap.push_back(std::make_pair(QLatin1String("All"), tr("All")));
432 
433  for (const auto& it : groupMap) {
434  if (combo->findData(it.first) < 0) {
435  combo->addItem(it.second);
436  combo->setItemData(combo->count() - 1, QVariant(it.first), Qt::UserRole);
437  }
438  }
439 }
