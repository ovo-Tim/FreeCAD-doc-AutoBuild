44 class PerfTestCase(unittest.TestCase):
45  """
46  Special Test Case that takes a list of filenames after the "--pass" parameter to FreeCAD, and
47  runs a performance test by opening them, starting instrumentation, calling recompute(), and
48  then saving results.
49 
50  Intended to be run as "<perf profiling> FreeCAD -t TestPerf --pass <modelname>
51 
52  External perf profiling requires Python 3.12 or better, and a linux platform.
53  cProfile profiling and guppy memory information can run anywhere.
54  """
55 
56  def setUp(self):
57  if "--pass" in sys.argv:
58  self.fileList = sys.argv[sys.argv.index("--pass") + 1 :]
59  else:
60  raise FileNotFoundError("Must provide filename parameter(s) via --pass")
61  if "--save" in sys.argv:
62  self.saveModels = True
63  self.fileList = sys.argv[sys.argv.index("--save") + 1 :]
64  else:
65  self.saveModels = False
66  if Part.Shape().ElementMapVersion == "":
67  self.tnp = ""
68  else:
69  self.tnp = ".tnp"
70  if Memtest:
71  # Use filename of first model with ".mprofile" appended for python memory use info.
72  self.memfile = open(self.fileList[0] + self.tnp + ".mprofile", "w", encoding="utf-8")
73 
74  def testAll(self):
75  if Pyprofile:
76  # Generate a cProfile file as a python only time profile.
77  profile = cProfile.Profile()
78  profile.enable()
79  try:
80  # This is Python 3.12 on supported platforms ( linux ) only so that if we are run under
81  # an external 'perf' command, we report the python data. This can be extremely useful,
82  # because it contains not only time consumed, but python and c++ calls that took place
83  # so deep analysis can be performed on the resulting file. See calling script in
84  # tools/profile/perftest.sh for a wrapper.
85  sys.activate_stack_trampoline("perf")
86  except AttributeError:
87  pass # Totally okay if we don't have that, we can use the cProfile if it's there.
88 
89  # Walk all files after the --pass. Normally one to avoid result intermingling.
90  for fileName in self.fileList:
91  doc = App.openDocument(fileName)
92  doc.recompute() # The heart of the performance measurement.
93  if Memtest:
94  # If guppy is available, take a heap snapshot and save it. Note that if multiple
95  # files are provided then their heap data sets will be appended to the same file.
96  dumpdata = hpy().heap()
97  dumpdata.stat.dump(self.memfile)
98  self.memfile.flush()
99  if self.saveModels:
100  filenametnp = App.ActiveDocument.Name + self.tnp
101  App.ActiveDocument.saveAs(filenametnp)
102  App.closeDocument(doc.Name)
103 
104  try:
105  sys.deactivate_stack_trampoline()
106  except AttributeError:
107  pass
108  if Pyprofile:
109  profile.disable()
110  # Use filename of first model with ".cprofile" appended for python profiling information.
111  profile.dump_stats(self.fileList[0] + self.tnp + ".cprofile")
112  if Memtest:
113  self.memfile.close()
