76 def compare(file1, file2):
77  with open(file1) as f1:
78  f1_text = f1.readlines()
79  with open(file2) as f2:
80  f2_text = f2.readlines()
81  res = [
82  l
83  for l in difflib.unified_diff(
84  f1_text, f2_text, fromfile=file1, tofile=file2, lineterm=""
85  )
86  ]
87  res = [l for l in res if l.startswith("+") or l.startswith("-")]
88  res = [l for l in res if not l.startswith("+++") and not l.startswith("---")]
89  return res
