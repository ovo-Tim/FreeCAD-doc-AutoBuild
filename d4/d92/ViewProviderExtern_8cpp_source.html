58 void ViewProviderExtern::setModeByFile(const char* name, const char* ivFileName)
59 {
60  SoInput in;
61  Base::ifstream file(Base::FileInfo(ivFileName), std::ios::in | std::ios::binary);
62  if (file){
63  std::streamoff size = 0;
64  std::streambuf* buf = file.rdbuf();
65  if (buf) {
66  std::streamoff curr;
67  curr = buf->pubseekoff(0, std::ios::cur, std::ios::in);
68  size = buf->pubseekoff(0, std::ios::end, std::ios::in);
69  buf->pubseekoff(curr, std::ios::beg, std::ios::in);
70  }
71 
72  // read in the file
73  std::vector<unsigned char> content;
74  content.reserve(size);
75  unsigned char c;
76  while (file.get((char&)c)) {
77  content.push_back(c);
78  }
79 
80  file.close();
81  in.setBuffer(&(content[0]),content.size());
82  setModeBySoInput(name,in);
83  }
84 }
