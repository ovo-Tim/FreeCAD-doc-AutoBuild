119 void CosmeticEdgePy::setFormat(Py::Dict arg)
120 {
121  Py::Tuple dummy;
122  Py::TupleN color(Py::Float(0.0), Py::Float(0.0), Py::Float(0.0), Py::Float(0.0));
123  int style = 1;
124  double weight = 0.5;
125  PyObject* pColor = color.ptr();
126  PyObject* visible = Py_True;
127  static const std::array<const char *, 5> kw{"style", "weight", "color", "visible", nullptr};
128  if (!Base::Wrapped_ParseTupleAndKeywords(dummy.ptr(), arg.ptr(), "|idO!O!", kw,
129  &style, &weight, &PyTuple_Type, &pColor, &PyBool_Type, &visible)) {
130  throw Py::ValueError("Expected {'style':int, 'weight':float, 'color':tuple, 'visible':bool} dict");
131  }
132 
133  TechDraw::LineFormat* format = &(this->getCosmeticEdgePtr()->m_format);
134  format->setStyle(style);
135  format->setWidth(weight);
136  format->setColor(DrawUtil::pyTupleToColor(pColor));
137  format->setVisible(Base::asBoolean(visible));
138 }
