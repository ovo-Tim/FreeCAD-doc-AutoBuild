204 {
205  if (dir == LINEAR) { // straight span
206  return 0;
207  }
208  double cosa = 1 - tolerance / radius;
209  if (cosa > NEARLY_ONE) {
210  cosa = NEARLY_ONE;
211  }
212  cosa = 2 * cosa * cosa - 1; /* double angle */
213  double sina = sqrt(1 - cosa * cosa) * dir;
214  double tempang = atan2(sina, cosa);
215  int num_vectors = (int)(fabs(angle / tempang)) + 1;
216  return num_vectors;
