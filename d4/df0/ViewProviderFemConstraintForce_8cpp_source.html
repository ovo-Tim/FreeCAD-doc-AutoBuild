79 void ViewProviderFemConstraintForce::transformSymbol(const Base::Vector3d& point,
80  const Base::Vector3d& normal,
81  SbMatrix& mat) const
82 {
83  auto obj = this->getObject<const Fem::ConstraintForce>();
84  bool rev = obj->Reversed.getValue();
85  float s = obj->getScaleFactor();
86  // Symbol length from .iv file
87  float symLen = 4.0f;
88  // Place each symbol outside the boundary
89  Base::Vector3d dir = (rev ? -1.0 : 1.0) * obj->DirectionVector.getValue();
90  float symTraY = dir.Dot(normal) < 0 ? -1 * symLen : 0.0f;
91  float rotAngle = rev ? F_PI : 0.0f;
92  SbMatrix mat0, mat1;
93  mat0.setTransform(SbVec3f(0, symTraY, 0),
94  SbRotation(SbVec3f(0, 0, 1), rotAngle),
95  SbVec3f(1, 1, 1),
96  SbRotation(SbVec3f(0, 0, 1), 0),
97  SbVec3f(0, symLen / 2.0f, 0));
98 
99  mat1.setTransform(SbVec3f(point.x, point.y, point.z),
100  SbRotation(SbVec3f(0, 1, 0), SbVec3f(dir.x, dir.y, dir.z)),
101  SbVec3f(s, s, s));
102 
103  mat = mat0 * mat1;
104 }
