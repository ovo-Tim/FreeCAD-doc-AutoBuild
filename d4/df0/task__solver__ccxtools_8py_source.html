183  def printCalculiXstdout(self):
184 
185  out = self.Calculix.readAllStandardOutput()
186  # print(type(out))
187  # <class 'PySide2.QtCore.QByteArray'>
188 
189  if out.isEmpty():
190  self.femConsoleMessage("CalculiX stdout is empty", "Error")
191  return False
192 
193  # https://forum.freecad.org/viewtopic.php?f=18&t=39195
194  # convert QByteArray to a binary string an decode it to "utf-8"
195  out = out.data().decode() # "utf-8" can be omitted
196  out = os.linesep.join([s for s in out.splitlines() if s])
197  out = out.replace("\n", "<br>")
198  # print(out)
199  self.femConsoleMessage(out)
200 
201  if "*ERROR in e_c3d: nonpositive jacobian" in out:
202  error_message = (
203  "\n\nCalculiX returned an error due to "
204  "nonpositive jacobian determinant in at least one element\n"
205  "Use the run button on selected solver to get a better error output.\n"
206  )
207  FreeCAD.Console.PrintError(error_message)
208 
209  if "*ERROR" in out:
210  return False
211  else:
212  return True
