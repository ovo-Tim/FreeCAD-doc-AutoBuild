304 template<class Precision>
305 inline BoundBox3<Precision>::BoundBox3(const Vector3<Precision>* pclVect, std::size_t ulCt)
306  : MinX(std::numeric_limits<Precision>::max())
307  , MinY(std::numeric_limits<Precision>::max())
308  , MinZ(std::numeric_limits<Precision>::max())
309  , MaxX(-std::numeric_limits<Precision>::max())
310  , MaxY(-std::numeric_limits<Precision>::max())
311  , MaxZ(-std::numeric_limits<Precision>::max())
312 {
313  const Vector3<Precision>* pI = nullptr;
314  const Vector3<Precision>* pEnd = pclVect + ulCt;
315  for (pI = pclVect; pI < pEnd; ++pI) {
316  MinX = std::min<Precision>(MinX, pI->x);
317  MinY = std::min<Precision>(MinY, pI->y);
318  MinZ = std::min<Precision>(MinZ, pI->z);
319  MaxX = std::max<Precision>(MaxX, pI->x);
320  MaxY = std::max<Precision>(MaxY, pI->y);
321  MaxZ = std::max<Precision>(MaxZ, pI->z);
322  }
323 }

478 template<class Precision>
479 inline bool BoundBox3<Precision>::GetOctantFromVector(const Vector3<Precision>& rclVct,
480  OCTANT& rclOctant) const
481 {
482  if (!IsInBox(rclVct)) {
483  return false;
484  }
485 
486  unsigned short usNdx = 0;
487  if (isOnRayS((MinX + MaxX) / 2, MaxX, rclVct.x)) { // left/RIGHT
488  usNdx |= 1;
489  }
490  if (isOnRayS((MinY + MaxY) / 2, MaxY, rclVct.y)) { // down/UP
491  usNdx |= 2;
492  }
493  if (isOnRayS((MinZ + MaxZ) / 2, MaxZ, rclVct.z)) { // back/FRONT
494  usNdx |= 4;
495  }
496  rclOctant = static_cast<OCTANT>(usNdx);
497  return true;
498 }

514  cOct.MinX = (cOct.MinX + cOct.MaxX) / 2;
515  cOct.MaxY = (cOct.MinY + cOct.MaxY) / 2;
516  cOct.MaxZ = (cOct.MinZ + cOct.MaxZ) / 2;
517  break;
518 
519  case OCT_LUB:
520  cOct.MaxX = (cOct.MinX + cOct.MaxX) / 2;
521  cOct.MinY = (cOct.MinY + cOct.MaxY) / 2;
522  cOct.MaxZ = (cOct.MinZ + cOct.MaxZ) / 2;
523  break;
524 
525  case OCT_RUB:
526  cOct.MinX = (cOct.MinX + cOct.MaxX) / 2;
527  cOct.MinY = (cOct.MinY + cOct.MaxY) / 2;
528  cOct.MaxZ = (cOct.MinZ + cOct.MaxZ) / 2;
529  break;
530 
531  case OCT_LDF:
532  cOct.MaxX = (cOct.MinX + cOct.MaxX) / 2;

534  cOct.MinZ = (cOct.MinZ + cOct.MaxZ) / 2;

694  BoundBox3<Precision> cCmpBound(*this);
695 
696  // enlarge bounding box by epsilon
697  cCmpBound.Enlarge(epsilon);
698 
699  // Is point inside?
700  if (cCmpBound.IsInBox(rcVct)) {
701  // test sides
702  for (unsigned short i = 0; (i < num) && (!rc); i++) {
703  rc = IntersectPlaneWithLine(i, rcVct, rcVctDir, cVctRes);
704 
705  if (!cCmpBound.IsInBox(cVctRes)) {
706  rc = false;
707  }
708 
709  if (rc) {
710  // does intersection point lie in desired direction
711  // or was found the opposing side?
712  // -> scalar product of both direction vectors > 0 (angle < 90)
713  rc = ((cVctRes - rcVct) * rcVctDir) >= 0.0;
714  }
715  }
716  }
717 
718  return rc;
719 }

721 template<class Precision>
722 inline bool BoundBox3<Precision>::IsCutLine(const Vector3<Precision>& rcBase,
723  const Vector3<Precision>& rcDir,
724  Precision fTolerance) const
725 {
726  const unsigned short num = 6;
727  Precision fDist;
728 
729  // at first only a rough and quick test by the
730  // Distance of the line to the center of the BB is calculated
731  // and with the maximum diagonal length + fTolerance
732  // will be compared.
733 
734  // Distance between center point and line
735  fDist = (rcDir % (GetCenter() - rcBase)).Length() / rcDir.Length();
736 
737  if (fDist > (CalcDiagonalLength() + fTolerance)) {
738  return false;
739  }
740 
741  // more detailed test here
742  Vector3<Precision> clVectRes;
743 
744  // intersect each face with the line
745  for (unsigned short i = 0; i < num; i++) {
746  if (IntersectPlaneWithLine(i, rcBase, rcDir, clVectRes)) {
747  // Check whether the intersection point is within BB limits + Tolerance
748  switch (i) {
749  case LEFT: // left and right plane
750  case RIGHT:
751  if ((isOnRayW(MinY - fTolerance, MaxY + fTolerance, clVectRes.y)
752  && isOnRayW(MinZ - fTolerance, MaxZ + fTolerance, clVectRes.z))) {
753  return true;
754  }
755  break;
756  case TOP: // top and bottom plane
757  case BOTTOM:
758  if ((isOnRayW(MinX - fTolerance, MaxX + fTolerance, clVectRes.x)
759  && isOnRayW(MinZ - fTolerance, MaxZ + fTolerance, clVectRes.z))) {
760  return true;
761  }
762  break;
763  case FRONT: // front and back plane
764  case BACK:
765  if ((isOnRayW(MinX - fTolerance, MaxX + fTolerance, clVectRes.x)
766  && isOnRayW(MinY - fTolerance, MaxY + fTolerance, clVectRes.y))) {
767  return true;
768  }
769  break;
770  }
771  }
772  }
773 
774  return false;
775 }

793 // NOLINTBEGIN(bugprone-easily-swappable-parameters)
794 template<class Precision>
795 inline bool BoundBox3<Precision>::IntersectWithLine(const Vector3<Precision>& rcBase,
796  const Vector3<Precision>& rcDir,
797  Vector3<Precision>& rcP0,
798  Vector3<Precision>& rcP1) const
799 // NOLINTEND(bugprone-easily-swappable-parameters)
800 {
801  const unsigned short num = 6;
802  Vector3<Precision> clVectRes;
803  std::array<Vector3<Precision>, num> clVect;
804  unsigned short numIntersect = 0;
805 
806  auto checkIntersect = [&](Base::Vector3<Precision> p1, Base::Vector3<Precision> p2) {
807  if (isOnRayS(p1.x, p1.y, p1.z) && isOnRayS(p2.x, p2.y, p2.z)) {
808  clVect[numIntersect] = clVectRes;
809  numIntersect++;
810  }
811  };
812 
813  // cut each face with the line
814  for (unsigned short i = 0; i < num; i++) {
815  if (IntersectPlaneWithLine(i, rcBase, rcDir, clVectRes)) {
816  // check if intersection point is inside
817  switch (i) {
818  case LEFT: // left and right plane
819  case RIGHT:
820  checkIntersect(Vector3<Precision> {MinY, MaxY, clVectRes.y},
821  Vector3<Precision> {MinZ, MaxZ, clVectRes.z});
822  break;
823  case TOP: // top and bottom plane
824  case BOTTOM:
825  checkIntersect(Vector3<Precision> {MinX, MaxX, clVectRes.x},
826  Vector3<Precision> {MinZ, MaxZ, clVectRes.z});
827  break;
828  case FRONT: // front and back plane
829  case BACK:
830  checkIntersect(Vector3<Precision> {MinX, MaxX, clVectRes.x},
831  Vector3<Precision> {MinY, MaxY, clVectRes.y});
832  break;
833  }
834  }
835  }
836 
837  if (numIntersect == 2) {
838  rcP0 = clVect[0];
839  rcP1 = clVect[1];
840  return true;
841  }
842 
843  if (numIntersect > 2) { // search two different intersection points
844  for (unsigned short i = 1; i < numIntersect; i++) {
845  if (clVect[i] != clVect[0]) {
846  rcP0 = clVect[0];
847  rcP1 = clVect[i];
848  return true;
849  }
850  }
851  }
852 
853  return false;
854 }

857 template<class Precision>
858 inline bool BoundBox3<Precision>::IntersectPlaneWithLine(unsigned short usSide,
859  const Vector3<Precision>& rcBase,
860  const Vector3<Precision>& rcDir,
861  Vector3<Precision>& rcP0) const
862 // NOLINTEND(bugprone-easily-swappable-parameters)
863 {
864  Precision value;
865  Vector3<Precision> cBase;
866  Vector3<Precision> cNormal;
867  Vector3<Precision> cDir(rcDir);
868  CalcPlane(usSide, cBase, cNormal);
869 
870  if ((cNormal * cDir) == 0.0F) {
871  return false; // no point of intersection
872  }
873 
874  value = (cNormal * (cBase - rcBase)) / (cNormal * cDir);
875  cDir.Scale(value, value, value);
876  rcP0 = rcBase + cDir;
877  return true;
878 }

889 template<class Precision>
890 inline typename BoundBox3<Precision>::SIDE
891 BoundBox3<Precision>::GetSideFromRay(const Vector3<Precision>& rclPt,
892  const Vector3<Precision>& rclDir,
893  Vector3<Precision>& rcInt) const
894 {
895  Vector3<Precision> cP0;
896  Vector3<Precision> cP1;
897  if (!IntersectWithLine(rclPt, rclDir, cP0, cP1)) {
898  return INVALID;
899  }
900 
901  Vector3<Precision> cOut;
902  // same orientation
903  if ((cP1 - cP0) * rclDir > 0) {
904  cOut = cP1;
905  }
906  else {
907  cOut = cP0;
908  }
909 
910  rcInt = cOut;
911 
912  Precision fMax = 1.0e-3F; // NOLINT
913  SIDE tSide = INVALID;
914 
915  if (fabs(cOut.x - MinX) < fMax) { // left plane
916  fMax = Precision(fabs(cOut.x - MinX));
917  tSide = LEFT;
918  }
919 
920  if (fabs(cOut.x - MaxX) < fMax) { // right plane
921  fMax = Precision(fabs(cOut.x - MaxX));
922  tSide = RIGHT;
923  }
924 
925  if (fabs(cOut.y - MinY) < fMax) { // bottom plane
926  fMax = Precision(fabs(cOut.y - MinY));
927  tSide = BOTTOM;
928  }
929 
930  if (fabs(cOut.y - MaxY) < fMax) { // top plane
931  fMax = Precision(fabs(cOut.y - MaxY));
932  tSide = TOP;
933  }
934 
935  if (fabs(cOut.z - MinZ) < fMax) { // front plane
936  fMax = Precision(fabs(cOut.z - MinZ));
937  tSide = FRONT;
938  }
939 
940  if (fabs(cOut.z - MaxZ) < fMax) { // back plane
941  fMax = Precision(fabs(cOut.z - MaxZ));
942  tSide = BACK;
943  }
944 
945  return tSide;
946 }

948 template<class Precision>
949 inline Vector3<Precision> BoundBox3<Precision>::ClosestPoint(const Vector3<Precision>& rclPt) const
950 {
951  Vector3<Precision> closest = rclPt;
952 
953  Vector3<Precision> center = GetCenter();
954  Precision devx = closest.x - center.x;
955  Precision devy = closest.y - center.y;
956  Precision devz = closest.z - center.z;
957 
958  Precision halfwidth = (MaxX - MinX) / 2;
959  Precision halfheight = (MaxY - MinY) / 2;
960  Precision halfdepth = (MaxZ - MinZ) / 2;
961 
962  // Move point to be on the nearest plane of the box.
963  if ((fabs(devx) > fabs(devy)) && (fabs(devx) > fabs(devz))) {
964  closest.x = center.x + halfwidth * ((devx < 0.0) ? -1.0 : 1.0);
965  }
966  else if (fabs(devy) > fabs(devz)) {
967  closest.y = center.y + halfheight * ((devy < 0.0) ? -1.0 : 1.0);
968  }
969  else {
970  closest.z = center.z + halfdepth * ((devz < 0.0) ? -1.0 : 1.0);
971  }
972 
973  // Clamp to be inside box.
974  closest.x = std::min<Precision>(std::max<Precision>(closest.x, MinX), MaxX);
975  closest.y = std::min<Precision>(std::max<Precision>(closest.y, MinY), MaxY);
976  closest.z = std::min<Precision>(std::max<Precision>(closest.z, MinZ), MaxZ);
977 
978  return closest;
979 }
