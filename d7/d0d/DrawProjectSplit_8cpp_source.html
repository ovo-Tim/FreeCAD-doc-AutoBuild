126 bool DrawProjectSplit::isOnEdge(TopoDS_Edge e, TopoDS_Vertex v, double& param, bool allowEnds)
127 {
128  param = -2;
129 
130  //eliminate obvious cases
131  Bnd_Box sBox;
132  BRepBndLib::AddOptimal(e, sBox);
133  sBox.SetGap(0.1);
134  if (!sBox.IsVoid()) {
135  gp_Pnt pt = BRep_Tool::Pnt(v);
136  if (sBox.IsOut(pt)) {
137  return false; // Out of box
138  }
139  }
140 
141  double dist = DrawUtil::simpleMinDist(v, e);
142  if (dist < 0.0) {
143  Base::Console().Error("DPS::isOnEdge - simpleMinDist failed: %.3f\n", dist);
144  return false;
145  } else if (dist < Precision::Confusion()) {
146  const gp_Pnt pt = BRep_Tool::Pnt(v); //have to duplicate method 3 to get param
147  BRepAdaptor_Curve adapt(e);
148  const Handle(Geom_Curve) c = adapt.Curve().Curve();
149  double maxDist = 0.000001; //magic number. less than this gives false positives.
150  //bool found =
151  (void) GeomLib_Tool::Parameter(c, pt, maxDist, param); //already know point it on curve
152 
153  TopoDS_Vertex v1 = TopExp::FirstVertex(e);
154  TopoDS_Vertex v2 = TopExp::LastVertex(e);
155  if (DrawUtil::isSamePoint(v, v1) || DrawUtil::isSamePoint(v, v2)) {
156  if (!allowEnds) {
157  return false;
158  }
159  }
160  return true;
161  }
162

167 std::vector<TopoDS_Edge> DrawProjectSplit::splitEdges(std::vector<TopoDS_Edge> edges, std::vector<splitPoint> splits)
168 {
169  std::vector<TopoDS_Edge> result;
170  std::vector<TopoDS_Edge> newEdges;
171  std::vector<splitPoint> edgeSplits; //splits for current edge
172  int iEdge = 0; //current edge index
173  int iSplit = 0; //current splitindex
174  int ii = 0; //i value of current split
175  int endEdge = edges.size();
176  int endSplit = splits.size();
177  int imax = std::numeric_limits<int>::max();
178 
179  while (iEdge < endEdge) {
180  if (iSplit < endSplit) {
181  ii = splits[iSplit].i;
182  } else {
183  ii = imax;
184  }
185  if (ii == iEdge) {
186  edgeSplits.push_back(splits[iSplit]);
187  iSplit++;
188  } else if (ii > iEdge) {
189  if (!edgeSplits.empty()) { //save *iedge's splits
190  newEdges = split1Edge(edges[iEdge], edgeSplits);
191  result.insert(result.end(), newEdges.begin(), newEdges.end());
192  edgeSplits.clear();
193  } else {
194  result.push_back(edges[iEdge]); //save *iedge
195  }
196  iEdge++; //next edge
197  } else if (iEdge > ii) {
198  iSplit++;
199  }
200  }
201 
202  if (!edgeSplits.empty()) { //handle last batch
203  newEdges = split1Edge(edges[iEdge], edgeSplits);
204  result.insert(result.end(), newEdges.begin(), newEdges.end());
205  edgeSplits.clear();
206  }
207 
208  return result;
209 }

212 std::vector<TopoDS_Edge> DrawProjectSplit::split1Edge(TopoDS_Edge e, std::vector<splitPoint> splits)
213 {
214  std::vector<TopoDS_Edge> result;
215  if (splits.empty()) {
216  return result;
217  }
218 
219  BRepAdaptor_Curve adapt(e);
220  Handle(Geom_Curve) c = adapt.Curve().Curve();
221  double first = BRepLProp_CurveTool::FirstParameter(adapt);
222  double last = BRepLProp_CurveTool::LastParameter(adapt);
223  if (first > last) {
224  //TODO parms.reverse();
225  Base::Console().Message("DPS::split1Edge - edge is backwards!\n");
226  return result;
227  }
228  std::vector<double> parms;
229  parms.push_back(first);
230  for (auto& s:splits) {
231  parms.push_back(s.param);
232  }
233 
234  parms.push_back(last);
235  std::vector<double>::iterator pfirst = parms.begin();
236  auto parms2 = parms.begin() + 1;
237  std::vector<double>::iterator psecond = parms2;
238  std::vector<double>::iterator pstop = parms.end();
239  for (; psecond != pstop; ++pfirst, ++psecond) {
240  try {
241  BRepBuilderAPI_MakeEdge mkEdge(c, *pfirst, *psecond);
242  if (mkEdge.IsDone()) {
243  TopoDS_Edge e1 = mkEdge.Edge();
244  result.push_back(e1);
245  }
246  }
247  catch (Standard_Failure&) {
248  Base::Console().Message("DPS::split1Edge failed building edge segment\n");
249  }
250  }
251  return result;
252 }

289 std::vector<TopoDS_Edge> DrawProjectSplit::removeDuplicateEdges(std::vector<TopoDS_Edge>& inEdges)
290 {
291  std::vector<TopoDS_Edge> result;
292  std::vector<edgeSortItem> temp;
293 
294  unsigned int idx = 0;
295  for (auto& e: inEdges) {
296  edgeSortItem item;
297  TopoDS_Vertex v1 = TopExp::FirstVertex(e);
298  TopoDS_Vertex v2 = TopExp::LastVertex(e);
299  item.start = DrawUtil::vertex2Vector(v1);
300  item.end = DrawUtil::vertex2Vector(v2);
301  item.startAngle = DrawUtil::angleWithX(e, v1);
302  item.endAngle = DrawUtil::angleWithX(e, v2);
303  //catch reverse-duplicates
304  if (DrawUtil::vectorLess(item.end, item.start)) {
305  Base::Vector3d vTemp = item.start;
306  item.start = item.end;
307  item.end = vTemp;
308  double aTemp = item.startAngle;
309  item.startAngle = item.endAngle;
310  item.endAngle = aTemp;
311  }
312  item.idx = idx;
313  temp.push_back(item);
314  idx++;
315  }
316 
317  std::vector<edgeSortItem> sorted = sortEdges(temp, true);
318  auto last = std::unique(sorted.begin(), sorted.end(), edgeSortItem::edgeEqual); //duplicates to back
319  sorted.erase(last, sorted.end()); //remove dupls
320 
321  for (const auto& e: sorted) {
322  if (e.idx < inEdges.size()) {
323  result.push_back(inEdges.at(e.idx));
324  } else {
325  Base::Console().Message("ERROR - DPS::removeDuplicateEdges - access: %d inEdges: %d\n", e.idx, inEdges.size());
326  //TODO: throw index error
327  }
328  }
329  return result;
330 }

358 {

370  }
371  } else if (e1.idx < e2.idx) {

399 std::vector<TopoDS_Edge> DrawProjectSplit::scrubEdges(const std::vector<TechDraw::BaseGeomPtr>& origEdges,
400  std::vector<TopoDS_Edge> &closedEdges)
401 {
402 // Base::Console().Message("DPS::scrubEdges() - BaseGeom in: %d\n", origEdges.size());
403  //make a copy of the input edges so the loose tolerances of face finding are
404  //not applied to the real edge geometry. See TopoDS_Shape::TShape().
405  std::vector<TopoDS_Edge> copyEdges;
406  bool copyGeometry = true;
407  bool copyMesh = false;
408  for (const auto& tdEdge: origEdges) {
409  if (!DrawUtil::isZeroEdge(tdEdge->getOCCEdge(), 2.0 * EWTOLERANCE)) {
410  BRepBuilderAPI_Copy copier(tdEdge->getOCCEdge(), copyGeometry, copyMesh);
411  copyEdges.push_back(TopoDS::Edge(copier.Shape()));
412  }
413  }
414  return scrubEdges(copyEdges, closedEdges);
415 }

417 //origEdges should be a copy of the original edges since the underlying TShape will be modified.
418 std::vector<TopoDS_Edge> DrawProjectSplit::scrubEdges(std::vector<TopoDS_Edge>& origEdges,
419  std::vector<TopoDS_Edge> &closedEdges)
420 {
421 // Base::Console().Message("DPS::scrubEdges() - TopoDS_Edges in: %d\n", origEdges.size());
422  std::vector<TopoDS_Edge> openEdges;
423 
424  // We must have at least 2 edges to perform the General Fuse operation
425  if (origEdges.size() < 2) {
426  if (origEdges.empty()) {
427  //how did this happen? if Scale is zero, all the edges will be zero length,
428  //but Scale property has constraint, so this shouldn't happen!
429  //Base::Console().Message("DPS::scrubEdges(2) - origEdges is empty\n");
430  }
431  else {
432  TopoDS_Edge &edge = origEdges.front();
433  if (BRep_Tool::IsClosed(edge)) {
434  closedEdges.push_back(edge);
435  }
436  else {
437  openEdges.push_back(edge);
438  }
439  }
440 
441  return openEdges;
442  }
443 
444  TopTools_ListOfShape edgeList;
445  for (auto edge : origEdges) {
446  edgeList.Append(edge);
447  }
448 
449  BOPAlgo_Builder bopBuilder;
450  bopBuilder.SetArguments(edgeList);
451  bopBuilder.SetFuzzyValue(FUZZYADJUST*EWTOLERANCE);
452  // Allow modifying edges in place, scrubEdges() caller is expected to back them up
453  bopBuilder.SetNonDestructive(Standard_False);
454  // Because we are interested only in edges, we do not need gluing
455  bopBuilder.SetGlue(BOPAlgo_GlueOff);
456  // No solids in the input list
457  bopBuilder.SetCheckInverted(Standard_False);
458  // Use oriented bound boxes
459  bopBuilder.SetUseOBB(Standard_True);
460  bopBuilder.SetRunParallel(Standard_True);
461 
462  bopBuilder.Perform();
463  if (bopBuilder.HasErrors()) {
464  Standard_SStream errorStream;
465  bopBuilder.DumpErrors(errorStream);
466  const std::string &errorStr = errorStream.str();
467  Base::Console().Error("DrawProjectSplit::scrubEdges - OCC fuse failed with error(s):\n%s\n", errorStr.c_str());
468  return std::vector<TopoDS_Edge>();
469  }
470 
471  if (bopBuilder.HasWarnings()) {
472  Standard_SStream warnStream;
473  bopBuilder.DumpWarnings(warnStream);
474  const std::string &warnStr = warnStream.str();
475  Base::Console().Warning("DrawProjectSplit::scrubEdges - OCC fuse raised warning(s):\n%s\n", warnStr.c_str());
476  }
477 
478  const TopoDS_Shape &bopResult = bopBuilder.Shape();
479  if (!bopResult.IsNull()) {
480  for (TopExp_Explorer explorer(bopResult, TopAbs_EDGE); explorer.More(); explorer.Next()) {
481  const TopoDS_Edge &edge = TopoDS::Edge(explorer.Current());
482  if (BRep_Tool::IsClosed(edge)) {
483  closedEdges.push_back(edge);
484  }
485  else {
486  openEdges.push_back(edge);
487  }
488  }
489  }
490 
491  //find all the unique vertices and count how many edges terminate at each, then
492  //remove edges that can't be part of a closed region since they are not connected at both ends
493  vertexMap verts = DrawProjectSplit::getUniqueVertexes(openEdges);
494  return DrawProjectSplit::pruneUnconnected(verts, openEdges);
495 }

524 //if an edge is not connected at both ends, then it can't be part of a face boundary
525 //and unconnected edges may confuse up the edge walker.
526 //note: closed edges have been removed by this point for later handling
527 std::vector<TopoDS_Edge> DrawProjectSplit::pruneUnconnected(vertexMap verts,
528  std::vector<TopoDS_Edge> edges)
529 {
530 // Base::Console().Message("DPS::pruneUnconnected() - edges in: %d\n", edges.size());
531  //check if edge ends are used at least twice => edge is joined to another edge
532  std::vector<TopoDS_Edge> newPile;
533  std::vector<TopoDS_Edge> deadEnds;
534  for (auto& edge: edges) {
535  gp_Pnt p = BRep_Tool::Pnt(TopExp::FirstVertex(edge));
536  Base::Vector3d v0(p.X(), p.Y(), p.Z());
537  int count0 = 0;
538  vertexMap::iterator it0(verts.find(v0));
539  if (it0 != verts.end()) {
540  count0 = it0->second;
541  }
542  p = BRep_Tool::Pnt(TopExp::LastVertex(edge));
543  Base::Vector3d v1(p.X(), p.Y(), p.Z());
544  int count1 = 0;
545  vertexMap::iterator it1(verts.find(v1));
546  if (it1 != verts.end()) {
547  count1 = it1->second;
548  }
549  if ((count0 > 1) && (count1 > 1)) { //connected at both ends
550  newPile.push_back(edge);
551  } else if ((count0 == 1) && (count1 == 1)) {
552  //completely disconnected edge. just drop it.
553  continue;
554  } else {
555  //only connected at 1 end
556  deadEnds.push_back(edge); //could separate dead ends here
557  }
558  }
559 
560  return newPile;
561 }

587 //eliminate edges that overlap another edge
588 std::vector<TopoDS_Edge> DrawProjectSplit::removeOverlapEdges(const std::vector<TopoDS_Edge> &inEdges)
589 {
590 // Base::Console().Message("DPS::removeOverlapEdges() - %d edges in\n", inEdges.size());
591  std::vector<TopoDS_Edge> outEdges;
592  std::vector<TopoDS_Edge> overlapEdges;
593  std::vector<bool> skipThisEdge(inEdges.size(), false);
594  int edgeCount = inEdges.size();
595  int ie0 = 0;
596  for (; ie0 < edgeCount; ie0++) {
597  if (skipThisEdge.at(ie0)) {
598  continue;
599  }
600  int ie1 = ie0 + 1;
601  for (; ie1 < edgeCount; ie1++) {
602  if (skipThisEdge.at(ie1)) {
603  continue;
604  }
605  int rc = isSubset(inEdges.at(ie0), inEdges.at(ie1));
606  if (rc == e0ISSUBSET) {
607  skipThisEdge.at(ie0) = true;
608  break; //stop checking ie0
609  } else if (rc == e1ISSUBSET) {
610  skipThisEdge.at(ie1) = true;
611  } else if (rc == EDGEOVERLAP) {
612  skipThisEdge.at(ie0) = true;
613  skipThisEdge.at(ie1) = true;
614  std::vector<TopoDS_Edge> olap = fuseEdges(inEdges.at(ie0), inEdges.at(ie1));
615  if (!olap.empty()) {
616  overlapEdges.insert(overlapEdges.end(), olap.begin(), olap.end());
617  }
618  break; //stop checking ie0
619  }
620  } //inner loop
621  } //outer loop
622 
623  int iOut = 0;
624  for (auto& e: inEdges) {
625  if (!skipThisEdge.at(iOut)) {
626  outEdges.push_back(e);
627  }
628  iOut++;
629  }
630 
631  if (!overlapEdges.empty()) {
632  outEdges.insert(outEdges.end(), overlapEdges.begin(), overlapEdges.end());
633  }
634 
635 // Base::Console().Message("DPS::removeOverlapEdges() - %d edges out\n", outEdges.size());
636 
637  return outEdges;
638 }

640 //determine if edge0 & edge1 are superimposed, and classify the type of overlap
641 int DrawProjectSplit::isSubset(const TopoDS_Edge &edge0, const TopoDS_Edge &edge1)
642 {
643  if (!boxesIntersect(edge0, edge1)) {
644  return NOTASUBSET; //boxes don't intersect, so edges do not overlap
645  }
646 
647  //bboxes of edges intersect
648  FCBRepAlgoAPI_Common anOp;
649  anOp.SetFuzzyValue (FUZZYADJUST * EWTOLERANCE);
650  TopTools_ListOfShape anArg1, anArg2;
651  anArg1.Append (edge0);
652  anArg2.Append (edge1);
653  anOp.SetArguments (anArg1);
654  anOp.SetTools (anArg2);
655  anOp.Build();
656  TopoDS_Shape aRes = anOp.Shape(); //always a compound
657  if (aRes.IsNull()) {
658  return NOTASUBSET; //no common segment
659  }
660  std::vector<TopoDS_Edge> commonEdgeList;
661  TopExp_Explorer edges(aRes, TopAbs_EDGE);
662  for (int i = 1; edges.More(); edges.Next(), i++) {
663  commonEdgeList.push_back(TopoDS::Edge(edges.Current()));
664  }
665  if (commonEdgeList.empty()) {
666  return NOTASUBSET;
667  }
668  //we're only going to deal with the situation where the common of the edges
669  //is a single edge. A really odd pair of edges could have >1 edge in their
670  //common.
671  TopoDS_Edge commonEdge = commonEdgeList.at(0);
672  if (sameEndPoints(edge1, commonEdge)) {
673  return e1ISSUBSET; //e1 is a subset of e0
674  }
675  if (sameEndPoints(edge0, commonEdge)) {
676  return e0ISSUBSET; //e0 is a subset of e1
677  }
678  // edge0 is not a subset of edge1, nor is edge1 a subset of edge0, but they have a common segment
679  return EDGEOVERLAP;
680 }
