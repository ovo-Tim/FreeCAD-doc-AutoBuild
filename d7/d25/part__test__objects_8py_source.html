59 def _create_frame():
60  """Draw a frame with information on the version of the software.
61 
62  It includes the date created, the version, the release type,
63  and the branch.
64  """
65  version = App.Version()
66  now = datetime.datetime.now().strftime("%Y/%m/%dT%H:%M:%S")
67 
68  _text = ["Part test file",
69  "Created: {}".format(now),
70  "\n",
71  "Version: " + ".".join(version[0:3]),
72  "Release: " + " ".join(version[3:5]),
73  "Branch: " + " ".join(version[5:])]
74  record = App.ActiveDocument.addObject("App::Annotation", "Description")
75  record.LabelText = _text
76  record.Position = Vector(0, -1000, 0)
77 
78  if App.GuiUp:
79  record.ViewObject.DisplayMode = "World"
80  record.ViewObject.FontSize = 400
81  record.ViewObject.TextColor = (0.0, 0.0, 0.0)
82 
83  p1 = Vector(-1000, -3500, 0)
84  p2 = Vector(20000, -3500, 0)
85  p3 = Vector(20000, 8500, 0)
86  p4 = Vector(-1000, 8500, 0)
87 
88  poly = Part.makePolygon([p1, p2, p3, p4, p1])
89  frame = App.ActiveDocument.addObject("Part::Feature", "Frame")
90  frame.Shape = poly

93 def create_test_file(file_name="part_test_objects",
94  file_path=os.environ["HOME"],
95  save=False):
96  """Create a complete test file of Part objects.
97 
98  It draws a frame with information on the software used to create
99  the test document, and fills it with every object that can be created.
100 
101  Parameters
102  ----------
103  file_name: str, optional
104  It defaults to `'part_test_objects'`.
105  It is the name of the document that is created.
106 
107  The `file_name` will be appended to `file_path`
108  to determine the actual path to save. The extension `.FCStd`
109  will be added automatically.
110 
111  file_path: str, optional
112  It defaults to the value of `os.environ['HOME']`
113  which in Linux is usually `'/home/user'`.
114 
115  If it is the empty string `''` it will use the value
116  returned by `App.getUserAppDataDir()`,
117  for example, `'/home/user/.FreeCAD/'`.
118 
119  save: bool, optional
120  It defaults to `False`. If it is `True` the new document
121  will be saved to disk after creating all objects.
122 
123  Returns
124  -------
125  App::Document
126  A reference to the test document that was created.
127  """
128  doc = App.newDocument(file_name)
129  _msg(16 * "-")
130  _msg("Filename: {}".format(file_name))
131  _msg("If the units tests fail, this script may fail as well")
132 
133  _create_frame()
