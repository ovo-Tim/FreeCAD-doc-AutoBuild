199  def clasifySub(self, bs, sub):
200  """clasifySub(bs, sub)...
201  Given a base and a sub-feature name, returns True
202  if the sub-feature is a horizontally oriented flat face.
203  """
204  face = bs.Shape.getElement(sub)
205 
206  if type(face.Surface) == Part.Plane:
207  Path.Log.debug("type() == Part.Plane")
208  if Path.Geom.isVertical(face.Surface.Axis):
209  Path.Log.debug(" -isVertical()")
210  # it's a flat horizontal face
211  self.horiz.append(face)
212  return True
213 
214  elif Path.Geom.isHorizontal(face.Surface.Axis):
215  Path.Log.debug(" -isHorizontal()")
216  self.vert.append(face)
217  return True
218 
219  else:
220  return False
221 
222  elif type(face.Surface) == Part.Cylinder and Path.Geom.isVertical(face.Surface.Axis):
223  Path.Log.debug("type() == Part.Cylinder")
224  # vertical cylinder wall
225  if any(e.isClosed() for e in face.Edges):
226  Path.Log.debug(" -e.isClosed()")
227  # complete cylinder
228  circle = Part.makeCircle(face.Surface.Radius, face.Surface.Center)
229  disk = Part.Face(Part.Wire(circle))
230  disk.translate(FreeCAD.Vector(0, 0, face.BoundBox.ZMin - disk.BoundBox.ZMin))
231  self.horiz.append(disk)
232  return True
233 
234  else:
235  Path.Log.debug(" -none isClosed()")
236  # partial cylinder wall
237  self.vert.append(face)
238  return True
239 
240  elif type(face.Surface) == Part.SurfaceOfExtrusion:
241  # extrusion wall
242  Path.Log.debug("type() == Part.SurfaceOfExtrusion")
243  # Save face to self.horiz for processing or display error
244  if self.isVerticalExtrusionFace(face):
245  self.vert.append(face)
246  return True
247  else:
248  Path.Log.error("Failed to identify vertical face from {}".format(sub))
249 
250  else:
251  Path.Log.debug(" -type(face.Surface): {}".format(type(face.Surface)))
252  return False
