107 const std::string& FileInfo::getTempPath()
108 {
109  static std::string tempPath;
110 
111  if (tempPath.empty()) {
112 #ifdef FC_OS_WIN32
113  wchar_t buf[MAX_PATH + 2];
114  GetTempPathW(MAX_PATH + 1, buf);
115  int neededSize = WideCharToMultiByte(CP_UTF8, 0, buf, -1, 0, 0, 0, 0);
116  char* dest = new char[static_cast<size_t>(neededSize)];
117  WideCharToMultiByte(CP_UTF8, 0, buf, -1, dest, neededSize, 0, 0);
118  tempPath = dest;
119  delete[] dest;
120 #else
121  const char* tmp = getenv("TMPDIR");
122  if (tmp && tmp[0] != '\0') {
123  tempPath = tmp;
124  FileInfo fi(tempPath);
125  if (tempPath.empty() || !fi.isDir()) { // still empty or non-existent
126  tempPath = "/tmp/";
127  }
128  else if (tempPath.at(tempPath.size() - 1) != '/') {
129  tempPath.append("/");
130  }
131  }
132  else {
133  tempPath = "/tmp/";
134  }
135 #endif
136  }
137 
138  return tempPath;
139 }

141 std::string FileInfo::getTempFileName(const char* FileName, const char* Path)
142 {
143  // FIXME: To avoid race conditions we should rather return a file pointer
144  // than a file name.
145 #ifdef FC_OS_WIN32
146  wchar_t buf[MAX_PATH + 2];
147 
148  // Path where the file is located
149  std::wstring path;
150  if (Path) {
151  path = ConvertToWideString(std::string(Path));
152  }
153  else {
154  path = ConvertToWideString(getTempPath());
155  }
156 
157  // File name in the path
158  std::wstring file;
159  if (FileName) {
160  file = ConvertToWideString(std::string(FileName));
161  }
162  else {
163  file = L"FCTempFile";
164  }
165 
166 
167  // this already creates the file
168  GetTempFileNameW(path.c_str(), file.c_str(), 0, buf);
169  DeleteFileW(buf);
170 
171  return std::string(ConvertFromWideString(std::wstring(buf)));
172 #else
173  std::string buf;
174 
175  // Path where the file is located
176  if (Path) {
177  buf = Path;
178  }
179  else {
180  buf = getTempPath();
181  }
182 
183  // File name in the path
184  if (FileName) {
185  buf += "/";
186  buf += FileName;
187  buf += "XXXXXX";
188  }
189  else {
190  buf += "/fileXXXXXX";
191  }
192 
193  std::vector<char> vec;
194  std::copy(buf.begin(), buf.end(), std::back_inserter(vec));
195  vec.push_back('\0');
196 
197  /* coverity[secure_temp] mkstemp uses 0600 as the mode and is safe */
198  int id = mkstemp(vec.data());
199  if (id > -1) {
200  FILE* file = fdopen(id, "w");
201  fclose(file);
202  vec.pop_back(); // remove '\0'
203  std::string str(vec.begin(), vec.end());
204  buf.swap(str);
205  unlink(buf.c_str());
206  }
207  return buf;
208 #endif
209 }

642 std::vector<Base::FileInfo> FileInfo::getDirectoryContent() const
643 {
644  std::vector<Base::FileInfo> List;
645 #if defined(FC_OS_WIN32)
646  struct _wfinddata_t dentry;
647 
648  intptr_t hFile;
649 
650  // Find first directory entry
651  std::wstring wstr = toStdWString();
652  wstr += L"/*";
653 
654  if ((hFile = _wfindfirst(wstr.c_str(), &dentry)) == -1L) {
655  return List;
656  }
657 
658  while (_wfindnext(hFile, &dentry) == 0) {
659  if (wcscmp(dentry.name, L"..") != 0) {
660  List.push_back(
661  FileInfo(FileName + "/" + ConvertFromWideString(std::wstring(dentry.name))));
662  }
663  }
664 
665  _findclose(hFile);
666 
667 #elif defined(FC_OS_LINUX) || defined(FC_OS_CYGWIN) || defined(FC_OS_MACOSX) || defined(FC_OS_BSD)
668  DIR* dp(nullptr);
669  struct dirent* dentry(nullptr);
670  if (!(dp = opendir(FileName.c_str()))) {
671  return List;
672  }
673 
674  while ((dentry = readdir(dp))) {
675  std::string dir = dentry->d_name;
676  if (dir != "." && dir != "..") {
677  List.emplace_back(FileName + "/" + dir);
678  }
679  }
680  closedir(dp);
681 #else
682 #error "FileInfo::getDirectoryContent() not implemented for this platform!"
683 #endif
684  return List;
685 }
