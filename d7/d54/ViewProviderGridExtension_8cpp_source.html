157 void GridExtensionP::getClosestGridPoint(double &x, double &y) const
158 {
159  auto closestdim = [](double &dim, double gridValue) {
160  dim = dim / gridValue;
161  dim = dim < 0.0 ? ceil(dim - 0.5) : floor(dim + 0.5);
162  dim *= gridValue;
163  };
164 
165  closestdim(x, computedGridValue);
166  closestdim(y, computedGridValue);

182 bool GridExtensionP::checkCameraTranslationChange(const Gui::View3DInventorViewer* viewer)
183 {
184  //Then we check if user moved by more than 10% of camera dimension (must be after updating camera dimension).
185  SbVec3f newCamCenterPointOnFocalPlane = viewer->getCenterPointOnFocalPlane();

195 void GridExtensionP::computeGridSize(const Gui::View3DInventorViewer* viewer)
196 {
197 
198  auto capGridSize = [](auto & value){
199  value = std::max(static_cast<float>(value), std::numeric_limits<float>::min());
200  value = std::min(static_cast<float>(value), std::numeric_limits<float>::max());
201  };
202 
203  if (!vp->GridAuto.getValue()) {
204  computedGridValue = vp->GridSize.getValue();
205  capGridSize(computedGridValue);
206  return;
207  }
208 
209  short pixelWidth = -1;
210  short pixelHeight = -1;
211  viewer->getViewportRegion().getViewportSizePixels().getValue(pixelWidth, pixelHeight);
212  if (pixelWidth < 0 || pixelHeight < 0) {
213  computedGridValue = vp->GridSize.getValue();
214  return;
215  }
216 
217  int numberOfLines = static_cast<int>(std::max(pixelWidth, pixelHeight)) / GridSizePixelThreshold;
218 
219  // If number of subdivision is 1, grid auto spacing can't work as it uses it as a factor
220  // In such case, we apply a default factor of 10
221  auto safeGridNumberSubdivision = GridNumberSubdivision <= 1 ? 10 : GridNumberSubdivision;
222 
223  computedGridValue = vp->GridSize.getValue() * pow(safeGridNumberSubdivision, 1 + floor(log(camMaxDimension / numberOfLines / vp->GridSize.getValue()) / log(safeGridNumberSubdivision)));
224 
225  //cap the grid size
226  capGridSize(computedGridValue);
227 }

229 void GridExtensionP::createGrid(bool cameraUpdate)
230 {
231  auto view = dynamic_cast<Gui::View3DInventor*>(Gui::Application::Instance->editDocument()->getActiveView());
232 
233  if(!view)
234  return;
235 
236  Gui::View3DInventorViewer* viewer = view->getViewer();
237 
238  bool cameraDimensionsChanged = checkCameraZoomChange(viewer);
239 
240  bool cameraCenterMoved = checkCameraTranslationChange(viewer);
241 
242  bool gridNeedUpdating = cameraDimensionsChanged || cameraCenterMoved;
243 
244  if (!gridNeedUpdating && cameraUpdate)
245  return;
246 
247  Gui::coinRemoveAllChildren(GridRoot);
248 
249  computeGridSize(viewer);
250 
251  auto getColor = [](auto unpackedcolor) {
252  SoBaseColor* lineColor = new SoBaseColor;
253  float transparency;
254  SbColor lineCol(0.7f, 0.7f, 0.7f);
255  lineCol.setPackedValue(unpackedcolor, transparency);
256  lineColor->rgb.setValue(lineCol);
257  return lineColor;
258  };
259 
260  //First we create the subdivision lines
261  createGridPart(GridNumberSubdivision, true,
262  (GridNumberSubdivision == 1), GridLinePattern,
263  getColor(GridLineColor), GridLineWidth);
264 
265  //Second we create the wider lines marking every nth lines
266  if (GridNumberSubdivision > 1) {
267  createGridPart(GridNumberSubdivision, false, true,
268  GridDivLinePattern, getColor(GridDivLineColor), GridDivLineWidth);
269  }
270 }

272 void GridExtensionP::createGridPart(int numberSubdiv, bool subDivLines, bool divLines, int pattern, SoBaseColor* color, int lineWidth)
273 {
274  auto* parent = new Gui::SoSkipBoundingGroup();
275  parent->mode = Gui::SoSkipBoundingGroup::EXCLUDE_BBOX;
276 
277  GridRoot->addChild(parent);
278  SoVertexProperty* vts;
279 
280  parent->addChild(color);
281 
282  SoDrawStyle* DefaultStyle = new SoDrawStyle;
283  DefaultStyle->lineWidth = lineWidth;
284  DefaultStyle->linePattern = pattern;
285  parent->addChild(DefaultStyle);
286 
287  SoPickStyle* PickStyle = new SoPickStyle;
288  PickStyle->style = SoPickStyle::UNPICKABLE;
289  parent->addChild(PickStyle);
290 
291  SoLineSet* grid = new SoLineSet;
292  vts = new SoVertexProperty;
293  grid->vertexProperty = vts;
294 
295  float gridDimension = 1.5 * camMaxDimension;
296  int vlines = static_cast<int>(gridDimension / computedGridValue); // total number of vertical lines
297  int nlines = 2 * vlines; // total number of lines
298 
299  if (nlines > 2000) {
300  if(!isTooManySegmentsNotified) {
301  Base::Console().Warning("The grid is too dense, so it is being disabled. Consider zooming in or changing the grid configuration\n");
302  isTooManySegmentsNotified = true;
303  }
304 
305  Gui::coinRemoveAllChildren(GridRoot);
306  return;
307  }
308  else {
309  isTooManySegmentsNotified = false;
310  }
311 
312  // set the grid indices
313  grid->numVertices.setNum(nlines);
314  auto * vertices = grid->numVertices.startEditing();
315  for (int i = 0; i < nlines; i++)
316  vertices[i] = 2;
317  grid->numVertices.finishEditing();
318 
319  // set the grid coordinates
320  vts->vertex.setNum(2 * nlines);
321  SbVec3f* vertex_coords = vts->vertex.startEditing();
322 
323  float minX, minY, maxX, maxY;
324  Base::Vector3d camCenterOnSketch = getCamCenterInSketchCoordinates();
325  minX = static_cast<float>(camCenterOnSketch.x);
326  minY = static_cast<float>(camCenterOnSketch.y);
327 
328  minX -= (gridDimension / 2);
329  minY -= (gridDimension / 2);
330  maxX = minX + gridDimension;
331  maxY = minY + gridDimension;
332 
333  // vertical lines
334  int i_offset_x = static_cast<int>(minX / computedGridValue);
335  for (int i = 0; i < vlines; i++) {
336  int iStep = (i + i_offset_x);
337  if (((iStep % numberSubdiv == 0) && divLines) || ((iStep % numberSubdiv != 0) && subDivLines)) {
338  vertex_coords[2 * i].setValue(iStep * computedGridValue, minY, 0);
339  vertex_coords[2 * i + 1].setValue(iStep * computedGridValue, maxY, 0);
340  }
341  else {
342  /*the number of vertices is defined before. To know the number of vertices ahead it would require
343  to run the loop once before, which would double computation time.
344  If vertices are not filled then there're visual bugs so there are here filled with dummy values.*/
345  vertex_coords[2 * i].setValue(0, 0, 0);
346  vertex_coords[2 * i + 1].setValue(0, 0, 0);
347  }
348  }
349 
350  // horizontal lines
351  int i_offset_y = static_cast<int>(minY / computedGridValue) - vlines;
352  for (int i = vlines; i < nlines; i++) {
353  int iStep = (i + i_offset_y);
354  if (((iStep % numberSubdiv == 0) && divLines) || ((iStep % numberSubdiv != 0) && subDivLines)) {
355  vertex_coords[2 * i].setValue(minX, iStep * computedGridValue, 0);
356  vertex_coords[2 * i + 1].setValue(maxX, iStep * computedGridValue, 0);
357  }
358  else {
359  vertex_coords[2 * i].setValue(0, 0, 0);
360  vertex_coords[2 * i + 1].setValue(0, 0, 0);
361  }
362  }
363  vts->vertex.finishEditing();
364 
365  parent->addChild(vts);
366  parent->addChild(grid);
367 }

369 Base::Vector3d GridExtensionP::getCamCenterInSketchCoordinates() const
370 {
371  Base::Vector3d xaxis(1, 0, 0), yaxis(0, 1, 0);
372 
373  gridRotation.multVec(xaxis,xaxis);
374  gridRotation.multVec(yaxis,yaxis);
375 
376  float x,y,z;
377  camCenterPointOnFocalPlane.getValue(x, y, z);
378 
379  Base::Vector3d center (x,y,z);
380 
381  center.TransformToCoordinateSystem(gridOrigin, xaxis, yaxis);
382 
383  return center;
384 }
