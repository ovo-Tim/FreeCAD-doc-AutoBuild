97 void StdCmdOpen::activated(int iMsg)
98 {
99  // clang-format off
100  auto checkPartialRestore = [](App::Document* doc) {
101  if (doc && doc->testStatus(App::Document::PartialRestore)) {
102  QMessageBox::critical(getMainWindow(), QObject::tr("Error"),
103  QObject::tr("There were errors while loading the file. Some data might have been "
104  "modified or not recovered at all. Look in the report view for more "
105  "specific information about the objects involved."));
106  }
107  };
108 
109  auto checkRestoreError = [](App::Document* doc) {
110  if (doc && doc->testStatus(App::Document::RestoreError)) {
111  QMessageBox::critical(getMainWindow(), QObject::tr("Error"),
112  QObject::tr("There were serious errors while loading the file. Some data might have "
113  "been modified or not recovered at all. Saving the project will most "
114  "likely result in loss of data."));
115  }
116  };
117  // clang-format on
118 
119  Q_UNUSED(iMsg);
120 
121  // fill the list of registered endings
122  QString formatList;
123  const char* supported = QT_TR_NOOP("Supported formats");
124  const char* allFiles = QT_TR_NOOP("All files (*.*)");
125  formatList = QObject::tr(supported);
126  formatList += QLatin1String(" (");
127 
128  std::vector<std::string> filetypes = App::GetApplication().getImportTypes();
129  std::vector<std::string>::iterator it;
130  // Make sure FCStd is the very first fileformat
131  it = std::find(filetypes.begin(), filetypes.end(), "FCStd");
132  if (it != filetypes.end()) {
133  filetypes.erase(it);
134  filetypes.insert(filetypes.begin(), "FCStd");
135  }
136  for (it=filetypes.begin();it != filetypes.end();++it) {
137  formatList += QLatin1String(" *.");
138  formatList += QLatin1String(it->c_str());
139  }
140 
141  formatList += QLatin1String(");;");
142 
143  std::map<std::string, std::string> FilterList = App::GetApplication().getImportFilters();
144  std::map<std::string, std::string>::iterator jt;
145  // Make sure the format name for FCStd is the very first in the list
146  for (jt=FilterList.begin();jt != FilterList.end();++jt) {
147  if (jt->first.find("*.FCStd") != std::string::npos) {
148  formatList += QLatin1String(jt->first.c_str());
149  formatList += QLatin1String(";;");
150  FilterList.erase(jt);
151  break;
152  }
153  }
154  for (jt=FilterList.begin();jt != FilterList.end();++jt) {
155  formatList += QLatin1String(jt->first.c_str());
156  formatList += QLatin1String(";;");
157  }
158  formatList += QObject::tr(allFiles);
159 
160  QString selectedFilter;
161  QStringList fileList = FileDialog::getOpenFileNames(getMainWindow(),
162  QObject::tr("Open document"), QString(), formatList, &selectedFilter);
163  if (fileList.isEmpty()) {
164  return;
165  }
166 
167  // load the files with the associated modules
168  SelectModule::Dict dict = SelectModule::importHandler(fileList, selectedFilter);
169  if (dict.isEmpty()) {
170  QMessageBox::critical(getMainWindow(),
171  qApp->translate("StdCmdOpen", "Cannot open file"),
172  qApp->translate("StdCmdOpen", "Loading the file %1 is not supported").arg(fileList.front()));
173  }
174  else {
175  for (SelectModule::Dict::iterator it = dict.begin(); it != dict.end(); ++it) {
176 
177  // Set flag indicating that this load/restore has been initiated by the user (not by a macro)
178  getGuiApplication()->setStatus(Gui::Application::UserInitiatedOpenDocument, true);
179 
180  getGuiApplication()->open(it.key().toUtf8(), it.value().toLatin1());
181 
182  getGuiApplication()->setStatus(Gui::Application::UserInitiatedOpenDocument, false);
183 
184  App::Document *doc = App::GetApplication().getActiveDocument();
185 
186  checkPartialRestore(doc);
187  checkRestoreError(doc);
188  }
189  }
190 }

315 QString createDefaultExportBasename()
316 {
317  QString defaultFilename;
318 
319  auto selection = Gui::Selection().getObjectsOfType(App::DocumentObject::getClassTypeId());
320  QString exportFormatString;
321  if (selection.size() == 1) {
322  exportFormatString = QString::fromStdString (App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")->
323  GetASCII("ExportDefaultFilenameSingle", "%F-%P-"));
324  }
325  else {
326  exportFormatString = QString::fromStdString (App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")->
327  GetASCII("ExportDefaultFilenameMultiple", "%F"));
328  }
329 
330  // For code simplicity, pull all values we might need
331 
332  // %F - the basename of the.FCStd file(or the label, if it is not saved yet)
333  QString docFilename = QString::fromUtf8(App::GetApplication().getActiveDocument()->getFileName());
334  QFileInfo fi(docFilename);
335  QString fcstdBasename = fi.completeBaseName();
336  if (fcstdBasename.isEmpty())
337  fcstdBasename = QString::fromStdString(App::GetApplication().getActiveDocument()->Label.getStrValue());
338 
339  // %L - the label of the selected object(s)
340  QStringList objectLabels;
341  for (const auto& object : selection)
342  objectLabels.push_back(QString::fromStdString(object->Label.getStrValue()));
343 
344  // %P - the label of the selected objects and their first parent
345  QStringList parentLabels;
346  for (const auto& object : selection) {
347  auto parents = object->getParents();
348  QString firstParent;
349  if (!parents.empty())
350  firstParent = QString::fromStdString(parents.front().first->Label.getStrValue());
351  parentLabels.append(firstParent + QString::fromStdString(object->Label.getStrValue()));
352  }
353 
354  // %U - the date and time, in UTC, ISO 8601
355  QDateTime utc = QDateTime(QDateTime::currentDateTimeUtc());
356  QString utcISO8601 = utc.toString(Qt::ISODate);
357 
358  // %D - the date and time, in local timezone, ISO 8601
359  QDateTime local = utc.toLocalTime();
360  QString localISO8601 = local.toString(Qt::ISODate);
361 
362  // Parse the format string one character at a time:
363  for (int i = 0; i < exportFormatString.size(); ++i) {
364  auto c = exportFormatString.at(i);
365  if (c != QLatin1Char('%')) {
366  // Anything that's not a format start character is just a literal
367  defaultFilename.append(c);
368  }
369  else {
370  // The format start character now requires us to look at at least the next single
371  // character (if there isn't another character, the % just gets eaten)
372  if (i < exportFormatString.size() - 1) {
373  ++i;
374  auto formatChar = exportFormatString.at(i);
375  QChar separatorChar = QLatin1Char('-');
376  // If this format type requires an additional char, read that now (or default to
377  // '-' if the format string ends)
378  if (formatChar == QLatin1Char('L') ||
379  formatChar == QLatin1Char('P')) {
380  if (i < exportFormatString.size() - 1) {
381  ++i;
382  separatorChar = exportFormatString.at(i);
383  }
384  }
385 
386  // Handle our format characters:
387  if (formatChar == QLatin1Char('F')) {
388  defaultFilename.append(fcstdBasename);
389  }
390  else if (formatChar == QLatin1Char('L')) {
391  defaultFilename.append(objectLabels.join(separatorChar));
392  }
393  else if (formatChar == QLatin1Char('P')) {
394  defaultFilename.append(parentLabels.join(separatorChar));
395  }
396  else if (formatChar == QLatin1Char('U')) {
397  defaultFilename.append(utcISO8601);
398  }
399  else if (formatChar == QLatin1Char('D')) {
400  defaultFilename.append(localISO8601);
401  }
402  else {
403  FC_WARN("When parsing default export filename format string, %"
404  << QString(formatChar).toStdString()
405  << " is not a known format string.");
406  }
407  }
408  }
409  }
410 
411  // Finally, clean the string so it's valid for all operating systems:
412  QString invalidCharacters = QLatin1String("/\\?%*:|\"<>");
413  for (const auto &c : invalidCharacters)
414  defaultFilename.replace(c,QLatin1String("_"));
415 
416  return defaultFilename;
417 }

419 void StdCmdExport::activated(int iMsg)
420 {
421  Q_UNUSED(iMsg);
422 
423  static QString lastExportFullPath = QString();
424  static App::DocumentObject* lastExportedObject = nullptr;
425  static bool lastExportUsedGeneratedFilename = true;
426  static QString lastExportFilterUsed = QString();
427  static Document* lastActiveDocument;
428 
429  auto selection = Gui::Selection().getObjectsOfType(App::DocumentObject::getClassTypeId());
430  if (selection.empty()) {
431  QMessageBox::warning(Gui::getMainWindow(),
432  QCoreApplication::translate("StdCmdExport", "No selection"),
433  QCoreApplication::translate("StdCmdExport", "Select the objects to export before choosing Export."));
434  return;
435  }
436 
437  // fill the list of registered suffixes
438  QStringList filterList;
439  std::map<std::string, std::string> filterMap = App::GetApplication().getExportFilters();
440  for (const auto &filter : filterMap) {
441  // ignore the project file format
442  if (filter.first.find("(*.FCStd)") == std::string::npos)
443  filterList << QString::fromStdString(filter.first);
444  }
445  QString formatList = filterList.join(QLatin1String(";;"));
446  Base::Reference<ParameterGrp> hPath =
447  App::GetApplication().GetUserParameter().GetGroup("BaseApp")->GetGroup("Preferences")->GetGroup("General");
448  QString selectedFilter = QString::fromStdString(hPath->GetASCII("FileExportFilter"));
449  if (!lastExportFilterUsed.isEmpty())
450  selectedFilter = lastExportFilterUsed;
451 
452  // Create a default filename for the export
453  // * If this is the first export this session default, generate a new default.
454  // * If this is a repeated export during the same session and file:
455  // * If the user accepted the default filename last time, regenerate a new
456  // default, potentially updating the object label.
457  // * If not, default to their previously-set export filename.
458  // * If this is an export of a different object than last time
459  QString defaultFilename = lastExportFullPath;
460 
461  bool filenameWasGenerated = false;
462  bool didActiveDocumentChange = lastActiveDocument != getActiveGuiDocument();
463  bool didExportedObjectChange = lastExportedObject != selection.front();
464  // We want to generate a new default name in four cases:
465  if (defaultFilename.isEmpty() || lastExportUsedGeneratedFilename || didActiveDocumentChange || didExportedObjectChange) {
466  // First, get the name and path of the current .FCStd file, if there is one:
467  QString docFilename = QString::fromUtf8(
468  App::GetApplication().getActiveDocument()->getFileName());
469 
470  // Find the default location for our exported file. Three possibilities:
471  QString defaultExportPath;
472  if (!lastExportFullPath.isEmpty()) {
473  QFileInfo fi(lastExportFullPath);
474  defaultExportPath = fi.path();
475  }
476  else if (!docFilename.isEmpty()) {
477  QFileInfo fi(docFilename);
478  defaultExportPath = fi.path();
479  }
480  else {
481  defaultExportPath = Gui::FileDialog::getWorkingDirectory();
482  }
483 
484  if (lastExportUsedGeneratedFilename || didActiveDocumentChange || didExportedObjectChange) { /*<- static, true on first call*/
485  defaultFilename = defaultExportPath + QLatin1Char('/') + createDefaultExportBasename();
486 
487  // Append the last extension used, if there is one.
488  if (!lastExportFullPath.isEmpty()) {
489  QFileInfo lastExportFile(lastExportFullPath);
490  if (!lastExportFile.suffix().isEmpty())
491  defaultFilename += QLatin1String(".") + lastExportFile.suffix();
492  }
493  filenameWasGenerated = true;
494  }
495  }
496 
497  // Launch the file selection modal dialog
498  QString fileName = FileDialog::getSaveFileName(getMainWindow(),
499  QObject::tr("Export file"), defaultFilename, formatList, &selectedFilter);
500  if (!fileName.isEmpty()) {
501  hPath->SetASCII("FileExportFilter", selectedFilter.toLatin1().constData());
502  lastExportFilterUsed = selectedFilter; // So we can select the same one next time
503  SelectModule::Dict dict = SelectModule::exportHandler(fileName, selectedFilter);
504  // export the files with the associated modules
505  for (SelectModule::Dict::iterator it = dict.begin(); it != dict.end(); ++it) {
506  getGuiApplication()->exportTo(it.key().toUtf8(),
507  getActiveGuiDocument()->getDocument()->getName(),
508  it.value().toLatin1());
509  }
510 
511  // Keep a record of if the user used our suggested generated filename. If they
512  // did, next time we can recreate it, which will update the object label if
513  // there is one.
514  QFileInfo defaultExportFI(defaultFilename);
515  QFileInfo thisExportFI(fileName);
516  if (filenameWasGenerated &&
517  thisExportFI.completeBaseName() == defaultExportFI.completeBaseName())
518  lastExportUsedGeneratedFilename = true;
519  else
520  lastExportUsedGeneratedFilename = false;
521 
522  lastExportFullPath = fileName;
523  lastActiveDocument = getActiveGuiDocument();
524  lastExportedObject = selection.front();
525  }
526 }

1232  for(auto &s : Selection().getCompleteSelection()) {
1233  if(s.pObject && s.pObject->isAttachedToDocument() && objSet.insert(s.pObject).second)
1234  sel.push_back(s.pObject);
1235  }
1236  if(sel.empty())
1237  return;
1238 
1239  bool hasXLink = false;
1240  Base::FileInfo fi(App::Application::getTempFileName());
1241  {
1242  auto all = App::Document::getDependencyList(sel);

1249  return;
1250  }

1252  hasXLink = App::PropertyXLink::hasXLink(sel,&unsaved);
1253  if(!unsaved.empty()) {
1254  QMessageBox::critical(getMainWindow(), QObject::tr("Unsaved document"),
1255  QObject::tr("The exported object contains external link. Please save the document"
1256  "at least once before exporting."));
1257  return;

1261  Base::ofstream str(fi, std::ios::out | std::ios::binary);
1262  App::Document* doc = sel.front()->getDocument();
1263  MergeDocuments mimeView(doc);
1264  doc->exportObjects(sel, str);
1265  str.close();
1266  }

1268  if (doc) {
1269  bool proceed = true;
1270  if(hasXLink && !doc->isSaved()) {
1271  auto ret = QMessageBox::question(getMainWindow(),
1272  qApp->translate("Std_DuplicateSelection","Object dependencies"),
1273  qApp->translate("Std_DuplicateSelection",
1274  "To link to external objects, the document must be saved at least once.\n"

1276  QMessageBox::Yes,QMessageBox::No);

1282  // restore objects from file and add to active document
1283  Base::ifstream str(fi, std::ios::in | std::ios::binary);
1284  MergeDocuments mimeView(doc);
