249 void CmdRobotAddToolShape::activated(int)
250 {
251  std::vector<App::DocumentObject*> robots =
252  getSelection().getObjectsOfType(Robot::RobotObject::getClassTypeId());
253  std::vector<App::DocumentObject*> shapes =
254  getSelection().getObjectsOfType(Base::Type::fromName("Part::Feature"));
255  std::vector<App::DocumentObject*> VRMLs =
256  getSelection().getObjectsOfType(Base::Type::fromName("App::VRMLObject"));
257 
258  if (robots.size() != 1 || (shapes.size() != 1 && VRMLs.size() != 1)) {
259  QMessageBox::warning(Gui::getMainWindow(),
260  QObject::tr("Wrong selection"),
261  QObject::tr("Select one robot and one shape or VRML object."));
262  return;
263  }
264 
265  std::string RoboName = robots.front()->getNameInDocument();
266  std::string ShapeName;
267  if (shapes.size() == 1) {
268  ShapeName = shapes.front()->getNameInDocument();
269  }
270  else {
271  ShapeName = VRMLs.front()->getNameInDocument();
272  }
273 
274  openCommand("Add tool to robot");
275  doCommand(Doc,
276  "App.activeDocument().%s.ToolShape = App.activeDocument().%s",
277  RoboName.c_str(),
278  ShapeName.c_str());
279  // doCommand(Gui,"Gui.activeDocument().hide(\"%s\")",ShapeName.c_str());
280  updateActive();
281  commitCommand();
282 }
