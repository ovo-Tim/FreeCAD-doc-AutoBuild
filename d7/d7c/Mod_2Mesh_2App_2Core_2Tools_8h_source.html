113 inline bool MeshSearchNeighbours::CheckDistToFacet(const MeshFacet& rclF)
114 {
115  bool bFound = false;
116 
117  for (PointIndex ulPIdx : rclF._aulPoints) {
118  if (!_rclPAry[ulPIdx].IsFlag(MeshPoint::MARKED)) {
119  if (Base::DistanceP2(_clCenter, _rclPAry[ulPIdx]) < _fMaxDistanceP2) {
120  bFound = true;
121  {
122  _aclResult.insert(ulPIdx);
123  _rclPAry[ulPIdx].SetFlag(MeshPoint::MARKED);
124  }
125  }
126  _aclOuter.insert(ulPIdx);
127  }
128  }
129 
130  return bFound;
131 }

138 inline bool MeshSearchNeighbours::TriangleCutsSphere(const MeshFacet& rclF) const
139 {
140  Base::Vector3f cP0 = _rclPAry[rclF._aulPoints[0]];
141  Base::Vector3f cP1 = _rclPAry[rclF._aulPoints[1]];
142  Base::Vector3f cP2 = _rclPAry[rclF._aulPoints[2]];
143 
144  Wm4::Vector3<float> akP0(cP0.x, cP0.y, cP0.z);
145  Wm4::Vector3<float> akP1(cP1.x, cP1.y, cP1.z);
146  Wm4::Vector3<float> akP2(cP2.x, cP2.y, cP2.z);
147 
148  Wm4::Triangle3<float> akTri(akP0, akP1, akP2);
149  Wm4::DistVector3Triangle3<float> akDistVecTri(_akSphere.Center, akTri);
150 
151  float fSqrDist = akDistVecTri.GetSquared();
152  float fRSqr = _akSphere.Radius * _akSphere.Radius;
153  return fSqrDist < fRSqr;
154 }
