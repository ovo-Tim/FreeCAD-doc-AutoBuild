497  static
498  void reorientCamera(SoCamera * cam, const SbRotation & rot)
499  {
500  if (!cam)
501  return;
502 
503  // Find global coordinates of focal point.
504  SbVec3f direction;
505  cam->orientation.getValue().multVec(SbVec3f(0, 0, -1), direction);
506  SbVec3f focalpoint = cam->position.getValue() +
507  cam->focalDistance.getValue() * direction;
508 
509  // Set new orientation value by accumulating the new rotation.
510  cam->orientation = rot * cam->orientation.getValue();
511 
512  // Reposition camera so we are still pointing at the same old focal point.
513  cam->orientation.getValue().multVec(SbVec3f(0, 0, -1), direction);
514  cam->position = focalpoint - cam->focalDistance.getValue() * direction;
515  }

517  static
518  void copyCameraSettings(SoCamera* cam1, SbRotation& rot_cam1, SbVec3f& pos_cam1,
519  SoCamera* cam2, SbRotation& rot_cam2, SbVec3f& pos_cam2)
520  {
521  Q_UNUSED(pos_cam2);
522 
523  // recompute the diff we have applied to the camera's orientation
524  SbRotation rot = cam1->orientation.getValue();
525  SbRotation dif = rot * rot_cam1.inverse();
526  rot_cam1 = rot;
527 
528  // copy the values
529  cam2->enableNotify(false);
530  cam2->nearDistance = cam1->nearDistance;
531  cam2->farDistance = cam1->farDistance;
532  cam2->focalDistance = cam1->focalDistance;
533  reorientCamera(cam2,dif);
534  rot_cam2 = cam2->orientation.getValue();
535 
536  // reverse engineer the translation part in wc
537  SbVec3f pos = cam1->position.getValue();
538  SbVec3f difpos = pos - pos_cam1;
539  pos_cam1 = pos;
540  // the translation in pixel coords
541  cam1->orientation.getValue().inverse().multVec(difpos,difpos);
542  // the translation again in wc for the second camera
543  cam2->orientation.getValue().multVec(difpos,difpos);
544  cam2->position.setValue(cam2->position.getValue()+difpos);
545 
546  if (cam1->getTypeId() == cam2->getTypeId()) {
547  if (cam1->getTypeId() == SoOrthographicCamera::getClassTypeId())
548  static_cast<SoOrthographicCamera*>(cam2)->height =
549  static_cast<SoOrthographicCamera*>(cam1)->height;
550  }
551 
552  cam2->enableNotify(true);
553  }
554  static
555  void syncCameraCB(void * data, SoSensor * s)
556  {
557  auto self = static_cast<ManualAlignment*>(data);
558  if (!self->myViewer)
559  return; // already destroyed
560  SoCamera* cam1 = self->myViewer->getViewer(0)->getSoRenderManager()->getCamera();
561  SoCamera* cam2 = self->myViewer->getViewer(1)->getSoRenderManager()->getCamera();
562  if (!cam1 || !cam2)
563  return; // missing camera
564  auto sensor = static_cast<SoNodeSensor*>(s);
565  SoNode* node = sensor->getAttachedNode();
566  if (node && node->getTypeId().isDerivedFrom(SoCamera::getClassTypeId())) {
567  if (node == cam1) {
568  Private::copyCameraSettings(cam1, self->d->rot_cam1, self->d->pos_cam1,
569  cam2, self->d->rot_cam2, self->d->pos_cam2);
570  self->myViewer->getViewer(1)->redraw();
571  }
572  else if (node == cam2) {
573  Private::copyCameraSettings(cam2, self->d->rot_cam2, self->d->pos_cam2,
574  cam1, self->d->rot_cam1, self->d->pos_cam1);
575  self->myViewer->getViewer(0)->redraw();
576  }
577  }

580  static Base::Placement
581  transformation2x2(const Base::Vector3d& plane1_base,
582  const Base::Vector3d& plane1_xaxis,
583  const Base::Vector3d& plane2_base,
584  const Base::Vector3d& plane2_xaxis)
585  {
586  // the transformation is:
587  // * move from plane1_base to plane2_base
588  // * rotate from plane1_zaxis to plane2_zaxis around plane2_base as center point
589  Base::Rotation rot(plane1_xaxis, plane2_xaxis);
590 
591  Base::Vector3d pln_base;
592  rot.multVec(plane1_base,pln_base);
593  Base::Vector3d dif = plane2_base - pln_base;
594  return {dif, rot};
595  }

597  static Base::Placement
598  transformation3x3(const Base::Vector3d& plane1_base,
599  const Base::Vector3d& plane1_zaxis,
600  const Base::Vector3d& plane1_xaxis,
601  const Base::Vector3d& plane2_base,
602  const Base::Vector3d& plane2_zaxis,
603  const Base::Vector3d& plane2_xaxis)
604  {
605  // the transformation is:
606  // * move from plane1_base to plane2_base
607  // * rotate from plane1_zaxis to plane2_zaxis around plane2_base as center point
608  Base::Rotation rot(plane1_zaxis, plane2_zaxis);
609 
610  // first transformation to align the plane normals and base points
611  Base::Vector3d dif1 = plane1_base;
612  rot.multVec(dif1,dif1);
613  dif1 = plane2_base - dif1;
614  Base::Placement plm1(dif1, rot);
615 
616  // second transformation to align the planes' x axes
617  Base::Vector3d pln_xaxis;
618  rot.multVec(plane1_xaxis,pln_xaxis);
619  Base::Rotation rot2(pln_xaxis, plane2_xaxis);
620  Base::Vector3d dif2 = plane2_base;
621  rot2.multVec(dif2,dif2);
622  dif2 = plane2_base - dif2;
623  Base::Placement plm2(dif2, rot2);
624  plm2 = plm2 * plm1;
625  return plm2;

1038 bool ManualAlignment::computeAlignment(const std::vector<PickedPoint>& movPts,

1069  Base::Vector3d n1 = (p2-p1) % (p3-p1);
1070  n1.Normalize();
1071

1077  Base::Vector3d n2 = (q2-q1) % (q3-q1);
1078  n2.Normalize();
