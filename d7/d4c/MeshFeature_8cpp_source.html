60 void Feature::onChanged(const App::Property* prop)
61 {
62  // if the placement has changed apply the change to the mesh data as well
63  if (prop == &this->Placement) {
64  this->Mesh.setTransform(this->Placement.getValue().toMatrix());
65  }
66  // if the mesh data has changed check and adjust the transformation as well
67  else if (prop == &this->Mesh) {
68  try {
69  Base::Placement p;
70  p.fromMatrix(this->Mesh.getTransform());
71  if (p != this->Placement.getValue()) {
72  this->Placement.setValue(p);
73  }
74  }
75  catch (const Base::ValueError&) {
76  }
77  }
78 
79  // Note: If the Mesh property has changed the property and this object are marked as 'touched'
80  // but no recomputation of this objects needs to be done because the Mesh property is regarded
81  // as output of a previous recomputation The mustExecute() method returns 0 in that case.
82  // However, objects that reference this object the Mesh property can be an input parameter.
83  // As this object and the property are touched such objects can check this and return a value 1
84  // (or -1) in their mustExecute() to be recomputed the next time the document recomputes itself.
85  DocumentObject::onChanged(prop);
86 }
