228 template<typename T>
229 auto toPointerVector(const std::vector<std::unique_ptr<T>>& vector)
230 {
231  std::vector<T*> vp(vector.size());
232 
233  std::transform(vector.begin(), vector.end(), vp.begin(), [](auto& p) {
234  return p.get();
235  });
236 
237  return vp;
238 }

245 template<typename T>
246 auto getSafeGeomLayerId(T geom)
247 {
248  int layerId = 0;
249 
250  if (geom->hasExtension(SketcherGui::ViewProviderSketchGeometryExtension::getClassTypeId())) {
251  auto vpext =
252  std::static_pointer_cast<const SketcherGui::ViewProviderSketchGeometryExtension>(
253  geom->getExtension(
254  SketcherGui::ViewProviderSketchGeometryExtension::getClassTypeId())
255  .lock());
256 
257  layerId = vpext->getVisualLayerId();
258  }
259 
260  return layerId;
261 }

267 template<typename T>
268 void setSafeGeomLayerId(T geom, int layerindex)
269 {
270  // create extension if none existing
271  if (!geom->hasExtension(SketcherGui::ViewProviderSketchGeometryExtension::getClassTypeId())) {
272  geom->setExtension(std::make_unique<SketcherGui::ViewProviderSketchGeometryExtension>());
273  }
274 
275  auto vpext = std::static_pointer_cast<SketcherGui::ViewProviderSketchGeometryExtension>(
276  geom->getExtension(SketcherGui::ViewProviderSketchGeometryExtension::getClassTypeId())
277  .lock());
278 
279  vpext->setVisualLayerId(layerindex);
280 }
