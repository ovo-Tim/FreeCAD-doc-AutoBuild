63  void validateBytes(XMLByte* const toFill, std::streamsize len)
64  {
65  QTextCodec* textCodec = QTextCodec::codecForName("UTF-8");
66  if (textCodec) {
67  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
68  const QString text = textCodec->toUnicode(reinterpret_cast<char*>(toFill),
69  static_cast<int>(len),
70  &state);
71  if (state.invalidChars > 0) {
72  // In case invalid characters were found decode back to 'utf-8' and replace
73  // them with '?'
74  // First, Qt replaces invalid characters with '\0' (see ConvertInvalidToNull)
75  // but Xerces doesn't like this because it handles this as termination. Thus,
76  // we have to go through the array and replace '\0' with '?'.
77  std::streamsize pos = 0;
78  QByteArray ba = textCodec->fromUnicode(text);
79  for (int i = 0; i < ba.length(); i++, pos++) {
80  if (pos < len && ba[i] == '\0') {
81  toFill[i] = '?';
82  }
83  }
84  }
85  }
86  }
87 };
88 #else

91  void validateBytes(XMLByte* const toFill, std::streamsize len)
92  {
93  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
94  QByteArray encodedString(reinterpret_cast<char*>(toFill), static_cast<int>(len));
95  auto toUtf16 = QStringDecoder(QStringDecoder::Utf8);
96  QString text = toUtf16(encodedString);
97  if (toUtf16.hasError()) {
98  // In case invalid characters were found decode back to 'utf-8' and replace
99  // them with '?'
100  // First, Qt replaces invalid characters with '\0'
101  // but Xerces doesn't like this because it handles this as termination. Thus,
102  // we have to go through the array and replace '\0' with '?'.
103  std::streamsize pos = 0;
104  auto fromUtf16 = QStringEncoder(QStringEncoder::Utf8);
105  QByteArray ba = fromUtf16(text);
106  for (int i = 0; i < ba.length(); i++, pos++) {
107  if (pos < len && ba[i] == '\0') {
108  toFill[i] = '?';

110  }
