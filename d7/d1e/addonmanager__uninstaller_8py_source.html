46 class AddonUninstaller(QObject):
47  """The core, non-GUI uninstaller class for non-macro addons. Usually instantiated
48  and moved to its own thread, otherwise it will block the GUI (if the GUI is
49  running) -- since all it does is delete files this is not a huge problem,
50  but in some cases the Addon might be quite large, and deletion may take a
51  non-trivial amount of time.
52 
53  In all cases in this class, the generic Python 'object' argument to the init
54  function is intended to be an Addon-like object that provides, at a minimum,
55  a 'name' attribute. The Addon manager uses the Addon class for this purpose,
56  but external code may use any other class that meets that criterion.
57 
58  Recommended Usage (when running with the GUI up, so you don't block the GUI thread):
59 
60  addon_to_remove = MyAddon() # Some class with 'name' attribute
61 
62  self.worker_thread = QThread()
63  self.uninstaller = AddonUninstaller(addon_to_remove)
64  self.uninstaller.moveToThread(self.worker_thread)
65  self.uninstaller.success.connect(self.removal_succeeded)
66  self.uninstaller.failure.connect(self.removal_failed)
67  self.uninstaller.finished.connect(self.worker_thread.quit)
68  self.worker_thread.started.connect(self.uninstaller.run)
69  self.worker_thread.start() # Returns immediately
70 
71  # On success, the connections above result in self.removal_succeeded being
72  emitted, and # on failure, self.removal_failed is emitted.
73 
74 
75  Recommended non-GUI usage (blocks until complete):
76 
77  addon_to_remove = MyAddon() # Some class with 'name' attribute
78  uninstaller = AddonInstaller(addon_to_remove)
79  uninstaller.run()
80 
81  """

99  def run(self) -> bool:
100  """Remove an addon. Returns True if the addon was removed cleanly, or False
101  if not. Emits either success or failure prior to returning."""
102  success = False
103  error_message = translate("AddonsInstaller", "An unknown error occurred")
104  if hasattr(self.addon_to_remove, "name") and self.addon_to_remove.name:
105  # Make sure we don't accidentally remove the Mod directory
106  path_to_remove = os.path.normpath(
107  os.path.join(self.installation_path, self.addon_to_remove.name)
108  )
109  if os.path.exists(path_to_remove) and not os.path.samefile(
110  path_to_remove, self.installation_path
111  ):
112  try:
113  self.run_uninstall_script(path_to_remove)
114  self.remove_extra_files(path_to_remove)
115  success = utils.rmdir(path_to_remove)
116  if (
117  hasattr(self.addon_to_remove, "contains_workbench")
118  and self.addon_to_remove.contains_workbench()
119  ):
120  self.addon_to_remove.desinstall_workbench()
121  except OSError as e:
122  error_message = str(e)
123  else:
124  error_message = translate(
125  "AddonsInstaller",
126  "Could not find addon {} to remove it.",
127  ).format(self.addon_to_remove.name)
128  if success:
129  self.success.emit(self.addon_to_remove)
130  else:
131  self.failure.emit(self.addon_to_remove, error_message)
132  self.addon_to_remove.set_status(Addon.Status.NOT_INSTALLED)
133  self.finished.emit()
134  return success

154  @staticmethod
155  def remove_extra_files(path_to_remove):
156  """When installing, an extra file called AM_INSTALLATION_DIGEST.txt may be
157  created, listing extra files that the installer put into place. Remove those
158  files."""
159  digest = os.path.join(path_to_remove, "AM_INSTALLATION_DIGEST.txt")
160  if not os.path.exists(digest):
161  return
162  with open(digest, encoding="utf-8") as f:
163  lines = f.readlines()
164  for line in lines:
165  stripped = line.strip()
166  if len(stripped) > 0 and stripped[0] != "#" and os.path.exists(stripped):
167  try:
168  os.unlink(stripped)
169  fci.Console.PrintMessage(
170  translate("AddonsInstaller", "Removed extra installed file {}").format(
171  stripped
172  )
173  + "\n"
174  )
175  except FileNotFoundError:
176  pass # Great, no need to remove then!
177  except OSError as e:
178  # Strange error to receive here, but just continue and print
179  # out an error to the console
180  fci.Console.PrintWarning(
181  translate(
182  "AddonsInstaller",
183  "Error while trying to remove extra installed file {}",
184  ).format(stripped)
185  + "\n"
186  )
187  fci.Console.PrintWarning(str(e) + "\n")

190 class MacroUninstaller(QObject):
191  """The core, non-GUI uninstaller class for macro addons. May be run directly on
192  the GUI thread if desired, since macros are intended to be relatively small and
193  shouldn't have too many files to delete. However, it is a QObject so may also be
194  moved into a QThread -- see AddonUninstaller documentation for details of that
195  implementation.
196 
197  The Python object passed in is expected to provide a "macro" subobject,
198  which itself is required to provide at least a "filename" attribute, and may also
199  provide an "icon", "xpm", and/or "other_files" attribute. All filenames provided
200  by those attributes are expected to be relative to the installed location of the
201  "filename" macro file (usually the main FreeCAD user macros directory)."""
202 
203  # Signals: success and failure Emitted when the removal process is complete. The
204  # object emitted is the object that the removal was requested for.
205  success = Signal(object)
206  failure = Signal(object, str)
207 
208  # Finished: regardless of the outcome, this is emitted when all work that is
209  # going to be done is done (i.e. whatever thread this is running in can quit).
210  finished = Signal()

212  def __init__(self, addon):
213  super().__init__()
214  self.installation_location = fci.DataPaths().macro_dir
215  self.addon_to_remove = addon
216  if (
217  not hasattr(self.addon_to_remove, "macro")
218  or not self.addon_to_remove.macro
219  or not hasattr(self.addon_to_remove.macro, "filename")
220  or not self.addon_to_remove.macro.filename
221  ):
222  raise InvalidAddon()

224  def run(self):
225  """Execute the removal process."""
226  success = True
227  errors = []
228  directories = set()
229  for f in self._get_files_to_remove():
230  normed = os.path.normpath(f)
231  if os.path.isabs(normed):
232  full_path = normed
233  else:
234  full_path = os.path.join(self.installation_location, normed)
235  if "/" in f:
236  directories.add(os.path.dirname(full_path))
237  try:
238  os.unlink(full_path)
239  fci.Console.PrintLog(f"Removed macro file {full_path}\n")
240  except FileNotFoundError:
241  pass # Great, no need to remove then!
242  except OSError as e:
243  # Probably permission denied, or something like that
244  errors.append(
245  translate(
246  "AddonsInstaller",
247  "Error while trying to remove macro file {}:",
248  ).format(full_path)
249  + " "
250  + str(e)
251  )
252  success = False
253  except Exception:
254  # Generic catch-all, just in case (because failure to catch an exception
255  # here can break things pretty badly)
256  success = False
257 
258  self._cleanup_directories(directories)
259 
260  if success:
261  self.success.emit(self.addon_to_remove)
262  else:
263  self.failure.emit(self.addon_to_remove, "\n".join(errors))
264  self.addon_to_remove.set_status(Addon.Status.NOT_INSTALLED)
265  self.finished.emit()

267  def _get_files_to_remove(self) -> List[str]:
268  """Get the list of files that should be removed"""
269  manifest_file = os.path.join(
270  self.installation_location, self.addon_to_remove.macro.filename + ".manifest"
271  )
272  if os.path.exists(manifest_file):
273  with open(manifest_file, "r", encoding="utf-8") as f:
274  manifest_data = f.read()
275  manifest = json.loads(manifest_data)
276  manifest.append(manifest_file) # Remove the manifest itself as well
277  return manifest
278  files_to_remove = [self.addon_to_remove.macro.filename]
279  if self.addon_to_remove.macro.icon:
280  files_to_remove.append(self.addon_to_remove.macro.icon)
281  if self.addon_to_remove.macro.xpm:
282  files_to_remove.append(self.addon_to_remove.macro.name.replace(" ", "_") + "_icon.xpm")
283  for f in self.addon_to_remove.macro.other_files:
284  files_to_remove.append(f)
285  return files_to_remove
