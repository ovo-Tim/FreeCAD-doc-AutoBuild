224 bool TaskTransformedParameters::originalSelected(const Gui::SelectionChanges& msg)
225 {
226  if (msg.Type == Gui::SelectionChanges::AddSelection
227  && ((selectionMode == SelectionMode::AddFeature)
228  || (selectionMode == SelectionMode::RemoveFeature))) {
229 
230  if (strcmp(msg.pDocName, getObject()->getDocument()->getName()) != 0) {
231  return false;
232  }
233 
234  PartDesign::Transformed* pcTransformed = getObject();
235  App::DocumentObject* selectedObject =
236  pcTransformed->getDocument()->getObject(msg.pObjectName);
237  if (selectedObject->isDerivedFrom<PartDesign::FeatureAddSub>()) {
238 
239  // Do the same like in TaskDlgTransformedParameters::accept() but without doCommand
240  std::vector<App::DocumentObject*> originals = pcTransformed->Originals.getValues();
241  auto or_iter = std::find(originals.begin(), originals.end(), selectedObject);
242  if (selectionMode == SelectionMode::AddFeature) {
243  if (or_iter == originals.end()) {
244  originals.push_back(selectedObject);
245  addObject(selectedObject);
246  }
247  else {
248  return false; // duplicate selection
249  }
250  }
251  else {
252  if (or_iter != originals.end()) {
253  originals.erase(or_iter);
254  removeObject(selectedObject);
255  }
256  else {
257  return false;
258  }
259  }
260  setupTransaction();
261  pcTransformed->Originals.setValues(originals);
262  recomputeFeature();
263 
264  return true;
265  }
266  }
267 
268  return false;
269 }

378  std::vector<App::DocumentObject*> originals = pcTransformed->Originals.getValues();
379  int currentRow = ui->listWidgetFeatures->currentRow();
380  if (currentRow < 0) {

385  setupTransaction();

594 void TaskTransformedParameters::indexesMoved()
595 {

597  if (!model) {
598  return;
599  }
600

602  std::vector<App::DocumentObject*> originals = pcTransformed->Originals.getValues();
603 
604  QByteArray name;
605  int rows = model->rowCount();

607  QModelIndex index = model->index(i, 0);

610  }
611 
612  setupTransaction();
