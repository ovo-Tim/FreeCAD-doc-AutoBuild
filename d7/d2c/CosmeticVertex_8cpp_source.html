146 void CosmeticVertex::Restore(Base::XMLReader &reader)
147 {
148  if (!CosmeticVertex::restoreCosmetic()) {
149  return;
150  }
151  TechDraw::Vertex::Restore(reader);
152  reader.readElement("PermaPoint");
153  permaPoint.x = reader.getAttributeAsFloat("X");
154  permaPoint.y = reader.getAttributeAsFloat("Y");
155  permaPoint.z = reader.getAttributeAsFloat("Z");
156  reader.readElement("LinkGeom");
157  linkGeom = reader.getAttributeAsInteger("value");
158  reader.readElement("Color");
159  std::string temp = reader.getAttribute("value");
160  color.fromHexString(temp);
161  reader.readElement("Size");
162  size = reader.getAttributeAsFloat("value");
163  reader.readElement("Style");
164  style = reader.getAttributeAsInteger("value");
165  reader.readElement("Visible");
166  visible = (int)reader.getAttributeAsInteger("value")==0?false:true;
167  reader.readElement("Tag");
168  temp = reader.getAttribute("value");
169  boost::uuids::string_generator gen;
170  boost::uuids::uuid u1 = gen(temp);
171  tag = u1;
172 }

180 Base::Vector3d CosmeticVertex::rotatedAndScaled(const double scale, const double rotDegrees)
181 {
182  Base::Vector3d scaledPoint = scaled(scale);
183  if (rotDegrees != 0.0) {
184  // invert the Y coordinate so the rotation math works out
185  // the stored point is inverted
186  scaledPoint = DU::invertY(scaledPoint);
187  scaledPoint.RotateZ(rotDegrees * M_PI / DegreesHalfCircle);
188  scaledPoint = DU::invertY(scaledPoint);
189  }
190  return scaledPoint;
191 }

196 Base::Vector3d CosmeticVertex::makeCanonicalPoint(DrawViewPart* dvp, Base::Vector3d point, bool unscale)
197 {
198  double rotDeg = dvp->Rotation.getValue();
199 
200  Base::Vector3d result = point;
201  if (rotDeg != 0.0) {
202  // unrotate the point
203  double rotRad = rotDeg * M_PI / DegreesHalfCircle;
204  // we always rotate around the origin.
205  result.RotateZ(-rotRad);
206  }
207 
208  if (unscale) {
209  // unrotated point is scaled and we need to unscale it (the normal situation)
210  double scale = dvp->getScale();
211  return result / scale;
212  }
213 
214  // return the unrotated version of input point without unscaling
215  return result;
216 }

238 void CosmeticVertex::createNewTag()
239 {
240  // Initialize a random number generator, to avoid Valgrind false positives.
241  // The random number generator is not threadsafe so we guard it. See
242  // https://www.boost.org/doc/libs/1_62_0/libs/uuid/uuid.html#Design%20notes
243  static boost::mt19937 ran;
244  static bool seeded = false;
245  static boost::mutex random_number_mutex;
246 
247  boost::lock_guard<boost::mutex> guard(random_number_mutex);
248 
249  if (!seeded) {
250  ran.seed(static_cast<unsigned int>(std::time(nullptr)));
251  seeded = true;
252  }
253  static boost::uuids::basic_random_generator<boost::mt19937> gen(&ran);
254 
255  tag = gen();
256 }
