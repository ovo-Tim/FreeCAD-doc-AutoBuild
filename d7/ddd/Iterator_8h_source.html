456 inline const MeshGeomFacet& MeshFacetIterator::Dereference()
457 {
458  MeshFacet rclF = *_clIter;
459  const PointIndex* paulPt = &(_clIter->_aulPoints[0]);
460  Base::Vector3f* pclPt = _clFacet._aclPoints;
461  *(pclPt++) = _rclPAry[*(paulPt++)];
462  *(pclPt++) = _rclPAry[*(paulPt++)];
463  *pclPt = _rclPAry[*paulPt];
464  _clFacet._ulProp = rclF._ulProp;
465  _clFacet._ucFlag = rclF._ucFlag;
466  _clFacet.NormalInvalid();
467  if (_bApply) {
468  _clFacet._aclPoints[0] = _clTrf * _clFacet._aclPoints[0];
469  _clFacet._aclPoints[1] = _clTrf * _clFacet._aclPoints[1];
470  _clFacet._aclPoints[2] = _clTrf * _clFacet._aclPoints[2];
471  }
472  return _clFacet;
473 }
