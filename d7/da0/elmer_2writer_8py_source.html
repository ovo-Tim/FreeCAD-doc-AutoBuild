118  def _handleUnits(self):
119  # Elmer solver writer no longer uses FreeCAD unit system
120  # to retrieve units for writing the sif file
121  #
122  # ATM Elmer writer uses SI units only
123  #
124  # see forum topic: https://forum.freecad.org/viewtopic.php?f=18&t=70150
125  #
126  # TODO: adapt method and comment
127  # should be only one system for all solver and not in each solver
128  # https://forum.freecad.org/viewtopic.php?t=47895
129  # https://forum.freecad.org/viewtopic.php?t=48451
130  # https://forum.freecad.org/viewtopic.php?f=10&t=48642
131  # The FreeCAD unit schema is only used to determine the schema number
132  # all definition are done here ATM
133  # keep in mind a unit schema might not be consistent:
134  # Length could be mm and Area could be m2 and Volume could be cm3
135  # as long as only the seven base units are retrieved from a unit schema
136  # the units are consistent
137  # TODO retrieve the seven base units from FreeCAD unit schema
138  # instead of hard coding them here for a second once
139  self.unit_schema = Units.Scheme.SI1
140  self.unit_system = { # standard FreeCAD Base units = unit schema 0
141  "L": "m",
142  "M": "kg",
143  "T": "s",
144  "I": "A",
145  "O": "K",
146  "N": "mol",
147  "J": "cd",
148  }
149  param = ParamGet("User parameter:BaseApp/Preferences/Units")
150  self.unit_schema = param.GetInt("UserSchema", Units.Scheme.SI1)
151  if self.unit_schema == Units.Scheme.SI1:
152  Console.PrintMessage(
153  "The FreeCAD standard unit schema mm/kg/s is used. "
154  "Elmer sif-file writing is however done in SI units.\n"
155  )
156  elif self.unit_schema == Units.Scheme.SI2:
157  Console.PrintMessage(
158  "The SI unit schema m/kg/s is used. "
159  "Elmer sif-file writing is done in SI-units.\n"
160  )
161  self.unit_system = {
162  "L": "m",
163  "M": "kg",
164  "T": "s",
165  "I": "A",
166  "O": "K",
167  "N": "mol",
168  "J": "cd",
169  }
170  elif self.unit_schema == Units.Scheme.FemMilliMeterNewton:
171  # see also unit comment in calculix writer
172  Console.PrintMessage(
173  "The FEM unit schema mm/N/s is used. "
174  "Elmer sif-file writing is however done in SI units.\n"
175  )
176  self.unit_system = {
177  "L": "m",
178  "M": "kg",
179  "T": "s",
180  "I": "A",
181  "O": "K",
182  "N": "mol",
183  "J": "cd",
184  }
185  elif (
186  self.unit_schema > Units.Scheme.SI2
187  and self.unit_schema != Units.Scheme.FemMilliMeterNewton
188  ):
189  Console.PrintMessage(
190  "Unit schema: {} not supported by Elmer writer. "
191  "The FreeCAD standard unit schema mm/kg/s is used. "
192  "Elmer sif-file writing is done in Standard FreeCAD units.\n".format(
193  Units.listSchemas(self.unit_schema)
194  )

201  def convert(self, quantityStr, unit):
202  quantity = Units.Quantity(quantityStr)
203  for key, setting in self.unit_system.items():
204  unit = unit.replace(key, setting)
205  return float(quantity.getValueAs(unit))

216  def _writeMesh(self):
217  mesh = self.getSingleMember("Fem::FemMeshObject")
218  unvPath = os.path.join(self.directory, "mesh.unv")
219  groups = []
220  groups.extend(self._builder.getBodyNames())
221  groups.extend(self._builder.getBoundaryNames())
222  self._exportToUnv(groups, mesh, unvPath)
223  if self.testmode:
224  Console.PrintMessage(
225  "Solver Elmer testmode, ElmerGrid will not be used. It might not be installed.\n"
226  )
227  else:
228  binary = settings.get_binary("ElmerGrid")
229  num_cores = settings.get_cores("ElmerGrid")
230  if binary is None:
231  raise WriteError("Could not find ElmerGrid binary.")
232  # for multithreading we first need a normal mesh creation run
233  # then a second to split the mesh into the number of used cores
234  argsBasic = [binary, _ELMERGRID_IFORMAT, _ELMERGRID_OFORMAT, unvPath]
235  args = argsBasic
236  args.extend(["-out", self.directory])
237  if system() == "Windows":
238  subprocess.call(
239  args, stdout=subprocess.DEVNULL, startupinfo=femutils.startProgramInfo("hide")
240  )
241  else:
242  subprocess.call(args, stdout=subprocess.DEVNULL)
243  if num_cores > 1:
244  args = argsBasic
245  args.extend(["-partdual", "-metiskway", str(num_cores), "-out", self.directory])
246  if system() == "Windows":
247  subprocess.call(
248  args,
249  stdout=subprocess.DEVNULL,
250  startupinfo=femutils.startProgramInfo("hide"),
251  )
252  else:
253  subprocess.call(args, stdout=subprocess.DEVNULL)

300  def _handleRedifinedConstants(self):
301  """
302  redefine constants in self.constsdef according constant redefine objects
303  """
304  objs = self.getMember("Fem::ConstantVacuumPermittivity")
305  if len(objs) == 1:

318  def _handleSimulation(self):
319  # check if we need to update the equation
320  self._updateSimulation(self.solver)
321  # output the equation parameters
322  # first check what equations we have
323 
324  # hasHeat ist not used, thus commented ATM
325  # hasHeat = False
326  # for equation in self.solver.Group:
327  # if femutils.is_of_type(equation, "Fem::EquationElmerHeat"):
328  # hasHeat = True
329 
330  self._simulation("Coordinate System", self.solver.CoordinateSystem)
331  self._simulation("Coordinate Mapping", (1, 2, 3))
332  # Elmer uses SI base units, but our mesh is in mm, therefore we must tell
333  # the solver that we have another scale
334  self._simulation("Coordinate Scaling", 0.001)
335  self._simulation("Simulation Type", self.solver.SimulationType)
336  if self.solver.SimulationType == "Steady State":
337  self._simulation("Steady State Max Iterations", self.solver.SteadyStateMaxIterations)
338  self._simulation("Steady State Min Iterations", self.solver.SteadyStateMinIterations)
339  if self.solver.SimulationType == "Scanning" or self.solver.SimulationType == "Transient":
340  self._simulation("BDF Order", self.solver.BDFOrder)
341  self._simulation("Output Intervals", self.solver.OutputIntervals)
342  self._simulation("Timestep Intervals", self.solver.TimestepIntervals)
343  self._simulation("Timestep Sizes", self.solver.TimestepSizes)
344  self._simulation("Timestepping Method", "BDF")
345  self._simulation("Use Mesh Names", True)
