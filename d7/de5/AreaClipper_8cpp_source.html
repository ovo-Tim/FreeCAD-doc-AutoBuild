48 static void AddVertex(const CVertex& vertex, const CVertex* prev_vertex)
49 {
50  if (vertex.m_type == 0 || prev_vertex == NULL) {
51  AddPoint(DoubleAreaPoint(vertex.m_p.x * CArea::m_units, vertex.m_p.y * CArea::m_units));
52  }
53  else {
54  if (vertex.m_p != prev_vertex->m_p) {
55  double phi, dphi, dx, dy;
56  int Segments;
57  int i;
58  double ang1, ang2, phit;
59 
60  dx = (prev_vertex->m_p.x - vertex.m_c.x) * CArea::m_units;
61  dy = (prev_vertex->m_p.y - vertex.m_c.y) * CArea::m_units;
62 
63  ang1 = atan2(dy, dx);
64  if (ang1 < 0) {
65  ang1 += 2.0 * PI;
66  }
67  dx = (vertex.m_p.x - vertex.m_c.x) * CArea::m_units;
68  dy = (vertex.m_p.y - vertex.m_c.y) * CArea::m_units;
69  ang2 = atan2(dy, dx);
70  if (ang2 < 0) {
71  ang2 += 2.0 * PI;
72  }
73 
74  if (vertex.m_type == -1) { // clockwise
75  if (ang2 > ang1) {
76  phit = 2.0 * PI - ang2 + ang1;
77  }
78  else {
79  phit = ang1 - ang2;
80  }
81  }
82  else { // counter_clockwise
83  if (ang1 > ang2) {
84  phit = -(2.0 * PI - ang1 + ang2);
85  }
86  else {
87  phit = -(ang2 - ang1);
88  }
89  }
90 
91  // what is the delta phi to get an accuracy of aber
92  double radius = sqrt(dx * dx + dy * dy);
93  dphi = 2 * acos((radius - CArea::m_accuracy) / radius);
94 
95  // set the number of segments
96  if (phit > 0) {
97  Segments = (int)ceil(phit / dphi);
98  }
99  else {
100  Segments = (int)ceil(-phit / dphi);
101  }
102 
103  if (Segments < CArea::m_min_arc_points) {
104  Segments = CArea::m_min_arc_points;
105  }
106  // if (Segments > CArea::m_max_arc_points)
107  // Segments=CArea::m_max_arc_points;
108 
109  dphi = phit / (Segments);
110 
111  double px = prev_vertex->m_p.x * CArea::m_units;
112  double py = prev_vertex->m_p.y * CArea::m_units;
113 
114  for (i = 1; i <= Segments; i++) {
115  dx = px - vertex.m_c.x * CArea::m_units;
116  dy = py - vertex.m_c.y * CArea::m_units;
117  phi = atan2(dy, dx);
118 
119  double nx = vertex.m_c.x * CArea::m_units + radius * cos(phi - dphi);
120  double ny = vertex.m_c.y * CArea::m_units + radius * sin(phi - dphi);
121 
122  AddPoint(DoubleAreaPoint(nx, ny));
123 
124  px = nx;
125  py = ny;
126  }
127  }
128  }
