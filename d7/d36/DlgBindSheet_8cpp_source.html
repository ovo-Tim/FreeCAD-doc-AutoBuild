42 DlgBindSheet::DlgBindSheet(Sheet* sheet, const std::vector<Range>& ranges, QWidget* parent)
43  : QDialog(parent)
44  , sheet(sheet)
45  , range(ranges.front())
46  , ui(new Ui::DlgBindSheet)
47 {
48  ui->setupUi(this);
49  // remove the automatic help button in dialog title since we don't use it
50  setWindowFlag(Qt::WindowContextHelpButtonHint, false);
51 
52  std::string toStart, toEnd;
53  ExpressionPtr pStart, pEnd;
54  App::ObjectIdentifier bindingTarget;
55  PropertySheet::BindingType type = sheet->getCellBinding(range, &pStart, &pEnd, &bindingTarget);
56  if (type == PropertySheet::BindingNone) {
57  if (ranges.size() > 1) {
58  toStart = ranges.back().from().toString();
59  toEnd = ranges.back().to().toString();
60  }
61  else {
62  CellAddress target(range.to().row() ? 0 : range.to().row() + 1, range.from().col());
63  toStart = target.toString();
64  target.setRow(target.row() + range.to().row() - range.from().row());
65  target.setCol(target.col() + range.to().col() - range.from().col());
66  toEnd = target.toString();
67  }
68  ui->btnDiscard->setDisabled(true);
69  }
70  else {
71  ui->lineEditFromStart->setReadOnly(true);
72  ui->lineEditFromEnd->setReadOnly(true);
73  ui->checkBoxHREF->setChecked(type == PropertySheet::BindingHiddenRef);
74  assert(pStart && pEnd);
75  if (!pStart->hasComponent() && pStart->isDerivedFrom(StringExpression::getClassTypeId())) {
76  toStart = static_cast<StringExpression*>(pStart.get())->getText();
77  }
78  else {
79  toStart = "=";
80  toStart += pStart->toString();
81  }
82  if (!pEnd->hasComponent() && pEnd->isDerivedFrom(StringExpression::getClassTypeId())) {
83  toEnd = static_cast<StringExpression*>(pEnd.get())->getText();
84  }
85  else {
86  toEnd = "=";
87  toEnd += pEnd->toString();
88  }
89  }
90 
91  ui->lineEditFromStart->setText(QString::fromLatin1(range.from().toString().c_str()));
92  ui->lineEditFromEnd->setText(QString::fromLatin1(range.to().toString().c_str()));
93 
94  ui->lineEditToStart->setDocumentObject(sheet, false);
95  ui->lineEditToStart->setPrefix('=');
96  ui->lineEditToEnd->setDocumentObject(sheet, false);
97  ui->lineEditToEnd->setPrefix('=');
98 
99  ui->lineEditToStart->setText(QLatin1String(toStart.c_str()));
100  ui->lineEditToEnd->setText(QLatin1String(toEnd.c_str()));
101 
102  ui->comboBox->addItem(
103  QString::fromLatin1(". (%1)").arg(QString::fromUtf8(sheet->Label.getValue())),
104  QByteArray(""));
105 
106  App::DocumentObject* target = bindingTarget.getDocumentObject();
107  for (auto obj : sheet->getDocument()->getObjectsOfType<Sheet>()) {
108  if (obj == sheet) {
109  continue;
110  }
111  QString label;
112  if (obj->Label.getStrValue() != obj->getNameInDocument()) {
113  label =
114  QString::fromLatin1("%1 (%2)").arg(QString::fromLatin1(obj->getNameInDocument()),
115  QString::fromUtf8(obj->Label.getValue()));
116  }
117  else {
118  label = QLatin1String(obj->getNameInDocument());
119  }
120  ui->comboBox->addItem(label, QByteArray(obj->getNameInDocument()));
121  if (obj == target) {
122  ui->comboBox->setCurrentIndex(ui->comboBox->count() - 1);
123  }
124  }
125  for (auto doc : GetApplication().getDocuments()) {
126  if (doc == sheet->getDocument()) {
127  continue;
128  }
129  for (auto obj : sheet->getDocument()->getObjectsOfType<Sheet>()) {
130  if (obj == sheet) {
131  continue;
132  }
133  std::string fullname = obj->getFullName();
134  QString label;
135  if (obj->Label.getStrValue() != obj->getNameInDocument()) {
136  label =
137  QString::fromLatin1("%1 (%2)").arg(QString::fromLatin1(fullname.c_str()),
138  QString::fromUtf8(obj->Label.getValue()));
139  }
140  else {
141  label = QLatin1String(fullname.c_str());
142  }
143  ui->comboBox->addItem(label, QByteArray(fullname.c_str()));
144  if (obj == target) {
145  ui->comboBox->setCurrentIndex(ui->comboBox->count() - 1);
146  }
147  }
148  }
149 
150  connect(ui->btnDiscard, &QPushButton::clicked, this, &DlgBindSheet::onDiscard);
151 }

158 void DlgBindSheet::accept()
159 {
160  bool commandActive = false;
161  try {
162  const char* ref = ui->comboBox->itemData(ui->comboBox->currentIndex())
163  .toByteArray()
164  .constData(); // clazy:exclude=returning-data-from-temporary
165  auto obj = sheet;
166  if (ref[0]) {
167  const char* sep = strchr(ref, '#');
168  if (sep) {
169  std::string docname(ref, sep);
170  auto doc = GetApplication().getDocument(docname.c_str());
171  if (!doc) {
172  FC_THROWM(Base::RuntimeError, "Cannot find document " << docname);
173  }
174  obj = Base::freecad_dynamic_cast<Sheet>(doc->getObject(sep + 1));
175  }
176  else {
177  obj = Base::freecad_dynamic_cast<Sheet>(sheet->getDocument()->getObject(ref));
178  }
179  if (!obj) {
180  FC_THROWM(Base::RuntimeError, "Cannot find Spreadsheet '" << ref << "'");
181  }
182  }
183 
184  auto checkAddress = [](std::string& addr, CellAddress& cell, bool quote) {
185  std::string copy(addr);
186  boost::to_upper(copy);
187  cell = App::stringToAddress(copy.c_str(), true);
188  if (!cell.isValid()) {
189  std::string msg("Invalid cell: ");
190  msg += addr;
191  throw Base::ValueError(msg.c_str());
192  }
193  if (quote) {
194  addr = std::string("<<") + copy + ">>";
195  }
196  else {
197  addr = copy;
198  }
199  };
200 
201  CellAddress fromCellStart, fromCellEnd, toCellStart, toCellEnd;
202  std::string fromStart(ui->lineEditFromStart->text().trimmed().toLatin1().constData());
203  std::string fromEnd(ui->lineEditFromEnd->text().trimmed().toLatin1().constData());
204  checkAddress(fromStart, fromCellStart, false);
205  checkAddress(fromEnd, fromCellEnd, false);
206 
207  std::string toStart(ui->lineEditToStart->text().trimmed().toLatin1().constData());
208  if (boost::starts_with(toStart, "=")) {
209  toStart.erase(toStart.begin());
210  }
211  else {
212  checkAddress(toStart, toCellStart, true);
213  }
214 
215  std::string toEnd(ui->lineEditToEnd->text().trimmed().toLatin1().constData());
216  if (boost::starts_with(toEnd, "=")) {
217  toEnd.erase(toEnd.begin());
218  }
219  else {
220  checkAddress(toEnd, toCellEnd, true);
221  if (toCellStart.isValid()) {
222  App::Range fromRange(fromCellStart, fromCellEnd, true);
223  App::Range toRange(toCellStart, toCellEnd, true);
224  if (fromRange.size() != toRange.size()) {
225  auto res = QMessageBox::warning(this,
226  tr("Bind cells"),
227  tr("Source and target cell count mismatch. "
228  "Partial binding may still work.\n\n"
229  "Do you want to continue?"),
230  QMessageBox::Yes | QMessageBox::No);
231  if (res == QMessageBox::No) {
232  return;
233  }
234  }
235  }
236  }
237 
238  Gui::Command::openCommand("Bind cells");
239  commandActive = true;
240 
241  if (ui->checkBoxHREF->isChecked()) {
242  Gui::cmdAppObjectArgs(sheet,
243  "setExpression('.cells.Bind.%s.%s', None)",
244  fromStart,
245  fromEnd);
246  Gui::cmdAppObjectArgs(
247  sheet,
248  "setExpression('.cells.BindHiddenRef.%s.%s', 'hiddenref(tuple(%s.cells, %s, %s))')",
249  fromStart,
250  fromEnd,
251  ref,
252  toStart,
253  toEnd);
254  }
255  else {
256  Gui::cmdAppObjectArgs(sheet,
257  "setExpression('.cells.BindHiddenRef.%s.%s', None)",
258  fromStart,
259  fromEnd);
260  Gui::cmdAppObjectArgs(sheet,
261  "setExpression('.cells.Bind.%s.%s', 'tuple(%s.cells, %s, %s)')",
262  fromStart,
263  fromEnd,
264  ref,
265  toStart,
266  toEnd);
267  }
268  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
269  Gui::Command::commitCommand();
270  QDialog::accept();
271  }
272  catch (Base::Exception& e) {
273  e.ReportException();
274  QMessageBox::critical(this,
275  tr("Bind Spreadsheet Cells"),
276  tr("Error:\n") + QString::fromUtf8(e.what()));
277  if (commandActive) {
278  Gui::Command::abortCommand();
279  }
280  }
281 }
