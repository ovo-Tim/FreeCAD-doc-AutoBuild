76 bool OpenGLBuffer::isVBOSupported(uint32_t ctx)
77 {
78  auto glue = cc_glglue_instance(ctx);
79  if (!glue || !cc_glglue_has_vertex_buffer_object(glue))
80  return false;
81  const GLubyte * str = glGetString(GL_EXTENSIONS);
82  if (!str)
83  return false;
84  std::string ext = reinterpret_cast<const char*>(str);
85  return (ext.find("GL_ARB_vertex_buffer_object") != std::string::npos);
86 }
