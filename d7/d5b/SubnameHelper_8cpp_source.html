101 // A.B.C. returns A.B.C
102 std::string SubnameHelper::pruneLastTerm(const std::string& inString)
103 {
104  auto result {inString};
105  if (result.back() == '.') {
106  // remove the trailing dot
107  result = result.substr(0, result.length() - 1);
108  }
109 
110  size_t lastDotPos = result.rfind('.');
111  if (lastDotPos != std::string::npos) {
112  result = result.substr(0, lastDotPos + 1);
113  }
114  else {
115  // no dot in string, remove everything!
116  result = "";
117  }
118 
119  return result;
120 }

125 std::string SubnameHelper::removeGeometryTerm(const std::string& longSubname)
126 {
127  auto lastTerm = getLastTerm(longSubname);
128  if (longSubname.empty() || longSubname.back() == '.') {
129  // not a geometric reference
130  return longSubname; // need a copy?
131  }
132 
133  // brute force check for geometry names in the last term
134  auto pos = lastTerm.find("Vertex");
135  if (pos != std::string::npos) {
136  return pruneLastTerm(longSubname);
137  }
138 
139  pos = lastTerm.find("Edge");
140  if (pos != std::string::npos) {
141  return pruneLastTerm(longSubname);
142  }
143 
144  pos = lastTerm.find("Face");
145  if (pos != std::string::npos) {
146  return pruneLastTerm(longSubname);
147  }
148 
149  pos = lastTerm.find("Shell");
150  if (pos != std::string::npos) {
151  return pruneLastTerm(longSubname);
152  }
153 
154  pos = lastTerm.find("Solid");
155  if (pos != std::string::npos) {
156  return pruneLastTerm(longSubname);
157  }
158 
159  return longSubname;
