36 def heal(objlist=None, delete=True, reparent=True):
37  """heal([objlist],[delete],[reparent])
38 
39  Recreate Draft objects that are damaged, for example if created from an
40  earlier version. If ran without arguments, all the objects in the document
41  will be healed if they are damaged.
42 
43  Parameters
44  ----------
45  objlist : list
46 
47  delete : Base.Vector or list of Base.Vector
48  If delete is True, the damaged objects are deleted (default).
49 
50  reparent : bool
51  If reparent is True (default), new objects go at the very same place
52  in the tree than their original.
53  """
54 
55  auto = False
56 
57  if not objlist:
58  objlist = App.ActiveDocument.Objects
59  print("Automatic mode: Healing whole document...")
60  auto = True
61  else:
62  print("Manual mode: Force-healing selected objects...")
63 
64  if not isinstance(objlist,list):
65  objlist = [objlist]
66 
67  dellist = []
68  got = False
69 
70  for obj in objlist:
71  dtype = utils.get_type(obj)
72  ftype = obj.TypeId
73  if ftype in ["Part::FeaturePython","App::FeaturePython","Part::Part2DObjectPython"]:
74  proxy = obj.Proxy
75  if hasattr(obj,"ViewObject"):
76  if hasattr(obj.ViewObject,"Proxy"):
77  proxy = obj.ViewObject.Proxy
78  if (proxy == 1) or (dtype in ["Unknown","Part"]) or (not auto):
79  got = True
80  dellist.append(obj.Name)
81  props = obj.PropertiesList
82  if ("Dimline" in props) and ("Start" in props):
83  print("Healing " + obj.Name + " of type Dimension")
84  nobj = make_copy(obj,force="Dimension",reparent=reparent)
85  elif ("Height" in props) and ("Length" in props):
86  print("Healing " + obj.Name + " of type Rectangle")
87  nobj = make_copy(obj,force="Rectangle",reparent=reparent)
88  elif ("Points" in props) and ("Closed" in props):
89  if "BSpline" in obj.Name:
90  print("Healing " + obj.Name + " of type BSpline")
91  nobj = make_copy(obj,force="BSpline",reparent=reparent)
92  else:
93  print("Healing " + obj.Name + " of type Wire")
94  nobj = make_copy(obj,force="Wire",reparent=reparent)
95  elif ("Radius" in props) and ("FirstAngle" in props):
96  print("Healing " + obj.Name + " of type Circle")
97  nobj = make_copy(obj,force="Circle",reparent=reparent)
98  elif ("DrawMode" in props) and ("FacesNumber" in props):
99  print("Healing " + obj.Name + " of type Polygon")
100  nobj = make_copy(obj,force="Polygon",reparent=reparent)
101  else:
102  dellist.pop()
103  print("Object " + obj.Name + " is not healable")
104 
105  if not got:
106  print("No object seems to need healing")
107  else:
108  print("Healed ",len(dellist)," objects")
109
