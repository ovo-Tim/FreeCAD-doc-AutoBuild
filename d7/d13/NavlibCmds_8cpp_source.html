48 NavlibInterface::ParsedData NavlibInterface::parseCommandId(const std::string& commandId) const
49 {
50  ParsedData result{"", "", 0};
51 
52  if (commandId.empty())
53  return result;
54 
55  auto groupDelimiter = std::find(commandId.cbegin(), commandId.cend(), '|');
56  if (groupDelimiter == commandId.cend())
57  return result;
58 
59  result.groupName = std::string(commandId.cbegin(), groupDelimiter);
60  groupDelimiter++;
61 
62  auto commandDelimiter = std::find(groupDelimiter, commandId.cend(), '|');
63 
64  result.commandName = std::string(groupDelimiter, commandDelimiter);
65  result.actionIndex = -1;
66 
67  if (commandDelimiter != commandId.cend()) {
68  QString indexString =
69  QString::fromStdString(std::string(++commandDelimiter, commandId.cend()).c_str());
70 
71  result.actionIndex = indexString.toInt();
72  }
73 
74  return result;
75 }

105 void NavlibInterface::removeMarkups(std::string& text) const
106 {
107  for (auto textBegin = text.cbegin(); textBegin != text.cend();) {
108  auto markupBegin = std::find(textBegin, text.cend(), '<');
109  if (markupBegin == text.cend())
110  return;
111 
112  auto markupEnd = std::find(markupBegin, text.cend(), '>');
113  if (markupEnd == text.cend())
114  return;
115 
116  const char enclosingChar = *std::prev(markupEnd);
117  textBegin = text.erase(markupBegin, ++markupEnd);
118 
119  if(enclosingChar == 'p')
120  textBegin = text.insert(textBegin, 2, '\n');
121  }
122 }

170 void NavlibInterface::unpackCommands(Gui::Command& command,
171  TDxCategory& category,
172  std::vector<TDx::CImage>& images)
173 {
174  if (command.getAction() == nullptr)
175  return;
176 
177  QList<QAction*> pQActions;
178  TDxCategory subCategory;
179  int32_t index = -1;
180  auto actionGroup = qobject_cast<Gui::ActionGroup*>(command.getAction());
181 
182  if (actionGroup != nullptr) {
183  pQActions = actionGroup->actions();
184  std::string subCategoryName = actionGroup->text().toStdString();
185  subCategory = TDxCategory(subCategoryName, subCategoryName);
186  index = 0;
187  }
188  else
189  pQActions.push_back(command.getAction()->action());
190 
191  for (QAction* pQAction : pQActions) {
192  if (pQAction->isSeparator())
193  continue;
194 
195  TDxCommand ccommand = getCCommand(command, *pQAction, index);
196 
197  if (ccommand.GetId().empty())
198  continue;
199 
200  if (!pQAction->icon().isNull()) {
201  TDxImage commandImage = getImage(*pQAction, ccommand.GetId());
202  images.push_back(commandImage);
203  }
204 
205  if (pQActions.size() > 1)
206  subCategory.push_back(std::move(ccommand));
207  else {
208  category.push_back(std::move(ccommand));
209  return;
210  }
211  index++;
212  }
213  category.push_back(std::move(subCategory));
214 }
