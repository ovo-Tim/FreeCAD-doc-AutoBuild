148 def remove_equal_vecs (vec_list):
149  """Remove equal vectors from a list.
150 
151  Parameters
152  ----------
153  vec_list: list of App.Vectors
154 
155  Returns

157  list of App.Vectors
158  """
159  res_list = []
160  for vec in vec_list:
161  for res in res_list:

168 def get_point_list(point_object):
169  """Extract a list of points from a point object.
170 
171  Parameters
172  ----------
173  point_object: Part::Feature, Sketcher::SketchObject, Mesh::Feature
174  or Points::FeatureCustom
175  The object must have vertices and/or points.
176 
177  Returns
178  -------
179  list of App.Vectors
180  """
181  if hasattr(point_object, "Shape") and hasattr(point_object.Shape, "Vertexes"):
182  pt_list = [v.Point for v in point_object.Shape.Vertexes]
183  # For compatibility with previous versions: add all points from sketch (including construction geometry):
184  if hasattr(point_object, 'Geometry'):
185  place = point_object.Placement
186  for geo in point_object.Geometry:
187  if geo.TypeId == "Part::GeomPoint":
188  pt_list.append(place.multVec(App.Vector(geo.X, geo.Y, geo.Z)))
189  elif hasattr(point_object, "Mesh"):
190  pt_list = [p.Vector for p in point_object.Mesh.Points]
191  elif hasattr(point_object, "Points"):
192  pt_list = point_object.Points.Points
193  else:
194  return []
195 
196  return remove_equal_vecs(pt_list)

198 def build_placements(base_object, pt_list=None, placement=App.Placement()):
199  """Build a placements from the base object and list of points.
200 
201  Returns
202  -------
203  list of App.Placements
204  """
205  if not pt_list:
206  _err(translate("Draft",
207  "Point object doesn't have a discrete point, "
208  "it cannot be used for an array."))
209  return []
210 
211  pls = list()
212 
213  for point in pt_list:
214  new_pla = base_object.Placement.copy()
215  original_rotation = new_pla.Rotation
216 
217  # Reset the position of the copy, and combine the original rotation
218  # with the provided rotation. Two rotations (quaternions)
219  # are combined by multiplying them.
220  new_pla.Base = placement.Base
221  new_pla.Rotation = original_rotation * placement.Rotation
222  new_pla.translate(point)
223 
224  pls.append(new_pla)
225 
226  return pls
