47 //===========================================================================
48 // DrawViewBalloon
49 //===========================================================================
50 // Balloon coordinates are relative to the position of the SourceView
51 // X, Y is the center of the balloon bubble
52 // OriginX, OriginY is the tip of the arrow
53 // these are in unscaled SourceView coordinates
54 // Note that if the SourceView coordinate system changes
55 // (ie location changes or additional items added to the View
56 // the location of the balloon may also change.

119 void DrawViewBalloon::handleChangedPropertyName(Base::XMLReader& reader, const char* TypeName,
120  const char* PropName)
121 {
122  Base::Type type = Base::Type::fromName(TypeName);
123  // was sourceView in the past, now is SourceView
124  if (SourceView.getClassTypeId() == type && strcmp(PropName, "sourceView") == 0) {
125  SourceView.Restore(reader);
126  }
127  else if (BubbleShape.getClassTypeId() == type && strcmp(PropName, "Symbol") == 0) {
128  // was Symbol, then Shape in the past, now is BubbleShape
129  BubbleShape.Restore(reader);
130  }
131  else if (BubbleShape.getClassTypeId() == type && strcmp(PropName, "Shape") == 0) {
132  // was Symbol, then Shape in the past, now is BubbleShape
133  BubbleShape.Restore(reader);
134  }
135  else if (ShapeScale.getClassTypeId() == type && strcmp(PropName, "SymbolScale") == 0) {
136  // was SymbolScale in the past, now is ShapeScale
137  ShapeScale.Restore(reader);
138  }
139  else {
140  DrawView::handleChangedPropertyName(reader, TypeName, PropName);
141  }
142 }
