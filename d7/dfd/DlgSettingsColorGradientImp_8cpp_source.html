48 DlgSettingsColorGradientImp::DlgSettingsColorGradientImp(const App::ColorGradient& cg,
49  QWidget* parent,
50  Qt::WindowFlags fl)
51  : QDialog(parent, fl)
52  , validator(nullptr)
53  , ui(new Ui_DlgSettingsColorGradient)
54 {
55  ui->setupUi(this);
56  ui->spinBoxLabel->setRange(5, 30);
57  ui->spinBoxDecimals->setMaximum(std::numeric_limits<float>::digits10);
58 
59  // remove the automatic help button in dialog title since we don't use it
60  setWindowFlag(Qt::WindowContextHelpButtonHint, false);
61 
62  // the elementary charge is 1.6e-19, since such values might be the result of
63  // simulations, use this as boundary for a scientific validator
64  validator = new QDoubleValidator(-2e19, 2e19, ui->spinBoxDecimals->maximum(), this);
65  validator->setNotation(QDoubleValidator::ScientificNotation);
66  ui->floatLineEditMax->setValidator(validator);
67  ui->floatLineEditMin->setValidator(validator);
68 
69  // assure that the LineEdit is as wide to contain numbers with 4 digits and 6 decimals
70  QFontMetrics fm(ui->floatLineEditMax->font());
71  ui->floatLineEditMax->setMinimumWidth(
72  QtTools::horizontalAdvance(fm, QString::fromLatin1("-8000.000000")));
73 
74  setColorModelNames(cg.getColorModelNames());
75  setProfile(cg.getProfile());
76  setupConnections();
77 }

199 void DlgSettingsColorGradientImp::setRange(float fMin, float fMax)
200 {
201  auto toString = [this](float value, int decimals) {
202  int pos = 0;
203  while (decimals > 0) {
204  QString str = QLocale().toString(value, 'g', decimals--);
205  if (validator->validate(str, pos) == QValidator::Acceptable) {
206  return str;
207  }
208  }
209 
210  return QLocale().toString(value, 'g', 1);
211  };
212 
213  ui->floatLineEditMax->blockSignals(true);
214  ui->floatLineEditMax->setText(toString(fMax, numberOfDecimals()));
215  ui->floatLineEditMax->blockSignals(false);
216 
217  ui->floatLineEditMin->blockSignals(true);
218  ui->floatLineEditMin->setText(toString(fMin, numberOfDecimals()));
219  ui->floatLineEditMin->blockSignals(false);
220 }
