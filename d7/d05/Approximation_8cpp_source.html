142 float PlaneFit::Fit()
143 {
144  _bIsFitted = true;
145  if (CountPoints() < 3) {
146  return FLOAT_MAX;
147  }
148 
149  double sxx {0.0};
150  double sxy {0.0};
151  double sxz {0.0};
152  double syy {0.0};
153  double syz {0.0};
154  double szz {0.0};
155  double mx {0.0};
156  double my {0.0};
157  double mz {0.0};
158 
159  for (const auto& vPoint : _vPoints) {
160  sxx += double(vPoint.x * vPoint.x);
161  sxy += double(vPoint.x * vPoint.y);
162  sxz += double(vPoint.x * vPoint.z);
163  syy += double(vPoint.y * vPoint.y);
164  syz += double(vPoint.y * vPoint.z);
165  szz += double(vPoint.z * vPoint.z);
166  mx += double(vPoint.x);
167  my += double(vPoint.y);
168  mz += double(vPoint.z);
169  }
170 
171  size_t nSize = _vPoints.size();
172  sxx = sxx - mx * mx / (double(nSize));
173  sxy = sxy - mx * my / (double(nSize));
174  sxz = sxz - mx * mz / (double(nSize));
175  syy = syy - my * my / (double(nSize));
176  syz = syz - my * mz / (double(nSize));
177  szz = szz - mz * mz / (double(nSize));
178 
179 #if defined(FC_USE_EIGEN)
180  Eigen::Matrix3d covMat = Eigen::Matrix3d::Zero();
181  covMat(0, 0) = sxx;
182  covMat(1, 1) = syy;
183  covMat(2, 2) = szz;
184  covMat(0, 1) = sxy;
185  covMat(1, 0) = sxy;
186  covMat(0, 2) = sxz;
187  covMat(2, 0) = sxz;
188  covMat(1, 2) = syz;
189  covMat(2, 1) = syz;
190  Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> eig(covMat);
191 
192  Eigen::Vector3d u = eig.eigenvectors().col(1);
193  Eigen::Vector3d v = eig.eigenvectors().col(2);
194  Eigen::Vector3d w = eig.eigenvectors().col(0);
195 
196  _vDirU.Set(u.x(), u.y(), u.z());
197  _vDirV.Set(v.x(), v.y(), v.z());
198  _vDirW.Set(w.x(), w.y(), w.z());
199  _vBase.Set(mx / (float)nSize, my / (float)nSize, mz / (float)nSize);
200 
201  float sigma = w.dot(covMat * w);
202 #else
203  // Covariance matrix
204  Wm4::Matrix3<double> akMat(sxx, sxy, sxz, sxy, syy, syz, sxz, syz, szz);
205  Wm4::Matrix3<double> rkRot, rkDiag;
206  try {
207  akMat.EigenDecomposition(rkRot, rkDiag);
208  }
209  catch (const std::exception&) {
210  return FLOAT_MAX;
211  }
212 
213  // We know the Eigenvalues are ordered
214  // rkDiag(0,0) <= rkDiag(1,1) <= rkDiag(2,2)
215  //
216  // points describe a line or even are identical
217  if (rkDiag(1, 1) <= 0) {
218  return FLOAT_MAX;
219  }
220 
221  Wm4::Vector3<double> U = rkRot.GetColumn(1);
222  Wm4::Vector3<double> V = rkRot.GetColumn(2);
223  Wm4::Vector3<double> W = rkRot.GetColumn(0);
224 
225  // It may happen that the result have nan values
226  for (int i = 0; i < 3; i++) {
227  if (boost::math::isnan(W[i])) {
228  return FLOAT_MAX;
229  }
230  }
231 
232  // In some cases when the points exactly lie on a plane it can happen that
233  // U or V have nan values but W is valid.
234  // In this case create an orthonormal basis
235  bool validUV = true;
236  for (int i = 0; i < 3; i++) {
237  if (boost::math::isnan(U[i]) || boost::math::isnan(V[i])) {
238  validUV = false;
239  break;
240  }
241  }
242 
243  if (!validUV) {
244  Wm4::Vector3<double>::GenerateOrthonormalBasis(U, V, W);
245  }
246 
247  _vDirU.Set(float(U.X()), float(U.Y()), float(U.Z()));
248  _vDirV.Set(float(V.X()), float(V.Y()), float(V.Z()));
249  _vDirW.Set(float(W.X()), float(W.Y()), float(W.Z()));
250  _vBase.Set(float(mx / nSize), float(my / nSize), float(mz / nSize));
251  float sigma = float(W.Dot(akMat * W));
252 #endif
253 
254  // In case sigma is nan
255  if (boost::math::isnan(sigma)) {
256  return FLOAT_MAX;
257  }
258 
259  // This must be caused by some round-off errors. Theoretically it's impossible
260  // that 'sigma' becomes negative because the covariance matrix is positive semi-definite.
261  if (sigma < 0) {
262  sigma = 0;
263  }
264 
265  // make a right-handed system
266  if ((_vDirU % _vDirV) * _vDirW < 0.0F) {
267  Base::Vector3f tmp = _vDirU;
268  _vDirU = _vDirV;
269  _vDirV = tmp;
270  }
271 
272  if (nSize > 3) {
273  sigma = sqrt(sigma / (nSize - 3));
274  }
275  else {
276  sigma = 0;
277  }
278 
279  _fLastResult = sigma;
280  return _fLastResult;
281 }

328 float PlaneFit::GetStdDeviation() const
329 {
330  // Mean: M=(1/N)*SUM Xi
331  // Variance: VAR=(N/N-1)*[(1/N)*SUM(Xi^2)-M^2]
332  // Standard deviation: SD=SQRT(VAR)
333  // Standard error of the mean: SE=SD/SQRT(N)
334  if (!_bIsFitted) {
335  return FLOAT_MAX;
336  }
337 
338  float fSumXi = 0.0F, fSumXi2 = 0.0F, fMean = 0.0F, fDist = 0.0F;
339 
340  float ulPtCt = float(CountPoints());
341  std::list<Base::Vector3f>::const_iterator cIt;
342 
343  for (cIt = _vPoints.begin(); cIt != _vPoints.end(); ++cIt) {
344  fDist = GetDistanceToPlane(*cIt);
345  fSumXi += fDist;
346  fSumXi2 += (fDist * fDist);
347  }
348 
349  fMean = (1.0F / ulPtCt) * fSumXi;
350  return sqrt((ulPtCt / (ulPtCt - 1.0F)) * ((1.0F / ulPtCt) * fSumXi2 - fMean * fMean));
351 }

353 float PlaneFit::GetSignedStdDeviation() const
354 {
355  // if the nearest point to the gravity is at the side
356  // of normal direction the value will be
357  // positive otherwise negative
358  if (!_bIsFitted) {
359  return FLOAT_MAX;
360  }
361 
362  float fSumXi = 0.0F, fSumXi2 = 0.0F, fMean = 0.0F, fDist = 0.0F;
363  float fMinDist = FLOAT_MAX;
364  float fFactor = 0.0F;
365 
366  float ulPtCt = float(CountPoints());
367  Base::Vector3f clGravity, clPt;
368  std::list<Base::Vector3f>::const_iterator cIt;
369  for (cIt = _vPoints.begin(); cIt != _vPoints.end(); ++cIt) {
370  clGravity += *cIt;
371  }
372  clGravity *= (1.0F / ulPtCt);
373 
374  for (cIt = _vPoints.begin(); cIt != _vPoints.end(); ++cIt) {
375  if ((clGravity - *cIt).Length() < fMinDist) {
376  fMinDist = (clGravity - *cIt).Length();
377  clPt = *cIt;
378  }
379  fDist = GetDistanceToPlane(*cIt);
380  fSumXi += fDist;
381  fSumXi2 += (fDist * fDist);
382  }
383 
384  // which side
385  if ((clPt - clGravity) * GetNormal() > 0) {
386  fFactor = 1.0F;
387  }
388  else {
389  fFactor = -1.0F;
390  }
391 
392  fMean = 1.0F / ulPtCt * fSumXi;
393 
394  return fFactor * sqrt((ulPtCt / (ulPtCt - 3.0F)) * ((1.0F / ulPtCt) * fSumXi2 - fMean * fMean));
395 }

529 void QuadraticFit::CalcEigenValues(double& dLambda1,
530  double& dLambda2,
531  double& dLambda3,
532  Base::Vector3f& clEV1,
533  Base::Vector3f& clEV2,
534  Base::Vector3f& clEV3) const
535 {
536  assert(_bIsFitted);
537 
538  /*
539  * F(x,y,z) = a11*x*x + a22*y*y + a33*z*z +2*a12*x*y + 2*a13*x*z + 2*a23*y*z + 2*a10*x + 2*a20*y
540  * + 2*a30*z * a00 = 0
541  *
542  * Form matrix:
543  *
544  * ( a11 a12 a13 )
545  * A = ( a21 a22 a23 ) where a[i,j] = a[j,i]
546  * ( a31 a32 a33 )
547  *
548  * Coefficients of the Quadrik-Fit based on the notation used here:
549  *
550  * 0 = C[0] + C[1]*X + C[2]*Y + C[3]*Z + C[4]*X^2 + C[5]*Y^2
551  * + C[6]*Z^2 + C[7]*X*Y + C[8]*X*Z + C[9]*Y*Z
552  *
553  * Square: a11 := c[4], a22 := c[5], a33 := c[6]
554  * Mixed: a12 := c[7]/2, a13 := c[8]/2, a23 := c[9]/2
555  * Linear: a10 := c[1]/2, a20 := c[2]/2, a30 := c[3]/2
556  * Constant: a00 := c[0]
557  *
558  */
559 
560  Wm4::Matrix3<double> akMat(_fCoeff[4],
561  _fCoeff[7] / 2.0,
562  _fCoeff[8] / 2.0,
563  _fCoeff[7] / 2.0,
564  _fCoeff[5],
565  _fCoeff[9] / 2.0,
566  _fCoeff[8] / 2.0,
567  _fCoeff[9] / 2.0,
568  _fCoeff[6]);
569 
570  Wm4::Matrix3<double> rkRot, rkDiag;
571  akMat.EigenDecomposition(rkRot, rkDiag);
572 
573  Wm4::Vector3<double> vEigenU = rkRot.GetColumn(0);
574  Wm4::Vector3<double> vEigenV = rkRot.GetColumn(1);
575  Wm4::Vector3<double> vEigenW = rkRot.GetColumn(2);
576 
577  clEV1 = Base::convertTo<Base::Vector3f>(vEigenU);
578  clEV2 = Base::convertTo<Base::Vector3f>(vEigenV);
579  clEV3 = Base::convertTo<Base::Vector3f>(vEigenW);
580 
581  dLambda1 = rkDiag[0][0];
582  dLambda2 = rkDiag[1][1];
583  dLambda3 = rkDiag[2][2];
584 }

586 void QuadraticFit::CalcZValues(double x, double y, double& dZ1, double& dZ2) const
587 {
588  assert(_bIsFitted);
589 
590  double dDisk = _fCoeff[3] * _fCoeff[3] + 2 * _fCoeff[3] * _fCoeff[8] * x
591  + 2 * _fCoeff[3] * _fCoeff[9] * y + _fCoeff[8] * _fCoeff[8] * x * x
592  + 2 * _fCoeff[8] * x * _fCoeff[9] * y + _fCoeff[9] * _fCoeff[9] * y * y
593  - 4 * _fCoeff[6] * _fCoeff[0] - 4 * _fCoeff[6] * _fCoeff[1] * x
594  - 4 * _fCoeff[6] * _fCoeff[2] * y - 4 * _fCoeff[6] * _fCoeff[7] * x * y
595  - 4 * _fCoeff[6] * _fCoeff[4] * x * x - 4 * _fCoeff[6] * _fCoeff[5] * y * y;
596 
597  if (fabs(_fCoeff[6]) < 0.000005) {
598  dZ1 = double(FLOAT_MAX);
599  dZ2 = double(FLOAT_MAX);
600  return;
601  }
602 
603  if (dDisk < 0.0) {
604  dZ1 = double(FLOAT_MAX);
605  dZ2 = double(FLOAT_MAX);
606  return;
607  }
608 
609  dDisk = sqrt(dDisk);
610 
611  dZ1 = 0.5 * ((-_fCoeff[3] - _fCoeff[8] * x - _fCoeff[9] * y + dDisk) / _fCoeff[6]);
612  dZ2 = 0.5 * ((-_fCoeff[3] - _fCoeff[8] * x - _fCoeff[9] * y - dDisk) / _fCoeff[6]);
613 }

672 double SurfaceFit::PolynomFit()
673 {
674  if (PlaneFit::Fit() >= FLOAT_MAX) {
675  return double(FLOAT_MAX);
676  }
677 
678  Base::Vector3d bs = Base::convertTo<Base::Vector3d>(this->_vBase);
679  Base::Vector3d ex = Base::convertTo<Base::Vector3d>(this->_vDirU);
680  Base::Vector3d ey = Base::convertTo<Base::Vector3d>(this->_vDirV);
681  // Base::Vector3d ez = Base::convertTo<Base::Vector3d>(this->_vDirW);
682 
683  // A*x = b
684  // See also www.cs.jhu.edu/~misha/Fall05/10.23.05.pdf
685  // z = f(x,y) = a*x^2 + b*y^2 + c*x*y + d*x + e*y + f
686  // z = P * Vi with Vi=(xi^2,yi^2,xiyi,xi,yi,1) and P=(a,b,c,d,e,f)
687  // To get the best-fit values the sum needs to be minimized:
688  // S = sum[(z-zi)^2} -> min with zi=z coordinates of the given points
689  // <=> S = sum[z^2 - 2*z*zi + zi^2] -> min
690  // <=> S(P) = sum[(P*Vi)^2 - 2*(P*Vi)*zi + zi^2] -> min
691  // To get the optimum the gradient of the expression must be the null vector
692  // Note: grad F(P) = (P*Vi)^2 = 2*(P*Vi)*Vi
693  // grad G(P) = -2*(P*Vi)*zi = -2*Vi*zi
694  // grad H(P) = zi^2 = 0
695  // => grad S(P) = sum[2*(P*Vi)*Vi - 2*Vi*zi] = 0
696  // <=> sum[(P*Vi)*Vi] = sum[Vi*zi]
697  // <=> sum[(Vi*Vi^t)*P] = sum[Vi*zi] where (Vi*Vi^t) is a symmetric matrix
698  // <=> sum[(Vi*Vi^t)]*P = sum[Vi*zi]
699  Eigen::Matrix<double, 6, 6> A = Eigen::Matrix<double, 6, 6>::Zero();
700  Eigen::Matrix<double, 6, 1> b = Eigen::Matrix<double, 6, 1>::Zero();
701  Eigen::Matrix<double, 6, 1> x = Eigen::Matrix<double, 6, 1>::Zero();
702 
703  std::vector<Base::Vector3d> transform;
704  transform.reserve(_vPoints.size());
705 
706  double dW2 = 0;
707  for (const auto& it : _vPoints) {
708  Base::Vector3d clPoint = Base::convertTo<Base::Vector3d>(it);
709  clPoint.TransformToCoordinateSystem(bs, ex, ey);
710  transform.push_back(clPoint);
711  double dU = clPoint.x;
712  double dV = clPoint.y;
713  double dW = clPoint.z;
714 
715  double dU2 = dU * dU;
716  double dV2 = dV * dV;
717  double dUV = dU * dV;
718 
719  dW2 += dW * dW;
720 
721  A(0, 0) = A(0, 0) + dU2 * dU2;
722  A(0, 1) = A(0, 1) + dU2 * dV2;
723  A(0, 2) = A(0, 2) + dU2 * dUV;
724  A(0, 3) = A(0, 3) + dU2 * dU;
725  A(0, 4) = A(0, 4) + dU2 * dV;
726  A(0, 5) = A(0, 5) + dU2;
727  b(0) = b(0) + dU2 * dW;
728 
729  A(1, 1) = A(1, 1) + dV2 * dV2;
730  A(1, 2) = A(1, 2) + dV2 * dUV;
731  A(1, 3) = A(1, 3) + dV2 * dU;
732  A(1, 4) = A(1, 4) + dV2 * dV;
733  A(1, 5) = A(1, 5) + dV2;
734  b(1) = b(1) + dV2 * dW;
735 
736  A(2, 2) = A(2, 2) + dUV * dUV;
737  A(2, 3) = A(2, 3) + dUV * dU;
738  A(2, 4) = A(2, 4) + dUV * dV;
739  A(2, 5) = A(2, 5) + dUV;
740  b(3) = b(3) + dUV * dW;
741 
742  A(3, 3) = A(3, 3) + dU * dU;
743  A(3, 4) = A(3, 4) + dU * dV;
744  A(3, 5) = A(3, 5) + dU;
745  b(3) = b(3) + dU * dW;
746 
747  A(4, 4) = A(4, 4) + dV * dV;
748  A(4, 5) = A(4, 5) + dV;
749  b(5) = b(5) + dV * dW;
750 
751  A(5, 5) = A(5, 5) + 1;
752  b(5) = b(5) + 1 * dW;
753  }
754 
755  // Mat is symmetric
756  //
757  A(1, 0) = A(0, 1);
758  A(2, 0) = A(0, 2);
759  A(3, 0) = A(0, 3);
760  A(4, 0) = A(0, 4);
761  A(5, 0) = A(0, 5);
762 
763  A(2, 1) = A(1, 2);
764  A(3, 1) = A(1, 3);
765  A(4, 1) = A(1, 4);
766  A(5, 1) = A(1, 5);
767 
768  A(3, 2) = A(2, 3);
769  A(4, 2) = A(2, 4);
770  A(5, 2) = A(2, 5);
771 
772  A(4, 3) = A(3, 4);
773  A(5, 3) = A(3, 5);
774 
775  A(5, 4) = A(4, 5);
776 
777  Eigen::HouseholderQR<Eigen::Matrix<double, 6, 6>> qr(A);
778  x = qr.solve(b);
779 
780  // FunctionContainer gets an implicit function F(x,y,z) = 0 of this form
781  // _fCoeff[0] +
782  // _fCoeff[1]*x + _fCoeff[2]*y + _fCoeff[3]*z +
783  // _fCoeff[4]*x^2 + _fCoeff[5]*y^2 + _fCoeff[6]*z^2 +
784  // _fCoeff[7]*x*y + _fCoeff[8]*x*z + _fCoeff[9]*y*z
785  //
786  // The bivariate polynomial surface we get here is of the form
787  // z = f(x,y) = a*x^2 + b*y^2 + c*x*y + d*x + e*y + f
788  // Writing it as implicit surface F(x,y,z) = 0 gives this form
789  // F(x,y,z) = f(x,y) - z = a*x^2 + b*y^2 + c*x*y + d*x + e*y - z + f
790  // Thus:
791  // _fCoeff[0] = f
792  // _fCoeff[1] = d
793  // _fCoeff[2] = e
794  // _fCoeff[3] = -1
795  // _fCoeff[4] = a
796  // _fCoeff[5] = b
797  // _fCoeff[6] = 0
798  // _fCoeff[7] = c
799  // _fCoeff[8] = 0
800  // _fCoeff[9] = 0
801 
802  _fCoeff[0] = x(5);
803  _fCoeff[1] = x(3);
804  _fCoeff[2] = x(4);
805  _fCoeff[3] = -1.0;
806  _fCoeff[4] = x(0);
807  _fCoeff[5] = x(1);
808  _fCoeff[6] = 0.0;
809  _fCoeff[7] = x(2);
810  _fCoeff[8] = 0.0;
811  _fCoeff[9] = 0.0;
812 
813  // Get S(P) = sum[(P*Vi)^2 - 2*(P*Vi)*zi + zi^2]
814  double sigma = 0;
815  FunctionContainer clFuncCont(_fCoeff);
816  for (const auto& it : transform) {
817  double u = it.x;
818  double v = it.y;
819  double z = clFuncCont.F(u, v, 0.0);
820  sigma += z * z;
821  }
822 
823  sigma += dW2 - 2 * x.dot(b);
824  // This must be caused by some round-off errors. Theoretically it's impossible
825  // that 'sigma' becomes negative.
826  if (sigma < 0) {
827  sigma = 0;
828  }
829  if (!_vPoints.empty()) {
830  sigma = sqrt(sigma / _vPoints.size());
831  }
832 
833  _fLastResult = static_cast<float>(sigma);
834  return double(_fLastResult);
835 }

925 std::vector<Base::Vector3d>
926 SurfaceFit::toBezier(double umin, double umax, double vmin, double vmax) const
927 {
928  std::vector<Base::Vector3d> pts;
929  pts.reserve(9);
930 
931  // the Bezier surface is defined by the 3x3 control points
932  // P11 P21 P31
933  // P12 P22 P32
934  // P13 P23 P33
935  //
936  // The surface goes through the points P11, P31, P31 and P33
937  // To get the four control points P21, P12, P32 and P23 we inverse
938  // the de-Casteljau algorithm used for Bezier curves of degree 2
939  // as we already know the points for the parameters
940  // (0, 0.5), (0.5, 0), (0.5, 1.0) and (1.0, 0.5)
941  // To get the control point P22 we inverse the de-Casteljau algorithm
942  // for the surface point on (0.5, 0.5)
943  double umid = 0.5 * (umin + umax);
944  double vmid = 0.5 * (vmin + vmax);
945 
946  // first row
947  double z11 = Value(umin, vmin);
948  double v21 = Value(umid, vmin);
949  double z31 = Value(umax, vmin);
950  double z21 = 2.0 * v21 - 0.5 * (z11 + z31);
951 
952  // third row
953  double z13 = Value(umin, vmax);
954  double v23 = Value(umid, vmax);
955  double z33 = Value(umax, vmax);
956  double z23 = 2.0 * v23 - 0.5 * (z13 + z33);
957 
958  // second row
959  double v12 = Value(umin, vmid);
960  double z12 = 2.0 * v12 - 0.5 * (z11 + z13);
961  double v32 = Value(umax, vmid);
962  double z32 = 2.0 * v32 - 0.5 * (z31 + z33);
963  double v22 = Value(umid, vmid);
964  double z22 = 4.0 * v22 - 0.25 * (z11 + z31 + z13 + z33 + 2.0 * (z12 + z21 + z32 + z23));
965 
966  // first row
967  pts.emplace_back(umin, vmin, z11);
968  pts.emplace_back(umid, vmin, z21);
969  pts.emplace_back(umax, vmin, z31);
970 
971  // second row
972  pts.emplace_back(umin, vmid, z12);
973  pts.emplace_back(umid, vmid, z22);
974  pts.emplace_back(umax, vmid, z32);
975 
976  // third row
977  pts.emplace_back(umin, vmax, z13);
978  pts.emplace_back(umid, vmax, z23);
979  pts.emplace_back(umax, vmax, z33);
980  return pts;
981 }

988 struct LMCylinderFunctor
989 {
990  Eigen::MatrixXd measuredValues;
991 
992  // Compute 'm' errors, one for each data point, for the given parameter values in 'x'
993  int operator()(const Eigen::VectorXd& xvec, Eigen::VectorXd& fvec) const
994  {
995  // 'xvec' has dimensions n x 1
996  // It contains the current estimates for the parameters.
997 
998  // 'fvec' has dimensions m x 1
999  // It will contain the error for each data point.
1000  double aParam = xvec(0); // dir_x
1001  double bParam = xvec(1); // dir_y
1002  double cParam = xvec(2); // dir_z
1003  double dParam = xvec(3); // cnt_x
1004  double eParam = xvec(4); // cnt_y
1005  double fParam = xvec(5); // cnt_z
1006  double gParam = xvec(6); // radius
1007 
1008  // use distance functions (fvec(i)) for cylinders as defined in the paper:
1009  // Least-Squares Fitting Algorithms of the NIST Algorithm Testing System
1010  for (int i = 0; i < values(); i++) {
1011  double xValue = measuredValues(i, 0);
1012  double yValue = measuredValues(i, 1);
1013  double zValue = measuredValues(i, 2);
1014 
1015  double a = aParam / (sqrt(aParam * aParam + bParam * bParam + cParam * cParam));
1016  double b = bParam / (sqrt(aParam * aParam + bParam * bParam + cParam * cParam));
1017  double c = cParam / (sqrt(aParam * aParam + bParam * bParam + cParam * cParam));
1018  double x = dParam;
1019  double y = eParam;
1020  double z = fParam;
1021  double u = c * (yValue - y) - b * (zValue - z);
1022  double v = a * (zValue - z) - c * (xValue - x);
1023  double w = b * (xValue - x) - a * (yValue - y);
1024 
1025  fvec(i) = sqrt(u * u + v * v + w * w) - gParam;
1026  }
1027  return 0;
1028  }
1029 
1030  // Compute the jacobian of the errors
1031  int df(const Eigen::VectorXd& x, Eigen::MatrixXd& fjac) const
1032  {
1033  // 'x' has dimensions n x 1
1034  // It contains the current estimates for the parameters.
1035 
1036  // 'fjac' has dimensions m x n
1037  // It will contain the jacobian of the errors, calculated numerically in this case.
1038 
1039  const double epsilon = 1e-5;
1040 
1041  for (int i = 0; i < x.size(); i++) {
1042  Eigen::VectorXd xPlus(x);
1043  xPlus(i) += epsilon;
1044  Eigen::VectorXd xMinus(x);
1045  xMinus(i) -= epsilon;
1046 
1047  Eigen::VectorXd fvecPlus(values());
1048  operator()(xPlus, fvecPlus);
1049 
1050  Eigen::VectorXd fvecMinus(values());
1051  operator()(xMinus, fvecMinus);
1052 
1053  Eigen::VectorXd fvecDiff(values());
1054  fvecDiff = (fvecPlus - fvecMinus) / (2.0F * epsilon);
1055 
1056  fjac.block(0, i, values(), 1) = fvecDiff;
1057  }
1058 
1059  return 0;
1060  }
1061 
1062  // Number of data points, i.e. values.
1063  int m;
1064 
1065  // Returns 'm', the number of values.
1066  int values() const
1067  {
1068  return m;
1069  }
1070 
1071  // The number of parameters, i.e. inputs.
1072  int n;
1073 
1074  // Returns 'n', the number of inputs.
1075  int inputs() const
1076  {
1077  return n;
1078  }
1079 };
