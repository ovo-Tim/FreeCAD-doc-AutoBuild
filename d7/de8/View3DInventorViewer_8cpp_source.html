220 class Gui::ViewerEventFilter : public QObject
221 {
222 public:
223  ViewerEventFilter() = default;
224  ~ViewerEventFilter() override = default;
225 
226  bool eventFilter(QObject* obj, QEvent* event) override {
227  // Bug #0000607: Some mice also support horizontal scrolling which however might
228  // lead to some unwanted zooming when pressing the MMB for panning.
229  // Thus, we filter out horizontal scrolling.
230  if (event->type() == QEvent::Wheel) {
231  auto we = static_cast<QWheelEvent*>(event); // NOLINT
232  if (qAbs(we->angleDelta().x()) > qAbs(we->angleDelta().y())) {
233  return true;
234  }
235  }
236  else if (event->type() == QEvent::KeyPress) {
237  auto ke = static_cast<QKeyEvent*>(event); // NOLINT
238  if (ke->matches(QKeySequence::SelectAll)) {
239  auto* viewer3d = static_cast<View3DInventorViewer*>(obj);
240  auto* editingVP = viewer3d->getEditingViewProvider();
241  if(!editingVP || !editingVP->selectAll())
242  {
243  viewer3d->selectAll();
244  }
245  return true;
246  }
247  }
248 
249  if (Base::Sequencer().isRunning() && Base::Sequencer().isBlocking()) {
250  return false;
251  }
252 
253  if (event->type() == Spaceball::ButtonEvent::ButtonEventType) {
254  auto buttonEvent = static_cast<Spaceball::ButtonEvent*>(event); // NOLINT
255  if (!buttonEvent) {
256  Base::Console().Log("invalid spaceball button event\n");
257  return true;
258  }
259  }
260  else if (event->type() == Spaceball::MotionEvent::MotionEventType) {
261  auto motionEvent = static_cast<Spaceball::MotionEvent*>(event); // NOLINT
262  if (!motionEvent) {
263  Base::Console().Log("invalid spaceball motion event\n");
264  return true;
265  }
266  }
267 
268  return false;
269  }

274  SpaceNavigatorDevice() : InputDevice(nullptr) {}
275  ~SpaceNavigatorDevice() override = default;
276  const SoEvent* translateEvent(QEvent* event) override {
277 
278  if (event->type() == Spaceball::MotionEvent::MotionEventType) {
279  auto motionEvent = static_cast<Spaceball::MotionEvent*>(event); // NOLINT
280  if (!motionEvent) {
281  Base::Console().Log("invalid spaceball motion event\n");
282  return nullptr;
283  }
284 
285  motionEvent->setHandled(true);
286 
287  float xTrans{};
288  float yTrans{};
289  float zTrans{};
290  xTrans = static_cast<float>(motionEvent->translationX());
291  yTrans = static_cast<float>(motionEvent->translationY());
292  zTrans = static_cast<float>(motionEvent->translationZ());
293  SbVec3f translationVector(xTrans, yTrans, zTrans);
294 
295  constexpr const float rotationConstant(.0001F);
296  SbRotation xRot;
297  SbRotation yRot;
298  SbRotation zRot;
299  xRot.setValue(SbVec3f(1.0, 0.0, 0.0), static_cast<float>(motionEvent->rotationX()) * rotationConstant);
300  yRot.setValue(SbVec3f(0.0, 1.0, 0.0), static_cast<float>(motionEvent->rotationY()) * rotationConstant);
301  zRot.setValue(SbVec3f(0.0, 0.0, 1.0), static_cast<float>(motionEvent->rotationZ()) * rotationConstant);
302 
303  auto motion3Event = new SoMotion3Event;
304  motion3Event->setTranslation(translationVector);
305  motion3Event->setRotation(xRot * yRot * zRot);
306  motion3Event->setPosition(this->mousepos);
307 
308  return motion3Event;
309  }
310 
311  return nullptr;
312  }

591 View3DInventorViewer::~View3DInventorViewer()

594  aboutToDestroyGLContext();
595 
596  // It can happen that a document has several MDI views and when the about to be
597  // closed 3D view is in edit mode the corresponding view provider must be restored
598  // because otherwise it might be left in a broken state
599  // See https://forum.freecad.org/viewtopic.php?f=3&t=39720
600  if (restoreEditingRoot) {
601  resetEditingRoot(false);
602  }
603

605  this->backgroundroot->unref();

608  this->foregroundroot = nullptr;

610  this->pcBackGround = nullptr;
611 
612  setSceneGraph(nullptr);
613  this->pEventCallback->unref();
614  this->pEventCallback = nullptr;
615  // Note: It can happen that there is still someone who references
616  // the root node but isn't destroyed when closing this viewer so
617  // that it prevents all children from being deleted. To reduce this
618  // likelihood we explicitly remove all child nodes now.
619  coinRemoveAllChildren(this->pcViewProviderRoot);
620  this->pcViewProviderRoot->unref();
621  this->pcViewProviderRoot = nullptr;
622  this->objectGroup->unref();
623  this->objectGroup = nullptr;

634  }

637

639  if (getMainWindow()) {
640  getMainWindow()->setPaneText(2, QLatin1String(""));

644

1803 SbVec2f View3DInventorViewer::screenCoordsOfPath(SoPath* path) const
1804 {
1805  // Generate a matrix (well, a SoGetMatrixAction) that
1806  // moves us to the picked object's coordinate space.
1807  SoGetMatrixAction gma(getSoRenderManager()->getViewportRegion());
1808  gma.apply(path);
1809 
1810  // Use that matrix to translate the origin in the picked
1811  // object's coordinate space into object space
1812  SbVec3f imageCoords(0, 0, 0);
1813  SbMatrix mat = gma.getMatrix().transpose();
1814  mat.multMatrixVec(imageCoords, imageCoords);
1815 
1816  // Now, project the object space coordinates of the object
1817  // into "normalized" screen coordinates.
1818  SbViewVolume vol = getSoRenderManager()->getCamera()->getViewVolume();
1819  vol.projectToScreen(imageCoords, imageCoords);
1820 
1821  // Translate "normalized" screen coordinates to pixel coords.
1822  //
1823  // Note: for some reason, projectToScreen() doesn't seem to
1824  // handle non-square viewports properly. The X and Y are
1825  // scaled such that [0,1] fits within the smaller of the window
1826  // width or height. For instance, in a window that's 400px
1827  // tall and 800px wide, the Y will be within [0,1], but X can
1828  // vary within [-0.5,1.5]...
1829  int width = getGLWidget()->width();
1830  int height = getGLWidget()->height();
1831 
1832  if (width >= height) {
1833  // "Landscape" orientation, to square
1834  imageCoords[0] *= height;
1835  imageCoords[0] += (width-height) / 2.0; // NOLINT
1836  imageCoords[1] *= height;
1837 
1838  }
1839  else {
1840  // "Portrait" orientation
1841  imageCoords[0] *= width;
1842  imageCoords[1] *= width;
1843  imageCoords[1] += (height-width) / 2.0; // NOLINT
1844  }
1845 
1846  return {imageCoords[0], imageCoords[1]};
1847 }
