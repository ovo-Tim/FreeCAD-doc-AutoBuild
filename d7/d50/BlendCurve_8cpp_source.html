58 Handle(Geom_BezierCurve) BlendCurve::compute()
59 {
60  size_t nb_pts = blendPoints.size();
61  try {
62  // Uniform Parametrization
63  TColStd_Array1OfReal params(1, nb_pts);
64  for (size_t i = 0; i < nb_pts; ++i) {
65  params(i + 1) = (double)i / ((double)nb_pts - 1);
66  }
67 
68  int num_poles = 0;
69  for (size_t i = 0; i < nb_pts; ++i) {
70  num_poles += blendPoints[i].nbVectors();
71  }
72 
73  Handle(Geom_BezierCurve) curve;
74  if (num_poles > (curve->MaxDegree() + 1)) { // use Geom_BezierCurve max degree
75  Standard_Failure::Raise("number of constraints exceeds bezier curve capacity");
76  }
77 
78  TColStd_Array1OfReal knots(1, 2 * num_poles);
79  for (int i = 1; i <= num_poles; ++i) {
80  knots(i) = params(1);
81  knots(num_poles + i) = params(nb_pts);
82  }
83 
84  math_Matrix OCCmatrix(1, num_poles, 1, num_poles, 0.0);
85  math_Vector res_x(1, num_poles, 0.0);
86  math_Vector res_y(1, num_poles, 0.0);
87  math_Vector res_z(1, num_poles, 0.0);
88  int row_idx = 1;
89  int cons_idx = 1;
90  for (size_t i = 0; i < nb_pts; ++i) {
91  math_Matrix bezier_eval(1, blendPoints[i].nbVectors(), 1, num_poles, 0.0);
92  Standard_Integer first_non_zero;
93  BSplCLib::EvalBsplineBasis(blendPoints[i].nbVectors() - 1,
94  num_poles,
95  knots,
96  params(cons_idx),
97  first_non_zero,
98  bezier_eval,
99  Standard_False);
100  int idx2 = 1;
101  for (int it2 = 0; it2 < blendPoints[i].nbVectors(); ++it2) {
102  OCCmatrix.SetRow(row_idx, bezier_eval.Row(idx2));
103  Base::Vector3d pnt = blendPoints[i].vectors[it2];
104  res_x(row_idx) = pnt.x;
105  res_y(row_idx) = pnt.y;
106  res_z(row_idx) = pnt.z;
107  idx2++;
108  row_idx++;
109  }
110  cons_idx++;
111  }
112  math_Gauss gauss(OCCmatrix);
113  gauss.Solve(res_x);
114  if (!gauss.IsDone()) {
115  Standard_Failure::Raise("Failed to solve equations");
116  }
117  gauss.Solve(res_y);
118  if (!gauss.IsDone()) {
119  Standard_Failure::Raise("Failed to solve equations");
120  }
121  gauss.Solve(res_z);
122  if (!gauss.IsDone()) {
123  Standard_Failure::Raise("Failed to solve equations");
124  }
125 
126  TColgp_Array1OfPnt poles(1, num_poles);
127  for (int idx = 1; idx <= num_poles; ++idx) {
128  poles.SetValue(idx, gp_Pnt(res_x(idx), res_y(idx), res_z(idx)));
129  }
130  Handle(Geom_BezierCurve) bezier = new Geom_BezierCurve(poles);
131  return bezier;
132  }
133  catch (Standard_Failure&) {
134  PyErr_SetString(Base::PyExc_FC_CADKernelError, "Failed to compute bezier curve");
135  }
136  return nullptr;
137 }
