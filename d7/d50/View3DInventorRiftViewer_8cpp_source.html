104 void View3DInventorRiftViewer::keyPressEvent(QKeyEvent *event)
105 {
106  static const float increment = 0.02; // move two centimeter per key
107  static const float rotIncrement = M_PI/4; // move two 90Â° per key
108 
109 
110  if (event->key() == Qt::Key_Plus) {
111  scale->scaleFactor.setValue(scale->scaleFactor.getValue() * 2.0f) ;
112  } else if (event->key() == Qt::Key_Minus) {
113  scale->scaleFactor.setValue(scale->scaleFactor.getValue() * 0.2f) ;
114  } else if (event->key() == Qt::Key_S) {
115  basePosition += SbVec3f(0,0,increment) ;
116  } else if (event->key() == Qt::Key_W) {
117  basePosition += SbVec3f(0,0,-increment) ;
118  } else if (event->key() == Qt::Key_Up) {
119  basePosition += SbVec3f(0,-increment,0) ;
120  } else if (event->key() == Qt::Key_Down) {
121  basePosition += SbVec3f(0,increment,0) ;
122  } else if (event->key() == Qt::Key_Left) {
123  rotation2->angle.setValue( rotation2->angle.getValue() + rotIncrement);
124  } else if (event->key() == Qt::Key_Right) {
125  rotation2->angle.setValue( rotation2->angle.getValue() - rotIncrement);
126  } else if (event->key() == Qt::Key_A) {
127  basePosition += SbVec3f(-increment,0,0) ;
128  } else if (event->key() == Qt::Key_D) {
129  basePosition += SbVec3f(increment,0,0) ;
130  } else {
131 
132  CoinRiftWidget::keyPressEvent(event);
133  }
134 }
