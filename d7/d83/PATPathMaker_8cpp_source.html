55 void PATPathMaker::lineSetToFillItems(LineSet& ls)
56 {
57  m_segCount = 0;
58  QPen pen = getPen();
59  for (auto& geom : ls.getGeoms()) {
60  //geom is a tdGeometry representation of 1 line in the pattern
61  if (ls.isDashed()) {
62  double offset = 0.0;
63  Base::Vector3d pStart = ls.getPatternStartPoint(geom, offset, m_fillScale);
64  offset = Rez::guiX(offset);
65  Base::Vector3d gStart(geom->getStartPoint().x,
66  geom->getStartPoint().y,
67  0.0);
68  Base::Vector3d gEnd(geom->getEndPoint().x,
69  geom->getEndPoint().y,
70  0.0);
71  if (DrawUtil::fpCompare(offset, 0.0, 0.00001)) { //no offset
72  QGraphicsPathItem* item1 = lineFromPoints(pStart, gEnd, ls.getDashSpec());
73  item1->setPen(pen);
74  m_fillItems.push_back(item1);
75  if (!pStart.IsEqual(gStart, 0.00001)) {
76  QGraphicsPathItem* item2 = lineFromPoints(pStart, gStart, ls.getDashSpec().reversed());
77  item2->setPen(pen);
78  m_fillItems.push_back(item2);
79  }
80  } else { //offset - pattern start not in g
81  double remain = dashRemain(decodeDashSpec(ls.getDashSpec()), offset);
82  QGraphicsPathItem* shortItem = geomToStubbyLine(geom, remain, ls);
83  shortItem->setPen(pen);
84  m_fillItems.push_back(shortItem);
85  }
86  } else { //not dashed
87  QGraphicsPathItem* fillItem = geomToLine(geom, ls);
88  fillItem->setPen(pen);
89  m_fillItems.push_back(fillItem);
90  }
91 
92  if (m_segCount > m_maxSeg) {
93  Base::Console().Warning("PAT segment count exceeded: %ld\n", m_segCount);
94  break;
95  }
96  }
97 }

129 QGraphicsPathItem* PATPathMaker::geomToStubbyLine(TechDraw::BaseGeomPtr base, double remain, LineSet& ls)
130 {
131  QGraphicsPathItem* fillItem = new QGraphicsPathItem(m_parent);
132  Base::Vector3d start(base->getStartPoint().x,
133  base->getStartPoint().y,
134  0.0);
135  Base::Vector3d end(base->getEndPoint().x,
136  base->getEndPoint().y,
137  0.0);
138  double origLen = (end - start).Length();
139 
140  double appRemain = Rez::appX(remain);
141  Base::Vector3d newEnd = start + (ls.getUnitDir() * appRemain);
142 
143  double newLen = (newEnd - start).Length();
144 
145  if (newLen > origLen) {
146  newEnd = end;
147  }
148 
149  double offset = Rez::guiX(m_fillScale * ls.getDashSpec().length()) - remain;
150 
151  fillItem->setPath(dashedPPath(offsetDash(decodeDashSpec(ls.getDashSpec()), offset),
152  Rez::guiX(start),
153  Rez::guiX(newEnd)));
154  m_fillItems.push_back(fillItem);
155  return fillItem;

160 std::vector<double> PATPathMaker::decodeDashSpec(DashSpec patDash)
161 {
162  double penWidth = Rez::guiX(m_lineWidth);
163  double scale = m_fillScale;
164  double minPen = 0.01; //avoid trouble with cosmetic pen (zero width)?
165  if (penWidth <= minPen) {
166  penWidth = minPen;
167  }
168  std::vector<double> result;
169  for (auto& d: patDash.get()) {
170  double strokeLength;
171  if (DrawUtil::fpCompare(d, 0.0)) { //pat dot
172  strokeLength = penWidth;
173  } else { //pat mark/space
174  strokeLength = Rez::guiX(d);
175  }
176  result.push_back(scale * strokeLength);
177  }
178  return result;
179 }

182 QPainterPath PATPathMaker::dashedPPath(const std::vector<double> dashPattern, const Base::Vector3d start, const Base::Vector3d end)
183 {
184  QPainterPath result;
185  Base::Vector3d dir = (end - start);
186  dir.Normalize();
187  result.moveTo(start.x, -start.y);
188  Base::Vector3d currentPos = start;
189  if (dashPattern.empty()) {
190  result.lineTo(end.x, -end.y);
191  m_segCount++;
192  } else {
193  double lineLength = (end - start).Length();
194  double travel = 0.0;
195  Base::Vector3d lineProgress;
196  while (travel < lineLength) {
197  bool stop = false;
198  if (m_segCount > 10000) {
199  Base::Console().Warning("PAT segment count exceeded: %ld\n", m_segCount);
200  break;
201  }
202 
203  for (auto& d: dashPattern) {
204  travel += fabs(d);
205  Base::Vector3d segmentEnd = (currentPos + dir * fabs(d));
206  if ((start - segmentEnd).Length() > lineLength) { //don't draw past end of line
207  segmentEnd = end;
208  stop = true;
209  }
210  if (d < 0.0) {
211  result.moveTo(segmentEnd.x, -segmentEnd.y); //space
212  } else {
213  result.lineTo(segmentEnd.x, -segmentEnd.y); //mark
214  }
215  if (stop) {
216  break;
217  }
218  m_segCount++;
219  currentPos = segmentEnd;
220  }
221  }
222  }
223  return result;
224 }

229 std::vector<double> PATPathMaker::offsetDash(const std::vector<double> dashPattern, const double offset)
230 {
231  std::vector<double> result;
232  double length = 0.0;
233  for (auto& d: dashPattern) {
234  length += fabs(d);
235  }
236  if (offset > length) {
237  result = dashPattern;
238  return result;
239  }
240  //find the dash cell that includes offset
241  double accum = 0;
242  int i = 0;
243  for (auto& d:dashPattern) {
244  accum += fabs(d);
245  if (accum > offset) {
246  break;
247  }
248  i++;
249  }
250 
251  double firstCell = accum - offset;
252  if (dashPattern.at(i) < 0.0) { //offset found in a space cell
253  result.push_back(-1.0* firstCell);
254  } else {
255  result.push_back(firstCell);
256  }
257  unsigned int iCell = i + 1;
258  for ( ; iCell < dashPattern.size() ; iCell++) {
259  result.push_back(dashPattern.at(iCell));
260  }
261 
262  return result;
263 }

267 double PATPathMaker::dashRemain(const std::vector<double> dashPattern, const double offset)
268 {
269  double length = 0.0;
270  for (auto& d: dashPattern) {
271  length += fabs(d);
272  }
273  if (offset > length) {
274  return 0.0;
275  }
276  return length - offset;
277 }
