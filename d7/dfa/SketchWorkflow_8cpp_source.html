95  void handleSelectedBody(PartDesign::Body* activeBody)
96  {
97  App::DocumentObject* object = faceSelection.getObject();
98  std::vector<std::string> elements = faceSelection.getSubNames();
99 
100  // In case the selected face belongs to the body then it means its
101  // Display Mode Body is set to Tip. But the body face is not allowed
102  // to be used as support because otherwise it would cause a cyclic
103  // dependency. So, instead we use the tip object as reference.
104  // https://forum.freecad.org/viewtopic.php?f=3&t=37448
105  if (object == activeBody) {
106  App::DocumentObject* tip = activeBody->Tip.getValue();
107  if (tip && tip->isDerivedFrom(Part::Feature::getClassTypeId()) && elements.size() == 1) {
108  Gui::SelectionChanges msg;
109  msg.pDocName = faceSelection.getDocName();
110  msg.pObjectName = tip->getNameInDocument();
111  msg.pSubName = elements[0].c_str();
112  msg.pTypeName = tip->getTypeId().getName();
113 
114  faceSelection = Gui::SelectionObject{msg};
115 
116  // automatically switch to 'Through' mode
117  setThroughModeOfBody(activeBody);
118  }
119  }
120  }
121 
122  void throwIfInvalid()
123  {
124  App::DocumentObject* object = faceSelection.getObject();
125  std::vector<std::string> elements = faceSelection.getSubNames();
126 
127  Part::Feature* partobject = dynamic_cast<Part::Feature*>(object);
128  if (!partobject) {
129  throw WrongSelectionException();
130  }
131 
132  if (elements.size() != 1) {
133  throw WrongSelectionException();
134  }
135 
136  // get the selected sub shape (a Face)
137  const Part::TopoShape &shape = partobject->Shape.getValue();
138  Part::TopoShape subshape(shape.getSubShape(elements[0].c_str()));
139  if (subshape.isNull()) {
140  throw WrongSupportException();
141  }
142 
143  if (!subshape.isPlanar()) {
144  throw SupportNotPlanarException();
145  }
146  }
147 
148  std::string getSupport() const
149  {
150  return faceSelection.getAsPropertyLinkSubString();
151  }
152 
153  App::DocumentObject* getObject() const
154  {
155  return faceSelection.getObject();
156  }

386  void findDatumPlanes()
387  {
388  App::GeoFeatureGroupExtension *geoGroup = getGroupExtensionOfBody();
389  auto datumPlanes( appdocument->getObjectsOfType(PartDesign::Plane::getClassTypeId()) );
390  for (auto plane : datumPlanes) {
391  planes.push_back ( plane );
392  // Check whether this plane belongs to the active body
393  if ( activeBody->hasObject(plane) ) {
394  if ( !activeBody->isAfterInsertPoint ( plane ) ) {
395  validPlaneCount++;
396  status.push_back(PartDesignGui::TaskFeaturePick::validFeature);
397  } else {
398  status.push_back(PartDesignGui::TaskFeaturePick::afterTip);
399  }
400  } else {
401  PartDesign::Body *planeBody = PartDesign::Body::findBodyOf (plane);
402  if ( planeBody ) {
403  if ( ( geoGroup && geoGroup->hasObject ( planeBody, true ) ) ||
404  !App::GeoFeatureGroupExtension::getGroupOfObject (planeBody) ) {
405  status.push_back ( PartDesignGui::TaskFeaturePick::otherBody );
406  } else {
407  status.push_back ( PartDesignGui::TaskFeaturePick::otherPart );
408  }
409  } else {
410  if ( ( geoGroup && geoGroup->hasObject ( plane, true ) ) ||
411  !App::GeoFeatureGroupExtension::getGroupOfObject ( plane ) ) {
412  status.push_back ( PartDesignGui::TaskFeaturePick::otherPart );
413  } else {
414  status.push_back ( PartDesignGui::TaskFeaturePick::notInBody );
415  }
416  }
417  }
418  }
419  }
420 
421  void findShapeBinderPlanes()
422  {
423 
424  // Collect also shape binders consisting of a single planar face
425  auto shapeBinders( appdocument->getObjectsOfType(PartDesign::ShapeBinder::getClassTypeId()) );
426  auto binders( appdocument->getObjectsOfType(PartDesign::SubShapeBinder::getClassTypeId()) );
427  shapeBinders.insert(shapeBinders.end(),binders.begin(),binders.end());
428  for (auto binder : shapeBinders) {
429  // Check whether this plane belongs to the active body
430  if (activeBody->hasObject(binder)) {
431  Part::TopoShape shape = static_cast<Part::Feature*>(binder)->Shape.getShape();
432  if (shape.isPlanar()) {
433  if (!activeBody->isAfterInsertPoint (binder)) {
434  validPlaneCount++;
435  planes.push_back(binder);
436  status.push_back(PartDesignGui::TaskFeaturePick::validFeature);
437  }
438  }
439  }
440  }
441  }

530  void findAndSelectPlane()
531  {
532  App::Document* appdocument = guidocument->getDocument();
533  PlaneFinder planeFinder{appdocument, activeBody};
534  planeFinder.findBasePlanes();
535  planeFinder.findDatumPlanes();
536  planeFinder.findShapeBinderPlanes();
537 
538  std::vector<App::DocumentObject*> planes = planeFinder.getPlanes();
539  std::vector<PartDesignGui::TaskFeaturePick::featureStatus> status = planeFinder.getStatus();
540  unsigned validPlaneCount = planeFinder.countValidPlanes();
541 
542  //
543  // Lambda definitions
544  //
545  App::Document* documentOfBody = appdocument;
546  PartDesign::Body* partDesignBody = activeBody;
547 
548  // Determines if user made a valid selection in dialog
549  auto acceptFunction = [](const std::vector<App::DocumentObject*>& features) -> bool {
550  return !features.empty();
551  };
552 
553  // Called by dialog when user hits "OK" and accepter returns true
554  auto processFunction = [documentOfBody, partDesignBody](const std::vector<App::DocumentObject*>& features) {
555  SketchRequestSelection::createSketch(documentOfBody, partDesignBody, features);
556  };
557 
558  // Called by dialog for "Cancel", or "OK" if accepter returns false
559  std::string docname = documentOfBody->getName();
560  auto rejectFunction = [docname]() {
561  Gui::Document* document = Gui::Application::Instance->getDocument(docname.c_str());
562  if (document)
563  document->abortCommand();
564  };
565 
566  //
567  // End of lambda definitions
568  //
569 
570  if (validPlaneCount == 0) {
571  throw MissingPlanesException();
572  }
573  else if (validPlaneCount == 1) {
574  processFunction(planes);
575  }
576  else if (validPlaneCount > 1) {
577  checkForShownDialog();
578  Gui::Selection().clearSelection();
579 
580  // Show dialog and let user pick plane
581  Gui::Control().showDialog(new PartDesignGui::TaskDlgFeaturePick(planes, status, acceptFunction,
582  processFunction, true, rejectFunction));
583  }
584  }

673 void SketchWorkflow::tryCreateSketch()
674 {
675  auto result = shouldCreateBody();
676  auto shouldMakeBody = std::get<0>(result);
677  activeBody = std::get<1>(result);
678  if (shouldAbort(shouldMakeBody)) {
679  return;
680  }
681 
682  auto faceOrPlaneFilter = getFaceAndPlaneFilter();
683  SketchPreselection sketchOnFace{ guidocument, activeBody, faceOrPlaneFilter };
684 
685  if (sketchOnFace.matches()) {
686  // create Sketch on Face or Plane
687  sketchOnFace.createSupport();
688  sketchOnFace.createSketchOnSupport(sketchOnFace.getSupport());
689  }
690  else {
691  SketchRequestSelection requestSelection{ guidocument, activeBody };
692  requestSelection.findSupport();
693  }
694 }

696 std::tuple<bool, PartDesign::Body*> SketchWorkflow::shouldCreateBody()
697 {
698  auto shouldMakeBody{false};
699 
700  // We need either an active Body, or for there to be no Body
701  // objects (in which case, just make one) to make a new sketch.
702  // If we are inside a link, we need to use its placement.
703  App::DocumentObject *topParent;
704  PartDesign::Body *pdBody = PartDesignGui::getBody(/* messageIfNot = */ false, true, true, &topParent);
705  if (pdBody && topParent->isLink()) {
706  auto *xLink = dynamic_cast<App::Link *>(topParent);
707  pdBody->Placement.setValue(xLink->Placement.getValue());
708  }
709  if (!pdBody) {
710  if (appdocument->countObjectsOfType<PartDesign::Body>() == 0) {
711  shouldMakeBody = true;
712  }
713  else {
714  PartDesignGui::DlgActiveBody dia(Gui::getMainWindow(), appdocument);
715  if (dia.exec() == QDialog::Accepted) {
716  pdBody = dia.getActiveBody();
717  }
718  }
719  }
720 
721  return std::make_tuple(shouldMakeBody, pdBody);
722 }

729 std::tuple<Gui::SelectionFilter, Gui::SelectionFilter> SketchWorkflow::getFaceAndPlaneFilter() const
730 {
731  // Hint:
