79 void ViewProviderDrawingView::attach(App::DocumentObject *pcFeat)
80 {
81 // Base::Console().Message("VPDV::attach(%s)\n", pcFeat->getNameInDocument());
82  ViewProviderDocumentObject::attach(pcFeat);
83 
84  //NOLINTBEGIN
85  auto bnd = std::bind(&ViewProviderDrawingView::onGuiRepaint, this, sp::_1);
86  auto bndProgressMessage = std::bind(&ViewProviderDrawingView::onProgressMessage, this, sp::_1, sp::_2, sp::_3);
87  //NOLINTEND
88  auto feature = getViewObject();
89  if (feature) {

93  m_myName = feature->getNameInDocument();
94  }
95  connectGuiRepaint = feature->signalGuiPaint.connect(bnd);
96  connectProgressMessage = feature->signalProgressMessage.connect(bndProgressMessage);
97  //TODO: would be good to start the QGIV creation process here, but no guarantee we actually have
98  // MDIVP or QGVP yet.
99  // but parent page might. we may not be part of the document yet though!
100  // :( we're not part of the page yet either!
101  } else {
102  Base::Console().Warning("VPDV::attach has no Feature!\n");

149 void ViewProviderDrawingView::hide()
150 {
151  TechDraw::DrawView* obj = getViewObject();
152  if (!obj || obj->isRestoring())
153  return;
154 
155  if (obj->isDerivedFrom<TechDraw::DrawView>()) {
156  QGIView* qView = getQView();
157  if (qView) {
158  //note: hiding an item in the scene clears its selection status
159  // this confuses Gui::Selection.
160  // So we block selection changes while we are hiding the qgiv
161  // in FC Tree hiding does not change selection state.
162  // block/unblock selection protects against crash in Gui::SelectionSingleton::setVisible
163  MDIViewPage* mdi = getMDIViewPage();
164  if (mdi) { //if there is no mdivp, there is nothing to hide!
165  mdi->blockSceneSelection(true);
166  qView->hide();
167  ViewProviderDocumentObject::hide();
168  mdi->blockSceneSelection(false);
169  }
170  }
171  }

223 void ViewProviderDrawingView::updateData(const App::Property* prop)
224 {
225  TechDraw::DrawView *obj = getViewObject();
226  App::PropertyLink *ownerProp = obj->getOwnerProperty();
227 
228  //only move the view on X, Y change
229  if (prop == &obj->X
230  || prop == &obj->Y) {
231  QGIView* qgiv = getQView();
232  if (qgiv && !qgiv->isSnapping()) {
233  qgiv->QGIView::updateView(true);
234 
235  // Update also the owner/parent view, if there is any
236  if (ownerProp) {
237  auto owner = dynamic_cast<TechDraw::DrawView *>(ownerProp->getValue());
238  if (owner) {
239  auto page = dynamic_cast<QGSPage *>(qgiv->scene());
240  if (page) {
241  QGIView *ownerView = page->getQGIVByName(owner->getNameInDocument());
242  if (ownerView) {
243  ownerView->updateView();
244  }
245  }
246  }
247  }
248  }
249  }
250  else if (ownerProp && prop == ownerProp) {
251  QGIView* qgiv = getQView();
252  if (qgiv) {
253  QGIView *ownerView = nullptr;
254  auto owner = dynamic_cast<TechDraw::DrawView *>(ownerProp->getValue());
255  if (owner) {
256  auto page = dynamic_cast<QGSPage *>(qgiv->scene());
257  if (page) {
258  ownerView = page->getQGIVByName(owner->getNameInDocument());
259  }
260  }
261 
262  qgiv->switchParentItem(ownerView);
263  qgiv->updateView();
264  }
265  }
266 
267  Gui::ViewProviderDocumentObject::updateData(prop);
268 }

309 void ViewProviderDrawingView::multiParentPaint(std::vector<TechDraw::DrawPage*>& pages)
310 {
311  for (auto& p : pages) {
312  std::vector<App::DocumentObject*> views = p->Views.getValues();
313  for (auto& v: views) {
314  if (v != getViewObject()) { //should this be dv from onGuiRepaint?
315  continue;
316  }
317  //view v belongs to this page p
318  ViewProviderPage* vpPage = getViewProviderPage();
319  if (!vpPage) {
320  continue;
321  }
322  if (vpPage->getQGSPage()) {
323  QGIView* qView = dynamic_cast<QGIView *>(vpPage->getQGSPage()->findQViewForDocObj(v));
324  if (qView) {
325  qView->updateView(true);
326  }
327  }
328  }
329  }

400 void ViewProviderDrawingView::stackTop()
401 {
402  QGIView* qView = getQView();
403  if (!qView || !getViewProviderPage()) {
404  //no view, nothing to stack
405  return;
406  }
407  int maxZ = INT_MIN;
408  auto parent = qView->parentItem();
409  if (parent) {
410  //if we have a parentItem, we have to stack within the parentItem, not within the page
411  auto siblings = parent->childItems();
412  for (auto& child : siblings) {
413  if (child->zValue() > maxZ) {
414  maxZ = child->zValue();
415  }
416  }
417  } else {
418  //if we have no parentItem, we are a top level QGIView and we need to stack
419  //with respect to the other top level views on this page
420  std::vector<App::DocumentObject*> peerObjects = getViewProviderPage()->claimChildren();
421  Gui::Document* gDoc = getDocument();
422  for (auto& peer: peerObjects) {
423  auto vpPeer = gDoc->getViewProvider(peer);
424  ViewProviderDrawingView* vpdv = static_cast<ViewProviderDrawingView*>(vpPeer);
425  int z = vpdv->StackOrder.getValue();
426  if (z > maxZ) {
427  maxZ = z;
428  }
429  }
430  }
431  StackOrder.setValue(maxZ + 1);
432  qView->setStack(maxZ + 1);
433 }
