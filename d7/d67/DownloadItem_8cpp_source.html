254 void DownloadItem::init()

343 QString DownloadItem::saveFileName(const QString& directory) const
344 {
345  // Move this function into QNetworkReply to also get file name sent from the server
346  QString path = m_url.path();
347  if (!m_fileName.isEmpty()) {
348  path = m_fileName;
349  }
350  QFileInfo info(path);
351  QString baseName = info.completeBaseName();
352  QString endName = info.suffix();
353 
354  if (baseName.isEmpty()) {
355  baseName = QLatin1String("unnamed_download");
356  qDebug() << "DownloadManager:: downloading unknown file:" << m_url;
357  }
358  QString name = directory + baseName + QLatin1Char('.') + endName;
359  if (QFile::exists(name)) {
360  // already exists, don't overwrite
361  int i = 1;
362  do {
363  name = directory + baseName + QLatin1Char('-') + QString::number(i++) + QLatin1Char('.')
364  + endName;
365  } while (QFile::exists(name));
366  }
367  return name;
368 }

486 void DownloadItem::metaDataChanged()
487 {
488  // https://tools.ietf.org/html/rfc6266
489  if (m_reply->hasRawHeader(QByteArray("Content-Disposition"))) {
490  QByteArray header = m_reply->rawHeader(QByteArray("Content-Disposition"));
491  int index = header.indexOf("filename=");
492  if (index >= 0) {
493  header = header.mid(index + 9);
494  if (header.startsWith("\"") || header.startsWith("'")) {
495  header = header.mid(1);
496  }
497  if ((index = header.lastIndexOf("\"")) > 0) {
498  header = header.left(index);
499  }
500  else if ((index = header.lastIndexOf("'")) > 0) {
501  header = header.left(index);
502  }
503  m_fileName = QUrl::fromPercentEncoding(header);
504  }
505  // Sometimes "filename=" and "filename*=UTF-8''" is set.
506  // So, search for this too.
507  index = header.indexOf("filename*=UTF-8''");
508  if (index >= 0) {
509  header = header.mid(index + 17);
510  if (header.startsWith("\"") || header.startsWith("'")) {
511  header = header.mid(1);
512  }
513  if ((index = header.lastIndexOf("\"")) > 0) {
514  header = header.left(index);
515  }
516  else if ((index = header.lastIndexOf("'")) > 0) {
517  header = header.left(index);
518  }
519  m_fileName = QUrl::fromPercentEncoding(header);
520  }
521  }
522 
523  QUrl url = m_reply->url();
524 
525  // If this is a redirected url use this instead
526  QUrl redirectUrl = m_reply->attribute(QNetworkRequest::RedirectionTargetAttribute).toUrl();
527  if (!redirectUrl.isEmpty()) {
528  QString s = redirectUrl.toString();
529  std::cout << "Redirected to " << s.toStdString() << std::endl;
530 
531  QVariant header = m_reply->header(QNetworkRequest::LocationHeader);
532  QString loc = header.toString();
533  Q_UNUSED(loc);
534 
535  if (url != redirectUrl) {
536  url = redirectUrl;
537 
538  disconnect(m_reply, &QNetworkReply::readyRead, this, &DownloadItem::downloadReadyRead);
539 #if QT_VERSION < QT_VERSION_CHECK(5, 15, 0)
540  disconnect(m_reply,
541  qOverload<QNetworkReply::NetworkError>(&QNetworkReply::error),
542  this,
543  &DownloadItem::error);
544 #else
545  disconnect(m_reply, &QNetworkReply::errorOccurred, this, &DownloadItem::error);
546 #endif
547  disconnect(m_reply,
548  &QNetworkReply::downloadProgress,
549  this,
550  &DownloadItem::downloadProgress);
551  disconnect(m_reply,
552  &QNetworkReply::metaDataChanged,
553  this,
554  &DownloadItem::metaDataChanged);
555  disconnect(m_reply, &QNetworkReply::finished, this, &DownloadItem::finished);
556  m_reply->close();
557  m_reply->deleteLater();
558 
559  m_reply =
560  DownloadManager::getInstance()->networkAccessManager()->get(QNetworkRequest(url));
561  init();
562  }
563  }

580 void DownloadItem::updateInfoLabel()
581 {
582  qint64 bytesTotal = progressBar->maximum();
583  bool running = !downloadedSuccessfully();
584 
585  // update info label
586  double speed = m_bytesReceived * 1000.0 / m_downloadTime.elapsed();
587  double timeRemaining = ((double)(bytesTotal - m_bytesReceived)) / speed;
588  QString timeRemainingString = tr("seconds");
589  if (timeRemaining > 60) {
590  timeRemaining = timeRemaining / 60;
591  timeRemainingString = tr("minutes");
592  }
593  timeRemaining = floor(timeRemaining);
594 
595  // When downloading the eta should never be 0
596  if (timeRemaining == 0) {
597  timeRemaining = 1;
598  }
599 
600  QString info;
601  if (running) {
602  QString remaining;
603  if (bytesTotal != 0) {
604  remaining = tr("- %4 %5 remaining").arg(timeRemaining).arg(timeRemainingString);
605  }
606  info = QString(tr("%1 of %2 (%3/sec) %4"))
607  .arg(dataString(m_bytesReceived),
608  bytesTotal == 0 ? tr("?") : dataString(bytesTotal),
609  dataString((int)speed),
610  remaining);
611  }
612  else {
613  if (m_bytesReceived == bytesTotal) {
614  info = dataString(m_output.size());
615  }
616  else {
617  info =
618  tr("%1 of %2 - Stopped").arg(dataString(m_bytesReceived), dataString(bytesTotal));
619  }
620  }
621  downloadInfoLabel->setText(info);
622 }
