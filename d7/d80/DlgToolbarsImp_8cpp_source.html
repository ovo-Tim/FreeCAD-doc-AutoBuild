190 void DlgCustomToolbars::importCustomToolbars(const QByteArray& name)
191 {
192  ParameterGrp::handle hGrp =
193  App::GetApplication().GetUserParameter().GetGroup("BaseApp")->GetGroup("Workbench");
194  const char* subgroup = (type == Toolbar ? "Toolbar" : "Toolboxbar");
195  if (!hGrp->HasGroup(name.constData())) {
196  return;
197  }
198  hGrp = hGrp->GetGroup(name.constData());
199  if (!hGrp->HasGroup(subgroup)) {
200  return;
201  }
202  hGrp = hGrp->GetGroup(subgroup);
203  std::string separator = "Separator";
204 
205  std::vector<Base::Reference<ParameterGrp>> hGrps = hGrp->GetGroups();
206  CommandManager& rMgr = Application::Instance->commandManager();
207  for (const auto& hGrp : hGrps) {
208  // create a toplevel item
209  auto toplevel = new QTreeWidgetItem(ui->toolbarTreeWidget);
210  bool active = hGrp->GetBool("Active", true);
211  toplevel->setCheckState(0, (active ? Qt::Checked : Qt::Unchecked));
212 
213  // get the elements of the subgroups
214  std::vector<std::pair<std::string, std::string>> items = hGrp->GetASCIIMap();
215  for (const auto& it2 : items) {
216  // since we have stored the separators to the user parameters as (key, pair) we had to
217  // make sure to use a unique key because otherwise we cannot store more than
218  // one.
219  if (it2.first.substr(0, separator.size()) == separator) {
220  auto item = new QTreeWidgetItem(toplevel);
221  item->setText(0, tr("<Separator>"));
222  item->setData(0, Qt::UserRole, QByteArray("Separator"));
223  item->setSizeHint(0, QSize(32, 32));
224  }
225  else if (it2.first == "Name") {
226  QString toolbarName = QString::fromUtf8(it2.second.c_str());
227  toplevel->setText(0, toolbarName);
228  }
229  else {
230  Command* pCmd = rMgr.getCommandByName(it2.first.c_str());
231  if (pCmd) {
232  // command name
233  auto* item = new QTreeWidgetItem(toplevel);
234  item->setText(0, Action::commandMenuText(pCmd));
235  item->setToolTip(0, Action::commandToolTip(pCmd));
236  item->setData(0, Qt::UserRole, QByteArray(it2.first.c_str()));
237  if (pCmd->getPixmap()) {
238  item->setIcon(0, BitmapFactory().iconFromTheme(pCmd->getPixmap()));
239  }
240  item->setSizeHint(0, QSize(32, 32));
241  }
242  else {
243  // If corresponding module is not yet loaded do not lose the entry
244  auto item = new QTreeWidgetItem(toplevel);
245  item->setText(
246  0,
247  tr("%1 module not loaded").arg(QString::fromStdString(it2.second)));
248  item->setData(0, Qt::UserRole, QByteArray(it2.first.c_str()));
249  item->setData(0, Qt::WhatsThisPropertyRole, QByteArray(it2.second.c_str()));
250  item->setSizeHint(0, QSize(32, 32));
251  }
252  }
253  }
254  }

257 void DlgCustomToolbars::exportCustomToolbars(const QByteArray& workbench)
258 {
259  ParameterGrp::handle hGrp =
260  App::GetApplication().GetUserParameter().GetGroup("BaseApp")->GetGroup("Workbench");
261  const char* subgroup = (type == Toolbar ? "Toolbar" : "Toolboxbar");
262  hGrp = hGrp->GetGroup(workbench.constData())->GetGroup(subgroup);
263  hGrp->Clear();
264 
265  CommandManager& rMgr = Application::Instance->commandManager();
266  for (int i = 0; i < ui->toolbarTreeWidget->topLevelItemCount(); i++) {
267  QTreeWidgetItem* toplevel = ui->toolbarTreeWidget->topLevelItem(i);
268  QString groupName = QString::fromLatin1("Custom_%1").arg(i + 1);
269  QByteArray toolbarName = toplevel->text(0).toUtf8();
270  ParameterGrp::handle hToolGrp = hGrp->GetGroup(groupName.toLatin1());
271  hToolGrp->SetASCII("Name", toolbarName.constData());
272  hToolGrp->SetBool("Active", toplevel->checkState(0) == Qt::Checked);
273 
274  // since we store the separators to the user parameters as (key, pair) we must
275  // make sure to use a unique key because otherwise we cannot store more than
276  // one.
277  int suffixSeparator = 1;
278  for (int j = 0; j < toplevel->childCount(); j++) {
279  QTreeWidgetItem* child = toplevel->child(j);
280  QByteArray commandName = child->data(0, Qt::UserRole).toByteArray();
281  if (commandName == "Separator") {
282  QByteArray key = commandName + QByteArray::number(suffixSeparator);
283  suffixSeparator++;
284  hToolGrp->SetASCII(key, commandName);
285  }
286  else {
287  Command* pCmd = rMgr.getCommandByName(commandName);
288  if (pCmd) {
289  hToolGrp->SetASCII(pCmd->getName(), pCmd->getAppModuleName());
290  }
291  else {
292  QByteArray moduleName = child->data(0, Qt::WhatsThisPropertyRole).toByteArray();
293  hToolGrp->SetASCII(commandName, moduleName);
294  }
295  }
296  }
297  }

329 void DlgCustomToolbars::onMoveActionLeftButtonClicked()
330 {
331  QTreeWidgetItem* item = ui->toolbarTreeWidget->currentItem();
332  if (item && item->parent() && item->isSelected()) {
333  QTreeWidgetItem* parent = item->parent();
334  int index = parent->indexOfChild(item);
335  parent->takeChild(index);
336 
337  // In case a separator should be moved we have to count the separators
338  // which come before this one.
339  // This is needed so that we can distinguish in removeCustomCommand
340  // which separator it is.
341  QByteArray data = item->data(0, Qt::UserRole).toByteArray();
342  if (data == "Separator") {
343  int countSep = 1;
344  for (int i = 0; i < index - 1; i++) {
345  QByteArray d = parent->child(i)->data(0, Qt::UserRole).toByteArray();
346  if (d == "Separator") {
347  countSep++;
348  }
349  }
350 
351  data += QByteArray::number(countSep);
352  }
353  removeCustomCommand(parent->text(0), data);
354  delete item;
355  }
356 
357  QVariant data = ui->workbenchBox->itemData(ui->workbenchBox->currentIndex(), Qt::UserRole);
358  QString workbench = data.toString();
359  exportCustomToolbars(workbench.toLatin1());
