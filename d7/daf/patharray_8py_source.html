28 """Provides the object code for the PathArray object.
29 
30 The copies will be placed along a path like a polyline, spline, or bezier
31 curve, and along the selected subelements.
32 
33 To Do
34 -----
35 The `'PathSubelements'` property must be changed in type, as it does not need
36 to be an `App::PropertyLinkSubList`.
37 A `LinkSubList` is to select multiple subelements (edges) from multiple
38 objects. However, since we need to select a `'Path Object'` already,
39 which is a single object, the subelements that we can choose must belong
40 to this `'Path Object'` only.
41 
42 Therefore, the correct property that must be used is `App::PropertyLinkSub`.
43 Then in the property editor we will be unable to select more than one object
44 thus preventing errors of the subelements not matching the `'PathObject'`.
45 
46 In fact, both `'PathObject'` and `'PathSubelements'`
47 could be handled with a single `App::PropertyLinkSub` property,
48 as this property can be used to select a single object,
49 or a single object with its subelements.
50 
51 In the future, we could migrate the properties, or outright break
52 compatibility with older objects by changing both properties
53 `'PathObject'` and `'PathSubelements'`.
54 
55 An alternative to this would be to use a single `App::PropertyLinkSubList`.
56 This would allow us to build PathArrays on multiple objects and multiple
57 subelements (edges) of those objects at the same time. However, to do this,
58 the logic in `execute` would have to be changed to account for multiple
59 objects. Therefore, the first solution is simpler, that is, using
60 a single property of type `App::PropertyLinkSub`.
61 """

84 class PathArray(DraftLink):
85  """The Draft Path Array object.
86 
87  The object distributes copies of an object along a path like a polyline,
88  spline, or bezier curve.
89 
90  Attributes
91  ----------
92  Align: bool
93  It defaults to `False`.
94  It sets whether the object will be specially aligned to the path.
95 
96  AlignMode: str
97  It defaults to `'Original'`.
98  Indicates the type of alignment that will be calculated when
99  `Align` is `True`.
100 
101  `'Original'` mode is the historic `'Align'` for old (v0.18) documents.
102  It is not really the Fernat alignment. It uses the normal parameter
103  from `getNormal` (or the default) as a constant, it does not calculate
104  curve normal.
105  `X` is curve tangent, `Y` is normal parameter, `Z` is the cross product
106  `X` x `Y`.
107 
108  `'Tangent'` mode is similar to `Original`, but includes a pre-rotation
109  (in execute) to align the `Base` object's `X` to `TangentVector`,
110  then `X` follows curve tangent, normal input parameter
111  is the Z component.
112 
113  If `ForceVertical` is `True`, the normal parameter from `getNormal`
114  is ignored, and `X` is curve tangent, `Z` is `VerticalVector`,
115  and `Y` is the cross product `X` x `Z`.
116 
117  `'Frenet'` mode orients the copies to a coordinate system
118  along the path.
119  `X` is tangent to curve, `Y` is curve normal, `Z` is curve binormal.
120  If normal cannot be computed, for example, in a straight line,
121  the default is used.
122 
123  ForceVertical: bool
124  It defaults to `False`.
125  If it is `True`, and `AlignMode` is `'Original'` or `'Tangent'`,
126  it will use the vector in `VerticalVector` as the `Z` axis.
127 
128  StartOffset: float
129  It defaults to 0.0.
130  It is the length from the start of the path to the first copy.
131 
132  EndOffset: float
133  It defaults to 0.0.
134  It is the length from the end of the path to the last copy.
135  """

309  """Execute when the object is created or recomputed."""
310  if self.props_changed_placement_only(obj) \
311  or not obj.Base \

438 def placements_on_path(shapeRotation, pathwire, count, xlate, align,
439  mode="Original", forceNormal=False,
440  normalOverride=None,
441  startOffset=0.0, endOffset=0.0):
442  """Calculate the placements of a shape along a given path.
443 
444  Copies will be distributed evenly.
445  """
446  if mode == "Frenet":
447  forceNormal = False
448 
449  if forceNormal and normalOverride:
450  normal = normalOverride
451  else:
452  normal = DraftGeomUtils.get_normal(pathwire)
453  if normal is None:
454  normal = App.Vector(0, 0, 1)
455 
456  path = Part.__sortEdges__(pathwire.Edges)
457 
458  # find cumulative edge end distance
459  cdist = 0
460  ends = []
461  for e in path:
462  cdist += e.Length
463  ends.append(cdist)
464 
465  if startOffset > (cdist - 1e-6):
466  if startOffset != 0:
467  _wrn(
468  translate(
469  "draft",
470  "Start Offset too large for path length. Using zero instead."
471  )
472  )
473  start = 0
474  else:
475  start = startOffset
476 
477  if endOffset > (cdist - start - 1e-6):
478  if endOffset != 0:
479  _wrn(
480  translate(
481  "draft",
482  "End Offset too large for path length minus Start Offset. Using zero instead."
483  )
484  )
485  end = 0
486  else:
487  end = endOffset
488 
489  cdist = cdist - start - end
490  count = max(count, 1)
491  n = count if (DraftGeomUtils.isReallyClosed(pathwire) and not (start or end)) else count - 1
492  n = max(n, 1)
493  step = cdist / n
494  remains = 0
495  travel = start
496  placements = []
497 
498  for i in range(count):
499  # which edge in path should contain this shape?
500  for j in range(len(ends)):
501  if travel <= ends[j]:
502  iend = j
503  remains = ends[iend] - travel
504  offset = path[iend].Length - remains
505  break
506  else:
507  # avoids problems with float math travel > ends[-1]
508  iend = len(ends) - 1
509  offset = path[iend].Length
510 
511  # place shape at proper spot on proper edge
512  pt = path[iend].valueAt(get_parameter_from_v0(path[iend], offset))
513  place = calculate_placement(shapeRotation,
514  path[iend], offset,
515  pt, xlate, align, normal,
516  mode, forceNormal)
517  placements.append(place)
518 
519  travel += step
520 
521  return placements

527 def calculate_placement(globalRotation,
528  edge, offset, RefPt, xlate, align,
529  normal=App.Vector(0.0, 0.0, 1.0),
530  mode="Original", overrideNormal=False):
531  """Orient shape in the local coordinate system at parameter offset.
532 
533  http://en.wikipedia.org/wiki/Euler_angles (previous version)
534  http://en.wikipedia.org/wiki/Quaternions
535  """
536  # Default Placement:
537  placement = App.Placement()
538  placement.Rotation = globalRotation
539  placement.Base = RefPt + placement.Rotation.multVec(xlate)
540 
541  if not align:
542  return placement
543 
544  tol = 1e-6 # App.Rotation() tolerance is 1e-7. Shorter vectors are ignored.
545  nullv = App.Vector()
546 
547  t = edge.tangentAt(get_parameter_from_v0(edge, offset))
548  if t.isEqual(nullv, tol):
549  _wrn(translate("draft", "Length of tangent vector is zero. Copy not aligned."))
550  return placement
551 
552  # If the length of the normal is zero or if it is parallel to the tangent,
553  # we make the vectors equal (n = t). The App.Rotation() algorithm will
554  # then replace the normal with a default axis.
555  # For the vector with the lowest App.Rotation() priority we use a null
556  # vector. Calculating this binormal would not make sense in the mentioned
557  # cases. And in all other cases calculating it is not necessary as
558  # App.Rotation() will ignore it.
559 
560  if mode in ("Original", "Tangent"):
561  n = normal
562  if n.isEqual(nullv, tol):
563  _wrn(translate("draft", "Length of normal vector is zero. Using a default axis instead."))
564  n = t
565  else:
566  n_nor = n.normalize()
567  t_nor = t.normalize()
568  if n_nor.isEqual(t_nor, tol) or n_nor.isEqual(t_nor.negative(), tol):
569  _wrn(translate("draft", "Tangent and normal vectors are parallel. Normal replaced by a default axis."))
570  n = t
571 
572  if overrideNormal:
573  newRot = App.Rotation(t, nullv, n, "XZY") # priority = "XZY"
574  else:
575  newRot = App.Rotation(t, n, nullv, "XYZ") # priority = "XYZ"
576 
577  elif mode == "Frenet":
578  try:
579  n = edge.normalAt(get_parameter_from_v0(edge, offset))
580  except App.Base.FreeCADError: # no/infinite normals here
581  _wrn(translate("draft", "Cannot calculate normal vector. Using the default normal instead."))
582  n = normal
583 
584  if n.isEqual(nullv, tol):
585  _wrn(translate("draft", "Length of normal vector is zero. Using a default axis instead."))
586  n = t
587  else:
588  n_nor = n.normalize()
589  t_nor = t.normalize()
590  if n_nor.isEqual(t_nor, tol) or n_nor.isEqual(t_nor.negative(), tol):
591  _wrn(translate("draft", "Tangent and normal vectors are parallel. Normal replaced by a default axis."))
592  n = t
593 
594  newRot = App.Rotation(t, n, nullv, "XYZ") # priority = "XYZ"
595 
596  else:
597  _err(translate("draft", "AlignMode {} is not implemented").format(mode))
598  return placement
599 
600  placement.Rotation = newRot.multiply(globalRotation)
601  placement.Base = RefPt + placement.Rotation.multVec(xlate)
602  return placement
