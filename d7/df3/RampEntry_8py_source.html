493  def createRampMethod1(self, rampedges, p0, projectionlen, rampangle):
494  """
495  This method generates ramp with following pattern:
496  1. Start from the original startpoint of the plunge
497  2. Ramp down along the path that comes after the plunge
498  3. When reaching the Z level of the original plunge, return back to the beginning
499  by going the path backwards until the original plunge end point is reached
500  4. Continue with the original path
501 
502  This method causes many unnecessary moves with tool down.
503  """
504  outedges = []
505  rampremaining = projectionlen
506  curPoint = p0 # start from the upper point of plunge
507  done = False
508  goingForward = True
509  i = 0
510  while not done:
511  for i, redge in enumerate(rampedges):
512  if redge.Length >= rampremaining:
513  # will reach end of ramp within this edge, needs to be split
514  p1 = self.getSplitPoint(redge, rampremaining)
515  splitEdge = Path.Geom.splitEdgeAt(redge, p1)
516  Path.Log.debug("Ramp remaining: {}".format(rampremaining))
517  Path.Log.debug(
518  "Got split edge (index: {}) (total len: {}) with lengths: {}, {}".format(
519  i, redge.Length, splitEdge[0].Length, splitEdge[1].Length
520  )
521  )
522  # ramp ends to the last point of first edge
523  p1 = splitEdge[0].valueAt(splitEdge[0].LastParameter)
524  outedges.append(self.createRampEdge(splitEdge[0], curPoint, p1))
525  # now we have reached the end of the ramp. Go back to plunge position with constant Z
526  # start that by going to the beginning of this splitEdge
527  if goingForward:
528  outedges.append(
529  self.createRampEdge(
530  splitEdge[0], p1, redge.valueAt(redge.FirstParameter)
531  )
532  )
533  else:
534  # if we were reversing, we continue to the same direction as the ramp
535  outedges.append(
536  self.createRampEdge(
537  splitEdge[0], p1, redge.valueAt(redge.LastParameter)
538  )
539  )
540  done = True
541  break
542  else:
543  deltaZ = redge.Length / math.tan(math.radians(rampangle))
544  newPoint = FreeCAD.Base.Vector(
545  redge.valueAt(redge.LastParameter).x,
546  redge.valueAt(redge.LastParameter).y,
547  curPoint.z - deltaZ,
548  )
549  outedges.append(self.createRampEdge(redge, curPoint, newPoint))
550  curPoint = newPoint
551  rampremaining = rampremaining - redge.Length
552 
553  if not done:
554  # we did not reach the end of the ramp going this direction, lets reverse.
555  rampedges = self.getreversed(rampedges)
556  Path.Log.debug("Reversing")
557  if goingForward:
558  goingForward = False
559  else:
560  goingForward = True
561  # now we need to return to original position.
562  if goingForward:
563  # if the ramp was going forward, the return edges are the edges we already covered in ramping,
564  # except the last one, which was already covered inside for loop. Direction needs to be reversed also
565  returnedges = self.getreversed(rampedges[:i])
566  else:
567  # if the ramp was already reversing, the edges needed for return are the ones
568  # which were not covered in ramp
569  returnedges = rampedges[(i + 1) :]
570 
571  # add the return edges:
572  outedges.extend(returnedges)
573 
574  return outedges

661  def createRampMethod2(self, rampedges, p0, projectionlen, rampangle):
662  """
663  This method generates ramp with following pattern:
664  1. Start from the original startpoint of the plunge
665  2. Calculate the distance on the path which is needed to implement the ramp
666  and travel that distance while maintaining start depth
667  3. Start ramping while traveling the original path backwards until reaching the
668  original plunge end point
669  4. Continue with the original path
670  """
671  outedges = []
672  rampremaining = projectionlen
673  curPoint = p0 # start from the upper point of plunge
674  if Path.Geom.pointsCoincide(
675  Path.Geom.xy(p0),
676  Path.Geom.xy(rampedges[-1].valueAt(rampedges[-1].LastParameter)),
677  ):
678  Path.Log.debug("The ramp forms a closed wire, needless to move on original Z height")
679  else:
680  for i, redge in enumerate(rampedges):
681  if redge.Length >= rampremaining:
682  # this edge needs to be split
683  p1 = self.getSplitPoint(redge, rampremaining)
684  splitEdge = Path.Geom.splitEdgeAt(redge, p1)
685  Path.Log.debug(
686  "Got split edges with lengths: {}, {}".format(
687  splitEdge[0].Length, splitEdge[1].Length
688  )
689  )
690  # ramp starts at the last point of first edge
691  p1 = splitEdge[0].valueAt(splitEdge[0].LastParameter)
692  p1.z = p0.z
693  outedges.append(self.createRampEdge(splitEdge[0], curPoint, p1))
694  # now we have reached the beginning of the ramp.
695  # start that by going to the beginning of this splitEdge
696  deltaZ = splitEdge[0].Length / math.tan(math.radians(rampangle))
697  newPoint = FreeCAD.Base.Vector(
698  splitEdge[0].valueAt(splitEdge[0].FirstParameter).x,
699  splitEdge[0].valueAt(splitEdge[0].FirstParameter).y,
700  p1.z - deltaZ,
701  )
702  outedges.append(self.createRampEdge(splitEdge[0], p1, newPoint))
703  curPoint = newPoint
704  elif i == len(rampedges) - 1:
705  # last ramp element but still did not reach the full length?
706  # Probably a rounding issue on floats.
707  # Lets start the ramp anyway
708  p1 = redge.valueAt(redge.LastParameter)
709  p1.z = p0.z
710  outedges.append(self.createRampEdge(redge, curPoint, p1))
711  # and go back that edge
712  deltaZ = redge.Length / math.tan(math.radians(rampangle))
713  newPoint = FreeCAD.Base.Vector(
714  redge.valueAt(redge.FirstParameter).x,
715  redge.valueAt(redge.FirstParameter).y,
716  p1.z - deltaZ,
717  )
718  outedges.append(self.createRampEdge(redge, p1, newPoint))
719  curPoint = newPoint
720 
721  else:
722  # we are traveling on start depth
723  newPoint = FreeCAD.Base.Vector(
724  redge.valueAt(redge.LastParameter).x,
725  redge.valueAt(redge.LastParameter).y,
726  p0.z,
727  )
728  outedges.append(self.createRampEdge(redge, curPoint, newPoint))
729  curPoint = newPoint
730  rampremaining = rampremaining - redge.Length
731 
732  # the last edge got handled previously
733  rampedges.pop()
734  # ramp backwards to the plunge position
735  for i, redge in enumerate(reversed(rampedges)):
736  deltaZ = redge.Length / math.tan(math.radians(rampangle))
737  newPoint = FreeCAD.Base.Vector(
738  redge.valueAt(redge.FirstParameter).x,
739  redge.valueAt(redge.FirstParameter).y,
740  curPoint.z - deltaZ,
741  )
742  if i == len(rampedges) - 1:
743  # make sure that the last point of the ramps ends to the original position
744  newPoint = redge.valueAt(redge.FirstParameter)
745  outedges.append(self.createRampEdge(redge, curPoint, newPoint))
746  curPoint = newPoint
