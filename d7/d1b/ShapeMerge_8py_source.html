57 def splitIntoGroupsBySharing(list_of_shapes, element_extractor, split_connections = []):
58  """splitIntoGroupsBySharing(list_of_shapes, element_type, split_connections = []): find,
59  which shapes in list_of_shapes are connected into groups by sharing elements.
60 
61  element_extractor: function that takes shape as input, and returns list of shapes.
62 
63  split_connections: list of shapes to exclude when testing for connections. Use to
64  split groups on purpose.
65 
66  return: list of lists of shapes. Top-level list is list of groups; bottom level lists
67  enumerate shapes of a group."""
68 
69  split_connections = set([HashableShape(element) for element in split_connections])
70 
71  groups = [] #list of tuples (shapes,elements). Shapes is a list of plain shapes. Elements is a set of HashableShapes - all elements of shapes in the group, excluding split_connections.
72 
73  # add shapes to the list of groups, one by one. If not connected to existing groups,
74  # new group is created. If connected, shape is added to groups, and the groups are joined.
75  for shape in list_of_shapes:
76  shape_elements = set([HashableShape(element) for element in element_extractor(shape)])
77  shape_elements.difference_update(split_connections)
78  #search if shape is connected to any groups
79  connected_to = []
80  not_in_connected_to = []
81  for iGroup in range(len(groups)):
82  connected = False
83  for element in shape_elements:
84  if element in groups[iGroup][1]:
85  connected_to.append(iGroup)
86  connected = True
87  break
88  else:
89  # `break` not invoked, so `connected` is false
90  not_in_connected_to.append(iGroup)
91 
92  # test if we need to join groups
93  if len(connected_to)>1:
94  #shape bridges a gap between some groups. Join them into one.
95  #rebuilding list of groups. First, add the new "supergroup", then add the rest
96  groups_new = []
97 
98  supergroup = (list(),set())
99  for iGroup in connected_to:
100  supergroup[0].extend( groups[iGroup][0] )# merge lists of shapes
101  supergroup[1].update( groups[iGroup][1] )# merge lists of elements
102  groups_new.append(supergroup)
103 
104  l_groups = len(groups)
105  groups_new.extend([groups[i_group] \
106  for i_group in not_in_connected_to \
107  if i_group < l_groups])
108  groups = groups_new
109  connected_to = [0]
110 
111  # add shape to the group it is connected to (if to many, the groups should have been unified by the above code snippet)
112  if len(connected_to) > 0:
113  iGroup = connected_to[0]
114  groups[iGroup][0].append(shape)
115  groups[iGroup][1].update( shape_elements )
116  else:
117  newgroup = ([shape], shape_elements)
118  groups.append(newgroup)
119 
120  # done. Discard unnecessary data and return result.
121  return [shapes for shapes,elements in groups]

123 def mergeSolids(list_of_solids_compsolids, flag_single = False, split_connections = [], bool_compsolid = False):
124  """mergeSolids(list_of_solids, flag_single = False): merges touching solids that share
125  faces. If flag_single is True, it is assumed that all solids touch, and output is a
126  single solid. If flag_single is False, the output is a compound containing all
127  resulting solids.
128 
129  Note. CompSolids are treated as lists of solids - i.e., merged into solids."""
130 
131  solids = []
132  for sh in list_of_solids_compsolids:
133  solids.extend(sh.Solids)
134  if flag_single:
135  cs = Part.CompSolid(solids)
136  return cs if bool_compsolid else Part.makeSolid(cs)
137  else:
138  if len(solids)==0:
139  return Part.Compound([])
140  groups = splitIntoGroupsBySharing(solids, lambda sh: sh.Faces, split_connections)
141  if bool_compsolid:
142  merged_solids = [Part.CompSolid(group) for group in groups]
143  else:
144  merged_solids = [Part.makeSolid(Part.CompSolid(group)) for group in groups]
145  return Part.makeCompound(merged_solids)

171 def mergeShapes(list_of_shapes, flag_single = False, split_connections = [], bool_compsolid = False):
172  """mergeShapes(list_of_shapes, flag_single = False, split_connections = [], bool_compsolid = False):
173  merges list of edges/wires into wires, faces/shells into shells, solids/compsolids
174  into solids or compsolids.
175 
176  list_of_shapes: shapes to merge. Shapes must share elements in order to be merged.
177 
178  flag_single: assume all shapes in list are connected. If False, return is a compound.
179  If True, return is the single piece (e.g. a shell).
180 
181  split_connections: list of shapes that are excluded when searching for connections.
182  This can be used for example to split a wire in two by supplying vertices where to
183  split. If flag_single is True, this argument is ignored.
184 
185  bool_compsolid: determines behavior when dealing with solids/compsolids. If True,
186  result is compsolid/compound of compsolids. If False, all touching solids and
187  compsolids are unified into single solids. If not merging solids/compsolids, this
188  argument is ignored."""
189 
190  if len(list_of_shapes)==0:
191  return Part.Compound([])
192  args = [list_of_shapes, flag_single, split_connections]
193  dim = dimensionOfShapes(list_of_shapes)
194  if dim == 0:
195  return mergeVertices(*args)
196  elif dim == 1:
197  return mergeWires(*args)
198  elif dim == 2:
199  return mergeShells(*args)
200  elif dim == 3:
201  args.append(bool_compsolid)
202  return mergeSolids(*args)
203  else:
204  assert(dim >= 0 and dim <= 3)

218 def dimensionOfShapes(list_of_shapes):
219  """dimensionOfShapes(list_of_shapes): returns dimension (0D, 1D, 2D, or 3D) of shapes
220  in the list. If dimension of shapes varies, TypeError is raised."""
221 
222  dimensions = [["Vertex"], ["Edge","Wire"], ["Face","Shell"], ["Solid","CompSolid"]]
223  dim = -1

228  if dim == -1:
