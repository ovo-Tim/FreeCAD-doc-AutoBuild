82 App::Property* DlgSheetConf::prepare(CellAddress& from,
83  CellAddress& to,
84  std::string& rangeConf,
85  ObjectIdentifier& path,
86  bool init)
87 {
88  from = sheet->getCellAddress(ui->lineEditStart->text().trimmed().toLatin1().constData());
89  to = sheet->getCellAddress(ui->lineEditEnd->text().trimmed().toLatin1().constData());
90 
91  if (from.col() >= to.col()) {
92  FC_THROWM(Base::RuntimeError, "Invalid cell range");
93  }
94 
95  // Setup row as parameters, and column as configurations
96  to.setRow(from.row());
97 
98  CellAddress confFrom(from.row() + 1, from.col());
99  rangeConf = confFrom.toString();
100  // rangeConf is supposed to hold the range of string cells, each
101  // holding the name of a configuration. The '|' below indicates a
102  // growing but continuous column, so that we can auto include new
103  // configurations. We'll bind the string list to a
104  // PropertyEnumeration for dynamical switching of the
105  // configuration.
106  rangeConf += ":|";
107 
108  if (!init) {
109  std::string exprTxt(ui->lineEditProp->text().trimmed().toUtf8().constData());
110  ExpressionPtr expr;
111  try {
112  expr.reset(App::Expression::parse(sheet, exprTxt));
113  }
114  catch (Base::Exception& e) {
115  e.ReportException();
116  FC_THROWM(Base::RuntimeError, "Failed to parse expression for property");
117  }
118  if (expr->hasComponent()
119  || !expr->isDerivedFrom(App::VariableExpression::getClassTypeId())) {
120  FC_THROWM(Base::RuntimeError, "Invalid property expression: " << expr->toString());
121  }
122 
123  path = static_cast<App::VariableExpression*>(expr.get())->getPath();
124  auto obj = path.getDocumentObject();
125  if (!obj) {
126  FC_THROWM(Base::RuntimeError, "Invalid object referenced in: " << expr->toString());
127  }
128 
129  int pseudoType;
130  auto prop = path.getProperty(&pseudoType);
131  if (pseudoType
132  || (prop
133  && (!prop->isDerivedFrom(App::PropertyEnumeration::getClassTypeId())
134  || !prop->testStatus(App::Property::PropDynamic)))) {
135  FC_THROWM(Base::RuntimeError, "Invalid property referenced in: " << expr->toString());
136  }
137  return prop;
138  }
139 
140  Cell* cell = sheet->getCell(from);
141  if (cell && cell->getExpression()) {
142  auto expr = cell->getExpression();
143  if (expr->isDerivedFrom(FunctionExpression::getClassTypeId())) {
144  auto fexpr = Base::freecad_dynamic_cast<FunctionExpression>(cell->getExpression());
145  if (fexpr
146  && (fexpr->getFunction() == FunctionExpression::HREF
147  || fexpr->getFunction() == FunctionExpression::HIDDENREF)
148  && fexpr->getArgs().size() == 1) {
149  expr = fexpr->getArgs().front();
150  }
151  }
152  auto vexpr = Base::freecad_dynamic_cast<VariableExpression>(expr);
153  if (vexpr) {
154  auto prop =
155  Base::freecad_dynamic_cast<PropertyEnumeration>(vexpr->getPath().getProperty());
156  if (prop) {
157  auto obj = Base::freecad_dynamic_cast<DocumentObject>(prop->getContainer());
158  if (obj && prop->hasName()) {
159  path = ObjectIdentifier(sheet);
160  path.setDocumentObjectName(obj, true);
161  path << ObjectIdentifier::SimpleComponent(prop->getName());
162  return prop;
163  }
164  }
165  }
166  }
167  return nullptr;
168 }

170 void DlgSheetConf::accept()
171 {
172  bool commandActive = false;
173  try {
174  std::string rangeConf;
175  CellAddress from, to;
176  ObjectIdentifier path;
177  App::Property* prop = prepare(from, to, rangeConf, path, false);
178 
179  Range range(from, to);
180 
181  // check rangeConf, make sure it is a sequence of string only
182  Range r(sheet->getRange(rangeConf.c_str()));
183  do {
184  auto cell = sheet->getCell(*r);
185  if (cell && cell->getExpression()) {
186  ExpressionPtr expr(cell->getExpression()->eval());
187  if (expr->isDerivedFrom(StringExpression::getClassTypeId())) {
188  continue;
189  }
190  }
191  FC_THROWM(Base::RuntimeError,
192  "Expects cell " << r.address() << " evaluates to string.\n"
193  << rangeConf
194  << " is supposed to contain a list of configuration names");
195  } while (r.next());
196 
197  std::string exprTxt(ui->lineEditProp->text().trimmed().toUtf8().constData());
198  App::ExpressionPtr expr(App::Expression::parse(sheet, exprTxt));
199  if (expr->hasComponent()
200  || !expr->isDerivedFrom(App::VariableExpression::getClassTypeId())) {
201  FC_THROWM(Base::RuntimeError, "Invalid property expression: " << expr->toString());
202  }
203 
204  AutoTransaction guard("Setup conf table");
205  commandActive = true;
206 
207  // unbind any previous binding
208  int count = range.rowCount() * range.colCount();
209  for (int i = 0; i < count; ++i) {
210  auto r = range;
211  auto binding = sheet->getCellBinding(r);
212  if (!binding) {
213  break;
214  }
215  Gui::cmdAppObjectArgs(sheet,
216  "setExpression('.cells.%s.%s.%s', None)",
217  binding == PropertySheet::BindingNormal ? "Bind"
218  : "BindHiddenRef",
219  r.from().toString(),
220  r.to().toString());
221  }
222 
223  auto obj = path.getDocumentObject();
224  if (!obj) {
225  FC_THROWM(Base::RuntimeError, "Object not found");
226  }
227 
228  // Add a dynamic PropertyEnumeration for user to switch the configuration
229  std::string propName = path.getPropertyName();
230  QString groupName = ui->lineEditGroup->text().trimmed();
231  if (!prop) {
232  prop = obj->addDynamicProperty("App::PropertyEnumeration",
233  propName.c_str(),
234  groupName.toUtf8().constData());
235  }
236  else if (groupName.size()) {
237  obj->changeDynamicProperty(prop, groupName.toUtf8().constData(), nullptr);
238  }
239  prop->setStatus(App::Property::CopyOnChange, true);
240 
241  // Bind the enumeration items to the column of configuration names
242  Gui::cmdAppObjectArgs(obj,
243  "setExpression('%s.Enum', '%s.cells[<<%s>>]')",
244  propName,
245  sheet->getFullName(),
246  rangeConf);
247 
248  Gui::cmdAppObjectArgs(obj, "recompute()");
249 
250  // Bind the first cell to string value of the PropertyEnumeration. We
251  // could have just bind the entire row as below, but binding the first
252  // cell separately using a simpler expression can make it easy for us
253  // to extract the name of the PropertyEnumeration for editing or unsetup.
254  Gui::cmdAppObjectArgs(sheet,
255  "set('%s', '=hiddenref(%s.String)')",
256  from.toString(CellAddress::Cell::ShowRowColumn),
257  prop->getFullName());
258 
259  // Adjust the range to skip the first cell
260  range = Range(from.row(), from.col() + 1, to.row(), to.col());
261 
262  // Formulate expression to calculate the row binding using
263  // PropertyEnumeration
264  Gui::cmdAppObjectArgs(
265  sheet,
266  "setExpression('.cells.Bind.%s.%s', "
267  "'tuple(.cells, <<%s>> + str(hiddenref(%s)+%d), <<%s>> + str(hiddenref(%s)+%d))')",
268  range.from().toString(CellAddress::Cell::ShowRowColumn),
269  range.to().toString(CellAddress::Cell::ShowRowColumn),
270  range.from().toString(CellAddress::Cell::ShowColumn),
271  prop->getFullName(),
272  from.row() + 2,
273  range.to().toString(CellAddress::Cell::ShowColumn),
274  prop->getFullName(),
275  from.row() + 2);
276 
277  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
278  Gui::Command::commitCommand();
279  QDialog::accept();
280  }
281  catch (Base::Exception& e) {
282  e.ReportException();
283  QMessageBox::critical(this, tr("Setup configuration table"), QString::fromUtf8(e.what()));
284  if (commandActive) {
285  Gui::Command::abortCommand();
286  }
287  }
288 }
