455 Document* Application::newDocument(const char * Name, const char * UserName, bool createView, bool tempDoc)
456 {
457  auto getNameAndLabel = [this](const char * Name, const char * UserName) -> std::tuple<std::string, std::string> {
458  bool defaultName = (!Name || Name[0] == '\0');
459 
460  // get a valid name anyway!
461  if (defaultName) {
462  Name = "Unnamed";
463  }
464 
465  std::string userName;
466  if (UserName && UserName[0] != '\0') {
467  userName = UserName;
468  }
469  else {
470  userName = defaultName ? QObject::tr("Unnamed").toStdString() : Name;
471 
472  std::vector<std::string> names;
473  names.reserve(DocMap.size());
474  for (const auto& pos : DocMap) {
475  names.emplace_back(pos.second->Label.getValue());
476  }
477 
478  if (!names.empty()) {
479  userName = Base::Tools::getUniqueName(userName, names);
480  }
481  }
482 
483  return std::make_tuple(std::string(Name), userName);
484  };
485 
486  auto tuple = getNameAndLabel(Name, UserName);
487  std::string name = std::get<0>(tuple);
488  std::string userName = std::get<1>(tuple);
489  name = getUniqueDocumentName(name.c_str(), tempDoc);
490 
491  // return the temporary document if it exists
492  if (tempDoc) {
493  auto it = DocMap.find(name);
494  if (it != DocMap.end() && it->second->testStatus(Document::TempDoc))
495  return it->second;
496  }
497 
498  // create the FreeCAD document
499  std::unique_ptr<Document> newDoc(new Document(name.c_str()));
500  newDoc->setStatus(Document::TempDoc, tempDoc);
501 
502  auto oldActiveDoc = _pActiveDoc;
503  auto doc = newDoc.release(); // now owned by the Application
504 
505  // add the document to the internal list
506  DocMap[name] = doc;
507  _pActiveDoc = doc;
508

510  // clang-format off
511  // connect the signals to the application for the new document

514  _pActiveDoc->signalNewObject.connect(std::bind(&App::Application::slotNewObject, this, sp::_1));

517  _pActiveDoc->signalChangedObject.connect(std::bind(&App::Application::slotChangedObject, this, sp::_1, sp::_2));
518  _pActiveDoc->signalRelabelObject.connect(std::bind(&App::Application::slotRelabelObject, this, sp::_1));
519  _pActiveDoc->signalActivatedObject.connect(std::bind(&App::Application::slotActivatedObject, this, sp::_1));
520  _pActiveDoc->signalUndo.connect(std::bind(&App::Application::slotUndoDocument, this, sp::_1));
521  _pActiveDoc->signalRedo.connect(std::bind(&App::Application::slotRedoDocument, this, sp::_1));
522  _pActiveDoc->signalRecomputedObject.connect(std::bind(&App::Application::slotRecomputedObject, this, sp::_1));
523  _pActiveDoc->signalRecomputed.connect(std::bind(&App::Application::slotRecomputed, this, sp::_1));

526  _pActiveDoc->signalCommitTransaction.connect(std::bind(&App::Application::slotCommitTransaction, this, sp::_1));

530  _pActiveDoc->signalChangePropertyEditor.connect(std::bind(&App::Application::slotChangePropertyEditor, this, sp::_1, sp::_2));
531  // clang-format on
532  //NOLINTEND
533 
534  // make sure that the active document is set in case no GUI is up
535  {
536  Base::PyGILStateLocker lock;
537  Py::Object active(_pActiveDoc->getPyObject(), true);
538  Py::Module("FreeCAD").setAttr(std::string("ActiveDocument"), active);
539  }
540 
541  signalNewDocument(*_pActiveDoc, createView);
542

544  _pActiveDoc->Label.setValue(userName);
545 
546  // set the old document active again if the new is temporary
547  if (tempDoc && oldActiveDoc)
548  setActiveDocument(oldActiveDoc);
549 
550  return doc;
551 }

746 std::vector<Document*> Application::openDocuments(const std::vector<std::string> &filenames,
747  const std::vector<std::string> *paths,
748  const std::vector<std::string> *labels,
749  std::vector<std::string> *errs,
750  bool createView)
751 {
752  std::vector<Document*> res(filenames.size(), nullptr);
753  if (filenames.empty())
754  return res;
755 
756  if (errs)
757  errs->resize(filenames.size());
758 
759  DocOpenGuard guard(_isRestoring, signalFinishOpenDocument);
760  _pendingDocs.clear();
761  _pendingDocsReopen.clear();
762  _pendingDocMap.clear();
763  _docReloadAttempts.clear();
764 
765  signalStartOpenDocument();
766 
767  ParameterGrp::handle hGrp = GetParameterGroupByPath("User parameter:BaseApp/Preferences/Document");
768  _allowPartial = !hGrp->GetBool("NoPartialLoading",false);
769 
770  for (auto &name : filenames)
771  _pendingDocs.emplace_back(name.c_str());
772 
773  std::map<DocumentT, DocTiming> timings;
774 
775  FC_TIME_INIT(t);
776 
777  std::vector<DocumentT> openedDocs;
778 
779  int pass = 0;
780  do {
781  std::set<App::DocumentT> newDocs;
782  for (std::size_t count=0;; ++count) {
783  std::string name = std::move(_pendingDocs.front());
784  _pendingDocs.pop_front();
785  bool isMainDoc = (pass == 0 && count < filenames.size());
786 
787  try {
788  _objCount = -1;
789  std::vector<std::string> objNames;
790  if (_allowPartial) {
791  auto it = _pendingDocMap.find(name);
792  if (it != _pendingDocMap.end()) {
793  if(isMainDoc)
794  it->second.clear();
795  else
796  objNames.swap(it->second);
797  _pendingDocMap.erase(it);
798  }
799  }
800 
801  FC_TIME_INIT(t1);
802  DocTiming timing;
803 
804  const char *path = name.c_str();
805  const char *label = nullptr;
806  if (isMainDoc) {
807  if (paths && paths->size()>count)
808  path = (*paths)[count].c_str();
809 
810  if (labels && labels->size()>count)
811  label = (*labels)[count].c_str();
812  }
813 
814  auto doc = openDocumentPrivate(path, name.c_str(), label, isMainDoc, createView, std::move(objNames));
815  FC_DURATION_PLUS(timing.d1,t1);
816  if (doc) {
817  timings[doc].d1 += timing.d1;
818  newDocs.emplace(doc);
819  }
820 
821  if (isMainDoc)
822  res[count] = doc;
823  _objCount = -1;
824  }
825  catch (const Base::Exception &e) {
826  e.ReportException();
827  if (!errs && isMainDoc)
828  throw;
829  if (errs && isMainDoc)
830  (*errs)[count] = e.what();
831  else
832  Base::Console().Error("Exception opening file: %s [%s]\n", name.c_str(), e.what());
833  }
834  catch (const std::exception &e) {
835  if (!errs && isMainDoc)
836  throw;
837  if (errs && isMainDoc)
838  (*errs)[count] = e.what();
839  else
840  Base::Console().Error("Exception opening file: %s [%s]\n", name.c_str(), e.what());
841  }
842  catch (...) {
843  if (errs) {
844  if (isMainDoc)
845  (*errs)[count] = "unknown error";
846  }
847  else {
848  _pendingDocs.clear();
849  _pendingDocsReopen.clear();
850  _pendingDocMap.clear();
851  throw;
852  }
853  }
854 
855  if (_pendingDocs.empty()) {
856  if(_pendingDocsReopen.empty())
857  break;
858  _pendingDocs = std::move(_pendingDocsReopen);
859  _pendingDocsReopen.clear();
860  for(const auto &file : _pendingDocs) {
861  auto doc = getDocumentByPath(file.c_str());
862  if(doc)
863  closeDocument(doc->getName());
864  }
865  }
866  }
867 
868  ++pass;
869  _pendingDocMap.clear();
870 
871  std::vector<Document*> docs;
872  docs.reserve(newDocs.size());
873  for(const auto &d : newDocs) {
874  auto doc = d.getDocument();
875  if(!doc)
876  continue;
877  // Notify PropertyXLink to attach newly opened documents and restore
878  // relevant external links
879  PropertyXLink::restoreDocument(*doc);
880  docs.push_back(doc);
881  }
882 
883  Base::SequencerLauncher seq("Postprocessing...", docs.size());
884 
885  // After external links has been restored, we can now sort the document
886  // according to their dependency order.
887  try {
888  docs = Document::getDependentDocuments(docs, true);
889  } catch (Base::Exception &e) {
890  e.ReportException();
891  }
892  for(auto it=docs.begin(); it!=docs.end();) {
893  auto doc = *it;
894 
895  // It is possible that the newly opened document depends on an existing
896  // document, which will be included with the above call to
897  // Document::getDependentDocuments(). Make sure to exclude that.
898  if(!newDocs.count(doc)) {
899  it = docs.erase(it);
900  continue;
901  }
902 
903  auto &timing = timings[doc];
904  FC_TIME_INIT(t1);
905  // Finalize document restoring with the correct order
906  if(doc->afterRestore(true)) {
907  openedDocs.emplace_back(doc);
908  it = docs.erase(it);
909  } else {
910  ++it;
911  // Here means this is a partial loaded document, and we need to
912  // reload it fully because of touched objects. The reason of
913  // reloading a partial document with touched object is because
914  // partial document is supposed to be readonly, while a
915  // 'touched' object requires recomputation. And an object may
916  // become touched during restoring if externally linked
917  // document time stamp mismatches with the stamp saved.
918  _pendingDocs.emplace_back(doc->FileName.getValue());
919  _pendingDocMap.erase(doc->FileName.getValue());
920  }
921  FC_DURATION_PLUS(timing.d2,t1);
922  seq.next();
923  }
924  // Close the document for reloading
925  for(const auto doc : docs)
926  closeDocument(doc->getName());
927 
928  }while(!_pendingDocs.empty());
929 
930  // Set the active document using the first successfully restored main
931  // document (i.e. documents explicitly asked for by caller).
932  for (auto doc : res) {
933  if (doc) {
934  setActiveDocument(doc);
935  break;
936  }
937  }
938 
939  for (auto &doc : openedDocs) {
940  auto &timing = timings[doc];
941  FC_DURATION_LOG(timing.d1, doc.getDocumentName() << " restore");
942  FC_DURATION_LOG(timing.d2, doc.getDocumentName() << " postprocess");
943  }
944  FC_TIME_LOG(t,"total");
945  _isRestoring = false;
946 
947  signalFinishOpenDocument();
948  return res;
949 }

951 Document* Application::openDocumentPrivate(const char * FileName,
952  const char *propFileName, const char *label,
953  bool isMainDoc, bool createView,
954  std::vector<std::string> &&objNames)
955 {
956  FileInfo File(FileName);
957 
958  if (!File.exists()) {
959  std::stringstream str;
960  str << "File '" << FileName << "' does not exist!";
961  throw Base::FileSystemError(str.str().c_str());
962  }
963 
964  // Before creating a new document we check whether the document is already open
965  auto doc = getDocumentByPath(File.filePath().c_str(), PathMatchMode::MatchCanonicalWarning);
966  if(doc) {
967  if(doc->testStatus(App::Document::PartialDoc)
968  || doc->testStatus(App::Document::PartialRestore)) {
969  // Here means a document is already partially loaded, but the document
970  // is requested again, either partial or not. We must check if the
971  // document contains the required object
972 
973  if(isMainDoc) {
974  // Main document must be open fully, so close and reopen
975  closeDocument(doc->getName());
976  doc = nullptr;
977  } else if(_allowPartial) {
978  bool reopen = false;
979  for(const auto &name : objNames) {
980  auto obj = doc->getObject(name.c_str());
981  if(!obj || obj->testStatus(App::PartialObject)) {
982  reopen = true;
983  // NOTE: We are about to reload this document with
984  // extra objects. However, it is possible to repeat
985  // this process several times, if it is linked by
986  // multiple documents and each with a different set of
987  // objects. To partially solve this problem, we do not
988  // close and reopen the document immediately here, but
989  // add it to _pendingDocsReopen to delay reloading.
990  for(auto obj : doc->getObjects())
991  objNames.emplace_back(obj->getNameInDocument());
992  _pendingDocMap[doc->FileName.getValue()] = std::move(objNames);
993  break;
994  }
995  }
996  if(!reopen)
997  return nullptr;
998  }
999 
1000  if(doc) {
1001  _pendingDocsReopen.emplace_back(FileName);
1002  return nullptr;
1003  }
1004  }
1005 
1006  if(!isMainDoc)
1007  return nullptr;
1008  else if(doc)
1009  return doc;
1010  }
1011 
1012  std::string name;
1013  if(propFileName != FileName) {
1014  FileInfo fi(propFileName);
1015  name = fi.fileNamePure();
1016  }else
1017  name = File.fileNamePure();
1018 
1019  // Use the same name for the internal and user name.
1020  // The file name is UTF-8 encoded which means that the internal name will be modified
1021  // to only contain valid ASCII characters but the user name will be kept.
1022  if(!label)
1023  label = name.c_str();
1024  Document* newDoc = newDocument(name.c_str(),label,isMainDoc && createView);
1025 
1026  newDoc->FileName.setValue(propFileName==FileName?File.filePath():propFileName);
1027 
1028  try {
1029  // read the document
1030  newDoc->restore(File.filePath().c_str(),true,objNames);
1031  if(!DocFileMap.empty())
1032  DocFileMap[FileInfo(newDoc->FileName.getValue()).filePath()] = newDoc;
1033  return newDoc;
1034  }
1035  // if the project file itself is corrupt then
1036  // close the document
1037  catch (const Base::FileException&) {
1038  closeDocument(newDoc->getName());
1039  throw;
1040  }
1041  catch (const std::ios_base::failure&) {
1042  closeDocument(newDoc->getName());
1043  throw;
1044  }
1045  // but for any other exceptions leave it open to give the
1046  // user a chance to fix it
1047  catch (...) {
1048  throw;
1049  }
