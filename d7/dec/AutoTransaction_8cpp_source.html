40 AutoTransaction::AutoTransaction(const char* name, bool tmpName)
41 {
42  auto& app = GetApplication();
43  if (name && app._activeTransactionGuard >= 0) {
44  if (!app.getActiveTransaction() || (!tmpName && app._activeTransactionTmpName)) {
45  FC_LOG("auto transaction '" << name << "', " << tmpName);
46  tid = app.setActiveTransaction(name);
47  app._activeTransactionTmpName = tmpName;
48  }
49  }
50  // We use negative transaction guard to disable auto transaction from here
51  // and any stack below. This is to support user setting active transaction
52  // before having any existing AutoTransaction on stack, or 'persist'
53  // transaction that can out live AutoTransaction.
54  if (app._activeTransactionGuard < 0) {
55  --app._activeTransactionGuard;
56  }
57  else if (tid || app._activeTransactionGuard > 0) {
58  ++app._activeTransactionGuard;
59  }
60  else if (app.getActiveTransaction()) {
61  FC_LOG("auto transaction disabled because of '" << app._activeTransactionName << "'");
62  --app._activeTransactionGuard;
63  }
64  else {
65  ++app._activeTransactionGuard;
66  }
67  FC_TRACE("construct auto Transaction " << app._activeTransactionGuard);
68 }

70 AutoTransaction::~AutoTransaction()
71 {
72  auto& app = GetApplication();
73  FC_TRACE("before destruct auto Transaction " << app._activeTransactionGuard);
74  if (app._activeTransactionGuard < 0) {
75  ++app._activeTransactionGuard;
76  }
77  else if (!app._activeTransactionGuard) {
78 #ifdef FC_DEBUG
79  FC_ERR("Transaction guard error");
80 #endif
81  }
82  else if (--app._activeTransactionGuard == 0) {
83  try {
84  // We don't call close() here, because close() only closes
85  // transaction that we opened during construction time. However,
86  // when _activeTransactionGuard reaches zero here, we are supposed
87  // to close any transaction opened.
88  app.closeActiveTransaction();
89  }
90  catch (Base::Exception& e) {
91  e.ReportException();
92  }
93  catch (...) {
94  }
95  }
96  FC_TRACE("destruct auto Transaction " << app._activeTransactionGuard);
97 }

107 void AutoTransaction::setEnable(bool enable)
108 {
109  auto& app = GetApplication();
110  if (!app._activeTransactionGuard) {
111  return;
112  }
113  if ((enable && app._activeTransactionGuard > 0)
114  || (!enable && app._activeTransactionGuard < 0)) {
115  return;
116  }
117  app._activeTransactionGuard = -app._activeTransactionGuard;
118  FC_TRACE("toggle auto Transaction " << app._activeTransactionGuard);
119  if (!enable && app._activeTransactionTmpName) {
120  bool close = true;
121  for (auto& v : app.DocMap) {
122  if (v.second->hasPendingTransaction()) {
123  close = false;
124  break;
125  }
126  }
127  if (close) {
128  app.closeActiveTransaction();
129  }
130  }
131 }

133 int Application::setActiveTransaction(const char* name, bool persist)
134 {
135  if (!name || !name[0]) {
136  name = "Command";
137  }
138 
139  if (_activeTransactionGuard > 0 && getActiveTransaction()) {
140  if (_activeTransactionTmpName) {
141  FC_LOG("transaction rename to '" << name << "'");
142  for (auto& v : DocMap) {
143  v.second->renameTransaction(name, _activeTransactionID);
144  }
145  }
146  else {
147  if (persist) {
148  AutoTransaction::setEnable(false);
149  }
150  return 0;
151  }
152  }
153  else if (_TransactionLock) {
154  if (FC_LOG_INSTANCE.isEnabled(FC_LOGLEVEL_LOG)) {
155  FC_WARN("Transaction locked, ignore new transaction '" << name << "'");
156  }
157  return 0;
158  }
159  else {
160  FC_LOG("set active transaction '" << name << "'");
161  _activeTransactionID = 0;
162  for (auto& v : DocMap) {
163  v.second->_commitTransaction();
164  }
165  _activeTransactionID = Transaction::getNewID();
166  }
167  _activeTransactionTmpName = false;
168  _activeTransactionName = name;
169  if (persist) {
170  AutoTransaction::setEnable(false);
171  }
172  return _activeTransactionID;
173 }

187 void Application::closeActiveTransaction(bool abort, int id)
188 {
189  if (!id) {
190  id = _activeTransactionID;
191  }
192  if (!id) {
193  return;
194  }
195 
196  if (_activeTransactionGuard > 0 && !abort) {
197  FC_LOG("ignore close transaction");
198  return;
199  }
200 
201  if (_TransactionLock) {
202  if (_TransactionClosed >= 0) {
203  _TransactionLock = abort ? -1 : 1;
204  }
205  FC_LOG("pending " << (abort ? "abort" : "close") << " transaction");
206  return;
207  }
208 
209  FC_LOG("close transaction '" << _activeTransactionName << "' " << abort);
210  _activeTransactionID = 0;
211 
212  TransactionSignaller signaller(abort, false);
213  for (auto& v : DocMap) {
214  if (v.second->getTransactionID(true) != id) {
215  continue;
216  }
217  if (abort) {
218  v.second->_abortTransaction();
219  }
220  else {
221  v.second->_commitTransaction();
222  }
223  }
224 }

259 void TransactionLocker::activate(bool enable)
260 {
261  if (active == enable) {
262  return;
263  }
264 
265  active = enable;
266  if (active) {
267  ++_TransactionLock;
268  return;
269  }
270

272  return;
273  }
274

277  _TransactionClosed = 0;
