166 void ViewProviderDocumentObject::onChanged(const App::Property* prop)
167 {
168  if (prop == &DisplayMode) {
169  setActiveMode();
170  }
171  else if (prop == &Visibility) {
172  // use this bit to check whether show() or hide() must be called
173  if (!Visibility.testStatus(App::Property::User2)) {
174  Visibility.setStatus(App::Property::User2, true);
175  Visibility.getValue() ? show() : hide();
176  Visibility.setStatus(App::Property::User2, false);
177  }
178  if (!Visibility.testStatus(App::Property::User1)
179  && getObject()
180  && getObject()->Visibility.getValue()!=Visibility.getValue())
181  {
182  // Changing the visibility of a document object will automatically set
183  // the document modified but if the 'TouchDocument' flag is not set then
184  // this is undesired behaviour. So, if this change marks the document as
185  // modified then it must be be reversed.
186  if (!testStatus(Gui::ViewStatus::TouchDocument)) {
187  // Note: reverting document modified status like that is not
188  // appropriate because we can't tell if there is any other
189  // property being changed due to the change of Visibility here.
190  // Temporary setting the Visibility property as 'NoModify' is
191  // the proper way.
192  Base::ObjectStatusLocker<App::Property::Status,App::Property> guard(
193  App::Property::NoModify, &Visibility);
194  // bool mod = false;
195  // if (pcDocument)
196  // mod = pcDocument->isModified();
197  getObject()->Visibility.setValue(Visibility.getValue());
198  // if (pcDocument)
199  // pcDocument->setModified(mod);
200  }
201  else {
202  getObject()->Visibility.setValue(Visibility.getValue());
203  }
204  }
205  }
206  else if (prop == &SelectionStyle) {
207  if(getRoot()->isOfType(SoFCSelectionRoot::getClassTypeId())) {
208  static_cast<SoFCSelectionRoot*>(getRoot())->selectionStyle = SelectionStyle.getValue()
209  ? SoFCSelectionRoot::Box : SoFCSelectionRoot::Full;
210  }
211  }
212 
213  if (prop && !prop->testStatus(App::Property::NoModify)
214  && pcDocument
215  && !pcDocument->isModified()
216  && testStatus(Gui::ViewStatus::TouchDocument)) {
217  if (prop)
218  FC_LOG(prop->getFullName() << " changed");
219  pcDocument->setModified(true);
220  }
221 
222  ViewProvider::onChanged(prop);
223 }

307 void ViewProviderDocumentObject::updateView()
308 {
309  if(!pcObject || testStatus(ViewStatus::UpdatingView))
310  return;
311 
312  Base::ObjectStatusLocker<ViewStatus,ViewProviderDocumentObject> lock(ViewStatus::UpdatingView,this);
313 
314  // Disable object visibility syncing
315  Base::ObjectStatusLocker<App::Property::Status,App::Property> lock2(App::Property::User1, &Visibility);
316 
317  std::map<std::string, App::Property*> Map;
318  pcObject->getPropertyMap(Map);
319 
320  // Hide the object temporarily to speed up the update
321  bool vis = ViewProvider::isShow();

323  for (const auto & it : Map) {
324  updateData(it.second);
325  }
326  if (vis && Visibility.getValue()) ViewProvider::show();

454 SoNode* ViewProviderDocumentObject::findFrontRootOfType(const SoType& type) const
455 {
456  if(!pcObject)
457  return nullptr;
458  // first get the document this object is part of and get its GUI counterpart
459  App::Document* pAppDoc = pcObject->getDocument();
460  Gui::Document* pGuiDoc = Gui::Application::Instance->getDocument(pAppDoc);
461 
462  SoSearchAction searchAction;
463  searchAction.setType(type);
464  searchAction.setInterest(SoSearchAction::FIRST);
465 
466  // search in all view providers for the node type
467  std::vector<App::DocumentObject*> obj = pAppDoc->getObjects();
468  for (auto & it : obj) {
469  const ViewProvider* vp = pGuiDoc->getViewProvider(it);
470  // Ignore 'this' view provider. It could also happen that vp is 0, e.g. when
471  // several objects have been added to the App::Document before notifying the
472  // Gui::Document
473  if (!vp || vp == this)
474  continue;
475  SoSeparator* front = vp->getFrontRoot();
476  //if (front && front->getTypeId() == type)
477  // return front;
478  if (front) {
479  searchAction.apply(front);
480  SoPath* path = searchAction.getPath();
481  if (path)
482  return path->getTail();
483  }
484  }
485 
486  return nullptr;
487 }

531 int ViewProviderDocumentObject::replaceObject(
532  App::DocumentObject *oldObj, App::DocumentObject *newObj)
533 {
534  if(!oldObj || !oldObj->isAttachedToDocument()
535  || !newObj || !newObj->isAttachedToDocument())
536  {
537  FC_THROWM(Base::RuntimeError,"Invalid object");
538  }
539 
540  auto obj = getObject();
541  if(!obj || !obj->isAttachedToDocument())
542  FC_THROWM(Base::RuntimeError,"View provider not attached");
543 
544  int res = ViewProvider::replaceObject(oldObj,newObj);
545  if(res>=0)
546  return res;
547 
548  std::vector<std::pair<App::DocumentObjectT, std::unique_ptr<App::Property> > > propChanges;
549  std::vector<App::Property*> props;
550  obj->getPropertyList(props);
551  for(auto prop : props) {
552  auto linkProp = Base::freecad_dynamic_cast<App::PropertyLinkBase>(prop);
553  if(!linkProp)
554  continue;
555  std::unique_ptr<App::Property> copy(linkProp->CopyOnLinkReplace(obj, oldObj,newObj));
556  if(!copy)
557  continue;
558  propChanges.emplace_back(prop,std::move(copy));
559  }
560 
561  if(propChanges.empty())
562  return 0;
563 
564  // Global search for affected links
565  for(auto doc : App::GetApplication().getDocuments()) {
566  for(auto o : doc->getObjects()) {
567  if(o == obj)
568  continue;
569  std::vector<App::Property*> props;
570  o->getPropertyList(props);
571  for(auto prop : props) {
572  auto linkProp = Base::freecad_dynamic_cast<App::PropertyLinkBase>(prop);
573  if(!linkProp)
574  continue;
575  std::unique_ptr<App::Property> copy(linkProp->CopyOnLinkReplace(obj,oldObj,newObj));
576  if(!copy)
577  continue;
578  propChanges.emplace_back(App::DocumentObjectT(prop),std::move(copy));
579  }
580  }
581  }
582 
583  for(auto &v : propChanges) {
584  auto prop = v.first.getProperty();
585  if(prop)
586  prop->Paste(*v.second.get());
587  }
588  return 1;
589 }

595 bool ViewProviderDocumentObject::getElementPicked(const SoPickedPoint *pp, std::string &subname) const
596 {
597  if(!isSelectable())
598  return false;
599  auto vector = getExtensionsDerivedFromType<Gui::ViewProviderExtension>();
600  for(Gui::ViewProviderExtension* ext : vector)
601  if(ext->extensionGetElementPicked(pp,subname))
602  return true;
603 
604  auto childRoot = getChildRoot();
605  int idx;
606  if(!childRoot ||
607  (idx=pcModeSwitch->whichChild.getValue())<0 ||
608  pcModeSwitch->getChild(idx)!=childRoot)
609  {
610  return ViewProvider::getElementPicked(pp,subname);
611  }
612 
613  SoPath* path = pp->getPath();
614  idx = path->findNode(childRoot);
615  if(idx<0 || idx+1>=path->getLength())
616  return false;
617  auto vp = getDocument()->getViewProvider(path->getNode(idx+1));
618  if(!vp)
619  return false;
620  auto obj = vp->getObject();
621  if(!obj || !obj->isAttachedToDocument())
622  return false;
623  std::ostringstream str;
624  str << obj->getNameInDocument() << '.';
625  if(vp->getElementPicked(pp,subname))
626  str << subname;
627  subname = str.str();
628  return true;

631 bool ViewProviderDocumentObject::getDetailPath(const char *subname, SoFullPath *path, bool append, SoDetail *&det) const
632 {
633  auto len = path->getLength();
634  if(!append && len>=2)
635  len -= 2;
636  if(ViewProvider::getDetailPath(subname,path,append,det)) {
637  if(det || !subname || !*subname)
638  return true;
639  }
640 
641  if(det) {
642  delete det;
643  det = nullptr;
644  }
645 
646  const char *dot = strchr(subname,'.');
647  if(!dot)
648  return false;
649  auto obj = getObject();
650  if(!obj || !obj->isAttachedToDocument())
651  return false;
652  auto sobj = obj->getSubObject(std::string(subname,dot-subname+1).c_str());
653  if(!sobj)
654  return false;
655  auto vp = Application::Instance->getViewProvider(sobj);
656  if(!vp)
657  return false;
658 
659  auto childRoot = getChildRoot();
660  if(!childRoot)
661  path->truncate(len);
662  else {
663  auto idx = pcModeSwitch->whichChild.getValue();
664  if(idx < 0 || pcModeSwitch->getChild(idx)!=childRoot)
665  return false;
666  path->append(childRoot);
667  }
668  bool ret = false;
669  if(path->getLength()) {
670  SoNode * tail = path->getTail();
671  const SoChildList * children = tail->getChildren();
672  if(children && children->find(vp->getRoot())>=0)
673  ret = vp->getDetailPath(dot+1,path,true,det);
674  }
