70 class Edit(gui_base_original.Modifier):
71  """The Draft_Edit FreeCAD command definition.
72 
73  A tool to graphically edit FreeCAD objects.
74  Current implementation use many parts of pivy graphics code by user "looo".
75  The tool collect editpoints from objects and display Trackers on them
76  to allow editing their Shape and their parameters.
77 
78  Callbacks
79  ---------
80  selection_callback
81  registered when tool is launched, identify
82  selected objects.
83 
84  editing_callbacks
85  self._keyPressedCB -> self.keyPressed
86  self._mouseMovedCB -> self._mouseMovedCB
87  if self._mousePressedCB -> self.mousePressed
88  when trackers are displayed for selected objects,
89  these callbacks capture user events and forward
90  them to related functions
91 
92  Task panel (Draft Toolbar)
93  ----------
94  self.ui = Gui.draftToolBar
95 
96  Draft_Edit uses the taskpanel in 2 ways:
97 
98  1 - the user can select select an object and close the operation
99  self.ui.editUi()
100 
101  2 - when editing, lineUi support clicking destination point
102  by self.startEditing
103  self.ui.lineUi()
104 
105  Tracker selection
106  -----------------
107  If the tool recognize mouse click as an attempt to startEditing,
108  using soRayPickAction, it identifies the selected editTracker and
109  start editing it. Here is where "looo" code was very useful.
110 
111  Editing preview
112  ---------------
113  When object editing begins, self.ghost is initiated with the
114  corresponding DraftTracker of the object type. The object Tracker
115  is deleted when user clicks again and endEditing.
116 
117  Context Menu
118  ------------
119  Activated with Alt+LeftClick or pressing key "e"
120  It's a custom context menu, that depends on clicked tracker
121  or on clicked object.
122 
123  display_tracker_menu
124  populates the menu with custom actions
125 
126  evaluate_menu_action
127  evaluate user chosen action and launch corresponding
128  function.
129 
130  Preferences
131  -----------
132  max_objects: Int
133  set by "DraftEditMaxObjects" in user preferences
134  The max number of FreeCAD objects the tool is
135  allowed to edit at the same time.
136 
137  pick_radius: Int
138  set by "DraftEditPickRadius" in user preferences
139  The pick radius during editing operation.
140  Increase if you experience problems in clicking
141  on a editTracker because of screen resolution.
142 
143  Attributes
144  ----------
145  obj: Edited object
146  I'm planning to discard this attribute.
147  In old implementation every function was supposed to
148  act on self.obj, self.editpoints, self.trackers,
149  self.pl, self.invpl.
150  Due to multiple object editing, i'm planning to keep
151  just self.trackers. Any other object will be identified
152  and processed starting from editTracker information.
153 
154  editing: Int
155  Index of the editTracker that has been clicked by the
156  user. Tracker selection mechanism is based on it.
157  if self.editing is None :
158  the user didn't click any node, and next click will
159  be processed as an attempt to start editing operation
160  if self.editing == o or 1 or 2 or 3 etc :
161  the user is editing corresponding node, so next click
162  will be processed as an attempt to end editing operation
163 
164  trackers: Dictionary {object.Name : [editTrackers]}
165  It records the list of DraftTrackers.editTracker.
166  {object.Name as String : [editTrackers for the object]}
167  Each tracker is created with (position,obj.Name,idx),
168  so it's possible to recall it
169  self.trackers[str(node.objectName.getValue())][ep]
170 
171  overNode: DraftTrackers.editTracker
172  It represent the editTracker under the cursor position.
173  It is used to preview the tracker selection action.
174 
175  ghost: DraftTrackers.*
176  Handles the tracker to preview editing operations.
177  it is initialized when user clicks on a editTracker
178  by self.startEditing() function.
179 
180  alt_edit_mode: Int
181  Allows alternative editing modes for objects.
182  ATM supported for:
183  - arcs: if 0 edit by 3 points, if 1 edit by center,
184  radius, angles
185 
186  supportedObjs: List
187  List of supported Draft Objects.
188  The tool use utils.get_type(obj) to compare object type
189  to the list.
190 
191  supportedCppObjs: List
192  List of supported Part Objects.
193  The tool use utils.get_type(obj) and obj.TypeId to compare
194  object type to the list.
195  """

429  """

435  if self.editing is None:
436 
437  pos = event.getPosition()

439  node_idx = self.getEditNodeIndex(node)
440  if node_idx is None:

444 
445  self.startEditing(obj, node_idx)
446  else:
447  self.endEditing(self.obj, self.editing)

533  def setTrackers(self, obj, points=None):
534  """Set Edit Trackers for editpoints collected from self.obj."""
535  if utils.get_type(obj) == "BezCurve":
536  return self.resetTrackersBezier(obj)
537  if points is None or len(points) == 0:
538  _wrn = translate("draft", "No edit point found for selected object")
539  App.Console.PrintWarning(_wrn + "\n")
540  # do not finish if some trackers are still present
541  if self.trackers == {'object': []}:
542  self.finish()
543  return
544  self.trackers[obj.Name] = []
545  if obj.Name in self.trackers:
546  self.removeTrackers(obj)
547  for ep in range(len(points)):
548  self.trackers[obj.Name].append(trackers.editTracker(pos=points[ep], name=obj.Name, idx=ep))

555  def resetTrackersBezier(self, obj):
556  # in future move tracker definition to DraftTrackers
557  size = params.get_param_view("MarkerSize")
558  knotmarkers = (
559  Gui.getMarkerIndex("DIAMOND_FILLED", size), # sharp
560  Gui.getMarkerIndex("SQUARE_FILLED", size), # tangent
561  Gui.getMarkerIndex("HOURGLASS_FILLED", size) # symmetric
562  )
563  polemarker = Gui.getMarkerIndex("CIRCLE_FILLED", size) # pole
564  self.trackers[obj.Name] = []
565  cont = obj.Continuity
566  firstknotcont = cont[-1] if (obj.Closed and cont) else 0
567  pointswithmarkers = [(obj.Shape.Edges[0].Curve.
568  getPole(1),knotmarkers[firstknotcont])]
569  for edgeindex, edge in enumerate(obj.Shape.Edges):
570  poles = edge.Curve.getPoles()
571  pointswithmarkers.extend([(point,polemarker) for point in poles[1:-1]])
572  if not obj.Closed or len(obj.Shape.Edges) > edgeindex +1:
573  knotmarkeri = cont[edgeindex] if len(cont) > edgeindex else 0
574  pointswithmarkers.append((poles[-1],knotmarkers[knotmarkeri]))
575  for index, pwm in enumerate(pointswithmarkers):
576  p, marker = pwm
577  p = obj.Placement.inverse().multVec(p)
578  p = obj.getGlobalPlacement().multVec(p)
579  self.trackers[obj.Name].append(trackers.editTracker(p, obj.Name,
580  index, obj.ViewObject.LineColor, marker=marker))
