362  def adaptiveLength(self, boneAngle, toolRadius):
363  theta = self.angle()
364  distance = self.distance(toolRadius)
365  # there is something weird happening if the boneAngle came from a horizontal/vertical t-bone
366  # for some reason pi/2 is not equal to pi/2
367  if math.fabs(theta - boneAngle) < 0.00001:
368  # moving directly towards the corner
369  Path.Log.debug("adaptive - on target: %.2f - %.2f" % (distance, toolRadius))
370  return distance - toolRadius
371  Path.Log.debug(
372  "adaptive - angles: corner=%.2f bone=%.2f diff=%.12f"
373  % (theta / math.pi, boneAngle / math.pi, theta - boneAngle)
374  )
375 
376  # The bones root and end point form a triangle with the intersection of the tool path
377  # with the toolRadius circle around the bone end point.
378  # In case the math looks questionable, look for "triangle ssa"
379  # c = distance
380  # b = self.toolRadius
381  # beta = fabs(boneAngle - theta)
382  beta = math.fabs(addAngle(boneAngle, -theta))
383  D = (distance / toolRadius) * math.sin(beta)
384  if D > 1: # no intersection
385  Path.Log.debug("adaptive - no intersection - no bone")
386  return 0
387  gamma = math.asin(D)
388  alpha = math.pi - beta - gamma
389  if Path.Geom.isRoughly(0.0, math.sin(beta)):
390  # it is not a good idea to divide by 0
391  length = 0.0
392  else:
393  length = toolRadius * math.sin(alpha) / math.sin(beta)
394  if D < 1 and toolRadius < distance: # there exists a second solution
395  beta2 = beta
396  gamma2 = math.pi - gamma
397  alpha2 = math.pi - beta2 - gamma2
398  length2 = toolRadius * math.sin(alpha2) / math.sin(beta2)
399  length = min(length, length2)
400 
401  Path.Log.debug(
402  "adaptive corner=%.2f * %.2f -> bone=%.2f * %.2f"
403  % (distance, theta, length, boneAngle)
404  )
405  return length
