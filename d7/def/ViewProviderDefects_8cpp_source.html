164 void ViewProviderMeshOrientation::showDefects(const std::vector<Mesh::ElementIndex>& inds)
165 {
166  const MeshCore::MeshKernel& rMesh = getMeshKernel();
167 
168  pcCoords->point.deleteValues(0);
169  pcCoords->point.setNum(int(3 * inds.size()));
170  MeshCore::MeshFacetIterator cF(rMesh);
171  int i = 0;
172  int j = 0;
173  for (Mesh::ElementIndex ind : inds) {
174  cF.Set(ind);
175  for (auto cP : cF->_aclPoints) {
176  // move a bit in opposite normal direction to overlay the original faces
177  cP -= 0.001F * cF->GetNormal();
178  pcCoords->point.set1Value(i++, cP.x, cP.y, cP.z);
179  }
180  pcFaces->numVertices.set1Value(j++, 3);
181  }
182 
183  setDisplayMaskMode("Face");
184 }

228 void ViewProviderMeshNonManifolds::showDefects(const std::vector<Mesh::ElementIndex>& inds)
229 {
230  if ((inds.size() % 2) != 0) {
231  return;
232  }
233  const MeshCore::MeshKernel& rMesh = getMeshKernel();
234 
235  pcCoords->point.deleteValues(0);
236  pcCoords->point.setNum(int(inds.size()));
237  MeshCore::MeshPointIterator cP(rMesh);
238  int i = 0;
239  int j = 0;
240  for (auto it = inds.begin(); it != inds.end(); ++it) {
241  cP.Set(*it);
242  pcCoords->point.set1Value(i++, cP->x, cP->y, cP->z);
243  ++it; // go to end point
244  cP.Set(*it);
245  pcCoords->point.set1Value(i++, cP->x, cP->y, cP->z);
246  pcLines->numVertices.set1Value(j++, 2);
247  }
248 
249  setDisplayMaskMode("Line");
250 }

479 void ViewProviderMeshDegenerations::showDefects(const std::vector<Mesh::ElementIndex>& inds)
480 {
481  const MeshCore::MeshKernel& rMesh = getMeshKernel();
482 
483  pcCoords->point.deleteValues(0);
484  pcCoords->point.setNum(int(2 * inds.size()));
485  MeshCore::MeshFacetIterator cF(rMesh);
486  int i = 0;
487  int j = 0;
488  for (Mesh::ElementIndex ind : inds) {
489  cF.Set(ind);
490  const MeshCore::MeshPoint& rE0 = cF->_aclPoints[0];
491  const MeshCore::MeshPoint& rE1 = cF->_aclPoints[1];
492  const MeshCore::MeshPoint& rE2 = cF->_aclPoints[2];
493 
494  // check if the points are coincident
495  if (rE0 == rE1 && rE0 == rE2) {
496  // set a small tolerance to get a non-degenerated line
497  float eps = 0.005F;
498  Base::Vector3f cP1;
499  Base::Vector3f cP2;
500  cP1.Set(rE1.x + eps, rE1.y + eps, rE1.z + eps);
501  cP2.Set(rE2.x - eps, rE2.y - eps, rE2.z - eps);
502  pcCoords->point.set1Value(i++, cP1.x, cP1.y, cP1.z);
503  pcCoords->point.set1Value(i++, cP2.x, cP2.y, cP2.z);
504  }
505  else if (rE0 == rE1) {
506  pcCoords->point.set1Value(i++, rE1.x, rE1.y, rE1.z);
507  pcCoords->point.set1Value(i++, rE2.x, rE2.y, rE2.z);
508  }
509  else if (rE1 == rE2) {
510  pcCoords->point.set1Value(i++, rE2.x, rE2.y, rE2.z);
511  pcCoords->point.set1Value(i++, rE0.x, rE0.y, rE0.z);
512  }
513  else if (rE2 == rE0) {
514  pcCoords->point.set1Value(i++, rE0.x, rE0.y, rE0.z);
515  pcCoords->point.set1Value(i++, rE1.x, rE1.y, rE1.z);
516  }
517  else {
518  for (int j = 0; j < 3; j++) {
519  Base::Vector3f cVec1 = cF->_aclPoints[(j + 1) % 3] - cF->_aclPoints[j];
520  Base::Vector3f cVec2 = cF->_aclPoints[(j + 2) % 3] - cF->_aclPoints[j];
521 
522  // adjust the neighbourhoods and point indices
523  if (cVec1 * cVec2 < 0.0F) {
524  pcCoords->point.set1Value(i++,
525  cF->_aclPoints[(j + 1) % 3].x,
526  cF->_aclPoints[(j + 1) % 3].y,
527  cF->_aclPoints[(j + 1) % 3].z);
528  pcCoords->point.set1Value(i++,
529  cF->_aclPoints[(j + 2) % 3].x,
530  cF->_aclPoints[(j + 2) % 3].y,
531  cF->_aclPoints[(j + 2) % 3].z);
532  break;
533  }
534  }
535  }
536 
537  pcLines->numVertices.set1Value(j++, 2);
538  }
539 
540  setDisplayMaskMode("Line");
541 }
