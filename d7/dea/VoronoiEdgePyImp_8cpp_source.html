51 Voronoi::point_type orthognalProjection(const Voronoi::point_type& point,
52  const Voronoi::segment_type& segment)
53 {
54  // move segment so it goes through the origin (s)
55  Voronoi::point_type offset;
56  {
57  offset.x(low(segment).x());
58  offset.y(low(segment).y());
59  }
60  Voronoi::point_type s;
61  {
62  s.x(high(segment).x() - offset.x());
63  s.y(high(segment).y() - offset.y());
64  }
65  // move point accordingly so it maintains it's relation to s (p)
66  Voronoi::point_type p;
67  {
68  p.x(point.x() - offset.x());
69  p.y(point.y() - offset.y());
70  }
71  // calculate the orthogonal projection of p onto s
72  // ((p dot s) / (s dot s)) * s
73  // (https://en.wikibooks.org/wiki/Linear_Algebra/Orthogonal_Projection_Onto_a_Line) and it back
74  // by original offset to get the projected point
75  const double proj = (p.x() * s.x() + p.y() * s.y())
76  / (s.x() * s.x() + s.y() * s.y() + std::numeric_limits<double>::epsilon());
77  Voronoi::point_type pt;
78  {
79  pt.x(offset.x() + proj * s.x());
80  pt.y(offset.y() + proj * s.y());
81  }
82  return pt;
83 }

90 int sideOf(const Voronoi::point_type& p, const Voronoi::segment_type& s)
91 {
92  Voronoi::coordinate_type dxp = p.x() - low(s).x();
93  Voronoi::coordinate_type dyp = p.y() - low(s).y();
94  Voronoi::coordinate_type dxs = high(s).x() - low(s).x();
95  Voronoi::coordinate_type dys = high(s).y() - low(s).y();
96 
97  double d = -dxs * dyp + dys * dxp;
98  if (d < 0) {
99  return -1;
100  }
101  if (d > 0) {
102  return +1;
103  }
104  return 0;
105 }

170 bool retrieveDistances(const VoronoiEdge* edge, Py::List* list)
171 {
172  const Voronoi::diagram_type::cell_type* c0 = edge->ptr->cell();
173  if (c0->contains_point()) {
174  return addDistancesToPoint(edge, edge->dia->retrievePoint(c0), list, edge->dia->getScale());
175  }
176  const Voronoi::diagram_type::cell_type* c1 = edge->ptr->twin()->cell();
177  if (c1->contains_point()) {
178  return addDistancesToPoint(edge, edge->dia->retrievePoint(c1), list, edge->dia->getScale());
179  }
180  // at this point both cells are sourced from segments and it does not matter which one we use
181  Voronoi::segment_type segment = edge->dia->retrieveSegment(c0);
182  addProjectedDistanceBetween(edge->ptr->vertex0(), segment, list, edge->dia->getScale());
183  addProjectedDistanceBetween(edge->ptr->vertex1(), segment, list, edge->dia->getScale());
184  return false;
185 }

273 PyObject* VoronoiEdgePy::richCompare(PyObject* lhs, PyObject* rhs, int op)
274 {
275  PyObject* cmp = (op == Py_EQ) ? Py_False : Py_True;
276  if (PyObject_TypeCheck(lhs, &VoronoiEdgePy::Type)
277  && PyObject_TypeCheck(rhs, &VoronoiEdgePy::Type) && (op == Py_EQ || op == Py_NE)) {
278  const VoronoiEdge* vl = static_cast<VoronoiEdgePy*>(lhs)->getVoronoiEdgePtr();
279  const VoronoiEdge* vr = static_cast<VoronoiEdgePy*>(rhs)->getVoronoiEdgePtr();
280  if (vl->dia == vr->dia && vl->index == vr->index) {
281  cmp = (op == Py_EQ) ? Py_True : Py_False;
282  }
283  }
284  Py_INCREF(cmp);
285  return cmp;

466 PyObject* VoronoiEdgePy::toShape(PyObject* args)
467 {
468  double z0 = 0.0;
469  double z1 = DBL_MAX;
470  int dbg = 0;
471  if (!PyArg_ParseTuple(args, "|ddp", &z0, &z1, &dbg)) {
472  throw Py::RuntimeError("no, one or two arguments of type double accepted");
473  }
474  if (z1 == DBL_MAX) {
475  z1 = z0;
476  }
477  VoronoiEdge* e = getVoronoiEdgePtr();
478  if (e->isBound()) {
479  if (e->ptr->is_linear()) {
480  if (e->ptr->is_finite()) {
481  auto v0 = e->ptr->vertex0();
482  auto v1 = e->ptr->vertex1();
483  if (v0 && v1) {
484  return makeLineSegment(e, *v0, z0, *v1, z1);
485  }
486  }
487  else {
488  // infinite linear, need to clip somehow
489  const Voronoi::diagram_type::cell_type* c0 = e->ptr->cell();
490  const Voronoi::diagram_type::cell_type* c1 = e->ptr->twin()->cell();
491  Voronoi::point_type origin;
492  Voronoi::point_type direction;
493  if (c0->contains_point() && c1->contains_point()) {
494  Voronoi::point_type p0 = e->dia->retrievePoint(c0);
495  Voronoi::point_type p1 = e->dia->retrievePoint(c1);
496  origin.x((p0.x() + p1.x()) / 2.);
497  origin.y((p0.y() + p1.y()) / 2.);
498  direction.x(p0.y() - p1.y());
499  direction.y(p1.x() - p0.x());
500  }
501  else {
502  origin = c0->contains_segment() ? e->dia->retrievePoint(c1)
503  : e->dia->retrievePoint(c0);
504  Voronoi::segment_type segment = c0->contains_segment()
505  ? e->dia->retrieveSegment(c0)
506  : e->dia->retrieveSegment(c1);
507  Voronoi::coordinate_type dx = high(segment).x() - low(segment).x();
508  Voronoi::coordinate_type dy = high(segment).y() - low(segment).y();
509  if ((low(segment) == origin) ^ c0->contains_point()) {
510  direction.x(dy);
511  direction.y(-dx);
512  }
513  else {
514  direction.x(-dy);
515  direction.y(dx);
516  }
517  }
518  double k = 2.5; // <-- need something smarter here
519  Voronoi::point_type begin;
520  Voronoi::point_type end;
521  if (e->ptr->vertex0()) {
522  begin.x(e->ptr->vertex0()->x());
523  begin.y(e->ptr->vertex0()->y());
524  }
525  else {
526  begin.x(origin.x() - direction.x() * k);
527  begin.y(origin.y() - direction.y() * k);
528  }
529  if (e->ptr->vertex1()) {
530  end.x(e->ptr->vertex1()->x());
531  end.y(e->ptr->vertex1()->y());
532  }
533  else {
534  end.x(origin.x() + direction.x() * k);
535  end.y(origin.y() + direction.y() * k);
536  }
537  return makeLineSegment(e, begin, z0, end, z1);
538  }
539  }
540  else {
541  // parabolic curve, which is always formed by a point and an edge
542  Voronoi::point_type point = e->ptr->cell()->contains_point()
543  ? e->dia->retrievePoint(e->ptr->cell())
544  : e->dia->retrievePoint(e->ptr->twin()->cell());
545  Voronoi::segment_type segment = e->ptr->cell()->contains_point()
546  ? e->dia->retrieveSegment(e->ptr->twin()->cell())
547  : e->dia->retrieveSegment(e->ptr->cell());
548  // the location is the mid point between the normal on the segment through point
549  // this is only the mid point of the segment if the parabola is symmetric
550 
551  if (isPointOnSegment(point, segment, e->dia->getScale())) {
552  return makeLineSegment(e, low(segment), z0, high(segment), z1);
553  }
554 
555  Voronoi::point_type loc;
556  {
557  Voronoi::point_type proj = orthognalProjection(point, segment);
558  // the location is the mid point between the projection on the segment and the point
559  loc.x((proj.x() + point.x()) / 2);
560  loc.y((proj.y() + point.y()) / 2);
561  }
562  Voronoi::point_type axis;
563  {
564  axis.x(point.x() - loc.x());
565  axis.y(point.y() - loc.y());
566  }
567  Voronoi::segment_type xaxis;
568  {
569  xaxis.low(point);
570  xaxis.high(loc);
571  }
572 
573  // determine distances of the end points from the x-axis, those are the parameters for
574  // the arc of the parabola in the horizontal plane
575  auto pt0 = pointFromVertex(*e->ptr->vertex0());
576  auto pt1 = pointFromVertex(*e->ptr->vertex1());
577  Voronoi::point_type pt0x = orthognalProjection(pt0, xaxis);
578  Voronoi::point_type pt1x = orthognalProjection(pt1, xaxis);
579  double dist0 = distanceBetween(pt0, pt0x, e->dia->getScale()) * sideOf(pt0, xaxis);
580  double dist1 = distanceBetween(pt1, pt1x, e->dia->getScale()) * sideOf(pt1, xaxis);
581  if (dist1 < dist0) {
582  // if the parabola is traversed in the revere direction we need to use the points
583  // on the other side of the parabola - 'beauty of symmetric geometries
584  dist0 = -dist0;
585  dist1 = -dist1;
586  }
587 
588  // at this point we have the direction of the x-axis and the two end points p0 and p1
589  // which means we know the plane of the parabola
590  auto p0 = e->dia->scaledVector(pt0, z0);
591  auto p1 = e->dia->scaledVector(pt1, z1);
592  // we get a third point by moving p0 along the axis of the parabola
593  auto p_ = p0 + e->dia->scaledVector(axis, 0);
594 
595  // normal of the plane defined by those 3 points
596  auto norm = ((p_ - p0).Cross(p1 - p0)).Normalize();
597 
598  // the next thing to figure out is the z level of the x-axis,
599  double zx = z0 - (dist0 / (dist0 - dist1)) * (z0 - z1);
600 
601  auto locn = e->dia->scaledVector(loc, zx);
602  auto xdir = e->dia->scaledVector(axis, zx);
603 
604  double focal;
605  if (z0 == z1) {
606  // focal length if parabola in the xy-plane is simply half the distance between the
607  // point and segment - aka the distance between point and location, aka the length
608  // of axis
609  focal = length(axis) / e->dia->getScale();
610  if (dbg) {
611  std::cerr << "focal = " << length(axis) << "/" << e->dia->getScale() << "\n";
612  }
613  }
614  else {
615  // if the parabola is not in the xy-plane we need to find the
616  // (x,y) coordinates of a point on the parabola in the parabola's
617  // coordinate system.
618  // see: http://amsi.org.au/ESA_Senior_Years/SeniorTopic2/2a/2a_2content_10.html
619  // note that above website uses Y as the symmetry axis of the parabola whereas
620  // OCC uses X as the symmetry axis. The math below is in the website's system.
621  // We already know 2 points on the parabola (p0 and p1), we know their X values
622  // (dist0 and dist1) if the parabola is in the xy-plane, and we know their
623  // orthogonal projection onto the parabola's symmetry axis Y (pt0x and pt1x). The
624  // resulting Y values are the distance between the parabola's location (loc) and the
625  // respective orthogonal projection. Pythagoras gives us the X values, using the X
626  // from the xy-plane and the difference in z. Note that this calculation also gives
627  // correct results if the parabola is in the xy-plane (z0 == z1), it's just that
628  // above calculation is so much simpler.
629  double flenX0 = sqrt(dist0 * dist0 + (z0 - zx) * (z0 - zx));
630  double flenX1 = sqrt(dist1 * dist1 + (zx - z1) * (zx - z1));
631  double flenX;
632  double flenY;
633  // if one of the points is the location, we have to use the other to get sensible
634  // values
635  if (fabs(dist0) > fabs(dist1)) {
636  flenX = flenX0;
637  flenY = distanceBetween(loc, pt0x, e->dia->getScale());
638  }
639  else {
640  flenX = flenX1;
641  flenY = distanceBetween(loc, pt1x, e->dia->getScale());
642  }
643  // parabola: (x - p)^2 = 4*focal*(y - q) | (p,q) ... location of parabola
644  focal = (flenX * flenX) / (4 * fabs(flenY));
645  if (dbg) {
646  std::cerr << "segment" << segment << ", point" << point << std::endl;
647  std::cerr << " loc" << loc << ", axis" << axis << std::endl;
648  std::cerr << " dist0(" << dist0 << " : " << flenX0 << ", dist1(" << dist1
649  << " : " << flenX1 << ")" << std::endl;
650  std::cerr << " z(" << z0 << ", " << zx << ", " << z1 << ")" << std::endl;
651  std::cerr << " focal = (" << flenX << " * " << flenX << ") / (4 * fabs("
652  << flenY << "))\n";
653  }
654  // use new X values to set the parameters
655  dist0 = dist0 >= 0 ? flenX0 : -flenX0;
656  dist1 = dist1 >= 0 ? flenX1 : -flenX1;
657  }
658 
659  gp_Pnt pbLocn(locn.x, locn.y, locn.z);
660  gp_Dir pbNorm(norm.x, norm.y, norm.z);
661  gp_Dir pbXdir(xdir.x, xdir.y, 0);
662 
663  gp_Ax2 pb(pbLocn, pbNorm, pbXdir);
664  Handle(Geom_Parabola) parabola = new Geom_Parabola(pb, focal);
665 
666  Part::GeomArcOfParabola arc;
667  arc.setHandle(parabola);
668  arc.setRange(dist0, dist1, false);
669 
670  // get a shape for the parabola arc
671  Handle(Geom_Curve) h = Handle(Geom_Curve)::DownCast(arc.handle());
672  BRepBuilderAPI_MakeEdge mkBuilder(h, h->FirstParameter(), h->LastParameter());
673  return new Part::TopoShapeEdgePy(new Part::TopoShape(mkBuilder.Shape()));
674  }
675  }
676  Py_INCREF(Py_None);
677  return Py_None;
678 }
