47 def write_meshdata_constraint(f, femobj, prs_obj, ccxwriter):
48 
49  # floats read from ccx should use {:.13G}, see comment in writer module
50 
51  rev = -1 if prs_obj.Reversed else 1
52  # the pressure has to be output in MPa
53  pressure_quantity = FreeCAD.Units.Quantity(prs_obj.Pressure.getValueAs("MPa"))
54  press_rev = rev * pressure_quantity
55 
56  f.write("*DLOAD\n")
57  for ref_shape in femobj["PressureFaces"]:
58  # the loop is needed for compatibility reason
59  # in deprecated method get_pressure_obj_faces_depreciated
60  # the face ids where per ref_shape
61  f.write("** " + ref_shape[0] + "\n")
62  for face, fno in ref_shape[1]:
63  if fno > 0: # solid mesh face
64  f.write(f"{face},P{fno},{press_rev}\n")
65  # on shell mesh face: fno == 0
66  # normal of element face == face normal
67  elif fno == 0:
68  f.write(f"{face},P,{press_rev}\n")
69  # on shell mesh face: fno == -1
70  # normal of element face opposite direction face normal
71  elif fno == -1:
72  f.write(f"{face},P,{-1 * press_rev}\n")
