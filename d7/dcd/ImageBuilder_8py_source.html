163  def build_image(self, obj, image_name):
164  """
165  Generates a headless picture of a 3D object and saves it as a PNG and optionally a PostScript file.
166 
167  Args:
168  - obj: The 3D object to generate an image for.
169  - image_name: Base name for the output image file without extension.
170 
171  Returns:
172  - A boolean indicating the success of the operation.
173  """
174  # Ensure the 'Part' and 'coin' modules are available, along with necessary attributes/methods.
175  if not hasattr(obj, "Shape") or not hasattr(obj.Shape, "writeInventor"):
176  Path.Log.debug("Object does not have the required attributes.")
177  return False
178 
179  try:
180  # Generate Inventor data from the object's shape
181  iv = obj.Shape.writeInventor()
182 
183  # Prepare Inventor data for rendering
184  inp = coin.SoInput()
185  inp.setBuffer(iv)
186  data = coin.SoDB.readAll(inp)
187 
188  if data is None:
189  Path.Log.debug("Failed to read Inventor data.")
190  return False
191 
192  # Setup the scene
193  base = coin.SoBaseColor()
194  base.rgb.setValue(0.6, 0.7, 1.0)
195  data.insertChild(base, 0)
196 
197  root = coin.SoSeparator()
198  light = coin.SoDirectionalLight()
199  cam = coin.SoOrthographicCamera()
200  root.addChild(cam)
201  root.addChild(light)
202  root.addChild(data)
203 
204  # Camera and rendering setup
205  axo = coin.SbRotation(-0.353553, -0.146447, -0.353553, -0.853553)
206  viewport = coin.SbViewportRegion(400, 400)
207  cam.orientation.setValue(axo)
208  cam.viewAll(root, viewport)
209  off = coin.SoOffscreenRenderer(viewport)
210  root.ref()
211  ret = off.render(root)
212  root.unref()
213 
214  # Saving the rendered image
215  if off.isWriteSupported("PNG"):
216  file_path = f"{self.file_path}{os.path.sep}{imageName}.png"
217  off.writeToFile(file_path, "PNG")
218  else:
219  Path.Log.debug("PNG format is not supported.")
220  # return False
221 
222  # Optionally save as PostScript if supported
223  file_path = f"{self.file_path}{os.path.sep}{imageName}.ps"
224  off.writeToPostScript(ps_file_path)
225 
226  return file_path
227 
228  except Exception as e:
229  print(f"An error occurred: {e}")
230  return False
