131 BlendPoint FeatureBlendCurve::GetBlendPoint(App::PropertyLinkSub& link,
132  App::PropertyFloatConstraint& param,
133  App::PropertyIntegerConstraint& continuity)
134 {
135  auto linked = link.getValue();
136 
137  TopoDS_Shape axEdge;
138  if (link.getSubValues().size() > 0 && link.getSubValues()[0].length() > 0) {
139  axEdge =
140  Feature::getTopoShape(linked, link.getSubValues()[0].c_str(), true /*need element*/)
141  .getShape();
142  }
143  else {
144  axEdge = Feature::getShape(linked);
145  }
146 
147  if (axEdge.IsNull()) {
148  throw Base::ValueError("DirLink shape is null");
149  }
150  if (axEdge.ShapeType() != TopAbs_EDGE) {
151  throw Base::TypeError("DirLink shape is not an edge");
152  }
153  const TopoDS_Edge& e = TopoDS::Edge(axEdge);
154  BRepAdaptor_Curve adapt(e);
155  double fp = adapt.FirstParameter();
156  double lp = adapt.LastParameter();
157 
158  double RealPar = RelativeToRealParameters(param.getValue(), fp, lp);
159 
160  std::vector<Base::Vector3d> constraints;
161  gp_Pnt Pt;
162 
163  adapt.D0(RealPar, Pt);
164  Base::Vector3d bv(Pt.X(), Pt.Y(), Pt.Z());
165  constraints.emplace_back(bv);
166 
167  for (int i = 1; i <= continuity.getValue(); i++) {
168  gp_Vec v1 = adapt.DN(RealPar, i);
169  Base::Vector3d bbv1(v1.X(), v1.Y(), v1.Z());
170  constraints.emplace_back(bbv1);
171  }
172 
173  BlendPoint bp(constraints);
174 
175  return bp;
176 }

178 App::DocumentObjectExecReturn* FeatureBlendCurve::execute()
179 {
180  BlendPoint bp1 = GetBlendPoint(StartEdge, StartParameter, StartContinuity);
181  BlendPoint bp2 = GetBlendPoint(EndEdge, EndParameter, EndContinuity);
182 
183  std::vector<BlendPoint> blendPointsList;
184 
185  blendPointsList.emplace_back(bp1);
186  blendPointsList.emplace_back(bp2);
187 
188  BlendCurve curve(blendPointsList);
189  curve.setSize(0, StartSize.getValue(), true);
190  curve.setSize(1, EndSize.getValue(), true);
191 
192  Handle(Geom_BezierCurve) bc(curve.compute());
193  BRepBuilderAPI_MakeEdge mkEdge(bc);
194 
195  Shape.setValue(mkEdge.Edge());
196 
197  return StdReturn;
198 }

206 void FeatureBlendCurve::onChanged(const App::Property* prop)

215  long maxValue = maxDegree - 2 - StartContinuity.getValue();

217  EndContinuity.setValue(maxValue);
218  }
219  }
220  if (prop == &StartContinuity || prop == &StartParameter || prop == &StartSize
221  || prop == &EndContinuity || prop == &EndParameter || prop == &EndSize) {
222  if (!isRestoring()) {

224  delete ret;
225  }
226  }
227  Part::Spline::onChanged(prop);
