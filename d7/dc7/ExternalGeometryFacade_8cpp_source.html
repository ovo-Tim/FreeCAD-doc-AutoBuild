121 void ExternalGeometryFacade::initExtensions() const
122 {
123  if (!Geo->hasExtension(SketchGeometryExtension::getClassTypeId())) {
124  THROWM(Base::ValueError,
125  "ExternalGeometryFacade for const::Geometry without SketchGeometryExtension");
126  }
127 
128  if (!Geo->hasExtension(ExternalGeometryExtension::getClassTypeId())) {
129  THROWM(Base::ValueError,
130  "ExternalGeometryFacade for const::Geometry without ExternalGeometryExtension");
131  }
132 
133  auto ext = std::static_pointer_cast<const SketchGeometryExtension>(
134  Geo->getExtension(SketchGeometryExtension::getClassTypeId()).lock());
135 
136  const_cast<ExternalGeometryFacade*>(this)->SketchGeoExtension = ext;
137 
138  auto extext = std::static_pointer_cast<const ExternalGeometryExtension>(
139  Geo->getExtension(ExternalGeometryExtension::getClassTypeId()).lock());
140 
141  const_cast<ExternalGeometryFacade*>(this)->ExternalGeoExtension = extext;
142 }
