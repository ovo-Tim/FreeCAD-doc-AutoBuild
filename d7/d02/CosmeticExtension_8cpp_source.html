66 void CosmeticExtension::deleteCosmeticElements(std::vector<std::string> removables)
67 {
68  // Base::Console().Message("CEx::deleteCosmeticElements(%d removables)\n", removables.size());
69  for (auto& name : removables) {
70  if (DU::getGeomTypeFromName(name) == "Vertex" &&
71  DU::isCosmeticVertex(getOwner(), name)) {
72  CosmeticVertex* vert = getCosmeticVertexBySelection(name);
73  removeCosmeticVertex(vert->getTagAsString());
74  continue;
75  }
76  if (DU::getGeomTypeFromName(name) == "Edge" &&
77  ( DU::isCosmeticEdge(getOwner(), name) ||
78  DU::isCenterLine(getOwner(), name) ) ) {
79  CosmeticEdge* edge = getCosmeticEdgeBySelection(name);
80  if (edge) {
81  // if not edge, something has gone very wrong!
82  removeCosmeticEdge(edge->getTagAsString());
83  continue;
84  }
85  CenterLine* line = getCenterLineBySelection(name);
86  if (line) {
87  removeCenterLine(line->getTagAsString());
88  continue;
89  }
90  }
91  }
92 }
