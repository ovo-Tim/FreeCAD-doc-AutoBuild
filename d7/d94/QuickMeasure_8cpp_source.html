78 void QuickMeasure::processSelection()
79 {
80  if (pendingProcessing) {
81  pendingProcessing = false;
82  try {
83  tryMeasureSelection();
84  }
85  catch (const Base::IndexError&) {
86  // ignore this exception because it can be caused by trying to access a non-existing
87  // sub-element e.g. when selecting a construction geometry in sketcher
88  }
89  catch (const Base::ValueError&) {
90  // ignore this exception because it can be caused by trying to access a non-existing
91  // sub-element e.g. when selecting a constraint in sketcher
92  }
93  catch (const Base::Exception& e) {
94  e.ReportException();
95  }
96  catch (const Standard_Failure& e) {
97  FC_ERR(e);
98  }
99  catch (...) {
100  FC_ERR("Unhandled unknown exception");
101  }
102  }
103 }

116 bool QuickMeasure::shouldMeasure(const Gui::SelectionChanges& msg) const
117 {
118 
119  // measure only IF
120  Gui::Document* doc = Gui::Application::Instance->activeDocument();
121  if (doc) {
122  // we have a document
123  if (msg.Type == Gui::SelectionChanges::AddSelection
124  || msg.Type == Gui::SelectionChanges::RmvSelection
125  || msg.Type == Gui::SelectionChanges::SetSelection
126  || msg.Type == Gui::SelectionChanges::ClrSelection) {
127  // the event is about a change in selected objects
128  return true;
129  }
130  }
131  return false;
132 }

144 void QuickMeasure::addSelectionToMeasurement()
145 {
146  int count = 0;
147  int limit = 100;
148 
149  auto selObjs = Gui::Selection().getSelectionEx(nullptr,
150  App::DocumentObject::getClassTypeId(),
151  Gui::ResolveMode::NoResolve);
152 
153  for (auto& selObj : selObjs) {
154  App::DocumentObject* rootObj = selObj.getObject();
155  const std::vector<std::string> subNames = selObj.getSubNames();
156 
157  // Check that there's not too many selection
158  count += subNames.empty() ? 1 : subNames.size();
159  if (count > limit) {
160  measurement->clear();
161  return;
162  }
163 
164  if (subNames.empty()) {
165  if (isObjAcceptable(rootObj)) {
166  measurement->addReference3D(rootObj, "");
167  }
168  continue;
169  }
170 
171  for (auto& subName : subNames) {
172  App::DocumentObject* obj = rootObj->getSubObject(subName.c_str());
173 
174  if (!isObjAcceptable(obj)) {
175  continue;
176  }
177  measurement->addReference3D(rootObj, subName);
178  }
179  }
180 }

194 void QuickMeasure::printResult()
195 {
196  MeasureType mtype = measurement->getType();
197  if (mtype == MeasureType::Surfaces) {
198  print(tr("Total area: %1").arg(areaStr(measurement->area())));
199  }
200  /* deactivated because computing the volumes/area of solids makes a significant
201  slow down in selection of complex solids.
202  else if (mtype == MeasureType::Volumes) {
203  Base::Quantity area(measurement->area(), Base::Unit::Area);
204  Base::Quantity vol(measurement->volume(), Base::Unit::Volume);
205  print(tr("Volume: %1, Area:
206  %2").arg(vol.getSafeUserString()).arg(area.getSafeUserString()));
207  }*/
208  else if (mtype == MeasureType::TwoPlanes) {
209  print(tr("Nominal distance: %1").arg(lenghtStr(measurement->planePlaneDistance())));
210  }
211  else if (mtype == MeasureType::Cone || mtype == MeasureType::Plane) {
212  print(tr("Area: %1").arg(areaStr(measurement->area())));
213  }
214  else if (mtype == MeasureType::Cylinder || mtype == MeasureType::Sphere
215  || mtype == MeasureType::Torus) {
216  print(tr("Area: %1, Radius: %2")
217  .arg(areaStr(measurement->area()), lenghtStr(measurement->radius())));
218  }
219  else if (mtype == MeasureType::Edges) {
220  print(tr("Total length: %1").arg(lenghtStr(measurement->length())));
221  }
222  else if (mtype == MeasureType::TwoParallelLines) {
223  print(tr("Nominal distance: %1").arg(lenghtStr(measurement->lineLineDistance())));
224  }
225  else if (mtype == MeasureType::TwoLines) {
226  print(tr("Angle: %1, Total length: %2")
227  .arg(lenghtStr(measurement->angle()), lenghtStr(measurement->length())));
228  }
229  else if (mtype == MeasureType::Line) {
230  print(tr("Length: %1").arg(lenghtStr(measurement->length())));
231  }
232  else if (mtype == MeasureType::Circle) {
233  print(tr("Radius: %1").arg(lenghtStr(measurement->radius())));
234  }
235  else if (mtype == MeasureType::PointToPoint) {
236  print(tr("Distance: %1").arg(lenghtStr(measurement->length())));
237  }
238  else if (mtype == MeasureType::PointToEdge || mtype == MeasureType::PointToSurface) {
239  print(tr("Minimum distance: %1").arg(lenghtStr(measurement->length())));
240  }
241  else {
242  print(QString::fromLatin1(""));
243  }
244 }
