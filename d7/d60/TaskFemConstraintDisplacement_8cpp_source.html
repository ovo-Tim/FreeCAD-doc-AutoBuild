235 void TaskFemConstraintDisplacement::addToSelection()
236 {
237  std::vector<Gui::SelectionObject> selection =
238  Gui::Selection().getSelectionEx(); // gets vector of selected objects of active document
239  if (selection.empty()) {
240  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
241  return;
242  }
243  Fem::ConstraintDisplacement* pcConstraint =
244  ConstraintView->getObject<Fem::ConstraintDisplacement>();
245  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
246  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
247 
248  for (auto& it : selection) { // for every selected object
249  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
250  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
251  return;
252  }
253  const std::vector<std::string>& subNames = it.getSubNames();
254  App::DocumentObject* obj = it.getObject();
255  for (const auto& subName : subNames) { // for every selected sub element
256  bool addMe = true;
257  for (std::vector<std::string>::iterator itr =
258  std::find(SubElements.begin(), SubElements.end(), subName);
259  itr != SubElements.end();
260  itr = std::find(++itr,
261  SubElements.end(),
262  subName)) { // for every sub element in selection that
263  // matches one in old list
264  if (obj
265  == Objects[std::distance(
266  SubElements.begin(),
267  itr)]) { // if selected sub element's object equals the one in old list
268  // then it was added before so don't add
269  addMe = false;
270  }
271  }
272  // limit constraint such that only vertexes or faces or edges can be used depending on
273  // what was selected first
274  std::string searchStr;
275  if (subName.find("Vertex") != std::string::npos) {
276  searchStr = "Vertex";
277  }
278  else if (subName.find("Edge") != std::string::npos) {
279  searchStr = "Edge";
280  }
281  else {
282  searchStr = "Face";
283  }
284  for (const auto& SubElement : SubElements) {
285  if (SubElement.find(searchStr) == std::string::npos) {
286  QString msg = tr("Only one type of selection (vertex, face or edge) per "
287  "analysis feature allowed!");
288  QMessageBox::warning(this, tr("Selection error"), msg);
289  addMe = false;
290  break;
291  }
292  }
293  if (addMe) {
294  QSignalBlocker block(ui->lw_references);
295  Objects.push_back(obj);
296  SubElements.push_back(subName);
297  ui->lw_references->addItem(makeRefText(obj, subName));
298  }
299  }
300  }
301  // Update UI
302  pcConstraint->References.setValues(Objects, SubElements);
303  updateUI();
304 }

306 void TaskFemConstraintDisplacement::removeFromSelection()
307 {
308  std::vector<Gui::SelectionObject> selection =
309  Gui::Selection().getSelectionEx(); // gets vector of selected objects of active document
310  if (selection.empty()) {
311  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
312  return;
313  }
314  Fem::ConstraintDisplacement* pcConstraint =
315  ConstraintView->getObject<Fem::ConstraintDisplacement>();
316  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
317  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
318  std::vector<size_t> itemsToDel;
319  for (const auto& it : selection) { // for every selected object
320  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
321  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
322  return;
323  }
324  const std::vector<std::string>& subNames = it.getSubNames();
325  const App::DocumentObject* obj = it.getObject();
326 
327  for (const auto& subName : subNames) { // for every selected sub element
328  for (std::vector<std::string>::iterator itr =
329  std::find(SubElements.begin(), SubElements.end(), subName);
330  itr != SubElements.end();
331  itr = std::find(++itr,
332  SubElements.end(),
333  subName)) { // for every sub element in selection that
334  // matches one in old list
335  if (obj
336  == Objects[std::distance(
337  SubElements.begin(),
338  itr)]) { // if selected sub element's object equals the one in old list
339  // then it was added before so mark for deletion
340  itemsToDel.push_back(std::distance(SubElements.begin(), itr));
341  }
342  }
343  }
344  }
345  std::sort(itemsToDel.begin(), itemsToDel.end());
346  while (!itemsToDel.empty()) {
347  Objects.erase(Objects.begin() + itemsToDel.back());
348  SubElements.erase(SubElements.begin() + itemsToDel.back());
349  itemsToDel.pop_back();
350  }
351  // Update UI
352  {
353  QSignalBlocker block(ui->lw_references);
354  ui->lw_references->clear();
355  for (unsigned int j = 0; j < Objects.size(); j++) {
356  ui->lw_references->addItem(makeRefText(Objects[j], SubElements[j]));
357  }
358  }
359  pcConstraint->References.setValues(Objects, SubElements);
360  updateUI();
361 }
