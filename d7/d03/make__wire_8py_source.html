40 def make_wire(pointslist, closed=False, placement=None, face=None, support=None, bs2wire=False):
41  """make_wire(pointslist, [closed], [placement])
42 
43  Creates a Wire object from the given list of vectors. If face is
44  true (and wire is closed), the wire will appear filled. Instead of
45  a pointslist, you can also pass a Part Wire.
46 
47  Parameters
48  ----------
49  pointslist : [Base.Vector]
50  List of points to create the polyline
51 
52  closed : bool
53  If closed is True or first and last points are identical,
54  the created polyline will be closed.
55 
56  placement : Base.Placement
57  If a placement is given, it is used.
58 
59  face : Bool
60  If face is False, the rectangle is shown as a wireframe,
61  otherwise as a face.
62 
63  support :
64  TODO: Describe
65 
66  bs2wire : bool
67  TODO: Describe
68  """
69  if not App.ActiveDocument:
70  App.Console.PrintError("No active document. Aborting\n")
71  return None
72 
73  import Part
74 
75  if isinstance(pointslist, (list,tuple)):
76  for pnt in pointslist:
77  if not isinstance(pnt, App.Vector):
78  App.Console.PrintError(
79  "Items must be Base.Vector objects, not {}\n".format(
80  type(pnt)))
81  return None
82 
83  elif isinstance(pointslist, Part.Wire):
84  for edge in pointslist.Edges:
85  if not DraftGeomUtils.is_straight_line(edge):
86  App.Console.PrintError("All edges must be straight lines\n")
87  return None
88  closed = pointslist.isClosed()
89  pointslist = [v.Point for v in pointslist.OrderedVertexes]
90 
91  else:
92  App.Console.PrintError("Can't make Draft Wire from {}\n".format(
93  type(pointslist)))
94  return None
95 
96 
97  if len(pointslist) == 0:
98  App.Console.PrintWarning("Draft Wire created with empty point list\n")
99 
100  if placement:
101  utils.type_check([(placement, App.Placement)], "make_wire")
102  ipl = placement.inverse()
103  if not bs2wire:
104  pointslist = [ipl.multVec(p) for p in pointslist]
105 
106  if len(pointslist) == 2:
107  fname = "Line"
108  else:

110
