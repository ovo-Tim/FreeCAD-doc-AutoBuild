83  // clang-format off
84  // This ExpressionCompleter model works without any physical items.
85  // Everything item related is stored inside QModelIndex.InternalPointer/InternalId(),
86  // using the following Info structure.
87  //
88  // The Info contains two indices, one for document and the other for object.
89  // For 32-bit system, the index is 16bit which limits the size to 64K. For
90  // 64-bit system, the index is 32bit.
91  //
92  // The "virtual" items are organized as a tree. The root items are special,
93  // which consists of three types in the following order,
94  //
95  // * Document, even index contains item using document's name, while
96  // odd index with quoted document label.
97  // * Objects of the current document, even index with object's internal
98  // name, and odd index with quoted object label.
99  // * Properties of the current object.
100  //
101  // Document item contains object item as child, and object item contains
102  // property item.
103  //
104  // The QModelIndex of a root item has both the doc field and obj field set
105  // to -1, and uses the row as the item index. We can figure out the type of
106  // the item solely based on this row index.
107  //
108  // QModelIndex of a non-root object item has doc field as the document
109  // index, and obj field set to -1.
110  //
111  // QModelIndex of a non-root property item has doc field as the document
112  // index, and obj field as the object index.
113  //
114  // An item is uniquely identified by the pair (row, father_link) in the QModelIndex
115  //
116  // The completion tree structure created takes into account the current document and object
117  //
118  // It is done as such, in order to have contextual completion (prop -> object -> files):
119  // * root (-1,-1)
120  // |
121  // |----- documents
122  // |----- current documents' objects [externally set]
123  // |----- current objects' props [externally set]
124  //
125  // This complicates the decoding schema for the root, where the childcount will be
126  // doc.size() + current_doc.Objects.size() + current_obj.Props.size().
127  //
128  // This is reflected in the complexity of the DATA function.
129  //
130  // Example encoding of a QMODEL Index
131  //
132  // ROOT (row -1, [-1,-1,-1,0]), info represented as [-1,-1,-1,0]
133  // |-- doc 1 (non contextual) - (row 0, [-1,-1,-1,0]) = encode as parent => [0,-1,-1,0]
134  // |-- doc 2 (non contextual) - (row 1, [-1,-1,-1,0]) = encode as parent => [1,-1,-1,0]
135  // | |- doc 2.obj1 - (row 0, [1,-1,-1,0]) = encode as parent => [1, 0,-1,0]
136  // | |- doc 2.obj2 - (row 1, [1,-1,-1,0]) = encode as parent => [1, 1,-1,0]
137  // | |- doc 2.obj3 - (row 2, [1,-1,-1,0]) = encode as parent => [1, 2,-1,0]
138  // | |- doc 2.obj3.prop1 - (row 0, [1, 2,-1,0]) = encode as parent => [1, 2, 0,0]
139  // | |- doc 2.obj3.prop2 - (row 1, [1, 2,-1,0]) = encode as parent => [1, 2, 1,0]
140  // | |- doc 2.obj3.prop3 - (row 2, [1, 2,-1,0]) = encode as parent => [1, 2, 2,0]
141  // | |- doc 2.obj3.prop3.path0 - (row 0, [1, 2, 2,0]) = encode as parent => INVALID, LEAF ITEM
142  // | |- doc 2.obj3.prop3.path1 - (row 1, [1, 2, 2,0]) = encode as parent => INVALID, LEAF ITEM
143  // |
144  // |
145  // |-- doc 3 (non contextual) - (row 2, [-1,-1,-1,0]) = encode as parent => [2,-1,-1,0]
146  // |
147  // |-- obj1 (current doc - contextual) - (row 3, [-1,-1,-1,0]) = encode as parent => [3,-1,-1,1]
148  // |-- obj2 (current doc - contextual) - (row 4, [-1,-1,-1,0]) = encode as parent => [4,-1,-1,1]
149  // | |- obj2.prop1 (contextual) - (row 0, [4,-1,-1,1]) = encode as parent => [4,-1,0,1]
150  // | |- obj2.prop2 (contextual) - (row 1, [4,-1,-1,1]) = encode as parent => [4,-1,1,1]
151  // | | - obj2.prop2.path1 (contextual) - (row 0, [4,-1,0 ,1]) = encode as parent => INVALID, LEAF ITEM
152  // | | - obj2.prop2.path2 (contextual) - (row 1, [4,-1,1 ,1]) = encode as parent => INVALID, LEAF ITEM
153  // |
154  // |-- prop1 (current obj - contextual) - (row 5, [-1,-1,-1,0]) = encode as parent => [5,-1,-1,1]
155  // |-- prop2 (current obj - contextual) - (row 6, [-1,-1,-1,0]) = encode as parent => [6,-1,-1,1]
156  // |-- prop2.path1 (contextual) - (row 0, [ 6,-1,-1,0]) = encode as parent => INVALID, LEAF ITEM
157  // |-- prop2.path2 (contextual) - (row 1, [ 6,-1,-1,0]) = encode as parent => INVALID, LEAF ITEM
158  //
159  // clang-format on

171  static const quint64 k_numBitsProp = 16ULL; // 0 .. 15
172  static const quint64 k_numBitsObj = 24ULL; // 16.. 39
173  static const quint64 k_numBitsContextualHierarchy = 1; // 40
174  static const quint64 k_numBitsDocuments = 23ULL; // 41.. 63
175 
176  static const quint64 k_offsetProp = 0;
177  static const quint64 k_offsetObj = k_offsetProp + k_numBitsProp;
178  static const quint64 k_offsetContextualHierarchy = k_offsetObj + k_numBitsObj;
179  static const quint64 k_offsetDocuments =
180  k_offsetContextualHierarchy + k_numBitsContextualHierarchy;
181 
182  static const quint64 k_maskProp = ((1ULL << k_numBitsProp) - 1);
183  static const quint64 k_maskObj = ((1ULL << k_numBitsObj) - 1);
184  static const quint64 k_maskContextualHierarchy = ((1ULL << k_numBitsContextualHierarchy) - 1);
185  static const quint64 k_maskDocuments = ((1ULL << k_numBitsDocuments) - 1);

187  union InfoPtrEncoding
188  {
189  quint64 d_enc;
190  struct
191  {
192  quint8 doc;
193  quint8 prop;
194  quint16 obj : 15;
195  quint16 contextualHierarchy : 1;
196  } d32;
197  void* ptr;
198 
199  InfoPtrEncoding(const Info& info)
200  : d_enc(0)
201  {
202  if (sizeof(void*) < sizeof(InfoPtrEncoding)) {
203  d32.doc = (quint8)(info.doc + 1);
204  d32.obj = (quint16)(info.obj + 1);
205  d32.prop = (quint8)(info.prop + 1);
206  d32.contextualHierarchy = info.contextualHierarchy;
207  }
208  else {
209  d_enc = ((quint64(info.doc + 1) & k_maskDocuments) << k_offsetDocuments)
210  | ((quint64(info.contextualHierarchy) & k_maskContextualHierarchy)
211  << k_offsetContextualHierarchy)
212  | ((quint64(info.obj + 1) & k_maskObj) << k_offsetObj)
213  | ((quint64(info.prop + 1) & k_maskProp) << k_offsetProp);
214  }
215  }
216  InfoPtrEncoding(void* pointer)
217  : d_enc(0)
218  {
219  this->ptr = pointer;
220  }
221 
222  Info DecodeInfo()
223  {
224  Info info;
225  if (sizeof(void*) < sizeof(InfoPtrEncoding)) {
226  info.doc = qint32(d32.doc) - 1;
227  info.obj = qint32(d32.obj) - 1;
228  info.prop = qint32(d32.prop) - 1;
229  info.contextualHierarchy = d32.contextualHierarchy;
230  }
231  else {
232  info.doc = ((d_enc >> k_offsetDocuments) & k_maskDocuments) - 1;
233  info.contextualHierarchy =
234  ((d_enc >> k_offsetContextualHierarchy) & k_maskContextualHierarchy);
235  info.obj = ((d_enc >> k_offsetObj) & k_maskObj) - 1;
236  info.prop = ((d_enc >> k_offsetProp) & k_maskProp) - 1;
237  }
238  return info;
239  }
240  };

283  // The completion tree structure created takes into account the current document and object
284  //
285  // It is done as such:
286  // * root (-1,-1)
287  // |
288  // |----- documents
289  // |----- current documents' objects [externally set]
290  // |----- current objects' props [externally set]
291  //
292  // This complicates the decoding schema for the root, where the childcount will be
293  // doc.size() + current_doc.Objects.size() + current_obj.Props.size().
294  //
295  // this function is called in two modes:
296  // - obtain the count of a node identified by Info,row => count != nullptr, v==nullptr
297  // - get the text of an item. This text will contain separators but NO full path
298  void _data(const Info& info, int row, QVariant* v, int* count, bool sep = false) const
299  {
300  int idx;
301  // identify the document index. For any children of the root, it is given by traversing
302  // the flat list and identified by [row]
303  idx = info.doc < 0 ? row : info.doc;
304  const auto& docs = App::GetApplication().getDocuments();
305  int docSize = (int)docs.size() * 2;
306  int objSize = 0;
307  int propSize = 0;
308  std::vector<std::pair<const char*, App::Property*>> props;
309  App::Document* doc = nullptr;
310  App::DocumentObject* obj = nullptr;
311  const char* propName = nullptr;
312  App::Property* prop = nullptr;
313  // check if the document is uniquely identified: either the correct index in info.doc
314  // OR if, the node is a descendant of the root, its row lands within 0...docsize
315  if (idx >= 0 && idx < docSize) {
316  doc = docs[idx / 2];
317  }
318  else {
319  // if we're looking at the ROOT, or the row identifies one of the other ROOT elements
320  // |----- current documents' objects, rows: docs.size ... docs.size +
321  // objs.size
322  // |----- current objects' props, rows: docs.size + objs.size ... docs.size +
323  // objs.size+ props.size
324  //
325  // We need to process the ROOT so we get the correct count for its children
326  doc = App::GetApplication().getDocument(currentDoc.c_str());
327  if (!doc) { // no current, there are no additional objects
328  return;
329  }
330 
331  // move to the current documents' objects' range
332  idx -= docSize;
333  if (info.doc < 0) {
334  row = idx;
335  }
336 
337  const auto& objs = doc->getObjects();
338  objSize = (int)objs.size() * 2;
339  // if this is a valid object, we found our object and break.
340  // if not, this may be the root or one of current object's properties
341  if (idx >= 0 && idx < objSize) {
342  obj = objs[idx / 2];
343  // if they are in the ignore list skip
344  if (inList.count(obj)) {
345  return;
346  }
347  }
348  else if (!noProperty) {
349  // need to check the current object's props range, or we're parsing the ROOT
350  auto cobj = doc->getObject(currentObj.c_str());
351  if (cobj) {
352  // move to the props range of the current object
353  idx -= objSize;
354  if (info.doc < 0) {
355  row = idx;
356  }
357  // get the properties
358  cobj->getPropertyNamedList(props);
359  propSize = (int)props.size();
360 
361  // if this is an invalid index, bail out
362  // if it's the ROOT break!
363  if (idx >= propSize) {
364  return;
365  }
366  if (idx >= 0) {
367  obj = cobj; // we only set the active object if we're not processing the
368  // root.
369  propName = props[idx].first;
370  prop = props[idx].second;
371  }
372  }
373  }
374  }
375  // the item is the ROOT or a CHILD of the root
376  if (info.doc < 0) {
377  // and we're asking for a count, compute it
378  if (count) {
379  // note that if we're dealing with a valid DOC node (row>0, ROOT_info)
380  // objSize and propSize will be zero because of the early exit above
381  *count = docSize + objSize + propSize;
382  }
383  if (idx >= 0 && v) {
384  // we're asking for this child's data, and IT's NOT THE ROOT
385  QString res;
386  // we resolved the property
387  if (propName) {
388  res = QString::fromLatin1(propName);
389  // resolve the property
390  if (sep && !noProperty && !retrieveSubPaths(prop).empty()) {
391  res += QLatin1Char('.');
392  }
393  }
394  else if (obj) {
395  // the object has been resolved, use the saved idx to figure out quotation or
396  // not.
397  if (idx & 1) {
398  res = QString::fromUtf8(quote(obj->Label.getStrValue()).c_str());
399  }
400  else {
401  res = QString::fromLatin1(obj->getNameInDocument());
402  }
403  if (sep && !noProperty) {
404  res += QLatin1Char('.');
405  }
406  }
407  else {
408  // the document has been resolved, use the saved idx to figure out quotation or
409  // not.
410  if (idx & 1) {
411  res = QString::fromUtf8(quote(doc->Label.getStrValue()).c_str());
412  }
413  else {
414  res = QString::fromLatin1(doc->getName());
415  }
416  if (sep) {
417  res += QLatin1Char('#');
418  }
419  }
420  v->setValue(res);
421  }
422  // done processing the ROOT or any child items
423  return;
424  }
425 
426  // object not resolved
427  if (!obj) {
428  // are we pointing to an object item, or our father (info) is an object
429  idx = info.obj < 0 ? row : info.obj;
430  const auto& objs = doc->getObjects();
431  objSize = (int)objs.size() * 2;
432  // if invalid index, or in the ignore list bail out
433  if (idx < 0 || idx >= objSize || inList.count(obj)) {
434  return;
435  }
436  obj = objs[idx / 2];
437 
438  if (info.obj < 0) {
439  // if this is AN actual Object item and not a root
440  if (count) {
441  *count = objSize; // set the correct count if requested
442  }
443  if (v) {
444  // resolve the name
445  QString res;
446  if (idx & 1) {
447  res = QString::fromUtf8(quote(obj->Label.getStrValue()).c_str());
448  }
449  else {
450  res = QString::fromLatin1(obj->getNameInDocument());
451  }
452  if (sep && !noProperty) {
453  res += QLatin1Char('.');
454  }
455  v->setValue(res);
456  }
457  return;
458  }
459  }
460 
461  if (noProperty) {
462  return;
463  }
464  if (!propName) {
465  idx = info.prop < 0 ? row : info.prop;
466  obj->getPropertyNamedList(props);
467  propSize = (int)props.size();
468  // return if the property is invalid
469  if (idx < 0 || idx >= propSize) {
470  return;
471  }
472  propName = props[idx].first;
473  prop = props[idx].second;
474  // if this is a root object item
475  if (info.prop < 0) {
476  // set the property size count
477  if (count) {
478  *count = propSize;
479  }
480  if (v) {
481  QString res = QString::fromLatin1(propName);
482 
483  // check to see if we have accessible paths from this prop name?
484  if (sep && !retrieveSubPaths(prop).empty()) {
485  res += QLatin1Char('.');
486  }
487  *v = res;
488  }
489  return;
490  }
491  }
492 
493  // resolve paths
494  if (prop) {
495  // idx identifies the path
496  idx = row;
497  std::vector<App::ObjectIdentifier> paths = retrieveSubPaths(prop);
498 
499  if (count) {
500  *count = paths.size();
501  }
502 
503  // check to see if this is a valid path
504  if (idx < 0 || idx >= static_cast<int>(paths.size())) {
505  return;
506  }
507 
508  if (v) {
509  auto str = paths[idx].getSubPathStr();
510  if (str.size() && (str[0] == '.' || str[0] == '#')) {
511  // skip the "."
512  *v = QString::fromLatin1(str.c_str() + 1);
513  }
514  else {
515  *v = QString::fromLatin1(str.c_str());
516  }
517  }
518  }
519  return;
520  }

522  QModelIndex parent(const QModelIndex& index) const override
523  {
524  if (!index.isValid()) {
525  return {};
526  }
527 
528  Info parentInfo = getInfo(index);
529  Info grandParentInfo = parentInfo;
530 
531  if (parentInfo.contextualHierarchy) {
532  // for contextual hierarchy we have this:
533  // ROOT -> Object in Current Doc -> Prop In Object -> PropPath
534  // ROOT -> prop in Current Object -> prop Path
535 
536  if (parentInfo.prop >= 0) {
537  grandParentInfo.prop = -1;
538  return createIndex(parentInfo.prop, 0, infoId(grandParentInfo));
539  }
540  // if the parent is the object or a prop attached to the root, we just need the below
541  // line
542  return createIndex(parentInfo.doc, 0, infoId(Info::root));
543  }
544  else {
545  if (parentInfo.prop >= 0) {
546  grandParentInfo.prop = -1;
547  return createIndex(parentInfo.prop, 0, infoId(grandParentInfo));
548  }
549  if (parentInfo.obj >= 0) {
550  grandParentInfo.obj = -1;
551  return createIndex(parentInfo.obj, 0, infoId(grandParentInfo));
552  }
553  if (parentInfo.doc >= 0) {
554  grandParentInfo.doc = -1;
555  return createIndex(parentInfo.doc, 0, infoId(grandParentInfo));
556  }
557  }

560  return {};

563  // returns true if successful, false if 'element' identifies a Leaf
564  bool modelIndexToParentInfo(QModelIndex element, Info& info) const
565  {
566  Info parentInfo;
567  info = Info::root;
568 
569  if (element.isValid()) {
570  parentInfo = getInfo(element);
571  info = parentInfo;
572 
573  // Our wonderful element is a child of the root
574  if (parentInfo.doc < 0) {
575  // need special casing to properly identify this model's object
576  const auto& docs = App::GetApplication().getDocuments();
577  auto docsSize = static_cast<int>(docs.size() * 2);
578 
579  info.doc = element.row();
580 
581  // if my element is a contextual descendant of root (current doc object list,
582  // current object prop list) mark it as such
583  if (element.row() >= docsSize) {
584  info.contextualHierarchy = 1;
585  }
586  }
587  else if (parentInfo.contextualHierarchy) {
588  const auto& docs = App::GetApplication().getDocuments();
589  auto cdoc = App::GetApplication().getDocument(currentDoc.c_str());
590 
591  if (cdoc) {
592  int objsSize = static_cast<int>(cdoc->getObjects().size() * 2);
593  int idx = parentInfo.doc - static_cast<int>(docs.size());
594  if (idx < objsSize) {
595  // |-- Parent (OBJECT) - (row 4, [-1,-1,-1,0]) = encode as element =>
596  // [parent.row,-1,-1,1]
597  // |- element (PROP) - (row 0, [parent.row,-1,-1,1]) = encode as
598  // element => [parent.row,-1,parent.row,1]
599 
600  info.doc = parentInfo.doc;
601  info.obj =
602  -1; // object information is determined by the DOC index actually
603  info.prop = element.row();
604  info.contextualHierarchy = 1;
605  }
606  else {
607  // if my parent (parentInfo) is a prop, it means that our element is a prop
608  // path and that is a leaf item (we don't split prop paths further) we can't
609  // encode leaf items into an "Info"
610  return false;
611  }
612  }
613  else {
614  // no contextual document
615  return false;
616  }
617  }
618  // regular hierarchy
619  else if (parentInfo.obj <= 0) {
620  info.obj = element.row();
621  }
622  else if (parentInfo.prop <= 0) {
623  info.prop = element.row();
624  }
625  else {
626  return false;
627  }
628  }
629  return true;
630  }

632  QModelIndex index(int row, int column, const QModelIndex& parent = QModelIndex()) const override
633  {
634  if (row < 0) {
635  return {};
636  }
637  Info myParentInfoEncoded = Info::root;
638

762 QStringList ExpressionCompleter::splitPath(const QString& input) const
763 {
764  QStringList resultList;
765  std::string path = input.toUtf8().constData();
766  if (path.empty()) {
767  return resultList;
768  }
769 
770  int retry = 0;
771  std::string lastElem; // used to recover in case of parse failure after ".".
772  std::string trim; // used to delete ._self added for another recovery path
773  while (true) {
774  try {
775  // this will not work for incomplete Tokens at the end
776  // "Sketch." will fail to parse and complete.
777 
778  App::ObjectIdentifier ident = ObjectIdentifier::parse(currentObj.getObject(), path);
779 
780  std::vector<std::string> stringList = ident.getStringList();
781  auto stringListIter = stringList.begin();
782  if (retry > 1 && !stringList.empty()) {
783  stringList.pop_back();
784  }
785 
786  if (!stringList.empty()) {
787  if (!trim.empty() && boost::ends_with(stringList.back(), trim)) {
788  stringList.back().resize(stringList.back().size() - trim.size());
789  }
790  while (stringListIter != stringList.end()) {
791  resultList << QString::fromStdString(*stringListIter);
792  ++stringListIter;
793  }
794  }
795  if (lastElem.size()) {
796  // if we finish in a trailing separator
797  if (!lastElem.empty()) {
798  // erase the separator
799  lastElem.erase(lastElem.begin());
800  resultList << QString::fromStdString(lastElem);
801  }
802  else {
803  // add empty string to allow completion after "." or "#"
804  resultList << QString();
805  }
806  }
807  FC_TRACE("split path " << path << " -> "
808  << resultList.join(QLatin1String("/")).toUtf8().constData());
809  return resultList;
810  }
811  catch (const Base::Exception& except) {
812  FC_TRACE("split path " << path << " error: " << except.what());
813  if (retry == 0) {
814  size_t lastElemStart = path.rfind('.');
815 
816  if (lastElemStart == std::string::npos) {
817  lastElemStart = path.rfind('#');
818  }
819  if (lastElemStart != std::string::npos) {
820  lastElem = path.substr(lastElemStart);
821  path = path.substr(0, lastElemStart);
822  }
823  retry++;
824  continue;
825  }
826  else if (retry == 1) {
827  // restore path from retry 0
828  if (lastElem.size() > 1) {
829  path = path + lastElem;
830  lastElem = "";
831  }
832  // else... we don't reset lastElem if it's a '.' or '#' to allow chaining
833  // completions
834  if (!path.empty()) {
835  char last = path[path.size() - 1];
836  if (last != '#' && last != '.' && path.find('#') != std::string::npos) {
837  path += "._self";
838  ++retry;
839  continue;
840  }
841  }
842  }
843  else if (retry == 2) {
844  if (path.size() >= 6) {
845  path.resize(path.size() - 6);
846  }
847  if (!path.empty()) {
848  char last = path[path.size() - 1];
849  if (last != '.' && last != '<' && path.find("#<<") != std::string::npos) {
850  path += ">>._self";
851  ++retry;
852  trim = ">>";
853  continue;
854  }
855  }
856  }
857  return QStringList() << input;
858  }
859  }
860 }
