226  struct ApproxPar
227  {
228  double weight1;
229  double weight2;
230  double weight3;
231  double tol3d;
232  int maxDegree;
233  GeomAbs_Shape cont;
234 
235  ApproxPar()
236  {
237  weight1 = 0.2;
238  weight2 = 0.4;
239  weight3 = 0.2;
240  tol3d = 1.0e-2;
241  maxDegree = 5;
242  cont = GeomAbs_C2;
243  }
244  };
245  Private()
246  : curve(new ViewProviderCurveOnMesh)
247  , editcursor(QPixmap(cursor_curveonmesh), 7, 7)
248  {}
249  ~Private()
250  {
251  delete curve;

254  static void vertexCallback(void* ud, SoEventCallback* n);
255  std::vector<SbVec3f> convert(const std::vector<Base::Vector3f>& points) const
256  {
257  std::vector<SbVec3f> pts;
258  pts.reserve(points.size());
259  for (const auto& it : points) {
260  pts.push_back(Base::convertTo<SbVec3f>(it));
261  }
262  return pts;
263  }
264  void createGrid()
265  {
266  Mesh::Feature* mf = mesh->getObject<Mesh::Feature>();
267  const Mesh::MeshObject& meshObject = mf->Mesh.getValue();
268  kernel = meshObject.getKernel();
269  kernel.Transform(meshObject.getTransform());
270 
271  MeshCore::MeshAlgorithm alg(kernel);
272  float fAvgLen = alg.GetAverageEdgeLength();
273  grid = new MeshCore::MeshFacetGrid(kernel, 5.0f * fAvgLen);
274  }
275  bool projectLineOnMesh(const PickedPoint& pick)
276  {
277  PickedPoint last = pickedPoints.back();
278  std::vector<Base::Vector3f> polyline;
279 
280  MeshCore::MeshProjection meshProjection(kernel);
281  Base::Vector3f v1 = Base::convertTo<Base::Vector3f>(last.point);
282  Base::Vector3f v2 = Base::convertTo<Base::Vector3f>(pick.point);
283  Base::Vector3f vd =
284  Base::convertTo<Base::Vector3f>(viewer->getViewer()->getViewDirection());
285  if (meshProjection.projectLineOnMesh(*grid, v1, last.facet, v2, pick.facet, vd, polyline)) {
286  if (polyline.size() > 1) {
287  if (cutLines.empty()) {
288  cutLines.push_back(polyline);
289  }
290  else {
291  SbVec3f dir1(0.0f, 0.0f, 0.0f);
292  SbVec3f dir2 = pick.point - last.point;
293  dir2.normalize();
294  std::size_t num = pickedPoints.size();
295  if (num >= 2) {
296  dir1 = pickedPoints[num - 1].point - pickedPoints[num - 2].point;
297  dir1.normalize();
298  }
299 
300  // if the angle between two line segments is greater than the angle
301  // split the curve in this position
302  if (dir1.dot(dir2) < cosAngle) {
303  cutLines.push_back(polyline);
304  }
305  else {
306  std::vector<Base::Vector3f>& segm = cutLines.back();
307  segm.insert(segm.end(), polyline.begin() + 1, polyline.end());
308  }
309  }
310 
311  return true;
312  }
313  }
314 
315  return false;
316  }

371 void CurveOnMeshHandler::onCreate()
372 {
373  for (auto it = d_ptr->cutLines.begin(); it != d_ptr->cutLines.end(); ++it) {
374  std::vector<SbVec3f> segm = d_ptr->convert(*it);
375  if (d_ptr->approximate) {
376  Handle(Geom_BSplineCurve) spline = approximateSpline(segm);
377  if (!spline.IsNull()) {
378  displaySpline(spline);
379  }
380  }
381  else {
382  TopoDS_Wire wire;
383  if (makePolyline(segm, wire)) {
384  displayPolyline(wire);
385  }
386  }
387  }
388 
389  d_ptr->curve->clearVertex();
390  d_ptr->curve->clearPoints();
391 
392  d_ptr->pickedPoints.clear();
393  d_ptr->cutLines.clear();
394  d_ptr->wireClosed = false;
395 
396  disableCallback();
397 }

476 Handle(Geom_BSplineCurve) CurveOnMeshHandler::approximateSpline(const std::vector<SbVec3f>& points)
477 {
478  TColgp_Array1OfPnt pnts(1, points.size());
479  Standard_Integer index = 1;
480  for (const auto& it : points) {
481  float x, y, z;
482  it.getValue(x, y, z);
483  pnts(index++) = gp_Pnt(x, y, z);
484  }
485 
486  try {
487  // GeomAPI_PointsToBSpline fit(pnts, 1, 2, GeomAbs_C0, 1.0e-3);
488  // GeomAPI_PointsToBSpline fit(pnts, d_ptr->par.weight1, d_ptr->par.weight2,
489  // d_ptr->par.weight3,
490  // d_ptr->par.maxDegree, d_ptr->par.cont, d_ptr->par.tol3d);
491  GeomAPI_PointsToBSpline fit(pnts,
492  1,
493  d_ptr->par.maxDegree,
494  d_ptr->par.cont,
495  d_ptr->par.tol3d);
496  Handle(Geom_BSplineCurve) spline = fit.Curve();
497  return spline;
498  }
499  catch (...) {
500  return Handle(Geom_BSplineCurve)();
501  }
502 }

504 void CurveOnMeshHandler::approximateEdge(const TopoDS_Edge& edge, double tolerance)
505 {
506  BRepMesh_IncrementalMesh(edge, tolerance);
507  TopLoc_Location loc;
508  Handle(Poly_Polygon3D) aPoly = BRep_Tool::Polygon3D(edge, loc);
509  if (!aPoly.IsNull()) {
510  int numNodes = aPoly->NbNodes();
511  const TColgp_Array1OfPnt& aNodes = aPoly->Nodes();
512  std::vector<SbVec3f> pts;
513  pts.reserve(numNodes);
514  for (int i = aNodes.Lower(); i <= aNodes.Upper(); i++) {
515  const gp_Pnt& p = aNodes.Value(i);
516  pts.emplace_back(static_cast<float>(p.X()),
517  static_cast<float>(p.Y()),
518  static_cast<float>(p.Z()));
519  }
520 
521  d_ptr->curve->setPoints(pts);
522  }
523 }

572 bool CurveOnMeshHandler::tryCloseWire(const SbVec3f& p) const
573 {
574  if (d_ptr->pickedPoints.size() >= 3) {
575  Private::PickedPoint first = d_ptr->pickedPoints.front();
576  // if the distance of the first and last points is small enough (~1mm)
577  // the curve can be closed.
578  float len = (first.point - p).length();
579  if (len < d_ptr->distance) {
580  return true;
581  }
582  }
583 
584  return false;
585 }

596 void CurveOnMeshHandler::Private::vertexCallback(void* ud, SoEventCallback* cb)
597 {
598  Gui::View3DInventorViewer* view = static_cast<Gui::View3DInventorViewer*>(cb->getUserData());
599  const SoEvent* ev = cb->getEvent();
600  if (ev->getTypeId() == SoMouseButtonEvent::getClassTypeId()) {
601  // set as handled
602  cb->setHandled();
603 
604  const SoMouseButtonEvent* mbe = static_cast<const SoMouseButtonEvent*>(ev);
605  if (mbe->getButton() == SoMouseButtonEvent::BUTTON1
606  && mbe->getState() == SoButtonEvent::DOWN) {
607  const SoPickedPoint* pp = cb->getPickedPoint();
608  if (pp) {
609  CurveOnMeshHandler* self = static_cast<CurveOnMeshHandler*>(ud);
610  if (!self->d_ptr->wireClosed) {
611  Gui::ViewProvider* vp = view->getViewProviderByPathFromTail(pp->getPath());
612  if (vp && vp->isDerivedFrom<MeshGui::ViewProviderMesh>()) {
613  MeshGui::ViewProviderMesh* mesh =
614  static_cast<MeshGui::ViewProviderMesh*>(vp);
615  const SoDetail* detail = pp->getDetail();
616  if (detail && detail->getTypeId() == SoFaceDetail::getClassTypeId()) {
617  // get the mesh and build a grid
618  if (!self->d_ptr->mesh) {
619  self->d_ptr->mesh = mesh;
620  self->d_ptr->createGrid();
621  }
622  else if (self->d_ptr->mesh != mesh) {
623  Gui::getMainWindow()->statusBar()->showMessage(
624  tr("Wrong mesh picked"));
625  return;
626  }
627 
628  const SbVec3f& p = pp->getPoint();
629  const SbVec3f& n = pp->getNormal();
630 
631  Private::PickedPoint pick;
632  pick.facet = static_cast<const SoFaceDetail*>(detail)->getFaceIndex();
633  pick.point = p;
634  pick.normal = n;
635 
636  if (self->d_ptr->pickedPoints.empty()) {
637  self->d_ptr->pickedPoints.push_back(pick);
638  self->d_ptr->curve->addVertex(p);
639  }
640  else {
641  // check to auto-complete the curve
642  if (self->tryCloseWire(p)) {
643  self->closeWire();
644  }
645  else if (self->d_ptr->projectLineOnMesh(pick)) {
646  self->d_ptr->curve->setPoints(self->getPoints());
647  self->d_ptr->pickedPoints.push_back(pick);
648  self->d_ptr->curve->addVertex(p);
649  }
650  }
651  }
652  }
653  // try to 'complete' the curve
654  else if (vp && vp->isDerivedFrom<ViewProviderCurveOnMesh>()) {
655  const SbVec3f& p = pp->getPoint();
656  if (self->tryCloseWire(p)) {
657  self->closeWire();
658  }
659  }
660  }
661  }
662  else {
663  Gui::getMainWindow()->statusBar()->showMessage(tr("No point was picked"));
