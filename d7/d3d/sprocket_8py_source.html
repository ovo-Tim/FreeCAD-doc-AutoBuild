21 def CreateSprocket(w, P, N, Dr):
22  """
23  Create a sprocket
24 
25  w is the wirebuilder object (in which the sprocket will be constructed)
26  P is the chain pitch
27  N is the number of teeth
28  Dr is the roller diameter
29 
30  Remaining variables can be found in Standard Handbook of Chains
31  """
32  Ds = 1.005 * Dr + (0.003 * 25.4)
33  R = Ds / 2
34  M = 0.8 * Dr * cos(radians(35) + radians(60/N))
35  T = 0.8 * Dr * sin(radians(35) + radians(60/N))
36  E = 1.3025 * Dr + (0.0015 * 25.4)
37  W = 1.4 * Dr * cos(radians(180/N))
38  V = 1.4 * Dr * sin(radians(180/N))
39  F = Dr * (0.8 * cos(radians(18) - radians(56)/N) + 1.4 *
40  cos(radians(17) - radians(64) / N) - 1.3025) - (0.0015 * 25.4)
41  PD = P / (sin(radians(180)/N))
42  # H = sqrt(F**2 - (1.4 * Dr - P/2)**2)
43  # OD = P * (0.6 + 1/tan(radians(180/N)))
44 
45  # The sprocket tooth gullet consists of four segments
46  x0 = 0
47  y0 = PD/2 - R
48 
49  # ---- Segment 1 -----
50  alpha = 35 + 60/N
51  x1 = -R * cos(radians(alpha))
52  y1 = PD/2 - R * sin(radians(alpha))
53 
54  # ---- Segment 2 -----
55  alpha = 35 + 60/N
56  beta = 18 - 56 / N
57  x2 = M - E * cos(radians(alpha-beta))
58  y2 = T - E * sin(radians(alpha-beta)) + PD/2
59 
60  # # ---- Segment 3 -----
61  y2o = y2 - PD/2
62  hyp = sqrt((-W-x2)**2 + (-V-y2o)**2)
63  AP = sqrt(hyp**2 - F**2)
64  gamma = atan((y2o + V)/(x2 + W))
65  alpha = asin(AP / hyp)
66  beta = 180 - (90 - degrees(alpha)) - (90 - degrees(gamma))
67  x3o = AP * sin(radians(beta))
68  y3o = AP * cos(radians(beta))
69  x3 = x2 - x3o
70  y3 = y2 + y3o
71 
72  # ---- Segment 4 -----
73  alpha = 180/N
74  m = -1/tan(radians(alpha))
75  yf = PD/2 - V
76  A = 1 + m**2
77  B = 2*m*yf - 2*W
78  C = W**2 + yf**2 - F**2
79  # x4a = (-B - sqrt(B**2 - 4 * A * C)) / (2*A)
80  x4b = (-B + sqrt(B**2 - 4 * A * C)) / (2*A)
81  x4 = -x4b
82  y4 = m * x4
83 
84  p0 = [x0,y0]
85  p1 = [x1,y1]
86  p2 = [x2,y2]
87  p3 = [x3,y3]
88  p4 = [x4,y4]
89  p5 = [-x1,y1]
90  p6 = [-x2,y2]
91  p7 = [-x3,y3]
92  p8 = [-x4,y4]
93 
94  w.move(p4) # vectors are lists [x,y]
95  w.arc(p3, F, 0)
96  w.line(p2)
97  w.arc(p1, E, 1)
98  w.arc(p0, R, 1)
99 
100  # ---- Mirror -----
101  w.arc(p5, R, 1)
102  w.arc(p6, E, 1)
103  w.line(p7)
104  w.arc(p8, F, 0)
105 
106  # ---- Polar Array ----
107  alpha = -radians(360/N)
108  for n in range(1,N):
109  # falling gullet slope
110  w.arc(rotate(p3, alpha*n), F, 0)
111  w.line(rotate(p2, alpha*n))
112  w.arc(rotate(p1, alpha*n), E, 1)
113  w.arc(rotate(p0, alpha*n), R, 1)
114 
115  # rising gullet slope
116  w.arc(rotate(p5, alpha*n), R, 1)
117  w.line(rotate(p6, alpha*n))
118  w.arc(rotate(p7, alpha*n), E, 0)
119  w.arc(rotate(p8, alpha*n), F, 0)
120 
121  w.close()
122 
123  return w

126 def rotate(pt, rads):
127  """
128  rotate pt by rads radians about origin
129  """
130  sinA = sin(rads)
131  cosA = cos(rads)
132  return (pt[0] * cosA - pt[1] * sinA,
133  pt[0] * sinA + pt[1] * cosA)
