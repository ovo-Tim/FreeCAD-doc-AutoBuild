91 const std::string TaskFemConstraint::getReferences(const std::vector<std::string>& items) const

111 void TaskFemConstraint::setSelection(QListWidgetItem* item)
112 {
113  // highlights the list item in the model
114 
115  // get the document name
116  std::string docName = ConstraintView->getObject()->getDocument()->getName();
117  // name of the item
118  std::string ItemName = item->text().toStdString();
119  std::string delimiter = ":";
120  size_t pos = 0;
121  pos = ItemName.find(delimiter);
122  // the objName is the name piece before the ':' of the item name
123  std::string objName = ItemName.substr(0, pos);
124  // the subName is the name piece behind the ':'
125  ItemName.erase(0, pos + delimiter.length());
126  // clear existing selection
127  Gui::Selection().clearSelection();
128  // highlight the selected item
129  Gui::Selection().addSelection(docName.c_str(), objName.c_str(), ItemName.c_str(), 0, 0, 0);
130 }

200 bool TaskDlgFemConstraint::accept()
201 {
202  std::string name = ConstraintView->getObject()->getNameInDocument();
203 
204  try {
205  std::string refs = parameter->getReferences();
206 
207  if (!refs.empty()) {
208  Gui::Command::doCommand(Gui::Command::Doc,
209  "App.ActiveDocument.%s.References = [%s]",
210  name.c_str(),
211  refs.c_str());
212  }
213  else {
214  QMessageBox::warning(parameter,
215  tr("Input error"),
216  tr("You must specify at least one reference"));
217  return false;
218  }
219 
220  std::string scale = parameter->getScale();
221  Gui::Command::doCommand(Gui::Command::Doc,
222  "App.ActiveDocument.%s.Scale = %s",
223  name.c_str(),
224  scale.c_str());
225  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
226  if (!ConstraintView->getObject()->isValid()) {
227  throw Base::RuntimeError(ConstraintView->getObject()->getStatusString());
228  }
229  Gui::Command::doCommand(Gui::Command::Gui, "Gui.activeDocument().resetEdit()");
230  Gui::Command::commitCommand();
231  }
232  catch (const Base::Exception& e) {
233  QMessageBox::warning(parameter, tr("Input error"), QString::fromLatin1(e.what()));
234  return false;
235  }
236 
237  return true;
