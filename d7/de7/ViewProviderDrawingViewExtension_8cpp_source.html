61 //let the page have any drops we receive.
62 bool ViewProviderDrawingViewExtension::extensionCanDropObject(App::DocumentObject* obj) const
63 {
64  // it can happen that if the tree gets badly corrupted, there can be loose
65  // objects that have no page or view provider, so we need to check that
66  // all these objects exist.
67  auto vpdv = getViewProviderDrawingView();
68  if (!vpdv) {
69  return false;
70  }
71  auto vpp = vpdv->getViewProviderPage();
72  if (!vpp) {
73  return false;
74  }
75  auto vppEx = vpp->getVPPExtension();
76  if (!vppEx) {
77  return false;
78  }
79  return getViewProviderDrawingView()
80  ->getViewProviderPage()
81  ->getVPPExtension()
82  ->extensionCanDropObject(obj);
83 }
