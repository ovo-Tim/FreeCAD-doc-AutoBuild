199 struct PropertyCleaner
200 {
201  explicit PropertyCleaner(Property* p)
202  : prop(p)
203  {
204  ++_PropCleanerCounter;
205  }
206  ~PropertyCleaner()
207  {
208  if (--_PropCleanerCounter) {
209  return;
210  }
211  bool found = false;
212  while (!_RemovedProps.empty()) {
213  auto p = _RemovedProps.back();
214  _RemovedProps.pop_back();
215  if (p != prop) {
216  p->setContainer(nullptr);
217  delete p;
218  }
219  else {
220  found = true;
221  }
222  }
223 
224  if (found) {
225  _RemovedProps.push_back(prop);
226  }
227  }
228  static void add(Property* prop)
229  {
230  _RemovedProps.push_back(prop);
231  }
232 
233  Property* prop;
234 
235  static std::vector<Property*> _RemovedProps;
236  static int _PropCleanerCounter;
237 };

308 void Property::setStatusValue(unsigned long status)
309 {
310  // clang-format off
311  static const unsigned long mask =
312  (1<<PropDynamic)
313  |(1<<PropNoRecompute)
314  |(1<<PropReadOnly)
315  |(1<<PropTransient)
316  |(1<<PropOutput)
317  |(1<<PropHidden)
318  |(1<<PropNoPersist)
319  |(1<<Busy);
320  // clang-format on
321 
322  status &= ~mask;
323  status |= StatusBits.to_ulong() & mask;
324  unsigned long oldStatus = StatusBits.to_ulong();
325  StatusBits = decltype(StatusBits)(status);
326 
327  if (father) {
328  static unsigned long _signalMask = (1 << ReadOnly) | (1 << Hidden);
329  if ((status & _signalMask) != (oldStatus & _signalMask)) {
330  father->onPropertyStatusChanged(*this, oldStatus);
331  }
332  }
333 }

362 void PropertyListsBase::_setPyObject(PyObject* value)
363 {
364  std::vector<int> indices;
365  std::vector<PyObject*> vals;
366  Py::Object pySeq;
367 
368  if (PyDict_Check(value)) {
369  Py::Dict dict(value);
370  auto size = dict.size();
371  vals.reserve(size);
372  indices.reserve(size);
373  int listSize = getSize();
374  for (auto it = dict.begin(); it != dict.end(); ++it) {
375  const auto& item = *it;
376  PyObject* key = item.first.ptr();
377  if (!PyLong_Check(key)) {
378  throw Base::TypeError("expect key type to be integer");
379  }
380  long idx = PyLong_AsLong(key);
381  if (idx < -1 || idx > listSize) {
382  throw Base::ValueError("index out of bound");
383  }
384  if (idx == -1 || idx == listSize) {
385  idx = listSize;
386  ++listSize;
387  }
388  indices.push_back(idx);
389  vals.push_back(item.second.ptr());
390  }
391  }
392  else {
393  if (PySequence_Check(value)) {
394  pySeq = value;
395  }
396  else {
397  PyObject* iter = PyObject_GetIter(value);
398  if (iter) {
399  Py::Object pyIter(iter, true);
400  pySeq = Py::asObject(PySequence_Fast(iter, ""));
401  }
402  else {
403  PyErr_Clear();
404  vals.push_back(value);
405  }
406  }
407  if (!pySeq.isNone()) {
408  Py::Sequence seq(pySeq);
409  vals.reserve(seq.size());
410  for (auto it = seq.begin(); it != seq.end(); ++it) {
411  vals.push_back((*it).ptr());
412  }
413  }
414  }
415  setPyValues(vals, indices);
416 }
