124 class MeshCrossSection
125 {
126 public:
127  MeshCrossSection(const MeshCore::MeshKernel& mesh,
128  const MeshCore::MeshFacetGrid& grid,
129  double x,
130  double y,
131  double z,
132  bool connectEdges,
133  double eps)
134  : mesh(mesh)
135  , grid(grid)
136  , x(x)
137  , y(y)
138  , z(z)
139  , connectEdges(connectEdges)
140  , epsilon(eps)
141  {}
142  std::list<TopoDS_Wire> section(double d)
143  {
144  Mesh::MeshObject::TPolylines polylines;
145  MeshCore::MeshAlgorithm algo(mesh);
146  Base::Vector3f p(x * d, y * d, z * d);
147  Base::Vector3f n(x, y, z);
148  algo.CutWithPlane(p, n, grid, polylines, epsilon, connectEdges);
149 
150  std::list<TopoDS_Wire> wires;
151  for (const auto& polyline : polylines) {
152  BRepBuilderAPI_MakePolygon mkPoly;
153  for (auto jt : polyline) {
154  mkPoly.Add(Base::convertTo<gp_Pnt>(jt));
155  }
156 
157  if (mkPoly.IsDone()) {
158  wires.push_back(mkPoly.Wire());
159  }
160  }
161 
162  return wires;
163  }
164 
165 private:
166  const MeshCore::MeshKernel& mesh;
167  const MeshCore::MeshFacetGrid& grid;
168  double x, y, z;
169  bool connectEdges;
170  double epsilon;
171 };

271 void CrossSections::apply()
272 {
273  std::vector<App::DocumentObject*> obj =
274  Gui::Selection().getObjectsOfType(Mesh::Feature::getClassTypeId());
275 
276  std::vector<double> d;
277  if (ui->sectionsBox->isChecked()) {
278  d = getPlanes();
279  }
280  else {
281  d.push_back(ui->position->value().getValue());
282  }
283  double a = 0, b = 0, c = 0;
284  switch (plane()) {
285  case CrossSections::XY:
286  c = 1.0;
287  break;
288  case CrossSections::XZ:
289  b = 1.0;
290  break;
291  case CrossSections::YZ:
292  a = 1.0;
293  break;
294  }
295 
296  bool connectEdges = ui->checkBoxConnect->isChecked();
297  double eps = ui->spinEpsilon->value();
298 
299 #if 1 // multi-threaded sections
300  for (auto it : obj) {
301  const Mesh::MeshObject& mesh = static_cast<Mesh::Feature*>(it)->Mesh.getValue();
302 
303  MeshCore::MeshKernel kernel(mesh.getKernel());
304  kernel.Transform(mesh.getTransform());
305 
306  MeshCore::MeshFacetGrid grid(kernel);
307 
308  // NOLINTBEGIN
309  MeshCrossSection cs(kernel, grid, a, b, c, connectEdges, eps);
310  QFuture<std::list<TopoDS_Wire>> future =
311  QtConcurrent::mapped(d, std::bind(&MeshCrossSection::section, &cs, sp::_1));
312  future.waitForFinished();
313  // NOLINTEND
314 
315  TopoDS_Compound comp;
316  BRep_Builder builder;
317  builder.MakeCompound(comp);
318 
319  for (const auto& w : future) {
320  for (const auto& wt : w) {
321  if (!wt.IsNull()) {
322  builder.Add(comp, wt);
323  }
324  }
325  }
326 
327  App::Document* doc = it->getDocument();
328  std::string s = it->getNameInDocument();
329  s += "_cs";
330  Part::Feature* section =
331  static_cast<Part::Feature*>(doc->addObject("Part::Feature", s.c_str()));
332  section->Shape.setValue(comp);
333  section->purgeTouched();
334  }
335 #else
336  try {
337  Gui::Command::runCommand(Gui::Command::App, "import Mesh, Part\n");
338  Gui::Command::runCommand(Gui::Command::App, "from FreeCAD import Base\n");
339 
340  std::stringstream str;
341  str << "[";
342  for (std::vector<double>::iterator jt = d.begin(); jt != d.end(); ++jt) {
343  double d = *jt;
344  str << "("
345  << "App.Vector(" << a * d << ", " << b * d << ", " << c * d << "), "
346  << "App.Vector(" << a << ", " << b << ", " << c << ")"
347  << "), ";
348  }
349  str << "]";
350 
351  QString planes = QString::fromStdString(str.str());
352  for (std::vector<App::DocumentObject*>::iterator it = obj.begin(); it != obj.end(); ++it) {
353  App::Document* doc = (*it)->getDocument();
354  std::string s = (*it)->getNameInDocument();
355  s += "_cs";
356  Gui::Command::runCommand(
357  Gui::Command::App,
358  QString::fromLatin1(
359  "points=FreeCAD.getDocument(\"%1\").%2.Mesh.crossSections(%3, %4, %5)\n"
360  "wires=[]\n"
361  "for i in points:\n"
362  " wires.extend([Part.makePolygon(j) for j in i])\n")
363  .arg(QLatin1String(doc->getName()))
364  .arg(QLatin1String((*it)->getNameInDocument()))
365  .arg(planes)
366  .arg(eps)
367  .arg(connectEdges ? QLatin1String("True") : QLatin1String("False"))
368  .toLatin1());
369 
370  Gui::Command::runCommand(
371  Gui::Command::App,
372  QString::fromLatin1(
373  "comp=Part.Compound(wires)\n"
374  "slice=FreeCAD.getDocument(\"%1\").addObject(\"Part::Feature\",\"%2\")\n"
375  "slice.Shape=comp\n"
376  "slice.purgeTouched()\n"
377  "del slice,comp,wires,points")
378  .arg(QLatin1String(doc->getName()))
379  .arg(QLatin1String(s.c_str()))
380  .toLatin1());
381  }
382  }
383  catch (const Base::Exception& e) {
384  QMessageBox::critical(this, tr("Failure"), QString::fromLatin1(e.what()));
385  }
386 #endif
387 }

568 std::vector<double> CrossSections::getPlanes() const
569 {
570  int count = ui->countSections->value();
571  double pos = ui->position->value().getValue();
572  double stp = ui->distance->value().getValue();
573  bool both = ui->checkBothSides->isChecked();
574 
575  std::vector<double> d;
576  if (both) {
577  double start = pos - 0.5f * (count - 1) * stp;
578  for (int i = 0; i < count; i++) {
579  d.push_back(start + i * stp);
580  }
581  }
582  else {
583  for (int i = 0; i < count; i++) {
584  d.push_back(pos + i * stp);
585  }
586  }
587  return d;

590 void CrossSections::makePlanes(Plane type, const std::vector<double>& d, double bound[4])
591 {
592  std::vector<Base::Vector3f> points;
593  for (double it : d) {
594  Base::Vector3f v[4];
595  switch (type) {
596  case XY:
597  v[0].Set(bound[0], bound[2], it);
598  v[1].Set(bound[1], bound[2], it);
599  v[2].Set(bound[1], bound[3], it);
600  v[3].Set(bound[0], bound[3], it);
601  break;
602  case XZ:
603  v[0].Set(bound[0], it, bound[2]);
604  v[1].Set(bound[1], it, bound[2]);
605  v[2].Set(bound[1], it, bound[3]);
606  v[3].Set(bound[0], it, bound[3]);
607  break;
608  case YZ:
609  v[0].Set(it, bound[0], bound[2]);
610  v[1].Set(it, bound[1], bound[2]);
611  v[2].Set(it, bound[1], bound[3]);
612  v[3].Set(it, bound[0], bound[3]);
613  break;
614  }
615 
616  points.push_back(v[0]);
617  points.push_back(v[1]);
618  points.push_back(v[2]);
619  points.push_back(v[3]);
620  points.push_back(v[0]);
621  }
622  vp->setCoords(points);
623 }
