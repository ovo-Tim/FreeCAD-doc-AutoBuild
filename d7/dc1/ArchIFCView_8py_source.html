149  def prefillMapConversionForm(self):
150  """Prefills each of the form entries with the existing value.
151 
152  Gets the existing value from the object's IfcData, specifically the complex
153  attribute, RepresentationContexts.
154  """
155  data = ArchIFC.IfcRoot.getObjIfcComplexAttribute(self, self.object, "RepresentationContexts")
156  for lineEdit in self.lineEditObjects:
