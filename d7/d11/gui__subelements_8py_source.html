115  def get_editable_objects_from_selection(self):
116  """Get editable Draft objects for the selection."""
117  for obj in Gui.Selection.getSelection():
118  if obj.isDerivedFrom("Part::Part2DObject"):
119  self.editable_objects.append(obj)

124  def highlight_editable_objects(self):
125  """Highlight editable Draft objects from the selection."""
126  for obj in self.editable_objects:
127  vobj = obj.ViewObject
128  self.original_view_settings[obj.Name] = {
129  'Visibility': vobj.Visibility,
130  'PointSize': vobj.PointSize,
131  'PointColor': vobj.PointColor,
132  'LineColor': vobj.LineColor}
133  vobj.Visibility = True
134  vobj.PointSize = 10
135  vobj.PointColor = (1.0, 0.0, 0.0)
136  vobj.LineColor = (1.0, 0.0, 0.0)
137  xray = coin.SoAnnotation()
138  switch = gui_utils.find_coin_node(vobj.RootNode, coin.SoSwitch)
139  if switch is not None:
140  xray.addChild(switch.getChild(0))
141  xray.setName("xray")
142  vobj.RootNode.addChild(xray)
