48  # Build a dictionary of edges according to their end points.
49  # Each entry is a set of edges that starts, or ends, at the
50  # given vertex hash.
51  sdict = dict()
52  edict = dict()
53  nedges = []
54  for e in edges:
55  if hasattr(e, "Length"):
56  if e.Length != 0:
57  sdict.setdefault(e.Vertexes[0].hashCode(), []).append(e)
58  edict.setdefault(e.Vertexes[-1].hashCode(), []).append(e)
59  nedges.append(e)
60 
61  if not nedges:
62  print("DraftGeomUtils.sortEdges: zero-length edges")
63  return edges
64 
65  # Find the start of the path. The start is the vertex that appears
66  # in the sdict dictionary but not in the edict dictionary, and has
67  # only one edge ending there.
68  startedge = None
69  for v, se in sdict.items():
70  if v not in edict and len(se) == 1:
71  startedge = se
72  break
73 
74  # The above may not find a start vertex; if the start edge is reversed,
75  # the start vertex will appear in edict (and not sdict).
76  if not startedge:
77  for v, se in edict.items():
78  if v not in sdict and len(se) == 1:
79  startedge = se
80  break
81 
82  # If we still have no start vertex, it was a closed path. If so, start
83  # with the first edge in the supplied list
84  if not startedge:
85  startedge = nedges[0]
86  v = startedge.Vertexes[0].hashCode()
87 
88  # Now build the return list by walking the edges starting at the start
89  # vertex we found. We're done when we've visited each edge, so the
90  # end check is simply the count of input elements (that works for closed
91  # as well as open paths).
92  ret = list()
93  # store the hash code of the last edge, to avoid picking the same edge back
94  eh = None
95  for _ in range(len(nedges)):
96  try:
97  eset = sdict[v]
98  e = eset.pop()
99  if not eset:
100  del sdict[v]
101  if e.hashCode() == eh:
102  raise KeyError
103  v = e.Vertexes[-1].hashCode()
104  eh = e.hashCode()
105  except KeyError:
106  try:
107  eset = edict[v]
108  e = eset.pop()
109  if not eset:
110  del edict[v]
111  if e.hashCode() == eh:
112  raise KeyError
113  v = e.Vertexes[0].hashCode()
114  eh = e.hashCode()
115  e = invert(e)
116  except KeyError:
117  print("DraftGeomUtils.sortEdges failed - running old version")
118  return sortEdgesOld(edges)
119  ret.append(e)
120 
121  return ret
