176 App::ElementNamePair Feature::getExportElementName(TopoShape shape,
177  const char* name) const
178 {
179  Data::MappedElement mapped = shape.getElementName(name);
180  auto res = shape.shapeTypeAndIndex(mapped.index);
181  static const int MinLowerTopoNames = 3;
182  static const int MaxLowerTopoNames = 10;
183  if (res.second && !mapped.name) {
184  // Here means valid index name, but no mapped name, check to see if
185  // we shall generate the high level topo name.
186  //
187  // The general idea of the algorithm is to find the minimum number of
188  // lower elements that can identify the given higher element, and
189  // combine their names to generate the name for the higher element.
190  //
191  // In theory, all it takes to find one lower element that only appear
192  // in the given higher element. To make the algorithm more robust
193  // against model changes, we shall take minimum MinLowerTopoNames lower
194  // elements.
195  //
196  // On the other hand, it may be possible to take too many elements for
197  // disambiguation. We shall limit to maximum MaxLowerTopoNames. If the
198  // chosen elements are not enough to disambiguate the higher element,
199  // we'll include an index for disambiguation.
200 
201  auto subshape = shape.getSubTopoShape(res.first, res.second, true);
202  TopAbs_ShapeEnum lower;
203  Data::IndexedName idxName;
204  if (!subshape.isNull()) {
205  switch (res.first) {
206  case TopAbs_WIRE:
207  lower = TopAbs_EDGE;
208  idxName = Data::IndexedName::fromConst("Edge", 1);
209  break;
210  case TopAbs_SHELL:
211  case TopAbs_SOLID:
212  case TopAbs_COMPOUND:
213  case TopAbs_COMPSOLID:
214  lower = TopAbs_FACE;
215  idxName = Data::IndexedName::fromConst("Face", 1);
216  break;
217  default:
218  lower = TopAbs_SHAPE;
219  }
220  if (lower != TopAbs_SHAPE) {
221  typedef std::pair<size_t, std::vector<int>> NameEntry;
222  std::vector<NameEntry> indices;
223  std::vector<Data::MappedName> names;
224  std::vector<int> ancestors;
225  int count = 0;
226  for (auto& ss : subshape.getSubTopoShapes(lower)) {
227  auto name = ss.getMappedName(idxName);
228  if (!name) {
229  continue;
230  }
231  indices.emplace_back(names.size(),
232  shape.findAncestors(ss.getShape(), res.first));
233  names.push_back(name);
234  if (indices.back().second.size() == 1 && ++count >= MinLowerTopoNames) {
235  break;
236  }
237  }
238 
239  if (names.size() >= MaxLowerTopoNames) {
240  std::stable_sort(indices.begin(),
241  indices.end(),
242  [](const NameEntry& a, const NameEntry& b) {
243  return a.second.size() < b.second.size();
244  });
245  std::vector<Data::MappedName> sorted;
246  auto pos = 0;
247  sorted.reserve(names.size());
248  for (auto& v : indices) {
249  size_t size = ancestors.size();
250  if (size == 0) {
251  ancestors = v.second;
252  }
253  else if (size > 1) {
254  for (auto it = ancestors.begin(); it != ancestors.end();) {
255  if (std::find(v.second.begin(), v.second.end(), *it)
256  == v.second.end()) {
257  it = ancestors.erase(it);
258  if (ancestors.size() == 1) {
259  break;
260  }
261  }
262  else {
263  ++it;
264  }
265  }
266  }
267  auto itPos = sorted.end();
268  if (size == 1 || size != ancestors.size()) {
269  itPos = sorted.begin() + pos;
270  ++pos;
271  }
272  sorted.insert(itPos, names[v.first]);
273  if (size == 1 && sorted.size() >= MinLowerTopoNames) {
274  break;
275  }
276  }
277  }
278 
279  names.resize(std::min((int)names.size(), MaxLowerTopoNames));
280  if (names.size()) {
281  std::string op;
282  if (ancestors.size() > 1) {
283  // The current chosen elements are not enough to
284  // identify the higher element, generate an index for
285  // disambiguation.
286  auto it = std::find(ancestors.begin(), ancestors.end(), res.second);
287  if (it == ancestors.end()) {
288  assert(0 && "ancestor not found"); // this shouldn't happen
289  }
290  else {
291  op = Data::POSTFIX_INDEX + std::to_string(it - ancestors.begin());
292  }
293  }
294 
295  // Note: setting names to shape will change its underlying
296  // shared element name table. This actually violates the
297  // const'ness of this function.
298  //
299  // To be const correct, we should have made the element
300  // name table to be implicit sharing (i.e. copy on change).
301  //
302  // Not sure if there is any side effect of indirectly
303  // change the element map inside the Shape property without
304  // recording the change in undo stack.
305  //
306  mapped.name = shape.setElementComboName(mapped.index,
307  names,
308  mapped.index.getType(),
309  op.c_str());
310  }
311  }
312  }
313  }
314  else if (!res.second && mapped.name) {
315  const char* dot = strchr(name, '.');
316  if (dot) {
317  ++dot;
318  // Here means valid mapped name, but cannot find the corresponding
319  // indexed name. This usually means the model has been changed. The
320  // original indexed name is usually appended to the mapped name
321  // separated by a dot. We use it as a clue to decode the combo name
322  // set above, and try to single out one sub shape that has all the
323  // lower elements encoded in the combo name. But since we don't
324  // always use all the lower elements for encoding, this can only be
325  // consider a heuristics.
326  if (Data::hasMissingElement(dot)) {
327  dot += strlen(Data::MISSING_PREFIX);
328  }
329  std::pair<TopAbs_ShapeEnum, int> occindex = shape.shapeTypeAndIndex(dot);
330  if (occindex.second > 0) {
331  auto idxName = Data::IndexedName::fromConst(shape.shapeName(occindex.first).c_str(),
332  occindex.second);
333  std::string postfix;
334  auto names =
335  shape.decodeElementComboName(idxName, mapped.name, idxName.getType(), &postfix);
336  std::vector<int> ancestors;
337  if ( names.empty() ) {
338  // Naming based heuristic has failed to find the element. Let's see if we can
339  // find it by matching either planes for faces or lines for edges.
340  auto searchShape = this->Shape.getShape();
341  // If we're still out at a Shell, Solid, CompSolid, or Compound drill in
342  while (!searchShape.getShape().IsNull() && searchShape.getShape().ShapeType() < TopAbs_FACE ) {
343  auto shapes = searchShape.getSubTopoShapes();
344  if ( shapes.empty() ) // No more subshapes, so don't continue
345  break;
346  searchShape = shapes.front(); // After the break, so we stopped at innermost container
347  }
348  auto newMapped = TopoShape::chooseMatchingSubShapeByPlaneOrLine(shape, searchShape);
349  if ( ! newMapped.name.empty() )
350  mapped = newMapped;
351  }
352  for (auto& name : names) {
353  auto index = shape.getIndexedName(name);
354  if (!index) {
355  ancestors.clear();
356  break;
357  }
358  auto oidx = shape.shapeTypeAndIndex(index);
359  auto subshape = shape.getSubShape(oidx.first, oidx.second);
360  if (subshape.IsNull()) {
361  ancestors.clear();
362  break;
363  }
364  auto current = shape.findAncestors(subshape, occindex.first);
365  if (ancestors.empty()) {
366  ancestors = std::move(current);
367  }
368  else {
369  for (auto it = ancestors.begin(); it != ancestors.end();) {
370  if (std::find(current.begin(), current.end(), *it) == current.end()) {
371  it = ancestors.erase(it);
372  }
373  else {
374  ++it;
375  }
376  }
377  if (ancestors.empty()) { // model changed beyond recognition, bail!
378  break;
379  }
380  }
381  }
382  if (ancestors.size() > 1 && boost::starts_with(postfix, Data::POSTFIX_INDEX)) {
383  std::istringstream iss(postfix.c_str() + strlen(Data::POSTFIX_INDEX));
384  int idx;
385  if (iss >> idx && idx >= 0 && idx < (int)ancestors.size()) {
386  ancestors.resize(1, ancestors[idx]);
387  }
388  }
389  if (ancestors.size() == 1) {
390  idxName.setIndex(ancestors.front());
391  mapped.index = idxName;
392  }
393  }
394  }
395  }
396  return App::GeoFeature::_getElementName(name, mapped);
397 }

399 App::DocumentObject* Feature::getSubObject(const char* subname,
400  PyObject** pyObj,
401  Base::Matrix4D* pmat,
402  bool transform,
403  int depth) const
404 {
405  while(subname && *subname=='.') ++subname; // skip leading .
406 
407  // having '.' inside subname means it is referencing some children object,
408  // instead of any sub-element from ourself
409  if (subname && !Data::isMappedElement(subname) && strchr(subname, '.')) {
410  return App::DocumentObject::getSubObject(subname, pyObj, pmat, transform, depth);
411  }
412 
413  Base::Matrix4D _mat;
414  auto& mat = pmat ? *pmat : _mat;
415  if (transform) {
416  mat *= Placement.getValue().toMatrix();
417  }
418 
419  if (!pyObj) {
420  // TopoShape::hasSubShape is kind of slow, let's cut outself some slack here.
421  return const_cast<Feature*>(this);
422  }
423 
424  try {
425  TopoShape ts(Shape.getShape());
426  bool doTransform = mat != ts.getTransform();
427  if (doTransform) {
428  ts.setShape(ts.getShape().Located(TopLoc_Location()), false);
429  }
430  if (subname && *subname && !ts.isNull()) {
431  ts = ts.getSubTopoShape(subname,true);
432  }
433  if (doTransform && !ts.isNull()) {
434  static int sCopy = -1;
435  if (sCopy < 0) {
436  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
437  "User parameter:BaseApp/Preferences/Mod/Part/General");
438  sCopy = hGrp->GetBool("CopySubShape", false) ? 1 : 0;
439  }
440  bool copy = sCopy ? true : false;
441  if (!copy) {
442  // Work around OCC bug on transforming circular edge with an
443  // offset surface. The bug probably affect other shape type,
444  // too.
445  TopExp_Explorer exp(ts.getShape(), TopAbs_EDGE);
446  if (exp.More()) {
447  auto edge = TopoDS::Edge(exp.Current());
448  exp.Next();
449  if (!exp.More()) {
450  BRepAdaptor_Curve curve(edge);
451  copy = curve.GetType() == GeomAbs_Circle;
452  }
453  }
454  }
455  ts.transformShape(mat, copy, true);
456  }
457  *pyObj = Py::new_reference_to(shape2pyshape(ts));
458  return const_cast<Feature*>(this);
459  }
460  catch (Standard_Failure& e) {
461  // FIXME: Do not handle the exception here because it leads to a flood of irrelevant and
462  // annoying error messages.
463  // For example: https://forum.freecad.org/viewtopic.php?f=19&t=42216
464  // Instead either raise a sub-class of Base::Exception and let it handle by the calling
465  // instance or do simply nothing. For now the error message is degraded to a log message.
466  std::ostringstream str;
467  Standard_CString msg = e.GetMessageString();
468 
469  // Avoid name mangling
470  str << e.DynamicType()->get_type_name() << " ";
471 
472  if (msg) {
473  str << msg;
474  }
475  else {
476  str << "No OCCT Exception Message";
477  }
478  str << ": " << getFullName();
479  if (subname) {
480  str << '.' << subname;
481  }
482  FC_LOG(str.str());
483  return nullptr;
484  }
485 }

487 static std::vector<std::pair<long, Data::MappedName>> getElementSource(App::DocumentObject* owner,
488  TopoShape shape,
489  const Data::MappedName& name,
490  char type)
491 {
492  std::set<std::pair<App::Document*, long>> tagSet;
493  std::vector<std::pair<long, Data::MappedName>> ret;
494  ret.emplace_back(0, name);
495  int depth = 0;
496  while (1) {
497  Data::MappedName original;
498  std::vector<Data::MappedName> history;
499  // It is possible the name does not belong to the shape, e.g. when user
500  // changes modeling order in PartDesign. So we try to assign the
501  // document hasher here in case getElementHistory() needs to de-hash
502  if (!shape.Hasher && owner) {
503  shape.Hasher = owner->getDocument()->getStringHasher();
504  }
505  long tag = shape.getElementHistory(ret.back().second, &original, &history);
506  if (!tag) {
507  break;
508  }
509  auto obj = owner;
510  App::Document* doc = nullptr;
511  if (owner) {
512  doc = owner->getDocument();
513  for (;; ++depth) {
514  auto linked = owner->getLinkedObject(false, nullptr, false, depth);
515  if (linked == owner) {
516  break;
517  }
518  owner = linked;
519  if (owner->getDocument() != doc) {
520  doc = owner->getDocument();
521  break;
522  }
523  }
524  if (owner->isDerivedFrom(App::GeoFeature::getClassTypeId())) {
525  auto ownerGeoFeature =
526  static_cast<App::GeoFeature*>(owner)->getElementOwner(ret.back().second);
527  if (ownerGeoFeature) {
528  doc = ownerGeoFeature->getDocument();
529  }
530  }
531  obj = doc->getObjectByID(tag < 0 ? -tag : tag);
532  if (type) {
533  for (auto& hist : history) {
534  if (shape.elementType(hist) != type) {
535  return ret;
536  }
537  }
538  }
539  }
540  owner = 0;
541  if (!obj) {
542  // Object maybe deleted, but it is still possible to extract the
543  // source element name from hasher table.
544  shape.setShape(TopoDS_Shape());
545  doc = nullptr;
546  }
547  else {
548  shape = Part::Feature::getTopoShape(obj, 0, false, 0, &owner);
549  }
550  if (type && shape.elementType(original) != type) {
551  break;
552  }
553 
554  if (std::abs(tag) != ret.back().first && !tagSet.insert(std::make_pair(doc, tag)).second) {
555  // Because an object might be deleted, which may be a link/binder
556  // that points to an external object that contain element name
557  // using external hash table. We shall prepare for circular element
558  // map due to looking up in the wrong table.
559  if (FC_LOG_INSTANCE.isEnabled(FC_LOGLEVEL_LOG)) {
560  FC_WARN("circular element mapping");
561  }
562  break;
563  }
564  ret.emplace_back(tag, original);
565  }
566  return ret;
567 }

569 std::list<Data::HistoryItem> Feature::getElementHistory(App::DocumentObject* feature,
570  const char* name,
571  bool recursive,
572  bool sameType)
573 {
574  std::list<Data::HistoryItem> ret;
575  TopoShape shape = getTopoShape(feature);
576  Data::IndexedName idx(name);
577  Data::MappedName element;
578  Data::MappedName prevElement;
579  if (idx) {
580  element = shape.getMappedName(idx, true);
581  }
582  else if (Data::isMappedElement(name)) {
583  element = Data::MappedName(Data::newElementName(name));
584  }
585  else {
586  element = Data::MappedName(name);
587  }
588  char element_type = 0;
589  if (sameType) {
590  element_type = shape.elementType(element);
591  }
592  int depth = 0;
593  do {
594  Data::MappedName original;
595  ret.emplace_back(feature, element);
596  long tag = shape.getElementHistory(element, &original, &ret.back().intermediates);
597 
598  ret.back().index = shape.getIndexedName(element);
599  if (!ret.back().index && prevElement) {
600  ret.back().index = shape.getIndexedName(prevElement);
601  if (ret.back().index) {
602  ret.back().intermediates.insert(ret.back().intermediates.begin(), element);
603  ret.back().element = prevElement;
604  }
605  }
606  if (ret.back().intermediates.size()) {
607  prevElement = ret.back().intermediates.back();
608  }
609  else {
610  prevElement = Data::MappedName();
611  }
612 
613  App::DocumentObject* obj = nullptr;
614  if (tag) {
615  App::Document* doc = feature->getDocument();
616  for (;; ++depth) {
617  auto linked = feature->getLinkedObject(false, nullptr, false, depth);
618  if (linked == feature) {
619  break;
620  }
621  feature = linked;
622  if (feature->getDocument() != doc) {
623  doc = feature->getDocument();
624  break;
625  }
626  }
627  if (feature->isDerivedFrom(App::GeoFeature::getClassTypeId())) {
628  auto ownerGeoFeature =
629  static_cast<App::GeoFeature*>(feature)->getElementOwner(element);
630  if (ownerGeoFeature) {
631  doc = ownerGeoFeature->getDocument();
632  }
633  }
634  obj = doc->getObjectByID(std::abs(tag));
635  }
636  if (!recursive) {
637  ret.emplace_back(obj, original);
638  ret.back().tag = tag;
639  return ret;
640  }
641  if (!obj) {
642  break;
643  }
644  if (element_type) {
645  for (auto& hist : ret.back().intermediates) {
646  if (shape.elementType(hist) != element_type) {
647  return ret;
648  }
649  }
650  }
651  feature = obj;
652  shape = Feature::getTopoShape(feature);
653  element = original;
654  if (element_type && shape.elementType(original) != element_type) {
655  break;
656  }
657  } while (feature);
658  return ret;
659 }

762  res.append(e);

764  break;
765  }
766  }
767  return res;
768  }

772  }

775  // result in multiple matches, e.g. a compound of array of the same

779  checkingSubname = Data::IndexedName::fromConst(shapetype, i + 1);
780  auto mapped = shape.getMappedName(checkingSubname);
781  tagChanges = 0;
782  shape.traceElement(mapped, checkHistory);
783  if (single && res.size()) {
784  break;
785  }
786  }
787  return res;
788 }
