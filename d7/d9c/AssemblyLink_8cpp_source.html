96 void AssemblyLink::onChanged(const App::Property* prop)
97 {
98  if (App::GetApplication().isRestoring()) {
99  App::Part::onChanged(prop);
100  return;
101  }
102 
103  if (prop == &Rigid) {
104  Base::Placement movePlc;
105 
106  if (Rigid.getValue()) {
107  // movePlc needs to be computed before updateContents.
108  if (!objLinkMap.empty()) {
109  auto firstElement = *objLinkMap.begin();
110 
111  App::DocumentObject* obj = firstElement.first;
112  App::DocumentObject* link = firstElement.second;
113  auto* prop =
114  dynamic_cast<App::PropertyPlacement*>(obj->getPropertyByName("Placement"));
115  auto* prop2 =
116  dynamic_cast<App::PropertyPlacement*>(link->getPropertyByName("Placement"));
117  if (prop && prop2) {
118  movePlc = prop2->getValue() * prop->getValue().inverse();
119  }
120  }
121  }
122 
123  updateContents();
124 
125  auto* propPlc = dynamic_cast<App::PropertyPlacement*>(getPropertyByName("Placement"));
126  if (!propPlc) {
127  return;
128  }
129 
130  if (!Rigid.getValue()) {
131  // when the assemblyLink becomes flexible, we need to make sure its placement is
132  // identity or it's going to mess up moving parts placement within.
133  Base::Placement plc = propPlc->getValue();
134  if (!plc.isIdentity()) {
135  propPlc->setValue(Base::Placement());
136 
137  // We need to apply the placement of the assembly link to the children or they will
138  // move.
139  std::vector<App::DocumentObject*> group = Group.getValues();
140  for (auto* obj : group) {
141  if (!obj->isDerivedFrom<App::Part>() && !obj->isDerivedFrom<PartApp::Feature>()
142  && !obj->isDerivedFrom<App::Link>()) {
143  continue;
144  }
145 
146  auto* prop =
147  dynamic_cast<App::PropertyPlacement*>(obj->getPropertyByName("Placement"));
148  if (prop) {
149  prop->setValue(plc * prop->getValue());
150  }
151  }
152 
153  AssemblyObject::redrawJointPlacements(getJoints());
154  }
155  }
156  else {
157  // For the assemblylink not to move to origin, we need to update its placement.
158  if (!movePlc.isIdentity()) {
159  propPlc->setValue(movePlc);
160  }
161  }
162 
163  return;
164  }
165  App::Part::onChanged(prop);
166 }

182 void AssemblyLink::synchronizeComponents()
183 {
184  App::Document* doc = getDocument();
185 
186  AssemblyObject* assembly = getLinkedAssembly();
187  if (!assembly) {
188  return;
189  }
190 
191  objLinkMap.clear();
192 
193  std::vector<App::DocumentObject*> assemblyGroup = assembly->Group.getValues();
194  std::vector<App::DocumentObject*> assemblyLinkGroup = Group.getValues();
195 
196  // We check if a component needs to be added to the AssemblyLink
197  for (auto* obj : assemblyGroup) {
198  if (!obj->isDerivedFrom<App::Part>() && !obj->isDerivedFrom<PartApp::Feature>()
199  && !obj->isDerivedFrom<App::Link>()) {
200  continue;
201  }
202 
203  // Note, the user can have nested sub-assemblies.
204  // In which case we need to add an AssemblyLink and not a Link.
205  App::DocumentObject* link = nullptr;
206  bool found = false;
207  for (auto* obj2 : assemblyLinkGroup) {
208  App::DocumentObject* linkedObj;
209 
210  auto* subAsmLink = dynamic_cast<AssemblyLink*>(obj2);
211  auto* link2 = dynamic_cast<App::Link*>(obj2);
212  if (subAsmLink) {
213  linkedObj = subAsmLink->getLinkedObject2(false); // not recursive
214  }
215  else if (link2) {
216  linkedObj = link2->getLinkedObject(false); // not recursive
217  }
218  else {
219  // We consider only Links and AssemblyLinks in the AssemblyLink.
220  continue;
221  }
222 
223 
224  if (linkedObj == obj) {
225  found = true;
226  link = obj2;
227  break;
228  }
229  }
230  if (!found) {
231  // Add a link or a AssemblyLink to it in the AssemblyLink.
232  if (obj->isDerivedFrom<AssemblyLink>()) {
233  auto* asmLink = static_cast<AssemblyLink*>(obj);
234  auto* subAsmLink = new AssemblyLink();
235  doc->addObject(subAsmLink, obj->getNameInDocument());
236  subAsmLink->LinkedObject.setValue(obj);
237  subAsmLink->Rigid.setValue(asmLink->Rigid.getValue());
238  subAsmLink->Label.setValue(obj->Label.getValue());
239  addObject(subAsmLink);
240  link = subAsmLink;
241  }
242  else {
243  auto* appLink = new App::Link();
244  doc->addObject(appLink, obj->getNameInDocument());
245  appLink->LinkedObject.setValue(obj);
246  appLink->Label.setValue(obj->Label.getValue());
247  addObject(appLink);
248  link = appLink;
249  }
250  }
251 
252  objLinkMap[obj] = link;
253  // If the assemblyLink is rigid, then we keep the placement synchronized.
254  if (isRigid()) {
255  auto* plcProp =
256  dynamic_cast<App::PropertyPlacement*>(obj->getPropertyByName("Placement"));
257  auto* plcProp2 =
258  dynamic_cast<App::PropertyPlacement*>(link->getPropertyByName("Placement"));
259  if (plcProp && plcProp2) {
260  if (!plcProp->getValue().isSame(plcProp2->getValue())) {
261  plcProp2->setValue(plcProp->getValue());
262  }
263  }
264  }
265  }
266 
267  // We check if a component needs to be removed from the AssemblyLink
268  for (auto* link : assemblyLinkGroup) {
269  // We don't need to update assemblyLinkGroup after the addition since we're not removing
270  // something we just added.
271 
272  if (objLinkMap.find(link) != objLinkMap.end()) {
273  doc->removeObject(link->getNameInDocument());
274  }
275 
276  /*if (!link->isDerivedFrom<App::Link>() && !link->isDerivedFrom<AssemblyLink>()) {
277  // AssemblyLink should contain only Links or assembly links.
278  continue;
279  }
280 
281  auto* linkedObj = link->getLinkedObject(false); // not recursive
282 
283  bool found = false;
284  for (auto* obj2 : assemblyGroup) {
285  if (obj2 == linkedObj) {
286  found = true;
287  break;
288  }
289  }
290  if (!found) {
291  doc->removeObject(link->getNameInDocument());
292  }*/
293  }

299 void copyPropertyIfDifferent(App::DocumentObject* source,
300  App::DocumentObject* target,
301  const char* propertyName)
302 {

304  auto targetProp = dynamic_cast<T*>(target->getPropertyByName(propertyName));

310 std::string removeUpToName(const std::string& sub, const std::string& name)
311 {
312  size_t pos = sub.find(name);
313  if (pos != std::string::npos) {
314  // Move the position to the character after the found substring and the following '.'
315  pos += name.length() + 1;
316  if (pos < sub.length()) {
317  return sub.substr(pos);
318  }
319  }
320  // If s2 is not found in s1, return the original string
321  return sub;
322 }
323 
324 std::string
325 replaceLastOccurrence(const std::string& str, const std::string& oldStr, const std::string& newStr)
326 {
327  size_t pos = str.rfind(oldStr);
328  if (pos != std::string::npos) {
329  std::string result = str;
330  result.replace(pos, oldStr.length(), newStr);
331  return result;
332  }
333  return str;
334 }

337 void AssemblyLink::synchronizeJoints()
338 {
339  App::Document* doc = getDocument();
340  AssemblyObject* assembly = getLinkedAssembly();
341  if (!assembly) {
342  return;
343  }
344 
345  JointGroup* jGroup = ensureJointGroup();
346 
347  std::vector<App::DocumentObject*> assemblyJoints =
348  assembly->getJoints(assembly->isTouched(), false, false);
349  std::vector<App::DocumentObject*> assemblyLinkJoints = getJoints();
350 
351  // We delete the excess of joints if any
352  for (size_t i = assemblyJoints.size(); i < assemblyLinkJoints.size(); ++i) {
353  doc->removeObject(assemblyLinkJoints[i]->getNameInDocument());
354  }
355 
356  // We make sure the joints match.
357  for (size_t i = 0; i < assemblyJoints.size(); ++i) {
358  App::DocumentObject* joint = assemblyJoints[i];
359  App::DocumentObject* lJoint;
360  if (i < assemblyLinkJoints.size()) {
361  lJoint = assemblyLinkJoints[i];
362  }
363  else {
364  auto ret = doc->copyObject({joint});
365  if (ret.size() != 1) {
366  continue;
367  }
368  lJoint = ret[0];
369  jGroup->addObject(lJoint);
370  }
371 
372  // Then we have to check the properties one by one.
373  copyPropertyIfDifferent<App::PropertyBool>(joint, lJoint, "Activated");
374  copyPropertyIfDifferent<App::PropertyFloat>(joint, lJoint, "Distance");
375  copyPropertyIfDifferent<App::PropertyFloat>(joint, lJoint, "Distance2");
376  copyPropertyIfDifferent<App::PropertyEnumeration>(joint, lJoint, "JointType");
377  copyPropertyIfDifferent<App::PropertyPlacement>(joint, lJoint, "Offset1");
378  copyPropertyIfDifferent<App::PropertyPlacement>(joint, lJoint, "Offset2");
379 
380  copyPropertyIfDifferent<App::PropertyBool>(joint, lJoint, "Detach1");
381  copyPropertyIfDifferent<App::PropertyBool>(joint, lJoint, "Detach2");
382 
383  copyPropertyIfDifferent<App::PropertyFloat>(joint, lJoint, "AngleMax");
384  copyPropertyIfDifferent<App::PropertyFloat>(joint, lJoint, "AngleMin");
385  copyPropertyIfDifferent<App::PropertyFloat>(joint, lJoint, "LengthMax");
386  copyPropertyIfDifferent<App::PropertyFloat>(joint, lJoint, "LengthMin");
387  copyPropertyIfDifferent<App::PropertyBool>(joint, lJoint, "EnableAngleMax");
388  copyPropertyIfDifferent<App::PropertyBool>(joint, lJoint, "EnableAngleMin");
389  copyPropertyIfDifferent<App::PropertyBool>(joint, lJoint, "EnableLengthMax");
390  copyPropertyIfDifferent<App::PropertyBool>(joint, lJoint, "EnableLengthMin");
391 
392  // The reference needs to be handled specifically
393  handleJointReference(joint, lJoint, "Reference1");
394  handleJointReference(joint, lJoint, "Reference2");
395  }
396 
397  assemblyLinkJoints = getJoints();
398 
399  AssemblyObject::recomputeJointPlacements(assemblyLinkJoints);
400 
401  for (auto* joint : assemblyLinkJoints) {
402  joint->purgeTouched();
403  }
404 }

407 void AssemblyLink::handleJointReference(App::DocumentObject* joint,
408  App::DocumentObject* lJoint,
409  const char* refName)
410 {
411  AssemblyObject* assembly = getLinkedAssembly();
412 
413  auto prop1 = dynamic_cast<App::PropertyXLinkSubHidden*>(joint->getPropertyByName(refName));
414  auto prop2 = dynamic_cast<App::PropertyXLinkSubHidden*>(lJoint->getPropertyByName(refName));
415  if (!prop1 || !prop2) {
416  return;
417  }
418 
419  App::DocumentObject* obj1 = nullptr;
420  App::DocumentObject* obj2 = prop2->getValue();
421  std::vector<std::string> subs1 = prop1->getSubValues();
422  std::vector<std::string> subs2 = prop2->getSubValues();
423  if (subs1.empty()) {
424  return;
425  }
426 
427  // Example :
428  // Obj1 = docA-Asm1 Subs1 = ["part1.body.pad.face0", "part1.body.pad.vertex1"]
429  // Obj1 = docA-Part Subs1 = ["Asm1.part1.body.pad.face0", "Asm1.part1.body.pad.vertex1"] // some
430  // user may put the assembly inside a part... should become : Obj2 = docB-Asm2 Subs2 =
431  // ["Asm1Link.part1.linkTobody.pad.face0", "Asm1Link.part1.linkTobody.pad.vertex1"] Obj2 =
432  // docB-Part Sub2 = ["Asm2.Asm1Link.part1.linkTobody.pad.face0",
433  // "Asm2.Asm1Link.part1.linkTobody.pad.vertex1"]
434 
435  std::string asmLink = getNameInDocument();
436  for (auto& sub : subs1) {
437  // First let's remove 'Asm1' name and everything before if any.
438  sub = removeUpToName(sub, assembly->getNameInDocument());
439  // Then we add the assembly link name.
440  sub = asmLink + "." + sub;
441  // Then the question is, is there more to prepend? Because the parent assembly may have some
442  // parents So we check assemblyLink parents and prepend necessary parents.
443  bool first = true;
444  std::vector<App::DocumentObject*> inList = getInList();
445  int limit = 0;
446  while (!inList.empty() && limit < 20) {
447  ++limit;
448  bool found = false;
449  for (auto* obj : inList) {
450  if (obj->isDerivedFrom<App::Part>()) {
451  found = true;
452  if (first) {
453  first = false;
454  }
455  else {
456  std::string obj1Name = obj1->getNameInDocument();
457  sub = obj1Name + "." + sub;
458  }
459  obj1 = obj;
460  break;
461  }
462  }
463  if (found) {
464  inList = obj1->getInList();
465  }
466  else {
467  inList = {};
468  }
469  }
470 
471  // Lastly we need to replace the object name by its link name.
472  auto* obj = getObjFromRef(prop1);
473  auto* link = objLinkMap[obj];
474  if (!obj || !link) {
475  return;
476  }
477  std::string objName = obj->getNameInDocument();
478  std::string linkName = link->getNameInDocument();
479  sub = replaceLastOccurrence(sub, objName, linkName);
480  }
481  // Now obj1 and the subs1 are what should be in obj2 and subs2 if the joint did not changed
482  if (obj1 != obj2) {
483  prop2->setValue(obj1);
484  }
485  bool changed = false;
486  for (size_t i = 0; i < subs1.size(); ++i) {
487  if (i >= subs2.size() || subs1[i] != subs2[i]) {
488  changed = true;
489  break;
490  }
491  }
492  if (changed) {
493  prop2->setSubValues(std::move(subs1));
494  }
495 }
