123  def execute(self,selfobj):
124  shapes = []
125  for obj in selfobj.Objects:
126  sh = obj.Shape.copy(True,False)
127  sh.limitTolerance(selfobj.minTolerance,selfobj.maxTolerance)
128  if selfobj.Refine:
129  sh.fix(selfobj.minTolerance,selfobj.minTolerance,selfobj.maxTolerance)
130  sh = sh.removeSplitter()
131  sh.fix(selfobj.minTolerance,selfobj.minTolerance,selfobj.maxTolerance)
132  shapes.append(sh)
133 
134  if len(shapes)>1:
135  rst = Part.makeCompound(shapes)
136  else:
137  rst = shapes[0]
138  selfobj.Shape = rst
