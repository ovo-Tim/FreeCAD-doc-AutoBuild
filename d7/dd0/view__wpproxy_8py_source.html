100  def writeCamera(self):
101  if hasattr(self,"Object"):
102  n = Gui.ActiveDocument.ActiveView.getCameraNode()
103  App.Console.PrintMessage(QT_TRANSLATE_NOOP("Draft","Writing camera position")+"\n")
104  cdata = list(n.position.getValue().getValue())
105  cdata.extend(list(n.orientation.getValue().getValue()))
106  cdata.append(n.nearDistance.getValue())
107  cdata.append(n.farDistance.getValue())
108  cdata.append(n.aspectRatio.getValue())
109  cdata.append(n.focalDistance.getValue())
110  if isinstance(n,coin.SoOrthographicCamera):
111  cdata.append(n.height.getValue())
112  cdata.append(0.0) # orthograhic camera
113  elif isinstance(n,coin.SoPerspectiveCamera):
114  cdata.append(n.heightAngle.getValue())
115  cdata.append(1.0) # perspective camera
116  self.Object.ViewObject.ViewData = cdata
