27 // Because the existence of hierarchical element maps, for the same document
28 // we may store an element map more than once in multiple objects. And because
29 // we may want to support partial loading, we choose to tolerate such redundancy
30 // for now.
31 //
32 // In order to not waste memory space when the file is loaded, we use the
33 // following two maps to assign a one-time id for each unique element map. The
34 // id will be saved together with the element map.
35 //
36 // When restoring, we'll read back the id and lookup for an existing element map
37 // with the same id, and skip loading the current map if one is found.
38 //
39 // TODO: Note that the same redundancy can be found when saving OCC shapes,
40 // because we currently save shapes for each object separately. After restoring,
41 // any shape sharing is lost. But again, we do want to keep separate shape files
42 // because of partial loading. The same technique used here can be applied to
43 // restore shape sharing.
44 static std::unordered_map<const ElementMap*, unsigned> _elementMapToId;
45 static std::unordered_map<unsigned, ElementMapPtr> _idToElementMap;

614 void ElementMap::encodeElementName(char element_type,
615  MappedName& name,
616  std::ostringstream& ss,
617  ElementIDRefs* sids,
618  long masterTag,
619  const char* postfix,
620  long tag,
621  bool forceTag) const
622 {
623  if (postfix && (postfix[0] != 0)) {
624  if (!boost::starts_with(postfix, ELEMENT_MAP_PREFIX)) {
625  ss << ELEMENT_MAP_PREFIX;
626  }
627  ss << postfix;
628  }
629  long inputTag = 0;
630  if (!forceTag && (ss.tellp() == 0)) {
631  if ((tag == 0) || tag == masterTag) {
632  return;
633  }
634  name.findTagInElementName(&inputTag, nullptr, nullptr, nullptr, true);
635  if (inputTag == tag) {
636  return;
637  }
638  }
639  else if ((tag == 0) || (!forceTag && tag == masterTag)) {
640  int pos = name.findTagInElementName(&inputTag, nullptr, nullptr, nullptr, true);
641  if (inputTag != 0) {
642  tag = inputTag;
643  // About to encode the same tag used last time. This usually means
644  // the owner object is doing multistep modeling. Let's not
645  // recursively encode the same tag too many times. It will be a
646  // waste of memory, because the intermediate shapes has no
647  // corresponding objects, so no real value for history tracing.
648  //
649  // On the other hand, we still need to distinguish the original name
650  // from the input object from the element name of the intermediate
651  // shapes. So we limit ourselves to encode only one extra level
652  // using the same tag. In order to do that, we need to de-hash the
653  // previous level name, and check for its tag.
654  Data::MappedName mappedName(name, 0, pos);
655  Data::MappedName prev = dehashElementName(mappedName);
656  long prevTag = 0;
657  prev.findTagInElementName(&prevTag, nullptr, nullptr, nullptr, true);
658  if (prevTag == inputTag || prevTag == -inputTag) {
659  name = mappedName;
660  }
661  }
662  }
663 
664  if (sids && this->hasher) {
665  name = hashElementName(name, *sids);
666  if (!forceTag && (tag == 0) && (ss.tellp() != 0)) {
667  forceTag = true;
668  }
669  }
670  if (forceTag || (tag != 0)) {
671  assert(element_type);
672  auto pos = ss.tellp();
673  boost::io::ios_flags_saver ifs(ss);
674  ss << POSTFIX_TAG << std::hex;
675  if (tag < 0) {
676  ss << '-' << -tag;
677  }
678  else if (tag != 0) {
679  ss << tag;
680  }

1083 void ElementMap::addChildElements(long masterTag, const std::vector<MappedChildElements>& children)
1084 {
1085  std::ostringstream ss;
1086  ss << std::hex;
1087 
1088  // To avoid possibly very long recursive child map lookup, resulting very
1089  // long mapped names, we try to resolve the grand child map now.
1090  std::vector<MappedChildElements> expansion;
1091  for (auto it = children.begin(); it != children.end(); ++it) {
1092  auto& child = *it;
1093  if (!child.elementMap || child.elementMap->childElements.empty()) {
1094  if (!expansion.empty()) {
1095  expansion.push_back(child);
1096  }
1097  continue;
1098  }
1099  auto& indices = child.elementMap->indexedNames[child.indexedName.getType()];
1100  if (indices.children.empty()) {
1101  if (!expansion.empty()) {
1102  expansion.push_back(child);
1103  }
1104  continue;
1105  }
1106 
1107  // Note that it is allowable to have both mapped names and child map. We
1108  // may have to split the current child mapping into pieces.
1109 
1110  int start = child.indexedName.getIndex();
1111  int end = start + child.count;
1112  for (auto iter = indices.children.upper_bound(start); iter != indices.children.end();
1113  ++iter) {
1114  auto& grandchild = iter->second;
1115  int istart = grandchild.indexedName.getIndex() + grandchild.offset;
1116  int iend = istart + grandchild.count;
1117  if (end <= istart) {
1118  break;
1119  }
1120  if (istart >= end) {
1121  if (!expansion.empty()) {
1122  expansion.push_back(child);
1123  expansion.back().indexedName.setIndex(start);
1124  expansion.back().count = end - start;
1125  }
1126  break;
1127  }
1128  if (expansion.empty()) {
1129  const int extra {10};
1130  expansion.reserve(children.size() + extra);
1131  expansion.insert(expansion.end(), children.begin(), it);
1132  }
1133  expansion.push_back(child);
1134  auto* entry = &expansion.back();
1135  if (istart > start) {
1136  entry->indexedName.setIndex(start);
1137  entry->count = istart - start;
1138 
1139  expansion.push_back(child);
1140  entry = &expansion.back();
1141  }
1142  else {
1143  istart = start;
1144  }
1145 
1146  if (iend > end) {
1147  iend = end;
1148  }
1149 
1150  entry->indexedName.setIndex(istart - grandchild.offset);
1151  entry->count = iend - istart;
1152  entry->offset += grandchild.offset;
1153  entry->elementMap = grandchild.elementMap;
1154  entry->sids += grandchild.sids;
1155  if (grandchild.postfix.size() != 0) {
1156  if ((entry->postfix.size() != 0)
1157  && !entry->postfix.startsWith(ELEMENT_MAP_PREFIX)) {
1158  entry->postfix = grandchild.postfix + ELEMENT_MAP_PREFIX + entry->postfix;
1159  }
1160  else {
1161  entry->postfix = grandchild.postfix + entry->postfix;
1162  }
1163  }
1164 
1165  start = iend;
1166  if (start >= end) {
1167  break;
1168  }
1169  }
1170  if (!expansion.empty() && start < end) {
1171  expansion.push_back(child);
1172  expansion.back().indexedName.setIndex(start);
1173  expansion.back().count = end - start;
1174  }
1175  }
1176 
1177  for (auto& child : expansion.empty() ? children : expansion) {
1178  if (!child.indexedName || (child.count == 0)) {
1179  if (FC_LOG_INSTANCE.isEnabled(FC_LOGLEVEL_LOG)) {
1180  FC_ERR("invalid mapped child element"); // NOLINT
1181  }
1182  continue;
1183  }
1184 
1185  ss.str("");
1186  MappedName tmp;
1187 
1188  ChildMapInfo* entry = nullptr;
1189 
1190  // do child mapping only if the child element count >= 5
1191  const int threshold {5};
1192  if (child.count >= threshold || !child.elementMap) {
1193  encodeElementName(child.indexedName[0],
1194  tmp,
1195  ss,
1196  nullptr,
1197  masterTag,
1198  child.postfix.constData(),
1199  child.tag,
1200  true);
1201 
1202  // Perform some disambiguation in case the same shape is mapped
1203  // multiple times, e.g. draft array.
1204  entry = &childElements[tmp.toBytes()];
1205  int mapIndex = entry->mapIndices[child.elementMap.get()]++;
1206  ++entry->index;
1207  if (entry->index != 1 && child.elementMap && mapIndex == 0) {
1208  // This child has duplicated 'tag' and 'postfix', but it
1209  // has its own element map. We'll expand this map now.
1210  entry = nullptr;
1211  }
1212  }
1213 
1214  if (!entry) {
1215  IndexedName childIdx(child.indexedName);
1216  IndexedName idx(childIdx.getType(), childIdx.getIndex() + child.offset);
1217  for (int i = 0; i < child.count; ++i, ++childIdx, ++idx) {
1218  ElementIDRefs sids;
1219  MappedName name = child.elementMap->find(childIdx, &sids);
1220  if (!name) {
1221  if ((child.tag == 0) || child.tag == masterTag) {
1222  if (FC_LOG_INSTANCE.isEnabled(FC_LOGLEVEL_LOG)) {
1223  FC_WARN("unmapped element"); // NOLINT
1224  }
1225  continue;
1226  }
1227  name = MappedName(childIdx);
1228  }
1229  ss.str("");
1230  encodeElementName(idx[0],
1231  name,
1232  ss,
1233  &sids,
1234  masterTag,
1235  child.postfix.constData(),
1236  child.tag);
1237  setElementName(idx, name, masterTag, &sids);
1238  }
1239  continue;
1240  }
1241 
1242  if (entry->index != 1) {
1243  // There is some ambiguity in child mapping. We need some
1244  // additional postfix for disambiguation. NOTE: We are not
1245  // using ComplexGeoData::indexPostfix() so we don't confuse
1246  // other code that actually uses this postfix for indexing
1247  // purposes. Here, we just need some postfix for
1248  // disambiguation. We don't need to extract the index.
1249  ss.str("");
1250  ss << ELEMENT_MAP_PREFIX << ":C" << entry->index - 1;
1251 
1252  tmp.clear();
1253  encodeElementName(child.indexedName[0],
1254  tmp,
1255  ss,
1256  nullptr,
1257  masterTag,
1258  child.postfix.constData(),
1259  child.tag,
1260  true);
1261 
1262  entry = &childElements[tmp.toBytes()];
1263  if (entry->childMap) {
1264  FC_ERR("duplicate mapped child element"); // NOLINT
1265  continue;
1266  }
1267  }
1268 
1269  auto& indices = this->indexedNames[child.indexedName.getType()];
1270  auto res =
1271  indices.children.emplace(child.indexedName.getIndex() + child.offset + child.count,
1272  child);
1273  if (!res.second) {
1274  if (!entry->childMap) {
1275  this->childElements.remove(tmp.toBytes());
1276  }
1277  FC_ERR("duplicate mapped child element"); // NOLINT
1278  continue;
1279  }
1280 
1281  auto& insertedChild = res.first->second;
1282  insertedChild.postfix = tmp.toBytes();
1283  entry->childMap = &insertedChild;
1284  childElementSize += insertedChild.count;
1285  }

1327 long ElementMap::getElementHistory(const MappedName& name,
1328  long masterTag,
1329  MappedName* original,
1330  std::vector<MappedName>* history) const
1331 {
1332  long tag = 0;
1333  int len = 0;
1334  int pos = name.findTagInElementName(&tag, &len, nullptr, nullptr, true);
1335  if (pos < 0) {
1336  if (original) {
1337  *original = name;
1338  }
1339  return tag;
1340  }
1341  if (!original && !history) {
1342  return tag;
1343  }
1344 
1345  MappedName tmp;
1346  MappedName& ret = original ? *original : tmp;
1347  if (name.startsWith(ELEMENT_MAP_PREFIX)) {
1348  unsigned offset = ELEMENT_MAP_PREFIX_SIZE;
1349  ret = MappedName::fromRawData(name, static_cast<int>(offset));
1350  }
1351  else {
1352  ret = name;
1353  }
1354 
1355  while (true) {
1356  if ((len == 0) || len > pos) {
1357  FC_WARN("invalid name length " << name); // NOLINT
1358  return 0;
1359  }
1360  bool deHashed = false;
1361  if (ret.startsWith(MAPPED_CHILD_ELEMENTS_PREFIX, len)) {
1362  int offset = (int)POSTFIX_TAG_SIZE;
1363  MappedName tmp2 = MappedName::fromRawData(ret, len + offset, pos - len - offset);
1364  MappedName postfix = dehashElementName(tmp2);
1365  if (postfix != tmp2) {
