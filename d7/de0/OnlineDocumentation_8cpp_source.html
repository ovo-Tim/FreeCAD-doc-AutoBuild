89 QByteArray PythonOnlineHelp::loadFavicon() const
90 {
91  // Return a resource icon in ico format
92  QByteArray res;
93  QBuffer buffer;
94  buffer.open(QBuffer::WriteOnly);
95  QImageWriter writer;
96  writer.setDevice(&buffer);
97  writer.setFormat("ICO");
98  if (writer.canWrite()) {
99  const int size = 24;
100  QPixmap px = qApp->windowIcon().pixmap(size, size); // NOLINT
101  writer.write(px.toImage());
102  buffer.close();
103  res = buffer.data();
104  }
105  else {
106  // fallback
107  res.reserve(navicon_data_len);
108  for (int i = 0; i < (int)navicon_data_len; i++) {
109  res[i] = navicon_data[i];
110  }
111  }
112 
113  return res;
114 }

117 {

132 QByteArray PythonOnlineHelp::tryInvoke(const std::function<std::string(Py::Module&)>& func) const
133 {
134  PyObject* module = PyImport_ImportModule("freecad.freecad_doc");
135  if (!module) {
136  throw Py::Exception();
137  }
138 
139  Py::Module mod(module, true);
140  std::string contents = func(mod);
141  QByteArray res;
142  res.append("HTTP/1.0 200 OK\n");
143  res.append("Content-type: text/html\n");

240 void HttpServer::incomingConnection(qintptr socket)
241 {
242  if (disabled) {
243  return;
244  }
245 
246  // When a new client connects the server constructs a QTcpSocket and all
247  // communication with the client is done over this QTcpSocket. QTcpSocket
248  // works asynchronously, this means that all the communication is done
249  // in the two slots readClient() and discardClient().
250  auto s = new QTcpSocket(this);
251  connect(s, &QTcpSocket::readyRead, this, &HttpServer::readClient);
252  connect(s, &QTcpSocket::disconnected, this, &HttpServer::discardClient);
253  s->setSocketDescriptor(socket);
254 }

266 void HttpServer::readClient()
267 {
268  if (disabled) {
269  return;
270  }
271 
272  // This slot is called when the client sent data to the server. The
273  // server looks if it was a GET request and sends back the
274  // corresponding HTML document from the ZIP file.
275  auto socket = qobject_cast<QTcpSocket*>(sender());
276  if (socket && socket->canReadLine()) {
277  // NOLINTBEGIN
278  QString httpRequestHeader = QString::fromLatin1(socket->readLine());
279  QStringList lst = httpRequestHeader.simplified().split(QLatin1String(" "));
280  QString method;
281  QString path;
282  if (lst.count() > 0) {
283  QString m = lst[0];
284  if (lst.count() > 1) {
285  QString p = lst[1];
286  if (lst.count() > 2) {
287  QString v = lst[2];
288  if (v.length() >= 8 && v.left(5) == QLatin1String("HTTP/") && v[5].isDigit()
289  && v[6] == QLatin1Char('.') && v[7].isDigit()) {
290  method = m;
291  path = p;
292  }
293  }
294  }
295  }
296  // NOLINTEND
297 
298  if (method == QLatin1String("GET")) {
299  socket->write(help.loadResource(path));
300  socket->close();
301  if (socket->state() == QTcpSocket::UnconnectedState) {
302  // mark the socket for deletion but do not destroy immediately
303  socket->deleteLater();
304  }
305  }
306  }
307 }

340 void StdCmdPythonHelp::activated(int iMsg)
341 {
342  Q_UNUSED(iMsg);
343  // try to open a connection over this port
344  const qint16 port = 7465;
345  if (!this->server) {
346  this->server = new HttpServer();
347  }
348 
349  // if server is not yet running try to open one
350  if (this->server->isListening()
351  || this->server->listen(QHostAddress(QHostAddress::LocalHost), port)) {
352  std::string url = "http://localhost:";
353  url += std::to_string(port);
354  OpenURLInBrowser(url.c_str());
355  }
356  else {
357  QMessageBox::critical(Gui::getMainWindow(),
358  QObject::tr("No Server"),
359  QObject::tr("Unable to start the server to port %1: %2.")
360  .arg(port)
361  .arg(server->errorString()));
362  }
363 }

365 bool Gui::OpenURLInBrowser(const char* URL)
366 {
367  // The webbrowser Python module allows one to start the system browser in an OS-independent way
368  Base::PyGILStateLocker lock;
369  try {
370  PyObject* module = PyImport_ImportModule("webbrowser");
371  if (module) {
372  Py::Module mod(module, true);
373  mod.callMemberFunction("open", Py::TupleN(Py::String(URL)));
374  return true;
375  }
376 
377  throw Py::Exception();
378  }
379  catch (Py::Exception& e) {
380  e.clear();
381  QMessageBox::critical(Gui::getMainWindow(),
382  QObject::tr("No Browser"),
383  QObject::tr("Unable to open your system browser."));
384  return false;
385  }
386 }
