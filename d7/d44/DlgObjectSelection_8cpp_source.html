64 static bool inline setCheckState(QTreeWidgetItem *item, Qt::CheckState state, bool forced=true)
65 {
66  if (!forced) {
67  if (item->isSelected()) {
68  if (state == Qt::Unchecked || item->checkState(0) == Qt::Unchecked)
69  return false;
70  }
71  if (item->checkState(0) == state)
72  return false;
73  }
74  // auto objT = qvariant_cast<App::SubObjectT>(item->data(0, Qt::UserRole));
75  // FC_MSG(objT.getObjectFullName() << (state == Qt::Unchecked ? " unchecked" :
76  // (state == Qt::Checked ? " checked" : " partial")));
77  item->setCheckState(0, state);
78  return true;
79 }

242 void DlgObjectSelection::updateAllItemState()
243 {
244  int count = 0;
245  for (const auto &v : itemMap) {
246  auto state = v.second[0]->checkState(0);
247  if (state == Qt::Unchecked) {
248  if (count) {
249  allItem->setCheckState(0, Qt::PartiallyChecked);
250  return;
251  }
252  } else {
253  if (state == Qt::PartiallyChecked) {
254  allItem->setCheckState(0, Qt::PartiallyChecked);
255  return;
256  }
257  ++count;
258  }
259  }
260  if (count && count == (int)itemMap.size())
261  allItem->setCheckState(0, Qt::Checked);
262  else if (!count)
263  allItem->setCheckState(0, Qt::Unchecked);
264 }

266 void DlgObjectSelection::setItemState(App::DocumentObject *obj,
267  Qt::CheckState state,
268  bool forced)
269 {
270  std::vector<QTreeWidgetItem*> *items = nullptr;
271  auto item = getItem(obj, &items);
272  if (!setCheckState(item, state, forced))
273  return;
274 
275  for (size_t i=1; i<items->size(); ++i)
276  setCheckState(items->at(i), state, true);
277 
278  std::vector<App::DocumentObject*> objs = {obj};
279 
280  if (ui->checkBoxAutoDeps->isChecked() && state == Qt::Checked) {
281  // If an object is newly checked, check all its dependencies
282  for (auto o : obj->getOutListRecursive()) {
283  if (!depSet.count(o) || itemChanged.count(o))
284  continue;
285  auto itItem = itemMap.find(o);
286  if (itItem == itemMap.end() || itItem->second[0]->checkState(0) == state)
287  continue;
288 
289  for (auto i : itItem->second)
290  setCheckState(i, state, true);
291  objs.push_back(o);
292  }
293  }
294 
295  for(auto obj : objs) {
296  auto it = inMap.find(obj);
297  if (it != inMap.end())
298  setCheckState(it->second, state);
299 
300  auto itDep = depMap.find(obj);
301  if (itDep != depMap.end())
302  setCheckState(itDep->second, state);
303 
304  // If an object toggles state, we need to revisit all its in-list
305  // object to update the partial/full checked state.
306  for (auto o : obj->getInList()) {
307  if (!depSet.count(o) ||itemChanged.count(o))
308  continue;
309  auto it = itemMap.find(o);
310  if (it == itemMap.end() || it->second[0]->checkState(0) == state)
311  continue;
312  int count = 0;
313  int selcount = 0;
314  for (auto sibling : o->getOutList()) {
315  if (!depSet.count(sibling))
316  continue;
317  ++count;
318  auto it = itemMap.find(sibling);
319  if (it == itemMap.end())
320  continue;
321  auto s = it->second[0]->checkState(0);
322  if (s == Qt::Unchecked)
323  continue;
324  if (it->second[0]->checkState(0) == Qt::PartiallyChecked) {
325  selcount = -1;
326  break;
327  }
328  ++selcount;
329  }
330  auto state = it->second[0]->checkState(0);
331  if (state == Qt::Checked && selcount != count)
332  setItemState(o, Qt::PartiallyChecked, true);
333  else if (state == Qt::PartiallyChecked && selcount == count)
334  setItemState(o, Qt::Checked, true);
335  }
336  }
337 }

339 std::vector<App::DocumentObject*> DlgObjectSelection::getSelections(SelectionOptions options) const {
340 
341  if (returnOriginals)
342  return initSels;
343 
344  std::vector<App::DocumentObject*> res;
345  Base::Flags<SelectionOptions> flags(options);
346  if (!flags.testFlag(SelectionOptions::Invert)) {
347  for (const auto &v : itemMap) {
348  if (v.second[0]->checkState(0) == Qt::Unchecked)
349  continue;
350  if (auto obj = v.first.getObject())
351  res.push_back(obj);
352  }
353  } else {
354  for (auto obj : deps) {
355  auto it = itemMap.find(obj);
356  if (it == itemMap.end() || it->second[0]->checkState(0) == Qt::Unchecked)
357  res.push_back(obj);
358  }
359  }
360  if (flags.testFlag(SelectionOptions::Sort))
361  std::sort(res.begin(), res.end());
362  return res;
363 }

365 void DlgObjectSelection::onDepItemChanged(QTreeWidgetItem * depItem, int column) {
366  if(column)
367  return;
368  QSignalBlocker blocker(ui->depList);
369  QSignalBlocker blocker2(ui->inList);
370  QSignalBlocker blocker3(ui->treeWidget);
371  auto state = depItem->checkState(0);
372  if (depItem->isSelected()) {
373  const auto items = depItem->treeWidget()->selectedItems();
374  for (auto item : items) {
375  auto objT = qvariant_cast<App::SubObjectT>(item->data(0, Qt::UserRole));
376  auto it = itemMap.find(objT);
377  if (it == itemMap.end())
378  continue;
379  setCheckState(item, state);
380  for (auto i : it->second)
381  setCheckState(i, state);
382  itemChanged[objT] = state;
383  }
384  } else {
385  auto objT = qvariant_cast<App::SubObjectT>(depItem->data(0, Qt::UserRole));
386  auto it = itemMap.find(objT);
387  if (it != itemMap.end()) {
388  itemChanged[objT] = state;
389  for (auto i : it->second)
390  setCheckState(i, state);
391  }
392  }
393  timer.start(10);
394 }

396 void DlgObjectSelection::onObjItemChanged(QTreeWidgetItem * objItem, int column) {
397  if(column != 0)
398  return;
399 
400  QSignalBlocker blocker3(ui->treeWidget);
401  auto state = objItem->checkState(0);
402  if (objItem == allItem) {
403  if (state == Qt::PartiallyChecked)
404  return;
405  ui->treeWidget->selectionModel()->clearSelection();
406  itemChanged.clear();
407  timer.stop();
408  onItemSelectionChanged();
409  if (state == Qt::Unchecked) {
410  for (const auto &v : itemMap) {
411  for (auto i : v.second)
412  setCheckState(i, Qt::Unchecked);
413  auto it = depMap.find(v.first);
414  if (it != depMap.end())
415  setCheckState(it->second, Qt::Unchecked);
416  it = inMap.find(v.first);
417  if (it != inMap.end())
418  setCheckState(it->second, Qt::Unchecked);
419  }
420  } else {
421  for (auto obj : initSels)
422  setCheckState(getItem(obj), Qt::Checked);
423  for (auto obj : deps) {
424  setCheckState(getItem(obj), Qt::Checked);
425  auto it = depMap.find(obj);
426  if (it != depMap.end())
427  setCheckState(it->second, Qt::Checked);
428  it = inMap.find(obj);
429  if (it != inMap.end())
430  setCheckState(it->second, Qt::Checked);
431  }
432  }
433  return;
434  }
435 
436  if (!objItem->isSelected()) {
437  ui->treeWidget->selectionModel()->clearSelection();
438  objItem->setSelected(true);
439  // We treat selected item in tree widget specially in case of checking
440  // items in depList or inList. To simplify logic, we change selection
441  // here if an unselected item has been checked.
442  itemChanged[qvariant_cast<App::SubObjectT>(objItem->data(0, Qt::UserRole))] = state;
443  onItemSelectionChanged();
444  }
445  else {
446  const auto items = ui->treeWidget->selectedItems();
447  for (auto item : items) {
448  setCheckState(item, state);
449  itemChanged[qvariant_cast<App::SubObjectT>(item->data(0, Qt::UserRole))] = state;
450  }
451  }
452  timer.start(10);
453 }

456 static bool getOutList(App::DocumentObject *obj,
457  std::set<App::DocumentObject*> &visited,
458  std::vector<App::DocumentObject*> &result)
459 {
460  if (!visited.insert(obj).second)
461  return false;
462 
463  for (auto o : obj->getOutList()) {
464  if (getOutList(o, visited, result))
465  result.push_back(o);
466  }
467  return true;

470 void DlgObjectSelection::checkItemChanged() {
471 
472  QSignalBlocker blocker(ui->depList);
473  QSignalBlocker blocker2(ui->inList);
474  QSignalBlocker blocker3(ui->treeWidget);
475 
476  std::set<App::DocumentObject*> unchecked;
477 
478  for (const auto &v : itemChanged) {
479  const auto &objT = v.first;
480  Qt::CheckState state = v.second;
481  if (auto obj = objT.getObject()) {
482  if (state == Qt::Unchecked) {
483  // We'll deal with unchecked item later
484  if (ui->checkBoxAutoDeps->isChecked())
485  unchecked.insert(obj);
486  } else {
487  // For checked item, setItemState will auto select its
488  // dependency
489  setItemState(obj, state, true);
490  }
491  }
492  }
493 
494  if (!unchecked.empty()) {
495  // When some item is unchecked by the user, we need to re-check the
496  // recursive outlist of the initially selected object, excluding all
497  // currently unchecked object. And then uncheck any item that does not
498  // appear in the returned outlist.
499 
500  for (const auto &v : itemMap) {
501  auto item = v.second[0];
502  if (item->checkState(0) == Qt::Unchecked) {
503  if (auto obj = qvariant_cast<App::SubObjectT>(item->data(0, Qt::UserRole)).getObject())
504  unchecked.insert(obj);
505  }
506  }
507 
508  auto outlist = initSels;
509  for (auto obj : initSels)
510  getOutList(obj, unchecked, outlist);
511  std::sort(outlist.begin(), outlist.end());
512 
513  for (const auto &v : itemMap) {
514  if (itemChanged.count(v.first) == 0 && v.second[0]->checkState(0) == Qt::Unchecked)
515  continue;
516  if (auto obj = v.first.getObject()) {
517  if (!std::binary_search(outlist.begin(), outlist.end(), obj))
518  setItemState(obj, Qt::Unchecked, true);
519  }
520  }
521  }
522 
523  itemChanged.clear();
524  updateAllItemState();
525 }

555 void DlgObjectSelection::onItemSelectionChanged() {
556  ui->depList->clear();
557  depMap.clear();
558  ui->inList->clear();
559  inMap.clear();
560 
561  std::vector<App::DocumentObject *> sels;
562  const auto items = ui->treeWidget->selectedItems();
563  for (auto item : items) {
564  if (item == allItem) {
565  sels.clear();
566  break;
567  }
568  auto obj = qvariant_cast<App::SubObjectT>(item->data(0, Qt::UserRole)).getObject();
569  if (obj)
570  sels.push_back(obj);
571  }
572 
573  std::vector<App::DocumentObject*> _deps;
574  if (!sels.empty()) {
575  std::sort(sels.begin(), sels.end());
576  for (auto dep : App::Document::getDependencyList(sels, App::Document::DepSort)) {
577  if (!std::binary_search(sels.begin(), sels.end(), dep))
578  _deps.push_back(dep);
579  }
580  }
581 
582  bool enabled = ui->depList->isSortingEnabled();
583  if (enabled)
584  ui->depList->setSortingEnabled(false);
585 
586  bool enabled2 = ui->inList->isSortingEnabled();
587  if (enabled2)
588  ui->inList->setSortingEnabled(false);
589 
590  {
591  QSignalBlocker blocker(ui->depList);
592  auto &objs = !sels.empty() ? _deps : deps;
593  for (auto it = objs.rbegin(); it != objs.rend(); ++it)
594  createDepItem(ui->depList, *it);
595  }
596 
597  std::set<App::DocumentObject*> inlist;
598  for (auto obj : sels)
599  obj->getInListEx(inlist, true);
600  for (auto it = inlist.begin(); it != inlist.end();) {
601  if (!depSet.count(*it) || std::binary_search(sels.begin(), sels.end(), *it))
602  it = inlist.erase(it);
603  else
604  ++it;
605  }
606  {
607  QSignalBlocker blocker2(ui->inList);
608  for (auto obj : inlist)
609  createDepItem(ui->inList, obj);
610  }
611 
612  if (enabled)
613  ui->depList->setSortingEnabled(true);
614  if (enabled2)
615  ui->inList->setSortingEnabled(true);
616 }
