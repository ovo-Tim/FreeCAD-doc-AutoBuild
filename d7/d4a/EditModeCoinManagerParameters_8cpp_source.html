32 int GeometryLayerParameters::getSubLayerIndex(const int geoId,
33  const Sketcher::GeometryFacade* geom) const
34 {
35  bool isConstruction = geom->getConstruction();
36  bool isInternal = geom->isInternalAligned();
37  bool isExternal = geoId <= Sketcher::GeoEnum::RefExt;
38  if (isExternal) {
39  auto egf = Sketcher::ExternalGeometryFacade::getFacade(geom->clone());
40  if (egf->testFlag(Sketcher::ExternalGeometryExtension::Defining)) {
41  // Defining external are added to the Normal sublayers because they
42  // share the same line style.
43  return static_cast<int>(SubLayer::Normal);
44  }
45  }
46 
47  return static_cast<int>(isExternal ? SubLayer::External
48  : isInternal ? SubLayer::Internal
49  : isConstruction ? SubLayer::Construction
50  : SubLayer::Normal);
51 }
