138 // Provide the ability to determine how big to draw constraint arrows etc.
139 // Try to get symbol size equal to 1/5 of the characteristic length of
140 // the object. Typical symbol size is 5, so use 1/25 of the characteristic length.
141 double Constraint::calcSizeFactor(double characLen) const
142 {
143  double l = characLen / 25.0;
144  l = ((round(l)) > 1) ? round(l) : l;
145  return (l > Precision::Confusion() ? l : 1);
146 }

155 void Constraint::onChanged(const App::Property* prop)
156 {
157  if (prop == &References) {
158  // If References are changed, recalculate the normal direction. If no useful reference is
159  // found, use z axis or previous value. If several faces are selected, only the first one is
160  // used
161  std::vector<App::DocumentObject*> Objects = References.getValues();
162  std::vector<std::string> SubElements = References.getSubValues();
163 
164  // Extract geometry from References
165  TopoDS_Shape sh;
166 
167  bool execute = this->isRecomputing();
168  for (std::size_t i = 0; i < Objects.size(); i++) {
169  App::DocumentObject* obj = Objects[i];
170  Part::Feature* feat = static_cast<Part::Feature*>(obj);
171  sh = Tools::getFeatureSubShape(feat, SubElements[i].c_str(), !execute);
172  if (!sh.IsNull() && sh.ShapeType() == TopAbs_FACE) {
173  // Get face normal in center point
174  TopoDS_Face face = TopoDS::Face(sh);
175  BRepGProp_Face props(face);
176  gp_Vec normal;
177  gp_Pnt center;
178  double u1, u2, v1, v2;
179  props.Bounds(u1, u2, v1, v2);
180  props.Normal((u1 + u2) / 2.0, (v1 + v2) / 2.0, center, normal);
181  normal.Normalize();
182  NormalDirection.setValue(normal.X(), normal.Y(), normal.Z());
183  // One face is enough...
184  break;
185  }
186  }
187 
188  std::vector<Base::Vector3d> points;
189  std::vector<Base::Vector3d> normals;
190  if (getPoints(points, normals, &sizeFactor)) {
191  Points.setValues(points);
192  Normals.setValues(normals);
193  Points.touch();
194  }
195  }
196 
197  App::DocumentObject::onChanged(prop);
198 }

251 bool Constraint::getPoints(std::vector<Base::Vector3d>& points,
252  std::vector<Base::Vector3d>& normals,
253  double* scale) const
254 {
255  std::vector<App::DocumentObject*> Objects = References.getValues();
256  std::vector<std::string> SubElements = References.getSubValues();
257 
258  // Extract geometry from References
259  TopoDS_Shape sh;
260 
261  for (std::size_t i = 0; i < Objects.size(); i++) {
262  Part::Feature* feat = static_cast<Part::Feature*>(Objects[i]);
263  sh = Tools::getFeatureSubShape(feat, SubElements[i].c_str(), true);
264  if (sh.IsNull()) {
265  return false;
266  }
267 
268  // Scale by bounding box of the object
269  Bnd_Box box;
270  BRepBndLib::Add(feat->Shape.getShape().getShape(), box);
271  double l = sqrt(box.SquareExtent() / 3.0);
272  *scale = this->calcSizeFactor(l);
273 
274  if (sh.ShapeType() == TopAbs_VERTEX) {
275  const TopoDS_Vertex& vertex = TopoDS::Vertex(sh);
276  gp_Pnt p = BRep_Tool::Pnt(vertex);
277  points.emplace_back(p.X(), p.Y(), p.Z());
278  normals.push_back(NormalDirection.getValue());
279  }
280  else if (sh.ShapeType() == TopAbs_EDGE) {
281  BRepAdaptor_Curve curve(TopoDS::Edge(sh));
282  double fp = curve.FirstParameter();
283  double lp = curve.LastParameter();
284  // Create points with 10 units distance, but at least one at the beginning and end of
285  // the edge
286  int steps;
287  // OvG: Increase 10 units distance proportionately to l for larger objects.
288  if (l >= 30) {
289  steps = static_cast<int>(round(l / (10 * (*scale))));
290  steps = steps < 3 ? 3 : steps;
291  }
292  else if (l >= 20) {
293  steps = static_cast<int>(round(l / 10));
294  }
295  else {
296  steps = 1;
297  }
298 
299  // OvG: Place upper limit on number of steps
300  steps = steps > CONSTRAINTSTEPLIMIT ? CONSTRAINTSTEPLIMIT : steps;
301  double step = (lp - fp) / steps;
302  for (int i = 0; i < steps + 1; i++) {
303  // Parameter values must be in the range [fp, lp] (#0003683)
304  gp_Pnt p = curve.Value(fp + i * step);
305  points.emplace_back(p.X(), p.Y(), p.Z());
306  normals.push_back(NormalDirection.getValue());
307  }
308  }
309  else if (sh.ShapeType() == TopAbs_FACE) {
310  TopoDS_Face face = TopoDS::Face(sh);
311 
312  // Surface boundaries
313  BRepAdaptor_Surface surface(face);
314  double ufp = surface.FirstUParameter();
315  double ulp = surface.LastUParameter();
316  double vfp = surface.FirstVParameter();
317  double vlp = surface.LastVParameter();
318  double l;
319  double lv, lu;
320 
321  // Surface normals
322  BRepGProp_Face props(face);
323  gp_Vec normal;
324  gp_Pnt center;
325 
326  // Get an estimate for the number of arrows by finding the average length of curves
327  Handle(Adaptor3d_HSurface) hsurf;
328  hsurf = new BRepAdaptor_HSurface(surface);
329 
330  Adaptor3d_IsoCurve isoc(hsurf);
331  try {
332  isoc.Load(GeomAbs_IsoU, ufp);
333  l = GCPnts_AbscissaPoint::Length(isoc, Precision::Confusion());
334  }
335  catch (const Standard_Failure&) {
336  gp_Pnt p1 = hsurf->Value(ufp, vfp);
337  gp_Pnt p2 = hsurf->Value(ufp, vlp);
338  l = p1.Distance(p2);
339  }
340 
341  try {
342  isoc.Load(GeomAbs_IsoU, ulp);
343  lv = (l + GCPnts_AbscissaPoint::Length(isoc, Precision::Confusion())) / 2.0;
344  }
345  catch (const Standard_Failure&) {
346  gp_Pnt p1 = hsurf->Value(ulp, vfp);
347  gp_Pnt p2 = hsurf->Value(ulp, vlp);
348  lv = (l + p1.Distance(p2)) / 2.0;
349  }
350 
351  try {
352  isoc.Load(GeomAbs_IsoV, vfp);
353  l = GCPnts_AbscissaPoint::Length(isoc, Precision::Confusion());
354  }
355  catch (const Standard_Failure&) {
356  gp_Pnt p1 = hsurf->Value(ufp, vfp);
357  gp_Pnt p2 = hsurf->Value(ulp, vfp);
358  l = p1.Distance(p2);
359  }
360 
361  try {
362  isoc.Load(GeomAbs_IsoV, vlp);
363  lu = (l + GCPnts_AbscissaPoint::Length(isoc, Precision::Confusion())) / 2.0;
364  }
365  catch (const Standard_Failure&) {
366  gp_Pnt p1 = hsurf->Value(ufp, vlp);
367  gp_Pnt p2 = hsurf->Value(ulp, vlp);
368  lu = (l + p1.Distance(p2)) / 2.0;
369  }
370 
371  // OvG: Increase 10 units distance proportionately to lv for larger objects.
372  int stepsv;
373  if (lv >= 30) {
374  stepsv = static_cast<int>(round(lv / (10 * (*scale))));
375  stepsv = stepsv < 3 ? 3 : stepsv;
376  }
377  else if (lv >= 20.0) {
378  stepsv = static_cast<int>(round(lv / 10));
379  }
380  else {
381  // Minimum of three arrows to ensure (as much as possible) that at
382  // least one is displayed
383  stepsv = 2;
384  }
385 
386  // OvG: Place upper limit on number of steps
387  stepsv = stepsv > CONSTRAINTSTEPLIMIT ? CONSTRAINTSTEPLIMIT : stepsv;
388  int stepsu;
389  // OvG: Increase 10 units distance proportionately to lu for larger objects.
390  if (lu >= 30) {
391  stepsu = static_cast<int>(round(lu / (10 * (*scale))));
392  stepsu = stepsu < 3 ? 3 : stepsu;
393  }
394  else if (lu >= 20.0) {
395  stepsu = static_cast<int>(round(lu / 10));
396  }
397  else {
398  stepsu = 2;
399  }
400 
401  // OvG: Place upper limit on number of steps
402  stepsu = stepsu > CONSTRAINTSTEPLIMIT ? CONSTRAINTSTEPLIMIT : stepsu;
403  double stepv = (vlp - vfp) / stepsv;
404  double stepu = (ulp - ufp) / stepsu;
405 
406  // Create points and normals
407  auto fillPointsAndNormals = [&](Standard_Real u, Standard_Real v) {
408  gp_Pnt p = surface.Value(u, v);
409  BRepClass_FaceClassifier classifier(face, p, Precision::Confusion());
410  if (classifier.State() != TopAbs_OUT) {
411  points.emplace_back(p.X(), p.Y(), p.Z());
412  props.Normal(u, v, center, normal);
413  if (normal.SquareMagnitude() > 0.0) {
414  normal.Normalize();
415  }
416  normals.emplace_back(normal.X(), normal.Y(), normal.Z());
417  }
418  };
419 
420  size_t prevSize = points.size();
421  for (int i = 0; i < stepsv + 1; i++) {
422  for (int j = 0; j < stepsu + 1; j++) {
423  double v = vfp + i * stepv;
424  double u = ufp + j * stepu;
425  fillPointsAndNormals(u, v);
426  }
427  }
428 
429  // it could happen that on a trimmed surface the steps on the iso-curves
430  // are outside the surface, so no points are added.
431  // In that case use points on the outer wire.
432  // https://github.com/FreeCAD/FreeCAD/issues/6073
433  if (prevSize == points.size()) {
434  BRepAdaptor_CompCurve compCurve(BRepTools::OuterWire(face), Standard_True);
435  GProp_GProps linProps;
436  BRepGProp::LinearProperties(compCurve.Wire(), linProps);
437  double outWireLength = linProps.Mass();
438  int stepWire = stepsu + stepsv;
439  // apply subshape transformation to the geometry
440  gp_Trsf faceTrans = face.Location().Transformation();
441  Handle(Geom_Geometry) transGeo =
442  surface.Surface().Surface()->Transformed(faceTrans);
443  ShapeAnalysis_Surface surfAnalysis(Handle(Geom_Surface)::DownCast(transGeo));
444  for (int i = 0; i < stepWire; ++i) {
445  gp_Pnt p = compCurve.Value(outWireLength * i / stepWire);
446  gp_Pnt2d pUV = surfAnalysis.ValueOfUV(p, Precision::Confusion());
447  fillPointsAndNormals(pUV.X(), pUV.Y());
448  }
449  }
450  }
451  }
452 
453  return true;
454 }

456 Base::Vector3d Constraint::getBasePoint(const Base::Vector3d& base,
457  const Base::Vector3d& axis,
458  const App::PropertyLinkSub& location,
459  const double& dist)
460 {
461  // Get the point specified by Location and Distance
462  App::DocumentObject* objLoc = location.getValue();
463  std::vector<std::string> names = location.getSubValues();
464  if (names.empty()) {
465  return Base::Vector3d(0, 0, 0);
466  }
467  std::string subName = names.front();
468  Part::Feature* featLoc = static_cast<Part::Feature*>(objLoc);
469  TopoDS_Shape shloc = featLoc->Shape.getShape().getSubShape(subName.c_str());
470 
471  // Get a plane from the Location reference
472  gp_Pln plane;
473  gp_Dir cylaxis(axis.x, axis.y, axis.z);
474  if (shloc.ShapeType() == TopAbs_FACE) {
475  BRepAdaptor_Surface surface(TopoDS::Face(shloc));
476  plane = surface.Plane();
477  }
478  else {
479  BRepAdaptor_Curve curve(TopoDS::Edge(shloc));
480  gp_Lin line = curve.Line();
481  gp_Dir tang = line.Direction().Crossed(cylaxis);
482  gp_Dir norm = line.Direction().Crossed(tang);
483  plane = gp_Pln(line.Location(), norm);
484  }
485 
486  // Translate the plane in direction of the cylinder (for positive values of Distance)
487  Handle(Geom_Plane) pln = new Geom_Plane(plane);
488  gp_Pnt cylbase(base.x, base.y, base.z);
489  GeomAPI_ProjectPointOnSurf proj(cylbase, pln);
490  if (!proj.IsDone()) {
491  return Base::Vector3d(0, 0, 0);
492  }
493 
494  gp_Pnt projPnt = proj.NearestPoint();
495  if ((fabs(dist) > Precision::Confusion())
496  && (projPnt.IsEqual(cylbase, Precision::Confusion()) == Standard_False)) {
497  plane.Translate(gp_Vec(projPnt, cylbase).Normalized().Multiplied(dist));
498  }
499  Handle(Geom_Plane) plnt = new Geom_Plane(plane);
500 
501  // Intersect translated plane with cylinder axis
502  Handle(Geom_Curve) crv = new Geom_Line(cylbase, cylaxis);
503  GeomAPI_IntCS intersector(crv, plnt);
504  if (!intersector.IsDone()) {
505  return Base::Vector3d(0, 0, 0);
506  }
507  gp_Pnt inter = intersector.Point(1);
508  return Base::Vector3d(inter.X(), inter.Y(), inter.Z());
509 }

511 const Base::Vector3d Constraint::getDirection(const App::PropertyLinkSub& direction)
512 {
513  App::DocumentObject* obj = direction.getValue();
514  if (!obj) {
515  return Base::Vector3d(0, 0, 0);
516  }
517 
518  if (obj->isDerivedFrom<App::Line>()) {
519  Base::Vector3d vec = static_cast<App::Line*>(obj)->getDirection();
520  return vec;
521  }
522 
523  if (obj->isDerivedFrom<App::Plane>()) {
524  Base::Vector3d vec = static_cast<App::Plane*>(obj)->getDirection();
525  return vec;
526  }
527 
528  if (!obj->isDerivedFrom<Part::Feature>()) {
529  std::stringstream str;
530  str << "Type is not a line, plane or Part object";
531  throw Base::TypeError(str.str());
532  }
533 
534  std::vector<std::string> names = direction.getSubValues();
535  if (names.empty()) {
536  return Base::Vector3d(0, 0, 0);
537  }
538  std::string subName = names.front();
539  Part::Feature* feat = static_cast<Part::Feature*>(obj);
540  const Part::TopoShape& shape = feat->Shape.getShape();
541  if (shape.isNull()) {
542  return Base::Vector3d(0, 0, 0);
543  }
544  TopoDS_Shape sh;
545  try {
546  sh = shape.getSubShape(subName.c_str());
547  }
548  catch (Standard_Failure&) {
549  std::stringstream str;
550  str << "No such sub-element '" << subName << "'";
551  throw Base::AttributeError(str.str());
552  }
553 
554  return Fem::Tools::getDirectionFromShape(sh);
555 }
