45 class SplittingSelection: public Gui::SelectionFilterGate
46 {
47  App::DocumentObject* object;
48 
49 public:
50  explicit SplittingSelection(App::DocumentObject* obj)
51  : Gui::SelectionFilterGate(nullPointer())
52  , object(obj)
53  {}
54 
55  bool allow(App::Document* /*pDoc*/, App::DocumentObject* pObj, const char* sSubName) override
56  {
57  if (pObj != this->object) {
58  return false;
59  }
60  if (!sSubName || sSubName[0] == '\0') {
61  return false;
62  }
63  std::string element(sSubName);
64  if (element.substr(0, 4) == "Edge") {
65  int GeoId = std::atoi(element.substr(4, 4000).c_str()) - 1;
66  Sketcher::SketchObject* Sketch = static_cast<Sketcher::SketchObject*>(object);
67  const Part::Geometry* geom = Sketch->getGeometry(GeoId);
68  if (geom->is<Part::GeomLineSegment>() || geom->is<Part::GeomCircle>()
69  || geom->is<Part::GeomEllipse>()
70  || geom->isDerivedFrom(Part::GeomArcOfConic::getClassTypeId())
71  || geom->is<Part::GeomBSplineCurve>()) {
72  return true;
73  }
74  }
75  else if (element.substr(0, 6) == "Vertex") {
76  int VertId = std::atoi(element.substr(6, 4000).c_str()) - 1;
77  int GeoId = Sketcher::GeoEnum::GeoUndef;
78  Sketcher::PointPos PosId = Sketcher::PointPos::none;
79  Sketcher::SketchObject* Sketch = static_cast<Sketcher::SketchObject*>(object);
80  Sketch->getGeoVertexIndex(VertId, GeoId, PosId);
81  if (isBsplineKnot(Sketch, GeoId)) {
82  return true;
83  }
84  }
85  return false;
86  }
87 };

90 class DrawSketchHandlerSplitting: public DrawSketchHandler
91 {
92 public:
93  DrawSketchHandlerSplitting() = default;
94  ~DrawSketchHandlerSplitting() override
95  {
96  Gui::Selection().rmvSelectionGate();
97  }
98 
99  void mouseMove(Base::Vector2d onSketchPos) override
100  {
101  Q_UNUSED(onSketchPos);
102  }
103 
104  bool pressButton(Base::Vector2d onSketchPos) override
105  {
106  Q_UNUSED(onSketchPos);
107  return true;
108  }
109 
110  bool releaseButton(Base::Vector2d onSketchPos) override
111  {
112  int GeoId = Sketcher::GeoEnum::GeoUndef;
113 
114  int curveGeoId = getPreselectCurve();
115  if (curveGeoId >= 0) {
116  const Part::Geometry* geom = sketchgui->getSketchObject()->getGeometry(curveGeoId);
117  if (geom->is<Part::GeomLineSegment>() || geom->is<Part::GeomCircle>()
118  || geom->is<Part::GeomEllipse>()
119  || geom->isDerivedFrom(Part::GeomArcOfConic::getClassTypeId())
120  || geom->is<Part::GeomBSplineCurve>()) {
121  GeoId = curveGeoId;
122  }
123  }
124  else {
125  // No curve of interest is pre-selected. Try pre-selected point.
126  int pointGeoId = getPreselectPoint();
127 
128  if (pointGeoId >= 0) {
129  // TODO: This has to be a knot. Find the spline.
130 
131  const auto& constraints = getSketchObject()->Constraints.getValues();
132  const auto& conIt =
133  std::find_if(constraints.begin(), constraints.end(), [pointGeoId](auto constr) {
134  return (constr->Type == Sketcher::InternalAlignment
135  && constr->AlignmentType == Sketcher::BSplineKnotPoint
136  && constr->First == pointGeoId);
137  });
138 
139  if (conIt != constraints.end()) {
140  GeoId = (*conIt)->Second;
141  }
142  }
143  }
144 
145  if (GeoId >= 0) {
146  try {
147  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Split edge"));
148  Gui::cmdAppObjectArgs(sketchgui->getObject(),
149  "split(%d,App.Vector(%f,%f,0))",
150  GeoId,
151  onSketchPos.x,
152  onSketchPos.y);
153  Gui::Command::commitCommand();
154  tryAutoRecompute(sketchgui->getObject<Sketcher::SketchObject>());
155  }
156  catch (const Base::Exception&) {
157  Gui::NotifyError(sketchgui,
158  QT_TRANSLATE_NOOP("Notifications", "Error"),
159  QT_TRANSLATE_NOOP("Notifications", "Failed to add edge"));
160 
161  Gui::Command::abortCommand();
162  }
163  }
164  else {
165  sketchgui->purgeHandler();
166  }
167 
168  return true;
169  }
170 
171 private:
172  void activated() override
173  {
174  Gui::Selection().clearSelection();
175  Gui::Selection().rmvSelectionGate();
176  Gui::Selection().addSelectionGate(new SplittingSelection(sketchgui->getObject()));
177  }
178 
179  QString getCrosshairCursorSVGName() const override
180  {
181  return QString::fromLatin1("Sketcher_Pointer_Splitting");
182  }
183 };
