59 void MeshAlgos::offsetSpecial2(MeshCore::MeshKernel* Mesh, float fSize)
60 {
61  Base::Builder3D builder;
62  std::vector<Base::Vector3f> PointNormals = Mesh->CalcVertexNormals();
63  std::vector<Base::Vector3f> FaceNormals;
64  std::set<MeshCore::FacetIndex> fliped;
65 
66  MeshFacetIterator it(*Mesh);
67  for (it.Init(); it.More(); it.Next()) {
68  FaceNormals.push_back(it->GetNormal().Normalize());
69  }
70 
71  unsigned int i = 0;
72 
73  // go through all the Vertex normals
74  for (std::vector<Base::Vector3f>::iterator It = PointNormals.begin(); It != PointNormals.end();
75  ++It, i++) {
76  Base::Line3f line {Mesh->GetPoint(i), Mesh->GetPoint(i) + It->Normalize() * fSize};
77  Base::DrawStyle drawStyle;
78  builder.addNode(Base::LineItem {line, drawStyle});
79  // and move each mesh point in the normal direction
80  Mesh->MovePoint(i, It->Normalize() * fSize);
81  }
82  Mesh->RecalcBoundBox();
83 
84  MeshTopoAlgorithm alg(*Mesh);
85 
86  for (int l = 0; l < 1; l++) {
87  for (it.Init(), i = 0; it.More(); it.Next(), i++) {
88  if (it->IsFlag(MeshFacet::INVALID)) {
89  continue;
90  }
91  // calculate the angle between them
92  float angle = acos((FaceNormals[i] * it->GetNormal())
93  / (it->GetNormal().Length() * FaceNormals[i].Length()));
94  if (angle > 1.6) {
95  Base::DrawStyle drawStyle;
96  drawStyle.pointSize = 4.0F;
97  Base::PointItem item {it->GetGravityPoint(),
98  drawStyle,
99  Base::ColorRGB {1.0F, 0.0F, 0.0F}};
100  builder.addNode(item);
101  fliped.insert(it.Position());
102  }
103  }
104 
105  if (fliped.empty()) {
106  break;
107  }
108 
109  for (MeshCore::FacetIndex It : fliped) {
110  alg.CollapseFacet(It);
111  }
112  fliped.clear();
113  }
114 
115  alg.Cleanup();
116 
117  // search for intersected facets
118  MeshCore::MeshEvalSelfIntersection eval(*Mesh);
119  std::vector<std::pair<MeshCore::FacetIndex, MeshCore::FacetIndex>> faces;
120  eval.GetIntersections(faces);
121 
122 
123  builder.saveToLog();
124 }

177 class _VertexCompare
178 {
179 public:
180  bool operator()(const TopoDS_Vertex& rclV1, const TopoDS_Vertex& rclV2) const
181  {
182  if (rclV1.IsSame(rclV2) == Standard_True) {
183  return false;
184  }
185 
186  gp_XYZ clP1 = BRep_Tool::Pnt(rclV1).XYZ();
187  gp_XYZ clP2 = BRep_Tool::Pnt(rclV2).XYZ();
188 
189  if (fabs(clP1.X() - clP2.X()) < dE) {
190  if (fabs(clP1.Y() - clP2.Y()) < dE) {
191  return clP1.Z() < clP2.Z();
192  }
193  else {
194  return clP1.Y() < clP2.Y();
195  }
196  }
197  else {
198  return clP1.X() < clP2.X();
199  }
200  }
201 
202  double dE = 1.0e-5;
203 };

206 void MeshAlgos::LoftOnCurve(MeshCore::MeshKernel& ResultMesh,
207  const TopoDS_Shape& Shape,
208  const std::vector<Base::Vector3f>& poly,
209  const Base::Vector3f& up,
210  float MaxSize)
211 {
212  TopExp_Explorer Ex;
213  Standard_Real fBegin, fEnd;
214  std::vector<MeshGeomFacet> cVAry;
215  std::map<TopoDS_Vertex, std::vector<Base::Vector3f>, _VertexCompare> ConnectMap;
216 
217  for (Ex.Init(Shape, TopAbs_EDGE); Ex.More(); Ex.Next()) {
218  // get the edge and the belonging Vertexes
219  TopoDS_Edge Edge = (TopoDS_Edge&)Ex.Current();
220  TopoDS_Vertex V1, V2;
221  TopExp::Vertices(Edge, V1, V2);
222  bool bBegin = false, bEnd = false;
223  // getting the geometric curve and the interval
224  GeomLProp_CLProps prop(BRep_Tool::Curve(Edge, fBegin, fEnd), 1, 0.0000000001);
225  int res = int((fEnd - fBegin) / MaxSize);
226  // do at least 2 segments
227  if (res < 2) {
228  res = 2;
229  }
230  gp_Dir Tangent;
231 
232  std::vector<Base::Vector3f> prePoint(poly.size());
233  std::vector<Base::Vector3f> actPoint(poly.size());
234 
235  // checking if there is already a end to connect
236  if (ConnectMap.find(V1) != ConnectMap.end()) {
237  bBegin = true;
238  prePoint = ConnectMap[V1];
239  }
240 
241  if (ConnectMap.find(V2) != ConnectMap.end()) {
242  bEnd = true;
243  }
244 
245  for (long i = 0; i < res; i++) {
246 
247  // get point and tangent at the position, up is fix for the moment
248  prop.SetParameter(fBegin + ((fEnd - fBegin) * float(i)) / float(res - 1));
249  prop.Tangent(Tangent);
250  Base::Vector3f Tng((float)Tangent.X(), (float)Tangent.Y(), (float)Tangent.Z());
251  Base::Vector3f Ptn((float)prop.Value().X(),
252  (float)prop.Value().Y(),
253  (float)prop.Value().Z());
254  Base::Vector3f Up(up);
255  // normalize and calc the third vector of the plane coordinatesystem
256  Tng.Normalize();
257  Up.Normalize();
258  Base::Vector3f Third(Tng % Up);
259 
260  unsigned int l = 0;
261  std::vector<Base::Vector3f>::const_iterator It;
262 
263  // got through the profile
264  for (It = poly.begin(); It != poly.end(); ++It, l++) {
265  actPoint[l] = ((Third * It->x) + (Up * It->y) + (Tng * It->z) + Ptn);
266  }
267 
268  if (i == res - 1 && !bEnd) {
269  // remember the last row to connect to a otger edge with the same vertex
270  ConnectMap[V2] = actPoint;
271  }
272 
273  if (i == 1 && bBegin) {
274  // using the end of an other edge as start
275  prePoint = ConnectMap[V1];
276  }
277 
278  if (i == 0 && !bBegin) {
279  // remember the first row for connection to a edge with the same vertex
280  ConnectMap[V1] = actPoint;
281  }
282 
283  if (i) // not the first row or something to connect to
284  {
285  for (l = 0; l < actPoint.size(); l++) {
286  if (l) // not first point in row
287  {
288  if (i == res - 1 && bEnd) { // if last row and a end to connect
289  actPoint = ConnectMap[V2];
290  }
291 
292  Base::Vector3f p1 = prePoint[l - 1], p2 = actPoint[l - 1], p3 = prePoint[l],
293  p4 = actPoint[l];
294 
295  cVAry.emplace_back(p1, p2, p3);
296  cVAry.emplace_back(p3, p2, p4);
297  }
298  }
299  }
300 
301  prePoint = actPoint;
302  }
303  }
304 
305  ResultMesh.AddFacets(cVAry);
306 }
