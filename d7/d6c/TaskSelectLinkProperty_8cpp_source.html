123 void TaskSelectLinkProperty::activate()
124 {
125  // first clear the selection
126  Gui::Selection().clearSelection();
127  // set the gate for the filter
128  Gui::Selection().addSelectionGate(new SelectionFilterGate(Filter));
129 
130  // In case of LinkSub property
131  if (LinkSub) {
132  // save the start values for a cnacel operation (reject())
133  StartValueBuffer = LinkSub->getSubValues();
134  StartObject = LinkSub->getValue();
135  if (StartObject) {
136  std::string ObjName = StartObject->getNameInDocument();
137  std::string DocName = StartObject->getDocument()->getName();
138 
139  for (const auto& it : StartValueBuffer) {
140  Gui::Selection().addSelection(DocName.c_str(), ObjName.c_str(), it.c_str());
141  }
142  }
143  }
144  // In case of LinkList property
145  else if (LinkList) {
146  // save the start values for a cnacel operation (reject())
147  const std::vector<App::DocumentObject*>& Values = LinkList->getValues();
148  for (const auto& Value : Values) {
149  std::string ObjName = Value->getNameInDocument();
150  std::string DocName = Value->getDocument()->getName();
151  Gui::Selection().addSelection(DocName.c_str(), ObjName.c_str());
152  }
153  }
154 
155  checkSelectionStatus();
156 }

182 void TaskSelectLinkProperty::sendSelection2Property()
183 {
184  if (LinkSub) {

186  assert(temp.size() >= 1);
187  LinkSub->setValue(temp[0].getObject(), temp[0].getSubNames());
188  }
189  else if (LinkList) {
190  std::vector<Gui::SelectionObject> sel = Gui::Selection().getSelectionEx();
191  std::vector<App::DocumentObject*> temp;
192  for (auto& it : sel) {
193  temp.push_back(it.getObject());
194  }
195 
196  LinkList->setValues(temp);
