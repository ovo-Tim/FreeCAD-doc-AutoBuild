159 bool SheetView::onMsg(const char* pMsg, const char**)
160 {
161  if (strcmp("Undo", pMsg) == 0) {
162  getGuiDocument()->undo(1);
163  App::Document* doc = getAppDocument();
164  if (doc) {
165  doc->recompute();
166  }
167  return true;
168  }
169  else if (strcmp("Redo", pMsg) == 0) {
170  getGuiDocument()->redo(1);
171  App::Document* doc = getAppDocument();
172  if (doc) {
173  doc->recompute();
174  }
175  return true;
176  }
177  else if (strcmp("Save", pMsg) == 0) {
178  getGuiDocument()->save();
179  return true;
180  }
181  else if (strcmp("SaveAs", pMsg) == 0) {
182  getGuiDocument()->saveAs();
183  return true;
184  }
185  else if (strcmp("Std_Delete", pMsg) == 0) {
186  std::vector<Range> ranges = selectedRanges();
187  if (sheet->hasCell(ranges)) {
188  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Clear cell(s)"));
189  std::vector<Range>::const_iterator i = ranges.begin();
190  for (; i != ranges.end(); ++i) {
191  FCMD_OBJ_CMD(sheet, "clear('" << i->rangeString() << "')");
192  }
193  Gui::Command::commitCommand();
194  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
195  }
196  return true;
197  }
198  else if (strcmp("Cut", pMsg) == 0) {
199  ui->cells->cutSelection();
200  return true;
201  }
202  else if (strcmp("Copy", pMsg) == 0) {
203  ui->cells->copySelection();
204  return true;
205  }
206  else if (strcmp("Paste", pMsg) == 0) {
207  ui->cells->pasteClipboard();
208  return true;
209  }
210  else {
211  return false;
212  }
213 }

432 void SheetView::confirmAliasChanged(const QString& text)
433 {
434  bool aliasOkay = true;
435 
436  ui->cellAlias->setDocumentObject(sheet);
437  if (text.length() != 0 && !sheet->isValidAlias(text.toStdString())) {
438  aliasOkay = false;
439  }
440 
441  QModelIndex i = ui->cells->currentIndex();
442  if (const auto* cell = sheet->getNewCell(CellAddress(i.row(), i.column()))) {
443  if (!aliasOkay) {
444  // do not show error message if failure to set new alias is because it is already the
445  // same string
446  std::string current_alias;
447  (void)cell->getAlias(current_alias);
448  if (text != QString::fromUtf8(current_alias.c_str())) {
449  Base::Console().Error("Unable to set alias: %s\n", text.toStdString().c_str());
450  }
451  }
452  else {
453  std::string address = CellAddress(i.row(), i.column()).toString();
454  Gui::cmdAppObjectArgs(sheet, "setAlias('%s', '%s')", address, text.toStdString());
455  Gui::cmdAppDocument(sheet->getDocument(), "recompute()");
456  ui->cells->setFocus();
457  }
458  }
459 }

469 void SheetView::aliasChanged(const QString& text)
470 {
471  // check live the input and highlight if the user input invalid characters
472 
473  bool aliasOk = true;
474  static auto originalStylesheet = ui->cellAlias->styleSheet();
475  QString warningColor;
476  if (qApp->styleSheet().contains(QLatin1String("dark"), Qt::CaseInsensitive)) {
477  warningColor = QLatin1String("rgb(255,90,90)"); // Light red for dark mode
478  }
479  else {
480  warningColor = QLatin1String("rgb(200,0,0)"); // Dark red for light mode
481  }
482 
483  if (!text.isEmpty() && !sheet->isValidAlias(text.toStdString())) {
484  aliasOk = false;
485  }
486 
487  if (!aliasOk) {
488  // change tooltip and make text color red
489  ui->cellAlias->setToolTip(QObject::tr("Alias contains invalid characters!"));
490  ui->cellAlias->setStyleSheet(QLatin1String("color:") + warningColor);
491  }
492  else {
493  // go back to normal
494  ui->cellAlias->setToolTip(
495  QObject::tr("Refer to cell by alias, for example\nSpreadsheet.my_alias_name instead of "
496  "Spreadsheet.B1"));
497  ui->cellAlias->setStyleSheet(originalStylesheet);
498  }

667  }
