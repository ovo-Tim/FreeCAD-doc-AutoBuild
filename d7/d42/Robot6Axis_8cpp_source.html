108 {

223 bool Robot6Axis::setTo(const Placement& To)
224 {
225  // Creation of the solvers:
226  ChainFkSolverPos_recursive fksolver1(Kinematic); // Forward position solver
227  ChainIkSolverVel_pinv iksolver1v(Kinematic); // Inverse velocity solver
228  ChainIkSolverPos_NR_JL iksolver1(Kinematic,
229  Min,
230  Max,
231  fksolver1,
232  iksolver1v,
233  100,
234  1e-6); // Maximum 100 iterations, stop at accuracy 1e-6
235 
236  // Creation of jntarrays:
237  JntArray result(Kinematic.getNrOfJoints());
238 
239  // Set destination frame
240  Frame F_dest =
241  Frame(KDL::Rotation::Quaternion(To.getRotation()[0],
242  To.getRotation()[1],
243  To.getRotation()[2],
244  To.getRotation()[3]),
245  KDL::Vector(To.getPosition()[0], To.getPosition()[1], To.getPosition()[2]));
246 
247  // solve
248  if (iksolver1.CartToJnt(Actual, F_dest, result) < 0) {
249  return false;
250  }
251  else {
252  Actual = result;
253  Tcp = F_dest;
254  return true;
255  }
256 }

266 bool Robot6Axis::calcTcp()
267 {
268  // Create solver based on kinematic chain
269  ChainFkSolverPos_recursive fksolver = ChainFkSolverPos_recursive(Kinematic);
270 
271  // Create the frame that will contain the results
272  KDL::Frame cartpos;
273 
274  // Calculate forward position kinematics
275  int kinematics_status;
276  kinematics_status = fksolver.JntToCart(Actual, cartpos);
277  if (kinematics_status >= 0) {
278  Tcp = cartpos;
279  return true;
280  }
281  else {
282  return false;
283  }
