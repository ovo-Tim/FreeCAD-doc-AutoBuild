170 void DlgCAMSimulator::GetMeshData(const Part::TopoShape& tshape,
171  float resolution,
172  std::vector<Vertex>& verts,
173  std::vector<GLushort>& indices)
174 {
175  std::vector<int> normalCount;
176  int nVerts = 0;
177  for (auto& shape : tshape.getSubTopoShapes(TopAbs_FACE)) {
178  std::vector<Base::Vector3d> points;
179  std::vector<Data::ComplexGeoData::Facet> facets;
180  shape.getFaces(points, facets, resolution);
181 
182  std::vector<Base::Vector3d> normals(points.size());
183  std::vector<int> normalCount(points.size());
184 
185  // copy triangle indices and calculate normals
186  for (auto face : facets) {
187  indices.push_back(face.I1 + nVerts);
188  indices.push_back(face.I2 + nVerts);
189  indices.push_back(face.I3 + nVerts);
190 
191  // calculate normal
192  Base::Vector3d vAB = points[face.I2] - points[face.I1];
193  Base::Vector3d vAC = points[face.I3] - points[face.I1];
194  Base::Vector3d vNorm = vAB.Cross(vAC).Normalize();
195 
196  normals[face.I1] += vNorm;
197  normals[face.I2] += vNorm;
198  normals[face.I3] += vNorm;
199 
200  normalCount[face.I1]++;
201  normalCount[face.I2]++;
202  normalCount[face.I3]++;
203  }
204 
205  // copy points and set normals
206  for (unsigned int i = 0; i < points.size(); i++) {
207  Base::Vector3d& point = points[i];
208  Base::Vector3d& normal = normals[i];
209  int count = normalCount[i];
210  normal /= count;
211  verts.push_back(Vertex(point.x, point.y, point.z, normal.x, normal.y, normal.z));
212  }
213 
214  nVerts = verts.size();
215  }
216 }

272 void DlgCAMSimulator::renderNow()
273 {
274  static unsigned int lastTime = 0;
275  static int frameCount = 0;
276  static int fps = 0;
277  if (!isExposed()) {
278  return;
279  }
280 
281  checkInitialization();
282 
283  frameCount++;
284  unsigned int curtime = QDateTime::currentMSecsSinceEpoch();
285  unsigned int timediff = curtime - lastTime;
286  if (timediff > 10000) {
287  fps = frameCount * 1000 / timediff; // for debug only. not used otherwise.
288  lastTime = curtime;
289  frameCount = 0;
290  }
291  render();
292  mContext->swapBuffers(this);
293 
294  if (mAnimating) {
295  renderLater();
296  }
297  (void)fps;
298 }

309 DlgCAMSimulator* DlgCAMSimulator::GetInstance()
310 {
311  if (mInstance == nullptr) {
312  QSurfaceFormat format;
313  format.setVersion(4, 1); // Request OpenGL 4.1 - for MacOS
314  format.setProfile(QSurfaceFormat::CoreProfile); // Use the core profile = for MacOS
315  int samples = Gui::View3DInventorViewer::getNumSamples();
316  if (samples > 1) {
317  format.setSamples(samples);
318  }
319  format.setSwapInterval(2);
320  format.setDepthBufferSize(24);
321  format.setStencilBufferSize(8);
322  mInstance = new DlgCAMSimulator();
323  mInstance->setFormat(format);
324  mInstance->resize(MillSim::gWindowSizeW, MillSim::gWindowSizeH);
325  mInstance->setModality(Qt::ApplicationModal);
326  mInstance->setMinimumWidth(700);
327  mInstance->setMinimumHeight(400);
328  }
329  return mInstance;
330 }
