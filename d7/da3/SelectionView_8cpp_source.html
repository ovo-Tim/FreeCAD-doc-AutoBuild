126 void SelectionView::onSelectionChanged(const SelectionChanges& Reason)
127 {
128  ParameterGrp::handle hGrp = App::GetApplication()
129  .GetUserParameter()
130  .GetGroup("BaseApp")
131  ->GetGroup("Preferences")
132  ->GetGroup("Selection");
133  bool autoShow = hGrp->GetBool("AutoShowSelectionView", false);
134  hGrp->SetBool("AutoShowSelectionView",
135  autoShow); // Remove this line once the preferences window item is implemented
136 
137  if (autoShow) {
138  if (!parentWidget()->isVisible() && Selection().hasSelection()) {
139  parentWidget()->show();
140  openedAutomatically = true;
141  }
142  else if (openedAutomatically && !Selection().hasSelection()) {
143  parentWidget()->hide();
144  openedAutomatically = false;
145  }
146  }
147 
148  QString selObject;
149  QTextStream str(&selObject);
150 
151  auto getSelectionName = [](QTextStream& str,
152  const char* docName,
153  const char* objName,
154  const char* subName,
155  App::DocumentObject* obj) {
156  str << docName;
157  str << "#";
158  str << objName;
159  if (subName != 0 && subName[0] != 0) {
160  str << ".";
161  /* Original code doesn't take account of histories in subelement names and displays
162  * them inadvertently. Let's not do that.
163  str << subName;
164  */
165  /* Remove the history from the displayed subelement name */
166  App::ElementNamePair elementName;
167  App::GeoFeature::resolveElement(obj, subName, elementName);
168  str << elementName.oldName.c_str(); // Use the shortened element name not the full one.
169  /* Mark it visually if there was a history as a "tell" for if a given selection has TNP
170  * fixes in it. */
171  if (elementName.newName.size() > 0) {
172  str << " []";
173  }
174  auto subObj = obj->getSubObject(subName);
175  if (subObj) {
176  obj = subObj;
177  }
178  }
179  str << " (";
180  str << QString::fromUtf8(obj->Label.getValue());
181  str << ")";
182  };
183 
184  if (Reason.Type == SelectionChanges::AddSelection) {
185  // save as user data
186  QStringList list;
187  list << QString::fromLatin1(Reason.pDocName);
188  list << QString::fromLatin1(Reason.pObjectName);
189  App::Document* doc = App::GetApplication().getDocument(Reason.pDocName);
190  App::DocumentObject* obj = doc->getObject(Reason.pObjectName);
191  getSelectionName(str, Reason.pDocName, Reason.pObjectName, Reason.pSubName, obj);
192 
193  // insert the selection as item
194  QListWidgetItem* item = new QListWidgetItem(selObject, selectionView);
195  item->setData(Qt::UserRole, list);
196  }
197  else if (Reason.Type == SelectionChanges::ClrSelection) {
198  if (!Reason.pDocName[0]) {
199  // remove all items
200  selectionView->clear();
201  }
202  else {
203  // build name
204  str << Reason.pDocName;
205  str << "#";
206  // remove all items
207  const auto items = selectionView->findItems(selObject, Qt::MatchStartsWith);
208  for (auto item : items) {
209  delete item;
210  }
211  }
212  }
213  else if (Reason.Type == SelectionChanges::RmvSelection) {
214  App::Document* doc = App::GetApplication().getDocument(Reason.pDocName);
215  App::DocumentObject* obj = doc->getObject(Reason.pObjectName);
216  getSelectionName(str, Reason.pDocName, Reason.pObjectName, Reason.pSubName, obj);
217  // remove all items
218  QList<QListWidgetItem*> l = selectionView->findItems(selObject, Qt::MatchStartsWith);
219  if (l.size() == 1) {
220  delete l[0];
221  }
222  }
223  else if (Reason.Type == SelectionChanges::SetSelection) {
224  // remove all items
225  selectionView->clear();
226  std::vector<SelectionSingleton::SelObj> objs =
227  Gui::Selection().getSelection(Reason.pDocName, ResolveMode::NoResolve);
228  for (const auto& it : objs) {
229  // save as user data
230  QStringList list;
231  list << QString::fromLatin1(it.DocName);
232  list << QString::fromLatin1(it.FeatName);
233 
234  App::Document* doc = App::GetApplication().getDocument(it.DocName);
235  App::DocumentObject* obj = doc->getObject(it.FeatName);
236  getSelectionName(str, it.DocName, it.FeatName, it.SubName, obj);
237  QListWidgetItem* item = new QListWidgetItem(selObject, selectionView);
238  item->setData(Qt::UserRole, list);
239  selObject.clear();
240  }
241  }
242  else if (Reason.Type == SelectionChanges::PickedListChanged) {
243  bool picking = Selection().needPickedList();
244  enablePickList->setChecked(picking);
245  pickList->setVisible(picking);
246  pickList->clear();
247  if (picking) {
248  const auto& sels = Selection().getPickedList(Reason.pDocName);
249  for (const auto& sel : sels) {
250  App::Document* doc = App::GetApplication().getDocument(sel.DocName);
251  if (!doc) {
252  continue;
253  }
254  App::DocumentObject* obj = doc->getObject(sel.FeatName);
255  if (!obj) {
256  continue;
257  }
258 
259  QString selObject;
260  QTextStream str(&selObject);
261  getSelectionName(str, sel.DocName, sel.FeatName, sel.SubName, obj);
262 
263  this->x = sel.x;
264  this->y = sel.y;
265  this->z = sel.z;
266 
267  new QListWidgetItem(selObject, pickList);
268  }
269  }
270  }
271 
272  countLabel->setText(QString::number(selectionView->count()));
273 }

275 void SelectionView::search(const QString& text)
276 {
277  if (!text.isEmpty()) {
278  searchList.clear();
279  App::Document* doc = App::GetApplication().getActiveDocument();
280  std::vector<App::DocumentObject*> objects;
281  if (doc) {
282  objects = doc->getObjects();
283  selectionView->clear();
284  for (auto it : objects) {
285  QString label = QString::fromUtf8(it->Label.getValue());
286  if (label.contains(text, Qt::CaseInsensitive)) {
287  searchList.push_back(it);
288  // save as user data
289  QString selObject;
290  QTextStream str(&selObject);
291  QStringList list;
292  list << QString::fromLatin1(doc->getName());
293  list << QString::fromLatin1(it->getNameInDocument());
294  // build name
295  str << QString::fromUtf8(doc->Label.getValue());
296  str << "#";
297  str << it->getNameInDocument();
298  str << " (";
299  str << label;
300  str << ")";
301  QListWidgetItem* item = new QListWidgetItem(selObject, selectionView);
302  item->setData(Qt::UserRole, list);
303  }
304  }
305  countLabel->setText(QString::number(selectionView->count()));
306  }
307  }

373 void SelectionView::toggleSelect(QListWidgetItem* item)
374 {
375  if (!item) {
376  return;
377  }
378  std::string name = item->text().toLatin1().constData();
379  char* docname = &name.at(0);
380  char* objname = std::strchr(docname, '#');
381  if (!objname) {
382  return;
383  }
384  *objname++ = 0;
385  char* subname = std::strchr(objname, '.');
386  if (subname) {
387  *subname++ = 0;
388  char* end = std::strchr(subname, ' ');
389  if (end) {
390  *end = 0;
391  }
392  }
393  else {
394  char* end = std::strchr(objname, ' ');
395  if (end) {
396  *end = 0;
397  }
398  }
399  QString cmd;
400  if (Gui::Selection().isSelected(docname, objname, subname)) {
401  cmd = QString::fromLatin1("Gui.Selection.removeSelection("
402  "App.getDocument('%1').getObject('%2'),'%3')")
403  .arg(QString::fromLatin1(docname),
404  QString::fromLatin1(objname),
405  QString::fromLatin1(subname));
406  }
407  else {
408  cmd = QString::fromLatin1("Gui.Selection.addSelection("
409  "App.getDocument('%1').getObject('%2'),'%3',%4,%5,%6)")
410  .arg(QString::fromLatin1(docname),
411  QString::fromLatin1(objname),
412  QString::fromLatin1(subname))
413  .arg(x)

569 QString SelectionView::getModule(const char* type) const
570 {
571  // go up the inheritance tree and find the module name of the first
572  // sub-class that has not the prefix "App::"
573  std::string prefix;
574  Base::Type typeId = Base::Type::fromName(type);
575 
576  while (!typeId.isBad()) {
577  std::string temp(typeId.getName());
578  std::string::size_type pos = temp.find_first_of("::");
579 
580  std::string module;
581  if (pos != std::string::npos) {
582  module = std::string(temp, 0, pos);
583  }
584  if (module != "App") {
585  prefix = module;
586  }
587  else {
588  break;
589  }
590  typeId = typeId.getParent();
591  }
592 
593  return QString::fromStdString(prefix);
594 }

611 bool SelectionView::supportPart(App::DocumentObject* obj, const QString& part) const
612 {
613  if (obj->isDerivedFrom<App::GeoFeature>()) {
614  App::GeoFeature* geo = static_cast<App::GeoFeature*>(obj);
615  const App::PropertyComplexGeoData* data = geo->getPropertyOfGeometry();
616  if (data) {
617  const Data::ComplexGeoData* geometry = data->getComplexData();
618  std::vector<const char*> types = geometry->getElementTypes();
619  for (auto it : types) {
620  if (part.startsWith(QString::fromLatin1(it))) {
621  return true;
622  }
623  }
624  }
625  }
626 
627  return false;
