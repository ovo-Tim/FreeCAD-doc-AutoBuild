140 App::DocumentObjectExecReturn* DrawViewDetail::execute()
141 {
142  // Base::Console().Message("DVD::execute() - %s\n", getNameInDocument());
143  if (!keepUpdated()) {
144  return DrawView::execute();
145  }
146 
147  App::DocumentObject* baseObj = BaseView.getValue();
148  if (!baseObj) {
149  return DrawView::execute();
150  }
151 
152  if (!baseObj->isDerivedFrom<TechDraw::DrawViewPart>()) {
153  //this can only happen via scripting?
154  return DrawView::execute();
155  }
156 
157  DrawViewPart* dvp = static_cast<DrawViewPart*>(baseObj);
158  TopoDS_Shape shape3d = dvp->getShapeForDetail();
159  DrawViewSection* dvs = nullptr;
160  if (dvp->isDerivedFrom(TechDraw::DrawViewSection::getClassTypeId())) {
161  dvs = static_cast<TechDraw::DrawViewSection*>(dvp);
162  }
163 
164  if (shape3d.IsNull()) {
165  return DrawView::execute();
166  }
167 
168  bool haveX = checkXDirection();
169  if (!haveX) {
170  //block touch/onChanged stuff
171  Base::Vector3d newX = getXDirection();
172  XDirection.setValue(newX);
173  XDirection.purgeTouched();//don't trigger updates!
174  //unblock
175  }
176 
177  detailExec(shape3d, dvp, dvs);
178  addPoints();
179 
180  dvp->requestPaint();//to refresh detail highlight in base view
181  return DrawView::execute();

186 void DrawViewDetail::detailExec(TopoDS_Shape& shape, DrawViewPart* dvp, DrawViewSection* dvs)
187 {
188  if (waitingForHlr() || waitingForDetail()) {
189  return;
190  }
191 
192  //note that &m_detailWatcher in the third parameter is not strictly required, but using the
193  //4 parameter signature instead of the 3 parameter signature prevents clazy warning:
194  //https://github.com/KDE/clazy/blob/1.11/docs/checks/README-connect-3arg-lambda.md
195  connectDetailWatcher =
196  QObject::connect(&m_detailWatcher, &QFutureWatcherBase::finished, &m_detailWatcher,
197  [this] { this->onMakeDetailFinished(); });
198 
199  // We create a lambda closure to hold a copy of shape.
200  // This is important because this variable might be local to the calling
201  // function and might get destructed before the parallel processing finishes.
202  // TODO: What about dvp and dvs? Do they live past makeDetailShape?
203  auto lambda = [this, shape, dvp, dvs]{this->makeDetailShape(shape, dvp, dvs);};
204  m_detailFuture = QtConcurrent::run(std::move(lambda));
205  m_detailWatcher.setFuture(m_detailFuture);
206  waitingForDetail(true);
207 }

209 //this runs in a separate thread since it can sometimes take a long time
210 //make a common of the input shape and a cylinder (or prism depending on
211 //the matting style)
212 void DrawViewDetail::makeDetailShape(const TopoDS_Shape& shape3d, DrawViewPart* dvp, DrawViewSection* dvs)
213 {
214  showProgressMessage(getNameInDocument(), "is making detail shape");
215 
216  Base::Vector3d dirDetail = dvp->Direction.getValue();
217  double radius = getFudgeRadius();
218 
219  //make a copy of the input shape so we don't inadvertently change it
220  BRepBuilderAPI_Copy BuilderCopy(shape3d);
221  TopoDS_Shape copyShape = BuilderCopy.Shape();
222  m_saveShape = copyShape;
223  m_saveDvp = dvp;
224  m_saveDvs = dvs;
225 
226  gp_Pnt gpCenter = ShapeUtils::findCentroid(copyShape, dirDetail);
227  Base::Vector3d shapeCenter = Base::Vector3d(gpCenter.X(), gpCenter.Y(), gpCenter.Z());
228  m_saveCentroid = shapeCenter;//centroid of original shape
229 
230  if (!dvs) {
231  //section cutShape should already be on origin
232  copyShape = ShapeUtils::moveShape(copyShape,//centre shape on origin
233  -shapeCenter);
234  }
235 
236  shapeCenter = Base::Vector3d(0.0, 0.0, 0.0);
237 
238 
239  m_viewAxis = dvp->getProjectionCS(shapeCenter);//save the CS for later
240  Base::Vector3d anchor = AnchorPoint.getValue();//this is a 2D point in base view local coords
241 
242  anchor = DrawUtil::toR3(m_viewAxis, anchor);//actual anchor coords in R3
243 
244  Bnd_Box bbxSource;
245  bbxSource.SetGap(0.0);
246  BRepBndLib::AddOptimal(copyShape, bbxSource);
247  double diag = sqrt(bbxSource.SquareExtent());
248 
249  Base::Vector3d toolPlaneOrigin = anchor + dirDetail * diag * -1.0;//center tool about anchor
250  double extrudeLength = 2.0 * toolPlaneOrigin.Length();
251 
252  gp_Pnt gpnt(toolPlaneOrigin.x, toolPlaneOrigin.y, toolPlaneOrigin.z);
253  gp_Dir gdir(dirDetail.x, dirDetail.y, dirDetail.z);
254 
255  TopoDS_Face extrusionFace;
256  Base::Vector3d extrudeVec = dirDetail * extrudeLength;
257  gp_Vec extrudeDir(extrudeVec.x, extrudeVec.y, extrudeVec.z);
258  TopoDS_Shape tool;
259  if (Preferences::mattingStyle()) {
260  //square mat
261  gp_Pln gpln(gpnt, gdir);
262  BRepBuilderAPI_MakeFace mkFace(gpln, -radius, radius, -radius, radius);
263  extrusionFace = mkFace.Face();
264  if (extrusionFace.IsNull()) {
265  Base::Console().Warning("DVD::makeDetailShape - %s - failed to create tool base face\n",
266  getNameInDocument());
267  return;
268  }
269  tool = BRepPrimAPI_MakePrism(extrusionFace, extrudeDir, false, true).Shape();
270  if (tool.IsNull()) {
271  Base::Console().Warning("DVD::makeDetailShape - %s - failed to create tool (prism)\n",
272  getNameInDocument());
273  return;
274  }
275  }
276  else {
277  //circular mat
278  gp_Ax2 cs(gpnt, gdir);
279  BRepPrimAPI_MakeCylinder mkTool(cs, radius, extrudeLength);
280  tool = mkTool.Shape();
281  if (tool.IsNull()) {
282  Base::Console().Warning("DVD::detailExec - %s - failed to create tool (cylinder)\n",
283  getNameInDocument());
284  return;
285  }
286  }
287 
288  //for each solid and shell in the input shape, make a common with the tool and
289  //add the result to a compound. This avoids issues with some geometry errors in the
290  //input shape.
291  BRep_Builder builder;
292  TopoDS_Compound pieces;
293  builder.MakeCompound(pieces);
294  TopExp_Explorer expl1(copyShape, TopAbs_SOLID);
295  for (; expl1.More(); expl1.Next()) {
296  const TopoDS_Solid& s = TopoDS::Solid(expl1.Current());
297  FCBRepAlgoAPI_Common mkCommon(s, tool);
298  if (!mkCommon.IsDone()) {
299  continue;
300  }
301  if (mkCommon.Shape().IsNull()) {
302  continue;
303  }
304  //Did we get a result?
305  TopExp_Explorer xp;
306  xp.Init(mkCommon.Shape(), TopAbs_SOLID);
307  if (xp.More() != Standard_True) {
308  continue;
309  }
310  builder.Add(pieces, mkCommon.Shape());
311  }
312 
313  TopExp_Explorer expl2(copyShape, TopAbs_SHELL, TopAbs_SOLID);
314  for (; expl2.More(); expl2.Next()) {
315  const TopoDS_Shell& s = TopoDS::Shell(expl2.Current());
316  FCBRepAlgoAPI_Common mkCommon(s, tool);
317  if (!mkCommon.IsDone()) {
318  continue;
319  }
320  if (mkCommon.Shape().IsNull()) {
321  continue;
322  }
323  //Did we get a result?
324  TopExp_Explorer xp;
325  xp.Init(mkCommon.Shape(), TopAbs_SHELL);
326  if (xp.More() != Standard_True) {
327  continue;
328  }
329  builder.Add(pieces, mkCommon.Shape());
330  }
331 
332  // now get any loose edges in the input
333  TopExp_Explorer expl3(copyShape, TopAbs_EDGE, TopAbs_FACE);
334  for (; expl3.More(); expl3.Next()) {
335  const TopoDS_Edge& e = TopoDS::Edge(expl3.Current());
336  FCBRepAlgoAPI_Common mkCommon(e, tool);
337  if (!mkCommon.IsDone()) {
338  continue;
339  }
340  if (mkCommon.Shape().IsNull()) {
341  continue;
342  }
343  //Did we get a result?
344  TopExp_Explorer xp;
345  xp.Init(mkCommon.Shape(), TopAbs_EDGE);
346  if (xp.More() != Standard_True) {
347  continue;
348  }
349  builder.Add(pieces, mkCommon.Shape());
350  }
351 
352  // save the detail shape for further processing
353  m_detailShape = pieces;
354 
355  if (debugDetail()) {
356  BRepTools::Write(tool, "DVDTool.brep"); //debug
357  BRepTools::Write(copyShape, "DVDCopy.brep");//debug
358  BRepTools::Write(m_detailShape, "DVDCommon.brep"); //debug
359  }
360 
361  gp_Pnt inputCenter = ShapeUtils::findCentroid(m_detailShape, dirDetail);
362  Base::Vector3d centroid(inputCenter.X(), inputCenter.Y(), inputCenter.Z());
363  m_saveCentroid += centroid;//center of massaged shape
364  //align shape with detail anchor
365  TopoDS_Shape centeredShape = ShapeUtils::moveShape(m_detailShape, anchor * -1.0);
366  m_scaledShape = ShapeUtils::scaleShape(centeredShape, getScale());
367 
368  if (debugDetail()) {
369  BRepTools::Write(m_scaledShape, "DVDScaled.brep");//debug
370  }
371 
372  Base::Vector3d stdOrg(0.0, 0.0, 0.0);
373  m_viewAxis = dvp->getProjectionCS(stdOrg);
374 
375  if (!DrawUtil::fpCompare(Rotation.getValue(), 0.0)) {
376  m_scaledShape = ShapeUtils::rotateShape(m_scaledShape, m_viewAxis, Rotation.getValue());
377  }
378 
379  showProgressMessage(getNameInDocument(), "has finished making detail shape");
380 }

418 TopoDS_Shape DrawViewDetail::projectEdgesOntoFace(TopoDS_Shape& edgeShape, TopoDS_Face& projFace,
419  gp_Dir& projDir)
420 {
421  BRep_Builder builder;
422  TopoDS_Compound edges;
423  builder.MakeCompound(edges);
424  TopExp_Explorer Ex(edgeShape, TopAbs_EDGE);
425  while (Ex.More()) {
426  TopoDS_Edge e = TopoDS::Edge(Ex.Current());
427  BRepProj_Projection mkProj(e, projFace, projDir);
428  if (mkProj.IsDone()) {
429  builder.Add(edges, mkProj.Shape());
430  }
431  Ex.Next();
432  }
433  if (debugDetail()) {
434  BRepTools::Write(edges, "DVDEdges.brep");//debug
435  }
436 
437  return TopoDS_Shape(std::move(edges));
438 }

441 Base::Vector3d DrawViewDetail::mapPoint3dToDetail(const Base::Vector3d& inPoint) const
442 {
443  auto baseObj = BaseView.getValue();
444  auto baseDvp = dynamic_cast<DrawViewPart*>(baseObj);
445  if (!baseDvp) {
446  throw Base::RuntimeError("Detail has no BaseView");
447  }
448  auto pointOnBase = baseDvp->projectPoint(inPoint, false);
449  auto detailCenter = AnchorPoint.getValue();
450  auto pointOnDetail = DU::invertY(pointOnBase - detailCenter);
451  return pointOnDetail;
452 }
