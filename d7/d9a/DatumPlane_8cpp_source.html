44 Plane::Plane()
45 {
46  // These properties are only relevant for the visual appearance.
47  // Since they are getting changed from within its view provider
48  // their type is set to "Output" to avoid that they are marked as
49  // touched all the time.
50  ADD_PROPERTY_TYPE(ResizeMode,(static_cast<long>(0)), "Size", App::Prop_Output, "Automatic or manual resizing");
51  ResizeMode.setEnums(ResizeModeEnums);
52  ADD_PROPERTY_TYPE(Length,(20), "Size", App::Prop_Output, "Length of the plane");
53  ADD_PROPERTY_TYPE(Width,(20), "Size", App::Prop_Output, "Width of the plane");
54 
55  Length.setReadOnly(true);
56  Width.setReadOnly(true);
57 
58  this->setAttacher(new AttachEnginePlane);
59  // Create a shape, which will be used by the Sketcher. Them main function is to avoid a dependency of
60  // Sketcher on the PartDesign module
61  BRepBuilderAPI_MakeFace builder(gp_Pln(gp_Pnt(0,0,0), gp_Dir(0,0,1)));
62  if (!builder.IsDone())
63  return;
64  TopoDS_Shape myShape = builder.Shape();
65  myShape.Infinite(Standard_True);
66  Shape.setValue(myShape);
67 }
