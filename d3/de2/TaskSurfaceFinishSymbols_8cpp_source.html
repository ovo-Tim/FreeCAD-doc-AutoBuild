88 TaskSurfaceFinishSymbols::TaskSurfaceFinishSymbols(const std::string &ownerName) :
89  currentIcon(nullptr),
90  ui(new Ui_TaskSurfaceFinishSymbols)
91 {
92  App::Document *doc = App::GetApplication().getActiveDocument();
93  if (doc) {
94  owner = doc->getObject(ownerName.c_str());
95  std::string subName;
96  if (!owner) {
97  size_t dot = ownerName.rfind('.');
98  if (dot != std::string::npos) {
99  subName = ownerName.substr(dot + 1);
100  owner = doc->getObject(ownerName.substr(0, dot).c_str());
101  }
102  }
103 
104  auto page = dynamic_cast<TechDraw::DrawPage *>(owner);
105  if (page) {
106  placement.x = page->getPageWidth()/2.0;
107  placement.y = page->getPageHeight()/2.0;
108  }
109 
110  auto viewPart = dynamic_cast<TechDraw::DrawViewPart *>(owner);
111  if (viewPart && !subName.empty()) {
112  std::string subType = DrawUtil::getGeomTypeFromName(subName);
113  if (subType == "Vertex") {
114  TechDraw::VertexPtr vertex = viewPart->getVertex(subName);
115  if (vertex) {
116  placement = vertex->point();
117  }
118  }
119  else if (subType == "Edge") {
120  TechDraw::BaseGeomPtr edge = viewPart->getEdge(subName);
121  if (edge) {
122  placement = edge->getMidPoint();
123  }
124  }
125  else if (subType == "Face") {
126  TechDraw::FacePtr face = viewPart->getFace(subName);
127  if (face) {
128  placement = face->getCenter();
129  }
130  }
131 
132  placement = DrawUtil::invertY(placement);
133  }
134  }
135 
136  raValues = {"Ra50", "Ra25", "Ra12, 5", "Ra6, 3",
137  "Ra3, 2", "Ra1, 6", "Ra0, 8", "Ra0, 4",
138  "Ra0, 2", "Ra0, 1", "Ra0, 05", "Ra0, 025"};
139  laySymbols = {"", "=", "âŸ‚", "X", "M", "C", "R"};
140  roughGrades = {"", "N1", "N2", "N3", "N4", "N5",
141  "N6", "N7", "N8", "N9", "N10", "N11"};
142 
143  ui->setupUi(this);
144  setUiEdit();
145 }

158 QPixmap TaskSurfaceFinishSymbols::baseSymbol(symbolType type)
159 // return QPixmap showing a base symbol
160 {
161  QImage img (50, 64, QImage::Format_ARGB32_Premultiplied);
162  img.fill(Qt::transparent);
163 
164  // TODO: color should depend on theme/background
165  QPainter painter;
166  painter.begin(&img);
167  painter.setPen(QPen(getPenColor(), 2, Qt::SolidLine,

171  QPainter::TextAntialiasing);
172  painter.drawLine(QLine(0, 40, 12, 60));

184 std::string TaskSurfaceFinishSymbols::completeSymbol()
185 // return string showing the complete symbol
186 {
187  SvgString symbol(150, 64);
188  symbol.addLine(0, 44, 12, 64);
189  symbol.addLine(12, 64, 42, 14);
190  int moveLeft(0), maxTextLength(0);
191  if (activeIcon == removeProhibit || activeIcon == removeProhibitAll)
192  symbol.addCircle(12, 46, 9);
193  if (activeIcon == removeRequired || activeIcon == removeRequiredAll)
194  symbol.addLine(0, 44, 24, 44);
195  if (activeIcon > removeRequired)
196  {
197  symbol.addCircle(42, 14, 6);
198  moveLeft = 5 ;
199  }
200  std::string methodText = leMethod->text().toStdString();
201  symbol.addText(42+moveLeft, 11, methodText);
202  int methodTextLength = methodText.length();
203  if (isISO)
204  {
205  std::string raText = cbRA->itemText(cbRA->currentIndex()).toStdString();
206  symbol.addText(42+moveLeft, 30, raText);
207  int raTextLength = raText.length();
208  maxTextLength = std::max(methodTextLength, raTextLength)*9.25+moveLeft;
209  }
210  else
211  {
212  std::string sampleText = leSamLength->text().toStdString();
213  symbol.addText(42+moveLeft, 30, sampleText);
214  int sampleTextLength = sampleText.length();
215  maxTextLength = std::max(methodTextLength, sampleTextLength)*9.25+moveLeft;
216  std::string minRoughtText = cbMinRought->itemText(cbMinRought->currentIndex()).toStdString();
217  symbol.addText(-10, 35, minRoughtText);
218  std::string maxRoughtText = cbMaxRought->itemText(cbMaxRought->currentIndex()).toStdString();
219  symbol.addText(-10, 20, maxRoughtText);
220  }
221  symbol.addLine(42, 14, 42+maxTextLength, 14);
222  symbol.addText(20, 60, cbLay->itemText(cbLay->currentIndex()).toStdString());
223  symbol.addText(-25, 60, leAddition->text().toStdString());
224  std::string symbolAsString = symbol.finish();
225  return symbolAsString;
226 }

348 void TaskSurfaceFinishSymbols::onIconChanged()
349 // Slot: change icon of surface finish symbol
350 {
351  QObject* senderObj(this->sender());
352  QPushButton* pressedButton = qobject_cast<QPushButton*>(senderObj);
353  if (!pressedButton) {
354  return;
355  }
356 
357  if (ui->pbIcon01 == pressedButton) activeIcon = anyMethod;
358  if (ui->pbIcon02 == pressedButton) activeIcon = removeProhibit;
359  if (ui->pbIcon03 == pressedButton) activeIcon = removeRequired;
360  if (ui->pbIcon04 == pressedButton) activeIcon = anyMethodAll;
361  if (ui->pbIcon05 == pressedButton) activeIcon = removeProhibitAll;
362  if (ui->pbIcon06 == pressedButton) activeIcon = removeRequiredAll;
363 
364  QIcon symbolIcon = pressedButton->icon();
365  if(currentIcon) {
366  symbolScene->removeItem(currentIcon);
367  }
368  currentIcon = new(QGraphicsPixmapItem);
369  currentIcon->setPixmap(symbolIcon.pixmap(50, 64));
370  currentIcon->setPos(-50, -126);
371  currentIcon->setZValue(-1);
372  symbolScene->addItem(currentIcon);
373 }

395 bool TaskSurfaceFinishSymbols::accept()
396 // Slot: dialog finished using OK
397 {
398  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Surface Finish Symbols"));
399  App::Document *doc = Application::Instance->activeDocument()->getDocument();
400  App::DocumentObject *docObject = doc->addObject("TechDraw::DrawViewSymbol", "SurfaceSymbol");
401  TechDraw::DrawViewSymbol *surfaceSymbol = dynamic_cast<TechDraw::DrawViewSymbol*>(docObject);
402  surfaceSymbol->Symbol.setValue(completeSymbol());
403  surfaceSymbol->Rotation.setValue(ui->leAngle->text().toDouble());
404 
405  auto view = dynamic_cast<TechDraw::DrawView *>(owner);
406  surfaceSymbol->Owner.setValue(view);
407  surfaceSymbol->X.setValue(placement.x);
408  surfaceSymbol->Y.setValue(placement.y);
409 
410  auto viewProvider = dynamic_cast<ViewProviderSymbol *>(QGIView::getViewProvider(surfaceSymbol));
411  if (viewProvider) {
412  viewProvider->StackOrder.setValue(ZVALUE::DIMENSION);
413  }
414 
415  auto page = dynamic_cast<TechDraw::DrawPage *>(owner);
416  if (!page && view) {
417  page = view->findParentPage();
418  }
419  if (page) {
420  page->addView(surfaceSymbol);
421  }
422 
423  Gui::Command::commitCommand();
424  return true;
425 }
