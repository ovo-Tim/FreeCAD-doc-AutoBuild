104  NumberOfNodes = len(femobjects[i]["Nodes"])
105  if obj.References:
106  for name in obj.References[0][1]:
107  if obj.ConstraintType == "Temperature":

138  def _outputHeatBodyForce(self, obj, name):
139  if obj.Mode == "Dissipation Rate":
140  heatSource = obj.DissipationRate.getValueAs("W/kg").Value
141 
142  elif obj.Mode == "Total Power":
143  ref = obj.References[0]
144  ref_feat = ref[0]
145  ref_sub_obj = ref[1][0]
146  density = None
147  for mat in self.write.getMember("App::MaterialObject"):
148  mat_ref = [
149  *itertools.chain(*[itertools.product([i[0]], i[1]) for i in mat.References])
150  ]
151  if (ref_feat, ref_sub_obj) in mat_ref:
152  density = FreeCAD.Units.Quantity(mat.Material["Density"])
153  break
154 
155  if not density:
156  # search material without references
157  for mat in self.write.getMember("App::MaterialObject"):
158  if not mat.References:
159  density = FreeCAD.Units.Quantity(mat.Material["Density"])
160  break
161  volume = ref_feat.getSubObject(ref_sub_obj).Volume
162  heatSource = (
163  (obj.TotalPower / (density * FreeCAD.Units.Quantity(volume, "mm^3")))
164  .getValueAs("W/kg")
165  .Value
166  )
167 
168  if heatSource == 0.0:
169  # a zero heat would break Elmer (division by zero)
170  raise general_writer.WriteError("The body heat source must not be zero!")
171  self.write.bodyForce(name, "Heat Source", heatSource)
