305 unsigned short HidToVirtualKey(unsigned long pid, unsigned short hidKeyCode)
306 {
307  unsigned short virtualkey = hidKeyCode;
308  for (size_t i = 0; i < sizeof(_3dmouseVirtualKeys) / sizeof(_3dmouseVirtualKeys[0]); ++i) {
309  if (pid == _3dmouseVirtualKeys[i].pid) {
310  if (hidKeyCode < _3dmouseVirtualKeys[i].nKeys)
311  virtualkey = _3dmouseVirtualKeys[i].vkeys[hidKeyCode];
312  else
313  virtualkey = V3DK_INVALID;
314  break;
315  }
316  }
317  // Remaining devices are unchanged
318  return virtualkey;
319 }

428 bool Gui::GuiNativeEvent::Is3dmouseAttached()
429 {
430  unsigned int numDevicesOfInterest = 0;
431  PRAWINPUTDEVICE devicesToRegister = GetDevicesToRegister(&numDevicesOfInterest);
432 
433  unsigned int nDevices = 0;
434 
435  if (::GetRawInputDeviceList(NULL, &nDevices, sizeof(RAWINPUTDEVICELIST)) != 0) {
436  return false;
437  }
438 
439  if (nDevices == 0)
440  return false;
441 
442  std::vector<RAWINPUTDEVICELIST> rawInputDeviceList(nDevices);
443  if (::GetRawInputDeviceList(&rawInputDeviceList[0], &nDevices, sizeof(RAWINPUTDEVICELIST))
444  == static_cast<unsigned int>(-1)) {
445  return false;
446  }
447 
448  for (unsigned int i = 0; i < nDevices; ++i) {
449  RID_DEVICE_INFO rdi = {sizeof(rdi)};
450  unsigned int cbSize = sizeof(rdi);
451 
452  if (GetRawInputDeviceInfo(rawInputDeviceList[i].hDevice, RIDI_DEVICEINFO, &rdi, &cbSize)
453  > 0) {
454  //skip non HID and non logitec (3DConnexion) devices
455  if (!(rdi.dwType == RIM_TYPEHID
456  && (rdi.hid.dwVendorId == LOGITECH_VENDOR_ID
457  || rdi.hid.dwVendorId == CONNEXION_VENDOR_ID))) {
458  continue;
459  }
460 
461  //check if devices matches Multi-axis Controller
462  for (unsigned int j = 0; j < numDevicesOfInterest; ++j) {
463  if (devicesToRegister[j].usUsage == rdi.hid.usUsage
464  && devicesToRegister[j].usUsagePage == rdi.hid.usUsagePage) {
465  return true;
466  }
467  }
468  }
469  }
470  return false;
471 }

518 UINT Gui::GuiNativeEvent::GetRawInputBuffer(PRAWINPUT pData, PUINT pcbSize, UINT cbSizeHeader)
519 {
520 #ifdef _WIN64
521  return ::GetRawInputBuffer(pData, pcbSize, cbSizeHeader);
522 #else
523  BOOL bIsWow64 = FALSE;
524  // 0002287: Entry Point Not Found IsWow64Process for Windows 2000
525  //::IsWow64Process(GetCurrentProcess(), &bIsWow64);
526  typedef BOOL(WINAPI * LPFN_ISWOW64PROCESS)(HANDLE, PBOOL);
527 
528  LPFN_ISWOW64PROCESS fnIsWow64Process =
529  (LPFN_ISWOW64PROCESS)GetProcAddress(GetModuleHandle("kernel32"), "IsWow64Process");
530 
531  if (NULL != fnIsWow64Process) {
532  fnIsWow64Process(GetCurrentProcess(), &bIsWow64);
533  }
534  if (!bIsWow64 || pData == NULL) {
535  return ::GetRawInputBuffer(pData, pcbSize, cbSizeHeader);
536  }
537  else {
538  HWND hwndTarget = fWindow;//fParent->winId();
539 
540  size_t cbDataSize = 0;
541  UINT nCount = 0;
542  PRAWINPUT pri = pData;
543 
544  MSG msg;
545  while (PeekMessage(&msg, hwndTarget, WM_INPUT, WM_INPUT, PM_NOREMOVE)) {
546  HRAWINPUT hRawInput = reinterpret_cast<HRAWINPUT>(msg.lParam);
547  size_t cbSize = *pcbSize - cbDataSize;
548  if (::GetRawInputData(hRawInput, RID_INPUT, pri, &cbSize, cbSizeHeader)
549  == static_cast<UINT>(-1)) {
550  if (nCount == 0) {
551  return static_cast<UINT>(-1);

553  else {
554  break;
555  }
556  }

558 
559  // Remove the message for the data just read
560  PeekMessage(&msg, hwndTarget, WM_INPUT, WM_INPUT, PM_REMOVE);
561 
562  pri = NEXTRAWINPUTBLOCK(pri);
563  cbDataSize = reinterpret_cast<ULONG_PTR>(pri) - reinterpret_cast<ULONG_PTR>(pData);
564  if (cbDataSize >= *pcbSize) {
565  cbDataSize = *pcbSize;
566  break;
567  }
568  }
569  return nCount;
570  }

581 void Gui::GuiNativeEvent::On3dmouseInput()
582 {
583  // Don't do any data processing in background
584  bool bIsForeground = (::GetActiveWindow() != NULL);
585  if (!bIsForeground) {
586  // set all cached data to zero so that a zero event is seen and the cached data deleted
587  for (std::map<HANDLE, TInputData>::iterator it = fDevice2Data.begin();
588  it != fDevice2Data.end(); it++) {
589  it->second.fAxes.assign(6, .0);
590  it->second.fIsDirty = true;
591  }
592  }
593 
594  DWORD dwNow = ::GetTickCount();// Current time;
595  DWORD dwElapsedTime; // Elapsed time since we were last here
596 
597  if (0 == fLast3dmouseInputTime) {
598  dwElapsedTime = 10;// System timer resolution
599  }
600  else {
601  dwElapsedTime = dwNow - fLast3dmouseInputTime;
602  if (fLast3dmouseInputTime > dwNow) {
603  dwElapsedTime = ~dwElapsedTime + 1;
604  }
605  if (dwElapsedTime < 1) {
606  dwElapsedTime = 1;
607  }
608  else if (dwElapsedTime > 500) {
609  // Check for wild numbers because the device was removed while sending data
610  dwElapsedTime = 10;
611  }
612  }
613 
614 #if _TRACE_3DINPUT_PERIOD
615  qDebug("On3DmouseInput() period is %dms\n", dwElapsedTime);
616 #endif
617 
618  float mouseData2Rotation;
619  // v = w * r, we don't know r yet so lets assume r=1.)
620  float mouseData2PanZoom;
621 
622  // Grab the I3dmouseParam interface
623  I3dMouseParam &i3dmouseParam = f3dMouseParams;
624  // Take a look at the users preferred speed setting and adjust the sensitivity accordingly
625  I3dMouseSensor::ESpeed speedSetting = i3dmouseParam.GetSpeed();
626  // See "Programming for the 3D Mouse", Section 5.1.3
627  float speed = (speedSetting == I3dMouseSensor::kLowSpeed ? 0.25f
628  : speedSetting == I3dMouseSensor::kHighSpeed ? 4.f
629  : 1.f);
630 
631  // Multiplying by the following will convert the 3d mouse data to real world units
632  mouseData2PanZoom = speed;
633  mouseData2Rotation = speed;
634 
635  std::map<HANDLE, TInputData>::iterator iterator = fDevice2Data.begin();
636  while (iterator != fDevice2Data.end()) {
637 
638  // If we have not received data for a while send a zero event
639  if ((--(iterator->second.fTimeToLive)) == 0) {
640  iterator->second.fAxes.assign(6, .0);
641  }
642  else if ( !iterator->second.fIsDirty) {
643  // If we are not polling then only handle the data that was actually received
644  ++iterator;
645  continue;
646  }
647  iterator->second.fIsDirty = false;
648 
649  // get a copy of the device
650  HANDLE hdevice = iterator->first;
651 
652  // get a copy of the motion vectors and apply the user filters
653  std::vector<float> motionData = iterator->second.fAxes;
654 
655  // apply the user filters
656 
657  // Pan Zoom filter
658  // See "Programming for the 3D Mouse", Section 5.1.2
659  if (!i3dmouseParam.IsPanZoom()) {
660  // Pan zoom is switched off so set the translation vector values to zero
661  motionData[0] = motionData[1] = motionData[2] = 0.;
662  }
663 
664  // Rotate filter
665  // See "Programming for the 3D Mouse", Section 5.1.1
666  if (!i3dmouseParam.IsRotate()) {
667  // Rotate is switched off so set the rotation vector values to zero
668  motionData[3] = motionData[4] = motionData[5] = 0.;
669  }
670 
671  // convert the translation vector into physical data
672  for (int axis = 0; axis < 3; axis++) { motionData[axis] *= mouseData2PanZoom; }
673  // convert the directed Rotate vector into physical data
674  // See "Programming for the 3D Mouse", Section 7.2.2
675  for (int axis = 3; axis < 6; axis++) { motionData[axis] *= mouseData2Rotation; }
676 
677  // Now that the data has had the filters and sensitivity settings applied
678  // calculate the displacements since the last view update
679  for (int axis = 0; axis < 6; axis++) { motionData[axis] *= dwElapsedTime; }
680 
681 
682  // Now a bit of book keeping before passing on the data
683  if (iterator->second.IsZero()) {
684  iterator = fDevice2Data.erase(iterator);
685  }
686  else {
687  ++iterator;
688  }
689 
690  // Work out which will be the next device
691  HANDLE hNextDevice = 0;
692  if (iterator != fDevice2Data.end()) {
693  hNextDevice = iterator->first;
694  }
695 
696  // Pass the 3dmouse input to the view controller
697  Move3d(hdevice, motionData);
698 
699  // Because we don't know what happened in the previous call, the cache might have
700  // changed so reload the iterator
701  iterator = fDevice2Data.find(hNextDevice);
702  }
703 
704  if (!fDevice2Data.empty()) {
705  fLast3dmouseInputTime = dwNow;
706  }
707  else {
708  fLast3dmouseInputTime = 0;
709  }
