110 void Rotation::evaluateVector()
111 {
112  // Taken from <http://de.wikipedia.org/wiki/Quaternionen>
113  //
114  // Note: -1 < w < +1 (|w| == 1 not allowed, with w:=quat[3])
115  if ((this->quat[3] > -1.0) && (this->quat[3] < 1.0)) {
116  double rfAngle = acos(this->quat[3]) * 2.0;
117  double scale = sin(rfAngle / 2.0);
118  // Get a normalized vector
119  double l = this->_axis.Length();
120  if (l < Base::Vector3d::epsilon()) {
121  l = 1;
122  }
123  this->_axis.x = this->quat[0] * l / scale;
124  this->_axis.y = this->quat[1] * l / scale;
125  this->_axis.z = this->quat[2] * l / scale;
126 
127  _angle = rfAngle;
128  }
129  else {
130  _axis.Set(0.0, 0.0, 1.0);
131  _angle = 0.0;
132  }
133 }

165 void Rotation::getValue(Matrix4D& matrix) const
166 {
167  // Taken from <http://de.wikipedia.org/wiki/Quaternionen>
168  //
169  const double l = sqrt(this->quat[0] * this->quat[0] + this->quat[1] * this->quat[1]
170  + this->quat[2] * this->quat[2] + this->quat[3] * this->quat[3]);
171  const double x = this->quat[0] / l;
172  const double y = this->quat[1] / l;
173  const double z = this->quat[2] / l;
174  const double w = this->quat[3] / l;
175 
176  matrix[0][0] = 1.0 - 2.0 * (y * y + z * z);
177  matrix[0][1] = 2.0 * (x * y - z * w);
178  matrix[0][2] = 2.0 * (x * z + y * w);
179  matrix[0][3] = 0.0;
180 
181  matrix[1][0] = 2.0 * (x * y + z * w);
182  matrix[1][1] = 1.0 - 2.0 * (x * x + z * z);
183  matrix[1][2] = 2.0 * (y * z - x * w);
184  matrix[1][3] = 0.0;
185 
186  matrix[2][0] = 2.0 * (x * z - y * w);
187  matrix[2][1] = 2.0 * (y * z + x * w);
188  matrix[2][2] = 1.0 - 2.0 * (x * x + y * y);
189  matrix[2][3] = 0.0;
190 
191  matrix[3][0] = 0.0;
192  matrix[3][1] = 0.0;
193  matrix[3][2] = 0.0;
194  matrix[3][3] = 1.0;
195 }

207 void Rotation::setValue(const Matrix4D& m)
208 {
209  // Get the rotation part matrix
210  Matrix4D mc = m.decompose()[2];
211  // Extract quaternion
212  double trace = (mc[0][0] + mc[1][1] + mc[2][2]);
213  if (trace > 0.0) {
214  double s = sqrt(1.0 + trace);
215  this->quat[3] = 0.5 * s;
216  s = 0.5 / s;
217  this->quat[0] = ((mc[2][1] - mc[1][2]) * s);
218  this->quat[1] = ((mc[0][2] - mc[2][0]) * s);
219  this->quat[2] = ((mc[1][0] - mc[0][1]) * s);
220  }
221  else {
222  // Described in RotationIssues.pdf from <http://www.geometrictools.com>
223  //
224  // Get the max. element of the trace
225  unsigned short i = 0;
226  if (mc[1][1] > mc[0][0]) {
227  i = 1;
228  }
229  if (mc[2][2] > mc[i][i]) {
230  i = 2;
231  }
232 
233  unsigned short j = (i + 1) % 3;
234  unsigned short k = (i + 2) % 3;
235 
236  double s = sqrt((mc[i][i] - (mc[j][j] + mc[k][k])) + 1.0);
237  this->quat[i] = s * 0.5;
238  s = 0.5 / s;
239  this->quat[3] = ((mc[k][j] - mc[j][k]) * s);
240  this->quat[j] = ((mc[j][i] + mc[i][j]) * s);
241  this->quat[k] = ((mc[k][i] + mc[i][k]) * s);
242  }
243 
244  this->evaluateVector();
245 }

247 void Rotation::setValue(const Vector3d& axis, double fAngle)
248 {
249  // Taken from <http://de.wikipedia.org/wiki/Quaternionen>
250  //
251  // normalization of the angle to be in [0, 2pi[
252  _angle = fAngle;
253  double theAngle = fAngle - floor(fAngle / (2.0 * D_PI)) * (2.0 * D_PI);
254  this->quat[3] = cos(theAngle / 2.0);
255 
256  Vector3d norm = axis;
257  norm.Normalize();
258  double l = norm.Length();
259  // Keep old axis in case the new axis is the null vector
260  if (l > 0.5) {
261  this->_axis = axis;
262  }
263  else {
264  norm = _axis;
265  norm.Normalize();
266  }
267 
268  double scale = sin(theAngle / 2.0);
269  this->quat[0] = norm.x * scale;
270  this->quat[1] = norm.y * scale;
271  this->quat[2] = norm.z * scale;
272 }

274 void Rotation::setValue(const Vector3d& rotateFrom, const Vector3d& rotateTo)
275 {
276  Vector3d u(rotateFrom);
277  u.Normalize();
278  Vector3d v(rotateTo);
279  v.Normalize();
280 
281  // The vector from x to is the rotation axis because it's the normal of the plane defined by
282  // (0,u,v)
283  const double dot = u * v;
284  Vector3d w = u % v;
285  const double wlen = w.Length();
286 
287  if (wlen == 0.0) { // Parallel vectors
288  // Check if they are pointing in the same direction.
289  if (dot > 0.0) {
290  this->setValue(0.0, 0.0, 0.0, 1.0);
291  }
292  else {
293  // We can use any axis perpendicular to u (and v)
294  Vector3d t = u % Vector3d(1.0, 0.0, 0.0);
295  if (t.Length() < Base::Vector3d::epsilon()) {
296  t = u % Vector3d(0.0, 1.0, 0.0);
297  }
298  this->setValue(t.x, t.y, t.z, 0.0);
299  }
300  }
301  else { // Vectors are not parallel
302  // Note: A quaternion is not well-defined by specifying a point and its transformed point.
303  // Every quaternion with a rotation axis having the same angle to the vectors of both points
304  // is okay.
305  double angle = acos(dot);
306  this->setValue(w, angle);
307  }
308 }

375 {

377  double x0 {};
378  double y0 {};

441 void Rotation::multVec(const Vector3d& src, Vector3d& dst) const
442 {
443  double x = this->quat[0];
444  double y = this->quat[1];
445  double z = this->quat[2];
446  double w = this->quat[3];
447  double x2 = x * x;
448  double y2 = y * y;
449  double z2 = z * z;
450  double w2 = w * w;
451 
452  double dx =
453  (x2 + w2 - y2 - z2) * src.x + 2.0 * (x * y - z * w) * src.y + 2.0 * (x * z + y * w) * src.z;
454  double dy =
455  2.0 * (x * y + z * w) * src.x + (w2 - x2 + y2 - z2) * src.y + 2.0 * (y * z - x * w) * src.z;
456  double dz =
457  2.0 * (x * z - y * w) * src.x + 2.0 * (x * w + y * z) * src.y + (w2 - x2 - y2 + z2) * src.z;
458  dst.x = dx;
459  dst.y = dy;
460  dst.z = dz;
461 }

485 Rotation Rotation::slerp(const Rotation& q0, const Rotation& q1, double t)
486 {
487  // Taken from <http://www.euclideanspace.com/maths/algebra/realNormedAlgebra/quaternions/slerp/>
488  if (t < 0.0) {
489  t = 0.0;
490  }
491  else if (t > 1.0) {
492  t = 1.0;
493  }
494 
495  double scale0 = 1.0 - t;
496  double scale1 = t;
497  double dot = q0.quat[0] * q1.quat[0] + q0.quat[1] * q1.quat[1] + q0.quat[2] * q1.quat[2]
498  + q0.quat[3] * q1.quat[3];
499  bool neg = false;
500  if (dot < 0.0) {
501  dot = -dot;
502  neg = true;
503  }
504 
505  if ((1.0 - dot) > Base::Vector3d::epsilon()) {
506  double angle = acos(dot);
507  double sinangle = sin(angle);
508  // If possible calculate spherical interpolation, otherwise use linear interpolation
509  if (sinangle > Base::Vector3d::epsilon()) {
510  scale0 = double(sin((1.0 - t) * angle)) / sinangle;
511  scale1 = double(sin(t * angle)) / sinangle;
512  }
513  }
514 
515  if (neg) {
516  scale1 = -scale1;
517  }
518 
519  double x = scale0 * q0.quat[0] + scale1 * q1.quat[0];
520  double y = scale0 * q0.quat[1] + scale1 * q1.quat[1];
521  double z = scale0 * q0.quat[2] + scale1 * q1.quat[2];
522  double w = scale0 * q0.quat[3] + scale1 * q1.quat[3];
523  return {x, y, z, w};
524 }

531 Rotation
532 Rotation::makeRotationByAxes(Vector3d xdir, Vector3d ydir, Vector3d zdir, const char* priorityOrder)
533 {
534  const double tol = Precision::Confusion();
535  enum dirIndex
536  {
537  X,
538  Y,
539  Z
540  };
541 
542  // convert priorityOrder string into a sequence of ints.
543  if (strlen(priorityOrder) != 3) {
544  THROWM(ValueError, "makeRotationByAxes: length of priorityOrder is not 3");
545  }
546  int order[3];
547  for (int i = 0; i < 3; ++i) {
548  order[i] = priorityOrder[i] - 'X';
549  if (order[i] < 0 || order[i] > 2) {
550  THROWM(ValueError,
551  "makeRotationByAxes: characters in priorityOrder must be uppercase X, Y, or Z. "
552  "Some other character encountered.")
553  }
554  }
555 
556  // ensure every axis is listed in priority list
557  if (order[0] == order[1] || order[1] == order[2] || order[2] == order[0]) {
558  THROWM(ValueError, "makeRotationByAxes: not all axes are listed in priorityOrder");
559  }
560 
561 
562  // group up dirs into an array, to access them by indexes stored in @order.
563  std::vector<Vector3d*> dirs = {&xdir, &ydir, &zdir};
564 
565 
566  auto dropPriority = [&order](int index) {
567  int tmp {};
568  if (index == 0) {
569  tmp = order[0];
570  order[0] = order[1];
571  order[1] = order[2];
572  order[2] = tmp;
573  }
574  else if (index == 1) {
575  tmp = order[1];
576  order[1] = order[2];
577  order[2] = tmp;
578  } // else if index == 2 do nothing
579  };
580 
581  // pick up the strict direction
582  Vector3d mainDir;
583  for (int i = 0; i < 3; ++i) {
584  mainDir = *(dirs[size_t(order[0])]);
585  if (mainDir.Length() > tol) {
586  break;
587  }
588 
589  dropPriority(0);
590 
591  if (i == 2) {
592  THROWM(ValueError, "makeRotationByAxes: all directions supplied are zero");
593  }
594  }
595  mainDir.Normalize();
596 
597  // pick up the 2nd priority direction, "hint" direction.
598  Vector3d hintDir;
599  for (int i = 0; i < 2; ++i) {
600  hintDir = *(dirs[size_t(order[1])]);
601  if ((hintDir.Cross(mainDir)).Length() > tol) {
602  break;
603  }
604 
605  dropPriority(1);
606 
607  if (i == 1) {
608  hintDir = Vector3d(); // no vector can be used as hint direction. Zero it out, to
609  // indicate that a guess is needed.
610  }
611  }
612  if (hintDir.Length() == 0.0) {
613  switch (order[0]) {
614  case X: { // xdir is main
615  // align zdir to OZ
616  order[1] = Z;
617  order[2] = Y;
618  hintDir = Vector3d(0, 0, 1);
619  if ((hintDir.Cross(mainDir)).Length() <= tol) {
620  // aligning to OZ is impossible, align to ydir to OY. Why so? I don't know, just
621  // feels right =)
622  hintDir = Vector3d(0, 1, 0);
623  order[1] = Y;
624  order[2] = Z;
625  }
626  } break;
627  case Y: { // ydir is main
628  // align zdir to OZ
629  order[1] = Z;
630  order[2] = X;
631  hintDir = mainDir.z > -tol ? Vector3d(0, 0, 1) : Vector3d(0, 0, -1);
632  if ((hintDir.Cross(mainDir)).Length() <= tol) {
633  // aligning zdir to OZ is impossible, align xdir to OX then.
634  hintDir = Vector3d(1, 0, 0);
635  order[1] = X;
636  order[2] = Z;
637  }
638  } break;
639  case Z: { // zdir is main
640  // align ydir to OZ
641  order[1] = Y;
642  order[2] = X;
643  hintDir = Vector3d(0, 0, 1);
644  if ((hintDir.Cross(mainDir)).Length() <= tol) {
645  // aligning ydir to OZ is impossible, align xdir to OX then.
646  hintDir = Vector3d(1, 0, 0);
647  order[1] = X;
648  order[2] = Y;
649  }
650  } break;
651  } // switch ordet[0]
652  }
653 
654  // ensure every axis is listed in priority list
655  assert(order[0] != order[1]);
656  assert(order[1] != order[2]);
657  assert(order[2] != order[0]);
658 
659  hintDir.Normalize();
660  // make hintDir perpendicular to mainDir. For that, we cross-product the two to obtain the third
661  // axis direction, and then recover back the hint axis by doing another cross product.
662  Vector3d lastDir = mainDir.Cross(hintDir);
663  lastDir.Normalize();
664  hintDir = lastDir.Cross(mainDir);
665  hintDir.Normalize(); // redundant?
666 
667  Vector3d finaldirs[3];
668  finaldirs[order[0]] = mainDir;
669  finaldirs[order[1]] = hintDir;
670  finaldirs[order[2]] = lastDir;
671 
672  // fix handedness
673  if (finaldirs[X].Cross(finaldirs[Y]) * finaldirs[Z] < 0.0) {
674  // handedness is wrong. Switch the direction of the least important axis
675  finaldirs[order[2]] = finaldirs[order[2]] * (-1.0);

683  m[0][i] = finaldirs[i].x;
684  m[1][i] = finaldirs[i].y;

712 void Rotation::getYawPitchRoll(double& y, double& p, double& r) const
713 {
714  double q00 = quat[0] * quat[0];
715  double q11 = quat[1] * quat[1];
716  double q22 = quat[2] * quat[2];
717  double q33 = quat[3] * quat[3];
718  double q01 = quat[0] * quat[1];
719  double q02 = quat[0] * quat[2];
720  double q03 = quat[0] * quat[3];
721  double q12 = quat[1] * quat[2];
722  double q13 = quat[1] * quat[3];
723  double q23 = quat[2] * quat[3];
724  double qd2 = 2.0 * (q13 - q02);
725 
726  // handle gimbal lock
727  if (fabs(qd2 - 1.0) <= 16 * DBL_EPSILON) { // Tolerance copied from OCC "gp_Quaternion.cxx"
728  // north pole
729  y = 0.0;
730  p = D_PI / 2.0;
731  r = 2.0 * atan2(quat[0], quat[3]);
732  }
733  else if (fabs(qd2 + 1.0)
734  <= 16 * DBL_EPSILON) { // Tolerance copied from OCC "gp_Quaternion.cxx"
735  // south pole
736  y = 0.0;
737  p = -D_PI / 2.0;
738  r = 2.0 * atan2(quat[0], quat[3]);
739  }
740  else {
741  y = atan2(2.0 * (q01 + q23), (q00 + q33) - (q11 + q22));
742  p = qd2 > 1.0 ? D_PI / 2.0 : (qd2 < -1.0 ? -D_PI / 2.0 : asin(qd2));
743  r = atan2(2.0 * (q12 + q03), (q22 + q33) - (q00 + q11));
744  }
745 
746  // convert to degree
747  y = (y / D_PI) * 180;
748  p = (p / D_PI) * 180;
749  r = (r / D_PI) * 180;
750 }

766 bool Rotation::isSame(const Rotation& q, double tol) const
767 {
768  // This follows the implementation of Coin3d where the norm
769  // (x1-y1)**2 + ... + (x4-y4)**2 is computed.
770  // This term can be simplified to
771  // 2 - 2*(x1*y1 + ... + x4*y4) so that for the equality we have to check
772  // 1 - tol/2 <= x1*y1 + ... + x4*y4
773  // This simplification only work if both quats are normalized
774  // Is it safe to assume that?
775  // Because a quaternion (x1,x2,x3,x4) is equal to (-x1,-x2,-x3,-x4) we use the
776  // absolute value of the scalar product
777  double dot =
778  q.quat[0] * quat[0] + q.quat[1] * quat[1] + q.quat[2] * quat[2] + q.quat[3] * quat[3];
779  return fabs(dot) >= 1.0 - tol / 2;
780 }

832 EulerSequence_Parameters translateEulerSequence(const Rotation::EulerSequence theSeq)
833 {
834  const bool F = false;
835  const bool T = true;
836 
837  switch (theSeq) {
838  case Rotation::Extrinsic_XYZ:
839  return {1, F, F, T};
840  case Rotation::Extrinsic_XZY:
841  return {1, T, F, T};
842  case Rotation::Extrinsic_YZX:
843  return {2, F, F, T};
844  case Rotation::Extrinsic_YXZ:
845  return {2, T, F, T};
846  case Rotation::Extrinsic_ZXY:
847  return {3, F, F, T};
848  case Rotation::Extrinsic_ZYX:
849  return {3, T, F, T};
850 
851  // Conversion of intrinsic angles is made by the same code as for extrinsic,
852  // using equivalence rule: intrinsic rotation is equivalent to extrinsic
853  // rotation by the same angles but with inverted order of elemental rotations.
854  // Swapping of angles (Alpha <-> Gamma) is done inside conversion procedure;
855  // sequence of axes is inverted by setting appropriate parameters here.
856  // Note that proper Euler angles (last block below) are symmetric for sequence of axes.
857  case Rotation::Intrinsic_XYZ:
858  return {3, T, F, F};
859  case Rotation::Intrinsic_XZY:
860  return {2, F, F, F};
861  case Rotation::Intrinsic_YZX:
862  return {1, T, F, F};
863  case Rotation::Intrinsic_YXZ:
864  return {3, F, F, F};
865  case Rotation::Intrinsic_ZXY:
866  return {2, T, F, F};
867  case Rotation::Intrinsic_ZYX:
868  return {1, F, F, F};
869 
870  case Rotation::Extrinsic_XYX:
871  return {1, F, T, T};
872  case Rotation::Extrinsic_XZX:
873  return {1, T, T, T};
874  case Rotation::Extrinsic_YZY:
875  return {2, F, T, T};
876  case Rotation::Extrinsic_YXY:
877  return {2, T, T, T};
878  case Rotation::Extrinsic_ZXZ:
879  return {3, F, T, T};
880  case Rotation::Extrinsic_ZYZ:
881  return {3, T, T, T};
882 
883  case Rotation::Intrinsic_XYX:
884  return {1, F, T, F};
885  case Rotation::Intrinsic_XZX:
886  return {1, T, T, F};
887  case Rotation::Intrinsic_YZY:
888  return {2, F, T, F};
889  case Rotation::Intrinsic_YXY:
890  return {2, T, T, F};
891  case Rotation::Intrinsic_ZXZ:

977 void Rotation::setEulerAngles(EulerSequence theOrder,
978  double theAlpha,
979  double theBeta,
980  double theGamma)
981 {
982  if (theOrder == Invalid || theOrder >= EulerSequenceLast) {
983  throw Base::ValueError("invalid euler sequence");
984  }
985 
986  EulerSequence_Parameters o = translateEulerSequence(theOrder);
987 
988  theAlpha *= D_PI / 180.0;
989  theBeta *= D_PI / 180.0;
990  theGamma *= D_PI / 180.0;
991 
992  double a = theAlpha;
993  double b = theBeta;
994  double c = theGamma;
995  if (!o.isExtrinsic) {
996  std::swap(a, c);
997  }
998 
999  if (o.isOdd) {
1000  b = -b;
1001  }
1002 
1003  double ti = 0.5 * a;
1004  double tj = 0.5 * b;
1005  double th = 0.5 * c;
1006  double ci = cos(ti);
1007  double cj = cos(tj);
1008  double ch = cos(th);
1009  double si = sin(ti);
1010  double sj = sin(tj);
1011  double sh = sin(th);
1012  double cc = ci * ch;
1013  double cs = ci * sh;
1014  double sc = si * ch;
1015  double ss = si * sh;
1016 
1017  double values[4]; // w, x, y, z
1018  if (o.isTwoAxes) {
1019  values[o.i] = cj * (cs + sc);
1020  values[o.j] = sj * (cc + ss);
1021  values[o.k] = sj * (cs - sc);
1022  values[0] = cj * (cc - ss);
1023  }
1024  else {
1025  values[o.i] = cj * sc - sj * cs;
1026  values[o.j] = cj * ss + sj * cc;
1027  values[o.k] = cj * cs - sj * sc;
1028  values[0] = cj * cc + sj * ss;
1029  }
1030  if (o.isOdd) {
1031  values[o.j] = -values[o.j];
1032  }
1033 
1034  quat[0] = values[1];
1035  quat[1] = values[2];
1036  quat[2] = values[3];
1037  quat[3] = values[0];
1038  this->evaluateVector();
1039 }

1041 void Rotation::getEulerAngles(EulerSequence theOrder,
1042  double& theAlpha,
1043  double& theBeta,
1044  double& theGamma) const
1045 {
1046  Mat M;
1047  getValue(M);
1048 
1049  EulerSequence_Parameters o = translateEulerSequence(theOrder);
1050  if (o.isTwoAxes) {
1051  double sy = sqrt(M(o.i, o.j) * M(o.i, o.j) + M(o.i, o.k) * M(o.i, o.k));
1052  if (sy > 16 * DBL_EPSILON) {
1053  theAlpha = atan2(M(o.i, o.j), M(o.i, o.k));
1054  theGamma = atan2(M(o.j, o.i), -M(o.k, o.i));
1055  }
1056  else {
1057  theAlpha = atan2(-M(o.j, o.k), M(o.j, o.j));
1058  theGamma = 0.;
1059  }
1060  theBeta = atan2(sy, M(o.i, o.i));
1061  }
1062  else {
1063  double cy = sqrt(M(o.i, o.i) * M(o.i, o.i) + M(o.j, o.i) * M(o.j, o.i));
1064  if (cy > 16 * DBL_EPSILON) {
1065  theAlpha = atan2(M(o.k, o.j), M(o.k, o.k));
1066  theGamma = atan2(M(o.j, o.i), M(o.i, o.i));
1067  }
1068  else {
1069  theAlpha = atan2(-M(o.j, o.k), M(o.j, o.j));
1070  theGamma = 0.;
1071  }
1072  theBeta = atan2(-M(o.k, o.i), cy);
1073  }
1074  if (o.isOdd) {
1075  theAlpha = -theAlpha;
1076  theBeta = -theBeta;
1077  theGamma = -theGamma;
1078  }
1079  if (!o.isExtrinsic) {
1080  double aFirst = theAlpha;
1081  theAlpha = theGamma;
1082  theGamma = aFirst;
1083  }
1084 
1085  theAlpha *= 180.0 / D_PI;
1086  theBeta *= 180.0 / D_PI;
1087  theGamma *= 180.0 / D_PI;
1088 }
