68 // shows error message if the shape is not an edge
69 void ShapeValidator::checkEdge(const TopoDS_Shape& shape)
70 {
71  if (shape.IsNull() || shape.ShapeType() != TopAbs_EDGE) {
72  Standard_Failure::Raise("Shape is not an edge.\n");
73  }
74 
75  TopoDS_Edge etmp = TopoDS::Edge(shape); // Curve TopoDS_Edge
76  TopLoc_Location heloc; // this will be output
77  Standard_Real u0; // contains output
78  Standard_Real u1; // contains output
79  Handle(Geom_Curve) c_geom = BRep_Tool::Curve(etmp, heloc, u0, u1); // The geometric curve
80  Handle(Geom_BezierCurve) bez_geom =
81  Handle(Geom_BezierCurve)::DownCast(c_geom); // Try to get Bezier curve
82 
83  // if not a Bezier then try to create a B-spline surface from the edges
84  if (bez_geom.IsNull()) {
85  willBezier = false;
86  }
87 
88  edgeCount++;
89 }

101 void ShapeValidator::checkAndAdd(const Part::TopoShape& ts,
102  const char* subName,
103  Handle(ShapeExtend_WireData) * aWD)
104 {
105  try {
106  if (subName && *subName != '\0') {
107  // we want only the subshape which is linked
108  checkAndAdd(ts.getSubShape(subName), aWD);
109  }
110  else if (!ts.getShape().IsNull() && ts.getShape().ShapeType() == TopAbs_WIRE) {
111  TopoDS_Wire wire = TopoDS::Wire(ts.getShape());
112  for (TopExp_Explorer xp(wire, TopAbs_EDGE); xp.More(); xp.Next()) {
113  checkAndAdd(xp.Current(), aWD);
114  }
115  }
116  else {
117  checkAndAdd(ts.getShape(), aWD);
118  }
119  }
120  catch (Standard_Failure&) { // any OCC exception means an inappropriate shape in the selection
121  Standard_Failure::Raise("Wrong shape type.\n");
122  }
123 }

163 App::DocumentObjectExecReturn* GeomFillSurface::execute()
164 {
165  try {
166  TopoDS_Wire aWire;
167 
168  // Gets the healed wire
169  if (getWire(aWire)) {
170  createBezierSurface(aWire);
171  }
172  else {
173  createBSplineSurface(aWire);
174  }
175 
176  return App::DocumentObject::StdReturn;
177  }
178  catch (Standard_ConstructionError&) {
179  // message is in a Latin language, show a normal one
180  return new App::DocumentObjectExecReturn("Curves are disjoint.");
181  }
182  catch (StdFail_NotDone&) {
183  return new App::DocumentObjectExecReturn(
184  "A curve was not a B-spline and could not be converted into one.");
185  }
186  catch (Standard_Failure& e) {
187  return new App::DocumentObjectExecReturn(e.GetMessageString());
188  }
189 }

206 bool GeomFillSurface::getWire(TopoDS_Wire& aWire)
207 {
208  Handle(ShapeFix_Wire) aShFW = new ShapeFix_Wire;
209  Handle(ShapeExtend_WireData) aWD = new ShapeExtend_WireData;
210 
211  std::vector<App::PropertyLinkSubList::SubSet> boundary = BoundaryList.getSubListValues();
212  if (boundary.size() > 4) { // if too many not even try
213  Standard_Failure::Raise("Only 2-4 curves are allowed\n");
214  }
215 
216  ShapeValidator validator;
217  for (const auto& set : boundary) {
218  if (set.first->isDerivedFrom<Part::Feature>()) {
219  for (const auto& jt : set.second) {
220  const Part::TopoShape& ts =
221  static_cast<Part::Feature*>(set.first)->Shape.getShape();
222  validator.checkAndAdd(ts, jt.c_str(), &aWD);
223  }
224  }
225  else {
226  Standard_Failure::Raise("Curve not from Part::Feature\n");
227  }
228  }
229 
230  if (validator.numEdges() < 2 || validator.numEdges() > 4) {
231  Standard_Failure::Raise("Only 2-4 curves are allowed\n");
232  }
233 
234  // Reorder the curves and fix the wire if required
235 
236  aShFW->Load(aWD); // Load in the wire
237  aShFW->FixReorder(); // Fix the order of the edges if required
238  aShFW->ClosedWireMode() = Standard_True; // Enables closed wire mode
239  aShFW->FixConnected(); // Fix connection between wires
240  aShFW->FixSelfIntersection(); // Fix Self Intersection
241  aShFW->Perform(); // Perform the fixes
242 
243  aWire = aShFW->Wire(); // Healed Wire
244 
245  if (aWire.IsNull()) {
246  Standard_Failure::Raise("Wire unable to be constructed\n");
247  }
248 
249  return validator.isBezier();
250 }

271 void GeomFillSurface::createBezierSurface(TopoDS_Wire& aWire)
272 {
273  std::vector<Handle(Geom_BezierCurve)> curves;
274  curves.reserve(4);
275 
276  Standard_Real u1, u2; // contains output
277  TopExp_Explorer anExp(aWire, TopAbs_EDGE);
278  for (; anExp.More(); anExp.Next()) {
279  const TopoDS_Edge hedge = TopoDS::Edge(anExp.Current());
280  TopLoc_Location heloc; // this will be output
281  Handle(Geom_Curve) c_geom = BRep_Tool::Curve(hedge, heloc, u1, u2); // The geometric curve
282  Handle(Geom_BezierCurve) bezier =
283  Handle(Geom_BezierCurve)::DownCast(c_geom); // Try to get Bezier curve
284 
285  if (!bezier.IsNull()) {
286  gp_Trsf transf = heloc.Transformation();
287  bezier->Transform(transf); // apply original transformation to control points
288  // Store Underlying Geometry
289  curves.push_back(bezier);
290  }
291  else {
292  Standard_Failure::Raise("Curve not a Bezier Curve");
293  }
294  }
295 
296  GeomFill_FillingStyle fstyle = getFillingStyle();
297  GeomFill_BezierCurves aSurfBuilder; // Create Surface Builder
298 
299  std::size_t edgeCount = curves.size();
300  const boost::dynamic_bitset<>& booleans = ReversedList.getValues();
301  if (edgeCount == booleans.size()) {
302  for (std::size_t i = 0; i < edgeCount; i++) {
303  if (booleans[i]) {
304  curves[i]->Reverse();
305  }
306  }
307  }
308 
309  if (edgeCount == 2) {
310  aSurfBuilder.Init(curves[0], curves[1], fstyle);
311  }
312  else if (edgeCount == 3) {
313  aSurfBuilder.Init(curves[0], curves[1], curves[2], fstyle);
314  }
315  else if (edgeCount == 4) {
316  aSurfBuilder.Init(curves[0], curves[1], curves[2], curves[3], fstyle);
317  }
318 
319  createFace(aSurfBuilder.Surface());
320 }

322 void GeomFillSurface::createBSplineSurface(TopoDS_Wire& aWire)
323 {
324  std::vector<Handle(Geom_BSplineCurve)> curves;
325  curves.reserve(4);
326  Standard_Real u1, u2; // contains output
327  TopExp_Explorer anExp(aWire, TopAbs_EDGE);
328  for (; anExp.More(); anExp.Next()) {
329  const TopoDS_Edge& edge = TopoDS::Edge(anExp.Current());
330  TopLoc_Location heloc; // this will be output
331  Handle(Geom_Curve) c_geom = BRep_Tool::Curve(edge, heloc, u1, u2); // The geometric curve
332  Handle(Geom_BSplineCurve) bspline =
333  Handle(Geom_BSplineCurve)::DownCast(c_geom); // Try to get BSpline curve
334  gp_Trsf transf = heloc.Transformation();
335  if (!bspline.IsNull()) {
336  bspline->Transform(transf); // apply original transformation to control points
337  // Store Underlying Geometry
338  curves.push_back(bspline);
339  }
340  else {
341  // try to convert it into a B-spline
342  Handle(Geom_TrimmedCurve) trim = new Geom_TrimmedCurve(c_geom, u1, u2);
343  // Approximate the curve to non-rational polynomial BSpline
344  // to avoid C0 continuity in output surface
345  GeomConvert conv;
346  Convert_ParameterisationType paratype = Convert_Polynomial;
347  Handle(Geom_BSplineCurve) bspline2 = conv.CurveToBSplineCurve(trim, paratype);
348  if (!bspline2.IsNull()) {
349  bspline2->Transform(transf); // apply original transformation to control points
350  curves.push_back(bspline2);
351  }
352  else {
353  // GeomConvert failed, try ShapeConstruct_Curve now
354  ShapeConstruct_Curve scc;
355  Handle(Geom_BSplineCurve) spline =
356  scc.ConvertToBSpline(c_geom, u1, u2, Precision::Confusion());
357  if (spline.IsNull()) {
358  Standard_Failure::Raise(
359  "A curve was not a B-spline and could not be converted into one.");
360  }
361  spline->Transform(transf); // apply original transformation to control points
362  curves.push_back(spline);
363  }
364  }
365  }
366 
367  GeomFill_FillingStyle fstyle = getFillingStyle();
368  GeomFill_BSplineCurves aSurfBuilder; // Create Surface Builder
369 
370  std::size_t edgeCount = curves.size();
371  const boost::dynamic_bitset<>& booleans = ReversedList.getValues();
372  if (edgeCount == booleans.size()) {
373  for (std::size_t i = 0; i < edgeCount; i++) {
374  if (booleans[i]) {
375  curves[i]->Reverse();
376  }
377  }
378  }
379  if (edgeCount == 2) {
380  aSurfBuilder.Init(curves[0], curves[1], fstyle);
381  }
382  else if (edgeCount == 3) {
383  aSurfBuilder.Init(curves[0], curves[1], curves[2], fstyle);
384  }
385  else if (edgeCount == 4) {
386  aSurfBuilder.Init(curves[0], curves[1], curves[2], curves[3], fstyle);
387  }
388 
389  createFace(aSurfBuilder.Surface());
390 }
