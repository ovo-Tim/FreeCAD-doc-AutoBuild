67 void PropertyEnumAttacherItem::openTask()
68 {
69  Gui::TaskView::TaskDialog* dlg = Gui::Control().activeDialog();
70  TaskDlgAttacher* task;
71  task = qobject_cast<TaskDlgAttacher*>(dlg);
72 
73  if (dlg && !task) {
74  // there is already another task dialog which must be closed first
75  Gui::Control().showDialog(dlg);
76  return;
77  }
78  if (!task) {
79  const App::Property* prop = getFirstProperty();
80  if (prop) {
81  App::PropertyContainer* parent = prop->getContainer();
82 
83  if (parent->isDerivedFrom<App::DocumentObject>()) {
84  App::DocumentObject* obj = static_cast<App::DocumentObject*>(parent);
85  Gui::ViewProvider* view = Gui::Application::Instance->getViewProvider(obj);
86 
87  if (view->isDerivedFrom<Gui::ViewProviderDocumentObject>()) {
88  task = new TaskDlgAttacher(static_cast<Gui::ViewProviderDocumentObject*>(view));
89  }
90  }
91  }
92  if (!task) {
93  return;
94  }
95  }
96 
97  Gui::Control().showDialog(task);
98 }
