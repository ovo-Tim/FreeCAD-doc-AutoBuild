194 class Preferences:
195  """Wrap access to all user preferences. If run within FreeCAD, user preferences are
196  persistent, otherwise they only exist per-run. All preferences are controlled by a
197  central JSON file defining their defaults."""
198 
199  preferences_defaults = {}
200 
201  def __init__(self, defaults_data=None):
202  """Set up the preferences, initializing the class statics if necessary. If
203  defaults_data is provided it is used as the preferences defaults. If it is not
204  provided, then the defaults are read in from the standard defaults file,
205  addonmanager_preferences_defaults.json, located in the same directory as this
206  Python file."""
207  if not self.preferences_defaults:
208  if defaults_data:
209  self.preferences_defaults = defaults_data
210  else:
211  self._load_preferences_defaults()
212  self.prefs = ParamGet("User parameter:BaseApp/Preferences/Addons")
213 
214  def get(self, name: str):
215  """Get the preference value for the given key"""
216  if name not in self.preferences_defaults:
217  raise RuntimeError(
218  f"Unrecognized preference {name} -- did you add "
219  + "it to addonmanager_preferences_defaults.json?"
220  )
221  if isinstance(self.preferences_defaults[name], bool):
222  return self.prefs.GetBool(name, self.preferences_defaults[name])
223  if isinstance(self.preferences_defaults[name], int):
224  return self.prefs.GetInt(name, self.preferences_defaults[name])
225  if isinstance(self.preferences_defaults[name], float):
226  return self.prefs.GetFloat(name, self.preferences_defaults[name])
227  if isinstance(self.preferences_defaults[name], str):
228  return self.prefs.GetString(name, self.preferences_defaults[name])
229  # We don't directly support any other types from the JSON file (e.g. arrays)
230  type_name = type(self.preferences_defaults[name])
231  raise RuntimeError(f"Unrecognized type for {name}: {type_name}")
232 
233  def set(self, name: str, value):
234  """Set the preference value for the given key. Must exist (e.g. must be in the
235  addonmanager_preferences_defaults.json file)."""
236  if name not in self.preferences_defaults:
237  raise RuntimeError(
238  f"Unrecognized preference {name} -- did you add "
239  + "it to addonmanager_preferences_defaults.json?"
240  )
241  if isinstance(self.preferences_defaults[name], bool):
242  self.prefs.SetBool(name, value)
243  elif isinstance(self.preferences_defaults[name], int):
244  self.prefs.SetInt(name, value)
245  elif isinstance(self.preferences_defaults[name], float):
246  self.prefs.SetFloat(name, value)
247  elif isinstance(self.preferences_defaults[name], str):
248  self.prefs.SetString(name, value)
249  else:
250  # We don't directly support any other types from the JSON file (e.g. arrays)
251  type_name = type(self.preferences_defaults[name])
252  raise RuntimeError(f"Unrecognized type for {name}: {type_name}")
253 
254  def rem(self, name: str):

257  if name not in self.preferences_defaults:
258  raise RuntimeError(

264  if isinstance(self.preferences_defaults[name], int):
265  return self.prefs.RemInt(name)
266  if isinstance(self.preferences_defaults[name], float):
267  return self.prefs.RemFloat(name)
268  if isinstance(self.preferences_defaults[name], str):
269  return self.prefs.RemString(name)
270  # We don't directly support any other types from the JSON file (e.g. arrays)

274  @classmethod
275  def _load_preferences_defaults(cls, filename=None):
276  """Loads the preferences defaults JSON file from either a specified file, or
277  from the standard addonmanager_preferences_defaults.json file."""
278
