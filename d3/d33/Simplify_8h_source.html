109 //
110 // target_count : target nr. of triangles
111 // tolerance : tolerance for the quadratic errors
112 // aggressiveness : sharpness to increase the threshold.
113 // 5..8 are good numbers
114 // more iterations yield higher quality
115 // If the passed tolerance is > 0 then this will be used to check
116 // the quadratic error metric of all triangles. If none of them is below
117 // the tolerance the algorithm will stop at this point. The number of the
118 // remaining triangles usually will be higher than \a target_count

120 void Simplify::simplify_mesh(int target_count, double tolerance, double aggressiveness)

123  //printf("%s - start\n",__FUNCTION__);

127  triangles[i].deleted=0;
128 
129  // main iteration loop
130 
131  int deleted_triangles=0;
132  std::vector<int> deleted0,deleted1;
133  int triangle_count=triangles.size();
134 
135  for (int iteration=0;iteration<100;++iteration)
136  {
137  // target number of triangles reached ? Then break
138  //printf("iteration %d - triangles %d\n",iteration,triangle_count-deleted_triangles);

140  break;

143  if (iteration%5==0)

154  //
155  // The following numbers works well for most models.
156  // If it does not, try to adjust the 3 parameters
157  //
158  double threshold = 0.000000001*pow(double(iteration+3),aggressiveness);
159  if (tolerance > 0.0)

161  bool canContinue = false;
162  for (std::size_t i=0;i<triangles.size();++i)
163  {
164  Triangle &t=triangles[i];
165  if (t.deleted)
166  continue;
167  if (t.dirty)
168  continue;
169  if (fabs(t.err[3])<tolerance)
170  {
171  canContinue = true;
172  break;
173  }
174  }
175 
176  if (!canContinue)
177  break;
178  }
179 
180  // remove vertices & mark deleted triangles
181  for (std::size_t i=0;i<triangles.size();++i)
182  {
183  Triangle &t=triangles[i];
184  if (t.err[3]>threshold)
185  continue;
186  if (t.deleted)
187  continue;
188  if (t.dirty)
189  continue;
190

192  {
193  if (t.err[j]<threshold)
194  {
195  int i0=t.v[ j ]; Vertex &v0 = vertices[i0];
196  int i1=t.v[(j+1)%3]; Vertex &v1 = vertices[i1];
197 
198  // Border check

200  continue;
201

203  vec3f p;
204  calculate_error(i0,i1,p);
205 
206  deleted0.resize(v0.tcount); // normals temporarily
207  deleted1.resize(v1.tcount); // normals temporarily
208 
209  // don't remove if flipped

219

221  update_triangles(i0,v1,deleted1,deleted_triangles);

224

226  {
227  // save ram
228  if (tcount)
229  memcpy(&refs[v0.tstart],&refs[tstart],tcount*sizeof(Ref));
230  }
231  else
232  {
233  // append
234  v0.tstart=tstart;
235  }
236 
237  v0.tcount=tcount;
238  break;
239  }
240  }
241 
242  // done?
243  if (triangle_count-deleted_triangles<=target_count)
244  break;
245  }
246  }
247 
248  // clean up mesh
249  compact_mesh();
250 
251  // ready
252  //int timeEnd=timeGetTime();

262 bool Simplify::flipped(vec3f p, int i0, int i1,
263  Vertex &v0,
264  Vertex &v1,
265  std::vector<int> &deleted)
266 {
267  (void)i0; (void)v1;
268  for (int k=0;k<v0.tcount;++k)
269  {
270  Triangle &t=triangles[refs[v0.tstart+k].tid];
271  if (t.deleted)
272  continue;
273 
274  int s=refs[v0.tstart+k].tvertex;
275  int id1=t.v[(s+1)%3];
276  int id2=t.v[(s+2)%3];
277 
278  if (id1==i1 || id2==i1) // delete ?
279  {
280  deleted[k]=1;
281  continue;
282  }
283  vec3f d1 = vertices[id1].p-p; d1.Normalize();
284  vec3f d2 = vertices[id2].p-p; d2.Normalize();
285  if (fabs(d1.Dot(d2))>0.999)
286  return true;
287  vec3f n;
288  n = d1.Cross(d2);
289  n.Normalize();
290  deleted[k]=0;
291  if (n.Dot(t.n)<0.2)
292  return true;
293  }
294  return false;
295 }

318  t.err[2]=calculate_error(t.v[2],t.v[0],p);

326 void Simplify::update_mesh(int iteration)
327 {
328  if(iteration>0) // compact triangles
329  {
330  int dst=0;
331  for (std::size_t i=0;i<triangles.size();++i)
332  {
333  if (!triangles[i].deleted)
334  {
335  triangles[dst++]=triangles[i];
336  }
337  }
338  triangles.resize(dst);
339  }
340  //
341  // Init Quadrics by Plane & Edge Errors
342  //
343  // required at the beginning ( iteration == 0 )
344  // recomputing during the simplification is not required,
345  // but mostly improves the result for closed meshes
346  //
347  if (iteration == 0)
348  {
349  for (std::size_t i=0;i<vertices.size();++i)
350  vertices[i].q=SymmetricMatrix(0.0);
351 
352  for (std::size_t i=0;i<triangles.size();++i)
353  {
354  Triangle &t=triangles[i];
355  vec3f n,p[3];
356  for (std::size_t j=0;j<3;++j)
357  p[j]=vertices[t.v[j]].p;
358  n = (p[1]-p[0]).Cross(p[2]-p[0]);
359  n.Normalize();
360  t.n=n;
361  for (std::size_t j=0;j<3;++j)
362  vertices[t.v[j]].q = vertices[t.v[j]].q+SymmetricMatrix(n.x,n.y,n.z,-n.Dot(p[0]));
363  }
364  for (std::size_t i=0;i<triangles.size();++i)
365  {
366  // Calc Edge Error
367  Triangle &t=triangles[i];vec3f p;
368  for (std::size_t j=0;j<3;++j)
369  t.err[j] = calculate_error(t.v[j],t.v[(j+1)%3],p);
370  t.err[3]=std::min(t.err[0],std::min(t.err[1],t.err[2]));
371  }
372  }
373 
374  // Init Reference ID list
375  for (std::size_t i=0;i<vertices.size();++i)
376  {
377  vertices[i].tstart=0;
378  vertices[i].tcount=0;
379  }
380  for (std::size_t i=0;i<triangles.size();++i)
381  {
382  Triangle &t=triangles[i];
383  for (std::size_t j=0;j<3;++j)
384  vertices[t.v[j]].tcount++;
385  }
386  int tstart=0;
387  for (std::size_t i=0;i<vertices.size();++i)
388  {
389  Vertex &v=vertices[i];
390  v.tstart=tstart;
391  tstart+=v.tcount;
392  v.tcount=0;
393  }
394 
395  // Write References
396  refs.resize(triangles.size()*3);
397  for (std::size_t i=0;i<triangles.size();++i)
398  {
399  Triangle &t=triangles[i];
400  for (std::size_t j=0;j<3;++j)
401  {
402  Vertex &v=vertices[t.v[j]];
403  refs[v.tstart+v.tcount].tid=i;
404  refs[v.tstart+v.tcount].tvertex=j;
405  v.tcount++;
406  }
407  }
408 
409  // Identify boundary : vertices[].border=0,1
410  if (iteration == 0)
411  {
412  std::vector<int> vcount,vids;
413 
414  for (std::size_t i=0;i<vertices.size();++i)
415  vertices[i].border=0;
416 
417  for (std::size_t i=0;i<vertices.size();++i)
418  {
419  Vertex &v=vertices[i];
420  vcount.clear();
421  vids.clear();
422  for (int j=0; j<v.tcount; ++j)
423  {
424  int k=refs[v.tstart+j].tid;
425  Triangle &t=triangles[k];
426  for (int k=0;k<3;++k)
427  {
428  std::size_t ofs=0; int id=t.v[k];
429  while(ofs<vcount.size())
430  {
431  if (vids[ofs]==id)
432  break;
433  ofs++;
434  }
435  if(ofs==vcount.size())
436  {
437  vcount.push_back(1);
438  vids.push_back(id);
439  }
440  else
441  {
442  vcount[ofs]++;
443  }
444  }
445  }
446  for (std::size_t j=0;j<vcount.size();++j) {
447  if (vcount[j]==1)
448  vertices[vids[j]].border=1;
449  }
450  }
451  }
452 }

504 double Simplify::calculate_error(int id_v1, int id_v2, vec3f &p_result)
505 {
506  // compute interpolated vertex
507 
508  SymmetricMatrix q = vertices[id_v1].q + vertices[id_v2].q;
509  bool border = vertices[id_v1].border & vertices[id_v2].border;
510  double error=0;
511  double det = q.det(0, 1, 2, 1, 4, 5, 2, 5, 7);
512 
513  if (det != 0 && !border)
514  {
515  // q_delta is invertible
516  p_result.x = -1/det*(q.det(1, 2, 3, 4, 5, 6, 5, 7 , 8)); // vx = A41/det(q_delta)
517  p_result.y = 1/det*(q.det(0, 2, 3, 1, 5, 6, 2, 7 , 8)); // vy = A42/det(q_delta)
518  p_result.z = -1/det*(q.det(0, 1, 3, 1, 4, 6, 2, 5, 8)); // vz = A43/det(q_delta)
519  error = vertex_error(q, p_result.x, p_result.y, p_result.z);
520  }
521  else
522  {
523  // det = 0 -> try to find best result
524  vec3f p1=vertices[id_v1].p;
525  vec3f p2=vertices[id_v2].p;
526  vec3f p3=(p1+p2)/2;
527  double error1 = vertex_error(q, p1.x,p1.y,p1.z);
528  double error2 = vertex_error(q, p2.x,p2.y,p2.z);
529  double error3 = vertex_error(q, p3.x,p3.y,p3.z);
530  error = std::min(error1, std::min(error2, error3));
531  if (error1 == error)
532  p_result=p1;
533  if (error2 == error)
534  p_result=p2;
535  if (error3 == error)
536  p_result=p3;
537  }
538  return error;
539 }
