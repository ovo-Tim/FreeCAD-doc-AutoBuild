49 std::string humanReadableSize(unsigned int bytes)
50 {
51  static const std::vector<std::string> siPrefix {
52  "b",
53  "kb",
54  "Mb",
55  "Gb",
56  "Tb",
57  "Pb",
58  "Eb" // I think it's safe to stop here (for the time being)...
59  };
60  size_t base = 0;
61  double inUnits = bytes;
62  constexpr double siFactor {1000.0};
63  while (inUnits > siFactor && base < siPrefix.size() - 1) {
64  ++base;
65  inUnits /= siFactor;
66  }
67  if (base == 0) {
68  // Don't include a decimal point for bytes
69  return fmt::format("{:.0f} {}", inUnits, siPrefix[base]);
70  }
71  // For all others, include one digit after the decimal place
72  return fmt::format("{:.1f} {}", inUnits, siPrefix[base]);
73 }

125 QString getMD5Hash(const std::string& path)
126 {
127  // Use MD5 hash as specified here:
128  // https://specifications.freedesktop.org/thumbnail-spec/0.8.0/thumbsave.html
129  QUrl url(QString::fromStdString(path));
130  url.setScheme(QString::fromLatin1("file"));
131  QCryptographicHash hash(QCryptographicHash::Md5);
132  hash.addData(url.toEncoded());
133  QByteArray ba = hash.result().toHex();
134  return QString::fromLatin1(ba);
135 }

160 QByteArray loadFCStdThumbnail(const std::string& pathToFCStdFile)
161 {
162  App::ProjectFile proj(pathToFCStdFile);
163  if (proj.loadDocument()) {
164  try {
165  std::string thumbnailFile = getUniquePNG(pathToFCStdFile).toStdString();
166  if (!useCachedPNG(thumbnailFile, pathToFCStdFile)) {
167  static std::string thumb = getThumbnailsImage();
168  if (proj.containsFile(thumb)) {
169  createThumbnailsDir();
170  Base::FileInfo fi(thumbnailFile);
171  Base::ofstream str(fi, std::ios::out | std::ios::binary);
172  proj.readInputFileDirect(thumb, str);
173  str.close();
174  }
175  }
176 
177  auto inputFile = QFile(QString::fromStdString(thumbnailFile));
178  if (inputFile.exists()) {
179  inputFile.open(QIODevice::OpenModeFlag::ReadOnly);
180  return inputFile.readAll();
181  }
182  }
183  catch (...) {
184  }
185  }
186  return {};
187 }

189 FileStats getFileInfo(const std::string& path)
190 {
191  FileStats result;
192  Base::FileInfo file(path);
193  if (file.hasExtension("FCStd")) {
194  result = fileInfoFromFreeCADFile(path);

196  else {
197  file.lastModified();
198  }
199  result.insert(std::make_pair(DisplayedFilesModelRoles::path, path));
200  result.insert(std::make_pair(DisplayedFilesModelRoles::size, humanReadableSize(file.size())));
201  result.insert(std::make_pair(DisplayedFilesModelRoles::baseName, file.fileName()));
202  return result;

230 QVariant DisplayedFilesModel::data(const QModelIndex& index, int roleAsInt) const
231 {
232  int row = index.row();
233  if (row < 0 || row >= static_cast<int>(_fileInfoCache.size())) {
234  return {};
235  }
236  auto mapEntry = _fileInfoCache.at(row);
237  auto role = static_cast<DisplayedFilesModelRoles>(roleAsInt);
238  switch (role) {
239  case DisplayedFilesModelRoles::author: // NOLINT(bugprone-branch-clone)
240  [[fallthrough]];
241  case DisplayedFilesModelRoles::baseName:
242  [[fallthrough]];
243  case DisplayedFilesModelRoles::company:
244  [[fallthrough]];
245  case DisplayedFilesModelRoles::creationTime:
246  [[fallthrough]];
247  case DisplayedFilesModelRoles::description:
248  [[fallthrough]];
249  case DisplayedFilesModelRoles::license:
250  [[fallthrough]];
251  case DisplayedFilesModelRoles::modifiedTime:
252  [[fallthrough]];
253  case DisplayedFilesModelRoles::path:
254  [[fallthrough]];
255  case DisplayedFilesModelRoles::size:
256  if (mapEntry.find(role) != mapEntry.end()) {
257  return QString::fromStdString(mapEntry.at(role));
258  }
259  else {
260  return {};
261  }
262  case DisplayedFilesModelRoles::image: {
263  auto path = QString::fromStdString(mapEntry.at(DisplayedFilesModelRoles::path));
264  if (_imageCache.contains(path)) {
265  return _imageCache[path];
266  }
267  break;
268  }
269  default:
270  break;
271  }
272  switch (roleAsInt) {
273  case Qt::ItemDataRole::ToolTipRole:
274  return QString::fromStdString(mapEntry.at(DisplayedFilesModelRoles::path));
275  }
276  return {};
277 }

279 void DisplayedFilesModel::addFile(const QString& filePath)
280 {
281  QFileInfo qfi(filePath);
282  if (!qfi.isReadable()) {
283  return;
284  }
285  if (!freecadCanOpen(qfi.suffix())) {
286  return;
287  }
288  _fileInfoCache.emplace_back(getFileInfo(filePath.toStdString()));
289  if (qfi.suffix().toLower() == QLatin1String("fcstd")) {
290  auto thumbnail = loadFCStdThumbnail(filePath.toStdString());
291  if (!thumbnail.isEmpty()) {
292  _imageCache.insert(filePath, thumbnail);
293  }
294  }
