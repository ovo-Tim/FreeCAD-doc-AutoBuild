170 void TaskCustomizeFormat::onFormatChanged()
171 {
172  // Slot: the LineEdit field has been changed
173  QString formatPreview = ui->leFormat->text();
174  if (isDimension)
175  {
176  constexpr int size(80);
177  char buffer[size];
178  std::string formatString = formatPreview.toUtf8().constData();
179  auto usedSize = snprintf(buffer, size, formatString.c_str(), dimRawValue);
180  formatPreview = QString::fromUtf8(buffer, usedSize);
181  }
182  ui->lbShowPreview->setText(formatPreview);
183 }

185 bool TaskCustomizeFormat::accept()
186 {
187  // Slot: the OK button has been pressed
188  QString formatPreview = ui->leFormat->text();
189  std::string formatString = formatPreview.toUtf8().constData();
190  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Customize Format"));
191  if (isDimension)
192  {
193  auto dim = dynamic_cast<TechDraw::DrawViewDimension*>(selectedObject);
194  dim->FormatSpec.setValue(formatString);
195  }
196  else
197  {
198  auto balloon = dynamic_cast<TechDraw::DrawViewBalloon*>(selectedObject);
199  balloon->Text.setValue(formatString);
200  }
201  Gui::Command::commitCommand();
202  return true;
203 }
