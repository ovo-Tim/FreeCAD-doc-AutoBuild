155 std::vector<MeasureType*> MeasureManager::getValidMeasureTypes(App::MeasureSelection selection,
156  std::string mode)
157 {
158  Base::PyGILStateLocker lock;
159 
160  // Convert selection to python list
161  Py::Tuple selectionPy = getSelectionPy(selection);
162 
163  // Store valid measure types
164  std::vector<MeasureType*> validTypes;
165  std::pair<int, MeasureType>();
166 
167 
168  // Loop through measure types and check if they work with given selection
169  for (App::MeasureType* mType : getMeasureTypes()) {
170 
171  if (mode != "" && mType->label != mode) {
172  continue;
173  }
174 
175 
176  if (mType->isPython) {
177  // Parse Python measure types
178  auto measurePyClass = Py::Object(mType->pythonClass);
179 
180  Py::Tuple args(1);
181  args.setItem(0, selectionPy);
182 
183  Py::Object isValid;
184  try {
185  isValid = measurePyClass.callMemberFunction(std::string("isValidSelection"), args);
186  }
187  catch (const Py::Exception&) {
188  Base::PyException e;
189  e.ReportException();
190  isValid = Py::False();
191  }
192 
193  if (isValid.as_bool()) {
194 
195  // Check priority
196  Py::Object isPriority;
197  try {
198  isPriority = measurePyClass.callMemberFunction("isPrioritySelection", args);
199  }
200  catch (const Py::Exception&) {
201  Base::PyException e;
202  e.ReportException();
203  isPriority = Py::False();
204  }
205 
206  if (isPriority.as_bool()) {
207  validTypes.insert(validTypes.begin(), mType);
208  }
209  else {
210  validTypes.push_back(mType);
211  }
212  }
213  }
214  else {
215  // Parse c++ measure types
216 
217  if (mType->validatorCb && !mType->validatorCb(selection)) {
218  continue;
219  }
220 
221  // Check if the measurement type prioritizes the given selection
222  if (mType->prioritizeCb && mType->prioritizeCb(selection)) {
223  validTypes.insert(validTypes.begin(), mType);
224  }
225  else {
226  validTypes.push_back(mType);
227  }
228  }
229  }
230 
231  return validTypes;
232 }
