39 class FirstRunDialog:
40  """Manage the display of the Addon Manager's first-run dialog, setting up some user
41  preferences and making sure they are aware that this connects to the internet, downloads
42  data, and possibly installs things that run code not affiliated with FreeCAD itself."""
43 
44  def __init__(self):
45  self.pref = FreeCAD.ParamGet("User parameter:BaseApp/Preferences/Addons")
46  self.readWarning = self.pref.GetBool("readWarning2022", False)
47 
48  def exec(self) -> bool:
49  """Display a first-run dialog if needed, and return True to indicate the Addon Manager
50  should continue loading, or False if the user cancelled the dialog and wants to exit."""
51  if not self.readWarning:
52  warning_dialog = FreeCADGui.PySideUic.loadUi(
53  os.path.join(os.path.dirname(__file__), "first_run.ui")
54  )
55 
56  # Set signal handlers for accept/reject buttons
57  warning_dialog.buttonContinue.clicked.connect(warning_dialog.accept)
58  warning_dialog.buttonQuit.clicked.connect(warning_dialog.reject)
59 
60  # Show the dialog and check whether the user accepted or canceled
61  if warning_dialog.exec() == QtWidgets.QDialog.Accepted:
62  # Store warning as read/accepted
63  self.readWarning = True
64  self.pref.SetBool("readWarning2022", True)
65 
66  return self.readWarning
