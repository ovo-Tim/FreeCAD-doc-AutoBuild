40 {
41  // offset kurve with simple span elimination
42  // direction 1 = left, -1 = right
43 
44  // ret = 0 - kurve offset ok
45  // = 1 - kurve has differential scale (not allowed)
46  // = 2 - offset failed
47  // = 3 - offset too large
48  if (this == &kOffset) {
49  FAILURE(L"Illegal Call - 'this' must not be kOffset");
50  }
51  double offset = (direction == GEOFF_LEFT) ? off : -off;
52 
53  if (fabs(offset) < geoff_geometry::TOLERANCE || m_nVertices < 2) {
54  kOffset = *this;
55  ret = 0;
56  return 1;
57  }
58

60  Span prevSpanOff; // previous offset span

62

65 
66  if (m_mirrored) {
67  offset = -offset;
68  }
69  int RollDir = (off < 0) ? direction : -direction; // Roll arc direction
70 
71  double scalex;
72  if (!GetScale(scalex)) {
73  ret = 1;
74  return 0; // differential scale
75  }
76  offset /= scalex;
77

79  int nspans = nSpans();

81  Get(nspans, curSpan, true); // assign previous span for closed
82

84  nspans++; // read first again
85  }
86

88  if (spannumber > nSpans()) {
89  Get(1, curSpan, true); // closed kurve - read first span again
90  }
91  else {
92  Get(spannumber, curSpan, true);
93  }
94

96  int numint = 0;
97  curSpanOff = curSpan.Offset(offset);
98  curSpanOff.ID = 0;
99  if (!kOffset.m_started) {
100  kOffset.Start(curSpanOff.p0);
101  kOffset.AddSpanID(0);
102  }
103 
104  if (spannumber > 1) {
105  // see if tangent
106  double d = curSpanOff.p0.Dist(prevSpanOff.p1);
107  if ((d > geoff_geometry::TOLERANCE)
108  && (!curSpanOff.NullSpan && !prevSpanOff.NullSpan)) {
109  // see if offset spans intersect
110 
111  double cp = prevSpanOff.ve ^ curSpanOff.vs;
112  bool inters =
113  (cp > 0 && direction == GEOFF_LEFT) || (cp < 0 && direction == GEOFF_RIGHT);
114 
115  if (inters) {
116  double t[4];
117  numint = prevSpanOff.Intof(curSpanOff, p0, p1, t);
118  }
119 
120  if (numint == 1) {
121  // intersection - modify previous endpoint
122  kOffset.Replace(kOffset.m_nVertices - 1,
123  prevSpanOff.dir,
124  p0,
125  prevSpanOff.pc,
126  prevSpanOff.ID);
127  }
128  else {
129  // 0 or 2 intersections, add roll around (remove -ve loops in elimination
130  // function)
131  if (kOffset.Add(RollDir, curSpanOff.p0, curSpan.p0, false)) {
132  kOffset.AddSpanID(ROLL_AROUND);
133  }
134  }
135  }
136  }
137 
138  // add span
139  if (spannumber < m_nVertices) {
140  curSpanOff.ID = spannumber;
141  kOffset.Add(curSpanOff, false);
142  }
143  else if (numint == 1) { // or replace the closed first span
144  kOffset.Replace(0, 0, p0, Point(0, 0), 0);
145  }
146  }
147  if (!curSpanOff.NullSpan) {
148  prevSpanOff = curSpanOff;
149  }
150  } // end of main pre-offsetting loop
151 
152 
153 #ifdef _DEBUG
154 // testDraw->AddKurve("", &kOffset, 0, GREEN);
155 // outXML oxml(L"c:\\temp\\eliminateLoops.xml");
156 // oxml.startElement(L"eliminateLoops");

160  // eliminate loops
161  if (method == NO_ELIMINATION) {
162  ret = 0;
163  return 1;
164  }
165  kOffset = eliminateLoops(kOffset, *this, offset, ret);
166 
167  if (ret == 0 && bClosed) {
168  // check for inverted offsets of closed kurves
169  if (kOffset.Closed()) {
170  double a = Area();
171  int dir = (a < 0);
172  double ao = kOffset.Area();
173  int dirOffset = ao < 0;
174 
175  if (dir != dirOffset) {
176  ret = 3;
177  }
178  else {

180  bool bigger = (a > 0 && offset > 0) || (a < 0 && offset < 0);
181  if (bigger && fabs(ao) < fabs(a)) {
182  ret = 2;
183  }
184  }
185  }
186  else {
187  ret = 2; // started closed but now open??
188  }
189  }
190  return (ret == 0) ? 1 : 0;
191 }

194 static Kurve eliminateLoops(const Kurve& k, const Kurve& originalk, double offset, int& ret)
195 {
196  // a simple loop elimination routine based on first offset ideas in Peps
197  // this needs extensive work for future
198  // start point mustn't disappear & only one valid offset is determined
199  //
200  // ret = 0 for ok
201  // ret = 2 for impossible geometry
202 
203  Span sp0, sp1;
204  Point pInt, pIntOther;
205 
206  Kurve ko; // eliminated output
207  ko = Matrix(k);
208  int kinVertex = 0;
209 
210  while (kinVertex <= k.nSpans()) {
211  bool clipped = false; // not in a clipped section (assumption with this simple method)
212 
213  sp0.dir = k.Get(kinVertex, sp0.p0, sp0.pc);
214  sp0.ID = k.GetSpanID(kinVertex++);
215  if (kinVertex == 1) {
216  ko.Start(sp0.p0); // start point mustn't disappear for this simple method
217  ko.AddSpanID(sp0.ID);
218  }
219  if (kinVertex <= k.nSpans()) { // any more?
220  int ksaveVertex = kinVertex;
221  sp0.dir = k.Get(kinVertex, sp0.p1, sp0.pc); // first span
222  sp0.ID = k.GetSpanID(kinVertex++);
223 
224  sp0.SetProperties(true);
225 
226  int ksaveVertex1 = kinVertex; // mark position AA
227  if (kinVertex <= k.nSpans()) { // get the next but one span
228  sp1.dir = k.Get(kinVertex, sp1.p0, sp1.pc);
229  sp1.ID = k.GetSpanID(kinVertex++);
230  int ksaveVertex2 = kinVertex; // mark position BB
231 
232  int fwdCount = 0;
233  while (kinVertex <= k.nSpans()) {
234  sp1.dir = k.Get(kinVertex, sp1.p1, sp1.pc); // check span
235  sp1.ID = k.GetSpanID(kinVertex++);
236  sp1.SetProperties(true);
237 
238  double t[4];
239  int numint = sp0.Intof(sp1, pInt, pIntOther, t); // find span intersections
240  if (numint && sp0.p0.Dist(pInt) < geoff_geometry::TOLERANCE) {
241  numint =
242  0; // check that intersection is not at the start of the check span
243  }
244  if (numint) {
245 
246  if (numint == 2) {
247  // choose first intercept on sp0
248  Span spd = sp0;
249  spd.p1 = pInt;
250  spd.SetProperties(true);
251  double dd = spd.length;
252 
253  spd.p1 = pIntOther;
254  spd.SetProperties(true);
255  if (dd > spd.length) {
256  pInt = pIntOther;
257  }
258  numint = 1;
259  }
260  ksaveVertex = ksaveVertex1;
261 
262  clipped = true; // in a clipped section
263  if (!DoesIntersInterfere(pInt, originalk, offset)) {
264  sp0.p1 = pInt; // ok so truncate this span to the intersection
265  clipped = false; // end of clipped section
266  break;
267  }
268  // no valid intersection found so carry on
269  }
270  sp1.p0 = sp1.p1; // next
271  ksaveVertex1 = ksaveVertex2; // pos AA = BB
272  ksaveVertex2 = kinVertex; // mark
273 
274  if ((kinVertex > k.nSpans() || fwdCount++ > 25) && !clipped) {
275  break;
276  }
277  }
278  }
279 
280  if (clipped) {
281  ret = 2; // still in a clipped section - error
282 
283  return ko;
284  }
285 
286  ko.Add(sp0, false);
287 
288  kinVertex = ksaveVertex;
289  }
290  }
291  ret = 0;
292 
293  return ko; // no more spans - seems ok
294 }

297 static bool DoesIntersInterfere(const Point& pInt, const Kurve& k, double offset)
298 {
299  // check that intersections don't interfere with the original kurve
300  Span sp;
301  Point dummy;
302  int kCheckVertex = 0;
303  k.Get(kCheckVertex++, sp.p0, sp.pc);
304 
305  offset = fabs(offset) - geoff_geometry::TOLERANCE;
306  while (kCheckVertex <= k.nSpans()) {
307  sp.dir = k.Get(kCheckVertex++, sp.p1, sp.pc);
308  sp.SetProperties(true);
309  // check for interference
310  if (Dist(sp, pInt, dummy) < offset) {
311  return true;
312  }
313  sp.p0 = sp.p1;
314  }
315  return false; // intersection is ok
316 }

327 int Kurve::OffsetISOMethod(Kurve& kOut, double off, int direction, bool BlendAll) const
328 {
329  // produces a special offset Kurve - observing so-called ISO radii
330  // eg line/arc/line tangent - keep arc radius constant
331  // this method also considers arc/arc/arc etc.
332  // interior radius must be smallest of triplet for above.
333 
334  // parameters:-
335  // Output kOut resulting kurve
336  // Input off offset amount
337  // Input direction offset direction (LEFT or RIGHT)
338  // Input BlendAall if false only consider ISO radius for LINE/ARC/LINE
339  // if true consider all blended radii (ARC/ARC/ARC etc.)
340  double offset = (direction == GEOFF_LEFT) ? off : -off;
341  if (FEQZ(off) || nSpans() < 1) {
342  kOut = *this;
343  return 1;
344  }
345  double cptol = 1.0e-05;
346  std::vector<iso> spans;
347  for (int i = 0; i < nSpans(); i++) { // store all spans and offsets
348  Get(i + 1, isodata.sp, true, true);
349  isodata.off = isodata.sp.Offset(offset);
350  spans.push_back(isodata);
351  }
352 
353  for (int i = 0; i < nSpans() - 1; i++) { // calculate intersections for none tangent spans
354  if (fabs(spans[i].off.ve ^ spans[i + 1].off.vs) > cptol) {
355  spans[i].off.JoinSeparateSpans(spans[i + 1].off);
356  }
357  }
358 
359  for (int i = 1; i < nSpans() - 1; i++) { // deal with isoradii
360  if (spans[i].off.dir) {
361  if (BlendAll) { // interior radius should be smaller than neighbours
362  if (spans[i - 1].sp.dir) {
363  if (spans[i - 1].sp.radius < spans[i].sp.radius) {
364  continue;
365  }
366  }
367  if (spans[i + 1].sp.dir) {
368  if (spans[i + 1].sp.radius < spans[i].sp.radius) {
369  continue;
370  }
371  }
372  }
373  else {
374  if ((spans[i - 1].off.dir || spans[i + 1].off.dir)) {
375  continue; // linear neighbours only
376  }
377  }
378 
379  if ((fabs(spans[i - 1].sp.ve ^ spans[i].sp.vs) < cptol)
380  && (fabs(spans[i].sp.ve ^ spans[i + 1].sp.vs) < cptol)) {
381  // isoradius - calculate the new offset radius and modify neighbouring spans
382  isoRadius(spans[i - 1].off, spans[i].off, spans[i + 1].off, spans[i].sp.radius);
383  }
384  }
385  }
386 
387  kOut.Start(spans[0].off.p0); // start point
388  for (int i = 0; i < nSpans(); i++) {
389  kOut.Add(spans[i].off.dir, spans[i].off.p1, spans[i].off.pc); // output all spans
390  }
391  return 1;
392 }

394 static void isoRadius(Span& before, Span& blend, Span& after, double radius)
395 {
396  // calculate the new offset radius and modify neighbouring spans
397  int direction = ((before.ve ^ after.vs) > 0) ? 1 : -1; // offset direction
398  Span beforeOff = before.Offset(direction * radius);
399  Span afterOff = after.Offset(direction * radius);
400  int turnLeft = ((before.ve ^ after.vs) > 0) ? 1 : -1;
401  if (before.dir == LINEAR) {
402  CLine b(beforeOff);
403  if (after.dir == LINEAR) {
404  CLine a(afterOff);
405  blend.pc = b.Intof(a);
406  }
407  else {
408  Circle a(afterOff);
409  b.Intof(turnLeft * after.dir, a, blend.pc);
410  }
411  }
412  else {
413  Circle b(beforeOff);
414 
415  if (after.dir == LINEAR) {
416  CLine a(afterOff);
417  a.Intof(-turnLeft * before.dir, b, blend.pc);
418  }
419  else {
420  // arc arc

423  b.Intof(leftright, a, blend.pc);
424  }
425  }

427  after.p0 = blend.p1 = after.Near(blend.pc);
