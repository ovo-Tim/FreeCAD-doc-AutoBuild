193 void DlgSettingsImageImp::adjustImageSize(float fRatio)
194 {
195  // if width has changed then adjust height and vice versa, if both has changed then adjust width
196  if (_height != ui->spinHeight->value()) {
197  _height = ui->spinHeight->value();
198  _width = (int)((float)_height * fRatio);
199  ui->spinWidth->setValue((int)_width);
200  }
201  else // if( _width != spinWidth->value() )
202  {
203  _width = ui->spinWidth->value();
204  _height = (int)((float)_width / fRatio);
205  ui->spinHeight->setValue((int)_height);
206  }
207 }

229 void DlgSettingsImageImp::onStandardSizeBoxActivated(int index)
230 {
231  if (index == 0) {
232  // we have set the user data for the 1st item
233  QSize s = ui->standardSizeBox->itemData(0).toSize();
234  ui->spinWidth->setValue(s.width());
235  ui->spinHeight->setValue(s.height());
236  }
237  else {
238  // try to extract from the string
239  QString text = ui->standardSizeBox->itemText(index);
240  QRegularExpression rx(QLatin1String(R"(\b\d{2,5}\b)"));
241  int pos = 0;
242  auto match = rx.match(text, pos);
243  if (match.hasMatch()) {
244  pos = match.capturedStart();
245  QString width = text.mid(pos, match.capturedLength());
246  ui->spinWidth->setValue(width.toInt());
247  pos += match.capturedLength();
248  }
249 
250  match = rx.match(text, pos);
251  if (match.hasMatch()) {
252  pos = match.capturedStart();
253  QString height = text.mid(pos, match.capturedLength());
254  ui->spinHeight->setValue(height.toInt());
255  }
256  }
257 }
