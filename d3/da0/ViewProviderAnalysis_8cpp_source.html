125 bool ViewProviderFemAnalysis::doubleClicked()
126 {
127  Gui::Command::assureWorkbench("FemWorkbench");
128  Gui::Command::addModule(Gui::Command::Gui, "FemGui");
129  Gui::Command::doCommand(Gui::Command::Gui,
130  "FemGui.setActiveAnalysis(App.activeDocument().%s)",
131  this->getObject()->getNameInDocument());
132  // After activation of the analysis the allowed FEM toolbar buttons should become active.
133  // To achieve this we must clear the object selection to trigger the selection observer.
134  Gui::Command::doCommand(Gui::Command::Gui, "Gui.Selection.clearSelection()");

136  // especially useful for files with 2 or more analyses but also
137  // necessary for the workflow with new files to add a solver as next object

139  selVector.push_back(this->getObject());
140  auto* docName = this->getObject()->getDocument()->getName();
141  Gui::Selection().setSelection(docName, selVector);
142  return true;

294 bool ViewProviderFemAnalysis::checkSelectedChildren(const std::vector<App::DocumentObject*> objs,
295  Gui::Document* docGui,
296  std::string objectName)
297 {
298  // warn the user if the object has unselected children
299  if (!objs.empty()) {
300  // check if all children are in the selection
301  bool found = false;
302  auto selectionList = Gui::Selection().getSelectionEx(docGui->getDocument()->getName());
303  for (auto child : objs) {
304  found = false;
305  for (Gui::SelectionObject selection : selectionList) {
306  if (std::string(child->getNameInDocument())
307  == std::string(selection.getFeatName())) {
308  found = true;
309  break;
310  }
311  }
312  if (!found) {
313  break;
314  }
315  }
316  if (found) { // all children are selected too
317  return true;
318  }
319 
320  // generate dialog
321  QString bodyMessage;
322  QTextStream bodyMessageStream(&bodyMessage);
323  bodyMessageStream << qApp->translate("Std_Delete",
324  ("The " + objectName
325  + " is not empty, therefore the\nfollowing "
326  "referencing objects might be lost:")
327  .c_str());
328  bodyMessageStream << '\n';
329  for (auto ObjIterator : objs) {
330  bodyMessageStream << '\n' << QString::fromUtf8(ObjIterator->Label.getValue());
331  }
332  bodyMessageStream << "\n\n" << QObject::tr("Are you sure you want to continue?");
333  // show and evaluate the dialog
334  int DialogResult =
335  QMessageBox::warning(Gui::getMainWindow(),
336  qApp->translate("Std_Delete", "Object dependencies"),
337  bodyMessage,
338  QMessageBox::Yes,
339  QMessageBox::No);
340  if (DialogResult == QMessageBox::Yes) {
341  return true;
342  }
343  else {
344  return false;
345  }
346  }
347  else {
348  return true;
349  }
350 }
