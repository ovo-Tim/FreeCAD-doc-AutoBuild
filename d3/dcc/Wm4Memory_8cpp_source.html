59 void* Memory::Allocate (size_t uiSize, char* acFile, unsigned int uiLine,
60  bool bIsArray)
61 {
62  ms_uiNumNewCalls++;
63 
64  // The 'assert' used to be enabled to trap attempts to allocate zero
65  // bytes. However, the DirectX function D3DXCheckNewDelete may pass in
66  // a value of zero.
67  // assert(uiSize > 0);
68 
69  // Allocate additional storage for the block header information.
70  size_t uiExtendedSize = sizeof(Block) + uiSize;
71  char* pcAddr = (char*)malloc(uiExtendedSize);
72 
73  // Save the allocation information.
74  Block* pkBlock = (Block*)pcAddr;
75  pkBlock->Size = uiSize;
76  pkBlock->File = acFile;
77  pkBlock->Line = uiLine;
78  pkBlock->IsArray = bIsArray;
79  InsertBlock(pkBlock);
80 
81  // Move the pointer to the start of what the user expects from 'new'.
82  pcAddr += sizeof(Block);
83 
84  // Keep track of number of allocated blocks and bytes.
85  ms_uiNumBlocks++;
86  ms_uiNumBytes += uiSize;
87 
88  if (ms_uiMaxAllowedBytes > 0 && ms_uiNumBytes > ms_uiMaxAllowedBytes)
89  {
90  // The allocation has exceeded the maximum number of bytes.
91  assert(false);
92  }
93 
94  // Keep track of the maximum number of bytes allocated.
95  if (ms_uiNumBytes > ms_uiMaxAllocatedBytes)
96  {
97  ms_uiMaxAllocatedBytes = ms_uiNumBytes;
98  }
99 
100  // Keep track of the distribution of sizes for allocations.
101  if (ms_bTrackSizes)
102  {
103  // Keep track of the largest block ever allocated.
104  if (uiSize > ms_uiMaxBlockSize)
105  {
106  ms_uiMaxBlockSize = uiSize;
107  }
108 
109  unsigned int uiTwoPowerI = 1;
110  int i;
111  for (i = 0; i <= 30; i++, uiTwoPowerI <<= 1)
112  {
113  if (uiSize <= uiTwoPowerI)
114  {
115  ms_auiHistogram[i]++;
116  break;
117  }
118  }
119  if (i == 31)
120  {
121  ms_auiHistogram[i]++;
122  }
123  }
124 
125  return (void*)pcAddr;
126 }
127 //----------------------------------------------------------------------------
128 void Memory::Deallocate (char* pcAddr, bool bIsArray)
129 {
130  ms_uiNumDeleteCalls++;
131 
132  if (!pcAddr)
133  {
134  return;
135  }
136 
137  // Move the pointer to the start of the actual allocated block.
138  pcAddr -= sizeof(Block);
139 
140  // Get the allocation information and remove the block. The removal
141  // only modifies the Prev and Next pointers, so the block information is
142  // accessible after the call.
143  Block* pkBlock = (Block*)pcAddr;
144  RemoveBlock(pkBlock);
145 
146 #ifdef WM4_ENABLE_NEW_DELETE_MISMATCH_ASSERT
147  // Check for correct pairing of new/delete or new[]/delete[].
148  assert(pkBlock->IsArray == bIsArray);
149 #endif
150 
151  // Keep track of number of allocated blocks and bytes. If the number of
152  // blocks is zero at this time, a delete has been called twice on the
153  // same pointer. If the number of bytes is too small at this time, some
154  // internal problem has occurred within this class and needs to be
155  // diagnosed.
156  assert(ms_uiNumBlocks > 0 && ms_uiNumBytes >= pkBlock->Size);
157  ms_uiNumBlocks--;
158  ms_uiNumBytes -= pkBlock->Size;
159 
160  // Deallocate the memory block.
161  free(pcAddr);
162 }
163 //----------------------------------------------------------------------------
164 void Memory::InsertBlock (Block* pkBlock)
165 {
166  // New blocks are inserted at the tail of the doubly linked list.
167  if (ms_pkTail)
168  {
169  pkBlock->Prev = ms_pkTail;
170  pkBlock->Next = 0;
171  ms_pkTail->Next = pkBlock;
172  ms_pkTail = pkBlock;
173  }
174  else
175  {
176  pkBlock->Prev = 0;
177  pkBlock->Next = 0;
178  ms_pkHead = pkBlock;
179  ms_pkTail = pkBlock;
180  }
181 }
182 //----------------------------------------------------------------------------
183 void Memory::RemoveBlock (Block* pkBlock)
184 {
185  if (pkBlock->Prev)
186  {
187  pkBlock->Prev->Next = pkBlock->Next;
188  }
189  else
190  {
191  ms_pkHead = pkBlock->Next;
192  }
193 
194  if (pkBlock->Next)
195  {
196  pkBlock->Next->Prev = pkBlock->Prev;
197  }
198  else
199  {
200  ms_pkTail = pkBlock->Prev;
201  }
202 }

204 void Memory::GenerateReport (const char* acFilename)
205 {
206  std::ofstream kOStr(acFilename);
207  assert(kOStr);
208  if (!kOStr)
209  {
210  return;
211  }
212 
213  // Total calls.
214  kOStr << "Total number of 'new' calls = "
215  << (unsigned int)ms_uiNumNewCalls << std::endl;
216  kOStr << "Total number of 'delete' calls = "
217  << (unsigned int)ms_uiNumDeleteCalls << std::endl;
218  kOStr << "Maximum number of allocated bytes = "
219  << (unsigned int)ms_uiMaxAllocatedBytes << std::endl << std::endl;
220 
221  // Remaining counts.
222  kOStr << "Remaining number of blocks = "
223  << (unsigned int)ms_uiNumBlocks << std::endl;
224  kOStr << "Remaining number of bytes = "
225  << (unsigned int)ms_uiNumBytes << std::endl << std::endl;
226 
227  // Count the blocks and bytes from known and unknown sources.
228  size_t uiNumKnownBlocks = 0;
229  size_t uiNumKnownBytes = 0;
230  size_t uiNumUnknownBlocks = 0;
231  size_t uiNumUnknownBytes = 0;
232  Block* pkBlock = ms_pkHead;
233  while (pkBlock)
234  {
235  if (pkBlock->File)
236  {
237  uiNumKnownBlocks++;
238  uiNumKnownBytes += pkBlock->Size;
239  }
240  else
241  {
242  uiNumUnknownBlocks++;
243  uiNumUnknownBytes += pkBlock->Size;
244  }
245  pkBlock = pkBlock->Next;
246  }
247 
248 #ifdef WM4_ENABLE_CONSISTENCY_CHECK
249  // consistency check
250  assert(uiNumKnownBlocks + uiNumUnknownBlocks == ms_uiNumBlocks);
251  assert(uiNumKnownBytes + uiNumUnknownBytes == ms_uiNumBytes);
252 #endif
253 
254  kOStr << "Remaining number of known blocks = "
255  << (unsigned int)uiNumKnownBlocks << std::endl;
256  kOStr << "Remaining number of known bytes = "
257  << (unsigned int)uiNumKnownBytes << std::endl << std::endl;
258 
259  kOStr << "Remaining number of unknown blocks = "
260  << (unsigned int)uiNumUnknownBlocks << std::endl;
261  kOStr << "Remaining number of unknown bytes = "
262  << (unsigned int)uiNumUnknownBytes << std::endl << std::endl;
263 
264  // Report the information for each block.
265  pkBlock = ms_pkHead;
266  size_t uiIndex = 0;
267  while (pkBlock)
268  {
269  kOStr << "block = " << (unsigned int)uiIndex << std::endl;
270  kOStr << "size = " << (unsigned int)pkBlock->Size << std::endl;
271  if (pkBlock->File)
272  {
273  kOStr << "file = " << pkBlock->File << std::endl;
274  kOStr << "line = " << pkBlock->Line << std::endl;
275  }
276  else
277  {
278  kOStr << "file = unknown" << std::endl;
279  kOStr << "line = unknown" << std::endl;
280  }
281  kOStr << "array = " << pkBlock->IsArray << std::endl << std::endl;
282  pkBlock = pkBlock->Next;
283  uiIndex++;
284  }
285  kOStr.close();
286 
287 #ifdef WM4_ENABLE_CONSISTENCY_CHECK
288  // consistency check
289  assert(uiIndex == ms_uiNumBlocks);
290 #endif
291 }
