56 App::Origin* OriginGroupExtension::getOrigin() const
57 {
58  App::DocumentObject* originObj = Origin.getValue();
59 
60  if (!originObj) {
61  std::stringstream err;
62  err << "Can't find Origin for \"" << getExtendedObject()->getFullName() << "\"";
63  throw Base::RuntimeError(err.str().c_str());
64  }
65  else if (!originObj->isDerivedFrom(App::Origin::getClassTypeId())) {
66  std::stringstream err;
67  err << "Bad object \"" << originObj->getFullName() << "\"("
68  << originObj->getTypeId().getName() << ") linked to the Origin of \""
69  << getExtendedObject()->getFullName() << "\"";
70  throw Base::RuntimeError(err.str().c_str());
71  }
72  else {
73  return static_cast<App::Origin*>(originObj);
74  }
75 }

77 bool OriginGroupExtension::extensionGetSubObject(DocumentObject*& ret,
78  const char* subname,
79  PyObject** pyObj,
80  Base::Matrix4D* mat,
81  bool transform,
82  int depth) const
83 {
84  App::DocumentObject* originObj = Origin.getValue();
85  const char* dot;
86  if (originObj && originObj->isAttachedToDocument() && subname && (dot = strchr(subname, '.'))) {
87  bool found;
88  if (subname[0] == '$') {
89  found = std::string(subname + 1, dot) == originObj->Label.getValue();
90  }
91  else {
92  found = std::string(subname, dot) == originObj->getNameInDocument();
93  }
94  if (found) {
95  if (mat && transform) {
96  *mat *= const_cast<OriginGroupExtension*>(this)->placement().getValue().toMatrix();
97  }
98  ret = originObj->getSubObject(dot + 1, pyObj, mat, true, depth + 1);
99  return true;
100  }
101  }
102  return GeoFeatureGroupExtension::extensionGetSubObject(ret,
103  subname,
104  pyObj,
105  mat,
106  transform,
107  depth);
108 }

110 App::DocumentObject* OriginGroupExtension::getGroupOfObject(const DocumentObject* obj)
111 {
112 
113  if (!obj) {
114  return nullptr;
115  }
116 
117  bool isOriginFeature = obj->isDerivedFrom(App::DatumElement::getClassTypeId());
118 
119  auto list = obj->getInList();
120  for (auto o : list) {
121  if (o->hasExtension(App::OriginGroupExtension::getExtensionClassTypeId())) {
122  return o;
123  }
124  else if (isOriginFeature
125  && o->isDerivedFrom(App::LocalCoordinateSystem::getClassTypeId())) {
126  auto result = getGroupOfObject(o);
127  if (result) {
128  return result;
129  }
130  }
131  }
132 
133  return nullptr;
134 }

189 void OriginGroupExtension::extensionOnChanged(const Property* p)
190 {
191  if (p == &Origin) {
192  App::DocumentObject* owner = getExtendedObject();
193  App::DocumentObject* origin = Origin.getValue();
194  // Document::Importing indicates the object is being imported (i.e.
195  // copied). So check the Origin ownership here to prevent copy without
196  // dependency
197  if (origin && owner && owner->getDocument()
198  && owner->getDocument()->testStatus(Document::Importing)) {
199  for (auto o : origin->getInList()) {
200  if (o != owner
201  && o->hasExtension(App::OriginGroupExtension::getExtensionClassTypeId())) {
202  App::Document* document = owner->getDocument();
203  // Temporarily reset 'Restoring' status to allow document to auto label new
204  // objects
205  Base::ObjectStatusLocker<Document::Status, Document> guard(Document::Restoring,
206  document,
207  false);
208  Origin.setValue(getLocalizedOrigin(document));
209  FC_WARN("Reset origin in " << owner->getFullName());
210  return;
211  }
212  }
213  }
214  }
215  GeoFeatureGroupExtension::extensionOnChanged(p);
216 }

218 void OriginGroupExtension::relinkToOrigin(App::DocumentObject* obj)
219 {
220  // we get all links and replace the origin objects if needed (subnames need not to change, they
221  // would stay the same)
222  std::vector<App::DocumentObject*> result;
223  std::vector<App::Property*> list;
224  obj->getPropertyList(list);
225  auto isOriginFeature = [](App::DocumentObject* obj) -> bool {
226  // Check if the object is a DatumElement
227  if (auto* datumElement = dynamic_cast<App::DatumElement*>(obj)) {
228  // Check if the DatumElement is an origin
229  return datumElement->isOriginFeature();
230  }
231  return false;
232  };
233 
234  for (App::Property* prop : list) {
235  if (prop->isDerivedFrom<App::PropertyLink>()) {
236 
237  auto p = static_cast<App::PropertyLink*>(prop);
238  if (!p->getValue() || !isOriginFeature(p->getValue())) {
239  continue;
240  }
241 
242  p->setValue(getOrigin()->getDatumElement(
243  static_cast<DatumElement*>(p->getValue())->Role.getValue()));
244  }
245  else if (prop->isDerivedFrom<App::PropertyLinkList>()) {
246  auto p = static_cast<App::PropertyLinkList*>(prop);
247  auto vec = p->getValues();
248  std::vector<App::DocumentObject*> result;
249  bool changed = false;
250  for (App::DocumentObject* o : vec) {
251  if (!isOriginFeature(o)) {
252  result.push_back(o);
253  }
254  else {
255  result.push_back(getOrigin()->getDatumElement(
256  static_cast<DatumElement*>(o)->Role.getValue()));
257  changed = true;
258  }
259  }
260  if (changed) {
261  static_cast<App::PropertyLinkList*>(prop)->setValues(result);
262  }
263  }
264  else if (prop->isDerivedFrom<App::PropertyLinkSub>()) {
265  auto p = static_cast<App::PropertyLinkSub*>(prop);
266  if (!p->getValue() || !isOriginFeature(p->getValue())) {
267  continue;
268  }
269 
270  std::vector<std::string> subValues = p->getSubValues();
271  p->setValue(getOrigin()->getDatumElement(
272  static_cast<DatumElement*>(p->getValue())->Role.getValue()),
273  subValues);
274  }
275  else if (prop->isDerivedFrom<App::PropertyLinkSubList>()) {
276  auto p = static_cast<App::PropertyLinkSubList*>(prop);
277  auto vec = p->getSubListValues();
278  bool changed = false;
279  for (auto& v : vec) {
280  if (isOriginFeature(v.first)) {
281  v.first = getOrigin()->getDatumElement(
282  static_cast<DatumElement*>(v.first)->Role.getValue());
283  changed = true;
284  }
285  }
286  if (changed) {
287  p->setSubListValues(vec);
288  }
289  }
290  }
291 }
