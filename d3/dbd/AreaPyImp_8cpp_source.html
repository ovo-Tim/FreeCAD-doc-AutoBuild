344 PyObject* AreaPy::add(PyObject* args, PyObject* keywds)
345 {
346  PARAM_PY_DECLARE_INIT(PARAM_FARG, AREA_PARAMS_OPCODE)
347  PyObject* pcObj;
348 
349  // Strangely, PyArg_ParseTupleAndKeywords requires all arguments to be keyword based,
350  // even non-optional ones? That doesn't make sense in python. Seems only in python 3
351  // they added '$' to address that issue.
352  static const std::array<const char*, 3> kwlist {"shape",
353  PARAM_FIELD_STRINGS(ARG, AREA_PARAMS_OPCODE),
354  nullptr};
355 
356  if (!Base::Wrapped_ParseTupleAndKeywords(args,
357  keywds,
358  "O|" PARAM_PY_KWDS(AREA_PARAMS_OPCODE),
359  kwlist,
360  &pcObj,
361  PARAM_REF(PARAM_FARG, AREA_PARAMS_OPCODE))) {
362  return nullptr;
363  }
364 
365  PY_TRY
366  {
367  if (PyObject_TypeCheck(pcObj, &(Part::TopoShapePy::Type))) {
368  getAreaPtr()->add(GET_TOPOSHAPE(pcObj), op);
369  Py_INCREF(this);
370  return this;
371  }
372  else if (PyObject_TypeCheck(pcObj, &(PyList_Type))
373  || PyObject_TypeCheck(pcObj, &(PyTuple_Type))) {
374  Py::Sequence shapeSeq(pcObj);
375  for (Py::Sequence::iterator it = shapeSeq.begin(); it != shapeSeq.end(); ++it) {
376  PyObject* item = (*it).ptr();
377  if (!PyObject_TypeCheck(item, &(Part::TopoShapePy::Type))) {
378  PyErr_SetString(PyExc_TypeError, "non-shape object in sequence");
379  return nullptr;
380  }
381  }

383  PyObject* item = (*it).ptr();

385  PARAM_PY_FIELDS(PARAM_FARG, AREA_PARAMS_OPCODE));

388  return this;
389  }
390  }

458 PyObject* AreaPy::makeSections(PyObject* args, PyObject* keywds)
459 {
460  static const std::array<const char*, 5> kwlist {
461  PARAM_FIELD_STRINGS(ARG, AREA_PARAMS_SECTION_EXTRA),
462  "heights",
463  "plane",
464  nullptr};
465  PyObject* heights = nullptr;
466  PyObject* plane = nullptr;
467 
468  PARAM_PY_DECLARE_INIT(PARAM_FARG, AREA_PARAMS_SECTION_EXTRA)
469 
470  if (!Base::Wrapped_ParseTupleAndKeywords(args,
471  keywds,
472  "|" PARAM_PY_KWDS(AREA_PARAMS_SECTION_EXTRA) "OO!",
473  kwlist,
474  PARAM_REF(PARAM_FARG, AREA_PARAMS_SECTION_EXTRA),
475  &heights,
476  &(Part::TopoShapePy::Type),
477  &plane)) {
478  return nullptr;
479  }
480 
481  PY_TRY
482  {
483  std::vector<double> h;
484  if (heights) {
485  if (PyObject_TypeCheck(heights, &(PyFloat_Type))) {
486  h.push_back(PyFloat_AsDouble(heights));
487  }
488  else if (PyObject_TypeCheck(heights, &(PyList_Type))
489  || PyObject_TypeCheck(heights, &(PyTuple_Type))) {
490  Py::Sequence shapeSeq(heights);
491  h.reserve(shapeSeq.size());
492  for (Py::Sequence::iterator it = shapeSeq.begin(); it != shapeSeq.end(); ++it) {
493  PyObject* item = (*it).ptr();
494  if (!PyObject_TypeCheck(item, &(PyFloat_Type))) {
495  PyErr_SetString(PyExc_TypeError, "heights must only contain float type");
496  return nullptr;
497  }
498  h.push_back(PyFloat_AsDouble(item));
499  }
500  }
501  else {
502  PyErr_SetString(PyExc_TypeError,
503  "heights must be of type float or list/tuple of float");
504  return nullptr;
505  }
506  }
507 
508  std::vector<std::shared_ptr<Area>> sections =
509  getAreaPtr()->makeSections(PARAM_PY_FIELDS(PARAM_FARG, AREA_PARAMS_SECTION_EXTRA),
510  h,
511  plane ? GET_TOPOSHAPE(plane) : TopoDS_Shape());
512 
513  Py::List ret;
514  for (auto& area : sections) {
515  ret.append(Py::asObject(new AreaPy(new Area(*area, true))));
516  }
517  return Py::new_reference_to(ret);
518  }
