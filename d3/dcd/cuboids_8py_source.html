39 def isCubic(shape):
40  """Return True if the shape is a parallelepiped (cuboid).
41 
42  A parallelepiped of cube-like shape has 8 vertices, 6 faces, 12 edges,
43  and all angles are 90 degrees between its edges.
44  """
45  # first we try fast methods
46  if (len(shape.Vertexes) != 8
47  or len(shape.Faces) != 6
48  or len(shape.Edges) != 12):
49  return False
50 
51  for e in shape.Edges:
52  if geomType(e) != "Line":
53  return False
54 
55  # if ok until now, let's do more advanced testing
56  for f in shape.Faces:
57  if len(f.Edges) != 4:
58  return False
59 
60  for i in range(4):
61  e1 = vec(f.Edges[i])
62  if i < 3:
63  e2 = vec(f.Edges[i+1])
64  else:
65  e2 = vec(f.Edges[0])
66  rpi = [0.0, round(math.pi/2, precision())]
67  if round(e1.getAngle(e2), precision()) not in rpi:
68  return False
69 
70  return True

73 def getCubicDimensions(shape):
74  """Return a list containing the placement, and dimensions of the shape.
75 
76  The dimensios are length, width and height of a the parallelepiped,
77  rounded to the value indicated by `precision`.
78  The placement point is the lowest corner of the shape.
79 
80  If it is not a parallelepiped (cuboid), return None.
81  """
82  if not isCubic(shape):
83  return None
84 
85  # determine lowest face, which will be our base
86  z = [10, 1000000000000]
87  for i in range(len(shape.Faces)):
88  if shape.Faces[i].CenterOfMass.z < z[1]:
89  z = [i, shape.Faces[i].CenterOfMass.z]
90 
91  if z[0] > 5:
92  return None
93 
94  base = shape.Faces[z[0]]
95  basepoint = base.Edges[0].Vertexes[0].Point
96  plpoint = base.CenterOfMass
97  # basenorm = base.normalAt(0.5, 0.5)
98 
99  # getting length and width
100  vx = vec(base.Edges[0])
101  vy = vec(base.Edges[1])
102  if round(vx.Length) == round(vy.Length):
103  vy = vec(base.Edges[2])
104 
105  # getting rotations
106  rotZ = DraftVecUtils.angle(vx)
107  rotY = DraftVecUtils.angle(vx, App.Vector(vx.x, vx.y, 0))
108  rotX = DraftVecUtils.angle(vy, App.Vector(vy.x, vy.y, 0))
109 
110  # getting height
111  vz = None
112  rpi = round(math.pi/2, precision())
113  for i in range(1, 6):
114  for e in shape.Faces[i].Edges:
115  if basepoint in [e.Vertexes[0].Point, e.Vertexes[1].Point]:
116  vtemp = vec(e)
117  # print(vtemp)
118  if round(vtemp.getAngle(vx), precision()) == rpi:
119  if round(vtemp.getAngle(vy), precision()) == rpi:
120  vz = vtemp
121 
122  if not vz:
123  return None
124 
125  mat = App.Matrix()
126  mat.move(plpoint)
127  mat.rotateX(rotX)
128  mat.rotateY(rotY)
129  mat.rotateZ(rotZ)
130 
131  return [App.Placement(mat),
132  round(vx.Length, precision()),
133  round(vy.Length, precision()),
134  round(vz.Length, precision())]
