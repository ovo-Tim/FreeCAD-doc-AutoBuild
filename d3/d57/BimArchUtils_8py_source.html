449  def Activated(self):
450  """Executed when Arch MergeWalls is called.
451 
452  Call ArchWall.joinWalls() on walls selected by the user, with the
453  delete option enabled. If the user has selected a single wall, check to
454  see if the wall has any Additions that are walls. If so, merges these
455  additions to the wall, deleting the additions.
456  """
457 
458  import Draft
459  walls = FreeCADGui.Selection.getSelection()
460  if len(walls) == 1:
461  if Draft.getType(walls[0]) == "Wall":
462  ostr = "FreeCAD.ActiveDocument."+ walls[0].Name
463  ok = False
464  for o in walls[0].Additions:
465  if Draft.getType(o) == "Wall":
466  ostr += ",FreeCAD.ActiveDocument." + o.Name
467  ok = True
468  if ok:
469  FreeCAD.ActiveDocument.openTransaction(translate("Arch","Merge Walls"))
470  FreeCADGui.addModule("Arch")
471  FreeCADGui.doCommand("Arch.joinWalls(["+ostr+"],delete=True)")
472  FreeCAD.ActiveDocument.commitTransaction()
473  return
474  else:
475  FreeCAD.Console.PrintWarning(translate("Arch","The selected wall contains no subwall to merge"))
476  return
477  else:
478  FreeCAD.Console.PrintWarning(translate("Arch","Please select only wall objects"))
479  return
480  for w in walls:
481  if Draft.getType(w) != "Wall":
482  FreeCAD.Console.PrintMessage(translate("Arch","Please select only wall objects"))
483  return
484  FreeCAD.ActiveDocument.openTransaction(translate("Arch","Merge Walls"))
485  FreeCADGui.addModule("Arch")
