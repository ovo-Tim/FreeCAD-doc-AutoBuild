245  def getLeadStart(self, obj, move, first):
246  lead = []
247  begin = move.positionBegin()
248 
249  def prepend(instr):
250  nonlocal lead
251  nonlocal begin
252  lead.insert(0, instr)
253  begin = lead[0].positionBegin()
254 
255  # tangent begin move
256  # <----_-----x-------------------x
257  # / |
258  # / | normal
259  # | |
260  # x v
261 
262  if obj.LeadIn:
263  length = obj.Length.Value
264  angle = move.anglesOfTangents()[0]
265  tangent = -self.angleToVector(angle) * length
266  normal = self.angleToVector(angle + self.getArcDirection(obj)) * length
267 
268  # prepend the selected lead-in
269  if obj.StyleOn == "Arc":
270  arcbegin = begin + tangent + normal
271  prepend(self.createArcMove(obj, arcbegin, begin, -tangent))
272  elif obj.StyleOn == "Tangent":
273  prepend(self.createStraightMove(obj, begin + tangent, begin))
274  else: # obj.StyleOn == "Perpendicular"
275  prepend(self.createStraightMove(obj, begin + normal, begin))
276 
277  extend = obj.ExtendLeadIn.Value
278  if extend != 0:
279  # prepend extension
280  extendbegin = begin + normal / length * extend
281  prepend(self.createStraightMove(obj, extendbegin, begin))
282 
283  # prepend travel moves
284  lead = self.getTravelStart(obj, begin, first) + lead
285 
286  return lead

288  def getLeadEnd(self, obj, move, last):
289  lead = []
290  end = move.positionEnd()
291 
292  def append(instr):
293  nonlocal lead
294  nonlocal end
295  lead.append(instr)
296  end = lead[-1].positionEnd()
297 
298  # move end tangent
299  # x-------------------x-----_---->
300  # | \
301  # normal | \
302  # | |
303  # v x
304 
305  if obj.LeadOut:
306  length = obj.LengthOut.Value
307  angle = move.anglesOfTangents()[1]
308  tangent = self.angleToVector(angle) * length
309  normal = self.angleToVector(angle + self.getArcDirection(obj)) * length
310 
311  # append the selected lead-out
312  if obj.StyleOff == "Arc":
313  arcend = end + tangent + normal
314  append(self.createArcMove(obj, end, arcend, normal))
315  elif obj.StyleOff == "Tangent":
316  append(self.createStraightMove(obj, end, end + tangent))
317  else: # obj.StyleOff == "Perpendicular"
318  append(self.createStraightMove(obj, end, end + normal))
319 
320  extend = obj.ExtendLeadOut.Value
321  if extend != 0:
322  # append extension
323  extendend = end + normal / length * extend
324  append(self.createStraightMove(obj, end, extendend))
325 
326  # append travel moves
327  lead += self.getTravelEnd(obj, end, last)
328 
329  return lead

344  def generateLeadInOutCurve(self, obj):
345  source = PathLanguage.Maneuver.FromPath(PathUtils.getPathWithPlacement(obj.Base)).instr
346  maneuver = PathLanguage.Maneuver()
347 
348  # Knowing weather a given instruction is the first cutting move is easy,
349  # we just use a flag and set it to false afterwards. To find the last
350  # cutting move we need to search the list in reverse order.
351  first = True
352  lastCuttingMoveIndex = self.findLastCuttingMoveIndex(obj, source)
353 
354  for i, instr in enumerate(source):
355  if not self.isCuttingMove(obj, instr):
356  # non-move instructions get added verbatim
357  if not instr.isMove():
358  maneuver.addInstruction(instr)
359 
360  # skip travel and plunge moves, travel moves will be added in
361  # getLeadStart and getLeadEnd
362  continue
363 
364  if first or not self.isCuttingMove(obj, source[i - 1]):
365  # add lead start and travel moves
366  maneuver.addInstructions(self.getLeadStart(obj, instr, first))
367  first = False
368 
369  # add current move
370  maneuver.addInstruction(instr)
371 
372  last = i == lastCuttingMoveIndex
373  if last or not self.isCuttingMove(obj, source[i + 1]):
374  # add lead end and travel moves
375  maneuver.addInstructions(self.getLeadEnd(obj, instr, last))
376 
377  return maneuver.toPath()
