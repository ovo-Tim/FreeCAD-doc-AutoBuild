45 static void print_curve(const CCurve& c)
46 {
47  std::size_t nvertices = c.m_vertices.size();
48 #if defined SIZEOF_SIZE_T && SIZEOF_SIZE_T == 4
49  printf("number of vertices = %d\n", nvertices);

51  printf("number of vertices = %Iu\n", nvertices);
52 #else
53  printf("number of vertices = %lu\n", nvertices);
54 #endif

57  It++, i++) {
58  const CVertex& vertex = *It;
59  printf("vertex %d type = %d, x = %g, y = %g",
60  i + 1,
61  vertex.m_type,
62  vertex.m_p.x / CArea::get_units(),
63  vertex.m_p.y / CArea::get_units());
64  if (vertex.m_type) {
65  printf(", xc = %g, yc = %g",
66  vertex.m_c.x / CArea::get_units(),
67  vertex.m_c.y / CArea::get_units());

69  printf("\n");
70  }
