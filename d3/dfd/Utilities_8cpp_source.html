113 std::vector<int> Tessellator::tessellate() const
114 {
115  std::vector<int> indices(polygon.size());
116  std::vector<int> face_indices;
117 
118  SbTesselator tessellator(tessCB, &face_indices);
119  tessellator.beginPolygon();
120 
121  int index = 0;
122  for (std::vector<SbVec2f>::const_iterator it = polygon.begin(); it != polygon.end(); ++it, index++) {
123  indices[index] = index;
124  tessellator.addVertex(SbVec3f((*it)[0], (*it)[1], 0.0f), &(indices[index]));
125  }
126 
127  // run the triangulation now
128  tessellator.endPolygon();
129  return face_indices;
130 }

150 void ItemViewSelection::applyFrom(const std::vector<App::DocumentObject*> objs)
151 {
152  QAbstractItemModel* model = view->model();
153  QItemSelection range;
154  for (int i=0; i<model->rowCount(); i++) {
155  QModelIndex item = model->index(i,0);
156  if (item.isValid()) {
157  QVariant name = model->data(item, Qt::UserRole);
158  std::vector<App::DocumentObject*>::const_iterator it;
159  it = std::find_if(objs.begin(), objs.end(), MatchName(name.toString()));
160  if (it != objs.end())
161  range.select(item, item);
162  }
163  }
164 
165  view->selectionModel()->select(range, QItemSelectionModel::Select);
