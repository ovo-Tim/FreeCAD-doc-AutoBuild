113 void TextEdit::complete()
114 {
115  QTextBlock block = textCursor().block();
116  if (!block.isValid())
117  return;
118  int cursorPos = textCursor().position()-block.position();
119  QString para = block.text();
120  int wordStart = cursorPos;
121  while (wordStart > 0 && para[wordStart - 1].isLetterOrNumber())
122  --wordStart;
123  wordPrefix = para.mid(wordStart, cursorPos - wordStart);
124  if (wordPrefix.isEmpty())
125  return;
126 
127  QStringList list = toPlainText().split(QRegularExpression(QLatin1String("\\W+")));
128  QMap<QString, QString> map;
129  QStringList::Iterator it = list.begin();
130  while (it != list.end()) {
131  if ((*it).startsWith(wordPrefix) && (*it).length() > wordPrefix.length())
132  map[(*it).toLower()] = *it;
133  ++it;
134  }
135 
136  if (map.count() == 1) {
137  insertPlainText((*map.begin()).mid(wordPrefix.length()));
138  } else if (map.count() > 1) {
139  if (!listBox)
140  createListBox();
141  listBox->clear();
142  listBox->addItems(map.values());
143  listBox->setFont(QFont(font().family(), 8));
144 
145  this->cursorPosition = textCursor().position();
146 
147  // get the minimum width and height of the box
148  int h = 0;
149  int w = 0;
150  for (int i = 0; i < listBox->count(); ++i) {
151  QRect r = listBox->visualItemRect(listBox->item(i));
152  w = qMax(w, r.width());
153  h += r.height();
154  }
155 
156  // Add an offset
157  w += 2*listBox->frameWidth();
158  h += 2*listBox->frameWidth();
159 
160  // get the start position of the word prefix
161  QTextCursor cursor = textCursor();
162  cursor.movePosition(QTextCursor::StartOfWord);
163  QRect rect = cursorRect(cursor);
164  int posX = rect.x();
165  int posY = rect.y();
166  int boxH = h;
167 
168  // Decide whether to show downstairs or upstairs
169  if (posY > viewport()->height()/2) {
170  h = qMin(qMin(h,posY), 250);
171  if (h < boxH)
172  w += style()->pixelMetric(QStyle::PM_ScrollBarExtent);
173  listBox->setGeometry(posX,posY-h, w, h);
174  } else {
175  h = qMin(qMin(h,viewport()->height()-fontMetrics().height()-posY), 250);
176  if (h < boxH)
177  w += style()->pixelMetric(QStyle::PM_ScrollBarExtent);
178  listBox->setGeometry(posX, posY+fontMetrics().height(), w, h);
179  }
180 
181  listBox->setCurrentRow(0);
182  listBox->show();
183  }
184 }

360 void TextEditor::lineNumberAreaPaintEvent(QPaintEvent *event)
361 {
362  if (!isVisibleLineNumbers()) {
363  return;
364  }
365  QPainter painter(lineNumberArea);
366  //painter.fillRect(event->rect(), Qt::lightGray);
367 
368  QTextBlock block = firstVisibleBlock();
369  int blockNumber = block.blockNumber();
370  int top = (int) blockBoundingGeometry(block).translated(contentOffset()).top();
371  int bottom = top + (int) blockBoundingRect(block).height();
372 
373  while (block.isValid() && top <= event->rect().bottom()) {
374  if (block.isVisible() && bottom >= event->rect().top()) {
375  QString number = QString::number(blockNumber + 1);
376  QPalette pal = palette();
377  QColor color = pal.windowText().color();
378  painter.setPen(color);
379  painter.drawText(0, top, lineNumberArea->width(), fontMetrics().height(),
380  Qt::AlignRight, number);
381  drawMarker(blockNumber + 1, 1, top, &painter);
382  }
383 
384  block = block.next();
385  top = bottom;
386  bottom = top + (int) blockBoundingRect(block).height();
387  ++blockNumber;
388  }

398 void TextEditor::OnChange(Base::Subject<const char*> &rCaller,const char* sReason)
399 {
400  Q_UNUSED(rCaller);
401  ParameterGrp::handle hPrefGrp = getWindowParameter();
402  if (strcmp(sReason, "FontSize") == 0 || strcmp(sReason, "Font") == 0) {
403 #ifdef FC_OS_LINUX
404  int fontSize = hPrefGrp->GetInt("FontSize", 15);
405 #else
406  int fontSize = hPrefGrp->GetInt("FontSize", 10);
407 #endif
408  QString fontFamily = QString::fromLatin1(hPrefGrp->GetASCII( "Font", "Courier" ).c_str());
409 
410  QFont font(fontFamily, fontSize);
411  setFont(font);
412  lineNumberArea->setFont(font);
413  }
414  else {
415  QMap<QString, QColor>::Iterator it = d->colormap.find(QString::fromLatin1(sReason));
416  if (it != d->colormap.end()) {
417  QColor color = it.value();
418  unsigned int col = App::Color::asPackedRGB<QColor>(color);
419  auto value = static_cast<unsigned long>(col);
420  value = hPrefGrp->GetUnsigned(sReason, value);
421  col = static_cast<unsigned int>(value);
422  color.setRgb((col>>24)&0xff, (col>>16)&0xff, (col>>8)&0xff);
423  if (this->highlighter)
424  this->highlighter->setColor(QLatin1String(sReason), color);
425  }
426  }
427 
428  if (strcmp(sReason, "TabSize") == 0 || strcmp(sReason, "FontSize") == 0) {
429  int tabWidth = hPrefGrp->GetInt("TabSize", 4);
430  QFontMetrics metric(font());
431  int fontSize = QtTools::horizontalAdvance(metric, QLatin1Char('0'));
432  setTabStopDistance(tabWidth * fontSize);
433  }
434 
435  // Enables/Disables Line number in the Macro Editor from Edit->Preferences->Editor menu.
436  if (strcmp(sReason, "EnableLineNumber") == 0) {
437  int width = 0;
438  QRect cr = contentsRect();
439  if (hPrefGrp->GetBool("EnableLineNumber", true)) {
440  width = lineNumberAreaWidth();
441  }
442  lineNumberArea->setGeometry(QRect(cr.left(), cr.top(), width, cr.height()));
443  }
444 }

456 void PythonTextEditor::keyPressEvent (QKeyEvent * e)
457 {
458  if ( e->key() == Qt::Key_Tab ) {
459  ParameterGrp::handle hPrefGrp = getWindowParameter();
460  bool space = hPrefGrp->GetBool("Spaces", true);
461  int indent = hPrefGrp->GetInt( "IndentSize", 4 );
462  QString ch = space ? QString(indent, QLatin1Char(' '))
463  : QString::fromLatin1("\t");
464 
465  QTextCursor cursor = textCursor();
466  if (!cursor.hasSelection()) {
467  // insert a single tab or several spaces
468  cursor.beginEditBlock();
469  cursor.insertText(ch);
470  cursor.endEditBlock();
471  } else {
472  // for each selected block insert a tab or spaces
473  int selStart = cursor.selectionStart();
474  int selEnd = cursor.selectionEnd();
475  QTextBlock block;
476  cursor.beginEditBlock();
477  for (block = document()->begin(); block.isValid(); block = block.next()) {
478  int pos = block.position();
479  int off = block.length()-1;
480  // at least one char of the block is part of the selection
481  if ( pos >= selStart || pos+off >= selStart) {
482  if ( pos+1 > selEnd )
483  break; // end of selection reached
484  cursor.setPosition(block.position());
485  cursor.insertText(ch);
486  selEnd += ch.length();
487  }
488  }
489 
490  cursor.endEditBlock();
491  }
492 
493  return;
494  }
495  else if (e->key() == Qt::Key_Backtab) {
496  QTextCursor cursor = textCursor();
497  if (!cursor.hasSelection())
498  return; // Shift+Tab should not do anything
499  // If some text is selected we remove a leading tab or
500  // spaces from each selected block
501  ParameterGrp::handle hPrefGrp = getWindowParameter();
502  int indent = hPrefGrp->GetInt( "IndentSize", 4 );
503 
504  int selStart = cursor.selectionStart();
505  int selEnd = cursor.selectionEnd();
506  QTextBlock block;
507  cursor.beginEditBlock();
508  for (block = document()->begin(); block.isValid(); block = block.next()) {
509  int pos = block.position();
510  int off = block.length()-1;
511  // at least one char of the block is part of the selection
512  if ( pos >= selStart || pos+off >= selStart) {
513  if ( pos+1 > selEnd )
514  break; // end of selection reached
515  // if possible remove one tab or several spaces
516  QString text = block.text();
517  if (text.startsWith(QLatin1String("\t"))) {
518  cursor.setPosition(block.position());
519  cursor.deleteChar();
520  selEnd--;
521  }
522  else {
523  cursor.setPosition(block.position());
524  for (int i=0; i<indent; i++) {
525  if (!text.startsWith(QLatin1String(" ")))
526  break;
527  text = text.mid(1);
528  cursor.deleteChar();
529  selEnd--;
530  }
531  }
532  }
533  }
534 
535  cursor.endEditBlock();
536  return;
537  }

596 bool CompletionList::eventFilter(QObject * watched, QEvent * event)
597 {
598  if (isVisible() && watched == textEdit->viewport()) {
599  if (event->type() == QEvent::MouseButtonPress)
600  hide();
601  } else if (isVisible() && watched == textEdit) {
602  if (event->type() == QEvent::KeyPress) {
603  auto ke = static_cast<QKeyEvent*>(event);
604  if (ke->key() == Qt::Key_Up || ke->key() == Qt::Key_Down) {
605  keyPressEvent(ke);
606  return true;
607  } else if (ke->key() == Qt::Key_PageUp || ke->key() == Qt::Key_PageDown) {
608  keyPressEvent(ke);
609  return true;
610  } else if (ke->key() == Qt::Key_Escape) {
611  hide();
612  return true;
613  } else if (ke->key() == Qt::Key_Space) {
614  hide();
615  return false;
616  } else if (ke->key() == Qt::Key_Return || ke->key() == Qt::Key_Enter) {
617  Q_EMIT itemActivated(currentItem());
618  return true;
619  }
620  } else if (event->type() == QEvent::FocusOut) {
621  if (!hasFocus())
622  hide();
623  }
624  }
625 
626  return QListWidget::eventFilter(watched, event);
627 }
