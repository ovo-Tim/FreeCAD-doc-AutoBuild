59 //DWS always has exactly 2 child tiles - ArrowSide and OtherSide.
60 //OtherSide tile may be hidden;
61 //once DWS has been added to the document, add 2x DrawTileWeld
62 //but if this is a restore of an existing DWS, the tiles will loaded elsewhere
63 void DrawWeldSymbol::onSettingDocument()
64 {
65 // Base::Console().Message("DWS::onSettingDocument() - doc: %s\n", getDocument()->getName());
66  App::Document* doc = getDocument();
67 
68  if (doc->testStatus(App::Document::Status::Restoring)) {
69 // Base::Console().Message("DWS::onSettingDocument() - restoring!\n");
70  return;
71  }
72 
73  std::vector<DrawTileWeld*> existingTiles = getTiles();
74  if (!existingTiles.empty()) {
75  return;
76  }
77 
78  std::string tileName1 = doc->getUniqueObjectName("TileWeld");
79  auto tile1Obj( doc->addObject( "TechDraw::DrawTileWeld", tileName1.c_str() ) );
80  DrawTileWeld* tile1 = dynamic_cast<DrawTileWeld*>(tile1Obj);
81  if (tile1) {
82  tile1->Label.setValue(DrawUtil::translateArbitrary("DrawTileWeld", "TileWeld", tileName1));
83  tile1->TileParent.setValue(this);
84  }
85 
86  std::string tileName2 = doc->getUniqueObjectName("TileWeld");
87  auto tile2Obj( doc->addObject( "TechDraw::DrawTileWeld", tileName2.c_str() ) );
88  DrawTileWeld* tile2 = dynamic_cast<DrawTileWeld*>(tile2Obj);
89  if (tile2) {
90  tile2->Label.setValue(DrawUtil::translateArbitrary("DrawTileWeld", "TileWeld", tileName2));
91  tile2->TileParent.setValue(this);
92  tile2->TileRow.setValue(-1); //other side is row -1
93  }
94 
95  DrawView::onSettingDocument();
96 }

125 std::vector<DrawTileWeld*> DrawWeldSymbol::getTiles() const
126 {
127 // Base::Console().Message("DWS::getTiles()\n");
128  std::vector<DrawTileWeld*> result;
129 
130  std::vector<App::DocumentObject*> tiles = getInList();
131  if (tiles.empty()) {
132  return result;
133  }
134 
135  for(std::vector<App::DocumentObject *>::iterator it = tiles.begin(); it != tiles.end(); it++) {
136  if ((*it)->isDerivedFrom<TechDraw::DrawTileWeld>()) {
137  App::DocumentObject* doTemp = (*it);
138  DrawTileWeld* temp = static_cast<DrawTileWeld*>(doTemp);
139  result.push_back(temp);
140  }
141  }
142  return result;
143 }

145 bool DrawWeldSymbol::isTailRightSide()
146 {
147  App::DocumentObject* obj = Leader.getValue();
148  TechDraw::DrawLeaderLine* realLeader = dynamic_cast<TechDraw::DrawLeaderLine*>(obj);
149  if (realLeader) {
150  Base::Vector3d tail = realLeader->getTailPoint();
151  Base::Vector3d kink = realLeader->getKinkPoint();
152  if (tail.x < kink.x) { //tail is to left
153  return false;
154  }
155  }
156  return true;
157 }
