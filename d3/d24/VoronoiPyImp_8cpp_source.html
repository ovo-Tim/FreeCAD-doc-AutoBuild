72 Voronoi::point_type getPointFromPy(PyObject* obj)
73 {
74  if (obj) {
75  if (PyObject_TypeCheck(obj, &Base::VectorPy::Type)) {
76  Base::Vector3d* vect = (static_cast<Base::VectorPy*>(obj))->getVectorPtr();
77  return Voronoi::point_type(vect->x, vect->y);
78  }
79  else if (PyObject_TypeCheck(obj, Base::Vector2dPy::type_object())) {
80  Base::Vector2d vect = Py::toVector2d(obj);
81  return Voronoi::point_type(vect.x, vect.y);

83  }
84  throw Py::TypeError("Points must be Base::Vector or Base::Vector2d");
85  return Voronoi::point_type();

174 using exterior_map_t = std::map<uintptr_t, bool>;
175 using coordinate_map_t = std::map<int32_t, std::set<int32_t>>;

177 #define VORONOI_USE_EXTERIOR_CACHE 1
178 
179 static bool callbackWithVertex(Voronoi::diagram_type* dia,
180  PyObject* callback,
181  const Voronoi::diagram_type::vertex_type* v,
182  bool& bail,
183  exterior_map_t& cache)
184 {
185  bool rc = false;
186  if (!bail && v->color() == 0) {
187 #if VORONOI_USE_EXTERIOR_CACHE
188  auto it = cache.find(uintptr_t(v));
189  if (it == cache.end()) {
190 #endif
191  PyObject* vx = new VoronoiVertexPy(new VoronoiVertex(dia, v));
192  PyObject* arglist = Py_BuildValue("(O)", vx);
193 #if PY_VERSION_HEX < 0x03090000
194  PyObject* result = PyEval_CallObject(callback, arglist);
195 #else
196  PyObject* result = PyObject_CallObject(callback, arglist);
197 #endif
198  Py_DECREF(arglist);
199  Py_DECREF(vx);
200  if (!result) {
201  bail = true;
202  }
203  else {
204  rc = result == Py_True;
205  Py_DECREF(result);
206  cache.insert(exterior_map_t::value_type(uintptr_t(v), rc));
207  }
208 #if VORONOI_USE_EXTERIOR_CACHE
209  }
210  else {
211  rc = it->second;
212  }
213 #else
214  (void)cache;
215 #endif
216  }
217  return rc;
218 }

220 PyObject* VoronoiPy::colorExterior(PyObject* args)
221 {
222  Voronoi::color_type color = 0;
223  PyObject* callback = nullptr;
224  if (!PyArg_ParseTuple(args, "k|O", &color, &callback)) {
225  throw Py::RuntimeError("colorExterior requires an integer (color) argument");
226  }
227  Voronoi* vo = getVoronoiPtr();
228  vo->colorExterior(color);
229  if (callback) {
230  exterior_map_t cache;
231  coordinate_map_t pts;
232  for (auto e = vo->vd->edges().begin(); e != vo->vd->edges().end(); ++e) {
233  if (e->is_finite() && e->color() == 0) {
234  const Voronoi::diagram_type::vertex_type* v0 = e->vertex0();
235  const Voronoi::diagram_type::vertex_type* v1 = e->vertex1();
236  bool bail = false;
237  if (callbackWithVertex(vo->vd, callback, v0, bail, cache)
238  && callbackWithVertex(vo->vd, callback, v1, bail, cache)) {
239  vo->colorExterior(&(*e), color);
240  }
241  else if (!bail && callbackWithVertex(vo->vd, callback, v1, bail, cache)) {
242  if (pts.empty()) {
243  for (auto s = vo->vd->segments.begin(); s != vo->vd->segments.end(); ++s) {
244  pts[low(*s).x()].insert(low(*s).y());
245  pts[high(*s).x()].insert(high(*s).y());
246  }
247  }
248  auto ys = pts.find(int32_t(v0->x()));
249  if (ys != pts.end() && ys->second.find(v0->y()) != ys->second.end()) {
250  vo->colorExterior(&(*e), color);
251  }
252  }
253  if (bail) {
254  return nullptr;
255  }
256  }
257  }
258  }
259 
260  Py_INCREF(Py_None);
261  return Py_None;
262 }
