79 std::vector<App::DocumentObject*> ViewProviderGeoFeatureGroupExtension::extensionClaimChildren() const {
80 
81  auto* obj = getExtendedViewProvider()->getObject();
82  if (!obj) {
83  return {};
84  }
85 
86  auto* group = obj->getExtensionByType<App::GeoFeatureGroupExtension>();
87  const std::vector<App::DocumentObject*> &model = group->Group.getValues ();
88  std::set<App::DocumentObject*> outSet; //< set of objects not to claim (childrens of childrens)
89 
90  // search for objects handled (claimed) by the features
91  for (auto obj: model) {
92  //stuff in another geofeaturegroup is not in the model anyway
93  if (!obj || obj->hasExtension(App::GeoFeatureGroupExtension::getExtensionClassTypeId())) { continue; }
94 
95  Gui::ViewProvider* vp = Gui::Application::Instance->getViewProvider ( obj );
96  if (!vp || vp == getExtendedViewProvider()) { continue; }
97 
98  auto children = vp->claimChildren();
99  std::remove_copy ( children.begin (), children.end (), std::inserter (outSet, outSet.begin () ), nullptr);
100  }
101 
102  // remove the otherwise handled objects, preserving their order so the order in the TreeWidget is correct
103  std::vector<App::DocumentObject*> Result;
104  for(auto obj : model) {
105  if(!obj || !obj->isAttachedToDocument())
106  continue;
107  if(outSet.count(obj))
108  obj->setStatus(App::ObjectStatus::GeoExcluded,true);
109  else {
110  obj->setStatus(App::ObjectStatus::GeoExcluded,false);
111  Result.push_back(obj);
112  }
113  }
114  return Result;
115 }
