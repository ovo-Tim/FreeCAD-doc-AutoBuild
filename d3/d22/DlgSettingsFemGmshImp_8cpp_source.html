98 void DlgSettingsFemGmshImp::populateLogVerbosity()
99 {
100  std::list<std::pair<std::string, int>> mapValues = {{"Silent", 0},
101  {"Errors", 1},
102  {"Warnings", 2},
103  {"Direct", 3},
104  {"Information", 4},
105  {"Status", 5},
106  {"Debug", 99}};
107 
108  for (const auto& val : mapValues) {
109  ui->cb_log_verbosity->addItem(QString::fromStdString(val.first),
110  QString::number(val.second));
111  }
112 
113  auto hGrp = App::GetApplication().GetParameterGroupByPath(
114  "User parameter:BaseApp/Preferences/Mod/Fem/Gmsh");
115  std::string current = hGrp->GetASCII("LogVerbosity", "3");
116  int index = ui->cb_log_verbosity->findData(QString::fromStdString(current));
117  ui->cb_log_verbosity->setCurrentIndex(index);
118 }
