31  def testXAxisLinearPattern(self):
32  self.Body = self.Doc.addObject('PartDesign::Body','Body')
33  self.Box = self.Doc.addObject('PartDesign::AdditiveBox','Box')
34  self.Body.addObject(self.Box)
35  self.Box.Length=10.00
36  self.Box.Width=10.00
37  self.Box.Height=10.00
38  self.Doc.recompute()
39  self.LinearPattern = self.Doc.addObject("PartDesign::LinearPattern","LinearPattern")
40  self.LinearPattern.Originals = [self.Box]
41  self.LinearPattern.Direction = (self.Doc.X_Axis,[""])
42  self.LinearPattern.Length = 90.0
43  self.LinearPattern.Occurrences = 10
44  self.LinearPattern.Refine = True
45  self.Body.addObject(self.LinearPattern)
46  self.Doc.recompute()
47  self.assertAlmostEqual(self.LinearPattern.Shape.Volume, 1e4)
48  # 44 + 84 + 42 = 170. 44 - 8 = 36 / 9 = 4. 84-12 = 72 / 9 = 8. 42 - 6 = 36 / 9 = 4
49  # We have the original 26 from the first shape, plus 4 more vertices, 8 more edges and
50  # 4 more faces for each additional copy. Since they have to touch ( single shape rule ),
51  # We're adding 4 points to define each additional prism's new points, 8 edges makes sense,
52  # and 4 faces makes sense since we're defining essentially a tube, not a box for each copy.
53  # self.assertNotEqual(self.LinearPattern.Shape.ElementReverseMap["Vertex44"], "")
54  # self.assertNotEqual(self.LinearPattern.Shape.ElementReverseMap["Edge84"], "")
55  # self.assertNotEqual(self.LinearPattern.Shape.ElementReverseMap["Face42"], "")
56  #
57  # self.assertEqual(self.LinearPattern.Shape.ElementMapSize, 170) # TODO
