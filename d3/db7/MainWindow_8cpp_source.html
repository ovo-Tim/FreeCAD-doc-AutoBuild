469 // Helper function to update dock widget according to the user parameter
470 // settings, e.g. register/unregister, enable/disable, show/hide.
471 template<class T>
472 static inline void _updateDockWidget(const char *name,
473  bool enabled,
474  bool show,
475  Qt::DockWidgetArea pos,
476  T callback)
477 {
478  auto pDockMgr = DockWindowManager::instance();
479  auto widget = pDockMgr->findRegisteredDockWindow(name);
480  if (!enabled) {
481  if(widget) {
482  pDockMgr->removeDockWindow(widget);
483  pDockMgr->unregisterDockWindow(name);
484  widget->deleteLater();
485  }
486  return;
487  }
488  // Use callback to perform specific update for each type of dock widget
489  widget = callback(widget);
490  if(!widget)
491  return;
492  DockWindowManager::instance()->registerDockWindow(name, widget);
493  if(show) {
494  auto dock = pDockMgr->addDockWindow(
495  widget->objectName().toUtf8().constData(), widget, pos);
496  if(dock) {
497  if(!dock->toggleViewAction()->isChecked())
498  dock->toggleViewAction()->activate(QAction::Trigger);
499  OverlayManager::instance()->refresh(dock);
500  }
501  }
502 }

1038 bool MainWindow::eventFilter(QObject* o, QEvent* e)
1039 {
1040  if (o != this) {
1041  if (e->type() == QEvent::WindowStateChange) {
1042  // notify all mdi views when the active view receives a show normal, show minimized
1043  // or show maximized event
1044  auto view = qobject_cast<MDIView*>(o);
1045  if (view) { // emit this signal
1046  Qt::WindowStates oldstate = static_cast<QWindowStateChangeEvent*>(e)->oldState();
1047  Qt::WindowStates newstate = view->windowState();
1048  if (oldstate != newstate)
1049  Q_EMIT windowStateChanged(view);
1050  }
1051  }
1052 
1053  // We don't want to show the bubble help for the what's this text but want to
1054  // start the help viewer with the according key word.
1055  // Thus, we have to observe WhatThis events if called for a widget, use its text and
1056  // must avoid to make the bubble widget visible.
1057  if (e->type() == QEvent::WhatsThis) {
1058  if (!o->isWidgetType())
1059  return false;
1060  // clicked on a widget in what's this mode
1061  auto w = static_cast<QWidget *>(o);
1062  d->whatstext = w->whatsThis();
1063  }
1064  if (e->type() == QEvent::WhatsThisClicked) {
1065  // if the widget is a top-level window
1066  if (o->isWidgetType() && qobject_cast<QWidget*>(o)->isWindow()) {
1067  // re-direct to the widget
1068  QApplication::sendEvent(this, e);
1069  }
1070  }
1071  // special treatment for menus because they directly call QWhatsThis::showText()
1072  // whereby we must be informed for which action the help should be shown
1073  if (o->inherits("QMenu") && QWhatsThis::inWhatsThisMode()) {
1074  bool whatthis = false;
1075  if (e->type() == QEvent::KeyPress) {
1076  auto ke = static_cast<QKeyEvent*>(e);
1077  if (ke->key() == Qt::Key_Return || ke->key() == Qt::Key_Enter || ke->key() == Qt::Key_F1)
1078  whatthis = true;
1079  }
1080  else if (e->type() == QEvent::MouseButtonRelease)
1081  whatthis = true;
1082  else if (e->type() == QEvent::EnterWhatsThisMode)
1083  whatthis = true;
1084  if (whatthis) {
1085  QAction* cur = static_cast<QMenu*>(o)->activeAction();
1086  if (cur) {
1087  // get the help text for later usage
1088  QString s = cur->whatsThis();
1089  if (s.isEmpty())
1090  s = static_cast<QMenu*>(o)->whatsThis();
1091  d->whatstext = s;
1092  }
1093  }
1094  }
1095  if (o->inherits("QWhatsThat") && e->type() == QEvent::Show) {
1096  // the bubble help should become visible which we avoid by marking the widget
1097  // that it is out of range. Instead of, we show the help viewer
1098  if (!d->whatstext.isEmpty()) {
1099  QWhatsThisClickedEvent e(d->whatstext);
1100  QApplication::sendEvent(this, &e);
1101  }
1102  static_cast<QWidget *>(o)->setAttribute(Qt::WA_OutsideWSRange);
1103  o->deleteLater();
1104  return true;
1105  }
1106  if (o->inherits("QWhatsThat") && e->type() == QEvent::Hide) {
1107  // leave what's this mode
1108  if (d->whatsthis) {
1109  d->whatsthis = false;
1110  d->whatstext.clear();
1111  qApp->removeEventFilter(this);
1112  }
1113  }
1114  }
1115 
1116  return QMainWindow::eventFilter(o, e);
1117 }
