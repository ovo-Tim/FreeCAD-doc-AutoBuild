46 def read(filename):
47  "reads a FCMat file and returns a dictionary from it"
48 
49  # the reader returns a dictionary in any case even if the file has problems
50  # an empty dict is returned in such case
51 
52  card_name_file = os.path.splitext(os.path.basename(filename))[0]
53  f = open(filename, encoding="utf8")
54  try:
55  content = f.readlines()
56  except Exception:
57  # https://forum.freecad.org/viewtopic.php?f=18&t=56912#p489721
58  # older FreeCAD do not write utf-8 for special character on windows
59  # I have seen "ISO-8859-15" or "windows-1252"
60  # explicit utf-8 writing, https://github.com/FreeCAD/FreeCAD/commit/9a564dd906f
61  print("Error on card loading. File might not utf-8.")
62  error_message = "Error on loading. Material file '{}' might not utf-8.".format(filename)
63  print("{}\n".format(error_message))
64  return {}
65  d = {}
66  d["Meta"] = {}
67  d["General"] = {}
68  d["Mechanical"] = {}
69  d["Fluidic"] = {}
70  d["Thermal"] = {}
71  d["Electromagnetic"] = {}
72  d["Architectural"] = {}
73  d["Rendering"] = {}
74  d["VectorRendering"] = {}
75  d["Cost"] = {}
76  d["UserDefined"] = {}
77  d["Meta"]["CardName"] = card_name_file # CardName is the MatCard file name
78  section = ''
79  for ln, line in enumerate(content):
80  # line numbers are used for CardName and AuthorAndLicense
81  # the use of line number is not smart for a data model
82  # a wrong user edit could break the file
83 
84  # comment
85  if line.startswith('#'):
86  # a '#' is assumed to be a comment which is ignored
87  continue
88  # CardName
89  if line.startswith(';') and ln == 0:
90  pass
91 
92  # AuthorAndLicense
93  elif line.startswith(';') and ln == 1:
94  v = line.split(";")[1].strip() # Line 2
95  if hasattr(v, "decode"):
96  v = v.decode('utf-8')
97  d["General"]["AuthorAndLicense"] = v # Move the field to the general group
98 
99  # rest
100  else:
101  # ; is a Comment
102  # [ is a Section
103  if line[0] == '[':
104  line = line[1:]
105  k = line.split("]", 1)
106  if len(k) >= 2:
107  v = k[0].strip()
108  if hasattr(v, "decode"):
109  v = v.decode('utf-8')
110  section = v
111  elif line[0] not in ";":
112  # split once on first occurrence
113  # a link could contain a '=' and thus would be split
114  k = line.split("=", 1)
115  if len(k) == 2:
116  v = k[1].strip()
117  if hasattr(v, "decode"):
118  v = v.decode('utf-8')
119  d[section][k[0].strip()] = v
120  return d

230 def saveYaml(card, output):
