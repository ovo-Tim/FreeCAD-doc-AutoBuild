51 void DlgPreferencePackManagementImp::showEvent(QShowEvent* event)
52 {
53  // Separate out user-saved packs from installed packs: we can remove individual user-saved packs,
54  // but can only disable individual installed packs (though we can completely uninstall the pack's
55  // containing Addon by redirecting to the Addon Manager).
56  auto savedPreferencePacksDirectory = Application::Instance->prefPackManager()->getSavedPreferencePacksPath();
57  auto modDirectories = Application::Instance->prefPackManager()->modPaths();
58  auto resourcePath = Application::Instance->prefPackManager()->getResourcePreferencePacksPath();
59 
60  // The displayed tree has two levels: at the toplevel is either "User-Saved Packs" or the name
61  // of the addon containing the pack. Beneath those are the individual packs themselves. The tree view shows
62  // "Hide"/"Show" for packs installed as a Mod, and "Delete" for packs in the user-saved pack
63  // section.
64  auto userPacks = Application::Instance->prefPackManager()->getPacksFromDirectory(savedPreferencePacksDirectory);
65 
66  auto builtinPacks = Application::Instance->prefPackManager()->getPacksFromDirectory(resourcePath);
67 
68  std::map<std::string, std::vector<std::string>> installedPacks;
69  for (const auto& modDirectory : modDirectories) {
70  if (fs::exists(modDirectory) && fs::is_directory(modDirectory)) {
71  for (const auto& mod : fs::directory_iterator(modDirectory)) {
72  auto packs = Application::Instance->prefPackManager()->getPacksFromDirectory(mod);
73  if (!packs.empty()) {
74  auto modName = Base::FileInfo::pathToString(mod.path().filename());
75  installedPacks.emplace(modName, packs);
76  }
77  }
78  }
79  }
80 
81  ui->treeWidget->clear(); // Begin by clearing whatever is there
82  ui->treeWidget->header()->setDefaultAlignment(Qt::AlignLeft);
83  ui->treeWidget->setColumnCount(2);
84  ui->treeWidget->setSelectionMode(QAbstractItemView::SelectionMode::NoSelection);
85  ui->treeWidget->header()->setStretchLastSection(false);
86  ui->treeWidget->header()->setSectionResizeMode(0, QHeaderView::ResizeMode::Stretch);
87  ui->treeWidget->header()->setSectionResizeMode(1, QHeaderView::ResizeMode::ResizeToContents);
88 
89  if (!userPacks.empty()) {
90  addTreeNode(tr("User-Saved Preference Packs").toStdString(), userPacks, TreeWidgetType::USER);
91  }
92 
93  if (!builtinPacks.empty()) {
94  addTreeNode(tr("Built-In Preference Packs").toStdString(), builtinPacks, TreeWidgetType::BUILTIN);
95  }
96 
97  for (const auto& installedPack : installedPacks) {
98  addTreeNode(installedPack.first, installedPack.second, TreeWidgetType::ADDON);
99  }
100 
101  QDialog::showEvent(event);
102 }

104 void DlgPreferencePackManagementImp::addTreeNode(const std::string &name, const std::vector<std::string> &contents, TreeWidgetType twt)
105 {
106  static const auto iconIsVisible = QIcon(QLatin1String(":/icons/dagViewVisible.svg"));
107  static const auto iconIsInvisible = QIcon(QLatin1String(":/icons/Invisible.svg"));
108  auto packRoot = new QTreeWidgetItem();
109  packRoot->setText(0, QString::fromStdString(name));
110  std::vector<QTreeWidgetItem*> items;
111  for (const auto& packName : contents) {
112  auto pack = new QTreeWidgetItem(packRoot);
113  pack->setText(0, QString::fromStdString(packName));
114  items.push_back(pack);
115  }
116  ui->treeWidget->addTopLevelItem(packRoot);
117  packRoot->setExpanded(true);
118  for (const auto item : items) {
119  auto button = new QPushButton();
120  button->setFlat(true);
121  switch (twt) {
122  break; case TreeWidgetType::BUILTIN:
123  // The button is a "hide" button
124  if (Application::Instance->prefPackManager()->isVisible("##BUILT_IN##", item->text(0).toStdString()))
125  button->setIcon(iconIsVisible);
126  else
127  button->setIcon(iconIsInvisible);
128  button->setToolTip(tr("Toggle visibility of built-in preference pack '%1'").arg(item->text(0)));
129  connect(button, &QPushButton::clicked, [this, name, item]() {
130  this->hideBuiltInPack(item->text(0).toStdString());
131  });
132  break; case TreeWidgetType::USER:
133  // The button is a "delete" button
134  button->setIcon(QIcon(QLatin1String(":/icons/delete.svg")));
135  button->setToolTip(tr("Delete user-saved preference pack '%1'").arg(item->text(0)));
136  connect(button, &QPushButton::clicked, [this, item]() {
137  this->deleteUserPack(item->text(0).toStdString());
138  });
139  break; case TreeWidgetType::ADDON:
140  // The button is a "hide" button
141  if (Application::Instance->prefPackManager()->isVisible(name, item->text(0).toStdString()))
142  button->setIcon(iconIsVisible);
143  else
144  button->setIcon(iconIsInvisible);
145  button->setToolTip(tr("Toggle visibility of Addon preference pack '%1' (use Addon Manager to permanently remove)").arg(item->text(0)));
146  connect(button, &QPushButton::clicked, [this, name, item]() {
147  this->hideInstalledPack(name, item->text(0).toStdString());
148  });
149  }
150  ui->treeWidget->setItemWidget(item, 1, button);
151  }
152 }

154 void DlgPreferencePackManagementImp::deleteUserPack(const std::string& name)
155 {
156  // Do the deletion here...
157  auto result = QMessageBox::warning(this, tr("Delete saved preference pack?"),
158  tr("Are you sure you want to delete the preference pack named '%1'? This cannot be undone.").arg(QString::fromStdString(name)),
159  QMessageBox::Yes | QMessageBox::Cancel, QMessageBox::Cancel);
