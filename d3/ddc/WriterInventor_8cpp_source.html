82  void addNormalNode(const MeshCore::MeshKernel& kernel)
83  {
84  MeshFacetIterator clIter(kernel), clEnd(kernel);
85  const MeshGeomFacet* geomFacet = nullptr;
86 
87  // write out the normals of the facets
88  std::vector<Base::Vector3f> normals;
89  normals.reserve(kernel.CountFacets());
90 
91  clIter.Begin();
92  clEnd.End();
93 
94  while (clIter < clEnd) {
95  geomFacet = &(*clIter);
96  normals.push_back(geomFacet->GetNormal());
97  ++clIter;
98  }
99 
100  builder.addNode(Base::NormalItem {normals});
101 
102  Base::NormalBindingItem binding;
103  binding.setValue(Base::BindingElement::Binding::PerFace);
104  builder.addNode(binding);
105  }

119  void addMaterialNode(const Material* material)
120  {
121  if (!material) {
122  return;
123  }
124 
125  auto transformColors = [](const std::vector<App::Color>& input) {
126  std::vector<Base::ColorRGB> output;
127  output.reserve(input.size());
128  std::transform(input.cbegin(),
129  input.cend(),
130  std::back_inserter(output),
131  [](const App::Color& col) {
132  return Base::ColorRGB {col.r, col.g, col.b};
133  });
134 
135  return output;
136  };
137 
138  Base::MaterialItem mat;
139  mat.setAmbientColor(transformColors(material->ambientColor));
140  mat.setDiffuseColor(transformColors(material->diffuseColor));
141  mat.setSpecularColor(transformColors(material->specularColor));
142  mat.setEmissiveColor(transformColors(material->emissiveColor));
143  mat.setShininess(material->shininess);
144  mat.setTransparency(material->transparency);
145  builder.addNode(mat);
146  }

148  void addMaterialBindingNode(const Material* material)

158  break;
159  case MeshIO::PER_VERTEX:
160  binding.setValue(Base::BindingElement::Binding::PerVertex);
161  break;
162  default:
163  binding.setValue(Base::BindingElement::Binding::Overall);
164  break;
165  }
166 
167  builder.addNode(binding);
168  }

170  void addIndexedFaceSetNode(const MeshCore::MeshKernel& kernel)
171  {
172  // and finally the facets with their point indices
173  const MeshFacetArray& faces = kernel.GetFacets();
174  std::vector<int> indices;
175  indices.reserve(4 * faces.size());
176  for (const auto& it : faces) {
177  indices.push_back(static_cast<int>(it._aulPoints[0]));
178  indices.push_back(static_cast<int>(it._aulPoints[1]));
179  indices.push_back(static_cast<int>(it._aulPoints[2]));
180  indices.push_back(-1);
181  }
182 
183  builder.addNode(Base::IndexedFaceSetItem {indices});
184  }

200 bool WriterInventor::Save(std::ostream& out)
201 {

203  return false;
204  }
205

210  writer.setupStream(out);

213  writer.addTransformNode(_transform, apply_transform);
214  writer.addNormalNode(_kernel);
215  writer.addCoordinateNode(_kernel);
216  writer.addMaterialNode(_material);
217  writer.addMaterialBindingNode(_material);
218  writer.addIndexedFaceSetNode(_kernel);
219 
220  builder.endSeparator();
221 
222  return true;
223 }
