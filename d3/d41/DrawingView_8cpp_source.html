96 void SvgView::openFile(const QFile& file)
97 {
98  if (!file.exists()) {
99  return;
100  }
101 
102  QGraphicsScene* s = scene();
103 
104  bool drawBackground = (m_backgroundItem ? m_backgroundItem->isVisible() : true);
105  bool drawOutline = (m_outlineItem ? m_outlineItem->isVisible() : false);
106 
107  s->clear();
108  resetTransform();
109 
110  m_svgItem = new QGraphicsSvgItem(file.fileName());
111  m_svgItem->setFlags(QGraphicsItem::ItemClipsToShape);
112  m_svgItem->setCacheMode(QGraphicsItem::NoCache);
113  m_svgItem->setZValue(0);
114 
115  m_backgroundItem = new QGraphicsRectItem(m_svgItem->boundingRect());
116  m_backgroundItem->setBrush(Qt::white);
117  m_backgroundItem->setPen(Qt::NoPen);
118  m_backgroundItem->setVisible(drawBackground);
119  m_backgroundItem->setZValue(-1);
120 
121  m_outlineItem = new QGraphicsRectItem(m_svgItem->boundingRect());
122  QPen outline(Qt::black, 2, Qt::DashLine);
123  outline.setCosmetic(true);
124  m_outlineItem->setPen(outline);
125  m_outlineItem->setBrush(Qt::NoBrush);
126  m_outlineItem->setVisible(drawOutline);
127  m_outlineItem->setZValue(1);
128 
129  s->addItem(m_backgroundItem);
130  s->addItem(m_svgItem);
131  s->addItem(m_outlineItem);
132 
133  // use the actual bounding box of the SVG template to avoid any scaling effect
134  // when printing the drawing (#0000932)
135  s->setSceneRect(m_outlineItem->boundingRect());
136 }

570 void DrawingView::print(QPrinter* printer)
571 {
572  // As size of the render area paperRect() should be used. When performing a real
573  // print pageRect() may also work but the output is cropped at the bottom part.
574  // So, independent whether pageRect() or paperRect() is used there is no scaling effect.
575  // However, when using a different paper size as set in the drawing template (e.g.
576  // DIN A5 instead of DIN A4) then the output is scaled.
577  //
578  // When creating a PDF file there seems to be no difference between pageRect() and
579  // paperRect().
580  //
581  // When showing the preview of a print paperRect() must be used because with pageRect()
582  // a certain scaling effect can be observed and the content becomes smaller.
583  QPaintEngine::Type paintType = printer->paintEngine()->type();
584  if (printer->outputFormat() == QPrinter::NativeFormat) {
585  int w = printer->widthMM();
586  int h = printer->heightMM();
587  QPageSize::PageSizeId realPaperSize = getPageSize(w, h);
588  QPageSize::PageSizeId curPaperSize = printer->pageLayout().pageSize().id();
589 
590  // for the preview a 'Picture' paint engine is used which we don't
591  // care if it uses wrong printer settings
592  bool doPrint = paintType != QPaintEngine::Picture;
593 
594  if (doPrint && printer->pageLayout().orientation() != this->m_orientation) {
595  int ret = QMessageBox::warning(
596  this,
597  tr("Different orientation"),
598  tr("The printer uses a different orientation than the drawing.\n"
599  "Do you want to continue?"),
600  QMessageBox::Yes | QMessageBox::No);
601  if (ret != QMessageBox::Yes) {
602  return;
603  }
604  }
605  else if (doPrint && realPaperSize != this->m_pageSize) {
606  int ret = QMessageBox::warning(
607  this,
608  tr("Different paper size"),
609  tr("The printer uses a different paper size than the drawing.\n"
610  "Do you want to continue?"),
611  QMessageBox::Yes | QMessageBox::No);
612  if (ret != QMessageBox::Yes) {
613  return;
614  }
615  }
616  else if (doPrint && curPaperSize != this->m_pageSize) {
617  int ret = QMessageBox::warning(
618  this,
619  tr("Different paper size"),
620  tr("The printer uses a different paper size than the drawing.\n"
621  "Do you want to continue?"),
622  QMessageBox::Yes | QMessageBox::No);
623  if (ret != QMessageBox::Yes) {
624  return;
625  }
626  }
627  }
628 
629  QPainter p(printer);
630  if (!p.isActive() && !printer->outputFileName().isEmpty()) {
631  qApp->setOverrideCursor(Qt::ArrowCursor);
632  QMessageBox::critical(
633  this,
634  tr("Opening file failed"),
635  tr("Can't open file '%1' for writing.").arg(printer->outputFileName()));
636  qApp->restoreOverrideCursor();
637  return;
638  }
639  QRect rect = printer->pageLayout().fullRectPixels(printer->resolution());
640 #ifdef Q_OS_WIN32
641  // On Windows the preview looks broken when using paperRect as render area.
642  // Although the picture is scaled when using pageRect, it looks just fine.
643  if (paintType == QPaintEngine::Picture) {
644  rect = printer->pageLayout().paintRectPixels(printer->resolution());
645  }
646 #endif
647  this->m_view->scene()->render(&p, rect);
648  p.end();
649 }

651 QPageSize::PageSizeId DrawingView::getPageSize(int w, int h) const
652 {
653  static const float paperSizes[][2] = {
