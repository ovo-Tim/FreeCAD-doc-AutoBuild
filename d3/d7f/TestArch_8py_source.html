400  def testWallMultiMatAlign(self):
401  App.Console.PrintLog ('Checking Arch Wall with MultiMaterial and 3 alignments...\n')
402  matA = Arch.makeMaterial()
403  matB = Arch.makeMaterial()
404  matMulti = Arch.makeMultiMaterial()
405  matMulti.Materials = [matA, matB]
406  matMulti.Thicknesses = [100, 200] # total width different from default 200
407  pts = [App.Vector( 0, 0, 0),
408  App.Vector(1000, 0, 0),
409  App.Vector(1000, 1000, 0),
410  App.Vector(2000, 1000, 0)]
411  # wall based on wire:
412  wire = Draft.makeWire(pts)
413  wallWire = Arch.makeWall(wire)
414  wallWire.Material = matMulti
415  # wall based on sketch:
416  sketch = App.activeDocument().addObject('Sketcher::SketchObject','Sketch')
417  sketch.addGeometry([Part.LineSegment(pts[0], pts[1]),
418  Part.LineSegment(pts[1], pts[2]),
419  Part.LineSegment(pts[2], pts[3])])
420  wallSketch = Arch.makeWall(sketch)
421  wallSketch.Material = matMulti
422 
423  alignLst = ["Left", "Center", "Right"]
424  checkLst = [[App.Vector(0, -300, 0), App.Vector(2000, 1000, 0)],
425  [App.Vector(0, -150, 0), App.Vector(2000, 1150, 0)],
426  [App.Vector(0, 0, 0), App.Vector(2000, 1300, 0)]]
427  for i in range(3):
428  wallWire.Align = alignLst[i]
429  wallSketch.Align = alignLst[i]
430  App.ActiveDocument.recompute()
431  for box in [wallWire.Shape.BoundBox, wallSketch.Shape.BoundBox]:
432  ptMin = App.Vector(box.XMin, box.YMin, 0)
433  self.assertTrue(ptMin.isEqual(checkLst[i][0], 1e-8),
434  "Arch Wall with MultiMaterial and 3 alignments failed")
435  ptMax = App.Vector(box.XMax, box.YMax, 0)
436  self.assertTrue(ptMax.isEqual(checkLst[i][1], 1e-8),
437  "Arch Wall with MultiMaterial and 3 alignments failed")

541  overhang = overhangsLst)
