128 void Tessellation::onComboFinenessCurrentIndexChanged(int index)
129 {

131  if (index == 5) {
132  ui->doubleGrading->setEnabled(true);
133  ui->spinEdgeElements->setEnabled(true);

137  ui->doubleGrading->setEnabled(false);
138  ui->spinEdgeElements->setEnabled(false);
139  ui->spinCurvatureElements->setEnabled(false);

141 
142  switch (index) {
143  case VeryCoarse:
144  ui->doubleGrading->setValue(0.7);
145  ui->spinEdgeElements->setValue(0.3);
146  ui->spinCurvatureElements->setValue(1.0);

149  ui->doubleGrading->setValue(0.5);
150  ui->spinEdgeElements->setValue(0.5);
151  ui->spinCurvatureElements->setValue(1.5);
152  break;

155  ui->spinEdgeElements->setValue(1.0);
156  ui->spinCurvatureElements->setValue(2.0);
157  break;
158  case Fine:
159  ui->doubleGrading->setValue(0.2);
160  ui->spinEdgeElements->setValue(2.0);
161  ui->spinCurvatureElements->setValue(3.0);
162  break;
163  case VeryFine:
164  ui->doubleGrading->setValue(0.1);
165  ui->spinEdgeElements->setValue(3.0);
166  ui->spinCurvatureElements->setValue(5.0);
167  break;
168  default:

206 void Tessellation::onEstimateMaximumEdgeLengthClicked()
207 {
208  App::Document* activeDoc = App::GetApplication().getActiveDocument();
209  if (!activeDoc) {
210  return;

232 bool Tessellation::accept()
233 {
234  std::list<App::SubObjectT> shapeObjects;
235  App::Document* activeDoc = App::GetApplication().getActiveDocument();
236  if (!activeDoc) {
237  QMessageBox::critical(this, windowTitle(), tr("No active document"));
238  return false;
239  }
240 
241  Gui::Document* activeGui = Gui::Application::Instance->getDocument(activeDoc);
242  if (!activeGui) {
243  QMessageBox::critical(this, windowTitle(), tr("No active document"));
244  return false;
245  }
246 
247  this->document = QString::fromLatin1(activeDoc->getName());
248 
249  bool bodyWithNoTip = false;
250  bool partWithNoFace = false;
251  for (auto& sel : Gui::Selection().getSelection("*", Gui::ResolveMode::NoResolve)) {
252  auto shape = Part::Feature::getTopoShape(sel.pObject, sel.SubName);
253  if (shape.hasSubShape(TopAbs_FACE)) {
254  shapeObjects.emplace_back(sel.pObject, sel.SubName);
255  }
256  else if (sel.pObject) {
257  if (sel.pObject->isDerivedFrom(Part::Feature::getClassTypeId())) {
258  partWithNoFace = true;
259  }
260  if (auto body = dynamic_cast<Part::BodyBase*>(sel.pObject)) {
261  if (!body->Tip.getValue()) {
262  bodyWithNoTip = true;
263  }
264  }
265  }
266  }
267 
268  if (shapeObjects.empty()) {
269  if (bodyWithNoTip) {
270  QMessageBox::critical(
271  this,
272  windowTitle(),
273  tr("You have selected a body without tip.\n"
274  "Either set the tip of the body or select a different shape, please."));
275  }
276  else if (partWithNoFace) {
277  QMessageBox::critical(this,
278  windowTitle(),
279  tr("You have selected a shape without faces.\n"
280  "Select a different shape, please."));
281  }
282  else {
283  QMessageBox::critical(this, windowTitle(), tr("Select a shape for meshing, first."));
284  }
285  return false;
286  }
287 
288  bool doClose = !ui->checkBoxDontQuit->isChecked();
289  int method = ui->stackedWidget->currentIndex();
290 
291  // For Gmsh the workflow is very different because it uses an executable
292  // and therefore things are asynchronous
293  if (method == Gmsh) {
294  gmsh->process(activeDoc, shapeObjects);
295  return false;
296  }
297 
298  process(method, activeDoc, shapeObjects);
299  return doClose;
300 }

302 void Tessellation::process(int method,
303  App::Document* doc,
304  const std::list<App::SubObjectT>& shapeObjects)
305 {
306  try {
307  Gui::WaitCursor wc;
308 
309  saveParameters(method);
310 
311  doc->openTransaction("Meshing");
312  for (auto& info : shapeObjects) {
313  QString subname = QString::fromLatin1(info.getSubName().c_str());
314  QString objname = QString::fromLatin1(info.getObjectName().c_str());
315 
316  auto obj = info.getObject();
317  if (!obj) {
318  continue;
319  }
320  auto sobj = obj->getSubObject(info.getSubName().c_str());
321  if (!sobj) {
322  continue;
323  }
324  sobj = sobj->getLinkedObject(true);
325  if (!sobj) {
326  continue;
327  }
328 
329  QString label = QString::fromUtf8(sobj->Label.getValue());
330 
331  QString param = getMeshingParameters(method, sobj);
332 
333  QString cmd =
334  QString::fromLatin1("__doc__=FreeCAD.getDocument(\"%1\")\n"
335  "__mesh__=__doc__.addObject(\"Mesh::Feature\",\"Mesh\")\n"
336  "__part__=__doc__.getObject(\"%2\")\n"
337  "__shape__=Part.getShape(__part__,\"%3\")\n"
338  "__mesh__.Mesh=MeshPart.meshFromShape(%4)\n"
339  "__mesh__.Label=\"%5 (Meshed)\"\n"
340  "del __doc__, __mesh__, __part__, __shape__\n")
341  .arg(this->document, objname, subname, param, label);
342 
343  Gui::Command::runCommand(Gui::Command::Doc, cmd.toUtf8());
344 
345  setFaceColors(method, doc, sobj);
346  }
347  doc->commitTransaction();
348  }
349  catch (const Base::Exception& e) {
350  doc->abortTransaction();
351  Base::Console().Error(e.what());
352  }
353 }

393 void Tessellation::addFaceColors(Mesh::Feature* mesh, const std::vector<App::Color>& colorPerSegm)
394 {
395  const Mesh::MeshObject& kernel = mesh->Mesh.getValue();
396  unsigned long numSegm = kernel.countSegments();
397  if (numSegm > 0 && numSegm == colorPerSegm.size()) {
398  unsigned long uCtFacets = kernel.countFacets();
399  std::vector<App::Color> colorPerFace(uCtFacets);
400  for (unsigned long i = 0; i < numSegm; i++) {
401  App::Color segmColor = colorPerSegm[i];
402  std::vector<Mesh::FacetIndex> segm = kernel.getSegment(i).getIndices();
403  for (Mesh::FacetIndex it : segm) {
404  colorPerFace[it] = segmColor;
405  }
406  }
407 
408  auto typeId = App::PropertyColorList::getClassTypeId();
409  if (auto prop = dynamic_cast<App::PropertyColorList*>(
410  mesh->addDynamicProperty(typeId.getName(), "FaceColors"))) {
411  prop->setValues(colorPerFace);
412  }
413  }
414 }

565 bool Mesh2ShapeGmsh::writeProject(QString& inpFile, QString& outFile)
566 {
567  if (!d->shapes.empty()) {
568  App::SubObjectT sub = d->shapes.front();
569  d->shapes.pop_front();
570 
571  App::DocumentObject* part = sub.getObject();
572  if (part) {
573  Part::TopoShape shape = Part::Feature::getTopoShape(part, sub.getSubName().c_str());
574  shape.exportBrep(d->cadFile.c_str());
575  d->label = part->Label.getStrValue() + " (Meshed)";
576 
577  // Parameters
578  int algorithm = meshingAlgorithm();
579  double maxSize = getMaxSize();
580  if (maxSize == 0.0) {
581  maxSize = 1.0e22;
582  }
583  double minSize = getMinSize();
584 
585  // Gmsh geo file
586  Base::FileInfo geo(d->geoFile);
587  Base::ofstream geoOut(geo, std::ios::out);
588  geoOut << "// geo file for meshing with Gmsh meshing software created by FreeCAD\n"
589  << "// open brep geometry\n"
590  << "Merge \"" << d->cadFile << "\";\n\n"
591  << "// Characteristic Length\n"
592  << "// no boundary layer settings for this mesh\n"
593  << "// min, max Characteristic Length\n"
594  << "Mesh.CharacteristicLengthMax = " << maxSize << ";\n"
595  << "Mesh.CharacteristicLengthMin = " << minSize << ";\n\n"
596  << "// optimize the mesh\n"
597  << "Mesh.Optimize = 1;\n"
598  << "Mesh.OptimizeNetgen = 0;\n"
599  << "// High-order meshes optimization (0=none, 1=optimization, "
600  "2=elastic+optimization, 3=elastic, 4=fast curving)\n"
601  << "Mesh.HighOrderOptimize = 0;\n\n"
602  << "// mesh order\n"
603  << "Mesh.ElementOrder = 2;\n"
604  << "// Second order nodes are created by linear interpolation instead by "
605  "curvilinear\n"
606  << "Mesh.SecondOrderLinear = 1;\n\n"
607  << "// mesh algorithm, only a few algorithms are usable with 3D boundary layer "
608  "generation\n"
609  << "// 2D mesh algorithm (1=MeshAdapt, 2=Automatic, 5=Delaunay, 6=Frontal, "
610  "7=BAMG, 8=DelQuad, 9=Packing of Parallelograms, 11=Quasi-structured Quad)\n"
611  << "Mesh.Algorithm = " << algorithm << ";\n"
612  << "// 3D mesh algorithm (1=Delaunay, 2=New Delaunay, 4=Frontal, 7=MMG3D, "
613  "9=R-tree, 10=HTX)\n"
614  << "Mesh.Algorithm3D = 1;\n\n"
615  << "// meshing\n"
616  << "// set geometrical tolerance (also used for merging nodes)\n"
617  << "Geometry.Tolerance = 1e-06;\n"
618  << "Mesh 2;\n"
619  << "Coherence Mesh; // Remove duplicate vertices\n";
620  geoOut.close();
621 
622  inpFile = QString::fromUtf8(d->geoFile.c_str());
623  outFile = QString::fromUtf8(d->stlFile.c_str());
624 
625  return true;
626  }
627  }
628  else {
629  App::Document* doc = d->doc.getDocument();
630  if (doc) {
631  doc->commitTransaction();
632  }
633 
634  Q_EMIT processed();
635  }
636 
637  return false;
638 }

640 bool Mesh2ShapeGmsh::loadOutput()
641 {
642  App::Document* doc = d->doc.getDocument();
643  if (!doc) {
644  return false;
645  }
646 
647  // Now read-in the mesh
648  Base::FileInfo stl(d->stlFile);
649  Base::FileInfo geo(d->geoFile);
650 
651  Mesh::MeshObject kernel;
652  MeshCore::MeshInput input(kernel.getKernel());
653  Base::ifstream stlIn(stl, std::ios::in | std::ios::binary);
654  input.LoadBinarySTL(stlIn);
655  stlIn.close();
656  kernel.harmonizeNormals();
