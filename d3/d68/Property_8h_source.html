367 template<class P>
368 class AtomicPropertyChangeInterface
369 {
370 protected:
371  AtomicPropertyChangeInterface() = default;
372 
373 public:
374  class AtomicPropertyChange
375  {
376  public:
384  explicit AtomicPropertyChange(P& prop, bool markChange = true)
385  : mProp(prop)
386  {
387  mProp.signalCounter++;
388  if (markChange) {
389  aboutToChange();
390  }
391  }
392 
398  void aboutToChange()
399  {
400  if (!mProp.hasChanged) {
401  mProp.hasChanged = true;
402  mProp.aboutToSetValue();
403  }
404  }
405 
412  ~AtomicPropertyChange()
413  {
414  // Signal counter == 1? meaning we are the last one. Invoke
415  // hasSetValue() before decrease counter to prevent recursive call
416  // triggered by another AtomicPropertyChange created inside
417  // hasSetValue(), as it has now been changed to a virtual function.
418  if (mProp.signalCounter == 1 && mProp.hasChanged) {
419  // Must make sure to not throw in a destructor
420  try {
421  mProp.hasSetValue();
422  }
423  catch (Base::Exception& e) {
424  e.ReportException();
425  }
426  catch (...) {
427  }
428  mProp.hasChanged = false;
429  }
430  if (mProp.signalCounter > 0) {
431  mProp.signalCounter--;
432  }
433  }
434 
440  // Destructor cannot throw. So we provide this function to allow error
441  // propagation.
442  void tryInvoke()
443  {
444  if (mProp.signalCounter == 1 && mProp.hasChanged) {
445  mProp.hasSetValue();
446  if (mProp.signalCounter > 0) {
447  --mProp.signalCounter;
448  }
449  mProp.hasChanged = false;
450  }
451  }
452 
453  private:
454  P& mProp;
455  };
456 
457 protected:
458  int signalCounter {0};
459  bool hasChanged {false};
460 };
