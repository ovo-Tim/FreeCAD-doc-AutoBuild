162 void TaskLinearPatternParameters::updateUI()
163 {
164  if (blockUpdate) {
165  return;
166  }
167  blockUpdate = true;
168 
169  auto pcLinearPattern = getObject<PartDesign::LinearPattern>();
170  auto mode = static_cast<PartDesign::LinearPatternMode>(pcLinearPattern->Mode.getValue());
171 
172  bool reverse = pcLinearPattern->Reversed.getValue();
173  double length = pcLinearPattern->Length.getValue();
174  double offset = pcLinearPattern->Offset.getValue();
175  unsigned occurrences = pcLinearPattern->Occurrences.getValue();
176 
177  if (dirLinks.setCurrentLink(pcLinearPattern->Direction) == -1) {
178  // failed to set current, because the link isn't in the list yet
179  dirLinks.addLink(pcLinearPattern->Direction,
180  getRefStr(pcLinearPattern->Direction.getValue(),
181  pcLinearPattern->Direction.getSubValues()));
182  dirLinks.setCurrentLink(pcLinearPattern->Direction);
183  }
184 
185  // Note: This block of code would trigger change signal handlers (e.g. onOccurrences())
186  // and another updateUI() if we didn't check for blockUpdate
187  ui->checkReverse->setChecked(reverse);
188  ui->comboMode->setCurrentIndex(static_cast<int>(mode));
189  ui->spinLength->setValue(length);
190  ui->spinOffset->setValue(offset);
191  ui->spinOccurrences->setValue(occurrences);
192 
193  blockUpdate = false;
194 }

218 void TaskLinearPatternParameters::onSelectionChanged(const Gui::SelectionChanges& msg)
219 {
220  if (selectionMode != SelectionMode::None && msg.Type == Gui::SelectionChanges::AddSelection) {
221  if (originalSelected(msg)) {
222  exitSelectionMode();
223  }
224  else {
225  auto pcLinearPattern = getObject<PartDesign::LinearPattern>();
226 
227  std::vector<std::string> directions;
228  App::DocumentObject* selObj = nullptr;
229  getReferencedSelection(pcLinearPattern, msg, selObj, directions);
230  if (!selObj) {
231  return;
232  }
233 
234  // Note: ReferenceSelection has already checked the selection for validity
235  if (selectionMode == SelectionMode::Reference || selObj->isDerivedFrom<App::Line>()) {
236  setupTransaction();
237  pcLinearPattern->Direction.setValue(selObj, directions);
238  recomputeFeature();
239  updateUI();
240  }
241  exitSelectionMode();
242  }
243  }
244 }
