69 float SoZoomTranslation::calculateScaleFactor(SoAction* action) const
70 {
71  // Dividing by 5 seems to work well
72  SbViewVolume vv = SoViewVolumeElement::get(action->getState());
73  float aspectRatio = SoViewportRegionElement::get(action->getState()).getViewportAspectRatio();
74  scaleFactor = vv.getWorldToScreenScale(SbVec3f(0.f, 0.f, 0.f), 0.1f) / (5 * aspectRatio);
75  return scaleFactor;
76 }

91 void SoZoomTranslation::doAction(SoAction* action)
92 {
93  SbVec3f v;
94  if (this->translation.getValue() == SbVec3f(0.0f, 0.0f, 0.0f)
95  && this->abPos.getValue() == SbVec3f(0.0f, 0.0f, 0.0f)) {
96  return;
97  }
98  else {
99  SbVec3f absVtr = this->abPos.getValue();
100  SbVec3f relVtr = this->translation.getValue();
101 
102  float sf = this->calculateScaleFactor(action);
103  // For Sketcher Keep Z value the same
104  relVtr[0] = (relVtr[0] != 0) ? sf * relVtr[0] : 0;
105  relVtr[1] = (relVtr[1] != 0) ? sf * relVtr[1] : 0;
106 
107  v = absVtr + relVtr;
108  }
109 
110  SoModelMatrixElement::translateBy(action->getState(), this, v);
111 }
