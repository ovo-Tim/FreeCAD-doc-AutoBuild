86 private:
87  void updateDataAndDrawToPosition(Base::Vector2d onSketchPos) override
88  {
89  Q_UNUSED(onSketchPos);
90  switch (state()) {
91  case SelectMode::SeekFirst: {
92  int VtId = getPreselectPoint();
93  int CrvId = getPreselectCurve();
94  int CrsId = getPreselectCross();
95  Sketcher::SketchObject* obj = sketchgui->getSketchObject();
96 
97  if (VtId >= 0) { // Vertex
98  SketchObject* Obj = sketchgui->getSketchObject();
99  Obj->getGeoVertexIndex(VtId, refGeoId, refPosId);
100  }
101  else if (CrsId == 0) { // RootPoint
102  refGeoId = Sketcher::GeoEnum::RtPnt;
103  refPosId = Sketcher::PointPos::start;
104  }
105  else if (CrsId == 1) { // H_Axis
106  refGeoId = Sketcher::GeoEnum::HAxis;
107  refPosId = Sketcher::PointPos::none;
108  }
109  else if (CrsId == 2) { // V_Axis
110  refGeoId = Sketcher::GeoEnum::VAxis;
111  refPosId = Sketcher::PointPos::none;
112  }
113  else if ((CrvId >= 0 || CrvId <= Sketcher::GeoEnum::RefExt)
114  && isLineSegment(*obj->getGeometry(CrvId))) { // Curves
115  refGeoId = CrvId;
116  refPosId = Sketcher::PointPos::none;
117  }
118  else {
119  refGeoId = Sketcher::GeoEnum::GeoUndef;
120  refPosId = Sketcher::PointPos::none;
121  }
122 
123 
124  CreateAndDrawShapeGeometry();
125  } break;
126  default:
127  break;
128  }
129  }

218  void deleteOriginalGeos()
219  {
220  std::stringstream stream;
221  for (size_t j = 0; j < listOfGeoIds.size() - 1; j++) {
222  stream << listOfGeoIds[j] << ",";
223  }
224  stream << listOfGeoIds[listOfGeoIds.size() - 1];
225  try {
226  Gui::cmdAppObjectArgs(sketchgui->getObject(),
227  "delGeometries([%s])",
228  stream.str().c_str());
229  }
230  catch (const Base::Exception& e) {

235  void createShape(bool onlyeditoutline) override
236  {
237  SketchObject* Obj = sketchgui->getSketchObject();
238 
239  ShapeGeometry.clear();
240 
241  if (refGeoId == Sketcher::GeoEnum::GeoUndef) {
242  return;
243  }
244 
245  if (onlyeditoutline) {
246  std::map<int, int> dummy1;
247  std::map<int, bool> dummy2;
248  std::vector<Part::Geometry*> symGeos =
249  Obj->getSymmetric(listOfGeoIds, dummy1, dummy2, refGeoId, refPosId);
250 
251  for (auto* geo : symGeos) {
252  ShapeGeometry.emplace_back(geo);
253  }
254  }
255  }
