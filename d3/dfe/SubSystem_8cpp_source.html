53 void SubSystem::initialize(VEC_pD& params, MAP_pD_pD& reductionmap)
54 {
55  csize = static_cast<int>(clist.size());
56 
57  // tmpplist will contain the subset of parameters from params that are
58  // relevant for the constraints listed in clist
59  VEC_pD tmpplist;
60  {
61  SET_pD s1(params.begin(), params.end());
62  SET_pD s2;
63  for (std::vector<Constraint*>::iterator constr = clist.begin(); constr != clist.end();
64  ++constr) {
65  (*constr)
66  ->revertParams(); // ensure that the constraint points to the original parameters
67  VEC_pD constr_params = (*constr)->params();
68  s2.insert(constr_params.begin(), constr_params.end());
69  }
70  std::set_intersection(s1.begin(),
71  s1.end(),
72  s2.begin(),
73  s2.end(),
74  std::back_inserter(tmpplist));
75  }
76 
77  plist.clear();
78  MAP_pD_I rindex;
79  if (!reductionmap.empty()) {
80  int i = 0;
81  MAP_pD_I pindex;
82  for (VEC_pD::const_iterator itt = tmpplist.begin(); itt != tmpplist.end(); ++itt) {
83  MAP_pD_pD::const_iterator itr = reductionmap.find(*itt);
84  if (itr != reductionmap.end()) {
85  MAP_pD_I::const_iterator itp = pindex.find(itr->second);
86  if (itp
87  == pindex.end()) { // the reduction target is not in plist yet, so add it now
88  plist.push_back(itr->second);
89  rindex[itr->first] = i;
90  pindex[itr->second] = i;
91  i++;
92  }
93  else { // the reduction target is already in plist, just inform rindex
94  rindex[itr->first] = itp->second;
95  }
96  }
97  else if (pindex.find(*itt) == pindex.end()) { // not in plist yet, so add it now
98  plist.push_back(*itt);
99  pindex[*itt] = i;
100  i++;
101  }
102  }
103  }
104  else {
105  plist = tmpplist;
106  }
107 
108  psize = static_cast<int>(plist.size());
109  pvals.resize(psize);
110  pmap.clear();
111  for (int j = 0; j < psize; j++) {
112  pmap[plist[j]] = &pvals[j];
113  pvals[j] = *plist[j];
114  }
115  for (MAP_pD_I::const_iterator itr = rindex.begin(); itr != rindex.end(); ++itr) {
116  pmap[itr->first] = &pvals[itr->second];
117  }
118 
119  c2p.clear();
120  p2c.clear();
121  for (std::vector<Constraint*>::iterator constr = clist.begin(); constr != clist.end();
122  ++constr) {
123  (*constr)->revertParams(); // ensure that the constraint points to the original parameters
124  VEC_pD constr_params_orig = (*constr)->params();
125  SET_pD constr_params;
126  for (VEC_pD::const_iterator p = constr_params_orig.begin(); p != constr_params_orig.end();
127  ++p) {
128  MAP_pD_pD::const_iterator pmapfind = pmap.find(*p);
129  if (pmapfind != pmap.end()) {
130  constr_params.insert(pmapfind->second);
131  }
132  }
133  for (SET_pD::const_iterator p = constr_params.begin(); p != constr_params.end(); ++p) {
134  // jacobi.set(*constr, *p, 0.);
135  c2p[*constr].push_back(*p);
136  p2c[*p].push_back(*constr);
137  }
138  // (*constr)->redirectParams(pmap); // redirect parameters to pvec
139  }
140 }

279 void SubSystem::calcGrad(VEC_pD& params, Eigen::VectorXd& grad)
280 {
281  assert(grad.size() == int(params.size()));
282 
283  grad.setZero();
284  for (int j = 0; j < int(params.size()); j++) {
285  MAP_pD_pD::const_iterator pmapfind = pmap.find(params[j]);
286  if (pmapfind != pmap.end()) {
287  std::vector<Constraint*> constrs = p2c[pmapfind->second];
288  for (std::vector<Constraint*>::const_iterator constr = constrs.begin();
289  constr != constrs.end();
290  ++constr) {
291  grad[j] += (*constr)->error() * (*constr)->grad(pmapfind->second);
292  }
293  }
