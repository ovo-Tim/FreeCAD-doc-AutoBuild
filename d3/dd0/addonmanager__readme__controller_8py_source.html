98  def _download_completed(self, index: int, code: int, data: QtCore.QByteArray) -> None:
99  """Callback for handling a completed README file download."""
100  if index == self.readme_request_index:
101  if code == 200: # HTTP success
102  self._process_package_download(data.data().decode("utf-8"))
103  else:
104  self.widget.setText(
105  translate(
106  "AddonsInstaller",
107  "Failed to download data from {} -- received response code {}.",
108  ).format(self.url, code)
109  )
110  elif index in self.resource_requests:
111  if code == 200:
112  self._process_resource_download(self.resource_requests[index], data.data())
113  else:
114  FreeCAD.Console.PrintLog(f"Failed to load {self.resource_requests[index]}\n")
115  self.resource_failures.append(self.resource_requests[index])
116  del self.resource_requests[index]
117  if not self.resource_requests:
118  if self.readme_data:
119  if self.readme_data_type == ReadmeDataType.Html:
120  self.widget.setHtml(self.readme_data)
121  elif self.readme_data_type == ReadmeDataType.Markdown:
122  self.widget.setMarkdown(self.readme_data)
123  else:
124  self.widget.setText(self.readme_data)
125  else:
126  self.set_addon(self.addon) # Trigger a reload of the page now with resources

155  def follow_link(self, url: str) -> None:
156  final_url = url

165  def _create_full_url(self, url: str) -> str:
166  if url.startswith("http"):
167  return url
168  if not self.url:
169  return url
170  lhs, slash, _ = self.url.rpartition("/")
171  return lhs + slash + url

173  def _create_markdown_url(self, file: str) -> str:
174  base_url = utils.get_readme_html_url(self.addon)

176  return lhs + slash + file

179 class WikiCleaner(HTMLParser):
180  """This HTML parser cleans up FreeCAD Macro Wiki Page for display in a
181  QTextBrowser widget (which does not deal will with tables used as formatting,
182  etc.) It strips out any tables, and extracts the mw-parser-output div as the only
183  thing that actually gets displayed. It also discards anything inside the [edit]
184  spans that litter wiki output."""
185 
186  class State(Enum):
187  BeforeMacroContent = auto()
188  InMacroContent = auto()
189  InTable = auto()
190  InEditSpan = auto()
191  AfterMacroContent = auto()
192 
193  def __init__(self):
194  super().__init__()
195  self.depth_in_div = 0
196  self.depth_in_span = 0
197  self.depth_in_table = 0
198  self.final_html = "<html><body>"
199  self.previous_state = WikiCleaner.State.BeforeMacroContent
200  self.state = WikiCleaner.State.BeforeMacroContent
201 
202  def handle_starttag(self, tag: str, attrs):
203  if tag == "div":
204  self.handle_div_start(attrs)
205  elif tag == "span":
206  self.handle_span_start(attrs)
207  elif tag == "table":
208  self.handle_table_start(attrs)
209  else:
210  if self.state == WikiCleaner.State.InMacroContent:
211  self.add_tag_to_html(tag, attrs)
212 
213  def handle_div_start(self, attrs):
214  for name, value in attrs:
215  if name == "class" and value == "mw-parser-output":
216  self.previous_state = self.state
217  self.state = WikiCleaner.State.InMacroContent
218  if self.state == WikiCleaner.State.InMacroContent:
219  self.depth_in_div += 1
220  self.add_tag_to_html("div", attrs)
221 
222  def handle_span_start(self, attrs):
223  for name, value in attrs:
224  if name == "class" and value == "mw-editsection":
225  self.previous_state = self.state
226  self.state = WikiCleaner.State.InEditSpan
227  break
228  if self.state == WikiCleaner.State.InEditSpan:
229  self.depth_in_span += 1
230  elif WikiCleaner.State.InMacroContent:
231  self.add_tag_to_html("span", attrs)
232 
233  def handle_table_start(self, unused):
234  if self.state != WikiCleaner.State.InTable:
235  self.previous_state = self.state
236  self.state = WikiCleaner.State.InTable
237  self.depth_in_table += 1
238 
239  def add_tag_to_html(self, tag, attrs=None):
240  self.final_html += f"<{tag}"
241  if attrs:
242  self.final_html += " "
243  for attr, value in attrs:
244  self.final_html += f"{attr}='{value}'"
245  self.final_html += ">\n"
246 
247  def handle_endtag(self, tag):
248  if tag == "table":
249  self.handle_table_end()
250  elif tag == "span":
251  self.handle_span_end()
252  elif tag == "div":
253  self.handle_div_end()
254  else:
255  if self.state == WikiCleaner.State.InMacroContent:
256  self.add_tag_to_html(f"/{tag}")
257 
258  def handle_span_end(self):
259  if self.state == WikiCleaner.State.InEditSpan:
260  self.depth_in_span -= 1
261  if self.depth_in_span <= 0:
262  self.depth_in_span = 0
263  self.state = self.previous_state
264  else:
265  self.add_tag_to_html(f"/span")
266 
267  def handle_div_end(self):
268  if self.state == WikiCleaner.State.InMacroContent:
269  self.depth_in_div -= 1
270  if self.depth_in_div <= 0:
271  self.depth_in_div = 0
272  self.state = WikiCleaner.State.AfterMacroContent
273  self.final_html += "</body></html>"
274  else:
275  self.add_tag_to_html(f"/div")
276 
277  def handle_table_end(self):
278  if self.state == WikiCleaner.State.InTable:
279  self.depth_in_table -= 1
280  if self.depth_in_table <= 0:
281  self.depth_in_table = 0
282  self.state = self.previous_state
283 
284  def handle_data(self, data):
285  if self.state == WikiCleaner.State.InMacroContent:
286  self.final_html += data
