90 bool PreferencePack::apply() const
91 {
92  // Run the pre.FCMacro, if it exists: if it raises an exception, abort the process
93  auto preMacroPath = _path / "pre.FCMacro";
94  if (fs::exists(preMacroPath)) {
95  try {
96  Base::Interpreter().runFile(Base::FileInfo::pathToString(preMacroPath).c_str(), false);
97  }
98  catch (...) {
99  Base::Console().Message("PreferencePack application aborted by the preferencePack's pre.FCMacro");
100  return false;
101  }
102  }
103 
104  // Back up the old config file
105  auto savedPreferencePacksDirectory = getSavedPrefPacksPath();
106  auto backupFile = savedPreferencePacksDirectory / "user.cfg.backup";
107  try {
108  fs::remove(backupFile);
109  }
110  catch (...) {}
111  App::GetApplication().GetUserParameter().SaveDocument(Base::FileInfo::pathToString(backupFile).c_str());
112 
113  // Apply the config settings
114  applyConfigChanges();
115 
116  // Run the Post.FCMacro, if it exists
117  auto postMacroPath = _path / "post.FCMacro";
118  if (fs::exists(postMacroPath)) {
119  try {
120  Base::Interpreter().runFile(Base::FileInfo::pathToString(postMacroPath).c_str(), false);
121  }
122  catch (...) {
123  Base::Console().Message("PreferencePack application reverted by the preferencePack's post.FCMacro");
124  App::GetApplication().GetUserParameter().LoadDocument(Base::FileInfo::pathToString(backupFile).c_str());
125  return false;
126  }
127  }
128 
129  return true;
130 }

178 void Gui::PreferencePackManager::AddPackToMetadata(const std::string &packName) const
179 {
180  std::lock_guard<std::mutex> lock(_mutex);
181  auto savedPreferencePacksDirectory = getSavedPreferencePacksPath();
182  fs::path preferencePackDirectory(savedPreferencePacksDirectory / packName);
183  if (fs::exists(preferencePackDirectory) && !fs::is_directory(preferencePackDirectory))
184  throw std::runtime_error("Cannot create " + savedPreferencePacksDirectory.string()
185  + ": file with that name exists already");
186 
187  if (!fs::exists(preferencePackDirectory)) fs::create_directories(preferencePackDirectory);
188 
189  // Create or update the saved user preferencePacks package.xml metadata file
190  std::unique_ptr<App::Metadata> metadata;
191  if (fs::exists(savedPreferencePacksDirectory / "package.xml")) {
192  metadata = std::make_unique<App::Metadata>(savedPreferencePacksDirectory / "package.xml");
193  }
194  else {
195  metadata = std::make_unique<App::Metadata>();
196  metadata->setName("User-Saved Preference Packs");
197  std::stringstream str;
198  str << "Generated automatically -- edits may be lost when saving new preference packs. To "
199  << "distribute one or more of these packs:\n"
200  << " 1) copy the entire SavedPreferencePacks directory to a convenient location,\n"
201  << " 2) rename the directory (usually to the name of the preference pack you are "
202  << "distributing),\n"
203  << " 3) delete any subfolders containing packs you don't want to distribute,\n"
204  << " 4) use git to initialize the directory as a git repository,\n"
205  << " 5) push it to a remote git host,\n"
206  << " 6) activate Developer Mode in the Addon Manager,\n"
207  << " 7) use Developer Tools in the Addon Manager to update the metadata file,\n"
208  << " 8) add, commit, and push the updated package.xml file,\n"
209  << " 9) add your remote host to the custom repositories list in the Addon Manager"
210  << " preferences,\n"
211  << " 10) use the Addon Manager to install your preference pack locally for testing.";
212  metadata->setDescription(str.str());
213  metadata->addLicense(App::Meta::License("All Rights Reserved", fs::path()));
214  }
215  for (const auto &item : metadata->content()) {
216  if (item.first == "preferencepack") {
217  if (item.second.name() == packName) {
218  // A pack with this name exists already, bail out
219  return;
220  }
221  }
222  }
223  App::Metadata newPreferencePackMetadata;
224  newPreferencePackMetadata.setName(packName);
225 
226  metadata->addContentItem("preferencepack", newPreferencePackMetadata);
227  metadata->write(savedPreferencePacksDirectory / "package.xml");
228 }

312 void PreferencePackManager::TryFindPreferencePacksInPackage(const boost::filesystem::path& mod)
313 {
314  auto packageMetadataFile = mod / "package.xml";
315  static const auto modDirectory = fs::path(Base::FileInfo::stringToPath(App::Application::getUserAppDataDir())) / "Mod" / "SavedPreferencePacks";
316  static const auto resourcePath = getResourcePreferencePacksPath();
317 
318  if (fs::exists(packageMetadataFile) && fs::is_regular_file(packageMetadataFile)) {
319  App::Metadata metadata(packageMetadataFile);
320  auto content = metadata.content();
321  auto basename = Base::FileInfo::pathToString(mod.filename());
322  if (mod == modDirectory)
323  basename = "##USER_SAVED##";
324  else if (mod == resourcePath)
325  basename = "##BUILT_IN##";
326  for (const auto& item : content) {
327  if (item.first == "preferencepack") {
328  if (isVisible(basename, item.second.name())) {
329  PreferencePack newPreferencePack(mod / item.second.name(), item.second);
330  _preferencePacks.insert(std::make_pair(newPreferencePack.name(), newPreferencePack));
331  }
332  }
333  }
334  }

403 void PreferencePackManager::toggleVisibility(const std::string& addonName, const std::string& preferencePackName)
404 {
405  if (preferencePackName.empty())
406  return;
407  auto pref = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/General/HiddenPreferencePacks");
408  auto hiddenPacks = pref->GetGroups();
409  auto hiddenPack = std::find_if(hiddenPacks.begin(), hiddenPacks.end(), [addonName,preferencePackName](ParameterGrp::handle handle) {
410  return (handle->GetASCII("addonName", "") == addonName) && (handle->GetASCII("preferencePackName", "") == preferencePackName);
411  });
412  if (hiddenPack == hiddenPacks.end()) {
413  auto name = findUnusedName("PreferencePack", pref);
414  auto group = pref->GetGroup(name.c_str());
415  group->SetASCII("addonName", addonName.c_str());
416  group->SetASCII("preferencePackName", preferencePackName.c_str());
417  }
418  else {
419  auto groupName = (*hiddenPack)->GetGroupName();
420  hiddenPacks.clear(); // To decrement the reference count of the group we are about the remove...
421  pref->RemoveGrp(groupName);
422  }
423  rescan();
424 }

446 static void copyTemplateParameters(Base::Reference<ParameterGrp> templateGroup, const std::string& path, Base::Reference<ParameterGrp> outputGroup)
447 {
448  auto userParameterHandle = App::GetApplication().GetParameterGroupByPath(path.c_str());
449 
450  // Ensure that the DockWindowManager has saved its current state:
451  Gui::DockWindowManager* pDockMgr = Gui::DockWindowManager::instance();
452  pDockMgr->saveState();
453 
454  // Do the same for ToolBars
455  Gui::ToolBarManager* pToolbarMgr = Gui::ToolBarManager::getInstance();
456  pToolbarMgr->saveState();
457 
458  auto boolMap = templateGroup->GetBoolMap();
459  for (const auto& kv : boolMap) {
460  auto currentValue = userParameterHandle->GetBool(kv.first.c_str(), kv.second);
461  outputGroup->SetBool(kv.first.c_str(), currentValue);
462  }
463 
464  auto intMap = templateGroup->GetIntMap();
465  for (const auto& kv : intMap) {
466  auto currentValue = userParameterHandle->GetInt(kv.first.c_str(), kv.second);
467  outputGroup->SetInt(kv.first.c_str(), currentValue);
468  }
469 
470  auto uintMap = templateGroup->GetUnsignedMap();
471  for (const auto& kv : uintMap) {
472  auto currentValue = userParameterHandle->GetUnsigned(kv.first.c_str(), kv.second);
473  outputGroup->SetUnsigned(kv.first.c_str(), currentValue);
474  }
475 
476  auto floatMap = templateGroup->GetFloatMap();
477  for (const auto& kv : floatMap) {
478  auto currentValue = userParameterHandle->GetFloat(kv.first.c_str(), kv.second);
479  outputGroup->SetFloat(kv.first.c_str(), currentValue);
480  }
481 
482  auto asciiMap = templateGroup->GetASCIIMap();
483  for (const auto& kv : asciiMap) {
484  auto currentValue = userParameterHandle->GetASCII(kv.first.c_str(), kv.second.c_str());
485  outputGroup->SetASCII(kv.first.c_str(), currentValue.c_str());
486  }
487 
488  // Recurse...
489  auto templateSubgroups = templateGroup->GetGroups();
490  for (auto& templateSubgroup : templateSubgroups) {
491  std::string sgName = templateSubgroup->GetGroupName();
492  auto outputSubgroupHandle = outputGroup->GetGroup(sgName.c_str());
493  copyTemplateParameters(templateSubgroup, path + "/" + sgName, outputSubgroupHandle);
494  }
495 }

534 static std::vector<fs::path> scanForTemplateFolders(const std::string& groupName, const fs::path& entry)
535 {
536  // From this location, find the folder(s) called "PreferencePackTemplates"
537  std::vector<fs::path> templateFolders;
538  if (fs::exists(entry)) {
539  if (fs::is_directory(entry)) {
540  if (entry.filename() == "PreferencePackTemplates" ||
541  entry.filename() == "preference_pack_templates") {
542  templateFolders.push_back(entry);
543  }
544  else {
545  std::string subgroupName = groupName + "/" + Base::FileInfo::pathToString(entry.filename());
546  for (const auto& subentry : fs::directory_iterator(entry)) {
547  auto contents = scanForTemplateFolders(subgroupName, subentry);
548  std::copy(contents.begin(), contents.end(), std::back_inserter(templateFolders));
549  }
550  }
551  }
552  }
553  return templateFolders;
554 }

556 static std::vector<PreferencePackManager::TemplateFile> scanForTemplateFiles(const std::string& groupName, const fs::path& entry)
557 {
558  auto templateFolders = scanForTemplateFolders(groupName, entry);
559 
560  std::vector<PreferencePackManager::TemplateFile> templateFiles;
561  for (const auto& templateDir : templateFolders) {
562  if (!fs::exists(templateDir) || !fs::is_directory(templateDir))
563  continue;
564  for (const auto& entry : fs::directory_iterator(templateDir)) {
565  if (entry.path().extension() == ".cfg") {
566  auto name = Base::FileInfo::pathToString(entry.path().filename().stem());
567  std::replace(name.begin(), name.end(), '_', ' ');
568  // Make sure we don't insert the same thing twice...
569  if (std::find_if(templateFiles.begin(), templateFiles.end(), [groupName, name](const auto &rhs)->bool {
570  return groupName == rhs.group && name == rhs.name;
571  } ) != templateFiles.end())
572  continue;
573  templateFiles.push_back({ groupName, name, entry });
574  }
575  }
576  }
577  return templateFiles;
578 }

580 std::vector<PreferencePackManager::TemplateFile> PreferencePackManager::templateFiles(bool rescan)
581 {
582  std::lock_guard<std::mutex> lock(_mutex);
583  if (!_templateFiles.empty() && !rescan)
584  return _templateFiles;
585 
586  // Locate all of the template files available on this system
587  // Template files end in ".cfg" -- They are located in:
588  // * $INSTALL_DIR/data/Gui/PreferencePackTemplates/(Appearance|Behavior)/*
589  // * $DATA_DIR/Mod/**/PreferencePackTemplates/(Appearance|Behavior)/*
590  // (alternate spellings are provided for packages using CamelCase and snake_case, and both major English dialects)
591 
592  auto resourcePath = fs::path(Base::FileInfo::stringToPath(App::Application::getResourceDir())) / "Gui";
593 
594  std::string group = "Built-In";
595  if (fs::exists(resourcePath) && fs::is_directory(resourcePath)) {
596  const auto localFiles = scanForTemplateFiles(group, resourcePath);
597  std::copy(localFiles.begin(), localFiles.end(), std::back_inserter(_templateFiles));
598  }
599 
600  for (const auto& modPath : modPaths()) {
601  if (fs::exists(modPath) && fs::is_directory(modPath)) {
602  for (const auto& mod : fs::directory_iterator(modPath)) {
