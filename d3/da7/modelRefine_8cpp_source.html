85 void ModelRefine::boundaryEdges(const FaceVectorType &faces, EdgeVectorType &edgesOut)
86 {
87  //this finds all the boundary edges. Maybe more than one boundary.
88  std::list<TopoDS_Edge> edges;
89  FaceVectorType::const_iterator faceIt;
90  for (faceIt = faces.begin(); faceIt != faces.end(); ++faceIt)
91  {
92  EdgeVectorType faceEdges;
93  EdgeVectorType::iterator faceEdgesIt;
94  getFaceEdges(*faceIt, faceEdges);
95  for (faceEdgesIt = faceEdges.begin(); faceEdgesIt != faceEdges.end(); ++faceEdgesIt)
96  {
97  bool foundSignal(false);
98  std::list<TopoDS_Edge>::iterator edgesIt;
99  for (edgesIt = edges.begin(); edgesIt != edges.end(); ++edgesIt)
100  {
101  if ((*edgesIt).IsSame(*faceEdgesIt))
102  {
103  edgesIt = edges.erase(edgesIt);
104  foundSignal = true;
105  break;
106  }
107  }
108  if (!foundSignal)
109  edges.push_back(*faceEdgesIt);
110  }
111  }
112 
113  edgesOut.reserve(edges.size());
114  std::copy(edges.begin(), edges.end(), back_inserter(edgesOut));
115 }

199 void FaceAdjacencySplitter::split(const FaceVectorType &facesIn)
200 {
201  facesInMap.Clear();
202  processedMap.Clear();
203  adjacencyArray.clear();
204 
205  FaceVectorType::const_iterator it;
206  for (it = facesIn.begin(); it != facesIn.end(); ++it)
207  facesInMap.Add(*it);
208  //the reserve call guarantees the vector will never get "pushed back" in the
209  //recursiveFind calls, thus invalidating the iterators. We can be sure of this as any one
210  //matched set can't be bigger than the set passed in. if we have seg faults, we will
211  //want to turn this tempFaces vector back into a std::list ensuring valid iterators
212  //at the expense of std::find speed.
213  FaceVectorType tempFaces;
214  tempFaces.reserve(facesIn.size() + 1);
215 
216  for (it = facesIn.begin(); it != facesIn.end(); ++it)
217  {
218  //skip already processed shapes.
219  if (processedMap.Contains(*it))
220  continue;
221 
222  tempFaces.clear();
223  processedMap.Add(*it);
224  recursiveFind(*it, tempFaces);
225  if (tempFaces.size() > 1)
226  {
227  adjacencyArray.push_back(tempFaces);
228  }
229  }

232 void FaceAdjacencySplitter::recursiveFind(const TopoDS_Face &face, FaceVectorType &outVector)
233 {
234  outVector.push_back(face);
235 
236  const TopTools_ListOfShape &edges = faceToEdgeMap.FindFromKey(face);
237  TopTools_ListIteratorOfListOfShape edgeIt;
238  for (edgeIt.Initialize(edges); edgeIt.More(); edgeIt.Next())

245  continue;
246  if (processedMap.Contains(faceIt.Value()))

248  processedMap.Add(faceIt.Value());
249  recursiveFind(TopoDS::Face(faceIt.Value()), outVector);

300 void FaceTypedBase::boundarySplit(const FaceVectorType &facesIn, std::vector<EdgeVectorType> &boundariesOut) const
301 {
302  EdgeVectorType bEdges;
303  boundaryEdges(facesIn, bEdges);
304 
305  std::list<TopoDS_Edge> edges;
306  std::copy(bEdges.begin(), bEdges.end(), back_inserter(edges));
307  while(!edges.empty())
308  {
309  TopoDS_Vertex destination = TopExp::FirstVertex(edges.front(), Standard_True);
310  TopoDS_Vertex lastVertex = TopExp::LastVertex(edges.front(), Standard_True);
311  EdgeVectorType boundary;
312  boundary.push_back(edges.front());
313  edges.pop_front();
314  //single edge closed check.
315  if (destination.IsSame(lastVertex))
316  {
317  boundariesOut.push_back(boundary);
318  continue;
319  }
320 
321  bool closedSignal(false);
322  std::list<TopoDS_Edge>::iterator it;
323  for (it = edges.begin(); it != edges.end();)
324  {
325  TopoDS_Vertex currentVertex = TopExp::FirstVertex(*it, Standard_True);
326  if (lastVertex.IsSame(currentVertex))
327  {
328  boundary.push_back(*it);
329  lastVertex = TopExp::LastVertex(*it, Standard_True);
330  edges.erase(it);
331  it = edges.begin();
332  if (lastVertex.IsSame(destination))
333  {
334  closedSignal = true;
335  break;
336  }
337  continue;
338  }
339  ++it;
340  }
341  if (closedSignal)
342  boundariesOut.push_back(boundary);
343  }
344 }

399  {
400  BRepLib_MakeWire wireMaker;
401  EdgeVectorType::iterator it;
402  for (it = (*splitIt).begin(); it != (*splitIt).end(); ++it)
403  wireMaker.Add(*it);

405  wires.push_back(currentWire);

414  if (wires.size() > 1)
415  {
416  ShapeFix_Face faceFix(current);
417  faceFix.SetContext(new ShapeBuild_ReShape());
418  for (size_t index(1); index<wires.size(); ++index)
419  faceFix.Add(wires.at(index));
420  faceFix.Perform();
421  if (faceFix.Status(ShapeExtend_FAIL))
422  return {};
423  faceFix.FixOrientation();
424  faceFix.Perform();
425  if(faceFix.Status(ShapeExtend_FAIL))
426  return {};
427  current = faceFix.Face();
428  }

505 // Detect whether a wire encircles the cylinder axis or not. This is done by calculating the
506 // wire length, oriented with respect to the cylinder axis
507 bool wireEncirclesAxis(const TopoDS_Wire& wire, const Handle(Geom_CylindricalSurface)& cylinder)
508 {
509  double radius = cylinder->Radius();
510  gp_Ax1 cylAxis = cylinder->Axis();
511  gp_Vec cv(cylAxis.Location().X(), cylAxis.Location().Y(), cylAxis.Location().Z()); // center of cylinder
512  gp_Vec av(cylAxis.Direction().X(), cylAxis.Direction().Y(), cylAxis.Direction().Z()); // axis of cylinder
513  Handle(Geom_Plane) plane = new Geom_Plane(gp_Ax3(cylAxis.Location(), cylAxis.Direction()));
514  double totalArc = 0.0;
515  bool firstSegment = false;
516  bool secondSegment = false;
517  gp_Pnt first, last;
518 
519  for (TopExp_Explorer ex(wire, TopAbs_EDGE); ex.More(); ex.Next()) {
520  TopoDS_Edge segment(TopoDS::Edge(ex.Current()));
521  BRepAdaptor_Curve adapt(segment);
522 
523  // Get curve data
524  double fp = adapt.FirstParameter();
525  double lp = adapt.LastParameter();
526  gp_Pnt segFirst, segLast;
527  gp_Vec segTangent;
528  adapt.D1(fp, segFirst, segTangent);
529  segLast = adapt.Value(lp);
530 
531  double length = 0.0;
532 
533  if (adapt.GetType() == GeomAbs_Line) {
534  // Any line on the cylinder must be parallel to the cylinder axis
535  length = 0.0;
536  } else if (adapt.GetType() == GeomAbs_Circle) {
537  // Arc segment
538  length = (lp - fp) * radius;
539  // Check orientation in relation to cylinder axis
540  GeomAPI_ProjectPointOnSurf proj(segFirst, plane);
541  gp_Vec bv = gp_Vec(proj.Point(1).X(), proj.Point(1).Y(), proj.Point(1).Z());
542  if ((bv - cv).Crossed(segTangent).IsOpposite(av, Precision::Confusion()))
543  length = -length;
544  } else {
545  // Linearize the edge. Idea taken from ShapeAnalysis.cxx ShapeAnalysis::TotCross2D()
546  TColgp_SequenceOfPnt SeqPnt;
547  ShapeAnalysis_Curve::GetSamplePoints(adapt.Curve().Curve(), fp, lp, SeqPnt);
548 
549  // Calculate the oriented length of the edge
550  gp_Pnt begin;
551  for (Standard_Integer j=1; j <= SeqPnt.Length(); j++) {
552  gp_Pnt end = SeqPnt.Value(j);
553 
554  // Project end point onto the plane
555  GeomAPI_ProjectPointOnSurf proj(end, plane);
556  if (!proj.IsDone())
557  return false; // FIXME: What else can we do?
558  gp_Pnt pend = proj.Point(1);
559 
560  if (j > 1) {
561  // Get distance between the points, equal to (linearised) arc length
562  gp_Vec bv = gp_Vec(begin.X(), begin.Y(), begin.Z());
563  gp_Vec dv = gp_Vec(pend.X(), pend.Y(), pend.Z()) - bv;
564  double dist = dv.Magnitude();
565 
566  if (dist > 0) {
567  // Check orientation of this piece in relation to cylinder axis
568  if ((bv - cv).Crossed(dv).IsOpposite(av, Precision::Confusion()))
569  dist = -dist;
570 
571  length += dist;
572  }
573  }
574 
575  begin = pend;
576  }
577  }
578 
579  if (!firstSegment) {
580  // First wire segment. Save the start and end point of the segment
581  firstSegment = true;
582  first = segFirst;
583  last = segLast;
584  } else if (!secondSegment) {
585  // Second wire segment. Determine whether the second segment continues from
586  // the first or from the last point of the first segment
587  secondSegment = true;
588  if (last.IsEqual(segFirst, Precision::Confusion())) {
589  last = segLast; // Third segment must begin here
590  } else if (last.IsEqual(segLast, Precision::Confusion())) {
591  last = segFirst;
592  length = -length;
593  } else if (first.IsEqual(segLast, Precision::Confusion())) {
594  last = segFirst;
595  totalArc = -totalArc;
596  length = -length;
597  } else { // first.IsEqual(segFirst)
598  last = segLast;
599  totalArc = -totalArc;
600  }
601  } else {
602  if (!last.IsEqual(segFirst, Precision::Confusion())) {
603  // The length was calculated in the opposite direction of the wire traversal
604  length = -length;
605  last = segFirst;
606  } else {
607  last = segLast;
608  }
609  }
610 
611  totalArc += length;
612  }
613 
614  // For an exact calculation, only two results would be possible:
615  // totalArc = 0.0: The wire does not encircle the axis
616  // totalArc = 2 * M_PI * radius: The wire encircles the axis
617  return (fabs(totalArc) > M_PI * radius);
618 }

623  std::vector<EdgeVectorType> boundaries;
624  boundarySplit(faces, boundaries);

627 
628  //make wires
629  std::vector<TopoDS_Wire> allWires;
630  std::vector<EdgeVectorType>::iterator boundaryIt;

643 
644  // Sort wires by size, that is, the innermost wire comes last
645  std::sort(allWires.begin(), allWires.end(), ModelRefine::WireSort());
646 
647  // Find outer boundary wires that cut the cylinder into segments. This will be the case f we
648  // have removed the seam edges of a complete (360 degrees) cylindrical face
649  Handle(Geom_CylindricalSurface) surface = getGeomCylinder(faces.at(0));
650  if (surface.IsNull())
651  return dummy;
652  std::vector<TopoDS_Wire> innerWires, encirclingWires;
653  std::vector<TopoDS_Wire>::iterator wireIt;
654  for (wireIt = allWires.begin(); wireIt != allWires.end(); ++wireIt) {
655  if (wireEncirclesAxis(*wireIt, surface))
656  encirclingWires.push_back(*wireIt);
657  else
658  innerWires.push_back(*wireIt);
659  }
660 
661  if (encirclingWires.empty()) {
662  // We can use the result of the bounding box sort. First wire is the outer wire
663  wireIt = allWires.begin();
664  BRepBuilderAPI_MakeFace faceMaker(surface, *wireIt);
665  if (!faceMaker.IsDone())
666  return dummy;
667 
668  // Add additional boundaries (inner wires).
669  for (wireIt++; wireIt != allWires.end(); ++wireIt)
670  {
671  faceMaker.Add(*wireIt);
672  if (!faceMaker.IsDone())
673  return dummy;
674  }
675 
676  return fixFace(faceMaker.Face());
677  } else {
678  if (encirclingWires.size() != 2)
679  return dummy;
680 
681  if (innerWires.empty()) {
682  // We have just two outer boundaries
683  BRepBuilderAPI_MakeFace faceMaker(surface, encirclingWires.front());
684  if (!faceMaker.IsDone())
685  return dummy;
686  faceMaker.Add(encirclingWires.back());
687  if (!faceMaker.IsDone())
688  return dummy;
689 
690  return fixFace(faceMaker.Face());
691  } else {
692  // Add the inner wires first, because otherwise those that cut the seam edge will fail
693  wireIt = innerWires.begin();
694  BRepBuilderAPI_MakeFace faceMaker(surface, *wireIt, false);
695  if (!faceMaker.IsDone())
696  return dummy;
697 
698  // Add additional boundaries (inner wires).
699  for (wireIt++; wireIt != innerWires.end(); ++wireIt)
700  {
701  faceMaker.Add(*wireIt);
702  if (!faceMaker.IsDone())
703  return dummy;
704  }
705 
706  // Add outer boundaries
707  faceMaker.Add(encirclingWires.front());
708  if (!faceMaker.IsDone())
709  return dummy;
710  faceMaker.Add(encirclingWires.back());
711  if (!faceMaker.IsDone())
712  return dummy;
713 
714  return fixFace(faceMaker.Face());
715  }

719 void FaceTypedCylinder::boundarySplit(const FaceVectorType &facesIn, std::vector<EdgeVectorType> &boundariesOut) const
720 {
721  //normal edges.
722  EdgeVectorType normalEdges;
723  ModelRefine::boundaryEdges(facesIn, normalEdges);
724 
725  std::list<TopoDS_Edge> sortedEdges;
726  std::copy(normalEdges.begin(), normalEdges.end(), back_inserter(sortedEdges));
727 
728  while (!sortedEdges.empty())
729  {
730  TopoDS_Vertex destination = TopExp::FirstVertex(sortedEdges.back(), Standard_True);
731  TopoDS_Vertex lastVertex = TopExp::LastVertex(sortedEdges.back(), Standard_True);
732  bool closedSignal(false);
733  std::list<TopoDS_Edge> boundary;
734  boundary.push_back(sortedEdges.back());
735  sortedEdges.pop_back();
736 
737  if (destination.IsSame(lastVertex)) {
738  // Single circular edge
739  closedSignal = true;
740  } else {
741  std::list<TopoDS_Edge>::iterator sortedIt;
742  for (sortedIt = sortedEdges.begin(); sortedIt != sortedEdges.end();)
743  {
744  TopoDS_Vertex currentVertex = TopExp::FirstVertex(*sortedIt, Standard_True);
745 
746  //Seam edges lie on top of each other. i.e. same. and we remove every match from the list
747  //so we don't actually ever compare the same edge.
748  if ((*sortedIt).IsSame(boundary.back()))
749  {
750  ++sortedIt;
751  continue;
752  }
753  if (lastVertex.IsSame(currentVertex))
754  {
755  boundary.push_back(*sortedIt);
756  lastVertex = TopExp::LastVertex(*sortedIt, Standard_True);
757  if (lastVertex.IsSame(destination))
758  {
759  closedSignal = true;
760  sortedEdges.erase(sortedIt);
761  break;
762  }
763  sortedEdges.erase(sortedIt);
764  sortedIt = sortedEdges.begin();
765  continue;
766  }
767  ++sortedIt;
768  }
769  }
770  if (closedSignal)
771  {
772  EdgeVectorType temp;
773  std::copy(boundary.begin(), boundary.end(), std::back_inserter(temp));
774  boundariesOut.push_back(temp);
775  }
776  }

814 bool FaceTypedBSpline::isEqual(const TopoDS_Face &faceOne, const TopoDS_Face &faceTwo) const

819  Handle(Geom_BSplineSurface) surfaceTwo = Handle(Geom_BSplineSurface)::DownCast(BRep_Tool::Surface(faceTwo));
820 
821  if (surfaceOne.IsNull() || surfaceTwo.IsNull())
822  return false;
823 
824  if (surfaceOne->IsURational() != surfaceTwo->IsURational())
825  return false;
826  if (surfaceOne->IsVRational() != surfaceTwo->IsVRational())
827  return false;
828  if (surfaceOne->IsUPeriodic() != surfaceTwo->IsUPeriodic())
829  return false;
830  if (surfaceOne->IsVPeriodic() != surfaceTwo->IsVPeriodic())
831  return false;
832  if (surfaceOne->IsUClosed() != surfaceTwo->IsUClosed())
833  return false;
834  if (surfaceOne->IsVClosed() != surfaceTwo->IsVClosed())
835  return false;
836  if (surfaceOne->UDegree() != surfaceTwo->UDegree())
837  return false;
838  if (surfaceOne->VDegree() != surfaceTwo->VDegree())
839  return false;
840 
841  //pole test
842  int uPoleCountOne(surfaceOne->NbUPoles());
843  int vPoleCountOne(surfaceOne->NbVPoles());
844  int uPoleCountTwo(surfaceTwo->NbUPoles());
845  int vPoleCountTwo(surfaceTwo->NbVPoles());
846 
847  if (uPoleCountOne != uPoleCountTwo || vPoleCountOne != vPoleCountTwo)
848  return false;
849 
850  TColgp_Array2OfPnt polesOne(1, uPoleCountOne, 1, vPoleCountOne);
851  TColgp_Array2OfPnt polesTwo(1, uPoleCountTwo, 1, vPoleCountTwo);
852  surfaceOne->Poles(polesOne);
853  surfaceTwo->Poles(polesTwo);
854 
855  for (int indexU = 1; indexU <= uPoleCountOne; ++indexU)
856  {
857  for (int indexV = 1; indexV <= vPoleCountOne; ++indexV)
858  {
859  if (!(polesOne.Value(indexU, indexV).IsEqual(polesTwo.Value(indexU, indexV), Precision::Confusion())))
860  return false;
861  }
862  }
863 
864  //knot test
865  int uKnotCountOne(surfaceOne->NbUKnots());
866  int vKnotCountOne(surfaceOne->NbVKnots());
867  int uKnotCountTwo(surfaceTwo->NbUKnots());
868  int vKnotCountTwo(surfaceTwo->NbVKnots());
869  if (uKnotCountOne != uKnotCountTwo || vKnotCountOne != vKnotCountTwo)
870  return false;
871  TColStd_Array1OfReal uKnotsOne(1, uKnotCountOne);
872  TColStd_Array1OfReal vKnotsOne(1, vKnotCountOne);
873  TColStd_Array1OfReal uKnotsTwo(1, uKnotCountTwo);
874  TColStd_Array1OfReal vKnotsTwo(1, vKnotCountTwo);
875  surfaceOne->UKnots(uKnotsOne);
876  surfaceOne->VKnots(vKnotsOne);
877  surfaceTwo->UKnots(uKnotsTwo);
878  surfaceTwo->VKnots(vKnotsTwo);
879  for (int indexU = 1; indexU <= uKnotCountOne; ++indexU)
880  if (uKnotsOne.Value(indexU) != uKnotsTwo.Value(indexU))
881  return false;
882  for (int indexV = 1; indexV <= vKnotCountOne; ++indexV)
883  if (vKnotsOne.Value(indexV) != vKnotsTwo.Value(indexV))
884  return false;
885 
886  //Formulas:
887  //Periodic: knots=poles+2*degree-mult(1)+2
888  //Non-periodic: knots=poles+degree+1
889  auto getUKnotSequenceSize = [](Handle(Geom_BSplineSurface) surface) {
890  int uPoleCount(surface->NbUPoles());
891  int uDegree(surface->UDegree());
892  if (surface->IsUPeriodic()) {
893  int uMult1(surface->UMultiplicity(1));
894  int size = uPoleCount + 2*uDegree - uMult1 + 2;
895  return size;
896  }
897  else {
898  int size = uPoleCount + uDegree + 1;
899  return size;
900  }
901  };
902  auto getVKnotSequenceSize = [](Handle(Geom_BSplineSurface) surface) {
903  int vPoleCount(surface->NbVPoles());
904  int vDegree(surface->VDegree());
905  if (surface->IsVPeriodic()) {
906  int vMult1(surface->VMultiplicity(1));
907  int size = vPoleCount + 2*vDegree - vMult1 + 2;
908  return size;
909  }
910  else {
911  int size = vPoleCount + vDegree + 1;
912  return size;
913  }
914  };
915 
916  //knot sequence.
917  int uKnotSequenceOneCount(getUKnotSequenceSize(surfaceOne));
918  int vKnotSequenceOneCount(getVKnotSequenceSize(surfaceOne));
919  int uKnotSequenceTwoCount(getUKnotSequenceSize(surfaceTwo));
920  int vKnotSequenceTwoCount(getVKnotSequenceSize(surfaceTwo));
921  if (uKnotSequenceOneCount != uKnotSequenceTwoCount || vKnotSequenceOneCount != vKnotSequenceTwoCount)
922  return false;
923  TColStd_Array1OfReal uKnotSequenceOne(1, uKnotSequenceOneCount);
924  TColStd_Array1OfReal vKnotSequenceOne(1, vKnotSequenceOneCount);
925  TColStd_Array1OfReal uKnotSequenceTwo(1, uKnotSequenceTwoCount);
926  TColStd_Array1OfReal vKnotSequenceTwo(1, vKnotSequenceTwoCount);
927  surfaceOne->UKnotSequence(uKnotSequenceOne);
928  surfaceOne->VKnotSequence(vKnotSequenceOne);
929  surfaceTwo->UKnotSequence(uKnotSequenceTwo);
930  surfaceTwo->VKnotSequence(vKnotSequenceTwo);
931  for (int indexU = 1; indexU <= uKnotSequenceOneCount; ++indexU)
932  if (uKnotSequenceOne.Value(indexU) != uKnotSequenceTwo.Value(indexU))
933  return false;
934  for (int indexV = 1; indexV <= vKnotSequenceOneCount; ++indexV)
935  if (vKnotSequenceOne.Value(indexV) != vKnotSequenceTwo.Value(indexV))
936  return false;
937  return true;
938  }
939  catch (Standard_Failure& e)
940  {
941  std::ostringstream stream;
942  if (e.GetMessageString())
943  stream << "FaceTypedBSpline::isEqual: OCC Error: " << e.GetMessageString() << std::endl;
944  else
945  stream << "FaceTypedBSpline::isEqual: Unknown OCC Error" << std::endl;
946  Base::Console().Message(stream.str().c_str());
947  }
948  catch (...)
949  {
950  std::ostringstream stream;
951  stream << "FaceTypedBSpline::isEqual: Unknown Error" << std::endl;
952  Base::Console().Message(stream.str().c_str());
953  }
954 
955  return false;
956 }

1029 bool FaceUniter::process()
1030 {

1032  return false;
1033  modifiedShapes.clear();
1034  deletedShapes.clear();
1035  typeObjects.push_back(&getPlaneObject());
1036  typeObjects.push_back(&getCylinderObject());
1037  typeObjects.push_back(&getBSplineObject());
1038  //add more face types.
1039 
1040  bool checkFinalShell = false;
1041  ModelRefine::FaceTypeSplitter splitter;
1042  splitter.addShell(workShell);
1043  std::vector<FaceTypedBase *>::iterator typeIt;
1044  for(typeIt = typeObjects.begin(); typeIt != typeObjects.end(); ++typeIt)
1045  splitter.registerType((*typeIt)->getType());
1046  splitter.split();
1047 
1048  ModelRefine::FaceVectorType facesToRemove;
1049  ModelRefine::FaceVectorType facesToSew;
1050 
1051  ModelRefine::FaceAdjacencySplitter adjacencySplitter(workShell);
1052 
1053  for(typeIt = typeObjects.begin(); typeIt != typeObjects.end(); ++typeIt)
1054  {
1055  ModelRefine::FaceVectorType typedFaces = splitter.getTypedFaceVector((*typeIt)->getType());
1056  ModelRefine::FaceEqualitySplitter equalitySplitter;
1057  equalitySplitter.split(typedFaces, *typeIt);
1058  for (std::size_t indexEquality(0); indexEquality < equalitySplitter.getGroupCount(); ++indexEquality)
1059  {
1060  adjacencySplitter.split(equalitySplitter.getGroup(indexEquality));
1061 // std::cout << " adjacency group count: " << adjacencySplitter.getGroupCount() << std::endl;
1062  for (std::size_t adjacentIndex(0); adjacentIndex < adjacencySplitter.getGroupCount(); ++adjacentIndex)
1063  {
1064 // std::cout << " face count is: " << adjacencySplitter.getGroup(adjacentIndex).size() << std::endl;
1065  TopoDS_Face newFace = (*typeIt)->buildFace(adjacencySplitter.getGroup(adjacentIndex));
1066  if (!newFace.IsNull())
1067  {
1068  // the created face should have the same orientation as the input faces
1069  const FaceVectorType& faces = adjacencySplitter.getGroup(adjacentIndex);
1070  if (!faces.empty() && newFace.Orientation() != faces[0].Orientation()) {
1071  checkFinalShell = true;
1072  }
1073  facesToSew.push_back(newFace);
1074  if (facesToRemove.capacity() <= facesToRemove.size() + adjacencySplitter.getGroup(adjacentIndex).size())
1075  facesToRemove.reserve(facesToRemove.size() + adjacencySplitter.getGroup(adjacentIndex).size());
1076  FaceVectorType temp = adjacencySplitter.getGroup(adjacentIndex);
1077  facesToRemove.insert(facesToRemove.end(), temp.begin(), temp.end());
1078  // the first shape will be marked as modified, i.e. replaced by newFace, all others are marked as deleted
1079  // jrheinlaender: IMHO this is not correct because references to the deleted faces will be broken, whereas they should
1080  // be replaced by references to the new face. To achieve this all shapes should be marked as
1081  // modified, producing one single new face. This is the inverse behaviour to faces that are split e.g.
1082  // by a boolean cut, where one old shape is marked as modified, producing multiple new shapes
1083  if (!temp.empty())
1084  {
1085  for (const auto & f : temp)
1086  modifiedShapes.emplace_back(f, newFace);
1087  }
1088  }
1089  }
1090  }
1091  }
1092  if (!facesToSew.empty())
1093  {
1094  modifiedSignal = true;
1095  workShell = ModelRefine::removeFaces(workShell, facesToRemove);
1096  TopExp_Explorer xp;
1097  bool emptyShell = true;
1098  for (xp.Init(workShell, TopAbs_FACE); xp.More(); xp.Next())
1099  {
1100  emptyShell = false;
1101  break;
1102  }
1103 
1104  if (!emptyShell || facesToSew.size() > 1)
1105  {
1106  BRepBuilderAPI_Sewing sew;
1107  sew.Add(workShell);
1108  FaceVectorType::iterator sewIt;
1109  for(sewIt = facesToSew.begin(); sewIt != facesToSew.end(); ++sewIt)
1110  sew.Add(*sewIt);
1111  sew.Perform();
1112  try {
1113  workShell = TopoDS::Shell(sew.SewedShape());
1114  } catch (Standard_Failure&) {
1115  return false;
1116  }
1117  // update the list of modifications
1118  for (auto & it : modifiedShapes)
1119  {
1120  if (sew.IsModified(it.second))
1121  {
1122  it.second = sew.Modified(it.second);
1123  break;
1124  }
1125  }
1126  }
1127  else
1128  {
1129  // workShell has no more faces and we add exactly one face
1130  BRep_Builder builder;
1131  builder.MakeShell(workShell);
1132  FaceVectorType::iterator sewIt;
1133  for(sewIt = facesToSew.begin(); sewIt != facesToSew.end(); ++sewIt)
1134  builder.Add(workShell, *sewIt);
1135  }
1136

1138 // TODO: change this version after occ fix. Freecad Mantis 1450
1139 #if OCC_VERSION_HEX <= 0x7fffff
1140  TopTools_IndexedMapOfShape map;

1142  edgeFuse.AvoidEdges(map);
1143 #endif

1145  edgeFuse.Faces(affectedFaces);
1146  TopTools_DataMapIteratorOfDataMapOfShapeShape mapIt;
1147  for (mapIt.Initialize(affectedFaces); mapIt.More(); mapIt.Next())
1148  {
1149  ShapeFix_Face faceFixer(TopoDS::Face(mapIt.Value()));
1150  faceFixer.Perform();
1151  }
1152  workShell = TopoDS::Shell(edgeFuse.Shape());
1153 
1154  // A difference of orientation between original and fused faces was previously detected
1155  if (checkFinalShell) {
1156  ShapeAnalysis_Shell check;
1157  check.LoadShells(workShell);
1158  if (!check.HasFreeEdges()) {
1159  // the shell is a solid, make sure that volume is positive
1160  GProp_GProps props;
1161  BRepGProp::VolumeProperties(workShell, props);
1162  if (props.Mass() < 0)
1163  workShell = TopoDS::Shell(workShell.Reversed());
1164  }
1165  }
1166  // update the list of modifications
1167  TopTools_DataMapOfShapeShape faceMap;
1168  edgeFuse.Faces(faceMap);
1169  for (mapIt.Initialize(faceMap); mapIt.More(); mapIt.Next())
1170  {
1171  bool isModifiedFace = false;
1172  for (auto & it : modifiedShapes)
1173  {
1174  if (mapIt.Key().IsSame(it.second)) {
1175  // Note: IsEqual() for some reason does not work
1176  it.second = mapIt.Value();
1177  isModifiedFace = true;
1178  }
1179  }
1180  if (!isModifiedFace)
1181  {
1182  // Catch faces that were not united but whose boundary was changed (probably because
1183  // several adjacent faces were united)
1184  // See https://sourceforge.net/apps/mantisbt/free-cad/view.php?id=873
1185  modifiedShapes.emplace_back(mapIt.Key(), mapIt.Value());
1186  }
1187  }
1188  // Handle edges that were fused. See https://sourceforge.net/apps/mantisbt/free-cad/view.php?id=873
1189  TopTools_DataMapOfIntegerListOfShape oldEdges;
1190  TopTools_DataMapOfIntegerShape newEdges;

1202  modifiedShapes.emplace_back(edgeIt.Value(), newEdges(idx));
1203  }
1204  // TODO: Handle vertices that have disappeared in the fusion of the edges
1205  }
1206  }
1207  return true;
