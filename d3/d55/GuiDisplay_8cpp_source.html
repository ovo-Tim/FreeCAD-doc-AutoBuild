74 bool GuiDisplay::GenerateGlItem(GuiItem* guiItem)
75 {
76  Vertex2D verts[4];
77  int x = guiItem->texItem.tx;
78  int y = guiItem->texItem.ty;
79  int w = guiItem->texItem.w;
80  int h = guiItem->texItem.h;
81 
82  verts[0] = {0, (float)h, mTexture.getTexX(x), mTexture.getTexY(y + h)};
83  verts[1] = {(float)w, (float)h, mTexture.getTexX(x + w), mTexture.getTexY(y + h)};
84  verts[2] = {0, 0, mTexture.getTexX(x), mTexture.getTexY(y)};
85  verts[3] = {(float)w, 0, mTexture.getTexX(x + w), mTexture.getTexY(y)};
86 
87  // vertex buffer
88  glGenBuffers(1, &(guiItem->vbo));
89  glBindBuffer(GL_ARRAY_BUFFER, guiItem->vbo);
90  glBufferData(GL_ARRAY_BUFFER, 4 * sizeof(Vertex2D), verts, GL_STATIC_DRAW);
91 
92  // glDrawElements(GL_TRIANGLES, numIndices, GL_UNSIGNED_SHORT, nullptr);
93  // vertex array
94  glGenVertexArrays(1, &(guiItem->vao));
95  glBindVertexArray(guiItem->vao);
96  glEnableVertexAttribArray(0);
97  glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, sizeof(Vertex2D), (void*)offsetof(Vertex2D, x));
98  glEnableVertexAttribArray(1);
99  glVertexAttribPointer(1,
100  2,
101  GL_FLOAT,
102  GL_FALSE,
103  sizeof(Vertex2D),
104  (void*)offsetof(Vertex2D, tx));
105  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, mIbo);
106  glBindVertexArray(0);
107 
108  return true;
109 }

117 bool GuiDisplay::InitGui()
118 {
119  if (guiInitiated) {
120  return true;
121  }
122  // index buffer
123  glGenBuffers(1, &mIbo);
124  GLshort indices[6] = {0, 2, 3, 0, 3, 1};
125  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, mIbo);
126  glBufferData(GL_ELEMENT_ARRAY_BUFFER, 6 * sizeof(GLushort), indices, GL_STATIC_DRAW);
127  TextureLoader tLoader(":/gl_simulator/", guiFileNames, TEX_SIZE);
128  unsigned int* buffer = tLoader.GetRawData();
129  if (buffer == nullptr) {
130  return false;
131  }
132  mTexture.LoadImage(buffer, TEX_SIZE, TEX_SIZE);
133  for (unsigned int i = 0; i < NUM_GUI_ITEMS; i++) {
134  guiItems[i].texItem = *tLoader.GetTextureItem(i);
135  GenerateGlItem(&(guiItems[i]));
136  }
137 
138  mThumbStartX = guiItems[eGuiItemSlider].posx() - guiItems[eGuiItemThumb].texItem.w / 2;
139  mThumbMaxMotion = (float)guiItems[eGuiItemSlider].texItem.w;
140 
141  // init shader
142  mShader.CompileShader("GuiDisplay", (char*)VertShader2DTex, (char*)FragShader2dTex);
143  mShader.UpdateTextureSlot(0);
144 
145  UpdateSimSpeed(1);
146  UpdateProjection();
147  guiInitiated = true;
148  return true;
149 }

162 void GuiDisplay::RenderItem(int itemId)
163 {
164  GuiItem* item = &(guiItems[itemId]);
165  if (item->hidden) {
166  return;
167  }
168  mat4x4 model;
169  mat4x4_translate(model, (float)item->posx(), (float)item->posy(), 0);
170  mShader.UpdateModelMat(model, nullptr);
171  if (item == mPressedItem) {
172  mShader.UpdateObjColor(mPressedColor);
173  }
174  else if (item->mouseOver) {
175  mShader.UpdateObjColor(mHighlightColor);
176  }
177  else if (itemId > 1 && item->actionKey == 0) {
178  mShader.UpdateObjColor(mTextColor);
179  }
180  else if (item->flags & GUIITEM_CHECKED) {
181  mShader.UpdateObjColor(mToggleColor);
182  }
183  else {
184  mShader.UpdateObjColor(mStdColor);
185  }
186 
187  glBindVertexArray(item->vao);
188  glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, mIbo);
189  glDrawElements(GL_TRIANGLES, 6, GL_UNSIGNED_SHORT, nullptr);
190 }

193 {
194  mMouseOverItem = nullptr;
195  for (unsigned int i = 0; i < NUM_GUI_ITEMS; i++) {
196  GuiItem* g = &(guiItems[i]);
197  if (g->actionKey == 0) {
198  continue;
199  }
200  bool mouseCursorContained = x > g->posx() && x < (g->posx() + g->texItem.w) && y > g->posy()
201  && y < (g->posy() + g->texItem.h);
202 
203  g->mouseOver = !g->hidden && mouseCursorContained;
204 
205  if (g->mouseOver) {
206  mMouseOverItem = g;
207  }
208  }

242 void GuiDisplay::MouseDrag(int /* buttons */, int dx, int /* dy */)
243 {
244  if (mPressedItem == nullptr) {
245  return;
246  }
247  if (mPressedItem->name == eGuiItemThumb) {
248  int newx = mPressedItem->posx() + dx;
249  if (newx < mThumbStartX) {
250  newx = mThumbStartX;
251  }
252  if (newx > ((int)mThumbMaxMotion + mThumbStartX)) {
253  newx = (int)mThumbMaxMotion + mThumbStartX;
254  }
255  if (newx != mPressedItem->posx()) {
256  mMillSim->SetSimulationStage((float)(newx - mThumbStartX) / mThumbMaxMotion);
257  mPressedItem->setPosx(newx);
258  }
259  }
