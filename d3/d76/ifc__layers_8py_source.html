53 def get_layer(layer, project):
54  """Returns (creates if necessary) a layer object in the given project"""
55 
56  group = ifc_tools.get_group(project, "IfcLayersGroup")
57  if not group:
58  return None

73 def populate_layer(obj):
74  """Attaches all the possible objects to this layer"""
75 
76  g = []
77  element = ifc_tools.get_ifc_element(obj)
78  for shape in getattr(element, "AssignedItems", []):
79  rep = getattr(shape, "OfProductRepresentation", None)
80  for prod in getattr(rep, "ShapeOfProduct", []):
81  obj = ifc_tools.get_object(prod)
82  if obj:
83  g.append(obj)
84  obj.Group = g

103 def add_to_layer(obj, layer):
104  """Adds the given object to the given layer"""
105 
106  if hasattr(obj, "StepId"):
107  obj_element = ifc_tools.get_ifc_element(obj)
108  elif hasattr(obj, "id"):

112  return
113  if hasattr(layer, "StepId"):
114  layer_element = ifc_tools.get_ifc_element(layer)
115  elif hasattr(layer, "id"):
116  layer_element = layer
117  layer = ifc_tools.get_object(layer_element)
118  else:
119  return
120  ifcfile = ifc_tools.get_ifcfile(obj)
121  if not ifcfile:
122  return
123  items = ()
124  if layer_element.AssignedItems:
125  items = layer_element.AssignedItems
126  if not obj_element in items:
127  cmd = "attribute.edit_attributes"
128  attribs = {"AssignedItems": items + (obj_element,)}
129  ifc_tools.api_run(cmd, ifcfile, product=layer_element, attributes=attribs)
130  if not obj in layer.Group:
131  layer.Proxy.addObject(layer, obj)
