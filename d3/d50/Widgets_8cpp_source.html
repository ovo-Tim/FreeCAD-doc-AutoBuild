81 void CommandIconView::startDrag (Qt::DropActions supportedActions)
82 {
83  Q_UNUSED(supportedActions);
84  QList<QListWidgetItem*> items = selectedItems();
85  QByteArray itemData;
86  QDataStream dataStream(&itemData, QIODevice::WriteOnly);
87 
88  QPixmap pixmap;
89  dataStream << items.count();
90  for (QList<QListWidgetItem*>::Iterator it = items.begin(); it != items.end(); ++it) {
91  if (it == items.begin())
92  pixmap = ((*it)->data(Qt::UserRole)).value<QPixmap>();
93  dataStream << (*it)->text();
94  }
95 
96  auto mimeData = new QMimeData;
97  mimeData->setData(QString::fromLatin1("text/x-action-items"), itemData);
98 
99  auto drag = new QDrag(this);
100  drag->setMimeData(mimeData);
101  drag->setHotSpot(QPoint(pixmap.width()/2, pixmap.height()/2));
102  drag->setPixmap(pixmap);
103  drag->exec(Qt::MoveAction);
104 }

379 void AccelLineEdit::keyPressEvent (QKeyEvent * e)
380 {
381  if (isReadOnly()) {
382  QLineEdit::keyPressEvent(e);
383  return;
384  }
385 
386  QString txtLine = text();
387 
388  int key = e->key();
389  Qt::KeyboardModifiers state = e->modifiers();
390 
391  // Backspace clears the shortcut if text is present, else sets Backspace as shortcut.
392  // If a modifier is pressed without any other key, return.
393  // AltGr is not a modifier but doesn't have a QString representation.
394  switch(key) {
395  case Qt::Key_Backspace:
396  case Qt::Key_Delete:
397  if (state == Qt::NoModifier) {
398  keyPressedCount = 0;
399  if (isNone()) {
400  QKeySequence ks(key);
401  setText(ks.toString(QKeySequence::NativeText));
402  }
403  else {
404  clear();
405  }
406  }
407  case Qt::Key_Control:
408  case Qt::Key_Shift:
409  case Qt::Key_Alt:
410  case Qt::Key_Meta:
411  case Qt::Key_AltGr:
412  return;
413  default:
414  break;
415  }
416 
417  if (txtLine.isEmpty()) {
418  // Text maybe cleared by QLineEdit's built in clear button
419  keyPressedCount = 0;
420  } else {
421  // 4 keys are allowed for QShortcut
422  switch (keyPressedCount) {
423  case 4:
424  keyPressedCount = 0;
425  txtLine.clear();
426  break;
427  case 0:
428  txtLine.clear();
429  break;
430  default:
431  txtLine += QString::fromLatin1(",");
432  break;
433  }
434  }
435 
436  // Handles modifiers applying a mask.
437  if ((state & Qt::ControlModifier) == Qt::ControlModifier) {
438  QKeySequence ks(Qt::CTRL);
439  txtLine += ks.toString(QKeySequence::NativeText);
440  }
441  if ((state & Qt::AltModifier) == Qt::AltModifier) {
442  QKeySequence ks(Qt::ALT);
443  txtLine += ks.toString(QKeySequence::NativeText);
444  }
445  if ((state & Qt::ShiftModifier) == Qt::ShiftModifier) {
446  QKeySequence ks(Qt::SHIFT);
447  txtLine += ks.toString(QKeySequence::NativeText);
448  }
449  if ((state & Qt::MetaModifier) == Qt::MetaModifier) {
450  QKeySequence ks(Qt::META);
451  txtLine += ks.toString(QKeySequence::NativeText);
452  }
453 
454  // Handles normal keys
455  QKeySequence ks(key);
456  txtLine += ks.toString(QKeySequence::NativeText);
457 
458  setText(txtLine);
459  keyPressedCount++;
460 }
