89  def holeDiameter(self, obj, base, sub):
90  """holeDiameter(obj, base, sub) ... returns the diameter of the specified hole."""
91  try:
92  shape = base.Shape.getElement(sub)
93  if shape.ShapeType == "Vertex":
94  return 0
95 
96  if shape.ShapeType == "Edge" and type(shape.Curve) == Part.Circle:
97  return shape.Curve.Radius * 2
98 
99  if shape.ShapeType == "Face":
100  for i in range(len(shape.Edges)):
101  if (
102  type(shape.Edges[i].Curve) == Part.Circle
103  and shape.Edges[i].Curve.Radius * 2 < shape.BoundBox.XLength * 1.1
104  and shape.Edges[i].Curve.Radius * 2 > shape.BoundBox.XLength * 0.9
105  ):
106  return shape.Edges[i].Curve.Radius * 2
107 
108  # for all other shapes the diameter is just the dimension in X.
109  # This may be inaccurate as the BoundBox is calculated on the tessellated geometry
110  Path.Log.warning(
111  translate(
112  "CAM",
113  "Hole diameter may be inaccurate due to tessellation on face. Consider selecting hole edge.",
114  )
115  )
116  return shape.BoundBox.XLength
117  except Part.OCCError as e:
118  Path.Log.error(e)
119 
120  return 0
