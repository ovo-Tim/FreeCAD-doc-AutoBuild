86 private:
87  void updateDataAndDrawToPosition(Base::Vector2d onSketchPos) override
88  {
89  switch (state()) {
90  case SelectMode::SeekFirst: {
91  toolWidgetManager.drawPositionAtCursor(onSketchPos);
92 
93  if (constructionMethod() == ConstructionMethod::Center) {
94  centerPoint = onSketchPos;
95  }
96  else {
97  firstPoint = onSketchPos;
98  }
99 
100  seekAndRenderAutoConstraint(sugConstraints[0],
101  onSketchPos,
102  Base::Vector2d(0.f, 0.f));
103  } break;
104  case SelectMode::SeekSecond: {
105  if (constructionMethod() == ConstructionMethod::Center) {
106  firstPoint = onSketchPos;
107  startAngle = (firstPoint - centerPoint).Angle();
108  }
109  else {
110  centerPoint = (onSketchPos - firstPoint) / 2 + firstPoint;
111  secondPoint = onSketchPos;
112  }
113 
114  radius = (onSketchPos - centerPoint).Length();
115 
116  CreateAndDrawShapeGeometry();
117 
118  if (constructionMethod() == ConstructionMethod::Center) {
119  toolWidgetManager.drawDirectionAtCursor(onSketchPos, centerPoint);
120  }
121  else {
122  toolWidgetManager.drawPositionAtCursor(onSketchPos);
123  }
124 
125  seekAndRenderAutoConstraint(sugConstraints[1],
126  onSketchPos,
127  Base::Vector2d(0.f, 0.f));
128  } break;
129  case SelectMode::SeekThird: {
130  double startAngleBackup = startAngle;
131 
132  if (constructionMethod() == ConstructionMethod::Center) {
133  secondPoint = onSketchPos;
134  double angle1 = (onSketchPos - centerPoint).Angle() - startAngle;
135  double angle2 = angle1 + (angle1 < 0. ? 2 : -2) * M_PI;
136  arcAngle = abs(angle1 - arcAngle) < abs(angle2 - arcAngle) ? angle1 : angle2;
137 
138  if (arcAngle > 0) {
139  endAngle = startAngle + arcAngle;
140  }
141  else {
142  endAngle = startAngle;
143  startAngle = startAngle + arcAngle;
144  }
145  }
146  else {
147  if (areCollinear(firstPoint, secondPoint, onSketchPos)) {
148  // If points are collinear then we can't calculate the center.
149  return;
150  }
151  centerPoint =
152  Part::Geom2dCircle::getCircleCenter(firstPoint, secondPoint, onSketchPos);
153  radius = (onSketchPos - centerPoint).Length();
154 
155  double angle1 = (firstPoint - centerPoint).Angle();
156  double angle2 = (secondPoint - centerPoint).Angle();
157  double angle3 = (onSketchPos - centerPoint).Angle();
158 
159  // Always build arc counter-clockwise
160  // Point 3 is between Point 1 and 2
161  if (angle3 > std::min(angle1, angle2) && angle3 < std::max(angle1, angle2)) {
162  if (angle2 > angle1) {
163  arcPos1 = Sketcher::PointPos::start;
164  arcPos2 = Sketcher::PointPos::end;
165  }
166  else {
167  arcPos1 = Sketcher::PointPos::end;
168  arcPos2 = Sketcher::PointPos::start;
169  }
170  startAngle = std::min(angle1, angle2);
171  endAngle = std::max(angle1, angle2);
172  arcAngle = endAngle - startAngle;
173  }
174  // Point 3 is not between Point 1 and 2
175  else {
176  if (angle2 > angle1) {
177  arcPos1 = Sketcher::PointPos::end;
178  arcPos2 = Sketcher::PointPos::start;
179  }
180  else {
181  arcPos1 = Sketcher::PointPos::start;
182  arcPos2 = Sketcher::PointPos::end;
183  }
184  startAngle = std::max(angle1, angle2);
185  endAngle = std::min(angle1, angle2);
186  arcAngle = 2 * M_PI - (startAngle - endAngle);
187  }
188  }
189 
190  CreateAndDrawShapeGeometry();
191 
192  if (constructionMethod() == ConstructionMethod::Center) {
193  startAngle = startAngleBackup;
194  }
195 
196  if (constructionMethod() == ConstructionMethod::Center) {
197  toolWidgetManager.drawDoubleAtCursor(onSketchPos, arcAngle, Base::Unit::Angle);
198  seekAndRenderAutoConstraint(sugConstraints[2],
199  onSketchPos,
200  Base::Vector2d(0.0, 0.0));
201  }
202  else {
203  toolWidgetManager.drawPositionAtCursor(onSketchPos);
204  seekAndRenderAutoConstraint(sugConstraints[2],
205  onSketchPos,
206  Base::Vector2d(0.f, 0.f),
207  AutoConstraint::CURVE);
208  }
209 
210  } break;
211  default:
212  break;
213  }

252  void generateAutoConstraints() override
253  {
254  int ArcGeoId = getHighestCurveIndex();
255 
256  auto& ac1 = sugConstraints[0];
257  auto& ac2 = sugConstraints[1];
258  auto& ac3 = sugConstraints[2];
259 
260  if (constructionMethod() == ConstructionMethod::Center) {
261  generateAutoConstraintsOnElement(
262  ac1,
263  ArcGeoId,
264  Sketcher::PointPos::mid); // add auto constraints for the center point
265  generateAutoConstraintsOnElement(ac2,
266  ArcGeoId,
267  (arcAngle > 0) ? Sketcher::PointPos::start
268  : Sketcher::PointPos::end);
269  generateAutoConstraintsOnElement(ac3,
270  ArcGeoId,
271  (arcAngle > 0) ? Sketcher::PointPos::end
272  : Sketcher::PointPos::start);
273  }
274  else {
275  generateAutoConstraintsOnElement(
276  ac1,
277  ArcGeoId,
278  arcPos1); // add auto constraints for the second picked point
279  generateAutoConstraintsOnElement(
280  ac2,
281  ArcGeoId,
282  arcPos2); // add auto constraints for thesecond picked point
283  generateAutoConstraintsOnElement(
284  ac3,
285  ArcGeoId,
286  Sketcher::PointPos::none); // add auto constraints for the third picked point
287  }
288 
289  // Ensure temporary autoconstraints do not generate a redundancy and that the geometry
290  // parameters are accurate This is particularly important for adding widget mandated
291  // constraints.
292  removeRedundantAutoConstraints();

362  void createShape(bool onlyeditoutline) override
363  {
364 
365  ShapeGeometry.clear();
366 
367  if (radius < Precision::Confusion()) {
368  return;
369  }
370 
371  if (state() == SelectMode::SeekSecond) {
372  addCircleToShapeGeometry(toVector3d(centerPoint), radius, isConstructionMode());
373  }
374  else {
375  if (fabs(arcAngle) < Precision::Confusion()) {
376  return;
377  }
378 
379  addArcToShapeGeometry(toVector3d(centerPoint),
380  startAngle,
381  endAngle,
382  radius,
383  isConstructionMode());
384  }
385 
386  if (onlyeditoutline) {
387  if (constructionMethod() == ConstructionMethod::Center) {
388  if (state() == SelectMode::SeekThird) {
389  const double scale = 0.8;
390  addLineToShapeGeometry(
391  toVector3d(centerPoint),
392  Base::Vector3d(centerPoint.x + cos(startAngle) * scale * radius,
393  centerPoint.y + sin(startAngle) * scale * radius,
394  0.),
395  isConstructionMode());
396 
397  addLineToShapeGeometry(
398  toVector3d(centerPoint),
399  Base::Vector3d(centerPoint.x + cos(endAngle) * scale * radius,
400  centerPoint.y + sin(endAngle) * scale * radius,
401  0.),
402  isConstructionMode());
403  }
404  }
405  else {
406  if (state() == SelectMode::SeekSecond) {
407  addLineToShapeGeometry(toVector3d(firstPoint),
408  toVector3d(secondPoint),
409  isConstructionMode());
410  }
411  else if (state() == SelectMode::SeekThird) {
412  const double scale = 0.8;
413  addLineToShapeGeometry(toVector3d(centerPoint),
414  toVector3d(centerPoint)
415  + (toVector3d(secondPoint) - toVector3d(centerPoint))
416  * scale,
417  isConstructionMode());
418 
419  addLineToShapeGeometry(toVector3d(centerPoint),
420  toVector3d(centerPoint)
421  + (toVector3d(firstPoint) - toVector3d(centerPoint))
422  * scale,
423  isConstructionMode());
424  }
425  }
426  }
427  }

509 template<>
510 void DSHArcControllerBase::doEnforceControlParameters(Base::Vector2d& onSketchPos)
511 {
512  switch (handler->state()) {
513  case SelectMode::SeekFirst: {
514  if (onViewParameters[OnViewParameter::First]->isSet) {
515  onSketchPos.x = onViewParameters[OnViewParameter::First]->getValue();
516  }
517 
518  if (onViewParameters[OnViewParameter::Second]->isSet) {
519  onSketchPos.y = onViewParameters[OnViewParameter::Second]->getValue();
520  }
521  } break;
522  case SelectMode::SeekSecond: {
523  if (handler->constructionMethod() == DrawSketchHandlerArc::ConstructionMethod::Center) {
524  Base::Vector2d dir = onSketchPos - handler->centerPoint;
525  if (dir.Length() < Precision::Confusion()) {
526  dir.x = 1.0; // if direction null, default to (1,0)
527  }
528  double radius = dir.Length();
529 
530  if (onViewParameters[OnViewParameter::Third]->isSet) {
531  radius = onViewParameters[OnViewParameter::Third]->getValue();
532  if (radius < Precision::Confusion()) {
533  unsetOnViewParameter(onViewParameters[OnViewParameter::Third].get());
534  return;
535  }
536 
537  onSketchPos = handler->centerPoint + radius * dir.Normalize();
538  }
539 
540  if (onViewParameters[OnViewParameter::Fourth]->isSet) {
541  double angle =
542  Base::toRadians(onViewParameters[OnViewParameter::Fourth]->getValue());
543  onSketchPos.x = handler->centerPoint.x + cos(angle) * radius;
544  onSketchPos.y = handler->centerPoint.y + sin(angle) * radius;
545  }
546  }
547  else {
548  if (onViewParameters[OnViewParameter::Third]->isSet) {
549  onSketchPos.x = onViewParameters[OnViewParameter::Third]->getValue();
550  }
551 
552  if (onViewParameters[OnViewParameter::Fourth]->isSet) {
553  onSketchPos.y = onViewParameters[OnViewParameter::Fourth]->getValue();
554  }
555  if (onViewParameters[OnViewParameter::Third]->isSet
556  && onViewParameters[OnViewParameter::Fourth]->isSet
557  && (onSketchPos - handler->firstPoint).Length() < Precision::Confusion()) {
558  unsetOnViewParameter(onViewParameters[OnViewParameter::Third].get());
559  unsetOnViewParameter(onViewParameters[OnViewParameter::Fourth].get());
560  }
561  }
562  } break;
563  case SelectMode::SeekThird: {
564  if (handler->constructionMethod() == DrawSketchHandlerArc::ConstructionMethod::Center) {
565  if (onViewParameters[OnViewParameter::Fifth]->isSet) {
566  double arcAngle =
567  Base::toRadians(onViewParameters[OnViewParameter::Fifth]->getValue());
568  if (fmod(fabs(arcAngle), 2 * M_PI) < Precision::Confusion()) {
569  unsetOnViewParameter(onViewParameters[OnViewParameter::Fifth].get());
570  return;
571  }
572  double angle = handler->startAngle + arcAngle;
573  onSketchPos.x = handler->centerPoint.x + cos(angle) * handler->radius;
574  onSketchPos.y = handler->centerPoint.y + sin(angle) * handler->radius;
575  }
576  }
577  else {
578  if (onViewParameters[OnViewParameter::Fifth]->isSet) {
579  onSketchPos.x = onViewParameters[OnViewParameter::Fifth]->getValue();
580  }
581 
582  if (onViewParameters[OnViewParameter::Sixth]->isSet) {
583  onSketchPos.y = onViewParameters[OnViewParameter::Sixth]->getValue();
584  }
585  }
586  } break;
587  default:
588  break;
589  }

592 template<>
593 void DSHArcController::adaptParameters(Base::Vector2d onSketchPos)
594 {
595  switch (handler->state()) {
596  case SelectMode::SeekFirst: {
597  if (!onViewParameters[OnViewParameter::First]->isSet) {
598  setOnViewParameterValue(OnViewParameter::First, onSketchPos.x);
599  }
600 
601  if (!onViewParameters[OnViewParameter::Second]->isSet) {
602  setOnViewParameterValue(OnViewParameter::Second, onSketchPos.y);
603  }
604 
605  bool sameSign = onSketchPos.x * onSketchPos.y > 0.;
606  onViewParameters[OnViewParameter::First]->setLabelAutoDistanceReverse(!sameSign);
607  onViewParameters[OnViewParameter::Second]->setLabelAutoDistanceReverse(sameSign);
608  onViewParameters[OnViewParameter::First]->setPoints(Base::Vector3d(),
609  toVector3d(onSketchPos));
610  onViewParameters[OnViewParameter::Second]->setPoints(Base::Vector3d(),
611  toVector3d(onSketchPos));
612  } break;
613  case SelectMode::SeekSecond: {
614  if (handler->constructionMethod() == DrawSketchHandlerArc::ConstructionMethod::Center) {
615 
616  if (!onViewParameters[OnViewParameter::Third]->isSet) {
617  setOnViewParameterValue(OnViewParameter::Third, handler->radius);
618  }
619  double range = Base::toDegrees(handler->startAngle);
620  if (!onViewParameters[OnViewParameter::Fourth]->isSet) {
621  setOnViewParameterValue(OnViewParameter::Fourth, range, Base::Unit::Angle);
622  }
623 
624  Base::Vector3d start = toVector3d(handler->centerPoint);
625  Base::Vector3d end = toVector3d(onSketchPos);
626 
627  onViewParameters[OnViewParameter::Third]->setPoints(start, end);
628  onViewParameters[OnViewParameter::Fourth]->setPoints(start, Base::Vector3d());
629  onViewParameters[OnViewParameter::Fourth]->setLabelRange(handler->startAngle);
630  }
631  else {
632  if (!onViewParameters[OnViewParameter::Third]->isSet) {
633  setOnViewParameterValue(OnViewParameter::Third, onSketchPos.x);
634  }
635 
636  if (!onViewParameters[OnViewParameter::Fourth]->isSet) {
637  setOnViewParameterValue(OnViewParameter::Fourth, onSketchPos.y);
638  }
639 
640  bool sameSign = onSketchPos.x * onSketchPos.y > 0.;
641  onViewParameters[OnViewParameter::Third]->setLabelAutoDistanceReverse(!sameSign);
642  onViewParameters[OnViewParameter::Fourth]->setLabelAutoDistanceReverse(sameSign);
643  onViewParameters[OnViewParameter::Third]->setPoints(Base::Vector3d(),
644  toVector3d(onSketchPos));
645  onViewParameters[OnViewParameter::Fourth]->setPoints(Base::Vector3d(),
646  toVector3d(onSketchPos));
647  }
648  } break;
649  case SelectMode::SeekThird: {
650  if (handler->constructionMethod() == DrawSketchHandlerArc::ConstructionMethod::Center) {
651  double range = Base::toDegrees(handler->arcAngle);
652 
653  if (!onViewParameters[OnViewParameter::Fifth]->isSet) {
654  setOnViewParameterValue(OnViewParameter::Fifth, range, Base::Unit::Angle);
655  }
656 
657  Base::Vector3d start = toVector3d(handler->centerPoint);
658  onViewParameters[OnViewParameter::Fifth]->setPoints(start, Base::Vector3d());
659 
660  onViewParameters[OnViewParameter::Fifth]->setLabelStartAngle(handler->startAngle);
661  onViewParameters[OnViewParameter::Fifth]->setLabelRange(handler->arcAngle);
662  }
663  else {
664  if (!onViewParameters[OnViewParameter::Fifth]->isSet) {
665  setOnViewParameterValue(OnViewParameter::Fifth, onSketchPos.x);
666  }
667 
668  if (!onViewParameters[OnViewParameter::Sixth]->isSet) {
669  setOnViewParameterValue(OnViewParameter::Sixth, onSketchPos.y);
670  }
671 
672  bool sameSign = onSketchPos.x * onSketchPos.y > 0.;
673  onViewParameters[OnViewParameter::Fifth]->setLabelAutoDistanceReverse(!sameSign);
674  onViewParameters[OnViewParameter::Sixth]->setLabelAutoDistanceReverse(sameSign);
675  onViewParameters[OnViewParameter::Fifth]->setPoints(Base::Vector3d(),
676  toVector3d(onSketchPos));

741  auto p5set = onViewParameters[OnViewParameter::Fifth]->isSet;
742 
743 
744  PointPos pos1 =
745  handler->constructionMethod() == DrawSketchHandlerArc::ConstructionMethod::Center
746  ? PointPos::mid
747  : handler->arcPos1;

750  ConstraintToAttachment(GeoElementId(firstCurve, pos1), GeoElementId::VAxis, x0, obj);
751  };
752 
753  auto constrainty0 = [&]() {

785

807  }

811  ConstraintToAttachment(GeoElementId(firstCurve, handler->arcPos2),
812  GeoElementId::RtPnt,
813  0.,
814  obj);
815  }

817  if (p3set) {
818  constraintp3x();
819  }
820 
821  if (p4set) {
822  constraintp4y();
823  }
824  }
825  }
826  }
827  else { // Valid diagnosis. Must check which constraints may be added.
828

831  return;
832  }
