164 PyObject* TopoShapeCompoundPy::setFaces(PyObject *args)
165 {
166  using Facet = Data::ComplexGeoData::Facet;
167  using Point = Base::Vector3d;
168 
169  std::vector<Point> points;
170  std::vector<Facet> facets;
171 
172  PyObject* data{};
173  double accuracy = 1.0e-06; // NOLINT
174  if (!PyArg_ParseTuple(args, "O!|d", &PyTuple_Type, &data, &accuracy)) {
175  return nullptr;
176  }
177 
178  Py::Tuple tuple(data);
179 
180  Py::Sequence pts(tuple.getItem(0));
181  points.reserve(pts.size());
182  for (const auto& pt : pts) {
183  Py::Vector vec(pt);
184  points.push_back(vec.toVector());
185  }
186 
187  std::size_t count = points.size();
188  auto checkFace = [count](const Facet& face) {
189  if (face.I1 >= count) {
190  return false;
191  }
192  if (face.I2 >= count) {
193  return false;
194  }
195  if (face.I3 >= count) {
196  return false;
197  }
198 
199  return true;
200  };
201 
202  Py::Sequence fts(tuple.getItem(1));
203  facets.reserve(fts.size());
204  Facet face;
205  for (const auto& ft : fts) {
206  Py::Tuple index(ft);
207  face.I1 = int32_t(static_cast<int>(Py::Long(index.getItem(0))));
208  face.I2 = int32_t(static_cast<int>(Py::Long(index.getItem(1))));
209  face.I3 = int32_t(static_cast<int>(Py::Long(index.getItem(2))));
210 
211  if (!checkFace(face)) {
212  PyErr_SetString(PyExc_ValueError, "Point index out of range");
213  return nullptr;
214  }
215 
216  facets.push_back(face);
217  }
218 
219  getTopoShapePtr()->setFaces(points, facets, accuracy);
220 
221  Py_Return;
