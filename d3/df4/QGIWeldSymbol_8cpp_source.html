161 void QGIWeldSymbol::drawTile(TechDraw::DrawTileWeld* tileFeat)
162 {
163  if (!tileFeat) {
164  Base::Console().Message("QGIWS::drawTile - tile is null\n");
165  return;
166  }
167 
168  const auto symbol = getFeature();
169  if (!symbol) {
170  return;
171  }
172  auto vp = dynamic_cast<ViewProviderWeld *>(getViewProvider(symbol));
173  if (!vp) {
174  return;
175  }
176 
177  std::string fontName = vp->Font.getValue();
178  int fontSize = QGIView::exactFontSize(vp->Font.getValue(),
179  vp->TileFontSize.getValue());
180 
181  double featScale = getLeader()->getScale();
182 
183  std::string tileTextL = tileFeat->LeftText.getValue();
184  std::string tileTextR = tileFeat->RightText.getValue();
185  std::string tileTextC = tileFeat->CenterText.getValue();
186  int row = (int)tileFeat->TileRow.getValue();
187  int col = (int)tileFeat->TileColumn.getValue();
188 
189  auto tile = new QGITile(tileFeat);
190  addToGroup(tile);
191 
192  QPointF org = getTileOrigin();
193  tile->setTilePosition(org, row, col);
194  tile->setFont(fontName, fontSize);
195  tile->setColor(getCurrentColor());
196  tile->setTileTextLeft(tileTextL);
197  tile->setTileTextRight(tileTextR);
198  tile->setTileTextCenter(tileTextC);
199  tile->setZValue(ZVALUE::DIMENSION);
200  tile->setTileScale(featScale);
201  tile->setTailRight(getFeature()->isTailRightSide());
202  tile->setAltWeld(getFeature()->AlternatingWeld.getValue());
203 
204  auto localAxes = getLocalAxes();
205  tile->setLocalAxes(localAxes.first, localAxes.second);
206  auto tilePos = tile->calcTilePosition(); // this is not the center! pos is left and up from center
207 
208  auto rotationDeg = getLastSegAngle();
209  Base::Vector3d stdX{1, 0, 0};
210  auto xdir = localAxes.first;
211  auto dot = stdX.Dot(xdir);
212  constexpr double DegreesHalfCircle{180.0};
213  if (dot < 0) {
214  rotationDeg -= DegreesHalfCircle;
215  }
216  tile->setRotation(rotationDeg); // Qt angles are clockwise
217 
218  tile->draw();
219  tile->setPos(tilePos);
220 
221 }

248 void QGIWeldSymbol::drawTailText()
249 {
250  constexpr double DegreesHalfCircle{180.0};
251  constexpr double verticalBRectAdjustFactor{1.2}; // text bounding rect is bigger than font size and text is not
252  // vertically aligned with brect midline, so we artificially
253  // bump the text size so the middle aligns with the leader
254 
255  QPointF textPos = getTailPoint();
256  m_tailText->setPos(textPos); //avoid messing up brect with empty item at 0, 0 !!!
257  std::string tText = getFeature()->TailText.getValue();
258  if (tText.empty()) {
259  m_tailText->hide();
260  return;
261  }
262 
263  m_tailText->show();
264 
265  const auto symbol = getFeature();
266  if (!symbol) {
267  return;
268  }
269  auto vp = dynamic_cast<ViewProviderWeld *>(getViewProvider(symbol));
270  if (!vp) {
271  return;
272  }
273  QString qFontName = QString::fromStdString(vp->Font.getValue());
274  int fontSize = QGIView::exactFontSize(vp->Font.getValue(),
275  vp->FontSize.getValue()); // this is different from the size used in the tiles
276 
277  m_font.setFamily(qFontName);
278  m_font.setPixelSize(fontSize);
279 
280  m_tailText->setFont(m_font);
281  m_tailText->setPlainText(
282  QString::fromUtf8(tText.c_str()));
283  m_tailText->setColor(getCurrentColor());
284  m_tailText->setZValue(ZVALUE::DIMENSION);
285 
286  auto brWidth = m_tailText->boundingRect().width();
287  auto brHeight = m_tailText->boundingRect().height();
288  auto brCharWidth = brWidth / tText.length();
289 
290  double hMargin = brCharWidth + prefArrowSize(); // leave a gap from the graphic
291  auto angleDeg = getLastSegAngle();
292 
293  auto localAxes = getLocalAxes();
294  auto xdir = localAxes.first;
295  auto ydir = localAxes.second;
296  Base::Vector3d stdX{1, 0, 0};
297  auto xdirStartOffset = DU::toQPointF(xdir * brWidth) * -1; // start of text in xDir when leader points left
298  auto dot = stdX.Dot(xdir);
299  if (dot < 0) {
300  angleDeg -= DegreesHalfCircle;
301  xdirStartOffset = QPointF(0, 0);
302  ydir = -ydir;
303  }
304 
305  auto xAdjust = DU::toQPointF(xdir * hMargin);
306  auto yAdjust = DU::toQPointF(ydir * (brHeight * verticalBRectAdjustFactor) / 2);
307 
308  // QGCustomText doesn't know about rotation so we can't use the justify methods
309  QPointF justPoint = textPos - xAdjust + yAdjust; // original
310  justPoint += xdirStartOffset;
311 
312  m_tailText->setPos(justPoint);
313 
314  auto kink = getLeader()->getKinkPoint();
315  m_tailText->setTransformOriginPoint(DU::toQPointF(kink));
316 
317  m_tailText->setRotation(angleDeg);
318 }

320 void QGIWeldSymbol::drawFieldFlag()
321 {
322  QPointF fieldFlagPos = getKinkPoint();
323  m_fieldFlag->setPos(fieldFlagPos);
324 
325  if (getFeature()->FieldWeld.getValue()) {
326  m_fieldFlag->show();
327  } else {
328  m_fieldFlag->hide();
329  return;
330  }
331 
332  auto localAxes = getLocalAxes();
333  auto xdir = localAxes.first;
334  auto ydir = localAxes.second;
335 
336  // TODO: similar code used here and in QGITiled to handle leader points left. different actions, though.
337  // used here and QGITile.
338  Base::Vector3d stdX{1, 0, 0};
339  auto dot = stdX.Dot(xdir);
340  if (dot < 0) {
341  // our leader points left, so yDir should be +Y, but we need -Y up
342  ydir = -ydir;
343  }
344 
345 // NOLINTBEGIN readability-magic-numbers
346  std::vector<QPointF> flagPoints = { QPointF(0.0, 0.0),
347  DU::toQPointF(ydir * 3.0),
348  DU::toQPointF(xdir * -2.0 + ydir * 2.5),
349  DU::toQPointF(ydir * 2.0) };
350 // NOLINTEND readability-magic-numbers
351 
352  double scale = (float)PreferencesGui::dimFontSizePX() / 2;
353  QPainterPath path;
354  path.moveTo(flagPoints.at(0) * scale);
355  size_t iSeg = 1;
356  for ( ; iSeg < flagPoints.size(); iSeg++) {
357  path.lineTo(flagPoints.at(iSeg) * scale);
358  }
359 
360  auto qgiLead = dynamic_cast<QGILeaderLine *>(getQGIVByName(getLeader()->getNameInDocument()));
361  if (qgiLead) {
362  m_fieldFlag->setWidth(qgiLead->getLineWidth());
363  }
364  m_fieldFlag->setZValue(ZVALUE::DIMENSION);
365 
366  m_fieldFlag->setPath(path);
367 }
