64 bool StdCmdMeasure::isActive()
65 {
66  App::Document* doc = App::GetApplication().getActiveDocument();
67  if (!doc || doc->countObjectsOfType<App::GeoFeature>() == 0) {
68  return false;
69  }
70 
71  Gui::MDIView* view = Gui::getMainWindow()->activeWindow();
72  if (view && view->isDerivedFrom(Gui::View3DInventor::getClassTypeId())) {
73  Gui::View3DInventorViewer* viewer = dynamic_cast<Gui::View3DInventor*>(view)->getViewer();
74  return !viewer->isEditing();
75  }
76  return false;
77 }
