23 """Provide the grid observer for the Draft and BIM workbenches.
24 """

26 import FreeCAD

29 # View observer code to update the Draft_ToggleGrid command button to reflect
30 # the grid's visibility status.
31 # Based on view observer code to update the Draft Tray.
32 if FreeCAD.GuiUp:
33  import FreeCADGui
34  from PySide import QtWidgets
35  from draftutils import gui_utils
36  from draftutils.todo import ToDo
37 
38  def _update_grid_gui():
39  """Update function executed by the callback to refresh the Toggle Grid
40  button state on all toolbars and menus.
41  """
42 
43  try:
44  # Get the active view.
45  view = gui_utils.get_3d_view()
46 
47  # If the active view is not a 3D view, uncheck the button.
48  if view is None:
49  _set_grid_button_state(False)
50  return
51 
52  # Otherwise, if there is a view, update the button's checked state.
53 
54  # Retrieve the associated grid for each MDI document.
55  # [1] is the index where the grid is stored in the
56  # trackers list. See setTrackers().
57  view_idx = FreeCADGui.Snapper.trackers[0].index(view)
58  grid = FreeCADGui.Snapper.trackers[1][view_idx]
59  if grid.Visible:
60  _set_grid_button_state(True)
61  else:
62  _set_grid_button_state(False)
63 
64  except Exception:
65  pass

67  def _view_observer_callback():
68  """Callback function to update the Toggle Grid button.
69 
70  The update will only happen if either the Draft or BIM
71  workbenches are active.
72  """
73 
74  ToDo.delay(_update_grid_gui, None)
75 
76  _view_observer_active = False
77 
78  def _view_observer_setup():
79  """Start or stop the grid observer.
80 
81  Start: it is intended to happen when either the Draft or BIM workbench
82  are activated. The _view_observer_callback is connected to Qt's
83  subWindowActivated signal. The MDI area emits the subWindowActivated()
84  signal when the active window changes.
85 
86  Stop: This happens when either the Draft or BIM workbenches are
87  deactivated and is the reverse of the start operation.
88  """
89 
90  mw = FreeCADGui.getMainWindow()
91  mdi = mw.findChild(QtWidgets.QMdiArea)
92  global _view_observer_active
93  if not _view_observer_active:
94  # Connect callback to subWindowActivated signal
95  mdi.subWindowActivated.connect(_view_observer_callback)
96  _view_observer_active = True
97  # Trigger initial grid button update
98  _view_observer_callback()
99  else:
100  mdi.subWindowActivated.disconnect(_view_observer_callback)
101  _view_observer_active = False
102 
103  def _set_grid_button_state(button_check):
104  """Sets the checked state of the Draft_ToggleGrid command.
105  This is then reflected on the associated "Toggle Grid" buttons of every
106  toolbar and menu.
107 
108  Args:
109  button_check (bool): if True, check the grid button.
110 
111  Returns:
112  None
113  """
114 
115  action = FreeCADGui.Command.get("Draft_ToggleGrid").getAction()[0]
116  action.setCheckable(True)
117  action.setChecked(button_check)
