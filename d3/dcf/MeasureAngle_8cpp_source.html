68 bool MeasureAngle::isValidSelection(const App::MeasureSelection& selection)
69 {
70  if (selection.size() != 2) {
71  return false;
72  }
73 
74  for (auto element : selection) {
75  auto type = App::MeasureManager::getMeasureElementType(element);

77  if (type == App::MeasureElementType::INVALID) {

89 bool MeasureAngle::isPrioritizedSelection(const App::MeasureSelection& selection)
90 {
91  if (selection.size() != 2) {
92  return false;
93  }
94 
95  // Check if the two elements are parallel
96  auto element1 = selection.at(0);
97  auto objT1 = element1.object;
98  App::DocumentObject* ob1 = objT1.getObject();
99  std::string sub1 = objT1.getSubName();
100  Base::Vector3d vec1;
101  getVec(*ob1, sub1, vec1);
102 
103  auto element2 = selection.at(1);
104  auto objT2 = element2.object;
105  App::DocumentObject* ob2 = objT2.getObject();
106  std::string sub2 = objT2.getSubName();
107  Base::Vector3d vec2;
108  getVec(*ob2, sub2, vec2);
109 
110 
111  double angle = std::fmod(vec1.GetAngle(vec2), D_PI);
112  return angle > Base::Precision::Angular();
113 }

214 App::DocumentObjectExecReturn* MeasureAngle::execute()
215 {
216  App::DocumentObject* ob1 = Element1.getValue();
217  std::vector<std::string> subs1 = Element1.getSubValues();
218 
219  App::DocumentObject* ob2 = Element2.getValue();
220  std::vector<std::string> subs2 = Element2.getSubValues();
221 
222  if (!ob1 || !ob1->isValid() || !ob2 || !ob2->isValid()) {
223  return new App::DocumentObjectExecReturn("Submitted object(s) is not valid");
224  }
225 
226  if (subs1.empty() || subs2.empty()) {
227  return new App::DocumentObjectExecReturn("No geometry element picked");
228  }
229 
230  Base::Vector3d vec1;
231  getVec(*ob1, subs1.at(0), vec1);
232 
233  Base::Vector3d vec2;
234  getVec(*ob2, subs2.at(0), vec2);
235 
236  Angle.setValue(Base::toDegrees(vec1.GetAngle(vec2)));
237 
238  return DocumentObject::StdReturn;
239 }
