91 void UnifiedDatumCommand(Gui::Command &cmd, Base::Type type, std::string name)
92 {
93  try{
94  std::string fullTypeName (type.getName());
95 
96  App::PropertyLinkSubList support;
97  cmd.getSelection().getAsPropertyLinkSubList(support);
98 
99  bool bEditSelected = false;
100  if (support.getSize() == 1 && support.getValue() ) {
101  if (support.getValue()->isDerivedFrom(type))
102  bEditSelected = true;
103  }
104 
105  PartDesign::Body *pcActiveBody = PartDesignGui::getBody(/*messageIfNot = */true);
106 
107  if (bEditSelected) {
108  std::string tmp = std::string("Edit ")+name;
109  cmd.openCommand(tmp.c_str());
110  PartDesignGui::setEdit(support.getValue(),pcActiveBody);
111  } else if (pcActiveBody) {
112 
113  // TODO Check how this will work outside of a body (2015-10-20, Fat-Zer)
114  std::string FeatName = cmd.getUniqueObjectName(name.c_str(), pcActiveBody);
115 
116  std::string tmp = std::string("Create ")+name;
117 
118  cmd.openCommand(tmp.c_str());
119  FCMD_OBJ_CMD(pcActiveBody,"newObject('" << fullTypeName << "','" << FeatName << "')");
120 
121  // remove the body from links in case it's selected as
122  // otherwise a cyclic dependency will be created
123  support.removeValue(pcActiveBody);
124 
125  auto Feat = pcActiveBody->getDocument()->getObject(FeatName.c_str());
126  if (!Feat)
127  return;
128 
129  //test if current selection fits a mode.
130  if (support.getSize() > 0) {
131  Part::AttachExtension* pcDatum = Feat->getExtensionByType<Part::AttachExtension>();
132  pcDatum->attacher().setReferences(support);
133  SuggestResult sugr;
134  pcDatum->attacher().suggestMapModes(sugr);
135  if (sugr.message == Attacher::SuggestResult::srOK) {
136  //fits some mode. Populate AttachmentSupport property.
137  FCMD_OBJ_CMD(Feat,"AttachmentSupport = " << support.getPyReprString());
138  FCMD_OBJ_CMD(Feat,"MapMode = '" << AttachEngine::getModeName(sugr.bestFitMode) << "'");
139  } else {
140  QMessageBox::information(Gui::getMainWindow(),QObject::tr("Invalid selection"), QObject::tr("There are no attachment modes that fit selected objects. Select something else."));
141  }
142  }
143  cmd.doCommand(Gui::Command::Doc,"App.activeDocument().recompute()"); // recompute the feature based on its references
144  PartDesignGui::setEdit(Feat,pcActiveBody);
145  } else {
146  QMessageBox::warning(Gui::getMainWindow(),QObject::tr("Error"), QObject::tr("There is no active body. Please make a body active before inserting a datum entity."));
147  }
148  } catch (Base::Exception &e) {
149  QMessageBox::warning(Gui::getMainWindow(),QObject::tr("Error"),QApplication::translate("Exception", e.what()));
150  } catch (Standard_Failure &e) {
151  QMessageBox::warning(Gui::getMainWindow(),QObject::tr("Error"),QString::fromLatin1(e.GetMessageString()));
152  }
153 }

353 void CmdPartDesignSubShapeBinder::activated(int iMsg)
354 {
355  Q_UNUSED(iMsg);
356 
357  App::DocumentObject *parent = nullptr;
358  std::string parentSub;
359  std::map<App::DocumentObject *, std::vector<std::string> > values;
360  for (auto &sel : Gui::Selection().getCompleteSelection(Gui::ResolveMode::NoResolve)) {
361  if (!sel.pObject) continue;
362  auto &subs = values[sel.pObject];
363  if (sel.SubName && sel.SubName[0])
364  subs.emplace_back(sel.SubName);
365  }
366 
367  std::string FeatName;
368  PartDesign::Body *pcActiveBody = PartDesignGui::getBody(false,true,true,&parent,&parentSub);
369  FeatName = getUniqueObjectName("Binder",pcActiveBody);
370  if (parent) {
371  decltype(values) links;
372  for (auto &v : values) {
373  App::DocumentObject *obj = v.first;
374  if (obj != parent) {
375  auto &subs = links[obj];
376  subs.insert(subs.end(),v.second.begin(),v.second.end());
377  continue;
378  }
379  for (auto &sub : v.second) {
380  auto link = obj;
381  auto linkSub = parentSub;
382  parent->resolveRelativeLink(linkSub,link,sub);
383  if (link && link != pcActiveBody)
384  links[link].push_back(sub);
385  }
386  }
387  values = std::move(links);
388  }
389 
390  PartDesign::SubShapeBinder *binder = nullptr;
391  try {
392  openCommand(QT_TRANSLATE_NOOP("Command", "Create SubShapeBinder"));
393  if (pcActiveBody) {
394  FCMD_OBJ_CMD(pcActiveBody,"newObject('PartDesign::SubShapeBinder','" << FeatName << "')");
395  binder = dynamic_cast<PartDesign::SubShapeBinder*>(pcActiveBody->getObject(FeatName.c_str()));
396  } else {
397  doCommand(Command::Doc,
398  "App.ActiveDocument.addObject('PartDesign::SubShapeBinder','%s')",FeatName.c_str());
399  binder = dynamic_cast<PartDesign::SubShapeBinder*>(
400  App::GetApplication().getActiveDocument()->getObject(FeatName.c_str()));
401  }
402  if (!binder)
403  return;
404  binder->setLinks(std::move(values));
405  updateActive();
406  commitCommand();
407  } catch (Base::Exception &e) {
408  e.ReportException();
409  QMessageBox::critical(Gui::getMainWindow(),
410  QObject::tr("Sub-Shape Binder"), QApplication::translate("Exception", e.what()));
411  abortCommand();
412  }
413 }

574 // Common utility functions for ProfileBased features
575 //===========================================================================

577 // Take a list of Part2DObjects and classify them for creating a
578 // ProfileBased feature. FirstFreeSketch is the first free sketch in the same body
579 // or sketches.end() if non available. The returned number is the amount of free sketches
580 unsigned validateSketches(std::vector<App::DocumentObject*>& sketches,
581  std::vector<PartDesignGui::TaskFeaturePick::featureStatus>& status,
582  std::vector<App::DocumentObject*>::iterator& firstFreeSketch)
583 {
584  // TODO Review the function for non-part bodies (2015-09-04, Fat-Zer)
585  PartDesign::Body* pcActiveBody = PartDesignGui::getBody(false);
586  App::Part* pcActivePart = PartDesignGui::getPartFor(pcActiveBody, false);
587 
588  // TODO: If the user previously opted to allow multiple use of sketches or use of sketches from other bodies,
589  // then count these as valid sketches!
590  unsigned freeSketches = 0;
591  firstFreeSketch = sketches.end();
592 
593  for (std::vector<App::DocumentObject*>::iterator s = sketches.begin(); s != sketches.end(); s++) {
594 
595  if (!pcActiveBody) {
596  // We work in the old style outside any body
597  if (PartDesign::Body::findBodyOf (*s)) {
598  status.push_back(PartDesignGui::TaskFeaturePick::otherPart);
599  continue;
600  }
601  } else if (!pcActiveBody->hasObject(*s)) {
602  // Check whether this plane belongs to a body of the same part
603  PartDesign::Body* b = PartDesign::Body::findBodyOf(*s);
604  if (!b)
605  status.push_back(PartDesignGui::TaskFeaturePick::notInBody);
606  else if (pcActivePart && pcActivePart->hasObject(b, true))
607  status.push_back(PartDesignGui::TaskFeaturePick::otherBody);
608  else
609  status.push_back(PartDesignGui::TaskFeaturePick::otherPart);
610 
611  continue;
612  }
613 
614  //Base::Console().Error("Checking sketch %s\n", (*s)->getNameInDocument());
615  // Check whether this sketch is already being used by another feature
616  // Body features don't count...
617  std::vector<App::DocumentObject*> inList = (*s)->getInList();
618  std::vector<App::DocumentObject*>::iterator o = inList.begin();
619  while (o != inList.end()) {
620  //Base::Console().Error("Inlist: %s\n", (*o)->getNameInDocument());
621  if ((*o)->isDerivedFrom<PartDesign::Body>())
622  o = inList.erase(o); //ignore bodies
623  else if (!( (*o)->isDerivedFrom<PartDesign::Feature>() ))
624  o = inList.erase(o); //ignore non-partDesign
625  else
626  ++o;
627  }
628  if (!inList.empty()) {
629  status.push_back(PartDesignGui::TaskFeaturePick::isUsed);
630  continue;
631  }
632 
633  if (pcActiveBody && pcActiveBody->isAfterInsertPoint(*s)){
634  status.push_back(PartDesignGui::TaskFeaturePick::afterTip);
635  continue;
636  }
637 
638  // Check whether the sketch shape is valid
639  Part::Part2DObject* sketch = static_cast<Part::Part2DObject*>(*s);
640  const TopoDS_Shape& shape = sketch->Shape.getValue();
641  if (shape.IsNull()) {
642  status.push_back(PartDesignGui::TaskFeaturePick::invalidShape);
643  continue;
644  }
645 
646  // count free wires
647  int ctWires=0;
648  TopExp_Explorer ex;
649  for (ex.Init(shape, TopAbs_WIRE); ex.More(); ex.Next()) {
650  ctWires++;
651  }
652  if (ctWires == 0) {
653  status.push_back(PartDesignGui::TaskFeaturePick::noWire);
654  continue;
655  }
656 
657  // All checks passed - found a valid sketch
658  if (firstFreeSketch == sketches.end())
659  firstFreeSketch = s;
660  freeSketches++;
661  status.push_back(PartDesignGui::TaskFeaturePick::validFeature);
662  }
663 
664  return freeSketches;

677 bool importExternalElements(App::PropertyLinkSub& prop, std::vector<App::SubObjectT> _sobjs)
678 {
679  if (!prop.getName() || !prop.getName()[0]) {
680  FC_THROWM(Base::RuntimeError, "Invalid property");
681  }
682  auto editObj = Base::freecad_dynamic_cast<App::DocumentObject>(prop.getContainer());
683  if (!editObj) {
684  FC_THROWM(Base::RuntimeError, "Editing object not found");
685  }
686  auto body = PartDesign::Body::findBodyOf(editObj);
687  if (!body) {
688  FC_THROWM(Base::RuntimeError,
689  "No body for editing object: " << editObj->getNameInDocument());
690  }
691  std::map<App::DocumentObject*, std::vector<std::string>> links;
692  std::vector<App::SubObjectT> sobjs;
693  auto docName = editObj->getDocument()->getName();
694  auto inList = editObj->getInListEx(true);
695  for (auto sobjT : _sobjs) {
696  auto sobj = sobjT.getSubObject();
697  if (sobj == editObj) {
698  continue;
699  }
700  if (!sobj) {
701  FC_THROWM(Base::RuntimeError,
702  "Object not found: " << sobjT.getSubObjectFullName(docName));
703  }
704  if (inList.count(sobj)) {
705  FC_THROWM(Base::RuntimeError,
706  "Cyclic dependency on object " << sobjT.getSubObjectFullName(docName));
707  }
708  sobjT.normalized();
709  // Make sure that if a subelement is chosen for some object,
710  // we exclude whole object reference for that object.
711  auto& subs = links[sobj];
712  std::string element = sobjT.getOldElementName();
713  if (element.size()) {
714  if (subs.size() == 1 && subs.front().empty()) {
715  for (auto it = sobjs.begin(); it != sobjs.end();) {
716  if (it->getSubObject() == sobj) {
717  sobjs.erase(it);
718  break;
719  }
720  }
721  }
722  }
723  else if (subs.size() > 0) {
724  continue;
725  }
726  subs.push_back(std::move(element));
727  sobjs.push_back(sobjT);
728  }
729 
730  int import = 0;
731  App::DocumentObject* obj = nullptr;
732  std::vector<std::string> subs;
733  for (const auto& sobjT : sobjs) {
734  auto sobj = sobjT.getSubObject();
735  if (PartDesign::Body::findBodyOf(sobj) != body) {
736  import = 1;
737  break;
738  }
739  if (!obj) {
740  obj = sobj;
741  }
742  else if (obj != sobj) {
743  if (!import) {
744  import = -1;
745  }
746  break;
747  }
748  subs.push_back(sobjT.getOldElementName());
749  }
750  if (!import) {
751  if (subs.empty()) {
752  subs.emplace_back();
753  }
754  if (obj == prop.getValue() && prop.getSubValues() == subs) {
755  return false;
756  }
757  prop.setValue(obj, std::move(subs));
758  return true;
759  }
760  return false;
761 }
