84  if preferences['SPLIT_LAYERS'] and hasattr(settings,"APPLY_LAYERSETS"):
85  settings.set(settings.APPLY_LAYERSETS,True)

90  docname = os.path.splitext(os.path.basename(filename))[0]
91  doc = FreeCAD.newDocument(docname)
92  doc.Label = docname
93  FreeCAD.setActiveDocument(doc.Name)
94

96  ifcfile = ifcopenshell.open(filename)
97  progressbar = Base.ProgressIndicator()
98  productscount = len(ifcfile.by_type("IfcProduct"))
99  progressbar.start("Importing "+str(productscount)+" products...",productscount)
100  cores = preferences["MULTICORE"]

102  iterator.initialize()
103  count = 0
104 
105  # process objects
106  for item in iterator:

108  ifcproduct = ifcfile.by_id(item.guid)

110  progressbar.next(True)
111  writeProgress(count,productscount,starttime)
112  count += 1
113

116  if annotations:
117  print("Processing",str(len(annotations)),"annotations...")
118  ifcscale = importIFCHelper.getScaling(ifcfile)

121

123  processRelationships()
124  storeColorDict()
125 
126  # finished
127  progressbar.stop()
128  FreeCAD.ActiveDocument.recompute()
129  endtime = round(time.time()-starttime,1)
130  fs = round(filesize,1)
131  ratio = int(endtime/filesize)
132  endtime = "%02d:%02d" % (divmod(endtime, 60))
133  writeProgress() # this cleans the line
134  print("Finished importing",fs,"Mb in",endtime,"s, or",ratio,"s/Mb")

139 
140  """write progress to console"""
