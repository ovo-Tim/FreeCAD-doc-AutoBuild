80 void FemPostBoxFunction::onChanged(const Property* prop)

82  if (prop == &Center || prop == &Length || prop == &Width || prop == &Height) {
83  const Base::Vector3d& vec = Center.getValue();
84  float l = Length.getValue();
85  float w = Width.getValue();

88  vec[0] + l / 2,
