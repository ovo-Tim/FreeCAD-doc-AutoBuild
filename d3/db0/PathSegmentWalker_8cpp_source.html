132 void PathSegmentWalker::walk(PathSegmentVisitor& cb, const Base::Vector3d& startPosition)
133 {
134  if (tp.getSize() == 0) {
135  return;
136  }
137 
138  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
139  "User parameter:BaseApp/Preferences/Mod/Part");
140  float deviation = hGrp->GetFloat("MeshDeviation", 0.2);
141 
142  Base::Vector3d rotCenter = tp.getCenter();
143  Base::Vector3d last(startPosition);
144  Base::Rotation lrot;
145  double A = 0.0;
146  double B = 0.0;
147  double C = 0.0;
148 
149  bool absolute = true;
150  bool absolutecenter = false;
151 
152  // for mapping the coordinates to XY plane
153  double Base::Vector3d::* pz = &Base::Vector3d::z;
154 
155  cb.setup(last);
156 
157  for (unsigned int i = 0; i < tp.getSize(); i++) {
158  std::deque<Base::Vector3d> points;
159 
160  const Path::Command& cmd = tp.getCommand(i);
161  const std::string& name = cmd.Name;
162  Base::Vector3d next = cmd.getPlacement().getPosition();
163  double a = A;
164  double b = B;
165  double c = C;
166 
167  if (!absolute) {
168  next = last + next;
169  }
170  if (!cmd.has("X")) {
171  next.x = last.x;
172  }
173  if (!cmd.has("Y")) {
174  next.y = last.y;
175  }
176  if (!cmd.has("Z")) {
177  next.z = last.z;
178  }
179  if (cmd.has("A")) {
180  a = cmd.getValue("A");
181  }
182  if (cmd.has("B")) {
183  b = cmd.getValue("B");
184  }
185  if (cmd.has("C")) {
186  c = cmd.getValue("C");
187  }
188 
189  Base::Rotation nrot = yawPitchRoll(a, b, c);
190 
191  Base::Vector3d rnext = compensateRotation(next, nrot, rotCenter);
192 
193  if ((name == "G0") || (name == "G00") || (name == "G1") || (name == "G01")) {
194  // straight line
195  if (nrot != lrot) {
196  double amax = std::max(fmod(fabs(a - A), 360),
197  std::max(fmod(fabs(b - B), 360), fmod(fabs(c - C), 360)));
198  double angle = amax / 180 * M_PI;
199  int segments = std::max(ARC_MIN_SEGMENTS, 3.0 / (deviation / angle));
200 
201  double da = (a - A) / segments;
202  double db = (b - B) / segments;
203  double dc = (c - C) / segments;
204 
205  Base::Vector3d dnext = (next - last) / segments;
206 
207  for (int j = 1; j < segments; j++) {
208  Base::Vector3d inter = last + dnext * j;
209 
210  Base::Rotation rot = yawPitchRoll(A + da * j, B + db * j, C + dc * j);
211  Base::Vector3d rinter = compensateRotation(inter, rot, rotCenter);
212 
213  points.push_back(rinter);
214  }
215  }
216 
217  if ("G0" == name || "G00" == name) {
218  cb.g0(i, last, rnext, points);
219  }
220  else {
221  cb.g1(i, last, rnext, points);
222  }
223 
224  last = next;
225  A = a;
226  B = b;
227  C = c;
228  lrot = nrot;
229  }
230  else if ((name == "G2") || (name == "G02") || (name == "G3") || (name == "G03")) {
231  // arc
232  Base::Vector3d norm;
233  Base::Vector3d center;
234 
235  if ((name == "G2") || (name == "G02")) {
236  norm.*pz = -1.0;
237  }
238  else {
239  norm.*pz = 1.0;
240  }
241 
242  if (absolutecenter) {
243  center = cmd.getCenter();
244  }
245  else {
246  center = (last + cmd.getCenter());
247  }
248  Base::Vector3d next0(next);
249  next0.*pz = 0.0;
250  Base::Vector3d last0(last);
251  last0.*pz = 0.0;
252  Base::Vector3d center0(center);
253  center0.*pz = 0.0;
254  // double radius = (last - center).Length();
255  double angle = (next0 - center0).GetAngle(last0 - center0);
256  // GetAngle will always return the minor angle. Switch if needed
257  Base::Vector3d anorm = (last0 - center0) % (next0 - center0);
258  if (anorm.*pz < 0) {
259  if (name == "G3" || name == "G03") {
260  angle = M_PI * 2 - angle;
261  }
262  }
263  else if (anorm.*pz > 0) {
264  if (name == "G2" || name == "G02") {
265  angle = M_PI * 2 - angle;
266  }
267  }
268  else if (angle == 0) {
269  angle = M_PI * 2;
270  }
271 
272  double amax = std::max(fmod(fabs(a - A), 360),
273  std::max(fmod(fabs(b - B), 360), fmod(fabs(c - C), 360)));
274 
275  int segments = std::max(
276  ARC_MIN_SEGMENTS,
277  3.0
278  / (deviation
279  / std::max(angle, amax))); // we use a rather simple rule here, provisorily
280  double dZ = (next.*pz - last.*pz) / segments; // How far each segment will helix in Z
281 
282  double dangle = angle / segments;
283  double da = (a - A) / segments;
284  double db = (b - B) / segments;
285  double dc = (c - C) / segments;
286 
287  for (int j = 1; j < segments; j++) {
288  Base::Vector3d inter;
289  Base::Rotation rot(norm, dangle * j);
290  rot.multVec((last0 - center0), inter);
291  inter.*pz = last.*pz + dZ * j; // Enable displaying helices
292 
293  Base::Rotation arot = yawPitchRoll(A + da * j, B + db * j, C + dc * j);
294  Base::Vector3d rinter = compensateRotation(center0 + inter, arot, rotCenter);
295 
296  points.push_back(rinter);
297  }
298 
299  cb.g23(i, last, rnext, points, center);
300 
301  last = next;
302  A = a;
303  B = b;
304  C = c;
305  lrot = nrot;
306  }
307  else if (name == "G90") {
308  // absolute mode
309  absolute = true;
310  }
311  else if (name == "G91") {
312  // relative mode
313  absolute = false;
314  }
315  else if (name == "G90.1") {
316  // absolute mode
317  absolutecenter = true;
318  }
319  else if (name == "G91.1") {
320  // relative mode
321  absolutecenter = false;
322  }
323  else if ((name == "G73") || (name == "G81") || (name == "G82") || (name == "G83")
324  || (name == "G84") || (name == "G85") || (name == "G86") || (name == "G89")) {
325  // drill,tap,bore
326  double r = 0;
327  if (cmd.has("R")) {
328  r = cmd.getValue("R");
329  }
330 
331  std::deque<Base::Vector3d> plist;
332  std::deque<Base::Vector3d> qlist;
333 
334  Base::Vector3d p1(next);
335  p1.*pz = last.*pz;
336 
337  if (nrot != lrot) {
338  double amax = std::max(fmod(fabs(a - A), 360),
339  std::max(fmod(fabs(b - B), 360), fmod(fabs(c - C), 360)));
340  double angle = amax / 180 * M_PI;
341  int segments = std::max(ARC_MIN_SEGMENTS, 3.0 / (deviation / angle));
342 
343  double da = (a - A) / segments;
344  double db = (b - B) / segments;
345  double dc = (c - C) / segments;
346 
347  Base::Vector3d dnext = (p1 - last) / segments;
348 
349  for (int j = 1; j < segments; j++) {
350  Base::Vector3d inter = last + dnext * j;
351 
352  Base::Rotation rot = yawPitchRoll(A + da * j, B + db * j, C + dc * j);
353  Base::Vector3d rinter = compensateRotation(inter, rot, rotCenter);
354 
355  points.push_back(rinter);
356  }
357  }
358 
359  Base::Vector3d p1r = compensateRotation(p1, nrot, rotCenter);
360  Base::Vector3d p2(next);
361  p2.*pz = r;
362  Base::Vector3d p2r = compensateRotation(p2, nrot, rotCenter);
363 
364  double q;
365  if (cmd.has("Q")) {
366  q = cmd.getValue("Q");
367  if (q > 0) {
368  Base::Vector3d temp(next);
369  for (temp.*pz = r; temp.*pz > next.*pz; temp.*pz -= q) {
370  Base::Vector3d pr = compensateRotation(temp, nrot, rotCenter);
371  qlist.push_back(pr);
372  }
373  }
374  }
375 
376  Base::Vector3d p3(next);
377  if (retract_mode == 99) { // G81,G83 need to account for G99 and retract to R only
378  p3.*pz = p2.*pz;
379  }
380  else {
381  p3.*pz = last.*pz;
382  }
383 
384 
385  Base::Vector3d p3r = compensateRotation(p3, nrot, rotCenter);
386 
387  plist.push_back(p1r);
388  plist.push_back(p2r);
389  plist.push_back(p3r);
390 
391  cb.g8x(i, last, next, points, plist, qlist);
392 
393  last = p3;
394  A = a;
395  B = b;
396  C = c;
397  lrot = nrot;
398  }
399  else if ((name == "G38.2") || (name == "G38.3") || (name == "G38.4") || (name == "G38.5")) {
400  // Straight probe
401  cb.g38(i, last, next);
402  last = next;
403  }
404  else if (name == "G17") {
405  pz = &Base::Vector3d::z;
406  }
407  else if (name == "G18") {
408  pz = &Base::Vector3d::y;
409  }
410  else if (name == "G19") {
411  pz = &Base::Vector3d::x;
412  }
413  else if (name == "G98") {
414  retract_mode = 98;
415  }
416  else if (name == "G99") {
417  retract_mode = 99;
418  }
419  }
420 }
