287 void Matrix::GetRotation(double& ax, double& ay, double& az) const
288 {
289  // return the rotations
290  if (m_unit) {
291  ax = ay = az = 0;
292  return;
293  }
294  double a; /* cos(bx) */
295  double b; /* sin(bx) */
296  double c; /* cos(by) */
297  double d; /* sin(by) */
298  double ee; /* cos(bz) */
299  double f; /* sin(bz) */
300  double sx, sy, sz;
301  GetScale(sx, sy, sz);
302  if (this->m_mirrored == -1) {
303  FAILURE(L"Don't know mirror - use IsMirrored method on object");
304  }
305  if (this->m_mirrored) {
306  sx = -sx;
307  }
308 
309  // solve for d and decide case and solve for a, b, c, e and f
310  d = -e[8] / sz;
311  if ((c = (1 - d) * (1 + d)) > 0.001) {
312  // case 1
313  c = sqrt(c);
314  a = e[10] / sz / c;
315  b = e[9] / sz / c;
316  ee = e[0] / sx / c;
317  f = e[4] / sy / c;
318  }
319  else {
320  // case 2
321  double coef;
322  double p, q;
323 
324  d = (d < 0) ? -1 : 1;
325  c = 0;
326  p = d * e[5] / sy - e[2] / sx;
327  q = d * e[6] / sy + e[1] / sx;
328  if ((coef = sqrt(p * p + q * q)) > 0.001) {
329  a = q / coef;
330  b = p / coef;
331  ee = b;
332  f = -d * b;
333  }
334  else {
335  /* dependent pairs */
336  a = e[5] / sy;
337  b = -e[6] / sy;
338  ee = 1;
339  f = 0;
340  }
341  }
342 
343  // solve and return ax, ay and az
344  ax = atan2(b, a);
345  ay = atan2(d, c);
346  az = atan2(f, ee);
347 }

349 Matrix Matrix::Inverse()
350 {
351  // matrix inversion routine
352 
353  // a is input matrix destroyed & replaced by inverse
354  // method used is gauss-jordan (ref ibm applications)
355 
356  double hold, biga;
357  int i, j, k, nk, kk, ij, iz;
358  int ki, ji, jp, jk, kj, jq, jr, ik;
359 
360  int n = 4; // 4 x 4 matrix only
361  Matrix a = *this;
362  int l[4], m[4];
363 
364  if (a.m_unit) { // unit matrix
365  return a;
366  }
367 
368  // search for largest element
369  nk = -n;
370  for (k = 0; k < n; k++) {
371  nk += n;
372  l[k] = m[k] = k;
373  kk = nk + k;
374  biga = a.e[kk];
375 
376  for (j = k; j < n; j++) {
377  iz = n * j;
378  for (i = k; i < n; i++) {
379  ij = iz + i;
380  if (fabs(biga) < fabs(a.e[ij])) {
381  biga = a.e[ij];
382  l[k] = i;
383  m[k] = j;
384  }
385  }
386  }
387 
388 
389  // interchange rows
390  j = l[k];
391  if (j > k) {
392  ki = k - n;
393 
394  for (i = 0; i < n; i++) {
395  ki += n;
396  hold = -a.e[ki];
397  ji = ki - k + j;
398  a.e[ki] = a.e[ji];
399  a.e[ji] = hold;
400  }
401  }
402 
403  // interchange columns
404  i = m[k];
405  if (i > k) {
406  jp = n * i;
407  for (j = 0; j < n; j++) {
408  jk = nk + j;
409  ji = jp + j;
410  hold = -a.e[jk];
411  a.e[jk] = a.e[ji];
412  a.e[ji] = hold;
413  }
414  }
415 
416  // divide columns by minus pivot (value of pivot element is contained in biga)
417  if (fabs(biga) < 1.0e-10) {
418  FAILURE(getMessage(L"Singular Matrix - Inversion failure")); // singular matrix
419  }
420 
421  for (i = 0; i < n; i++) {
422  if (i != k) {
423  ik = nk + i;
424  a.e[ik] = -a.e[ik] / biga;
425  }
426  }
427 
428  // reduce matrix
429  for (i = 0; i < n; i++) {
430  ik = nk + i;
431  hold = a.e[ik];
432  ij = i - n;
433 
434  for (j = 0; j < n; j++) {
435  ij = ij + n;
436  if (i != k && j != k) {
437  kj = ij - i + k;
438  a.e[ij] = hold * a.e[kj] + a.e[ij];
439  }
440  }
441  }
442 
443  // divide row by pivot
444  kj = k - n;
445  for (j = 0; j < n; j++) {
446  kj = kj + n;
447  if (j != k) {
448  a.e[kj] = a.e[kj] / biga;
449  }
450  }
451 
452  // replace pivot by reciprocal
453  a.e[kk] = 1 / biga;
454  }
455 
456  // final row and column interchange
457  k = n - 1;
458 
459  while (k > 0) {
460  i = l[--k];
461  if (i > k) {
462  jq = n * k;
463  jr = n * i;
464 
465  for (j = 0; j < n; j++) {
466  jk = jq + j;
467  hold = a.e[jk];
468  ji = jr + j;
469  a.e[jk] = -a.e[ji];
470  a.e[ji] = hold;
471  }
472  }
473 
474  j = m[k];
475  if (j > k) {
476  ki = k - n;
477 
478  for (i = 1; i <= n; i++) {
479  ki = ki + n;
480  hold = a.e[ki];
481  ji = ki - k + j;
482  a.e[ki] = -a.e[ji];
483  a.e[ji] = hold;
484  }
485  }
486  }
487 
488  return a;
489 }
