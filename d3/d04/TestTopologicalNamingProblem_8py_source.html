44  def countFacesEdgesVertexes(self, map):
45  """Helper to return a tuple of the counts of Faces, Edges, and Vertexes in a map for
46  easier test writing"""
47  faces = [name for name in map.keys() if name.startswith("Face")]
48  edges = [name for name in map.keys() if name.startswith("Edge")]
49  vertexes = [name for name in map.keys() if name.startswith("Vertex")]
50  return (len(faces), len(edges), len(vertexes))

52  def testPadsOnBaseObject(self):
53  """Simple TNP test case
54  By creating three Pads dependent on each other in succession, and then moving the
55  middle one we can determine if the last one breaks because of a broken reference
56  to the middle one. This is the essence of a TNP. Pretty much a duplicate of the
57  steps at https://wiki.freecad.org/Topological_naming_problem"""
58 
59  # Arrange
60  self.Body = self.Doc.addObject("PartDesign::Body", "Body")
61  # Make first offset cube Pad
62  self.PadSketch = self.Doc.addObject("Sketcher::SketchObject", "SketchPad")
63  self.Body.addObject(self.PadSketch)
64  TestSketcherApp.CreateRectangleSketch(self.PadSketch, (0, 0), (1, 1))
65  self.Doc.recompute()
66  self.Pad = self.Doc.addObject("PartDesign::Pad", "Pad")
67  self.Body.addObject(self.Pad)
68  self.Pad.Profile = self.PadSketch
69  self.Pad.Length = 1
70  self.Doc.recompute()
71 
72  # Attach a second pad to the top of the first.
73  self.PadSketch1 = self.Doc.addObject("Sketcher::SketchObject", "SketchPad1")
74  self.Body.addObject(self.PadSketch1)
75  self.PadSketch1.MapMode = "FlatFace"
76  self.PadSketch1.AttachmentSupport = [(self.Doc.getObject("Pad"), "Face6")]
77  TestSketcherApp.CreateRectangleSketch(self.PadSketch1, (0, 0), (1, 1))
78  self.Doc.recompute()
79  self.Pad1 = self.Doc.addObject("PartDesign::Pad", "Pad1")
80  self.Body.addObject(self.Pad1)
81  self.Pad1.Profile = self.PadSketch1
82  self.Pad1.Length = 1
83  self.Doc.recompute()
84 
85  # Attach a third pad to the top of the second.
86  self.PadSketch2 = self.Doc.addObject("Sketcher::SketchObject", "SketchPad2")
87  self.Body.addObject(self.PadSketch2)
88  self.PadSketch2.MapMode = "FlatFace"
89  self.PadSketch2.AttachmentSupport = [(self.Doc.getObject("Pad1"), "Face6")]
90  TestSketcherApp.CreateRectangleSketch(self.PadSketch2, (0, 0), (1, 1))
91  self.Doc.recompute()
92  self.Pad2 = self.Doc.addObject("PartDesign::Pad", "Pad2")
93  self.Body.addObject(self.Pad2)
94  self.Pad2.Profile = self.PadSketch2
95  self.Pad2.Length = 1
96  self.Doc.recompute()
97 
98  # Assert everything is valid
99  self.assertTrue(self.Pad.isValid())
100  self.assertTrue(self.Pad1.isValid())
101  self.assertTrue(self.Pad2.isValid())
102 
103  # Act
104  # Move the second pad ( the sketch attachment point )
105  self.PadSketch1.AttachmentOffset = App.Placement(
106  App.Vector(0.5000000000, 0.0000000000, 0.0000000000),
107  App.Rotation(0.0000000000, 0.0000000000, 0.0000000000),
108  )
109  self.Doc.recompute()
110 
111  # Assert everything is still valid.
112  self.assertTrue(self.Pad.isValid())
113  self.assertTrue(self.Pad1.isValid())
114 
115  if self.Body.Shape.ElementMapVersion == "": # Should be '4' as of Mar 2023.
116  self.assertFalse(
117  self.Pad2.isValid()
118  ) # TNP problem is present without ElementMaps
119  else:
120  self.assertTrue(
121  self.Pad2.isValid()
122  ) # TNP problem is not present with ElementMaps
