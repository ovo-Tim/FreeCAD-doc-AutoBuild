347 {
348  std::vector<QGIView*> items = getViews();
349  QString qsName = QString::fromUtf8(name);
350  bool found = false;
351  QGIView* ourItem = nullptr;
352  for (auto& i : items) {
353  if (qsName == i->data(1).toString()) {//is there a QGIV with this name in scene?
354  found = true;
355  ourItem = i;
356  break;
357  }
358  }
359 
360  if (found) {

364  balloon->disconnect();
365  }
366  removeQViewFromScene(ourItem);
367  delete ourItem;//commenting this prevents crash but means a small memory waste.

390 bool QGSPage::attachView(App::DocumentObject* obj)
391 {
392  // Base::Console().Message("QGSP::attachView(%s)\n", obj->getNameInDocument());
393  QGIView* existing = findQViewForDocObj(obj);
394  if (existing)

398 
399  QGIView* qview(nullptr);
400 
401  if (typeId.isDerivedFrom(TechDraw::DrawViewSection::getClassTypeId())) {
402  qview = addViewSection(static_cast<TechDraw::DrawViewSection*>(obj));
403  }
404  else if (typeId.isDerivedFrom(TechDraw::DrawViewPart::getClassTypeId())) {
405  qview = addViewPart(static_cast<TechDraw::DrawViewPart*>(obj));
406  }
407  else if (typeId.isDerivedFrom(TechDraw::DrawProjGroup::getClassTypeId())) {
408  qview = addProjectionGroup(static_cast<TechDraw::DrawProjGroup*>(obj));
409  }
410  else if (typeId.isDerivedFrom(TechDraw::DrawViewCollection::getClassTypeId())) {
411  qview = addDrawView(static_cast<TechDraw::DrawViewCollection*>(obj));
412  }
413  else if (typeId.isDerivedFrom(TechDraw::DrawViewDimension::getClassTypeId())) {
414  qview = addViewDimension(static_cast<TechDraw::DrawViewDimension*>(obj));
415  }
416  else if (typeId.isDerivedFrom(TechDraw::DrawViewBalloon::getClassTypeId())) {
417  qview = addViewBalloon(static_cast<TechDraw::DrawViewBalloon*>(obj));
418  }
419  else if (typeId.isDerivedFrom(TechDraw::DrawViewAnnotation::getClassTypeId())) {
420  qview = addAnnotation(static_cast<TechDraw::DrawViewAnnotation*>(obj));
421  }
422  else if (typeId.isDerivedFrom(TechDraw::DrawViewSymbol::getClassTypeId())) {
423  qview = addDrawViewSymbol(static_cast<TechDraw::DrawViewSymbol*>(obj));
424  }
425  else if (typeId.isDerivedFrom(TechDraw::DrawViewClip::getClassTypeId())) {
426  qview = addDrawViewClip(static_cast<TechDraw::DrawViewClip*>(obj));
427  }
428  else if (typeId.isDerivedFrom(TechDraw::DrawViewSpreadsheet::getClassTypeId())) {
429  qview = addDrawViewSpreadsheet(static_cast<TechDraw::DrawViewSpreadsheet*>(obj));
430  }
431  else if (typeId.isDerivedFrom(TechDraw::DrawViewImage::getClassTypeId())) {
432  qview = addDrawViewImage(static_cast<TechDraw::DrawViewImage*>(obj));
433  }
434  else if (typeId.isDerivedFrom(TechDraw::DrawLeaderLine::getClassTypeId())) {
435  qview = addViewLeader(static_cast<TechDraw::DrawLeaderLine*>(obj));

448  return (qview != nullptr);

466 void QGSPage::addItemToParent(QGIView* item, QGIView* parent)
467 {
468  // not every view uses the same remapping?? spreadsheets, image, RTA, anno?
469  // anything that was originally designed to have its position
470  // defined relative to the Page should not use the dimension/balloon mapping.
471  assert(item);
472  assert(parent);
473  // TODO: make custom user types retrievable by name (see QGIUserTypes.h)
474  constexpr int QGIVDimensionType {QGraphicsItem::UserType + 106};
475  constexpr int QGIVBalloonType {QGraphicsItem::UserType + 140};
476  constexpr int QGIWeldSymbolType {QGraphicsItem::UserType + 340};
477  // constexpr int QGIViewAnnotationType {QGraphicsItem::UserType + 120};
478 
479  if (item->type() == QGIWeldSymbolType) {
480  // don't touch these
481  return;
482  }
483 
484  // original parenting logic here
485  QPointF posRef(0., 0.);
486  if (item->type() == QGIVDimensionType || item->type() == QGIVBalloonType) {
487  QPointF mapPos = item->mapToItem(parent, posRef);
488  item->moveBy(-mapPos.x(), -mapPos.y());
489  parent->addToGroup(item);
490  return;
491  }
492 
493  // positioning logic for objects (leader/rta/etc) that normally draw relative to the page goes
494  // here
495  //
496  QPointF itemPosition {item->getViewObject()->X.getValue(), // millimetres on page
497  -item->getViewObject()->Y.getValue()};
498  parent->addToGroup(item);
499  item->setPos(Rez::guiX(itemPosition));
500 
501  item->setZValue(ZVALUE::DIMENSION);
502 }

922 void QGSPage::fixOrphans(bool force)
923 {

925

927  // if we ever have collections of collections, we'll need to revisit this
928  TechDraw::DrawPage* thisPage = m_vpPage->getDrawPage();
929 
930  if (!thisPage->isAttachedToDocument())
931  return;
932 
933  std::vector<App::DocumentObject*> pChildren = thisPage->getAllViews();
934 
935  // if dv doesn't have a graphic, make one
936  for (auto& dv : pChildren) {
937  if (dv->isRemoving())
938  continue;
939  QGIView* qv = findQViewForDocObj(dv);
940  if (!qv)
941  attachView(dv);
942  }
943  // if qView doesn't have a Feature on this Page, delete it
944  std::vector<QGIView*> qvss = getViews();
945  // qvss may contain an item and its child item(s) and to avoid to access a deleted item a QPointer is needed
946  std::vector<QPointer<QGIView>> qvs;
947  std::for_each(qvss.begin(), qvss.end(), [&qvs](QGIView* v) { qvs.emplace_back(v); });
948  App::Document* doc = m_vpPage->getDrawPage()->getDocument();
949  for (auto& qv : qvs) {
950  if (!qv)
951  continue;// already deleted?
952 
953  App::DocumentObject* obj = doc->getObject(qv->getViewName());
954  if (!obj) {
955  //no DrawView anywhere in Document
956  removeQView(qv);
957  }
958  else {
959  //DrawView exists in Document. Does it belong to this DrawPage?
960  TechDraw::DrawView* viewObj = qv->getViewObject();
961  int numParentPages = viewObj->countParentPages();
962  if (numParentPages == 0) {
963  //DrawView does not belong to any DrawPage
964  //remove QGItem from QGScene
965  removeQView(qv);
966  }
967  else if (numParentPages == 1) {
968  //Does DrawView belong to this DrawPage?
969  TechDraw::DrawPage* parentPage = viewObj->findParentPage();
970  if (thisPage != parentPage) {
971  // The view could be a child of a link (ea Dimension).
972  TechDraw::DrawView* parentView = viewObj->claimParent();
973  for (auto* v : thisPage->getViews()) {
974  if (v == parentView) {
975  continue;
976  }
977  }
978 
979  //DrawView does not belong to this DrawPage
980  //remove QGItem from QGScene
981  removeQView(qv);
982  }
983  }
984  else if (numParentPages > 1) {
985  //DrawView belongs to multiple DrawPages
986  //check if this MDIViewPage corresponds to any parent DrawPage
987  //if not, delete the QGItem
988  std::vector<TechDraw::DrawPage*> potentialParentPages =
989  qv->getViewObject()->findAllParentPages();
990  bool found = false;
991  for (auto p : potentialParentPages) {
992  if (thisPage == p) {
993  found = true;
994  break;
995  }
996  }
997  if (!found) {
998  //none of the parent Pages for View correspond to this Page
999  removeQView(qv);
1000  }
1001  }
1002  }
1003  }
