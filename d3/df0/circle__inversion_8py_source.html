47 def pointInversion(circle, point):
48  """Return the circle inversion of a point.
49 
50  It will return `None` if the given point is equal to the center
51  of the circle.
52  """
53  if geomType(circle) != "Circle" or isinstance(point, App.Vector):
54  print("debug: pointInversion bad parameters!")
55  return None
56 
57  cen = circle.Curve.Center
58  rad = circle.Curve.Radius
59 
60  if DraftVecUtils.equals(cen, point):
61  return None
62 
63  # Inverse the distance of the point
64  # dist(cen -> P) = r^2 / dist(cen -> invP)
65 
66  dist = DraftVecUtils.dist(point, cen)
67  invDist = rad**2 / dist
68 
69  invPoint = App.Vector(0, 0, point.z)
70  invPoint.x = cen.x + (point.x - cen.x) * invDist / dist
71  invPoint.y = cen.y + (point.y - cen.y) * invDist / dist
72 
73  return invPoint

76 def polarInversion(circle, edge):
77  """Return the inversion pole of a line. The edge is the polar.
78 
79  The nearest point on the line is inversed.
80 
81  http://mathworld.wolfram.com/InversionPole.html
82  """
83  if geomType(circle) != "Circle" or geomType(edge) != "Line":
84  print("debug: circleInversionPole bad parameters! Must be a circle.")
85  return None
86 
87  nearest = circle.Curve.Center.add(findDistance(circle.Curve.Center,
88  edge,
89  False))
90  if nearest:
91  inversionPole = pointInversion(circle, nearest)
92  if inversionPole:
93  return inversionPole
94 
95  return None

98 def circleInversion(circle, circle2):
99  """Circle inversion of a circle, inverting the center point.
100 
101  Returns the new circle created from the inverted center of circle2.
102  """
103  if geomType(circle) != "Circle" or geomType(circle2) != "Circle":
104  print("debug: circleInversion bad parameters! Must be circles.")
105  return None
106 
107  cen1 = circle.Curve.Center
108 
109  cen2 = circle2.Curve.Center
110  rad2 = circle2.Curve.Radius
111 
112  if DraftVecUtils.equals(cen1, cen2):
113  return None
114 
115  invCen2 = pointInversion(circle, cen2)
116 
117  pointOnCircle2 = App.Vector.add(cen2, App.Vector(rad2, 0, 0))
118  invPointOnCircle2 = pointInversion(circle, pointOnCircle2)
119 
120  return Part.Circle(invCen2,
121  NORM,
122  DraftVecUtils.dist(invCen2, invPointOnCircle2))
