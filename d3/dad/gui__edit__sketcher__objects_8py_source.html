46  def get_edit_points(self, obj):
47  """Return the list of edipoints for the given single line sketch.
48  (WallTrace)
49  0 : startpoint
50  1 : endpoint
51  """
52  import Part
53  editpoints = []
54  if (obj.ConstraintCount == 0
55  and obj.GeometryCount == 1
56  and type(obj.Geometry[0]) == Part.LineSegment):
57  editpoints.append(obj.getPoint(0,1))
58  editpoints.append(obj.getPoint(0,2))
59  return editpoints
60  else:
61  _wrn = translate("draft", "Sketch is too complex to edit: "
62  "it is suggested to use the default Sketcher editor")
63  App.Console.PrintWarning(_wrn + "\n")
64  return None
