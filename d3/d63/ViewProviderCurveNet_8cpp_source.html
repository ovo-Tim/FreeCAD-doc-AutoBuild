153 bool ViewProviderCurveNet::handleEvent(const SoEvent * const ev, Gui::View3DInventorViewer &Viewer)
154 {
155  SbVec3f point, norm;
156 
157  // get the position of the mouse
158  const SbVec2s pos(ev->getPosition());
159 
160  // Keyboard events
161  if (ev->getTypeId().isDerivedFrom(SoKeyboardEvent::getClassTypeId())) {
162  const SoKeyboardEvent * ke = static_cast<const SoKeyboardEvent *>(ev);
163  switch (ke->getKey()) {
164  case SoKeyboardEvent::LEFT_ALT:
165  case SoKeyboardEvent::RIGHT_ALT:
166  case SoKeyboardEvent::LEFT_CONTROL:
167  case SoKeyboardEvent::RIGHT_CONTROL:
168  case SoKeyboardEvent::LEFT_SHIFT:
169  case SoKeyboardEvent::RIGHT_SHIFT:
170  break;
171  default:
172  break;
173  }
174  }
175 
176  // switching the mouse buttons
177  if (ev->getTypeId().isDerivedFrom(SoMouseButtonEvent::getClassTypeId())) {
178  const SoMouseButtonEvent * const event = static_cast<const SoMouseButtonEvent *>(ev);
179  const int button = event->getButton();
180  const SbBool press = event->getState() == SoButtonEvent::DOWN ? true : false;
181 
182  // which button pressed?
183  switch (button) {
184  case SoMouseButtonEvent::BUTTON1:
185  if (press) {
186  Base::Console().Log("ViewProviderCurveNet::handleEvent() press left\n");
187 
188  bool bIsNode = false;
189  for (const auto & It : NodeList)
190  {
191  if (It.pcHighlight->isHighlighted())
192  {
193  bIsNode = true;
194  PointToMove = It;
195  break;
196  }
197  }
198 
199  if (bIsNode)
200  {
201  // set the provider in point move mode and remember the point
202  bMovePointMode = true;
203  // PointToMove = *It; ### Error 'It' is out of scope->move inside the loop
204 
205  return true;
206  }
207  else if(Viewer.pickPoint(pos,point,norm))
208  {
209  Node n;
210  Base::Console().Log("Picked(%f,%f,%f)\n",point[0],point[1],point[2]);
211 
212  SoSeparator *TransRoot = new SoSeparator();
213  n.pcTransform = new SoTransform();
214  TransRoot->addChild(n.pcTransform);
215  n.pcTransform->translation.setValue(point);
216  n.pcHighlight = new Gui::SoFCSelection();
217  SoSphere * sphere = new SoSphere;
218  sphere->radius = (float)pcLineStyle->pointSize.getValue();
219  n.pcHighlight->addChild(sphere);
220  TransRoot->addChild(n.pcHighlight);
221  VertexRoot->addChild(TransRoot);
222 
223  NodeList.push_back(n);
224 
225  return true;
226  }
227  }
228  else // if(pressd)..
229  {
230  if (bMovePointMode)
231  {
232  bMovePointMode = false;
233  return true;
234  }
235  }
236  break;
237  }
238  }
239 
240  // Mouse Movement handling
241  if (ev->getTypeId().isDerivedFrom(SoLocation2Event::getClassTypeId())) {
242  // const SoLocation2Event * const event = (const SoLocation2Event *) ev;
243 
244  if (bMovePointMode && Viewer.pickPoint(pos,point,norm) ){
245  PointToMove.pcTransform->translation.setValue(point);
246  return true;
247  }
248  }
249 
250  // event not processed
251  return false;
252 }

254 Standard_Boolean ViewProviderCurveNet::computeEdges(SoSeparator* root, const TopoDS_Shape &myShape)
255 {
256  unsigned long ulNbOfPoints = 50;
257 
258  TopExp_Explorer ex;
259  SoSeparator *EdgeRoot = new SoSeparator();
260  root->addChild(EdgeRoot);
261 
262  EdgeRoot->addChild(pcLineStyle);
263  EdgeRoot->addChild(pcLineMaterial);
264 
265  for (ex.Init(myShape, TopAbs_EDGE); ex.More(); ex.Next())
266  {
267  // get the shape and mesh it
268  const TopoDS_Edge& aEdge = TopoDS::Edge(ex.Current());
269 
270  Standard_Real fBegin, fEnd;
271  SbVec3f* vertices = new SbVec3f[ulNbOfPoints];
272 
273  Handle(Geom_Curve) hCurve = BRep_Tool::Curve(aEdge,fBegin,fEnd);
274  float fLen = float(fEnd - fBegin);
275 
276  for (unsigned long i = 0; i < ulNbOfPoints; i++)
277  {
278  gp_Pnt gpPt = hCurve->Value(fBegin + (fLen * float(i)) / float(ulNbOfPoints-1));
279  vertices[i].setValue((float)(gpPt.X()),(float)(gpPt.Y()),(float)(gpPt.Z()));
280  }
281 
282  // define vertices
283  SoCoordinate3 * coords = new SoCoordinate3;
284  coords->point.setValues(0,ulNbOfPoints, vertices);
285  EdgeRoot->addChild(coords);
286 
287  // define the indexed face set
288  SoLocateHighlight* h = new SoLocateHighlight();
289  h->color.setValue((float)0.2,(float)0.5,(float)0.2);
290 
291  SoLineSet * lineset = new SoLineSet;
292  h->addChild(lineset);
293  EdgeRoot->addChild(h);
294  }
295 
296  return true;
297 }

299 Standard_Boolean ViewProviderCurveNet::computeVertices(SoSeparator* root, const TopoDS_Shape &myShape)
300 {
301  TopExp_Explorer ex;
302  SoSeparator *VertexRoot = new SoSeparator();
303  root->addChild(VertexRoot);
304 
305  VertexRoot->addChild(pcPointMaterial);
306 
307  SoComplexity *copl = new SoComplexity();
308  copl->value = (float)0.2;
309  VertexRoot->addChild(copl);
310 
311  for (ex.Init(myShape, TopAbs_VERTEX); ex.More(); ex.Next())
312  {
313  // get the shape
314  const TopoDS_Vertex& aVertex = TopoDS::Vertex(ex.Current());
315  gp_Pnt gpPt = BRep_Tool::Pnt (aVertex);
316 
317  SoSeparator *TransRoot = new SoSeparator();
318  SoTransform *Trans = new SoTransform();
319  TransRoot->addChild(Trans);
320  Trans->translation.setValue((float)(gpPt.X()),(float)(gpPt.Y()),(float)(gpPt.Z()));
321 
322  SoLocateHighlight* h = new SoLocateHighlight();
323  h->color.setValue((float)0.2,(float)0.5,(float)0.2);
324 
325  SoSphere * sphere = new SoSphere;
326  sphere->radius = (float)pcPointStyle->pointSize.getValue();
327 
328  h->addChild(sphere);
329  TransRoot->addChild(h);
330  VertexRoot->addChild(TransRoot);
331  }
332 
333  return true;
