108 ImageEdit::ImageEdit(const QString& propertyName,
109  const std::shared_ptr<Materials::Material>& material,
110  QWidget* parent)
111  : QDialog(parent)
112  , ui(new Ui_ImageEdit)
113  , _material(material)
114  , _pixmap(QString::fromStdString(":/images/default_image.png"))
115 {
116  ui->setupUi(this);
117 
118  if (material->hasPhysicalProperty(propertyName)) {
119  _property = material->getPhysicalProperty(propertyName);
120  }
121  else if (material->hasAppearanceProperty(propertyName)) {
122  _property = material->getAppearanceProperty(propertyName);
123  }
124  else {
125  Base::Console().Log("Property '%s' not found\n", propertyName.toStdString().c_str());
126  _property = nullptr;
127  }
128  if (_property) {
129  if (_property->getType() == Materials::MaterialValue::SVG) {
130  _svg = _property->getString();
131  showSVG();
132  }
133  else {
134  QString value = _property->getString();
135  if (!value.isEmpty()) {
136  QByteArray by = QByteArray::fromBase64(value.toUtf8());
137  QImage img = QImage::fromData(by, "PNG");
138  _pixmap = QPixmap::fromImage(img);
139  }
140  showPixmap();
141  }
142  }
143  else {
144  Base::Console().Log("No value loaded\n");
145  showPixmap();
146  }
147 
148  connect(ui->buttonFileSelect, &QPushButton::clicked, this, &ImageEdit::onFileSelect);
149 
150  connect(ui->standardButtons, &QDialogButtonBox::accepted, this, &ImageEdit::accept);
151  connect(ui->standardButtons, &QDialogButtonBox::rejected, this, &ImageEdit::reject);
152 }

216 void ImageEdit::onFileSelectSVG()
217 {
218  QString fn = selectFile(tr("Image files (*.svg);;All files (*)"));
219  if (!fn.isEmpty()) {
220  fn = QDir::fromNativeSeparators(fn);
221 
222  _pixmap = QPixmap();
223  QFile file(fn);
224  if (!file.open(QIODevice::ReadOnly | QIODevice::Text)) {
225  _svg.clear();
226  }
227  else {
228  QTextStream stream(&file);
229  _svg = stream.readAll();
230  }
231  showSVG();
232  }
233 }
