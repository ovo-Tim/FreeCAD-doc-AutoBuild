110 void Filling::addConstraints(BRepFill_Filling& builder,
111  const App::PropertyLinkSubList& edges,
112  const App::PropertyStringList& faces,
113  const App::PropertyIntegerList& orders,
114  Standard_Boolean bnd)
115 {
116  auto edge_obj = edges.getValues();
117  auto edge_sub = edges.getSubValues();
118  auto face_sub = faces.getValues();
119  auto contvals = orders.getValues();
120 
121  // if the number of continuities doesn't match then fall back to C0
122  if (edge_sub.size() != contvals.size()) {
123  contvals.resize(edge_sub.size());
124  std::fill(contvals.begin(), contvals.end(), static_cast<long>(GeomAbs_C0));
125  }
126 
127  // if the number of faces doesn't match then fall back to empty strings
128  // an empty face string indicates that there is no face associated to an edge
129  if (face_sub.size() != edge_sub.size()) {
130  face_sub.resize(edge_obj.size());
131  std::fill(face_sub.begin(), face_sub.end(), std::string());
132  }
133 
134  if (edge_obj.size() == edge_sub.size()) {
135  // BRepFill_Filling crashes if the boundary edges are not added in a consecutive order.
136  // these edges are first added to a test wire to check that they can be securely added to
137  // the Filling algo
138  BRepBuilderAPI_MakeWire testWire;
139  for (std::size_t index = 0; index < edge_obj.size(); index++) {
140  // get the part object
141  App::DocumentObject* obj = edge_obj[index];
142  const std::string& sub = edge_sub[index];
143 
144  if (obj && obj->isDerivedFrom<Part::Feature>()) {
145  // get the sub-edge of the part's shape
146  const Part::TopoShape& shape = static_cast<Part::Feature*>(obj)->Shape.getShape();
147  TopoDS_Shape edge = shape.getSubShape(sub.c_str());
148  if (!edge.IsNull() && edge.ShapeType() == TopAbs_EDGE) {
149  GeomAbs_Shape cont = static_cast<GeomAbs_Shape>(contvals[index]);
150 
151  // check for an adjacent face of the edge
152  std::string subFace = face_sub[index];
153 
154  // edge doesn't have set an adjacent face
155  if (subFace.empty()) {
156  if (!bnd) {
157  // not a boundary edge: safe to add it directly
158  builder.Add(TopoDS::Edge(edge), cont, bnd);
159  }
160  else {
161  // boundary edge: try to add it to the test wire first
162  testWire.Add(TopoDS::Edge(edge));
163  if (testWire.IsDone()) {
164  builder.Add(TopoDS::Edge(edge), cont, bnd);
165  }
166  else {
167  Standard_Failure::Raise(
168  "Boundary edges must be added in a consecutive order");
169  }
170  }
171  }
172  else {
173  TopoDS_Shape face = shape.getSubShape(subFace.c_str());
174  if (!face.IsNull() && face.ShapeType() == TopAbs_FACE) {
175  if (!bnd) {
176  // not a boundary edge: safe to add it directly
177  builder.Add(TopoDS::Edge(edge), TopoDS::Face(face), cont, bnd);
178  }
179  else {
180  // boundary edge: try to add it to the test wire first
181  testWire.Add(TopoDS::Edge(edge));
182  if (testWire.IsDone()) {
183  builder.Add(TopoDS::Edge(edge), TopoDS::Face(face), cont, bnd);
184  }
185  else {
186  Standard_Failure::Raise(
187  "Boundary edges must be added in a consecutive order");
188  }
189  }
190  }
191  else {
192  Standard_Failure::Raise("Sub-shape is not a face");
193  }
194  }
195  }
196  else {
197  Standard_Failure::Raise("Sub-shape is not an edge");
198  }
199  }
200  }
201  }
202  else {
203  Standard_Failure::Raise("Number of links doesn't match with number of orders");
204  }
205 }

207 // Add free support faces with their continuities
208 void Filling::addConstraints(BRepFill_Filling& builder,
209  const App::PropertyLinkSubList& faces,
210  const App::PropertyIntegerList& orders)
211 {
212  auto face_obj = faces.getValues();
213  auto face_sub = faces.getSubValues();
214  auto contvals = orders.getValues();
215 
216  if (face_obj.size() == face_sub.size() && face_obj.size() == contvals.size()) {
217  for (std::size_t index = 0; index < face_obj.size(); index++) {
218  App::DocumentObject* obj = face_obj[index];
219  const std::string& sub = face_sub[index];
220  if (obj && obj->isDerivedFrom<Part::Feature>()) {
221  const Part::TopoShape& shape = static_cast<Part::Feature*>(obj)->Shape.getShape();
222  TopoDS_Shape face = shape.getSubShape(sub.c_str());
223  if (!face.IsNull() && face.ShapeType() == TopAbs_FACE) {
224  GeomAbs_Shape cont = static_cast<GeomAbs_Shape>(contvals[index]);
225  builder.Add(TopoDS::Face(face), cont);
226  }
227  else {
228  Standard_Failure::Raise("Sub-shape is not a face");
229  }
230  }
231  }
232  }
233  else {
234  Standard_Failure::Raise("Number of links doesn't match with number of orders");
235  }

238 void Filling::addConstraints(BRepFill_Filling& builder, const App::PropertyLinkSubList& pointsList)
239 {
240  auto points = pointsList.getSubListValues();
241  for (const auto& it : points) {
242  App::DocumentObject* obj = it.first;
243  std::vector<std::string> sub = it.second;
244  if (obj && obj->isDerivedFrom<Part::Feature>()) {
245  const Part::TopoShape& shape = static_cast<Part::Feature*>(obj)->Shape.getShape();
246  for (const auto& jt : sub) {
247  TopoDS_Shape subShape = shape.getSubShape(jt.c_str());
248  if (!subShape.IsNull() && subShape.ShapeType() == TopAbs_VERTEX) {
249  gp_Pnt pnt = BRep_Tool::Pnt(TopoDS::Vertex(subShape));
250  builder.Add(pnt);
251  }
252  }
253  }
254  }

257 App::DocumentObjectExecReturn* Filling::execute()
258 {
259  // Assign Variables
260  unsigned int degree = Degree.getValue();
261  unsigned int ptsoncurve = PointsOnCurve.getValue();
262  unsigned int numIter = Iterations.getValue();
263  bool anisotropy = Anisotropy.getValue();
264  double tol2d = Tolerance2d.getValue();
265  double tol3d = Tolerance3d.getValue();
266  double tolG1 = TolAngular.getValue();
267  double tolG2 = TolCurvature.getValue();
268  unsigned int maxdeg = MaximumDegree.getValue();
269  unsigned int maxseg = MaximumSegments.getValue();
270 
271  try {
272  BRepFill_Filling builder(degree,
273  ptsoncurve,
274  numIter,
275  anisotropy,
276  tol2d,
277  tol3d,
278  tolG1,
279  tolG2,
280  maxdeg,
281  maxseg);
282 
283  if ((BoundaryEdges.getSize()) < 1) {
284  return new App::DocumentObjectExecReturn(
285  "Border must have at least one curve defined.");
286  }
287 
288  // Load the initial surface if set
289  App::DocumentObject* initFace = InitialFace.getValue();
290  if (initFace && initFace->isDerivedFrom<Part::Feature>()) {
291  const Part::TopoShape& shape = static_cast<Part::Feature*>(initFace)->Shape.getShape();
292  std::vector<std::string> subNames = InitialFace.getSubValues();
293  for (const auto& it : subNames) {
294  TopoDS_Shape subShape = shape.getSubShape(it.c_str());
295  if (!subShape.IsNull() && subShape.ShapeType() == TopAbs_FACE) {
296  builder.LoadInitSurface(TopoDS::Face(subShape));
297  break;
298  }
299  }
300  }
301 
302  // Add the constraints of border curves/faces (bound)
303  int numBoundaries = BoundaryEdges.getSize();
304  addConstraints(builder, BoundaryEdges, BoundaryFaces, BoundaryOrder, Standard_True);
305 
306  // Add additional edge constraints if available (unbound)
307  if (UnboundEdges.getSize() > 0) {
308  addConstraints(builder, UnboundEdges, UnboundFaces, UnboundOrder, Standard_False);
309  }
310 
311  // Add additional constraint on free faces
312  if (FreeFaces.getSize() > 0) {
313  addConstraints(builder, FreeFaces, FreeOrder);
314  }
315 
316  // App point constraints
317  if (Points.getSize() > 0) {
318  addConstraints(builder, Points);
319  }
320 
321  // Build the face
322  if (numBoundaries > 1) {
323  builder.Build();
324  }
325  if (!builder.IsDone()) {
326  Standard_Failure::Raise("Failed to create a face from constraints");
327  }
328 
329  // Return the face
330  TopoDS_Face aFace = builder.Face();
331  this->Shape.setValue(aFace);
332  return App::DocumentObject::StdReturn;
333  }
334  catch (Standard_Failure& e) {
335  return new App::DocumentObjectExecReturn(e.GetMessageString());
336  }
337 }
