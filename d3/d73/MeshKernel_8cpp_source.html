134 void MeshKernel::AddFacet(const MeshGeomFacet& rclSFacet)
135 {
136  MeshFacet clFacet;
137 
138  // set corner points
139  for (int i = 0; i < 3; i++) {
140  _clBoundBox.Add(rclSFacet._aclPoints[i]);
141  clFacet._aulPoints[i] = _aclPointArray.GetOrAddIndex(rclSFacet._aclPoints[i]);
142  }
143 
144  // adjust orientation to normal
145  AdjustNormal(clFacet, rclSFacet.GetNormal());
146 
147  FacetIndex ulCt = _aclFacetArray.size();
148 
149  // set neighbourhood
150  PointIndex ulP0 = clFacet._aulPoints[0];
151  PointIndex ulP1 = clFacet._aulPoints[1];
152  PointIndex ulP2 = clFacet._aulPoints[2];
153  FacetIndex ulCC = 0;
154  for (auto pF = _aclFacetArray.begin(); pF != _aclFacetArray.end(); ++pF, ulCC++) {
155  for (int i = 0; i < 3; i++) {
156  PointIndex ulP = pF->_aulPoints[i];
157  PointIndex ulQ = pF->_aulPoints[(i + 1) % 3];
158  if (ulQ == ulP0 && ulP == ulP1) {
159  clFacet._aulNeighbours[0] = ulCC;
160  pF->_aulNeighbours[i] = ulCt;
161  }
162  else if (ulQ == ulP1 && ulP == ulP2) {
163  clFacet._aulNeighbours[1] = ulCC;
164  pF->_aulNeighbours[i] = ulCt;
165  }
166  else if (ulQ == ulP2 && ulP == ulP0) {
167  clFacet._aulNeighbours[2] = ulCC;
168  pF->_aulNeighbours[i] = ulCt;
169  }
170  }
171  }
172 
173  // insert facet into array
174  _aclFacetArray.push_back(clFacet);
175 }

193 unsigned long MeshKernel::AddFacets(const std::vector<MeshFacet>& rclFAry, bool checkManifolds)
194 {
195  // Build map of edges of the referencing facets we want to append
196 #ifdef FC_DEBUG
197  unsigned long countPoints = CountPoints();
198 #endif
199 
200  // if the manifold check shouldn't be done then just add all faces
201  if (!checkManifolds) {
202  FacetIndex countFacets = CountFacets();
203  FacetIndex countValid = rclFAry.size();
204  _aclFacetArray.reserve(countFacets + countValid);
205 
206  // just add all faces now
207  for (const auto& pF : rclFAry) {
208  _aclFacetArray.push_back(pF);
209  }
210 
211  RebuildNeighbours(countFacets);
212  return _aclFacetArray.size();
213  }
214 
215  this->_aclPointArray.ResetInvalid();
216  FacetIndex k = CountFacets();
217  std::map<std::pair<PointIndex, PointIndex>, std::list<FacetIndex>> edgeMap;
218  for (auto pF = rclFAry.begin(); pF != rclFAry.end(); ++pF, k++) {
219  // reset INVALID flag for all candidates
220  pF->ResetFlag(MeshFacet::INVALID);
221  for (int i = 0; i < 3; i++) {
222 #ifdef FC_DEBUG
223  assert(pF->_aulPoints[i] < countPoints);
224 #endif
225  this->_aclPointArray[pF->_aulPoints[i]].SetFlag(MeshPoint::INVALID);
226  PointIndex ulT0 = pF->_aulPoints[i];
227  PointIndex ulT1 = pF->_aulPoints[(i + 1) % 3];
228  PointIndex ulP0 = std::min<PointIndex>(ulT0, ulT1);
229  PointIndex ulP1 = std::max<PointIndex>(ulT0, ulT1);
230  edgeMap[std::make_pair(ulP0, ulP1)].push_front(k);
231  }
232  }
233 
234  // Check for the above edges in the current facet array
235  k = 0;
236  for (auto pF = _aclFacetArray.begin(); pF != _aclFacetArray.end(); ++pF, k++) {
237  // if none of the points references one of the edges ignore the facet
238  if (!this->_aclPointArray[pF->_aulPoints[0]].IsFlag(MeshPoint::INVALID)
239  && !this->_aclPointArray[pF->_aulPoints[1]].IsFlag(MeshPoint::INVALID)
240  && !this->_aclPointArray[pF->_aulPoints[2]].IsFlag(MeshPoint::INVALID)) {
241  continue;
242  }
243  for (int i = 0; i < 3; i++) {
244  PointIndex ulT0 = pF->_aulPoints[i];
245  PointIndex ulT1 = pF->_aulPoints[(i + 1) % 3];
246  PointIndex ulP0 = std::min<PointIndex>(ulT0, ulT1);
247  PointIndex ulP1 = std::max<PointIndex>(ulT0, ulT1);
248  std::pair<PointIndex, PointIndex> edge = std::make_pair(ulP0, ulP1);
249  auto pI = edgeMap.find(edge);
250  // Does the current facet share the same edge?
251  if (pI != edgeMap.end()) {
252  pI->second.push_front(k);
253  }
254  }
255  }
256 
257  this->_aclPointArray.ResetInvalid();
258 
259  // Now let's see for which edges we might get manifolds, if so we don't add the corresponding
260  // candidates
261  FacetIndex countFacets = CountFacets();
262  std::map<std::pair<PointIndex, PointIndex>, std::list<FacetIndex>>::iterator pE;
263  for (pE = edgeMap.begin(); pE != edgeMap.end(); ++pE) {
264  if (pE->second.size() > 2) {
265  for (FacetIndex it : pE->second) {
266  if (it >= countFacets) {
267  // this is a candidate
268  FacetIndex index = it - countFacets;
269  rclFAry[index].SetFlag(MeshFacet::INVALID);
270  }
271  }
272  }
273  }
274 
275  // Do not insert directly to the data structure because we should get the correct size of new
276  // facets, otherwise std::vector reallocates too much memory which can't be freed so easily
277  MeshIsNotFlag<MeshFacet> flag;
278  FacetIndex countValid =
279  std::count_if(rclFAry.begin(), rclFAry.end(), [flag](const MeshFacet& f) {
280  return flag(f, MeshFacet::INVALID);
281  });
282  _aclFacetArray.reserve(_aclFacetArray.size() + countValid);
283  // now start inserting the facets to the data structure and set the correct neighbourhood as
284  // well
285  FacetIndex startIndex = CountFacets();
286  for (const auto& pF : rclFAry) {
287  if (!pF.IsFlag(MeshFacet::INVALID)) {
288  _aclFacetArray.push_back(pF);
289  pF.SetProperty(startIndex++);
290  }
291  }
292 
293  // resolve neighbours
294  for (pE = edgeMap.begin(); pE != edgeMap.end(); ++pE) {
295  PointIndex ulP0 = pE->first.first;
296  PointIndex ulP1 = pE->first.second;
297  if (pE->second.size() == 1) // border facet
298  {
299  FacetIndex ulF0 = pE->second.front();
300  if (ulF0 >= countFacets) {
301  ulF0 -= countFacets;
302  std::vector<MeshFacet>::const_iterator pF = rclFAry.begin() + ulF0;
303  if (!pF->IsFlag(MeshFacet::INVALID)) {
304  ulF0 = pF->_ulProp;
305  }
306  else {
307  ulF0 = FACET_INDEX_MAX;
308  }
309  }
310 
311  if (ulF0 != FACET_INDEX_MAX) {
312  unsigned short usSide = _aclFacetArray[ulF0].Side(ulP0, ulP1);
313  assert(usSide != USHRT_MAX);
314  _aclFacetArray[ulF0]._aulNeighbours[usSide] = FACET_INDEX_MAX;
315  }
316  }
317  else if (pE->second.size() == 2) // normal facet with neighbour
318  {
319  // we must check if both facets are part of the mesh now
320  FacetIndex ulF0 = pE->second.front();
321  if (ulF0 >= countFacets) {
322  ulF0 -= countFacets;
323  std::vector<MeshFacet>::const_iterator pF = rclFAry.begin() + ulF0;
324  if (!pF->IsFlag(MeshFacet::INVALID)) {
325  ulF0 = pF->_ulProp;
326  }
327  else {
328  ulF0 = FACET_INDEX_MAX;
329  }
330  }
331  FacetIndex ulF1 = pE->second.back();
332  if (ulF1 >= countFacets) {
333  ulF1 -= countFacets;
334  std::vector<MeshFacet>::const_iterator pF = rclFAry.begin() + ulF1;
335  if (!pF->IsFlag(MeshFacet::INVALID)) {
336  ulF1 = pF->_ulProp;
337  }
338  else {
339  ulF1 = FACET_INDEX_MAX;
340  }
341  }
342 
343  if (ulF0 != FACET_INDEX_MAX) {
344  unsigned short usSide = _aclFacetArray[ulF0].Side(ulP0, ulP1);
345  assert(usSide != USHRT_MAX);
346  _aclFacetArray[ulF0]._aulNeighbours[usSide] = ulF1;
347  }
348 
349  if (ulF1 != FACET_INDEX_MAX) {
350  unsigned short usSide = _aclFacetArray[ulF1].Side(ulP0, ulP1);
351  assert(usSide != USHRT_MAX);
352  _aclFacetArray[ulF1]._aulNeighbours[usSide] = ulF0;
353  }
354  }
355  }
356 
357  return _aclFacetArray.size();
358 }

380 void MeshKernel::Merge(const MeshPointArray& rPoints, const MeshFacetArray& rFaces)
381 {
382  if (rPoints.empty() || rFaces.empty()) {
383  return; // nothing to do
384  }
385  std::vector<PointIndex> increments(rPoints.size());
386 
387  FacetIndex countFacets = this->_aclFacetArray.size();
388  // Reserve the additional memory to append the new facets
389  this->_aclFacetArray.reserve(this->_aclFacetArray.size() + rFaces.size());
390 
391  // Copy the new faces immediately to the facet array
392  MeshFacet face;
393  for (const auto& it : rFaces) {
394  face = it;
395  for (PointIndex point : it._aulPoints) {
396  increments[point]++;
397  }
398 
399  // append to the facet array
400  this->_aclFacetArray.push_back(face);
401  }
402 
403  std::size_t countNewPoints =
404  std::count_if(increments.begin(), increments.end(), [](PointIndex v) {
405  return v > 0;
406  });
407  // Reserve the additional memory to append the new points
408  PointIndex index = this->_aclPointArray.size();
409  this->_aclPointArray.reserve(this->_aclPointArray.size() + countNewPoints);
410 
411  // Now we can start inserting the points and adjust the point indices of the faces
412  for (auto it = increments.begin(); it != increments.end(); ++it) {
413  if (*it > 0) {
414  // set the index of the point array
415  *it = index++;
416  const MeshPoint& rPt = rPoints[it - increments.begin()];
417  this->_aclPointArray.push_back(rPt);
418  _clBoundBox.Add(rPt);
419  }
420  }
421 
422  for (auto pF = this->_aclFacetArray.begin() + countFacets; pF != this->_aclFacetArray.end();
423  ++pF) {
424  for (PointIndex& index : pF->_aulPoints) {
425  index = increments[index];
426  }
427  }
428 
429  // Since rFaces could consist of a subset of the actual facet array the
430  // neighbour indices could be totally wrong so they must be rebuilt from
431  // scratch. Fortunately, this needs only to be done for the newly inserted
432  // facets -- not for all
433  RebuildNeighbours(countFacets);
434 }

454 bool MeshKernel::DeleteFacet(const MeshFacetIterator& rclIter)
455 {
456  FacetIndex ulNFacet {}, ulInd {};
457 
458  if (rclIter._clIter >= _aclFacetArray.end()) {
459  return false;
460  }
461 
462  // index of the facet to delete
463  ulInd = rclIter._clIter - _aclFacetArray.begin();
464 
465  // invalidate neighbour indices of the neighbour facet to this facet
466  for (FacetIndex nbIndex : rclIter._clIter->_aulNeighbours) {
467  ulNFacet = nbIndex;
468  if (ulNFacet != FACET_INDEX_MAX) {
469  for (FacetIndex& nbOfNb : _aclFacetArray[ulNFacet]._aulNeighbours) {
470  if (nbOfNb == ulInd) {
471  nbOfNb = FACET_INDEX_MAX;
472  break;
473  }
474  }
475  }
476  }
477 
478  // erase corner point if needed
479  for (int i = 0; i < 3; i++) {
480  if ((rclIter._clIter->_aulNeighbours[i] == FACET_INDEX_MAX)
481  && (rclIter._clIter->_aulNeighbours[(i + 1) % 3] == FACET_INDEX_MAX)) {
482  // no neighbours, possibly delete point
483  ErasePoint(rclIter._clIter->_aulPoints[(i + 1) % 3], ulInd);
484  }
485  }
486 
487  // remove facet from array
488  _aclFacetArray.Erase(_aclFacetArray.begin() + rclIter.Position());
489 
490  return true;
491 }

551 {

553  std::vector<MeshFacetIterator> clToDel;
554  PointIndex ulInd {};

556  // index of the point to delete
557  ulInd = rclIter._clIter - _aclPointArray.begin();

559  pFIter.Begin();

562  // check corner points of all facets
563  while (pFIter < pFEnd) {

619 void MeshKernel::ErasePoint(PointIndex ulIndex, FacetIndex ulFacetIndex, bool bOnlySetInvalid)
620 {
621  std::vector<MeshFacet>::iterator pFIter, pFEnd, pFNot;
622 
623  pFIter = _aclFacetArray.begin();
624  pFNot = _aclFacetArray.begin() + ulFacetIndex;
625  pFEnd = _aclFacetArray.end();
626 
627  // check all facets
628  while (pFIter < pFNot) {
629  for (PointIndex ptIndex : pFIter->_aulPoints) {
630  if (ptIndex == ulIndex) {
631  return; // point still referenced ==> do not delete
632  }
633  }
634  ++pFIter;
635  }
636 
637  ++pFIter;
638  while (pFIter < pFEnd) {
639  for (PointIndex ptIndex : pFIter->_aulPoints) {
640  if (ptIndex == ulIndex) {
641  return; // point still referenced ==> do not delete
642  }
643  }
644  ++pFIter;
645  }
646 
647 
648  if (!bOnlySetInvalid) {
649  // completely remove point
650  _aclPointArray.erase(_aclPointArray.begin() + ulIndex);
651 
652  // correct point indices of the facets
653  pFIter = _aclFacetArray.begin();
654  while (pFIter < pFEnd) {
655  for (PointIndex& ptIndex : pFIter->_aulPoints) {
656  if (ptIndex > ulIndex) {
657  ptIndex--;
658  }
659  }
660  ++pFIter;
661  }

918 void MeshKernel::Read(std::istream& rclIn)
919 {
920  if (!rclIn || rclIn.bad()) {
921  return;
922  }
923 
924  // get header
925  Base::InputStream str(rclIn);
926 
927  // Read the header with a "magic number" and a version
928  uint32_t magic {}, version {}, swap_magic {}, swap_version {};
929  str >> magic >> version;
930  swap_magic = magic;
931  Base::SwapEndian(swap_magic);
932  swap_version = version;
933  Base::SwapEndian(swap_version);
934  uint32_t open_edge = 0xffffffff; // value to mark an open edge
935 
936  // is it the new or old format?
937  bool new_format = false;
938  if (magic == 0xA0B0C0D0 && version == 0x010000) {
939  new_format = true;
940  }
941  else if (swap_magic == 0xA0B0C0D0 && swap_version == 0x010000) {
942  new_format = true;
943  str.setByteOrder(Base::Stream::BigEndian);
944  }
945 
946  if (new_format) {
947  char szInfo[256];
948  rclIn.read(szInfo, 256);
949 
950  // read the number of points and facets
951  uint32_t uCtPts = 0, uCtFts = 0;
952  str >> uCtPts >> uCtFts;
953 
954  try {
955  // read the data
956  MeshPointArray pointArray;
957  pointArray.resize(uCtPts);
958  for (auto& it : pointArray) {
959  str >> it.x >> it.y >> it.z;
960  }
961 
962  MeshFacetArray facetArray;
963  facetArray.resize(uCtFts);
964 
965  uint32_t v1 {}, v2 {}, v3 {};
966  for (auto& it : facetArray) {
967  str >> v1 >> v2 >> v3;
968 
969  // make sure to have valid indices
970  if (v1 >= uCtPts || v2 >= uCtPts || v3 >= uCtPts) {
971  throw Base::BadFormatError("Invalid data structure");
972  }
973 
974  it._aulPoints[0] = v1;
975  it._aulPoints[1] = v2;
976  it._aulPoints[2] = v3;
977 
978  // On systems where an 'unsigned long' is a 64-bit value
979  // the empty neighbour must be explicitly set to 'FACET_INDEX_MAX'
980  // because in algorithms this value is always used to check
981  // for open edges.
982  str >> v1 >> v2 >> v3;
983 
984  // make sure to have valid indices
985  if (v1 >= uCtFts && v1 < open_edge) {
986  throw Base::BadFormatError("Invalid data structure");
987  }
988  if (v2 >= uCtFts && v2 < open_edge) {
989  throw Base::BadFormatError("Invalid data structure");
990  }
991  if (v3 >= uCtFts && v3 < open_edge) {
992  throw Base::BadFormatError("Invalid data structure");
993  }
994 
995  if (v1 < open_edge) {
996  it._aulNeighbours[0] = v1;
997  }
998  else {
999  it._aulNeighbours[0] = FACET_INDEX_MAX;
1000  }
1001 
1002  if (v2 < open_edge) {
1003  it._aulNeighbours[1] = v2;
1004  }
1005  else {
1006  it._aulNeighbours[1] = FACET_INDEX_MAX;
1007  }
1008 
1009  if (v3 < open_edge) {
1010  it._aulNeighbours[2] = v3;
1011  }
1012  else {
1013  it._aulNeighbours[2] = FACET_INDEX_MAX;
1014  }
1015  }
1016 
1017  str >> _clBoundBox.MinX >> _clBoundBox.MaxX;
1018  str >> _clBoundBox.MinY >> _clBoundBox.MaxY;
1019  str >> _clBoundBox.MinZ >> _clBoundBox.MaxZ;
1020 
1021  // If we reach this block no exception occurred and we can safely assign the mesh
1022  _aclPointArray.swap(pointArray);
1023  _aclFacetArray.swap(facetArray);
1024  }
1025  catch (std::exception&) {
1026  // Special handling of std::length_error
1027  throw Base::BadFormatError("Reading from stream failed");
1028  }
1029  }
1030  else {
1031  // The old formats
1032  unsigned long uCtPts = magic, uCtFts = version;
1033  MeshPointArray pointArray;
1034  MeshFacetArray facetArray;
1035 
1036  float ratio = 0;
1037  if (uCtPts > 0) {
1038  ratio = static_cast<float>(uCtFts) / static_cast<float>(uCtPts);
1039  }
1040 
1041  // without edge array
1042  if (ratio < 2.5F) {
1043  // the stored mesh kernel might be empty
1044  if (uCtPts > 0) {
1045  pointArray.resize(uCtPts);
1046  rclIn.read((char*)pointArray.data(), uCtPts * sizeof(MeshPoint));
1047  }
1048  if (uCtFts > 0) {
1049  facetArray.resize(uCtFts);
1050  rclIn.read((char*)facetArray.data(), uCtFts * sizeof(MeshFacet));
1051  }
1052  rclIn.read((char*)&_clBoundBox, sizeof(Base::BoundBox3f));
1053  }
1054  else {
1055  // with edge array
1056  unsigned long uCtEdges = uCtFts;
1057  str >> magic;
1058  uCtFts = magic;
1059  pointArray.resize(uCtPts);
1060  for (auto& it : pointArray) {
1061  str >> it.x >> it.y >> it.z;
1062  }
1063  uint32_t dummy {};
1064  for (unsigned long i = 0; i < uCtEdges; i++) {
1065  str >> dummy;
1066  }
1067  uint32_t v1 {}, v2 {}, v3 {};
1068  facetArray.resize(uCtFts);
1069  for (auto& it : facetArray) {
1070  str >> v1 >> v2 >> v3;
1071  it._aulNeighbours[0] = v1;
1072  it._aulNeighbours[1] = v2;
1073  it._aulNeighbours[2] = v3;
1074  str >> v1 >> v2 >> v3;
1075  it._aulPoints[0] = v1;
1076  it._aulPoints[1] = v2;
1077  it._aulPoints[2] = v3;
1078  str >> it._ucFlag;
1079  }
1080 
1081  str >> _clBoundBox.MinX >> _clBoundBox.MinY >> _clBoundBox.MinZ >> _clBoundBox.MaxX
1082  >> _clBoundBox.MaxY >> _clBoundBox.MaxZ;
1083  }
1084 
1085  for (auto& it : facetArray) {
1086  for (int i = 0; i < 3; i++) {
1087  if (it._aulPoints[i] >= uCtPts) {
1088  throw Base::BadFormatError("Invalid data structure");
1089  }
1090  if (it._aulNeighbours[i] < FACET_INDEX_MAX && it._aulNeighbours[i] >= uCtFts) {
1091  throw Base::BadFormatError("Invalid data structure");
1092  }
1093  }
1094  }
1095 
1096  _aclPointArray.swap(pointArray);
1097  _aclFacetArray.swap(facetArray);
1098  }
