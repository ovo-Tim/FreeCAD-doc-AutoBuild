56 def _collectVoronoiWires(vd):
57  edges = [e for e in vd.Edges if e.Color == PRIMARY]
58  vertex = {}
59  for e in edges:
60  for v in e.Vertices:
61  i = v.Index
62  j = vertex.get(i, [])
63  j.append(e)
64  vertex[i] = j
65 
66  # knots are the start and end points of a wire
67  knots = [i for i in vertex if len(vertex[i]) == 1]
68  knots.extend([i for i in vertex if len(vertex[i]) > 2])
69  if len(knots) == 0:
70  for i in vertex:
71  if len(vertex[i]) > 0:
72  knots.append(i)
73  break
74 
75  def consume(v, edge):
76  vertex[v] = [e for e in vertex[v] if e.Index != edge.Index]
77  return len(vertex[v]) == 0
78 
79  def traverse(vStart, edge, edges):
80  if vStart == edge.Vertices[0].Index:
81  vEnd = edge.Vertices[1].Index
82  edges.append(edge)
83  else:
84  vEnd = edge.Vertices[0].Index
85  edges.append(edge.Twin)
86 
87  consume(vStart, edge)
88  if consume(vEnd, edge):
89  return None
90  return vEnd
91 
92  wires = []
93  while knots:
94  we = []
95  vFirst = knots[0]
96  vStart = vFirst
97  vLast = vFirst
98  if len(vertex[vStart]):
99  while vStart is not None:
100  vLast = vStart
101  edges = vertex[vStart]
102  if len(edges) > 0:
103  edge = edges[0]
104  vStart = traverse(vStart, edge, we)
105  else:
106  vStart = None
107  wires.append(we)
108  if len(vertex[vFirst]) == 0:
109  knots = [v for v in knots if v != vFirst]
110  if len(vertex[vLast]) == 0:
111  knots = [v for v in knots if v != vLast]
112  return wires

116  def closestTo(start, point):
117  p = None
118  l = None
119  for i in point:
120  if l is None or l > start.distanceToPoint(point[i]):
121  l = start.distanceToPoint(point[i])

165  def incrementStepDownDepth(self, maximumUsableDepth):
166  """
167  Increase stepDown depth before staring new carving pass.
168  :returns: True if successful, False if maximum depth achieved
169  """
170 
171  # do not allow one to increase depth if we are already at stop depth
172  if self.maximumDepth == self.stop:
173  return False
174 
175  # do not allow one to increase depth if we are already at
176  # maximum usable depth
177 
178  if self.maximumDepth <= maximumUsableDepth:
179  return False
180 
181  self.stepDownPass += 1

188  """

353  def insert_many_wires(vd, wires):
354  for wire in wires:
355  Path.Log.debug("discretize value: {}".format(obj.Discretize))
356  pts = wire.discretize(QuasiDeflection=obj.Discretize)
357  ptv = [FreeCAD.Vector(p.x, p.y) for p in pts]
358  # Check over the last point before just closing the polygon
359  # by adding the start again. If the discretizer was aiming
360  # for the last point and missed by a little bit, closing the
361  # polygon as is could result in OpenVoronoi truncating the
362  # coordinates to a self-intersecting polygon which is invalid.
363  # Instead, if the last point is close to the first, remove it
364  # and let the final append close the polygon.
365  # See issue 8064
366  if len(ptv) > 0:
367  dist = ptv[-1].distanceToPoint(ptv[0])
368  if dist < FreeCAD.Base.Precision.confusion():
369  Path.Log.debug(
370  "Removing bad carve point: {} from polygon origin".format(dist)
371  )
372  del ptv[-1]
373  ptv.append(ptv[0])
374 
375  for i in range(len(ptv) - 1):
376  vd.addSegment(ptv[i], ptv[i + 1])
377 
378  for f in faces:
379  voronoiWires = []
380  vd = Path.Voronoi.Diagram()
381  insert_many_wires(vd, f.Wires)
382 
383  vd.construct()
384 
385  for e in vd.Edges:
386  if e.isPrimary():
387  if e.isBorderline():
388  e.Color = BORDERLINE
389  else:

391  else:
