167 void ViewProviderMeshCurvature::init(const Mesh::PropertyCurvatureList* prop)
168 {
169  // NOLINTBEGIN(readability-magic-numbers)
170  std::vector<float> aMinValues;
171  std::vector<float> aMaxValues;
172  const std::vector<Mesh::CurvatureInfo>& fCurvInfo = prop->getValues();
173  aMinValues.reserve(fCurvInfo.size());
174  aMaxValues.reserve(fCurvInfo.size());
175 
176  for (const auto& jt : fCurvInfo) {
177  aMinValues.push_back(jt.fMinCurvature);
178  aMaxValues.push_back(jt.fMaxCurvature);
179  }
180 
181  if (aMinValues.empty() || aMaxValues.empty()) {
182  return; // no values inside
183  }
184 
185  float fMin = *std::min_element(aMinValues.begin(), aMinValues.end());
186  float fMax = *std::max_element(aMinValues.begin(), aMinValues.end());
187 
188  // histogram over all values
189  std::map<int, int> aHistogram;
190  for (float aMinValue : aMinValues) {
191  int grp = (int)(10.0F * (aMinValue - fMin) / (fMax - fMin));
192  aHistogram[grp]++;
193  }
194 
195  float fRMin = -1.0F;
196  for (const auto& mIt : aHistogram) {
197  if ((float)mIt.second / (float)aMinValues.size() > 0.15F) {
198  fRMin = float(mIt.first) * (fMax - fMin) / 10.0F + fMin;
199  break;
200  }
201  }
202 
203  fMin = *std::min_element(aMaxValues.begin(), aMaxValues.end());
204  fMax = *std::max_element(aMaxValues.begin(), aMaxValues.end());
205 
206  // histogram over all values
207  aHistogram.clear();
208  for (float aMaxValue : aMaxValues) {
209  int grp = (int)(10.0F * (aMaxValue - fMin) / (fMax - fMin));
210  aHistogram[grp]++;
211  }
212 
213  float fRMax = 1.0F;
214  for (auto rIt2 = aHistogram.rbegin(); rIt2 != aHistogram.rend(); ++rIt2) {
215  if ((float)rIt2->second / (float)aMaxValues.size() > 0.15F) {
216  fRMax = float(rIt2->first) * (fMax - fMin) / 10.0F + fMin;
217  break;
218  }
219  }
220  // NOLINTEND(readability-magic-numbers)
221 
222  float fAbs = std::max<float>(fabs(fRMin), fabs(fRMax));
223  fRMin = -fAbs;
224  fRMax = fAbs;
225  fMin = fRMin;
226  fMax = fRMax;
227  pcColorBar->setRange(fMin, fMax, 3);
228 }

337 void ViewProviderMeshCurvature::setVertexCurvatureMode(int mode)
338 {
339  using PropertyMap = std::map<std::string, App::Property*>;
340  PropertyMap Map;
341  pcObject->getPropertyMap(Map);
342 
343  auto it = std::find_if(Map.begin(), Map.end(), [](const PropertyMap::value_type& v) {
344  Base::Type type = v.second->getTypeId();
345  return (type == Mesh::PropertyCurvatureList::getClassTypeId());
346  });
347 
348  if (it == Map.end()) {
349  return; // cannot display this feature type due to missing curvature property
350  }
351 
352  auto pCurvInfo = dynamic_cast<Mesh::PropertyCurvatureList*>(it->second);
353 
354  // curvature values
355  std::vector<float> fValues = pCurvInfo->getCurvature(mode);
356  pcColorMat->diffuseColor.setNum(int(fValues.size()));
357  pcColorMat->transparency.setNum(int(fValues.size()));
358 
359  SbColor* diffcol = pcColorMat->diffuseColor.startEditing();
360  float* transp = pcColorMat->transparency.startEditing();
361 
362  for (auto const& value : fValues | boost::adaptors::indexed(0)) {
363  App::Color c = pcColorBar->getColor(value.value());
364  // NOLINTBEGIN
365  diffcol[value.index()].setValue(c.r, c.g, c.b);
366  transp[value.index()] = c.transparency();
367  // NOLINTEND
368  }
369 
370  pcColorMat->diffuseColor.finishEditing();
371  pcColorMat->transparency.finishEditing();
372 
373  // In order to apply the transparency changes the IndexFaceSet node must be touched
374  touchShapeNode();
375 }

511 void ViewProviderMeshCurvature::curvatureInfoCallback(void* ud, SoEventCallback* n)
512 {
513  auto view = static_cast<Gui::View3DInventorViewer*>(n->getUserData());
514  const SoEvent* ev = n->getEvent();
515  if (ev->getTypeId() == SoMouseButtonEvent::getClassTypeId()) {
516  const auto mbe = static_cast<const SoMouseButtonEvent*>(ev); // NOLINT
517 
518  // Mark all incoming mouse button events as handled, especially, to deactivate the
519  // selection node
520  n->getAction()->setHandled();
521  if (mbe->getButton() == SoMouseButtonEvent::BUTTON2
522  && mbe->getState() == SoButtonEvent::UP) {
523  n->setHandled();
524  // context-menu
525  QMenu menu;
526  QAction* fl = menu.addAction(QObject::tr("Annotation"));
527  fl->setCheckable(true);
528  fl->setChecked(addflag);
529  QAction* cl = menu.addAction(QObject::tr("Leave info mode"));
530  QAction* id = menu.exec(QCursor::pos());
531  if (fl == id) {
532  addflag = fl->isChecked();
533  }
534  else if (cl == id) {
535  view->setEditing(false);
536  view->getWidget()->setCursor(QCursor(Qt::ArrowCursor));
537  view->setRedirectToSceneGraph(false);
538  view->setSelectionEnabled(true);
539  view->removeEventCallback(SoEvent::getClassTypeId(), curvatureInfoCallback, ud);
540  }
541  }
542  else if (mbe->getButton() == SoMouseButtonEvent::BUTTON1
543  && mbe->getState() == SoButtonEvent::UP) {
544  const SoPickedPoint* point = n->getPickedPoint();
545  if (!point) {
546  Base::Console().Message("No facet picked.\n");
547  return;
548  }
549 
550  n->setHandled();
551 
552  // By specifying the indexed mesh node 'pcFaceSet' we make sure that the picked point is
553  // really from the mesh we render and not from any other geometry
554  Gui::ViewProvider* vp = view->getViewProviderByPathFromTail(point->getPath());
555  if (auto self = dynamic_cast<ViewProviderMeshCurvature*>(vp)) {
556  const SoDetail* detail = point->getDetail(point->getPath()->getTail());
557  if (detail && detail->getTypeId() == SoFaceDetail::getClassTypeId()) {
558  const auto facedetail = static_cast<const SoFaceDetail*>(detail); // NOLINT
559  // get the curvature info of the three points of the picked facet
560  int index1 = facedetail->getPoint(0)->getCoordinateIndex();
561  int index2 = facedetail->getPoint(1)->getCoordinateIndex();
562  int index3 = facedetail->getPoint(2)->getCoordinateIndex();
563  std::string info = self->curvatureInfo(true, index1, index2, index3);
564  QString text = QString::fromLatin1(info.c_str());
565  if (addflag) {
566  SbVec3f pt = point->getPoint();
567  SbVec3f nl = point->getNormal();
568  auto anno = new Annotation(self, text, pt, nl);
569  auto sensor = new SoIdleSensor(Annotation::run, anno);
570  sensor->schedule();
571  }
572  else {
573  Gui::ToolTip::showText(QCursor::pos(), text);
574  }
575  }
576  }
577  }
578  }
579  else if (ev->getTypeId().isDerivedFrom(SoLocation2Event::getClassTypeId())) {
580  const SoPickedPoint* point = n->getPickedPoint();
581  if (!point) {
582  return;
583  }
584  n->setHandled();
585 
586  // By specifying the indexed mesh node 'pcFaceSet' we make sure that the picked point is
587  // really from the mesh we render and not from any other geometry
588  Gui::ViewProvider* vp = view->getViewProviderByPathFromTail(point->getPath());
589  if (auto self = dynamic_cast<ViewProviderMeshCurvature*>(vp)) {
590  const SoDetail* detail = point->getDetail(point->getPath()->getTail());
591  if (detail && detail->getTypeId() == SoFaceDetail::getClassTypeId()) {
592  const auto facedetail = static_cast<const SoFaceDetail*>(detail); // NOLINT
593  // get the curvature info of the three points of the picked facet
594  int index1 = facedetail->getPoint(0)->getCoordinateIndex();
595  int index2 = facedetail->getPoint(1)->getCoordinateIndex();
596  int index3 = facedetail->getPoint(2)->getCoordinateIndex();
597  std::string info = self->curvatureInfo(false, index1, index2, index3);
598  Gui::getMainWindow()->setPaneText(1, QString::fromLatin1(info.c_str()));
599  }
600  }
601  }

604 std::string
605 ViewProviderMeshCurvature::curvatureInfo(bool detail, int index1, int index2, int index3) const
606 {
607  // get the curvature info of the three points of the picked facet
608  App::Property* prop = pcObject->getPropertyByName("CurvInfo");
609  std::stringstream str;
610  if (auto curv = dynamic_cast<Mesh::PropertyCurvatureList*>(prop)) {
611  const Mesh::CurvatureInfo& cVal1 = (*curv)[index1];
612  const Mesh::CurvatureInfo& cVal2 = (*curv)[index2];
613  const Mesh::CurvatureInfo& cVal3 = (*curv)[index3];
614  float fVal1 = 0.0F;
615  float fVal2 = 0.0F;
616  float fVal3 = 0.0F;
617
