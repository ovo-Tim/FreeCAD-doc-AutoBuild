110 def get_pynastran_mesh(
111  femnodes_mesh,
112  femelement_table,
113  export_element_type,
114 ):
115  if export_element_type is None:
116  Console.PrintError("Error: wrong export_element_type.\n")
117  return
118 
119  # nodes
120  pynas_nodes = "# grid cards, geometric mesh points\n"
121  for node in femnodes_mesh:
122  vec = femnodes_mesh[node]
123  pynas_nodes += f"model.add_grid({node}, [{vec.x}, {vec.y}, {vec.z}])\n"
124  # print(pynas_nodes)
125 
126  # elements
127  # Nastran seems to have the same node order as SMESH (FreeCAD) has
128  # thus just write the nodes at once
129  pynas_elements = "# elements cards\n"
130  for element in femelement_table:
131  nodes = femelement_table[element]
132  # print(element) # eleid
133  # print(n) # tuple of nodes
134  if export_element_type == "cbar":
135  pynas_elements += (
136  "model.add_{ele_keyword}({eid}, {pid}, {nodes}, "
137  "{orientation_vec}, {gnull})\n".format(
138  ele_keyword=export_element_type,
139  eid=element,
140  pid=1,
141  nodes=nodes,
142  orientation_vec="x=[0.0, 0.0, 1.0]",
143  gnull="g0=None",
144  )
145  )
146  else:
147  if export_element_type == "ctetra4":
148  ele_keyword = "ctetra"
149  # N1, N3, N2, N4
150  the_nodes = [nodes[0], nodes[2], nodes[1], nodes[3]]
151  elif export_element_type == "ctetra10":
152  ele_keyword = "ctetra"
153  # N1, N3, N2, N4, N7, N6, N5, N8, N10, N9
154  the_nodes = [
155  nodes[0],
156  nodes[2],
157  nodes[1],
158  nodes[3],
159  nodes[6],
160  nodes[5],
161  nodes[4],
162  nodes[7],
163  nodes[9],
164  nodes[8],
165  ]
166  else:
167  ele_keyword = export_element_type
168  the_nodes = nodes
169  pynas_elements += "model.add_{ele_keyword}({eid}, {pid}, {nodes})\n".format(
170  ele_keyword=ele_keyword, eid=element, pid=1, nodes=the_nodes
171  )
172  # print(pynas_elements)
173 
174  mesh_pynas_code = f"{pynas_nodes}\n\n{pynas_elements}\n\n"
175  return mesh_pynas_code
