80  int ChainIkSolverVel_wdls::CartToJnt(const JntArray& q_in, const Twist& v_in, JntArray& qdot_out)
81  {
82  jnt2jac.JntToJac(q_in,jac);
83 
84  double sum;
85  unsigned int i,j;
86 
87  // Initialize (internal) return values
88  nrZeroSigmas = 0 ;
89  sigmaMin = 0.;
90  lambda_scaled = 0.;
91 
92  /*
93  for (i=0;i<jac.rows();i++) {
94  for (j=0;j<jac.columns();j++)
95  tmp_jac(i,j) = jac(i,j);
96  }
97  */
98 
99  // Create the Weighted jacobian
100  tmp_jac_weight1 = jac.data.lazyProduct(weight_js);
101  tmp_jac_weight2 = weight_ts.lazyProduct(tmp_jac_weight1);
102 
103  // Compute the SVD of the weighted jacobian
104  svdResult = svd_eigen_HH(tmp_jac_weight2,U,S,V,tmp,maxiter);
105  if (0 != svdResult)
106  {
107  qdot_out.data.setZero() ;
108  return (error = E_SVD_FAILED);
109  }
110 
111  //Pre-multiply U and V by the task space and joint space weighting matrix respectively
112  tmp_ts = weight_ts.lazyProduct(U.topLeftCorner(6,6));
113  tmp_js = weight_js.lazyProduct(V);
114 
115  // Minimum of six largest singular values of J is S(5) if number of joints >=6 and 0 for <6
116  if ( jac.columns() >= 6 ) {
117  sigmaMin = S(5);
118  }
119  else {
120  sigmaMin = 0.;
121  }
122 
123  // tmp = (Si*U'*Ly*y),
124  for (i=0;i<jac.columns();i++) {
125  sum = 0.0;
126  for (j=0;j<jac.rows();j++) {
127  if(i<6)
128  sum+= tmp_ts(j,i)*v_in(j);
129  else
130  sum+=0.0;
131  }
132  // If sigmaMin > eps, then wdls is not active and lambda_scaled = 0 (default value)
133  // If sigmaMin < eps, then wdls is active and lambda_scaled is scaled from 0 to lambda
134  // Note: singular values are always positive so sigmaMin >=0
135  if ( sigmaMin < eps )
136  {
137  lambda_scaled = sqrt(1.0-(sigmaMin/eps)*(sigmaMin/eps))*lambda ;
138  }
139  if(fabs(S(i))<eps) {
140  if (i<6) {
141  // Scale lambda to size of singular value sigmaMin
142  tmp(i) = sum*((S(i)/(S(i)*S(i)+lambda_scaled*lambda_scaled)));
143  }
144  else {
145  tmp(i)=0.0; // S(i)=0 for i>=6 due to cols>rows
146  }
147  // Count number of singular values near zero
148  ++nrZeroSigmas ;
149  }
150  else {
151  tmp(i) = sum/S(i);
152  }
153  }
154 
155  /*
156  // x = Lx^-1*V*tmp + x
157  for (i=0;i<jac.columns();i++) {
158  sum = 0.0;
159  for (j=0;j<jac.columns();j++) {
160  sum+=tmp_js(i,j)*tmp(j);
161  }
162  qdot_out(i)=sum;
163  }
164  */
165  qdot_out.data=tmp_js.lazyProduct(tmp);
166 
167  // If number of near zero singular values is greater than the full rank
168  // of jac, then wdls is active
169  if ( nrZeroSigmas > (jac.columns()-jac.rows()) ) {
170  return (error = E_CONVERGE_PINV_SINGULAR); // converged but pinv singular
171  } else {
172  return (error = E_NOERROR); // have converged
173  }
174  }
