129 {
130  try {
131  Handle(Geom_Geometry) g = getGeometryPtr()->handle();
132  Handle(Geom_Curve) c = Handle(Geom_Curve)::DownCast(g);
133  if (c.IsNull()) {
134  PyErr_SetString(PartExceptionOCCError, "Geometry is not a curve");
135  return nullptr;
136  }
137 
138  GeomAdaptor_Curve adapt(c);
139  bool uniformAbscissaPoints = false;
140  bool uniformAbscissaDistance = false;
141  int numPoints = -1;
142  double distance = -1;
143  double first = adapt.FirstParameter();
144  double last = adapt.LastParameter();
145 
146  // use no kwds
147  PyObject* dist_or_num;
148  if (PyArg_ParseTuple(args, "O", &dist_or_num)) {
149  if (PyLong_Check(dist_or_num)) {
150  numPoints = PyLong_AsLong(dist_or_num);
151  uniformAbscissaPoints = true;
152  }
153  else if (PyFloat_Check(dist_or_num)) {
154  distance = PyFloat_AsDouble(dist_or_num);
155  uniformAbscissaDistance = true;
156  }
157  else {
158  PyErr_SetString(PyExc_TypeError, "Either int or float expected");
159  return nullptr;
160  }
161  }
162  else {
163  // use Number kwds
164  static const std::array<const char *, 4> kwds_numPoints {"Number", "First", "Last", nullptr};
165  PyErr_Clear();
166  if (Base::Wrapped_ParseTupleAndKeywords(args, kwds, "i|dd", kwds_numPoints, &numPoints, &first, &last)) {
167  uniformAbscissaPoints = true;
168  }
169  else {
170  // use Abscissa kwds
171  static const std::array<const char *, 4> kwds_Distance {"Distance", "First", "Last", nullptr};
172  PyErr_Clear();

178

181  if (uniformAbscissaPoints)

183  else
184  discretizer.Initialize (adapt, distance, first, last);
185

187  Py::List points;
188  int nbPoints = discretizer.NbPoints ();
189  for (int i=1; i<=nbPoints; i++) {
190  gp_Pnt p = adapt.Value (discretizer.Parameter (i));
191  points.append(Py::Vector(Base::Vector3d(p.X(),p.Y(),p.Z())));
192  }

560 PyObject* GeometryCurvePy::projectPoint(PyObject *args, PyObject* kwds)
561 {
562  PyObject *v;
563  const char *meth = "NearestPoint";
564  static const std::array<const char *, 3> kwlist{"Point", "Method", nullptr};
565  if (!Base::Wrapped_ParseTupleAndKeywords(args, kwds, "O!|s", kwlist,
566  &Base::VectorPy::Type, &v, &meth)) {
567  return nullptr;
568  }
569 
570  try {
571  Base::Vector3d vec = Py::Vector(v, false).toVector();
572  gp_Pnt pnt(vec.x, vec.y, vec.z);
573  std::string method = meth;
574 
575  Handle(Geom_Geometry) geom = getGeometryPtr()->handle();
576  Handle(Geom_Curve) curve = Handle(Geom_Curve)::DownCast(geom);
577 
578  GeomAPI_ProjectPointOnCurve proj(pnt, curve);
579  if (method == "NearestPoint") {
580  pnt = proj.NearestPoint();
581  vec.Set(pnt.X(), pnt.Y(), pnt.Z());
582  return new Base::VectorPy(vec);
583  }
584  else if (method == "LowerDistance") {
585  Py::Float dist(proj.LowerDistance());
586  return Py::new_reference_to(dist);
587  }
588  else if (method == "LowerDistanceParameter") {
589  Py::Float par(proj.LowerDistanceParameter());
590  return Py::new_reference_to(par);
591  }
592  else if (method == "Distance") {
593  Standard_Integer num = proj.NbPoints();
594  Py::List list;
595  for (Standard_Integer i=1; i <= num; i++) {
596  list.append(Py::Float(proj.Distance(i)));
597  }
598  return Py::new_reference_to(list);
599  }
600  else if (method == "Parameter") {
601  Standard_Integer num = proj.NbPoints();
602  Py::List list;
603  for (Standard_Integer i=1; i <= num; i++) {
604  list.append(Py::Float(proj.Parameter(i)));
605  }
606  return Py::new_reference_to(list);
607  }
608  else if (method == "Point") {
609  Standard_Integer num = proj.NbPoints();
610  Py::List list;
611  for (Standard_Integer i=1; i <= num; i++) {
612  gp_Pnt pnt = proj.Point(i);
613  Base::Vector3d vec(pnt.X(), pnt.Y(), pnt.Z());
