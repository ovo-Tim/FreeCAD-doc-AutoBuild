151 def threadRadii(internal, majorDia, minorDia, toolDia, toolCrest):
152  """threadRadii(majorDia, minorDia, toolDia, toolCrest) ... returns the minimum and maximum radius for thread."""
153  Path.Log.track(internal, majorDia, minorDia, toolDia, toolCrest)
154  if toolCrest is None:
155  toolCrest = 0.0
156  # As it turns out metric and imperial standard threads follow the same rules.
157  # The determining factor is the height of the full 60 degree triangle H.
158  # - The minor diameter is 1/4 * H smaller than the pitch diameter.
159  # - The major diameter is 3/8 * H bigger than the pitch diameter
160  # Since we already have the outer diameter it's simpler to just add 1/8 * H
161  # to get the outer tip of the thread.
162  H = ((majorDia - minorDia) / 2.0) * 1.6 # (D - d)/2 = 5/8 * H
163  if internal:
164  # mill inside out
165  outerTip = majorDia / 2.0 + H / 8.0
166  # Compensate for the crest of the tool
167  toolTip = outerTip - toolCrest * SQRT_3_DIVIDED_BY_2
168  radii = ((minorDia - toolDia) / 2.0, toolTip - toolDia / 2.0)
169  else:
170  # mill outside in
171  innerTip = minorDia / 2.0 - H / 4.0
172  # Compensate for the crest of the tool
173  toolTip = innerTip - toolCrest * SQRT_3_DIVIDED_BY_2
174  radii = ((majorDia + toolDia) / 2.0, toolTip + toolDia / 2.0)
175  Path.Log.track(radii)
176  return radii

179 def threadPasses(count, radii, internal, majorDia, minorDia, toolDia, toolCrest):
180  Path.Log.track(count, radii, internal, majorDia, minorDia, toolDia, toolCrest)
181  # the logic goes as follows, total area to be removed:
182  # A = H * W ... where H is the depth and W is half the width of a thread
183  # H = k * sin(30) = k * 1/2 -> k = 2 * H
184  # W = k * cos(30) = k * sqrt(3)/2
185  # -> W = (2 * H) * sqrt(3) / 2 = H * sqrt(3)
186  # A = sqrt(3) * H^2
187  # Each pass has to remove the same area
188  # An = A / count = sqrt(3) * H^2 / count
189  # Because each successive pass doesn't have to remove the aera of the previous
190  # passes the result for the height:
191  # Ai = (i + 1) * An = (i + 1) * sqrt(3) * Hi^2 = sqrt(3) * H^2 / count
192  # Hi = sqrt(H^2 * (i + 1) / count)
193  # Hi = H * sqrt((i + 1) / count)
194  minor, major = radii(internal, majorDia, minorDia, toolDia, toolCrest)
195  H = float(major - minor)
196  Hi = [H * math.sqrt((i + 1) / count) for i in range(count)]
197 
198  # For external threads threadRadii returns the radii in reverse order because that's
199  # the order in which they have to get milled. As a result H ends up being negative
200  # and the math for internal and external threads is identical.
201  passes = [minor + h for h in Hi]
202  Path.Log.debug(f"threadPasses({minor}, {major}) -> H={H} : {Hi} --> {passes}")
203 
204  return passes
