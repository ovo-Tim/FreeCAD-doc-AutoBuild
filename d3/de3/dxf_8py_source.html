63 def get_dxf(obj, direction=None):

65 
66  If direction is given, the object is projected in 2D.
67  """
68  plane = None
69  result = ""
70  if obj.isDerivedFrom("TechDraw::DrawView"):
71  if obj.Source.isDerivedFrom("App::DocumentObjectGroup"):
72  for o in obj.Source.Group:
73  result += get_dxf(o, obj.Direction)
74  else:
75  result += get_dxf(obj.Source, obj.Direction)
76  return result
77 
78  if direction and isinstance(direction, App.Vector):
79  if direction != App.Vector(0, 0, 0):
80  plane = WorkingPlane.PlaneBase()
81  plane.align_to_point_and_axis(App.Vector(0, 0, 0), direction)
82 
83  if utils.get_type(obj) in ("Dimension", "LinearDimension"):
84  p1 = _get_proj(obj.Start, plane=plane)
85  p2 = _get_proj(obj.End, plane=plane)
86  p3 = _get_proj(obj.Dimline, plane=plane)
87  result += "0\nDIMENSION\n8\n0\n62\n0\n3\nStandard\n70\n1\n"
88  result += "10\n"+str(p3.x)+"\n20\n"+str(p3.y)+"\n30\n"+str(p3.z)+"\n"
89  result += "13\n"+str(p1.x)+"\n23\n"+str(p1.y)+"\n33\n"+str(p1.z)+"\n"
90  result += "14\n"+str(p2.x)+"\n24\n"+str(p2.y)+"\n34\n"+str(p2.z)+"\n"
91 
92  elif utils.get_type(obj) == "Annotation":
93  # Only for App::Annotation
94  p = _get_proj(obj.Position, plane=plane)
95  count = 0
