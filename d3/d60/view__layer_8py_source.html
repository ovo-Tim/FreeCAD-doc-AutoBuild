424  def update_groups_after_drag_drop(self):
425  """Workaround function to improve the drag and drop behavior of Layer
426  objects.
427 
428  The function processes the parent group data stored in the
429  old_parent_data dictionary by canDragObject and canDropObject.
430  """
431 
432  # The function can be called multiple times, old_parent_data will be
433  # empty after the first call.
434  if (not hasattr(self, "old_parent_data")) or (not self.old_parent_data):
435  return
436 
437  # List to collect parents whose Group must be updated.
438  # This has to happen later in a separate loop as we need the unmodified
439  # InList properties of the children in the main loop.
440  parents_to_update = []
441 
442  # Main loop:
443  for child, old_data in self.old_parent_data.items():
444 
445  # We assume a single old and a single new layer...
446 
447  old_layer = None
448  for old_parent, old_parent_group in old_data:
449  if utils.get_type(old_parent) == "Layer":
450  old_layer = old_parent
451  break
452 
453  new_layer = self._get_layer(child)
454  if new_layer == old_layer:
455  continue
456 
457  elif new_layer is None:
458  # An object was dragged out of a layer.
459  # We need to check if it was put in a new group. If that is
460  # the case the content of old_layer should be restored.
461  # If the object was not put in a new group it was dropped on
462  # the document node, in that case we do nothing.
463  old_parents = [sub[0] for sub in old_data]
464  for new_parent in child.InList:
465  if (hasattr(new_parent, "Group")
466  and new_parent not in old_parents): # New group check.
467  for old_parent, old_parent_group in old_data:
468  if old_parent == old_layer:
469  parents_to_update.append([old_parent, old_parent_group])
470  break
471  break
472 
473  else:
474  # A new layer was assigned.
475  # The content of all `non-layer` groups should be restored.
476  for old_parent, old_parent_group in old_data:
477  if utils.get_type(old_parent) != "Layer":
478  parents_to_update.append([old_parent, old_parent_group])
479 
480  # Update parents:
481  if parents_to_update:
482  for old_parent, old_parent_group in parents_to_update:
483  old_parent.Group = old_parent_group
484  App.ActiveDocument.recompute()
485 
486  self.old_parent_data = {}

550  def merge_by_name(self):
551  """Merge the layers that have the same base label."""
552  doc = App.ActiveDocument
553  doc.openTransaction(translate("draft", "Merge layer duplicates"))
554 
555  layer_container = self.Object
556  layers = []
557  for obj in layer_container.Group:
558  if utils.get_type(obj) == "Layer":
559  layers.append(obj)
560 
561  to_delete = []
562  for layer in layers:
563  # Remove trailing digits (usually 3 but there might be more) and
564  # trailing spaces from Label before comparing:
565  base_label = layer.Label.rstrip("0123456789 ")
566 
567  # Try to find the `'base'` layer:
568  base = None
569  for other_layer in layers:
570  if ((not other_layer in to_delete) # Required if there are duplicate labels.
571  and other_layer != layer
572  and other_layer.Label.upper() == base_label.upper()):
573  base = other_layer
574  break
575 
576  if base:
577  if layer.Group:
578  base_group = base.Group
579  for obj in layer.Group:
580  if not obj in base_group:
581  base_group.append(obj)
582  base.Group = base_group
583  to_delete.append(layer)
584  elif layer.Label != base_label:
585  layer.Label = base_label
586 
587  for layer in to_delete:
588  doc.removeObject(layer.Name)
