43 class Draft_Snap_Base():
44  """Base Class inherited by all Draft Snap commands."""
45 
46  def Activated(self, status=0):
47  # _log("GuiCommand: {}".format(self.__class__.__name__))
48 
49  if hasattr(Gui, "Snapper"):
50  Gui.Snapper.toggle_snap(self.__class__.__name__[11:], bool(status))
51 
52  def IsActive(self):
53  return hasattr(Gui, "Snapper") and Gui.Snapper.isEnabled("Lock")
54 
55  def isChecked(self):
56  """Return true if the given snap is active in Snapper."""
57  return hasattr(Gui, "Snapper") and self.__class__.__name__[11:] in Gui.Snapper.active_snaps
