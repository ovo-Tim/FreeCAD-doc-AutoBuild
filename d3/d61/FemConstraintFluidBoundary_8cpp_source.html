110 void ConstraintFluidBoundary::onChanged(const App::Property* prop)
111 {
112  // Note: If we call this at the end, then the arrows are not oriented correctly initially
113  // because the NormalDirection has not been calculated yet
114  Constraint::onChanged(prop);
115 
116  if (prop == &BoundaryType) {
117  std::string boundaryType = BoundaryType.getValueAsString();
118  if (boundaryType == "wall") {
119  Subtype.setEnums(WallSubtypes);
120  }
121  else if (boundaryType == "interface") {
122  Subtype.setEnums(InterfaceSubtypes);
123  }
124  else if (boundaryType == "freestream") {
125  Subtype.setEnums(FreestreamSubtypes);
126  }
127  else if (boundaryType == "inlet") {
128  Subtype.setEnums(InletSubtypes);
129  }
130  else if (boundaryType == "outlet") {
131  Subtype.setEnums(OutletSubtypes);
132  }
133  else {
134  Base::Console().Message(boundaryType.c_str());
135  Base::Console().Message(" Error: this boundaryType is not defined\n");
136  }
137 
138  // must set a default (0 or 1) as freestream has only 2 subtypes
139  Subtype.setValue(1);
140  // need to trigger ViewProvider::updateData() for redraw in 3D view after this method
141  }
142  else if (prop == &Direction) {
143  Base::Vector3d direction = getDirection(Direction);
144  // if Direct has no link provided return Base::Vector3d(0,0,0);
145  if (direction.Length() < Precision::Confusion()) {
146  return;
147  }
148  naturalDirectionVector = direction;
149  if (Reversed.getValue()) {
150  direction = -direction;
151  }
152  DirectionVector.setValue(direction);
153  }
154  else if (prop == &Reversed) {
155  // if the direction is invalid try to compute it again
156  if (naturalDirectionVector.Length() < Precision::Confusion()) {
157  naturalDirectionVector = getDirection(Direction);
158  }
159  if (naturalDirectionVector.Length() >= Precision::Confusion()) {
160  if (Reversed.getValue() && (DirectionVector.getValue() == naturalDirectionVector)) {
161  DirectionVector.setValue(-naturalDirectionVector);
162  }
163  else if (!Reversed.getValue()
164  && (DirectionVector.getValue() != naturalDirectionVector)) {
165  DirectionVector.setValue(naturalDirectionVector);
166  }
167  }
168  }
169  else if (prop == &NormalDirection) {
170  // Set a default direction if no direction reference has been given
171  if (!Direction.getValue()) {
172  Base::Vector3d direction = NormalDirection.getValue();
173  if (Reversed.getValue()) {
174  direction = -direction;
175  }
176  DirectionVector.setValue(direction);
177  naturalDirectionVector = direction;
178  }
179  }
180 }
