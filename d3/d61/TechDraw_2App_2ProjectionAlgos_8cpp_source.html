57  const TopoDS_Shape& build3dCurves(const TopoDS_Shape &shape)
58  {
59  TopExp_Explorer it;
60  for (it.Init(shape, TopAbs_EDGE); it.More(); it.Next())
61  BRepLib::BuildCurve3d(TopoDS::Edge(it.Current()));
62  return shape;
63  }

77 void ProjectionAlgos::execute()
78 {
79  Handle( HLRBRep_Algo ) brep_hlr = new HLRBRep_Algo;
80  brep_hlr->Add(Input);
81 
82  gp_Ax2 transform(gp_Pnt(0, 0, 0), gp_Dir(Direction.x, Direction.y, Direction.z));
83  HLRAlgo_Projector projector( transform );
84  brep_hlr->Projector(projector);
85  brep_hlr->Update();
86  brep_hlr->Hide();
87 
88  // extracting the result sets:
89  HLRBRep_HLRToShape shapes( brep_hlr );
90 
91  V = build3dCurves(shapes.VCompound ());// hard edge visibly
92  V1 = build3dCurves(shapes.Rg1LineVCompound());// Smoth edges visibly
93  VN = build3dCurves(shapes.RgNLineVCompound());// contour edges visibly
94  VO = build3dCurves(shapes.OutLineVCompound());// contours apparents visibly
95  VI = build3dCurves(shapes.IsoLineVCompound());// isoparamtriques visibly
96  H = build3dCurves(shapes.HCompound ());// hard edge invisibly
97  H1 = build3dCurves(shapes.Rg1LineHCompound());// Smoth edges invisibly
98  HN = build3dCurves(shapes.RgNLineHCompound());// contour edges invisibly
99  HO = build3dCurves(shapes.OutLineHCompound());// contours apparents invisibly
100  HI = build3dCurves(shapes.IsoLineHCompound());// isoparamtriques invisibly
101 }
