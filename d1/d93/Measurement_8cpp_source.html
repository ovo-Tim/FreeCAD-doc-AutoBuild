102 MeasureType Measurement::findType()
103 {
104  const std::vector<App::DocumentObject*>& objects = References3D.getValues();
105  const std::vector<std::string>& subElements = References3D.getSubValues();
106 
107  std::vector<App::DocumentObject*>::const_iterator obj = objects.begin();
108  std::vector<std::string>::const_iterator subEl = subElements.begin();
109 
110  MeasureType mode;
111 
112  int verts = 0;
113  int edges = 0;
114  int lines = 0;
115  int circles = 0;
116  int faces = 0;
117  int planes = 0;
118  int cylinders = 0;
119  int cones = 0;
120  int torus = 0;
121  int spheres = 0;
122  int vols = 0;
123  int other = 0;
124 
125  for (; obj != objects.end(); ++obj, ++subEl) {
126  TopoDS_Shape refSubShape;
127  try {
128  refSubShape = Part::Feature::getShape(*obj, (*subEl).c_str(), true);
129  if (refSubShape.IsNull()) {
130  return MeasureType::Invalid;
131  }
132  }
133  catch (Standard_Failure& e) {
134  std::stringstream errorMsg;
135 
136  errorMsg << "Measurement - getType - " << e.GetMessageString() << std::endl;
137  throw Base::CADKernelError(e.GetMessageString());
138  }
139 
140  switch (refSubShape.ShapeType()) {
141  case TopAbs_VERTEX: {
142  verts++;
143  } break;
144  case TopAbs_EDGE: {
145  edges++;
146  TopoDS_Edge edge = TopoDS::Edge(refSubShape);
147  BRepAdaptor_Curve sf(edge);
148 
149  if (sf.GetType() == GeomAbs_Line) {
150  lines++;
151  }
152  else if (sf.GetType() == GeomAbs_Circle) {
153  circles++;
154  }
155  } break;
156  case TopAbs_FACE: {
157  faces++;
158  TopoDS_Face face = TopoDS::Face(refSubShape);
159  BRepAdaptor_Surface sf(face);
160 
161  if (sf.GetType() == GeomAbs_Plane) {
162  planes++;
163  }
164  else if (sf.GetType() == GeomAbs_Cylinder) {
165  cylinders++;
166  }
167  else if (sf.GetType() == GeomAbs_Sphere) {
168  spheres++;
169  }
170  else if (sf.GetType() == GeomAbs_Cone) {
171  cones++;
172  }
173  else if (sf.GetType() == GeomAbs_Torus) {
174  torus++;
175  }
176  } break;
177  case TopAbs_SOLID: {
178  vols++;
179  } break;
180  default:
181  other++;
182  break;
183  }
184  }
185 
186  if (other > 0) {
187  mode = MeasureType::Invalid;
188  }
189  else if (vols > 0) {
190  if (verts > 0 || edges > 0 || faces > 0) {
191  mode = MeasureType::Invalid;
192  }
193  else {
194  mode = MeasureType::Volumes;
195  }
196  }
197  else if (faces > 0) {
198  if (verts > 0 || edges > 0) {
199  if (faces == 1 && verts == 1) {
200  mode = MeasureType::PointToSurface;
201  }
202  else {
203  mode = MeasureType::Invalid;
204  }
205  }
206  else {
207  if (planes == 1 && faces == 1) {
208  mode = MeasureType::Plane;
209  }
210  else if (planes == 2 && faces == 2) {
211  if (planesAreParallel()) {
212  mode = MeasureType::TwoPlanes;
213  }
214  else {
215  mode = MeasureType::Surfaces;
216  }
217  }
218  else if (cylinders == 1 && faces == 1) {
219  mode = MeasureType::Cylinder;
220  }
221  else if (cones == 1 && faces == 1) {
222  mode = MeasureType::Cone;
223  }
224  else if (spheres == 1 && faces == 1) {
225  mode = MeasureType::Sphere;
226  }
227  else if (torus == 1 && faces == 1) {
228  mode = MeasureType::Torus;
229  }
230  else {
231  mode = MeasureType::Surfaces;
232  }
233  }
234  }
235  else if (edges > 0) {
236  if (verts > 0) {
237  if (verts > 1 && edges > 0) {
238  mode = MeasureType::Invalid;
239  }
240  else {
241  mode = MeasureType::PointToEdge;
242  }
243  }
244  else if (lines == 1 && edges == 1) {
245  mode = MeasureType::Line;
246  }
247  else if (lines == 2 && edges == 2) {
248  if (linesAreParallel()) {
249  mode = MeasureType::TwoParallelLines;
250  }
251  else {
252  mode = MeasureType::TwoLines;
253  }
254  }
255  else if (circles == 1 && edges == 1) {
256  mode = MeasureType::Circle;
257  }
258  else {
259  mode = MeasureType::Edges;
260  }
261  }
262  else if (verts > 0) {
263  if (verts == 2) {
264  mode = MeasureType::PointToPoint;
265  }
266  else {
267  mode = MeasureType::Points;
268  }
269  }
270  else {
271  mode = MeasureType::Invalid;
272  }
273 
274  return mode;
275 }

288 // TODO:: add lengthX, lengthY (and lengthZ??) support
289 // Methods for distances (edge length, two points, edge and a point
290 double Measurement::length() const
291 {
292  double result = 0.0;
293  int numRefs = References3D.getSize();
294  if (numRefs == 0) {
295  Base::Console().Error("Measurement::length - No 3D references available\n");
296  }
297  else if (measureType == MeasureType::Invalid) {
298  Base::Console().Error("Measurement::length - measureType is Invalid\n");
299  }
300  else {
301  const std::vector<App::DocumentObject*>& objects = References3D.getValues();
302  const std::vector<std::string>& subElements = References3D.getSubValues();
303 
304  if (measureType == MeasureType::Points || measureType == MeasureType::PointToPoint
305  || measureType == MeasureType::PointToEdge
306  || measureType == MeasureType::PointToSurface) {
307 
308  Base::Vector3d diff = this->delta();
309  result = diff.Length();
310  }
311  else if (measureType == MeasureType::Edges || measureType == MeasureType::Line
312  || measureType == MeasureType::TwoLines || measureType == MeasureType::Circle) {
313 
314  // Iterate through edges and calculate each length
315  std::vector<App::DocumentObject*>::const_iterator obj = objects.begin();
316  std::vector<std::string>::const_iterator subEl = subElements.begin();
317 
318  for (; obj != objects.end(); ++obj, ++subEl) {
319 
320  // Get the length of one edge
321  TopoDS_Shape shape = getShape(*obj, (*subEl).c_str());
322  const TopoDS_Edge& edge = TopoDS::Edge(shape);
323  BRepAdaptor_Curve curve(edge);
324 
325  switch (curve.GetType()) {
326  case GeomAbs_Line: {
327  gp_Pnt P1 = curve.Value(curve.FirstParameter());
328  gp_Pnt P2 = curve.Value(curve.LastParameter());
329  gp_XYZ diff = P2.XYZ() - P1.XYZ();
330  result += diff.Modulus();
331  break;
332  }
333  case GeomAbs_Circle: {
334  double u = curve.FirstParameter();
335  double v = curve.LastParameter();
336  double radius = curve.Circle().Radius();
337  if (u > v) { // if arc is reversed
338  std::swap(u, v);
339  }
340 
341  double range = v - u;
342  result += radius * range;
343  break;
344  }
345  case GeomAbs_Ellipse:
346  case GeomAbs_BSplineCurve:
347  case GeomAbs_Hyperbola:
348  case GeomAbs_BezierCurve: {
349  result += GCPnts_AbscissaPoint::Length(curve);
350  break;
351  }
352  default: {
353  throw Base::RuntimeError(
354  "Measurement - length - Curve type not currently handled");
355  }
356  } // end switch
357  } // end for
358  }
359  }
360  return result;
361 }

363 double Measurement::lineLineDistance() const
364 {
365  // We don't use delta() because BRepExtrema_DistShapeShape return minimum length between line
366  // segment. Here we get the nominal distance between the infinite lines.
367  double distance = 0.0;
368 
369  if (measureType != MeasureType::TwoParallelLines || References3D.getSize() != 2) {
370  return distance;
371  }
372 
373  const std::vector<App::DocumentObject*>& objects = References3D.getValues();
374  const std::vector<std::string>& subElements = References3D.getSubValues();
375 
376  // Get the first line
377  TopoDS_Shape shape1 = getShape(objects[0], subElements[0].c_str());
378  const TopoDS_Edge& edge1 = TopoDS::Edge(shape1);
379  BRepAdaptor_Curve curve1(edge1);
380 
381  // Get the second line
382  TopoDS_Shape shape2 = getShape(objects[1], subElements[1].c_str());
383  const TopoDS_Edge& edge2 = TopoDS::Edge(shape2);
384  BRepAdaptor_Curve curve2(edge2);
385 
386  if (curve1.GetType() == GeomAbs_Line && curve2.GetType() == GeomAbs_Line) {
387  gp_Lin line1 = curve1.Line();
388  gp_Lin line2 = curve2.Line();
389 
390  gp_Pnt p1 = line1.Location();
391  gp_Pnt p2 = line2.Location();
392 
393  // Create a vector from a point on line1 to a point on line2
394  gp_Vec lineVec(p1, p2);
395 
396  // The direction vector of one of the lines
397  gp_Dir lineDir = line1.Direction();
398 
399  // Project lineVec onto lineDir
400  gp_Vec parallelComponent = lineVec.Dot(lineDir) * lineDir;
401 
402  // Compute the perpendicular component
403  gp_Vec perpendicularComponent = lineVec - parallelComponent;
404 
405  // Distance is the magnitude of the perpendicular component
406  distance = perpendicularComponent.Magnitude();
407  }
408  else {
409  Base::Console().Error("Measurement::length - TwoLines measureType requires two lines\n");
410  }
411  return distance;
412 }

414 double Measurement::planePlaneDistance() const
415 {
416  if (measureType != MeasureType::TwoPlanes || References3D.getSize() != 2) {
417  return 0.0;
418  }
419 
420  const auto& objects = References3D.getValues();
421  const auto& subElements = References3D.getSubValues();
422 
423  std::vector<gp_Pln> planes;
424 
425  // Get the first plane
426  TopoDS_Shape shape1 = getShape(objects[0], subElements[0].c_str());
427  const TopoDS_Face& face1 = TopoDS::Face(shape1);
428  BRepAdaptor_Surface surface1(face1);
429  const gp_Pln& plane1 = surface1.Plane();
430 
431  // Get the second plane
432  TopoDS_Shape shape2 = getShape(objects[1], subElements[1].c_str());
433  const TopoDS_Face& face2 = TopoDS::Face(shape2);
434  BRepAdaptor_Surface surface2(face2);
435  const gp_Pln& plane2 = surface2.Plane();
436 
437  // Distance between two parallel planes
438  gp_Pnt pointOnPlane1 = plane1.Location();
439  gp_Dir normalToPlane1 = plane1.Axis().Direction();
440 
441  gp_Pnt pointOnPlane2 = plane2.Location();
442 
443  // Create a vector from a point on plane1 to a point on plane2
444  gp_Vec vectorBetweenPlanes(pointOnPlane1, pointOnPlane2);
445 
446  // Project this vector onto the plane normal
447  double distance = Abs(vectorBetweenPlanes.Dot(normalToPlane1));
448 
449  return distance;
450 }

452 double Measurement::angle(const Base::Vector3d& /*param*/) const
453 {
454  // TODO: do these references arrive as obj+sub pairs or as a struct of obj + [subs]?
455  const std::vector<App::DocumentObject*>& objects = References3D.getValues();
456  const std::vector<std::string>& subElements = References3D.getSubValues();
457  int numRefs = objects.size();
458  if (numRefs == 0) {
459  throw Base::RuntimeError("No references available for angle measurement");
460  }
461  else if (measureType == MeasureType::Invalid) {
462  throw Base::RuntimeError("MeasureType is Invalid for angle measurement");
463  }
464  else if (measureType == MeasureType::TwoLines) {
465  // Only case that is supported is edge to edge
466  // The angle between two skew lines is measured by the angle between one line (A)
467  // and a line (B) with the direction of the second through a point on the first line.
468  // Since we don't know if the directions of the lines point in the same general direction
469  // we could get the angle we want or the supplementary angle.
470  if (numRefs == 2) {
471  TopoDS_Shape shape1 = getShape(objects.at(0), subElements.at(0).c_str());
472  TopoDS_Shape shape2 = getShape(objects.at(1), subElements.at(1).c_str());
473 
474  BRepAdaptor_Curve curve1(TopoDS::Edge(shape1));
475  BRepAdaptor_Curve curve2(TopoDS::Edge(shape2));
476 
477  if (curve1.GetType() == GeomAbs_Line && curve2.GetType() == GeomAbs_Line) {
478 
479  gp_Pnt pnt1First = curve1.Value(curve1.FirstParameter());
480  gp_Dir dir1 = curve1.Line().Direction();
481  gp_Dir dir2 = curve2.Line().Direction();
482  gp_Dir dir2r = curve2.Line().Direction().Reversed();
483 
484  gp_Lin l1 = gp_Lin(pnt1First, dir1); // (A)
485  gp_Lin l2 = gp_Lin(pnt1First, dir2); // (B)
486  gp_Lin l2r = gp_Lin(pnt1First, dir2r); // (B')
487  Standard_Real aRad = l1.Angle(l2);
488  double aRadr = l1.Angle(l2r);
489  return Base::toDegrees<double>(std::min(aRad, aRadr));
490  }
491  else {
492  throw Base::RuntimeError("Measurement references must both be lines");
493  }
494  }
495  else {
496  throw Base::RuntimeError("Can not compute angle measurement - too many references");
497  }
498  }
499  else if (measureType == MeasureType::Points) {
500  // NOTE: we are calculating the 3d angle here, not the projected angle
501  // ASSUMPTION: the references are in end-apex-end order
502  if (numRefs == 3) {
503  TopoDS_Shape shape0 = getShape(objects.at(0), subElements.at(0).c_str());
504  TopoDS_Shape shape1 = getShape(objects.at(1), subElements.at(1).c_str());
505  TopoDS_Shape shape2 = getShape(objects.at(1), subElements.at(2).c_str());
506  if (shape0.ShapeType() != TopAbs_VERTEX || shape1.ShapeType() != TopAbs_VERTEX
507  || shape2.ShapeType() != TopAbs_VERTEX) {
508  throw Base::RuntimeError("Measurement references for 3 point angle are not Vertex");
509  }
510  gp_Pnt gEnd0 = BRep_Tool::Pnt(TopoDS::Vertex(shape0));
511  gp_Pnt gApex = BRep_Tool::Pnt(TopoDS::Vertex(shape1));
512  gp_Pnt gEnd1 = BRep_Tool::Pnt(TopoDS::Vertex(shape2));
513  gp_Dir gDir0 = gp_Dir(gEnd0.XYZ() - gApex.XYZ());
514  gp_Dir gDir1 = gp_Dir(gEnd1.XYZ() - gApex.XYZ());
515  gp_Lin line0 = gp_Lin(gEnd0, gDir0);
516  gp_Lin line1 = gp_Lin(gEnd1, gDir1);
517  double radians = line0.Angle(line1);
518  return Base::toDegrees<double>(radians);
519  }
520  }
521  throw Base::RuntimeError("Unexpected error for angle measurement");
522 }

524 double Measurement::radius() const

527  const std::vector<std::string>& subElements = References3D.getSubValues();
528 
529  int numRefs = References3D.getSize();
530  if (numRefs == 0) {
531  Base::Console().Error("Measurement::radius - No 3D references available\n");

535  const TopoDS_Edge& edge = TopoDS::Edge(shape);

541  }

562 Base::Vector3d Measurement::delta() const
563 {
564  Base::Vector3d result;
565  int numRefs = References3D.getSize();
566  if (numRefs == 0) {
567  Base::Console().Error("Measurement::delta - No 3D references available\n");
568  }
569  else if (measureType == MeasureType::Invalid) {
570  Base::Console().Error("Measurement::delta - measureType is Invalid\n");
571  }
572  else {
573  const std::vector<App::DocumentObject*>& objects = References3D.getValues();
574  const std::vector<std::string>& subElements = References3D.getSubValues();
575 
576  if (measureType == MeasureType::PointToPoint) {
577  if (numRefs == 2) {
578  // Keep separate case for two points to reduce need for complex algorithm
579  TopoDS_Shape shape1 = getShape(objects.at(0), subElements.at(0).c_str());
580  TopoDS_Shape shape2 = getShape(objects.at(1), subElements.at(1).c_str());
581 
582  const TopoDS_Vertex& vert1 = TopoDS::Vertex(shape1);
583  const TopoDS_Vertex& vert2 = TopoDS::Vertex(shape2);
584 
585  gp_Pnt P1 = BRep_Tool::Pnt(vert1);
586  gp_Pnt P2 = BRep_Tool::Pnt(vert2);
587  gp_XYZ diff = P2.XYZ() - P1.XYZ();
588  return Base::Vector3d(diff.X(), diff.Y(), diff.Z());
589  }
590  }
591  else if (measureType == MeasureType::PointToEdge
592  || measureType == MeasureType::PointToSurface) {
593  // BrepExtema can calculate minimum distance between any set of topology sets.
594  if (numRefs == 2) {
595  TopoDS_Shape shape1 = getShape(objects.at(0), subElements.at(0).c_str());
596  TopoDS_Shape shape2 = getShape(objects.at(1), subElements.at(1).c_str());
597 
598  BRepExtrema_DistShapeShape extrema(shape1, shape2);
599 
600  if (extrema.IsDone()) {
601  // Found the nearest point between point and curve
602  // NOTE we will assume there is only 1 solution (cyclic topology will create
603  // multiple solutions.
604  gp_Pnt P1 = extrema.PointOnShape1(1);
605  gp_Pnt P2 = extrema.PointOnShape2(1);
606  gp_XYZ diff = P2.XYZ() - P1.XYZ();
607  result = Base::Vector3d(diff.X(), diff.Y(), diff.Z());
608  }
609  }
610  }
611  else if (measureType == MeasureType::Edges) {
612  // Only case that is supported is straight line edge
613  if (numRefs == 1) {
614  TopoDS_Shape shape = getShape(objects.at(0), subElements.at(0).c_str());
615  const TopoDS_Edge& edge = TopoDS::Edge(shape);
616  BRepAdaptor_Curve curve(edge);
617 
618  if (curve.GetType() == GeomAbs_Line) {
619  gp_Pnt P1 = curve.Value(curve.FirstParameter());
620  gp_Pnt P2 = curve.Value(curve.LastParameter());
621  gp_XYZ diff = P2.XYZ() - P1.XYZ();
622  result = Base::Vector3d(diff.X(), diff.Y(), diff.Z());
623  }
624  }
625  else if (numRefs == 2) {
626  TopoDS_Shape shape1 = getShape(objects.at(0), subElements.at(0).c_str());
627  TopoDS_Shape shape2 = getShape(objects.at(1), subElements.at(1).c_str());
628 
629  BRepAdaptor_Curve curve1(TopoDS::Edge(shape1));
630  BRepAdaptor_Curve curve2(TopoDS::Edge(shape2));
631 
632  // Only permit line to line distance
633  if (curve1.GetType() == GeomAbs_Line && curve2.GetType() == GeomAbs_Line) {
634  BRepExtrema_DistShapeShape extrema(shape1, shape2);
635 
636  if (extrema.IsDone()) {
637  // Found the nearest point between point and curve
638  // NOTE we will assume there is only 1 solution (cyclic topology will create
639  // multiple solutions.
640  gp_Pnt P1 = extrema.PointOnShape1(1);
641  gp_Pnt P2 = extrema.PointOnShape2(1);
