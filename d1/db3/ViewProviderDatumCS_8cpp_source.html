212 void ViewProviderDatumCoordinateSystem::setExtents (Base::BoundBox3d bbox) {
213  // Axis length of the CS is 1/3 of maximum bbox dimension, any smarter sizing will make it only worse
214  double axisLength;
215  // Empty gap at the origin of the coordinate system, this is a nice experimental value
216  double centerGap;
217 
218  if(Zoom.getValue()) {
219  axisLength = 6 * Zoom.getValue();
220  }else{
221  axisLength = std::max ( { bbox.LengthX (), bbox.LengthY(), bbox.LengthZ() } );
222  axisLength *= (1 + marginFactor ()) / 3;
223  }
224 
225  centerGap = axisLength / 8.;
226 
227  coord->point.set1Value ( 0, 0, 0, 0 );
228  coord->point.set1Value ( 1, centerGap, 0, 0 );
229  coord->point.set1Value ( 2, axisLength, 0, 0 );
230  coord->point.set1Value ( 3, 0, centerGap, 0 );
231  coord->point.set1Value ( 4, 0, axisLength, 0 );
232  coord->point.set1Value ( 5, 0, 0, centerGap );
233  coord->point.set1Value ( 6, 0, 0, axisLength );
234 
235  double labelPos = axisLength;
236  double labelOffset = 0;
237 
238  // offset 1 pixel
239  axisLabelXTrans->translation.setValue ( SbVec3f( labelPos, labelOffset, 0) );
240  axisLabelXToYTrans->translation.setValue ( SbVec3f( -labelPos + labelOffset, labelPos - labelOffset, 0) );
241  axisLabelYToZTrans->translation.setValue ( SbVec3f( -labelOffset, -labelPos + labelOffset, labelPos) );
242 }
