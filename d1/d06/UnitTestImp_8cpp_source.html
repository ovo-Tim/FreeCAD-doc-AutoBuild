133 void UnitTestDialog::onTreeViewFailureItemDoubleClicked(QTreeWidgetItem* item, int column)
134 {
135  Q_UNUSED(column);
136 
137  QString text = item->data(0, Qt::UserRole).toString();
138 
139  QMessageBox msgBox(this);
140  msgBox.setIcon(QMessageBox::Information);
141  msgBox.setWindowTitle(item->text(0));
142  msgBox.setDetailedText(text);
143 
144  // truncate the visible text when it's too long
145  if (text.count(QLatin1Char('\n')) > 20) {
146  QStringList lines = text.split(QLatin1Char('\n'));
147  lines.erase(lines.begin() + 20, lines.end());
148  text = lines.join(QLatin1String("\n"));
149  }
150  if (text.size() > 1000) {
151  text = text.left(1000);
152  }
153 
154  msgBox.setText(text);
155  msgBox.exec();
156 }

188 void UnitTestDialog::onStartButtonClicked()
189 {
190  reset();
191  setProgressColor(QColor(40, 210, 43)); // a darker green
192  ui->startButton->setDisabled(true);
193  try {
194  Base::Interpreter().runString("import qtunittest, gc\n"
195  "__qt_test__=qtunittest.QtTestRunner(0,\"\")\n"
196  "__qt_test__.runClicked()\n"
197  "del __qt_test__\n"
198  "gc.collect()\n");
199  }
200  catch (const Base::PyException& e) {
201  std::string msg = e.what();
202  msg += "\n\n";
203  msg += e.getStackTrace();
204  showErrorDialog("Exception", msg.c_str());
205  }
206  catch (const Base::Exception& e) {
207  showErrorDialog("Exception", e.what());
208  }
209  catch (const std::exception& e) {
210  showErrorDialog("C++ standard exception", e.what());
211  }
212  catch (...) {
213  showErrorDialog("Unknown exception", "Unknown exception raised");
214  }
215  ui->startButton->setEnabled(true);
216 }
