811 bool TaskBoxPrimitives::setPrimitive(App::DocumentObject* obj)
812 {
813  try {
814  App::Document* doc = App::GetApplication().getActiveDocument();
815  if (!doc) {
816  return false;
817  }
818 
819  std::string cmd;
820  std::string name(Gui::Command::getObjectCmd(obj));
821  Base::QuantityFormat format(Base::QuantityFormat::Fixed, Base::UnitsApi::getDecimals());
822  switch (ui->widgetStack->currentIndex()) {
823  case 1: // box
824  cmd = fmt::format("{0}.Length='{1}'\n"
825  "{0}.Width='{2}'\n"
826  "{0}.Height='{3}'\n",
827  name,
828  ui->boxLength->value().getSafeUserString(),
829  ui->boxWidth->value().getSafeUserString(),
830  ui->boxHeight->value().getSafeUserString());
831  break;
832 
833  case 2: // cylinder
834  cmd = fmt::format("{0}.Radius='{1}'\n"
835  "{0}.Height='{2}'\n"
836  "{0}.Angle='{3}'\n"
837  "{0}.FirstAngle='{4}'\n"
838  "{0}.SecondAngle='{5}'\n",
839  name,
840  ui->cylinderRadius->value().getSafeUserString(),
841  ui->cylinderHeight->value().getSafeUserString(),
842  ui->cylinderAngle->value().getSafeUserString(),
843  ui->cylinderXSkew->value().getSafeUserString(),
844  ui->cylinderYSkew->value().getSafeUserString());
845  break;
846 
847  case 3: // cone
848  cmd = fmt::format("{0}.Radius1='{1}'\n"
849  "{0}.Radius2='{2}'\n"
850  "{0}.Height='{3}'\n"
851  "{0}.Angle='{4}'\n",
852  name,
853  ui->coneRadius1->value().getSafeUserString(),
854  ui->coneRadius2->value().getSafeUserString(),
855  ui->coneHeight->value().getSafeUserString(),
856  ui->coneAngle->value().getSafeUserString());
857  break;
858 
859  case 4: // sphere
860  cmd = fmt::format("{0}.Radius='{1}'\n"
861  "{0}.Angle1='{2}'\n"
862  "{0}.Angle2='{3}'\n"
863  "{0}.Angle3='{4}'\n",
864  name,
865  ui->sphereRadius->value().getSafeUserString(),
866  ui->sphereAngle1->value().getSafeUserString(),
867  ui->sphereAngle2->value().getSafeUserString(),
868  ui->sphereAngle3->value().getSafeUserString());
869  break;
870  case 5: // ellipsoid
871  cmd = fmt::format("{0}.Radius1='{1}'\n"
872  "{0}.Radius2='{2}'\n"
873  "{0}.Radius3='{3}'\n"
874  "{0}.Angle1='{4}'\n"
875  "{0}.Angle2='{5}'\n"
876  "{0}.Angle3='{6}'\n",
877  name,
878  ui->ellipsoidRadius1->value().getSafeUserString(),
879  ui->ellipsoidRadius2->value().getSafeUserString(),
880  ui->ellipsoidRadius3->value().getSafeUserString(),
881  ui->ellipsoidAngle1->value().getSafeUserString(),
882  ui->ellipsoidAngle2->value().getSafeUserString(),
883  ui->ellipsoidAngle3->value().getSafeUserString());
884  break;
885 
886  case 6: // torus
887  cmd = fmt::format("{0}.Radius1='{1}'\n"
888  "{0}.Radius2='{2}'\n"
889  "{0}.Angle1='{3}'\n"
890  "{0}.Angle2='{4}'\n"
891  "{0}.Angle3='{5}'\n",
892  name,
893  ui->torusRadius1->value().getSafeUserString(),
894  ui->torusRadius2->value().getSafeUserString(),
895  ui->torusAngle1->value().getSafeUserString(),
896  ui->torusAngle2->value().getSafeUserString(),
897  ui->torusAngle3->value().getSafeUserString());
898  break;
899  case 7: // prism
900  cmd = fmt::format("{0}.Polygon={1}\n"
901  "{0}.Circumradius='{2}'\n"
902  "{0}.Height='{3}'\n"
903  "{0}.FirstAngle='{4}'\n"
904  "{0}.SecondAngle='{5}'\n",
905  name,
906  ui->prismPolygon->value(),
907  ui->prismCircumradius->value().getSafeUserString(),
908  ui->prismHeight->value().getSafeUserString(),
909  ui->prismXSkew->value().getSafeUserString(),
910  ui->prismYSkew->value().getSafeUserString());
911  break;
912  case 8: // wedge
913  // Xmin/max, Ymin/max and Zmin/max must each not be equal
914  if (ui->wedgeXmin->value().getValue() == ui->wedgeXmax->value().getValue()) {
915  QMessageBox::warning(Gui::getMainWindow(),
916  tr("Invalid wedge parameters"),
917  tr("X min must not be equal to X max!"));
918  return false;
919  }
920  else if (ui->wedgeYmin->value().getValue() == ui->wedgeYmax->value().getValue()) {
921  QMessageBox::warning(Gui::getMainWindow(),
922  tr("Invalid wedge parameters"),
923  tr("Y min must not be equal to Y max!"));
924  return false;
925  }
926  else if (ui->wedgeZmin->value().getValue() == ui->wedgeZmax->value().getValue()) {
927  QMessageBox::warning(Gui::getMainWindow(),
928  tr("Invalid wedge parameters"),
929  tr("Z min must not be equal to Z max!"));
930  return false;
931  }
932  cmd = fmt::format("{0}.Xmin='{1}'\n"
933  "{0}.Ymin='{2}'\n"
934  "{0}.Zmin='{3}'\n"
935  "{0}.X2min='{4}'\n"
936  "{0}.Z2min='{5}'\n"
937  "{0}.Xmax='{6}'\n"
938  "{0}.Ymax='{7}'\n"
939  "{0}.Zmax='{8}'\n"
940  "{0}.X2max='{9}'\n"
941  "{0}.Z2max='{10}'\n",
942  name,
943  ui->wedgeXmin->value().getSafeUserString(),
944  ui->wedgeYmin->value().getSafeUserString(),
945  ui->wedgeZmin->value().getSafeUserString(),
946  ui->wedgeX2min->value().getSafeUserString(),
947  ui->wedgeZ2min->value().getSafeUserString(),
948  ui->wedgeXmax->value().getSafeUserString(),
949  ui->wedgeYmax->value().getSafeUserString(),
950  ui->wedgeZmax->value().getSafeUserString(),
951  ui->wedgeX2max->value().getSafeUserString(),
952  ui->wedgeZ2max->value().getSafeUserString());
953  break;
954 
955  default:
956  break;
957  }
958 
959  // Execute the Python block
960  // No need to open a transaction because this is already done in the command
961  // class or when starting to edit a primitive.
962  Gui::Command::runCommand(Gui::Command::Doc, cmd.c_str());
963  Gui::Command::runCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
964  }
965  catch (const Base::PyException& e) {
966  QMessageBox::warning(this,
967  tr("Create primitive"),
968  QApplication::translate("Exception", e.what()));
969  return false;
970  }
971  return true;
972 }
