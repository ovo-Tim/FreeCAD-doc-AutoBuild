69 App::DocumentObjectExecReturn* FeaturePrimitive::execute(const TopoDS_Shape& primitive)
70 {
71  if (onlyHasToRefine()){
72  TopoShape result = refineShapeIfActive(rawShape);
73  Shape.setValue(result);
74  return App::DocumentObject::StdReturn;
75  }
76 
77  try {
78  //transform the primitive in the correct coordinance
79  FeatureAddSub::execute();
80 
81  //if we have no base we just add the standard primitive shape
82  TopoShape primitiveShape;
83  primitiveShape.setShape(primitive);
84 
85  TopoShape base;
86  try {
87  // if we have a base shape we need to make sure that it does not get our transformation
88  // to
89  base = getBaseTopoShape().moved(getLocation().Inverted());
90  primitiveShape.Tag = -this->getID();
91  }
92 
93  catch (const Base::Exception&) {
94 
95  //as we use this for preview we can add it even if useless for subtractive
96  AddSubShape.setValue(primitiveShape);
97 
98  if(getAddSubType() == FeatureAddSub::Additive)
99  Shape.setValue(getSolid(primitiveShape));
100  else
101  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Cannot subtract primitive feature without base feature"));
102 
103  return App::DocumentObject::StdReturn;
104  }
105  AddSubShape.setValue(primitiveShape);
106 
107  TopoShape boolOp(0);
108 
109  const char* maker;
110  switch (getAddSubType()) {
111  case Additive:
112  maker = Part::OpCodes::Fuse;
113  break;
114  case Subtractive:
115  maker = Part::OpCodes::Cut;
116  break;
117  default:
118  return new App::DocumentObjectExecReturn(
119  QT_TRANSLATE_NOOP("Exception", "Unknown operation type"));
120  }
121  try {
122  boolOp.makeElementBoolean(maker, {base, primitiveShape});
123  }
124  catch (Standard_Failure&) {
125  return new App::DocumentObjectExecReturn(
126  QT_TRANSLATE_NOOP("Exception", "Failed to perform boolean operation"));
127  }
128 
129  TopoShape solidBoolOp = getSolid(boolOp);
130  // lets check if the result is a solid
131  if (solidBoolOp.isNull()) {
132  return new App::DocumentObjectExecReturn(
133  QT_TRANSLATE_NOOP("Exception", "Resulting shape is not a solid"));
134  }
135  // store shape before refinement
136  this->rawShape = boolOp;
137 
138  if (solidBoolOp == base){
139  //solidBoolOp is misplaced but boolOp is ok
140  Shape.setValue(boolOp);
141  return App::DocumentObject::StdReturn;
142  }
143  solidBoolOp = refineShapeIfActive(solidBoolOp);
144  Shape.setValue(getSolid(solidBoolOp));
145  }
146  catch (Standard_Failure& e) {
147 
148  return new App::DocumentObjectExecReturn(e.GetMessageString());
149  }
150 
151  return App::DocumentObject::StdReturn;
152 }

425 App::DocumentObjectExecReturn* Ellipsoid::execute()
426 {
427  // Build a sphere
428  if (Radius1.getValue() < Precision::Confusion())
429  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Radius of ellipsoid too small"));
430  if (Radius2.getValue() < Precision::Confusion())
431  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Radius of ellipsoid too small"));
432 
433  try {
434  gp_Pnt pnt(0.0,0.0,0.0);
435  gp_Dir dir(0.0,0.0,1.0);
436  gp_Ax2 ax2(pnt,dir);
437  BRepPrimAPI_MakeSphere mkSphere(ax2,
438  Radius2.getValue(),
439  Base::toRadians<double>(Angle1.getValue()),
440  Base::toRadians<double>(Angle2.getValue()),
441  Base::toRadians<double>(Angle3.getValue()));
442  Standard_Real scaleX = 1.0;
443  Standard_Real scaleZ = Radius1.getValue()/Radius2.getValue();
444  // issue #1798: A third radius has been introduced. To be backward
445  // compatible if Radius3 is 0.0 (default) it's handled to be the same
446  // as Radius2
447  Standard_Real scaleY = 1.0;
448  if (Radius3.getValue() >= Precision::Confusion())
449  scaleY = Radius3.getValue()/Radius2.getValue();
450  gp_GTrsf mat;
451  mat.SetValue(1,1,scaleX);
452  mat.SetValue(2,1,0.0);
453  mat.SetValue(3,1,0.0);
454  mat.SetValue(1,2,0.0);
455  mat.SetValue(2,2,scaleY);
456  mat.SetValue(3,2,0.0);
457  mat.SetValue(1,3,0.0);
458  mat.SetValue(2,3,0.0);
459  mat.SetValue(3,3,scaleZ);
460  BRepBuilderAPI_GTransform mkTrsf(mkSphere.Shape(), mat);
461  return FeaturePrimitive::execute(mkTrsf.Shape());
462  }
463  catch (Standard_Failure& e) {
464  return new App::DocumentObjectExecReturn(e.GetMessageString());
465  }
466 
467  return App::DocumentObject::StdReturn;
468 }

574 App::DocumentObjectExecReturn* Prism::execute()
575 {
576  // Build a prism
577  if (Polygon.getValue() < 3)
578  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Polygon of prism is invalid, must have 3 or more sides"));
579  if (Circumradius.getValue() < Precision::Confusion())
580  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Circumradius of the polygon, of the prism, is too small"));
581  if (Height.getValue() < Precision::Confusion())
582  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Height of prism is too small"));
583  try {
584  long nodes = Polygon.getValue();
585 
586  Base::Matrix4D mat;
587  mat.rotZ(Base::toRadians(360.0/nodes));
588 
589  // create polygon
590  BRepBuilderAPI_MakePolygon mkPoly;
591  Base::Vector3d v(Circumradius.getValue(),0,0);
592  for (long i=0; i<nodes; i++) {
593  mkPoly.Add(gp_Pnt(v.x,v.y,v.z));
594  v = mat * v;
595  }
596  mkPoly.Add(gp_Pnt(v.x,v.y,v.z));
597  BRepBuilderAPI_MakeFace mkFace(mkPoly.Wire());
598  // the direction vector for the prism is the height for z and the given angle
599  TopoDS_Shape prism = makePrism(Height.getValue(), mkFace.Face());
600  return FeaturePrimitive::execute(prism);
601  }
602  catch (Standard_Failure& e) {
603  return new App::DocumentObjectExecReturn(e.GetMessageString());
604  }
605 
606  return App::DocumentObject::StdReturn;
607 }

643 App::DocumentObjectExecReturn* Wedge::execute()
644 {
645  double xmin = Xmin.getValue();
646  double ymin = Ymin.getValue();
647  double zmin = Zmin.getValue();
648  double z2min = Z2min.getValue();
649  double x2min = X2min.getValue();
650  double xmax = Xmax.getValue();
651  double ymax = Ymax.getValue();
652  double zmax = Zmax.getValue();
653  double z2max = Z2max.getValue();
654  double x2max = X2max.getValue();
655 
656  double dx = xmax-xmin;
657  double dy = ymax-ymin;
658  double dz = zmax-zmin;
659  double dz2 = z2max-z2min;
660  double dx2 = x2max-x2min;
661 
662  if (dx < Precision::Confusion())
663  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "delta x of wedge too small"));
664 
665  if (dy < Precision::Confusion())
666  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "delta y of wedge too small"));
667 
668  if (dz < Precision::Confusion())
669  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "delta z of wedge too small"));
670 
671  if (dz2 < 0)
672  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "delta z2 of wedge is negative"));
673 
674  if (dx2 < 0)
675  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "delta x2 of wedge is negative"));
676 
677  try {
678  gp_Pnt pnt(0.0,0.0,0.0);
679  gp_Dir dir(0.0,0.0,1.0);
680  BRepPrim_Wedge mkWedge(gp_Ax2(pnt,dir),
681  xmin, ymin, zmin, z2min, x2min,
682  xmax, ymax, zmax, z2max, x2max);
683  BRepBuilderAPI_MakeSolid mkSolid;
684  mkSolid.Add(mkWedge.Shell());
685  return FeaturePrimitive::execute(mkSolid.Solid());
686  }
687  catch (Standard_Failure& e) {
688  return new App::DocumentObjectExecReturn(e.GetMessageString());
689  }
690 
691  return App::DocumentObject::StdReturn;
692 }
