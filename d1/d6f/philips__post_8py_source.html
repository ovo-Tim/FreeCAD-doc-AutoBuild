293 def angleUnder180(command, lastX, lastY, x, y, i, j):
294  # radius R can be used iff angle is < 180.
295  # This is the case
296  # if the previous point is left of the current and the center is below (or on) the connection line
297  # or if the previous point is right of the current and the center is above
298  # (or on) the connection line
299  middleOfLineY = (lastY + y) / 2
300  centerY = lastY + j
301  if (
302  command == "G2"
303  and (
304  (lastX == x and ((lastY < y and i >= 0) or (lastY > y and i <= 0)))
305  or (lastX < x and centerY <= middleOfLineY)
306  or (lastX > x and centerY >= middleOfLineY)
307  )
308  ) or (
309  command == "G3"
310  and (
311  (lastX == x and ((lastY < y and i <= 0) or (lastY > y and i >= 0)))
312  or (lastX < x and centerY >= middleOfLineY)
313  or (lastX > x and centerY <= middleOfLineY)
314  )
315  ):
316  return True
317  else:
318  return False

334 def linenumberify(GCodeString):
335  # add a linenumber at every beginning of line
