287  def handleElasticityBndConditions(self):

299  self.write.boundary(name, "Displacement 1", 0.0)

301  self.write.boundary(name, "Displacement 3", 0.0)
302  self.write.handled(obj)
303  for obj in self.write.getMember("Fem::ConstraintForce"):
304  if obj.References:
305  for name in obj.References[0][1]:
306  force = float(obj.Force.getValueAs("N"))

308  self.write.boundary(name, "Force 2", obj.DirectionVector.y * force)
309  self.write.boundary(name, "Force 3", obj.DirectionVector.z * force)
310  self.write.boundary(name, "Force 1 Normalize by Area", True)
311  self.write.boundary(name, "Force 2 Normalize by Area", True)
312  self.write.boundary(name, "Force 3 Normalize by Area", True)

353  def handleElasticityBodyForces(self, bodies):
354  obj = self.write.getSingleMember("Fem::ConstraintSelfWeight")
355  if obj is not None:
356  for name in bodies:
357  gravity = self.write.convert(obj.GravityAcceleration.toStr(), "L/T^2")
358  if self.write.getBodyMaterial(name) is None:
359  raise general_writer.WriteError(
360  f"The body {name} is not referenced in any material.\n\n"
361  )
362  m = self.write.getBodyMaterial(name).Material
363 
364  densityQuantity = Units.Quantity(m["Density"])
365  dimension = "M/L^3"
366  if name.startswith("Edge"):
367  # not tested, bernd
368  # TODO: test
369  densityQuantity.Unit = Units.Unit(-2, 1)
370  dimension = "M/L^2"
371  density = self.write.convert(densityQuantity, dimension)
372 
373  force1 = gravity * obj.GravityDirection.x * density
374  force2 = gravity * obj.GravityDirection.y * density
375  force3 = gravity * obj.GravityDirection.z * density
376  self.write.bodyForce(name, "Stress Bodyforce 1", force1)
377  self.write.bodyForce(name, "Stress Bodyforce 2", force2)
378  self.write.bodyForce(name, "Stress Bodyforce 3", force3)

381  def handleElasticityMaterial(self, bodies):
382  # density
383  # is needed for self weight constraints and frequency analysis
384  density_needed = False
385  for equation in self.solver.Group:
386  if femutils.is_of_type(equation, "Fem::EquationElmerElasticity"):
387  if equation.EigenAnalysis is True:
388  density_needed = True
389  break # there could be a second equation without frequency
390  gravObj = self.write.getSingleMember("Fem::ConstraintSelfWeight")
391  if gravObj is not None:
392  density_needed = True
393  # temperature
394  tempObj = self.write.getSingleMember("Fem::ConstraintInitialTemperature")
395  if tempObj is not None:
396  refTemp = float(tempObj.initialTemperature.getValueAs("K"))
397  for name in bodies:
398  self.write.material(name, "Reference Temperature", refTemp)
399  # get the material data for all bodies
400  for obj in self.write.getMember("App::MaterialObject"):
401  m = obj.Material
402  refs = obj.References[0][1] if obj.References else self.write.getAllBodies()
403  for name in (n for n in refs if n in bodies):
404  # don't evaluate fluid material
405  if self.write.isBodyMaterialFluid(name):
406  break
407  if "YoungsModulus" not in m:
408  Console.PrintMessage(f"m: {m}\n")
409  # it is no fluid but also no solid
410  # -> user set no material reference at all
411  # that now material is known
412  raise general_writer.WriteError(
413  "There are two or more materials with empty references.\n\n"
414  "Set for the materials to what solid they belong to.\n"
415  )
416  self.write.material(name, "Name", m["Name"])
417  if density_needed is True:
418  self.write.material(name, "Density", self.write.getDensity(m))
419  self.write.material(name, "Youngs Modulus", self._getYoungsModulus(m))
420  self.write.material(name, "Poisson ratio", float(m["PoissonRatio"]))
421  if tempObj:
422  self.write.material(
423  name,
424  "Heat expansion Coefficient",
425  self.write.convert(m["ThermalExpansionCoefficient"], "O^-1"),
426  )
