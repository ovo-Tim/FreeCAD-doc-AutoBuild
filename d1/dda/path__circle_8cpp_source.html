51 Path_Circle::Path_Circle(const Frame& F_base_start,
52  const Vector& _V_base_center,
53  const Vector& V_base_p,
54  const Rotation& R_base_end,
55  double alpha,
56  RotationalInterpolation* _orient,
57  double _eqradius,
58  bool _aggregate) :
59  orient(_orient) ,
60  eqradius(_eqradius),
61  aggregate(_aggregate)
62  {
63  F_base_center.p = _V_base_center;
64  orient->SetStartEnd(F_base_start.M,R_base_end);
65  double oalpha = orient->Angle();
66 
67  Vector x(F_base_start.p - F_base_center.p);
68  radius = x.Normalize();
69  if (radius < epsilon) throw Error_MotionPlanning_Circle_ToSmall();
70  Vector tmpv(V_base_p-F_base_center.p);
71  tmpv.Normalize();
72  Vector z( x * tmpv);
73  double n = z.Normalize();
74  if (n < epsilon) throw Error_MotionPlanning_Circle_No_Plane();
75  F_base_center.M = Rotation(x,z*x,z);
76  double dist = alpha*radius;
77  // See what has the slowest eq. motion, and adapt
78  // the other to this slower motion
79  // use eqradius to transform between rot and transl.
80  // the same as for lineair motion
81  if (oalpha*eqradius > dist) {
82  // rotational_interpolation is the limitation
83  pathlength = oalpha*eqradius;
84  scalerot = 1/eqradius;
85  scalelin = dist/pathlength;
86  } else {
87  // translation is the limitation
88  pathlength = dist;
89  scalerot = oalpha/pathlength;
90  scalelin = 1;
91  }
92  }

121 Twist Path_Circle::Acc(double s,double sd,double sdd) const {
122  double p = s*scalelin / radius;
123  double cp = cos(p);
124  double sp = sin(p);
125  double v = sd*scalelin / radius;
126  double a = sdd*scalelin / radius;
127  return Twist( F_base_center.M*Vector(
128  -radius*cp*v*v - radius*sp*a,
129  -radius*sp*v*v + radius*cp*a,
130  0
131  ),
132  orient->Acc(s*scalerot,sd*scalerot,sdd*scalerot)
133  );
134 }
