210 Measure::MeasureBase* TaskMeasure::createObject(const App::MeasureType* measureType)
211 {
212  App::Document* doc = App::GetApplication().getActiveDocument();
213  if (!doc) {
214  return nullptr;
215  }
216 
217  if (measureType->isPython) {
218  Base::PyGILStateLocker lock;
219  auto pyMeasureClass = measureType->pythonClass;
220 
221  // Create a MeasurePython instance
222  auto featurePython = doc->addObject("Measure::MeasurePython", measureType->label.c_str());
223  _mMeasureObject = dynamic_cast<Measure::MeasureBase*>(featurePython);
224 
225  // Create an instance of the pyMeasureClass, the classe's initializer sets the object as
226  // proxy
227  Py::Tuple args(1);
228  args.setItem(0, Py::asObject(featurePython->getPyObject()));
229  PyObject* result = PyObject_CallObject(pyMeasureClass, args.ptr());
230  Py_XDECREF(result);
231  }
232  else {
233  // Create measure object
234  _mMeasureObject = dynamic_cast<Measure::MeasureBase*>(
235  doc->addObject(measureType->measureObject.c_str(), measureType->label.c_str()));
236  }
237 
238  return _mMeasureObject;
239 }

242 void TaskMeasure::update()
243 {
244  App::Document* doc = App::GetApplication().getActiveDocument();
245 
246  // Reset selection if the selected object is not valid
247  for (auto sel : Gui::Selection().getSelection()) {
248  App::DocumentObject* ob = sel.pObject;
249  App::DocumentObject* sub = ob->getSubObject(sel.SubName);
250 
251  // Resolve App::Link
252  if (sub->isDerivedFrom<App::Link>()) {
253  auto link = static_cast<App::Link*>(sub);
254  sub = link->getLinkedObject(true);
255  }
256 
257  std::string mod = Base::Type::getModuleName(sub->getTypeId().getName());
258  if (!App::MeasureManager::hasMeasureHandler(mod.c_str())) {
259  Base::Console().Message("No measure handler available for geometry of module: %s\n",
260  mod);
261  clearSelection();
262  return;
263  }
264  }
265 
266  valueResult->setText(QString::asprintf("-"));
267 
268  std::string mode = explicitMode ? modeSwitch->currentText().toStdString() : "";
269 
270  App::MeasureSelection selection;
271  for (auto s : Gui::Selection().getSelection(doc->getName(), ResolveMode::NoResolve)) {
272  App::SubObjectT sub(s.pObject, s.SubName);
273 
274  App::MeasureSelectionItem item = {sub, Base::Vector3d(s.x, s.y, s.z)};
275  selection.push_back(item);
276  }
277 
278  // Get valid measure type
279  App::MeasureType* measureType = nullptr;
280  auto measureTypes = App::MeasureManager::getValidMeasureTypes(selection, mode);
281  if (measureTypes.size() > 0) {
282  measureType = measureTypes.front();
283  }
284 
285 
286  if (!measureType) {
287 
288  // Note: If there's no valid measure type we might just restart the selection,
289  // however this requires enough coverage of measuretypes that we can access all of them
290 
291  // std::tuple<std::string, std::string> sel = selection.back();
292  // clearSelection();
293  // addElement(measureModule.c_str(), get<0>(sel).c_str(), get<1>(sel).c_str());
294 
295  // Reset measure object
296  if (!explicitMode) {
297  setModeSilent(nullptr);
298  }
299  removeObject();
300  enableAnnotateButton(false);
301  return;
302  }
303 
304  // Update tool mode display
305  setModeSilent(measureType);
306 
307  if (!_mMeasureObject || measureType->measureObject != _mMeasureObject->getTypeId().getName()
308  || _mMeasureObject->getDocument() != doc) {
309  // we don't already have a measureobject or it isn't the same type as the new one
310  removeObject();
311  createObject(measureType);
312  }
313 
314  // we have a valid measure object so we can enable the annotate button
315  enableAnnotateButton(true);
316 
317  // Fill measure object's properties from selection
318  _mMeasureObject->parseSelection(selection);
319 
320  // Get result
321  valueResult->setText(_mMeasureObject->getResultString());
322 
323  // Initialite the measurement's viewprovider
324  initViewObject();
325 }

456 void TaskMeasure::onSelectionChanged(const Gui::SelectionChanges& msg)
457 {
458  // Skip non-relevant events
459  if (msg.Type != SelectionChanges::AddSelection && msg.Type != SelectionChanges::RmvSelection
460  && msg.Type != SelectionChanges::SetSelection
461  && msg.Type != SelectionChanges::ClrSelection) {
462 
463  return;
464  }
465 
466  // If the control modifier is pressed, the object is just added to the current measurement
467  // If the control modifier is not pressed, a new measurement will be started. If autosave is on,
468  // the old measurement will be saved otherwise discharded. Shift inverts the autosave behaviour
469  // temporarily
470  const auto modifier = QGuiApplication::keyboardModifiers();
471  const bool ctrl = (modifier & Qt::ControlModifier) > 0;
472  const bool shift = (modifier & Qt::ShiftModifier) > 0;
473  // shift inverts the current state temporarily
474  const auto autosave = (mAutoSave && !shift) || (!mAutoSave && shift);
475  if ((!ctrl && Selection().getSelectionStyle() == SelectionStyle::NormalSelection)
476  || (ctrl && Selection().getSelectionStyle() == SelectionStyle::GreedySelection)) {
477  if (autosave && this->buttonBox->button(QDialogButtonBox::Apply)->isEnabled()) {
478  apply(false);
479  }
480  }
481  update();
482 }

484 bool TaskMeasure::eventFilter(QObject* obj, QEvent* event)
485 {
486 
487  if (event->type() == QEvent::KeyPress) {
488  auto keyEvent = static_cast<QKeyEvent*>(event);
489 
490  if (keyEvent->key() == Qt::Key_Escape) {
491 
492  if (this->hasSelection()) {
493  this->reset();
494  }
495  else {
496  this->reject();
497  }
498 
499  return true;
500  }
501 
502  if (keyEvent->key() == Qt::Key_Return || keyEvent->key() == Qt::Key_Enter) {
503  // Save object. Indirectly dependent on whether the apply button is enabled
504  // enabled if valid measurement object.
505  this->buttonBox->button(QDialogButtonBox::Apply)->click();
506  return true;
507  }
508  }
509 
510  return TaskDialog::eventFilter(obj, event);
511 }
