46 void CInnerCurves::Insert(shared_ptr<CCurve> pcurve)
47 {
48  std::list<shared_ptr<CInnerCurves>> outside_of_these;
49  std::list<shared_ptr<CInnerCurves>> crossing_these;
50 
51  // check all inner curves
52  for (shared_ptr<CInnerCurves> c : m_inner_curves) {
53 
54  switch (GetOverlapType(*pcurve, *(c->m_curve))) {
55  case eOutside:
56  outside_of_these.push_back(c);
57  break;
58 
59  case eInside:
60  // insert in this inner curve
61  c->Insert(pcurve);
62  return;
63 
64  case eSiblings:
65  break;
66 
67  case eCrossing:
68  crossing_these.push_back(c);
69  break;
70  }
71  }
72 
73  // add as a new inner
74  shared_ptr<CInnerCurves> new_item(new CInnerCurves(shared_from_this(), pcurve));
75  this->m_inner_curves.insert(new_item);
76 
77  for (const shared_ptr<CInnerCurves>& c : outside_of_these) {
78  // move items
79  c->m_pOuter = new_item;
80  new_item->m_inner_curves.insert(c);
81  this->m_inner_curves.erase(c);
82  }
83 
84  for (const shared_ptr<CInnerCurves>& c : crossing_these) {
85  // unite these
86  new_item->Unite(c);
87  this->m_inner_curves.erase(c);
88  }
89 }

91 void CInnerCurves::GetArea(CArea& area, bool outside, bool use_curve)
92 {

119 void CInnerCurves::Unite(shared_ptr<CInnerCurves> c)
120 {
121  // unite all the curves in c, with this one
122  shared_ptr<CArea> new_area(new CArea());
123  new_area->m_curves.push_back(*m_curve);
124  m_unite_area = new_area;
125 
126  CArea a2;
127  c->GetArea(a2);
128 
129  m_unite_area->Union(a2);
130  m_unite_area->Reorder();
131  for (std::list<CCurve>::iterator It = m_unite_area->m_curves.begin();
132  It != m_unite_area->m_curves.end();
133  It++) {
134  CCurve& curve = *It;
135  if (It == m_unite_area->m_curves.begin()) {
136  m_curve = make_shared<CCurve>(curve);
137  }
138  else {
139  if (curve.IsClockwise()) {
140  curve.Reverse();
141  }
142  Insert(std::make_shared<CCurve>(curve));
143  }
144  }
145 }
