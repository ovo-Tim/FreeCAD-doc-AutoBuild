118 bool FileHandler::openInternal()
119 {
120  if (activateEditor()) {
121  return true;
122  }
123 
124  QFileInfo fi;
125  fi.setFile(filename);
126  QString ext = fi.suffix().toLower();
127 
128  auto hasExtension = [ext](const QStringList& suffixes) {
129  return suffixes.contains(ext);
130  };
131 
132  if (hasExtension(QStringList() << QLatin1String("iv"))) {
133  openInventor();
134  return true;
135  }
136 
137  if (hasExtension(QStringList() << QLatin1String("wrl")
138  << QLatin1String("wrz")
139  << QLatin1String("vrml"))) {
140  openVRML();
141  return true;
142  }
143 
144  if (hasExtension(QStringList() << QLatin1String("py")
145  << QLatin1String("fcmacro")
146  << QLatin1String("fcscript"))) {
147  openPython();
148  return true;
149  }
150 
151  QStringList supportedFormats;
152  auto imageFormats = QImageReader::supportedImageFormats();
153  std::transform(imageFormats.cbegin(), imageFormats.cend(),
154  std::back_inserter(supportedFormats), [](const QByteArray& format) {
155  return QString::fromLatin1(format);
156  });
157 
158  if (hasExtension(supportedFormats)) {
159  openImage();
160  return true;
161  }
162 
163  return false;
164 }
