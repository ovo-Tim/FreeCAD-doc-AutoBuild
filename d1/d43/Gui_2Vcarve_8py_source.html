57  def addBaseGeometry(self, selection):
58  Path.Log.track(selection)
59  added = False
60  shapes = self.obj.BaseShapes
61  for sel in selection:
62  job = PathUtils.findParentJob(self.obj)
63  base = job.Proxy.resourceClone(job, sel.Object)
64  if not base:
65  Path.Log.notice(
66  (translate("CAM", "%s is not a Base Model object of the job %s") + "\n")
67  % (sel.Object.Label, job.Label)
68  )
69  continue
70  if base in shapes:
71  Path.Log.notice("Base shape %s already in the list".format(sel.Object.Label))
72  continue
73  if base.isDerivedFrom("Part::Part2DObject"):
74  if sel.HasSubObjects:
75  # selectively add some elements of the drawing to the Base
76  for sub in sel.SubElementNames:
77  if "Vertex" in sub:
78  Path.Log.info("Ignoring vertex")
79  else:
80  self.obj.Proxy.addBase(self.obj, base, sub)
81  else:
82  # when adding an entire shape to BaseShapes we can take its sub shapes out of Base
83  self.obj.Base = [(p, el) for p, el in self.obj.Base if p != base]
84  shapes.append(base)
85  self.obj.BaseShapes = shapes
86  added = True
87 
88  if not added:
89  # user wants us to engrave an edge of face of a base model
90  Path.Log.info(" call default")
91  base = self.super().addBaseGeometry(selection)
92  added = added or base
93 
94  return added
