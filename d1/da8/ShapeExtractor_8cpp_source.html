86 TopoDS_Shape ShapeExtractor::getShapes(const std::vector<App::DocumentObject*> links, bool include2d)
87 {
88  std::vector<TopoDS_Shape> sourceShapes;
89 
90  for (auto& l:links) {
91  if (is2dObject(l) && !include2d) {
92  continue;
93  }
94 
95  // Copy the pointer as not const so it can be changed if needed.
96  App::DocumentObject* obj = l;
97 
98  bool isExplodedView = false;
99  auto proxy = dynamic_cast<App::PropertyPythonObject*>(l->getPropertyByName("Proxy"));
100  Base::PyGILStateLocker lock;
101  if (proxy && proxy->getValue().hasAttr("saveAssemblyAndExplode")) {
102  isExplodedView = true;
103 
104  Py::Object explodedViewPy = proxy->getValue();
105  Py::Object attr = explodedViewPy.getAttr("saveAssemblyAndExplode");
106 
107  if (attr.ptr() && attr.isCallable()) {
108  Py::Tuple args(1);
109  args.setItem(0, Py::asObject(l->getPyObject()));
110  Py::Callable methode(attr);
111  Py::Object pyResult = methode.apply(args);
112 
113  if (PyObject_TypeCheck(pyResult.ptr(), &(Part::TopoShapePy::Type))) {
114  auto* shapepy = static_cast<Part::TopoShapePy*>(pyResult.ptr());
115  const TopoDS_Shape& shape = shapepy->getTopoShapePtr()->getShape();
116  sourceShapes.push_back(shape);
117  }
118  }
119 
120  for (auto* inObj : l->getInList()) {
121  if (inObj->isDerivedFrom(App::Part::getClassTypeId())) {
122  // we replace obj by the assembly
123  obj = inObj;
124  break;
125  }
126  }
127  }
128 
129  if (obj->isDerivedFrom<App::Link>()) {
130  App::Link* xLink = dynamic_cast<App::Link*>(obj);
131  std::vector<TopoDS_Shape> xShapes = getXShapes(xLink);
132  if (!xShapes.empty()) {
133  sourceShapes.insert(sourceShapes.end(), xShapes.begin(), xShapes.end());
134  continue;
135  }
136  }
137  else {
138  auto shape = Part::Feature::getShape(obj);
139  // if source obj has a shape, we use that shape.
140  if(!SU::isShapeReallyNull(shape)) {
141  if (checkShape(obj, shape)) {
142  sourceShapes.push_back(getLocatedShape(obj));
143  }
144  }
145  else {
146  std::vector<TopoDS_Shape> shapeList = getShapesFromObject(obj);
147  sourceShapes.insert(sourceShapes.end(), shapeList.begin(), shapeList.end());
148  }
149  }
150 
151  if (isExplodedView) {
152  Py::Object explodedViewPy = proxy->getValue();
153 
154  Py::Object attr = explodedViewPy.getAttr("restoreAssembly");
155  if (attr.ptr() && attr.isCallable()) {
156  Py::Tuple args(1);
157  args.setItem(0, Py::asObject(l->getPyObject()));
158  Py::Callable(attr).apply(args);
159  }
160  }
161  }
162 
163  BRep_Builder builder;
164  TopoDS_Compound comp;
165  builder.MakeCompound(comp);
166  for (auto& s:sourceShapes) {
167  if (SU::isShapeReallyNull(s)) {
168  continue;
169  } else if (s.ShapeType() < TopAbs_SOLID) {
170  //clean up composite shapes
171  TopoDS_Shape cleanShape = ShapeFinder::ShapeFinder::stripInfiniteShapes(s);
172  if (!cleanShape.IsNull()) {
173  builder.Add(comp, cleanShape);
174  }
175  } else if (Part::TopoShape(s).isInfinite()) {
176  continue; //simple shape is infinite
177  } else {
178  //a simple shape - add to compound
179  builder.Add(comp, s);
180  }
181  }
182  //it appears that an empty compound is !IsNull(), so we need to check a different way
183  if (!SU::isShapeReallyNull(comp)) {
184  return comp;
185  }
186 
187  return TopoDS_Shape();
188 }

190 std::vector<TopoDS_Shape> ShapeExtractor::getXShapes(const App::Link* xLink)
191 {
192  std::vector<TopoDS_Shape> xSourceShapes;
193  if (!xLink) {
194  return xSourceShapes;
195  }
196 
197  bool needsTransform = false;
198  std::vector<App::DocumentObject*> children = xLink->getLinkedChildren();
199  Base::Placement xLinkPlacement; // default constructor is an identity placement, i.e. no rotation nor translation
200  if (xLink->hasPlacement()) {
201  xLinkPlacement = xLink->getLinkPlacementProperty()->getValue();
202  needsTransform = true;
203  }
204  Base::Matrix4D linkScale; // default constructor is an identity matrix, possibly scale it with link's scale
205  if(xLink->getScaleProperty() || xLink->getScaleVectorProperty()) {
206  linkScale.scale(xLink->getScaleVector());
207  needsTransform = true;
208  }
209 
210  Base::Matrix4D netTransform;
211  if (!children.empty()) {
212  // this link points to other links???
213  for (auto& l:children) {
214  bool childNeedsTransform = false;
215  Base::Placement childPlm;
216  Base::Matrix4D childScale;
217  if (l->isDerivedFrom<App::LinkElement>()) {
218  App::LinkElement* cLinkElem = static_cast<App::LinkElement*>(l);
219  if (cLinkElem->hasPlacement()) {
220  childPlm = cLinkElem->getLinkPlacementProperty()->getValue();
221  childNeedsTransform = true;
222  }
223  if(cLinkElem->getScaleProperty() || cLinkElem->getScaleVectorProperty()) {
224  childScale.scale(cLinkElem->getScaleVector());
225  childNeedsTransform = true;
226  }
227  }
228  auto shape = Part::Feature::getShape(l); // TODO: getTopoShape() ?
229  Part::TopoShape ts(shape);
230  if (ts.isInfinite()) {
231  shape = ShapeFinder::stripInfiniteShapes(shape);
232  }
233  if (!checkShape(l, shape)) {
234  continue;
235  }
236  // copying the shape prevents "non-orthogonal GTrsf" errors in some versions
237  // of OCC. Something to do with triangulation of shape??
238  // it may be that incremental mesh would work here too.
239  BRepBuilderAPI_Copy copier(shape);
240  ts = Part::TopoShape(copier.Shape());
241  if(!ts.isNull()) {
242  if (needsTransform || childNeedsTransform) {
243  // Multiplication is associative, but the braces show the idea of combining the two transforms:
244  // ( link placement and scale ) combined to ( child placement and scale )
245  netTransform = (xLinkPlacement.toMatrix() * linkScale) * (childPlm.toMatrix() * childScale);
246  ts.transformGeometry(netTransform);
247  shape = ts.getShape();
248  }
249  xSourceShapes.push_back(shape);
250  } else {
251  Base::Console().Message("SE::getXShapes - no shape from getXShape\n");
252  }
253  }
254  } else {
255  // link points to a regular object, not another link? no sublinks?
256  TopoDS_Shape xLinkShape = getShapeFromXLink(xLink);
257  if (!xLinkShape.IsNull() &&
258  checkShape(xLink, xLinkShape)) {
259  // copying the shape prevents "non-orthogonal GTrsf" errors in some versions
260  // of OCC. Something to do with triangulation of shape??
261  BRepBuilderAPI_Copy copier(xLinkShape);
262  xSourceShapes.push_back(copier.Shape());
263  }
264  }
265  return xSourceShapes;
266 }

269 TopoDS_Shape ShapeExtractor::getShapeFromXLink(const App::Link* xLink)
270 {
271  Base::Placement xLinkPlacement;
272  if (xLink->hasPlacement()) {
273  xLinkPlacement = xLink->getLinkPlacementProperty()->getValue();
274  }
275  Base::Matrix4D linkScale; // default constructor is an identity matrix, possibly scale it with link's scale
276  if(xLink->getScaleProperty() || xLink->getScaleVectorProperty()) {
277  linkScale.scale(xLink->getScaleVector());
278  }
279  int depth = 0; //0 is default value, related to recursion of Links???
280  App::DocumentObject* linkedObject = xLink->getLink(depth);
281  if (linkedObject) {
282  // have a linked object, get the shape
283  TopoDS_Shape shape = Part::Feature::getShape(linkedObject);
284  if (shape.IsNull()) {
285  // this is where we need to parse the target for objects with a shape??
286  return TopoDS_Shape();
287  }
288  Part::TopoShape ts(shape);
289  if (ts.isInfinite()) {
290  shape = ShapeFinder::stripInfiniteShapes(shape);
291  ts = Part::TopoShape(shape);
292  }
293  //ts might be garbage now, better check
294  try {
295  if (!ts.isNull()) {
296  ts.setPlacement(xLinkPlacement);
297  }
298  }
299  catch (...) {
300  Base::Console().Error("ShapeExtractor failed to retrieve shape from %s\n", xLink->getNameInDocument());
301  return TopoDS_Shape();
302  }
303  if (checkShape(linkedObject, ts.getShape())) {
304  return ts.getShape();
305  }
306  }
307  return TopoDS_Shape();
308 }

310 std::vector<TopoDS_Shape> ShapeExtractor::getShapesFromObject(const App::DocumentObject* docObj)
311 {
312  std::vector<TopoDS_Shape> result;
313 
314  const App::GroupExtension* gex = dynamic_cast<const App::GroupExtension*>(docObj);
315  App::Property* gProp = docObj->getPropertyByName("Group");
316  App::Property* sProp = docObj->getPropertyByName("Shape");
317  if (docObj->isDerivedFrom<Part::Feature>()) {
318  if (checkShape(docObj, getLocatedShape(docObj))) {
319  result.push_back(getLocatedShape(docObj));
320  }
321  } else if (gex) { //is a group extension
322  std::vector<App::DocumentObject*> objs = gex->Group.getValues();
323  std::vector<TopoDS_Shape> shapes;
324  for (auto& d: objs) {
325  shapes = getShapesFromObject(d);
326  if (!shapes.empty()) {
327  result.insert(result.end(), shapes.begin(), shapes.end());
328  }
329  }
330  //the next 2 bits are mostly for Arch module objects
331  } else if (gProp) { //has a Group property
332  App::PropertyLinkList* list = dynamic_cast<App::PropertyLinkList*>(gProp);
333  if (list) {
334  std::vector<App::DocumentObject*> objsAll = list->getValues();
335  std::vector<TopoDS_Shape> shapesAll;
336  for (auto& obj : objsAll) {
337  shapesAll = getShapesFromObject(obj);
338  result.insert(result.end(), shapesAll.begin(), shapesAll.end());
339  }
340  }
341  } else if (sProp) { //has a Shape property
342  Part::PropertyPartShape* shapeProperty = dynamic_cast<Part::PropertyPartShape*>(sProp);
343  if (shapeProperty &&
344  checkShape(docObj, getLocatedShape(docObj))) {
345  result.push_back(getLocatedShape(docObj));
346  }
347  }
348  return result;
349 }

351 TopoDS_Shape ShapeExtractor::getShapesFused(const std::vector<App::DocumentObject*> links)
352 {
353  // get only the 3d shapes and fuse them
354  TopoDS_Shape baseShape = getShapes(links, false);
355  if (!baseShape.IsNull()) {
356  TopoDS_Iterator it(baseShape);
357  TopoDS_Shape fusedShape = it.Value();
358  it.Next();
359  for (; it.More(); it.Next()) {
360  const TopoDS_Shape& aChild = it.Value();
361  FCBRepAlgoAPI_Fuse mkFuse(fusedShape, aChild);
362  // Let's check if the fusion has been successful
363  if (!mkFuse.IsDone()) {
364  Base::Console().Error("SE - Fusion failed\n");
365  return baseShape;
366  }
367  fusedShape = mkFuse.Shape();
368  }
369  baseShape = fusedShape;
370  }
371 
372  // if there are 2d shapes in the links they will not fuse with the 3d shapes,
373  // so instead we return a compound of the fused 3d shapes and the 2d shapes
374  std::vector<TopoDS_Shape> shapes2d = getShapes2d(links);
375 
376  if (!shapes2d.empty()) {
377  shapes2d.push_back(baseShape);
378  return DrawUtil::shapeVectorToCompound(shapes2d, false);
379  }
380 
381  return baseShape;
382 }
