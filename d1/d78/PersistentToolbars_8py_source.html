24 def addTop(name, toolbars):
25  """addTop("name", ["toolbars"])
26 
27  Example for Sketcher workbench (InitGui.py):
28 
29  try:
30  import PersistentToolbars
31 
32  toolbars = ["Sketcher constraints",
33  "Break",
34  "Sketcher geometries"]
35 
36  PersistentToolbars.addRight("SketcherWorkbench", toolbars)
37  except ImportError:
38  pass
39 
40  Description:
41  Provides ability to preset toolbar position for any workbench.
42  Settings are deleted when FreeCAD exits normally. After user
43  customizes workbench toolbar position this setting has no
44  effect anymore. Customized toolbar position set by the user is
45  used instead."""
46 
47  p = App.ParamGet("User parameter:Tux/PersistentToolbars/System")
48  p.GetGroup(name).SetBool("Saved", 1)
49  p.GetGroup(name).SetString("Top", ",".join(toolbars))
