93 View3DInventor::View3DInventor(Gui::Document* pcDocument, QWidget* parent,
94  const QtGLWidget* sharewidget, Qt::WindowFlags wflags)
95  : MDIView(pcDocument, parent, wflags), _viewerPy(nullptr)
96 {
97  stack = new QStackedWidget(this);
98  // important for highlighting
99  setMouseTracking(true);
100  // accept drops on the window, get handled in dropEvent, dragEnterEvent
101  setAcceptDrops(true);
102 
103  //anti-aliasing settings
104  bool smoothing = false;
105  bool glformat = false;
106  int samples = View3DInventorViewer::getNumSamples();
107  QtGLFormat f;
108 
109  if (samples > 1) {
110  glformat = true;
111  f.setSamples(samples);
112  }
113  else if (samples > 0) {
114  smoothing = true;
115  }
116 
117  if (glformat)
118  _viewer = new View3DInventorViewer(f, this, sharewidget);
119  else
120  _viewer = new View3DInventorViewer(this, sharewidget);
121 
122  if (smoothing)
123  _viewer->getSoRenderManager()->getGLRenderAction()->setSmoothing(true);
124 
125  // create the inventor widget and set the defaults
126  _viewer->setDocument(this->_pcDocument);
127  stack->addWidget(_viewer->getWidget());
128  // https://forum.freecad.org/viewtopic.php?f=3&t=6055&sid=150ed90cbefba50f1e2ad4b4e6684eba
129  // describes a minor error but trying to fix it leads to a major issue
130  // https://forum.freecad.org/viewtopic.php?f=3&t=6085&sid=3f4bcab8007b96aaf31928b564190fd7
131  // so the change is commented out
132  // By default, the wheel events are processed by the 3d view AND the mdi area.
133  //_viewer->getGLWidget()->setAttribute(Qt::WA_NoMousePropagation);
134  setCentralWidget(stack);
135 
136  // apply the user settings
137  applySettings();
138 
139  stopSpinTimer = new QTimer(this);
140  connect(stopSpinTimer, &QTimer::timeout, this, &View3DInventor::stopAnimating);
141 
142  setWindowIcon(Gui::BitmapFactory().pixmap("Document"));

145 View3DInventor::~View3DInventor()
146 {
147  if(_pcDocument) {
148  SoCamera * Cam = _viewer->getSoRenderManager()->getCamera();
149  if (Cam)
150  _pcDocument->saveCameraSettings(SoFCDB::writeNodesToString(Cam).c_str());
151  }
152 
153  viewSettings.reset();
154 
155  //If we destroy this viewer by calling 'delete' directly the focus proxy widget which is defined
156  //by a widget in SoQtViewer isn't reset. This widget becomes a dangling pointer and makes
157  //the application crash. (Probably it's better to destroy this viewer by calling close().)
158  //See also Gui::Document::~Document().
159  QWidget* foc = qApp->focusWidget();
160  if (foc) {
161  QWidget* par = foc->parentWidget();
162  while (par) {
163  if (par == this) {
164  foc->setFocusProxy(nullptr);
165  foc->clearFocus();
166  break;
167  }
168  par = par->parentWidget();
169  }
170  }
171 
172  if (_viewerPy) {
173  Base::PyGILStateLocker lock;
174  Py_DECREF(_viewerPy);
175  }
176 
177  // here is from time to time trouble!!!
178  delete _viewer;

538 bool View3DInventor::setCamera(const char* pCamera)
539 {
540  SoCamera * CamViewer = _viewer->getSoRenderManager()->getCamera();
541  if (!CamViewer) {
542  throw Base::RuntimeError("No camera set so far...");
543  }
544 
545  SoInput in;
546  in.setBuffer((void*)pCamera,std::strlen(pCamera));
547 
548  SoNode * Cam;
549  SoDB::read(&in,Cam);
550 
551  if (!Cam || !Cam->isOfType(SoCamera::getClassTypeId())) {
552  throw Base::RuntimeError("Camera settings failed to read");
553  }
554 
555  // this is to make sure to reliably delete the node
556  CoinPtr<SoNode> camPtr(Cam, true);
557 
558  // toggle between perspective and orthographic camera
559  if (Cam->getTypeId() != CamViewer->getTypeId()) {
560  _viewer->setCameraType(Cam->getTypeId());
561  CamViewer = _viewer->getSoRenderManager()->getCamera();
562  }
563 
564  SoPerspectiveCamera * CamViewerP = nullptr;
565  SoOrthographicCamera * CamViewerO = nullptr;
566 
567  if (CamViewer->getTypeId() == SoPerspectiveCamera::getClassTypeId()) {
568  CamViewerP = static_cast<SoPerspectiveCamera *>(CamViewer); // safe downward cast, knows the type
569  }
570  else if (CamViewer->getTypeId() == SoOrthographicCamera::getClassTypeId()) {
571  CamViewerO = static_cast<SoOrthographicCamera *>(CamViewer); // safe downward cast, knows the type
572  }
573 
574  if (Cam->getTypeId() == SoPerspectiveCamera::getClassTypeId()) {
575  if (CamViewerP){
576  CamViewerP->position = static_cast<SoPerspectiveCamera *>(Cam)->position;
577  CamViewerP->orientation = static_cast<SoPerspectiveCamera *>(Cam)->orientation;
578  CamViewerP->nearDistance = static_cast<SoPerspectiveCamera *>(Cam)->nearDistance;
579  CamViewerP->farDistance = static_cast<SoPerspectiveCamera *>(Cam)->farDistance;
580  CamViewerP->focalDistance = static_cast<SoPerspectiveCamera *>(Cam)->focalDistance;
581  }
582  else {
583  throw Base::TypeError("Camera type mismatch");
584  }
585  }
586  else if (Cam->getTypeId() == SoOrthographicCamera::getClassTypeId()) {
587  if (CamViewerO){
588  CamViewerO->viewportMapping = static_cast<SoOrthographicCamera *>(Cam)->viewportMapping;
589  CamViewerO->position = static_cast<SoOrthographicCamera *>(Cam)->position;
590  CamViewerO->orientation = static_cast<SoOrthographicCamera *>(Cam)->orientation;
591  CamViewerO->nearDistance = static_cast<SoOrthographicCamera *>(Cam)->nearDistance;
592  CamViewerO->farDistance = static_cast<SoOrthographicCamera *>(Cam)->farDistance;
593  CamViewerO->focalDistance = static_cast<SoOrthographicCamera *>(Cam)->focalDistance;
594  CamViewerO->aspectRatio = static_cast<SoOrthographicCamera *>(Cam)->aspectRatio ;
595  CamViewerO->height = static_cast<SoOrthographicCamera *>(Cam)->height;
596  }
597  else {
598  throw Base::TypeError("Camera type mismatch");
599  }
600  }
601 
602  return true;
603 }

642 void View3DInventor::dump(const char* filename, bool onlyVisible)
643 {
644  SoGetPrimitiveCountAction action;
645  action.setCanApproximate(true);
646  action.apply(_viewer->getSceneGraph());
647 
648  SoNode* node = _viewer->getSceneGraph();
649  if (onlyVisible) {
650  node = replaceSwitchesInSceneGraph(node);
651  node->ref();
652  }
653 
654  if ( action.getTriangleCount() > 100000 || action.getPointCount() > 30000 || action.getLineCount() > 10000 )
655  _viewer->dumpToFile(node, filename, true);
656  else
657  _viewer->dumpToFile(node, filename, false);
658 
659  if (onlyVisible) {
660  node->unref();
661  }

719 void View3DInventor::setCurrentViewMode(ViewMode newmode)
720 {
721  ViewMode oldmode = MDIView::currentViewMode();
722  if (oldmode == newmode)
723  return;
724 
725  if (newmode == Child) {
726  // Fix in two steps:
727  // The mdi view got a QWindow when it became a top-level widget and when resetting it to a child widget
728  // the QWindow must be deleted because it has an impact on resize events and may break the layout of
729  // mdi view inside the QMdiSubWindow.
730  // In the second step below the layout must be invalidated after it's again a child widget to make sure
731  // the mdi view fits into the QMdiSubWindow.
732  QWindow* winHandle = this->windowHandle();
733  if (winHandle)
734  winHandle->destroy();
735  }
736 
737  MDIView::setCurrentViewMode(newmode);
738 
739  // Internally the QOpenGLWidget switches of the multi-sampling and there is no
740  // way to switch it on again. So as a workaround we just re-create a new viewport
741  // The method is private but defined as slot to avoid to call it by accident.
742  //int index = _viewer->metaObject()->indexOfMethod("replaceViewport()");
743  //if (index >= 0) {
744  // _viewer->qt_metacall(QMetaObject::InvokeMetaMethod, index, 0);
745  //}
746 
747  // This widget becomes the focus proxy of the embedded GL widget if we leave
748  // the 'Child' mode. If we reenter 'Child' mode the focus proxy is reset to 0.
749  // If we change from 'TopLevel' mode to 'Fullscreen' mode or vice versa nothing
750  // happens.
751  // Grabbing keyboard when leaving 'Child' mode (as done in a recent version) should
752  // be avoided because when two or more windows are either in 'TopLevel' or 'Fullscreen'
753  // mode only the last window gets all key event even if it is not the active one.
754  //
755  // It is important to set the focus proxy to get all key events otherwise we would lose
756  // control after redirecting the first key event to the GL widget.
757  if (oldmode == Child) {
758  // To make a global shortcut working from this window we need to add
759  // all existing actions from the mainwindow and its sub-widgets
760  QList<QAction*> acts = getMainWindow()->findChildren<QAction*>();
761  this->addActions(acts);
762  _viewer->getGLWidget()->setFocusProxy(this);
763  // To be notfified for new actions
764  qApp->installEventFilter(this);
765  }
766  else if (newmode == Child) {
767  _viewer->getGLWidget()->setFocusProxy(nullptr);
768  qApp->removeEventFilter(this);
769  QList<QAction*> acts = this->actions();
770  for (QAction* it : acts)
771  this->removeAction(it);
772 
773  // Step two
774  auto mdi = qobject_cast<QMdiSubWindow*>(parentWidget());
775  if (mdi && mdi->layout())
776  mdi->layout()->invalidate();
777  }
778 }

780 RayPickInfo View3DInventor::getObjInfoRay(Base::Vector3d* startvec, Base::Vector3d* dirvec)
781 {
782  double vsx, vsy, vsz;
783  double vdx, vdy, vdz;
784  vsx = startvec->x;
785  vsy = startvec->y;
786  vsz = startvec->z;
787  vdx = dirvec->x;
788  vdy = dirvec->y;
789  vdz = dirvec->z;
790  // near plane clipping is required to avoid false intersections
791  float nearClippingPlane = 0.1;
792 
793  RayPickInfo ret = {false,
794  Base::Vector3d(),
795  "",
796  "",
797  std::nullopt,
798  std::nullopt,
799  std::nullopt};
800  SoRayPickAction action(getViewer()->getSoRenderManager()->getViewportRegion());
801  action.setRay(SbVec3f(vsx, vsy, vsz), SbVec3f(vdx, vdy, vdz), nearClippingPlane);
802  action.apply(getViewer()->getSoRenderManager()->getSceneGraph());
803  SoPickedPoint* Point = action.getPickedPoint();
804 
805  if (!Point) {
806  return ret;
807  }
808 
809  ret.point = Base::convertTo<Base::Vector3d>(Point->getPoint());
810  ViewProvider* vp = getViewer()->getViewProviderByPath(Point->getPath());
811  if (vp && vp->isDerivedFrom(ViewProviderDocumentObject::getClassTypeId())) {
812  if (!vp->isSelectable()) {
813  return ret;
814  }
815  auto vpd = static_cast<ViewProviderDocumentObject*>(vp);
816  if (vp->useNewSelectionModel()) {
817  std::string subname;
818  if (!vp->getElementPicked(Point, subname)) {
819  return ret;
820  }
821  auto obj = vpd->getObject();
822  if (!obj) {
823  return ret;
824  }
825  if (!subname.empty()) {
826  App::ElementNamePair elementName;
827  auto sobj = App::GeoFeature::resolveElement(obj, subname.c_str(), elementName);
828  if (!sobj) {
829  return ret;
830  }
831  if (sobj != obj) {
832  ret.parentObject = obj->getExportName();
833  ret.subName = subname;
834  obj = sobj;
835  }
836  subname = !elementName.oldName.empty() ? elementName.oldName : elementName.newName;
837  }
838  ret.document = obj->getDocument()->getName();
839  ret.object = obj->getNameInDocument();
840  ret.component = subname;
841  ret.isValid = true;
842  }
843  else {
844  ret.document = vpd->getObject()->getDocument()->getName();
845  ret.object = vpd->getObject()->getNameInDocument();
846  // search for a SoFCSelection node
847  SoFCDocumentObjectAction objaction;
848  objaction.apply(Point->getPath());
849  if (objaction.isHandled()) {
850  ret.component = objaction.componentName.getString();
851  }
852  }
853  // ok, found the node of interest
854  ret.isValid = true;
855  }
856  else {
857  // custom nodes not in a VP: search for a SoFCSelection node
858  SoFCDocumentObjectAction objaction;
859  objaction.apply(Point->getPath());
860  if (objaction.isHandled()) {
861  ret.document = objaction.documentName.getString();
862  ret.object = objaction.objectName.getString();
863  ret.component = objaction.componentName.getString();
864  // ok, found the node of interest
865  ret.isValid = true;
866  }
867  }
868  return ret;
869 }

871 bool View3DInventor::eventFilter(QObject* watched, QEvent* e)
872 {
873  // As long as this widget is a top-level window (either in 'TopLevel' or 'FullScreen' mode) we
874  // need to be notified when an action is added to a widget. This action must also be added to
875  // this window to allow one to make use of its shortcut (if defined).
876  // Note: We don't need to care about removing an action if its parent widget gets destroyed.
877  // This does the action itself for us.
878  if (watched != this && e->type() == QEvent::ActionAdded) {
879  auto a = static_cast<QActionEvent*>(e);
880  QAction* action = a->action();
881 
882  if (!action->isSeparator()) {
883  QList<QAction*> actions = this->actions();
884  if (!actions.contains(action))
885  this->addAction(action);
886  }
887  }
888 
889  return false;
890 }
