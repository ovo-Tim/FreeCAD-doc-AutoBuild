202 bool ImportOCAF2::getColor(const TopoDS_Shape& shape, Info& info, bool check, bool noDefault)
203 {
204  bool ret = false;
205  Quantity_ColorRGBA aColor;
206  if (aColorTool->GetColor(shape, XCAFDoc_ColorSurf, aColor)) {
207  App::Color c = Tools::convertColor(aColor);
208  if (!check || info.faceColor != c) {
209  info.faceColor = c;
210  info.hasFaceColor = true;
211  ret = true;
212  }
213  }
214  if (!noDefault && !info.hasFaceColor && aColorTool->GetColor(shape, XCAFDoc_ColorGen, aColor)) {
215  App::Color c = Tools::convertColor(aColor);
216  if (!check || info.faceColor != c) {
217  info.faceColor = c;
218  info.hasFaceColor = true;
219  ret = true;
220  }
221  }
222  if (aColorTool->GetColor(shape, XCAFDoc_ColorCurv, aColor)) {
223  App::Color c = Tools::convertColor(aColor);
224  // Some STEP include a curve color with the same value of the face
225  // color. And this will look weird in FC. So for shape with face
226  // we'll ignore the curve color, if it is the same as the face color.
227  if ((c != info.faceColor || !TopExp_Explorer(shape, TopAbs_FACE).More())
228  && (!check || info.edgeColor != c)) {
229  info.edgeColor = c;
230  info.hasEdgeColor = true;
231  ret = true;
232  }
233  }
234  if (!check) {
235  if (!info.hasFaceColor) {
236  info.faceColor = options.defaultFaceColor;
237  }
238  if (!info.hasEdgeColor) {
239  info.edgeColor = options.defaultEdgeColor;
240  }
241  }
242  return ret;
243 }

246 ImportOCAF2::expandShape(App::Document* doc, TDF_Label label, const TopoDS_Shape& shape)
247 {
248  if (shape.IsNull() || !TopExp_Explorer(shape, TopAbs_VERTEX).More()) {
249  return nullptr;
250  }
251 
252  std::vector<App::DocumentObject*> objs;
253 
254  if (shape.ShapeType() == TopAbs_COMPOUND) {
255  for (TopoDS_Iterator it(shape, Standard_False, Standard_False); it.More(); it.Next()) {
256  TDF_Label childLabel;
257  if (!label.IsNull()) {
258  aShapeTool->FindSubShape(label, it.Value(), childLabel);
259  }
260  auto child = expandShape(doc, childLabel, it.Value());
261  if (child) {
262  objs.push_back(child);
263  Info info;
264  info.free = false;
265  info.obj = child;
266  myShapes.emplace(it.Value().Located(TopLoc_Location()), info);
267  }
268  }
269  if (objs.empty()) {
270  return nullptr;
271  }
272  auto compound =
273  static_cast<Part::Compound2*>(doc->addObject("Part::Compound2", "Compound"));
274  compound->Links.setValues(objs);
275  setPlacement(&compound->Placement, shape);
276  return compound;
277  }
278  Info info;
279  info.obj = nullptr;
280  createObject(doc, label, shape, info, false);
281  return info.obj;
282 }

284 bool ImportOCAF2::createObject(App::Document* doc,
285  TDF_Label label,
286  const TopoDS_Shape& shape,
287  Info& info,
288  bool newDoc)
289 {
290  if (shape.IsNull() || !TopExp_Explorer(shape, TopAbs_VERTEX).More()) {
291  FC_WARN(Tools::labelName(label) << " has empty shape");
292  return false;
293  }
294 
295  getColor(shape, info);
296  bool hasFaceColors = false;
297  bool hasEdgeColors = false;
298 
299  Part::TopoShape tshape(shape);
300  std::vector<App::Color> faceColors;
301  std::vector<App::Color> edgeColors;
302 
303  TDF_LabelSequence seq;
304  if (!label.IsNull() && aShapeTool->GetSubShapes(label, seq)) {
305 
306  TopTools_IndexedMapOfShape faceMap, edgeMap;
307  TopExp::MapShapes(tshape.getShape(), TopAbs_FACE, faceMap);
308  TopExp::MapShapes(tshape.getShape(), TopAbs_EDGE, edgeMap);
309 
310  faceColors.assign(faceMap.Extent(), info.faceColor);
311  edgeColors.assign(edgeMap.Extent(), info.edgeColor);
312  // Two passes to get sub shape colors. First pass, look for solid, and
313  // second pass look for face and edges. This allows lower level
314  // subshape to override color of higher level ones.
315  for (int j = 0; j < 2; ++j) {
316  for (int i = 1; i <= seq.Length(); ++i) {
317  TDF_Label l = seq.Value(i);
318  TopoDS_Shape subShape = aShapeTool->GetShape(l);
319  if (subShape.IsNull()) {
320  continue;
321  }
322  if (subShape.ShapeType() == TopAbs_FACE || subShape.ShapeType() == TopAbs_EDGE) {
323  if (j == 0) {
324  continue;
325  }
326  }
327  else if (j != 0) {
328  continue;
329  }
330 
331  bool foundFaceColor = false, foundEdgeColor = false;
332  App::Color faceColor, edgeColor;
333  Quantity_ColorRGBA aColor;
334  if (aColorTool->GetColor(l, XCAFDoc_ColorSurf, aColor)
335  || aColorTool->GetColor(l, XCAFDoc_ColorGen, aColor)) {
336  faceColor = Tools::convertColor(aColor);
337  foundFaceColor = true;
338  }
339  if (aColorTool->GetColor(l, XCAFDoc_ColorCurv, aColor)) {
340  edgeColor = Tools::convertColor(aColor);
341  foundEdgeColor = true;
342  if (j == 0 && foundFaceColor && !faceColors.empty() && edgeColor == faceColor) {
343  // Do not set edge the same color as face
344  foundEdgeColor = false;
345  }
346  }
347 
348  if (foundFaceColor) {
349  for (TopExp_Explorer exp(subShape, TopAbs_FACE); exp.More(); exp.Next()) {
350  int idx = faceMap.FindIndex(exp.Current()) - 1;
351  if (idx >= 0 && idx < (int)faceColors.size()) {
352  faceColors[idx] = faceColor;
353  hasFaceColors = true;
354  info.hasFaceColor = true;
355  }
356  }
357  }
358  if (foundEdgeColor) {
359  for (TopExp_Explorer exp(subShape, TopAbs_EDGE); exp.More(); exp.Next()) {
360  int idx = edgeMap.FindIndex(exp.Current()) - 1;
361  if (idx >= 0 && idx < (int)edgeColors.size()) {
362  edgeColors[idx] = edgeColor;
363  hasEdgeColors = true;
364  info.hasEdgeColor = true;
365  }
366  }
367  }
368  }
369  }
370  }
371 
372  Part::Feature* feature;
373 
374  if (newDoc && (options.mode == ObjectPerDoc || options.mode == ObjectPerDir)) {
375  doc = getDocument(doc, label);
376  }
377 
378  if (options.expandCompound
379  && (tshape.countSubShapes(TopAbs_SOLID) > 1
380  || (!tshape.countSubShapes(TopAbs_SOLID) && tshape.countSubShapes(TopAbs_SHELL) > 1))) {
381  feature = dynamic_cast<Part::Feature*>(expandShape(doc, label, shape));
382  assert(feature);
383  }
384  else {
385  feature = static_cast<Part::Feature*>(
386  doc->addObject("Part::Feature", tshape.shapeName().c_str()));
387  feature->Shape.setValue(shape);
388  }
389  applyFaceColors(feature, {info.faceColor});
390  applyEdgeColors(feature, {info.edgeColor});
391  if (hasFaceColors) {
392  applyFaceColors(feature, faceColors);
393  }
394  if (hasEdgeColors) {
395  applyEdgeColors(feature, edgeColors);
396  }
397 
398  info.propPlacement = &feature->Placement;
399  info.obj = feature;
400  return true;
401 }

403 App::Document* ImportOCAF2::getDocument(App::Document* doc, TDF_Label label)
404 {
405  if (filePath.empty() || options.mode == SingleDoc || options.merge) {
406  return doc;
407  }
408 
409  auto name = getLabelName(label);
410  if (name.empty()) {
411  return doc;
412  }
413 
414  auto newDoc = App::GetApplication().newDocument(name.c_str(), name.c_str(), false);
415  std::ostringstream ss;
416  Base::FileInfo fi(doc->FileName.getValue());
417  std::string path = fi.dirPath();
418  if (options.mode == GroupPerDir || options.mode == ObjectPerDir) {
419  for (int i = 0; i < 1000; ++i) {
420  ss.str("");
421  ss << path << '/' << fi.fileNamePure() << "_parts";
422  if (i > 0) {
423  ss << '_' << std::setfill('0') << std::setw(3) << i;
424  }
425  Base::FileInfo fi2(ss.str());
426  if (fi2.exists()) {
427  if (!fi2.isDir()) {
428  continue;
429  }
430  }
431  else if (!fi2.createDirectory()) {
432  FC_WARN("Failed to create directory " << fi2.filePath());
433  break;
434  }
435  path = fi2.filePath();
436  break;
437  }
438  }
439  for (int i = 0; i < 1000; ++i) {
440  ss.str("");
441  ss << path << '/' << newDoc->getName() << ".fcstd";
442  if (i > 0) {
443  ss << '_' << std::setfill('0') << std::setw(3) << i;
444  }
445  Base::FileInfo fi(ss.str());
446  if (!fi.exists()) {
447  if (!newDoc->saveAs(fi.filePath().c_str())) {
448  break;
449  }
450  return newDoc;
451  }
452  }
453 
454  FC_WARN("Cannot save document for part '" << name << "'");
455  return doc;
456 }

458 bool ImportOCAF2::createGroup(App::Document* doc,
459  Info& info,
460  const TopoDS_Shape& shape,
461  std::vector<App::DocumentObject*>& children,
462  const boost::dynamic_bitset<>& visibilities,
463  bool canReduce)
464 {
465  assert(children.size() == visibilities.size());
466  if (children.empty()) {
467  return false;
468  }
469  bool hasColor = getColor(shape, info, false, true);
470  if (canReduce && !hasColor && options.reduceObjects && children.size() == 1
471  && visibilities[0]) {
472  info.obj = children.front();
473  info.free = true;
474  info.propPlacement =
475  dynamic_cast<App::PropertyPlacement*>(info.obj->getPropertyByName("Placement"));
476  myCollapsedObjects.emplace(info.obj, info.propPlacement);
477  return true;
478  }
479  auto group = static_cast<App::LinkGroup*>(doc->addObject("App::LinkGroup", "LinkGroup"));
480  for (auto& child : children) {
481  if (child->getDocument() != doc) {
482  auto link = static_cast<App::Link*>(doc->addObject("App::Link", "Link"));
483  link->Label.setValue(child->Label.getValue());
484  link->setLink(-1, child);
485  auto pla = Base::freecad_dynamic_cast<App::PropertyPlacement>(
486  child->getPropertyByName("Placement"));
487  if (pla) {
488  link->Placement.setValue(pla->getValue());
489  }
490  child = link;
491  }
492  }
493  group->ElementList.setValues(children);
494  group->VisibilityList.setValue(visibilities);
495  info.obj = group;
496  info.propPlacement = &group->Placement;
497  if (getColor(shape, info, false, true)) {
498  if (info.hasFaceColor) {
499  applyLinkColor(group, -1, info.faceColor);
500  }
501  }
502  return true;
503 }

505 App::DocumentObject* ImportOCAF2::loadShapes()
506 {
507  if (!options.useLinkGroup) {
508  ImportLegacy legacy(*this);
509  legacy.setMerge(options.merge);
510  legacy.loadShapes();
511  return nullptr;
512  }
513 
514  if (FC_LOG_INSTANCE.isEnabled(FC_LOGLEVEL_LOG)) {
515  Tools::dumpLabels(pDoc->Main(), aShapeTool, aColorTool);
516  }
517 
518  TDF_LabelSequence labels;
519  aShapeTool->GetShapes(labels);
520  Base::SequencerLauncher seq("Importing...", labels.Length());
521  FC_LOG("free shape count " << labels.Length());
522  sequencer = options.showProgress ? &seq : nullptr;
523 
524  labels.Clear();
525  myShapes.clear();
526  myNames.clear();
527  myCollapsedObjects.clear();
528 
529  std::vector<App::DocumentObject*> objs;
530  aShapeTool->GetFreeShapes(labels);
531  boost::dynamic_bitset<> vis;
532  int count = 0;
533  for (Standard_Integer i = 1; i <= labels.Length(); i++) {
534  auto label = labels.Value(i);
535  if (!options.importHidden && !aColorTool->IsVisible(label)) {
536  continue;
537  }
538  ++count;
539  }
540  for (Standard_Integer i = 1; i <= labels.Length(); i++) {
541  auto label = labels.Value(i);
542  if (!options.importHidden && !aColorTool->IsVisible(label)) {
543  continue;
544  }
545  auto obj = loadShape(pDocument, label, aShapeTool->GetShape(label), false, count > 1);
546  if (obj) {
547  objs.push_back(obj);
548  vis.push_back(aColorTool->IsVisible(label));
549  }
550  }
551  App::DocumentObject* ret = nullptr;
552  if (objs.size() == 1) {
553  ret = objs.front();
554  }
555  else {
556  Info info;
557  if (createGroup(pDocument, info, TopoDS_Shape(), objs, vis)) {
558  ret = info.obj;
559  }
560  }
561  if (ret) {
562  ret->recomputeFeature(true);
563  }
564  if (options.merge && ret && !ret->isDerivedFrom(Part::Feature::getClassTypeId())) {
565  auto shape = Part::Feature::getTopoShape(ret);
566  auto feature =
567  static_cast<Part::Feature*>(pDocument->addObject("Part::Feature", "Feature"));
568  auto name = Tools::labelName(pDoc->Main());
569  feature->Label.setValue(name.empty() ? default_name.c_str() : name.c_str());
570  feature->Shape.setValue(shape);
571  applyFaceColors(feature, {});
572 
573  std::vector<std::pair<App::Document*, std::string>> objNames;
574  for (auto obj : App::Document::getDependencyList(objs, App::Document::DepSort)) {
575  objNames.emplace_back(obj->getDocument(), obj->getNameInDocument());
576  }
577  for (auto& v : objNames) {
578  v.first->removeObject(v.second.c_str());
579  }
580  ret = feature;
581  ret->recomputeFeature(true);
582  }
583  sequencer = nullptr;
584  return ret;
585 }

587 void ImportOCAF2::getSHUOColors(TDF_Label label,
588  std::map<std::string, App::Color>& colors,
589  bool appendFirst)
590 {
591  TDF_AttributeSequence seq;
592  if (label.IsNull() || !aShapeTool->GetAllComponentSHUO(label, seq)) {
593  return;
594  }
595  std::ostringstream ss;
596  for (int i = 1; i <= seq.Length(); ++i) {
597  Handle(XCAFDoc_GraphNode) shuo = Handle(XCAFDoc_GraphNode)::DownCast(seq.Value(i));
598  if (shuo.IsNull()) {
599  continue;
600  }
601 
602  TDF_Label slabel = shuo->Label();
603 
604  // We only want to process the main shuo, i.e. those without upper_usage
605  TDF_LabelSequence uppers;
606  aShapeTool->GetSHUOUpperUsage(slabel, uppers);
607  if (uppers.Length()) {
608  continue;
609  }
610 
611  // appendFirst tells us whether we shall append the object name of the first label
612  bool skipFirst = !appendFirst;
613  ss.str("");
614  while (true) {
615  if (skipFirst) {
616  skipFirst = false;
617  }
618  else {
619  TDF_Label l = shuo->Label().Father();
620  auto it = myNames.find(l);
621  if (it == myNames.end()) {
622  FC_WARN("Failed to find object of label " << Tools::labelName(l));
623  ss.str("");
624  break;
625  }
626  if (!it->second.empty()) {
627  ss << it->second << '.';
628  }
629  }
630  if (!shuo->NbChildren()) {
631  break;
632  }
633  shuo = shuo->GetChild(1);
634  }
635  std::string subname = ss.str();
636  if (subname.empty()) {
637  continue;
638  }
639  if (!aColorTool->IsVisible(slabel)) {
640  subname += App::DocumentObject::hiddenMarker();
641  colors.emplace(subname, App::Color());
642  }
643  else {
644  Quantity_ColorRGBA aColor;
645  if (aColorTool->GetColor(slabel, XCAFDoc_ColorSurf, aColor)
646  || aColorTool->GetColor(slabel, XCAFDoc_ColorGen, aColor)) {
647  colors.emplace(subname, Tools::convertColor(aColor));
648  }
649  }
650  }
651 }

653 App::DocumentObject* ImportOCAF2::loadShape(App::Document* doc,
654  TDF_Label label,
655  const TopoDS_Shape& shape,
656  bool baseOnly,
657  bool newDoc)
658 {
659  if (shape.IsNull()) {
660  return nullptr;
661  }
662 
663  auto baseShape = shape.Located(TopLoc_Location());
664  auto it = myShapes.find(baseShape);
665  if (it == myShapes.end()) {

744 bool ImportOCAF2::createAssembly(App::Document* _doc,
745  TDF_Label label,
746  const TopoDS_Shape& shape,
747  Info& info,
748  bool newDoc)
749 {
750  (void)label;
751 
752  std::vector<App::DocumentObject*> children;
753  std::map<App::DocumentObject*, ChildInfo> childrenMap;
754  boost::dynamic_bitset<> visibilities;
755  std::map<std::string, App::Color> shuoColors;
756 
757  auto doc = _doc;
758  if (newDoc) {
759  doc = getDocument(_doc, label);
760  }
761 
762  for (TopoDS_Iterator it(shape, Standard_False, Standard_False); it.More(); it.Next()) {
763  TopoDS_Shape childShape = it.Value();
764  if (childShape.IsNull()) {
765  continue;
766  }
767  TDF_Label childLabel;
768  aShapeTool->Search(childShape, childLabel, Standard_True, Standard_True, Standard_False);
769  if (!childLabel.IsNull() && !options.importHidden && !aColorTool->IsVisible(childLabel)) {
770  continue;
771  }
772  auto obj = loadShape(doc, childLabel, childShape, options.reduceObjects);
773  if (!obj) {
774  continue;
775  }
776  bool vis = true;
777  if (!childLabel.IsNull() && aShapeTool->IsComponent(childLabel)) {
778  vis = aColorTool->IsVisible(childLabel);
779  }
780  if (!options.reduceObjects) {
781  visibilities.push_back(vis);
782  children.push_back(obj);
783  getSHUOColors(childLabel, shuoColors, true);
784  continue;
785  }
786 
787  auto& childInfo = childrenMap[obj];
788  if (childInfo.plas.empty()) {
789  children.push_back(obj);
790  visibilities.push_back(vis);
791  childInfo.shape = childShape;
792  }
793 
794  childInfo.vis.push_back(vis);
795  childInfo.labels.push_back(childLabel);
796  childInfo.plas.emplace_back(
797  Part::TopoShape::convert(childShape.Location().Transformation()));
798  Quantity_ColorRGBA aColor;
799  if (aColorTool->GetColor(childShape, XCAFDoc_ColorSurf, aColor)) {
800  childInfo.colors[childInfo.plas.size() - 1] = Tools::convertColor(aColor);
801  }
802  }
803  assert(visibilities.size() == children.size());
804 
805  if (children.empty()) {
806  if (doc != _doc) {
807  App::GetApplication().closeDocument(doc->getName());
808  }
809  return false;
810  }
811 
812  if (options.reduceObjects) {
813  int i = -1;
814  for (auto& child : children) {
815  ++i;
816  auto& childInfo = childrenMap[child];
817  if (childInfo.plas.size() == 1) {
818  child = loadShape(doc, childInfo.labels.front(), childInfo.shape);
819  getSHUOColors(childInfo.labels.front(), shuoColors, true);
820  continue;
821  }
822 
823  visibilities[i] = true;
824 
825  // Okay, we are creating a link array
826  auto link = static_cast<App::Link*>(doc->addObject("App::Link", "Link"));
827  link->setLink(-1, child);
828  link->ShowElement.setValue(false);
829  link->ElementCount.setValue(childInfo.plas.size());
830  auto it = myCollapsedObjects.find(child);
831  if (it != myCollapsedObjects.end()) {
832  // child is a single component assembly that has been
833  // collapsed, so we have to honour its placement
834  for (auto& pla : childInfo.plas) {
835  pla *= it->second->getValue();
836  }
837  }
838  link->PlacementList.setValue(childInfo.plas);
839  link->VisibilityList.setValue(childInfo.vis);
840 
841  for (auto& v : childInfo.colors) {
842  applyLinkColor(link, v.first, v.second);
843  }
844 
845  int i = 0;
846  std::string name = link->getNameInDocument();
847  name += '.';
848  for (auto childLabel : childInfo.labels) {
849  myNames.emplace(childLabel, name + std::to_string(i++));
850  getSHUOColors(childLabel, shuoColors, true);
851  }
852 
853  child = link;
854  Info objInfo;
855  objInfo.obj = child;
856  setObjectName(objInfo, childInfo.labels.front());
857  }
858  }
859 
860  if (children.empty()) {
861  return false;
862  }
863 
864  if (!createGroup(doc, info, shape, children, visibilities, shuoColors.empty())) {
865  return false;
866  }
867  if (!shuoColors.empty()) {
868  applyElementColors(info.obj, shuoColors);
869  }
870  return true;
871 }
