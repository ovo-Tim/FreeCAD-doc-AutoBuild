178 void Workbench::activated()
179 {
180  /* When the workbench is activated, it may happen that we are in edit mode or not.
181  * If we are not in edit mode, the function enterEditMode (called by the ViewProvider) takes
182  * care to save the state of toolbars outside of edit mode. We cannot do it here, as we are
183  * coming from another WB.
184  *
185  * If we moved to another WB from edit mode, the new WB was activated before deactivating this.
186  * Therefore we had no chance to tidy up the save state. We assume a loss of any CHANGE to
187  * toolbar configuration since last entering edit mode in this case (for any change in
188  * configuration to be stored, the edit mode must be left while the selected Workbench is the
189  * sketcher WB).
190  *
191  * However, now that we are back (from another WB), we need to make the toolbars available.
192  * These correspond to the last saved state.
193  */
194  Gui::Document* doc = Gui::Application::Instance->activeDocument();
195  if (isSketchInEdit(doc)) {
196  Gui::ToolBarManager::getInstance()->setState(editModeToolbarNames(),
197  Gui::ToolBarManager::State::ForceAvailable);
198  }
199 }

201 void Workbench::enterEditMode()
202 {
203  /* Ensure the state left by the non-edit mode toolbars is saved (in case of changing to edit
204  * mode) without changing workbench
205  */
206  Gui::ToolBarManager::getInstance()->setState(nonEditModeToolbarNames(),
207  Gui::ToolBarManager::State::SaveState);
208 
209  Gui::ToolBarManager::getInstance()->setState(editModeToolbarNames(),
210  Gui::ToolBarManager::State::ForceAvailable);
211  Gui::ToolBarManager::getInstance()->setState(nonEditModeToolbarNames(),
212  Gui::ToolBarManager::State::ForceHidden);
213 }
214 
215 void Workbench::leaveEditMode()
216 {
217  /* Ensure the state left by the edit mode toolbars is saved (in case of changing to edit mode)
218  * without changing workbench.
219  *
220  * However, do not save state if the current workbench is not the Sketcher WB, because otherwise
221  * we would be saving the state of the currently activated workbench, and the toolbars would
222  * disappear (as the toolbars of that other WB are only visible).
223  */
224  auto* workbench = Gui::WorkbenchManager::instance()->active();
225 
226  if (workbench->name() == "SketcherWorkbench") {
227  Gui::ToolBarManager::getInstance()->setState(editModeToolbarNames(),
228  Gui::ToolBarManager::State::SaveState);
229  }
230 
231  Gui::ToolBarManager::getInstance()->setState(editModeToolbarNames(),
232  Gui::ToolBarManager::State::RestoreDefault);
233  Gui::ToolBarManager::getInstance()->setState(nonEditModeToolbarNames(),
234  Gui::ToolBarManager::State::RestoreDefault);
235 }
