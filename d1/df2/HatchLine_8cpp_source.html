80 Base::Vector3d LineSet::calcApparentStart(TechDraw::BaseGeomPtr g)
81 {
82  Base::Vector3d start(g->getStartPoint().x, g->getStartPoint().y, 0.0);
83  double angle = getPATLineSpec().getAngle();
84  if (angle == 0.0) { //horizontal
85  return Base::Vector3d(getMinX(), start.y, 0.0);
86  } else if ((angle == 90.0) ||
87  (angle == -90.0)) { //vertical
88  return Base::Vector3d(start.x, getMinY(), 0.0);
89  } else {
90  double slope = getPATLineSpec().getSlope();
91  double y = getMinY();
92  double x = ((y - start.y) / slope) + start.x;
93  return Base::Vector3d(x, y,0);
94  }
95 }

111 Base::Vector3d LineSet::getUnitOrtho()
112 {
113  Base::Vector3d result;
114  Base::Vector3d unit = getUnitDir();
115  Base::Vector3d X(1.0, 0.0, 0.0);
116  Base::Vector3d Y(0.0, 1.0, 0.0);
117  if (unit.IsEqual(X, 0.000001)) {
118  result = Y;
119  } else if (unit.IsEqual(Y, 0.000001)) {
120  result = X;
121  } else {
122  double unitX = unit.x;
123  double unitY = unit.y;
124  result = Base::Vector3d(unitY, -unitX, 0.0); //perpendicular
125  }
126  result.Normalize(); //probably redundant
127  return result;

131 Base::Vector3d LineSet::findAtomStart()
132 {
133  Base::Vector3d origin = getOrigin();
134  double angle = getAngle();
135  if (angle == 0.0) {
136  return Base::Vector3d(getMinX(), origin.y, 0.0);
137  } else if ( (angle == 90.0) ||
138  (angle == -90.0) ) {
139  return Base::Vector3d(origin.x, getMinY(), 0.0);

147 Base::Vector3d LineSet::getPatternStartPoint(TechDraw::BaseGeomPtr g, double &offset, double scale)
148 {
149  Base::Vector3d result = getOrigin();
150  Base::Vector3d atomStart = findAtomStart();
151  Base::Vector3d thisStart = calcApparentStart(g);
152  double angle = getAngle();
153  double patternLength = scale * getPatternLength();
154  Base::Vector3d lineOrigin;
155  double distX, distY;
156  int overlayIndex = 0;
157  //interval & offset are not patternscaled
158 
159 //figure out which line this is in the overlay
160  if (angle == 0.0) { //odd case - horizontal line
161  distY = (thisStart.y - atomStart.y); //this is patternscaled
162  overlayIndex = (int) round(distY/(scale * getIntervalY())); //this is +/-
163  lineOrigin = getOrigin() + distY * Base::Vector3d(0.0, 1.0, 0.0); //move u/d
164  } else {
165  distX = (thisStart.x - atomStart.x); //this is patternscaled
166  overlayIndex = (int) round(distX/(scale * getIntervalX()));
167  lineOrigin = getOrigin() + overlayIndex * (scale * getInterval()) * getUnitOrtho();
168  }
169 
170  lineOrigin = lineOrigin + (overlayIndex * (scale * getOffset())) * getUnitDir(); //move along line
171 
172  //is lineOrigin on line of g? should be within fp error
173  Base::Vector3d gStart(g->getStartPoint().x,
174  g->getStartPoint().y,
175  0.0);
176  Base::Vector3d gEnd(g->getEndPoint().x,
177  g->getEndPoint().y,
178  0.0);
179  double lenStartOrg = (gStart - lineOrigin).Length();
180  double lenEndOrg = (gEnd - lineOrigin).Length();
181  double lenStartEnd = (gStart - gEnd).Length();
182  if ( (lenStartOrg <= lenStartEnd) &&
183  (lenEndOrg <= lenStartEnd) ) { //origin is in g
184  result = lineOrigin;
185  offset = 0.0;
186  } else {
187  //find a point where pattern repeats within g
188  double patsStartOrg = lenStartOrg/patternLength; //# pattern repeats from lineOrigin to start
189  double patsEndOrg = lenEndOrg/patternLength;
190  if (lenStartOrg < lenEndOrg) { //origin is before start
191  double c = ceil(patsStartOrg);
192  if (c <= patsEndOrg) { //c is an integer pattern count in [patsStartOrg, patsEndOrg]
193  result = lineOrigin + c*patternLength*getUnitDir();
194  offset = 0.0;
195  } else {
196 // //ugly case - partial pattern
197  result = gStart;
198  offset = patsStartOrg - (int)patsStartOrg; //fraction of a patternLength
199  offset = offset * patternLength;
200  }
201  } else if (lenStartOrg > lenEndOrg) { //origin is after end
202  double c = ceil(patsEndOrg);
203  if (c <= patsStartOrg) { //c is an integer pattern count in [patsStartOrg, patsEndOrg]
204  result = lineOrigin - c*patternLength*getUnitDir();
205  offset = 0.0;
206  } else {
207  result = gStart;
208  offset = ceil(patsStartOrg) - patsStartOrg; //fraction of a patternLength patstartorg to repeat point
209  offset = offset * patternLength;
210  }
211  }
212  }
213  return result;
214 }

241 void PATLineSpec::load(std::string& lineSpec)
242 {
243  std::vector<double> values = split(lineSpec);
244  if (values.size() < 5) {
245  Base::Console().Message( "PATLineSpec::load(%s) invalid entry in pattern\n", lineSpec.c_str() );
246  return;
247  }
248  m_angle = values[0];
249  m_origin = Base::Vector3d(values[1], values[2], 0.0);
250  m_offset = values[3];
251  m_interval = values[4];
252  if (values.size() > 5) {
253  std::vector<double> dash;
254  dash.insert(std::end(dash), std::begin(values) + 5, std::end(values));
255  m_dashParms = DashSpec(dash);
256  }
257 }

259 std::vector<double> PATLineSpec::split(std::string line)
260 {
261  std::vector<double> result;
262  std::stringstream lineStream(line);
263  std::string cell;
264 
265  while(std::getline(lineStream, cell, ','))
266  {
267  try {
268  result.push_back(std::stod(cell));
269  }
270  catch (const std::invalid_argument& ia) {
271  Base::Console().Warning("Invalid number in cell: %s (%s) \n", cell.c_str(), ia.what());
272  result.push_back(0.0);
273  }
274  }
275  return result;

296 {
297  std::vector<std::string> lineSpecs;

299  Base::ifstream inFile;

303  return std::vector<PATLineSpec>();

305 
306  //get all the definition lines for this pattern

311  lineSpecs = loadPatternDef(inFile);
312 
313  //decode definition lines into PATLineSpec objects
314  std::vector<PATLineSpec> result;

316  PATLineSpec hl(l);
317  result.push_back(hl);
318  }
319  return result;
320 }

322 bool PATLineSpec::findPatternStart(std::ifstream& inFile, std::string& parmName)
323 {
324 // Base::Console().Message("HL::findPatternStart() - parmName: %s\n", parmName.c_str());
325  while (inFile.good() ){
326  std::string line;
327  std::getline(inFile, line);
328  std::string nameTag = line.substr(0, 1);
329  std::string patternName;
330  std::size_t commaPos;
331  if (nameTag == ";" ||
332  nameTag == " " ||
333  line.empty()) { //is cr/lf empty?
334  continue;
335  } else if (nameTag == "*") {
336  commaPos = line.find(',',1);
337  if (commaPos != std::string::npos) {
338  patternName = line.substr(1, commaPos-1);
339  } else {
340  patternName = line.substr(1);
341  }
342  if (patternName == parmName) {
343  //this is our pattern
344  return true;
345  }
346  }
347  } //endwhile
348  return false;
349 }
