125 template<typename T>
126 inline std::size_t base64_decode(T& out, char const* in, std::size_t len)
127 {
128  std::size_t size = out.size();
129  out.resize(size + base64_decode_size(len));
130  std::pair<std::size_t, std::size_t> res = base64_decode(&out[size], in, len);
