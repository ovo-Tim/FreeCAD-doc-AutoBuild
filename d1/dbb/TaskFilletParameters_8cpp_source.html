48 TaskFilletParameters::TaskFilletParameters(ViewProviderDressUp* DressUpView, QWidget* parent)
49  : TaskDressUpParameters(DressUpView, true, true, parent)
50  , ui(new Ui_TaskFilletParameters)
51 {
52  // we need a separate container widget to add all controls to
53  proxy = new QWidget(this);
54  ui->setupUi(proxy);
55  this->groupLayout()->addWidget(proxy);
56 
57  PartDesign::Fillet* pcFillet = DressUpView->getObject<PartDesign::Fillet>();
58  bool useAllEdges = pcFillet->UseAllEdges.getValue();
59  ui->checkBoxUseAllEdges->setChecked(useAllEdges);
60  ui->buttonRefSel->setEnabled(!useAllEdges);
61  ui->listWidgetReferences->setEnabled(!useAllEdges);
62  double r = pcFillet->Radius.getValue();
63 
64  ui->filletRadius->setUnit(Base::Unit::Length);
65  ui->filletRadius->setValue(r);
66  ui->filletRadius->setMinimum(0);
67  ui->filletRadius->selectNumber();
68  ui->filletRadius->bind(pcFillet->Radius);
69  QMetaObject::invokeMethod(ui->filletRadius, "setFocus", Qt::QueuedConnection);
70  std::vector<std::string> strings = pcFillet->Base.getSubValues();
71  for (const auto& string : strings) {
72  ui->listWidgetReferences->addItem(QString::fromStdString(string));
73  }
74 
75  QMetaObject::connectSlotsByName(this);
76 
77  // clang-format off
78  connect(ui->filletRadius, qOverload<double>(&Gui::QuantitySpinBox::valueChanged),
79  this, &TaskFilletParameters::onLengthChanged);
80  connect(ui->buttonRefSel, &QToolButton::toggled,
81  this, &TaskFilletParameters::onButtonRefSel);
82  connect(ui->checkBoxUseAllEdges, &QToolButton::toggled,
83  this, &TaskFilletParameters::onCheckBoxUseAllEdgesToggled);
84 
85  // Create context menu
86  createDeleteAction(ui->listWidgetReferences);
87  connect(deleteAction, &QAction::triggered, this, &TaskFilletParameters::onRefDeleted);
88 
89  createAddAllEdgesAction(ui->listWidgetReferences);
90  connect(addAllEdgesAction, &QAction::triggered, this, &TaskFilletParameters::onAddAllEdges);
91 
92  connect(ui->listWidgetReferences, &QListWidget::currentItemChanged,
93  this, &TaskFilletParameters::setSelection);
94  connect(ui->listWidgetReferences, &QListWidget::itemClicked,
95  this, &TaskFilletParameters::setSelection);
96  connect(ui->listWidgetReferences, &QListWidget::itemDoubleClicked,
97  this, &TaskFilletParameters::doubleClicked);
98  // clang-format on
99 
100  if (strings.empty()) {
101  setSelectionMode(refSel);
102  }
103  else {
104  hideOnError();
105  }
106 }
