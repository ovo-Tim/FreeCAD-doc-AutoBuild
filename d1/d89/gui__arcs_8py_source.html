104  def updateAngle(self, angle):
105  """Update the angle with the new value."""
106  # previous absolute angle
107  lastangle = self.firstangle + self.angle
108  if lastangle <= -2 * math.pi:
109  lastangle += 2 * math.pi
110  if lastangle >= 2 * math.pi:
111  lastangle -= 2 * math.pi
112  # compute delta = change in angle:
113  d0 = angle - lastangle
114  d1 = d0 + 2 * math.pi
115  d2 = d0 - 2 * math.pi
116  if abs(d0) < min(abs(d1), abs(d2)):
117  delta = d0
118  elif abs(d1) < abs(d2):
119  delta = d1
120  else:
121  delta = d2
122  newangle = self.angle + delta
123  # normalize angle, preserving direction
124  if newangle >= 2 * math.pi:
125  newangle -= 2 * math.pi
126  if newangle <= -2 * math.pi:
127  newangle += 2 * math.pi
128  self.angle = newangle

506  def getPoint(self, point, info):
507  """Get the point by clicking on the 3D view.

537  # the arc tracker to show the preview of the final curve.

541  callback=self.getPoint,
542  movecallback=self.drawArc)

545  icon="Draft_Arc_3Points")
546  Gui.Snapper.ui.continueCmd.show()
547 
548  else:
