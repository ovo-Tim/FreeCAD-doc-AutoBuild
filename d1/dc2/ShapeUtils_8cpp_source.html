81 gp_Ax2 ShapeUtils::getViewAxis(const Base::Vector3d origin, const Base::Vector3d& direction,
82  const bool flip)
83 {
84  // Base::Console().Message("GO::getViewAxis() - 1 - use only with getLegacyX\n");
85  (void)flip;
86  gp_Ax2 viewAxis;
87  gp_Pnt inputCenter(origin.x, origin.y, origin.z);
88  Base::Vector3d stdZ(0.0, 0.0, 1.0);
89  Base::Vector3d stdOrg(0.0, 0.0, 0.0);
90  Base::Vector3d cross = direction;
91  if (DU::checkParallel(direction, stdZ)) {
92  cross = Base::Vector3d(1.0, 0.0, 0.0);
93  }
94  else {
95  cross.Normalize();
96  cross = cross.Cross(stdZ);
97  }
98 
99  if (cross.IsEqual(stdOrg, FLT_EPSILON)) {
100  viewAxis = gp_Ax2(inputCenter, gp_Dir(direction.x, direction.y, direction.z));
101  return viewAxis;
102  }
103 
104  viewAxis = gp_Ax2(inputCenter, gp_Dir(direction.x, direction.y, direction.z),
105  gp_Dir(cross.x, cross.y, cross.z));
106  return viewAxis;
107 }

124 gp_Ax2 ShapeUtils::legacyViewAxis1(const Base::Vector3d origin, const Base::Vector3d& direction,
125  const bool flip)
126 {
127  // Base::Console().Message("GO::legacyViewAxis1()\n");
128  gp_Pnt inputCenter(origin.x, origin.y, origin.z);
129  Base::Vector3d stdZ(0.0, 0.0, 1.0);
130  Base::Vector3d stdOrg(0.0, 0.0, 0.0);
131  Base::Vector3d flipDirection(direction.x, -direction.y, direction.z);
132  if (!flip) {
133  flipDirection = Base::Vector3d(direction.x, direction.y, direction.z);
134  }
135  Base::Vector3d cross = flipDirection;
136  // //special case
137  if (DU::checkParallel(flipDirection, stdZ)) {
138  cross = Base::Vector3d(1.0, 0.0, 0.0);
139  }
140  else {
141  cross.Normalize();
142  cross = cross.Cross(stdZ);
143  }
144 
145  if (cross.IsEqual(stdOrg, FLT_EPSILON)) {
146  return gp_Ax2(inputCenter, gp_Dir(flipDirection.x, flipDirection.y, flipDirection.z));
147  }
148 
149  gp_Ax2 viewAxis = gp_Ax2(inputCenter,
150  gp_Dir(flipDirection.x, flipDirection.y, flipDirection.z),
151  gp_Dir(cross.x, cross.y, cross.z));
152 
153  //this bit is to handle the old mirror Y logic, but it messes up
154  //some old files.
155  gp_Trsf mirrorXForm;
156  gp_Ax2 mirrorCS(inputCenter, gp_Dir(0, -1, 0));
157  mirrorXForm.SetMirror(mirrorCS);
158  viewAxis = viewAxis.Transformed(mirrorXForm);
159 
160  return viewAxis;
161 }

165 {

171  tBounds.Get(xMin, yMin, zMin, xMax, yMax, zMax);

188 gp_Pnt ShapeUtils::findCentroid(const TopoDS_Shape& shape, const gp_Ax2& viewAxis)
189 {
190  // Base::Console().Message("GO::findCentroid() - 2\n");
191 
192  gp_Trsf tempTransform;
193  tempTransform.SetTransformation(viewAxis);
194  BRepBuilderAPI_Transform builder(shape, tempTransform);
195 
196  Bnd_Box tBounds;
197  tBounds.SetGap(0.0);
198  BRepBndLib::AddOptimal(builder.Shape(), tBounds, true, false);
199 
200  Standard_Real xMin, yMin, zMin, xMax, yMax, zMax;
201  tBounds.Get(xMin, yMin, zMin, xMax, yMax, zMax);
202 
203  Standard_Real x = (xMin + xMax) / 2.0, y = (yMin + yMax) / 2.0, z = (zMin + zMax) / 2.0;
204 
205  // Get centroid back into object space
206  tempTransform.Inverted().Transforms(x, y, z);
207 
208  return gp_Pnt(x, y, z);
209 }

234 TopoDS_Shape ShapeUtils::mirrorShape(const TopoDS_Shape& input, const gp_Pnt& inputCenter,
235  double scale)
236 {
237  TopoDS_Shape transShape;
238  if (input.IsNull()) {
239  return transShape;
240  }
241  try {
242  // Make tempTransform scale the object around its centre point and
243  // mirror about the Y axis
244  gp_Trsf tempTransform;
245  //BRepBuilderAPI_Transform will loop forever if asked to use 0.0 as scale
246  if (scale <= 0.0) {
247  tempTransform.SetScale(inputCenter, 1.0);
248  }
249  else {
250  tempTransform.SetScale(inputCenter, scale);
251  }
252  gp_Trsf mirrorTransform;
253  mirrorTransform.SetMirror(gp_Ax2(inputCenter, gp_Dir(0, -1, 0)));
254  tempTransform.Multiply(mirrorTransform);
255 
256  // Apply that transform to the shape. This should preserve the centre.
257  BRepBuilderAPI_Transform mkTrf(input, tempTransform);
258  transShape = mkTrf.Shape();
259  }
260  catch (...) {
261  return transShape;
262  }
263  return transShape;
264 }

416 bool ShapeUtils::edgesAreParallel(TopoDS_Edge edge0, TopoDS_Edge edge1)
417 {
418  std::pair<Base::Vector3d, Base::Vector3d> ends0 = getEdgeEnds(edge0);
419  Base::Vector3d vec0 = ends0.second - ends0.first;
420  vec0.Normalize();
421  std::pair<Base::Vector3d, Base::Vector3d> ends1 = getEdgeEnds(edge1);
422  Base::Vector3d vec1 = ends1.second - ends1.first;
423  vec1.Normalize();
424  double dot = fabs(vec0.Dot(vec1));
425  if (DU::fpCompare(dot, 1.0, EWTOLERANCE)) {
426  // parallel vectors
427  return true;
428  }
429  return false;
430 
431 }
