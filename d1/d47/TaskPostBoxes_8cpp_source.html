277 void TaskDlgPost::connectSlots()
278 {
279  // Connect emitAddedFunction() with slotAddedFunction()
280  QObject* sender = nullptr;
281  int indexSignal = 0;
282  for (const auto dlg : m_boxes) {
283  indexSignal = dlg->metaObject()->indexOfSignal(
284  QMetaObject::normalizedSignature("emitAddedFunction()"));
285  if (indexSignal >= 0) {
286  sender = dlg;
287  break;
288  }
289  }
290 
291  if (sender) {
292  for (const auto dlg : m_boxes) {
293  int indexSlot = dlg->metaObject()->indexOfSlot(
294  QMetaObject::normalizedSignature("slotAddedFunction()"));
295  if (indexSlot >= 0) {
296  connect(sender,
297  sender->metaObject()->method(indexSignal),
298  dlg,
299  dlg->metaObject()->method(indexSlot));
300  }
301  }
302  }
303 }
