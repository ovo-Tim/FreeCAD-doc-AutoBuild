307  void setFocusToParameter(unsigned int parameterindex)
308  {
309  // To be able to cycle through OVP and widget, we use a parameter index that goes from
310  // 0 to (onViewParameters.size() + nParameter)
311  if (!ControllerBase::setFocusToOnViewParameter(parameterindex)) {
312  parameterindex = parameterindex - ControllerBase::onViewParameters.size();
313 
314  if (parameterindex < static_cast<unsigned int>(nParameter)) {
315  toolWidget->setParameterFocus(parameterindex);
316  ControllerBase::parameterWithFocus =
317  ControllerBase::onViewParameters.size() + parameterindex;
318  }
319  }
320  }
321 
323  void passFocusToNextParameter()
324  {
325  unsigned int index = ControllerBase::parameterWithFocus + 1;
326 
327  if (index >= ControllerBase::onViewParameters.size() + nParameter) {
328  index = 0;
329  }
330 
331  auto trySetFocus = [this](unsigned int& idx) -> bool {
332  while (idx < ControllerBase::onViewParameters.size()) {
333  if (ControllerBase::isOnViewParameterOfCurrentMode(idx)
334  && ControllerBase::isOnViewParameterVisible(idx)) {
335  setFocusToParameter(idx);
336  return true;
337  }
338  idx++;
339  }
340  if (idx < ControllerBase::onViewParameters.size() + nParameter) {
341  setFocusToParameter(idx);
342  return true;
343  }
344  return false;
345  };
346 
347  if (!trySetFocus(index)) {
348  // We have not found a parameter in this mode after current.
349  // So we go back to start and retry.
350  index = 0;
351  trySetFocus(index);
352  }
353 
354  // At that point if no onViewParameter is found, there is none.
355  }

389  void resetDefaultWidget()
390  {
391  boost::signals2::shared_connection_block parameter_focus_block(
392  connectionParameterTabOrEnterPressed);
393  boost::signals2::shared_connection_block parameter_block(connectionParameterValueChanged);
394  boost::signals2::shared_connection_block checkbox_block(connectionCheckboxCheckedChanged);
395  boost::signals2::shared_connection_block combobox_block(connectionComboboxSelectionChanged);
396 
397  nParameter = WidgetParametersT::size(handler->constructionMethod());
398  nCheckbox = WidgetCheckboxesT::size(handler->constructionMethod());
399  nCombobox = WidgetComboboxesT::size(handler->constructionMethod());
400 
401  toolWidget->initNParameters(nParameter, ControllerBase::getKeyManager());
402  toolWidget->initNCheckboxes(nCheckbox);
403  toolWidget->initNComboboxes(nCombobox);
404 
405  configureToolWidget();
406 
407  // update the combobox only if necessary (if the change was not triggered by the
408  // combobox)
409  if constexpr (PFirstComboboxIsConstructionMethod == true) {
410  auto currentindex = toolWidget->getComboboxIndex(WCombobox::FirstCombo);
411  auto methodint = static_cast<int>(handler->constructionMethod());
412 
413  if (currentindex != methodint) {
414  // avoid triggering of method change
415  boost::signals2::shared_connection_block combobox_block(
416  connectionComboboxSelectionChanged);
417  toolWidget->setComboboxIndex(WCombobox::FirstCombo, methodint);
418  }
