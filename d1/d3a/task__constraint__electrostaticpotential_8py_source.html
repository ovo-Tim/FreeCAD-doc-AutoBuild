169  def _applyPotentialChanges(self, enabledBox, potentialQSB):
170  enabled = enabledBox.isChecked()
171  potential = None
172  try:
173  potential = potentialQSB.property("value")
174  except ValueError:
175  FreeCAD.Console.PrintMessage(
