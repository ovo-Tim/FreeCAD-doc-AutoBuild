66  def addBaseGeometry(self, selection):
67  added = False
68  shapes = self.obj.BaseShapes
69  for sel in selection:
70  job = PathUtils.findParentJob(self.obj)
71  base = job.Proxy.resourceClone(job, sel.Object)
72  if not base:
73  Path.Log.notice(
74  (translate("CAM", "%s is not a Base Model object of the job %s") + "\n")
75  % (sel.Object.Label, job.Label)
76  )
77  continue
78  if base in shapes:
79  Path.Log.notice(
80  (translate("CAM", "Base shape %s already in the list") + "\n")
81  % (sel.Object.Label)
82  )
83  continue
84  if base.isDerivedFrom("Part::Part2DObject"):
85  if sel.HasSubObjects:
86  # selectively add some elements of the drawing to the Base
87  for sub in sel.SubElementNames:
88  if "Vertex" in sub:
89  Path.Log.info("Ignoring vertex")
90  else:
91  self.obj.Proxy.addBase(self.obj, base, sub)
92  else:
93  # when adding an entire shape to BaseShapes we can take its sub shapes out of Base
94  self.obj.Base = [(p, el) for p, el in self.obj.Base if p != base]
95  shapes.append(base)
96  self.obj.BaseShapes = shapes
97  added = True
98  else:
99  # user wants us to engrave an edge of face of a base model
100  base = self.super().addBaseGeometry(selection)
101  added = added or base
102 
103  return added
