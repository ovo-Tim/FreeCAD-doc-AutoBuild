83 void GeometryFacade::initExtension()
84 {
85  if (!Geo->hasExtension(SketchGeometryExtension::getClassTypeId())) {

87  getGeo()->setExtension(std::make_unique<SketchGeometryExtension>()); // Create getExtension

94  (Geo->getExtension(SketchGeometryExtension::getClassTypeId())).lock());
95 }

97 void GeometryFacade::initExtension() const
98 {
99  // const Geometry without SketchGeometryExtension cannot initialise a GeometryFacade
100  if (!Geo->hasExtension(SketchGeometryExtension::getClassTypeId())) {
101  THROWM(Base::ValueError,
102  "Cannot create a GeometryFacade out of a const Geometry pointer not having a "
103  "SketchGeometryExtension!");
104  }
105 
106  auto ext = std::static_pointer_cast<const SketchGeometryExtension>(
107  Geo->getExtension(SketchGeometryExtension::getClassTypeId()).lock());
108 
109  const_cast<GeometryFacade*>(this)->SketchGeoExtension = ext;
110 }
