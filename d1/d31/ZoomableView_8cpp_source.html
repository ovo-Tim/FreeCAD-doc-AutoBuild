169 void ZoomableView::updateView(void)
170 {
171  /* QGraphicsView has hardcoded margins therefore we have to avoid fitInView
172  * Find more information at https://bugreports.qt.io/browse/QTBUG-42331 */
173 
174  const qreal scale_factor = static_cast<qreal>(m_zoomLevel) / 100.0,
175  new_w = static_cast<qreal>(viewport()->rect().width()) / scale_factor,
176  new_h = static_cast<qreal>(viewport()->rect().height()) / scale_factor;
177 
178  const QRectF new_geometry {0.0, 0.0, new_w, new_h};
179 
180  const QRect old_geometry {stv->geometry()};
181  stv->setGeometry(1, 1, old_geometry.width() - 1, old_geometry.height() - 1);
182 
183  resetTransform();
184  qpw->setGeometry(new_geometry);
185  setSceneRect(new_geometry);
186  scale(scale_factor, scale_factor);
187  centerOn(new_geometry.center());
188 }

190 void ZoomableView::keyPressEvent(QKeyEvent* event)
191 {
192  if (event->modifiers() & Qt::ControlModifier) {

196  event->accept();
197  break;
198  case Qt::Key_Minus:
199  zoomOut();
200  event->accept();
201  break;
202  case Qt::Key_0:
203  resetZoom();
204  event->accept();
205  break;
206  default:
207  QGraphicsView::keyPressEvent(event);
208  }
209  }
210  else {
211  QGraphicsView::keyPressEvent(event);
212  }
213 }
