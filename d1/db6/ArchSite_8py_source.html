64 def toNode(shape):
65 
66  """builds a linear pivy node from a shape"""
67 
68  from pivy import coin
69  buf = shape.writeInventor(2,0.01).replace("\n","")
70  buf = re.findall(r"point \[(.*?)\]",buf)
71  pts = []
72  for c in buf:
73  pts.extend(zip(*[iter( c.split() )]*3) )
74  pc = []
75  for v in pts:
76  v = [float(v[0]),float(v[1]),float(v[2])]
77  if (not pc) or (pc[-1] != v):
78  pc.append(v)
79  coords = coin.SoCoordinate3()
80  coords.point.setValues(0,len(pc),pc)

82  line.numVertices.setValue(-1)
83  item = coin.SoSeparator()
84  item.addChild(coords)
85  item.addChild(line)
86  return item

274 def makeWindRose(epwfile,scale=1,sectors=24):
275 
276  """makeWindRose(site,sectors):
277  returns a wind rose diagram as a pivy node"""
278 
279  try:
280  import ladybug
281  from ladybug import epw
282  except Exception:
283  FreeCAD.Console.PrintError("The ladybug module was not found. Unable to generate solar diagrams\n")
284  return None
285  if not epwfile:
286  FreeCAD.Console.PrintWarning("No EPW file, unable to generate wind rose.\n")
287  return None
288  epw_data = ladybug.epw.EPW(epwfile)
289  baseangle = 360/sectors
290  sectorangles = [i * baseangle for i in range(sectors)] # the divider angles between each sector
291  basebissect = baseangle/2
292  angles = [basebissect] # build a list of central direction for each sector
293  for i in range(1,sectors):
294  angles.append(angles[-1]+baseangle)
295  windsbysector = [0 for i in range(sectors)] # prepare a holder for values for each sector
296  for hour in epw_data.wind_direction:
297  sector = min(angles, key=lambda x:abs(x-hour)) # find the closest sector angle
298  sectorindex = angles.index(sector)
299  windsbysector[sectorindex] = windsbysector[sectorindex] + 1
300  maxwind = max(windsbysector)
301  windsbysector = [wind/maxwind for wind in windsbysector] # normalize
302  vectors = [] # create 3D vectors
303  dividers = []
304  for i in range(sectors):
305  angle = math.radians(90 + angles[i])
306  x = math.cos(angle) * windsbysector[i] * scale
307  y = math.sin(angle) * windsbysector[i] * scale
308  vectors.append(FreeCAD.Vector(x,y,0))
309  secangle = math.radians(90 + sectorangles[i])
310  x = math.cos(secangle) * scale
311  y = math.sin(secangle) * scale
312  dividers.append(FreeCAD.Vector(x,y,0))
313  vectors.append(vectors[0])
314 
315  # build coin node
316  import Part
317  from pivy import coin
318  masternode = coin.SoSeparator()
319  for r in (0.25,0.5,0.75,1.0):
320  c = Part.makeCircle(r * scale)
321  masternode.addChild(toNode(c))
322  for divider in dividers:
323  l = Part.makeLine(FreeCAD.Vector(),divider)
324  masternode.addChild(toNode(l))
325  ds = coin.SoDrawStyle()
326  ds.lineWidth = 2.0
327  masternode.addChild(ds)
328  d = Part.makePolygon(vectors)
329  masternode.addChild(toNode(d))
330  return masternode
