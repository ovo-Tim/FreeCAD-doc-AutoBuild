73 std::string Tools::labelName(TDF_Label label)

88 void Tools::printLabel(TDF_Label label,
89  Handle(XCAFDoc_ShapeTool) aShapeTool,
90  Handle(XCAFDoc_ColorTool) aColorTool,
91  const char* msg)
92 {
93  if (label.IsNull() || !FC_LOG_INSTANCE.isEnabled(FC_LOGLEVEL_LOG)) {
94  return;
95  }
96  if (!msg) {
97  msg = "Label: ";
98  }
99  TCollection_AsciiString entry;
100  TDF_Tool::Entry(label, entry);
101  std::ostringstream ss;
102  ss << msg << entry << ", " << labelName(label) << (aShapeTool->IsShape(label) ? ", shape" : "")
103  << (aShapeTool->IsTopLevel(label) ? ", topLevel" : "")
104  << (aShapeTool->IsFree(label) ? ", free" : "")
105  << (aShapeTool->IsAssembly(label) ? ", assembly" : "")
106  << (aShapeTool->IsSimpleShape(label) ? ", simple" : "")
107  << (aShapeTool->IsCompound(label) ? ", compound" : "")
108  << (aShapeTool->IsReference(label) ? ", reference" : "")
109  << (aShapeTool->IsComponent(label) ? ", component" : "")
110  << (aShapeTool->IsSubShape(label) ? ", subshape" : "");
111  if (aShapeTool->IsSubShape(label)) {
112  auto shape = aShapeTool->GetShape(label);
113  if (!shape.IsNull()) {
114  ss << ", " << Part::TopoShape::shapeName(shape.ShapeType(), true);
115  }
116  }
117  if (aShapeTool->IsShape(label)) {
118  Quantity_ColorRGBA c;
119  if (aColorTool->GetColor(label, XCAFDoc_ColorGen, c)) {
120  ss << ", gc: " << c;
121  }
122  if (aColorTool->GetColor(label, XCAFDoc_ColorSurf, c)) {
123  ss << ", sc: " << c;
124  }
125  if (aColorTool->GetColor(label, XCAFDoc_ColorCurv, c)) {
126  ss << ", cc: " << c;
127  }
128  }
129 
130  ss << std::endl;
131  Base::Console().Notify<Base::LogStyle::Log>("ImportOCAF2", ss.str().c_str());
132 }
