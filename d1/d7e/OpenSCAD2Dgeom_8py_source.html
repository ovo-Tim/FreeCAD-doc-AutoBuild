250 def endpointdistance(edges):
251  '''return the distance of vertices in path (list of edges) as
252  maximum, minimum and distance between start and endpoint
253  it expects the edges to be traversed forward starting from Vertex 0'''
254  numedges=len(edges)
255  if numedges == 1 and len(edges[0].Vertexes) == 1:
256  return 0.0,0.0,0.0
257  outerdistance = edges[0].Vertexes[0].Point.sub(\
258  edges[-1].Vertexes[-1].Point).Length
259  if numedges > 1:
260  innerdistances=[edges[i].Vertexes[-1].Point.sub(edges[i+1].\
261  Vertexes[0].Point).Length for i in range(numedges-1)]
262  return max(innerdistances),min(innerdistances),outerdistance
263  else:
264  return 0.0,0.0,outerdistance
