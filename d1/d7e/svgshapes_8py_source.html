51 
52  TODO: check if the same function for SVG and DXF projection can be used
53  so that this function is not just duplicated code.
54  This function may also be present elsewhere, like `WorkingPlane`
55  or `DraftGeomUtils`, so we should avoid code duplication.
56 
57  Parameters
58  ----------
59  vec: Base::Vector3
60  An arbitrary vector that will be projected on the U and V directions.
61 
62  plane: WorkingPlane.PlaneBase
63  Working plane.
64  """
65  if not plane:
66  return vec

69  lx = nx.Length
70

72  lx = -lx
73 
74  ny = DraftVecUtils.project(vec, plane.v)

76 
77  if abs(ny.getAngle(plane.v)) > 0.1:
78  ly = -ly
79 
80  # if techdraw: buggy - we now simply do it at the end
81  # ly = -ly
82  return App.Vector(lx, ly, 0)

156  # TODO: trap only specific exception.
157  # Check the problem. Split didn't produce a two element list?
158  # _wrn("Circle or ellipse: "
159  # "cannot split the projection snip "
160  # "obtained by 'projectToSVG', "
161  # "continue manually.")

173  # otherwise it will process the `edata` further

176  # Complete ellipse not only arc
177  # svg = get_ellipse(plane,
178  # fill, stroke, linewidth,
179  # lstyle, edge)
180  # return svg

183  _diff = (center.LastParameter - center.FirstParameter)/2.0
184  endpoints = [get_proj(center.value(_diff), plane),
185  get_proj(verts[-1].Point, plane)]
186  else:
187  endpoints = [get_proj(verts[-1].Point, plane)]

192  rot = 0
193  else: # ellipse

215  t1 = edge.tangentAt(edge.FirstParameter)
216  t2 = edge.tangentAt(edge.FirstParameter + _diff/10)
217  flag_sweep = DraftVecUtils.angle(t1, t2, drawing_plane_normal) < 0
218 
219  for v in endpoints:
220  edata += ('A {} {} {} '

222  '{} {} '.format(rx, ry, rot,
223  int(flag_large_arc),
224  int(flag_sweep),
225  v.x, v.y))
226

230 def _get_path_bspline(plane, edge, edata):
231  """Convert the edge to a BSpline and discretize it."""
232  bspline = edge.Curve.toBSpline(edge.FirstParameter, edge.LastParameter)
233  if bspline.Degree > 3 or bspline.isRational():
234  try:
235  bspline = bspline.approximateBSpline(0.05, 50, 3, 'C0')
236  except RuntimeError:
237  _wrn("Debug: unable to approximate bspline from edge")
238 
239  if bspline.Degree <= 3 and not bspline.isRational():
240  for bezierseg in bspline.toBezier():
241  if bezierseg.Degree > 3: # should not happen
242  _wrn("Bezier segment of degree > 3")
243  raise AssertionError
244  elif bezierseg.Degree == 1:
245  edata += 'L '
246  elif bezierseg.Degree == 2:
247  edata += 'Q '
248  elif bezierseg.Degree == 3:
249  edata += 'C '
250 
251  for pole in bezierseg.getPoles()[1:]:
252  v = get_proj(pole, plane)
253  edata += '{} {} '.format(v.x, v.y)
254  else:
255  _msg("Debug: one edge (hash {}) "
256  "has been discretized "
257  "with parameter 0.1".format(edge.hashCode()))
258 
259  for linepoint in bspline.discretize(0.1)[1:]:
260  v = get_proj(linepoint, plane)
261  edata += 'L {} {} '.format(v.x, v.y)
262 
263  return edata

338 def get_path(obj, plane,
339  fill, pathdata, stroke, linewidth, lstyle,
340  fill_opacity=None,
341  edges=[], wires=[], pathname=None):
342  """Get the SVG representation from an object's edges or wires.
343 
344  TODO: the `edges` and `wires` must not default to empty list `[]`
345  but to `None`. Verify that the code doesn't break with this change.
346 
347  `edges` and `wires` are mutually exclusive. If no `wires` are provided,
348  sort the `edges`, and use them. If `wires` are provided, sort the edges
349  in these `wires`, and use them.
350  """
351  svg = "<path "
352 
353  if pathname is None:
354  svg += 'id="{}" '.format(obj.Name)
355  elif pathname != "":
356  svg += 'id="{}" '.format(pathname)
357 
358  svg += ' d="'
359 
360  if not wires:
361  egroups = Part.sortEdges(edges)
362  else:
363  egroups = []
364  first = True
365  for w in wires:
366  wire = w.copy()
367  if first:
368  first = False
369  else:
370  # invert further wires to create holes
371  wire = DraftGeomUtils.invert(wire)
372 
373  wire.fixWire()
374  egroups.append(Part.__sortEdges__(wire.Edges))
375 
376  for _edges in egroups:
377  edata = ""
378 
379  for edgeindex, edge in enumerate(_edges):
380  if edgeindex == 0:
381  verts = edge.Vertexes
382  if len(_edges) > 1:
383  last_pt = verts[-1].Point
384  nextverts = _edges[1].Vertexes
385  if (last_pt - nextverts[0].Point).Length > 1e-6 \
386  and (last_pt - nextverts[-1].Point).Length > 1e-6:
387  verts.reverse()
388  v = get_proj(verts[0].Point, plane)
389  edata += 'M {} {} '.format(v.x, v.y)
390  else:
391  previousverts = verts
392  verts = edge.Vertexes
393  if (verts[0].Point - previousverts[-1].Point).Length > 1e-6:
394  verts.reverse()
395  if (verts[0].Point - previousverts[-1].Point).Length > 1e-6:
396  raise ValueError('edges not ordered')
397 
398  iscircle = DraftGeomUtils.geomType(edge) == "Circle"
399  isellipse = DraftGeomUtils.geomType(edge) == "Ellipse"
400 
401  if iscircle or isellipse:
402  _type, data = _get_path_circ_ellipse(plane, edge, verts,
403  edata,
404  iscircle, isellipse,
405  fill, stroke,
406  linewidth, lstyle)
407  if _type == "svg":
408  # final svg string already calculated, so just return it
409  return data
410 
411  # else the `edata` was properly augmented, so re-assing it
412  edata = data
413  elif DraftGeomUtils.geomType(edge) == "Line":
414  v = get_proj(verts[-1].Point, plane)
415  edata += 'L {} {} '.format(v.x, v.y)
416  else:
417  # If it's not a circle nor ellipse nor straight line
418  # convert the curve to BSpline
419  edata = _get_path_bspline(plane, edge, edata)
420 
421  if fill != 'none':
422  edata += 'Z '
423 
424  if edata in pathdata:
425  # do not draw a path on another identical path
426  return ""
427  else:
428  svg += edata
429  pathdata.append(edata)
430 
431  svg += '" '
432  svg += 'stroke="{}" '.format(stroke)
433  svg += 'stroke-width="{} px" '.format(linewidth)
434  svg += 'style="'
435  svg += 'stroke-width:{};'.format(linewidth)
436  svg += 'stroke-miterlimit:4;'
437  svg += 'stroke-dasharray:{};'.format(lstyle)
438  svg += 'stroke-linecap:square;'
439  svg += 'fill:{};'.format(fill)
440  # fill_opacity must be a number, but if it's `None` it is omitted
441  if fill_opacity is not None:
442  svg += 'fill-opacity:{};'.format(fill_opacity)
443 
444  svg += 'fill-rule: evenodd"'
445  svg += '/>\n'
446  return svg
