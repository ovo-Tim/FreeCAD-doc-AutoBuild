117 void TaskCosmeticCircle::setUiPrimary()
118 {
119  setWindowTitle(QObject::tr("Create Cosmetic Circle"));
120  std::vector<Base::Vector3d> displayPoints;
121  std::vector<Base::Vector3d> mathPoints;
122  for (auto& point : m_points) {
123  // use conventional coordinates for calculations
124  mathPoints.push_back(DU::invertY(point));
125  }
126 
127  if (!m_points.empty()) {
128  m_center = m_points.front();
129  }
130 
131  Base::Vector3d displayCenter;
132  if (m_is3d) {
133  ui->rb2d1->setChecked(false);
134  ui->rb3d1->setChecked(true);
135  // center, project and invert the 3d point
136  Base::Vector3d centroid = m_partFeat->getCurrentCentroid();
137  displayCenter = m_partFeat->projectPoint(mathPoints[0] - centroid, false);
138  } else {
139  ui->rb2d1->setChecked(true);
140  ui->rb3d1->setChecked(false);
141  // if the points are selected from 2d, they are already inverted
142  // unscale and unrotate the selected 2d point
143  displayCenter = CosmeticVertex::makeCanonicalPointInverted(m_partFeat, m_center);
144  displayCenter = DU::invertY(displayCenter);
145  }
146 
147  ui->qsbCenterX->setUnit(Base::Unit::Length);
148  ui->qsbCenterX->setValue(displayCenter.x);
149  ui->qsbCenterY->setUnit(Base::Unit::Length);
150  ui->qsbCenterY->setValue(displayCenter.y);
151  ui->qsbCenterY->setUnit(Base::Unit::Length);
152  ui->qsbCenterZ->setValue(displayCenter.z);
153 
154  double radius = (mathPoints[1] - mathPoints[0]).Length() / m_partFeat->getScale();
155  ui->qsbRadius->setValue(radius);
156 
157  ui->qsbStartAngle->setValue(Base::toDegrees(DU::angleWithX(mathPoints[1] - mathPoints[0])));
158  ui->qsbEndAngle->setValue(Base::toDegrees(DU::angleWithX(mathPoints[2] - mathPoints[0])));
159 }

189 void TaskCosmeticCircle::createCosmeticCircle(void)
190 {
191 // Base::Console().Message("TCL::createCosmeticCircle()\n");
192 
193  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create Cosmetic Circle"));
194 
195  // point from Page/View is conventional coordinates (Y+ up), unscaled, unrotated, but centered (Csriz)
196  // this is Canonical form with out inversion.
197  // point from 3d is OXYZ and needs to be projected.
198  double x = ui->qsbCenterX->value().getValue();
199  double y = ui->qsbCenterY->value().getValue();
200  double z = ui->qsbCenterZ->value().getValue();
201  Base::Vector3d center(x, y, z);
202  if (ui->rb3d1->isChecked()) {
203  Base::Vector3d centroid = m_partFeat->getCurrentCentroid();
204  center = m_partFeat->projectPoint(center - centroid);
205  }
206 
207  TechDraw::BaseGeomPtr bg;
208  if (ui->qsbStartAngle->value().getValue() == 0.0 &&
209  ui->qsbEndAngle->value().getValue() == 0.0) {
210  bg = std::make_shared<TechDraw::Circle> (center, ui->qsbRadius->value().getValue());
211  } else {
212  bg = std::make_shared<TechDraw::AOC>(center, ui->qsbRadius->value().getValue(),
213  ui->qsbStartAngle->value().getValue(),
214  ui->qsbEndAngle->value().getValue());
215  }
216 
217  // after all the calculations are done, we invert the geometry
218  m_tag = m_partFeat->addCosmeticEdge(bg->inverted());
219  m_ce = m_partFeat->getCosmeticEdge(m_tag);
220  m_ce->setFormat(LineFormat::getCurrentLineFormat());
221 
222  Gui::Command::commitCommand();
223 }

225 void TaskCosmeticCircle::updateCosmeticCircle(void)
226 {
227  // Base::Console().Message("TCL::updateCosmeticCircle()\n");
228  double x = ui->qsbCenterX->value().getValue();
229  double y = ui->qsbCenterY->value().getValue();
230  double z = ui->qsbCenterZ->value().getValue();
231  Base::Vector3d p0(x, y, z);
232 
233  //replace the geometry
234  m_ce->permaRadius = ui->qsbRadius->value().getValue();
235 
236  TechDraw::BaseGeomPtr bg;
237  if (ui->qsbStartAngle->value().getValue() == 0.0 &&
238  ui->qsbEndAngle->value().getValue() == 0.0) {
239  bg = std::make_shared<TechDraw::Circle> (p0, m_ce->permaRadius);
240  } else {
241  bg = std::make_shared<TechDraw::AOC>(p0, ui->qsbRadius->value().getValue(),
242  ui->qsbStartAngle->value().getValue(),
243  ui->qsbEndAngle->value().getValue());
244  }
245  m_ce->m_geometry = bg->inverted();
246  m_ce->permaStart = p0;
247  m_ce->permaEnd = p0;
248 }

252 bool TaskCosmeticCircle::accept()
253 {
254  if (ui->qsbRadius->value().getValue() <= 0.0) {
255  // this won't work!
256  Base::Console().Error("TaskCosmeticCircle - can not create a circle with radius: %.3f\n",
257  ui->qsbRadius->value().getValue());
258  return false;
259  }
260  if (m_createMode) {
261  createCosmeticCircle();
262  m_partFeat->add1CEToGE(m_tag);
263  m_partFeat->refreshCEGeoms();
264  m_partFeat->requestPaint();
265  } else {
266  //update mode
267  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Update CosmeticCircle"));
268  updateCosmeticCircle();
269  m_partFeat->refreshCEGeoms();
270  m_partFeat->requestPaint();
271  Gui::Command::updateActive();
272  Gui::Command::commitCommand();
273  }
274 
275  Gui::Command::doCommand(Gui::Command::Gui, "Gui.ActiveDocument.resetEdit()");
276 
277  return true;
278 }
