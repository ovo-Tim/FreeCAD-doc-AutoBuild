52 unsigned long
53 MeshSearchNeighbours::NeighboursFromFacet(FacetIndex ulFacetIdx,
54  float fDistance,
55  unsigned long ulMinPoints,
56  std::vector<Base::Vector3f>& raclResultPoints)
57 {
58  bool bAddPoints = false;
59 
60  _fMaxDistanceP2 = fDistance * fDistance;
61  _clCenter = _rclMesh.GetFacet(ulFacetIdx).GetGravityPoint();
62 
63  unsigned long ulVisited = 1;
64  std::vector<MeshFacetArray::_TConstIterator> aclTestedFacet;
65 
66  _aclResult.clear();
67  _aclOuter.clear();
68 
69  // add start facet
70  bool bFound = CheckDistToFacet(_rclFAry[ulFacetIdx]);
71  _rclFAry[ulFacetIdx].SetFlag(MeshFacet::MARKED);
72  aclTestedFacet.push_back(_rclFAry.begin() + ulFacetIdx);
73 
74  if (!bFound && (_aclResult.size() < ulMinPoints)) {
75  bAddPoints = true;
76  bFound = ExpandRadius(ulMinPoints);
77  }
78 
79  int nCtExpandRadius = 0;
80  // search neighbours, add not marked facets, test distance, add outer points
81  MeshFacetArray::_TConstIterator f_beg = _rclFAry.begin();
82  while (bFound && (nCtExpandRadius < 10)) {
83  bFound = false;
84 
85  std::set<PointIndex> aclTmp;
86  aclTmp.swap(_aclOuter);
87  for (PointIndex pI : aclTmp) {
88  const std::set<FacetIndex>& rclISet = _clPt2Fa[pI];
89  // search all facets hanging on this point
90  for (FacetIndex pJ : rclISet) {
91  const MeshFacet& rclF = f_beg[pJ];
92 
93  if (!rclF.IsFlag(MeshFacet::MARKED)) {
94  bool bLF = CheckDistToFacet(rclF);
95  bFound = bFound || bLF;
96  rclF.SetFlag(MeshFacet::MARKED);
97  aclTestedFacet.push_back(f_beg + pJ);
98  }
99  }
100  ulVisited += rclISet.size();
101  }
102 
103  // too few points inside radius found -> expand radius
104  if (!bFound && (_aclResult.size() < ulMinPoints)) {
105  nCtExpandRadius++;
106  bAddPoints = true;
107  bFound = ExpandRadius(ulMinPoints);
108  }
109  else {
110  nCtExpandRadius = 0;
111  }
112  }
113 
114  // reset marked facets, points
115  for (auto& pF : aclTestedFacet) {
116  pF->ResetFlag(MeshFacet::MARKED);
117  }
118  for (PointIndex pR : _aclResult) {
119  _rclPAry[pR].ResetFlag(MeshPoint::MARKED);
120  }
121 
122 
123  // copy points in result container
124  raclResultPoints.resize(_aclResult.size());
125  size_t i = 0;
126  for (auto pI = _aclResult.begin(); pI != _aclResult.end(); ++pI, i++) {
127  raclResultPoints[i] = _rclPAry[*pI];
128  }
129 
130  if (bAddPoints) {
131  // sort points, remove points lying furthest from center
132  std::sort(raclResultPoints.begin(), raclResultPoints.end(), CDistRad(_clCenter));
133  raclResultPoints.erase(raclResultPoints.begin() + ulMinPoints, raclResultPoints.end());
134  }
135 
136  return ulVisited;
137 }

139 void MeshSearchNeighbours::SampleAllFacets()
140 {
141  if (_aclSampledFacets.size() == _rclMesh.CountFacets()) {
142  return; // already sampled, do nothing
143  }
144 
145  _aclSampledFacets.resize(_rclMesh.CountFacets());
146  MeshFacetIterator clFIter(_rclMesh);
147  size_t i = 0;
148  for (clFIter.Init(); clFIter.More(); clFIter.Next(), i++) {
149  std::vector<Base::Vector3f> clPoints;
150  clFIter->SubSample(_fSampleDistance, clPoints);
151  _aclSampledFacets[i].resize(clPoints.size());
152  std::copy(clPoints.begin(), clPoints.end(), _aclSampledFacets[i].begin());
153  }
154 }

223 bool MeshSearchNeighbours::AccumulateNeighbours(const MeshFacet& rclF, FacetIndex ulFIdx)
224 {
225  int k = 0;
226 
227  for (PointIndex ulPIdx : rclF._aulPoints) {
228  _aclOuter.insert(ulPIdx);
229  _aclResult.insert(ulPIdx);
230 
231  if (Base::DistanceP2(_clCenter, _rclPAry[ulPIdx]) < _fMaxDistanceP2) {
232  k++;
233  }
234  }
235 
236  bool bFound = false;
237  if (k == 3) { // add all sample points
238  _aclPointsResult.insert(_aclPointsResult.end(),
239  _aclSampledFacets[ulFIdx].begin(),
240  _aclSampledFacets[ulFIdx].end());
241  bFound = true;
242  }
243  else { // add points inner radius
244  bFound = TriangleCutsSphere(rclF);
245 
246  if (bFound) {
247  const std::vector<Base::Vector3f>& rclT = _aclSampledFacets[ulFIdx];
248  std::vector<Base::Vector3f> clTmp;
249  clTmp.reserve(rclT.size());
250  for (const auto& pI : rclT) {
251  if (InnerPoint(pI)) {
252  clTmp.push_back(pI);
253  }
254  }
255  _aclPointsResult.insert(_aclPointsResult.end(), clTmp.begin(), clTmp.end());
256  }
257  }
258 
259  return bFound;
260 }

262 bool MeshSearchNeighbours::ExpandRadius(unsigned long ulMinPoints)
263 {
264  // add facets from current level
265  _aclResult.insert(_aclOuter.begin(), _aclOuter.end());
266  for (PointIndex pI : _aclOuter) {
267  _rclPAry[pI].SetFlag(MeshPoint::MARKED);
268  }
269 
270  if (_aclResult.size() < ulMinPoints) {
271  _fMaxDistanceP2 *= float(ulMinPoints) / float(_aclResult.size());
272  return true;
273  }
274 
275  return false;
276 }
