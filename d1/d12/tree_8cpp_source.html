55 bool Tree::addSegment(const Segment& segment, const std::string& hook_name) {
56  SegmentMap::iterator parent = segments.find(hook_name);
57  //check if parent exists
58  if (parent == segments.end())
59  return false;
60  pair<SegmentMap::iterator, bool> retval;
61  //insert new element
62  unsigned int q_nr = segment.getJoint().getType() != Joint::None ? nrOfJoints : 0;
63 
64 #ifdef KDL_USE_NEW_TREE_INTERFACE
65  retval = segments.insert(make_pair(segment.getName(), TreeElementType( new TreeElement(segment, parent, q_nr))));
66 #else //#ifdef KDL_USE_NEW_TREE_INTERFACE
67  retval = segments.insert(make_pair(segment.getName(), TreeElementType(segment, parent, q_nr)));
68 #endif //#ifdef KDL_USE_NEW_TREE_INTERFACE
69 
70  //check if insertion succeeded
71  if (!retval.second)
72  return false;
73  //add iterator to new element in parents children list
74  GetTreeElementChildren(parent->second).push_back(retval.first);
75  //increase number of segments
76  nrOfSegments++;
77  //increase number of joints
78  if (segment.getJoint().getType() != Joint::None)
79  nrOfJoints++;
80  return true;
81 }

117  bool Tree::getChain(const std::string& chain_root, const std::string& chain_tip, Chain& chain)const
118  {
119  // clear chain
120  chain = Chain();
121 
122  // walk down from chain_root and chain_tip to the root of the tree
123  vector<SegmentMap::key_type> parents_chain_root, parents_chain_tip;
124  for (SegmentMap::const_iterator s=getSegment(chain_root); s!=segments.end(); s = GetTreeElementParent(s->second)){
125  parents_chain_root.push_back(s->first);
126  if (s->first == root_name) break;
127  }
128  if (parents_chain_root.empty() || parents_chain_root.back() != root_name)
129  return false;
130  for (SegmentMap::const_iterator s=getSegment(chain_tip); s!=segments.end(); s = GetTreeElementParent(s->second)){
131  parents_chain_tip.push_back(s->first);
132  if (s->first == root_name) break;
133  }
134  if (parents_chain_tip.empty() || parents_chain_tip.back() != root_name)
135  return false;
136 
137  // remove common part of segment lists
138  SegmentMap::key_type last_segment = root_name;
139  while (!parents_chain_root.empty() && !parents_chain_tip.empty() &&
140  parents_chain_root.back() == parents_chain_tip.back()){
141  last_segment = parents_chain_root.back();
142  parents_chain_root.pop_back();
143  parents_chain_tip.pop_back();
144  }
145  parents_chain_root.push_back(last_segment);
146 
147 
148  // add the segments from the root to the common frame
149  for (unsigned int s=0; s<parents_chain_root.size()-1; s++){
150  Segment seg = GetTreeElementSegment(getSegment(parents_chain_root[s])->second);
151  Frame f_tip = seg.pose(0.0).Inverse();
152  Joint jnt = seg.getJoint();
153  if (jnt.getType() == Joint::RotX || jnt.getType() == Joint::RotY || jnt.getType() == Joint::RotZ || jnt.getType() == Joint::RotAxis)
154  jnt = Joint(jnt.getName(), f_tip*jnt.JointOrigin(), f_tip.M*(-jnt.JointAxis()), Joint::RotAxis);
155  else if (jnt.getType() == Joint::TransX || jnt.getType() == Joint::TransY || jnt.getType() == Joint::TransZ || jnt.getType() == Joint::TransAxis)
156  jnt = Joint(jnt.getName(),f_tip*jnt.JointOrigin(), f_tip.M*(-jnt.JointAxis()), Joint::TransAxis);
157  chain.addSegment(Segment(GetTreeElementSegment(getSegment(parents_chain_root[s+1])->second).getName(),
158  jnt, f_tip, GetTreeElementSegment(getSegment(parents_chain_root[s+1])->second).getInertia()));
159  }
160 
161  // add the segments from the common frame to the tip frame
162  for (int s=parents_chain_tip.size()-1; s>-1; s--){
163  chain.addSegment(GetTreeElementSegment(getSegment(parents_chain_tip[s])->second));
164  }
165  return true;
166  }
