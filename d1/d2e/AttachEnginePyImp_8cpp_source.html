60  PyErr_Clear();
61  if (PyArg_ParseTuple(args, "O!", &(AttachEnginePy::Type), &o)) {
62  AttachEngine* attacher = static_cast<AttachEnginePy*>(o)->getAttachEnginePtr();
63  AttachEngine* oldAttacher = this->getAttachEnginePtr();
64  this->_pcTwinPointer = attacher->copy();
65  delete oldAttacher;
66  return 0;
67  }
68 
69  PyErr_Clear();
70  char* typeName;
71  if (PyArg_ParseTuple(args, "s", &typeName)) {
72  Base::Type t = Base::Type::fromName(typeName);
73  AttachEngine* pNewAttacher = nullptr;
74  if (t.isDerivedFrom(AttachEngine::getClassTypeId())){
75  pNewAttacher = static_cast<Attacher::AttachEngine*>(Base::Type::createInstanceByName(typeName));
76  }
77  if (!pNewAttacher) {
78  std::stringstream errMsg;
79  errMsg << "Object if this type is not derived from AttachEngine: " << typeName;
80  PyErr_SetString(Base::PyExc_FC_GeneralError, errMsg.str().c_str());
81  return -1;
82  }
83  AttachEngine* oldAttacher = this->getAttachEnginePtr();
84  this->_pcTwinPointer = pNewAttacher;
85  delete oldAttacher;
86  return 0;
87  }
88 
89  PyErr_SetString(PyExc_TypeError, "Wrong set of constructor arguments. Can be: (), ('Attacher::AttachEngine3D'), ('Attacher::AttachEnginePlane'), ('Attacher::AttachEngineLine'), ('Attacher::AttachEnginePoint'), (other_attacher_instance).");

264 PyObject* AttachEnginePy::getModeInfo(PyObject* args)
265 {
266  char* modeName;
267  if (!PyArg_ParseTuple(args, "s", &modeName))
268  return nullptr;
269 
270  try {
271  AttachEngine &attacher = *(this->getAttachEnginePtr());
272  eMapMode mmode = attacher.getModeByName(modeName);
273  Py::List pyListOfCombinations;
274  Py::List pyCombination;
275  refTypeStringList &listOfCombinations = attacher.modeRefTypes.at(mmode);
276  for(const refTypeString &combination: listOfCombinations){
277  pyCombination = Py::List(combination.size());
278  for(std::size_t iref = 0; iref < combination.size(); iref++) {
279  pyCombination[iref] = Py::String(AttachEngine::getRefTypeName(combination[iref]));
280  }
281  pyListOfCombinations.append(pyCombination);
282  }
283  Py::Dict ret;
284  ret["ReferenceCombinations"] = pyListOfCombinations;
285  ret["ModeIndex"] = Py::Long(mmode);
286  try {
287  Py::Module module(PyImport_ImportModule("PartGui"),true);
288  if (module.isNull() || !module.hasAttr("AttachEngineResources")) {
289  // in v0.14+, the GUI module can be loaded in console mode (but doesn't have all its document methods)
290  throw Py::RuntimeError("Gui is not up");//DeepSOIC: wanted to throw ImportError here, but it's not defined, so I don't know...
291  }
292  Py::Object submod(module.getAttr("AttachEngineResources"));
293  Py::Callable method(submod.getAttr("getModeStrings"));
294  Py::Tuple arg(2);
295  arg.setItem(0, Py::String(this->getAttachEnginePtr()->getTypeId().getName()));
296  arg.setItem(1, Py::Long(mmode));
297  Py::List strs = method.apply(arg);
298  assert(strs.size() == 2);
299  ret["UserFriendlyName"] = strs[0];
300  ret["BriefDocu"] = strs[1];
301  } catch (Py::Exception& e) {
302  if (PyErr_ExceptionMatches(PyExc_ImportError)) {
303  // the GUI is not up.
304  Base::Console().Warning("AttachEngine: Gui not up, so no gui-related entries in getModeInfo.\n");
305  e.clear();
306  } else {
307  Base::Console().Warning("AttachEngine.getModeInfo: error obtaining GUI strings\n");
308  e.clear();
309  }
310  } catch (Base::Exception &e){
311  Base::Console().Warning("AttachEngine.getModeInfo: error obtaining GUI strings:");
312  Base::Console().Warning(e.what());
313  Base::Console().Warning("\n");
314  }
315 
316  return Py::new_reference_to(ret);
317  } ATTACHERPY_STDCATCH_METH;
318 }

432 {
433  if (!PyArg_ParseTuple(args, ""))
434  return nullptr;
435 
436  try {
437  AttachEngine &attacher = *(this->getAttachEnginePtr());
438  SuggestResult sugr;
439  attacher.suggestMapModes(sugr);
440  Py::Dict result;
441  { //sugr.allApplicableModes
442  Py::List pyList;
443  for(eMapMode mmode: sugr.allApplicableModes){
444  pyList.append(Py::String(AttachEngine::getModeName(mmode)));
445  }
446  result["allApplicableModes"] = pyList;
447  }
448  { //sugr.bestFitMode
449  result["bestFitMode"] = Py::String(AttachEngine::getModeName(sugr.bestFitMode));

451  {//sugr.error

454  result["error"] = Py::String(isError ? sugr.error.what() : "");
455  }
456  {//sugr.message
457  std::string msg;
458  switch(sugr.message){
459  case SuggestResult::srIncompatibleGeometry:

461  break;
462  case SuggestResult::srLinkBroken:
463  msg = "LinkBroken";
464  break;
465  case SuggestResult::srNoModesFit:
466  msg = "NoModesFit";
467  break;
468  case SuggestResult::srOK:
469  msg = "OK";
470  break;
471  case SuggestResult::srUnexpectedError:
472  msg = "UnexpectedError";
473  break;
474  default:
475  msg = "<message index out of range>";
476  }
477  result["message"] = Py::String(msg);
478  }
479  {//sugr.nextRefTypeHint
480  Py::List pyList;
481  for(eRefType rt : sugr.nextRefTypeHint){
482  pyList.append(Py::String(AttachEngine::getRefTypeName(rt)));

498  }
