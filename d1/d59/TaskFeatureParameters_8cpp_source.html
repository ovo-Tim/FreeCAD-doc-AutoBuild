90 bool TaskDlgFeatureParameters::accept()
91 {
92  App::DocumentObject* feature = getObject();
93  bool isUpdateBlocked = false;
94  try {
95  // Iterate over parameter dialogs and apply all parameters from them
96  for ( QWidget *wgt : Content ) {
97  TaskFeatureParameters *param = qobject_cast<TaskFeatureParameters *> (wgt);
98  if(!param)
99  continue;
100 
101  param->saveHistory ();
102  param->apply ();
103  isUpdateBlocked |= param->isUpdateBlocked();
104  }
105  // Make sure the feature is what we are expecting
106  // Should be fine but you never know...
107  if ( !feature->isDerivedFrom<PartDesign::Feature>() ) {
108  throw Base::TypeError("Bad object processed in the feature dialog.");
109  }
110 
111  if(isUpdateBlocked){
112  Gui::cmdAppDocument(feature, "recompute()");
113  } else {
114  // object was already computed, nothing more to do with it...
115  Gui::cmdAppDocument(feature, "purgeTouched()");
116 
117  // ...but touch parents to signal the change...
118  for (auto obj : feature->getInList()){
119  obj->touch();
120  }
121  // ...and recompute them
122  Gui::cmdAppDocument(feature->getDocument(), "recompute()");
123  }
124 
125  if (!feature->isValid()) {
126  throw Base::RuntimeError(getObject()->getStatusString());
127  }
128 
129  App::DocumentObject* previous = static_cast<PartDesign::Feature*>(feature)->getBaseObject(/* silent = */ true );
130  Gui::cmdAppObjectHide(previous);
131 
132  // detach the task panel from the selection to avoid to invoke
133  // eventually onAddSelection when the selection changes
134  std::vector<QWidget*> subwidgets = getDialogContent();
135  for (auto it : subwidgets) {
136  TaskSketchBasedParameters* param = qobject_cast<TaskSketchBasedParameters*>(it);
137  if (param)
138  param->detachSelection();
139  }
140 
141  Gui::cmdGuiDocument(feature, "resetEdit()");
142  Gui::Command::commitCommand();
143  } catch (const Base::Exception& e) {
144  // Generally the only thing that should fail is feature->isValid() others should be fine
145  QString errorText = QApplication::translate(feature->getTypeId().getName(), e.what());
146  QMessageBox::warning(Gui::getMainWindow(), tr("Input error"), errorText);
147  return false;
148  }
149 
150  return true;
151 }

153 bool TaskDlgFeatureParameters::reject()
154 {
155  auto feature = getObject<PartDesign::Feature>();
156  App::DocumentObjectWeakPtrT weakptr(feature);
157  App::Document* document = feature->getDocument();
158 
159  PartDesign::Body* body = PartDesign::Body::findBodyOf(feature);
160 
161  // Find out previous feature we won't be able to do it after abort
162  // (at least in the body case)
163  App::DocumentObject* previous = feature->getBaseObject(/* silent = */ true );
164 
165  // detach the task panel from the selection to avoid to invoke
166  // eventually onAddSelection when the selection changes
167  std::vector<QWidget*> subwidgets = getDialogContent();
168  for (auto it : subwidgets) {
169  TaskSketchBasedParameters* param = qobject_cast<TaskSketchBasedParameters*>(it);
170  if (param)
171  param->detachSelection();
172  }
173 
174  // roll back the done things which may delete the feature
175  Gui::Command::abortCommand();
176 
177  // if abort command deleted the object make the previous feature visible again
178  if (weakptr.expired()) {
179  // Make the tip or the previous feature visible again with preference to the previous one
180  // TODO: ViewProvider::onDelete has the same code. May be this one is excess?
181  if (previous && Gui::Application::Instance->getViewProvider(previous)) {
182  Gui::Application::Instance->getViewProvider(previous)->show();
183  }
184  else if (body) {
185  App::DocumentObject* tip = body->Tip.getValue();
186  if (tip && Gui::Application::Instance->getViewProvider(tip)) {
187  Gui::Application::Instance->getViewProvider(tip)->show();
188  }
189  }
190  }
191 
192  Gui::cmdAppDocument(document, "recompute()");
193  Gui::cmdGuiDocument(document, "resetEdit()");
194 
195  return true;
196 }
