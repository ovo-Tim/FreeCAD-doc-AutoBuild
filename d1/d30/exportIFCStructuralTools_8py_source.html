136 def createStructuralMember(ifcfile, ifcbin, obj):
137 
138  """Creates a structural member if possible. Returns the member"""
139 
140  global structural_nodes, structural_curves
141  structuralMember = None
142  import Draft
143  import Part
144  import ifcopenshell
145  import FreeCAD
146  uid = ifcopenshell.guid.new
147  ownerHistory = ifcfile.by_type("IfcOwnerHistory")[0]
148  structContext = getStructuralContext(ifcfile)
149 
150  # find edges to convert into structural members
151  edges = None
152  if Draft.getType(obj) not in ["Structure"]:
153  # for non structural elements
154  if ALLOW_LINEAR_OBJECTS and obj.isDerivedFrom("Part::Feature"):
155  # for objects created with Part workbench
156  if obj.Shape.Faces:
157  # for objects with faces
158  return None
159  elif not obj.Shape.Edges:
160  # for objects without edges
161  return None
162  else:
163  edges = obj.Shape.Edges
164  else:
165  # for structural elements with nodes
166  nodes = [obj.Placement.multVec(n) for n in obj.Nodes]
167  if len(nodes) > 2:
168  # when there are more then 2 nodes (i.e. for slabs) append closing node to produce closing edge
169  nodes.append(nodes[0])
170  wire = Part.makePolygon(nodes)
171  edges = wire.Edges
172  if not edges:
173  return None
174 
175  # OBJECT CLASSIFICATION by edge number
176  # Linear elements for edge_number = 1, Surface elements for edge_number > 1
177  # we don't care about curved edges just now...
178 
179  if len(edges) == 1:
180  # LINEAR OBJECTS: beams, columns
181  # ATM limitations:
182  # - no profile properties are taken into account
183  # - no materials properties are takein into account
184  # -
185  # create geometry
186  verts = [None for _ in range(len(edges)+1)]
187  verts[0] = tuple(edges[0].Vertexes[ 0].Point.multiply(scaling))
188  verts[1] = tuple(edges[0].Vertexes[-1].Point.multiply(scaling))
189  cartPnt1 = ifcfile.createIfcCartesianPoint(verts[0])
190  cartPnt2 = ifcfile.createIfcCartesianPoint(verts[1])
191  vertPnt1 = ifcfile.createIfcVertexPoint(cartPnt1)
192  vertPnt2 = ifcfile.createIfcVertexPoint(cartPnt2)
193  newEdge = ifcfile.createIfcEdge(vertPnt1, vertPnt2)
194  topologyRep = ifcfile.createIfcTopologyRepresentation(structContext, "Analysis", "Edge", (newEdge,))
195  prodDefShape = ifcfile.createIfcProductDefinitionShape(None, None, (topologyRep,))
196  # set local coordinate system
197  localPlacement = ifcbin.createIfcLocalPlacement()
198  localZAxis = ifcbin.createIfcDirection((0, 0, 1))
199  # create structural member
200  if ifcfile.wrapped_data.schema_name() == "IFC2X3":
201  structuralMember = ifcfile.createIfcStructuralCurveMember(
202  uid(), ownerHistory, obj.Label, None, None, localPlacement, prodDefShape, "RIGID_JOINED_MEMBER")
203  else:
204  localZAxis = ifcbin.createIfcDirection((0, 0, 1))
205  structuralMember = ifcfile.createIfcStructuralCurveMember(
206  uid(), ownerHistory, obj.Label, None, None, localPlacement, prodDefShape, "RIGID_JOINED_MEMBER", localZAxis)
207 
208  elif len(edges) > 1:
209  # SURFACE OBJECTS: slabs (horizontal, vertical, inclined)
210  # ATM limitations:
211  # - mo material properties are taken into account
212  # - walls don't work because they miss a node system
213  # -
214  # creates geometry
215  verts = [None for _ in range(len(edges))]
216  for i, edge in enumerate(edges):
217  verts[i] = tuple(edge.Vertexes[0].Point.multiply(scaling))
218  cartPnt = ifcfile.createIfcCartesianPoint(verts[i])
219  vertPnt = ifcfile.createIfcVertexPoint(cartPnt)
220  orientedEdges = [None for _ in range(len(edges))]
221  for i, vert in enumerate(verts):
222  v2Index = (i + 1) if i < len(verts) - 1 else 0
223  cartPnt1 = ifcfile.createIfcCartesianPoint(vert)
224  cartPnt2 = ifcfile.createIfcCartesianPoint(verts[v2Index])
225  vertPnt1 = ifcfile.createIfcVertexPoint(cartPnt1)
226  vertPnt2 = ifcfile.createIfcVertexPoint(cartPnt2)
227  edge = ifcfile.createIfcEdge(vertPnt1, vertPnt2)
228  orientedEdges[i] = ifcfile.createIfcOrientedEdge(None, None, edge, True)
229  edgeLoop = ifcfile.createIfcEdgeLoop(tuple(orientedEdges))
230  # sets local coordinate system
231  localPlacement = ifcbin.createIfcLocalPlacement()
232  # sets face origin to the first vertex point of the planar surface
233  origin = cartPnt2
234  # find crossVect that is perpendicular to the planar surface
235  vect0 = FreeCAD.Vector(verts[0])
236  vect1 = FreeCAD.Vector(verts[1])
237  vectn = FreeCAD.Vector(verts[-1])
238  vect01 = vect1.sub(vect0)
239  vect0n = vectn.sub(vect0)
240  crossVect = vect01.cross(vect0n)
241  # normalize crossVect
242  normVect = crossVect.normalize()
243  xAxis = ifcfile.createIfcDirection(tuple([vect01.x, vect01.y, vect01.z]))
244  zAxis = ifcfile.createIfcDirection(tuple([normVect.x, normVect.y, normVect.z]))
245  localAxes = ifcfile.createIfcAxis2Placement3D(origin, zAxis, xAxis)
246  plane = ifcfile.createIfcPlane(localAxes)
247  faceBound = ifcfile.createIfcFaceBound(edgeLoop, True)
248  face = ifcfile.createIfcFaceSurface((faceBound,), plane, True)
249  topologyRep = ifcfile.createIfcTopologyRepresentation(structContext, "Analysis", "Face", (face,))
250  prodDefShape = ifcfile.createIfcProductDefinitionShape(None, None, (topologyRep,))
251  # sets surface thickness
252  # TODO: ATM limitations
253  # - for vertical slabs (walls) or inclined slabs (ramps) the thickness is taken from the Height property
254  thickness = float(obj.Height)*scaling
255  # creates structural member
256  structuralMember = ifcfile.createIfcStructuralSurfaceMember(
257  uid(), ownerHistory, obj.Label, None, None, localPlacement, prodDefShape, "SHELL", thickness)
258 
259  # check for existing connection nodes
260  for vert in verts:
261  vertCoord = tuple(vert)
262  if vertCoord in structural_nodes:
263  if structural_nodes[vertCoord]:
264  # there is already another member using this point
265  structPntConn = structural_nodes[vertCoord]
266  else:
267  # there is another member with same point, create a new node connection
268  structPntConn = createStructuralNode(ifcfile, ifcbin, vert)
269  structural_nodes[vertCoord] = structPntConn
270  ifcfile.createIfcRelConnectsStructuralMember(
271  uid(), ownerHistory, None, None, structuralMember, structPntConn, None, None, None, None)
272  else:
273  # just add the point, no other member using it yet
274  structural_nodes[vertCoord] = None
275 
276  # check for existing connection curves
277  for edge in edges:
278  verts12 = tuple([edge.Vertexes[ 0].Point.x, edge.Vertexes[ 0].Point.y, edge.Vertexes[ 0].Point.z,
279  edge.Vertexes[-1].Point.x, edge.Vertexes[-1].Point.y, edge.Vertexes[-1].Point.z])
280  verts21 = tuple([edge.Vertexes[-1].Point.x, edge.Vertexes[-1].Point.y, edge.Vertexes[-1].Point.z,
281  edge.Vertexes[ 0].Point.x, edge.Vertexes[ 0].Point.y, edge.Vertexes[ 0].Point.z])
282  verts12_in_curves = verts12 in structural_curves
283  verts21_in_curves = verts21 in structural_curves
284  if verts21_in_curves:
285  verts = verts21
286  else:
287  verts = verts12
288  if (verts12_in_curves or verts21_in_curves):
289  if structural_curves[verts]:
290  # there is already another member using this curve
291  strucCrvConn = structural_curves[verts]
292  else:
293  # there is another member with same edge, create a new curve connection
294  strucCrvConn = createStructuralCurve(ifcfile, ifcbin, edge)
295  structural_curves[verts] = strucCrvConn
296  ifcfile.createIfcRelConnectsStructuralMember(
297  uid(), None, None, None, structuralMember, strucCrvConn, None, None, None, None)
298  else:
299  # just add the curve, no other member using it yet
300  structural_curves[verts] = None
301  return structuralMember
