234 QPixmap BitmapFactoryInst::pixmap(const char* name) const
235 {
236  if (!name || *name == '\0')
237  return {};
238 
239  // as very first test check whether the pixmap is in the cache
240  QMap<std::string, QPixmap>::Iterator it = d->xpmCache.find(name);
241  if (it != d->xpmCache.end())
242  return it.value();
243 
244  QPixmap icon;
245 
246  // Try whether an absolute path is given
247  QString fn = QString::fromUtf8(name);
248  loadPixmap(fn, icon);
249 
250  // try to find it in the 'icons' search paths
251  if (icon.isNull()) {
252  QList<QByteArray> formats = QImageReader::supportedImageFormats();
253  formats.prepend("SVG"); // check first for SVG to use special import mechanism
254 
255  QString fileName = QString::fromLatin1("icons:") + fn;
256  if (!loadPixmap(fileName, icon)) {
257  // Go through supported file formats
258  for (QList<QByteArray>::iterator fm = formats.begin(); fm != formats.end(); ++fm) {
259  QString path = QString::fromLatin1("%1.%2").arg(fileName,
260  QString::fromLatin1((*fm).toLower().constData()));
261  if (loadPixmap(path, icon)) {
262  break;
263  }
264  }
265  }
266  }
267 
268  if (!icon.isNull()) {
269  d->xpmCache[name] = icon;
270  return icon;
271  }
272 
273  Base::Console().Warning("Cannot find icon: %s\n", name);
274  return QPixmap(Gui::BitmapFactory().pixmapFromSvg("help-browser", QSize(16, 16)));
275 }

364 QPixmap BitmapFactoryInst::resize(int w, int h, const QPixmap& p, Qt::BGMode bgmode) const
365 {
366  if (bgmode == Qt::TransparentMode) {
367  if (p.width() == 0 || p.height() == 0)
368  w = 1;
369 
370  QPixmap pix = p;
371  int x = pix.width () > w ? 0 : (w - pix.width ())/2;
372  int y = pix.height() > h ? 0 : (h - pix.height())/2;
373 
374  if (x == 0 && y == 0)
375  return pix;
376 
377  QPixmap pm (w,h);
378  QBitmap mask (w,h);
379  mask.fill(Qt::color0);
380 
381  QBitmap bm = pix.mask();
382  if (!bm.isNull())
383  {
384  QPainter painter(&mask);
385  painter.drawPixmap(QPoint(x, y), bm, QRect(0, 0, pix.width(), pix.height()));
386  pm.setMask(mask);
387  }
388  else
389  {
390  pm.setMask(mask);
391  pm = fillRect(x, y, pix.width(), pix.height(), pm, Qt::OpaqueMode);
392  }
393 
394  QPainter pt;
395  pt.begin( &pm );
396  pt.drawPixmap(x, y, pix);
397  pt.end();
398  return pm;
399  } else { // Qt::OpaqueMode
400  QPixmap pix = p;
401 
402  if (pix.width() == 0 || pix.height() == 0)
403  return pix; // do not resize a null pixmap
404 
405  QPalette pal = qApp->palette();
406  QColor dl = pal.color(QPalette::Disabled, QPalette::Light);
407  QColor dt = pal.color(QPalette::Disabled, QPalette::Text);
408 
409  QPixmap pm(w,h);
410  pm.fill(dl);
411 
412  QPainter pt;
413  pt.begin( &pm );
414  pt.setPen( dl );
415  pt.drawPixmap(1, 1, pix);
416  pt.setPen( dt );
417  pt.drawPixmap(0, 0, pix);
418  pt.end();
419  return pm;
420  }

445 QPixmap BitmapFactoryInst::merge(const QPixmap& p1, const QPixmap& p2, bool vertical) const
446 {
447  int width = 0;
448  int height = 0;
449 
450  int x = 0;
451  int y = 0;
452 
453  // get the size for the new pixmap
454  if (vertical) {
455  y = p1.height();
456  width = qMax( p1.width(), p2.width() );
457  height = p1.height() + p2.height();
458  } else {
459  x = p1.width();
460  width = p1.width() + p2.width();
461  height = qMax( p1.height(), p2.height() );
462  }
463 
464  QPixmap res( width, height );
465  QBitmap mask( width, height );
466  QBitmap mask1 = p1.mask();
467  QBitmap mask2 = p2.mask();
468  mask.fill( Qt::color0 );
469 
470  auto* pt1 = new QPainter(&res);
471  pt1->drawPixmap(0, 0, p1);
472  pt1->drawPixmap(x, y, p2);
473  delete pt1;
474 
475  auto* pt2 = new QPainter(&mask);
476  pt2->drawPixmap(0, 0, mask1);
477  pt2->drawPixmap(x, y, mask2);
478  delete pt2;
479 
480  res.setMask(mask);
481  return res;

484 QPixmap BitmapFactoryInst::merge(const QPixmap& p1, const QPixmap& p2, Position pos) const
485 {
486  // does the similar as the method above except that this method does not resize the resulting pixmap
487  int x = 0, y = 0;
488  qreal dpr1 = p1.devicePixelRatio();
489  qreal dpr2 = p2.devicePixelRatio();
490 
491  switch (pos)
492  {
493  case TopLeft:
494  break;
495  case TopRight:
496  x = p1.width ()/dpr1 - p2.width ()/dpr2;
497  break;
498  case BottomLeft:
499  y = p1.height()/dpr1 - p2.height()/dpr2;
500  break;
501  case BottomRight:
502  x = p1.width ()/dpr1 - p2.width ()/dpr2;
503  y = p1.height()/dpr1 - p2.height()/dpr2;
504  break;
505  }
506 
507  QPixmap p = p1;
508  p = fillRect(x, y, p2.width(), p2.height(), p, Qt::OpaqueMode);
509 
510  QPainter pt;
511  pt.begin( &p );
512  pt.setPen(Qt::NoPen);
513  pt.drawRect(x, y, p2.width(), p2.height());
514  pt.drawPixmap(x, y, p2);
515  pt.end();
516 
517  return p;
518 }

527 void BitmapFactoryInst::convert(const QImage& p, SoSFImage& img) const
528 {
529  SbVec2s size;
530  size[0] = p.width();
531  size[1] = p.height();
532 
533 #if QT_VERSION >= QT_VERSION_CHECK(5, 10, 0)
534  int buffersize = static_cast<int>(p.sizeInBytes());
535 #else
536  int buffersize = p.byteCount();
537 #endif
538  int numcomponents = 0;
539  QVector<QRgb> table = p.colorTable();
540  if (!table.isEmpty()) {
541  if (p.hasAlphaChannel()) {
542  if (p.allGray()) {
543  numcomponents = 2;
544  }
545  else {
546  numcomponents = 4;
547  }
548  }
549  else {
550  if (p.allGray()) {
551  numcomponents = 1;
552  }
553  else {
554  numcomponents = 3;
555  }
556  }
557  }
558  else {
559  numcomponents = buffersize / (size[0] * size[1]);
560  }
561 
562  int depth = numcomponents;
563 
564  // Coin3D only supports up to 32-bit images
565  if (numcomponents == 8) {
566  numcomponents = 4;
567  }
568 
569  // allocate image data
570  img.setValue(size, numcomponents, nullptr);
571 
572  unsigned char* bytes = img.startEditing(size, numcomponents);
573 
574  int width = (int)size[0];
575  int height = (int)size[1];
576 
577  for (int y = 0; y < height; y++) {
578  unsigned char* line = &bytes[width * numcomponents * (height - (y + 1))];
579  for (int x = 0; x < width; x++) {
580  QColor col = p.pixelColor(x,y);
581  switch (depth) {
582  default:
583  break;
584  case 1:
585  {
586  QRgb rgb = col.rgb();
587  line[0] = qGray(rgb);
588  } break;
589  case 2:
590  {
591  QRgb rgb = col.rgba();
592  line[0] = qGray(rgb);
593  line[1] = qAlpha(rgb);
594  } break;
595  case 3:
596  {
597  QRgb rgb = col.rgb();
598  line[0] = qRed(rgb);
599  line[1] = qGreen(rgb);
600  line[2] = qBlue(rgb);
601  } break;
602  case 4:
603  {
604  QRgb rgb = col.rgba();
605  line[0] = qRed(rgb);
606  line[1] = qGreen(rgb);
607  line[2] = qBlue(rgb);
608  line[3] = qAlpha(rgb);
609  } break;
610  case 8:
611  {
612  QRgba64 rgb = col.rgba64();
613  line[0] = qRed(rgb);
614  line[1] = qGreen(rgb);
615  line[2] = qBlue(rgb);
616  line[3] = qAlpha(rgb);
617  } break;
618  }
619 
620  line += numcomponents;
621  }
622  }
623 
624  img.finishEditing();
625 }

628 {
629  SbVec2s size;
630  int numcomponents;
631 
632  const unsigned char * bytes = p.getValue(size, numcomponents);
633  if (!bytes)
634  return;
635 
636  int width = (int)size[0];
637  int height = (int)size[1];
638 
639  img = QImage(width, height, QImage::Format_RGB32);

643  {
644  const unsigned char * line = &bytes[width*numcomponents*(height-(y+1))];
645  for (int x = 0; x < width; x++)
646  {
647  switch (numcomponents)

649  default:
