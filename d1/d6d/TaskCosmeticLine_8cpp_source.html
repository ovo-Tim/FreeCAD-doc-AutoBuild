111 void TaskCosmeticLine::setUiPrimary()
112 {
113  setWindowTitle(QObject::tr("Create Cosmetic Line"));
114 
115  // double rotDeg = m_partFeat->Rotation.getValue();
116  // double rotRad = rotDeg * M_PI / 180.0;
117  Base::Vector3d centroid = m_partFeat->getCurrentCentroid();
118  Base::Vector3d p1, p2;
119  if (m_is3d.front()) {
120  // center, project and invert the 3d point
121  p1 = m_partFeat->projectPoint(m_points.front() - centroid);
122  } else {
123  // if the points are selected from 2d, they are already inverted
124  // unscale and unrotate the selected 2d point
125  p1 = CosmeticVertex::makeCanonicalPointInverted(m_partFeat, m_points.front());
126  }
127 
128  if (m_is3d.back()) {
129  p2 = m_partFeat->projectPoint(m_points.back() - centroid);
130  } else {
131  p2 = CosmeticVertex::makeCanonicalPointInverted(m_partFeat, m_points.back());
132  }
133 
134  // 3d points are projected above, so they are now 2d point and we need to set the radio buttons appropriately
135  ui->rb2d1->setChecked(true);
136  ui->rb3d1->setChecked(false);
137 
138  ui->qsbx1->setUnit(Base::Unit::Length);
139  ui->qsbx1->setValue(p1.x);
140  ui->qsby1->setUnit(Base::Unit::Length);
141  ui->qsby1->setValue(-p1.y);
142  ui->qsby1->setUnit(Base::Unit::Length);
143  ui->qsbz1->setValue(p1.z);
144 
145  ui->qsbx2->setUnit(Base::Unit::Length);
146  ui->qsbx2->setValue(p2.x);
147  ui->qsby2->setUnit(Base::Unit::Length);
148  ui->qsby2->setValue(-p2.y);
149  ui->qsbz2->setUnit(Base::Unit::Length);
150  ui->qsbz2->setValue(p2.z);
151 }

174 void TaskCosmeticLine::createCosmeticLine()
175 {
176 // Base::Console().Message("TCL::createCosmeticLine()\n");
177  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create Cosmetic Line"));
178 
179  // ui 2d points are interpreted as unscaled, unrotated, uninverted
180  double x = ui->qsbx1->value().getValue();
181  double y = ui->qsby1->value().getValue();
182  double z = ui->qsbz1->value().getValue();
183  Base::Vector3d p0(x, y, z);
184  if (ui->rb3d1->isChecked()) {
185  Base::Vector3d centroid = m_partFeat->getCurrentCentroid();
186  p0 = m_partFeat->projectPoint(p0 - centroid);
187  } else {
188  p0 = DU::invertY(p0);
189  }
190 
191  x = ui->qsbx2->value().getValue();
192  y = ui->qsby2->value().getValue();
193  z = ui->qsbz2->value().getValue();
194  Base::Vector3d p1(x, y, z);
195  if (ui->rb3d2->isChecked()) {
196  Base::Vector3d centroid = m_partFeat->getCurrentCentroid();
197  p1 = m_partFeat->projectPoint(p1 - centroid);
198  } else {
199  p1 = DU::invertY(p1);
200  }
201  m_tag = m_partFeat->addCosmeticEdge(p0, p1);
202  m_ce = m_partFeat->getCosmeticEdge(m_tag);
203  m_ce->setFormat(LineFormat::getCurrentLineFormat());
204 
205  Gui::Command::commitCommand();
206 }

208 void TaskCosmeticLine::updateCosmeticLine()
209 {
210 // Base::Console().Message("TCL::updateCosmeticLine()\n");
211  double x = ui->qsbx1->value().getValue();
212  double y = ui->qsby1->value().getValue();
213  double z = ui->qsbz1->value().getValue();
214  Base::Vector3d p0(x, y, z);
215  if (ui->rb3d1->isChecked()) {
216  Base::Vector3d centroid = m_partFeat->getCurrentCentroid();
217  p0 = m_partFeat->projectPoint(p0 - centroid);
218  } else {
219  p0 = DU::invertY(p0);
220  }
221 
222  x = ui->qsbx2->value().getValue();
223  y = ui->qsby2->value().getValue();
224  z = ui->qsbz2->value().getValue();
225  Base::Vector3d p1(x, y, z);
226  if (ui->rb3d2->isChecked()) {
227  Base::Vector3d centroid = m_partFeat->getCurrentCentroid();
228  p1 = m_partFeat->projectPoint(p1 - centroid);
229  } else {
230  p1 = DU::invertY(p1);
231  }
232  //replace the geometry
233  m_ce->permaStart = p0;
234  m_ce->permaEnd = p1;
235 
236  gp_Pnt gp1(p0.x, p0.y, p0.z);
237  gp_Pnt gp2(p1.x, p1.y, p1.z);
238  TopoDS_Edge edge = BRepBuilderAPI_MakeEdge(gp1, gp2);
239  m_ce->m_geometry = TechDraw::BaseGeom::baseFactory(edge);
240 }

244 bool TaskCosmeticLine::accept()
245 {
246  if (m_createMode) {
247  createCosmeticLine();
248  m_partFeat->add1CEToGE(m_tag);
249  m_partFeat->refreshCEGeoms();
250  m_partFeat->requestPaint();
251  } else {
252  //update mode
253  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Update CosmeticLine"));
254  updateCosmeticLine();
255  m_partFeat->refreshCEGeoms();
256  m_partFeat->requestPaint();
257  Gui::Command::updateActive();
258  Gui::Command::commitCommand();
259  }
260 
261  Gui::Command::doCommand(Gui::Command::Gui, "Gui.ActiveDocument.resetEdit()");
262 
263  return true;
264 }
