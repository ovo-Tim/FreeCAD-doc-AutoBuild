52  inline size_t operator()(const TopoDS_Shape& s) const
53  {
54 #if OCC_VERSION_HEX >= 0x070800
55  return std::hash<TopoDS_Shape> {}(s);
56 #else
57  return s.HashCode(INT_MAX);
58 #endif
59  }
60  inline bool operator()(const TopoShape& a, const TopoShape& b) const
61  {
62  return a.getShape().IsSame(b.getShape());
63  }
64  inline bool operator()(const TopoDS_Shape& a, const TopoDS_Shape& b) const
65  {
66  return a.IsSame(b);
67  }
68  template<class T>
69  static inline void hash_combine(std::size_t& seed, const T& v)
70  {
71  // copied from boost::hash_combine
72  std::hash<T> hasher;
73  seed ^= hasher(v) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
74  }
75  inline size_t operator()(const std::pair<TopoShape, TopoShape>& s) const
76  {
77 #if OCC_VERSION_HEX >= 0x070800
78  size_t res = std::hash<TopoDS_Shape> {}(s.first.getShape());
79  hash_combine(res, std::hash<TopoDS_Shape> {}(s.second.getShape()));
80 #else
81  size_t res = s.first.getShape().HashCode(INT_MAX);
82  hash_combine(res, s.second.getShape().HashCode(INT_MAX));
83 #endif
84  return res;
85  }
86  inline size_t operator()(const std::pair<TopoDS_Shape, TopoDS_Shape>& s) const
87  {
88 #if OCC_VERSION_HEX >= 0x070800
89  size_t res = std::hash<TopoDS_Shape> {}(s.first);
90  hash_combine(res, std::hash<TopoDS_Shape> {}(s.second));
91 #else
92  size_t res = s.first.HashCode(INT_MAX);
93  hash_combine(res, s.second.HashCode(INT_MAX));
94 #endif
95  return res;
96  }
97  inline bool operator()(const std::pair<TopoShape, TopoShape>& a,
98  const std::pair<TopoShape, TopoShape>& b) const
99  {
100  return a.first.getShape().IsSame(b.first.getShape())
101  && a.second.getShape().IsSame(b.second.getShape());
102  }
103  inline bool operator()(const std::pair<TopoDS_Shape, TopoDS_Shape>& a,
104  const std::pair<TopoDS_Shape, TopoDS_Shape>& b) const
105  {
106  return a.first.IsSame(b.first) && a.second.IsSame(b.second);
107  }
108 };
