588  def rebase(self,shape,hint=None):
589  """Copy a shape to the (0,0,0) origin.
590 
591  Create a copy of a shape, such that its center of mass is in the
592  (0,0,0) origin.
593 
594  TODO Determine the way the shape is rotated by this method.
595 
596  Return the copy of the shape, and the <Base.Placement> needed to move
597  the copy back to its original location/orientation.
598 
599  Parameters
600  ----------
601  shape: <Part.Shape>
602  The shape to copy.
603  hint: <Base.Vector>, optional
604  If the angle between the normal vector of the shape, and the hint
605  vector is greater than 90 degrees, the normal will be reversed
606  before being rotated.
607  """
608 
609  import DraftGeomUtils,math
610 
611  # Get the object's center.
612  if not isinstance(shape,list):
613  shape = [shape]
614  if hasattr(shape[0],"CenterOfMass"):
615  v = shape[0].CenterOfMass
616  else:
617  v = shape[0].BoundBox.Center
618 
619  # Get the object's normal.
620  n = DraftGeomUtils.getNormal(shape[0])
621  if (not n) or (not n.Length):
622  n = FreeCAD.Vector(0, 0, 1)
623 
624  # Reverse the normal if the hint vector and the normal vector have more
625  # than a 90 degree angle between them.
626  if hint and hint.getAngle(n) > 1.58:
627  n = n.negative()
628 
629  r = FreeCAD.Rotation(FreeCAD.Vector(0, 0, 1), n)
630  if round(abs(r.Angle),8) == round(math.pi,8):
631  r = FreeCAD.Rotation()
632 
633  shapes = []
634  for s in shape:
635 
636  s = s.copy()
637  s.translate(v.negative())
638  s.rotate(FreeCAD.Vector(0, 0, 0),
639  r.Axis,
640  math.degrees(-r.Angle))
641  shapes.append(s)
642  p = FreeCAD.Placement()
643  p.Base = v
644  p.Rotation = r
645  if len(shapes) == 1:
646  return (shapes[0],p)
647  else:
648  return(shapes,p)
