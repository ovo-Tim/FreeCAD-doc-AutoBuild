45 def isClockwise(edge, ref=None):
46  """Return True if a circle-based edge has a clockwise direction.
47 
48  Parameters
49  ----------
50  edge :
51  The edge to be analyzed.
52 
53  ref : Vector, optional
54  The normal around which the direction of the edge is to be determined.
55  Defaults to the Z normal vector.
56 
57  Returns
58  -------
59  bool
60  Returns True if the edge is clockwise oriented around the ref Vector
61  or not.
62  """
63  if not geomType(edge) == "Circle":
64  return True
65 
66  v1 = edge.Curve.tangent(edge.ParameterRange[0])[0]
67  if DraftVecUtils.isNull(v1):
68  return True
69 
70  # we take an arbitrary other point on the edge that has little chances
71  # to be aligned with the first one
72  v2 = edge.Curve.tangent(edge.ParameterRange[0] + 0.01)[0]
73  n = edge.Curve.Axis
74  # if that axis points "the wrong way" from the reference, we invert it
75  if not ref:
76  ref = App.Vector(0, 0, 1)
77  if n.getAngle(ref) > math.pi/2:
78  n = n.negative()
79 
80  if DraftVecUtils.angle(v1, v2, n) < 0:
81  return False
82 
83  if n.z < 0:
84  return False
85 
86  return True

89 def isWideAngle(edge):
90  """Return True if the given edge is an arc with angle > 180 degrees."""
91  if geomType(edge) != "Circle":
92  return False
93 
94  r = edge.Curve.Radius
95  total = 2*r*math.pi
96 
97  if edge.Length > total/2:
98  return True
99 
100  return False

103 def arcFrom2Pts(firstPt, lastPt, center, axis=None):
104  """Build an arc with center and 2 points, can be oriented with axis."""
105  radius1 = firstPt.sub(center).Length
106  radius2 = lastPt.sub(center).Length
107 
108  # (PREC = 4 = same as Part Module), Is it possible?
109  if round(radius1-radius2, 4) != 0:
110  return None
111 
112  thirdPt = App.Vector(firstPt.sub(center).add(lastPt).sub(center))
113  thirdPt.normalize()
114  thirdPt.scale(radius1, radius1, radius1)
115  thirdPt = thirdPt.add(center)
116  newArc = Part.Edge(Part.Arc(firstPt, thirdPt, lastPt))
117 
118  if axis and newArc.Curve.Axis.dot(axis) < 0:
119  thirdPt = thirdPt.sub(center)
120  thirdPt.scale(-1, -1, -1)
121  thirdPt = thirdPt.add(center)
122  newArc = Part.Edge(Part.Arc(firstPt, thirdPt, lastPt))
123 
124  return newArc

127 def arcFromSpline(edge):
128  """Turn given edge into a circular arc from three points.
129 
130  Takes its first point, midpoint and endpoint. It works best with bspline
131  segments such as those from imported svg files. Use this only
132  if you are sure your edge is really an arc.
133 
134  It returns None if there is a problem, including passing straight edges.
135  """
136  if geomType(edge) == "Line":
137  print("This edge is straight, cannot build an arc on it")
138  return None
139 
140  if len(edge.Vertexes) > 1:
141  # 2-point arc
142  p1 = edge.Vertexes[0].Point
143  p2 = edge.Vertexes[-1].Point
144  ml = edge.Length/2
145  p3 = edge.valueAt(ml)
146  try:
147  return Part.Arc(p1, p3, p2).toShape()
148  except Part.OCCError:
149  print("Couldn't make an arc out of this edge")
150  return None
151  else:
152  # circle
153  p1 = edge.Vertexes[0].Point
154  p2 = findMidpoint(edge)
155  ray = p2.sub(p1)
156  ray.scale(0.5, 0.5, 0.5)
157  center = p1.add(ray)
158  radius = ray.Length
159  try:
160  return Part.makeCircle(radius, center)
161  except Part.OCCError:
162  print("couldn't make a circle out of this edge")
163  return None
