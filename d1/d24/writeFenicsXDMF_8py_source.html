63 def numpy_array_to_str(npa):
64  res = ""
65  dt = str(npa.dtype)
66  if "int" in dt:
67  res = "\n".join([" ".join([("%d" % s) for s in a]) for a in npa.tolist()])
68  elif "float" in dt:
69  res = "\n".join([" ".join([("%3.6f" % s) for s in a]) for a in npa.tolist()])
70  return res

81 def write_fenics_mesh_points_xdmf(fem_mesh_obj, geometrynode, encoding=ENCODING_ASCII):
82  """
83  Writes either into hdf5 file or into open mesh file
84  """
85 
86  numnodes = fem_mesh_obj.FemMesh.NodeCount
87 
88  dim = get_MaxDimElementFromList(get_FemMeshObjectElementTypes(fem_mesh_obj))[2]
89  effective_dim = dim
90  if dim <= 2:
91  effective_dim = 2 # effective dim is 2 for dim==1
92  geometrynode.set("GeometryType", "XY")
93  elif dim == 3:
94  geometrynode.set("GeometryType", "XYZ")
95 
96  recalc_nodes_ind_dict = {}
97 
98  if encoding == ENCODING_ASCII:
99  dataitem = ET.SubElement(
100  geometrynode, "DataItem", Dimensions="%d %d" % (numnodes, effective_dim), Format="XML"
101  )
102  nodes = []
103  for ind, (key, node) in enumerate(list(fem_mesh_obj.FemMesh.Nodes.items())):
104  nodes.append(node)
105  recalc_nodes_ind_dict[key] = ind
106 
107  dataitem.text = numpy_array_to_str(points_to_numpy(nodes, dim=effective_dim))
108  elif encoding == ENCODING_HDF5:
109  pass
110 
111  return recalc_nodes_ind_dict

114 def write_fenics_mesh_codim_xdmf(
115  fem_mesh_obj, topologynode, nodes_dict, codim=0, encoding=ENCODING_ASCII
116 ):
117  mesh_dimension = get_FemMeshObjectDimension(fem_mesh_obj)
118 
119  element_types = get_FemMeshObjectElementTypes(fem_mesh_obj, remove_zero_element_entries=True)
120  element_order = get_FemMeshObjectOrder(fem_mesh_obj)
121  # we get all elements from mesh to decide which one to write by selection of codim
122  """
123  nodeindices = [(
124  nodes_dict[ind] for ind in fem_mesh_obj.FemMesh.getElementNodes(fc_volume_ind)
125  ) for (fen_ind, fc_volume_ind) in enumerate(fc_cells)]
126  """
127  writeout_element_dimension = mesh_dimension - codim
128 
129  (num_topo, name_topo, dim_topo) = (0, "", 0)
130  for num, name, dim in element_types:
131  if writeout_element_dimension == dim:
132  (num_topo, name_topo, dim_topo) = (num, name, dim)
133 
134  (topology_type, nodes_per_element) = FreeCAD_to_Fenics_XDMF_dict[(name_topo, element_order)]
135 
136  topologynode.set("TopologyType", topology_type)
137  topologynode.set("NumberOfElements", str(num_topo))
138  topologynode.set("NodesPerElement", str(nodes_per_element))
139 
140  if dim_topo == 3:
141  fc_topo = fem_mesh_obj.FemMesh.Volumes
142  elif dim_topo == 2:
143  fc_topo = fem_mesh_obj.FemMesh.Faces
144  elif dim_topo == 1:
145  fc_topo = fem_mesh_obj.FemMesh.Edges
146  elif dim_topo == 0:
147  fc_topo = fem_mesh_obj.FemMesh.Nodes
148  else:
149  fc_topo = []
150  Console.PrintError(
151  "Dimension of mesh incompatible with export" + f" XDMF function: {dim_topo}\n"
152  )
153 
154  nodeindices = [
155  (nodes_dict[ind] for ind in fem_mesh_obj.FemMesh.getElementNodes(fc_topo_ind))
156  for (fen_ind, fc_topo_ind) in enumerate(fc_topo)
157  ]
158 
159  if encoding == ENCODING_ASCII:
160  dataitem = ET.SubElement(
161  topologynode,
162  "DataItem",
163  NumberType="UInt",
164  Dimensions="%d %d" % (num_topo, nodes_per_element),
165  Format="XML",
166  )
167  dataitem.text = numpy_array_to_str(tuples_to_numpy(nodeindices, nodes_per_element))
168  elif encoding == ENCODING_HDF5:
169  pass
170 
171  return fc_topo

231 def write_fenics_mesh_xdmf(fem_mesh_obj, outputfile, group_values_dict={}, encoding=ENCODING_ASCII):
232  """
233  For the export of xdmf.
234  """
235 
236  Console.PrintMessage(f"Converting {fem_mesh_obj.Label} to fenics XDMF File\n")
237  Console.PrintMessage(f"Dimension of mesh: {get_FemMeshObjectDimension(fem_mesh_obj)}\n")
238 
239  elements_in_mesh = get_FemMeshObjectElementTypes(fem_mesh_obj)
240  Console.PrintMessage(f"Elements appearing in mesh: {str(elements_in_mesh)}\n")
241  celltype_in_mesh = get_MaxDimElementFromList(elements_in_mesh)
242  (num_cells, cellname_fc, dim_cell) = celltype_in_mesh
243 
244  root = ET.Element("Xdmf", Version="3.0")
245  domain = ET.SubElement(root, "Domain")
246  base_grid = ET.SubElement(domain, "Grid", Name="base_mesh", GridType="Uniform")
247  base_topology = ET.SubElement(base_grid, "Topology")
248  base_geometry = ET.SubElement(base_grid, "Geometry")
249 
250  # TODO: for the general mesh: write out topology and geometry in grid node
251  # TODO: for every marked group write own grid node with topology (ref if cells)
252  # geometry ref, attribute
253 
254  # ***********************************
255  # write base topo and geometry
256  nodes_dict = write_fenics_mesh_points_xdmf(fem_mesh_obj, base_geometry, encoding=encoding)
257  write_fenics_mesh_codim_xdmf(
258  fem_mesh_obj, base_topology, nodes_dict, codim=0, encoding=encoding
259  )
260  # ***********************************
261 
262  fem_mesh = fem_mesh_obj.FemMesh
263  gmshgroups = get_FemMeshObjectMeshGroups(fem_mesh_obj)
264 
265  if gmshgroups != ():
266  Console.PrintMessage("found mesh groups\n")
267 
268  for g in gmshgroups:
269  mesh_function_type = fem_mesh.getGroupElementType(g)
270  mesh_function_codim = dim_cell - FreeCAD_Group_Dimensions[mesh_function_type]
271  mesh_function_name = fem_mesh.getGroupName(g)
272 
273  Console.PrintMessage(
274  f"group id: {g} (label: {mesh_function_name})"
275  + f" with element type {mesh_function_type} and"
276  + " codim {mesh_function_codim}\n"
277  )
278 
279  mesh_function_grid = ET.SubElement(
280  domain, "Grid", Name=mesh_function_name + "_mesh", GridType="Uniform"
281  )
282  mesh_function_topology = ET.SubElement(mesh_function_grid, "Topology")
283 
284  mesh_function_topology_description = write_fenics_mesh_codim_xdmf(
285  fem_mesh_obj,
286  mesh_function_topology,
287  nodes_dict,
288  codim=mesh_function_codim,
289  encoding=encoding,
290  )
291 
292  mesh_function_geometry = ET.SubElement(mesh_function_grid, "Geometry", Reference="XML")
293  mesh_function_geometry.text = "/Xdmf/Domain/Grid/Geometry"
294  mesh_function_attribute = ET.SubElement(mesh_function_grid, "Attribute")
295 
296  elem_dict = {}
297  (elem_mark_group, elem_mark_default) = group_values_dict.get(g, (1, 0))
298 
299  # TODO: is it better to save all groups each at once or collect all codim equal
300  # groups to put them into one function?
301  # TODO: nevertheless there has to be a dialog
302  # which fixes the default value and the mark value
303 
304  for e in fem_mesh.getGroupElements(g):
305  elem_dict[e] = elem_mark_group
306 
307  val_array = np.array(
308  [elem_dict.get(e, elem_mark_default) for e in mesh_function_topology_description]
309  )
310  topo_array = np.vstack((val_array,)).T
311  write_fenics_mesh_scalar_cellfunctions(
312  mesh_function_name, topo_array, mesh_function_attribute, encoding=ENCODING_ASCII
313  )
314 
315  # TODO: improve cell functions support
316 
317  fp = open(outputfile, "wb")
318  fp.write(b"""<?xml version="1.0"?>\n<!DOCTYPE Xdmf SYSTEM "Xdmf.dtd" []>\n""")
319  fp.write(ET.tostring(root))
320  # xml core functionality does not support pretty printing
321  # so the output file looks quite ugly
322  fp.close()
