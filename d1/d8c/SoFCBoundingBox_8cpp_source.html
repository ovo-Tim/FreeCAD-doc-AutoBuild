78  root = new SoSeparator();

80

83  bboxSep->addChild(bboxCoords);
84  root->addChild(bboxSep);
85 
86  // the lines of the box
87  bboxLines = new SoIndexedLineSet();
88  bboxLines->coordIndex.setNum(36);
89  bboxLines->coordIndex.setValues(0, 36, bBoxEdges);
90  bboxSep->addChild(bboxLines);
91 
92 
93  // create the text nodes, including a transform for each vertice offset
94  textSep = new SoSeparator();
95  for (int i = 0; i < 8; i++) {
96  auto temp = new SoSeparator();
97  auto trans = new SoTransform();
98  temp->addChild(trans);
99  auto text = new SoText2();
100  text->justification.setValue(SoText2::CENTER);
101  temp->addChild(text);
102  textSep->addChild(temp);
103  }
104 
105  // create the text nodes, including a transform for each dimension
106  dimSep = new SoSeparator();
107  for (int i = 0; i < 3; i++) {
108  auto temp = new SoSeparator();
109  auto trans = new SoTransform();
110  temp->addChild(trans);
111  auto text = new SoText2();
112  text->justification.setValue(SoText2::CENTER);
113  temp->addChild(text);
114  dimSep->addChild(temp);
115  }
116 
117  root->addChild(textSep);
118  root->addChild(dimSep);
119  root->ref();
120 }

127 void SoFCBoundingBox::GLRender (SoGLRenderAction *action)
128 {
129  SbVec3f corner[2], ctr, *vptr;
130  SbBool coord, dimension;
131 
132  // grab the current state
133  //SoState *state = action->getState();
134 
135  if (!shouldGLRender(action))
136  return;
137 
138  // get the latest values from the fields
139  corner[0] = minBounds.getValue();
140  corner[1] = maxBounds.getValue();
141  coord = coordsOn.getValue();
142  dimension = dimensionsOn.getValue();
143 
144  // set the coordinates for the LineSet to point to
145  vptr = bboxCoords->point.startEditing();
146  for (int i = 0; i < 8; i++) {
147  for (int j = 0; j < 3; j++) {
148  vptr[i][j] = corner[bBoxVerts[i][j]][j];
149  }
150  }
151 
152  // if coord is true then set the text nodes
153  if (coord) {
154  ctr = (corner[1] - corner[0]) / 2.0f;
155  for (int i = 0; i < 8; i++) {
156  // create the string for the text
157  std::stringstream str;
158  str.precision(2);
159  str.setf(std::ios::fixed | std::ios::showpoint);
160  str << "(" << vptr[i][0] << "," << vptr[i][1] << "," << vptr[i][2] << ")";
161 
162  SoSeparator *sep = static_cast<SoSeparator *>(textSep->getChild(i));
163  SoTransform *trans = static_cast<SoTransform *>(sep->getChild(0));
164 
165  trans->translation.setValue(vptr[i].getValue());
166  SoText2* t = static_cast<SoText2 *>(sep->getChild(1));
167  t->string.setValue(str.str().c_str());
168  }
169 
170  textSep->ref();
171  if (root->findChild(textSep) < 0)
172  root->addChild(textSep);
173  } else {
174  if (root->findChild(textSep) >= 0)
175  root->removeChild(textSep);
176  }
177 
178  // if dimension is true then set the text nodes
179  if (dimension) {
180  ctr = (corner[1] - corner[0]) / 2.0f;
181  for (int i = 0; i < 3; i++) {
182  // create the string for the text
183  std::stringstream str;
184  str.precision(2);
185  str.setf(std::ios::fixed | std::ios::showpoint);
186  str << (2.0f * ctr[i]);
187 
188  SoSeparator *sep = static_cast<SoSeparator *>(dimSep->getChild(i));
189  SoTransform *trans = static_cast<SoTransform *>(sep->getChild(0));
190 
191  SbVec3f point = corner[0];
192  point[i] += ctr[i];
193  trans->translation.setValue(point.getValue());
194  SoText2* t = static_cast<SoText2 *>(sep->getChild(1));
195  t->string.setValue(str.str().c_str());
196  }
197 
198  dimSep->ref();
199  if (root->findChild(dimSep) < 0)
200  root->addChild(dimSep);
201  } else {
202  if (root->findChild(dimSep) >= 0)
203  root->removeChild(dimSep);
204  }
205 
206  bboxCoords->point.finishEditing();
207 
208  // Avoid shading
209  SoState * state = action->getState();
210  state->push();
211  SoLazyElement::setLightModel(state, SoLazyElement::BASE_COLOR);
212  root->GLRender(action);
213  state->pop();
214 }
