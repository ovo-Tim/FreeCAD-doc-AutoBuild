113 // Similar to XCAFDoc_ShapeTool::FindSHUO but return only main SHUO, i.e. SHUO
114 // with no upper_usage. It should not be necessary if we strictly export from
115 // bottom up, but let's make sure of it.
116 static Standard_Boolean FindSHUO(const TDF_LabelSequence& theLabels,
117  Handle(XCAFDoc_GraphNode) & theSHUOAttr)
118 {
119  assert(theLabels.Length() > 1);
120  theSHUOAttr.Nullify();
121  TDF_AttributeSequence SHUOAttrs;
122  TDF_Label aCompLabel = theLabels.Value(1);
123  if (!::XCAFDoc_ShapeTool::GetAllComponentSHUO(aCompLabel, SHUOAttrs)) {
124  return Standard_False;
125  }
126  for (Standard_Integer i = 1; i <= SHUOAttrs.Length(); i++) {
127  Handle(XCAFDoc_GraphNode) anSHUO = Handle(XCAFDoc_GraphNode)::DownCast(SHUOAttrs.Value(i));
128  TDF_LabelSequence aUpLabels;
129  // check for any upper_usage
130  ::XCAFDoc_ShapeTool::GetSHUOUpperUsage(anSHUO->Label(), aUpLabels);
131  if (aUpLabels.Length() > 0) {
132  continue; // reject if there is one
133  }
134  int j = 2;
135  for (; anSHUO->NbChildren(); ++j) {
136  if (j > theLabels.Length()) {
137  j = 0;
138  break;
139  }
140  anSHUO = anSHUO->GetChild(1);
141  if (theLabels.Value(j) != anSHUO->Label().Father()) {
142  j = 0;
143  break;
144  }
145  }
146  if (j != theLabels.Length() + 1) {
147  continue;
148  }
149 
150  theSHUOAttr = Handle(XCAFDoc_GraphNode)::DownCast(SHUOAttrs.Value(i));
151  break;
152  }
153  return (!theSHUOAttr.IsNull());
154 }

156 TDF_Label
157 ExportOCAF2::findComponent(const char* subname, TDF_Label label, TDF_LabelSequence& labels)
158 {
159  const char* dot = strchr(subname, '.');
160  if (!dot) {
161  if (labels.Length() == 1) {
162  return labels.Value(1);
163  }
164  Handle(XCAFDoc_GraphNode) ret;
165  if (labels.Length() && (FindSHUO(labels, ret) || aShapeTool->SetSHUO(labels, ret))) {
166  return ret->Label();
167  }
168  return {};
169  }
170  TDF_LabelSequence components;
171  TDF_Label ref;
172  if (!aShapeTool->GetReferredShape(label, ref)) {
173  ref = label;
174  }
175  if (aShapeTool->GetComponents(ref, components)) {
176  for (int i = 1; i <= components.Length(); ++i) {
177  auto component = components.Value(i);
178  if (std::isdigit((int)subname[0])) {
179  auto n = std::to_string(i - 1) + ".";
180  if (boost::starts_with(subname, n)) {
181  labels.Append(component);
182  return findComponent(subname + n.size(), component, labels);
183  }
184  }
185  auto it = myNames.find(component);
186  if (it == myNames.end()) {
187  continue;
188  }
189  for (auto& n : it->second) {
190  if (boost::starts_with(subname, n)) {
191  labels.Append(component);
192  return findComponent(subname + n.size(), component, labels);
193  }
194  }
195  }
196  }
197  return {};
198 }

200 void ExportOCAF2::setupObject(TDF_Label label,
201  App::DocumentObject* obj,
202  const Part::TopoShape& shape,
203  const std::string& prefix,
204  const char* name,
205  bool force)
206 {
207  setName(label, obj, name);
208  if (aShapeTool->IsComponent(label)) {
209  auto& names = myNames[label];
210  // The subname reference may contain several possible namings.
211  if (!name) {
212  // simple object internal name
213  names.push_back(prefix + obj->getNameInDocument() + ".");
214  }
215  else {
216  // name is not NULL in case this is a collapsed link array element.
217  // Collapsed means that the element is not an actual object, and
218  // 'obj' here is actually the parent. The given 'name' is in fact
219  // the element index
220  names.push_back(prefix + name + ".");
221  // In case the subname reference is created when the link array is
222  // previously expanded, the element object will be named as the
223  // parent object internal name + '_i<index>'
224  names.push_back(prefix + obj->getNameInDocument() + "_i" + name + ".");
225  }
226  // Finally, the subname reference allows one to use the label for naming
227  // with preceding '$'
228  names.push_back(prefix + "$" + obj->Label.getValue() + ".");
229  }
230 
231  if (!getShapeColors || (!force && !mySetups.emplace(obj, name ? name : "").second)) {
232  return;
233  }
234 
235  std::map<std::string, std::map<std::string, App::Color>> colors;
236  static std::string marker(App::DocumentObject::hiddenMarker() + "*");
237  static std::array<const char*, 3> keys = {"Face*", "Edge*", marker.c_str()};
238  std::string childName;
239  if (name) {
240  childName = name;
241  childName += '.';
242  }
243  for (auto key : keys) {
244  for (auto& v : getShapeColors(obj, key)) {
245  const char* subname = v.first.c_str();
246  if (name) {
247  if (!boost::starts_with(v.first, childName)) {
248  continue;
249  }
250  subname += childName.size();
251  }
252  const char* dot = strrchr(subname, '.');
253  if (!dot) {
254  colors[""].emplace(subname, v.second);
255  }
256  else {
257  ++dot;
258  colors[std::string(subname, dot - subname)].emplace(dot, v.second);
259  }
260  }
261  }
262 
263  bool warned = false;
264 
265  for (auto& v : colors) {
266  TDF_Label nodeLabel = label;
267  if (!v.first.empty()) {
268  TDF_LabelSequence labels;
269  if (aShapeTool->IsComponent(label)) {
270  labels.Append(label);
271  }
272  nodeLabel = findComponent(v.first.c_str(), label, labels);
273  if (nodeLabel.IsNull()) {
274  FC_WARN("Failed to find component " << v.first);
275  continue;
276  }
277  }
278  for (auto& vv : v.second) {
279  if (vv.first == App::DocumentObject::hiddenMarker()) {
280  aColorTool->SetVisibility(nodeLabel, Standard_False);
281  continue;
282  }
283  const App::Color& c = vv.second;
284  Quantity_ColorRGBA color = Tools::convertColor(c);
285  auto colorType = vv.first[0] == 'F' ? XCAFDoc_ColorSurf : XCAFDoc_ColorCurv;
286  if (vv.first == "Face" || vv.first == "Edge") {
287  aColorTool->SetColor(nodeLabel, color, colorType);
288  continue;
289  }
290 
291  if (nodeLabel != label || aShapeTool->IsComponent(label)) {
292  // OCCT 7 seems to only support "Recommended practices for
293  // model styling and organization" version 1.2
294  // (https://www.cax-if.org/documents/rec_prac_styling_org_v12.pdf).
295  // The SHUO described in section 5.3 does not mention the
296  // capability of overriding context-depdendent element color,
297  // only whole shape color. Newer version of the same document
298  // (https://www.cax-if.org/documents/rec_prac_styling_org_v15.pdf)
299  // does support this, in section 5.1.
300  //
301  // The above observation is confirmed by further inspection of
302  // OCCT code, XCAFDoc_ShapeTool.cxx and STEPCAFControl_Writer.cxx.
303  if (!warned) {
304  warned = true;
305  FC_WARN("Current OCCT does not support element color override, for object "
306  << obj->getFullName());
307  }
308  // continue;
309  }
310 
311  auto subShape = shape.getSubShape(vv.first.c_str(), true);
312  if (subShape.IsNull()) {
313  FC_WARN("Failed to get subshape " << vv.first);
314  continue;
315  }
316 
317  // The following code is copied from OCCT 7.3 and is a work around
318  // a bug in previous versions
319  Handle(XCAFDoc_ShapeMapTool) A;
320  if (!nodeLabel.FindAttribute(XCAFDoc_ShapeMapTool::GetID(), A)) {
321  TopoDS_Shape aShape = aShapeTool->GetShape(nodeLabel);
322  if (!aShape.IsNull()) {
323  A = XCAFDoc_ShapeMapTool::Set(nodeLabel);
324  A->SetShape(aShape);
325  }
326  }
327 
328  TDF_Label subLabel = aShapeTool->AddSubShape(nodeLabel, subShape);
329  if (subLabel.IsNull()) {
330  FC_WARN("Failed to add subshape " << vv.first);
331  continue;
332  }
333  aColorTool->SetColor(subLabel, color, colorType);
334  }
335  }
336 }

338 void ExportOCAF2::exportObjects(std::vector<App::DocumentObject*>& objs, const char* name)
339 {
340  if (objs.empty()) {
341  return;
342  }
343  myObjects.clear();
344  myNames.clear();
345  mySetups.clear();
346  if (objs.size() == 1) {
347  exportObject(objs.front(), nullptr, TDF_Label());
348  }
349  else {
350  auto label = aShapeTool->NewShape();
351  App::Document* doc = nullptr;
352  bool sameDoc = true;
353  for (auto obj : objs) {
354  if (doc) {
355  sameDoc = sameDoc && doc == obj->getDocument();
356  }
357  else {
358  doc = obj->getDocument();
359  }
360  exportObject(obj, nullptr, label);
361  }
362 
363  if (!name && doc && sameDoc) {
364  name = doc->getName();
365  }
366  setName(label, nullptr, name);
367  }
368 
369  if (FC_LOG_INSTANCE.isEnabled(FC_LOGLEVEL_LOG)) {
370  Tools::dumpLabels(pDoc->Main(), aShapeTool, aColorTool);
371  }
372 
373  // Update is not performed automatically anymore:
374  // https://tracker.dev.opencascade.org/view.php?id=28055
375  aShapeTool->UpdateAssemblies();
376 }

378 TDF_Label ExportOCAF2::exportObject(App::DocumentObject* parentObj,
379  const char* sub,
380  TDF_Label parent,
381  const char* name)
382 {
383  App::DocumentObject* obj;
384  auto shape = Part::Feature::getTopoShape(parentObj, sub, false, nullptr, &obj, false, !sub);
385  if (!obj || shape.isNull()) {
386  if (obj) {
387  FC_WARN(obj->getFullName() << " has null shape");
388  }
389  return {};
390  }
391 
392  // sub may contain more than one hierarchy, e.g. Assembly container may use
393  // getSubObjects to skip some hierarchy containing constraints and stuff
394  // when exporting. We search for extra '.', and set it as prefix if found.
395  // When setting SHUO's, we'll need this prefix for matching.
396  std::string prefix;
397  if (sub) {
398  auto len = strlen(sub);
399  if (len > 1) {
400  --len;
401  // The prefix ends with the second last '.', so search for it.
402  for (int i = 0; len != 0; --len) {
403  if (sub[len] == '.' && ++i == 2) {
404  prefix = std::string(sub, len + 1);
405  break;
406  }
407  }
408  }
409  }
410 
411  TDF_Label label;
412  std::vector<App::DocumentObject*> links;
413 
414  int depth = 0;
415  auto linked = obj;
416  auto linkedShape = shape;
417  while (true) {
418  auto s = Part::Feature::getTopoShape(linked);
419  if (s.isNull() || !s.getShape().IsPartner(shape.getShape())) {
420  break;
421  }
422  linkedShape = s;
423  // Search using our own cache. We can't rely on ShapeTool::FindShape()
424  // in case this is an assembly. Because FindShape() search among its
425  // own computed shape, i.e. its own created compound, and thus will
426  // never match ours.
427  auto it = myObjects.find(linked);
428  if (it != myObjects.end()) {
429  for (auto l : links) {
430  myObjects.emplace(l, it->second);
431  }
432  // Note: OCAF does not seem to support reference of references. We
433  // have to flaten all multi-level link without scales. In other
434  // word, all link will all be forced to refer to the same
435  // non-located shape
436 
437  // retrieve OCAF computed shape, in case the current object returns
438  // a new shape every time Part::Feature::getTopoShape() is called.
439  auto baseShape = aShapeTool->GetShape(it->second);
440  shape.setShape(baseShape.Located(shape.getShape().Location()));
441  if (!parent.IsNull()) {
442  label = aShapeTool->AddComponent(parent, shape.getShape(), Standard_False);
443  }
444  else {
445  label = aShapeTool->AddShape(shape.getShape(), Standard_False, Standard_False);
446  }
447  setupObject(label, name ? parentObj : obj, shape, prefix, name);
448  return label;
449  }
450  auto next = linked->getLinkedObject(false, nullptr, false, depth++);
451  if (!next || linked == next) {
452  break;
453  }
454  linked = next;
455  links.push_back(linked);
456  }
457 
458  auto subs = obj->getSubObjects();
459  // subs empty means obj is not a container.
460  if (subs.empty()) {
461 
462  if (!parent.IsNull()) {
463  // Search for non-located shape to see if we've stored the original shape before
464  if (!aShapeTool->FindShape(shape.getShape(), label)) {
465  auto baseShape = linkedShape;
466  auto linked = links.empty() ? obj : links.back();
467  baseShape.setShape(baseShape.getShape().Located(TopLoc_Location()));
468  label = aShapeTool->NewShape();
469  aShapeTool->SetShape(label, baseShape.getShape());
470  setupObject(label, linked, baseShape, prefix);
471  }
472 
473  label = aShapeTool->AddComponent(parent, shape.getShape(), Standard_False);
474  setupObject(label, name ? parentObj : obj, shape, prefix, name);
475  }
476  else {
477  // Here means we are exporting a single non-assembly object. We must
478  // not call setupObject() on a non-located baseshape like above,
479  // because OCCT does not respect shape style sharing when not
480  // exporting assembly
481  if (!options.keepPlacement || shape.getPlacement() == Base::Placement()) {
482  shape.setShape(shape.getShape().Located(TopLoc_Location()));
483  }
484  else {
485  Base::Matrix4D mat = shape.getTransform();
486  shape.setShape(shape.getShape().Located(TopLoc_Location()));
487  // Transform with copy to conceal the transformation
488  shape.transformShape(mat, true);
489  // Even if the shape has no transformation, TopoShape still sets
490  // a TopLoc_Location, so we need to clear it again.
491  shape.setShape(shape.getShape().Located(TopLoc_Location()));
492  }
493  label = aShapeTool->AddShape(shape.getShape(), Standard_False, Standard_False);
494  auto o = name ? parentObj : obj;
495  if (o != linked) {
496  setupObject(label, linked, shape, prefix, nullptr, true);
497  }
498  setupObject(label, o, shape, prefix, name, true);
499  }
500 
501  myObjects.emplace(obj, label);
502  for (auto link : links) {
503  myObjects.emplace(link, label);
504  }
505  return label;
506  }
507 
508  if (obj->getExtensionByType<App::LinkBaseExtension>(true)
509  || obj->getExtensionByType<App::GeoFeatureGroupExtension>(true)) {
510  groupLinks.push_back(obj);
511  }
512 
513  // Create a new assembly
514  label = aShapeTool->NewShape();
515 
516  // check for link array
517  auto linkArray = obj->getLinkedObject(true)->getExtensionByType<App::LinkBaseExtension>(true);
518  if (linkArray && (linkArray->getShowElementValue() || !linkArray->getElementCountValue())) {
519  linkArray = nullptr;
520  }
521  for (auto& subobj : subs) {
522  App::DocumentObject* parentGrp = nullptr;
523  std::string childName;
524  auto sobj = obj->resolve(subobj.c_str(), &parentGrp, &childName);
525  if (!sobj) {
526  FC_WARN("Cannot find object " << obj->getFullName() << '.' << subobj);
527  continue;
528  }
529  int vis = -1;
530  if (parentGrp) {
531  if (!groupLinks.empty()
532  && parentGrp->getExtensionByType<App::GroupExtension>(true, false)) {
533  vis = groupLinks.back()->isElementVisible(childName.c_str());
534  }
535  else {
536  vis = parentGrp->isElementVisible(childName.c_str());
537  }
538  }
539 
540  if (vis < 0) {
541  vis = sobj->Visibility.getValue() ? 1 : 0;
542  }
543 
544  if (!vis && !options.exportHidden) {
545  continue;
546  }
547 
548  TDF_Label childLabel =
549  exportObject(obj, subobj.c_str(), label, linkArray ? childName.c_str() : nullptr);
550  if (childLabel.IsNull()) {
551  continue;
552  }
553 
554  if (!vis) {
555  // Work around OCCT bug. If no color setting here, it will crash.
556  // The culprit is at STEPCAFControl_Writer::1093 as shown below
557  //
558  // surfColor = Styles.EncodeColor(Quantity_Color(1,1,1,OCC_COLOR_SPACE),DPDCs,ColRGBs);
559  // PSA = Styles.MakeColorPSA ( item, surfColor, curvColor, isComponent );
560  // if ( isComponent )
561  // setDefaultInstanceColor( override, PSA);
562  //
563  // Can be fixed with following
564  // if ( !override.IsNull() && isComponent )
565  // setDefaultInstanceColor( override, PSA);
566  //
567  auto childShape = aShapeTool->GetShape(childLabel);
568  Quantity_ColorRGBA col;
569  if (!aColorTool->GetInstanceColor(childShape, XCAFDoc_ColorGen, col)
570  && !aColorTool->GetInstanceColor(childShape, XCAFDoc_ColorSurf, col)
571  && !aColorTool->GetInstanceColor(childShape, XCAFDoc_ColorCurv, col)) {
572  auto& c = options.defaultColor;
573  aColorTool->SetColor(childLabel, Tools::convertColor(c), XCAFDoc_ColorGen);
574  FC_WARN(Tools::labelName(childLabel) << " set default color");
575  }
576  aColorTool->SetVisibility(childLabel, Standard_False);
577  }
578  }
579 
580  if (!groupLinks.empty() && groupLinks.back() == obj) {
581  groupLinks.pop_back();
582  }
583 
584  // Finished adding components. Now retrieve the computed non-located shape
585  auto baseShape = shape;
586  baseShape.setShape(aShapeTool->GetShape(label));
587 
588  myObjects.emplace(obj, label);
589  for (auto link : links) {
590  myObjects.emplace(link, label);
591  }
592 
593  if (!parent.IsNull() && !links.empty()) {
594  linked = links.back();
595  }
596  else {
597  linked = obj;
598  }
599  setupObject(label, linked, baseShape, prefix);
600 
601  if (!parent.IsNull()) {
602  // If we are a component, swap in the base shape but keep our location.
603  shape.setShape(baseShape.getShape().Located(shape.getShape().Location()));
604  label = aShapeTool->AddComponent(parent, label, shape.getShape().Location());
605  setupObject(label, name ? parentObj : obj, shape, prefix, name);
606  }
607  return label;
608 }
