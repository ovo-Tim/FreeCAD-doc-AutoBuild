227  def file_to_base64_with_tag(self, file_path):

235  }
236  extension = os.path.splitext(file_path)[1]
237  mime_type = mime_types.get(
238  extension, "application/octet-stream"
239  ) # Default to binary data type if unknown
240 
241  if not os.path.exists(file_path):
242  Path.Log.error(f"File not found: {file_path}")
243  return "", ""
244 
245  try:
246  # Encode file to base64
247  with open(file_path, "rb") as file:
248  encoded_string = base64.b64encode(file.read()).decode()
249 
250  # Generate HTML tag based on file type
251  if extension in [".jpg", ".jpeg", ".png", ".svg"]:
252  html_tag = f'<img src="data:{mime_type};base64,{encoded_string}">'
253  elif extension in [".gcode", ".nc", ".tap", ".cnc"]:
254  html_tag = f'<a href="data:{mime_type};base64,{encoded_string}" download="{os.path.basename(file_path)}">Download G-code File</a>'
255  else:

257 
258  return encoded_string, html_tag
