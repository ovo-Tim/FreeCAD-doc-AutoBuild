214 void PropertiesDialog::displayUnitChanged(const QString& text)
215 {
216  if (text.isEmpty()) {
217  displayUnit = DisplayUnit();
218  ui->buttonBox->button(QDialogButtonBox::Ok)->setEnabled(true);
219  return;
220  }
221 
222  QPalette palette = ui->displayUnit->palette();
223  try {
224  std::unique_ptr<UnitExpression> e(
225  App::ExpressionParser::parseUnit(sheet, text.toUtf8().constData()));
226 
227  displayUnit = DisplayUnit(text.toUtf8().constData(), e->getUnit(), e->getScaler());
228  palette.setColor(QPalette::Text, Qt::black);
229  displayUnitOk = true;
230  }
231  catch (...) {
232  displayUnit = DisplayUnit();
233  palette.setColor(QPalette::Text, text.size() == 0 ? Qt::black : Qt::red);
234  displayUnitOk = false;
235  }
236  ui->buttonBox->button(QDialogButtonBox::Ok)->setEnabled(displayUnitOk && aliasOk);
237  ui->displayUnit->setPalette(palette);
238 }

257 void PropertiesDialog::apply()
258 {
259  if (!ranges.empty()) {
260  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Set cell properties"));
261  std::vector<Range>::const_iterator i = ranges.begin();
262  bool changes = false;
263 
264  for (; i != ranges.end(); ++i) {
265  if (orgAlignment != alignment) {
266  Gui::cmdAppObjectArgs(sheet,
267  "setAlignment('%s', '%s')",
268  i->rangeString().c_str(),
269  Cell::encodeAlignment(alignment).c_str());
270  changes = true;
271  }
272  if (orgStyle != style) {
273  Gui::cmdAppObjectArgs(sheet,
274  "setStyle('%s', '%s')",
275  i->rangeString().c_str(),
276  Cell::encodeStyle(style).c_str());
277  changes = true;
278  }
279  if (orgForegroundColor != foregroundColor) {
280  Gui::cmdAppObjectArgs(sheet,
281  "setForeground('%s', (%f,%f,%f,%f))",
282  i->rangeString().c_str(),
283  foregroundColor.r,
284  foregroundColor.g,
285  foregroundColor.b,
286  foregroundColor.a);
287  changes = true;
288  }
289  if (orgBackgroundColor != backgroundColor) {
290  Gui::cmdAppObjectArgs(sheet,
291  "setBackground('%s', (%f,%f,%f,%f))",
292  i->rangeString().c_str(),
293  backgroundColor.r,
294  backgroundColor.g,
295  backgroundColor.b,
296  backgroundColor.a);
297  changes = true;
298  }
299  if (orgDisplayUnit != displayUnit) {
300  std::string escapedstr =
301  Base::Tools::escapedUnicodeFromUtf8(displayUnit.stringRep.c_str());
302  escapedstr = Base::Tools::escapeQuotesFromString(escapedstr);
303  Gui::cmdAppObjectArgs(sheet,
304  "setDisplayUnit('%s', '%s')",
305  i->rangeString().c_str(),
306  escapedstr.c_str());
307  changes = true;
308  }
309  if (ranges.size() == 1 && ranges[0].size() == 1 && orgAlias != alias) {
310  Gui::cmdAppObjectArgs(sheet,
311  "setAlias('%s', '%s')",
312  i->address().c_str(),
313  alias.c_str());
314  changes = true;
315  }
316  }
317  if (changes) {
318  Gui::Command::commitCommand();
319  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
320  }
321  else {
322  Gui::Command::abortCommand();
323  }
324  }
325 }
