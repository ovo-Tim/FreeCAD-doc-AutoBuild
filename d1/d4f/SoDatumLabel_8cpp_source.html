216  void computeBBox(SbBox3f& box, SbVec3f& center) const
217  {
218  std::vector<SbVec3f> corners;
219  if (label->datumtype.getValue() == SoDatumLabel::DISTANCE ||
220  label->datumtype.getValue() == SoDatumLabel::DISTANCEX ||
221  label->datumtype.getValue() == SoDatumLabel::DISTANCEY ) {
222  corners = computeDistanceBBox();
223  }
224  else if (label->datumtype.getValue() == SoDatumLabel::RADIUS ||
225  label->datumtype.getValue() == SoDatumLabel::DIAMETER) {
226  corners = computeRadiusDiameterBBox();
227  }
228  else if (label->datumtype.getValue() == SoDatumLabel::ANGLE) {
229  corners = computeAngleBBox();
230  }
231  else if (label->datumtype.getValue() == SoDatumLabel::SYMMETRIC) {
232  corners = computeSymmetricBBox();
233  }
234  else if (label->datumtype.getValue() == SoDatumLabel::ARCLENGTH) {
235  corners = computeArcLengthBBox();
236  }
237 
238  getBBox(corners, box, center);
239  }

241 private:
242  void getBBox(const std::vector<SbVec3f>& corners, SbBox3f& box, SbVec3f& center) const
243  {
244  if (corners.size() > 1) {
245  float minX = FLT_MAX;
246  float minY = FLT_MAX;
247  float maxX = -FLT_MAX;
248  float maxY = -FLT_MAX;
249  for (SbVec3f it : corners) {
250  minX = (it[0] < minX) ? it[0] : minX;
251  minY = (it[1] < minY) ? it[1] : minY;
252  maxX = (it[0] > maxX) ? it[0] : maxX;
253  maxY = (it[1] > maxY) ? it[1] : maxY;
254  }
255 
256  // Store the bounding box
257  box.setBounds(SbVec3f(minX, minY, 0.0F), SbVec3f (maxX, maxY, 0.0F));
258  center = box.getCenter();
259  }

261  std::vector<SbVec3f> computeDistanceBBox() const
262  {
263  SbVec2s imgsize;
264  int nc {};
265  int srcw = 1;
266  int srch = 1;
267 
268  const unsigned char * dataptr = label->image.getValue(imgsize, nc);
269  if (dataptr) {
270  srcw = imgsize[0];
271  srch = imgsize[1];
272  }
273 
274  float aspectRatio = (float) srcw / (float) srch;
275  float imgHeight = scale * (float) (srch);
276  float imgWidth = aspectRatio * imgHeight;
277 
278  // Get the points stored in the pnt field
279  const SbVec3f *points = label->pnts.getValues(0);
280  if (label->pnts.getNum() < 2) {
281  return {};
282  }
283 
284  SbVec3f textOffset;
285 
286  float length = label->param1.getValue();
287  float length2 = label->param2.getValue();
288 
289  SbVec3f p1 = points[0];
290  SbVec3f p2 = points[1];
291 
292  SbVec3f dir;
293  SbVec3f normal;
294  if (label->datumtype.getValue() == SoDatumLabel::DISTANCE) {
295  dir = (p2-p1);
296  }
297  else if (label->datumtype.getValue() == SoDatumLabel::DISTANCEX) {
298  dir = SbVec3f( (p2[0] - p1[0] >= FLT_EPSILON) ? 1 : -1, 0, 0);
299  }
300  else if (label->datumtype.getValue() == SoDatumLabel::DISTANCEY) {
301  dir = SbVec3f(0, (p2[1] - p1[1] >= FLT_EPSILON) ? 1 : -1, 0);
302  }
303 
304  dir.normalize();
305  normal = SbVec3f (-dir[1], dir[0], 0);
306 
307  // when the datum line is not parallel to p1-p2 the projection of
308  // p1-p2 on normal is not zero, p2 is considered as reference and p1
309  // is replaced by its projection p1_
310  float normproj12 = (p2 - p1).dot(normal);
311  SbVec3f p1_ = p1 + normproj12 * normal;
312 
313  SbVec3f midpos = (p1_ + p2)/2;
314 
315  float offset1 = ((length + normproj12 < 0.0F) ? -1.0F : 1.0F) * float(srch);
316  float offset2 = ((length < 0.0F) ? -1.0F : 1.0F) * float(srch);
317 
318  textOffset = midpos + normal * length + dir * length2;
319  float margin = imgHeight / 4.0F;
320 
321  SbVec3f perp1 = p1_ + normal * (length + offset1 * scale);
322  SbVec3f perp2 = p2 + normal * (length + offset2 * scale);
323 
324  // Finds the mins and maxes
325  std::vector<SbVec3f> corners;
326  corners.push_back(p1);
327  corners.push_back(p2);
328  corners.push_back(perp1);
329  corners.push_back(perp2);
330 
331  // Make sure that the label is inside the bounding box
332  corners.push_back(textOffset + dir * (imgWidth / 2.0F + margin) + normal * (srch + margin));
333  corners.push_back(textOffset - dir * (imgWidth / 2.0F + margin) + normal * (srch + margin));
334  corners.push_back(textOffset + dir * (imgWidth / 2.0F + margin) - normal * margin);
335  corners.push_back(textOffset - dir * (imgWidth / 2.0F + margin) - normal * margin);
336 
337  return corners;
338  }

361  }
362 
363  // Get the Points

441  SbVec3f v2(cos(endangle), sin(endangle), 0);
442 
443  SbVec3f pnt1 = p0+(len2-margin)*v1;
444  SbVec3f pnt2 = p0+(len2+margin)*v1;
445  SbVec3f pnt3 = p0+(len2-margin)*v2;
446  SbVec3f pnt4 = p0+(len2+margin)*v2;
447 
448  // Finds the mins and maxes

450

452  SbVec3f img2 = SbVec3f(-imgWidth / 2.0F, imgHeight / 2, 0.0F);

457  img2 += textOffset;
458  img3 += textOffset;
459  img4 += textOffset;

463  corners.push_back(pnt2);
464  corners.push_back(pnt3);
465  corners.push_back(pnt4);
466  corners.push_back(img1);
467  corners.push_back(img2);

493  std::vector<SbVec3f> computeArcLengthBBox() const
494  {
495  SbVec2s imgsize;
496  int nc {};
497  int srcw = 1;
498  int srch = 1;
499 
500  const unsigned char * dataptr = label->image.getValue(imgsize, nc);
501  if (dataptr) {
502  srcw = imgsize[0];
503  srch = imgsize[1];
504  }
505 
506  float aspectRatio = (float) srcw / (float) srch;
507  float imgHeight = scale * (float) (srch);
508  float imgWidth = aspectRatio * imgHeight;
509 
510  // Get the points stored in the pnt field
511  const SbVec3f *points = label->pnts.getValues(0);
512  if (label->pnts.getNum() < 3) {
513  return {};
514  }
515 
516  SbVec3f ctr = points[0];
517  SbVec3f p1 = points[1];
518  SbVec3f p2 = points[2];
519 
520  SbVec3f img1 = SbVec3f(-imgWidth / 2, -imgHeight / 2, 0.F);
521  SbVec3f img2 = SbVec3f(-imgWidth / 2, imgHeight / 2, 0.F);
522  SbVec3f img3 = SbVec3f( imgWidth / 2, -imgHeight / 2, 0.F);
523  SbVec3f img4 = SbVec3f( imgWidth / 2, imgHeight / 2, 0.F);
524 
525  //Text orientation
526  SbVec3f dir = (p2 - p1);
527  dir.normalize();
528  SbVec3f normal = SbVec3f (-dir[1], dir[0], 0);
529  float angle = atan2f(dir[1],dir[0]);
530 
531  // Rotate through an angle
532  float s = sin(angle);
533  float c = cos(angle);
534  img1 = SbVec3f((img1[0] * c) - (img1[1] * s), (img1[0] * s) + (img1[1] * c), 0.F);
535  img2 = SbVec3f((img2[0] * c) - (img2[1] * s), (img2[0] * s) + (img2[1] * c), 0.F);
536  img3 = SbVec3f((img3[0] * c) - (img3[1] * s), (img3[0] * s) + (img3[1] * c), 0.F);
537  img4 = SbVec3f((img4[0] * c) - (img4[1] * s), (img4[0] * s) + (img4[1] * c), 0.F);
538 
539  float length = label->param1.getValue();
540 
541 
542  // Text location
543  SbVec3f vm = (p1+p2)/2 - ctr;
544  vm.normalize();
545 
546  SbVec3f vc1 = (p1 - ctr);
547  SbVec3f vc2 = (p2 - ctr);
548 
549  float startangle = atan2f(vc1[1], vc1[0]);
550  float endangle = atan2f(vc2[1], vc2[0]);
551  if (endangle < startangle) {
552  endangle += 2. * M_PI;
553  }
554 
555  SbVec3f textCenter;
556  if (endangle - startangle <= M_PI) {
557  textCenter = ctr + vm * (length + imgHeight);
558  } else {
559  textCenter = ctr - vm * (length + 2. * imgHeight);
560  }
561  img1 += textCenter;
562  img2 += textCenter;
563  img3 += textCenter;
564  img4 += textCenter;
565 
566  // Finds the mins and maxes
567  std::vector<SbVec3f> corners;
568  float margin = imgHeight / 4.0F;
569  corners.push_back(textCenter + dir * (imgWidth / 2.0F + margin) - normal * (imgHeight / 2.0F + margin));
570  corners.push_back(textCenter - dir * (imgWidth / 2.0F + margin) - normal * (imgHeight / 2.0F + margin));
571  corners.push_back(textCenter + dir * (imgWidth / 2.0F + margin) + normal * (imgHeight / 2.0F + margin));
572  corners.push_back(textCenter - dir * (imgWidth / 2.0F + margin) + normal * (imgHeight / 2.0F + margin));
573 
574  return corners;
