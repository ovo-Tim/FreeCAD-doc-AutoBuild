106 void QGTracker::mousePressEvent(QGraphicsSceneMouseEvent *event)
107 {
108  QPointF myScenePos = event->scenePos();
109  if (!m_sleep) {
110  double someLimit = Rez::guiX(1.0);
111  QPointF manhat = myScenePos - m_lastClick;
112 // Base::Console().Message("QGT::mousePressEvent - scenePos: (%.3f, %.3f) lastClick:(%.3f, %.3f)\n",
113 // myScenePos.x(), myScenePos.y(), m_lastClick.x(), m_lastClick.y());
114 // Base::Console().Message("QGT::mousePressEvent - manhat(%.3f, %.3f) mLength: %.3f\n",
115 // manhat.x(), manhat.y(), manhat.manhattanLength());
116  if (manhat.manhattanLength() < someLimit) {
117 // Base::Console().Message("QGT::mousePressEvent - too close to last click\n");
118  } else {
119  if (event->button() == Qt::LeftButton) {
120  if (event->modifiers() & Qt::ControlModifier) {
121  myScenePos = snapToAngle(myScenePos);
122  }
123  onMousePress(myScenePos);
124 
125  } else if (event->button() == Qt::RightButton) {
126  terminateDrawing();
127  }
128  }
129  }
130  m_lastClick = myScenePos;
131  QGIPrimPath::mousePressEvent(event);
132 }

185 QPointF QGTracker::snapToAngle(QPointF dumbPt)
186 {
187  // If no point selected yet, snapping has no sense
188  if (m_points.empty())
189  return dumbPt;
190 
191  QPointF result(dumbPt);
192  double angleIncr = M_PI / 8.0; //15*
193  //mirror last clicked point and event point to get sensible coords
194  QPointF last(m_points.back().x(), -m_points.back().y());
195  QPointF pt(dumbPt.x(), -dumbPt.y());
196 
197  //let's do math with sensible coord system.
198  QPointF qVec = last - pt; //vec from end of track to end of tail
199  double actual = atan2(-qVec.y(), qVec.x());
200  if (actual < 0.0) {
201  actual = (2 * M_PI) + actual; //map to +ve angle
202  }
203 
204  double intPart;
205  double remain = modf(actual/angleIncr, &intPart);
206  if (!TechDraw::DrawUtil::fpCompare(remain, 0.0)) { //not n*15
207  double low = intPart * angleIncr;
208  double high = (intPart + 1) * angleIncr;
209  double lowGap = actual - low;
210  double highGap = high - actual;
211  if (lowGap > highGap) {
212  actual = high;
213  } else {
214  actual = low;
215  }
216 
217  double vecLen = sqrt(pow(qVec.x(), 2) + pow(qVec.y(), 2));
218  double xPart = vecLen * cos(actual) * -1.0; //actual is not in Qt coords
219  double yPart = vecLen * sin(actual) * -1.0;
220  result = QPointF(last.x() + xPart, (-last.y()) + yPart); //back to silly qt coords!
221  }
222  return result;
223 }

226 void QGTracker::onMousePress(QPointF pos)
227 {
228 // Base::Console().Message("QGT::onMousePress(%s)\n", TechDraw::DrawUtil::formatVector(pos).c_str());
229  m_points.push_back(pos);
230  TrackerMode mode = getTrackerMode();
231  if (m_points.size() > 1) {
232  switch (mode) {
233  case TrackerMode::None:
234  break;
235  case TrackerMode::Line:
236  setPathFromPoints(m_points);
237  break;
238  case TrackerMode::Rectangle:
239  setSquareFromPoints(m_points);
240  break;
241  case TrackerMode::Circle:
242  setCircleFromPoints(m_points);
243  break;
244  case TrackerMode::Point:
245  //do nothing
246 // setPoint(m_points);
247  break;
248  }
249  } else if (m_points.size() == 1) { //first point selected
250  //just return pos to caller
251  getPickedQGIV(pos);
252  setCursor(Qt::CrossCursor); //why cross??
253 // Q_EMIT qViewPicked(pos, m_qgParent); //not in use yet.
254  if (mode == TrackerMode::Point) {
255  setPoint(m_points); //first point is mouse click scene pos
256  terminateDrawing();
257  }
258  }
259 
260  if ( (m_points.size() == 2) &&
261  ( (getTrackerMode() == TrackerMode::Circle) ||
262  (getTrackerMode() == TrackerMode::Rectangle) ) ) { //only 2 points for square/circle
263  terminateDrawing();
264  }
265 }

300 void QGTracker::getPickedQGIV(QPointF pos)
301 {
302  setVisible(false);
303  m_qgParent = nullptr;
304  QList<QGraphicsView *> views = scene()->views();
305  QGraphicsView* ourView = views.front(); //only 1 view / 1 scene / 1 mdipage
306  QTransform viewXForm = ourView->transform();
307  QGraphicsItem* pickedItem = scene()->itemAt(pos, viewXForm);
308  if (pickedItem) {
309  QGraphicsItem* topItem = pickedItem->topLevelItem();
310  if (topItem != pickedItem) {
311  pickedItem = topItem;
312  } //pickedItem sb a QGIV
313  QGIView* qgParent = dynamic_cast<QGIView*>(pickedItem);
314  if (qgParent) {
315  m_qgParent = qgParent;
316  }
317  }
318  setVisible(true);
319  return;
320 }
