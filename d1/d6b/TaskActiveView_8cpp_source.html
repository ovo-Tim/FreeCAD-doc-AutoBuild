102 TechDraw::DrawViewImage* TaskActiveView::createActiveView()
103 {
104  // Base::Console().Message("TAV::createActiveView()\n");
105 
106  //make sure there is an 3D MDI to grab!!
107  if (!Gui::getMainWindow()) {
108  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("No Main Window"),
109  QObject::tr("Can not find the main window"));
110  return nullptr;
111  }
112 
113  App::Document* pageDocument = m_pageFeat->getDocument();
114  std::string documentName = m_pageFeat->getDocument()->getName();
115  Gui::Document* pageGuiDocument =
116  Gui::Application::Instance->getDocument(pageDocument->getName());
117 
118  //if the active view is a 3d window, use that.
119  View3DInventor* view3d = qobject_cast<View3DInventor*>(Gui::getMainWindow()->activeWindow());
120  if (!view3d) {
121  // active view is not a 3D view, try to find one in the current document
122  auto views3dAll = pageGuiDocument->getMDIViewsOfType(Gui::View3DInventor::getClassTypeId());
123  if (!views3dAll.empty()) {
124  view3d = qobject_cast<View3DInventor*>(views3dAll.front());
125  }
126  else {
127  //this code is only for the rare case where the page's document does not have a
128  //3D window. It might occur if the user closes the 3D window, but leaves, for
129  //example, a DrawPage window open.
130  //the active window is not a 3D view, and the page's document does not have a
131  //3D view, so try to find one somewhere among the open windows.
132  auto mdiWindows = Gui::getMainWindow()->windows();
133  for (auto& mdi : mdiWindows) {
134  auto mdiView = qobject_cast<View3DInventor*>(mdi);
135  if (mdiView) {
136  view3d = mdiView;
137  break;
138  }
139  }
140  }
141  }
142  if (!view3d) {
143  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("No 3D Viewer"),
144  QObject::tr("Can not find a 3D viewer"));
145  return nullptr;
146  }
147 
148  //we are sure we have a 3D window!
149 
150  const std::string objectName{"ActiveView"};
151  std::string imageName = m_pageFeat->getDocument()->getUniqueObjectName(objectName.c_str());
152  std::string generatedSuffix {imageName.substr(objectName.length())};
153  std::string imageType = "TechDraw::DrawViewImage";
154 
155  std::string pageName = m_pageFeat->getNameInDocument();
156 
157  //the Page's document may not be the active one, so we need to get the right
158  //document by name instead of using ActiveDocument
159  Command::doCommand(Command::Doc, "App.getDocument('%s').addObject('%s','%s')",
160  documentName.c_str(), imageType.c_str(), imageName.c_str());
161 
162  Command::doCommand(Command::Doc, "App.activeDocument().%s.translateLabel('DrawActiveView', 'ActiveView', '%s')",
163  imageName.c_str(), imageName.c_str());
164 
165  Command::doCommand(Command::Doc, "App.getDocument('%s').%s.addView(App.getDocument('%s').%s)",
166  documentName.c_str(), pageName.c_str(), documentName.c_str(),
167  imageName.c_str());
168 
169  App::Document* doc = m_pageFeat->getDocument();
170  std::string special = "/" + imageName + "image.png";
171  std::string dir = doc->TransientDir.getValue();
172  std::string fileSpec = dir + special;
173 
174  //fixes fail to create 2nd Active view with same name in old docs
175  Base::FileInfo fi(fileSpec);
176  if (fi.exists()) {
177  //old filename were unique by pageName + imageName only
178  fi.deleteFile();
179  }
180 
181  //better way of making temp file name
182  std::string baseName = pageName + imageName;
183  std::string tempName =
184  Base::FileInfo::getTempFileName(baseName.c_str(), doc->TransientDir.getValue()) + ".png";
185 
186  QColor bg = ui->ccBgColor->color();
187  if (ui->cbUse3d->isChecked()) {
188  bg = QColor();
189  }
190  else if (ui->cbNoBG->isChecked()) {
191  bg = QColor(Qt::transparent);
192  }
193 
194  int imageWidth{SXGAWidth};
195  int imageHeight{SXGAHeight};
196  if (ui->cbCrop->isChecked()) {
197  imageWidth = Rez::guiX(ui->qsbWidth->rawValue());
198  imageHeight = Rez::guiX(ui->qsbHeight->rawValue());
199  }
200 
201  QImage image(imageWidth, imageHeight,
202  QImage::Format_RGB32); //arbitrary initial image size.
203  image.fill(QColor(Qt::transparent));
204  Grabber3d::quickView(view3d, bg, image);
205  bool success = image.save(QString::fromStdString(tempName));
206 
207  if (!success) {
208  Base::Console().Error("ActiveView could not save file: %s\n", fileSpec.c_str());
209  }
210 
211  tempName = DU::cleanFilespecBackslash(tempName);
212  Command::doCommand(Command::Doc, "App.getDocument('%s').%s.ImageFile = '%s'",
213  documentName.c_str(), imageName.c_str(), tempName.c_str());
214  Command::doCommand(Command::Doc, "App.getDocument('%s').%s.Width = %.5f", documentName.c_str(),
215  imageName.c_str(), ui->qsbWidth->rawValue());
216  Command::doCommand(Command::Doc, "App.getDocument('%s').%s.Height = %.5f", documentName.c_str(),
217  imageName.c_str(), ui->qsbHeight->rawValue());
218 
219  App::DocumentObject* newObj = m_pageFeat->getDocument()->getObject(imageName.c_str());
220  TechDraw::DrawViewImage* newImg = dynamic_cast<TechDraw::DrawViewImage*>(newObj);
221  if (!newObj || !newImg)
222  throw Base::RuntimeError("TaskActiveView - new image object not found");
223  Gui::Document* guiDoc = Gui::Application::Instance->getDocument(newImg->getDocument());
224  if (guiDoc) {
225  Gui::ViewProvider* vp = guiDoc->getViewProvider(newImg);
226  if (vp) {
227  auto vpImage = dynamic_cast<ViewProviderImage*>(vp);
228  if (vpImage) {
229  vpImage->Crop.setValue(ui->cbCrop->isChecked());
230  }
231  }
232  }
233 
234  return newImg;
235 }
