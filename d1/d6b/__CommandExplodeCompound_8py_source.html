79 def cmdExplode():
80  FreeCAD.ActiveDocument.openTransaction("Explode")
81  try:
82  sel = FreeCADGui.Selection.getSelectionEx()
83  if len(sel) != 1:
84  raise ValueError("Bad selection","More than one object selected. You have selected {num} objects.".format(num= len(sel)))
85  obj = sel[0].Object
86  FreeCADGui.addModule("CompoundTools.Explode")

88  FreeCADGui.doCommand("CompoundTools.Explode.explodeCompound(input_obj)")
