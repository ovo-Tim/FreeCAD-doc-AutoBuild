504 // More efficient algorithm to find the recursive inList of an object,
505 // including possible external parents. One shortcoming of this algorithm is
506 // it does not detect cyclic reference, althgouth it won't crash either.
507 void DocumentObject::getInListEx(std::set<App::DocumentObject*>& inSet,
508  bool recursive,
509  std::vector<App::DocumentObject*>* inList) const
510 {
511 #ifdef USE_OLD_DAG
512  std::map<DocumentObject*, std::set<App::DocumentObject*>> outLists;
513 
514  // Old DAG does not have pre-built InList, and must calculate The InList by
515  // going through all objects' OutLists. So we collect all objects and their
516  // outLists first here.
517  for (auto doc : GetApplication().getDocuments()) {
518  for (auto obj : doc->getObjects()) {
519  if (!obj || !obj->isAttachedToDocument() || obj == this) {
520  continue;
521  }
522  const auto& outList = obj->getOutList();
523  outLists[obj].insert(outList.begin(), outList.end());
524  }
525  }
526 
527  std::stack<DocumentObject*> pendings;
528  pendings.push(const_cast<DocumentObject*>(this));
529  while (pendings.size()) {
530  auto obj = pendings.top();
531  pendings.pop();
532  for (auto& v : outLists) {
533  if (v.first == obj) {
534  continue;
535  }
536  auto& outList = v.second;
537  // Check the outList to see if the object is there, and pend the
538  // object for recursive check if it's not already in the inList
539  if (outList.find(obj) != outList.end() && inSet.insert(v.first).second && recursive) {
540  pendings.push(v.first);
541  }
542  }
543  }
544 #else // USE_OLD_DAG
545 
546  if (!recursive) {
547  inSet.insert(_inList.begin(), _inList.end());
548  if (inList) {
549  *inList = _inList;
550  }
551  return;
552  }
553 
554  std::stack<DocumentObject*> pendings;
555  pendings.push(const_cast<DocumentObject*>(this));
556  while (!pendings.empty()) {
557  auto obj = pendings.top();
558  pendings.pop();
559  for (auto o : obj->getInList()) {
560  if (o && o->isAttachedToDocument() && inSet.insert(o).second) {
561  pendings.push(o);
562  if (inList) {
563  inList->push_back(o);
564  }
565  }
566  }
567  }
568 
569 #endif
570 }

579 void _getOutListRecursive(std::set<DocumentObject*>& objSet,
580  const DocumentObject* obj,
581  const DocumentObject* checkObj,
582  int depth)
583 {
584  for (const auto objIt : obj->getOutList()) {
585  // if the check object is in the recursive inList we have a cycle!
586  if (objIt == checkObj || depth <= 0) {
587  throw Base::BadGraphError(
588  "DocumentObject::getOutListRecursive(): cyclic dependency detected!");
589  }
590 
591  // if the element was already in the set then there is no need to process it again
592  auto pair = objSet.insert(objIt);
593  if (pair.second) {
594  _getOutListRecursive(objSet, objIt, checkObj, depth - 1);
595  }
596  }
597 }

1310 DocumentObject* DocumentObject::resolve(const char* subname,
1311  App::DocumentObject** parent,
1312  std::string* childName,
1313  const char** subElement,
1314  PyObject** pyObj,
1315  Base::Matrix4D* pmat,
1316  bool transform,
1317  int depth) const
1318 {
1319  auto self = const_cast<DocumentObject*>(this);
1320  if (parent) {
1321  *parent = nullptr;
1322  }
1323  if (subElement) {
1324  *subElement = nullptr;
1325  }
1326 
1327  auto obj = getSubObject(subname, pyObj, pmat, transform, depth);
1328  if (!obj || !subname || *subname == 0) {
1329  return self;
1330  }
1331 
1332  if (!parent && !subElement) {
1333  return obj;
1334  }
1335 
1336  // NOTE, the convention of '.' separated SubName demands a mandatory ending
1337  // '.' for each object name in SubName, even if there is no subelement
1338  // following it. So finding the last dot will give us the end of the last
1339  // object name.
1340  const char* dot = nullptr;
1341  if (Data::isMappedElement(subname) || !(dot = strrchr(subname, '.')) || dot == subname) {
1342  if (subElement) {
1343  *subElement = dot ? dot + 1 : subname;
1344  }
1345  return obj; // this means no parent object reference in SubName
1346  }
1347 
1348  if (parent) {
1349  *parent = self;
1350  }
1351 
1352  bool elementMapChecked = false;
1353  const char* lastDot = dot;
1354  for (--dot;; --dot) {
1355  // check for the second last dot, which is the end of the last parent object
1356  if (*dot == '.' || dot == subname) {
1357  // We can't get parent object by its name, because the object may be
1358  // externally linked (i.e. in a different document). So go through
1359  // getSubObject again.
1360  if (!elementMapChecked) {
1361  elementMapChecked = true;
1362  const char* sub = dot == subname ? dot : dot + 1;
1363  if (Data::isMappedElement(sub)) {
1364  lastDot = dot;
1365  if (dot == subname) {
1366  break;
1367  }
1368  else {
1369  continue;
1370  }
1371  }
1372  }
1373  if (dot == subname) {
1374  break;
1375  }
1376  auto sobj = getSubObject(std::string(subname, dot - subname + 1).c_str());
1377  if (sobj != obj) {
1378  if (parent) {
1379  // Link/LinkGroup has special visibility handling of plain
1380  // group, so keep ascending
1381  if (!sobj->hasExtension(GroupExtension::getExtensionClassTypeId(), false)) {
1382  *parent = sobj;
1383  break;
1384  }
1385  for (auto ddot = dot - 1; ddot != subname; --ddot) {
1386  if (*ddot != '.') {
1387  continue;
1388  }
1389  auto sobj = getSubObject(std::string(subname, ddot - subname + 1).c_str());
1390  if (!sobj->hasExtension(GroupExtension::getExtensionClassTypeId(), false)) {
1391  *parent = sobj;
1392  break;
1393  }
1394  }
1395  }
1396  break;
1397  }
1398  }
1399  }
1400  if (childName && lastDot != dot) {
1401  if (*dot == '.') {
1402  ++dot;
1403  }
1404  const char* nextDot = strchr(dot, '.');
1405  assert(nextDot);
1406  *childName = std::string(dot, nextDot - dot);
1407  }
1408  if (subElement) {
1409  *subElement = *lastDot == '.' ? lastDot + 1 : lastDot;
1410  }
1411  return obj;
1412 }
