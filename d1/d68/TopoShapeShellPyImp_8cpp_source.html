54 std::string TopoShapeShellPy::representation() const
55 {
56  // Note: As the return type is 'const char*' we cannot create a temporary
57  // char array neither on the stack because the array would be freed when
58  // leaving the scope nor on the heap because we would have a memory leak.

60  // This, however, is not a problem as long as we only use this method in
61  // _repr().
62 
63  std::stringstream str;
64  str << "<Shell object at " << getTopoShapePtr() << ">";

96 PyObject* TopoShapeShellPy::add(PyObject *args)
97 {
98  PyObject *obj;
99  if (!PyArg_ParseTuple(args, "O!", &(Part::TopoShapeFacePy::Type), &obj))
100  return nullptr;
101 
102  BRep_Builder builder;
103  TopoDS_Shape shell = getTopoShapePtr()->getShape();
104  if (shell.IsNull()) {
105  builder.MakeShell(TopoDS::Shell(shell));
106  }
107 
108  try {
109  const TopoShape& shape = *static_cast<TopoShapeFacePy*>(obj)->getTopoShapePtr();
110  const auto &sh = shape.getShape();
111  if (!sh.IsNull()) {
112  builder.Add(shell, sh);
113  BRepCheck_Analyzer check(shell);
114  getTopoShapePtr()->mapSubElement(shape);
115  if (!check.IsValid()) {
116  ShapeUpgrade_ShellSewing sewShell;
117  getTopoShapePtr()->setShape(sewShell.ApplySewing(shell));
118  }
119  }
120  else {
121  Standard_Failure::Raise("cannot add empty shape");
122  }
123  }
124  catch (Standard_Failure& e) {
125  PyErr_SetString(PartExceptionOCCError, e.GetMessageString());
126  return nullptr;
127  }
128 
129  getTopoShapePtr()->setShape(shell);
130 
131  Py_Return;
132 }
