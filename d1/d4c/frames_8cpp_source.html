111  double Vector::Norm() const
112  {
113  double tmp1;
114  double tmp2;
115  tmp1 = fabs(data[0]);
116  tmp2 = fabs(data[1]);
117  if (tmp1 >= tmp2) {

119  if (tmp1 >= tmp2) {

122  return 0;

198  void Rotation::GetQuaternion(double& x,double& y,double& z, double& w) const
199  {
200  double trace = (*this)(0,0) + (*this)(1,1) + (*this)(2,2);
201  double epsilon=1E-12;
202  if( trace > epsilon ){
203  double s = 0.5 / sqrt(trace + 1.0);
204  w = 0.25 / s;
205  x = ( (*this)(2,1) - (*this)(1,2) ) * s;
206  y = ( (*this)(0,2) - (*this)(2,0) ) * s;
207  z = ( (*this)(1,0) - (*this)(0,1) ) * s;
208  }else{
209  if ( (*this)(0,0) > (*this)(1,1) && (*this)(0,0) > (*this)(2,2) ){
210  double s = 2.0 * sqrt( 1.0 + (*this)(0,0) - (*this)(1,1) - (*this)(2,2));
211  w = ((*this)(2,1) - (*this)(1,2) ) / s;
212  x = 0.25 * s;
213  y = ((*this)(0,1) + (*this)(1,0) ) / s;
214  z = ((*this)(0,2) + (*this)(2,0) ) / s;
215  } else if ((*this)(1,1) > (*this)(2,2)) {
216  double s = 2.0 * sqrt( 1.0 + (*this)(1,1) - (*this)(0,0) - (*this)(2,2));
217  w = ((*this)(0,2) - (*this)(2,0) ) / s;
218  x = ((*this)(0,1) + (*this)(1,0) ) / s;
219  y = 0.25 * s;
220  z = ((*this)(1,2) + (*this)(2,1) ) / s;
221  }else {
222  double s = 2.0 * sqrt( 1.0 + (*this)(2,2) - (*this)(0,0) - (*this)(1,1) );
223  w = ((*this)(1,0) - (*this)(0,1) ) / s;
224  x = ((*this)(0,2) + (*this)(2,0) ) / s;
225  y = ((*this)(1,2) + (*this)(2,1) ) / s;
226  z = 0.25 * s;
227  }
228  }

297 Rotation Rotation::Rot2(const Vector& rotvec,double angle) {
298  // rotvec should be normalized !
299  // The formula is
300  // V.(V.tr) + st*[V x] + ct*(I-V.(V.tr))
301  // can be found by multiplying it with an arbitrary vector p
302  // and noting that this vector is rotated.
303  double ct = cos(angle);
304  double st = sin(angle);
305  double vt = 1-ct;
306  double m_vt_0=vt*rotvec(0);
307  double m_vt_1=vt*rotvec(1);
308  double m_vt_2=vt*rotvec(2);
309  double m_st_0=rotvec(0)*st;
310  double m_st_1=rotvec(1)*st;
311  double m_st_2=rotvec(2)*st;
312  double m_vt_0_1=m_vt_0*rotvec(1);
313  double m_vt_0_2=m_vt_0*rotvec(2);
314  double m_vt_1_2=m_vt_1*rotvec(2);
315  return Rotation(
316  ct + m_vt_0*rotvec(0),
317  -m_st_2 + m_vt_0_1,
318  m_st_1 + m_vt_0_2,
319  m_st_2 + m_vt_0_1,
320  ct + m_vt_1*rotvec(1),
321  -m_st_0 + m_vt_1_2,
322  -m_st_1 + m_vt_0_2,
323  m_st_0 + m_vt_1_2,
324  ct + m_vt_2*rotvec(2)
325  );
326 }

352 double Rotation::GetRotAngle(Vector& axis,double eps) const {
353  double ca = (data[0]+data[4]+data[8]-1)/2.0;
354  double t= eps*eps/2.0;
355  if (ca>1-t) {
356  // undefined choose the Z-axis, and angle 0
357  axis = Vector(0,0,1);
358  return 0;
359  }
360  if (ca < -1+t) {
361  // The case of angles consisting of multiples of M_PI:
362  // two solutions, choose a positive Z-component of the axis
363  double x = sqrt( (data[0]+1.0)/2);
364  double y = sqrt( (data[4]+1.0)/2);
365  double z = sqrt( (data[8]+1.0)/2);
366  if ( data[2] < 0) x=-x;
367  if ( data[7] < 0) y=-y;
368  if ( x*y*data[1] < 0) x=-x; // this last line can be necessary when z is 0
369  // z always >= 0
370  // if z equal to zero
371  axis = Vector( x,y,z );
372  return PI;
373  }
374  double angle;
375  double mod_axis;
376  double axisx, axisy, axisz;
377  axisx = data[7]-data[5];
378  axisy = data[2]-data[6];
379  axisz = data[3]-data[1];
380  mod_axis = sqrt(axisx*axisx+axisy*axisy+axisz*axisz);
381  axis = Vector(axisx/mod_axis,
382  axisy/mod_axis,
383  axisz/mod_axis );
384  angle = atan2(mod_axis/2,ca);
385  return angle;
386 }
