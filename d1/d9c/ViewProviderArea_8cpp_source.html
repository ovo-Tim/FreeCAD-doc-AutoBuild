58 void ViewProviderArea::dragObject(App::DocumentObject* obj)
59 {
60  Path::FeatureArea* area = getObject<Path::FeatureArea>();
61  std::vector<App::DocumentObject*> sources = area->Sources.getValues();
62  for (std::vector<App::DocumentObject*>::iterator it = sources.begin(); it != sources.end();
63  ++it) {
64  if (*it == obj) {
65  sources.erase(it);
66  area->Sources.setValues(sources);
