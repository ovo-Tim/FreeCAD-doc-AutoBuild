57  def __init__(self, name: str, code: str = ""):
58  """Create a parser for the macro named "name". Note that the name is only
59  used as the context for error messages, it is not otherwise important."""
60  self.name = name
61  self.parse_results = {
62  "comment": "",
63  "url": "",
64  "wiki": "",
65  "version": "",
66  "other_files": [""],
67  "author": "",
68  "date": "",
69  "license": "",
70  "icon": "",
71  "xpm": "",
72  }
73  self.remaining_item_map = {}
74  self.console = None if FreeCAD is None else FreeCAD.Console
75  self.current_thread = DummyThread() if QtCore is None else QtCore.QThread.currentThread()
76  if code:
77  self.fill_details_from_code(code)

79  def _reset_map(self):
80  """This map tracks which items we've already read. If the same parser is used
81  twice, it has to be reset."""
82  self.remaining_item_map = {
83  "__comment__": "comment",
84  "__web__": "url",
85  "__wiki__": "wiki",
86  "__version__": "version",
87  "__files__": "other_files",
88  "__author__": "author",
89  "__date__": "date",
90  "__license__": "license",
91  "__licence__": "license", # accept either spelling
92  "__icon__": "icon",
93  "__xpm__": "xpm",

96  def fill_details_from_code(self, code: str) -> None:
97  """Reads in the macro code from the given string and parses it for its
98  metadata."""
99 
100  self._reset_map()
101  line_counter = 0
102  content_lines = io.StringIO(code)
103  while content_lines and line_counter < self.MAX_LINES_TO_SEARCH:
104  line = content_lines.readline()
105  if not line:
106  break
107  if self.current_thread.isInterruptionRequested():
108  return
109  line_counter += 1
110  if not line.startswith("__"):
111  # Speed things up a bit... this comparison is very cheap
112  continue
113  try:
114  self._process_line(line, content_lines)
115  except SyntaxError as e:
116  err_string = f"Syntax error when parsing macro {self.name}:\n{str(e)}"
117  if self.console:
118  self.console.PrintWarning(err_string)
119  else:
120  print(err_string)

122  def _process_line(self, line: str, content_lines: io.StringIO):
123  """Given a single line of the macro file, see if it matches one of our items,
124  and if so, extract the data."""
125 
126  lowercase_line = line.lower()
127  for key in self.remaining_item_map:
128  if lowercase_line.startswith(key):
129  self._process_key(key, line, content_lines)
130  break
131 
132  def _process_key(self, key: str, line: str, content_lines: io.StringIO):
133  """Given a line that starts with a known key, extract the data for that key,
134  possibly reading in additional lines (if it contains a line continuation
135  character, or is a triple-quoted string)."""
136 
137  line = self._handle_backslash_continuation(line, content_lines)
138  line, was_triple_quoted = self._handle_triple_quoted_string(line, content_lines)
139 
140  _, _, line = line.partition("=")
141  if not was_triple_quoted:
142  line, _, _ = line.partition("#")
143  self._detect_illegal_content(line)
144  final_content_line = line.strip()
145 
146  stripped_of_quotes = self._strip_quotes(final_content_line)
147  if stripped_of_quotes is not None:
148  self._standard_extraction(self.remaining_item_map[key], stripped_of_quotes)
149  self.remaining_item_map.pop(key)
150  else:
151  self._apply_special_handling(key, line)

153  @staticmethod
154  def _handle_backslash_continuation(line, content_lines) -> str:
155  while line.strip().endswith("\\"):
156  line = line.strip()[:-1]
157  concat_line = content_lines.readline()
158  line += concat_line.strip()
159  return line
160 
161  @staticmethod
162  def _handle_triple_quoted_string(line, content_lines) -> Tuple[str, bool]:
163  result = line
164  was_triple_quoted = False
165  if '"""' in result:
166  was_triple_quoted = True
167  while True:
168  new_line = content_lines.readline()
169  if not new_line:
170  raise SyntaxError("Syntax error while reading macro")
171  if '"""' in new_line:
172  last_line, _, _ = new_line.partition('"""')
173  result += last_line + '"""'
174  break
175  result += new_line
176  return result, was_triple_quoted
177 
178  @staticmethod
179  def _strip_quotes(line) -> str:
180  line = line.strip()
181  stripped_of_quotes = None
182  if line.startswith('"""') and line.endswith('"""'):
183  stripped_of_quotes = line[3:-3]
184  elif (line[0] == '"' and line[-1] == '"') or (line[0] == "'" and line[-1] == "'"):
185  stripped_of_quotes = line[1:-1]
186  return stripped_of_quotes

188  def _standard_extraction(self, value: str, match_group: str):
189  """For most macro metadata values, this extracts the required data"""
190  if isinstance(self.parse_results[value], str):
191  self.parse_results[value] = match_group
192  if value == "comment":
193  self._cleanup_comment()
194  elif value == "license":
195  self._cleanup_license()
196  elif isinstance(self.parse_results[value], list):
197  self.parse_results[value] = [of.strip() for of in match_group.split(",")]
198  else:
199  raise SyntaxError(f"Conflicting data type for {value}")

201  def _cleanup_comment(self):
202  """Remove HTML from the comment line, and truncate it at 512 characters."""
203 
204  self.parse_results["comment"] = re.sub(r"<.*?>", "", self.parse_results["comment"])
205  if len(self.parse_results["comment"]) > 512:
206  self.parse_results["comment"] = self.parse_results["comment"][:511] + "…"

208  def _cleanup_license(self):
209  if get_license_manager is not None:
210  lm = get_license_manager()
211  self.parse_results["license"] = lm.normalize(self.parse_results["license"])

213  def _apply_special_handling(self, key: str, line: str):
214  # Macro authors are supposed to be providing strings here, but in some
215  # cases they are not doing so. If this is the "__version__" tag, try
216  # to apply some special handling to accept numbers, and "__date__"
217  if key == "__version__":
218  self._process_noncompliant_version(line)
219  self.remaining_item_map.pop(key)
220  return
221 
222  raise SyntaxError(f"Failed to process {key} from {line}")
223 
224  def _process_noncompliant_version(self, after_equals):
225  if is_float(after_equals):
226  self.parse_results["version"] = str(after_equals).strip()
227  elif "__date__" in after_equals.lower() and self.parse_results["date"]:
228  self.parse_results["version"] = self.parse_results["date"]
229  else:
230  self.parse_results["version"] = "(Unknown)"
231  raise SyntaxError(f"Unrecognized version string {after_equals}")

233  @staticmethod
234  def _detect_illegal_content(line: str):
235  """Raise a syntax error if this line contains something we can't handle"""
236 
237  lower_line = line.strip().lower()
238  if lower_line.startswith("'") and lower_line.endswith("'"):
239  return
240  if lower_line.startswith('"') and lower_line.endswith('"'):
241  return
242  if is_float(lower_line):
243  return
244  if lower_line == "__date__":
245  return
246  raise SyntaxError(f"Metadata is expected to be a static string, but got {line}")
