41 constexpr int UnitSignatureLengthBits = 4;
42 constexpr int UnitSignatureMassBits = 4;
43 constexpr int UnitSignatureTimeBits = 4;
44 constexpr int UnitSignatureElectricCurrentBits = 4;
45 constexpr int UnitSignatureThermodynamicTemperatureBits = 4;

47 constexpr int UnitSignatureLuminousIntensityBits = 4;
48 constexpr int UnitSignatureAngleBits = 4;

50 struct UnitSignature {
51  int32_t Length: UnitSignatureLengthBits;
52  int32_t Mass: UnitSignatureMassBits;
53  int32_t Time: UnitSignatureTimeBits;
54  int32_t ElectricCurrent: UnitSignatureElectricCurrentBits;
55  int32_t ThermodynamicTemperature: UnitSignatureThermodynamicTemperatureBits;
56  int32_t AmountOfSubstance: UnitSignatureAmountOfSubstanceBits;
57  int32_t LuminousIntensity: UnitSignatureLuminousIntensityBits;
58  int32_t Angle: UnitSignatureAngleBits;
59 };

61 static inline uint32_t sigVal(const std::string &op,
62  int length, int mass, int time, int electricCurrent,
63  int thermodynamicTemperature, int amountOfSubstance, int luminousIntensity, int angle)
64 {
65  if ( ( length >= (1 << (UnitSignatureLengthBits - 1)) ) ||
66  ( mass >= (1 << (UnitSignatureMassBits - 1)) ) ||
67  ( time >= (1 << (UnitSignatureTimeBits - 1)) ) ||
68  ( electricCurrent >= (1 << (UnitSignatureElectricCurrentBits - 1)) ) ||
69  ( thermodynamicTemperature >= (1 << (UnitSignatureThermodynamicTemperatureBits - 1)) ) ||
70  ( amountOfSubstance >= (1 << (UnitSignatureAmountOfSubstanceBits - 1)) ) ||
71  ( luminousIntensity >= (1 << (UnitSignatureLuminousIntensityBits - 1)) ) ||
72  ( angle >= (1 << (UnitSignatureAngleBits - 1)) ) ) {
73  throw Base::OverflowError(("Unit overflow in " + op).c_str());
74  }
75  if ( ( length < -(1 << (UnitSignatureLengthBits - 1)) ) ||
76  ( mass < -(1 << (UnitSignatureMassBits - 1)) ) ||
77  ( time < -(1 << (UnitSignatureTimeBits - 1)) ) ||
78  ( electricCurrent < -(1 << (UnitSignatureElectricCurrentBits - 1)) ) ||
79  ( thermodynamicTemperature < -(1 << (UnitSignatureThermodynamicTemperatureBits - 1)) ) ||
80  ( amountOfSubstance < -(1 << (UnitSignatureAmountOfSubstanceBits - 1)) ) ||
81  ( luminousIntensity < -(1 << (UnitSignatureLuminousIntensityBits - 1)) ) ||
82  ( angle < -(1 << (UnitSignatureAngleBits - 1)) ) ) {
83  throw Base::UnderflowError(("Unit underflow in " + op).c_str());
84  }
85 
86  UnitSignature Sig;
87  Sig.Length = length;
88  Sig.Mass = mass;
89  Sig.Time = time;
90  Sig.ElectricCurrent = electricCurrent;
91  Sig.ThermodynamicTemperature = thermodynamicTemperature;
92  Sig.AmountOfSubstance = amountOfSubstance;
93  Sig.LuminousIntensity = luminousIntensity;
94  Sig.Angle = angle;
95 
96  uint32_t ret;
97  memcpy(&ret, &Sig, sizeof(ret));
98  return ret;
99 }

138 Unit Unit::pow(double exp) const
139 {
140  UnitSignature sig;
141  memcpy(&sig, &Val, sizeof(Val));
142  auto isInt = [](double value) {
143  return std::fabs(std::round(value) - value) < std::numeric_limits<double>::epsilon();
144  };
145  if (!isInt(sig.Length * exp) ||
146  !isInt(sig.Mass * exp) ||
147  !isInt(sig.Time * exp) ||
148  !isInt(sig.ElectricCurrent * exp) ||
149  !isInt(sig.ThermodynamicTemperature * exp) ||
150  !isInt(sig.AmountOfSubstance * exp) ||
151  !isInt(sig.LuminousIntensity * exp) ||
152  !isInt(sig.Angle * exp))
153  throw Base::UnitsMismatchError("pow() of unit not possible");
154 
155  Unit result;
156  result.Val = sigVal("pow()",
157  sig.Length * exp,
158  sig.Mass * exp,
159  sig.Time * exp,
160  sig.ElectricCurrent * exp,
161  sig.ThermodynamicTemperature * exp,
162  sig.AmountOfSubstance * exp,
163  sig.LuminousIntensity * exp,
164  sig.Angle * exp);
165 
166  return result;
167 }

169 Unit Unit::sqrt() const
170 {
171  UnitSignature sig;
172  memcpy(&sig, &Val, sizeof(Val));
173  // All components of unit must be either zero or dividable by 2
174  if (!((sig.Length % 2) == 0) &&
175  ((sig.Mass % 2) == 0) &&
176  ((sig.Time % 2) == 0) &&
177  ((sig.ElectricCurrent % 2) == 0) &&
178  ((sig.ThermodynamicTemperature % 2) == 0) &&
179  ((sig.AmountOfSubstance % 2) == 0) &&
180  ((sig.LuminousIntensity % 2) == 0) &&
181  ((sig.Angle % 2) == 0))
182  throw Base::UnitsMismatchError("sqrt() needs even dimensions");
183 
184  Unit result;
185  result.Val = sigVal("sqrt()",
186  sig.Length >> 1,
187  sig.Mass >> 1,
188  sig.Time >> 1,
189  sig.ElectricCurrent >> 1,
190  sig.ThermodynamicTemperature >> 1,
191  sig.AmountOfSubstance >> 1,
192  sig.LuminousIntensity >> 1,
193  sig.Angle >> 1);
194 
195  return result;
196 }

361 std::string Unit::getString() const
362 {
363  if (isEmpty()) {
364  return {};
365  }
366 
367  std::stringstream ret;
368  UnitSignature sig;
369  memcpy(&sig, &Val, sizeof(Val));
370 
371  if (sig.Length > 0 ||
372  sig.Mass > 0 ||
373  sig.Time > 0 ||
374  sig.ElectricCurrent > 0 ||
375  sig.ThermodynamicTemperature > 0 ||
376  sig.AmountOfSubstance > 0 ||
377  sig.LuminousIntensity > 0 ||
378  sig.Angle > 0 ) {
379 
380  bool mult = false;
381  if (sig.Length > 0) {
382  mult = true;
383  ret << "mm";
384  if (sig.Length > 1) {
385  ret << "^" << sig.Length;
386  }
387  }
388 
389  if (sig.Mass > 0) {
390  if (mult) {
391  ret << '*';
392  }
393  mult = true;
394  ret << "kg";
395  if (sig.Mass > 1) {
396  ret << "^" << sig.Mass;
397  }
398  }
399 
400  if (sig.Time > 0) {
401  if (mult) {
402  ret << '*';
403  }
404  mult = true;
405  ret << "s";
406  if (sig.Time > 1) {
407  ret << "^" << sig.Time;
408  }
409  }
410 
411  if (sig.ElectricCurrent > 0) {
412  if (mult) {
413  ret << '*';
414  }

419  }
420  }
421

424  ret << '*';

426  mult = true;

464  }
465  }
466  else {
467  ret << "1";
468  }
469 
470  if (sig.Length < 0 ||
471  sig.Mass < 0 ||
472  sig.Time < 0 ||
473  sig.ElectricCurrent < 0 ||
474  sig.ThermodynamicTemperature < 0 ||
475  sig.AmountOfSubstance < 0 ||
476  sig.LuminousIntensity < 0 ||
477  sig.Angle < 0 ) {
478  ret << "/";
479 
480  int nnom = 0;
481  nnom += sig.Length < 0 ? 1 : 0;
482  nnom += sig.Mass < 0 ? 1 : 0;
483  nnom += sig.Time < 0 ? 1 : 0;
484  nnom += sig.ElectricCurrent < 0 ? 1 : 0;
485  nnom += sig.ThermodynamicTemperature < 0 ? 1 : 0;
486  nnom += sig.AmountOfSubstance < 0 ? 1 : 0;
487  nnom += sig.LuminousIntensity < 0 ? 1 : 0;
488  nnom += sig.Angle < 0 ? 1 : 0;
489 
490  if (nnom > 1) {
491  ret << '(';
492  }
493 
494  bool mult = false;
495  if (sig.Length < 0) {
496  ret << "mm";
497  mult = true;
498  if (sig.Length < -1) {
499  ret << "^" << abs(sig.Length);
500  }
501  }
502 
503  if (sig.Mass < 0) {
504  if (mult) {
505  ret << '*';
506  }
507  mult = true;
508  ret << "kg";
509  if (sig.Mass < -1) {
510  ret << "^" << abs(sig.Mass);
511  }
512  }
513 
514  if (sig.Time < 0) {

517  }

520  if (sig.Time < -1) {

522  }
