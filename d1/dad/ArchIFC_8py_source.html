44 class IfcRoot:
45  """This class defines the common methods and properties for managing IFC data.
46 
47  IFC, or Industry Foundation Classes are a standardised way to digitally
48  describe the built environment. The ultimate goal of IFC is to provide
49  better interoperability between software that deals with the built
50  environment. You can learn more here:
51  https://technical.buildingsmart.org/standards/ifc/
52 
53  You can learn more about the technical details of the IFC schema here:
54  https://standards.buildingsmart.org/IFC/RELEASE/IFC4/FINAL/HTML/
55 
56  This class is further segmented down into IfcProduct and IfcContext.
57  """

101  def setupIfcAttributes(self, obj):
102  """Set up the IFC attributes in the object's properties.
103 
104  Add the attributes specified in the object's IFC type schema, to the
105  object's properties. Do not re-add them if they're already present.
106  Also remove old IFC attribute properties that no longer appear in the
107  schema for backwards compatibility.
108 
109  Do so using the .addIfcAttributes() and
110  .purgeUnusedIfcAttributesFromPropertiesList() methods.
111 
112  Learn more about IFC attributes here:
113  https://standards.buildingsmart.org/IFC/RELEASE/IFC4/FINAL/HTML/schema/chapter-3.htm#attribute
114  """
115 
116  ifcTypeSchema = self.getIfcTypeSchema(obj.IfcType)
117  if ifcTypeSchema is None:
118  return
119  self.purgeUnusedIfcAttributesFromPropertiesList(ifcTypeSchema, obj)
120  self.addIfcAttributes(ifcTypeSchema, obj)

180  def getCanonicalisedIfcTypes(self):
181  """Get the names of IFC types, converted to the form used in Arch.
182 
183  Change the names of all IFC types to a more human readable form which
184  is used instead throughout Arch instead of the raw type names. The
185  names have the "Ifc" stripped from the start of their name, and spaces
186  inserted between the words.
187 
188  Returns
189  -------
190  list of str

192  will have names in the same order as they appear in the schema's
193  JSON, as per the .keys() method of dicts.
194

199  def getIfcAttributeSchema(self, ifcTypeSchema, name):
200  """Get the schema of an IFC attribute with the given name.
201 
202  Convert the IFC attribute's name from the human readable version Arch
203  uses, and convert it to the less readable name it has in the IFC
204  schema.
205 
206  Parameters

209  The schema of the IFC type to access the attribute of.
210  name: str
211  The name the attribute has in Arch.
212 
213  Returns
214  -------
215  dict
216  Returns the schema of the attribute.
217  None
218  Returns None if the IFC type does not have the attribute requested.

227  def addIfcAttributes(self, ifcTypeSchema, obj):
228  """Add the attributes of the IFC type's schema to the object's properties.
229 
230  Add the attributes as properties of the object. Also add the
231  attribute's schema within the object's IfcData property. Do so using
232  the .addIfcAttribute() method.
233 
234  Also add expressions to copy data from the object's editable
235  properties. This means the IFC properties will remain accurate with
236  the actual values of the object. Do not do so for all IFC properties.
237  Do so using the .addIfcAttributeValueExpressions() method.
238 
239  Learn more about expressions here:
240  https://wiki.freecad.org/Expressions
241 
242  Do not add the attribute if the object has a property with the
243  attribute's name. Also do not add the attribute if its name is
244  RefLatitude, RefLongitude, or Name.
245 
246  Parameters
247  ----------
248  ifcTypeSchema: dict
249  The schema of the IFC type.
250  """
251 
252  for attribute in ifcTypeSchema["attributes"]:
253  if attribute["name"] in obj.PropertiesList \
254  or attribute["name"] == "RefLatitude" \
255  or attribute["name"] == "RefLongitude" \
256  or attribute["name"] == "Name":
257  continue
258  self.addIfcAttribute(obj, attribute)
259  self.addIfcAttributeValueExpressions(obj, attribute)

299  def addIfcAttributeValueExpressions(self, obj, attribute):
300  """Add expressions for IFC attributes, so they stay accurate with the object.
301 
302  Add expressions to the object that copy data from the editable
303  properties of the object. This ensures that the IFC attributes will
304  remain accurate with the actual values of the object.
305 
306  Currently, add expressions for the following IFC attributes:
307 
308  - OverallWidth
309  - OverallHeight
310  - ElevationWithFlooring
311  - Elevation
312  - NominalDiameter
313  - BarLength
314  - RefElevation
315  - LongName
316 
317  Learn more about expressions here:
318  https://wiki.freecad.org/Expressions
319 
320  Parameters
321  ----------
322  attribute: dict
323  The schema of the attribute to add the expression for.
324  """
325 
326  if obj.getGroupOfProperty(attribute["name"]) != "IFC Attributes" \
327  or attribute["name"] not in obj.PropertiesList:
328  return
329  if attribute["name"] == "OverallWidth":
330  if "Length" in obj.PropertiesList:
331  obj.setExpression("OverallWidth", "Length.Value")
332  elif "Width" in obj.PropertiesList:
333  obj.setExpression("OverallWidth", "Width.Value")
334  elif obj.Shape and (obj.Shape.BoundBox.XLength > obj.Shape.BoundBox.YLength):
335  obj.setExpression("OverallWidth", "Shape.BoundBox.XLength")
336  elif obj.Shape:
337  obj.setExpression("OverallWidth", "Shape.BoundBox.YLength")
338  elif attribute["name"] == "OverallHeight":
339  if "Height" in obj.PropertiesList:
340  obj.setExpression("OverallHeight", "Height.Value")
341  else:
342  obj.setExpression("OverallHeight", "Shape.BoundBox.ZLength")
343  elif attribute["name"] == "ElevationWithFlooring" and "Shape" in obj.PropertiesList:
344  obj.setExpression("ElevationWithFlooring", "Shape.BoundBox.ZMin")
345  elif attribute["name"] == "Elevation" and "Placement" in obj.PropertiesList:
346  obj.setExpression("Elevation", "Placement.Base.z")
347  elif attribute["name"] == "NominalDiameter" and "Diameter" in obj.PropertiesList:
348  obj.setExpression("NominalDiameter", "Diameter.Value")
349  elif attribute["name"] == "BarLength" and "Length" in obj.PropertiesList:
350  obj.setExpression("BarLength", "Length.Value")
351  elif attribute["name"] == "RefElevation" and "Elevation" in obj.PropertiesList:
352  obj.setExpression("RefElevation", "Elevation.Value")
353  elif attribute["name"] == "LongName":
354  obj.LongName = obj.Label

410  def purgeUnusedIfcAttributesFromPropertiesList(self, ifcTypeSchema, obj):
411  """Remove properties representing IFC attributes if they no longer appear.
412 
413  Remove the property representing an IFC attribute, if it does not
414  appear in the schema of the IFC type provided. Also, remove the
415  property if its attribute is an enum type, presumably for backwards
416  compatibility.
417 
418  Learn more about IFC enums here:
419  https://standards.buildingsmart.org/IFC/RELEASE/IFC4/FINAL/HTML/schema/chapter-3.htm#enumeration
420  """
421 
422  for property in obj.PropertiesList:
423  if obj.getGroupOfProperty(property) != "IFC Attributes":
424  continue
425  ifcAttribute = self.getIfcAttributeSchema(ifcTypeSchema, property)
426  if ifcAttribute is None or ifcAttribute["is_enum"] is True:
427  obj.removeProperty(property)
