186 def arcAdjustmentAngle(arc1, arc2):
187  """arcAdjustmentAngle(arc1, arc2)...
188  Return adjustment angle to apply to arc2 in order to align it with arc1.
189  Arcs must have same center point."""
190  center = arc1.Curve.Center
191  cntr2 = arc2.Curve.Center
192 
193  # Verify centers of arcs are same
194  if center.sub(cntr2).Length > 0.0000001:
195  return None
196 
197  # Calculate midpoint of arc1, and standard angle from center to that midpoint
198  midPntArc1 = arc1.valueAt(
199  arc1.FirstParameter + (arc1.LastParameter - arc1.FirstParameter) / 2.0
200  )
201  midPntVect1 = midPntArc1.sub(center)
202  ang1 = getStandardAngle(midPntVect1.x, midPntVect1.y)
203 
204  # Calculate midpoint of arc2, and standard angle from center to that midpoint
205  midPntArc2 = arc2.valueAt(
206  arc2.FirstParameter + (arc2.LastParameter - arc2.FirstParameter) / 2.0
207  )
208  midPntVect2 = midPntArc2.sub(center)
209  ang2 = getStandardAngle(midPntVect2.x, midPntVect2.y)
210 
211  # Return adjustment angle to apply to arc2 in order to align with arc1
212  return ang1 - ang2

527  """
