302 
303  Base::Type type = Base::Type::fromName(o);

307  std::shared_ptr<const Part::GeometryExtension> ext(

321  "Geometry extension does not exist anymore.");

470 PyObject* ExternalGeometryFacadePy::getExtensions(PyObject* args)
471 {
472  if (!PyArg_ParseTuple(args, "")) {
473  PyErr_SetString(Part::PartExceptionOCCError, "No arguments were expected");
474  return nullptr;
475  }
476 
477  try {
478  const std::vector<std::weak_ptr<const Part::GeometryExtension>> ext =
479  this->getExternalGeometryFacadePtr()->getExtensions();
480 
481  Py::List list;
482 
483  for (std::size_t i = 0; i < ext.size(); ++i) {
484 
485  std::shared_ptr<const Part::GeometryExtension> p = ext[i].lock();
486 
487  if (p) {
488  // we create a python copy and add it to the list
489  try {
490  list.append(Py::asObject(p->copyPyObject()));
491  }
492  catch (Base::NotImplementedError&) {
493  // silently ignoring extensions not having a Python object
494  }
495  }
496  }
497 
498  return Py::new_reference_to(list);
499  }
500  catch (const Base::ValueError& e) {
501  PyErr_SetString(Part::PartExceptionOCCError, e.what());
502  return nullptr;
503  }
