113 void TaskSelectLineAttributes::setUiEdit()
114 {
115  setWindowTitle(tr("Select line attributes"));
116  int lineStyle = LineFormat::getCurrentLineFormat().getStyle();
117  // line numbering starts at 1, not 0
118  DrawGuiUtil::loadLineStyleChoices(ui->cbLineStyle, m_lineGenerator);
119  if (ui->cbLineStyle->count() >= lineStyle ) {
120  ui->cbLineStyle->setCurrentIndex(lineStyle - 1);
121  }
122 
123  // TODO: how to handle translation of a string with arg parameters in it?
124  ui->rbThin->setText(QString::fromUtf8("Thin %1").arg(QString::number(TechDraw::LineGroup::getDefaultWidth("Thin"))));
125  ui->rbMiddle->setText(QString::fromUtf8("Middle %1").arg(QString::number(TechDraw::LineGroup::getDefaultWidth("Graphic"))));
126  ui->rbThick->setText(QString::fromUtf8("Thick %1").arg(QString::number(TechDraw::LineGroup::getDefaultWidth("Thick"))));
127 
128  double lineWidth = LineFormat::getCurrentLineFormat().getWidth();
129  if (lineWidth <= TechDraw::LineGroup::getDefaultWidth("Thin")) {
130  ui->rbThin->setChecked(true);
131  } else if (lineWidth <= TechDraw::LineGroup::getDefaultWidth("Graphic")) {
132  ui->rbMiddle->setChecked(true);
133  } else if (lineWidth <= TechDraw::LineGroup::getDefaultWidth("Thick")) {
134  ui->rbThick->setChecked(true);
135  } else {
136  ui->rbMiddle->setChecked(true);
137  }
138 
139  QColor lineColor = LineFormat::getCurrentLineFormat().getQColor();
140  ui->cbColor->setColor(lineColor);
141 
142  double cascadeSpacing = activeDimAttributes.getCascadeSpacing();
143  ui->sbSpacing->setValue(cascadeSpacing);
144  double lineStretching = activeDimAttributes.getLineStretch();
145  ui->sbStretch->setValue(lineStretching);
146 
147 }

149 bool TaskSelectLineAttributes::accept()
150 {
151  LineFormat::getCurrentLineFormat().setStyle(ui->cbLineStyle->currentIndex() + 1);
152  LineFormat::getCurrentLineFormat().setLineNumber(ui->cbLineStyle->currentIndex() + 1);
153 
154  if (ui->rbThin->isChecked()){
155  LineFormat::getCurrentLineFormat().setWidth(TechDraw::LineGroup::getDefaultWidth("Thin"));
156  }
157  else if (ui->rbMiddle->isChecked()){
158  LineFormat::getCurrentLineFormat().setWidth(TechDraw::LineGroup::getDefaultWidth("Graphic"));
159  }
160  else if (ui->rbThick->isChecked()){
161  LineFormat::getCurrentLineFormat().setWidth(TechDraw::LineGroup::getDefaultWidth("Thick"));
162  }
163  else {
164  LineFormat::getCurrentLineFormat().setWidth(TechDraw::LineGroup::getDefaultWidth("Graphic"));
165  }
166 
167  QColor qTemp = ui->cbColor->color();
168  App::Color temp;
169  temp.set(qTemp.redF(), qTemp.greenF(), qTemp.blueF(), 1.0 - qTemp.alphaF());
170  LineFormat::getCurrentLineFormat().setColor(temp);
171 
172  double cascadeSpacing = ui->sbSpacing->value();
173  activeDimAttributes.setCascadeSpacing(cascadeSpacing);
174  double lineStretching = ui->sbStretch->value();
175  activeDimAttributes.setLineStretch(lineStretching);
176 
177  return true;
178 }
