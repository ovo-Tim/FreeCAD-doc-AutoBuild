86 void ViewProviderImagePlane::attach(App::DocumentObject *pcObj)
87 {
88  ViewProviderGeometryObject::attach(pcObj);
89 
90  // NOTE: SoFCSelection node has beem removed because it led to
91  // problems using the image as a construction plane with the
92  // draft commands
93  SoSeparator* shading = new SoSeparator;
94  shading->addChild(pcCoords);
95 
96  SoTextureCoordinate2 *textCoord = new SoTextureCoordinate2;
97  textCoord->point.set1Value(0,0,0);
98  textCoord->point.set1Value(1,1,0);
99  textCoord->point.set1Value(2,1,1);
100  textCoord->point.set1Value(3,0,1);
101  shading->addChild(textCoord);
102 
103  // texture
104  texture->model = SoTexture2::MODULATE;
105  shading->addChild(texture);
106 
107  shading->addChild(shapeHints);
108  shading->addChild(pcShapeMaterial);
109 
110  // plane
111  pcCoords->point.set1Value(0,0,0,0);
112  pcCoords->point.set1Value(1,1,0,0);
113  pcCoords->point.set1Value(2,1,1,0);
114  pcCoords->point.set1Value(3,0,1,0);
115  SoFaceSet *faceset = new SoFaceSet;
116  faceset->numVertices.set1Value(0,4);
117  shading->addChild(faceset);
118 
119  addDisplayMaskMode(shading, "Shading");
120 
121  SoLightModel* lightmodel = new SoLightModel;
122  lightmodel->model = SoLightModel::BASE_COLOR;
123 
124  SoSeparator* noshading = new SoSeparator;
125  noshading->addChild(pcCoords);
126  noshading->addChild(textCoord);
127  noshading->addChild(texture);
128  noshading->addChild(shapeHints);
129  noshading->addChild(pcShapeMaterial);
130  noshading->addChild(lightmodel);
131  noshading->addChild(faceset);
132 
133  addDisplayMaskMode(noshading, "No shading");
134 }

199 void ViewProviderImagePlane::loadImage()
200 {
201  Image::ImagePlane* imagePlane = static_cast<Image::ImagePlane*>(pcObject);
202  std::string fileName = imagePlane->ImageFile.getValue();
203 
204  if (!fileName.empty()) {
205  QImage impQ;
206  if (isSvgFile(fileName.c_str())) {
207  impQ = loadSvg(fileName.c_str());
208  }
209  else {
210  impQ = loadRaster(fileName.c_str());
211  }
212 
213  QSizeF size = getSizeInMM(impQ);
214  setPlaneSize(size, impQ);
215  convertToSFImage(impQ);
216  }
217 }

219 void ViewProviderImagePlane::setPlaneSize(const QSizeF& size, const QImage& img)
220 {
221  if (!img.isNull()) {
222  Image::ImagePlane* imagePlane = static_cast<Image::ImagePlane*>(pcObject);
223  // When restoring the document or importing a ImagePlane by eg. pasting it
224  // preserve the X and Y size.
225  if (!isRestoring() && !pcObject->testStatus(App::ObjectStatus::ObjImporting)) {
226  imagePlane->XSize.setValue(size.width());
227  imagePlane->YSize.setValue(size.height());
228  }
229 
230  imagePlane->XPixelsPerMeter = img.dotsPerMeterX();
231  imagePlane->YPixelsPerMeter = img.dotsPerMeterY();
232  }
233 }

283 QSizeF ViewProviderImagePlane::getSizeInMM(const QImage& img) const
284 {
285  double xPixelsPerM = img.dotsPerMeterX();
286  double width = img.width();
287  width = width * 1000 / xPixelsPerM;
288 
289  double yPixelsPerM = img.dotsPerMeterY();
290  double height = img.height();
291  height = height * 1000 / yPixelsPerM;
292 
293  QSizeF sizef;
294  sizef.setWidth(width);
295  sizef.setHeight(height);
296  return sizef;
297 }

299 QSizeF ViewProviderImagePlane::pixelSize(const char* filename, const QSizeF& size) const
300 {
301  QImage tmp;
302  if (tmp.load(QString::fromUtf8(filename))) {
303  double xPixelsPerM = tmp.dotsPerMeterX();
304  double yPixelsPerM = tmp.dotsPerMeterY();
305 
306  double width = size.width();
307  double height = size.height();
308 
309  width = width * xPixelsPerM / 1000;
310  height = height * yPixelsPerM / 1000;
311 
312  return {width, height};
313  }
314 
315  return size;
316 }
