409 Int128 Int128Mul(long64 lhs, long64 rhs)
410 {
411  bool negate = (lhs < 0) != (rhs < 0);
412 
413  if (lhs < 0) {
414  lhs = -lhs;
415  }
416  ulong64 int1Hi = ulong64(lhs) >> 32;
417  ulong64 int1Lo = ulong64(lhs & 0xFFFFFFFF);
418 
419  if (rhs < 0) {
420  rhs = -rhs;
421  }
422  ulong64 int2Hi = ulong64(rhs) >> 32;
423  ulong64 int2Lo = ulong64(rhs & 0xFFFFFFFF);
424 
425  // nb: see comments in clipper.pas
426  ulong64 a = int1Hi * int2Hi;
427  ulong64 b = int1Lo * int2Lo;
428  ulong64 c = int1Hi * int2Lo + int1Lo * int2Hi;
429 
430  Int128 tmp;
431  tmp.hi = long64(a + (c >> 32));
432  tmp.lo = long64(c << 32);
433  tmp.lo += long64(b);
434  if (tmp.lo < b) {
435  tmp.hi++;
436  }
437  if (negate) {
438  tmp = -tmp;
439  }
440  return tmp;

505 int PointInPolygon(const IntPoint& pt, const Path& path)
506 {
507  // returns 0 if false, +1 if true, -1 if pt ON polygon boundary
508  // See "The Point in Polygon Problem for Arbitrary Polygons" by Hormann & Agathos
509  // http://citeseerx.ist.psu.edu/viewdoc/download?doi=10.1.1.88.5498&rep=rep1&type=pdf
510  int result = 0;
511  size_t cnt = path.size();
512  if (cnt < 3) {
513  return 0;
514  }
515  IntPoint ip = path[0];
516  for (size_t i = 1; i <= cnt; ++i) {
517  IntPoint ipNext = (i == cnt ? path[0] : path[i]);
518  if (ipNext.Y == pt.Y) {
519  if ((ipNext.X == pt.X) || (ip.Y == pt.Y && ((ipNext.X > pt.X) == (ip.X < pt.X)))) {
520  return -1;
521  }
522  }
523  if ((ip.Y < pt.Y) != (ipNext.Y < pt.Y)) {
524  if (ip.X >= pt.X) {
525  if (ipNext.X > pt.X) {
526  result = 1 - result;
527  }
528  else {
529  double d = (double)(ip.X - pt.X) * (ipNext.Y - pt.Y)
530  - (double)(ipNext.X - pt.X) * (ip.Y - pt.Y);
531  if (!d) {
532  return -1;
533  }
534  if ((d > 0) == (ipNext.Y > ip.Y)) {
535  result = 1 - result;
536  }
537  }
538  }
539  else {
540  if (ipNext.X > pt.X) {
541  double d = (double)(ip.X - pt.X) * (ipNext.Y - pt.Y)
542  - (double)(ipNext.X - pt.X) * (ip.Y - pt.Y);
543  if (!d) {
544  return -1;
545  }
546  if ((d > 0) == (ipNext.Y > ip.Y)) {
547  result = 1 - result;
548  }
549  }
550  }
551  }
552  ip = ipNext;
553  }
554  return result;
555 }

559 {
560  // returns 0 if false, +1 if true, -1 if pt ON polygon boundary
561  int result = 0;
562  OutPt* startOp = op;
563  for (;;) {
564  if (op->Next->Pt.Y == pt.Y) {
565  if ((op->Next->Pt.X == pt.X)
566  || (op->Pt.Y == pt.Y && ((op->Next->Pt.X > pt.X) == (op->Pt.X < pt.X)))) {
567  return -1;
568  }
569  }
570  if ((op->Pt.Y < pt.Y) != (op->Next->Pt.Y < pt.Y)) {
571  if (op->Pt.X >= pt.X) {
572  if (op->Next->Pt.X > pt.X) {
573  result = 1 - result;
574  }
575  else {
576  double d = (double)(op->Pt.X - pt.X) * (op->Next->Pt.Y - pt.Y)
577  - (double)(op->Next->Pt.X - pt.X) * (op->Pt.Y - pt.Y);
578  if (!d) {

584  }
585  }

712 void IntersectPoint(TEdge& Edge1, TEdge& Edge2, IntPoint& ip)
713 {
714 #ifdef use_xyz
715  ip.Z = 0;
716 #endif
717 
718  double b1, b2;
719  if (Edge1.Dx == Edge2.Dx) {
720  ip.Y = Edge1.Curr.Y;
721  ip.X = TopX(Edge1, ip.Y);
722  return;
723  }
724  else if (Edge1.Delta.X == 0) {
725  ip.X = Edge1.Bot.X;
726  if (IsHorizontal(Edge2)) {
727  ip.Y = Edge2.Bot.Y;
728  }
729  else {
730  b2 = Edge2.Bot.Y - (Edge2.Bot.X / Edge2.Dx);
731  ip.Y = Round(ip.X / Edge2.Dx + b2);
732  }
733  }
734  else if (Edge2.Delta.X == 0) {
735  ip.X = Edge2.Bot.X;
736  if (IsHorizontal(Edge1)) {
737  ip.Y = Edge1.Bot.Y;
738  }
739  else {
740  b1 = Edge1.Bot.Y - (Edge1.Bot.X / Edge1.Dx);
741  ip.Y = Round(ip.X / Edge1.Dx + b1);
742  }
743  }
744  else {
745  b1 = Edge1.Bot.X - Edge1.Bot.Y * Edge1.Dx;
746  b2 = Edge2.Bot.X - Edge2.Bot.Y * Edge2.Dx;
747  double q = (b2 - b1) / (Edge1.Dx - Edge2.Dx);
748  ip.Y = Round(q);
749  if (std::fabs(Edge1.Dx) < std::fabs(Edge2.Dx)) {
750  ip.X = Round(Edge1.Dx * q + b1);
751  }
752  else {
753  ip.X = Round(Edge2.Dx * q + b2);
754  }
755  }
756 
757  if (ip.Y < Edge1.Top.Y || ip.Y < Edge2.Top.Y) {
758  if (Edge1.Top.Y > Edge2.Top.Y) {
759  ip.Y = Edge1.Top.Y;
760  }
761  else {
762  ip.Y = Edge2.Top.Y;
763  }
764  if (std::fabs(Edge1.Dx) < std::fabs(Edge2.Dx)) {
765  ip.X = TopX(Edge1, ip.Y);
766  }
767  else {
768  ip.X = TopX(Edge2, ip.Y);

1238  --highI;

1764 void Clipper::SetWindingCount(TEdge& edge)
1765 {
1766  TEdge* e = edge.PrevInAEL;
1767  // find the edge of the same polytype that immediately precedes 'edge' in AEL
1768  while (e && ((e->PolyTyp != edge.PolyTyp) || (e->WindDelta == 0))) {
1769  e = e->PrevInAEL;
1770  }
1771  if (!e) {
1772  edge.WindCnt = (edge.WindDelta == 0 ? 1 : edge.WindDelta);
1773  edge.WindCnt2 = 0;
1774  e = m_ActiveEdges; // ie get ready to calc WindCnt2
1775  }
1776  else if (edge.WindDelta == 0 && m_ClipType != ctUnion) {
1777  edge.WindCnt = 1;
1778  edge.WindCnt2 = e->WindCnt2;
1779  e = e->NextInAEL; // ie get ready to calc WindCnt2
1780  }
1781  else if (IsEvenOddFillType(edge)) {
1782  // EvenOdd filling ...
1783  if (edge.WindDelta == 0) {
1784  // are we inside a subj polygon ...
1785  bool Inside = true;
1786  TEdge* e2 = e->PrevInAEL;
1787  while (e2) {
1788  if (e2->PolyTyp == e->PolyTyp && e2->WindDelta != 0) {
1789  Inside = !Inside;
1790  }
1791  e2 = e2->PrevInAEL;
1792  }
1793  edge.WindCnt = (Inside ? 0 : 1);
1794  }
1795  else {
1796  edge.WindCnt = edge.WindDelta;
1797  }
1798  edge.WindCnt2 = e->WindCnt2;
1799  e = e->NextInAEL; // ie get ready to calc WindCnt2
1800  }
1801  else {
1802  // nonZero, Positive or Negative filling ...
1803  if (e->WindCnt * e->WindDelta < 0) {
1804  // prev edge is 'decreasing' WindCount (WC) toward zero
1805  // so we're outside the previous polygon ...
1806  if (Abs(e->WindCnt) > 1) {
1807  // outside prev poly but still inside another.
1808  // when reversing direction of prev poly use the same WC
1809  if (e->WindDelta * edge.WindDelta < 0) {
1810  edge.WindCnt = e->WindCnt;
1811  }
1812  // otherwise continue to 'decrease' WC ...
1813  else {
1814  edge.WindCnt = e->WindCnt + edge.WindDelta;
1815  }
1816  }
1817  else {
1818  // now outside all polys of same polytype so set own WC ...
1819  edge.WindCnt = (edge.WindDelta == 0 ? 1 : edge.WindDelta);
1820  }
1821  }
1822  else {
1823  // prev edge is 'increasing' WindCount (WC) away from zero
1824  // so we're inside the previous polygon ...
1825  if (edge.WindDelta == 0) {
1826  edge.WindCnt = (e->WindCnt < 0 ? e->WindCnt - 1 : e->WindCnt + 1);
1827  }
1828  // if wind direction is reversing prev then use same WC
1829  else if (e->WindDelta * edge.WindDelta < 0) {
1830  edge.WindCnt = e->WindCnt;
1831  }
1832  // otherwise add to WC ...
1833  else {
1834  edge.WindCnt = e->WindCnt + edge.WindDelta;
1835  }
1836  }
1837  edge.WindCnt2 = e->WindCnt2;
1838  e = e->NextInAEL; // ie get ready to calc WindCnt2
1839  }
1840 
1841  // update WindCnt2 ...
1842  if (IsEvenOddAltFillType(edge)) {
1843  // EvenOdd filling ...
1844  while (e != &edge) {
1845  if (e->WindDelta != 0) {
1846  edge.WindCnt2 = (edge.WindCnt2 == 0 ? 1 : 0);
1847  }
1848  e = e->NextInAEL;
1849  }
1850  }
1851  else {
1852  // nonZero, Positive or Negative filling ...
1853  while (e != &edge) {
1854  edge.WindCnt2 += e->WindDelta;
1855  e = e->NextInAEL;
1856  }
1857  }
1858 }

4646 double DistanceFromLineSqrd(const IntPoint& pt, const IntPoint& ln1, const IntPoint& ln2)
4647 {
4648  // The equation of a line in general form (Ax + By + C = 0)
4649  // given 2 points (x¹,y¹) & (x²,y²) is ...
4650  //(y¹ - y²)x + (x² - x¹)y + (y² - y¹)x¹ - (x² - x¹)y¹ = 0
4651  // A = (y¹ - y²); B = (x² - x¹); C = (y² - y¹)x¹ - (x² - x¹)y¹
4652  // perpendicular distance of point (x³,y³) = (Ax³ + By³ + C)/Sqrt(A² + B²)
4653  // see http://en.wikipedia.org/wiki/Perpendicular_distance
4654  double A = double(ln1.Y - ln2.Y);
4655  double B = double(ln2.X - ln1.X);
4656  double C = A * ln1.X + B * ln1.Y;
4657  C = A * pt.X + B * pt.Y - C;
4658  return (C * C) / (A * A + B * B);
4659 }
