81  short int getOffsetTo(const App::Property* prop) const {
82  auto *pt = (const char*)prop;
83  auto *base = (const char *)m_container;
84  if(pt<base || pt>base+SHRT_MAX)
85  return -1;
86  return (short) (pt-base);
87  }
88  char* getOffset() const {return (char*) m_container;}

91  const void* m_container;
92  };
