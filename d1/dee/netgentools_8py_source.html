140  brep_file,

143  params,
144  second_order,

148  verbosity,

152 
153  shape = geom.shape
154  for items, l in mesh_region:
155  for t, n in items:
156  if t == "Vertex":
157  shape.vertices.vertices[n - 1].maxh = l
158  elif t == "Edge":
159  shape.edges.edges[n - 1].maxh = l
160  elif t == "Face":
161  shape.faces.faces[n - 1].maxh = l
162  elif t == "Solid":
163  shape.solids.solids[n - 1].maxh = l
164 
165  with ngcore.TaskManager():
166  meshing.SetMessageImportance(verbosity)
167  geom = occ.OCCGeometry(shape)
168  if heal:
169  geom.Heal()
170  mesh = geom.GenerateMesh(mp=meshing.MeshingParameters(**params))
171 
172  result = {{
173  "coords": [],
174  "Edges": [[], []],
175  "Faces": [[], []],
176  "Volumes": [[], []],
177  }}
178  groups = {{"Edges": [], "Faces": [], "Solids": []}}
179 
180  # save empty data if last step is geometry analysis
181  if params["perfstepsend"] == 1:
182  np.save(result_file, [result, groups])
183  return None
184 
185  if second_order:
186  if second_order_linear:
187  mesh.SetGeometry(None)
188  mesh.SecondOrder()
189 
190  coords = mesh.Coordinates()
191 
192  edges = mesh.Elements1D().NumPy()
193  faces = mesh.Elements2D().NumPy()
194  volumes = mesh.Elements3D().NumPy()
195 
196  nod_edges = edges["nodes"]
197  nod_faces = faces["nodes"]
198  nod_volumes = volumes["nodes"]
199 
200  np_edges = (nod_edges != 0).sum(axis=1).tolist()
201  np_faces = faces["np"].tolist()
202  np_volumes = volumes["np"].tolist()
203 
204  # set smesh node order
205  for i in range(faces.size):
206  nod_faces[i] = nod_faces[i][order_face[np_faces[i]]]
207 
208  for i in range(volumes.size):
209  nod_volumes[i] = nod_volumes[i][order_volume[np_volumes[i]]]
210 
211  flat_edges = nod_edges[nod_edges != 0].tolist()
212  flat_faces = nod_faces[nod_faces != 0].tolist()
213  flat_volumes = nod_volumes[nod_volumes != 0].tolist()
214 
215  result = {{
216  "coords": coords,
217  "Edges": [flat_edges, np_edges],
218  "Faces": [flat_faces, np_faces],
219  "Volumes": [flat_volumes, np_volumes],
220  }}
221 
222  # create groups
223  nb_edges = edges.size
224  nb_faces = faces.size
225  nb_volumes = volumes.size
226 
227  idx_edges = edges["index"]
228  idx_faces = faces["index"]
229  idx_volumes = volumes["index"]
230 
231  for i in np.unique(idx_edges):
232  edge_i = (np.nonzero(idx_edges == i)[0] + 1).tolist()
233  groups["Edges"].append([i, edge_i])
234  for i in np.unique(idx_faces):
235  face_i = (np.nonzero(idx_faces == i)[0] + nb_edges + 1).tolist()
236  groups["Faces"].append([i, face_i])
237 
238  for i in np.unique(idx_volumes):
239  volume_i = (np.nonzero(idx_volumes == i)[0] + nb_edges + nb_faces + 1).tolist()
240  groups["Solids"].append([i, volume_i])
241 
242  np.save(result_file, [result, groups])
243 
244 run_netgen(**{kwds})
