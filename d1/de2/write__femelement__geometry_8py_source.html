29 def write_femelement_geometry(f, ccxwriter):
30 
31  # floats read from ccx should use {:.13G}, see comment in writer module
32 
33  f.write("\n{}\n".format(59 * "*"))
34  f.write("** Sections\n")
35  for matgeoset in ccxwriter.mat_geo_sets:
36  if matgeoset["ccx_elset"]:
37  elsetdef = "ELSET={}, ".format(matgeoset["ccx_elset_name"])
38  material = "MATERIAL={}".format(matgeoset["mat_obj_name"])
39 
40  if "beamsection_obj" in matgeoset: # beam mesh
41  beamsec_obj = matgeoset["beamsection_obj"]
42  beam_axis_m = matgeoset["beam_axis_m"]
43  # in CalxuliX called the 1direction
44  # see meshtools.get_beam_main_axis_m(beam_direction, defined_angle)
45  section_nor = "{:.13G}, {:.13G}, {:.13G}\n".format(
46  beam_axis_m[0], beam_axis_m[1], beam_axis_m[2]
47  )
48  if beamsec_obj.SectionType == "Rectangular":
49  # see meshtools.get_beam_main_axis_m(beam_direction, defined_angle)
50  # the method get_beam_main_axis_m() which calculates the beam_axis_m vector
51  # unless rotated, this vector points towards +y axis
52  # doesn't follow 1,2-direction order of CalculiX
53  # ^ (n, 2-direction)
54  # |
55  # |
56  # .----> (m, 1-direction)
57  #
58  len_beam_axis_n = beamsec_obj.RectHeight.getValueAs("mm").Value
59  len_beam_axis_m = beamsec_obj.RectWidth.getValueAs("mm").Value
60  section_type = ", SECTION=RECT"
61  section_geo = f"{len_beam_axis_m:.13G},{len_beam_axis_n:.13G}\n"
62  section_def = f"*BEAM SECTION, {elsetdef}{material}{section_type}\n"
63  elif beamsec_obj.SectionType == "Circular":
64  diameter = beamsec_obj.CircDiameter.getValueAs("mm").Value
65  section_type = ", SECTION=CIRC"
66  section_geo = f"{diameter:.13G}\n"
67  section_def = f"*BEAM SECTION, {elsetdef}{material}{section_type}\n"
68  elif beamsec_obj.SectionType == "Elliptical":
69  axis1 = beamsec_obj.Axis1Length.getValueAs("mm").Value
70  axis2 = beamsec_obj.Axis2Length.getValueAs("mm").Value
71  section_type = ", SECTION=CIRC"
72  section_geo = f"{axis1:.13G},{axis2:.13G}\n"
73  section_def = f"*BEAM SECTION, {elsetdef}{material}{section_type}\n"
74  elif beamsec_obj.SectionType == "Pipe":
75  radius = 0.5 * beamsec_obj.PipeDiameter.getValueAs("mm").Value
76  thickness = beamsec_obj.PipeThickness.getValueAs("mm").Value
77  section_type = ", SECTION=PIPE"
78  section_geo = f"{radius:.13G},{thickness:.13G}\n"
79  section_def = f"*BEAM SECTION, {elsetdef}{material}{section_type}\n"
80  elif beamsec_obj.SectionType == "Box":
81  box_width = beamsec_obj.BoxWidth.getValueAs("mm").Value
82  box_height = beamsec_obj.BoxHeight.getValueAs("mm").Value
83  box_t1 = beamsec_obj.BoxT1.getValueAs("mm").Value
84  box_t2 = beamsec_obj.BoxT2.getValueAs("mm").Value
85  box_t3 = beamsec_obj.BoxT3.getValueAs("mm").Value
86  box_t4 = beamsec_obj.BoxT4.getValueAs("mm").Value
87  section_type = ", SECTION=BOX"
88  section_geo = f"{box_width:.13G},{box_height:.13G},{box_t1:.13G},{box_t2:.13G},{box_t3:.13G},{box_t4:.13G}\n"
89  section_def = f"*BEAM SECTION, {elsetdef}{material}{section_type}\n"
90 
91  f.write(section_def)
92  f.write(section_geo)
93  f.write(section_nor)
94  elif "fluidsection_obj" in matgeoset: # fluid mesh
95  fluidsec_obj = matgeoset["fluidsection_obj"]
96  if fluidsec_obj.SectionType == "Liquid":
97  section_type = fluidsec_obj.LiquidSectionType
98  if (section_type == "PIPE INLET") or (section_type == "PIPE OUTLET"):
99  section_type = "PIPE INOUT"
100  section_def = "*FLUID SECTION, {}TYPE={}, {}\n".format(
101  elsetdef, section_type, material
102  )
103  section_geo = liquid_section_def(fluidsec_obj, section_type)
104  """
105  # deactivate as it would result in section_def and section_geo not defined
106  # deactivated in the App and Gui object and thus in the task panel as well
107  elif fluidsec_obj.SectionType == "Gas":
108  section_type = fluidsec_obj.GasSectionType
109  elif fluidsec_obj.SectionType == "Open Channel":
110  section_type = fluidsec_obj.ChannelSectionType
111  """
112  f.write(section_def)
113  f.write(section_geo)
114  elif "shellthickness_obj" in matgeoset: # shell mesh
115  shellth_obj = matgeoset["shellthickness_obj"]
116  if ccxwriter.solver_obj.ModelSpace == "3D":
117  section_def = f"*SHELL SECTION, {elsetdef}{material}\n"
118  else:
119  section_def = f"*SOLID SECTION, {elsetdef}{material}\n"
120  thickness = shellth_obj.Thickness.getValueAs("mm").Value
121  section_geo = f"{thickness:.13G}\n"
122  f.write(section_def)
123  f.write(section_geo)
124  else: # solid mesh
125  section_def = f"*SOLID SECTION, {elsetdef}{material}\n"
126  f.write(section_def)

131 def liquid_section_def(obj, section_type):
132  if section_type == "PIPE MANNING":
133  manning_area = obj.ManningArea.getValueAs("mm^2").Value
134  manning_radius = obj.ManningRadius.getValueAs("mm").Value
135  manning_coefficient = obj.ManningCoefficient
136  section_geo = "{:.13G},{:.13G},{:.13G}\n".format(
137  manning_area, manning_radius, manning_coefficient
138  )
139  return section_geo
140  elif section_type == "PIPE ENLARGEMENT":
141  enlarge_area1 = obj.EnlargeArea1.getValueAs("mm^2").Value
142  enlarge_area2 = obj.EnlargeArea2.getValueAs("mm^2").Value
143  section_geo = f"{enlarge_area1:.13G},{enlarge_area2:.13G}\n"
144  return section_geo
145  elif section_type == "PIPE CONTRACTION":
146  contract_area1 = obj.ContractArea1.getValueAs("mm^2").Value
147  contract_area2 = obj.ContractArea2.getValueAs("mm^2").Value
148  section_geo = f"{contract_area1:.13G},{contract_area2:.13G}\n"
149  return section_geo
150  elif section_type == "PIPE ENTRANCE":
151  entrance_pipe_area = obj.EntrancePipeArea.getValueAs("mm^2").Value
152  entrance_area = obj.EntranceArea.getValueAs("mm^2").Value
153  section_geo = f"{entrance_pipe_area:.13G},{entrance_area:.13G}\n"
154  return section_geo
155  elif section_type == "PIPE DIAPHRAGM":
156  diaphragm_pipe_area = obj.DiaphragmPipeArea.getValueAs("mm^2").Value
157  diaphragm_area = obj.DiaphragmArea.getValueAs("mm^2").Value
158  section_geo = f"{diaphragm_pipe_area:.13G},{diaphragm_area:.13G}\n"
159  return section_geo
160  elif section_type == "PIPE BEND":
161  bend_pipe_area = obj.BendPipeArea.getValueAs("mm^2").Value
162  bend_radius_diameter = obj.BendRadiusDiameter
163  bend_angle = obj.BendAngle
164  bend_loss_coefficient = obj.BendLossCoefficient
165  section_geo = "{:.13G},{:.13G},{:.13G},{:.13G}\n".format(
166  bend_pipe_area, bend_radius_diameter, bend_angle, bend_loss_coefficient
167  )
168  return section_geo
169  elif section_type == "PIPE GATE VALVE":
170  gatevalve_pipe_area = obj.GateValvePipeArea.getValueAs("mm^2").Value
171  gatevalve_closing_coeff = obj.GateValveClosingCoeff
172  section_geo = f"{gatevalve_pipe_area:.13G},{gatevalve_closing_coeff:.13G}\n"
173  return section_geo
174  elif section_type == "PIPE WHITE-COLEBROOK":
175  colebrooke_area = obj.ColebrookeArea.getValueAs("mm^2").Value
176  colebrooke_diameter = 2 * obj.ColebrookeRadius.getValueAs("mm")
177  colebrooke_grain_diameter = obj.ColebrookeGrainDiameter.getValueAs("mm")
178  colebrooke_form_factor = obj.ColebrookeFormFactor
179  section_geo = "{:.13G},{:.13G},{},{:.13G},{:.13G}\n".format(
180  colebrooke_area,
181  colebrooke_diameter,
182  "-1",
183  colebrooke_grain_diameter,
184  colebrooke_form_factor,
185  )
186  return section_geo
187  elif section_type == "LIQUID PUMP":
188  section_geo = ""
189  for i in range(len(obj.PumpFlowRate)):
190  flow_rate = obj.PumpFlowRate[i]
191  top = obj.PumpHeadLoss[i]
192  section_geo = f"{section_geo + flow_rate:.13G},{top:.13G},\n"
193  section_geo = f"{section_geo}\n"
194  return section_geo
195  else:
196  return ""
