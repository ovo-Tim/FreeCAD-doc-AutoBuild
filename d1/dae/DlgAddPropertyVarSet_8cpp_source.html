70 void DlgAddPropertyVarSet::initializeGroup()
71 {
72  comboBoxGroup.setObjectName(QString::fromUtf8("comboBoxGroup"));
73  comboBoxGroup.setInsertPolicy(QComboBox::InsertAtTop);
74  comboBoxGroup.setSizePolicy(QSizePolicy::Preferred, QSizePolicy::Preferred);
75  auto formLayout = qobject_cast<QFormLayout*>(layout());
76  formLayout->setWidget(1, QFormLayout::FieldRole, &comboBoxGroup);
77 
78  std::vector<App::Property*> properties;
79  varSet->getPropertyList(properties);
80  std::unordered_set<std::string> groupNames;
81  for (auto prop : properties) {
82  const char* groupName = varSet->getPropertyGroup(prop);
83  groupNames.insert(groupName ? groupName : GROUP_BASE);
84  }
85  std::vector<std::string> groupNamesSorted(groupNames.begin(), groupNames.end());
86  std::sort(groupNamesSorted.begin(), groupNamesSorted.end(), [](std::string& a, std::string& b) {
87  // prefer anything else other than Base, so move it to the back
88  if (a == GROUP_BASE) {
89  return false;
90  }
91  else if (b == GROUP_BASE) {
92  return true;
93  }
94  else {
95  return a < b;
96  }
97  });
98 
99  for (const auto& groupName : groupNamesSorted) {
100  comboBoxGroup.addItem(QString::fromStdString(groupName));
101  }
102 
103  comboBoxGroup.setEditText(QString::fromStdString(groupNamesSorted[0]));
104  connComboBoxGroup = connect(&comboBoxGroup, &EditFinishedComboBox::editFinished,
105  this, &DlgAddPropertyVarSet::onEditFinished);
106 }

108 void DlgAddPropertyVarSet::getSupportedTypes(std::vector<Base::Type>& types)
109 {
110  std::vector<Base::Type> proptypes;
111  Base::Type::getAllDerivedFrom(Base::Type::fromName("App::Property"), proptypes);
112  std::copy_if(proptypes.begin(), proptypes.end(), std::back_inserter(types), [](const Base::Type& type) {
113  return type.canInstantiate();
114  });
115  std::sort(types.begin(), types.end(), [](Base::Type a, Base::Type b) {
116  return strcmp(a.getName(), b.getName()) < 0;
117  });
118 }

272 void DlgAddPropertyVarSet::createProperty()
273 {
274  std::string name = ui->lineEditName->text().toStdString();
275  std::string group = comboBoxGroup.currentText().toStdString();
276  std::string type = ui->comboBoxType->currentText().toStdString();
277  std::string doc = ui->lineEditToolTip->text().toStdString();
278 
279  App::Property* prop;
280  try {
281  prop = varSet->addDynamicProperty(type.c_str(), name.c_str(),
282  group.c_str(), doc.c_str());
283  }
284  catch (Base::Exception& e) {
285  e.ReportException();
286  critical(QObject::tr("Add property"),
287  QObject::tr("Failed to add property to '%1': %2").arg(
288  QString::fromLatin1(varSet->getFullName().c_str()),
289  QString::fromUtf8(e.what())));
290  clearEditors();
291  return;
292  }
293 
294  namePropertyToAdd = name;
295 
296  objectIdentifier = std::make_unique<App::ObjectIdentifier>(*prop);
297  // creating a propertyItem here because it has all kinds of logic for
298  // editors that we can reuse
299  removeEditor();
300  propertyItem.reset(createPropertyItem(prop));
301  if (propertyItem && isTypeWithEditor(type)) {
302  propertyItem->setPropertyData({prop});
303  propertyItem->bind(*objectIdentifier);
304  addEditor(propertyItem.get(), type);
305  }
306 
307  setOkEnabled(true);
308 }

321 void DlgAddPropertyVarSet::changePropertyToAdd() {
322  // we were already adding a new property, the only option to get here
323  // is a change of type or group.
324 
325  std::string name = ui->lineEditName->text().toStdString();
326  assert(name == namePropertyToAdd);
327 
328  // performs a check for nullptr
329  App::Property* prop = getPropertyToAdd();
330 
331  std::string group = comboBoxGroup.currentText().toStdString();
332  std::string doc = ui->lineEditToolTip->text().toStdString();
333  if (prop->getGroup() != group) {
334  varSet->changeDynamicProperty(prop, group.c_str(), doc.c_str());
335  }
336 
337  std::string type = ui->comboBoxType->currentText().toStdString();
338  if (prop->getTypeId() != Base::Type::fromName(type.c_str())) {
339  // the property should have a different type
340  varSet->removeDynamicProperty(namePropertyToAdd.c_str());
341  createProperty();
342  }
343 }

397 void DlgAddPropertyVarSet::checkName() {
398  std::string name = ui->lineEditName->text().toStdString();
399  if(name.empty() || name != Base::Tools::getIdentifier(name)) {
400  QMessageBox::critical(getMainWindow(),
401  QObject::tr("Invalid name"),
402  QObject::tr("The property name must only contain alpha numericals, "
403  "underscore, and must not start with a digit."));
404  clearEditors(!CLEAR_NAME);
405  throw CreatePropertyException("Invalid name");
406  }
407 
408  if(App::ExpressionParser::isTokenAUnit(name) || App::ExpressionParser::isTokenAConstant(name)) {
409  critical(QObject::tr("Invalid name"),
410  QObject::tr("The property name is a reserved word."));
411  clearEditors(!CLEAR_NAME);
412  throw CreatePropertyException("Invalid name");
413  }
414 
415  if (namePropertyToAdd.empty()) {
416  // we are adding a new property, check whether it doesn't already exist
417  auto prop = varSet->getPropertyByName(name.c_str());
418  if(prop && prop->getContainer() == varSet) {
419  critical(QObject::tr("Invalid name"),
420  QObject::tr("The property '%1' already exists in '%2'").arg(
421  QString::fromLatin1(name.c_str()),
422  QString::fromLatin1(varSet->getFullName().c_str())));
423  clearEditors(!CLEAR_NAME);
424  throw CreatePropertyException("Invalid name");
425  }
426  }
427 }

449 void DlgAddPropertyVarSet::onEditFinished() {
450  /* The editor for the value is dynamically created if 1) the name has been
451  * determined and 2) if the type of the property has been determined. The
452  * group of the property is important too, but it is not essential, because
453  * we can change the group after the property has been created.
454  *
455  * In this function we check whether we can create a property and therefore
456  * an editor.
457  */
458 
459  try {
460  checkName();
461  checkGroup();
462  checkType();
463  // no check for tooltip, we accept any string
464  }
465  catch (const CreatePropertyException&) {
466  if (!namePropertyToAdd.empty()) {
467  clearCurrentProperty();
468  }
469  return;
470  }
471 
472  if (namePropertyToAdd.empty()) {
473  // we are adding a new property
474  openTransaction();
475  createProperty();
476  }
477  else {
478  // we were already adding a new property that should now be changed
479  changePropertyToAdd();
480  }
481 }

549 void DlgAddPropertyVarSet::reject()
550 {
551  // On reject we can disconnect the signal handlers because nothing useful
552  // is to be done. Otherwise, signals may activate the handlers that assume
553  // that a new property has been created, an assumption that will be
554  // violated by aborting the transaction because it will remove the newly
555  // created property.
556  disconnect(connComboBoxGroup);
557  disconnect(connComboBoxType);
558  disconnect(connLineEditNameEditFinished);
559  disconnect(connLineEditNameTextChanged);
560 
561  // a transaction is not pending if a name has not been determined.
562  if (hasPendingTransaction()) {
563  closeTransaction(ABORT);
564  }
565  QDialog::reject();
566 }
