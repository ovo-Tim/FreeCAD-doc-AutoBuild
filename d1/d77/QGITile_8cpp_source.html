119 void QGITile::makeText()
120 {
121  prepareGeometryChange();
122  double verticalFudge = 0.10;
123 
124  m_qgTextL->setFont(m_font);
125  m_qgTextL->setPlainText(m_textL);
126  m_qgTextL->setColor(m_colCurrent);
127  double textWidth = m_qgTextL->boundingRect().width();
128  double charWidth = textWidth / m_textL.size(); //not good for non-ASCII chars
129  double hMargin = 1;
130  if (!m_textL.isEmpty()) {
131  hMargin = (m_wide / 2.0) + (charWidth * 1.5);
132  }
133 
134  double vertAdjust = 0.0;
135  double minVertAdjust = PreferencesGui::labelFontSizePX() * 0.1;
136  if (m_font.pixelSize() > m_high) {
137  vertAdjust = ((m_font.pixelSize() - m_high) / 2.0) + minVertAdjust;
138  }
139 
140  double textHeightL = m_qgTextL->boundingRect().height();
141  double vOffset = 0.0;
142  if (m_row < 0) { // below line
143  vOffset = textHeightL * verticalFudge;
144  } else { // above line
145  vOffset = -vertAdjust;
146  }
147  m_qgTextL->justifyRightAt(-hMargin, vOffset, true); //sb vCentered at 0.5 m_high
148 
149  m_qgTextR->setFont(m_font);
150  m_qgTextR->setPlainText(m_textR);
151  m_qgTextR->setColor(m_colCurrent);
152  textWidth = m_qgTextR->boundingRect().width();
153  charWidth = textWidth / m_textR.size();
154  hMargin = 1;
155  if (!m_textR.isEmpty()) {
156  hMargin = (m_wide / 2) + (charWidth);
157  }
158  double textHeightR = m_qgTextR->boundingRect().height();
159  if (m_row < 0) { // below line
160  vOffset = textHeightR * verticalFudge;
161  } else {
162  vOffset = -vertAdjust;
163  }
164  m_qgTextR->justifyLeftAt(hMargin, vOffset, true);
165 
166  m_qgTextC->setFont(m_font);
167  m_qgTextC->setPlainText(m_textC);
168  m_qgTextC->setColor(m_colCurrent);
169  double textHeightC = m_qgTextC->boundingRect().height();
170  if (m_row < 0) { // below line
171  vOffset = m_high * (1 + verticalFudge);
172  } else {
173  vOffset = -0.5 * (m_high + textHeightC);
174  }
175  m_qgTextC->centerAt(0.0, vOffset);
176 }

330 QPointF QGITile::calcTilePosition()
331 {
332  constexpr double OneHalf{0.5};
333  auto xDir = m_leaderXDirection; // "right"
334  auto yDir = m_leaderYDirection; // "up"
335  Base::Vector3d stdX{1, 0, 0};
336  auto dot = stdX.Dot(xDir);
337  if (dot < 0) {
338  // our leader points left, so yDir should be +Y, but we need -Y up
339  yDir = -yDir;
340  }
341 
342  double textWidthL = m_qgTextL->boundingRect().width();
343  double textWidthR = m_qgTextR->boundingRect().width();
344  double totalWidth = m_wide + textWidthL + textWidthR;
345 
346  if (m_row == 0) { //arrowSide
347  double offsetDistanceY = getSymbolHeight() * OneHalf;
348  auto vOffset = yDir * offsetDistanceY;
349  auto netPosition = m_origin + DU::toQPointF(vOffset);
350  return netPosition;
351  }
352 
353  if (m_row == -1) { //otherSide
354  if (getAltWeld()) {
355  if (isTailRight()) {
356  auto hOffset = DU::toQPointF(xDir * (OneHalf * totalWidth)); //move to right 1/2 tile width
357  auto vOffset = DU::toQPointF(yDir * (getSymbolHeight() * OneHalf)); // move down 1/2 tile height
358  auto netPosition = m_origin + hOffset - vOffset;
359  return netPosition;
360  }
361  auto hOffset = DU::toQPointF(xDir * (OneHalf * totalWidth));
362  auto vOffset = DU::toQPointF(yDir * (OneHalf * getSymbolHeight()));
363  auto netPosition = m_origin - hOffset - vOffset; // left and down
364  return netPosition;
365  }
366 
367  auto vOffset = DU::toQPointF(yDir * (OneHalf * getSymbolHeight())); // down 1/2 tile height
368  auto netPosition = m_origin - vOffset;
369  return netPosition;
370  }
371 
372  auto hOffset = DU::toQPointF(xDir * (m_col * totalWidth));
373  auto vOffset = DU::toQPointF(yDir * ((m_row * m_high) + (m_high * OneHalf)));
374  auto netPosition = m_origin + hOffset - vOffset;
375 
376  return netPosition;
377 }
