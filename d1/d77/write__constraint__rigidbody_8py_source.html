64 def write_constraint(f, femobj, rb_obj, ccxwriter):
65 
66  rb_obj_idx = ccxwriter.analysis.Group.index(rb_obj)
67  node_count = ccxwriter.mesh_object.FemMesh.NodeCount
68  # factor 2 is to prevent conflict with other rigid body constraint
69  ref_node_idx = node_count + 2 * rb_obj_idx + 1
70  rot_node_idx = node_count + 2 * rb_obj_idx + 2
71 
72  f.write("*NODE\n")
73  f.write("{},{},{},{}\n".format(ref_node_idx, *rb_obj.ReferenceNode))
74  f.write("{},{},{},{}\n".format(rot_node_idx, *rb_obj.ReferenceNode))
75 
76  f.write(f"*NSET,NSET={rb_obj.Name}_RefNode\n")
77  f.write(f"{ref_node_idx},\n")
78  f.write(f"*NSET,NSET={rb_obj.Name}_RotNode\n")
79  f.write(f"{rot_node_idx},\n")
80 
81  kw_line = f"*RIGID BODY, NSET={rb_obj.Name}, REF NODE={ref_node_idx}, ROT NODE={rot_node_idx}"
82 
83  f.write(kw_line + "\n")
