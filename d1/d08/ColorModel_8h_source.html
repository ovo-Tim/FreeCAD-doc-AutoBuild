354 inline Color ColorField::getColor(float fVal) const
355 {
356  // if the value is outside or at the border of the range
357  std::size_t ct = colorModel.getCountColors() - 1;
358  if (fVal <= fMin) {
359  return colorModel.colors[0];
360  }
361  else if (fVal >= fMax) {
362  return colorModel.colors[ct];
363  }
364 
365  // get the color field position (with 0 < t < 1)
366  float t = (fVal - fMin) / (fMax - fMin);
367  Color col(1.0f, 1.0f, 1.0f); // white as default
368  for (std::size_t i = 0; i < ct; i++) {
369  float r = (float)(i + 1) / (float)ct;
370  if (t < r) {
371  // calculate the exact position in the subrange
372  float s = t * float(ct) - float(i);
373  Color c1 = colorModel.colors[i];
374  Color c2 = colorModel.colors[i + 1];
375  col.r = (1.0f - s) * c1.r + s * c2.r;
376  col.g = (1.0f - s) * c1.g + s * c2.g;
377  col.b = (1.0f - s) * c1.b + s * c2.b;
378  break;
379  }
380  }
381 
382  return col;
383 }
