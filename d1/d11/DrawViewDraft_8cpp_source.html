80 App::DocumentObjectExecReturn *DrawViewDraft::execute()
81 {
82 // Base::Console().Message("DVDr::execute() \n");
83  if (!keepUpdated()) {
84  return App::DocumentObject::StdReturn;
85  }
86 
87  App::DocumentObject* sourceObj = Source.getValue();
88  if (sourceObj) {
89  std::string svgFrag;
90  std::string svgHead = getSVGHead();
91  std::string svgTail = getSVGTail();
92  std::string FeatName = getNameInDocument();
93  std::string SourceName = sourceObj->getNameInDocument();
94  // Draft.get_svg(obj, scale=1, linewidth=0.35, fontsize=12, fillstyle="shape color", direction=None, linestyle=None, color=None, linespacing=None, techdraw=False)
95 
96  std::stringstream paramStr;
97  App::Color col = Color.getValue();
98  paramStr << ", scale=" << getScale()
99  << ", linewidth=" << LineWidth.getValue()
100  << ", fontsize=" << FontSize.getValue()
101  // TODO treat fillstyle here
102  << ", direction=FreeCAD.Vector(" << Direction.getValue().x << ", " << Direction.getValue().y << ", " << Direction.getValue().z << ")"
103  << ", linestyle=\"" << LineStyle.getValue() << "\""
104  << ", color=\"" << col.asHexString() << "\""
105  << ", linespacing=" << LineSpacing.getValue()
106  // We must set techdraw to "true" becausea couple of things behave differently than in Drawing
107  << ", techdraw=True"
108  << ", override=" << (OverrideStyle.getValue() ? "True" : "False");
109 
110 // this is ok for a starting point, but should eventually make dedicated Draft functions that build the svg for all the special cases
111 // (Arch section, etc)
112 // like Draft.makeDrawingView, but we don't need to create the actual document objects in Draft, just the svg.
113  Base::Interpreter().runString("import Draft");
114  Base::Interpreter().runStringArg("svgBody = Draft.get_svg(App.activeDocument().%s %s)",
115  SourceName.c_str(), paramStr.str().c_str());
116 // Base::Interpreter().runString("print svgBody");
117  Base::Interpreter().runStringArg("App.activeDocument().%s.Symbol = '%s' + svgBody + '%s'",
118  FeatName.c_str(), svgHead.c_str(), svgTail.c_str());
119  }
120 
121  overrideKeepUpdated(false);
122  return DrawView::execute();
123 }
