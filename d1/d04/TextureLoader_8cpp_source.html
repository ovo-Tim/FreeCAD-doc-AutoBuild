50  std::vector<std::string> fileNames,
51  int textureSize)
52  : mImageFolder(imgFolder)
53 {

56  if (mRawData == nullptr) {

66 // parse compressed image into a texture buffer
67 bool TextureLoader::AddImage(TextureItem* texItem,
68  QImage& pixmap,
69  unsigned int* buffPos,
70  int stride)
71 {
72  int width = pixmap.width();
73  int height = pixmap.height();
74  buffPos += stride * texItem->ty + texItem->tx;
75  for (int i = 0; i < height; i++) {
76  unsigned int* line = reinterpret_cast<unsigned int*>(pixmap.scanLine(i));
77  for (int j = 0; j < width; j++) {
78  buffPos[j] = line[j];
79  }
80  buffPos += stride;
81  }
82  texItem->w = width;
83  texItem->h = height;
84  return true;
85 }
