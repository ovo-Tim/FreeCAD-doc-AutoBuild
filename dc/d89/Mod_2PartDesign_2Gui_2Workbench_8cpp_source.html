78 void Workbench::setupContextMenu(const char* recipient, Gui::MenuItem* item) const
79 {
80  auto selection = Gui::Selection().getSelection();
81  // Add move Tip Command
82  if (!selection.empty()) {
83  App::DocumentObject *feature = selection.front().pObject;
84  PartDesign::Body *body = nullptr;
85 
86  body = PartDesignGui::getBodyFor (feature, false, false, true);
87  // lote of assertion so feature should be marked as a tip
88  if ( selection.size() == 1 && feature && body && (
89  feature->isDerivedFrom<PartDesign::Feature>() ||
90  ( feature->isDerivedFrom<Part::Feature>() &&
91  body->BaseFeature.getValue() == feature )
92  ) ) {
93  *item << "PartDesign_MoveTip";
94  }
95 
96  if (strcmp(recipient, "Tree") == 0) {
97  Gui::MDIView *activeView = Gui::Application::Instance->activeView();
98 
99  if (activeView ) {
100  if (feature && feature->isDerivedFrom<PartDesign::Body>()){
101  *item << "Std_ToggleFreeze";
102  }
103 
104  if (activeView->getAppDocument()->countObjectsOfType<PartDesign::Body>() > 0) {
105  bool addMoveFeature = true;
106  bool addMoveFeatureInTree = (body != nullptr);
107  for (auto sel : selection) {
108  // if at least one selected feature cannot be moved to a body
109  // disable the entry
110  if ( addMoveFeature && !PartDesign::Body::isAllowed ( sel.pObject ) ) {
111  addMoveFeature = false;
112  }
113  // if all at least one selected feature doesn't belong to the same body
114  // disable the menu entry
115  if ( addMoveFeatureInTree && !body->hasObject ( sel.pObject ) ) {
116  addMoveFeatureInTree = false;
117  }
118 
119  if ( !addMoveFeatureInTree && !addMoveFeature ) {
120  break;
121  }
122  }
123  if (addMoveFeature) {
124  *item << "PartDesign_MoveFeature";
125  }
126  if (addMoveFeatureInTree) {
127  *item << "PartDesign_MoveFeatureInTree";
128  }
129  }
130  }
131 
132  if (Gui::Selection().countObjectsOfType<PartDesign::Transformed>() -
133  Gui::Selection().countObjectsOfType<PartDesign::MultiTransform>() == 1 )
134  *item << "PartDesign_MultiTransform";
135 
136  if (Gui::Selection().countObjectsOfType<App::DocumentObject>() > 0) {
137  *item << "Std_Placement"
138  << "Std_ToggleVisibility"
139  << "Std_ShowSelection"
140  << "Std_HideSelection"
141  << "Std_ToggleSelectability"
142  << "Std_TreeSelectAllInstances"
143  << "Separator"
144  << "Std_RandomColor"
145  << "Std_ToggleTransparency"
146  << "Std_Cut"
147  << "Std_Copy"
148  << "Std_Paste"
149  << "Std_Delete"
150  << "Std_SendToPythonConsole"
151  << "Separator";
152  }
153  }
154  }
155 
156  if (strcmp(recipient, "View") == 0) {
157  if (item->hasItems())
158  *item << "Separator";
159  Gui::StdWorkbench::setupContextMenu(recipient, item);
160  }
161 }
