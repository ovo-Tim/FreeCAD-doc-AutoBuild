157  // but can't do this since otherwise it will complain about almost any

159  //surfaceFormat.setMajorVersion(3);
160  //surfaceFormat.setMinorVersion(2);
161  //surfaceFormat.setProfile(QSurfaceFormat::CoreProfile);
162 #if defined (_DEBUG) && 0
163  surfaceFormat.setOption(QSurfaceFormat::DebugContext);
164 #endif

169  void initializeGL() override
170  {
171 #if defined (_DEBUG) && 0
172  QOpenGLContext *context = QOpenGLContext::currentContext();
173  if (context && context->hasExtension(QByteArrayLiteral("GL_KHR_debug"))) {
174  QOpenGLDebugLogger *logger = new QOpenGLDebugLogger(this);
175  connect(logger, &QOpenGLDebugLogger::messageLogged, this, &CustomGLWidget::handleLoggedMessage);
176 
177  if (logger->initialize())
178  logger->startLogging(QOpenGLDebugLogger::SynchronousLogging);
179  }
180 #endif
181 
182  connect(this, &CustomGLWidget::resized, this, &CustomGLWidget::slotResized);
183  }
184  // paintGL() is invoked when e.g. using the method grabFramebuffer of this class
185  // \code
186  // from PySide import QtWidgets
187  // mw = Gui.getMainWindow()
188  // mdi = mw.findChild(QtWidgets.QMdiArea)
189  // gl = mdi.findChild(QtWidgets.QOpenGLWidget)
190  // img = gl.grabFramebuffer()
191  // \endcode
192  void paintGL() override
193  {
194  QuarterWidget* qw = qobject_cast<QuarterWidget*>(parentWidget());
195  if (qw) {
196  qw->redraw();
197  }
198  }

200  bool event(QEvent *e) override
201  {
202  // If a debug logger is activated then Qt's default implementation
203  // first releases the context before stopping the logger. However,
204  // the logger needs the active context and thus crashes because it's
205  // null.
206  if (e->type() == QEvent::WindowChangeInternal) {
207  if (!qApp->testAttribute(Qt::AA_ShareOpenGLContexts)) {
208  QOpenGLDebugLogger* logger = this->findChild<QOpenGLDebugLogger*>();

646  PRIVATE(this)->soeventmanager->setSceneGraph(superscene);
647  PRIVATE(this)->sorendermanager->setSceneGraph(superscene);
648 
649  if (viewall) { this->viewAll(); }
650  if (superscene) { superscene->touch(); }

839 void QuarterWidget::paintEvent(QPaintEvent* event)
840 {
841  if (updateDevicePixelRatio()) {
842  qreal dev_pix_ratio = devicePixelRatio();
843  int width = static_cast<int>(dev_pix_ratio * this->width());
844  int height = static_cast<int>(dev_pix_ratio * this->height());
845  SbViewportRegion vp(width, height);
846  PRIVATE(this)->sorendermanager->setViewportRegion(vp);
847  PRIVATE(this)->soeventmanager->setViewportRegion(vp);
848  }
849 
850  if(!initialized) {
851  this->getSoRenderManager()->reinitialize();
852  initialized = true;
853  }
854 
855  getSoRenderManager()->activate();
856 
857  glMatrixMode(GL_PROJECTION);
858 
859  QtGLWidget* w = static_cast<QtGLWidget*>(this->viewport());
860  if (!w->isValid()) {
861  qWarning() << "No valid GL context found!";
862  return;
863  }
864  //assert(w->isValid() && "No valid GL context found!");
865  // We might have to process the delay queue here since we don't know
866  // if paintGL() is called from Qt, and we might have some sensors
867  // waiting to trigger (the redraw sensor has a lower priority than a
868  // normal field sensor to guarantee that your sensor is processed
869  // before the next redraw). Disable autorendering while we do this
870  // to avoid recursive redraws.
871 
872  // We set the PRIVATE(this)->processdelayqueue = false in redraw()
873  // to avoid processing the delay queue when paintGL() is triggered
874  // by us, and we don't want to process the delay queue in those
875  // cases
876 
877  PRIVATE(this)->autoredrawenabled = false;
878 
879  if(PRIVATE(this)->processdelayqueue && SoDB::getSensorManager()->isDelaySensorPending()) {
880  // processing the sensors might trigger a redraw in another
881  // context. Release this context temporarily
882  w->doneCurrent();
883  SoDB::getSensorManager()->processDelayQueue(false);
884  w->makeCurrent();
885  }
886 
887  assert(w->isValid() && "No valid GL context found!");
888 
889  // Causes an OpenGL error on resize
890  //glDrawBuffer(w->format().swapBehavior() == QSurfaceFormat::DoubleBuffer ? GL_BACK : GL_FRONT);
891 
892  w->makeCurrent();
893  this->actualRedraw();
894 
895  //start the standard graphics view processing for all widgets and graphic items. As
896  //QGraphicsView initaliizes a QPainter which changes the Opengl context in an unpredictable
897  //manner we need to store the context and recreate it after Qt is done.
898  glPushAttrib(GL_MULTISAMPLE_BIT_EXT);
899  inherited::paintEvent(event);
900  glPopAttrib();

913 bool QuarterWidget::viewportEvent(QEvent* event)
914 {
915 
916  // If no item is selected still let the graphics scene handle it but
917  // additionally handle it by this viewer. This is e.g. needed when
918  // resizing a widget item because the cursor may already be outside
919  // this widget.
920  if (event->type() == QEvent::MouseButtonDblClick ||
921  event->type() == QEvent::MouseButtonPress) {
922  QMouseEvent* mouse = static_cast<QMouseEvent*>(event);

924  if (!item) {
925  QGraphicsView::viewportEvent(event);
926  return false;
927  }
928  }
929  else if (event->type() == QEvent::MouseMove ||
930  event->type() == QEvent::MouseButtonRelease) {
931  QGraphicsScene* glScene = this->scene();
932  if (!(glScene && glScene->mouseGrabberItem())) {
933  QGraphicsView::viewportEvent(event);
934  return false;
935  }
936  }
937  else if (event->type() == QEvent::Wheel) {
938  auto wheel = static_cast<QWheelEvent*>(event);
939 #if QT_VERSION < QT_VERSION_CHECK(5, 15, 0)
940  QPoint pos = wheel->pos();
941 #else
942  QPoint pos = wheel->position().toPoint();
943 #endif
944  QGraphicsItem* item = itemAt(pos);
945  if (!item) {
946  QGraphicsView::viewportEvent(event);
947  return false;
948  }
949  }

962 void
963 QuarterWidget::redraw()
964 {
965  // we're triggering the next paintGL(). Set a flag to remember this
966  // to avoid that we process the delay queue in paintGL()
967  PRIVATE(this)->processdelayqueue = false;
968 
969  // When stylesheet is used, there is recursive repaint warning caused by
970  // repaint() here. It happens when switching active documents. Based on call
971  // stacks, it happens like this, the repaint event first triggers a series
972  // calls of QWidgetPrivate::paintSiblingsRecrusive(), and then reaches one of
973  // the QuarterWidget. From its paintEvent(), it calls
974  // SoSensorManager::processDelayQueue(), which triggers redraw() of another
975  // QuarterWidget. And if repaint() is called here, it will trigger another
976  // series call of QWidgetPrivate::paintSiblingRecursive(), and eventually
977  // back to the first QuarterWidget, at which time the "Recursive repaint
978  // detected" Qt warning message will be printed.
979  //
980  // Note that, the recursive repaint is not infinite due to setting
981  // 'processdelayqueue = false' above. However, it does cause annoying
982  // flickering, and actually crash on Windows.
983  this->viewport()->update();
984 }

1203 void
1204 QuarterWidget::setNavigationModeFile(const QUrl & url)
1205 {
1206  QString filename;
1207 
1208  if (url.scheme()=="coin") {
1209  filename = url.path();
1210 
1211  //Workaround for differences between url scheme, and Coin internal
1212  //scheme in Coin 3.0.
1213  if (filename[0]=='/') {
1214  filename.remove(0,1);
1215  }
1216 
1217  filename = url.scheme()+':'+filename;
1218  }
1219  else if (url.scheme()=="file")
1220  filename = url.toLocalFile();
1221  else if (url.isEmpty()) {
1222  if (PRIVATE(this)->currentStateMachine) {
1223  this->removeStateMachine(PRIVATE(this)->currentStateMachine);
1224  delete PRIVATE(this)->currentStateMachine;
1225  PRIVATE(this)->currentStateMachine = nullptr;
1226  PRIVATE(this)->navigationModeFile = url;
1227  }
1228  return;
1229  }
1230  else {
1231  qDebug()<<url.scheme()<<"is not recognized";
1232  return;
1233  }
1234 
1235  QByteArray filenametmp = filename.toLocal8Bit();
1236  ScXMLStateMachine * stateMachine = nullptr;
1237 
1238  if (filenametmp.startsWith("coin:")){
1239  stateMachine = ScXML::readFile(filenametmp.data());
1240  }
1241  else {
1242  //Use Qt to read the file in case it is a Qt resource
1243  QFile file(filenametmp);
1244  if (file.open(QIODevice::ReadOnly)){
1245  QByteArray fileContents = file.readAll();
1246 #if COIN_MAJOR_VERSION >= 4
1247  stateMachine = ScXML::readBuffer(SbByteBuffer(fileContents.size(), fileContents.constData()));
1248 #else
1249  stateMachine = ScXML::readBuffer(fileContents.constData());
1250 #endif
1251  file.close();
1252  }
1253  }
1254 
1255  if (stateMachine &&
1256  stateMachine->isOfType(SoScXMLStateMachine::getClassTypeId())) {
1257  SoScXMLStateMachine * newsm =
1258  static_cast<SoScXMLStateMachine *>(stateMachine);
1259  if (PRIVATE(this)->currentStateMachine) {
1260  this->removeStateMachine(PRIVATE(this)->currentStateMachine);
1261  delete PRIVATE(this)->currentStateMachine;
1262  }
1263  this->addStateMachine(newsm);
1264  newsm->initialize();
1265  PRIVATE(this)->currentStateMachine = newsm;
1266  }
1267  else {
1268  delete stateMachine;
1269  stateMachine = nullptr;
1270  qDebug()<<filename;
1271  qDebug()<<"Unable to load"<<url;
1272  return;
1273  }
1274 
1275  //If we have gotten this far, we have successfully loaded the
1276  //navigation file, so we set the property
1277  PRIVATE(this)->navigationModeFile = url;
1278 
1279  if (QUrl(DEFAULT_NAVIGATIONFILE) == PRIVATE(this)->navigationModeFile ) {
1280 
1281  // set up default cursors for the examiner navigation states
1282  //FIXME: It may be overly restrictive to not do this for arbitrary
1283  //navigation systems? - BFG 20090117
1284  setupDefaultCursors();
1285  }
1286 }
