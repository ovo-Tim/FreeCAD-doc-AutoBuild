106 TechDraw::LineFormat *TaskLineDecor::getFormatAccessPtr(const std::string &edgeName, std::string *newFormatTag)
107 {
108  BaseGeomPtr bg = m_partFeat->getEdge(edgeName);
109  if (bg) {
110  if (bg->getCosmetic()) {
111  if (bg->source() == SourceType::COSEDGE) {
112  TechDraw::CosmeticEdge *ce = m_partFeat->getCosmeticEdgeBySelection(edgeName);
113  if (ce) {
114  return &ce->m_format;
115  }
116  }
117  else if (bg->source() == SourceType::CENTERLINE) {
118  TechDraw::CenterLine *cl = m_partFeat->getCenterLineBySelection(edgeName);
119  if (cl) {
120  return &cl->m_format;
121  }
122  }
123  }
124  else {
125  TechDraw::GeomFormat *gf = m_partFeat->getGeomFormatBySelection(edgeName);
126  if (gf) {
127  return &gf->m_format;
128  }
129  else {
130  ViewProviderViewPart *viewPart = dynamic_cast<ViewProviderViewPart *>(QGIView::getViewProvider(m_partFeat));
131  if (viewPart) {
132  TechDraw::LineFormat lineFormat(Qt::SolidLine, viewPart->LineWidth.getValue(), LineFormat::getDefEdgeColor(), true);
133  TechDraw::GeomFormat geomFormat(DrawUtil::getIndexFromName(edgeName), lineFormat);
134 
135  std::string formatTag = m_partFeat->addGeomFormat(&geomFormat);
136  if (newFormatTag) {
137  *newFormatTag = formatTag;
138  }
139 
140  return &m_partFeat->getGeomFormat(formatTag)->m_format;
141  }
142  }
143  }
144  }
145  return {};
146 }

148 void TaskLineDecor::initializeRejectArrays()
149 {
150  m_originalFormats.resize(m_edges.size());
151  m_createdFormatTags.resize(m_edges.size());
152 
153  for (size_t i = 0; i < m_edges.size(); ++i) {
154  std::string newTag;
155  TechDraw::LineFormat *accessPtr = getFormatAccessPtr(m_edges[i], &newTag);
156 
157  if (accessPtr) {
158  m_originalFormats[i] = *accessPtr;
159  if (!newTag.empty()) {
160  m_createdFormatTags[i] = newTag;
161  }
162  }
163  }
164 }

214 void TaskLineDecor::applyDecorations()
215 {

217  for (auto& e: m_edges) {

248 bool TaskLineDecor::reject()
249 {
250 // Base::Console().Message("TLD::reject()\n");
251  Gui::Document* doc = Gui::Application::Instance->getDocument(m_partFeat->getDocument());
252  if (!doc)
253  return false;
254 
255  for (size_t i = 0; i < m_originalFormats.size(); ++i) {
256  std::string &formatTag = m_createdFormatTags[i];
257  if (formatTag.empty()) {
258  LineFormat *lf = getFormatAccessPtr(m_edges[i]);
259  if (lf) {
260  *lf = m_originalFormats[i];
261  }
262  }
263  else {
264  m_partFeat->removeGeomFormat(formatTag);
265  }
266  }
267 
268  m_partFeat->requestPaint();
269 
270  Gui::Command::doCommand(Gui::Command::Gui, "Gui.ActiveDocument.resetEdit()");
271  return false;
272 }
