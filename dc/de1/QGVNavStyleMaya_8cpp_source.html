66 void QGVNavStyleMaya::handleMouseMoveEvent(QMouseEvent *event)
67 {
68  if (getViewer()->isBalloonPlacing()) {
69  balloonCursorMovement(event);
70  return;
71  }
72 
73  //pan mode alt + MMB + mouse movement
74  if (QGuiApplication::mouseButtons() & Qt::MiddleButton &&
75  QApplication::keyboardModifiers().testFlag(Qt::AltModifier)) {
76  if (panningActive) {
77  pan(event->pos());
78  } else {
79  startPan(event->pos());
80  }
81  event->accept();
82  }
83 
84  //zoom mode 2 ALT + RMB
85  if (QGuiApplication::mouseButtons() & Qt::RightButton &&
86  QApplication::keyboardModifiers().testFlag(Qt::AltModifier)) {
87  if (zoomingActive) {
88  zoom(mouseZoomFactor(event->pos()));
89  } else {
90  startZoom(event->pos());
91  }
92  event->accept();
93  }
94 }

109

111  //zoom mode 2 ALT + RMB

113  zoomingActive = false;
114  event->accept();
115  }
