328 Base::BoundBox3d SectionCut::collectObjects()
329 {
330  Base::BoundBox3d BoundCompound;
331  if (doc->getObject(BoxXName) || doc->getObject(BoxYName) || doc->getObject(BoxZName)) {
332 
333  // automatic coloring must be disabled
334  setAutoColoringChecked(false);
335 
336  // get the object with the right name
337  if (auto compoundObject = doc->getObject(CompoundName)) {
338  // to later store the childs
339  std::vector<App::DocumentObject*> compoundChilds;
340 
341  // check if this is a BooleanFragments or a Part::Compound
342  // Part::Compound is the case when there was only one object
343  auto pcCompoundPart = dynamic_cast<Part::Compound*>(compoundObject);
344  auto pcPartFeature = dynamic_cast<Part::Feature*>(compoundObject);
345  if (!pcCompoundPart && pcPartFeature) {
346  // for more security check for validity accessing its ViewProvider
347  auto pcCompoundBF = Gui::Application::Instance->getViewProvider(pcPartFeature);
348  compoundChilds = pcCompoundBF->claimChildren();
349  BoundCompound = pcPartFeature->Shape.getBoundingBox();
350 
351  auto pcCompoundBFGO = dynamic_cast<Gui::ViewProviderGeometryObject*>(pcCompoundBF);
352  initBooleanFragmentControls(pcCompoundBFGO);
353  }
354  else if (pcCompoundPart) {
355  BoundCompound = pcCompoundPart->Shape.getBoundingBox();
356  pcCompoundPart->Links.getLinks(compoundChilds);
357  }
358 
359  collectAndShowLinks(compoundChilds);
360  }
361  }
362 
363  return BoundCompound;

480 void SectionCut::deleteObejcts()
481 {
482  App::DocumentObject* anObject = nullptr;
483 
484  // lambda function to delete objects
485  auto deleteObject = [&](const char* objectName) {
486  anObject = doc->getObject(objectName);
487  // the deleted object might have been visible before, thus check and delete it from the list
488  auto found = std::find_if(
489  ObjectsListVisible.begin(), ObjectsListVisible.end(),
490  [anObject](const App::DocumentObjectT &obj) { return (obj.getObject() == anObject);
491  });
492  if (found != ObjectsListVisible.end()) {
493  ObjectsListVisible.erase(found);
494  }
495  doc->removeObject(objectName);
496  };
497 
498  int compoundTransparency = -1;
499  // lambda to store the compoundTransparency
500  auto storeTransparency = [&](App::DocumentObject* cutObject) {
501  auto CompoundVP = dynamic_cast<Gui::ViewProviderGeometryObject*>(
502  Gui::Application::Instance->getViewProvider(cutObject));
503  if (CompoundVP && compoundTransparency == -1) {
504  compoundTransparency = int(CompoundVP->Transparency.getValue());
505  }
506  };
507 
508  // delete the objects we might have already created to cut
509  // we must do this because we support several cuts at once and
510  // it is dangerous to deal with the fact that the user is free
511  // to uncheck cutting planes and to add/remove objects while this dialog is open
512  // We must remove in this order because the tree hierary of the features is Z->Y->X and Cut->Box
513  if (doc->getObject(CutZName)) {
514  // the topmost cut transparency determines the overall transparency
515  storeTransparency(doc->getObject(CutZName));
516  deleteObject(CutZName);
517  }
518  if (doc->getObject(BoxZName)) {
519  deleteObject(BoxZName);
520  }
521  if (doc->getObject(CutYName)) {
522  storeTransparency(doc->getObject(CutYName));
523  deleteObject(CutYName);
524  }
525  if (doc->getObject(BoxYName)) {
526  deleteObject(BoxYName);
527  }
528  if (doc->getObject(CutXName)) {
529  storeTransparency(doc->getObject(CutXName));
530  deleteObject(CutXName);
531  }
532  if (doc->getObject(BoxXName)) {
533  deleteObject(BoxXName);
534  }
535 }

537 void SectionCut::deleteCompound()
538 {
539  // get the object with the right name
540  if (auto compoundObject = doc->getObject(CompoundName)) {
541  App::DocumentObject* anObject = compoundObject;
542  // to later store the childs
543  std::vector<App::DocumentObject*> compoundChilds;
544 
545  // check if this is a BooleanFragments or a Part::Compound
546  auto pcCompoundDelPart = dynamic_cast<Part::Compound*>(compoundObject);
547  Gui::ViewProvider* pcCompoundDelBF{};
548  if (!pcCompoundDelPart) {
549  // check for BooleanFragments
550  pcCompoundDelBF = Gui::Application::Instance->getViewProvider(compoundObject);
551  if (!pcCompoundDelBF) {
552  Base::Console().Error(
553  "SectionCut error: compound is incorrectly named, cannot proceed\n");
554  return;
555  }
556  compoundChilds = pcCompoundDelBF->claimChildren();
557  }
558  else {
559  pcCompoundDelPart->Links.getLinks(compoundChilds);
560  }
561 
562  // first delete the compound
563  auto foundObj = std::find_if(
564  ObjectsListVisible.begin(), ObjectsListVisible.end(),
565  [anObject](const App::DocumentObjectT &obj) { return (obj.getObject() == anObject);
566  });
567  if (foundObj != ObjectsListVisible.end()) {
568  ObjectsListVisible.erase(foundObj);
569  }
570  doc->removeObject(CompoundName);
571  // now delete the objects that have been part of the compound
572  for (auto aChild : compoundChilds) {
573  anObject = doc->getObject(aChild->getNameInDocument());
574  auto foundObjInner = std::find_if(ObjectsListVisible.begin(), ObjectsListVisible.end(),
575  [anObject](const App::DocumentObjectT &objInner) {
576  return (objInner.getObject() == anObject);
577  });
578  if (foundObjInner != ObjectsListVisible.end()) {
579  ObjectsListVisible.erase((foundObjInner));
580  }
581  doc->removeObject(aChild->getNameInDocument());
582  }
583  }
584 }

836 bool SectionCut::findObjects(std::vector<App::DocumentObject*>& objects)
837 {
838  bool isLinkAssembly = false;
839  for (auto& aVisObject : ObjectsListVisible) {
840  App::DocumentObject* object = aVisObject.getObject();
841  if (!object) {
842  continue;
843  }
844  // we need all Link objects in App::Part for example for Assembly 4
845  if (auto pcPart = dynamic_cast<App::Part*>(object)) {
846  // collect all its link objects
847  auto groupObjects = pcPart->Group.getValue();
848  for (auto aGroupObject : groupObjects) {
849  if (aGroupObject->getTypeId() == Base::Type::fromName("App::Link")) {
850  objects.push_back(aGroupObject);
851  // we assume that App::Links inside a App::Part are an assembly
852  isLinkAssembly = true;
853  }
854  }
855  }
856  // get all shapes that are also Part::Features
857  if (object->getPropertyByName("Shape") != nullptr
858  && object->getTypeId().isDerivedFrom(
859  Base::Type::fromName("Part::Feature"))) {
860  // sort out 2D objects, datums, App:Parts, compounds and objects that are
861  // part of a PartDesign body
862  if (!object->getTypeId().isDerivedFrom(
863  Base::Type::fromName("Part::Part2DObject"))
864  && !object->getTypeId().isDerivedFrom(
865  Base::Type::fromName("Part::Datum"))
866  && !object->getTypeId().isDerivedFrom(
867  Base::Type::fromName("PartDesign::Feature"))
868  && !object->getTypeId().isDerivedFrom(
869  Base::Type::fromName("Part::Compound"))
870  && object->getTypeId() != Base::Type::fromName("App::Part")) {
871  objects.push_back(object);
872  }
873  }

875  if (auto pcLink = dynamic_cast<App::Link*>(object)) {
876  auto linkedObject = doc->getObject(pcLink->LinkedObject.getObjectName());
877  if (linkedObject != nullptr
878  && linkedObject->getTypeId().isDerivedFrom(Base::Type::fromName("Part::Feature"))) {
