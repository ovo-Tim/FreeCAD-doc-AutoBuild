55 void DlgPrefsTechDrawDimensionsImp::saveSettings()
56 {
57  ui->pcbStandardAndStyle->onSave();
58  ui->cbGlobalDecimals->onSave();
59  ui->cbShowUnits->onSave();
60  ui->sbAltDecimals->onSave();
61  ui->plsb_FontSize->onSave();
62  ui->pdsbToleranceScale->onSave();
63  ui->leDiameter->onSave();
64  ui->pcbArrow->onSave();
65  ui->plsb_ArrowSize->onSave();
66  ui->leFormatSpec->onSave();
67  ui->pdsbGapISO->onSave();
68  ui->pdsbGapASME->onSave();
69  ui->pdsbLineSpacingFactorISO->onSave();
70 
71  enum
72  {
73  DimensionSingleTool,
74  DimensionSeparateTools,
75  DimensionBoth
76  };
77 
78  // Dimensioning constraints mode
79  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
80  "User parameter:BaseApp/Preferences/Mod/TechDraw/dimensioning");
81  bool singleTool = true;
82  bool SeparatedTools = false;
83  int index = ui->dimensioningMode->currentIndex();
84  switch (index) {
85  case DimensionSeparateTools:
86  singleTool = false;
87  SeparatedTools = true;
88  break;
89  case DimensionBoth:
90  singleTool = true;
91  SeparatedTools = true;
92  break;
93  }
94  hGrp->SetBool("SingleDimensioningTool", singleTool);
95  hGrp->SetBool("SeparatedDimensioningTools", SeparatedTools);
96 
97  ui->radiusDiameterMode->setEnabled(index != 1);
98 
99  enum
100  {
101  DimensionAutoRadiusDiam,
102  DimensionDiameter,
103  DimensionRadius
104  };
105 
106  bool Diameter = true;
107  bool Radius = true;
108  index = ui->radiusDiameterMode->currentIndex();
109  switch (index) {
110  case DimensionDiameter:
111  Diameter = true;
112  Radius = false;
113  break;
114  case DimensionRadius:
115  Diameter = false;
116  Radius = true;
117  break;
118  }
119  hGrp->SetBool("DimensioningDiameter", Diameter);
120  hGrp->SetBool("DimensioningRadius", Radius);
121 
122  if (property("dimensioningMode").toInt() != ui->dimensioningMode->currentIndex()) {
123  requireRestart();
124  }
125 }

127 void DlgPrefsTechDrawDimensionsImp::loadSettings()
128 {
129  //set defaults for Quantity widgets if property not found
130  //Quantity widgets do not use preset value since they are based on
131  //QAbstractSpinBox
132  double fontDefault = Preferences::dimFontSizeMM();
133  double arrowDefault = Preferences::dimArrowSize();
134  ui->plsb_FontSize->setValue(fontDefault);
135 // double arrowDefault = 5.0;
136 // plsb_ArrowSize->setValue(arrowDefault);
137  ui->plsb_ArrowSize->setValue(arrowDefault);
138 
139  ui->pcbStandardAndStyle->onRestore();
140  ui->cbGlobalDecimals->onRestore();
141  ui->cbShowUnits->onRestore();
142  ui->sbAltDecimals->onRestore();
143  ui->plsb_FontSize->onRestore();
144  ui->pdsbToleranceScale->onRestore();
145  ui->leDiameter->onRestore();
146  ui->pcbArrow->onRestore();
147  ui->plsb_ArrowSize->onRestore();
148 
149  DrawGuiUtil::loadArrowBox(ui->pcbArrow);
150  ui->pcbArrow->setCurrentIndex(prefArrowStyle());
151 
152  ui->leFormatSpec->setText(QString::fromStdString(Preferences::formatSpec()));
153  ui->leFormatSpec->onRestore();
154 
155  ui->pdsbGapISO->onRestore();
156  ui->pdsbGapASME->onRestore();
157  ui->pdsbLineSpacingFactorISO->onRestore();
158 
159 
160  // Dimensioning constraints mode
161  ui->dimensioningMode->clear();
162  ui->dimensioningMode->addItem(tr("Single tool"));
163  ui->dimensioningMode->addItem(tr("Separated tools"));
164  ui->dimensioningMode->addItem(tr("Both"));
165 
166  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
167  "User parameter:BaseApp/Preferences/Mod/TechDraw/dimensioning");
168  bool singleTool = hGrp->GetBool("SingleDimensioningTool", true);
169  bool SeparatedTools = hGrp->GetBool("SeparatedDimensioningTools", false);
170  int index = SeparatedTools ? (singleTool ? 2 : 1) : 0;
171  ui->dimensioningMode->setCurrentIndex(index);
172  setProperty("dimensioningMode", index);
173  connect(ui->dimensioningMode,
174  QOverload<int>::of(&QComboBox::currentIndexChanged),
175  this,
176  &DlgPrefsTechDrawDimensionsImp::dimensioningModeChanged);
177 
178  ui->radiusDiameterMode->setEnabled(index != 1);
179 
180  // Dimensioning constraints mode
181  ui->radiusDiameterMode->clear();
182  ui->radiusDiameterMode->addItem(tr("Auto"));
183  ui->radiusDiameterMode->addItem(tr("Diameter"));
184  ui->radiusDiameterMode->addItem(tr("Radius"));
185 
186  bool Diameter = hGrp->GetBool("DimensioningDiameter", true);
187  bool Radius = hGrp->GetBool("DimensioningRadius", true);
188  index = Diameter ? (Radius ? 0 : 1) : 2;
189  ui->radiusDiameterMode->setCurrentIndex(index);
190 }
