36  def getNudgeValue(self, mode):
37  "mode can be dist, up, down, left, right. dist returns a float in mm, other modes return a 3D vector"
38 
39  from PySide import QtCore, QtGui
40 
41  mw = FreeCADGui.getMainWindow()
42  if mw:
43  st = mw.statusBar()
44  statuswidget = st.findChild(QtGui.QToolBar, "BIMStatusWidget")
45  if statuswidget:
46  nudgeValue = statuswidget.nudge.text().replace("&", "")
47  dist = 0
48  if "auto" in nudgeValue.lower():
49  unit = FreeCAD.ParamGet(
50  "User parameter:BaseApp/Preferences/Units"
51  ).GetInt("UserSchema", 0)
52  if unit in [2, 3, 5, 7]:
53  scale = [1.5875, 3.175, 6.35, 25.4, 152.4, 304.8]
54  else:
55  scale = [1, 5, 10, 50, 100, 500]
56  viewsize = (
57  FreeCADGui.ActiveDocument.ActiveView.getCameraNode()
58  .getViewVolume()
59  .getWidth()
60  )
61  if viewsize < 250:
62  dist = scale[0]
63  elif viewsize < 750:
64  dist = scale[1]
65  elif viewsize < 4500:
66  dist = scale[2]
67  elif viewsize < 8000:
68  dist = scale[3]
69  elif viewsize < 25000:
70  dist = scale[4]
71  else:
72  dist = scale[5]
73  # u = FreeCAD.Units.Quantity(dist,FreeCAD.Units.Length).UserString
74  statuswidget.nudge.setText(translate("BIM", "Auto"))
75  else:
76  try:
77  dist = FreeCAD.Units.Quantity(nudgeValue)
78  except ValueError:
79  try:
80  dist = float(nudgeValue)
81  except ValueError:
82  return None
83  else:
84  dist = dist.Value
85  if not dist:
86  return None
87  if mode == "dist":
88  return dist
89  elif mode == "up":
90  return FreeCAD.Vector(FreeCAD.DraftWorkingPlane.v).multiply(dist)
91  elif mode == "down":
92  return (
93  FreeCAD.Vector(FreeCAD.DraftWorkingPlane.v)
94  .negative()
95  .multiply(dist)
96  )
97  elif mode == "right":
98  return FreeCAD.Vector(FreeCAD.DraftWorkingPlane.u).multiply(dist)
99  elif mode == "left":
100  return (
101  FreeCAD.Vector(FreeCAD.DraftWorkingPlane.u)
102  .negative()
103  .multiply(dist)
104  )
105  return None

107  def toStr(self, objs):
108  "builds a string which is a list of objects"
109 
110  return (
111  "[" + ",".join(["FreeCAD.ActiveDocument." + obj.Name for obj in objs]) + "]"
112  )

114  def getCenter(self, objs):
115  "returns the center point of a group of objects"
116 
117  bb = None
118  for obj in objs:
119  if hasattr(obj, "Shape") and hasattr(obj.Shape, "BoundBox"):
120  if not bb:
121  bb = obj.Shape.BoundBox
122  else:
123  bb.add(obj.Shape.BoundBox)
124  if bb:
125  return bb.Center
126  else:
127  return None
