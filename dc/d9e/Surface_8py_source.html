1736  def _stepTransitionCmds(self, obj, p1, p2, safePDC, tolrnc):
1737  """Generate transition commands / paths between two dropcutter steps or
1738  passes, as well as other kinds of breaks. When
1739  OptimizeStepOverTransitions is enabled, uses safePDC to safely optimize
1740  short (~order of cutter diameter) transitions."""
1741  cmds = []
1742  rtpd = False
1743  height = obj.SafeHeight.Value
1744  # Allow cutter-down transitions with a distance up to 2x cutter
1745  # diameter. We might be able to extend this further to the
1746  # full-retract-and-rapid break even point in the future, but this will
1747  # require a safeSTL that has all non-cut surfaces raised sufficiently
1748  # to avoid inadvertent cutting.
1749  maxXYDistanceSqrd = (self.cutter.getDiameter() * 2) ** 2
1750 
1751  if obj.OptimizeStepOverTransitions:
1752  if p1 and p2:
1753  # Short distance within step over
1754  xyDistanceSqrd = (p1.x - p2.x) ** 2 + (p1.y - p2.y) ** 2
1755  # Try to keep cutting for short distances.
1756  if xyDistanceSqrd <= maxXYDistanceSqrd:
1757  # Try to keep cutting, following the model shape
1758  (transLine, minZ, maxZ) = self._getTransitionLine(safePDC, p1, p2, obj)
1759  # For now, only optimize moderate deviations in Z direction, and
1760  # no dropping below the min of p1 and p2, primarily for multi
1761  # layer path safety.
1762  zFloor = min(p1.z, p2.z)
1763  if abs(minZ - maxZ) < self.cutter.getDiameter():
1764  for pt in transLine[1:-1]:
1765  cmds.append(
1766  Path.Command(
1767  "G1",
1768  {
1769  "X": pt.x,
1770  "Y": pt.y,

1773  "F": self.horizFeed,
1774  },
1775  )

1782  )
1783  )
1784  return cmds

1786  # to SafeHeight for lack of an accurate stock model, but then
1787  # speed up the drop back down when using multi pass, dropping
1788  # quickly to *previous* layer depth.
1789  stepDown = obj.StepDown.Value if hasattr(obj, "StepDown") else 0
1790  rtpd = min(height, p2.z + stepDown + 2)
