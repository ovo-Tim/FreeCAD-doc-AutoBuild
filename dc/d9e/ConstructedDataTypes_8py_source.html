48 class ENUMERATION(object):
49  """
50  EXPRESS definition:
51  ===================
52  An ENUMERATION data type has as its domain an ordered set of names. The names represent
53  values of the enumeration data type.
54 
55  Python implementation:
56  ======================
57  An enumeration is initialized from strings defining the types.
58  For instance, some EXPRESS definition:
59  TYPE ahead_or_behind = ENUMERATION OF
60  (ahead,
61  behind);
62  END_TYPE; -- ahead_or_behind
63 
64  is implemented in python with the line:
65  >>> ahead_of_behind = ENUMERATION('ahead','behind', the_current_scope)
66  >>> ahead_or_behind.ahead
67  >>> ahead_of_behind.behind
68 
69  And, if and only if ahead and/or behind are not in scope (e.g. they are not entity names,
70  and/or many enums define the same enumeration identifier):
71  >>> ahead
72  >>> behind
73  """
74 
75  def __init__(self, *kargs, **args):
76  # first defining the scope
77  if "scope" in args:
78  self._scope = args["scope"]
79  else:
80  self._scope = None
81  # store passed enum identifiers
82  self._enum_id_names = list(kargs)
83  self._enum_ids = []
84  # we create enums id from names, and create attributes
85  # for instance, from the identifier name 'ahead',
86  # we create an attribute ahead with which is a new
87  # instance of EnumerationId
88  for enum_id_name in self._enum_id_names:
89  setattr(self, enum_id_name, EnumerationId())
90  # we store this new attributes to the enum_ids list, which
91  # will be accessed by the type checker with the get_enum_ids method
92  self._enum_ids.append(self.__getattribute__(enum_id_name))
93  #
94  # Then we check if the enums names can be added to the current scope:
95  # if the name is already in the scope, then another enums id or select
96  # has the same name -> we do nothing, enums will be called
97  # with ahead_of_behind.ahead or ahead_or_behind.behind.
98  # otherwise, they can be called as only ahead or behind
99  # Note: since ENUMERATIONS are defined *before* entities, if an entity
100  # has the same name as an enum id, it will replace it in the current scope.
101  #
102  for enum_id_name in self._enum_id_names:
103  if enum_id_name not in vars(self._scope):
104  vars(self._scope)[enum_id_name] = self.__getattribute__(enum_id_name)
105 
106  def get_enum_ids(self):
107  return self._enum_ids

110 class SELECT(object):
111  """A select data type has as its domain the union of the domains of the named data types in
112  its select list. The select data type is a generalization of each of the named data types in its
113  select list.
114  """
115

119  self._scope = args["scope"]

123  self._base_types = []
