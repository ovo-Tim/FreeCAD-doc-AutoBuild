107  It2 != curve_tree->inners.end();

118  if (prev_vertex) {
119  Span span(prev_vertex->m_p, vertex);
120 
121  // order inners on this span

123  for (std::list<CurveTree*>::iterator It2 = inners_to_visit.begin();
124  It2 != inners_to_visit.end();) {
125  CurveTree* inner = *It2;
126  double t;
127  if (span.On(inner->point_on_parent, &t)) {
128  ordered_inners.insert(std::make_pair(t, inner));
129  It2 = inners_to_visit.erase(It2);
130  }
131  else {
132  It2++;
133  }

135  return;
136  }
137  }

205 static Point GetNearestPoint(CurveTree* curve_tree,
206  std::list<CurveTree*>& islands_added,
207  const CCurve& test_curve,
208  CurveTree** best_curve_tree)
209 {
210  // find nearest point to test_curve, from curve and all the islands in
211  double best_dist;
212  Point best_point = curve_tree->curve.NearestPoint(test_curve, &best_dist);
213  *best_curve_tree = curve_tree;
214  for (std::list<CurveTree*>::iterator It = islands_added.begin(); It != islands_added.end();
215  It++) {
216  CurveTree* island = *It;
217  double dist;
218  Point p = island->curve.NearestPoint(test_curve, &dist);
219  if (dist < best_dist) {
220  *best_curve_tree = island;
221  best_point = p;
222  best_dist = dist;
223  }
224  }
225 
226  return best_point;

440 void recur(std::list<CArea>& arealist, const CArea& a1, const CAreaPocketParams& params, int level)
441 {
442  // if(level > 3)return;
443 
444  // this makes arealist by recursively offsetting a1 inwards
445 
446  if (a1.m_curves.size() == 0) {
447  return;
448  }
449 
450  if (params.from_center) {
451  arealist.push_front(a1);
452  }
453  else {
454  arealist.push_back(a1);
455  }
456 
457  CArea a_offset = a1;
458  a_offset.Offset(params.stepover);
459 
460  // split curves into new areas
461  if (CArea::HolesLinked()) {
462  for (std::list<CCurve>::iterator It = a_offset.m_curves.begin();
463  It != a_offset.m_curves.end();
464  It++) {
465  CArea a2;
466  a2.m_curves.push_back(*It);
467  recur(arealist, a2, params, level + 1);
468  }
469  }
470  else {
471  // split curves into new areas
472  a_offset.Reorder();
473  CArea* a2 = NULL;
474 
475  for (std::list<CCurve>::iterator It = a_offset.m_curves.begin();
476  It != a_offset.m_curves.end();
477  It++) {
478  CCurve& curve = *It;
479  if (curve.IsClockwise()) {
480  if (a2 != NULL) {
481  a2->m_curves.push_back(curve);
482  }
483  }
484  else {
485  if (a2 != NULL) {
486  recur(arealist, *a2, params, level + 1);
487  }
488  else {
489  a2 = new CArea();
490  }
491  a2->m_curves.push_back(curve);
492  }
493  }
494 
495  if (a2 != NULL) {
496  recur(arealist, *a2, params, level + 1);
497  }
498  }
