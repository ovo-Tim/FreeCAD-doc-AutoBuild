65 void DlgPageChooser::fillList(std::vector<std::string> labels, std::vector<std::string> names)
66 {
67  QListWidgetItem* item;
68  QString qLabel;
69  QString qName;
70  QString qText;
71  int labelCount = labels.size();
72  int i = 0;
73  for (; i < labelCount; i++) {
74  qLabel = QString::fromStdString(labels[i]);
75  qName = QString::fromStdString(names[i]);
76  qText = QString::fromUtf8("%1 (%2)").arg(qLabel, qName);
77  item = new QListWidgetItem(qText, ui->lwPages);
78  item->setData(Qt::UserRole, qName);
79  }
80 }
