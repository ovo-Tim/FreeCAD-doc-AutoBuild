100 void ViewProviderCoordinateSystem::setTemporaryVisibility(bool axis, bool plane, bool points) {
101  auto origin = static_cast<App::Origin*>( getObject() );
102 
103  bool saveState = tempVisMap.empty();
104 
105  try {
106  // Remember & Set axis visibility
107  for(App::DocumentObject* obj : origin->axes()) {
108  if (obj) {
109  Gui::ViewProvider* vp = Gui::Application::Instance->getViewProvider(obj);
110  if(vp) {
111  if (saveState) {
112  tempVisMap[vp] = vp->isVisible();
113  }
114  vp->setVisible(axis);
115  }
116  }
117  }
118 
119  // Remember & Set plane visibility
120  for(App::DocumentObject* obj : origin->planes()) {
121  if (obj) {
122  Gui::ViewProvider* vp = Gui::Application::Instance->getViewProvider(obj);
123  if(vp) {
124  if (saveState) {
125  tempVisMap[vp] = vp->isVisible();
126  }
127  vp->setVisible(plane);
128  }
129  }
130  }
131 
132  // Remember & Set origin point visibility
133  App::DocumentObject* obj = origin->getOrigin();
134  if (obj) {
135  Gui::ViewProvider* vp = Gui::Application::Instance->getViewProvider(obj);
136  if (vp) {
137  if (saveState) {
138  tempVisMap[vp] = vp->isVisible();
139  }
140  vp->setVisible(points);
141  }
142  }
143 
144 
145  }
146  catch (const Base::Exception &ex) {
147  Base::Console().Error ("%s\n", ex.what() );
148  }
149 
150  // Remember & Set self visibility
151  tempVisMap[this] = isVisible();
152  setVisible(true);
153 
154 }

222 bool ViewProviderCoordinateSystem::onDelete(const std::vector<std::string> &) {
223  auto lcs = getObject<App::LocalCoordinateSystem>();
224 
225  auto origin = dynamic_cast<App::Origin*>(lcs);
226  if (origin && !origin->getInList().empty()) {
227  // Do not allow deletion of origin objects that are not lost.
228  return false;
229  }
230 
231  auto objs = lcs->OriginFeatures.getValues();
232  lcs->OriginFeatures.setValues({});
233 
234  for (auto obj: objs ) {
235  Gui::Command::doCommand( Gui::Command::Doc, "App.getDocument(\"%s\").removeObject(\"%s\")",
236  obj->getDocument()->getName(), obj->getNameInDocument() );
237  }
238 
239  return true;
240 }
