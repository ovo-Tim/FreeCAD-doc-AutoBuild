88 App::DocumentObjectExecReturn *LandmarkDimension::execute()
89 {
90  // Base::Console().Message("LD::execute() - %s\n", getNameInDocument());
91  if (!okToProceed()) {
92  return App::DocumentObject::StdReturn;
93  }
94 
95  DrawViewPart* dvp = getViewPart();
96  References2D.setValue(dvp);
97 
98  std::vector<DocumentObject*> features = References3D.getValues();
99  // Base::Console().Message("LD::execute - features: %d\n", features.size());
100  //if distance, required size = 2
101  //if angle, required size = 3; //not implemented yet
102  unsigned int requiredSize = 2;
103  if (features.size() < requiredSize) {
104  return App::DocumentObject::StdReturn;
105  }
106 
107  std::vector<Base::Vector3d> points;
108  std::vector<std::string> reprs = ReferenceTags.getValues();
109  if (reprs.empty()) {
110  //add verts to dvp & vert tags to RD
111  for (auto& f: features) {
112  Base::Vector3d loc3d = ShapeExtractor::getLocation3dFromFeat(f);
113  Base::Vector3d loc2d = projectPoint(loc3d, dvp) * dvp->getScale();
114  points.push_back(loc2d);
115  std::string tag = dvp->addReferenceVertex(loc2d);
116  reprs.push_back(tag);
117  }
118  ReferenceTags.setValues(reprs);
119  } else {
120  //update dvp referenceverts locations
121  int index = 0;
122  for (auto& f: features) {
123  Base::Vector3d loc3d = ShapeExtractor::getLocation3dFromFeat(f);
124  Base::Vector3d loc2d = projectPoint(loc3d, dvp) * dvp->getScale();
125  points.push_back(loc2d);
126  dvp->updateReferenceVert(reprs.at(index), loc2d); //sb by tag
127  index++;
128  }
129  }
130  // Base::Console().Message("LD::execute - front: %s back: %s\n",
131  // DrawUtil::formatVector(points.front()).c_str(),
132  // DrawUtil::formatVector(points.back()).c_str());
133  setLinearPoints(points.front(), points.back());
134 
135  // App::DocumentObjectExecReturn* dvdResult = DrawViewDimension::execute();
136 
137  dvp->addReferencesToGeom();
138  // dvp->requestPaint();
139 
140  overrideKeepUpdated(false);
141  return DrawView::execute();
142 }

144 Base::Vector3d LandmarkDimension::projectPoint(const Base::Vector3d& pt, DrawViewPart* dvp) const
145 {
146  Base::Vector3d stdOrg(0.0, 0.0, 0.0);
147  gp_Ax2 viewAxis = dvp->getProjectionCS(stdOrg);
148  Base::Vector3d alignedPt = pt - dvp->getOriginalCentroid();
149  gp_Pnt gPt(alignedPt.x, alignedPt.y, alignedPt.z);
150 
151  HLRAlgo_Projector projector( viewAxis );
152  gp_Pnt2d prjPnt;
153  projector.Project(gPt, prjPnt);
154  Base::Vector3d result(prjPnt.X(), prjPnt.Y(), 0.0);
155  return DrawUtil::invertY(result);
156 }
