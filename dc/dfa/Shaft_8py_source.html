295  # Initialization of structures. All three axes are handled separately so everything is 3-fold
296  # dictionaries of (location : outer force/moment) with reverse sign, which means that the segment functions for the section force and section moment
297  # created from them will have signs as by the convention in
298  # http://www.umwelt-campus.de/ucb/fileadmin/users/90_t.preussler/dokumente/Skripte/TEMECH/TMI/Ebene_Balkenstatik.pdf (page 10)

302  # Boundary conditions for shaft bending line
303  tangents = [[], [], []] # Tangents to shaft bending line
304  translations = [[], [], []] # Shaft displacement
305  # Variable names, e.g. Fx, Mz. Because the system must be exactly determined, not more than two independent variables for each
306  # force/moment per axis are possible (if there are more no solution is calculated)
307  variableNames = [[""], [""], [""]]
308  # # dictionary of (variableName : location) giving the x-coordinate at which the force/moment represented by the variable acts on the shaft
309  locations = {}
310  # Coefficients of the equilibrium equations in the form a = b * F1 + c * F2 and d = e * M1 + f * M2
311  # LHS (variables a1, a2, a3, d3) initialized to zero
312  coefficientsF = [[0], [0], [0]]
313  coefficientsM = [[0], [0], [0]]
