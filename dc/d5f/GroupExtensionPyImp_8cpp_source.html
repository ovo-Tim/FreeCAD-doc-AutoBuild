59 PyObject* GroupExtensionPy::addObject(PyObject* args)
60 {
61  PyObject* object;
62  if (!PyArg_ParseTuple(args, "O!", &(DocumentObjectPy::Type), &object)) {
63  return nullptr;
64  }
65 
66  DocumentObjectPy* docObj = static_cast<DocumentObjectPy*>(object);
67  if (!docObj->getDocumentObjectPtr()
68  || !docObj->getDocumentObjectPtr()->isAttachedToDocument()) {
69  PyErr_SetString(Base::PyExc_FC_GeneralError, "Cannot add an invalid object");
70  return nullptr;
71  }
72 
73  if (docObj->getDocumentObjectPtr()->getDocument()
74  != getGroupExtensionPtr()->getExtendedObject()->getDocument()) {
75  PyErr_SetString(Base::PyExc_FC_GeneralError,
76  "Cannot add an object from another document to this group");
77  return nullptr;
78  }
79  if (docObj->getDocumentObjectPtr() == this->getGroupExtensionPtr()->getExtendedObject()) {
80  PyErr_SetString(Base::PyExc_FC_GeneralError, "Cannot add a group object to itself");
81  return nullptr;
82  }
83  if (docObj->getDocumentObjectPtr()->hasExtension(GroupExtension::getExtensionClassTypeId())) {
84  App::GroupExtension* docGrp =
85  docObj->getDocumentObjectPtr()->getExtensionByType<GroupExtension>();
86  if (docGrp->hasObject(getGroupExtensionPtr()->getExtendedObject())) {
87  PyErr_SetString(Base::PyExc_FC_GeneralError,
88  "Cannot add a group object to a child group");
89  return nullptr;
90  }
91  }
92 
93  GroupExtension* grp = getGroupExtensionPtr();
94 
95  auto vec = grp->addObject(docObj->getDocumentObjectPtr());
96  Py::List list;
97  for (App::DocumentObject* obj : vec) {
98  list.append(Py::asObject(obj->getPyObject()));
99  }
100 
101  return Py::new_reference_to(list);
102 }
