69 void Path_RoundedComposite::Add(const Frame& F_base_point) {
70  double eps = 1E-7;
71  if (nrofpoints == 0) {
72  F_base_start = F_base_point;
73  } else if (nrofpoints == 1) {
74  F_base_via = F_base_point;
75  } else {
76  // calculate rounded segment : line + circle,
77  // determine the angle between the line segments :
78  Vector ab = F_base_via.p - F_base_start.p;
79  Vector bc = F_base_point.p - F_base_via.p;
80  double abdist = ab.Norm();
81  double bcdist = bc.Norm();
82  if (abdist < eps) {
83  throw Error_MotionPlanning_Not_Feasible(2);
84  }
85  if (bcdist < eps) {
86  throw Error_MotionPlanning_Not_Feasible(3);
87  }
88  double alpha = acos(dot(ab, bc) / abdist / bcdist);
89  if ((PI - alpha) < eps) {
90  throw Error_MotionPlanning_Not_Feasible(4);
91  }
92  if (alpha < eps) {
93  // no rounding is done in the case of parallel line segments
94  comp->Add(
95  new Path_Line(F_base_start, F_base_via, orient->Clone(),
96  eqradius));
97  F_base_start = F_base_via;
98  F_base_via = F_base_point;
99  } else {
100  double d = radius / tan((PI - alpha) / 2); // tan. is guaranteed not to return zero.
101  if (d >= abdist)
102  throw Error_MotionPlanning_Not_Feasible(5);
103 
104  if (d >= bcdist)
105  throw Error_MotionPlanning_Not_Feasible(6);
106 
107  std::unique_ptr < Path
108  > line1(
109  new Path_Line(F_base_start, F_base_via,
110  orient->Clone(), eqradius));
111  std::unique_ptr < Path
112  > line2(
113  new Path_Line(F_base_via, F_base_point,
114  orient->Clone(), eqradius));
115  Frame F_base_circlestart = line1->Pos(line1->LengthToS(abdist - d));
116  Frame F_base_circleend = line2->Pos(line2->LengthToS(d));
117  // end of circle segment, beginning of next line
118  Vector V_base_t = ab * (ab * bc);
119  V_base_t.Normalize();
120  comp->Add(
121  new Path_Line(F_base_start, F_base_circlestart,
122  orient->Clone(), eqradius));
123  comp->Add(
124  new Path_Circle(F_base_circlestart,
125  F_base_circlestart.p - V_base_t * radius,
126  F_base_circleend.p, F_base_circleend.M, alpha,
127  orient->Clone(), eqradius));
128  // shift for next line
129  F_base_start = F_base_circleend; // end of the circle segment
130  F_base_via = F_base_point;
131  }
132  }
133 
134  nrofpoints++;
135 }
