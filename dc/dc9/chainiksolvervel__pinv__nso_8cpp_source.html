68  int ChainIkSolverVel_pinv_nso::CartToJnt(const JntArray& q_in, const Twist& v_in, JntArray& qdot_out)
69  {
70  //Let the ChainJntToJacSolver calculate the jacobian "jac" for
71  //the current joint positions "q_in"
72  jnt2jac.JntToJac(q_in,jac);
73 
74  //Do a singular value decomposition of "jac" with maximum
75  //iterations "maxiter", put the results in "U", "S" and "V"
76  //jac = U*S*Vt
77  int svdResult = svd_eigen_HH(jac.data,U,S,V,tmp,maxiter);
78  if (0 != svdResult)
79  {
80  qdot_out.data.setZero() ;
81  return svdResult;
82  }
83 
84  unsigned int i;
85 
86  // We have to calculate qdot_out = jac_pinv*v_in
87  // Using the svd decomposition this becomes(jac_pinv=V*S_pinv*Ut):
88  // qdot_out = V*S_pinv*Ut*v_in
89 
90  // S^-1
91  for (i = 0; i < nj; ++i) {
92  Sinv(i) = fabs(S(i))<eps ? 0.0 : 1.0/S(i);
93  }
94  for (i = 0; i < 6; ++i) {
95  tmp(i) = v_in(i);
96  }
97 
98  qdot_out.data = V * Sinv.asDiagonal() * U.transpose() * tmp.head(6);
99 
100  // Now onto NULL space
101  // Given the cost function g, and the current joints q, desired joints qd, and weights w:
102  // t = g(q) = 1/2 * Sum( w_i * (q_i - qd_i)^2 )
103  //
104  // The jacobian Jc is:
105  // t_dot = Jc(q) * q_dot
106  // Jc = dt/dq = w_j * (q_i - qd_i) [1 x nj vector]
107  //
108  // The pseudo inverse (^-1) is
109  // Jc^-1 = w_j * (q_i - qd_i) / A [nj x 1 vector]
110  // A = Sum( w_i^2 * (q_i - qd_i)^2 )
111  //
112  // We can set the change as the step needed to remove the error times a value alpha:
113  // t_dot = -2 * alpha * t
114  //
115  // When we put it together and project into the nullspace, the final result is
116  // q'_out += (I_n - J^-1 * J) * Jc^-1 * (-2 * alpha * g(q))
117 
118  double g = 0; // g(q)
119  double A = 0; // normalizing term
120  for (i = 0; i < nj; ++i) {
121  double qd = q_in(i) - opt_pos(i);
122  g += 0.5 * qd*qd * weights(i);
123  A += qd*qd * weights(i)*weights(i);
124  }
125 
126  if (A > 1e-9) {
127  // Calculate inverse Jacobian Jc^-1
128  for (i = 0; i < nj; ++i) {
129  tmp(i) = weights(i)*(q_in(i) - opt_pos(i)) / A;
130  }
131 
132  // Calculate J^-1 * J * Jc^-1 = V*S^-1*U' * U*S*V' * tmp
133  tmp2 = V * Sinv.asDiagonal() * U.transpose() * U * S.asDiagonal() * V.transpose() * tmp;
134 
135  for (i = 0; i < nj; ++i) {
136  //std::cerr << i <<": "<< qdot_out(i) <<", "<< -2*alpha*g * (tmp(i) - tmp2(i)) << std::endl;
137  qdot_out(i) += -2*alpha*g * (tmp(i) - tmp2(i));
138  }
139  }
140  //return the return value of the svd decomposition
141  return svdResult;
142  }
