188  // *** Find closest quaternions with constrained angles.
189  class Constraints
190  {
191  public:
192  Constraints ()
193  {
194  // Members are uninitialized.
195  }
196 
197  Constraints (Real fMinAngle, Real fMaxAngle)
198  {
199  SetAngles(fMinAngle,fMaxAngle);
200  }
201 
202  void SetAngles (Real fMinAngle, Real fMaxAngle)
203  {
204  m_fMinAngle = fMinAngle;
205  m_fMaxAngle = fMaxAngle;
206  m_fCosMinAngle = Math<Real>::Cos(m_fMinAngle);
207  m_fSinMinAngle = Math<Real>::Sin(m_fMinAngle);
208  m_fCosMaxAngle = Math<Real>::Cos(m_fMaxAngle);
209  m_fSinMaxAngle = Math<Real>::Sin(m_fMaxAngle);
210  m_fDiffCosMaxMin = m_fCosMaxAngle - m_fCosMinAngle;
211  m_fDiffSinMaxMin = m_fSinMaxAngle - m_fSinMinAngle;
212  Real fAvrAngle = ((Real)0.5)*(m_fMinAngle + m_fMaxAngle);
213  m_fCosAvrAngle = Math<Real>::Cos(fAvrAngle);
214  m_fSinAvrAngle = Math<Real>::Sin(fAvrAngle);
215  }
216 
217  bool IsValid (Real fX, Real fY) const
218  {
219  // (x,y) must be unit-length.
220 
221  // Test whether (x,y) satisfies the constraints.
222  Real fXm = fX - m_fCosMinAngle;
223  Real fYm = fY - m_fSinMinAngle;
224  if (fXm*m_fDiffSinMaxMin >= fYm*m_fDiffCosMaxMin)
225  {
226  return true;
227  }
228 
229  // Test whether (-x,-y) satisfies the constraints.
230  Real fXp = fX + m_fCosMinAngle;
231  Real fYp = fY + m_fSinMinAngle;
232  if (fXp*m_fDiffSinMaxMin <= fYp*m_fDiffCosMaxMin)
233  {
234  return true;
235  }
236 
237  return false;
238  }
239 
240  Real m_fMinAngle; // in [-PI/2,PI/2]
241  Real m_fMaxAngle; // in [m_fMinAngle/2,PI/2]
242  Real m_fCosMinAngle; // = cos(m_fMinAngle)
243  Real m_fSinMinAngle; // = sin(m_fMinAngle)
244  Real m_fCosMaxAngle; // = cos(m_fMaxAngle)
245  Real m_fSinMaxAngle; // = sin(m_fMaxAngle)
246  Real m_fDiffCosMaxMin; // = cos(m_fMaxAngle) - cos(m_fMinAngle)
247  Real m_fDiffSinMaxMin; // = sin(m_fMaxAngle) - sin(m_fMinAngle)
248  Real m_fCosAvrAngle; // = cos((m_fMinAngle + m_fMaxAngle)/2)
249  Real m_fSinAvrAngle; // = sin((m_fMinAngle + mM_faxAngle)/2)
250  };
