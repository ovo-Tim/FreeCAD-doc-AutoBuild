101  if face.findPlane(): # Only planar faces.
102  face = face.copy()
103  if obj.Translate:
104  mtx = None
105  w = face.normalAt(0, 0)
106  # Try to base a matrix on the first straight edge with
107  # a reasonable length (> 0.001):
108  for e in face.Edges:
109  if geomType(e) == "Line":
110  sta = e.firstVertex().Point
111  end = e.lastVertex().Point
112  u = end.sub(sta)
113  if u.Length > 0.001:
114  u = u.normalize()
115  v = w.cross(u)
116  mtx = App.Matrix(u.x, v.x, w.x, sta.x,
117  u.y, v.y, w.y, sta.y,
118  u.z, v.z, w.z, sta.z,
119  0.0, 0.0, 0.0, 1.0)
120  break
121  # If no suitable straight edge was found use a default matrix:
122  if not mtx:
123  cen = face.CenterOfMass
124  rot = App.Rotation(App.Vector(0, 0, 1), w)
125  mtx = App.Placement(cen, rot).Matrix
126  face = face.transformShape(mtx.inverse()).Faces[0]

128  # In TechDraw edges longer than 9999.9 (ca. 10m) are considered 'crazy'.
129  # Lines in a hatch pattern are also checked. In the code below 9999 is
130  # used. With extra scaling based on that value a 1000m x 1000m rectangle
131  # can be hatched. Tested with pattern: Diagonal4, and pattern scale: 1000.
132  # With a limit of 9999.9 the same test fails.
133  if face.BoundBox.DiagonalLength > 9999:
