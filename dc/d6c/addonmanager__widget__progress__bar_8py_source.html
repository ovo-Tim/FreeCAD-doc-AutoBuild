105  @property
106  def current_task_progress(self):
107  """Current task progress, guaranteed to be in the range [0.0, 100.0]. Attempts to set a
108  value outside that range are clamped to the range."""
109  return self._current_task_progress
110 
111  @current_task_progress.setter
112  def current_task_progress(self, value: float):
113  """Set the current task's progress. Rather than raising an exception when the value is
114  outside the expected range of [0,100], clamp the task progress to allow for some
115  floating point imprecision in its calculation."""
116  if value < 0.0:
117  value = 0.0
118  elif value > 100.0:
119  value = 100.0
120  self._current_task_progress = value
121 
122  def next_task(self) -> None:
123  """Increment the task counter and reset the progress"""
124  self.current_task += 1
125  self.current_task_progress = 0.0
126 
127  def overall_progress(self) -> float:
128  """Gets the overall progress as a fractional value in the range [0, 1]"""
129  base = self._current_task / self._number_of_tasks
130  fraction = self._current_task_progress / (100.0 * self._number_of_tasks)
131  return base + fraction
