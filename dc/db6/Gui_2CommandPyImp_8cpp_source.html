91 PyObject* CommandPy::listByShortcut(PyObject *args)
92 {
93  char* shortcut_to_find;
94  PyObject* bIsRegularExp = Py_False;
95  if (!PyArg_ParseTuple(args, "s|O!", &shortcut_to_find, &PyBool_Type, &bIsRegularExp))
96  return nullptr;
97 
98  std::vector <Command*> cmds = Application::Instance->commandManager().getAllCommands();
99  std::vector <std::string> matches;
100  for (Command* c : cmds) {
101  Action* action = c->getAction();
102  if (action) {
103  QString spc = QString::fromLatin1(" ");
104  if (Base::asBoolean(bIsRegularExp)) {
105  QRegularExpression re(QString::fromLatin1(shortcut_to_find), QRegularExpression::CaseInsensitiveOption);
106  if (!re.isValid()) {
107  std::stringstream str;
108  str << "Invalid regular expression:" << ' ' << shortcut_to_find;
109  throw Py::RuntimeError(str.str());
110  }
111 
112  if (re.match(action->shortcut().toString().remove(spc).toUpper()).hasMatch()) {
113  matches.emplace_back(c->getName());
114  }
115  }
116  else if (action->shortcut().toString().remove(spc).toUpper() ==
117  QString::fromLatin1(shortcut_to_find).remove(spc).toUpper()) {
118  matches.emplace_back(c->getName());
119  }
120  }
121  }
122 
123  PyObject* pyList = PyList_New(matches.size());
124  int i=0;
125  for (const std::string& match : matches) {
126  PyObject* str = PyUnicode_FromString(match.c_str());
127  PyList_SetItem(pyList, i++, str);
128  }
129  return pyList;
130 }
