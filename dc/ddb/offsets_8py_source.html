131 def offset(edge, vector, trim=False):
132  """Return a copy of the edge at a certain vector offset.
133 
134  If the edge is an arc, the vector will be added at its first point
135  and a complete circle will be returned.
136 
137  None if there is a problem.
138 
139  Parameters
140  ----------
141  edge: Part.Shape
142  the edge to offset
143  vector: Base::Vector3
144  the vector by which the edge is to be offset
145  trim: bool, optional
146  If `edge` is an arc and `trim` is `True`, the resulting
147  arc will be trimmed to the proper angle
148 
149  Returns
150  -------
151  Part.Shape
152  The offset shape
153  """
154  if (not isinstance(edge, Part.Shape)
155  or not isinstance(vector, App.Vector)):
156  return None
157 
158  if geomType(edge) == "Line":
159  v1 = App.Vector.add(edge.Vertexes[0].Point, vector)
160  v2 = App.Vector.add(edge.Vertexes[-1].Point, vector)
161  return Part.LineSegment(v1, v2).toShape()
162 
163  elif geomType(edge) == "Circle":
164  rad = edge.Vertexes[0].Point.sub(edge.Curve.Center)
165  curve = Part.Circle(edge.Curve)
166  curve.Radius = App.Vector.add(rad, vector).Length
167  if trim:
168  return Part.ArcOfCircle(curve,
169  edge.FirstParameter,
170  edge.LastParameter).toShape()
171  elif geomType(edge) == "Ellipse":
172  rad = edge.Vertexes[0].Point.sub(edge.Curve.Center)
173  curve = edge.Curve.copy()
174  if vector.getAngle(rad) < 1:
175  curve.MajorRadius = curve.MajorRadius+vector.Length
176  curve.MinorRadius = curve.MinorRadius+vector.Length
177  else:
178  curve.MajorRadius = curve.MajorRadius-vector.Length
179  curve.MinorRadius = curve.MinorRadius-vector.Length
180  return curve.toShape()
181  else:
182  return None
