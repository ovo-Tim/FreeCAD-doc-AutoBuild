55 void CmdRobotExportKukaCompact::activated(int)
56 {
57  unsigned int n1 = getSelection().countObjectsOfType<Robot::RobotObject>();
58  unsigned int n2 = getSelection().countObjectsOfType<Robot::TrajectoryObject>();
59 
60  if (n1 != 1 || n2 != 1) {
61  QMessageBox::warning(Gui::getMainWindow(),
62  QObject::tr("Wrong selection"),
63  QObject::tr("Select one Robot and one Trajectory object."));
64  return;
65  }
66 
67  std::vector<Gui::SelectionSingleton::SelObj> Sel = getSelection().getSelection();
68 
69 
70  Robot::RobotObject* pcRobotObject = nullptr;
71  if (Sel[0].pObject->is<Robot::RobotObject>()) {
72  pcRobotObject = static_cast<Robot::RobotObject*>(Sel[0].pObject);
73  }
74  else if (Sel[1].pObject->is<Robot::RobotObject>()) {
75  pcRobotObject = static_cast<Robot::RobotObject*>(Sel[1].pObject);
76  }
77  std::string RoboName = pcRobotObject->getNameInDocument();
78 
79  Robot::TrajectoryObject* pcTrajectoryObject = nullptr;
80  if (Sel[0].pObject->is<Robot::TrajectoryObject>()) {
81  pcTrajectoryObject = static_cast<Robot::TrajectoryObject*>(Sel[0].pObject);
82  }
83  else if (Sel[1].pObject->is<Robot::TrajectoryObject>()) {
84  pcTrajectoryObject = static_cast<Robot::TrajectoryObject*>(Sel[1].pObject);
85  }
86  // std::string TrakName = pcTrajectoryObject->getNameInDocument();
87 
88  QStringList filter;
89  filter << QString::fromLatin1("%1 (*.src)").arg(QObject::tr("KRL file"));
90  filter << QString::fromLatin1("%1 (*.*)").arg(QObject::tr("All Files"));
91  QString fn = Gui::FileDialog::getSaveFileName(Gui::getMainWindow(),
92  QObject::tr("Export program"),
93  QString(),
94  filter.join(QLatin1String(";;")));
95  if (fn.isEmpty()) {
96  return;
97  }
98 
99  doCommand(Doc, "from KukaExporter import ExportCompactSub");
100  doCommand(Doc,
101  "ExportCompactSub(App.activeDocument().%s,App.activeDocument().%s,'%s')",
102  pcRobotObject->getNameInDocument(),
103  pcTrajectoryObject->getNameInDocument(),
104  (const char*)fn.toLatin1());
105 }
