126 int TopoShapeFacePy::PyInit(PyObject* args, PyObject* /*kwd*/)
127 {
128  if (PyArg_ParseTuple(args, "")) {
129  // Undefined Face
130  getTopoShapePtr()->setShape(TopoDS_Face());
131  return 0;
132  }
133 
134  PyErr_Clear();
135  PyObject *pW;
136  if (PyArg_ParseTuple(args, "O!", &(Part::TopoShapePy::Type), &pW)) {
137  try {
138  const TopoDS_Shape& sh = static_cast<Part::TopoShapePy*>(pW)->getTopoShapePtr()->getShape();
139  if (sh.IsNull()) {
140  PyErr_SetString(PartExceptionOCCError, "cannot create face out of empty wire");
141  return -1;
142  }
143 
144  if (sh.ShapeType() == TopAbs_WIRE) {
145  BRepBuilderAPI_MakeFace mkFace(TopoDS::Wire(sh));
146  if (!mkFace.IsDone()) {
147  PyErr_SetString(PartExceptionOCCError, "Failed to create face from wire");
148  return -1;
149  }
150  getTopoShapePtr()->setShape(mkFace.Face());
151  return 0;
152  }
153  else if (sh.ShapeType() == TopAbs_FACE) {
154  getTopoShapePtr()->setShape(sh);
155  return 0;
156  }
157  }
158  catch (Standard_Failure& e) {
159  PyErr_SetString(PartExceptionOCCError, e.GetMessageString());
160  return -1;
161  }
162  }
163 
164  PyErr_Clear();
165  PyObject *face, *wire;
166  if (PyArg_ParseTuple(args, "O!O!", &(Part::TopoShapeFacePy::Type), &face,
167  &(Part::TopoShapeWirePy::Type), &wire)) {
168  try {
169  const TopoDS_Shape& f = static_cast<Part::TopoShapePy*>(face)->getTopoShapePtr()->getShape();
170  if (f.IsNull()) {
171  PyErr_SetString(PartExceptionOCCError, "cannot create face out of empty support face");
172  return -1;
173  }
174  const TopoDS_Shape& w = static_cast<Part::TopoShapePy*>(wire)->getTopoShapePtr()->getShape();
175  if (w.IsNull()) {
176  PyErr_SetString(PartExceptionOCCError, "cannot create face out of empty boundary wire");
177  return -1;
178  }
179 
180  const TopoDS_Face& supportFace = TopoDS::Face(f);
181  const TopoDS_Wire& boundaryWire = TopoDS::Wire(w);
182  BRepBuilderAPI_MakeFace mkFace(supportFace, boundaryWire);
183  if (!mkFace.IsDone()) {
184  PyErr_SetString(PartExceptionOCCError, "Failed to create face from wire");
185  return -1;
186  }
187  getTopoShapePtr()->setShape(mkFace.Face());
188  return 0;
189  }
190  catch (Standard_Failure& e) {
191  PyErr_SetString(PartExceptionOCCError, e.GetMessageString());
192  return -1;
193  }
194  }
195 
196  PyErr_Clear();
197  PyObject *surf;
198  if (PyArg_ParseTuple(args, "O!O!", &(Part::GeometrySurfacePy::Type), &surf,
199  &(Part::TopoShapeWirePy::Type), &wire)) {
200  try {
201  Handle(Geom_Surface) S = Handle(Geom_Surface)::DownCast
202  (static_cast<GeometryPy*>(surf)->getGeometryPtr()->handle());
203  if (S.IsNull()) {
204  PyErr_SetString(PyExc_TypeError, "geometry is not a valid surface");
205  return -1;
206  }
207  const TopoDS_Shape& w = static_cast<Part::TopoShapePy*>(wire)->getTopoShapePtr()->getShape();
208  if (w.IsNull()) {
209  PyErr_SetString(PartExceptionOCCError, "cannot create face out of empty boundary wire");
210  return -1;
211  }
212 
213  const TopoDS_Wire& boundaryWire = TopoDS::Wire(w);
214  BRepBuilderAPI_MakeFace mkFace(S, boundaryWire);
215  if (!mkFace.IsDone()) {
216  PyErr_SetString(PartExceptionOCCError, "Failed to create face from wire");
217  return -1;
218  }
219  getTopoShapePtr()->setShape(mkFace.Face());
220  return 0;
221  }
222  catch (Standard_Failure& e) {
223  PyErr_SetString(PartExceptionOCCError, e.GetMessageString());
224  return -1;
225  }
226  }
227 
228  PyErr_Clear();
229  PyObject *bound=nullptr;
230  if (PyArg_ParseTuple(args, "O!|O!", &(GeometryPy::Type), &surf, &(PyList_Type), &bound)) {
231  try {
232  Handle(Geom_Surface) S = Handle(Geom_Surface)::DownCast
233  (static_cast<GeometryPy*>(surf)->getGeometryPtr()->handle());
234  if (S.IsNull()) {
235  PyErr_SetString(PyExc_TypeError, "geometry is not a valid surface");
236  return -1;
237  }
238 
239  BRepBuilderAPI_MakeFace mkFace(S, Precision::Confusion());
240  if (bound) {
241  Py::List list(bound);
242  for (Py::List::iterator it = list.begin(); it != list.end(); ++it) {
243  PyObject* item = (*it).ptr();
244  if (PyObject_TypeCheck(item, &(Part::TopoShapePy::Type))) {
245  const TopoDS_Shape& sh = static_cast<Part::TopoShapePy*>(item)->getTopoShapePtr()->getShape();
246  if (sh.ShapeType() == TopAbs_WIRE)
247  mkFace.Add(TopoDS::Wire(sh));
248  else {
249  PyErr_SetString(PyExc_TypeError, "shape is not a wire");
250  return -1;
251  }
252  }
253  else {
254  PyErr_SetString(PyExc_TypeError, "item is not a shape");
255  return -1;
256  }
257  }
258  }
259 
260  getTopoShapePtr()->setShape(mkFace.Face());
261  return 0;
262  }
263  catch (Standard_Failure& e) {
264  PyErr_SetString(PartExceptionOCCError, e.GetMessageString());
265  return -1;
266  }
267  }
268 
269  PyErr_Clear();
270  if (PyArg_ParseTuple(args, "O!", &(PyList_Type), &bound)) {
271  try {
272  std::vector<TopoDS_Wire> wires;
273  Py::List list(bound);
274  for (Py::List::iterator it = list.begin(); it != list.end(); ++it) {
275  PyObject* item = (*it).ptr();
276  if (PyObject_TypeCheck(item, &(Part::TopoShapePy::Type))) {
277  const TopoDS_Shape& sh = static_cast<Part::TopoShapePy*>(item)->getTopoShapePtr()->getShape();
278  if (sh.ShapeType() == TopAbs_WIRE)
279  wires.push_back(TopoDS::Wire(sh));
280  else
281  Standard_Failure::Raise("shape is not a wire");
282  }
283  else
284  Standard_Failure::Raise("shape is not a wire");
285  }
286 
287  if (!wires.empty()) {
288  BRepBuilderAPI_MakeFace mkFace(wires.front());
289  if (!mkFace.IsDone()) {
290  switch (mkFace.Error()) {
291  case BRepBuilderAPI_NoFace:
292  Standard_Failure::Raise("No face");
293  break;
294  case BRepBuilderAPI_NotPlanar:
295  Standard_Failure::Raise("Not planar");
296  break;
297  case BRepBuilderAPI_CurveProjectionFailed:
298  Standard_Failure::Raise("Curve projection failed");
299  break;
300  case BRepBuilderAPI_ParametersOutOfRange:
301  Standard_Failure::Raise("Parameters out of range");
302  break;
303  default:
304  Standard_Failure::Raise("Unknown failure");
305  break;
306  }
307  }
308  for (std::vector<TopoDS_Wire>::iterator it = wires.begin()+1; it != wires.end(); ++it)
309  mkFace.Add(*it);
310  getTopoShapePtr()->setShape(mkFace.Face());
311  return 0;
312  }
313  else {
314  Standard_Failure::Raise("no wires in list");
315  }
316  }
317  catch (Standard_Failure& e) {
318  PyErr_SetString(PartExceptionOCCError, e.GetMessageString());
319  return -1;
320  }
321  }
322 
323  char* className = nullptr;
324  PyObject* pcPyShapeOrList = nullptr;
325  PyErr_Clear();
326  if (PyArg_ParseTuple(args, "Os", &pcPyShapeOrList, &className)) {
327  try {
328  getTopoShapePtr()->makeElementFace(getPyShapes(pcPyShapeOrList),0,className);
329  return 0;
330  } catch (Base::Exception &e) {
331  e.setPyException();
332  return -1;
333  } catch (Standard_Failure& e){
334  PyErr_SetString(PartExceptionOCCError, e.GetMessageString());
335  return -1;
336  }
337  }
338 
339  PyErr_SetString(PartExceptionOCCError,
340  "Argument list signature is incorrect.\n\nSupported signatures:\n"
341  "(face)\n"
342  "(wire)\n"
343  "(face, wire)\n"
344  "(surface, wire)\n"
345  "(list_of_wires)\n"
346  "(wire, facemaker_class_name)\n"
347  "(list_of_wires, facemaker_class_name)\n"
348  "(surface, list_of_wires)\n"
349  );
350  return -1;
351 }

444 PyObject* TopoShapeFacePy::getUVNodes(PyObject *args)
445 {
446  if (!PyArg_ParseTuple(args, ""))
447  return nullptr;
448 
449  auto f = getTopoDSFace(this);
450  TopLoc_Location aLoc;
451  Handle (Poly_Triangulation) mesh = BRep_Tool::Triangulation(f,aLoc);
452  if (mesh.IsNull()) {
453  PyErr_SetString(PyExc_RuntimeError, "Face has no triangulation");
454  return nullptr;
455  }
456 
457  Py::List list;
458  if (!mesh->HasUVNodes()) {
459  return Py::new_reference_to(list);
460  }
461 
462 #if OCC_VERSION_HEX >= 0x070600
463  for (int i=1; i<=mesh->NbNodes(); i++) {
464  gp_Pnt2d pt2d = mesh->UVNode(i);
465  Py::Tuple uv(2);
466  uv.setItem(0, Py::Float(pt2d.X()));
467  uv.setItem(1, Py::Float(pt2d.Y()));
468  list.append(uv);
469  }
470 #else
471  const TColgp_Array1OfPnt2d& aNodesUV = mesh->UVNodes();
472  for (int i=aNodesUV.Lower(); i<=aNodesUV.Upper(); i++) {
473  gp_Pnt2d pt2d = aNodesUV(i);
474  Py::Tuple uv(2);
475  uv.setItem(0, Py::Float(pt2d.X()));
476  uv.setItem(1, Py::Float(pt2d.Y()));
477  list.append(uv);
478  }
479 #endif
480 
481  return Py::new_reference_to(list);
482 }
