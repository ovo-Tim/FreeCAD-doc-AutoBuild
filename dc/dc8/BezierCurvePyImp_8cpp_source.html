385 PyObject* BezierCurvePy::interpolate(PyObject * args)
386 {
387  PyObject* obj;
388  PyObject* par=nullptr;
389  if (!PyArg_ParseTuple(args, "O|O", &obj, &par))
390  return nullptr;
391  try {
392  Handle(Geom_BezierCurve) curve = Handle(Geom_BezierCurve)::DownCast
393  (getGeometryPtr()->handle());
394  Py::Sequence constraints(obj);
395  int nb_pts = constraints.size();
396  if (nb_pts < 2)
397  Standard_Failure::Raise("not enough points given");
398 
399  TColStd_Array1OfReal params(1, nb_pts);
400  if (par) {
401  Py::Sequence plist(par);
402  int param_size = plist.size();
403  if (param_size != nb_pts)
404  Standard_Failure::Raise("number of points and parameters don't match");
405  int idx=1;
406  for (Py::Sequence::iterator pit = plist.begin(); pit != plist.end(); ++pit) {
407  Py::Float val(*pit);
408  params(idx++) = (double)val;
409  }
410  }
411  else {
412  for (int idx=0; idx<nb_pts; ++idx) {
413  params(idx+1) = (double)idx/((double)nb_pts-1);
414  }
415  }
416 
417  int num_poles = 0;
418  for (Py::Sequence::iterator it1 = constraints.begin(); it1 != constraints.end(); ++it1) {
419  Py::Sequence row(*it1);
420  num_poles += (int)row.size();
421  }
422  if (num_poles > curve->MaxDegree())
423  Standard_Failure::Raise("number of constraints exceeds bezier curve capacity");
424  // create a bezier-type knot sequence
425  TColStd_Array1OfReal knots(1, 2*num_poles);
426  for (int idx=1; idx<=num_poles; ++idx) {
427  knots(idx) = params(1);
428  knots(num_poles+idx) = params(nb_pts);
429  }
430  math_Matrix OCCmatrix(1, num_poles, 1, num_poles, 0.0);
431  math_Vector res_x(1, num_poles, 0.0);
432  math_Vector res_y(1, num_poles, 0.0);
433  math_Vector res_z(1, num_poles, 0.0);
434  int row_idx = 1;
435  int cons_idx = 1;
436  for (Py::Sequence::iterator it1 = constraints.begin(); it1 != constraints.end(); ++it1) {
437  Py::Sequence row(*it1);
438  math_Matrix bezier_eval(1, row.size(), 1, num_poles, 0.0);
439  Standard_Integer first_non_zero;
440  BSplCLib::EvalBsplineBasis(row.size()-1, num_poles, knots, params(cons_idx), first_non_zero, bezier_eval, Standard_False);
441  int idx2 = 1;
442  for (Py::Sequence::iterator it2 = row.begin(); it2 != row.end(); ++it2) {
443  OCCmatrix.SetRow(row_idx, bezier_eval.Row(idx2));
444  Py::Vector v(*it2);
445  Base::Vector3d pnt = v.toVector();
446  res_x(row_idx) = pnt.x;
447  res_y(row_idx) = pnt.y;
448  res_z(row_idx) = pnt.z;
449  idx2++;
450  row_idx++;
451  }
452  cons_idx++;
453  }
454  math_Gauss gauss(OCCmatrix);
455  gauss.Solve(res_x);
456  if (!gauss.IsDone())
457  Standard_Failure::Raise("Failed to solve equations");
458  gauss.Solve(res_y);
459  if (!gauss.IsDone())
460  Standard_Failure::Raise("Failed to solve equations");
461  gauss.Solve(res_z);
462  if (!gauss.IsDone())
463  Standard_Failure::Raise("Failed to solve equations");
464 
465  TColgp_Array1OfPnt poles(1,num_poles);
466  for (int idx=1; idx<=num_poles; ++idx) {
467  poles.SetValue(idx, gp_Pnt(res_x(idx),res_y(idx),res_z(idx)));
468  }
469 
470  Handle(Geom_BezierCurve) bezier = new Geom_BezierCurve(poles);
471  this->getGeomBezierCurvePtr()->setHandle(bezier);
472  Py_Return;
473  }
474  catch (Standard_Failure& e) {
475  PyErr_SetString(PartExceptionOCCError, e.GetMessageString());
476  return nullptr;
477  }
478 }
