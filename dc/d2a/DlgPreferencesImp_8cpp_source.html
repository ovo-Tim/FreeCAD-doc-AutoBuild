206 QPixmap DlgPreferencesImp::loadIconForGroup(const std::string &name) const
207 {
208  // normalize file name
209  auto normalizeName = [](std::string str) {
210  std::transform(str.begin(), str.end(), str.begin(),
211  [](unsigned char ch) {
212  return ch == ' ' ? '_' : std::tolower(ch);
213  });
214  return str;
215  };
216 
217  std::string fileName = normalizeName(name);
218  fileName = std::string("preferences-") + fileName;
219 
220  const int px = 24;
221  QSize iconSize(px, px);
222  QPixmap icon = Gui::BitmapFactory().pixmapFromSvg(fileName.c_str(), iconSize);
223 
224  if (icon.isNull()) {
225  icon = Gui::BitmapFactory().pixmap(fileName.c_str());
226  if (icon.isNull()) {
227  qWarning() << "No group icon found for " << fileName.c_str();
228  }
229  else if (icon.size() != iconSize) {
230  icon = icon.scaled(iconSize, Qt::KeepAspectRatio, Qt::SmoothTransformation);
231  }
232  }
233 
234  return icon;
235 }

282 PreferencePage* DlgPreferencesImp::createPreferencePage(const std::string& pageName, const std::string& groupName)
283 {
284  PreferencePage* page = WidgetFactory().createPreferencePage(pageName.c_str());
285 
286  if (!page) {
287  return nullptr;
288  }
289 
290  auto resetMargins = [](QWidget* widget) {
291  widget->setContentsMargins(0, 0, 0, 0);
292  if (auto layout = widget->layout()) {
293  layout->setContentsMargins(0, 0, 0, 0);
294  }
295  };
296 
297  // settings layout already takes care for margins, we need to reset everything to 0
298  resetMargins(page);
299 
300  // special handling for PreferenceUiForm to reset margins for forms too
301  if (auto uiFormPage = qobject_cast<PreferenceUiForm*>(page)) {
302  resetMargins(uiFormPage->form());
303  }
304 
305  page->setProperty(GroupNameProperty, QString::fromStdString(groupName));
306  page->setProperty(PageNameProperty, QString::fromStdString(pageName));
307 
308  return page;
309 }

314 void DlgPreferencesImp::createPageInGroup(PreferencesPageItem *groupItem, const std::string &pageName)
315 {
316  try {
317  PreferencePage* page = createPreferencePage(pageName, groupItem->data(GroupNameRole).toString().toStdString());
318 
319  if (!page) {
320  Base::Console().Warning("%s is not a preference page\n", pageName.c_str());
321 
322  return;
323  }
324 
325  auto pageItem = new PreferencesPageItem;
326 
327  pageItem->setText(page->windowTitle());
328  pageItem->setEditable(false);
329  pageItem->setData(groupItem->data(GroupNameRole), GroupNameRole);
330  pageItem->setData(QString::fromStdString(pageName), PageNameRole);
331  pageItem->setWidget(page);
332 
333  groupItem->appendRow(pageItem);
334 
335  page->loadSettings();
336 
337  auto pages = qobject_cast<QStackedWidget*>(groupItem->getWidget());
338 
339  if (pages->count() > 0) {
340  page->setSizePolicy(QSizePolicy::Ignored, QSizePolicy::Ignored);
341  }
342 
343  pages->addWidget(page);
344  addSizeHint(page);
345  }
346  catch (const Base::Exception& e) {
347  Base::Console().Error("Base exception thrown for '%s'\n", pageName.c_str());
348  e.ReportException();
349  }
350  catch (const std::exception& e) {
351  Base::Console().Error("C++ exception thrown for '%s' (%s)\n", pageName.c_str(), e.what());
352  }
353 }

402 void DlgPreferencesImp::addPage(const std::string& className, const std::string& group)
403 {
404  auto groupToAddTo = _pages.end();
405  for (auto it = _pages.begin(); it != _pages.end(); ++it) {
406  if (it->first == group) {
407  groupToAddTo = it;
408  break;
409  }
410  }
411 
412  if (groupToAddTo != _pages.end()) {
413  // The group exists: add this page to the end of the list
414  groupToAddTo->second.push_back(className);
415  }
416  else {
417  // This is a new group: create it, with its one page
418  std::list<std::string> pages;
419  pages.push_back(className);
420  _pages.emplace_back(group, pages);
421  }
422 
423  if (DlgPreferencesImp::_activeDialog) {
424  // If the dialog is currently showing, tell it to insert the new page
425  _activeDialog->reloadPages();
426  }
427 }

429 void DlgPreferencesImp::removePage(const std::string& className, const std::string& group)
430 {
431  for (auto it = _pages.begin(); it != _pages.end(); ++it) {
432  if (it->first == group) {
433  if (className.empty()) {
434  _pages.erase(it);
435  return;
436  }
437 
438  std::list<std::string>& p = it->second;
439  for (auto jt = p.begin(); jt != p.end(); ++jt) {
440  if (*jt == className) {
441  p.erase(jt);
442  if (p.empty()) {
443  _pages.erase(it);
444  }
445  return;
446  }
447  }
448  }
449  }

646 void DlgPreferencesImp::reloadPages()
647 {
648  // Make sure that pages are ready to create
649  GetWidgetFactorySupplier();
650 
651  for (const auto &[ group, pages ] : _pages) {
652  QString groupName = QString::fromStdString(group);
653 
654  // First, does this group already exist?
655  PreferencesPageItem* groupItem = nullptr;
656 
657  auto root = _model.invisibleRootItem();
658  for (int i = 0; i < root->rowCount(); i++) {
659  auto currentGroupItem = static_cast<PreferencesPageItem*>(root->child(i));
660  auto currentGroupName = currentGroupItem->data(GroupNameRole).toString();
661 
662  if (currentGroupName == groupName) {
663  groupItem = currentGroupItem;
664  break;
665  }
666  }
667 
668  // This is a new group that wasn't there when we started this instance of the dialog:
669  if (!groupItem) {
670  groupItem = createGroup(group);
671  }
672 
673  // Move on to the pages in the group to see if we need to add any
674  for (const auto& page : pages) {
675  // Does this page already exist?
676  QString pageName = QString::fromStdString(page);
677 
678  bool pageExists = false;
679 
680  for (int i = 0; i < groupItem->rowCount(); i++) {
681  auto currentPageItem = static_cast<PreferencesPageItem*>(groupItem->child(i));
682 
683  if (currentPageItem->data(PageNameRole).toString() == pageName) {
684  pageExists = true;
685  break;
686  }
687  }
688 
689  // This is a new page that wasn't there when we started this instance of the dialog:
690  if (!pageExists) {
691  createPageInGroup(groupItem, page);
692  }
693  }
694  }
695 }

697 void DlgPreferencesImp::applyChanges()
698 {
699  // Checks if any of the classes that represent several pages of settings
700  // (DlgSettings*.*) implement checkSettings() method. If any of them do,
701  // call it to validate if user input is correct. If something fails (i.e.,
702  // not correct), shows a messageBox and set this->invalidParameter = true to
703  // cancel further operation in other methods (like in accept()).
704 
705  for (int i = 0; i < ui->groupWidgetStack->count(); i++) {
706  auto pagesStackWidget = qobject_cast<QStackedWidget*>(ui->groupWidgetStack->widget(i));
707 
708  for (int j = 0; j < pagesStackWidget->count(); j++) {
709  QWidget* page = pagesStackWidget->widget(j);
710 
711  int index = page->metaObject()->indexOfMethod("checkSettings()");
712 
713  if (index >= 0) {
714  try {
715  page->qt_metacall(QMetaObject::InvokeMetaMethod, index, nullptr);
716  }
717  catch (const Base::Exception& e) {
718  ui->groupWidgetStack->setCurrentIndex(i);
719  pagesStackWidget->setCurrentIndex(j);
720 
721  QMessageBox::warning(this,
722  tr("Wrong parameter"),
723  QString::fromLatin1(e.what()));
724 
725  this->invalidParameter = true;
726 
727  // exit early due to found errors
728  return;
729  }
730  }
731  }
732  }
733 
734  // If everything is ok (i.e., no validation problem), call method
735  // saveSettings() in every subpage (DlgSetting*) object.
736  for (int i = 0; i < ui->groupWidgetStack->count(); i++) {
737  auto pageStackWidget = qobject_cast<QStackedWidget*>(ui->groupWidgetStack->widget(i));
738 
739  for (int j = 0; j < pageStackWidget->count(); j++) {
740  auto page = qobject_cast<PreferencePage*>(pageStackWidget->widget(j));
741 
742  if (page) {
743  page->saveSettings();
744  restartRequired = restartRequired || page->isRestartRequired();
745  }
746  }
747  }
748 
749  bool saveParameter = App::GetApplication()
750  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")
751  ->GetBool("SaveUserParameter", true);
752 
753  if (saveParameter) {
754  ParameterManager* parmgr = App::GetApplication().GetParameterSet("User parameter");
755  parmgr->SaveDocument(App::Application::Config()["UserParameter"].c_str());
756  }
757 }

759 void DlgPreferencesImp::restartIfRequired()
760 {
761  if (restartRequired) {
762  QMessageBox restartBox;
763 
764  restartBox.setIcon(QMessageBox::Warning);
765  restartBox.setWindowTitle(tr("Restart required"));
766  restartBox.setText(tr("You must restart FreeCAD for changes to take effect."));
767  restartBox.setStandardButtons(QMessageBox::Ok | QMessageBox::Cancel);
768  restartBox.setDefaultButton(QMessageBox::Cancel);
769  auto okBtn = restartBox.button(QMessageBox::Ok);
770  auto cancelBtn = restartBox.button(QMessageBox::Cancel);
771  okBtn->setText(tr("Restart now"));
772  cancelBtn->setText(tr("Restart later"));
773 
774  int exec = restartBox.exec();
775 
776  if (exec == QMessageBox::Ok) {
777  //restart FreeCAD after a delay to give time to this dialog to close
778  const int ms = 1000;
779  QTimer::singleShot(ms, []()
780  {
781  QStringList args = QApplication::arguments();
782  args.pop_front();
783  if (getMainWindow()->close()) {
784  QProcess::startDetached(QApplication::applicationFilePath(), args);
785  }
786  });
787  }
788  }
