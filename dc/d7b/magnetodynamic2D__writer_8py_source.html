44  def getMagnetodynamic2DSolver(self, equation):
45  # output the equation parameters
46  s = self.write.createNonlinearSolver(equation)
47  if not equation.IsHarmonic:
48  s["Equation"] = "MgDyn2D"

54  s["Variable"] = "Potential[Potential Re:1 Potential Im:1]"

105  def handleMagnetodynamic2DMaterial(self, bodies):
106  # check that all bodies have a set material
107  for name in bodies:
108  if self.write.getBodyMaterial(name) is None:
109  raise general_writer.WriteError(
110  f"The body {name} is not referenced in any material.\n\n"
111  )
112  for obj in self.write.getMember("App::MaterialObject"):
113  m = obj.Material
114  refs = obj.References[0][1] if obj.References else self.write.getAllBodies()
115  for name in (n for n in refs if n in bodies):
116  if "ElectricalConductivity" not in m:
117  Console.PrintMessage(f"m: {m}\n")
118  raise general_writer.WriteError(
119  "The electrical conductivity must be specified for all materials.\n\n"
120  )
121  if "RelativePermeability" not in m:
122  Console.PrintMessage(f"m: {m}\n")
123  raise general_writer.WriteError(
124  "The relative permeability must be specified for all materials.\n\n"
125  )
126  self.write.material(name, "Name", m["Name"])
127  conductivity = self.write.convert(m["ElectricalConductivity"], "T^3*I^2/(L^3*M)")
128  conductivity = round(conductivity, 10) # to get rid of numerical artifacts
129  self.write.material(name, "Electric Conductivity", conductivity)
130  self.write.material(name, "Relative Permeability", float(m["RelativePermeability"]))
131  # permittivity might be necessary for the post processor
132  if "RelativePermittivity" in m:
133  self.write.material(
134  name, "Relative Permittivity", float(m["RelativePermittivity"])
135  )
