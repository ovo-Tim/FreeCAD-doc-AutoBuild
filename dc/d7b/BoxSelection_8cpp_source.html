84 void BoxSelection::selectionCallback(void * ud, SoEventCallback * cb)
85 {
86  Gui::View3DInventorViewer* view = static_cast<Gui::View3DInventorViewer*>(cb->getUserData());
87  view->removeEventCallback(SoMouseButtonEvent::getClassTypeId(), selectionCallback, ud);
88  view->setSelectionEnabled(true);
89 
90  std::vector<SbVec2f> picked = view->getGLPolygon();
91  SoCamera* cam = view->getSoRenderManager()->getCamera();
92  SbViewVolume vv = cam->getViewVolume();
93  Gui::ViewVolumeProjection proj(vv);
94  Base::Polygon2d polygon;
95  if (picked.size() == 2) {
96  SbVec2f pt1 = picked[0];
97  SbVec2f pt2 = picked[1];
98  polygon.Add(Base::Vector2d(pt1[0], pt1[1]));
99  polygon.Add(Base::Vector2d(pt1[0], pt2[1]));
100  polygon.Add(Base::Vector2d(pt2[0], pt2[1]));
101  polygon.Add(Base::Vector2d(pt2[0], pt1[1]));
102  }
103  else {
104  for (const auto& it : picked)
105  polygon.Add(Base::Vector2d(it[0],it[1]));
106  }
107 
108  BoxSelection* self = static_cast<BoxSelection*>(ud);
109  App::Document* doc = App::GetApplication().getActiveDocument();
110  if (doc) {
111  cb->setHandled();
112 
113  std::vector<Part::Feature*> geom = doc->getObjectsOfType<Part::Feature>();
114  for (auto it : geom) {
115  Gui::ViewProvider* vp = Gui::Application::Instance->getViewProvider(it);
116  if (!vp->isVisible())
117  continue;
118  const TopoDS_Shape& shape = it->Shape.getValue();
119  self->addShapeToSelection(doc->getName(), it->getNameInDocument(), proj, polygon, shape, self->shapeEnum);
120  }
121  view->redraw();
122  }
123 
124  Gui::Selection().rmvSelectionGate();
125 
126  if (self->isAutoDelete()) {
127  delete self;
128  }
129 }

145 void BoxSelection::addShapeToSelection(const char* doc, const char* obj,
146  const Gui::ViewVolumeProjection& proj,
147  const Base::Polygon2d& polygon,
148  const TopoDS_Shape& shape,
149  TopAbs_ShapeEnum subtype)
150 {
151  try {
152  const char* subname = nameFromShapeType(subtype);
153  if (!subname)
154  return;
155 
156  TopTools_IndexedMapOfShape M;
157  TopExp_Explorer xp(shape, subtype);
158  while (xp.More()) {
159  M.Add(xp.Current());
160  xp.Next();
161  }
162 
163  for (Standard_Integer k = 1; k <= M.Extent(); k++) {
164  const TopoDS_Shape& subshape = M(k);
165 
166  TopExp_Explorer xp_vertex(subshape, TopAbs_VERTEX);
167  while (xp_vertex.More()) {
168  gp_Pnt p = BRep_Tool::Pnt(TopoDS::Vertex(xp_vertex.Current()));
169  Base::Vector3d pt2d;
170  pt2d = proj(Base::Vector3d(p.X(), p.Y(), p.Z()));
171  if (polygon.Contains(Base::Vector2d(pt2d.x, pt2d.y))) {
172  std::stringstream str;
173  str << subname << k;
174  Gui::Selection().addSelection(doc, obj, str.str().c_str());
175  break;
176  }
177  xp_vertex.Next();
178  }
179  }
180  }
181  catch (...) {
182  }
183 }
