72 int CommandPy::PyInit(PyObject* args, PyObject* kwd)
73 {
74  PyObject* parameters = nullptr;
75  const char* name = "";
76  static const std::array<const char*, 3> kwlist {"name", "parameters", nullptr};
77  if (Base::Wrapped_ParseTupleAndKeywords(args,
78  kwd,
79  "|sO!",
80  kwlist,
81  &name,
82  &PyDict_Type,
83  &parameters)) {
84  std::string sname(name);
85  boost::to_upper(sname);
86  try {
87  if (!sname.empty()) {
88  getCommandPtr()->setFromGCode(name);
89  }
90  }
91  catch (const Base::Exception& e) {
92  PyErr_SetString(PyExc_ValueError, e.what());
93  return -1;
94  }
95 
96  PyObject *key, *value;
97  Py_ssize_t pos = 0;
98  while (parameters && PyDict_Next(parameters, &pos, &key, &value)) {
99  std::string ckey;
100  if (PyUnicode_Check(key)) {
101  ckey = PyUnicode_AsUTF8(key);
102  }
103  else {
104  PyErr_SetString(PyExc_TypeError, "The dictionary can only contain string keys");
105  return -1;
106  }
107 
108  boost::to_upper(ckey);
109  double cvalue;
110  if (PyObject_TypeCheck(value, &(PyLong_Type))) {
111  cvalue = (double)PyLong_AsLong(value);
112  }
113  else if (PyObject_TypeCheck(value, &(PyFloat_Type))) {
114  cvalue = PyFloat_AsDouble(value);
115  }
116  else {
117  PyErr_SetString(PyExc_TypeError, "The dictionary can only contain number values");
118  return -1;
119  }
120  getCommandPtr()->Parameters[ckey] = cvalue;
121  }
122  parameters_copy_dict.clear();
123  return 0;
124  }
125  PyErr_Clear(); // set by PyArg_ParseTuple()
126 
127  if (Base::Wrapped_ParseTupleAndKeywords(args,
128  kwd,
129  "|sO!",
130  kwlist,
131  &name,
132  &(Base::PlacementPy::Type),
133  &parameters)) {
134  std::string sname(name);
135  boost::to_upper(sname);
136  try {
137  if (!sname.empty()) {
138  getCommandPtr()->setFromGCode(name);
139  }
140  }
141  catch (const Base::Exception& e) {
142  PyErr_SetString(PyExc_ValueError, e.what());
143  return -1;
144  }
145  Base::PlacementPy* p = static_cast<Base::PlacementPy*>(parameters);
146  getCommandPtr()->setFromPlacement(*p->getPlacementPtr());
147  return 0;
148  }
149  return -1;
150 }
