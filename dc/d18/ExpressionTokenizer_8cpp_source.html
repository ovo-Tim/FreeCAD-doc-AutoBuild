38 QString ExpressionTokenizer::perform(const QString& prefix, int pos)
39 {
40  // ExpressionParser::tokenize() only supports std::string but we need a tuple QString
41  // because due to UTF-8 encoding a std::string may be longer than a QString
42  // See https://forum.freecad.org/viewtopic.php?f=3&t=69931
43  auto tokenizeExpression = [](const QString& expr) {
44  std::vector<std::tuple<int, int, std::string>> result =
45  ExpressionParser::tokenize(expr.toStdString());
46  std::vector<std::tuple<int, int, QString>> tokens;
47  std::transform(
48  result.cbegin(),
49  result.cend(),
50  std::back_inserter(tokens),
51  [&](const std::tuple<int, int, std::string>& item) {
52  return std::make_tuple(
53  std::get<0>(item),
54  QString::fromStdString(expr.toStdString().substr(0, std::get<1>(item))).size(),
55  QString::fromStdString(std::get<2>(item)));
56  });
57  return tokens;
58  };
59 
60  QString completionPrefix;
61 
62  // Compute start; if prefix starts with =, start parsing from offset 1.
63  int start = (prefix.size() > 0 && prefix.at(0) == QChar::fromLatin1('=')) ? 1 : 0;
64 
65  // Tokenize prefix
66  std::vector<std::tuple<int, int, QString>> tokens = tokenizeExpression(prefix.mid(start));
67 
68  // No tokens
69  if (tokens.empty()) {
70  return {};
71  }
72 
73  prefixEnd = prefix.size();
74 
75  // Pop those trailing tokens depending on the given position, which may be
76  // in the middle of a token, and we shall include that token.
77  for (auto it = tokens.begin(); it != tokens.end(); ++it) {
78  if (std::get<1>(*it) >= pos) {
79  // Include the immediately followed '.' or '#', because we'll be
80  // inserting these separators too, in ExpressionCompleteModel::pathFromIndex()
81  if (it != tokens.begin() && std::get<0>(*it) != '.' && std::get<0>(*it) != '#') {
82  it = it - 1;
83  }
84  tokens.resize(it - tokens.begin() + 1);
85  prefixEnd = start + std::get<1>(*it) + (int)std::get<2>(*it).size();
86  break;
87  }
88  }
89 
90  int trim = 0;
91  if (prefixEnd > pos) {
92  trim = prefixEnd - pos;
93  }
94 
95  // Extract last tokens that can be rebuilt to a variable
96  long i = static_cast<long>(tokens.size()) - 1;
97 
98  // First, check if we have unclosing string starting from the end
99  bool stringing = false;
100  for (; i >= 0; --i) {
101  int token = std::get<0>(tokens[i]);
102  if (token == ExpressionParser::STRING) {
103  stringing = false;
104  break;
105  }
106 
107  if (token == ExpressionParser::LT && i > 0
108  && std::get<0>(tokens[i - 1]) == ExpressionParser::LT) {
109  --i;
110  stringing = true;
111  break;
112  }
113  }
114 
115  // Not an unclosed string and the last character is a space
116  if (!stringing && !prefix.isEmpty() && prefixEnd > 0 && prefixEnd <= prefix.size()
117  && prefix[prefixEnd - 1] == QChar(32)) {
118  return {};
119  }
120 
121  if (!stringing) {
122  i = static_cast<long>(tokens.size()) - 1;
123  for (; i >= 0; --i) {
124  int token = std::get<0>(tokens[i]);
125  if (token != '.' && token != '#' && token != ExpressionParser::IDENTIFIER
126  && token != ExpressionParser::INTEGER && token != ExpressionParser::STRING
127  && token != ExpressionParser::UNIT && token != ExpressionParser::ONE) {
128  break;
129  }
130  }
131  ++i;
132  }
133 
134  // Set prefix start for use when replacing later
135  if (i == static_cast<long>(tokens.size())) {
136  prefixStart = prefixEnd;
137  }
138  else {
139  prefixStart = start + std::get<1>(tokens[i]);
140  }
141 
142  // Build prefix from tokens
143  while (i < static_cast<long>(tokens.size())) {
144  completionPrefix += std::get<2>(tokens[i]);
145  ++i;
146  }
147 
148  if (trim && trim < int(completionPrefix.size())) {
149  completionPrefix.resize(completionPrefix.size() - trim);
150  }
151 
152  return completionPrefix;
153 }
