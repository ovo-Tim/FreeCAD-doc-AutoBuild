39 void SimDisplay::InitShaders()
40 {
41  // use shaders
42  // standard diffuse shader
43  shader3D.CompileShader("StdDiffuse", VertShader3DNorm, FragShaderNorm);
44  shader3D.UpdateEnvColor(lightPos, lightColor, ambientCol, 0.0f);
45 
46  // invarted normal diffuse shader for inner mesh
47  shaderInv3D.CompileShader("InvertNormal", VertShader3DInvNorm, FragShaderNorm);
48  shaderInv3D.UpdateEnvColor(lightPos, lightColor, ambientCol, 0.0f);
49 
50  // null shader to calculate meshes only (simulation stage)
51  shaderFlat.CompileShader("Null", VertShader3DNorm, FragShaderFlat);
52 
53  // texture shader to render Simulator FBO
54  shaderSimFbo.CompileShader("Texture", VertShader2DFbo, FragShader2dFbo);
55  shaderSimFbo.UpdateTextureSlot(0);
56 
57  // geometric shader - generate texture with all geometric info for further processing
58  shaderGeom.CompileShader("Geometric", VertShaderGeom, FragShaderGeom);
59  shaderGeomCloser.CompileShader("GeomCloser", VertShaderGeom, FragShaderGeom);
60 
61  // SSAO shader - generate SSAO info and embed in texture buffer
62  shaderSSAO.CompileShader("SSAO", VertShader2DFbo, FragShaderSSAO);
63  shaderSSAO.UpdateRandomTexSlot(0);
64  shaderSSAO.UpdatePositionTexSlot(1);
65  shaderSSAO.UpdateNormalTexSlot(2);
66 
67  // SSAO blur shader - smooth generated SSAO texture
68  shaderSSAOBlur.CompileShader("Blur", VertShader2DFbo, FragShaderSSAOBlur);
69  shaderSSAOBlur.UpdateSsaoTexSlot(0);
70 
71  // SSAO lighting shader - apply lightig modified by SSAO calculations
72  shaderSSAOLighting.CompileShader("SsaoLighting", VertShader2DFbo, FragShaderSSAOLighting);
73  shaderSSAOLighting.UpdateColorTexSlot(0);
74  shaderSSAOLighting.UpdatePositionTexSlot(1);
75  shaderSSAOLighting.UpdateNormalTexSlot(2);
76  shaderSSAOLighting.UpdateSsaoTexSlot(3);
77  shaderSSAOLighting.UpdateEnvColor(lightPos, lightColor, ambientCol, 0.01f);
78 
79  // Mill Path Line Shader
80  shaderLinePath.CompileShader("PathLine", VertShader3DLine, FragShader3DLine);
81 }

117 void SimDisplay::UniformHemisphere(vec3& randVec)
118 {
119  float x1 = distr01(generator);
120  float x2 = distr01(generator);
121  float s = sqrt(1.0f - x1 * x1);
122  randVec[0] = cosf(PI2 * x2) * s;
123  randVec[1] = sinf(PI2 * x2) * s;
124  randVec[2] = x1;
125 }

127 void SimDisplay::UniformCircle(vec3& randVec)
128 {
129  float x = distr01(generator);
130  randVec[0] = cosf(PI2 * x);
131  randVec[1] = sinf(PI2 * x);
132  randVec[2] = 0;
133 }

137 {

144  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, mFboColTexture, 0);
145 
146  // a position texture for the frame buffer
147  CreateGBufTex(GL_TEXTURE1, GL_RGB32F, GL_RGBA, GL_FLOAT, mFboPosTexture);
148  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, mFboPosTexture, 0);
149 
150  // a normal texture for the frame buffer
151  CreateGBufTex(GL_TEXTURE2, GL_RGB32F, GL_RGBA, GL_FLOAT, mFboNormTexture);
152  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT2, GL_TEXTURE_2D, mFboNormTexture, 0);
153 
154 
155  unsigned int attachments[3] = {GL_COLOR_ATTACHMENT0,
156  GL_COLOR_ATTACHMENT1,
157  GL_COLOR_ATTACHMENT2};
158  glDrawBuffers(3, attachments);
159 
160  glGenRenderbuffers(1, &mRboDepthStencil);
161  glBindRenderbuffer(GL_RENDERBUFFER, mRboDepthStencil);
162  glRenderbufferStorage(
163  GL_RENDERBUFFER,
164  GL_DEPTH24_STENCIL8,
165  mWidth,
166  mHeight); // use a single renderbuffer object for both a depth AND stencil buffer.
167  glFramebufferRenderbuffer(GL_FRAMEBUFFER,
168  GL_DEPTH_STENCIL_ATTACHMENT,
169  GL_RENDERBUFFER,
170  mRboDepthStencil); // now actually attach it
171 
172  if (glCheckFramebufferStatus(GL_FRAMEBUFFER) != GL_FRAMEBUFFER_COMPLETE) {
173  return;
174  }
175 
176  glBindFramebuffer(GL_FRAMEBUFFER, 0);
177 }

179 void SimDisplay::CreateSsaoFbos()
180 {
181 
182  mSsaoValid = true;
183 
184  // setup framebuffer for SSAO processing
185  glGenFramebuffers(1, &mSsaoFbo);
186  glBindFramebuffer(GL_FRAMEBUFFER, mSsaoFbo);
187  // SSAO color buffer
188  CreateGBufTex(GL_TEXTURE0, GL_R16F, GL_RED, GL_FLOAT, mFboSsaoTexture);
189  glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, mFboSsaoTexture, 0);
190  if (glCheckFramebufferStatus(GL_FRAMEBUFFER) != GL_FRAMEBUFFER_COMPLETE) {
191  mSsaoValid = false;
192  return;
193  }
194 
195  // setup framebuffer for SSAO blur processing
196  glGenFramebuffers(1, &mSsaoBlurFbo);
197  glBindFramebuffer(GL_FRAMEBUFFER, mSsaoBlurFbo);
198  CreateGBufTex(GL_TEXTURE0, GL_R16F, GL_RED, GL_FLOAT, mFboSsaoBlurTexture);
199  glFramebufferTexture2D(GL_FRAMEBUFFER,
200  GL_COLOR_ATTACHMENT0,
201  GL_TEXTURE_2D,
202  mFboSsaoBlurTexture,
203  0);
204  if (glCheckFramebufferStatus(GL_FRAMEBUFFER) != GL_FRAMEBUFFER_COMPLETE) {
205  mSsaoValid = false;
206  return;
207  }
208 
209  glBindFramebuffer(GL_FRAMEBUFFER, 0);
210 
211  // generate sample kernel
212  int kernSize = 64;
213  for (int i = 0; i < kernSize; i++) {
214  vec3 sample;
215  UniformHemisphere(sample);
216  float scale = ((float)(i * i)) / (kernSize * kernSize);
217  float interpScale = 0.1f * (1.0f - scale) + scale;
218  vec3_scale(sample, sample, interpScale);
219  mSsaoKernel.push_back(*(Point3D*)sample);
220  }
221  shaderSSAO.Activate();
222  shaderSSAO.UpdateKernelVals(mSsaoKernel.size(), &mSsaoKernel[0].x);
223 
224  // generate random direction texture
225  int randSize = 4 * 4;
226  std::vector<Point3D> randDirections;
227  for (int i = 0; i < randSize; i++) {
228  vec3 randvec;
229  UniformCircle(randvec);
230  randDirections.push_back(*(Point3D*)randvec);
231  }
232 
233  glGenTextures(1, &mFboRandTexture);
234  glBindTexture(GL_TEXTURE_2D, mFboRandTexture);
235  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA32F, 4, 4, 0, GL_RGB, GL_FLOAT, &randDirections[0].x);
236  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
237  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
238  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
239  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
240 }
