87 def python_package_updates_are_available() -> bool:
88  """Returns True if any of the Python packages installed into the AdditionalPythonPackages
89  directory have updates available, or False if they are all up-to-date."""
90 
91  vendor_path = os.path.join(fci.DataPaths().data_dir, "AdditionalPythonPackages")
92  package_counter = 0
93  try:
94  outdated_packages_stdout = call_pip(["list", "-o", "--path", vendor_path])
95  except PipFailed as e:
96  fci.Console.PrintError(str(e) + "\n")
97  return False
98  fci.Console.PrintLog("Output from pip -o:\n")
99  for line in outdated_packages_stdout:
100  if len(line) > 0:
101  package_counter += 1
102  fci.Console.PrintLog(f" {line}\n")
103  return package_counter > 0

106 def call_pip(args: List[str]) -> List[str]:
107  """Tries to locate the appropriate Python executable and run pip with version checking
108  disabled. Fails if Python can't be found or if pip is not installed."""
109 
110  python_exe = get_python_exe()
111  pip_failed = False
112  if python_exe:
113  call_args = [python_exe, "-m", "pip", "--disable-pip-version-check"]
114  call_args.extend(args)
115  proc = None
116  try:
117  proc = utils.run_interruptable_subprocess(call_args)
118  except subprocess.CalledProcessError:
119  pip_failed = True
120 
121  if not pip_failed:
122  data = proc.stdout
123  return data.split("\n")
124  elif proc:
125  raise PipFailed(proc.stderr)
126  else:
127  raise PipFailed("pip timed out")
128  else:
129  raise PipFailed("Could not locate Python executable on this system")

132 def parse_pip_list_output(all_packages, outdated_packages) -> Dict[str, Dict[str, str]]:
133  """Parses the output from pip into a dictionary with update information in it. The pip
134  output should be an array of lines of text."""
135 
136  # All Packages output looks like this:
137  # Package Version
138  # ---------- -------
139  # gitdb 4.0.9
140  # setuptools 41.2.0
141 
142  # Outdated Packages output looks like this:
143  # Package Version Latest Type
144  # ---------- ------- ------ -----
145  # pip 21.0.1 22.1.2 wheel
146  # setuptools 41.2.0 63.2.0 wheel
147 
148  packages = {}
149  skip_counter = 0
150  for line in all_packages:
151  if skip_counter < 2:
152  skip_counter += 1
153  continue
154  entries = line.split()
155  if len(entries) > 1:
156  package_name = entries[0]
157  installed_version = entries[1]
158  packages[package_name] = {
159  "installed_version": installed_version,
160  "available_version": "",
161  }
162 
163  skip_counter = 0
164  for line in outdated_packages:
165  if skip_counter < 2:
166  skip_counter += 1
167  continue
168  entries = line.split()
169  if len(entries) > 1:
170  package_name = entries[0]
171  installed_version = entries[1]
172  available_version = entries[2]
173  packages[package_name] = {
174  "installed_version": installed_version,
175  "available_version": available_version,
176  }
177 
178  return packages

394  @classmethod
395  def migrate_old_am_installations(cls) -> bool:
396  """Move packages installed before the Addon Manager switched to a versioned directory
397  structure into the versioned structure. Returns True if a migration was done, or false
398  if no migration was needed."""
399 
400  migrated = False
401 
402  old_directory = os.path.join(fci.DataPaths().data_dir, "AdditionalPythonPackages")
403 
404  new_directory = utils.get_pip_target_directory()
405  new_directory_name = new_directory.rsplit(os.path.sep, 1)[1]
406 
407  if not os.path.exists(old_directory) or os.path.exists(
408  os.path.join(old_directory, "MIGRATION_COMPLETE")
409  ):
410  # Nothing to migrate
411  return False
412 
413  os.makedirs(new_directory, mode=0o777, exist_ok=True)
414 
415  for content_item in os.listdir(old_directory):
416  if content_item == new_directory_name:
417  continue
418  old_path = os.path.join(old_directory, content_item)
419  new_path = os.path.join(new_directory, content_item)
420  fci.Console.PrintLog(
421  f"Moving {content_item} into the new (versioned) directory structure\n"
422  )
423  fci.Console.PrintLog(f" {old_path} --> {new_path}\n")
424  shutil.move(old_path, new_path)
425  migrated = True
426 
427  sys.path.append(new_directory)
428  cls._add_current_python_version()
429 
430  with open(os.path.join(old_directory, "MIGRATION_COMPLETE"), "w", encoding="utf-8") as f:
431  f.write("Files originally installed in this directory have been migrated to:\n")
432  f.write(new_directory)
433  f.write(
434  "\nThe existence of this file prevents the Addon Manager from "
435  "attempting the migration again.\n"
436  )
437  return migrated
