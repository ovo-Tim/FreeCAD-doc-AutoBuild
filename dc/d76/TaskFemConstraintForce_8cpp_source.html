121 void TaskFemConstraintForce::addToSelection()
122 {
123  std::vector<Gui::SelectionObject> selection =
124  Gui::Selection().getSelectionEx(); // gets vector of selected objects of active document
125  if (selection.empty()) {
126  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
127  return;
128  }
129  Fem::ConstraintForce* pcConstraint = ConstraintView->getObject<Fem::ConstraintForce>();
130  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
131  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
132 
133  for (auto& it : selection) { // for every selected object
134  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
135  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
136  return;
137  }
138  const std::vector<std::string>& subNames = it.getSubNames();
139  App::DocumentObject* obj = it.getObject();
140  for (const auto& subName : subNames) { // for every selected sub element
141  bool addMe = true;
142  for (std::vector<std::string>::iterator itr =
143  std::find(SubElements.begin(), SubElements.end(), subName);
144  itr != SubElements.end();
145  itr = std::find(++itr,
146  SubElements.end(),
147  subName)) { // for every sub element in selection that
148  // matches one in old list
149  if (obj
150  == Objects[std::distance(
151  SubElements.begin(),
152  itr)]) { // if selected sub element's object equals the one in old list
153  // then it was added before so don't add
154  addMe = false;
155  }
156  }
157  // limit constraint such that only vertexes or faces or edges can be used depending on
158  // what was selected first
159  std::string searchStr;
160  if (subName.find("Vertex") != std::string::npos) {
161  searchStr = "Vertex";
162  }
163  else if (subName.find("Edge") != std::string::npos) {
164  searchStr = "Edge";
165  }
166  else {
167  searchStr = "Face";
168  }
169 
170  for (const auto& SubElement : SubElements) {
171  if (SubElement.find(searchStr) == std::string::npos) {
172  QString msg = tr("Only one type of selection (vertex, face or edge) per "
173  "analysis feature allowed!");
174  QMessageBox::warning(this, tr("Selection error"), msg);
175  addMe = false;
176  break;
177  }
178  }
179  if (addMe) {
180  QSignalBlocker block(ui->listReferences);
181  Objects.push_back(obj);
182  SubElements.push_back(subName);
183  ui->listReferences->addItem(makeRefText(obj, subName));
184  }
185  }
186  }
187  // Update UI
188  pcConstraint->References.setValues(Objects, SubElements);
189  updateUI();
190 }

192 void TaskFemConstraintForce::removeFromSelection()
193 {
194  std::vector<Gui::SelectionObject> selection =
195  Gui::Selection().getSelectionEx(); // gets vector of selected objects of active document
196  if (selection.empty()) {
197  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
198  return;
199  }
200  Fem::ConstraintForce* pcConstraint = ConstraintView->getObject<Fem::ConstraintForce>();
201  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
202  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
203  std::vector<size_t> itemsToDel;
204  for (const auto& it : selection) { // for every selected object
205  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
206  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
207  return;
208  }
209  const std::vector<std::string>& subNames = it.getSubNames();
210  const App::DocumentObject* obj = it.getObject();
211 
212  for (const auto& subName : subNames) { // for every selected sub element
213  for (std::vector<std::string>::iterator itr =
214  std::find(SubElements.begin(), SubElements.end(), subName);
215  itr != SubElements.end();
216  itr = std::find(++itr,
217  SubElements.end(),
218  subName)) { // for every sub element in selection that
219  // matches one in old list
220  if (obj
221  == Objects[std::distance(
222  SubElements.begin(),
223  itr)]) { // if selected sub element's object equals the one in old list
224  // then it was added before so mark for deletion
225  itemsToDel.push_back(std::distance(SubElements.begin(), itr));
226  }
227  }
228  }
229  }
230  std::sort(itemsToDel.begin(), itemsToDel.end());
231  while (!itemsToDel.empty()) {
232  Objects.erase(Objects.begin() + itemsToDel.back());
233  SubElements.erase(SubElements.begin() + itemsToDel.back());
234  itemsToDel.pop_back();
235  }
236  // Update UI
237  {
238  QSignalBlocker block(ui->listReferences);
239  ui->listReferences->clear();
240  for (unsigned int j = 0; j < Objects.size(); j++) {
241  ui->listReferences->addItem(makeRefText(Objects[j], SubElements[j]));
242  }
243  }
244  pcConstraint->References.setValues(Objects, SubElements);
245  updateUI();
246 }

254 std::pair<App::DocumentObject*, std::string>
255 TaskFemConstraintForce::getDirection(const std::vector<Gui::SelectionObject>& selection) const
256 {
257  std::pair<App::DocumentObject*, std::string> link;
258  if (selection.empty()) {
259  return link;
260  }
261 
262  // we only handle the first selected object
263  Gui::SelectionObject selectionElement = selection.at(0);
264 
265  // Line or Plane
266  Base::Type line = Base::Type::fromName("PartDesign::Line");
267  Base::Type plane = Base::Type::fromName("PartDesign::Plane");
268  if (selectionElement.isObjectTypeOf(App::Line::getClassTypeId())
269  || selectionElement.isObjectTypeOf(App::Plane::getClassTypeId())) {
270  link = std::make_pair(selectionElement.getObject(), std::string());
271  }
272  else if (selectionElement.isObjectTypeOf(line)) {
273  link = std::make_pair(selectionElement.getObject(), std::string("Edge1"));
274  }
275  else if (selectionElement.isObjectTypeOf(plane)) {
276  link = std::make_pair(selectionElement.getObject(), std::string("Face1"));
277  }
278  // Sub-element of Part object
279  else if (selectionElement.isObjectTypeOf(Part::Feature::getClassTypeId())) {
280  const std::vector<std::string>& subNames = selectionElement.getSubNames();
281  if (subNames.size() != 1) {
282  return link;
283  }
284 
285  std::string subNamesElement = subNames[0];
286 
287  const Part::Feature* feat = static_cast<const Part::Feature*>(selectionElement.getObject());
288  TopoDS_Shape ref = feat->Shape.getShape().getSubShape(subNamesElement.c_str());
289 
290  if (ref.ShapeType() == TopAbs_EDGE) {
291  if (Fem::Tools::isLinear(TopoDS::Edge(ref))) {
292  link = std::make_pair(selectionElement.getObject(), subNamesElement);
293  }
294  }
295  else if (ref.ShapeType() == TopAbs_FACE) {
296  if (Fem::Tools::isPlanar(TopoDS::Face(ref))) {
297  link = std::make_pair(selectionElement.getObject(), subNamesElement);
298  }
299  }
300  }
301 
302  return link;
303 }
