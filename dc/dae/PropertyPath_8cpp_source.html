103 void PropertyPath::Restore(Base::XMLReader& reader)
104 {
105  reader.readElement("Path");
106 
107  std::string file(reader.getAttribute("file"));
108  if (!file.empty()) {
109  // initiate a file read
110  reader.addFile(file.c_str(), this);
111  }
112 
113  if (reader.hasAttribute("version")) {
114  int version = reader.getAttributeAsInteger("version");
115  if (version >= Toolpath::SchemaVersion) {
116  reader.readElement("Center");
117  double x = reader.getAttributeAsFloat("x");
118  double y = reader.getAttributeAsFloat("y");
119  double z = reader.getAttributeAsFloat("z");
120  Base::Vector3d center(x, y, z);
121  _Path.setCenter(center);
122  }
123  }
124 }
