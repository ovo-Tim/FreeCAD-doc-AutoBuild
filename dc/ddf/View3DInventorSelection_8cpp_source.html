96 void View3DInventorSelection::checkGroupOnTop(const SelectionChanges &Reason)
97 {
98  if (Reason.Type == SelectionChanges::SetSelection || Reason.Type == SelectionChanges::ClrSelection) {
99  clearGroupOnTop();
100  if(Reason.Type == SelectionChanges::ClrSelection)
101  return;
102  }
103  if(Reason.Type == SelectionChanges::RmvPreselect ||
104  Reason.Type == SelectionChanges::RmvPreselectSignal)
105  {
106  SoSelectionElementAction action(SoSelectionElementAction::None,true);
107  action.apply(pcGroupOnTopPreSel);
108  coinRemoveAllChildren(pcGroupOnTopPreSel);
109  objectsOnTopPreSel.clear();
110  return;
111  }
112  if(!getDocument() || !Reason.pDocName || !Reason.pDocName[0] || !Reason.pObjectName)
113  return;
114  auto obj = getDocument()->getDocument()->getObject(Reason.pObjectName);
115  if(!obj || !obj->isAttachedToDocument())
116  return;
117  std::string key(obj->getNameInDocument());
118  key += '.';
119  auto subname = Reason.pSubName;
120  App::ElementNamePair element;
121  App::GeoFeature::resolveElement(obj, Reason.pSubName, element);
122  if (Data::isMappedElement(subname)
123  && !element.oldName.empty()) { // If we have a shortened element name
124  subname = element.oldName.c_str(); // use if
125  }
126  if(subname)
127  key += subname;
128  if(Reason.Type == SelectionChanges::RmvSelection) {
129  auto &objs = objectsOnTop;
130  auto pcGroup = pcGroupOnTopSel;
131  auto it = objs.find(key.c_str());
132  if(it == objs.end())
133  return;
134  int index = pcGroup->findChild(it->second);
135  if(index >= 0) {
136  auto node = static_cast<SoFCPathAnnotation*>(it->second);
137  SoSelectionElementAction action(node->getDetail()?
138  SoSelectionElementAction::Remove:SoSelectionElementAction::None,true);
139  auto path = node->getPath();
140  SoTempPath tmpPath(2 + (path ? path->getLength() : 0));
141  tmpPath.ref();
142  tmpPath.append(pcGroup);
143  tmpPath.append(node);
144  tmpPath.append(node->getPath());
145  action.setElement(node->getDetail());
146  action.apply(&tmpPath);
147  tmpPath.unrefNoDelete();
148  pcGroup->removeChild(index);
149  FC_LOG("remove annotation " << Reason.Type << " " << key);
150  }else
151  FC_LOG("remove annotation object " << Reason.Type << " " << key);
152  objs.erase(it);
153  return;
154  }
155 
156  auto &objs = Reason.Type==SelectionChanges::SetPreselect?objectsOnTopPreSel:objectsOnTop;
157  auto pcGroup = Reason.Type==SelectionChanges::SetPreselect?pcGroupOnTopPreSel:pcGroupOnTopSel;
158 
159  if(objs.find(key.c_str())!=objs.end())
160  return;
161  auto vp = dynamic_cast<ViewProviderDocumentObject*>(
162  Application::Instance->getViewProvider(obj));
163  if(!vp || !vp->isSelectable() || !vp->isShow())
164  return;
165  auto svp = vp;
166  if(subname && *subname) {
167  auto sobj = obj->getSubObject(subname);
168  if(!sobj || !sobj->isAttachedToDocument())
169  return;
170  if(sobj!=obj) {
171  svp = dynamic_cast<ViewProviderDocumentObject*>(
172  Application::Instance->getViewProvider(sobj));
173  if(!svp || !svp->isSelectable())
174  return;
175  }
176  }
177  int onTop;
178  // onTop==2 means on top only if whole object is selected,
179  // onTop==3 means on top only if some sub-element is selected
180  // onTop==1 means either
181  if(Gui::Selection().needPickedList())
182  onTop = 1;
183  else if(vp->OnTopWhenSelected.getValue())
184  onTop = vp->OnTopWhenSelected.getValue();
185  else
186  onTop = svp->OnTopWhenSelected.getValue();
187  if(Reason.Type == SelectionChanges::SetPreselect) {
188  SoHighlightElementAction action;
189  action.setHighlighted(true);
190  action.setColor(selectionRoot->colorHighlight.getValue());
191  action.apply(pcGroupOnTopPreSel);
192  if(!onTop)
193  onTop = 2;
194  }else {
195  if(!onTop)
196  return;
197  SoSelectionElementAction action(SoSelectionElementAction::All);
198  action.setColor(selectionRoot->colorHighlight.getValue());
199  action.apply(pcGroupOnTopSel);
200  }
201  if(onTop==2 || onTop==3) {
202  if(subname && *subname) {
203  size_t len = strlen(subname);
204  if(subname[len-1]=='.') {
205  // ending with '.' means whole object selection
206  if(onTop == 3)
207  return;
208  }else if(onTop==2)
209  return;
210  }else if(onTop==3)
211  return;
212  }
213 
214  std::vector<ViewProvider*> groups;
215  auto grpVp = vp;
216  std::set<ViewProvider*> visited;
217  for(auto childVp=vp;;childVp=grpVp) {
218  auto grp = App::GeoFeatureGroupExtension::getGroupOfObject(childVp->getObject());
219  if (!grp || !grp->isAttachedToDocument()) {
220  break;
221  }
222 
223  grpVp = dynamic_cast<ViewProviderDocumentObject*>(
224  Application::Instance->getViewProvider(grp));
225  if (!grpVp) {
226  break;
227  }
228 
229  // avoid endless-loops
230  if (!visited.insert(childVp).second) {
231  break;
232  }
233 
234  auto childRoot = grpVp->getChildRoot();
235  auto modeSwitch = grpVp->getModeSwitch();
236  auto idx = modeSwitch->whichChild.getValue();
237  if(idx<0 || idx>=modeSwitch->getNumChildren() ||
238  modeSwitch->getChild(idx)!=childRoot)
239  {
240  FC_LOG("skip " << obj->getFullName() << '.' << (subname?subname:"")
241  << ", hidden inside geo group");
242  return;
243  }
244  if(childRoot->findChild(childVp->getRoot())<0) {
245  FC_LOG("cannot find '" << childVp->getObject()->getFullName()
246  << "' in geo group '" << grp->getNameInDocument() << "'");
247  break;
248  }
249  groups.push_back(grpVp);
250  }
251 
252  SoTempPath path(10);
253  path.ref();
254 
255  for(auto it=groups.rbegin();it!=groups.rend();++it) {
256  auto grpVp = *it;
257  path.append(grpVp->getRoot());
258  path.append(grpVp->getModeSwitch());
259  path.append(grpVp->getChildRoot());
260  }
261 
262  SoDetail *det = nullptr;
263  if(vp->getDetailPath(subname, &path,true,det) && path.getLength()) {
264  auto node = new SoFCPathAnnotation;
265  node->setPath(&path);
266  pcGroup->addChild(node);
267  if(det) {
268  SoSelectionElementAction action(SoSelectionElementAction::Append,true);
269  action.setElement(det);
270  SoTempPath tmpPath(path.getLength()+2);
271  tmpPath.ref();
272  tmpPath.append(pcGroup);
273  tmpPath.append(node);
274  tmpPath.append(&path);
275  action.apply(&tmpPath);
276  tmpPath.unrefNoDelete();
277  node->setDetail(det);
278  det = nullptr;
279  }
280  FC_LOG("add annotation " << Reason.Type << " " << key);
281  objs[key.c_str()] = node;
282  }
283  delete det;
284  path.unrefNoDelete();
285 }
