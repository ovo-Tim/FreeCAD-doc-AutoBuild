117 void TaskChamferParameters::setUpUI(PartDesign::Chamfer* pcChamfer)
118 {
119  const int index = pcChamfer->ChamferType.getValue();
120  ui->chamferType->setCurrentIndex(index);
121 
122  ui->flipDirection->setEnabled(index != 0); // Enable if type is not "Equal distance"
123  ui->flipDirection->setChecked(pcChamfer->FlipDirection.getValue());
124 
125  ui->chamferSize->setUnit(Base::Unit::Length);
126  ui->chamferSize->setMinimum(0);
127  ui->chamferSize->setValue(pcChamfer->Size.getValue());
128  ui->chamferSize->bind(pcChamfer->Size);
129  ui->chamferSize->selectNumber();
130 
131  ui->chamferSize2->setUnit(Base::Unit::Length);
132  ui->chamferSize2->setMinimum(0);
133  ui->chamferSize2->setValue(pcChamfer->Size2.getValue());
134  ui->chamferSize2->bind(pcChamfer->Size2);
135 
136  ui->chamferAngle->setUnit(Base::Unit::Angle);
137  ui->chamferAngle->setMinimum(pcChamfer->Angle.getMinimum());
138  ui->chamferAngle->setMaximum(pcChamfer->Angle.getMaximum());
139  ui->chamferAngle->setValue(pcChamfer->Angle.getValue());
140  ui->chamferAngle->bind(pcChamfer->Angle);
141 
142  ui->stackedWidget->setFixedHeight(ui->chamferSize2->sizeHint().height());
143 
144  QFontMetrics fm(ui->typeLabel->font());
145  int minWidth = Gui::QtTools::horizontalAdvance(fm, ui->typeLabel->text());
146  minWidth = std::max<int>(minWidth, Gui::QtTools::horizontalAdvance(fm, ui->sizeLabel->text()));
147  minWidth = std::max<int>(minWidth, Gui::QtTools::horizontalAdvance(fm, ui->size2Label->text()));
148  minWidth = std::max<int>(minWidth, Gui::QtTools::horizontalAdvance(fm, ui->angleLabel->text()));
149  minWidth = minWidth + 5; // spacing
150  ui->typeLabel->setMinimumWidth(minWidth);
151  ui->sizeLabel->setMinimumWidth(minWidth);
152  ui->size2Label->setMinimumWidth(minWidth);
153  ui->angleLabel->setMinimumWidth(minWidth);
154 }

304 void TaskChamferParameters::apply()
305 {
306  auto chamfer = getObject<PartDesign::Chamfer>();
307 
308  const int chamfertype = chamfer->ChamferType.getValue();
309 
310  switch (chamfertype) {
311 
312  case 0: // "Equal distance"
313  ui->chamferSize->apply();
314  break;
315  case 1: // "Two distances"
316  ui->chamferSize->apply();
317  ui->chamferSize2->apply();
318  break;
319  case 2: // "Distance and Angle"
320  ui->chamferSize->apply();
321  ui->chamferAngle->apply();
322  break;
323  }
324 
325  // Alert user if he created an empty feature
326  if (ui->listWidgetReferences->count() == 0) {
327  Base::Console().Warning(tr("Empty chamfer created !\n").toStdString().c_str());
328  }
329 }
