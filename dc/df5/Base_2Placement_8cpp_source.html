55 {

79 DualQuat Placement::toDualQuaternion() const
80 {
81  DualQuat posqq(_pos.x, _pos.y, _pos.z, 0.0);
82  DualQuat rotqq;
83  _rot.getValue(rotqq.x.re, rotqq.y.re, rotqq.z.re, rotqq.w.re);
84  DualQuat ret(rotqq, 0.5 * posqq * rotqq);
85  return ret;
