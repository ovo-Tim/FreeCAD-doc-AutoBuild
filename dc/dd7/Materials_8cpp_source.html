201 void MaterialProperty::setType(const QString& type)
202 {

289 void MaterialProperty::setValue(const QString& value)
290 {
291  if (_valuePtr->getType() == MaterialValue::Boolean) {
292  setBoolean(value);
293  }
294  else if (_valuePtr->getType() == MaterialValue::Integer) {

305  // This value can't be directly assigned

313  e.what(),
314  value.toStdString().c_str());

316  setString(value);
317  }
318  }
319  else {
320  setString(value);

638 void Material::addPhysical(const QString& uuid)
639 {

641  return;
642  }
643 
644  ModelManager manager;
645 
646  try {
647  auto model = manager.getModel(uuid);
648 
649  auto& inheritance = model->getInheritance();
650  for (auto& it : inheritance) {
651  // Inherited models may already have the properties, so just
652  // remove the uuid
653  removeUUID(_physicalUuids, it);
