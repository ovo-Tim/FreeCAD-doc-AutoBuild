145 void DlgInspectMaterial::update(std::vector<Gui::ViewProvider*>& views)
146 {
147  clipboardText = QLatin1String("");
148  clipboardIndent = 0;
149  App::Document* doc = App::GetApplication().getActiveDocument();
150  if (doc) {
151  appendClip(tr("Document: ") + QString::fromUtf8(doc->Label.getValue()));
152  ui->editDocument->setText(QString::fromUtf8(doc->Label.getValue()));
153 
154  if (views.size() == 1) {
155  auto view = dynamic_cast<Gui::ViewProviderDocumentObject*>(views[0]);
156  if (!view) {
157  return;
158  }
159  auto* obj = view->getObject();
160  if (!obj) {
161  return;
162  }
163  auto* labelProp = dynamic_cast<App::PropertyString*>(obj->getPropertyByName("Label"));
164  if (labelProp) {
165  ui->editObjectLabel->setText(QString::fromUtf8(labelProp->getValue()));
166  appendClip(tr("Label: ") + QString::fromUtf8(labelProp->getValue()));
167  }
168  else {
169  ui->editObjectLabel->setText(QLatin1String(""));
170  }
171  ui->editObjectName->setText(QLatin1String(obj->getNameInDocument()));
172  appendClip(tr("Internal Name: ") + QString::fromUtf8(obj->getNameInDocument()));
173 
174  auto subElement = Gui::Selection().getSelectionEx();
175  if (subElement.size() > 0) {
176  auto& subObject = subElement[0];
177  if (subObject.getSubNames().size() > 0) {
178  ui->editSubShape->setText(QString::fromStdString(subObject.getSubNames()[0]));
179  }
180  else {
181  ui->editSubShape->setText(QLatin1String(""));
182  }
183  }
184  else {
185  ui->editSubShape->setText(QLatin1String(""));
186  }
187 
188  auto subShapeType = QString::fromUtf8(obj->getTypeId().getName());
189  subShapeType.remove(subShapeType.indexOf(QLatin1String("::")), subShapeType.size());
190  appendClip(tr("Type: ") + subShapeType);
191  ui->editSubShapeType->setText(subShapeType);
192  appendClip(tr("TypeID: ") + QString::fromUtf8(obj->getTypeId().getName()));
193  ui->editShapeType->setText(QString::fromUtf8(obj->getTypeId().getName()));
194 
195  if (labelProp && QString::fromUtf8(labelProp->getValue()).size() > 0) {
196  auto* prop = dynamic_cast<Materials::PropertyMaterial*>(
197  obj->getPropertyByName("ShapeMaterial"));
198  if (prop) {
199  updateMaterialTree(prop->getValue());
200  }
201  }
202  }
203  }
204 }

352 
353  indent();
354  auto parentUUID = material.getParentUUID();
