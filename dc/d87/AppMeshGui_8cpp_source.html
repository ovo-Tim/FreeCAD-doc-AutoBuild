76  : Py::ExtensionModule<Module>("MeshGui")

79  initialize("This module is the MeshGui module."); // register with Python
80  }
81 
82 private:
83  Py::Object convertToSTL(const Py::Tuple& args)
84  {
85  char* inname {};
86  char* outname {};
87  if (!PyArg_ParseTuple(args.ptr(), "etet", "utf-8", &inname, "utf-8", &outname)) {
88  throw Py::Exception();
89  }
90  std::string inputName = std::string(inname);
91  PyMem_Free(inname);
92  std::string outputName = std::string(outname);
93  PyMem_Free(outname);
94 
95  bool ok = false;
96  SoInput in;
97  if (in.openFile(inputName.c_str())) {
98  SoSeparator* node = SoDB::readAll(&in);
99  if (node) {
100  node->ref();
101  SoSTLFileKit* stlKit = new SoSTLFileKit();
102  stlKit->ref();
103  ok = stlKit->readScene(node);
104  stlKit->writeFile(outputName.c_str());
105  stlKit->unref();
106  node->unref();
107  }
108  }
109 
110  return Py::Boolean(ok); // NOLINT
111  }
112 };
