51  def exec(self) -> Tuple[str, str, str]:
52  """Run the dialog, and return a tuple of the person's record type, their name, and their
53  email address. Email may be None. If the others are None it's because the user cancelled
54  the interaction."""
55  result = self.dialog.exec()
56  if result == QDialog.Accepted:
57  return (
58  self.dialog.comboBox.currentData(),
59  self.dialog.nameLineEdit.text(),
60  self.dialog.emailLineEdit.text(),
61  )
62  return "", "", ""

69  index = 0
70  self.dialog.comboBox.setCurrentIndex(index)
