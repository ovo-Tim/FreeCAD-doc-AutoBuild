87 static std::pair<int, int> selectedMinMaxRows(QModelIndexList list)
88 {
89  int min = std::numeric_limits<int>::max();
90  int max = 0;
91  for (const auto& item : list) {
92  int row = item.row();
93  min = std::min(row, min);
94  max = std::max(row, max);
95  }
96  return {min, max};
97 }

270 std::vector<Range> SheetTableView::selectedRanges() const
271 {
272  std::vector<Range> result;
273 
274  if (!sheet->getCells()->hasSpan()) {
275  for (const auto& sel : selectionModel()->selection()) {
276  result.emplace_back(sel.top(), sel.left(), sel.bottom(), sel.right());
277  }
278  }
279  else {
280  // If there is spanning cell, QItemSelection returned by
281  // QTableView::selection() does not merge selected indices into ranges.
282  // So we have to do it by ourselves. Qt records selection in the order
283  // of column first and then row.
284  //
285  // Note that there will always be ambiguous cases with the available
286  // information, where multiple user selected ranges are merged
287  // together. For example, consecutive single column selections that
288  // form a rectangle will be merged together, but single row selections
289  // will not be merged.
290  for (const auto& sel : selectionModel()->selection()) {
291  if (!result.empty() && sel.bottom() == sel.top() && sel.right() == sel.left()) {
292  auto& last = result.back();
293  if (last.colCount() == 1 && last.from().col() == sel.left()
294  && sel.top() == last.to().row() + 1) {
295  // This is the case of rectangle selection. We keep
296  // accumulating the last column, and try to merge the
297  // column to previous range whenever possible.
298  last = Range(last.from(), CellAddress(sel.top(), sel.left()));
299  if (result.size() > 1) {
300  auto& secondLast = result[result.size() - 2];
301  if (secondLast.to().col() + 1 == last.to().col()
302  && secondLast.from().row() == last.from().row()
303  && secondLast.rowCount() == last.rowCount()) {
304  secondLast = Range(secondLast.from(), last.to());
305  result.pop_back();
306  }
307  }
308  continue;
309  }
310  else if (last.rowCount() == 1 && last.from().row() == sel.top()
311  && last.to().col() + 1 == sel.left()) {
312  // This is the case of single row selection
313  last = Range(last.from(), CellAddress(sel.top(), sel.left()));
314  continue;
315  }
316  }
317  result.emplace_back(sel.top(), sel.left(), sel.bottom(), sel.right());
318  }
319  }
320  return result;
321 }

328 void SheetTableView::insertRows()
329 {
330  assert(sheet);
331 
332  QModelIndexList rows = selectionModel()->selectedRows();
333  std::vector<int> sortedRows;
334 
335  /* Make sure rows are sorted in ascending order */
336  for (const auto& it : rows) {
337  sortedRows.push_back(it.row());
338  }
339  std::sort(sortedRows.begin(), sortedRows.end());
340 
341  /* Insert rows */
342  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Insert rows"));
343  std::vector<int>::const_reverse_iterator it = sortedRows.rbegin();
344  while (it != sortedRows.rend()) {
345  int prev = *it;
346  int count = 1;
347 
348  /* Collect neighbouring rows into one chunk */
349  ++it;
350  while (it != sortedRows.rend()) {
351  if (*it == prev - 1) {
352  prev = *it;
353  ++count;
354  ++it;
355  }
356  else {
357  break;
358  }
359  }
360 
361  Gui::cmdAppObjectArgs(sheet, "insertRows('%s', %d)", rowName(prev).c_str(), count);
362  }
363  Gui::Command::commitCommand();
364  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
365 }

483 void SheetTableView::updateCellSpan()
484 {
485  int rows, cols;
486 
487  // Unspan first to avoid overlap
488  for (const auto& addr : spanChanges) {
489  if (rowSpan(addr.row(), addr.col()) > 1 || columnSpan(addr.row(), addr.col()) > 1) {
490  setSpan(addr.row(), addr.col(), 1, 1);
491  }
492  }
493 
494  for (const auto& addr : spanChanges) {
495  sheet->getSpans(addr, rows, cols);
496  if (rows > 1 || cols > 1) {
497  setSpan(addr.row(), addr.col(), rows, cols);
498  }
499  }
500  spanChanges.clear();
501 }

813 void SheetTableView::finishEditWithMove(int keyPressed,
814  Qt::KeyboardModifiers modifiers,
815  bool handleTabMotion)
816 {
817  // A utility lambda for finding the beginning and ending of data regions
818  auto scanForRegionBoundary = [this](int& r, int& c, int dr, int dc) {
819  auto startAddress = CellAddress(r, c);
820  auto startCell = sheet->getCell(startAddress);
821  bool startedAtEmptyCell = startCell ? !startCell->isUsed() : true;
822  const int maxRow = this->model()->rowCount() - 1;
823  const int maxCol = this->model()->columnCount() - 1;
824  while (c + dc >= 0 && r + dr >= 0 && c + dc <= maxCol && r + dr <= maxRow) {
825  r += dr;
826  c += dc;
827  auto cell = sheet->getCell(CellAddress(r, c));
828  auto cellIsEmpty = cell ? !cell->isUsed() : true;
829  if (cellIsEmpty && !startedAtEmptyCell) {
830  // Don't stop at the empty cell, stop at the last non-empty cell
831  r -= dr;
832  c -= dc;
833  break;
834  }
835  else if (!cellIsEmpty && startedAtEmptyCell) {
836  break;
837  }
838  }
839  if (r == startAddress.row() && c == startAddress.col()) {
840  // Always move at least one cell:
841  r += dr;
842  c += dc;
843  }
844  r = std::max(0, std::min(r, maxRow));
845  c = std::max(0, std::min(c, maxCol));
846  };
847 
848  int targetRow = currentIndex().row();
849  int targetColumn = currentIndex().column();
850  int colSpan;
851  int rowSpan;
852  sheet->getSpans(CellAddress(targetRow, targetColumn), rowSpan, colSpan);
853  switch (keyPressed) {
854  case Qt::Key_Return:
855  case Qt::Key_Enter:
856  if (modifiers == Qt::NoModifier) {
857  targetRow += rowSpan;
858  targetColumn -= tabCounter;
859  }
860  else if (modifiers == Qt::ShiftModifier) {
861  targetRow -= 1;
862  targetColumn -= tabCounter;
863  }
864  else {
865  // For an unrecognized modifier, just go down
866  targetRow += rowSpan;
867  }
868  tabCounter = 0;
869  break;
870 
871  case Qt::Key_Home:
872  // Home: row 1, same column
873  // Ctrl-Home: row 1, column 1
874  targetRow = 0;
875  if (modifiers == Qt::ControlModifier) {
876  targetColumn = 0;
877  }
878  tabCounter = 0;
879  break;
880 
881  case Qt::Key_End: {
882  // End should take you to the last occupied cell in the current column
883  // Ctrl-End takes you to the last cell in the sheet
884  auto usedCells = sheet->getCells()->getNonEmptyCells();
885  for (const auto& cell : usedCells) {
886  if (modifiers == Qt::NoModifier) {
887  if (cell.col() == targetColumn) {
888  targetRow = std::max(targetRow, cell.row());
889  }
890  }
891  else if (modifiers == Qt::ControlModifier) {
892  targetRow = std::max(targetRow, cell.row());
893  targetColumn = std::max(targetColumn, cell.col());
894  }
895  }
896  tabCounter = 0;
897  break;
898  }
899 
900  case Qt::Key_Left:
901  if (targetColumn == 0) {
902  break; // Nothing to do, we're already in the first column
903  }
904  if (modifiers == Qt::NoModifier || modifiers == Qt::ShiftModifier) {
905  targetColumn--;
906  }
907  else if (modifiers == Qt::ControlModifier
908  || modifiers == (Qt::ControlModifier | Qt::ShiftModifier)) {
909  scanForRegionBoundary(targetRow, targetColumn, 0, -1);
910  }
911  else {
912  targetColumn--; // Unrecognized modifier combination: default to just moving one
913  // cell
914  }
915  tabCounter = 0;
916  break;
917  case Qt::Key_Right:
918  if (targetColumn >= this->model()->columnCount() - 1) {
919  break; // Nothing to do, we're already in the last column
920  }
921  if (modifiers == Qt::NoModifier || modifiers == Qt::ShiftModifier) {
922  targetColumn += colSpan;
923  }
924  else if (modifiers == Qt::ControlModifier
925  || modifiers == (Qt::ControlModifier | Qt::ShiftModifier)) {
926  scanForRegionBoundary(targetRow, targetColumn, 0, 1);
927  }
928  else {
929  targetColumn +=
930  colSpan; // Unrecognized modifier combination: default to just moving one cell
931  }
932  tabCounter = 0;
933  break;
934  case Qt::Key_Up:
935  if (targetRow == 0) {
936  break; // Nothing to do, we're already in the first column
937  }
938  if (modifiers == Qt::NoModifier || modifiers == Qt::ShiftModifier) {
939  targetRow--;
940  }
941  else if (modifiers == Qt::ControlModifier
942  || modifiers == (Qt::ControlModifier | Qt::ShiftModifier)) {
943  scanForRegionBoundary(targetRow, targetColumn, -1, 0);
944  }
945  else {
946  targetRow--; // Unrecognized modifier combination: default to just moving one cell
947  }
948  tabCounter = 0;
949  break;
950  case Qt::Key_Down:
951  if (targetRow >= this->model()->rowCount() - 1) {
952  break; // Nothing to do, we're already in the last row
953  }
954  if (modifiers == Qt::NoModifier || modifiers == Qt::ShiftModifier) {
955  targetRow += rowSpan;
956  }
957  else if (modifiers == Qt::ControlModifier
958  || modifiers == (Qt::ControlModifier | Qt::ShiftModifier)) {
959  scanForRegionBoundary(targetRow, targetColumn, 1, 0);
960  }
961  else {
962  targetRow +=
963  rowSpan; // Unrecognized modifier combination: default to just moving one cell
964  }
965  tabCounter = 0;
966  break;
967  case Qt::Key_Tab:
968  if (modifiers == Qt::NoModifier) {
969  tabCounter++;
970  if (handleTabMotion) {
971  targetColumn += colSpan;
972  }
973  }
974  else if (modifiers == Qt::ShiftModifier) {
975  tabCounter = 0;
976  if (handleTabMotion) {
977  targetColumn--;
978  }
979  }
980  break;
981  case Qt::Key_Backtab:
982  if (modifiers == Qt::NoModifier) {
983  targetColumn--;
984  }
985  tabCounter = 0;
986  break;
987  default:
988  break;
989  }
990 
991  if (this->sheet->isMergedCell(CellAddress(targetRow, targetColumn))) {
992  auto anchor = this->sheet->getAnchor(CellAddress(targetRow, targetColumn));
993  targetRow = anchor.row();
994  targetColumn = anchor.col();
995  }
996 
997  // Overflow/underflow protection:
998  const int maxRow = this->model()->rowCount() - 1;
999  const int maxCol = this->model()->columnCount() - 1;
1000  targetRow = std::max(0, std::min(targetRow, maxRow));
1001  targetColumn = std::max(0, std::min(targetColumn, maxCol));
1002 
1003  if (!(modifiers & Qt::ShiftModifier) || keyPressed == Qt::Key_Tab || keyPressed == Qt::Key_Enter
1004  || keyPressed == Qt::Key_Return) {
1005  // We have to use this method so that Ctrl-modifier combinations don't result in multiple
1006  // selection
1007  this->selectionModel()->setCurrentIndex(model()->index(targetRow, targetColumn),
1008  QItemSelectionModel::ClearAndSelect);
1009  }
1010  else if (modifiers & Qt::ShiftModifier) {
1011  // With shift down, this motion becomes a block selection command, rather than just simple
1012  // motion:
1013  ModifyBlockSelection(targetRow, targetColumn);
1014  }
1015 }

1017 void SheetTableView::ModifyBlockSelection(int targetRow, int targetCol)
1018 {
1019  int startingRow = currentIndex().row();
1020  int startingCol = currentIndex().column();
1021 
1022  // Get the current block selection size:
1023  auto selection = this->selectionModel()->selection();
1024  for (const auto& range : selection) {
1025  if (range.contains(currentIndex())) {
1026  // This range contains the current cell, so it's the one we're going to modify (assuming
1027  // we're at one of the corners)
1028  int rangeMinRow = range.top();
1029  int rangeMaxRow = range.bottom();
1030  int rangeMinCol = range.left();
1031  int rangeMaxCol = range.right();
1032  if ((startingRow == rangeMinRow || startingRow == rangeMaxRow)
1033  && (startingCol == rangeMinCol || startingCol == rangeMaxCol)) {
1034  if (range.contains(model()->index(targetRow, targetCol))) {
1035  // If the range already contains the target cell, then we're making the range
1036  // smaller
1037  if (startingRow == rangeMinRow) {
1038  rangeMinRow = targetRow;
1039  }
1040  if (startingRow == rangeMaxRow) {
1041  rangeMaxRow = targetRow;
1042  }
1043  if (startingCol == rangeMinCol) {
1044  rangeMinCol = targetCol;
1045  }
1046  if (startingCol == rangeMaxCol) {
1047  rangeMaxCol = targetCol;
1048  }
1049  }
1050  else {
1051  // We're making the range bigger
1052  rangeMinRow = std::min(rangeMinRow, targetRow);
1053  rangeMaxRow = std::max(rangeMaxRow, targetRow);
1054  rangeMinCol = std::min(rangeMinCol, targetCol);
1055  rangeMaxCol = std::max(rangeMaxCol, targetCol);
1056  }
1057  QItemSelection oldRange(range.topLeft(), range.bottomRight());
1058  this->selectionModel()->select(oldRange, QItemSelectionModel::Deselect);
1059  QItemSelection newRange(model()->index(rangeMinRow, rangeMinCol),
1060  model()->index(rangeMaxRow, rangeMaxCol));
1061  this->selectionModel()->select(newRange, QItemSelectionModel::Select);
1062  }
1063  break;
1064  }
1065  }
1066 
1067  this->selectionModel()->setCurrentIndex(model()->index(targetRow, targetCol),
1068  QItemSelectionModel::Current);
1069 }
