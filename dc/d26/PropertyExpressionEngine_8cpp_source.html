136 void PropertyExpressionEngine::hasSetValue()
137 {
138  App::DocumentObject* owner = dynamic_cast<App::DocumentObject*>(getContainer());
139  if (!owner || !owner->isAttachedToDocument() || owner->isRestoring()
140  || testFlag(LinkDetached)) {
141  PropertyExpressionContainer::hasSetValue();
142  return;
143  }
144 
145  std::map<App::DocumentObject*, bool> deps;
146  std::vector<std::string> labels;
147  unregisterElementReference();
148  UpdateElementReferenceExpressionVisitor<PropertyExpressionEngine> v(*this);
149  for (auto& e : expressions) {
150  auto expr = e.second.expression;
151  if (expr) {
152  expr->getDepObjects(deps, &labels);
153  if (!restoring) {
154  expr->visit(v);
155  }
156  }
157  }
158  registerLabelReferences(std::move(labels));
159 
160  updateDeps(std::move(deps));
161 
162  if (pimpl) {
163  pimpl->conns.clear();
164  pimpl->propMap.clear();
165  }
166  // check if there is any hidden references
167  bool hasHidden = false;
168  for (auto& v : _Deps) {
169  if (v.second) {
170  hasHidden = true;
171  break;
172  }
173  }
174  if (hasHidden) {
175  if (!pimpl) {
176  pimpl = std::make_unique<Private>();
177  }
178  for (auto& e : expressions) {
179  auto expr = e.second.expression;
180  if (!expr) {
181  continue;
182  }
183  for (auto& dep : expr->getIdentifiers()) {
184  if (!dep.second) {
185  continue;
186  }
187  const ObjectIdentifier& var = dep.first;
188  for (auto& vdep : var.getDep(true)) {
189  auto obj = vdep.first;
190  auto objName = obj->getFullName() + ".";
191  for (auto& propName : vdep.second) {
192  std::string key = objName + propName;
193  auto& propDeps = pimpl->propMap[key];
194  if (propDeps.empty()) {
195  // NOLINTBEGIN
196  if (!propName.empty()) {
197  pimpl->conns.emplace_back(obj->signalChanged.connect(
198  std::bind(&PropertyExpressionEngine::slotChangedProperty,
199  this,
200  sp::_1,
201  sp::_2)));
202  }
203  else {
204  pimpl->conns.emplace_back(obj->signalChanged.connect(
205  std::bind(&PropertyExpressionEngine::slotChangedObject,
206  this,
207  sp::_1,
208  sp::_2)));
209  }
210  // NOLINTEND
211  }
212  propDeps.push_back(e.first);
213  }
214  }
215  }
216  }
217  }

222 void PropertyExpressionEngine::updateHiddenReference(const std::string& key)
223 {
224  if (!pimpl) {
225  return;
226  }
227  auto it = pimpl->propMap.find(key);
228  if (it == pimpl->propMap.end()) {
229  return;
230  }
231  for (auto& var : it->second) {
232  auto it = expressions.find(var);
233  if (it == expressions.end() || it->second.busy) {
234  continue;
235  }
236  Property* myProp = var.getProperty();
237  if (!myProp) {
238  continue;
239  }
240  Base::StateLocker guard(it->second.busy);
241  App::any value;
242  try {
243  value = it->second.expression->getValueAsAny();
244  if (!isAnyEqual(value, myProp->getPathValue(var))) {
245  myProp->setPathValue(var, value);
246  }
247  }
248  catch (Base::Exception& e) {
249  e.ReportException();
250  FC_ERR("Failed to evaluate property binding " << myProp->getFullName()
251  << " on change of " << key);
252  }
253  catch (std::bad_cast&) {
254  FC_ERR("Invalid type '" << value.type().name() << "' in property binding "
255  << myProp->getFullName() << " on change of " << key);
256  }
257  catch (std::exception& e) {
258  FC_ERR(e.what());
259  FC_ERR("Failed to evaluate property binding " << myProp->getFullName()
260  << " on change of " << key);
261  }
262  }
263 }

362 void PropertyExpressionEngine::buildGraphStructures(
363  const ObjectIdentifier& path,
364  const std::shared_ptr<Expression> expression,
365  boost::unordered_map<ObjectIdentifier, int>& nodes,
366  boost::unordered_map<int, ObjectIdentifier>& revNodes,
367  std::vector<Edge>& edges) const
368 {
369  /* Insert target property into nodes structure */
370  if (nodes.find(path) == nodes.end()) {
371  int s = nodes.size();
372 
373  revNodes[s] = path;
374  nodes[path] = s;
375  }
376  else {
377  revNodes[nodes[path]] = path;
378  }
379 
380  /* Insert dependencies into nodes structure */
381  ExpressionDeps deps;
382  if (expression) {
383  deps = expression->getDeps();
384  }
385 
386  for (auto& dep : deps) {
387  for (auto& info : dep.second) {
388  if (info.first.empty()) {
389  continue;
390  }
391  for (auto& oid : info.second) {
392  ObjectIdentifier cPath(oid.canonicalPath());
393  if (nodes.find(cPath) == nodes.end()) {
394  int s = nodes.size();
395  nodes[cPath] = s;
396  }
397  edges.emplace_back(nodes[path], nodes[cPath]);
398  }
399  }
400  }
401 }

409 ObjectIdentifier PropertyExpressionEngine::canonicalPath(const ObjectIdentifier& p) const
410 {
411  DocumentObject* docObj = freecad_dynamic_cast<DocumentObject>(getContainer());
412 
413  // Am I owned by a DocumentObject?
414  if (!docObj) {
415  throw Base::RuntimeError("PropertyExpressionEngine must be owned by a DocumentObject.");
416  }
417 
418  int ptype;
419  Property* prop = p.getProperty(&ptype);
420 
421  // p pointing to a property...?
422  if (!prop) {
423  throw Base::RuntimeError(p.resolveErrorString().c_str());
424  }
425 
426  if (ptype || prop->getContainer() != getContainer()) {
427  return p;
428  }

431  if (prop->isDerivedFrom(PropertyExpressionEngine::classTypeId)) {
432  return p;
433  }
434

436  return p.canonicalPath();

514 void PropertyExpressionEngine::setValue(const ObjectIdentifier& path,
515  std::shared_ptr<Expression> expr)
516 {
517  ObjectIdentifier usePath(canonicalPath(path));
518  const Property* prop = usePath.getProperty();
519 
520  // Try to access value; it should trigger an exception if it is not supported, or if the path is
521  // invalid
522  prop->getPathValue(usePath);
523 
524  // Check if the current expression equals the new one and do nothing if so to reduce unneeded
525  // computations
526  auto it = expressions.find(usePath);
527  if (it != expressions.end()
528  && (expr == it->second.expression
529  || (expr && it->second.expression && expr->isSame(*it->second.expression)))) {
530  return;
531  }
532 
533  if (expr) {
534  std::string error = validateExpression(usePath, expr);
535  if (!error.empty()) {
536  throw Base::RuntimeError(error.c_str());
537  }
538  AtomicPropertyChange signaller(*this);
539  expressions[usePath] = ExpressionInfo(expr);
540  expressionChanged(usePath);
541  signaller.tryInvoke();
542  }
543  else if (it != expressions.end()) {
544  AtomicPropertyChange signaller(*this);
545  expressions.erase(it);
546  expressionChanged(usePath);
547  signaller.tryInvoke();
548  }

583 void PropertyExpressionEngine::buildGraph(const ExpressionMap& exprs,
584  boost::unordered_map<int, ObjectIdentifier>& revNodes,
585  DiGraph& g,
586  ExecuteOption option) const
587 {
588  boost::unordered_map<ObjectIdentifier, int> nodes;
589  std::vector<Edge> edges;
590 
591  // Build data structure for graph
592  for (const auto& expr : exprs) {
593  if (option != ExecuteAll) {
594  auto prop = expr.first.getProperty();
595  if (!prop) {
596  throw Base::RuntimeError("Path does not resolve to a property.");
597  }
598  bool is_output =
599  prop->testStatus(App::Property::Output) || (prop->getType() & App::Prop_Output);
600  if ((is_output && option == ExecuteNonOutput)
601  || (!is_output && option == ExecuteOutput)) {
602  continue;
603  }
604  if (option == ExecuteOnRestore && !prop->testStatus(Property::Transient)
605  && !(prop->getType() & Prop_Transient)
606  && !prop->testStatus(Property::EvalOnRestore)) {
607  continue;
608  }
609  }
610  buildGraphStructures(expr.first, expr.second.expression, nodes, revNodes, edges);
611  }
612 
613  // Create graph
614  g = DiGraph(nodes.size());
615 
616  // Add edges to graph
617  for (const auto& edge : edges) {
618  add_edge(edge.first, edge.second, g);
619  }
620 
621  // Check for cycles
622  bool has_cycle = false;
623  int src = -1;
624  cycle_detector vis(has_cycle, src);
625  depth_first_search(g, visitor(vis));
626 
627  if (has_cycle) {
628  std::string s = revNodes[src].toString() + " reference creates a cyclic dependency.";
629 
630  throw Base::RuntimeError(s.c_str());
631  }
632 }

640 std::vector<App::ObjectIdentifier>
641 PropertyExpressionEngine::computeEvaluationOrder(ExecuteOption option)
642 {
643  std::vector<App::ObjectIdentifier> evaluationOrder;
644  boost::unordered_map<int, ObjectIdentifier> revNodes;
645  DiGraph g;
646 
647  buildGraph(expressions, revNodes, g, option);
648 
649  /* Compute evaluation order for expressions */
650  std::vector<int> c;
651  topological_sort(g, std::back_inserter(c));
652 
653  for (int i : c) {
654  // we return the evaluation order for our properties, not the dependencies
655  // the topo sort will contain node ids for both our props and their deps
656  if (revNodes.find(i) != revNodes.end()) {
657  evaluationOrder.push_back(revNodes[i]);
658  }
659  }
660 
661  return evaluationOrder;
662 }

669 DocumentObjectExecReturn* App::PropertyExpressionEngine::execute(ExecuteOption option,
670  bool* touched)
671 {
672  DocumentObject* docObj = freecad_dynamic_cast<DocumentObject>(getContainer());
673 
674  if (!docObj) {
675  throw Base::RuntimeError("PropertyExpressionEngine must be owned by a DocumentObject.");
676  }
677 
678  if (running) {
679  return DocumentObject::StdReturn;
680  }
681 
682  if (option == ExecuteOnRestore) {
683  bool found = false;
684  for (auto& e : expressions) {
685  auto prop = e.first.getProperty();
686  if (!prop) {
687  continue;
688  }
689  if (prop->testStatus(App::Property::Transient)
690  || (prop->getType() & App::Prop_Transient)
691  || prop->testStatus(App::Property::EvalOnRestore)) {
692  found = true;
693  break;
694  }
695  }
696  if (!found) {
697  return DocumentObject::StdReturn;
698  }

701  /* Resetter class, to ensure that the "running" variable gets set to false, even if
702  * an exception is thrown.
703  */
704 
705  class resetter
706  {
707  public:
708  explicit resetter(bool& b)
709  : _b(b)
710  {
711  _b = true;
712  }
713  ~resetter()
714  {
715  _b = false;
716  }
717 
718  private:
719  bool& _b;
720  };
721 
722  resetter r(running);
723 
724  // Compute evaluation order
725  std::vector<App::ObjectIdentifier> evaluationOrder = computeEvaluationOrder(option);
726  std::vector<ObjectIdentifier>::const_iterator it = evaluationOrder.begin();
727 
728 #ifdef FC_PROPERTYEXPRESSIONENGINE_LOG
729  std::clog << "Computing expressions for " << getName() << std::endl;
730 #endif
731 
732  /* Evaluate the expressions, and update properties */
733  for (; it != evaluationOrder.end(); ++it) {
734 
735  // Get property to update
736  Property* prop = it->getProperty();
737 
738  if (!prop) {
739  throw Base::RuntimeError("Path does not resolve to a property.");
740  }
741 
742  DocumentObject* parent = freecad_dynamic_cast<DocumentObject>(prop->getContainer());
743 
744  /* Make sure property belongs to the same container as this PropertyExpressionEngine */
745  if (parent != docObj) {
746  throw Base::RuntimeError("Invalid property owner.");
747  }
748 
749  /* Set value of property */
750  App::any value;
751  try {
752  // Evaluate expression
753  std::shared_ptr<App::Expression> expression = expressions[*it].expression;
754  if (expression) {
755  value = expression->getValueAsAny();
756 
757  // Enable value comparison for all expression bindings to reduce
758  // unnecessary touch and recompute.
759  //
760  // This optimization is necessary for some hidden reference to
761  // work because it introduce dependency loop. The repeativtive
762  // recompute can be stopped if the expression evaluates the same
763  // value.
764  //
765  // In the future, we can generalize the optimization to all
766  // property modification, i.e. do not touch unless value change
767  //
768  // if (option == ExecuteOnRestore && prop->testStatus(Property::EvalOnRestore))
769  {
770  if (isAnyEqual(value, prop->getPathValue(*it))) {
771  continue;
772  }
773  if (touched) {
774  *touched = true;
775  }
776  }
777  prop->setPathValue(*it, value);
778  }
779  }
780  catch (Base::Exception& e) {
781  std::ostringstream ss;
782  ss << e.what() << std::endl << "in property binding '" << prop->getFullName() << "'";
783  e.setMessage(ss.str());
784  throw;
785  }
786  catch (std::bad_cast&) {
787  std::ostringstream ss;
788  ss << "Invalid type '" << value.type().name() << "'";
789  ss << "\nin property binding '" << prop->getFullName() << "'";
790  throw Base::TypeError(ss.str().c_str());
791  }
792  catch (std::exception& e) {
793  std::ostringstream ss;
794  ss << e.what() << "\nin property binding '" << prop->getFullName() << "'";
795  throw Base::RuntimeError(ss.str().c_str());
796  }
797  }
798  return DocumentObject::StdReturn;
799 }

855 std::string
856 PropertyExpressionEngine::validateExpression(const ObjectIdentifier& path,
857  std::shared_ptr<const Expression> expr) const
858 {
859  std::string error;
860  ObjectIdentifier usePath(canonicalPath(path));
