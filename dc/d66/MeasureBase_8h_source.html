114  static Part::MeasureInfoPtr getMeasureInfo(App::SubObjectT& subObjT)
115  {
116 
117  // Resolve App::Link
118  App::DocumentObject* sub = subObjT.getSubObject();
119  if (!sub) {
120  return nullptr;
121  }
122 
123  if (sub->isDerivedFrom<App::Link>()) {
124  auto link = static_cast<App::Link*>(sub);
125  sub = link->getLinkedObject(true);
126  }
127 
128  // Get the Geometry handler based on the module
129  const char* className = sub->getTypeId().getName();
130  std::string mod = Base::Type::getModuleName(className);
131 
132  auto handler = getGeometryHandler(mod);
133  if (!handler) {
134  Base::Console().Log("MeasureBaseExtendable::getMeasureInfo: No geometry handler "
135  "available for submitted element type");
136  return nullptr;
137  }
138 
139  return handler(subObjT);
140  }
141 
142  static void addGeometryHandlers(const std::vector<std::string>& modules,
143  GeometryHandler callback)
144  {
145  // TODO: this will replace a callback with a later one. Should we check that there isn't
146  // already a handler defined for this module?
147  for (auto& mod : modules) {
148  _mGeometryHandlers[mod] = callback;
149  }
150  }
151 
152 
153  static bool hasGeometryHandler(const std::string& module)
154  {
155  return (_mGeometryHandlers.count(module) > 0);
156  }
157 
158 private:
159  inline static HandlerMap _mGeometryHandlers = MeasureBaseExtendable<T>::HandlerMap();
160 };
