95  import numpy as np

98  face = Gui.Selection.getSelectionEx()[0].SubObjects[0]
99  shape = face.toNurbs()
100  face = shape.Faces[0]
101  nurbs = face.Surface
102  nurbs.setUNotPeriodic()
103  nurbs.setVNotPeriodic()
104  bs = nurbs.toBSpline(1, "C0", "C0", 3, 3, 10)
105  face = bs.toShape()
106  face.tessellate(0.01)
107  flattener = flatmesh.FaceUnwrapper(face)
108  flattener.findFlatNodes(5, 0.99)
109  poles = flattener.interpolateFlatFace(face)
110  num_u_poles = len(bs.getPoles())
111  num_v_poles = len(bs.getPoles()[0])
112  i = 0
113  for u in range(num_u_poles):
114  for v in range(num_v_poles):
115  bs.setPole(u + 1, v + 1, App.Vector(poles[i]))
116  i += 1
117  Part.show(bs.toShape())
