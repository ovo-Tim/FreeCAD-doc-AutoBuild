73 void DlgSettings3DViewPart::onMaxDeviationValueChanged(double vMaxDev)
74 {
75  if (!this->isVisible()) {
76  return;
77  }
78  const double maxDevMinThreshold = 0.01;
79  if (vMaxDev < maxDevMinThreshold && !checkValue) {
80  checkValue = true;
81  QMessageBox::warning(
82  this,
83  tr("Deviation"),
84  tr("Setting a too small deviation causes the tessellation to take longer"
85  " and thus freezes or slows down the GUI."));

110 void DlgSettings3DViewPart::saveSettings()
111 {
112  ui->maxDeviation->onSave();
113  ui->maxAngularDeflection->onSave();
114 
115  // search for Part view providers and apply the new settings
116  std::vector<App::Document*> docs = App::GetApplication().getDocuments();
117  for (auto it : docs) {
118  Gui::Document* doc = Gui::Application::Instance->getDocument(it);
119  std::vector<Gui::ViewProvider*> views =
120  doc->getViewProvidersOfType(ViewProviderPart::getClassTypeId());
121  for (auto view : views) {
122  static_cast<ViewProviderPart*>(view)->reload();
123  }
124  }
