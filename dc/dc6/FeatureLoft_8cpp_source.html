68 std::vector<Part::TopoShape>
69 Loft::getSectionShape(const char *name,
70  App::DocumentObject *obj,
71  const std::vector<std::string> &subs,
72  size_t expected_size)
73 {
74  std::vector<TopoShape> shapes;
75  // Be smart. If part of a sketch is selected, use the entire sketch unless it is a single vertex -
76  // backward compatibility (#16630)
77  auto subName = subs.empty() ? "" : subs.front();
78  auto useEntireSketch = obj->isDerivedFrom(Part::Part2DObject::getClassTypeId()) && subName.find("Vertex") != 0;
79  if (subs.empty() || std::find(subs.begin(), subs.end(), std::string()) != subs.end() || useEntireSketch ) {
80  shapes.push_back(Part::Feature::getTopoShape(obj));
81  if (shapes.back().isNull())
82  FC_THROWM(Part::NullShapeException, "Failed to get shape of "
83  << name << " " << App::SubObjectT(obj, "").getSubObjectFullName(obj->getDocument()->getName()));
84  } else {
85  for (const auto &sub : subs) {
86  shapes.push_back(Part::Feature::getTopoShape(obj, sub.c_str(), /*needSubElement*/true));
87  if (shapes.back().isNull())
88  FC_THROWM(Part::NullShapeException, "Failed to get shape of " << name << " "
89  << App::SubObjectT(obj, sub.c_str()).getSubObjectFullName(obj->getDocument()->getName()));
90  }
91  }
92  auto compound = TopoShape(0).makeElementCompound(shapes, "", TopoShape::SingleShapeCompoundCreationPolicy::returnShape);
93  auto wires = compound.getSubTopoShapes(TopAbs_WIRE);
94  auto edges = compound.getSubTopoShapes(TopAbs_EDGE, TopAbs_WIRE); // get free edges and make wires from it
95  if ( ! edges.empty()) {
96  auto extra = TopoShape(0).makeElementWires(edges).getSubTopoShapes(TopAbs_WIRE);
97  wires.insert(wires.end(), extra.begin(), extra.end());
98  }
99  const char *msg = "Sections need to have the same amount of wires or vertices as the base section";
100  if (!wires.empty()) {
101  if (expected_size && expected_size != wires.size())
102  FC_THROWM(Base::CADKernelError, msg);
103  return wires;
104  }
105  auto vertices = compound.getSubTopoShapes(TopAbs_VERTEX);
106  if (vertices.empty())
107  FC_THROWM(Base::CADKernelError, "Invalid " << name << " shape, expecting either wires or vertices");
108  if (expected_size && expected_size != vertices.size())
109  FC_THROWM(Base::CADKernelError, msg);
110  return vertices;
111 }

113 App::DocumentObjectExecReturn *Loft::execute()
114 {
115  if (onlyHasToRefine()){
116  TopoShape result = refineShapeIfActive(rawShape);
117  Shape.setValue(result);
118  return App::DocumentObject::StdReturn;
119  }
120 
121  std::vector<TopoShape> wires;
122  try {
123  wires = getSectionShape("Profile", Profile.getValue(), Profile.getSubValues());
124  } catch (const Base::Exception& e) {
125  return new App::DocumentObjectExecReturn(e.what());
126  }
127 
128  // if the Base property has a valid shape, fuse the pipe into it
129  TopoShape base;
130  try {
131  base = getBaseTopoShape();
132  } catch (const Base::Exception&) {
133  }
134 
135  auto hasher = getDocument()->getStringHasher();
136 
137  try {
138  // setup the location
139  this->positionByPrevious();
140  auto invObjLoc = this->getLocation().Inverted();
141  if(!base.isNull())
142  base.move(invObjLoc);
143 
144  // build up multisections
145  auto multisections = Sections.getSubListValues();
146  if(multisections.empty())
147  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Loft: At least one section is needed"));
148 
149  std::vector<std::vector<TopoShape>> wiresections;
150  wiresections.reserve(wires.size());
151  for(auto& wire : wires)
152  wiresections.emplace_back(1, wire);
153 
154  for (const auto &subSet : multisections) {
155  int i=0;
156  for (const auto &s : getSectionShape("Section", subSet.first, subSet.second, wiresections.size()))
157  wiresections[i++].push_back(s);
158  }
159 
160  bool closed = Closed.getValue();
161  // invalid for less then 3 sections
162  if (multisections.size() < 2) {
163  closed = false;
164  }
165 
166  TopoShape result(0,hasher);
167  std::vector<TopoShape> shapes;
168 
169  // build all shells
170  std::vector<TopoShape> shells;
171  for (auto &sectionWires : wiresections) {
172  for(auto& wire : sectionWires)
173  wire.move(invObjLoc);
174  shells.push_back(TopoShape(0, hasher).makeElementLoft(
175  sectionWires, Part::IsSolid::notSolid, Ruled.getValue()? Part::IsRuled::ruled : Part::IsRuled::notRuled, closed ? Part::IsClosed::closed : Part::IsClosed::notClosed));
176  }
177 
178  // build the top and bottom face, sew the shell and build the final solid
179  TopoShape front;
180  if (wiresections[0].front().shapeType() != TopAbs_VERTEX) {
181  front = getTopoShapeVerifiedFace();
182  if (front.isNull())
183  return new App::DocumentObjectExecReturn(
184  QT_TRANSLATE_NOOP("Exception", "Loft: Creating a face from sketch failed"));
185  front.move(invObjLoc);
186  }
187 
188  TopoShape back;
189  if (wiresections[0].back().shapeType() != TopAbs_VERTEX) {
190  std::vector<TopoShape> backwires;
191  for(auto& sectionWires : wiresections)
192  backwires.push_back(sectionWires.back());
193  back = TopoShape(0).makeElementFace(backwires);
194  }
195 
196  if (!front.isNull() || !back.isNull()) {
197  BRepBuilderAPI_Sewing sewer;
198  sewer.SetTolerance(Precision::Confusion());
199  if (!front.isNull())
200  sewer.Add(front.getShape());
201  if (!back.isNull())
202  sewer.Add(back.getShape());
203  for(auto& s : shells)
204  sewer.Add(s.getShape());
205 
206  sewer.Perform();
207 
208  if (!front.isNull())
209  shells.push_back(front);
210  if (!back.isNull())
211  shells.push_back(back);
212  // equivalent of the removed: result = result.makeElementShape(sewer,shells);
213  result = result.makeShapeWithElementMap(sewer.SewedShape(), Part::MapperSewing(sewer), shells, Part::OpCodes::Sewing);
214  }
215 
216  if(!result.countSubShapes(TopAbs_SHELL))
217  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Loft: Failed to create shell"));
218  shapes = result.getSubTopoShapes(TopAbs_SHELL);
219 
220  for (auto &s : shapes) {
221  // build the solid
222  s = s.makeElementSolid();
223  BRepClass3d_SolidClassifier SC(s.getShape());
224  SC.PerformInfinitePoint(Precision::Confusion());
225  if ( SC.State() == TopAbs_IN)
226  s.setShape(s.getShape().Reversed(),false);
227  }
228 
229  AddSubShape.setValue(result.makeElementCompound(shapes, nullptr, Part::TopoShape::SingleShapeCompoundCreationPolicy::returnShape));
230 
231  if (shapes.size() > 1)
232  result.makeElementFuse(shapes);
233  else
234  result = shapes.front();
235 
236  if(base.isNull()) {
237  Shape.setValue(getSolid(result));
238  return App::DocumentObject::StdReturn;
239  }
240 
241  result.Tag = -getID();
242  TopoShape boolOp(0,getDocument()->getStringHasher());
243 
244  const char *maker;
245  switch(getAddSubType()) {
246  case Additive:
247  maker = Part::OpCodes::Fuse;
248  break;
249  case Subtractive:
250  maker = Part::OpCodes::Cut;
251  break;
252  default:
253  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Unknown operation type"));
254  }
255  try {
256  boolOp.makeElementBoolean(maker, {base,result});
257  }
258  catch(Standard_Failure&) {
259  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Failed to perform boolean operation"));
260  }
261  boolOp = this->getSolid(boolOp);
262  // lets check if the result is a solid
263  if (boolOp.isNull())
264  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Resulting shape is not a solid"));
265 
266  // store shape before refinement
267  this->rawShape = boolOp;
268  boolOp = refineShapeIfActive(boolOp);
269  boolOp = getSolid(boolOp);
270  if (!isSingleSolidRuleSatisfied(boolOp.getShape())) {
271  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Result has multiple solids: that is not currently supported."));
272  }
273  Shape.setValue(boolOp);
274  return App::DocumentObject::StdReturn;
275  }
276  catch (Standard_Failure& e) {
277  return new App::DocumentObjectExecReturn(e.GetMessageString());
278  }
279  catch (const Base::Exception& e) {
280  return new App::DocumentObjectExecReturn(e.what());
281  }
282  catch (...) {
283  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Loft: A fatal error occurred when making the loft"));
284  }
285 }
