124 void TaskCreateNodeSet::DefineNodesCallback(void* ud, SoEventCallback* n)
125 {
126  // show the wait cursor because this could take quite some time
127  Gui::WaitCursor wc;
128 
129  TaskCreateNodeSet* taskBox = static_cast<TaskCreateNodeSet*>(ud);
130 
131 
132  // When this callback function is invoked we must in either case leave the edit mode
133  Gui::View3DInventorViewer* view = static_cast<Gui::View3DInventorViewer*>(n->getUserData());
134  view->setEditing(false);
135  view->removeEventCallback(SoMouseButtonEvent::getClassTypeId(), DefineNodesCallback, ud);
136  n->setHandled();
137 
138  Gui::SelectionRole role;
139  std::vector<SbVec2f> clPoly = view->getGLPolygon(&role);
140  if (clPoly.size() < 3) {
141  return;
142  }
143  if (clPoly.front() != clPoly.back()) {
144  clPoly.push_back(clPoly.front());
145  }
146 
147  SoCamera* cam = view->getSoRenderManager()->getCamera();
148  SbViewVolume vv = cam->getViewVolume();
149  Gui::ViewVolumeProjection proj(vv);
150  Base::Polygon2d polygon;
151  for (auto it : clPoly) {
152  polygon.Add(Base::Vector2d(it[0], it[1]));
153  }
154 
155  taskBox->DefineNodes(polygon, proj, role == Gui::SelectionRole::Inner ? true : false);
156 }

158 void TaskCreateNodeSet::DefineNodes(const Base::Polygon2d& polygon,
159  const Gui::ViewVolumeProjection& proj,
160  bool inner)
161 {
162  const SMESHDS_Mesh* data = pcObject->FemMesh.getValue<Fem::FemMeshObject*>()
163  ->FemMesh.getValue()
164  .getSMesh()
165  ->GetMeshDS();
166 
167  SMDS_NodeIteratorPtr aNodeIter = data->nodesIterator();
168  Base::Vector3f pt2d;
169 
170  if (!ui->checkBox_Add->isChecked()) {
171  tempSet.clear();
172  }
173 
174  while (aNodeIter->more()) {
175  const SMDS_MeshNode* aNode = aNodeIter->next();
176  Base::Vector3f vec(aNode->X(), aNode->Y(), aNode->Z());
177  pt2d = proj(vec);
178  if (polygon.Contains(Base::Vector2d(pt2d.x, pt2d.y)) == inner) {
179  tempSet.insert(aNode->GetID());
180  }
181  }
182 
183  MeshViewProvider->setHighlightNodes(tempSet);
184 }

186 void TaskCreateNodeSet::onSelectionChanged(const Gui::SelectionChanges& msg)
187 {
188  if (selectionMode == none) {
189  return;
190  }
191 
192  if (msg.Type == Gui::SelectionChanges::AddSelection) {
193  std::string subName(msg.pSubName);
194  unsigned int i = 0;
195  for (; i < subName.size(); i++) {
196  if (msg.pSubName[i] == 'F') {
197  break;
198  }
199  }
200 
201  int elem = atoi(subName.substr(4).c_str());
202  int face = atoi(subName.substr(i + 1).c_str());
203 
204  tempSet.clear();
205 
206  Base::Console().Message("Picked Element:%i Face:%i\n", elem, face);
207 
208 
209  if (!ui->checkBox_Add->isChecked()) {
210  std::set<long> tmp = pcObject->FemMesh.getValue<Fem::FemMeshObject*>()
211  ->FemMesh.getValue()
212  .getSurfaceNodes(elem, face);
213  tempSet.insert(tmp.begin(), tmp.end());
214  }
215  else {
216  tempSet = pcObject->FemMesh.getValue<Fem::FemMeshObject*>()
217  ->FemMesh.getValue()
218  .getSurfaceNodes(elem, face);
219  }
220 
221  selectionMode = none;
222  Gui::Selection().rmvSelectionGate();
223 
224  MeshViewProvider->setHighlightNodes(tempSet);
225  }
226 }
