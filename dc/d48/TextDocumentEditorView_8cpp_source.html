148 bool TextDocumentEditorView::onHasMsg(const char* msg) const
149 {
150  // don't allow any actions if the editor is being closed
151  if (aboutToClose) {
152  return false;
153  }
154 
155  if (strcmp(msg, "Save") == 0) {
156  return true;
157  }
158  if (strcmp(msg, "Cut") == 0) {
159  return (!getEditor()->isReadOnly() && getEditor()->textCursor().hasSelection());
160  }
161  if (strcmp(msg, "Copy") == 0) {
162  return (getEditor()->textCursor().hasSelection());
163  }
164  if (strcmp(msg, "Paste") == 0) {
165  if (getEditor()->isReadOnly()) {
166  return false;
167  }
168  QClipboard* cb = QApplication::clipboard();
169  QString text = cb->text();
170  return !text.isEmpty();
171  }
172  if (strcmp(msg, "Undo") == 0) {
173  return (getEditor()->document()->isUndoAvailable());
174  }
175  if (strcmp(msg, "Redo") == 0) {
176  return (getEditor()->document()->isRedoAvailable());
177  }
178  return false;
179 }

181 bool TextDocumentEditorView::onMsg(const char* msg, const char** output)
182 {
183  Q_UNUSED(output)
184 
185  // don't allow any actions if the editor is being closed
186  if (aboutToClose) {
187  return false;
188  }
189 
190  if (strcmp(msg, "Save") == 0) {
191  saveToObject();
192  getGuiDocument()->save();
193  return true;
194  }
195  if (strcmp(msg, "Cut") == 0) {
196  getEditor()->cut();
197  return true;
198  }
199  if (strcmp(msg, "Copy") == 0) {
200  getEditor()->copy();
201  return true;
202  }
203  if (strcmp(msg, "Paste") == 0) {
204  getEditor()->paste();
205  return true;
206  }
207  if (strcmp(msg, "Undo") == 0) {
208  getEditor()->undo();
209  return true;
210  }
211  if (strcmp(msg, "Redo") == 0) {
212  getEditor()->redo();
213  return true;
214  }
215  return false;
216 }
