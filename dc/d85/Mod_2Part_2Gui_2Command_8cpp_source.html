235 bool checkForSolids(const TopoDS_Shape& shape)
236 {
237  TopExp_Explorer xp;
238  xp.Init(shape, TopAbs_FACE, TopAbs_SHELL);
239  if (xp.More()) {
240  return false;
241  }
242  xp.Init(shape, TopAbs_WIRE, TopAbs_FACE);
243  if (xp.More()) {
244  return false;
245  }
246  xp.Init(shape, TopAbs_EDGE, TopAbs_WIRE);
247  if (xp.More()) {
248  return false;
249  }
250  xp.Init(shape, TopAbs_VERTEX, TopAbs_EDGE);
251  if (xp.More()) {
252  return false;
253  }
254 
255  return true;
256 }

313  getSelection().getSelectionEx(nullptr, App::DocumentObject::getClassTypeId(), Gui::ResolveMode::FollowLink);
