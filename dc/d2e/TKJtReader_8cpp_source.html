58 void TKJtReader::open(const std::string& filename)
59 {
60  clear();
61 
62  Base::FileInfo file(filename);
63  jtDir = TCollection_ExtendedString(TCollection_AsciiString((file.dirPath() + '/').c_str()));
64 
65  TCollection_ExtendedString aFileName(filename.c_str());
66  Handle(JtData_Model) aModel = new JtData_Model(aFileName);
67 
68  if (!aModel.IsNull()) {
69  Handle(JtNode_Partition) aNode = aModel->Init();
70  if (!aNode.IsNull()) {
71  rootNode = aNode;
72 
73  builder.addHeader();
74  builder.beginSeparator();
75  Base::ShapeHintsItem shapeHints;
76  shapeHints.setShapeType(Base::ShapeType::Type::UnknownShapeType);
77  shapeHints.setVertexOrdering(Base::VertexOrdering::Ordering::CounterClockwise);
78  builder.addNode(shapeHints);
79  traverseGraph(aNode);
80  builder.endSeparator();
81  rootNode.Nullify();
82  }
83  }
84 }

91 void TKJtReader::readMaterialAttribute(const Handle(JtAttribute_Material) & aMaterial)
92 {
93  const Jt_F32* ambient = aMaterial->AmbientColor();
94  const Jt_F32* diffuse = aMaterial->DiffuseColor();
95  const Jt_F32* specular = aMaterial->SpecularColor();
96  // const Jt_F32* emissive = aMaterial->EmissionColor();
97  Jt_F32 shininess = aMaterial->Shininess() / 100.0F;
98 
99  // NOLINTBEGIN
100  Base::MaterialItem item;
101  item.setAmbientColor({Base::ColorRGB(ambient[0], ambient[1], ambient[2])});
102  item.setDiffuseColor({Base::ColorRGB(diffuse[0], diffuse[1], diffuse[2])});
103  item.setSpecularColor({Base::ColorRGB(specular[0], specular[1], specular[2])});
104  // Ignore the emissive color as there are sometimes some weird values
105  // item.setEmissiveColor({Base::ColorRGB(emissive[0], emissive[1], emissive[2])});
106  item.setShininess({shininess});
107  // NOLINTEND
108 
109  builder.addNode(item);
110 }

112 void TKJtReader::readTransformAttribute(const Handle(JtAttribute_GeometricTransform) & aTransform)
113 {
114  gp_Trsf trsf;
115  aTransform->GetTrsf(trsf);
116  gp_Mat mat = trsf.VectorialPart();
117  gp_XYZ xyz = trsf.TranslationPart();
118  Base::Matrix4D mtrx;
119  // NOLINTBEGIN
120  mtrx[0][0] = mat(1, 1);
121  mtrx[0][1] = mat(1, 2);
122  mtrx[0][2] = mat(1, 3);
123 
124  mtrx[1][0] = mat(2, 1);
125  mtrx[1][1] = mat(2, 2);
126  mtrx[1][2] = mat(2, 3);
127 
128  mtrx[2][0] = mat(3, 1);
129  mtrx[2][1] = mat(3, 2);
130  mtrx[2][2] = mat(3, 3);
131 
132  // set pos vector
133  mtrx[0][3] = xyz.X();
134  mtrx[1][3] = xyz.Y();
135  mtrx[2][3] = xyz.Z();
136  // NOLINTEND
137 
138  transformations.push_back(mtrx);
139 }

141 void TKJtReader::readShapeVertex(const Handle(JtNode_Shape_Vertex) & aShape)
142 {
143  const JtData_Object::VectorOfLateLoads& aLateLoaded = aShape->LateLoads();
144  for (std::size_t index = 0; index < aLateLoaded.Count(); ++index) {
145  Handle(JtData_Object) anObject = aLateLoaded[index]->DefferedObject();
146  if (anObject.IsNull()) {
147  aLateLoaded[index]->Load();
148  anObject = aLateLoaded[index]->DefferedObject();
149  }
150  if (!anObject.IsNull()) {
151  Handle(JtElement_ShapeLOD_TriStripSet) aLOD =
152  Handle(JtElement_ShapeLOD_TriStripSet)::DownCast(anObject);
153  if (!aLOD.IsNull()) {
154  getTriangleStripSet(aLOD);
155  }
156  }
157  }

160 void TKJtReader::getTriangleStripSet(const Handle(JtElement_ShapeLOD_TriStripSet) & aLOD)
161 {
162  const int pointsPerFace = 3;
163  std::vector<Base::Vector3f> points;
164  std::vector<int> faces;
165  const JtElement_ShapeLOD_Vertex::VertexData& vertices = aLOD->Vertices();
166  const JtElement_ShapeLOD_Vertex::IndicesVec& indices = aLOD->Indices();
167  float* data = vertices.Data();
168  // NOLINTBEGIN
169  for (int index = 0; index < indices.Count(); index += 3) {
170  int coordIndex = indices[index] * 3;
171  points.emplace_back(data[coordIndex], data[coordIndex + 1], data[coordIndex + 2]);
172  coordIndex = indices[index + 1] * 3;
173  points.emplace_back(data[coordIndex], data[coordIndex + 1], data[coordIndex + 2]);
174  coordIndex = indices[index + 2] * 3;
175  points.emplace_back(data[coordIndex], data[coordIndex + 1], data[coordIndex + 2]);
176  faces.push_back(pointsPerFace);
177  }
178  // NOLINTEND
179  builder.addNode(Base::Coordinate3Item {points});
180  builder.addNode(Base::FaceSetItem {faces});
181 }

197 void TKJtReader::readGroup(const Handle(JtNode_Group) & aGroup)
198 {
199  if (!transformations.empty()) {
200  builder.addNode(Base::TransformItem {transformations.back()});
201  transformations.pop_back();
202  }
203  const auto& aChildren = aGroup->Children();
204  for (std::size_t aChildIdx = 0; aChildIdx < aChildren.Count(); ++aChildIdx) {
205  Handle(JtNode_Base) aChild = Handle(JtNode_Base)::DownCast(aChildren[aChildIdx]);
206  traverseGraph(aChild);
207  }

223 void TKJtReader::readAttributes(const JtData_Object::VectorOfObjects& attr)
224 {
225  for (std::size_t aAttrIdx = 0; aAttrIdx < attr.Count(); ++aAttrIdx) {
226  Handle(JtData_Object) anAttr = attr[aAttrIdx];
227  if (anAttr->IsKind(Type_JtAttribute_GeometricTransform)) {
228  Handle(JtAttribute_GeometricTransform) aTransform =
229  Handle(JtAttribute_GeometricTransform)::DownCast(anAttr);
230  readTransformAttribute(aTransform);
231  }
232  else if (anAttr->IsKind(Type_JtAttribute_Material)) {
233  Handle(JtAttribute_Material) aMaterial = Handle(JtAttribute_Material)::DownCast(anAttr);
234  readMaterialAttribute(aMaterial);
235  }
236  }

239 void TKJtReader::traverseGraph(const Handle(JtNode_Base) & aNode)
240 {
241  readAttributes(aNode->Attributes());
242 
243  if (aNode->IsKind(Type_JtNode_Partition)) {
244  Handle(JtNode_Partition) aPart = Handle(JtNode_Partition)::DownCast(aNode);
245  builder.beginSeparator();
246  readPartition(aPart);
247  readGroup(aPart);
248  builder.endSeparator();
249  }
250  else if (aNode->IsKind(Type_JtNode_RangeLOD)) {
251  Handle(JtNode_RangeLOD) aRangeLOD = Handle(JtNode_RangeLOD)::DownCast(aNode);
252  builder.beginSeparator();
253  readRangeLOD(aRangeLOD);
254  builder.endSeparator();
255  }
256  else if (aNode->IsKind(Type_JtNode_Group)) {
257  Handle(JtNode_Group) aGroup = Handle(JtNode_Group)::DownCast(aNode);
258  builder.beginSeparator();
259  readGroup(aGroup);
260  builder.endSeparator();
261  }
262  else if (aNode->IsKind(Type_JtNode_Instance)) {
263  Handle(JtNode_Instance) anInstance = Handle(JtNode_Instance)::DownCast(aNode);
264  readInstance(anInstance);
265  }
266  else if (aNode->IsKind(Type_JtNode_Shape_Vertex)) {
267  Handle(JtNode_Shape_Vertex) aShape = Handle(JtNode_Shape_Vertex)::DownCast(aNode);
268  readShapeVertex(aShape);
269  }
