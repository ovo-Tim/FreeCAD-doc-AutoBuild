55  """Calculates the path direction from cut mode and cut side"""
56  # NB: at the time of writing, we need py3.8 compat, thus not using py3.10 pattern machting
57  if mode == "Conventional" and side == "Inside":
58  return "CW"
59  elif mode == "Conventional" and side == "Outside":
60  return "CCW"
61  elif mode == "Climb" and side == "Inside":
62  return "CCW"

64  return "CW"
