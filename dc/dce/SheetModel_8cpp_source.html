88 QVariant formatCellDisplay(QString value, const Cell* cell)
89 {
90  std::string alias;
91  static auto hGrpSpreadsheet =
92  App::GetApplication().GetUserParameter().GetGroup("BaseApp/Preferences/Mod/Spreadsheet");
93  if (cell->getAlias(alias) && hGrpSpreadsheet->GetBool("showAliasName", false)) {
94  QString formatStr = QString::fromStdString(
95  hGrpSpreadsheet->GetASCII("DisplayAliasFormatString", "%V = %A"));
96  if (formatStr.contains(QLatin1String("%V")) || formatStr.contains(QLatin1String("%A"))) {
97  formatStr.replace(QLatin1String("%A"), QString::fromStdString(alias));
98  formatStr.replace(QLatin1String("%V"), value);
99  return QVariant(formatStr);
100  }
101  }
102  return QVariant(value);
103 }

106 QVariant SheetModel::data(const QModelIndex& index, int role) const
107 {
108  static const Cell* emptyCell = new Cell(CellAddress(0, 0), nullptr);
109  int row = index.row();
110  int col = index.column();
111  const Cell* cell = sheet->getCell(CellAddress(row, col));
112 
113  if (!cell) {
114  cell = emptyCell;
115  }
116 
117 // #define DEBUG_DEPS
118 #ifdef DEBUG_DEPS
119  if (role == Qt::ToolTipRole) {
120  QString v;
121 
122  std::set<std::string> deps = sheet->dependsOn(CellAddress(row, col));
123  std::set<std::string> provides;
124 
125  sheet->providesTo(CellAddress(row, col), provides);
126 
127  if (deps.size() > 0) {
128  v += QString::fromUtf8("Depends on:");
129  for (std::set<std::string>::const_iterator i = deps.begin(); i != deps.end(); ++i) {
130  v += QString::fromUtf8("\n\t") + Tools::fromStdString(*i);
131  }
132  v += QString::fromUtf8("\n");
133  }
134  if (provides.size() > 0) {
135  v += QString::fromUtf8("Used by:");
136  for (std::set<std::string>::const_iterator i = provides.begin(); i != provides.end();
137  ++i) {
138  v += QString::fromUtf8("\n\t") + Tools::fromStdString(*i);
139  }
140  v += QString::fromUtf8("\n");
141  }
142  return QVariant(v);
143  }
144 #else
145  if (!cell->hasException() && role == Qt::ToolTipRole) {
146  std::string alias;
147  if (cell->getAlias(alias)) {
148  return QVariant(QString::fromStdString(alias));
149  }
150  }
151 #endif
152 
153  if (cell->hasException()) {
154  switch (role) {
155  case Qt::ToolTipRole: {
156  QString txt(QString::fromStdString(cell->getException()).toHtmlEscaped());
157  return QVariant(QString::fromLatin1("<pre>%1</pre>").arg(txt));
158  }
159  case Qt::DisplayRole: {
160 #ifdef DEBUG_DEPS
161  return QVariant::fromValue(
162  QString::fromUtf8("#ERR: %1").arg(Tools::fromStdString(cell->getException())));
163 #else
164  std::string str;
165  if (cell->getStringContent(str)) {
166  return QVariant::fromValue(QString::fromUtf8(str.c_str()));
167  }
168  return QVariant::fromValue(QString::fromUtf8("#ERR"));
169 #endif
170  }
171  case Qt::ForegroundRole:
172  return QVariant::fromValue(QColor(255.0, 0, 0));
173  case Qt::TextAlignmentRole:
174  return QVariant(Qt::AlignVCenter | Qt::AlignLeft);
175  default:
176  break;
177  }
178  }
179 
180  // Get edit value by querying the sheet
181  if (role == Qt::EditRole || role == Qt::StatusTipRole) {
182  std::string str;
183 
184  if (cell->getStringContent(str)) {
185  return QVariant(QString::fromUtf8(str.c_str()));
186  }
187  return {};
188  }
189 
190  // Get display value as computed property
191  std::string address = CellAddress(row, col).toString();
192  Property* prop = sheet->getPropertyByName(address.c_str());
193 
194  if (role == Qt::BackgroundRole) {
195  Color color;
196 
197  if (cell->getBackground(color)) {
198  return QVariant::fromValue(
199  QColor(255.0 * color.r, 255.0 * color.g, 255.0 * color.b, 255.0 * color.a));
200  }
201  else {
202  std::string alias;
203  if (cell->getAlias(alias)) {
204  return QVariant::fromValue(aliasBgColor);
205  }
206  return {};
207  }
208  }
209 
210  int qtAlignment = 0;
211 
212  int alignment;
213  cell->getAlignment(alignment);
214 
215  if (alignment & Cell::ALIGNMENT_LEFT) {
216  qtAlignment |= Qt::AlignLeft;
217  }
218  if (alignment & Cell::ALIGNMENT_HCENTER) {
219  qtAlignment |= Qt::AlignHCenter;
220  }
221  if (alignment & Cell::ALIGNMENT_RIGHT) {
222  qtAlignment |= Qt::AlignRight;
223  }
224  if (alignment & Cell::ALIGNMENT_TOP) {
225  qtAlignment |= Qt::AlignTop;
226  }
227  if (alignment & Cell::ALIGNMENT_VCENTER) {
228  qtAlignment |= Qt::AlignVCenter;
229  }
230  if (alignment & Cell::ALIGNMENT_BOTTOM) {
231  qtAlignment |= Qt::AlignBottom;
232  }
233 
234  std::set<std::string> style;
235  if (role == Qt::FontRole && cell->getStyle(style)) {
236  QFont f;
237 
238  for (const auto& i : style) {
239  if (i == "bold") {
240  f.setBold(true);
241  }
242  else if (i == "italic") {
243  f.setItalic(true);
244  }
245  else if (i == "underline") {
246  f.setUnderline(true);
247  }
248  }
249 
250  return QVariant::fromValue(f);
251  }
252 
253  auto dirtyCells = sheet->getCells()->getDirty();
254  auto dirty = (dirtyCells.find(CellAddress(row, col)) != dirtyCells.end());
255 
256  if (!prop || dirty) {
257  switch (role) {
258  case Qt::ForegroundRole: {
259  return QColor(0,
260  0,
261  255.0); // TODO: Remove this hardcoded color, replace with preference
262  }
263  case Qt::TextAlignmentRole: {
264  qtAlignment = Qt::AlignHCenter | Qt::AlignVCenter;
265  return QVariant::fromValue(qtAlignment);
266  }
267  case Qt::DisplayRole:
268  if (cell->getExpression()) {
269  std::string str;
270  if (cell->getStringContent(str)) {
271  if (!str.empty() && str[0] == '=') {
272  // If this is a real computed value, indicate that a recompute is
273  // needed before we can display it
274  return QVariant(QLatin1String("#PENDING"));
275  }
276  else {
277  // If it's just a simple value, display the new value, but still
278  // format it as a pending value to indicate to the user that
279  // a recompute is needed
280  return QVariant(QString::fromUtf8(str.c_str()));
281  }
282  }
283  }
284  return {};
285  default:
286  return {};
287  }
288  }
289  else if (prop->isDerivedFrom(App::PropertyString::getClassTypeId())) {
290  /* String */

292 
293  switch (role) {

303  }
304  }

308  }

311  qtAlignment &= ~(Qt::AlignLeft | Qt::AlignHCenter | Qt::AlignRight);
312  qtAlignment |= Qt::AlignLeft;
313  }

315  qtAlignment &= ~(Qt::AlignTop | Qt::AlignVCenter | Qt::AlignBottom);

319  }

321  return {};
322  }
323  }

325  /* Number */
326  const App::PropertyQuantity* floatProp = static_cast<const App::PropertyQuantity*>(prop);
327 
328  switch (role) {
329  case Qt::ForegroundRole: {
330  Color color;
331 
332  if (cell->getForeground(color)) {
333  return QVariant::fromValue(
334  QColor(255.0 * color.r, 255.0 * color.g, 255.0 * color.b, 255.0 * color.a));
335  }
336  else {
337  if (floatProp->getValue() < 0) {
338  return QVariant::fromValue(QColor(negativeFgColor));
339  }
340  else {
341  return QVariant::fromValue(QColor(positiveFgColor));
342  }
343  }
344  }
345  case Qt::TextAlignmentRole: {

347  qtAlignment &= ~(Qt::AlignLeft | Qt::AlignHCenter | Qt::AlignRight);

352  qtAlignment |= Qt::AlignVCenter;
353  }
354  return QVariant::fromValue(qtAlignment);
355  }

357  QString v;
358  const Base::Unit& computedUnit = floatProp->getUnit();
359  DisplayUnit displayUnit;
360

362  if (cell->getDisplayUnit(displayUnit)) {

384  }

387  }

513 {
