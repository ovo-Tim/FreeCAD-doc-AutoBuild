66 App::DocumentObjectExecReturn* HarmonizeNormals::execute()
67 {
68  App::DocumentObject* link = Source.getValue();
69  if (!link) {
70  return new App::DocumentObjectExecReturn("No mesh linked");
71  }
72  App::Property* prop = link->getPropertyByName("Mesh");
73  if (prop && prop->is<Mesh::PropertyMeshKernel>()) {
74  Mesh::PropertyMeshKernel* kernel = static_cast<Mesh::PropertyMeshKernel*>(prop);
75  std::unique_ptr<MeshObject> mesh(new MeshObject);
76  *mesh = kernel->getValue();
77  mesh->harmonizeNormals();
78  this->Mesh.setValuePtr(mesh.release());
79  }
80 
81  return App::DocumentObject::StdReturn;
82 }
