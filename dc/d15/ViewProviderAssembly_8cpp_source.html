169 bool ViewProviderAssembly::canDragObjectToTarget(App::DocumentObject* obj,
170  App::DocumentObject* target) const
171 {
172  // If a solid is removed from the assembly, its joints need to be removed.
173  bool prompted = false;
174  auto* assemblyPart = getObject<AssemblyObject>();
175 
176  // If target is null then it's being dropped on a doc.
177  if (target && assemblyPart->hasObject(target)) {
178  // If the obj stays in assembly then its ok.
179  return true;
180  }
181 
182  // Combine the joints and groundedJoints vectors into one for simplicity.
183  std::vector<App::DocumentObject*> allJoints = assemblyPart->getJoints();
184  std::vector<App::DocumentObject*> groundedJoints = assemblyPart->getGroundedJoints();
185  allJoints.insert(allJoints.end(), groundedJoints.begin(), groundedJoints.end());
186 
187  for (auto joint : allJoints) {
188  // getLinkObjFromProp returns nullptr if the property doesn't exist.
189  App::DocumentObject* part1 = getMovingPartFromRef(assemblyPart, joint, "Reference1");
190  App::DocumentObject* part2 = getMovingPartFromRef(assemblyPart, joint, "Reference2");
191  App::DocumentObject* obj1 = getObjFromRef(joint, "Reference1");
192  App::DocumentObject* obj2 = getObjFromRef(joint, "Reference2");
193  App::DocumentObject* obj3 = getObjFromProp(joint, "ObjectToGround");
194  if (obj == obj1 || obj == obj2 || obj == part1 || obj == part2 || obj == obj3) {
195  if (!prompted) {
196  prompted = true;
197  QMessageBox msgBox;
198  msgBox.setText(tr("The object is associated to one or more joints."));
199  msgBox.setInformativeText(
200  tr("Do you want to move the object and delete associated joints?"));
201  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
202  msgBox.setDefaultButton(QMessageBox::No);
203  int ret = msgBox.exec();
204 
205  if (ret == QMessageBox::No) {
206  return false;
207  }
208  }
209  Gui::Command::doCommand(Gui::Command::Gui,
210  "App.activeDocument().removeObject('%s')",
211  joint->getNameInDocument());
212  }
213  }
214  return true;
215 }

350 bool ViewProviderAssembly::tryMouseMove(const SbVec2s& cursorPos, Gui::View3DInventorViewer* viewer)
351 {
352  if (!isInEditMode()) {
353  return false;
354  }
355 
356  // Initialize or cancel the dragging of parts
357  if (canStartDragging) {
358  canStartDragging = false;
359 
360  if (enableMovement && getSelectedObjectsWithinAssembly()) {
361  initMove(cursorPos, viewer);
362  }
363  }
364 
365  // Do the dragging of parts
366  if (partMoving) {
367  Base::Vector3d newPos, newPosRot;
368  if (dragMode == DragMode::RotationOnPlane) {
369  SbVec3f vec = viewer->getPointOnXYPlaneOfPlacement(cursorPos, jcsGlobalPlc);
370  newPosRot = Base::Vector3d(vec[0], vec[1], vec[2]);
371  }
372  else if (dragMode == DragMode::TranslationOnAxis) {
373  Base::Vector3d zAxis = jcsGlobalPlc.getRotation().multVec(Base::Vector3d(0., 0., 1.));
374  Base::Vector3d pos = jcsGlobalPlc.getPosition();
375  SbVec3f axisCenter(pos.x, pos.y, pos.z);
376  SbVec3f axis(zAxis.x, zAxis.y, zAxis.z);
377  SbVec3f vec = viewer->getPointOnLine(cursorPos, axisCenter, axis);
378  newPos = Base::Vector3d(vec[0], vec[1], vec[2]);
379  }
380  else if (dragMode == DragMode::TranslationOnAxisAndRotationOnePlane) {
381  SbVec3f vec = viewer->getPointOnXYPlaneOfPlacement(cursorPos, jcsGlobalPlc);
382  newPosRot = Base::Vector3d(vec[0], vec[1], vec[2]);
383 
384  Base::Vector3d zAxis = jcsGlobalPlc.getRotation().multVec(Base::Vector3d(0., 0., 1.));
385  Base::Vector3d pos = jcsGlobalPlc.getPosition();
386  SbVec3f axisCenter(pos.x, pos.y, pos.z);
387  SbVec3f axis(zAxis.x, zAxis.y, zAxis.z);
388  vec = viewer->getPointOnLine(cursorPos, axisCenter, axis);
389  newPos = Base::Vector3d(vec[0], vec[1], vec[2]);
390  }
391  else if (dragMode == DragMode::TranslationOnPlane) {
392  SbVec3f vec = viewer->getPointOnXYPlaneOfPlacement(cursorPos, jcsGlobalPlc);
393  newPos = Base::Vector3d(vec[0], vec[1], vec[2]);
394  }
395  else {
396  SbVec3f vec = viewer->getPointOnFocalPlane(cursorPos);
397  newPos = Base::Vector3d(vec[0], vec[1], vec[2]);
398  }
399 
400  for (auto& objToMove : docsToMove) {
401  App::DocumentObject* obj = objToMove.obj;
402  auto* propPlacement =
403  dynamic_cast<App::PropertyPlacement*>(obj->getPropertyByName("Placement"));
404  if (propPlacement) {
405  Base::Placement plc = objToMove.plc;
406 
407  if (dragMode == DragMode::RotationOnPlane) {
408  Base::Vector3d center = jcsGlobalPlc.getPosition();
409  Base::Vector3d norm =
410  jcsGlobalPlc.getRotation().multVec(Base::Vector3d(0., 0., -1.));
411  double angle =
412  (newPosRot - center).GetAngleOriented(initialPositionRot - center, norm);
413  Base::Rotation zRotation = Base::Rotation(Base::Vector3d(0., 0., 1.), angle);
414  Base::Placement rotatedGlovalJcsPlc =
415  jcsGlobalPlc * Base::Placement(Base::Vector3d(), zRotation);
416  Base::Placement jcsPlcRelativeToPart = plc.inverse() * jcsGlobalPlc;
417  plc = rotatedGlovalJcsPlc * jcsPlcRelativeToPart.inverse();
418  }
419  else if (dragMode == DragMode::TranslationOnAxis) {
420  Base::Vector3d pos = plc.getPosition() + (newPos - initialPosition);
421  plc.setPosition(pos);
422  }
423  else if (dragMode == DragMode::TranslationOnAxisAndRotationOnePlane) {
424  Base::Vector3d pos = plc.getPosition() + (newPos - initialPosition);
425  plc.setPosition(pos);
426 
427  Base::Placement newJcsGlobalPlc = jcsGlobalPlc;
428  newJcsGlobalPlc.setPosition(jcsGlobalPlc.getPosition()
429  + (newPos - initialPosition));
430 
431  Base::Vector3d center = newJcsGlobalPlc.getPosition();
432  Base::Vector3d norm =
433  newJcsGlobalPlc.getRotation().multVec(Base::Vector3d(0., 0., -1.));
434 
435  Base::Vector3d projInitialPositionRot =
436  initialPositionRot.ProjectToPlane(newJcsGlobalPlc.getPosition(), norm);
437  boost::ignore_unused(projInitialPositionRot);
438  double angle =
439  (newPosRot - center).GetAngleOriented(initialPositionRot - center, norm);
440  Base::Rotation zRotation = Base::Rotation(Base::Vector3d(0., 0., 1.), angle);
441  Base::Placement rotatedGlovalJcsPlc =
442  newJcsGlobalPlc * Base::Placement(Base::Vector3d(), zRotation);
443  Base::Placement jcsPlcRelativeToPart = plc.inverse() * newJcsGlobalPlc;
444  plc = rotatedGlovalJcsPlc * jcsPlcRelativeToPart.inverse();
445  }
446  else if (dragMode == DragMode::TranslationOnPlane) {
447  Base::Vector3d pos = plc.getPosition() + (newPos - initialPosition);
448  plc.setPosition(pos);
449  }
450  else { // DragMode::Translation
451  Base::Vector3d delta = newPos - prevPosition;
452 
453  Base::Vector3d pos = propPlacement->getValue().getPosition() + delta;
454  plc.setPosition(pos);
455  }
456  propPlacement->setValue(plc);
457  }
458  }
459 
460  prevPosition = newPos;
461 
462  auto* assemblyPart = getObject<AssemblyObject>();
463  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
464  "User parameter:BaseApp/Preferences/Mod/Assembly");
465  bool solveOnMove = hGrp->GetBool("SolveOnMove", true);
466  if (solveOnMove && dragMode != DragMode::TranslationNoSolve) {
467  assemblyPart->doDragStep();
468  }
469  else {
470  assemblyPart->redrawJointPlacements(assemblyPart->getJoints());
471  }
472  }
473  return false;
474 }

476 bool ViewProviderAssembly::mouseButtonPressed(int Button,
477  bool pressed,
478  const SbVec2s& cursorPos,
479  const Gui::View3DInventorViewer* viewer)
480 {
481  Q_UNUSED(cursorPos);
482  Q_UNUSED(viewer);
483 
484  if (!isInEditMode()) {
485  return false;
486  }
487 
488  // Left Mouse button ****************************************************
489  if (Button == 1) {
490  if (pressed && !getDraggerVisibility()) {
491  // Check for double-click
492  auto now = std::chrono::steady_clock::now();
493  long nowMillis =
494  std::chrono::duration_cast<std::chrono::milliseconds>(now.time_since_epoch())
495  .count();
496  if (nowMillis - lastClickTime < 500) {
497  auto* joint = getSelectedJoint();
498  if (joint) {
499  // Double-click detected
500  // We start by clearing selection such that the second click selects the joint
501  // and not the assembly.
502  Gui::Selection().clearSelection();
503  // singleShot timer to make sure this happens after the release of the click.
504  // Else the release will trigger a removeSelection of what
505  // doubleClickedIn3dView adds to the selection.
506  QTimer::singleShot(50, [this]() {
507  doubleClickedIn3dView();
508  });
509  return true;
510  }
511  }
512  // First click detected
513  lastClickTime = nowMillis;
514 
515  canStartDragging = true;
516  }
517  else { // Button 1 released
518  // release event is not received when user click on a part for selection.
519  // So we use SelectionObserver to know if something got selected.
520 
521  canStartDragging = false;
522  if (partMoving) {
523  endMove();
524  return true;
525  }
526  }
527  }
528  return false;
529 }

550 bool ViewProviderAssembly::canDragObjectIn3d(App::DocumentObject* obj) const
551 {
552  if (!obj) {
553  return false;
554  }
555 
556  auto* assemblyPart = getObject<AssemblyObject>();
557 
558  // Check if the selected object is a child of the assembly
559  if (!assemblyPart->hasObject(obj, true)) {
560  // hasObject does not detect LinkElements (see
561  // https://github.com/FreeCAD/FreeCAD/issues/16113) the following block can be removed if
562  // the issue is fixed :
563  auto* linkEl = dynamic_cast<App::LinkElement*>(obj);
564  if (linkEl) {
565  auto* linkGroup = linkEl->getLinkGroup();
566  if (assemblyPart->hasObject(linkGroup, true)) {
567  return true;
568  }
569  }
570  return false;
571  }
572 
573  auto* propPlacement =
574  dynamic_cast<App::PropertyPlacement*>(obj->getPropertyByName("Placement"));
575  if (!propPlacement) {
576  return false;
577  }
578 
579  // We have to exclude Grounded joints as they happen to have a Placement prop
580  auto* propLink = dynamic_cast<App::PropertyLink*>(obj->getPropertyByName("ObjectToGround"));
581  if (propLink) {
582  return false;
583  }
584 
585  // We have to exclude grounded objects as they should not move.
586  if (assemblyPart->isPartGrounded(obj)) {
587  return false;
588  }
589  return true;
590 }

609 bool ViewProviderAssembly::getSelectedObjectsWithinAssembly(bool addPreselection, bool onlySolids)
610 {
611  // check the current selection, and check if any of the selected objects are within this
612  // App::Part
613  // If any, put them into the vector docsToMove and return true.
614  // Get the document
615  docsToMove.clear();
616 
617  // Get the assembly object for this ViewProvider
618  auto* assemblyPart = getObject<AssemblyObject>();
619 
620  if (!assemblyPart) {
621  return false;
622  }
623 
624  if (!moveOnlyPreselected) {
625  for (auto& selObj : Gui::Selection().getSelectionEx("",
626  App::DocumentObject::getClassTypeId(),
627  Gui::ResolveMode::NoResolve)) {
628  // getSubNames() returns ["Body001.Pad.Face14", "Body002.Pad.Face7"]
629  // if you have several objects within the same assembly selected.
630 
631  std::vector<std::string> objsSubNames = selObj.getSubNames();
632  for (auto& subNamesStr : objsSubNames) {
633  std::vector<std::string> subNames = Base::Tools::splitSubName(subNamesStr);
634  if (subNames.empty()) {
635  continue;
636  }
637  if (onlySolids && subNames.back() != "") {
638  continue;
639  }
640 
641  App::DocumentObject* selRoot = selObj.getObject();
642  App::DocumentObject* obj = getObjFromRef(selRoot, subNamesStr);
643  if (!obj) {
644  // In case of sub-assembly, the jointgroup would trigger the dragger.
645  continue;
646  }
647  App::DocumentObject* part =
648  getMovingPartFromRef(assemblyPart, selRoot, subNamesStr);
649 
650  if (!canDragObjectIn3d(part)) {
651  continue;
652  }
653 
654  auto* pPlc =
655  dynamic_cast<App::PropertyPlacement*>(part->getPropertyByName("Placement"));
656 
657  MovingObject movingObj(part, pPlc->getValue(), selRoot, subNamesStr);
658 
659  docsToMove.emplace_back(movingObj);
660  }
661  }
662  }
663 
664  // This function is called before the selection is updated. So if a user click and drag a part

701 {
702  auto addPartsToMove = [&](const std::vector<Assembly::ObjRef>& refs) {
703  for (auto& partRef : refs) {
704  auto* pPlc =
705  dynamic_cast<App::PropertyPlacement*>(partRef.obj->getPropertyByName("Placement"));
706  if (pPlc) {
707  App::DocumentObject* selRoot = partRef.ref->getValue();
708  if (!selRoot) {

710  }
711  std::vector<std::string> subs = partRef.ref->getSubValues();
712  if (subs.empty()) {
713  continue;
714  }
715 
716  docsToMove.emplace_back(partRef.obj, pPlc->getValue(), selRoot, subs[0]);
717  }
718  }
719  };
720 
721  if (docsToMove.size() == 1) {
722  auto* assemblyPart = getObject<AssemblyObject>();
723  std::string pName;
724  movingJoint = assemblyPart->getJointOfPartConnectingToGround(docsToMove[0].obj, pName);
725 
726  if (!movingJoint) {
727  // In this case the user is moving an object that is not grounded
728  // Then we want to also move other parts that may be connected to it.
729  // In particular for case of flexible subassemblies or it looks really weird
730  std::vector<Assembly::ObjRef> connectedParts =
731  assemblyPart->getDownstreamParts(docsToMove[0].obj, movingJoint);
732 
733  addPartsToMove(connectedParts);
734  return DragMode::TranslationNoSolve;
735  }
736 
737  JointType jointType = getJointType(movingJoint);
738  if (jointType == JointType::Fixed) {
739  // If fixed joint we need to find the upstream joint to find move mode.
740  // For example : Gnd -(revolute)- A -(fixed)- B : if user try to move B, then we should
741  // actually move A
742  auto* upPart =
743  assemblyPart->getUpstreamMovingPart(docsToMove[0].obj, movingJoint, pName);
744  if (!movingJoint) {
745  return DragMode::Translation;
746  }
747  docsToMove.clear();
748  if (!upPart) {
749  return DragMode::None;
750  }
751 
752  auto* pPlc =
753  dynamic_cast<App::PropertyPlacement*>(upPart->getPropertyByName("Placement"));
754  if (pPlc) {
755  auto* ref = dynamic_cast<App::PropertyXLinkSub*>(
756  movingJoint->getPropertyByName(pName.c_str()));
757 
758  App::DocumentObject* selRoot = ref->getValue();
759  if (!selRoot) {
760  return DragMode::None;
761  }
762  std::vector<std::string> subs = ref->getSubValues();
763  if (subs.empty()) {
764  return DragMode::None;
765  }
766 
767  docsToMove.emplace_back(upPart, pPlc->getValue(), selRoot, subs[0]);
768  }
769 
770  jointType = getJointType(movingJoint);
771  }
772 
773  const char* plcPropName = (pName == "Reference1") ? "Placement1" : "Placement2";
774 
775  // jcsPlc is relative to the Object
776  jcsPlc = App::GeoFeature::getPlacementFromProp(movingJoint, plcPropName);
777 
778  // Make jcsGlobalPlc relative to the origin of the doc
779  auto* ref =
780  dynamic_cast<App::PropertyXLinkSub*>(movingJoint->getPropertyByName(pName.c_str()));
781  if (!ref) {
782  return DragMode::Translation;
783  }
784  auto* obj = getObjFromRef(movingJoint, pName.c_str());
785  Base::Placement global_plc = App::GeoFeature::getGlobalPlacement(obj, ref);
786  jcsGlobalPlc = global_plc * jcsPlc;
787 
788  // Add downstream parts so that they move together
789  std::vector<Assembly::ObjRef> downstreamParts =
790  assemblyPart->getDownstreamParts(docsToMove[0].obj, movingJoint);
791  addPartsToMove(downstreamParts);
792 
793  if (jointType == JointType::Revolute) {
794  return DragMode::RotationOnPlane;
795  }
796  else if (jointType == JointType::Slider) {
797  return DragMode::TranslationOnAxis;
798  }
799  else if (jointType == JointType::Cylindrical) {
800  return DragMode::TranslationOnAxisAndRotationOnePlane;
801  }
802  else if (jointType == JointType::Ball) {
803  // return DragMode::Ball;
804  }
805  else if (jointType == JointType::Distance) {
806  // depends on the type of distance. For example plane-plane:
807  DistanceType distanceType = getDistanceType(movingJoint);
808  if (distanceType == DistanceType::PlanePlane || distanceType == DistanceType::Other) {
809  return DragMode::TranslationOnPlane;
810  }
811  }
812  }
813  return DragMode::Translation;
