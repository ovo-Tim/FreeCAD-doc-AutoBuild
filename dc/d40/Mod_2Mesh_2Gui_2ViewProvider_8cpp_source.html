108 void ViewProviderMeshBuilder::buildNodes(const App::Property* prop,
109  std::vector<SoNode*>& nodes) const
110 {
111  SoCoordinate3* pcPointsCoord = nullptr;
112  SoIndexedFaceSet* pcFaces = nullptr;
113 
114  if (nodes.empty()) {
115  pcPointsCoord = new SoCoordinate3();
116  nodes.push_back(pcPointsCoord);
117  pcFaces = new SoIndexedFaceSet();
118  nodes.push_back(pcFaces);
119  }
120  else if (nodes.size() == 2) {
121  if (nodes[0]->getTypeId() == SoCoordinate3::getClassTypeId()) {
122  pcPointsCoord = static_cast<SoCoordinate3*>(nodes[0]);
123  }
124  if (nodes[1]->getTypeId() == SoIndexedFaceSet::getClassTypeId()) {
125  pcFaces = static_cast<SoIndexedFaceSet*>(nodes[1]);
126  }
127  }
128 
129  if (pcPointsCoord && pcFaces) {
130  createMesh(prop, pcPointsCoord, pcFaces);
131  }
132 }

143 void ViewProviderMeshBuilder::createMesh(const MeshCore::MeshKernel& kernel,
144  SoCoordinate3* coords,
145  SoIndexedFaceSet* faces) const
146 {
147 
148  // set the point coordinates
149  const MeshCore::MeshPointArray& cP = kernel.GetPoints();
150  coords->point.setNum(kernel.CountPoints());
151  SbVec3f* verts = coords->point.startEditing();
152  int i = 0;
153  for (auto it = cP.begin(); it != cP.end(); ++it, i++) {
154  verts[i].setValue(it->x, it->y, it->z);
155  }
156  coords->point.finishEditing();
157 
158  // set the face indices
159  int j = 0;
160  const MeshCore::MeshFacetArray& cF = kernel.GetFacets();
161  faces->coordIndex.setNum(4 * kernel.CountFacets());
162  int32_t* indices = faces->coordIndex.startEditing();
163  for (auto it = cF.begin(); it != cF.end(); ++it, j++) {
164  for (int i = 0; i < 3; i++) {
165  indices[4 * j + i] = it->_aulPoints[i];
166  }
167  indices[4 * j + 3] = SO_END_FACE_INDEX;
168  }
169  faces->coordIndex.finishEditing();
170 }

325 void ViewProviderMesh::onChanged(const App::Property* prop)
326 {
327  // we're going to change the number of colors to one
328  if (prop == &ShapeAppearance) {
329  pcMatBinding->value = SoMaterialBinding::OVERALL;
330  }
331  if (prop == &LineTransparency) {
332  float trans = LineTransparency.getValue() / 100.0F;
333  pLineColor->transparency = trans;
334  }
335  else if (prop == &LineWidth) {
336  pcLineStyle->lineWidth = LineWidth.getValue();
337  }
338  else if (prop == &PointSize) {
339  pcPointStyle->pointSize = PointSize.getValue();
340  }
341  else if (prop == &CreaseAngle) {
342  pShapeHints->creaseAngle = Base::toRadians<float>(CreaseAngle.getValue());
343  }
344  else if (prop == &OpenEdges) {
345  showOpenEdges(OpenEdges.getValue());
346  }
347  else if (prop == &Lighting) {
348  if (Lighting.getValue() == 0) {
349  pShapeHints->vertexOrdering = SoShapeHints::UNKNOWN_ORDERING;
350  }
351  else {
352  pShapeHints->vertexOrdering = SoShapeHints::COUNTERCLOCKWISE;
353  }
354  }
355  else if (prop == &LineColor) {
356  const App::Color& c = LineColor.getValue();
357  pLineColor->diffuseColor.setValue(c.r, c.g, c.b);
358  }
359  else if (prop == &Coloring) {
360  tryColorPerVertexOrFace(Coloring.getValue());
361  }
362  else if (prop == &SelectionStyle) {
363  pcHighlight->style =
364  SelectionStyle.getValue() ? Gui::SoFCSelection::BOX : Gui::SoFCSelection::EMISSIVE;
365  }
366  else {
367  // Set the inverse color for open edges
368  if (prop == &ShapeAppearance) {
369  setOpenEdgeColorFrom(ShapeAppearance.getDiffuseColor());
370  }
371  }
372 
373  ViewProviderGeometryObject::onChanged(prop);

416 void ViewProviderMesh::attach(App::DocumentObject* obj)
417 {
418  ViewProviderGeometryObject::attach(obj);
419 
420  pcHighlight->objectName = obj->getNameInDocument();
421  pcHighlight->documentName = obj->getDocument()->getName();
422  pcHighlight->subElementName = "Main";

424  // Note: Since for mesh data the SoFCSelection node has no SoSeparator but
425  // an SoGroup as parent the EMISSIVE style if set has fundamentally no effect.
426  // This behaviour is given due to the fact that SoFCSelection inherits from
427  // SoGroup (formerly SoSeparator). If we wanted to enable emissive overlay for
428  // highlighting or selection we would need an SoSeparator as parent node below.
429 
430  // faces

433  pcFlatRoot->addChild(pcShapeMaterial);

438  // points
439  auto pcPointRoot = new SoGroup();
440  pcPointRoot->addChild(pcPointStyle);
441  pcPointRoot->addChild(pShapeHints);
442  pcPointRoot->addChild(pcShapeMaterial);
443  pcPointRoot->addChild(pcMatBinding);
444  pcPointRoot->addChild(pcHighlight);
445  addDisplayMaskMode(pcPointRoot, "Point");
446 
447  // wires
448  auto pcLightModel = new SoLightModel();
449  pcLightModel->model = SoLightModel::BASE_COLOR;
450  auto pcWireRoot = new SoGroup();
451  pcWireRoot->addChild(pcLineStyle);
452  pcWireRoot->addChild(pcLightModel);

469  pcWireSep->addChild(pcLineStyle);
470  pcWireSep->addChild(pcLightModel);
471  pcWireSep->addChild(binding);
472  pcWireSep->addChild(pLineColor);

478  pcFlatWireRoot->addChild(pShapeHints);
479  pcFlatWireRoot->addChild(pcShapeMaterial);
480  pcFlatWireRoot->addChild(pcMatBinding);

531 void ViewProviderMesh::tryColorPerVertexOrFace(bool on)
532 {
533  if (on) {
534  const Mesh::MeshObject& mesh = getMeshObject();
535  int numPoints = static_cast<int>(mesh.countPoints());
536  int numFacets = static_cast<int>(mesh.countFacets());
537 
538  if (App::PropertyColorList* colors = getColorProperty()) {
539  if (colors->getSize() == numPoints) {
540  setColorPerVertex(colors);
541  }
542  else if (colors->getSize() == numFacets) {
543  setColorPerFace(colors);
544  }
545  }
546  else if (Mesh::PropertyMaterial* material = getMaterialProperty()) {
547  auto bind = material->getBinding();
548  if (bind == MeshCore::MeshIO::Binding::OVERALL) {
549  pcMatBinding->value = SoMaterialBinding::OVERALL;
550 
551  if (!material->getDiffuseColor().empty()) {
552  auto c = material->getDiffuseColor()[0];
553  pcShapeMaterial->diffuseColor.setValue(c.r, c.g, c.b);
554  }
555  if (!material->getTransparency().empty()) {
556  pcShapeMaterial->transparency.setValue(material->getTransparency()[0]);
557  }
558  }
559  else if (bind == MeshCore::MeshIO::Binding::PER_VERTEX) {
560  if (material->getDiffuseColor().size() == std::size_t(numPoints)) {
561  pcMatBinding->value = SoMaterialBinding::PER_VERTEX_INDEXED;
562  setDiffuseColor(material->getDiffuseColor());
563  }
564  }
565  else if (bind == MeshCore::MeshIO::Binding::PER_FACE) {
566  if (material->getAmbientColor().size() == std::size_t(numFacets)) {
567  pcMatBinding->value = SoMaterialBinding::PER_FACE;
568  setAmbientColor(material->getAmbientColor());
569  }
570  if (material->getDiffuseColor().size() == std::size_t(numFacets)) {
571  pcMatBinding->value = SoMaterialBinding::PER_FACE;
572  setDiffuseColor(material->getDiffuseColor());
573  }
574  if (material->getEmissiveColor().size() == std::size_t(numFacets)) {
575  pcMatBinding->value = SoMaterialBinding::PER_FACE;
576  setEmissiveColor(material->getEmissiveColor());
577  }
578  if (material->getSpecularColor().size() == std::size_t(numFacets)) {
579  pcMatBinding->value = SoMaterialBinding::PER_FACE;
580  setSpecularColor(material->getSpecularColor());
581  }
582  if (material->getTransparency().size() == std::size_t(numFacets)) {
583  pcMatBinding->value = SoMaterialBinding::PER_FACE;
584  setFacetTransparency(material->getTransparency());
585  }
586  }
587  }
588  }
589  else {
590  pcMatBinding->value = SoMaterialBinding::OVERALL;
591  const App::Color& c = ShapeAppearance.getDiffuseColor();
592  pcShapeMaterial->diffuseColor.setValue(c.r, c.g, c.b);
593  pcShapeMaterial->transparency.setValue(Transparency.getValue() / 100.0F);
594  }
595 }

841 bool ViewProviderMesh::createToolMesh(const std::vector<SbVec2f>& rclPoly,
842  const SbViewVolume& vol,
843  const Base::Vector3f& rcNormal,
844  std::vector<MeshCore::MeshGeomFacet>& aFaces)
845 {
846  float fX {};
847  float fY {};
848  float fZ {};
849  SbVec3f pt1;
850  SbVec3f pt2;
851  SbVec3f pt3;
852  SbVec3f pt4;
853  MeshGeomFacet face;
854  std::vector<Base::Vector3f> top;
855  std::vector<Base::Vector3f> bottom;
856  std::vector<Base::Vector3f> polygon;
857 
858  for (auto it = rclPoly.begin(); it != rclPoly.end(); ++it) {
859  // the following element
860  auto nt = it + 1;
861  if (nt == rclPoly.end()) {
862  nt = rclPoly.begin();
863  }
864  else if (*it == *nt) {
865  continue; // two adjacent vertices are equal
866  }
867 
868  vol.projectPointToLine(*it, pt1, pt2);
869  vol.projectPointToLine(*nt, pt3, pt4);
870 
871  // 1st facet
872  pt1.getValue(fX, fY, fZ);
873  face._aclPoints[0].Set(fX, fY, fZ);
874  pt4.getValue(fX, fY, fZ);
875  face._aclPoints[1].Set(fX, fY, fZ);
876  pt3.getValue(fX, fY, fZ);
877  face._aclPoints[2].Set(fX, fY, fZ);
878  if (face.Area() > 0) {
879  aFaces.push_back(face);
880  }
881 
882  // 2nd facet
883  pt1.getValue(fX, fY, fZ);
884  face._aclPoints[0].Set(fX, fY, fZ);
885  pt2.getValue(fX, fY, fZ);
886  face._aclPoints[1].Set(fX, fY, fZ);
887  pt4.getValue(fX, fY, fZ);
888  face._aclPoints[2].Set(fX, fY, fZ);
889  if (face.Area() > 0) {
890  aFaces.push_back(face);
891  }
892 
893  if (it + 1 < rclPoly.end()) {
894  pt1.getValue(fX, fY, fZ);
895  top.emplace_back(fX, fY, fZ);
896  pt2.getValue(fX, fY, fZ);
897  bottom.emplace_back(fX, fY, fZ);
898  // polygon we need to triangulate (in x,y-plane)
899  it->getValue(fX, fY);
900  polygon.emplace_back(fX, fY, 0.0F);
901  }
902  }
903 
904  // now create the lids
905  std::vector<MeshGeomFacet> aLid;
906  MeshCore::EarClippingTriangulator cTria;
907  cTria.SetPolygon(polygon);
908  bool ok = cTria.TriangulatePolygon();
909 
910  std::vector<MeshFacet> faces = cTria.GetFacets();
911  for (const auto& face : faces) {
912  MeshGeomFacet topFacet;
913  topFacet._aclPoints[0] = top[face._aulPoints[0]];
914  topFacet._aclPoints[1] = top[face._aulPoints[1]];
915  topFacet._aclPoints[2] = top[face._aulPoints[2]];
916  if (topFacet.GetNormal() * rcNormal < 0) {
917  std::swap(topFacet._aclPoints[1], topFacet._aclPoints[2]);
918  topFacet.CalcNormal();
919  }
920  aFaces.push_back(topFacet);
921 
922  MeshGeomFacet botFacet;
923  botFacet._aclPoints[0] = bottom[face._aulPoints[0]];
924  botFacet._aclPoints[1] = bottom[face._aulPoints[1]];
925  botFacet._aclPoints[2] = bottom[face._aulPoints[2]];
926  if (botFacet.GetNormal() * rcNormal > 0) {
927  std::swap(botFacet._aclPoints[1], botFacet._aclPoints[2]);
928  botFacet.CalcNormal();
929  }
930  aFaces.push_back(botFacet);
931  }
932 
933  return ok;
934 }

1914 void ViewProviderMesh::fillHole(Mesh::FacetIndex uFacet)
1915 {
1916  // get parameter from user settings
1917  Base::Reference<ParameterGrp> hGrp =
1918  Gui::WindowParameter::getDefaultParameter()->GetGroup("Mod/Mesh");
1919  int level = (int)hGrp->GetInt("FillHoleLevel", 2);
1920 
1921  // get the boundary to the picked facet
1922  std::list<Mesh::PointIndex> aBorder;
1923  const MeshCore::MeshKernel& rKernel = getMeshObject().getKernel();
1924  MeshCore::MeshRefPointToFacets cPt2Fac(rKernel);
1925  MeshCore::MeshAlgorithm meshAlg(rKernel);
1926  meshAlg.GetFacetBorder(uFacet, aBorder);
1927  std::vector<Mesh::PointIndex> boundary(aBorder.begin(), aBorder.end());
1928  std::list<std::vector<Mesh::PointIndex>> boundaries;
1929  boundaries.push_back(boundary);
1930  meshAlg.SplitBoundaryLoops(boundaries);
1931 
1932  std::vector<MeshCore::MeshFacet> newFacets;
1933  std::vector<Base::Vector3f> newPoints;
1934  unsigned long numberOfOldPoints = rKernel.CountPoints();
1935  for (const auto& it : boundaries) {
1936  if (it.size() < 3 /* || it->size() > 200*/) {
1937  continue;
1938  }
1939  boundary = it;
1940  MeshCore::MeshFacetArray faces;
1941  MeshCore::MeshPointArray points;
1942  MeshCore::QuasiDelaunayTriangulator cTria /*(0.05f)*/;
1943  cTria.SetVerifier(new MeshCore::TriangulationVerifierV2);
1944  if (meshAlg.FillupHole(boundary, cTria, faces, points, level, &cPt2Fac)) {
1945  if (boundary.front() == boundary.back()) {
1946  boundary.pop_back();
1947  }
1948  // the triangulation may produce additional points which we must take into account when
1949  // appending to the mesh
1950  unsigned long countBoundaryPoints = boundary.size();
1951  unsigned long countDifference = points.size() - countBoundaryPoints;
1952  if (countDifference > 0) {
1953  auto pt = points.begin() + countBoundaryPoints;
1954  for (unsigned long i = 0; i < countDifference; i++, pt++) {
1955  boundary.push_back(numberOfOldPoints++);
1956  newPoints.push_back(*pt);
1957  }
1958  }
1959  for (auto& face : faces) {
1960  face._aulPoints[0] = boundary[face._aulPoints[0]];
1961  face._aulPoints[1] = boundary[face._aulPoints[1]];
1962  face._aulPoints[2] = boundary[face._aulPoints[2]];
1963  newFacets.push_back(face);
1964  }
1965  }
1966  }
1967 
1968  if (newFacets.empty()) {
1969  return; // nothing to do
1970  }
1971 
1972  // add the facets to the mesh and open a transaction object for the undo/redo stuff
1973  Gui::Application::Instance->activeDocument()->openCommand(
1974  QT_TRANSLATE_NOOP("Command", "Fill hole"));

1977  kernel->addFacets(newFacets, newPoints, true);
