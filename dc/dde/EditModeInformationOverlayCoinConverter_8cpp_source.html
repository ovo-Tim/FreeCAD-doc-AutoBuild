60 void EditModeInformationOverlayCoinConverter::convert(const Part::Geometry* geometry, int geoid)
61 {
62 
63  if (geometry->is<Part::GeomBSplineCurve>()) {
64  if (geoid < 0) {
65  return;
66  }
67  calculate<CalculationType::BSplineDegree>(geometry, geoid);
68  calculate<CalculationType::BSplineControlPolygon>(geometry, geoid);
69  calculate<CalculationType::BSplineCurvatureComb>(geometry, geoid);
70  calculate<CalculationType::BSplineKnotMultiplicity>(geometry, geoid);
71  calculate<CalculationType::BSplinePoleWeight>(geometry, geoid);
72 
73  addUpdateNode(degree);
74  addUpdateNode(controlPolygon);
75  addUpdateNode(curvatureComb);
76  addUpdateNode(knotMultiplicity);
77  addUpdateNode(poleWeights);
78  }
79  else if (geometry->is<Part::GeomArcOfCircle>()) {
80  // at this point all calculations relate to ArcOfCircle
81  calculate<CalculationType::ArcCircleHelper>(geometry, geoid);
82  addUpdateNode(circleHelper);
83  }
84 }

92 template<EditModeInformationOverlayCoinConverter::CalculationType calculation>
93 void EditModeInformationOverlayCoinConverter::calculate(const Part::Geometry* geometry,
94  [[maybe_unused]] int geoid)
95 {
96 
97  if constexpr (calculation == CalculationType::ArcCircleHelper) {
98  const Part::GeomArcOfCircle* arc = static_cast<const Part::GeomArcOfCircle*>(geometry);
99  clearCalculation(circleHelper);
100 
101  Base::Vector3d center = arc->getCenter();
102  double radius = arc->getRadius();
103  Part::GeomCircle circle;
104  circle.setRadius(radius);
105  circle.setCenter(center);
106 
107  const int ndiv = drawingParameters.curvedEdgeCountSegments;
108 
109  circleHelper.coordinates.reserve(ndiv);
110  for (int i = 0; i < ndiv; i++) {
111  double param = i * std::atan(1) * 8 / ndiv;
112  circleHelper.coordinates.emplace_back(circle.value(param));
113  }
114  circleHelper.coordinates.emplace_back(circle.value(0.0));
115  circleHelper.indices.push_back(ndiv + 1);
116  }
117  else {
118  const Part::GeomBSplineCurve* spline = static_cast<const Part::GeomBSplineCurve*>(geometry);
119 
120  if constexpr (calculation == CalculationType::BSplineDegree) {
121  clearCalculation(degree);
122 
123  std::vector<Base::Vector3d> poles = spline->getPoles();
124 
125  degree.strings.clear();
126  degree.positions.clear();
127 
128  Base::Vector3d midp = Base::Vector3d(0, 0, 0);
129 
130  for (auto val : poles) {
131  midp += val;
132  }
133 
134  midp /= poles.size();
135 
136  degree.strings.emplace_back(std::to_string(spline->getDegree()));
137  degree.positions.emplace_back(midp);
138  }
139  else if constexpr (calculation == CalculationType::BSplineControlPolygon) {
140 
141  clearCalculation(controlPolygon);
142 
143  std::vector<Base::Vector3d> poles = spline->getPoles();
144 
145  controlPolygon.coordinates.clear();
146  controlPolygon.indices.clear();
147 
148  size_t nvertices;
149 
150  if (spline->isPeriodic()) {
151  nvertices = poles.size() + 1;
152  }
153  else {
154  nvertices = poles.size();
155  }
156 
157  controlPolygon.coordinates.reserve(nvertices);
158 
159  for (auto& v : poles) {
160  controlPolygon.coordinates.emplace_back(v);
161  }
162 
163  if (spline->isPeriodic()) {
164  controlPolygon.coordinates.emplace_back(poles[0]);
165  }
166 
167  controlPolygon.indices.push_back(
168  nvertices); // single continuous polygon starting at index 0
169  }
170  else if constexpr (calculation == CalculationType::BSplineCurvatureComb) {
171 
172  clearCalculation(curvatureComb);
173  // curvature graph --------------------------------------------------------
174 
175  // based on python source
176  // https://github.com/tomate44/CurvesWB/blob/master/freecad/Curves/ParametricComb.py
177  // by FreeCAD user Chris_G
178 
179  std::vector<Base::Vector3d> poles = spline->getPoles();
180  auto knots = spline->getKnots();
181  auto mults = spline->getMultiplicities();
182 
183  const int ndivPerPiece = 64; // heuristic of number of division to fill in
184  const int ndiv = ndivPerPiece * (knots.size() - 1);
185 
186  std::vector<Base::Vector3d> pointatcurvelist;
187  std::vector<double> curvaturelist;
188  std::vector<Base::Vector3d> normallist;
189 
190  pointatcurvelist.reserve(ndiv);
191  curvaturelist.reserve(ndiv);
192  normallist.reserve(ndiv);
193 
194  // go through the polynomial pieces (i.e. from one knot to next)
195  for (size_t k = 0; k < knots.size() - 1; ++k) {
196  // first and last params are a little off to account for possible discontinuity at
197  // knots
198  double firstparam =
199  knots[k] + Precision::Approximation() * (knots[k + 1] - knots[k]);
200  double lastparam =
201  knots[k + 1] - Precision::Approximation() * (knots[k + 1] - knots[k]);
202 
203  // TODO: Maybe this can be improved, specifically adapted for each piece
204  double step = (lastparam - firstparam) / (ndivPerPiece - 1);
205 
206  for (int i = 0; i < ndivPerPiece; ++i) {
207  double param = firstparam + i * step;
208  pointatcurvelist.emplace_back(spline->value(param));
209 
210  try {
211  curvaturelist.emplace_back(spline->curvatureAt(param));
212  }
213  catch (Base::CADKernelError& e) {
214  // it is "just" a visualisation matter OCC could not calculate the curvature
215  // terminating here would mean that the other shapes would not be drawn.
216  // Solution: Report the issue and set dummy curvature to 0
217  e.ReportException();
218  Base::Console().DeveloperError(
219  "EditModeInformationOverlayCoinConverter",
220  "Curvature graph for B-spline with GeoId=%d could not be calculated.\n",
221  geoid);
222  curvaturelist.emplace_back(0);
223  }
224 
225  Base::Vector3d normal;
226  try {
227  spline->normalAt(param, normal);
228  normallist.emplace_back(normal);
229  }
230  catch (Base::Exception&) {
231  normallist.emplace_back(0, 0, 0);
232  }
233  }
234  }
235 
236  std::vector<Base::Vector3d> pointatcomblist;
237  pointatcomblist.reserve(ndiv);
238 
239  for (int i = 0; i < ndiv; i++) {
240  pointatcomblist.emplace_back(
241  pointatcurvelist[i]
242  - overlayParameters.currentBSplineCombRepresentationScale * curvaturelist[i]
243  * normallist[i]);
244  }
245 
246  curvatureComb.coordinates.reserve(3 * ndiv); // 2*ndiv +1 points of ndiv separate
247  // segments + ndiv points for last segment
248  curvatureComb.indices.reserve(
249  ndiv + 1); // ndiv separate segments of radials + 1 segment connecting at comb end
250 
251  auto zInfoH = ViewProviderSketchCoinAttorney::getViewOrientationFactor(viewProvider)
252  * drawingParameters.zInfo;
253 
254  for (int i = 0; i < ndiv; i++) {
255  // note emplace emplaces on the position BEFORE the iterator given.
256  curvatureComb.coordinates.emplace_back(pointatcurvelist[i].x,
257  pointatcurvelist[i].y,
258  zInfoH); // radials
259  curvatureComb.coordinates.emplace_back(pointatcomblist[i].x,
260  pointatcomblist[i].y,
261  zInfoH); // radials
262 
263  curvatureComb.indices.emplace_back(2); // line
264  }
265 
266  for (int i = 0; i < ndiv; i++) {
267  curvatureComb.coordinates.emplace_back(pointatcomblist[i].x,
268  pointatcomblist[i].y,
269  zInfoH); // // comb endpoint closing segment
270  }
271 
272  curvatureComb.indices.emplace_back(ndiv); // Comb line
273  }
274  else if constexpr (calculation == CalculationType::BSplineKnotMultiplicity) {
275 
276  clearCalculation(knotMultiplicity);
277  std::vector<double> knots = spline->getKnots();
278  std::vector<int> mult = spline->getMultiplicities();
279 
280  for (size_t i = 0; i < knots.size(); i++) {
281  knotMultiplicity.positions.emplace_back(spline->pointAtParameter(knots[i]));
282 
283  std::ostringstream stringStream;
284  stringStream << "(" << mult[i] << ")";
285 
286  knotMultiplicity.strings.emplace_back(stringStream.str());
287  }
288  }
289  else if constexpr (calculation == CalculationType::BSplinePoleWeight) {
290 
291  clearCalculation(poleWeights);
292  std::vector<Base::Vector3d> poles = spline->getPoles();
293  auto weights = spline->getWeights();
294 
295  for (size_t i = 0; i < poles.size(); i++) {
296  poleWeights.positions.emplace_back(poles[i]);
297 
298  QString WeightString =
299  QString::fromLatin1("[%1]").arg(weights[i],
300  0,
301  'f',
302  Base::UnitsApi::getDecimals());
303 
304  poleWeights.strings.emplace_back(WeightString.toStdString());
305  }
306  }
307  }
308 }

346 void EditModeInformationOverlayCoinConverter::setPolygon(const Result& result,
347  SoLineSet* polygonlineset,
348  SoCoordinate3* polygoncoords)
349 {
350 
351  polygoncoords->point.setNum(result.coordinates.size());
352  polygonlineset->numVertices.setNum(result.indices.size());
353 
354  int32_t* index = polygonlineset->numVertices.startEditing();
355  SbVec3f* vts = polygoncoords->point.startEditing();
356 
357  for (size_t i = 0; i < result.coordinates.size(); i++) {
358  vts[i].setValue(result.coordinates[i].x,
359  result.coordinates[i].y,
360  ViewProviderSketchCoinAttorney::getViewOrientationFactor(viewProvider)
361  * drawingParameters.zInfo);
362  }
363 
364  for (size_t i = 0; i < result.indices.size(); i++) {
365  index[i] = result.indices[i];
366  }
367 
368  polygoncoords->point.finishEditing();
369  polygonlineset->numVertices.finishEditing();

405 template<typename Result>
406 void EditModeInformationOverlayCoinConverter::addNode(const Result& result)
407 {
408 
409  if constexpr (Result::visualisationType == VisualisationType::Text) {
410 
411  for (size_t i = 0; i < result.strings.size(); i++) {
412 
413  SoSwitch* sw = new SoSwitch();
414 
415  sw->whichChild = isVisible<Result::calculationType>() ? SO_SWITCH_ALL : SO_SWITCH_NONE;
416 
417  SoSeparator* sep = new SoSeparator();
418  sep->ref();
419  // no caching for frequently-changing data structures
420  sep->renderCaching = SoSeparator::OFF;
421 
422  // every information visual node gets its own material for to-be-implemented
423  // preselection and selection
424  SoMaterial* mat = new SoMaterial;
425  mat->ref();
426  mat->diffuseColor = drawingParameters.InformationColor;
427 
428  SoTranslation* translate = new SoTranslation;
429 
430  translate->translation.setValue(
431  result.positions[i].x,
432  result.positions[i].y,
433  ViewProviderSketchCoinAttorney::getViewOrientationFactor(viewProvider)
434  * drawingParameters.zInfo);
435 
436  SoFont* font = new SoFont;
437  font->name.setValue("Helvetica");
438  font->size.setValue(drawingParameters.coinFontSize);
439 
440  SoText2* text = new SoText2;
441 
442  // since the first and last control point of a spline is also treated as knot and thus
443  // can also have a displayed multiplicity, we must assure the multiplicity is not
444  // visibly overwritten therefore be output the weight in a second line
445  //
446  // This could be made into a more generic form, but it is probably not worth the effort
447  // at this time.
448  if constexpr (Result::calculationType == CalculationType::BSplinePoleWeight) {
449  setText<2>(result.strings[i], text);
450  }
451  else {
452  setText(result.strings[i], text);
453  }
454 
455  sep->addChild(mat);
456  sep->addChild(font);
457  sep->addChild(translate);
458  sep->addChild(text);
459 
460  sw->addChild(sep);
461 
462  addToInfoGroup(sw);
463  sep->unref();
464  mat->unref();
465  }
466  }
467  else if constexpr (Result::visualisationType == VisualisationType::Polygon) {
468 
469  SoSwitch* sw = new SoSwitch();
470 
471  // hGrpsk->GetBool("BSplineControlPolygonVisible", true)
472  sw->whichChild = isVisible<Result::calculationType>() ? SO_SWITCH_ALL : SO_SWITCH_NONE;
473 
474  SoSeparator* sep = new SoSeparator();
475  sep->ref();
476  // no caching for frequently-changing data structures
477  sep->renderCaching = SoSeparator::OFF;
478 
479  // every information visual node gets its own material for to-be-implemented preselection
480  // and selection
481  SoMaterial* mat = new SoMaterial;
482  mat->ref();
483  mat->diffuseColor = drawingParameters.InformationColor;
484 
485  SoLineSet* polygonlineset = new SoLineSet;
486  SoCoordinate3* polygoncoords = new SoCoordinate3;
487 
488  setPolygon<Result>(result, polygonlineset, polygoncoords);
489 
490  sep->addChild(mat);
491  sep->addChild(polygoncoords);
492  sep->addChild(polygonlineset);
493 
494  sw->addChild(sep);
495 
496  addToInfoGroup(sw);
497  sep->unref();
498  mat->unref();
499  }

502 template<typename Result>
503 void EditModeInformationOverlayCoinConverter::updateNode(const Result& result)
504 {
505 
506  if constexpr (Result::visualisationType == VisualisationType::Text) {
507 
508  for (size_t i = 0; i < result.strings.size(); i++) {
509  SoSwitch* sw = static_cast<SoSwitch*>(infoGroup->getChild(nodeId));
510 
511  if (overlayParameters.visibleInformationChanged) {
512  sw->whichChild =
513  isVisible<Result::calculationType>() ? SO_SWITCH_ALL : SO_SWITCH_NONE;

515 
516  SoSeparator* sep = static_cast<SoSeparator*>(sw->getChild(0));
517 
518  static_cast<SoTranslation*>(
519  sep->getChild(static_cast<int>(TextNodePosition::TextCoordinates)))
520  ->translation.setValue(
521  result.positions[i].x,
522  result.positions[i].y,
523  ViewProviderSketchCoinAttorney::getViewOrientationFactor(viewProvider)
524  * drawingParameters.zInfo);
525 
526  // since the first and last control point of a spline is also treated as knot and thus
527  // can also have a displayed multiplicity, we must assure the multiplicity is not
528  // visibly overwritten therefore be output the weight in a second line
529  //
530  // This could be made into a more generic form, but it is probably not worth the effort
531  // at this time.
532  if constexpr (Result::calculationType == CalculationType::BSplinePoleWeight) {
533  setText<2>(result.strings[i],
534  static_cast<SoText2*>(
535  sep->getChild(static_cast<int>(TextNodePosition::TextInformation))));
536  }
537  else {
538  setText(result.strings[i],
539  static_cast<SoText2*>(
540  sep->getChild(static_cast<int>(TextNodePosition::TextInformation))));
541  }
542 
543  nodeId++;
544  }
545  }
546  else if constexpr (Result::visualisationType == VisualisationType::Polygon) {
547 
548  SoSwitch* sw = static_cast<SoSwitch*>(infoGroup->getChild(nodeId));
549 
550  if (overlayParameters.visibleInformationChanged) {
551  sw->whichChild = isVisible<Result::calculationType>() ? SO_SWITCH_ALL : SO_SWITCH_NONE;
552  }
553 
554  SoSeparator* sep = static_cast<SoSeparator*>(sw->getChild(0));
555 
556  SoCoordinate3* polygoncoords = static_cast<SoCoordinate3*>(
557  sep->getChild(static_cast<int>(PolygonNodePosition::PolygonCoordinates)));
558 
559  SoLineSet* polygonlineset = static_cast<SoLineSet*>(
560  sep->getChild(static_cast<int>(PolygonNodePosition::PolygonLineSet)));
561 
562  setPolygon(result, polygonlineset, polygoncoords);
563 
564  nodeId++;
