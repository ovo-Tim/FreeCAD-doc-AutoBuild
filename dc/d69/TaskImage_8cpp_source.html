289 void TaskImage::restoreAngles(const Base::Rotation& rot)
290 {
291  Base::Vector3d vec(0, 0, 1);
292  rot.multVec(vec, vec);
293 
294  double yaw {};
295  double pitch {};
296  double roll {};
297  rot.getYawPitchRoll(yaw, pitch, roll);
298 
299  bool reverse = false;
300 
301  const double tol = 1.0e-5;
302  const double angle1 = 90.0;
303  const double angle2 = 180.0;
304 
305  auto isTopOrBottom = [=](bool& reverse) {
306  if (std::fabs(vec.z - 1.0) < tol) {
307  return true;
308  }
309  if (std::fabs(vec.z + 1.0) < tol) {
310  reverse = true;
311  return true;
312  }
313 
314  return false;
315  };
316 
317  auto isFrontOrRear = [&](bool& reverse) {
318  if (std::fabs(vec.y + 1.0) < tol) {
319  if (std::fabs(yaw - angle2) < tol) {
320  pitch = -angle2 - pitch;
321  }
322  return true;
323  }
324  if (std::fabs(vec.y - 1.0) < tol) {
325  if (std::fabs(yaw) < tol) {
326  pitch = -angle2 - pitch;
327  }
328  reverse = true;
329  return true;
330  }
331 
332  return false;
333  };
334 
335  auto isRightOrLeft = [&](bool& reverse) {
336  if (std::fabs(vec.x - 1.0) < tol) {
337  if (std::fabs(yaw + angle1) < tol) {
338  pitch = -angle2 - pitch;
339  }
340  return true;
341  }
342  if (std::fabs(vec.x + 1.0) < tol) {
343  if (std::fabs(yaw - angle1) < tol) {
344  pitch = -angle2 - pitch;
345  }
346  reverse = true;
347  return true;
348  }
349 
350  return false;
351  };
352 
353  if (isTopOrBottom(reverse)) {
354  int inv = reverse ? -1 : 1;
355  ui->XY_radioButton->setChecked(true);
356  ui->spinBoxRotation->setValue(yaw * inv);
357  }
358  else if (isFrontOrRear(reverse)) {
359  ui->XZ_radioButton->setChecked(true);
360  ui->spinBoxRotation->setValue(-pitch);
361  }
362  else if (isRightOrLeft(reverse)) {
363  ui->YZ_radioButton->setChecked(true);
364  ui->spinBoxRotation->setValue(-pitch);
365  }
366  ui->Reverse_checkBox->setChecked(reverse);
367 }

369 void TaskImage::restore(const Base::Placement& plm)
370 {
371  if (feature.expired()) {
372  return;
373  }
374 
375  QSignalBlocker blockW(ui->spinBoxWidth);
376  QSignalBlocker blockH(ui->spinBoxHeight);
377  ui->spinBoxWidth->setValue(feature->XSize.getValue());
378  ui->spinBoxHeight->setValue(feature->YSize.getValue());
379 
380  Base::Rotation rot = plm.getRotation(); // NOLINT
381  Base::Vector3d pos = plm.getPosition();
382 
383  restoreAngles(rot);
384 
385  Base::Vector3d R0(0, 0, 0);
386  Base::Vector3d RX(1, 0, 0);
387  Base::Vector3d RY(0, 1, 0);
388 
389  RX = rot.multVec(RX);
390  RY = rot.multVec(RY);
391  pos.TransformToCoordinateSystem(R0, RX, RY);
392  ui->spinBoxX->setValue(pos.x);
393  ui->spinBoxY->setValue(pos.y);
394  ui->spinBoxZ->setValue(pos.z);
395 
396  onPreview();
397 }

399 void TaskImage::updatePlacement()
400 {
401  double angle = ui->spinBoxRotation->value().getValue();
402  bool reverse = ui->Reverse_checkBox->isChecked();
403 
404  // NOLINTBEGIN
405  Base::Placement Pos;
406  Base::Rotation rot;
407  double dir = reverse ? 180. : 0.;
408  int inv = reverse ? -1 : 1;
409 
410  if (ui->XY_radioButton->isChecked()) {
411  rot.setYawPitchRoll(inv * angle, 0., dir);
412  }
413  else if (ui->XZ_radioButton->isChecked()) {
414  rot.setYawPitchRoll(dir, -angle, 90.);
415  }
416  else if (ui->YZ_radioButton->isChecked()) {
417  rot.setYawPitchRoll(90. - dir, -angle, 90.);
418  }
419  else if (!feature.expired()) {
420  Base::Placement plm = feature->Placement.getValue();
421  rot = plm.getRotation();
422  }
423  // NOLINTEND
424 
425  Base::Vector3d offset = Base::Vector3d(ui->spinBoxX->value().getValue(),
426  ui->spinBoxY->value().getValue(),
427  ui->spinBoxZ->value().getValue());
428  offset = rot.multVec(offset);
429  Pos = Base::Placement(offset, rot);
430 
431  if (!feature.expired()) {
432  feature->Placement.setValue(Pos);
433  if (scale) {
434  scale->setPlacement(feature->globalPlacement());
435  }
436  }
437 }

556 void InteractiveScale::collectPoint(const SbVec3f& pos3d)
557 {
558  if (points.empty()) {
559  points.push_back(pos3d);
560 
561  measureLabel->label->setPoints(getCoordsOnImagePlane(pos3d), getCoordsOnImagePlane(pos3d));
562  measureLabel->activate();
563  }
564  else if (points.size() == 1) {
565  double distance = getDistance(pos3d);
566  if (distance > Base::Precision::Confusion()) {
567  points.push_back(pos3d);
568 
569  midPoint = (points[0] + points[1]) / 2;
570 
571  measureLabel->startEdit(getDistance(points[1]), this, true);
572 
573  Q_EMIT enableApplyBtn();
574  }
575  else {
576  Base::Console().Warning(std::string("Image scale"),
577  "The second point is too close. Retry!\n");
578  }
579  }

599 void InteractiveScale::soEventFilter(void* ud, SoEventCallback* ecb)
600 {
601  auto scale = static_cast<InteractiveScale*>(ud);
602 
603  const SoEvent* soEvent = ecb->getEvent();
604  if (soEvent->isOfType(SoKeyboardEvent::getClassTypeId())) {
605  /* If user presses escape, then we cancel the tool.*/
606  const auto kbe = static_cast<const SoKeyboardEvent*>(soEvent); // NOLINT
607 
608  if (kbe->getKey() == SoKeyboardEvent::ESCAPE && kbe->getState() == SoButtonEvent::UP) {
609  ecb->setHandled();
610  Q_EMIT scale->scaleCanceled();
611  }
612  }
613  else if (soEvent->isOfType(SoMouseButtonEvent::getClassTypeId())) {
614  const auto mbe = static_cast<const SoMouseButtonEvent*>(soEvent); // NOLINT
615 
616  if (mbe->getButton() == SoMouseButtonEvent::BUTTON1
617  && mbe->getState() == SoButtonEvent::DOWN) {
618  ecb->setHandled();
619  scale->findPointOnImagePlane(ecb);
620  }
621  if (mbe->getButton() == SoMouseButtonEvent::BUTTON2
622  && mbe->getState() == SoButtonEvent::DOWN) {
623  ecb->setHandled();
624  Q_EMIT scale->scaleCanceled();
625  }
626  }
627 }

629 bool InteractiveScale::eventFilter(QObject* object, QEvent* event)
630 {
631  if (event->type() == QEvent::KeyRelease) {
632  auto keyEvent = static_cast<QKeyEvent*>(event); // NOLINT
633 
634  /* If user press enter in the spinbox, then we validate the tool.*/
635  if ((keyEvent->key() == Qt::Key_Enter || keyEvent->key() == Qt::Key_Return)
636  && dynamic_cast<QuantitySpinBox*>(object)) {
637  Q_EMIT scaleRequired();
638  }
639 
640  /* If user press escape, then we cancel the tool. Required here as well for when checkbox
641  * has focus.*/
642  if (keyEvent->key() == Qt::Key_Escape) {
643  Q_EMIT scaleCanceled();
644  }
645  }
646  return false;
647 }

655 SbVec3f InteractiveScale::getCoordsOnImagePlane(const SbVec3f& point)
656 {
657  // Plane form
658  Base::Vector3d RX(1, 0, 0);
659  Base::Vector3d RY(0, 1, 0);
660 
661  // move to position of Sketch
662  Base::Rotation tmp(placement.getRotation());
663  RX = tmp.multVec(RX);
664  RY = tmp.multVec(RY);
665  Base::Vector3d pos = placement.getPosition();
666 
667  // we use pos as the Base because in setPlacement we set transform->translation using
668  // placement.getPosition() to fix the Zoffset. But this applies the X & Y translation too.
669  Base::Vector3d pnt(point[0], point[1], point[2]);
670  pnt.TransformToCoordinateSystem(pos, RX, RY);
671 
672  return {float(pnt.x), float(pnt.y), 0.0F};
673 }
