126 //remove a subElement(Face) from Source PropertyLinkSub
127 bool DrawHatch::removeSub(std::string toRemove)
128 {
129 // Base::Console().Message("DH::removeSub(%s)\n", toRemove.c_str());
130  bool removed = false;
131  const std::vector<std::string> &sourceNames = Source.getSubValues();
132  std::vector<std::string> newList;
133  App::DocumentObject* sourceFeat = Source.getValue();
134  for (auto& s: sourceNames) {
135  if (s == toRemove) {
136  removed = true;
137  } else {
138  newList.push_back(s);
139  }
140  }
141  if (removed) {
142  Source.setValue(sourceFeat, newList);
143  }
144  return removed;
145 }
