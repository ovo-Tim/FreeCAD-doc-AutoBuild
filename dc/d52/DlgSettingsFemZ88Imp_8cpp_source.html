103 void DlgSettingsFemZ88Imp::onfileNameChanged(QString FileName)
104 {
105  if (!QFileInfo::exists(FileName)) {
106  QMessageBox::critical(this,
107  tr("File does not exist"),
108  tr("The specified z88r executable\n'%1'\n does not exist!\n"
109  "Specify another file please.")
110  .arg(FileName));
111  return;
112  }
113 
114  // since the Z88 folder is full of files like "z88h", "z88o" etc. one can easily make a
115  // mistake and is then lost why the solver fails. Therefore check for the correct filename.
116  auto strName = FileName.toStdString();
117 #if defined(FC_OS_WIN32)
118  if (strName.substr(strName.length() - 8) != "z88r.exe") {
119  QMessageBox::critical(this,
120  tr("Wrong file"),
121  tr("You must specify the path to the z88r.exe!"));
122  return;
123  }
124 #elif defined(FC_OS_LINUX) || defined(FC_OS_CYGWIN) || defined(FC_OS_MACOSX) || defined(FC_OS_BSD)
125  if (strName.substr(strName.length() - 4) != "z88r") {
126  QMessageBox::critical(this, tr("Wrong file"), tr("You must specify the path to the z88r!"));
127  return;
128  }
129 #endif
130 }
