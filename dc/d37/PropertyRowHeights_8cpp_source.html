121  int Cnt;
122 
123  // Row info
124  reader.readElement("RowInfo");
125  Cnt = reader.hasAttribute("Count") ? reader.getAttributeAsInteger("Count") : 0;
126  for (int i = 0; i < Cnt; i++) {
127  reader.readElement("Row");
128  const char* name = reader.hasAttribute("name") ? reader.getAttribute("name") : nullptr;
129  const char* height =
130  reader.hasAttribute("height") ? reader.getAttribute("height") : nullptr;
131 
132  try {
133  if (name && height) {
134  int row = App::decodeRow(name);
135  int rowHeight = atoi(height);
136

139  }
