141 int BSplineBasis::FindSpan(double fParam)
142 {
143  int n = _vKnotVector.Length() - _iOrder - 1;
144  if (fParam == _vKnotVector(n + 1)) {
145  return n;
146  }
147 
148  int low = _iOrder - 1;
149  int high = n + 1;
150  int mid = (low + high) / 2; // Binary search
151 
152  while (fParam < _vKnotVector(mid) || fParam >= _vKnotVector(mid + 1)) {
153  if (fParam < _vKnotVector(mid)) {
154  high = mid;
155  }
156  else {
157  low = mid;
158  }
159  mid = (low + high) / 2;
160  }
161 
162  return mid;
163 }

165 void BSplineBasis::AllBasisFunctions(double fParam, TColStd_Array1OfReal& vFuncVals)
166 {
167  if (vFuncVals.Length() != _iOrder) {
168  Standard_RangeError::Raise("BSplineBasis");
169  }
170 
171  int iIndex = FindSpan(fParam);
172 
173  TColStd_Array1OfReal zaehler_left(1, _iOrder - 1);
174  TColStd_Array1OfReal zaehler_right(1, _iOrder - 1);
175  vFuncVals(0) = 1.0;
176 
177  for (int j = 1; j < _iOrder; j++) {
178  zaehler_left(j) = fParam - _vKnotVector(iIndex + 1 - j);
179  zaehler_right(j) = _vKnotVector(iIndex + j) - fParam;
180  double saved = 0.0;
181  for (int r = 0; r < j; r++) {
182  double tmp = vFuncVals(r) / (zaehler_right(r + 1) + zaehler_left(j - r));
183  vFuncVals(r) = saved + zaehler_right(r + 1) * tmp;
184  saved = zaehler_left(j - r) * tmp;
185  }
186 
187  vFuncVals(j) = saved;
188  }

208 double BSplineBasis::BasisFunction(int iIndex, double fParam)
209 {
210  int m = _vKnotVector.Length() - 1;
211  int p = _iOrder - 1;
212  double saved;
213  TColStd_Array1OfReal N(0, p);
214 
215  if ((iIndex == 0 && fParam == _vKnotVector(0))
216  || (iIndex == m - p - 1 && fParam == _vKnotVector(m))) {
217  return 1.0;
218  }
219 
220  if (fParam < _vKnotVector(iIndex) || fParam >= _vKnotVector(iIndex + p + 1)) {
221  return 0.0;
222  }
223 
224  for (int j = 0; j <= p; j++) {
225  if (fParam >= _vKnotVector(iIndex + j) && fParam < _vKnotVector(iIndex + j + 1)) {
226  N(j) = 1.0;
227  }
228  else {
229  N(j) = 0.0;
230  }
231  }
232 
233  for (int k = 1; k <= p; k++) {
234  if (N(0) == 0.0) {
235  saved = 0.0;
236  }
237  else {
238  saved = ((fParam - _vKnotVector(iIndex)) * N(0))
239  / (_vKnotVector(iIndex + k) - _vKnotVector(iIndex));
240  }
241 
242  for (int j = 0; j < p - k + 1; j++) {
243  double Tleft = _vKnotVector(iIndex + j + 1);
244  double Tright = _vKnotVector(iIndex + j + k + 1);
245 
246  if (N(j + 1) == 0.0) {
247  N(j) = saved;
248  saved = 0.0;
249  }
250  else {
251  double tmp = N(j + 1) / (Tright - Tleft);
252  N(j) = saved + (Tright - fParam) * tmp;
253  saved = (fParam - Tleft) * tmp;
254  }
255  }
256  }
257 
258  return N(0);

261 void BSplineBasis::DerivativesOfBasisFunction(int iIndex,
262  int iMaxDer,
263  double fParam,
264  TColStd_Array1OfReal& Derivat)
265 {
266  int iMax = iMaxDer;
267  if (Derivat.Length() != iMax + 1) {
268  Standard_RangeError::Raise("BSplineBasis");
269  }
270  // kth derivatives (k> degrees) are zero
271  if (iMax >= _iOrder) {
272  for (int i = _iOrder; i <= iMaxDer; i++) {
273  Derivat(i) = 0.0;
274  }
275  iMax = _iOrder - 1;
276  }
277 
278  TColStd_Array1OfReal ND(0, iMax);
279  int p = _iOrder - 1;
280  math_Matrix N(0, p, 0, p);
281  double saved;
282 
283  // if value is outside the interval, then function value and all derivatives equal null
284  if (fParam < _vKnotVector(iIndex) || fParam >= _vKnotVector(iIndex + p + 1)) {
285  for (int k = 0; k <= iMax; k++) {
286  Derivat(k) = 0.0;
287  }
288  return;
289  }
290 
291  // Calculate the basis functions of Order 1
292  for (int j = 0; j < _iOrder; j++) {
293  if (fParam >= _vKnotVector(iIndex + j) && fParam < _vKnotVector(iIndex + j + 1)) {
294  N(j, 0) = 1.0;
295  }
296  else {
297  N(j, 0) = 0.0;
298  }
299  }
300 
301  // Calculate a triangular table of the function values
302  for (int k = 1; k < _iOrder; k++) {
303  if (N(0, k - 1) == 0.0) {
304  saved = 0.0;
305  }
306  else {
307  saved = ((fParam - _vKnotVector(iIndex)) * N(0, k - 1))
308  / (_vKnotVector(iIndex + k) - _vKnotVector(iIndex));
309  }
310  for (int j = 0; j < p - k + 1; j++) {
311  double Tleft = _vKnotVector(iIndex + j + 1);
312  double Tright = _vKnotVector(iIndex + j + k + 1);
313 
314  if (N(j + 1, k - 1) == 0.0) {
315  N(j, k) = saved;
316  saved = 0.0;
317  }
318  else {
319  double tmp = N(j + 1, k - 1) / (Tright - Tleft);
320  N(j, k) = saved + (Tright - fParam) * tmp;
321  saved = (fParam - Tleft) * tmp;
322  }
323  }
324  }
325 
326  // Function value
327  Derivat(0) = N(0, p);
328  // Calculate the derivatives from the triangle table
329  for (int k = 1; k <= iMax; k++) {
330  for (int j = 0; j <= k; j++) {
331  // Load the (p-k)th column
332  ND(j) = N(j, p - k);
333  }
334 
335  for (int jj = 1; jj <= k; jj++) {
336  if (ND(0) == 0.0) {
337  saved = 0.0;
338  }
339  else {
340  saved = ND(0) / (_vKnotVector(iIndex + p - k + jj) - _vKnotVector(iIndex));
341  }
342 
343  for (int j = 0; j < k - jj + 1; j++) {
344  double Tleft = _vKnotVector(iIndex + j + 1);
345  double Tright = _vKnotVector(iIndex + j + p - k + jj + 1);
346  if (ND(j + 1) == 0.0) {
347  ND(j) = (p - k + jj) * saved;
348  saved = 0.0;
349  }
350  else {
351  double tmp = ND(j + 1) / (Tright - Tleft);
352  ND(j) = (p - k + jj) * (saved - tmp);
353  saved = tmp;
354  }
355  }
356  }
357 
358  Derivat(k) = ND(0); // kth derivative
359  }
360 }

362 double BSplineBasis::DerivativeOfBasisFunction(int iIndex, int iMaxDer, double fParam)
363 {
364  int iMax = iMaxDer;
365 
366  // Function value (0th derivative)
367  if (iMax == 0) {
368  return BasisFunction(iIndex, fParam);
369  }
370 
371  // The kth derivatives (k>degrees) are null
372  if (iMax >= _iOrder) {
373  return 0.0;
374  }
375 
376  TColStd_Array1OfReal ND(0, iMax);
377  int p = _iOrder - 1;
378  math_Matrix N(0, p, 0, p);
379  double saved;
380 
381  // If value is outside the interval, then function value and derivatives equal null
382  if (fParam < _vKnotVector(iIndex) || fParam >= _vKnotVector(iIndex + p + 1)) {
383  return 0.0;
384  }
385 
386  // Calculate the basis functions of Order 1
387  for (int j = 0; j < _iOrder; j++) {
388  if (fParam >= _vKnotVector(iIndex + j) && fParam < _vKnotVector(iIndex + j + 1)) {
389  N(j, 0) = 1.0;
390  }
391  else {
392  N(j, 0) = 0.0;
393  }
394  }
395 
396  // Calculate triangular table of function values
397  for (int k = 1; k < _iOrder; k++) {
398  if (N(0, k - 1) == 0.0) {
399  saved = 0.0;
400  }
401  else {
402  saved = ((fParam - _vKnotVector(iIndex)) * N(0, k - 1))
403  / (_vKnotVector(iIndex + k) - _vKnotVector(iIndex));
404  }
405 
406  for (int j = 0; j < p - k + 1; j++) {
407  double Tleft = _vKnotVector(iIndex + j + 1);
408  double Tright = _vKnotVector(iIndex + j + k + 1);
409 
410  if (N(j + 1, k - 1) == 0.0) {
411  N(j, k) = saved;
412  saved = 0.0;
413  }
414  else {
415  double tmp = N(j + 1, k - 1) / (Tright - Tleft);
416  N(j, k) = saved + (Tright - fParam) * tmp;
417  saved = (fParam - Tleft) * tmp;
418  }
419  }
420  }
421 
422  // Use the triangular table to calculate the derivatives
423  for (int j = 0; j <= iMax; j++) {
424  // Loading (p-iMax)th column
425  ND(j) = N(j, p - iMax);
426  }
427 
428  for (int jj = 1; jj <= iMax; jj++) {
429  if (ND(0) == 0.0) {
430  saved = 0.0;
431  }
432  else {
433  saved = ND(0) / (_vKnotVector(iIndex + p - iMax + jj) - _vKnotVector(iIndex));
434  }
435 
436  for (int j = 0; j < iMax - jj + 1; j++) {
437  double Tleft = _vKnotVector(iIndex + j + 1);
438  double Tright = _vKnotVector(iIndex + j + p - iMax + jj + 1);
439  if (ND(j + 1) == 0.0) {
440  ND(j) = (p - iMax + jj) * saved;
441  saved = 0.0;
442  }
443  else {
444  double tmp = ND(j + 1) / (Tright - Tleft);
445  ND(j) = (p - iMax + jj) * (saved - tmp);
446  saved = tmp;
447  }
448  }
449  }
450 
451  return ND(0); // iMax-th derivative
452 }

695 bool ParameterCorrection::GetUVParameters(double fSizeFactor)
696 {
697  // Eigenvectors as a new base
698  Base::Vector3d e[3];
699  e[0] = _clU;
700  e[1] = _clV;
701  e[2] = _clW;
702 
703  // Canonical base of R^3
704  Base::Vector3d b[3];
705  b[0] = Base::Vector3d(1.0, 0.0, 0.0);
706  b[1] = Base::Vector3d(0.0, 1.0, 0.0);
707  b[2] = Base::Vector3d(0.0, 0.0, 1.0);
708  // Create a right system from the orthogonal eigenvectors
709  if ((e[0] % e[1]) * e[2] < 0) {
710  Base::Vector3d tmp = e[0];
711  e[0] = e[1];
712  e[1] = tmp;
713  }
714 
715  // Now generate the transpon. Rotation matrix
716  Wm4::Matrix3d clRotMatTrans;
717  for (int i = 0; i < 3; i++) {
718  for (int j = 0; j < 3; j++) {
719  clRotMatTrans[i][j] = b[j] * e[i];
720  }
721  }
722 
723  std::vector<Base::Vector2d> vcProjPts;
724  Base::BoundBox2d clBBox;
725 
726  // Calculate the coordinates of the transf. Points and project
727  // these on to the x,y-plane of the new coordinate system
728  for (int ii = _pvcPoints->Lower(); ii <= _pvcPoints->Upper(); ii++) {
729  const gp_Pnt& pnt = (*_pvcPoints)(ii);
730  Wm4::Vector3d clProjPnt = clRotMatTrans * Wm4::Vector3d(pnt.X(), pnt.Y(), pnt.Z());
731  vcProjPts.emplace_back(clProjPnt.X(), clProjPnt.Y());
732  clBBox.Add(Base::Vector2d(clProjPnt.X(), clProjPnt.Y()));
733  }
734 
735  if ((clBBox.MaxX == clBBox.MinX) || (clBBox.MaxY == clBBox.MinY)) {
736  return false;
737  }
738  double tx = fSizeFactor * clBBox.MinX - (fSizeFactor - 1.0) * clBBox.MaxX;
739  double ty = fSizeFactor * clBBox.MinY - (fSizeFactor - 1.0) * clBBox.MaxY;
740  double fDeltaX = (2 * fSizeFactor - 1.0) * (clBBox.MaxX - clBBox.MinX);
741  double fDeltaY = (2 * fSizeFactor - 1.0) * (clBBox.MaxY - clBBox.MinY);
742 
743  // Calculate the u,v parameters with u,v from [0,1]
744  _pvcUVParam->Init(gp_Pnt2d(0.0, 0.0));
745  int ii = 0;
746  if (clBBox.MaxX - clBBox.MinX >= clBBox.MaxY - clBBox.MinY) {
747  for (const auto& pt : vcProjPts) {
748  (*_pvcUVParam)(ii) = gp_Pnt2d((pt.x - tx) / fDeltaX, (pt.y - ty) / fDeltaY);
749  ii++;
750  }
751  }
752  else {
753  for (const auto& pt : vcProjPts) {
754  (*_pvcUVParam)(ii) = gp_Pnt2d((pt.y - ty) / fDeltaY, (pt.x - tx) / fDeltaX);
755  ii++;
756  }
757  }
758 
759  return true;
760 }

961 void BSplineParameterCorrection::DoParameterCorrection(int iIter)
962 {
963  int i = 0;
964  double fMaxDiff = 0.0, fMaxScalar = 1.0;
965  double fWeight = _fSmoothInfluence;
966 
967  Base::SequencerLauncher seq("Calc surface...", iIter * _pvcPoints->Length());
968 
969  do {
970  fMaxScalar = 1.0;
971  fMaxDiff = 0.0;
972 
973  Handle(Geom_BSplineSurface) pclBSplineSurf = new Geom_BSplineSurface(_vCtrlPntsOfSurf,
974  _vUKnots,
975  _vVKnots,
976  _vUMults,
977  _vVMults,
978  _usUOrder - 1,
979  _usVOrder - 1);
980 
981  for (int ii = _pvcPoints->Lower(); ii <= _pvcPoints->Upper(); ii++) {
982  double fDeltaU, fDeltaV, fU, fV;
983  const gp_Pnt& pnt = (*_pvcPoints)(ii);
984  gp_Vec P(pnt.X(), pnt.Y(), pnt.Z());
985  gp_Pnt PntX;
986  gp_Vec Xu, Xv, Xuv, Xuu, Xvv;
987  // Calculate the first two derivatives and point at (u,v)
988  gp_Pnt2d& uvValue = (*_pvcUVParam)(ii);
989  pclBSplineSurf->D2(uvValue.X(), uvValue.Y(), PntX, Xu, Xv, Xuu, Xvv, Xuv);
990  gp_Vec X(PntX.X(), PntX.Y(), PntX.Z());
991  gp_Vec ErrorVec = X - P;
992 
993  // Calculate Xu x Xv the normal in X(u,v)
994  gp_Dir clNormal = Xu ^ Xv;
995 
996  // Check, if X = P
997  if (!(X.IsEqual(P, 0.001, 0.001))) {
998  ErrorVec.Normalize();
999  if (fabs(clNormal * ErrorVec) < fMaxScalar) {
1000  fMaxScalar = fabs(clNormal * ErrorVec);
1001  }
1002  }
1003 
1004  fDeltaU = ((P - X) * Xu) / ((P - X) * Xuu - Xu * Xu);
1005  if (fabs(fDeltaU) < Precision::Confusion()) {
1006  fDeltaU = 0.0;
1007  }
1008  fDeltaV = ((P - X) * Xv) / ((P - X) * Xvv - Xv * Xv);
1009  if (fabs(fDeltaV) < Precision::Confusion()) {
1010  fDeltaV = 0.0;
1011  }
1012 
1013  // Replace old u/v values with new ones
1014  fU = uvValue.X() - fDeltaU;
1015  fV = uvValue.Y() - fDeltaV;
1016  if (fU <= 1.0 && fU >= 0.0 && fV <= 1.0 && fV >= 0.0) {
1017  uvValue.SetX(fU);
1018  uvValue.SetY(fV);
1019  fMaxDiff = std::max<double>(fabs(fDeltaU), fMaxDiff);
1020  fMaxDiff = std::max<double>(fabs(fDeltaV), fMaxDiff);
1021  }
1022 
1023  seq.next();
1024  }
1025 
1026  if (_bSmoothing) {
1027  fWeight *= 0.5f;
1028  SolveWithSmoothing(fWeight);
1029  }
1030  else {
1031  SolveWithoutSmoothing();
1032  }
1033 
1034  i++;
1035  } while (i < iIter && fMaxDiff > Precision::Confusion() && fMaxScalar < 0.99);
1036 }

1038 bool BSplineParameterCorrection::SolveWithoutSmoothing()
1039 {
1040  unsigned ulSize = _pvcPoints->Length();
1041  unsigned ulDim = _usUCtrlpoints * _usVCtrlpoints;
1042  math_Matrix M(0, ulSize - 1, 0, ulDim - 1);
1043  math_Matrix Xx(0, ulDim - 1, 0, 0);
1044  math_Matrix Xy(0, ulDim - 1, 0, 0);
1045  math_Matrix Xz(0, ulDim - 1, 0, 0);
1046  math_Vector bx(0, ulSize - 1);
1047  math_Vector by(0, ulSize - 1);
1048  math_Vector bz(0, ulSize - 1);
1049 
1050  // Determining the coefficient matrix of the overdetermined LGS
1051  for (unsigned i = 0; i < ulSize; i++) {
1052  const gp_Pnt2d& uvValue = (*_pvcUVParam)(i);
1053  double fU = uvValue.X();
1054  double fV = uvValue.Y();
1055  unsigned ulIdx = 0;
1056 
1057  // Pre-calculation of the values of the base functions
1058  std::vector<double> basisU(_usUCtrlpoints);
1059  for (unsigned j = 0; j < _usUCtrlpoints; j++) {
1060  basisU[j] = _clUSpline.BasisFunction(j, fU);
1061  }
1062  std::vector<double> basisV(_usVCtrlpoints);
1063  for (unsigned k = 0; k < _usVCtrlpoints; k++) {
1064  basisV[k] = _clVSpline.BasisFunction(k, fV);
1065  }
1066 
1067  for (unsigned j = 0; j < _usUCtrlpoints; j++) {
1068  double valueU = basisU[j];
1069  if (valueU == 0.0) {
1070  for (unsigned k = 0; k < _usVCtrlpoints; k++) {
1071  M(i, ulIdx) = 0.0;
1072  ulIdx++;
1073  }
1074  }
1075  else {
1076  for (unsigned k = 0; k < _usVCtrlpoints; k++) {
1077  M(i, ulIdx) = valueU * basisV[k];
1078  ulIdx++;
1079  }
1080  }
1081  }
1082  }
1083 
1084  // Determine the right side
1085  for (int ii = _pvcPoints->Lower(); ii <= _pvcPoints->Upper(); ii++) {
1086  const gp_Pnt& pnt = (*_pvcPoints)(ii);
1087  bx(ii) = pnt.X();
1088  by(ii) = pnt.Y();
1089  bz(ii) = pnt.Z();
1090  }
1091 
1092  // Solve the over-determined LGS with Householder-Transformation
1093  math_Householder hhX(M, bx);
1094  math_Householder hhY(M, by);
1095  math_Householder hhZ(M, bz);
1096 
1097  if (!(hhX.IsDone() && hhY.IsDone() && hhZ.IsDone())) {
1098  // LGS could not be solved
1099  return false;
1100  }
1101  Xx = hhX.AllValues();
1102  Xy = hhY.AllValues();
1103  Xz = hhZ.AllValues();
1104 
1105  unsigned ulIdx = 0;
1106  for (unsigned j = 0; j < _usUCtrlpoints; j++) {
1107  for (unsigned k = 0; k < _usVCtrlpoints; k++) {
1108  _vCtrlPntsOfSurf(j, k) = gp_Pnt(Xx(ulIdx, 0), Xy(ulIdx, 0), Xz(ulIdx, 0));
1109  ulIdx++;
1110  }
1111  }
1112 
1113  return true;
1114 }
