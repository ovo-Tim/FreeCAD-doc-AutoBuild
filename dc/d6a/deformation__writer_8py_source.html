175  def handleDeformationMaterial(self, bodies):

179  for equation in self.solver.Group:

189  if tempObj is not None:

191  for name in bodies:

195  m = obj.Material

197  for name in (n for n in refs if n in bodies):
198  # don't evaluate fluid material

200  break
