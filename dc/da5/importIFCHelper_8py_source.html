50 def dd2dms(dd):
51  """Convert decimal degrees to degrees, minutes, seconds.
52 
53  Used in export.
54  """
55  sign = 1 if dd >= 0 else -1
56  dd = abs(dd)
57  minutes, seconds = divmod(dd * 3600, 60)
58  degrees, minutes = divmod(minutes, 60)
59 
60  if dd < 0:
61  degrees = -degrees
62 
63  return (int(degrees) * sign,
64  int(minutes) * sign,
65  int(seconds) * sign)

71  Used in import.

73  dd = float(degrees) + float(minutes)/60 + float(seconds)/3600

464 def getColorFromStyledItem(styled_item):
465  """Get color from the IfcStyledItem.
466 
467  Returns
468  -------
469  float, float, float, int
470  A tuple with the red, green, blue, and transparency values.
471  If the `IfcStyledItem` is a `IfcDraughtingPreDefinedColour`
472  the transparency is set to 0.
473  The first three values range from 0 to 1.0, while the transparency
474  varies from 0 to 100.
475

477  Return `None` if `styled_item` is not of type `'IfcStyledItem'`
478  or if there is any other problem getting a color.
479  """
480 
481  if styled_item.is_a("IfcStyledRepresentation"):
482  styled_item = styled_item.Items[0]

485  return None

488  transparency = None
489  col = None

492  # either explicitly for an `IfcGeometricRepresentationItem` being part of
493  # an `IfcShapeRepresentation` assigned to a product, or by assigning
494  # presentation information to `IfcMaterial` being assigned
495  # as other representation for a product.
496 
497  # In current IFC release (IFC2x3) only one presentation style
498  # assignment shall be assigned.
499  # In IFC4 `IfcPresentationStyleAssignment` is deprecated
500  # In IFC4 multiple styles are assigned to style in 'IfcStyleItem' instead
501 
502  # print(ifcfile[p])

506  # IN IFC2x3, only one element in `Styles` should be available.

510  # this is an error in the IFC file in my opinion
511  else:
512  # never seen an ifc with more than one Styles in IfcStyledItem
513  # the above seems to only apply for IFC2x3, IFC4 can have them
514  # see https://forum.freecad.org/viewtopic.php?f=39&t=33560&p=437056#p437056
515 
516  # Get the `IfcPresentationStyleAssignment`, there should only be one,
517  if styled_item.Styles[0].is_a('IfcPresentationStyleAssignment'):
518  assign_style = styled_item.Styles[0]

540  if (len(assign_style.Styles) == 2
541  and assign_style.Styles[1].is_a("IfcSurfaceStyle")):

547  # print(rgb_color)

549  # 2x Annotations in 210_King_Merged.ifc
