213  Py::Object interactiveFilletArc(const Py::Tuple& /*args*/)
214  {
215  Gui::Document* doc = Gui::Application::Instance->activeDocument();
216  if (doc) {
217  Gui::View3DInventor* view = qobject_cast<Gui::View3DInventor*>(doc->getActiveView());
218  if (view) {
219  Gui::View3DInventorViewer* viewer = view->getViewer();
220  SoSeparator* scene = static_cast<SoSeparator*>(viewer->getSceneGraph());
221  SoSeparator* node = new SoSeparator();
222  SoBaseColor* rgb = new SoBaseColor();
223  rgb->rgb.setValue(1,1,0);
224  node->addChild(rgb);
225  SoCoordinate3* coords = new SoCoordinate3();
226  node->addChild(coords);
227  node->addChild(new SoLineSet());
228  scene->addChild(node);
229 
230  ObjectObserver* obs = new ObjectObserver(doc->getDocument()->getActiveObject(), coords);
231  obs->attachDocument(doc->getDocument());
232  }
233  }
234  return Py::None();
235  }
236  Py::Object xmlReader(const Py::Tuple& args)
237  {
238  std::string file = static_cast<std::string>(Py::String(args[0]));
239  App::Document* doc = App::GetApplication().newDocument();

241  std::ifstream str(file.c_str(), std::ios::in);
242  Base::XMLReader reader(file.c_str(), str);
