157  def updateShape(self, pts):
158  """Create shape for display during creation process."""
159  import Part
160  edges = []
161  if len(pts) >= 2: # allow lower degree segment
162  poles = pts[1:]
163  else:
164  poles = []
165  if self.degree:
166  segpoleslst = [poles[x:x+self.degree] for x in range(0, len(poles), (self.degree or 1))]
167  else:
168  segpoleslst = [pts]
169  startpoint = pts[0]
170  for segpoles in segpoleslst:

173  c.setPoles([startpoint] + segpoles)
174  edges.append(Part.Edge(c))
175  startpoint = segpoles[-1]
176  w = Part.Wire(edges)
