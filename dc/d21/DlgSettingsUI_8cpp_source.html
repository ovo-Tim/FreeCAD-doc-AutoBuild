117 void DlgSettingsUI::populateStylesheets(const char *key,
118  const char *path,
119  PrefComboBox *combo,
120  const char *def,
121  QStringList filter)
122 {
123  auto hGrp = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/MainWindow");
124  // List all .qss/.css files
125  QMap<QString, QString> cssFiles;
126  QDir dir;
127  if (filter.isEmpty()) {
128  filter << QStringLiteral("*.qss");
129  filter << QStringLiteral("*.css");
130  }
131  QFileInfoList fileNames;
132 
133  // read from user, resource and built-in directory
134  QStringList qssPaths = QDir::searchPaths(QString::fromUtf8(path));
135  for (QStringList::iterator it = qssPaths.begin(); it != qssPaths.end(); ++it) {
136  dir.setPath(*it);
137  fileNames = dir.entryInfoList(filter, QDir::Files, QDir::Name);
138  for (QFileInfoList::iterator jt = fileNames.begin(); jt != fileNames.end(); ++jt) {
139  if (cssFiles.find(jt->baseName()) == cssFiles.end()) {
140  cssFiles[jt->baseName()] = jt->fileName();
141  }
142  }
143  }
144 
145  combo->clear();
146 
147  // now add all unique items
148  combo->addItem(tr(def), QStringLiteral(""));
149  for (QMap<QString, QString>::iterator it = cssFiles.begin(); it != cssFiles.end(); ++it) {
150  combo->addItem(it.key(), it.value());
151  }
152 
153  QString selectedStyleSheet = QString::fromUtf8(hGrp->GetASCII(key).c_str());
154  int index = combo->findData(selectedStyleSheet);
155 
156  // might be an absolute path name
157  if (index < 0 && !selectedStyleSheet.isEmpty()) {
158  QFileInfo fi(selectedStyleSheet);
159  if (fi.isAbsolute()) {
160  QString path = fi.absolutePath();
161  if (qssPaths.indexOf(path) >= 0) {
162  selectedStyleSheet = fi.fileName();
163  }
164  else {
165  selectedStyleSheet = fi.absoluteFilePath();
166  combo->addItem(fi.baseName(), selectedStyleSheet);
167  }
168  }
169  }
170 
171  combo->setCurrentIndex(index);
172  combo->onRestore();
173 }
