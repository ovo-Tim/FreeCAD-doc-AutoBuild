309 QString Action::createToolTip(QString helpText,
310  const QString & title,
311  const QFont &font,
312  const QString &shortCut,
313  const Command *command)
314 {
315  QString text = cleanTitle(title);
316 
317  if (text.isEmpty()) {
318  return helpText;
319  }
320 
321  // The following code tries to make a more useful tooltip by inserting at
322  // the beginning of the tooltip the action title in bold followed by the
323  // shortcut.
324  //
325  // The long winding code is to deal with the fact that Qt will auto wrap
326  // a rich text tooltip but the width is too short. We can escape the auto
327  // wrapping using <p style='white-space:pre'>.
328 
329  QString shortcut = shortCut;
330  if (!shortcut.isEmpty() && helpText.endsWith(shortcut)) {
331  helpText.resize(helpText.size() - shortcut.size());
332  }
333  if (!shortcut.isEmpty()) {
334  shortcut = QString::fromLatin1(" (%1)").arg(shortcut);
335  }
336 
337  QString tooltip = QString::fromLatin1(
338  "<p style='white-space:pre; margin-bottom:0.5em;'><b>%1</b>%2</p>").arg(
339  text.toHtmlEscaped(), shortcut.toHtmlEscaped());
340 
341  QString cmdName;
342  if (command && command->getName()) {
343  cmdName = QString::fromLatin1(command->getName());
344  if (auto groupcmd = dynamic_cast<const GroupCommand*>(command)) {
345  if (auto act = command->getAction()) {
346  int idx = act->property("defaultAction").toInt();
347  auto cmd = groupcmd->getCommand(idx);
348  if (cmd && cmd->getName()) {
349  cmdName = QStringLiteral("%1 (%2:%3)")
350  .arg(QString::fromLatin1(cmd->getName()), cmdName)
351  .arg(idx);
352  }
353  }
354  }
355  cmdName = QStringLiteral("<p style='white-space:pre; margin-top:0.5em;'><i>%1</i></p>")
356  .arg(cmdName.toHtmlEscaped());
357  }
358 
359  if (!shortcut.isEmpty() && helpText.endsWith(shortcut)) {
360  helpText.resize(helpText.size() - shortcut.size());
361  }
362 
363  if (helpText.isEmpty()
364  || helpText == text
365  || helpText == title)
366  {
367  return tooltip + cmdName;
368  }
369  if (Qt::mightBeRichText(helpText)) {
370  // already rich text, so let it be to avoid duplicated unwrapping
371  return tooltip + helpText + cmdName;
372  }
373 
374  tooltip += QString::fromLatin1(
375  "<p style='white-space:pre; margin:0;'>");
376 
377  // If the user supplied tooltip contains line break, we shall honour it.
378  if (helpText.indexOf(QLatin1Char('\n')) >= 0) {
379  tooltip += helpText.toHtmlEscaped() + QString::fromLatin1("</p>") ;
380  }
381  else {
382  // If not, try to end the non wrapping paragraph at some pre defined
383  // width, so that the following text can wrap at that width.
384  float tipWidth = 400;
385  QFontMetrics fm(font);
386  int width = QtTools::horizontalAdvance(fm, helpText);
387  if (width <= tipWidth) {
388  tooltip += helpText.toHtmlEscaped() + QString::fromLatin1("</p>") ;
389  }
390  else {
391  int index = tipWidth / width * helpText.size();
392  // Try to only break at white space
393  for(int i=0; i<50 && index<helpText.size(); ++i, ++index) {
394  if (helpText[index] == QLatin1Char(' ')) {
395  break;
396  }
397  }
398  tooltip += helpText.left(index).toHtmlEscaped()
399  + QString::fromLatin1("</p>")
400  + helpText.right(helpText.size()-index).trimmed().toHtmlEscaped();
401  }
402  }
403  return tooltip + cmdName;
404 }

851 static QString numberToLabel(int number) {
852  if (number > 0 && number < 10) { // NOLINT: *-magic-numbers
853  return QString::fromLatin1("&%1").arg(number);
854  }
855  if (number == 10) { // NOLINT: *-magic-numbers
856  return QString::fromLatin1("1&0");
857  }
858  // If we have a number greater than 10, we start using the alphabet.
859  // So 11 becomes 'A' and so on.
860  constexpr char lettersStart = 11;
861  constexpr char lettersEnd = lettersStart + ('Z' - 'A');
862  if (number >= lettersStart && number < lettersEnd) {
863  QChar letter = QChar::fromLatin1('A' + (number - lettersStart));
864  return QString::fromLatin1("%1 (&%2)").arg(number).arg(letter);
865  }
866  // Not enough accelerators to cover this number.
867  return QString::fromLatin1("%1").arg(number);
868 }
