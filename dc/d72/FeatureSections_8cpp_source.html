49 App::DocumentObjectExecReturn* Sections::execute()
50 {
51  TColGeom_SequenceOfCurve curveSeq;
52  auto edge_obj = NSections.getValues();
53  auto edge_sub = NSections.getSubValues();
54  if (edge_obj.size() == edge_sub.size()) {
55  for (std::size_t index = 0; index < edge_obj.size(); index++) {
56  // get the part object
57  App::DocumentObject* obj = edge_obj[index];
58  const std::string& sub = edge_sub[index];
59  if (obj && obj->isDerivedFrom<Part::Feature>()) {
60  // get the sub-edge of the part's shape
61  const Part::TopoShape& shape = static_cast<Part::Feature*>(obj)->Shape.getShape();
62  TopoDS_Shape edge = shape.getSubShape(sub.c_str());
63  if (!edge.IsNull() && edge.ShapeType() == TopAbs_EDGE) {
64  BRepAdaptor_Curve curve_adapt(TopoDS::Edge(edge));
65  const TopLoc_Location& loc = edge.Location();
66  Handle(Geom_TrimmedCurve) hCurve =
67  new Geom_TrimmedCurve(curve_adapt.Curve().Curve(),
68  curve_adapt.FirstParameter(),
69  curve_adapt.LastParameter());
70  if (!loc.IsIdentity()) {
71  hCurve->Transform(loc.Transformation());
72  }
73  curveSeq.Append(hCurve);
74  }
75  }
76  }
77  }
78 
79  if (curveSeq.Length() < 2) {
80  return new App::DocumentObjectExecReturn("At least two sections are required.");
81  }
82 
83  GeomFill_NSections fillOp(curveSeq);
84  fillOp.ComputeSurface();
85 
86  Handle(Geom_BSplineSurface) aSurf = fillOp.BSplineSurface();
87  if (aSurf.IsNull()) {
88  return new App::DocumentObjectExecReturn("Failed to create surface from sections.");
89  }
90 
91  BRepBuilderAPI_MakeFace mkFace(aSurf, Precision::Confusion());
92 
93  Shape.setValue(mkFace.Face());
94  return StdReturn;
95 }
