74 // project the points onto the dvp's paper plane.
75 void pointPair::project(const DrawViewPart* dvp)
76 {
77  auto detailView = dynamic_cast<const DrawViewDetail*>(dvp);
78  if (detailView) {
79  m_first = detailView->mapPoint3dToDetail(m_first) * detailView->getScale();
80  m_second = detailView->mapPoint3dToDetail(m_second) * detailView->getScale();
81  m_overrideFirst = detailView->mapPoint3dToDetail(m_overrideFirst) * detailView->getScale();
82  m_overrideSecond = detailView->mapPoint3dToDetail(m_overrideSecond) * detailView->getScale();
83  return;
84  }
85  m_first = dvp->projectPoint(m_first) * dvp->getScale();
86  m_second = dvp->projectPoint(m_second) * dvp->getScale();
87  m_overrideFirst = dvp->projectPoint(m_overrideFirst) * dvp->getScale();
88  m_overrideSecond = dvp->projectPoint(m_overrideSecond) * dvp->getScale();
89 }

103  m_second = DU::toVector3d(gvSecond);

136 pointPair pointPair::toDisplayForm(DrawViewPart* dvp) const
137 {

139 
140  // invert points?

143  result.m_overrideFirst = m_overrideFirst * dvp->getScale();

145  auto rotationDeg = dvp->Rotation.getValue();
