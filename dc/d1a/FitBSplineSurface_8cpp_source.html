106 void FitBSplineSurfaceWidget::onMakePlacementClicked()
107 {
108  try {
109  App::GeoFeature* geo = d->obj.getObjectAs<App::GeoFeature>();
110  if (geo) {
111  const App::PropertyComplexGeoData* geom = geo->getPropertyOfGeometry();
112  if (geom) {
113  std::vector<Base::Vector3d> points, normals;
114  geom->getComplexData()->getPoints(points, normals, 0.001);
115 
116  std::vector<Base::Vector3f> data;
117  std::transform(points.begin(),
118  points.end(),
119  std::back_inserter(data),
120  [](const Base::Vector3d& v) {
121  return Base::convertTo<Base::Vector3f>(v);
122  });
123  MeshCore::PlaneFit fit;
124  fit.AddPoints(data);
125  if (fit.Fit() < FLOAT_MAX) {
126  Base::Vector3f base = fit.GetBase();
127  Base::Vector3f dirU = fit.GetDirU();
128  Base::Vector3f norm = fit.GetNormal();
129 
130  Base::CoordinateSystem cs;
131  cs.setPosition(Base::convertTo<Base::Vector3d>(base));
132  cs.setAxes(Base::convertTo<Base::Vector3d>(norm),
133  Base::convertTo<Base::Vector3d>(dirU));
134  Base::Placement pm = Base::CoordinateSystem().displacement(cs);
135  double q0, q1, q2, q3;
136  pm.getRotation().getValue(q0, q1, q2, q3);
137 
138  QString argument = QString::fromLatin1("Base.Placement(Base.Vector(%1, %2, "
139  "%3), Base.Rotation(%4, %5, %6, %7))")
140  .arg(base.x)
141  .arg(base.y)
142  .arg(base.z)
143  .arg(q0)
144  .arg(q1)
145  .arg(q2)
146  .arg(q3);
147 
148  QString document = QString::fromStdString(d->obj.getDocumentPython());
149  QString command =
150  QString::fromLatin1(
151  R"(%1.addObject("App::Placement", "Placement").Placement = %2)")
152  .arg(document, argument);
153 
154  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Placement"));
155  Gui::Command::runCommand(Gui::Command::Doc, "from FreeCAD import Base");
156  Gui::Command::runCommand(Gui::Command::Doc, command.toLatin1());
157  Gui::Command::commitCommand();
158  Gui::Command::updateActive();
159  }
160  }
161  }
162  }
163  catch (const Base::Exception& e) {
164  Gui::Command::abortCommand();
165  QMessageBox::warning(this, tr("Input error"), QString::fromLatin1(e.what()));
166  }
167 }

169 bool FitBSplineSurfaceWidget::accept()
170 {
171  try {
172  QString document = QString::fromStdString(d->obj.getDocumentPython());
173  QString object = QString::fromStdString(d->obj.getObjectPython());
174 
175  QString argument =
176  QString::fromLatin1("Points=getattr(%1, %1.getPropertyNameOfGeometry()), "
177  "UDegree=%2, VDegree=%3, "
178  "NbUPoles=%4, NbVPoles=%5, "
179  "Smooth=%6, "
180  "Weight=%7, "
181  "Grad=%8, "
182  "Bend=%9, "
183  "Curv=%10, "
184  "Iterations=%11, "
185  "PatchFactor=%12, "
186  "Correction=True")
187  .arg(object)
188  .arg(d->ui.degreeU->value())
189  .arg(d->ui.degreeV->value())
190  .arg(d->ui.polesU->value())
191  .arg(d->ui.polesV->value())
192  .arg(d->ui.groupBoxSmooth->isChecked() ? QLatin1String("True")
193  : QLatin1String("False"))
194  .arg(d->ui.totalWeight->value())
195  .arg(d->ui.gradient->value())
196  .arg(d->ui.bending->value())
197  .arg(d->ui.curvature->value())
198  .arg(d->ui.iterations->value())
199  .arg(d->ui.sizeFactor->value());
200  if (d->ui.uvdir->isChecked()) {
201  std::vector<App::Placement*> selection =
202  Gui::Selection().getObjectsOfType<App::Placement>();
203  if (selection.size() != 1) {
204  QMessageBox::warning(
205  this,
206  tr("Wrong selection"),
207  tr("Please select a single placement object to get local orientation."));
208  return false;
209  }
210 
211  Base::Rotation rot = selection.front()->GeoFeature::Placement.getValue().getRotation();
212  Base::Vector3d u(1, 0, 0);
213  Base::Vector3d v(0, 1, 0);
214  rot.multVec(u, u);
215  rot.multVec(v, v);
216  argument +=
217  QString::fromLatin1(", UVDirs=(FreeCAD.Vector(%1,%2,%3), FreeCAD.Vector(%4,%5,%6))")
218  .arg(u.x)
219  .arg(u.y)
220  .arg(u.z)
221  .arg(v.x)
222  .arg(v.y)
223  .arg(v.z);
224  }
225  QString command = QString::fromLatin1("%1.addObject(\"Part::Spline\", \"Spline\").Shape = "
226  "ReverseEngineering.approxSurface(%2).toShape()")
227  .arg(document, argument);
228 
229  Gui::WaitCursor wc;
230  Gui::Command::addModule(Gui::Command::App, "ReverseEngineering");
231  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Fit B-spline"));
232  Gui::Command::runCommand(Gui::Command::Doc, command.toLatin1());
233  Gui::Command::commitCommand();
234  Gui::Command::updateActive();
235  }
236  catch (const Base::Exception& e) {
237  Gui::Command::abortCommand();
238  QMessageBox::warning(this, tr("Input error"), QString::fromLatin1(e.what()));
239  return false;
240  }
241 
242  return true;
243 }
