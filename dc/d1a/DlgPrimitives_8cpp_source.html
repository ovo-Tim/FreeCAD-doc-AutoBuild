137 QString Picker::toPlacement(const gp_Ax2& axis) const
138 {
139  gp_Dir dir = axis.Direction();
140  gp_Pnt pnt = gp_Pnt(0.0,0.0,0.0);
141  gp_Ax3 ax3(pnt, dir, axis.XDirection());
142 
143  gp_Trsf Trf;
144  Trf.SetTransformation(ax3);
145  Trf.Invert();
146 
147  gp_XYZ theAxis(0,0,1);
148  Standard_Real theAngle = 0.0;
149  Trf.GetRotation(theAxis,theAngle);
150 
151  Base::Rotation rot(Base::convertTo<Base::Vector3d>(theAxis), theAngle);
152  gp_Pnt loc = axis.Location();
153 
154  return QString::fromLatin1("Base.Placement(Base.Vector(%1,%2,%3),Base.Rotation(%4,%5,%6,%7))")
155  .arg(loc.X(),0,'g',Base::UnitsApi::getDecimals())
156  .arg(loc.Y(),0,'g',Base::UnitsApi::getDecimals())
157  .arg(loc.Z(),0,'g',Base::UnitsApi::getDecimals())
158  .arg(rot[0],0,'g',Base::UnitsApi::getDecimals())
159  .arg(rot[1],0,'g',Base::UnitsApi::getDecimals())
160  .arg(rot[2],0,'g',Base::UnitsApi::getDecimals())
161  .arg(rot[3],0,'g',Base::UnitsApi::getDecimals());
162 }

164 class CircleFromThreePoints : public Picker
165 {
166 public:
167  CircleFromThreePoints() : Picker()
168  {
169  }
170  bool pickedPoint(const SoPickedPoint * point) override
171  {
172  SbVec3f pnt = point->getPoint();
173  points.emplace_back(pnt[0],pnt[1],pnt[2]);
174  return points.size() == 3;
175  }
176  QString command(App::Document* doc) const override
177  {
178  GC_MakeArcOfCircle arc(points[0], points[1], points[2]);
179  if (!arc.IsDone())
180  throw Base::CADKernelError(gce_ErrorStatusText(arc.Status()));
181  Handle(Geom_TrimmedCurve) trim = arc.Value();
182  Handle(Geom_Circle) circle = Handle(Geom_Circle)::DownCast(trim->BasisCurve());
183 
184  QString name = QString::fromLatin1(doc->getUniqueObjectName("Circle").c_str());
185  return QString::fromLatin1(
186  "App.ActiveDocument.addObject(\"Part::Circle\",\"%1\")\n"
187  "App.ActiveDocument.%1.Radius=%2\n"
188  "App.ActiveDocument.%1.Angle1=%3\n"
189  "App.ActiveDocument.%1.Angle2=%4\n"
190  "App.ActiveDocument.%1.Placement=%5\n")
191  .arg(name)
192  .arg(circle->Radius(),0,'g',Base::UnitsApi::getDecimals())
193  .arg(Base::toDegrees(trim->FirstParameter()),0,'g',Base::UnitsApi::getDecimals())
194  .arg(Base::toDegrees(trim->LastParameter ()),0,'g',Base::UnitsApi::getDecimals())
195  .arg(toPlacement(circle->Position()));
196  }
197 
198 private:
199  std::vector<gp_Pnt> points;
200 };
