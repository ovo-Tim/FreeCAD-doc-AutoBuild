79 {
80  // add resources and reloads the translators
81  Q_INIT_RESOURCE(TechDraw);
82  Q_INIT_RESOURCE(TechDraw_translation);
83  Gui::Translator::instance()->refresh();
84 
85  // add fonts
86  std::string fontDir = App::Application::getResourceDir() + "Mod/TechDraw/Resources/fonts/";
87 
88  std::vector<std::string> fontsAll(
89  {"osifont-lgpl3fe.ttf", "osifont-italic.ttf", "Y14.5-2018.ttf", "Y14.5-FreeCAD.ttf"});
90 
91  for (auto& font : fontsAll) {
92  QString fontFile = QString::fromStdString(fontDir + font);
93  int rc = QFontDatabase::addApplicationFont(fontFile);
94  if (rc < 0) {
95  Base::Console().Warning(
96  "TechDraw failed to load font file: %d from: %s\n", rc, qPrintable(fontFile));
97  }
98  }
99 }
