328 def parse(pathobj):
329  global PRECISION
330  global MODAL
331  global OUTPUT_DOUBLES
332  global UNIT_FORMAT
333  global UNIT_SPEED_FORMAT
334  global RETRACT_MODE
335  global DWELL_TIME
336  global clearanceHeight
337 
338  lastX = 0
339  lastY = 0
340  lastZ = 0
341  out = ""
342  lastcommand = None
343  precision_string = "." + str(PRECISION) + "f"
344  currLocation = {} # keep track for no doubles
345 
346  # the order of parameters
347  # Added O and L since these are needed by Dynapath.
348  # "O" is used as a 2nd reference plane in canned Cycles.
349  # "L" is equivalent to "P" for dwell entries.
350  params = [
351  "X",
352  "Y",
353  "Z",
354  "A",
355  "B",
356  "C",
357  "I",
358  "J",
359  "K",
360  "F",
361  "S",
362  "T",
363  "Q",
364  "O",
365  "R",
366  "L",
367  "P",
368  ]
369 
370  firstmove = Path.Command("G0", {"X": -1, "Y": -1, "Z": -1, "F": 0.0})
371  currLocation.update(firstmove.Parameters) # set first location parameters
372 
373  if hasattr(pathobj, "Group"): # We have a compound or project.
374  if OUTPUT_COMMENTS:
375  out += linenumber() + "(T)" + str.upper("compound: " + pathobj.Label) + "$\n"
376  for p in pathobj.Group:
377  out += parse(p)
378  return out
379  else: # parsing simple path
380 
381  # groups might contain non-path things like stock.
382  if not hasattr(pathobj, "Path"):
383  return out
384 
385  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
386  outstring = []
387  command = c.Name
388  # Convert G54-G59 Fixture offsets to E01-E06 for Dynapath Delta Control
389  if command in GCODE_MAP:
390  command = GCODE_MAP[command]
391 
392  # Inserts "(T)" in front of comments to signify textfield (Dynapath)
393  if command.startswith("("):
394  if OUTPUT_COMMENTS:
395  command = "(T)" + str.upper(command) + "$"
396 
397  outstring.append(command)
398 
399  # if modal: suppress the command if it is the same as the last one
400  if MODAL is True:
401  if command == lastcommand:
402  outstring.pop(0)
403 
404  if c.Name[0] == "(" and not OUTPUT_COMMENTS: # command is a comment
405  continue
406 
407  # Now add the remaining parameters in order
408  for param in params:
409  if param in c.Parameters:
410  if param == "F" and (
411  currLocation[param] != c.Parameters[param] or OUTPUT_DOUBLES
412  ):
413  speed = Units.Quantity(c.Parameters["F"], FreeCAD.Units.Velocity)
414  if speed.getValueAs(UNIT_SPEED_FORMAT) > 0.0:
415  outstring.append(
416  param
417  + format(
418  float(speed.getValueAs(UNIT_SPEED_FORMAT)),
419  precision_string,
420  )
421  )
422  # Inserts "X0 Y0" in front of a G0 Z + clearanceHeight movement.
423  # This fixes an error thrown by Dynapath due to missing and
424  # required XYZ move after Tool change.
425  elif param == "Z" and (
426  c.Parameters["Z"] == clearanceHeight and c.Parameters["Z"] != lastZ
427  ):
428  x = 0
429  y = 0
430  outstring.insert(
431  1,
432  "X"
433  + PostUtils.fmt(x, PRECISION, UNITS)
434  + "Y"
435  + PostUtils.fmt(y, PRECISION, UNITS),
436  )
437  outstring.append(param + PostUtils.fmt(c.Parameters["Z"], PRECISION, UNITS))
438  elif param == "X" and (command in QCYCLE_RANGE):
439  pos = Units.Quantity(c.Parameters["X"], FreeCAD.Units.Length)
440  outstring.append(
441  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
442  )
443  elif param == "Y" and (command in QCYCLE_RANGE):
444  pos = Units.Quantity(c.Parameters["Y"], FreeCAD.Units.Length)
445  outstring.append(
446  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
447  )
448  # Remove X and Y between QCYCLE's since we already included them.
449  # This is needed to prevent Path of inserting additional XY codes between
450  # Canned cycle holes.
451  elif lastcommand in QCYCLE_RANGE and (param == "X" or "Y"):
452  outstring = []
453  elif param == "S":
454  SPINDLE_SPEED = c.Parameters["S"]
455  outstring.append(
456  param + "{:.0f}".format(c.Parameters["S"])
457  ) # Added formatting to strip trailing .000 from RPM (needed by dynapath)
458  elif param == "T":
459  outstring.append(
460  param + "{:.0f}".format(c.Parameters["T"])
461  ) # Added formatting to strip trailing .000 from Tool number (needed by dynapath)
462  elif param == "I" and (command == "G2" or command == "G3"):
463  # Convert incremental arc center to absolute in I and J
464  # Dynapath requires "absolute" arcs in (G2,G3)
465  i = c.Parameters["I"]
466  if ABSOLUTE_CIRCLE_CENTER:
467  i += lastX
468  outstring.append(param + PostUtils.fmt(i, PRECISION, UNITS))
469  elif param == "J" and (command == "G2" or command == "G3"):
470  # Convert incremental arc center to absolute in I and J
471  j = c.Parameters["J"]
472  if ABSOLUTE_CIRCLE_CENTER:
473  j += lastY
474  outstring.append(param + PostUtils.fmt(j, PRECISION, UNITS))
475  elif param == "K" and (command == "G2" or command == "G3"):
476  # Convert incremental arc center to absolute in K (Z axis arc)
477  k = c.Parameters["K"]
478  if ABSOLUTE_CIRCLE_CENTER:
479  k += lastZ
480  if command == (
481  "G18" or "G19"
482  ): # Dynapath supports G18/G19 for Z axis arcs in Y or X.
483  outstring.append(param + PostUtils.fmt(k, PRECISION, UNITS))
484  # Converts "Q" to "K" as needed by Dynapath.
485  elif param == "Q":
486  pos = Units.Quantity(c.Parameters["Q"], FreeCAD.Units.Length)
487  outstring.append(
488  "K" + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
489  )
490  # Following inserts a 2nd reference plane in all canned cycles (dynapath).
491  # This provides the ability to manually go in and bump up the "O" offset in
492  # order to avoid obstacles. The "O" overrides "R", so set them both equal if you
493  # don't need the 2nd reference plane.
494  elif (param == "R") and ((command in QCYCLE_RANGE)):
495  pos = Units.Quantity(pathobj.ClearanceHeight.Value, FreeCAD.Units.Length)
496  outstring.insert(
497  6,
498  "O" + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string),
499  ) # Insert "O" param for 2nd reference plane (Clearance Height)
500  pos = Units.Quantity(c.Parameters["R"], FreeCAD.Units.Length)
501  outstring.append(
502  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
503  ) # First Reference plan (Safe Height)
504  elif param == "P":
505  outstring.append(
506  "L" + format(c.Parameters[param], precision_string)
507  ) # Converts "P" to "L" for dynapath.
508  else:
509  if (
510  (not OUTPUT_DOUBLES)
511  and (param in currLocation)
512  and (currLocation[param] == c.Parameters[param])
513  ):
514  continue
515  else:
516  pos = Units.Quantity(c.Parameters[param], FreeCAD.Units.Length)
517  outstring.append(
518  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
519  )
520  # save the last X, Y values
521  if "X" in c.Parameters:
522  lastX = c.Parameters["X"]
523  if "Y" in c.Parameters:
524  lastY = c.Parameters["Y"]
525  if "Z" in c.Parameters:
526  lastZ = c.Parameters["Z"]
527 
528  # store the latest command
529  lastcommand = command
530  currLocation.update(c.Parameters)
531 
532  # Check for Tool Change:
533  if command == "M6":
534  if OUTPUT_COMMENTS:
535  out += linenumber() + "(T)" + "BEGIN TOOLCHANGE$\n"
536 
537  if command == "message":
538  if OUTPUT_COMMENTS is False:
539  out = []
540  else:
541  outstring.pop(0) # remove the command
542  # G98 is not used by dynapath and G99 is not used in Drilling/Boring/Tapping.
543  if c.Name == "G98" or (c.Name == "G99" and pathobj.Label == "Drilling"):
544  outstring = []
545 
546  # prepend a line number and append a newline
547  if len(outstring) >= 1:
548  if OUTPUT_LINE_NUMBERS:
549  outstring.insert(0, (linenumber()))
550 
551  # append the line to the final output
552  for w in outstring:
553  out += w.strip() + COMMAND_SPACE
554  out += "\n"
555 
556  # Add a DWELL after spindle start based on RPM.
557  # Used by Dynapath for implementing a scalable dwell time.
558  if DWELL_TIME > 0.0:
559  if command in ["M3", "M03", "M4", "M04"]:
560  DWELL_TIME = (SPINDLE_SPEED / 100) / 10
561  out += linenumber() + "L%.1f\n" % DWELL_TIME
562 
563  return out
