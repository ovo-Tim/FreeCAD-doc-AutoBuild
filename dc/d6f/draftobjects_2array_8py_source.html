448 def rect_placements(base_placement,
449  xvector, yvector, zvector,
450  xnum, ynum, znum):
451  """Determine the placements where the rectangular copies will be."""
452  pl = base_placement
453  placements = [pl.copy()]
454 
455  for xcount in range(xnum):
456  currentxvector = App.Vector(xvector).multiply(xcount)
457  if xcount != 0:
458  npl = pl.copy()
459  npl.translate(currentxvector)
460  placements.append(npl)
461 
462  for ycount in range(ynum):
463  currentyvector = App.Vector(currentxvector)
464  _y_shift = App.Vector(yvector).multiply(ycount)
465  currentyvector = currentyvector.add(_y_shift)
466  if ycount != 0:
467  npl = pl.copy()
468  npl.translate(currentyvector)
469  placements.append(npl)
470 
471  for zcount in range(znum):
472  currentzvector = App.Vector(currentyvector)
473  _z_shift = App.Vector(zvector).multiply(zcount)
474  currentzvector = currentzvector.add(_z_shift)
475  if zcount != 0:
476  npl = pl.copy()
477  npl.translate(currentzvector)
478  placements.append(npl)
479 
480  return placements

483 def polar_placements(base_placement,
484  center, angle,
485  number, axis, axisvector):
486  """Determine the placements where the polar copies will be."""
487  # print("angle ",angle," num ",num)
488  placements = [base_placement.copy()]
489 
490  if number <= 1:
491  return placements
492 
493  if angle == 360:
494  fraction = float(angle) / number
495  else:
496  fraction = float(angle) / (number - 1)
497 
498  for i in range(number - 1):
499  currangle = fraction + (i * fraction)
500  npl = base_placement.copy()
501  npl.rotate(center, axis, currangle, comp=True)
502  if axisvector:
503  if not DraftVecUtils.isNull(axisvector):
504  npl.translate(App.Vector(axisvector).multiply(i + 1))
505  placements.append(npl)
506 
507  return placements

510 def circ_placements(base_placement,
511  r_distance, tan_distance,
512  axis, center,
513  circle_number, symmetry):
514  """Determine the placements where the circular copies will be."""
515  symmetry = max(1, symmetry)
516  lead = (0, 1, 0)
517 
518  if axis.x == 0 and axis.z == 0:
519  lead = (1, 0, 0)
520 
521  direction = axis.cross(App.Vector(lead)).normalize()
522  placements = [base_placement.copy()]
523 
524  for xcount in range(1, circle_number):
525  rc = xcount * r_distance
526  trans = App.Vector(direction).multiply(rc)
527  c = 2 * rc * math.pi
528  n = math.floor(c / tan_distance)
529  n = int(math.floor(n / symmetry) * symmetry)
530  if n == 0:
531  continue
532 
533  angle = 360.0 / n
534  for ycount in range(0, n):
535  npl = base_placement.copy()
536  npl.translate(trans)
537  npl.rotate(center, axis, ycount * angle, comp=True)
538  placements.append(npl)
539 
540  return placements
