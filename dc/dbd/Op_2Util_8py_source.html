105 def _orientEdges(inEdges):
106  """_orientEdges(inEdges) ... internal worker function to orient edges so the last vertex of one edge connects to the first vertex of the next edge.
107  Assumes the edges are in an order so they can be connected."""
108  Path.Log.track()
109  # orient all edges of the wire so each edge's last value connects to the next edge's first value
110  e0 = inEdges[0]
111  # well, even the very first edge could be misoriented, so let's try and connect it to the second
112  if 1 < len(inEdges):
113  last = e0.valueAt(e0.LastParameter)
114  e1 = inEdges[1]
115  if not Path.Geom.pointsCoincide(
116  last, e1.valueAt(e1.FirstParameter)
117  ) and not Path.Geom.pointsCoincide(last, e1.valueAt(e1.LastParameter)):
118  debugEdge("# _orientEdges - flip first", e0)
119  e0 = Path.Geom.flipEdge(e0)
120 
121  edges = [e0]
122  last = e0.valueAt(e0.LastParameter)
123  for e in inEdges[1:]:
124  edge = (
125  e
126  if Path.Geom.pointsCoincide(last, e.valueAt(e.FirstParameter))
127  else Path.Geom.flipEdge(e)
128  )
129  edges.append(edge)
130  last = edge.valueAt(edge.LastParameter)
131  return edges

134 def _isWireClockwise(w):
135  """_isWireClockwise(w) ... return True if wire is oriented clockwise.
136  Assumes the edges of w are already properly oriented - for generic access use isWireClockwise(w).
137  """
138  # handle wires consisting of a single circle or 2 edges where one is an arc.
139  # in both cases, because the edges are expected to be oriented correctly, the orientation can be
140  # determined by looking at (one of) the circle curves.
141  if 2 >= len(w.Edges) and Part.Circle == type(w.Edges[0].Curve):
142  return 0 > w.Edges[0].Curve.Axis.z
143  if 2 == len(w.Edges) and Part.Circle == type(w.Edges[1].Curve):
144  return 0 > w.Edges[1].Curve.Axis.z
145 
146  # for all other wires we presume they are polygonial and refer to Gauss
147  # https://en.wikipedia.org/wiki/Shoelace_formula
148  area = 0
149  for e in w.Edges:
150  v0 = e.valueAt(e.FirstParameter)
151  v1 = e.valueAt(e.LastParameter)
152  area = area + (v0.x * v1.y - v1.x * v0.y)
153  Path.Log.track(area)
154  return area < 0

163  """orientWire(w, forward=True) ... orients given wire in a specific direction.
164  If forward = True (the default) the wire is oriented clockwise, looking down the negative Z axis.
165  If forward = False the wire is oriented counter clockwise.
166  If forward = None the orientation is determined by the order in which the edges appear in the wire.

172  Path.Log.track("orientWire - needs flipping")
