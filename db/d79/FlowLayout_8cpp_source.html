146 int FlowLayout::doLayout(const QRect& rect, bool testOnly) const
147 {
148  int left {};
149  int top {};
150  int right {};
151  int bottom {};
152  getContentsMargins(&left, &top, &right, &bottom);
153  QRect effectiveRect = rect.adjusted(+left, +top, -right, -bottom);
154  int x = effectiveRect.x();
155  int y = effectiveRect.y();
156  int lineHeight = 0;
157 
158  for (auto item : std::as_const(itemList)) {
159  QWidget* wid = item->widget();
160  int spaceX = horizontalSpacing();
161  if (spaceX == -1) {
162  spaceX = wid->style()->layoutSpacing(QSizePolicy::PushButton,
163  QSizePolicy::PushButton,
164  Qt::Horizontal);
165  }
166 
167  int spaceY = verticalSpacing();
168  if (spaceY == -1) {
169  spaceY = wid->style()->layoutSpacing(QSizePolicy::PushButton,
170  QSizePolicy::PushButton,
171  Qt::Vertical);
172  }
173 
174  int nextX = x + item->sizeHint().width() + spaceX;
175  if (nextX - spaceX > effectiveRect.right() && lineHeight > 0) {
176  x = effectiveRect.x();
177  y = y + lineHeight + spaceY;
178  nextX = x + item->sizeHint().width() + spaceX;
179  lineHeight = 0;
180  }
181 
182  if (!testOnly) {
183  item->setGeometry(QRect(QPoint(x, y), item->sizeHint()));
184  }
185 
186  x = nextX;
187  lineHeight = std::max(lineHeight, item->sizeHint().height());
188  }
189 
190  return y + lineHeight - rect.y() + bottom;
191 }
