66 Extension::~Extension()
67 {
68  if (!ExtensionPythonObject.is(Py::_None())) {
69  // Remark: The API of Py::Object has been changed to set whether the wrapper owns the passed
70  // Python object or not. In the constructor we forced the wrapper to own the object so we
71  // need not to dec'ref the Python object any more. But we must still invalidate the Python
72  // object because it need not to be destructed right now because the interpreter can own
73  // several references to it.
74  Base::PyObjectBase* obj = static_cast<Base::PyObjectBase*>(ExtensionPythonObject.ptr());
75  // Call before decrementing the reference counter, otherwise a heap error can occur
76  obj->setInvalid();
77  }
78 }

88 void Extension::initExtension(ExtensionContainer* obj)
89 {
90  if (m_extensionType.isBad()) {
91  throw Base::RuntimeError("Extension: Extension type not set");
92  }
93 
94  // all properties are initialised without PropertyContainer father. Now that we know it we can
95  // finally finish the property initialisation
96  std::vector<Property*> list;
97  extensionGetPropertyData().getPropertyList(this, list);
98  for (Property* prop : list) {
99  prop->setContainer(obj);
100  }
101 
102  m_base = obj;
103  m_base->registerExtension(m_extensionType, this);
104 }

188 {
