136 void ViewProviderShapeBinder::highlightReferences(bool on)
137 {
138  App::GeoFeature* obj = nullptr;
139  std::vector<std::string> subs;
140 
141  if (getObject()->isDerivedFrom(PartDesign::ShapeBinder::getClassTypeId()))
142  PartDesign::ShapeBinder::getFilteredReferences(&getObject<PartDesign::ShapeBinder>()->Support, obj, subs);
143  else
144  return;
145 
146  // stop if not a Part feature was found
147  if (!obj || !obj->isDerivedFrom<Part::Feature>())
148  return;
149 
150  PartGui::ViewProviderPart* svp = dynamic_cast<PartGui::ViewProviderPart*>(
151  Gui::Application::Instance->getViewProvider(obj));
152  if (!svp)
153  return;
154 
155  if (on) {
156  if (!subs.empty() && originalLineColors.empty()) {
157  TopTools_IndexedMapOfShape eMap;
158  TopExp::MapShapes(static_cast<Part::Feature*>(obj)->Shape.getValue(), TopAbs_EDGE, eMap);
159  originalLineColors = svp->LineColorArray.getValues();
160  std::vector<App::Color> lcolors = originalLineColors;
161  lcolors.resize(eMap.Extent(), svp->LineColor.getValue());
162 
163  TopExp::MapShapes(static_cast<Part::Feature*>(obj)->Shape.getValue(), TopAbs_FACE, eMap);
164  originalFaceAppearance = svp->ShapeAppearance.getValues();
165  std::vector<App::Material> fcolors = originalFaceAppearance;
166  fcolors.resize(eMap.Extent(), svp->ShapeAppearance[0]);
167 
168  for (const std::string& e : subs) {
169  // Note: stoi may throw, but it strictly shouldn't happen
170  if (e.compare(0, 4, "Edge") == 0) {
171  int idx = std::stoi(e.substr(4)) - 1;
172  assert(idx >= 0);
173  if (idx < static_cast<int>(lcolors.size()))
174  lcolors[idx] = App::Color(1.0, 0.0, 1.0); // magenta
175  }
176  else if (e.compare(0, 4, "Face") == 0) {
177  int idx = std::stoi(e.substr(4)) - 1;
178  assert(idx >= 0);
179  if (idx < static_cast<int>(fcolors.size()))
180  fcolors[idx].diffuseColor = App::Color(1.0, 0.0, 1.0); // magenta
181  }
182  }
183  svp->LineColorArray.setValues(lcolors);
184  svp->ShapeAppearance.setValues(fcolors);
185  }
186  }
187  else {
188  if (!subs.empty() && !originalLineColors.empty()) {
189  svp->LineColorArray.setValues(originalLineColors);
190  originalLineColors.clear();
191 
192  svp->ShapeAppearance.setValues(originalFaceAppearance);
193  originalFaceAppearance.clear();
194  }
195  }
196 }

278 std::string ViewProviderSubShapeBinder::dropObjectEx(App::DocumentObject* obj, App::DocumentObject* owner,
279  const char* subname, const std::vector<std::string>& elements)
280 {
281  auto self = getObject<PartDesign::SubShapeBinder>();
282  if (!self)
283  return {};
284  std::map<App::DocumentObject*, std::vector<std::string> > values;
285  if (!subname) subname = "";
286  std::string sub(subname);
287  if (sub.empty())
288  values[owner ? owner : obj] = elements;
289  else {
290  std::vector<std::string> subs;
291  if (!elements.empty()) {
292  subs.reserve(elements.size());
293  for (auto& element : elements)
294  subs.push_back(sub + element);
295  }
296  else
297  subs.push_back(sub);
298  values[owner ? owner : obj] = std::move(subs);
299  }
300 
301  self->setLinks(std::move(values), QApplication::keyboardModifiers() == Qt::ControlModifier);
302  if (self->Relative.getValue())
303  updatePlacement(false);
304  return {};
305 }

323 bool ViewProviderSubShapeBinder::setEdit(int ModNum) {
324 
325  switch (ModNum) {
326  case Synchronize:
327  updatePlacement(true);
328  break;
329  case SelectObject: {
330  auto self = getObject<PartDesign::SubShapeBinder>();
331  if (!self || !self->Support.getValue())
332  break;
333 
334  Gui::Selection().selStackPush();
335  Gui::Selection().clearSelection();
336  for (auto& link : self->Support.getSubListValues()) {
337  auto obj = link.getValue();
338  if (!obj || !obj->isAttachedToDocument())
339  continue;
340  const auto& subs = link.getSubValues();
341  if (!subs.empty())
342  Gui::Selection().addSelections(obj->getDocument()->getName(),
343  obj->getNameInDocument(), subs);
344  else
345  Gui::Selection().addSelection(obj->getDocument()->getName(),
346  obj->getNameInDocument());
347  }
348  Gui::Selection().selStackPush();
349  break;
350  }
351  default:
352  return ViewProviderPart::setEdit(ModNum);
353  }
354  return false;
355 }

357 void ViewProviderSubShapeBinder::updatePlacement(bool transaction) {
358  auto self = getObject<PartDesign::SubShapeBinder>();
359  if (!self || !self->Support.getValue())
360  return;
361 
362  std::vector<Base::Matrix4D> mats;
363  bool relative = self->Relative.getValue();
364  App::DocumentObject* parent = nullptr;
365  std::string parentSub;
366  if (relative && !self->getParents().empty()) {
367  const auto& sel = Gui::Selection().getSelection("", Gui::ResolveMode::NoResolve);
368  if (sel.size() != 1 || !sel[0].pObject ||
369  sel[0].pObject->getSubObject(sel[0].SubName) != self)
370  {
371  FC_WARN("invalid selection");
372  }
373  else {
374  parent = sel[0].pObject;
375  parentSub = sel[0].SubName;
376  }
377  }
378 
379  if (!transaction) {
380  if (relative)
381  self->Context.setValue(parent, parentSub.c_str());
382  try {
383  self->update(PartDesign::SubShapeBinder::UpdateForced);
384  }
385  catch (Base::Exception& e) {
386  e.ReportException();
387  }
388  return;
389  }
390 
391  App::GetApplication().setActiveTransaction("Sync binder");
392  try {
393  if (relative)
394  self->Context.setValue(parent, parentSub.c_str());
395  self->update(PartDesign::SubShapeBinder::UpdateForced);
396  App::GetApplication().closeActiveTransaction();
397  return;
398  }
399  catch (Base::Exception& e) {
400  e.ReportException();
401  }
402  catch (Standard_Failure& e) {
403  std::ostringstream str;
404  Standard_CString msg = e.GetMessageString();
405  str << typeid(e).name() << " ";
406  if (msg) { str << msg; }
407  else { str << "No OCCT Exception Message"; }
408  FC_ERR(str.str());
409  }
410  App::GetApplication().closeActiveTransaction(true);
411 }

413 std::vector<App::DocumentObject*> ViewProviderSubShapeBinder::claimChildren() const {
414  std::vector<App::DocumentObject*> ret;
415  auto self = Base::freecad_dynamic_cast<PartDesign::SubShapeBinder>(getObject());
416  if (self && self->ClaimChildren.getValue() && self->Support.getValue()) {
417  std::set<App::DocumentObject*> objSet;
418  for (auto& l : self->Support.getSubListValues()) {
419  auto obj = l.getValue();
420  if (!obj)
421  continue;
422  const auto& subs = l.getSubValues();
423  if (subs.empty()) {
424  if (objSet.insert(obj).second)
425  ret.push_back(obj);
426  continue;
427  }
428  for (auto& sub : subs) {
429  auto sobj = obj->getSubObject(sub.c_str());
430  if (sobj && objSet.insert(sobj).second)
431  ret.push_back(sobj);
432  }
433  }
434  }
435  return ret;
436 }
