98 App::DocumentObjectExecReturn *Draft::execute()
99 {
100  // Get parameters
101  // Base shape
102  Part::TopoShape TopShape;
103  try {
104  TopShape = getBaseTopoShape();
105  }
106  catch (Base::Exception& e) {
107  return new App::DocumentObjectExecReturn(e.what());
108  }
109 
110  // Faces where draft should be applied
111  // Note: Cannot be const reference currently because of BRepOffsetAPI_DraftAngle::Remove() bug, see below
112  std::vector<std::string> SubVals = Base.getSubValuesStartsWith("Face");
113 
114  //If no element is selected, then we use a copy of previous feature.
115  if (SubVals.empty()) {
116  this->positionByBaseFeature();
117  this->Shape.setValue(TopShape);
118  return App::DocumentObject::StdReturn;
119  }
120 
121  // Draft angle
122  double angle = Base::toRadians(Angle.getValue());
123 
124  // Pull direction
125  gp_Dir pullDirection;
126  App::DocumentObject* refDirection = PullDirection.getValue();
127  if (refDirection) {
128  if (refDirection->isDerivedFrom<PartDesign::Line>()) {
129  PartDesign::Line* line = static_cast<PartDesign::Line*>(refDirection);
130  Base::Vector3d d = line->getDirection();
131  pullDirection = gp_Dir(d.x, d.y, d.z);
132  } else if (refDirection->isDerivedFrom<Part::Feature>()) {
133  std::vector<std::string> subStrings = PullDirection.getSubValues();
134  if (subStrings.empty() || subStrings[0].empty())
135  throw Base::ValueError("No pull direction reference specified");
136 
137  Part::Feature* refFeature = static_cast<Part::Feature*>(refDirection);
138  Part::TopoShape refShape = refFeature->Shape.getShape();
139  TopoDS_Shape ref = refShape.getSubShape(subStrings[0].c_str());
140 
141  if (ref.ShapeType() == TopAbs_EDGE) {
142  TopoDS_Edge refEdge = TopoDS::Edge(ref);
143  if (refEdge.IsNull())
144  throw Base::ValueError("Failed to extract pull direction reference edge");
145  BRepAdaptor_Curve adapt(refEdge);
146  if (adapt.GetType() != GeomAbs_Line)
147  throw Base::TypeError("Pull direction reference edge must be linear");
148 
149  pullDirection = adapt.Line().Direction();
150  } else {
151  throw Base::TypeError("Pull direction reference must be an edge or a datum line");
152  }
153  } else {
154  throw Base::TypeError("Pull direction reference must be an edge of a feature or a datum line");
155  }
156 
157  TopLoc_Location invObjLoc = this->getLocation().Inverted();
158  pullDirection.Transform(invObjLoc.Transformation());
159  }
160 
161  // Neutral plane
162  gp_Pln neutralPlane;
163  App::DocumentObject* refPlane = NeutralPlane.getValue();
164  if (!refPlane) {
165  // Try to guess a neutral plane from the first selected face
166  // Get edges of first selected face
167  TopoDS_Shape face = TopShape.getSubShape(SubVals[0].c_str());
168  TopTools_IndexedMapOfShape mapOfEdges;
169  TopExp::MapShapes(face, TopAbs_EDGE, mapOfEdges);
170  bool found = false;
171 
172  for (int i = 1; i <= mapOfEdges.Extent(); i++) {
173  // Note: What happens if mapOfEdges(i) is the degenerated edge of a cone?
174  // But in that case the draft is not possible anyway!
175  BRepAdaptor_Curve c(TopoDS::Edge(mapOfEdges(i)));
176  gp_Pnt p1 = c.Value(c.FirstParameter());
177  gp_Pnt p2 = c.Value(c.LastParameter());
178 
179  if (c.IsClosed()) {
180  // Edge is a circle or a circular arc (other types are not allowed for drafting)
181  if (c.GetType() == GeomAbs_Circle) {
182  neutralPlane = gp_Pln(p1, c.Circle().Axis().Direction());
183  found = true;
184  break;
185  }
186  } else {
187  // Edge is linear
188  // Find midpoint of edge and create auxiliary plane through midpoint normal to edge
189  gp_Pnt pm = c.Value((c.FirstParameter() + c.LastParameter()) / 2.0);
190  Handle(Geom_Plane) aux = new Geom_Plane(pm, gp_Dir(p2.X() - p1.X(), p2.Y() - p1.Y(), p2.Z() - p1.Z()));
191  // Intersect plane with face. Is there no easier way?
192  BRepAdaptor_Surface adapt(TopoDS::Face(face), Standard_False);
193  Handle(Geom_Surface) sf = adapt.Surface().Surface();
194  GeomAPI_IntSS intersector(aux, sf, Precision::Confusion());
195  if (!intersector.IsDone() || intersector.NbLines() < 1)
196  continue;
197  Handle(Geom_Curve) icurve = intersector.Line(1);
198  if (!icurve->IsKind(STANDARD_TYPE(Geom_Line)))
199  continue;
200  // TODO: How to extract the line from icurve without creating an edge first?
201  TopoDS_Edge edge = BRepBuilderAPI_MakeEdge(icurve);
202  BRepAdaptor_Curve c(edge);
203  neutralPlane = gp_Pln(pm, c.Line().Direction());
204  found = true;
205  break;
206  }
207  }
208 
209  if (!found)
210  throw Base::RuntimeError("No neutral plane specified and none can be guessed");
211  } else {
212  if (refPlane->isDerivedFrom<PartDesign::Plane>()) {
213  PartDesign::Plane* plane = static_cast<PartDesign::Plane*>(refPlane);
214  Base::Vector3d b = plane->getBasePoint();
215  Base::Vector3d n = plane->getNormal();
216  neutralPlane = gp_Pln(gp_Pnt(b.x, b.y, b.z), gp_Dir(n.x, n.y, n.z));
217  } else if (refPlane->isDerivedFrom<App::Plane>()) {
218  neutralPlane = Feature::makePlnFromPlane(refPlane);
219  } else if (refPlane->isDerivedFrom<Part::Feature>()) {
220  std::vector<std::string> subStrings = NeutralPlane.getSubValues();
221  if (subStrings.empty() || subStrings[0].empty())
222  throw Base::ValueError("No neutral plane reference specified");
223 
224  Part::Feature* refFeature = static_cast<Part::Feature*>(refPlane);
225  Part::TopoShape refShape = refFeature->Shape.getShape();
226  TopoDS_Shape ref = refShape.getSubShape(subStrings[0].c_str());
227 
228  if (ref.ShapeType() == TopAbs_FACE) {
229  TopoDS_Face refFace = TopoDS::Face(ref);
230  if (refFace.IsNull())
231  throw Base::ValueError("Failed to extract neutral plane reference face");
232  BRepAdaptor_Surface adapt(refFace);
233  if (adapt.GetType() != GeomAbs_Plane)
234  throw Base::TypeError("Neutral plane reference face must be planar");
235 
236  neutralPlane = adapt.Plane();
237  } else if (ref.ShapeType() == TopAbs_EDGE) {
238  if (refDirection) {
239  // Create neutral plane through edge normal to pull direction
240  TopoDS_Edge refEdge = TopoDS::Edge(ref);
241  if (refEdge.IsNull())
242  throw Base::ValueError("Failed to extract neutral plane reference edge");
243  BRepAdaptor_Curve c(refEdge);
244  if (c.GetType() != GeomAbs_Line)
245  throw Base::TypeError("Neutral plane reference edge must be linear");
246  double a = c.Line().Angle(gp_Lin(c.Value(c.FirstParameter()), pullDirection));
247  if (std::fabs(a - M_PI_2) > Precision::Confusion())
248  throw Base::ValueError("Neutral plane reference edge must be normal to pull direction");
249  neutralPlane = gp_Pln(c.Value(c.FirstParameter()), pullDirection);
250  } else {
251  throw Base::TypeError("Neutral plane reference can only be an edge if pull direction is defined");
252  }
253  } else {
254  throw Base::TypeError("Neutral plane reference must be a face");
255  }
256  } else {
257  throw Base::TypeError("Neutral plane reference must be face of a feature or a datum plane");
258  }
259 
260  TopLoc_Location invObjLoc = this->getLocation().Inverted();
261  neutralPlane.Transform(invObjLoc.Transformation());
262  }
263 
264  if (!refDirection) {
265  // Choose pull direction normal to neutral plane
266  pullDirection = neutralPlane.Axis().Direction();
267  }
268 
269  // Reversed pull direction
270  bool reversed = Reversed.getValue();
271  if (reversed)
272  angle *= -1.0;
273 
274  this->positionByBaseFeature();
275  // create an untransformed copy of the base shape
276  Part::TopoShape baseShape(TopShape);
277  baseShape.setTransform(Base::Matrix4D());
278  try {
279  BRepOffsetAPI_DraftAngle mkDraft;
280  // Note:
281  // LocOpe_SplitDrafts can split a face with a wire and apply draft to both parts
282  // Not clear though whether the face must have free boundaries
283  // LocOpe_DPrism can create a stand-alone draft prism. The sketch can only have a single
284  // wire, though.
285  // BRepFeat_MakeDPrism requires a support for the operation but will probably support multiple
286  // wires in the sketch
287 
288  bool success;
289 
290  do {
291  success = true;
292  mkDraft.Init(baseShape.getShape());
293 
294  for (std::vector<std::string>::iterator it=SubVals.begin(); it != SubVals.end(); ++it) {
295  TopoDS_Face face = TopoDS::Face(baseShape.getSubShape(it->c_str()));
296  // TODO: What is the flag for?
297  mkDraft.Add(face, pullDirection, angle, neutralPlane);
298  if (!mkDraft.AddDone()) {
299  // Note: the function ProblematicShape returns the face on which the error occurred
300  // Note: mkDraft.Remove() stumbles on a bug in Draft_Modification::Remove() and is
301  // therefore unusable. See https://forum.freecad.org/viewtopic.php?f=10&t=3209&start=10#p25341
302  // The only solution is to discard mkDraft and start over without the current face
303  // mkDraft.Remove(face);
304  Base::Console().Error("Adding face failed on %s. Omitted\n", it->c_str());
305  success = false;
306  SubVals.erase(it);
307  break;
308  }
309  }
310  }
311  while (!success);
312 
313  mkDraft.Build();
314  if (!mkDraft.IsDone())
315  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Failed to create draft"));
316 
317  TopoDS_Shape shape = mkDraft.Shape();
318  if (shape.IsNull())
319  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Resulting shape is null"));
320 
321  if (!isSingleSolidRuleSatisfied(shape)) {
322  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Result has multiple solids: that is not currently supported."));
323  }
324 
325  this->Shape.setValue(getSolid(shape));
326  return App::DocumentObject::StdReturn;
327  }
328  catch (Standard_Failure& e) {
329 
330  return new App::DocumentObjectExecReturn(e.GetMessageString());
331  }
332 }
