121  def getPoint(self,point=None,obj=None):
122  """Callback for clicks during interactive mode.
123 
124  When method _CommandWall.Activated() has entered the interactive mode,
125  this callback runs when the user clicks.
126 
127  Parameters
128  ----------
129  point: <class 'Base.Vector'>
130  The point the user has selected.
131  obj: <Part::PartFeature>, optional
132  The object the user's cursor snapped to, if any.
133  """
134 
135  import Draft
136  import Part
137  import Arch
138  import ArchWall
139  from draftutils import gui_utils
140  if obj:
141  if Draft.getType(obj) == "Wall":
142  if not obj in self.existing:
143  self.existing.append(obj)
144  if point is None:
145  self.tracker.finalize()
146  FreeCAD.activeDraftCommand = None
147  FreeCADGui.Snapper.off()
148  return
149  self.points.append(point)
150  if len(self.points) == 1:
151  self.tracker.width(self.Width)
152  self.tracker.height(self.Height)
153  self.tracker.on()
154  FreeCADGui.Snapper.getPoint(last=self.points[0],
155  callback=self.getPoint,
156  movecallback=self.update,
157  extradlg=self.taskbox(),
158  title=translate("Arch","Next point")+":",mode="line")
159 
160  elif len(self.points) == 2:
161  l = Part.LineSegment(self.wp.get_local_coords(self.points[0]),
162  self.wp.get_local_coords(self.points[1]))
163  self.tracker.off()
164  FreeCAD.activeDraftCommand = None
165  FreeCADGui.Snapper.off()
166  FreeCAD.ActiveDocument.openTransaction(translate("Arch","Create Wall"))
167  FreeCADGui.addModule("Arch")
168  FreeCADGui.doCommand('import Part')
169  FreeCADGui.doCommand('trace=Part.LineSegment(FreeCAD.'+str(l.StartPoint)+',FreeCAD.'+str(l.EndPoint)+')')
170  if not self.existing:
171  # no existing wall snapped, just add a default wall
172  self.addDefault()
173  else:
174  if self.JOIN_WALLS_SKETCHES:
175  # join existing subwalls first if possible, then add the new one
176  w = Arch.joinWalls(self.existing)
177  if w:
178  if ArchWall.areSameWallTypes([w,self]):
179  FreeCADGui.doCommand('FreeCAD.ActiveDocument.'+w.Name+'.Base.addGeometry(trace)')
180  else:
181  # if not possible, add new wall as addition to the existing one
182  self.addDefault()
183  if self.AUTOJOIN:
184  FreeCADGui.doCommand('Arch.addComponents(FreeCAD.ActiveDocument.'+FreeCAD.ActiveDocument.Objects[-1].Name+',FreeCAD.ActiveDocument.'+w.Name+')')
185  else:
186  self.addDefault()
187  else:
188  # add new wall as addition to the first existing one
189  self.addDefault()
190  if self.AUTOJOIN:
191  FreeCADGui.doCommand('Arch.addComponents(FreeCAD.ActiveDocument.'+FreeCAD.ActiveDocument.Objects[-1].Name+',FreeCAD.ActiveDocument.'+self.existing[0].Name+')')
192  FreeCAD.ActiveDocument.commitTransaction()
193  FreeCAD.ActiveDocument.recompute()
194  # gui_utils.end_all_events() # Causes a crash on Linux.
195  self.tracker.finalize()
196  if self.continueCmd:
197  self.Activated()
