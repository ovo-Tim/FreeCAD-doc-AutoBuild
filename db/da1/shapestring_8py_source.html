125  def execute(self, obj):
126  if self.props_changed_placement_only():
127  obj.positionBySupport()
128  self.props_changed_clear()
129  return
130 
131  if obj.String and obj.FontFile:
132  plm = obj.Placement
133 
134  if obj.FontFile[0] == ".":
135  # FontFile path relative to the FreeCAD file directory.
136  font_file = os.path.join(os.path.dirname(obj.Document.FileName), obj.FontFile)
137  else:
138  font_file = obj.FontFile
139 
140  fill = obj.MakeFace
141  if fill is True:
142  # Test a simple letter to know if we have a sticky font or not.
143  # If the font is sticky change fill to `False`.
144  # The 0.03 total area minimum is based on tests with:
145  # 1CamBam_Stick_0.ttf and 1CamBam_Stick_0C.ttf.
146  # See the make_faces function for more information.
147  char = Part.makeWireString("L", font_file, 1, 0)[0]
148  shapes = self.make_faces(char) # char is list of wires
149  if not shapes:
150  fill = False
151  else:
152  fill = sum([shape.Area for shape in shapes]) > 0.03\
153  and math.isclose(Part.Compound(char).BoundBox.DiagonalLength,
154  Part.Compound(shapes).BoundBox.DiagonalLength,
155  rel_tol=1e-7)
156 
157  chars = Part.makeWireString(obj.String, font_file, obj.Size, obj.Tracking)
158  shapes = []
159 
160  for char in chars:
161  if fill is False:
162  shapes.extend(char)
163  elif char:
164  shapes.extend(self.make_faces(char))
165  if shapes:
166  if fill and obj.Fuse:
167  ss_shape = shapes[0].fuse(shapes[1:])
168  ss_shape = faces.concatenate(ss_shape)
169  # Concatenate returns a Face or a Compound. We always
170  # need a Compound as we use ss_shape.SubShapes later.
171  if ss_shape.ShapeType == "Face":
172  ss_shape = Part.Compound([ss_shape])
173  else:
174  ss_shape = Part.Compound(shapes)
175  cap_char = Part.makeWireString("M", font_file, obj.Size, obj.Tracking)[0]
176  cap_height = Part.Compound(cap_char).BoundBox.YMax
177  if obj.ScaleToSize:
178  ss_shape.scale(obj.Size / cap_height)
179  cap_height = obj.Size
180  if obj.ObliqueAngle:
181  if -80 <= obj.ObliqueAngle <= 80:
182  mtx = App.Matrix()
183  mtx.A12 = math.tan(math.radians(obj.ObliqueAngle))
184  ss_shape = ss_shape.transformGeometry(mtx)
185  else:

208  def justification_vector(self, ss_shape, cap_height, just, just_ref, keep_left_margin): # ss_shape is a compound
209  box = ss_shape.optimalBoundingBox()
210  if keep_left_margin is True and "Left" in just:
211  vec = App.Vector(0, 0, 0)
212  else:
213  vec = App.Vector(-box.XMin, 0, 0) # remove left margin caused by kerning and white space characters
214  width = box.XLength
215  if "Shape" in just_ref:
216  vec = vec + App.Vector(0, -box.YMin, 0)
217  height = box.YLength
218  else:
219  height = cap_height
220  if "Top" in just:
221  vec = vec + App.Vector(0, -height, 0)
222  elif "Middle" in just:
223  vec = vec + App.Vector(0, -height/2, 0)
224  if "Right" in just:
225  vec = vec + App.Vector(-width, 0, 0)
226  elif "Center" in just:
227  vec = vec + App.Vector(-width/2, 0, 0)
228  return vec
