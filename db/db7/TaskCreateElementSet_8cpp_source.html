332 void writeToFile(std::string fileName,
333  SMESHDS_Mesh* newMesh,
334  int* nodeNumbers,
335  double** nodeCoords,
336  int rows,
337  int requiredType)
338 {
339  std::map<int, std::string> elType2D;
340  std::map<int, std::string> elType3D;
341 
342  elType2D[3] = "S3"; // 3 node triangle
343  elType2D[6] = "S6"; // 4 node quadrilateral
344  elType2D[4] = "S4"; // 6 node triangle
345  elType2D[8] = "S8"; // 8 node quadrilateral
346 
347  elType3D[4] = "C3D4"; // 4 node tetrahedron
348  elType3D[6] = "C3D6"; // 6 node pentahedron
349  elType3D[8] = "C3D8"; // 8 node hexahedron
350  elType3D[10] = "C3D10"; // 10 node tetrahedron
351  elType3D[15] = "C3D15"; // 15 node pentahedron
352  elType3D[20] = "C3D20"; // 20 node hexahedron
353  // no pyramid elements
354  FILE* fptr = fopen(fileName.c_str(), "w");
355  if (fptr == NULL) {
356  return;
357  }
358  fprintf(fptr, "** written by Erase Elements inp file writer for CalculiX,Abaqus meshes\n");
359  fprintf(fptr, "** all mesh elements.\n");
360 
361  fprintf(fptr, "\n");
362  fprintf(fptr, "\n");
363  fprintf(fptr, "** Nodes\n");
364  fprintf(fptr, "*Node, NSET=Nall\n");
365 
366  for (int i = 1; i < rows + 1; i++) {
367  if (nodeNumbers[i] > 0) {
368  fprintf(fptr,
369  "%d, %e, %e, %e\n",
370  nodeNumbers[i],
371  nodeCoords[i][0],
372  nodeCoords[i][1],
373  nodeCoords[i][2]);
374  }
375  }
376 
377  SMDS_ElemIteratorPtr srcElemIt;
378  SMDS_NodeIteratorPtr srcNodeIt;
379  srcElemIt = newMesh->elementsIterator();
380  srcNodeIt = newMesh->nodesIterator();
381  const SMDS_MeshNode* nSrc;
382  int NID, EID;
383  while (srcNodeIt->more()) {
384  const SMDS_MeshNode* node = srcNodeIt->next();
385  NID = node->GetID();
386  }
387  int numberNodes = -1;
388  while (srcElemIt->more()) {
389  const SMDS_MeshElement* elem = srcElemIt->next();
390  EID = elem->GetID();
391  if (elem->GetType() != requiredType) {
392  continue;
393  }
394 
395  if (numberNodes != elem->NbNodes()) {
396  if (requiredType == 4) {
397  fprintf(fptr, "\n");
398  fprintf(fptr, "\n");
399  fprintf(fptr, "%s", "** Volume elements\n");
400  fprintf(fptr,
401  "*Element, TYPE=%s, ELSET=Evolumes\n",
402  elType3D[elem->NbNodes()].c_str());
403  }
404  else if (requiredType == 3) {
405  fprintf(fptr, "%s", "** Face elements\n");
406  fprintf(fptr,
407  "*Element, TYPE=%s, ELSET=Efaces\n",
408  elType2D[elem->NbNodes()].c_str());
409  }
410  numberNodes = elem->NbNodes();
411  }
412  SMDS_ElemIteratorPtr nIt = elem->nodesIterator();
413  fprintf(fptr, "%d", EID);
414  while (nIt->more()) {
415  nSrc = static_cast<const SMDS_MeshNode*>(nIt->next());
416  NID = nSrc->GetID();
417  fprintf(fptr, ", %d", NID);
418  }
419  fprintf(fptr, "\n");
420  } // while print
421  if (requiredType == 4) {
422  fprintf(fptr, "\n");
423  fprintf(fptr, "\n");
424  fprintf(fptr, "%s", "** Define element set Eall\n");
425  fprintf(fptr, "%s", "*ELSET, ELSET=Eall\n");
426  fprintf(fptr, "%s", "Evolumes\n");
427  }
428  else if (requiredType == 3) {
429  fprintf(fptr, "%s", "** Define element set Eall\n");
430  fprintf(fptr, "%s", "*ELSET, ELSET=Eall\n");
431  fprintf(fptr, "%s", "Efaces\n");
432  }
433 
434  fclose(fptr);
435  return;
436 }
