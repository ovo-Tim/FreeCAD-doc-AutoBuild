67 App::DocumentObjectExecReturn *Thickness::execute() {
68  if (onlyHasToRefine()){
69  TopoShape result = refineShapeIfActive(rawShape);
70  Shape.setValue(result);
71  return App::DocumentObject::StdReturn;
72  }
73 
74  // Base shape
75  Part::TopoShape TopShape;
76  try {
77  TopShape = getBaseTopoShape();
78  }
79  catch (Base::Exception& e) {
80  return new App::DocumentObjectExecReturn(e.what());
81  }
82 
83  const std::vector<std::string>& subStrings = Base.getSubValues(true);
84 
85  // If the base has no sub elements listed just return a copy of the base.
86  if (subStrings.empty()) {
87  // We must set the placement of the feature in case it's empty.
88  this->positionByBaseFeature();
89  this->Shape.setValue(TopShape);
90  return App::DocumentObject::StdReturn;
91  }
92 
93  /* If the feature was ever empty, then Placement was set by positionByBaseFeature. However,
94  * makeThickSolid apparently requires the placement to be empty, so we have to clear it */
95  this->Placement.setValue(Base::Placement());
96 
97  std::map<int, std::vector<TopoShape>> closeFaces;
98  for ( const auto& it : subStrings ) {
99  TopoDS_Shape face;
100  try {
101  face = TopShape.getSubShape(it.c_str());
102  }
103  catch (...) {
104  }
105  if (face.IsNull())
106  return new App::DocumentObjectExecReturn(
107  QT_TRANSLATE_NOOP("Exception", "Invalid face reference"));
108  // We found the sub element (face) so let's get its history index in our shape
109  int index = TopShape.findAncestor(face, TopAbs_SOLID);
110  if (!index) {
111  FC_WARN(getFullName() << ": Ignore non-solid face " << it);
112  continue;
113  }
114  closeFaces[index].emplace_back(face);
115  }
116 
117  bool reversed = Reversed.getValue();
118  bool intersection = Intersection.getValue();
119  double thickness = (reversed ? -1. : 1.) * Value.getValue();
120  double tol = Precision::Confusion();
121  auto mode = static_cast<int16_t>(Mode.getValue());
122  auto join = Join.getValue();
123 
124  std::vector<TopoShape> shapes;
125  auto count = static_cast<int>(TopShape.countSubShapes(TopAbs_SOLID));
126  if (!count)
127  return new App::DocumentObjectExecReturn("No solid");
128  // we do not offer tangent join type
129  if (join == 1)
130  join = 2;
131 
132  if (fabs(thickness) > 2 * tol) {
133  auto mapIterator = closeFaces.begin();
134  for (auto loopIndex = 1; loopIndex <= count; ++loopIndex) {
135  std::vector<TopoShape> dummy;
136  const auto* faces = &dummy;
137  TopoShape solid = TopShape;
138  // expect the sub element indexes in the map to be in order and matching our loop index,
139  // and effectively ignore them if they are not.
140  if (mapIterator != closeFaces.end() && loopIndex >= mapIterator->first) {
141  faces = &mapIterator->second;
142  solid = TopShape.getSubTopoShape(TopAbs_SOLID, mapIterator->first);
143  }
144  TopoShape res(0);
145  try {
146  res = solid.makeElementThickSolid(*faces,
147  thickness,
148  tol,
149  intersection,
150  false,
151  mode,
152  static_cast<Part::JoinType>(join));
153  shapes.push_back(res);
154  }
155  catch (Standard_Failure& e) {
156  FC_ERR("Exception on making thick solid: " << e.GetMessageString());
157  return new App::DocumentObjectExecReturn("Failed to make thick solid");
158  }
159  if (mapIterator != closeFaces.end()) {
160  ++mapIterator;
161  }
162  }
163  }
164 
165  TopoShape result(0);
166  if (shapes.size() > 1) {
167  result.makeElementFuse(shapes);
168  } else if (shapes.empty()) {
169  result = TopShape;
170  } else {
171  result = shapes.front();
172  }
173  // store shape before refinement
174  this->rawShape = result;
175  result = refineShapeIfActive(result);
176  this->Shape.setValue(getSolid(result));
177  return App::DocumentObject::StdReturn;
178 }
