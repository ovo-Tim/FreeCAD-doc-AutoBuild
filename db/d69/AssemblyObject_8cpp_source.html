136 int AssemblyObject::solve(bool enableRedo, bool updateJCS)
137 {
138  ensureIdentityPlacements();
139 
140  mbdAssembly = makeMbdAssembly();
141  objectPartMap.clear();
142  motions.clear();
143 
144  auto groundedObjs = fixGroundedParts();
145  if (groundedObjs.empty()) {
146  // If no part fixed we can't solve.
147  return -6;
148  }
149 
150  std::vector<App::DocumentObject*> joints = getJoints(updateJCS);
151 
152  removeUnconnectedJoints(joints, groundedObjs);
153 
154  jointParts(joints);
155 
156  if (enableRedo) {
157  savePlacementsForUndo();
158  }
159 
160  try {
161  // mbdAssembly->runPreDrag(); // solve() is causing some issues with limits.
162  mbdAssembly->runKINEMATIC();
163  }
164  catch (const std::exception& e) {
165  FC_ERR("Solve failed: " << e.what());
166  return -1;
167  }
168  catch (...) {
169  FC_ERR("Solve failed: unhandled exception");
170  return -1;
171  }
172 
173  setNewPlacements();
174 
175  redrawJointPlacements(joints);
176 
177  return 0;
178 }

253 void AssemblyObject::preDrag(std::vector<App::DocumentObject*> dragParts)
254 {
255  bundleFixed = true;
256  solve();
257  bundleFixed = false;
258 
259  draggedParts.clear();
260  for (auto part : dragParts) {
261  // make sure no duplicate
262  if (std::find(draggedParts.begin(), draggedParts.end(), part) != draggedParts.end()) {
263  continue;
264  }
265 
266  // Some objects have been bundled, we don't want to add these to dragged parts
267  Base::Placement plc;
268  for (auto& pair : objectPartMap) {
269  App::DocumentObject* parti = pair.first;
270  if (parti != part) {
271  continue;
272  }
273  plc = pair.second.offsetPlc;
274  }
275  if (!plc.isIdentity()) {
276  // If not identity, then it's a bundled object. Some bundled objects may
277  // have identity placement if they have the same position as the main object of
278  // the bundle. But they're not going to be a problem.
279  continue;
280  }
281 
282  draggedParts.push_back(part);
283  }
284 
285  mbdAssembly->runPreDrag();
286 }

288 void AssemblyObject::doDragStep()
289 {
290  try {
291  std::vector<std::shared_ptr<MbD::ASMTPart>> dragMbdParts;
292 
293  for (auto& part : draggedParts) {
294  if (!part) {
295  continue;
296  }
297 
298  auto mbdPart = getMbDPart(part);
299  dragMbdParts.push_back(mbdPart);
300 
301  // Update the MBD part's position
302  Base::Placement plc = getPlacementFromProp(part, "Placement");
303  Base::Vector3d pos = plc.getPosition();
304  mbdPart->updateMbDFromPosition3D(
305  std::make_shared<FullColumn<double>>(ListD {pos.x, pos.y, pos.z}));
306 
307  // Update the MBD part's rotation
308  Base::Rotation rot = plc.getRotation();
309  Base::Matrix4D mat;
310  rot.getValue(mat);
311  Base::Vector3d r0 = mat.getRow(0);
312  Base::Vector3d r1 = mat.getRow(1);
313  Base::Vector3d r2 = mat.getRow(2);
314  mbdPart
315  ->updateMbDFromRotationMatrix(r0.x, r0.y, r0.z, r1.x, r1.y, r1.z, r2.x, r2.y, r2.z);
316  }
317 
318  // Timing mbdAssembly->runDragStep()
319  auto dragPartsVec = std::make_shared<std::vector<std::shared_ptr<ASMTPart>>>(dragMbdParts);
320  mbdAssembly->runDragStep(dragPartsVec);
321 
322  // Timing the validation and placement setting
323  if (validateNewPlacements()) {
324  setNewPlacements();
325 
326  auto joints = getJoints(false);
327  for (auto* joint : joints) {
328  if (joint->Visibility.getValue()) {
329  // redraw only the moving joint as its quite slow as its python code.
330  redrawJointPlacement(joint);
331  }
332  }
333  }
334  }
335  catch (...) {
336  // We do nothing if a solve step fails.
337  }
338 }

341 {
342  if (!mbdPart) {

344  }

357 bool AssemblyObject::validateNewPlacements()
358 {
359  // First we check if a grounded object has moved. It can happen that they flip.
360  auto groundedParts = getGroundedParts();
361  for (auto* obj : groundedParts) {
362  auto* propPlacement =
363  dynamic_cast<App::PropertyPlacement*>(obj->getPropertyByName("Placement"));
364  if (propPlacement) {
365  Base::Placement oldPlc = propPlacement->getValue();
366 
367  auto it = objectPartMap.find(obj);
368  if (it != objectPartMap.end()) {
369  std::shared_ptr<MbD::ASMTPart> mbdPart = it->second.part;
370  Base::Placement newPlacement = getMbdPlacement(mbdPart);
371  if (!it->second.offsetPlc.isIdentity()) {
372  newPlacement = newPlacement * it->second.offsetPlc;
373  }
374 
375  if (!oldPlc.isSame(newPlacement)) {
376  Base::Console().Warning(
377  "Assembly : Ignoring bad solve, a grounded object (%s) moved.\n",
378  obj->getFullLabel());
379  return false;
380  }
381  }
382  }
383  }
384 
385  // TODO: We could do further tests
386  // For example check if the joints connectors are correctly aligned.
387 
388  return true;
389 }

523 void AssemblyObject::recomputeJointPlacements(std::vector<App::DocumentObject*> joints)
524 {
525  // The Placement1 and Placement2 of each joint needs to be updated as the parts moved.
526  Base::PyGILStateLocker lock;
527  for (auto* joint : joints) {
528  if (!joint) {
529  continue;
530  }
531 
532  App::PropertyPythonObject* proxy = joint
533  ? dynamic_cast<App::PropertyPythonObject*>(joint->getPropertyByName("Proxy"))
534  : nullptr;
535 
536  if (!proxy) {
537  continue;
538  }
539 
540  Py::Object jointPy = proxy->getValue();
541 
542  if (!jointPy.hasAttr("updateJCSPlacements")) {
543  continue;
544  }
545 
546  Py::Object attr = jointPy.getAttr("updateJCSPlacements");
547  if (attr.ptr() && attr.isCallable()) {
548  Py::Tuple args(1);
549  args.setItem(0, Py::asObject(joint->getPyObject()));
550  Py::Callable(attr).apply(args);
551  }
552  }

638 std::vector<App::DocumentObject*>
639 AssemblyObject::getJoints(bool updateJCS, bool delBadJoints, bool subJoints)
640 {
641  std::vector<App::DocumentObject*> joints = {};
642 
643  JointGroup* jointGroup = getJointGroup();
644  if (!jointGroup) {
645  return {};
646  }
647 
648  Base::PyGILStateLocker lock;
649  for (auto joint : jointGroup->getObjects()) {
650  if (!joint) {
651  continue;
652  }
653 
654  auto* prop = dynamic_cast<App::PropertyBool*>(joint->getPropertyByName("Activated"));
655  if (!prop || !prop->getValue()) {
656  // Filter grounded joints and deactivated joints.
657  continue;
658  }
659 
660  auto* part1 = getMovingPartFromRef(this, joint, "Reference1");
661  auto* part2 = getMovingPartFromRef(this, joint, "Reference2");
662  if (!part1 || !part2 || part1->getFullName() == part2->getFullName()) {
663  // Remove incomplete joints. Left-over when the user deletes a part.
664  // Remove incoherent joints (self-pointing joints)
665  if (delBadJoints) {
666  getDocument()->removeObject(joint->getNameInDocument());
667  }
668  continue;
669  }
670 
671  auto proxy = dynamic_cast<App::PropertyPythonObject*>(joint->getPropertyByName("Proxy"));
672  if (proxy) {
673  if (proxy->getValue().hasAttr("setJointConnectors")) {
674  joints.push_back(joint);
675  }
676  }
677  }
678 
679  // add sub assemblies joints.
680  if (subJoints) {
681  for (auto& assembly : getSubAssemblies()) {
682  auto subJoints = assembly->getJoints();
683  joints.insert(joints.end(), subJoints.begin(), subJoints.end());
684  }
685  }
686 
687  // Make sure the joints are up to date.
688  if (updateJCS) {
689  recomputeJointPlacements(joints);
690  }
691 
692  return joints;
693 }

1698  std::shared_ptr<ASMTPart> mbdPart = data1.part;

1718 {

1731  if (getJointType(jt) == JointType::Slider) {

1734  int found = 0;

1736  if (jpart1 == part1 || jpart1 == part2) {
1737  found = (jpart1 == part1) ? 1 : 2;
1738  plci = (jpart1 == part1) ? plc1 : plc2;
1739  plcjt = getPlacementFromProp(jt, "Placement1");
1740  }
1741  else if (jpart2 == part1 || jpart2 == part2) {
1742  found = (jpart2 == part1) ? 1 : 2;
1743  plci = (jpart2 == part1) ? plc1 : plc2;
1744  plcjt = getPlacementFromProp(jt, "Placement2");

1751  plcjt.getRotation().getYawPitchRoll(y1, p1, r1);
1752  plci.getRotation().getYawPitchRoll(y2, p2, r2);

1758  }
