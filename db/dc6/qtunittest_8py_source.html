193 class RollbackImporter:
194  """This tricky little class is used to make sure that modules under test
195  will be reloaded the next time they are imported.
196  """
197 
198  def __init__(self):
199  self.previousModules = sys.modules.copy()
200 
201  def rollbackImports(self):
202  for modname in sys.modules.keys():
203  if modname not in self.previousModules:
204  # Force reload when modname next imported
205  del sys.modules[modname]
