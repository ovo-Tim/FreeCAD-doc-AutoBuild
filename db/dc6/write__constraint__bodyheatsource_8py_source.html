69 def write_constraint(f, femobj, bodyheatsource_obj, ccxwriter):
70 
71  # floats read from ccx should use {:.13G}, see comment in writer module
72  # search referenced material
73  ref = bodyheatsource_obj.References[0]
74  ref_feat = ref[0]
75  ref_sub_obj = ref[1][0]
76  density = None
77  for mat in ccxwriter.member.mats_linear:
78  mat_ref = [
79  *itertools.chain(*[itertools.product([i[0]], i[1]) for i in mat["Object"].References])
80  ]
81  if (ref_feat, ref_sub_obj) in mat_ref:
82  density = FreeCAD.Units.Quantity(mat["Object"].Material["Density"])
83  break
84 
85  if not density:
86  # search material without references
87  for mat in ccxwriter.member.mats_linear:
88  if not mat["Object"].References:
89  density = FreeCAD.Units.Quantity(mat["Object"].Material["Density"])
90  break
91 
92  # get data from the bodyheatsource_obj (DissipationRate is in power per unit mass)
93  if bodyheatsource_obj.Mode == "Dissipation Rate":
94  heat = bodyheatsource_obj.DissipationRate * density
95  elif bodyheatsource_obj.Mode == "Total Power":
96  volume = ref_feat.getSubObject(ref_sub_obj).Volume
97  heat = bodyheatsource_obj.TotalPower / FreeCAD.Units.Quantity(volume, "mm^3")
98  # write to file
99  f.write("*DFLUX\n")
100  f.write("{},BF,{:.13G}\n".format(bodyheatsource_obj.Name, heat.getValueAs("t/(mm*s^3)").Value))
101  f.write("\n")
