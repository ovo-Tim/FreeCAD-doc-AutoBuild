93 Part::Feature* Transformed::getBaseObject(bool silent) const
94 {
95  Part::Feature* rv = Feature::getBaseObject(/* silent = */ true);
96  if (rv) {
97  return rv;
98  }
99 
100  const char* err = nullptr;
101  const std::vector<App::DocumentObject*>& originals = Originals.getValues();
102  // NOTE: may be here supposed to be last origin but in order to keep the old behaviour keep here
103  // first
104  App::DocumentObject* firstOriginal = originals.empty() ? nullptr : originals.front();
105  if (firstOriginal) {
106  rv = Base::freecad_dynamic_cast<Part::Feature>(firstOriginal);
107  if (!rv) {
108  err = QT_TRANSLATE_NOOP("Exception",
109  "Transformation feature Linked object is not a Part object");
110  }
111  }
112  else {
113  err = QT_TRANSLATE_NOOP("Exception", "No originals linked to the transformed feature.");
114  }
115 
116  if (!silent && err) {
117  throw Base::RuntimeError(err);
118  }
119 
120  return rv;

207 App::DocumentObjectExecReturn* Transformed::execute()
208 {
209  if (isMultiTransformChild()) {
210  return App::DocumentObject::StdReturn;
211  }
212 
213  std::vector<App::DocumentObject*> originals;
214  auto const mode = static_cast<Mode>(TransformMode.getValue());
215  if (mode == Mode::TransformBody) {
216  Originals.setStatus(App::Property::Status::Hidden, true);
217  } else {
218  Originals.setStatus(App::Property::Status::Hidden, false);
219  originals = Originals.getValues();
220  }
221  // Remove suppressed features from the list so the transformations behave as if they are not
222  // there
223  auto eraseIter =
224  std::remove_if(originals.begin(), originals.end(), [](App::DocumentObject const* obj) {
225  auto feature = Base::freecad_dynamic_cast<PartDesign::Feature>(obj);
226  return feature != nullptr && feature->Suppressed.getValue();
227  });
228  originals.erase(eraseIter, originals.end());
229 
230  if (mode == Mode::TransformToolShapes && originals.empty()) {
231  return App::DocumentObject::StdReturn;
232  }
233 
234  if (!this->BaseFeature.getValue()) {
235  auto body = getFeatureBody();
236  if (body) {
237  body->setBaseProperty(this);
238  }
239  }
240 
241  this->positionBySupport();
242 
243  // get transformations from subclass by calling virtual method
244  std::vector<gp_Trsf> transformations;
245  try {
246  std::list<gp_Trsf> t_list = getTransformations(originals);
247  transformations.insert(transformations.end(), t_list.begin(), t_list.end());
248  }
249  catch (Base::Exception& e) {
250  return new App::DocumentObjectExecReturn(e.what());
251  }
252  catch (const Standard_Failure& e) {
253  return new App::DocumentObjectExecReturn(e.GetMessageString());
254  }
255 
256  if (transformations.empty()) {
257  return App::DocumentObject::StdReturn; // No transformations defined, exit silently
258  }
259 
260  // Get the support
261  Part::Feature* supportFeature = nullptr;
262 
263  try {
264  supportFeature = getBaseObject();
265  }
266  catch (Base::Exception& e) {
267  return new App::DocumentObjectExecReturn(e.what());
268  }
269 
270  const Part::TopoShape& supportTopShape = supportFeature->Shape.getShape();
271  if (supportTopShape.getShape().IsNull()) {
272  return new App::DocumentObjectExecReturn(
273  QT_TRANSLATE_NOOP("Exception", "Cannot transform invalid support shape"));
274  }
275 
276  // create an untransformed copy of the support shape
277  Part::TopoShape supportShape(supportTopShape);
278 
279  gp_Trsf trsfInv = supportShape.getShape().Location().Transformation().Inverted();
280 
281  supportShape.setTransform(Base::Matrix4D());
282 
283  auto getTransformedCompShape = [&](const auto& supportShape, const auto& origShape) {
284  std::vector<TopoShape> shapes = {supportShape};
285  TopoShape shape (origShape);
286  int idx=1;
287  auto transformIter = transformations.cbegin();
288  transformIter++;
289  for ( ; transformIter != transformations.end(); transformIter++) {
290  auto opName = Data::indexSuffix(idx++);
291  shapes.emplace_back(shape.makeElementTransform(*transformIter, opName.c_str()));
292  }
293  return shapes;
294  };
295 
296  switch (mode) {
297  case Mode::TransformToolShapes:
298  // NOTE: It would be possible to build a compound from all original addShapes/subShapes
299  // and then transform the compounds as a whole. But we choose to apply the
300  // transformations to each Original separately. This way it is easier to discover what
301  // feature causes a fuse/cut to fail. The downside is that performance suffers when
302  // there are many originals. But it seems safe to assume that in most cases there are
303  // few originals and many transformations
304  for (auto original : originals) {
305  // Extract the original shape and determine whether to cut or to fuse
306  Part::TopoShape fuseShape;
307  Part::TopoShape cutShape;
308 
309  auto feature = Base::freecad_dynamic_cast<PartDesign::FeatureAddSub>(original);
310  if (!feature) {
311  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP(
312  "Exception",
313  "Only additive and subtractive features can be transformed"));
314  }
315 
316  feature->getAddSubShape(fuseShape, cutShape);
317  if (fuseShape.isNull() && cutShape.isNull()) {
318  return new App::DocumentObjectExecReturn(
319  QT_TRANSLATE_NOOP("Exception",
320  "Shape of additive/subtractive feature is empty"));
321  }
322  gp_Trsf trsf = feature->getLocation().Transformation().Multiplied(trsfInv);
323  if (!fuseShape.isNull()) {
324  fuseShape = fuseShape.makeElementTransform(trsf);
325  }
326  if (!cutShape.isNull()) {
327  cutShape = cutShape.makeElementTransform(trsf);
328  }
329  if (!fuseShape.isNull()) {
330  supportShape.makeElementFuse(getTransformedCompShape(supportShape, fuseShape));
331  }
332  if (!cutShape.isNull()) {
333  supportShape.makeElementCut(getTransformedCompShape(supportShape, cutShape));
334  }
335  }
336  break;
337  case Mode::TransformBody: {
338  supportShape.makeElementFuse(getTransformedCompShape(supportShape, supportShape));
339  break;
340  }
341  }
342 
343  supportShape = refineShapeIfActive((supportShape));
344  if (!isSingleSolidRuleSatisfied(supportShape.getShape())) {
345  Base::Console().Warning("Transformed: Result has multiple solids. Only keeping the first.\n");
346  }
347 
348  this->Shape.setValue(getSolid(supportShape)); // picking the first solid
349  rejected = getRemainingSolids(supportShape.getShape());
350 
351  return App::DocumentObject::StdReturn;
352 }

363 TopoDS_Shape Transformed::getRemainingSolids(const TopoDS_Shape& shape)
364 {
365  BRep_Builder builder;
366  TopoDS_Compound compShape;
367  builder.MakeCompound(compShape);
368 
369  if (shape.IsNull()) {
370  Standard_Failure::Raise("Shape is null");
371  }
372  TopExp_Explorer xp;
373  xp.Init(shape, TopAbs_SOLID);
374  xp.Next(); // skip the first
375 
376  for (; xp.More(); xp.Next()) {
377  builder.Add(compShape, xp.Current());
378  }
379 
380  return {std::move(compShape)};
381 }
