24 """Provides the ToDo static class to run commands with a time delay.
25 
26 The `ToDo` class is used to delay the commit of commands for later execution.
27 This is necessary when a GUI command needs to manipulate the 3D view

57 class ToDo:
58  """A static class that delays execution of functions.
59 
60  It calls `QtCore.QTimer.singleShot(0, doTasks)`
61  where `doTasks` is a static method which executes
62  the commands stored in the list attributes.
63 
64  Attributes
65  ----------
66  itinerary: list of tuples
67  Each tuple is of the form `(name, arg)`.
68  The `name` is a reference (pointer) to a function,
69  and `arg` is the corresponding argument that is passed
70  to that function.
71  It then tries executing the function with the argument,
72  if available, or without it, if not available.
73  ::
74  name(arg)
75  name()
76 
77  commitlist: list of tuples
78  Each tuple is of the form `(name, command_list)`.
79  The `name` is a string identifier or description of the commands
80  that will be run, and `command_list` is a list of strings
81  that indicate the Python instructions that will be executed,
82  or a reference to a single function that will be executed.
83 
84  If `command_list` is a list, the program opens a transaction,
85  then runs all commands in the list in sequence,
86  and finally commits the transaction.
87  ::
88  command_list = ["command1", "command2", "..."]
89  App.activeDocument().openTransaction(name)
90  Gui.doCommand("command1")
91  Gui.doCommand("command2")
92  Gui.doCommand("...")
93  App.activeDocument().commitTransaction()
94 
95  If `command_list` is a reference to a function
96  the function is executed directly.
97  ::
98  command_list = function
99  App.activeDocument().openTransaction(name)
100  function()
101  App.activeDocument().commitTransaction()
102 
103  afteritinerary: list of tuples
104  Each tuple is of the form `(name, arg)`.
105  This list is used just like `itinerary`.
106 
107  Lists
108  -----
109  The lists contain tuples. Each tuple contains a `name` which is just
110  a string to identify the operation, and a `command_list` which is
111  a list of strings, each string an individual Python instruction.
112  """
113 
114  itinerary = []
115  commitlist = []
116  afteritinerary = []

119  def doTasks():
120  """Execute the commands stored in the lists.
121 
122  The lists are `itinerary`, `commitlist` and `afteritinerary`.
123  """
124  if _DEBUG:
125  _msg("Debug: doing delayed tasks.\n"
126  "itinerary: {0}\n"
127  "commitlist: {1}\n"
128  "afteritinerary: {2}\n".format(todo.itinerary,
129  todo.commitlist,
130  todo.afteritinerary))
131  try:
132  for f, arg in ToDo.itinerary:
133  try:
134  if _DEBUG_inner:
135  _msg("Debug: executing.\n"
136  "function: {}\n".format(f))
137  if arg or (arg is False):
138  f(arg)
139  else:
140  f()
141  except Exception:
142  _log(traceback.format_exc())
143  _err(traceback.format_exc())
144  wrn = ("ToDo.doTasks, Unexpected error:\n"
145  "{0}\n"
146  "in {1}({2})".format(sys.exc_info()[0], f, arg))
147  _wrn(wrn)
148  except ReferenceError:
149  _wrn("Debug: ToDo.doTasks: "
150  "queue contains a deleted object, skipping")
151  ToDo.itinerary = []
152 
153  if ToDo.commitlist:
154  commit_list = ToDo.commitlist
155  ToDo.commitlist = [] # Reset immediately to avoid race condition.
156  for name, func in commit_list:
157  if _DEBUG_inner:
158  _msg("Debug: committing.\n"
159  "name: {}\n".format(name))
160  try:
161  name = str(name)
162  App.activeDocument().openTransaction(name)
163  if isinstance(func, list):
164  for string in func:
165  Gui.doCommand(string)
166  else:
167  func()
168  App.activeDocument().commitTransaction()
169  except Exception:
170  _log(traceback.format_exc())
171  _err(traceback.format_exc())
172  wrn = ("ToDo.doTasks, Unexpected error:\n"
173  "{0}\n"
174  "in {1}".format(sys.exc_info()[0], func))
175  _wrn(wrn)
176  # Restack Draft screen widgets after creation
177  if hasattr(Gui, "Snapper"):
178  Gui.Snapper.restack()

180 
181  for f, arg in ToDo.afteritinerary:

185  "function: {}\n".format(f))

199  @staticmethod
200  def delay(f, arg):
201  """Add the function and argument to the itinerary list.
202 
203  Schedule geometry manipulation that would crash Coin if done
204  in the event callback.
205 
206  If the `itinerary` list is empty, it will call
207  `QtCore.QTimer.singleShot(0, ToDo.doTasks)`
208  to execute the commands in the other lists.
209 
210  Finally, it will build the tuple `(f, arg)`
211  and append it to the `itinerary` list.
212 
213  Parameters
214  ----------
215  f: function reference
216  A reference (pointer) to a Python command
217  which can be executed directly.
218  ::
219  f()
220 
221  arg: argument reference
222  A reference (pointer) to the argument to the `f` function.
223  ::
224  f(arg)
225  """
226  if _DEBUG:
227  _msg("Debug: delaying.\n"
228  "function: {}\n".format(f))
229  if ToDo.itinerary == []:
230  QtCore.QTimer.singleShot(0, ToDo.doTasks)
231  ToDo.itinerary.append((f, arg))

246  Parameters

248  cl: list of tuples
