52 class _PropHelper:
53  """
54  Helper class to manage property data inside proxy objects.
55  Initialization keywords are the same used with PropertyContainer
56  to add dynamics properties plus "value" for the initial value.
57  """
58 
59  def __init__(self, **kwds):
60  self.value = kwds.pop("value")
61  self.info = kwds
62  self.name = kwds["name"]
63 
64  def add_to_object(self, obj):
65  obj.addProperty(**self.info)
66  obj.setPropertyStatus(self.name, "LockDynamic")
67  setattr(obj, self.name, self.value)
68 
69  def handle_change_type(self, obj, old_type, convert_old_value=lambda x: x):
70  if obj.getTypeIdOfProperty(self.name) == old_type:
71  new_value = convert_old_value(obj.getPropertyByName(self.name))
72  obj.setPropertyStatus(self.name, "-LockDynamic")
73  obj.removeProperty(self.name)
74  self.add_to_object(obj)
75  setattr(obj, self.name, new_value)
