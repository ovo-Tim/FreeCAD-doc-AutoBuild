119 void DlgSettingsFemElmerImp::onfileNameChangedMT(QString FileName)
120 {
121  ui->sb_elmer_num_cores->setMaximum(processor_count);
122 
123  if (ui->sb_elmer_num_cores->value() == 1) {
124  return;
125  }
126 
127 #if defined(FC_OS_WIN32)
128  // name ends with "_mpi.exe"
129  if (!FileName.endsWith(QLatin1String("_mpi.exe"))) {
130  QMessageBox::warning(this,
131  tr("FEM Elmer: Not suitable for multithreading"),
132  tr("Wrong Elmer setting: You use more than one CPU core.\n"
133  "Therefore an executable with the suffix '_mpi.exe' is required."));
134  ui->sb_elmer_num_cores->setValue(1);
135  ui->sb_elmer_num_cores->setMaximum(1);
136  return;
137  }
138 #elif defined(FC_OS_LINUX) || defined(FC_OS_CYGWIN) || defined(FC_OS_MACOSX) || defined(FC_OS_BSD)
139  // name ends with "_mpi"
140  if (!FileName.endsWith(QLatin1String("_mpi"))) {
141  QMessageBox::warning(this,
142  tr("FEM Elmer: Not suitable for multithreading"),
143  tr("Wrong Elmer setting: You use more than one CPU core.\n"
144  "Therefore an executable with the suffix '_mpi' is required."));
145  ui->sb_elmer_num_cores->setValue(1);
146  ui->sb_elmer_num_cores->setMaximum(1);
