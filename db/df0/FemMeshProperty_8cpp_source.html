103 void PropertyFemMesh::setPyObject(PyObject* value)
104 {
105  if (PyObject_TypeCheck(value, &(FemMeshPy::Type))) {
106  FemMeshPy* pcObject = static_cast<FemMeshPy*>(value);
107  setValue(*pcObject->getFemMeshPtr());
108  }
109  else if (PyObject_TypeCheck(value, &(Base::PlacementPy::Type))) {
110  Base::PlacementPy* pcObject = static_cast<Base::PlacementPy*>(value);
111  transformGeometry(pcObject->getPlacementPtr()->toMatrix());
112  }
113  else {
114  std::string error = std::string("type must be 'FemMesh', not ");
