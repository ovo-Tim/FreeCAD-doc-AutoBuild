69 void ConstraintBearing::onChanged(const App::Property* prop)
70 {
71  // Note: If we call this at the end, then the symbol is not oriented correctly initially
72  // because the NormalDirection has not been calculated yet
73  Constraint::onChanged(prop);
74 
75  if (prop == &References) {
76  // Find data of cylinder
77  std::vector<App::DocumentObject*> ref = References.getValues();
78  std::vector<std::string> subRef = References.getSubValues();
79  if (ref.empty()) {
80  return;
81  }
82 
83  Part::Feature* feat = static_cast<Part::Feature*>(ref.front());
84  TopoDS_Shape sh = Tools::getFeatureSubShape(feat, subRef.front().c_str(), true);
85  double radius, height;
86  Base::Vector3d base, axis;
87  if (sh.IsNull() || !Tools::getCylinderParams(sh, base, axis, height, radius)) {
88  return;
89  }
90  Radius.setValue(radius);
91  Axis.setValue(axis);
92  Height.setValue(height);
93  // Update base point
94  base = base + axis * height / 2;
95  if (Location.getValue()) {
96  base = getBasePoint(base, axis, Location, Dist.getValue());
97  }
98  BasePoint.setValue(base);
99  BasePoint.touch(); // This triggers ViewProvider::updateData()
100  }
101  else if ((prop == &Location) || (prop == &Dist)) {
102  App::DocumentObject* obj = Location.getValue();
103  std::vector<std::string> names = Location.getSubValues();
104  if (names.empty()) {
105  return;
106  }
107  std::string subName = names.front();
108  Part::Feature* feat = static_cast<Part::Feature*>(obj);
109  TopoDS_Shape sh = feat->Shape.getShape().getSubShape(subName.c_str());
110 
111  if (sh.ShapeType() == TopAbs_FACE) {
112  BRepAdaptor_Surface surface(TopoDS::Face(sh));
113  if (surface.GetType() != GeomAbs_Plane) {
114  return; // "Location must be a planar face or linear edge"
115  }
116  }
117  else if (sh.ShapeType() == TopAbs_EDGE) {
118  BRepAdaptor_Curve line(TopoDS::Edge(sh));
119  if (line.GetType() != GeomAbs_Line) {
120  return; // "Location must be a planar face or linear edge"
121  }
122  }
123 
124  std::vector<App::DocumentObject*> ref = References.getValues();
125  std::vector<std::string> subRef = References.getSubValues();
126  if (ref.empty()) {
127  return;
128  }
129 
130  feat = static_cast<Part::Feature*>(ref.front());
131  sh = Tools::getFeatureSubShape(feat, subRef.front().c_str(), true);
132  double radius, height;
133  Base::Vector3d base, axis;
134  if (!Tools::getCylinderParams(sh, base, axis, height, radius)) {
135  return;
136  }
137  BasePoint.setValue(base);
138  BasePoint.touch();
139  }
140 }
