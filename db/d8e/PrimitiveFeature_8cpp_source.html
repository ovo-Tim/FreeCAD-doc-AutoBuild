114 void Primitive::handleChangedPropertyType(Base::XMLReader &reader, const char * TypeName, App::Property * prop)
115 {
116  // For #0001652 the property types of many primitive features have changed
117  // from PropertyFloat or PropertyFloatConstraint to a more meaningful type.
118  // In order to load older project files there must be checked in case the
119  // types don't match if both inherit from PropertyFloat because all derived
120  // classes do not re-implement the Save/Restore methods.
121  Base::Type inputType = Base::Type::fromName(TypeName);
122  if (prop->isDerivedFrom<App::PropertyFloat>() &&
123  inputType.isDerivedFrom(App::PropertyFloat::getClassTypeId())) {
124  // Do not directly call the property's Restore method in case the implementation
125  // has changed. So, create a temporary PropertyFloat object and assign the value.
126  App::PropertyFloat floatProp;
127  floatProp.Restore(reader);
128  static_cast<App::PropertyFloat*>(prop)->setValue(floatProp.getValue());
129  }
130  else {
131  Part::Feature::handleChangedPropertyType(reader, TypeName, prop);
132  }
133 }

281 App::DocumentObjectExecReturn *Plane::execute()
282 {
283  double L = this->Length.getValue();
284  double W = this->Width.getValue();
285 
286  if (L < Precision::Confusion())
287  return new App::DocumentObjectExecReturn("Length of plane too small");
288  if (W < Precision::Confusion())
289  return new App::DocumentObjectExecReturn("Width of plane too small");
290 
291  gp_Pnt pnt(0.0,0.0,0.0);
292  gp_Dir dir(0.0,0.0,1.0);
293  Handle(Geom_Plane) aPlane = new Geom_Plane(pnt, dir);
294  BRepBuilderAPI_MakeFace mkFace(aPlane, 0.0, L, 0.0, W, Precision::Confusion());
295 
296  const char *error=nullptr;
297  switch (mkFace.Error())
298  {
299  case BRepBuilderAPI_FaceDone:
300  break; // ok
301  case BRepBuilderAPI_NoFace:
302  error = "no face";
303  break;
304  case BRepBuilderAPI_NotPlanar:
305  error = "not planar";
306  break;
307  case BRepBuilderAPI_CurveProjectionFailed:
308  break;
309  case BRepBuilderAPI_ParametersOutOfRange:
310  error = "parameters out of range";
311  break;
312  default:
313  error = "unknown error";
314  break;
315  }
316  // Error ?
317  if (error) {
318  return new App::DocumentObjectExecReturn(error);
319  }
320 
321  TopoDS_Shape ResultShape = mkFace.Shape();
322  this->Shape.setValue(ResultShape);
323 
324  return Primitive::execute();
325 }

410 App::DocumentObjectExecReturn *Ellipsoid::execute()
411 {
412  // Build a sphere
413  if (Radius1.getValue() < Precision::Confusion())
414  return new App::DocumentObjectExecReturn("Radius of ellipsoid too small");
415  if (Radius2.getValue() < Precision::Confusion())
416  return new App::DocumentObjectExecReturn("Radius of ellipsoid too small");
417 
418  try {
419  gp_Pnt pnt(0.0,0.0,0.0);
420  gp_Dir dir(0.0,0.0,1.0);
421  gp_Ax2 ax2(pnt,dir);
422  BRepPrimAPI_MakeSphere mkSphere(ax2,
423  Radius2.getValue(),
424  Base::toRadians<double>(Angle1.getValue()),
425  Base::toRadians<double>(Angle2.getValue()),
426  Base::toRadians<double>(Angle3.getValue()));
427  Standard_Real scaleX = 1.0;
428  Standard_Real scaleZ = Radius1.getValue()/Radius2.getValue();
429  // issue #1798: A third radius has been introduced. To be backward
430  // compatible if Radius3 is 0.0 (default) it's handled to be the same
431  // as Radius2
432  Standard_Real scaleY = 1.0;
433  if (Radius3.getValue() >= Precision::Confusion())
434  scaleY = Radius3.getValue()/Radius2.getValue();
435  gp_GTrsf mat;
436  mat.SetValue(1,1,scaleX);
437  mat.SetValue(2,1,0.0);
438  mat.SetValue(3,1,0.0);
439  mat.SetValue(1,2,0.0);
440  mat.SetValue(2,2,scaleY);
441  mat.SetValue(3,2,0.0);
442  mat.SetValue(1,3,0.0);
443  mat.SetValue(2,3,0.0);
444  mat.SetValue(3,3,scaleZ);
445  BRepBuilderAPI_GTransform mkTrsf(mkSphere.Shape(), mat);
446  TopoDS_Shape ResultShape = mkTrsf.Shape();
447  this->Shape.setValue(ResultShape);
448  }
449  catch (Standard_Failure& e) {
450 
451  return new App::DocumentObjectExecReturn(e.GetMessageString());
452  }
453 
454  return Primitive::execute();
455 }

531 App::DocumentObjectExecReturn *Prism::execute()
532 {
533  // Build a prism
534  if (Polygon.getValue() < 3)
535  return new App::DocumentObjectExecReturn("Polygon of prism is invalid, must have 3 or more sides");
536  if (Circumradius.getValue() < Precision::Confusion())
537  return new App::DocumentObjectExecReturn("Circumradius of the polygon, of the prism, is too small");
538  if (Height.getValue() < Precision::Confusion())
539  return new App::DocumentObjectExecReturn("Height of prism is too small");
540  try {
541  long nodes = Polygon.getValue();
542 
543  Base::Matrix4D mat;
544  mat.rotZ(Base::toRadians(360.0/nodes));
545 
546  // create polygon
547  BRepBuilderAPI_MakePolygon mkPoly;
548  Base::Vector3d v(Circumradius.getValue(),0,0);
549  for (long i=0; i<nodes; i++) {
550  mkPoly.Add(gp_Pnt(v.x,v.y,v.z));
551  v = mat * v;
552  }
553  mkPoly.Add(gp_Pnt(v.x,v.y,v.z));
554  BRepBuilderAPI_MakeFace mkFace(mkPoly.Wire());
555  // the direction vector for the prism is the height for z and the given angle
556  this->Shape.setValue(makePrism(Height.getValue(), mkFace.Face()));
557  }
558  catch (Standard_Failure& e) {
559  return new App::DocumentObjectExecReturn(e.GetMessageString());
560  }
561 
562  return Primitive::execute();
563 }
