500  void onChanged(const App::Property* prop) override {
501  if (prop == &Proxy) {
502  imp->init(Proxy.getValue().ptr());
503  if (ViewProviderT::pcObject && !Proxy.getValue().is(Py::_None())) {
504  if (!_attached) {
505  _attached = true;
506  imp->attach(ViewProviderT::pcObject);
507  ViewProviderT::attach(ViewProviderT::pcObject);
508  // needed to load the right display mode after they're known now
509  ViewProviderT::DisplayMode.touch();
510  ViewProviderT::setOverrideMode(viewerMode);
511  }
512  if (!this->testStatus(Gui::isRestoring) &&
513  !canAddToSceneGraph()) {
514  this->getDocument()->toggleInSceneGraph(this);
515  }
516  ViewProviderT::updateView();
517  }
