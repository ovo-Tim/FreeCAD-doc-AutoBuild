43 def rotate(objectslist, angle, center=App.Vector(0, 0, 0),
44  axis=App.Vector(0, 0, 1), copy=False):
45  """rotate(objects,angle,[center,axis,copy])
46 
47  Rotates the objects contained in objects (that can be a list of objects
48  or an object) of the given angle (in degrees) around the center, using
49  axis as a rotation axis.
50 
51  Parameters
52  ----------
53  objectslist : list
54 
55  angle : rotation angle (in degrees)
56 
57  center : Base.Vector
58 
59  axis : Base.Vector
60  If axis is omitted, the rotation will be around the vertical Z axis.
61 
62  copy : bool
63  If copy is True, the actual objects are not moved, but copies
64  are created instead.
65 
66  Return
67  ----------
68  The objects (or their copies) are returned.
69  """
70  import Part
71  utils.type_check([(copy,bool)], "rotate")
72  if not isinstance(objectslist,list):
73  objectslist = [objectslist]
74 
75  objectslist.extend(groups.get_movable_children(objectslist))
76  newobjlist = []
77  newgroups = {}
78  objectslist = utils.filter_objects_for_modifiers(objectslist, copy)
79 
80  if copy:
81  doc = App.ActiveDocument
82  for obj in objectslist:
83  if obj.isDerivedFrom("App::DocumentObjectGroup") \
84  and obj.Name not in newgroups:
85  newgroups[obj.Name] = doc.addObject(obj.TypeId,
86  utils.get_real_name(obj.Name))
87 
88  for obj in objectslist:
89  newobj = None
90 
91  # real_center and real_axis are introduced to take into account
92  # the possibility that object is inside an App::Part
93  if hasattr(obj, "getGlobalPlacement"):
94  ci = obj.getGlobalPlacement().inverse().multVec(center)
95  real_center = obj.Placement.multVec(ci)
96  ai = obj.getGlobalPlacement().inverse().Rotation.multVec(axis)
97  real_axis = obj.Placement.Rotation.multVec(ai)
98  else:
99  real_center = center
100  real_axis = axis
101 
102  if obj.isDerivedFrom("App::Annotation"):
103  # TODO: this is very different from how move handle annotations
104  # maybe we can uniform the two methods
105  if copy:
106  newobj = make_copy.make_copy(obj)
107  else:
108  newobj = obj
109  if axis.normalize() == App.Vector(1, 0, 0):
110  newobj.ViewObject.RotationAxis = "X"
111  newobj.ViewObject.Rotation = angle
112  elif axis.normalize() == App.Vector(0, 1, 0):
113  newobj.ViewObject.RotationAxis = "Y"
114  newobj.ViewObject.Rotation = angle
115  elif axis.normalize() == App.Vector(0, -1, 0):
116  newobj.ViewObject.RotationAxis = "Y"
117  newobj.ViewObject.Rotation = -angle
118  elif axis.normalize() == App.Vector(0, 0, 1):
119  newobj.ViewObject.RotationAxis = "Z"
120  newobj.ViewObject.Rotation = angle
121  elif axis.normalize() == App.Vector(0, 0, -1):
122  newobj.ViewObject.RotationAxis = "Z"
123  newobj.ViewObject.Rotation = -angle
124 
125  elif obj.isDerivedFrom("App::DocumentObjectGroup"):
126  if copy:
127  newobj = newgroups[obj.Name]
128  else:
129  newobj = obj
130 
131  elif hasattr(obj, "Placement"):
132  # App.Console.PrintMessage("placement rotation\n")
133  if copy:
134  newobj = make_copy.make_copy(obj)
135  else:
136  newobj = obj
137  newobj.Placement.rotate(real_center, real_axis, angle, comp=True)
138 
139  elif hasattr(obj, "Shape"):
140  if copy:
141  newobj = make_copy.make_copy(obj)
142  else:
143  newobj = obj
144  shape = newobj.Shape.copy()
145  shape.rotate(real_center, real_axis, angle)
146  newobj.Shape = shape
147 
148  if newobj is not None:
149  newobjlist.append(newobj)
150  if copy:
151  for parent in obj.InList:
152  if parent.isDerivedFrom("App::DocumentObjectGroup") \
153  and (parent in objectslist):
154  newgroups[parent.Name].addObject(newobj)
155  if utils.get_type(parent) == "Layer":
156  parent.Proxy.addObject(parent ,newobj)
157 
158  if copy and params.get_param("selectBaseObjects"):
159  gui_utils.select(objectslist)
160  else:
161  gui_utils.select(newobjlist)
162 
163  if len(newobjlist) == 1:
164  return newobjlist[0]
165  return newobjlist

185 def rotate_vector_from_center(vector, angle, axis, center):
186  """
187  Needed for SubObjects modifiers.
188  Implemented by Dion Moult during 0.19 dev cycle (works only with Draft Wire).
189  """
190  rv = vector.sub(center)
191  rv = DraftVecUtils.rotate(rv, math.radians(angle), axis)
192  return center.add(rv)
