64 static QString prettyProductInfoWrapper()
65 {
66  auto productName = QSysInfo::prettyProductName();
67 #if QT_VERSION < QT_VERSION_CHECK(6, 5, 0)
68 #ifdef FC_OS_MACOSX
69  auto macosVersionFile =
70  QString::fromUtf8("/System/Library/CoreServices/.SystemVersionPlatform.plist");
71  auto fi = QFileInfo(macosVersionFile);
72  if (fi.exists() && fi.isReadable()) {
73  auto plistFile = QFile(macosVersionFile);
74  plistFile.open(QIODevice::ReadOnly);
75  while (!plistFile.atEnd()) {
76  auto line = plistFile.readLine();
77  if (line.contains("ProductUserVisibleVersion")) {
78  auto nextLine = plistFile.readLine();
79  if (nextLine.contains("<string>")) {
80  QRegularExpression re(QString::fromUtf8("\\s*<string>(.*)</string>"));
81  auto matches = re.match(QString::fromUtf8(nextLine));
82  if (matches.hasMatch()) {
83  productName = QString::fromUtf8("macOS ") + matches.captured(1);
84  break;
85  }
86  }
87  }
88  }
89  }
90 #endif
91 #endif
92 #ifdef FC_OS_WIN64
93  QSettings regKey {
94  QString::fromUtf8("HKEY_LOCAL_MACHINE\\SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion"),
95  QSettings::NativeFormat};
96  if (regKey.contains(QString::fromUtf8("CurrentBuildNumber"))) {
97  auto buildNumber = regKey.value(QString::fromUtf8("CurrentBuildNumber")).toInt();
98  if (buildNumber > 0) {
99  if (buildNumber < 9200) {
100  productName = QString::fromUtf8("Windows 7 build %1").arg(buildNumber);
101  }
102  else if (buildNumber < 10240) {
103  productName = QString::fromUtf8("Windows 8 build %1").arg(buildNumber);
104  }
105  else if (buildNumber < 22000) {
106  productName = QString::fromUtf8("Windows 10 build %1").arg(buildNumber);
107  }
108  else {
109  productName = QString::fromUtf8("Windows 11 build %1").arg(buildNumber);
110  }
111  }
112  }
113 #endif
114  return productName;
115 }

155 AboutDialog::AboutDialog(QWidget* parent)
156  : QDialog(parent)
157  , ui(new Ui_AboutApplication)
158 {
159  setModal(true);
160  ui->setupUi(this);
161  connect(ui->copyButton, &QPushButton::clicked, this, &AboutDialog::copyToClipboard);
162 
163  // remove the automatic help button in dialog title since we don't use it
164  setWindowFlag(Qt::WindowContextHelpButtonHint, false);
165 
166  layout()->setSizeConstraint(QLayout::SetFixedSize);
167  QRect rect = QApplication::primaryScreen()->availableGeometry();
168 
169  // See if we have a custom About screen image set
170  QPixmap image = aboutImage();
171 
172  // Fallback to the splashscreen image
173  if (image.isNull()) {
174  image = SplashScreen::splashImage();
175  }
176 
177  // Make sure the image is not too big
178  int denom = 2;
179  if (image.height() > rect.height() / denom || image.width() > rect.width() / denom) {
180  float scale = static_cast<float>(image.width()) / static_cast<float>(image.height());
181  int width = std::min(image.width(), rect.width() / denom);
182  int height = std::min(image.height(), rect.height() / denom);
183  height = std::min(height, static_cast<int>(width / scale));
184  width = static_cast<int>(scale * height);
185 
186  image = image.scaled(width, height);
187  }
188  ui->labelSplashPicture->setPixmap(image);
189  ui->tabWidget->setCurrentIndex(0); // always start on the About tab
190 
191  setupLabels();
192  showCredits();
193  showLicenseInformation();
194  showLibraryInformation();
195  showCollectionInformation();
196  showPrivacyPolicy();
197  showOrHideImage(rect);
198 }

209 QPixmap AboutDialog::aboutImage() const
210 {
211  // See if we have a custom About screen image set
212  QPixmap about_image;
213  QFileInfo fi(QString::fromLatin1("images:about_image.png"));
214  if (fi.isFile() && fi.exists()) {
215  about_image.load(fi.filePath(), "PNG");
216  }
217 
218  std::string about_path = App::Application::Config()["AboutImage"];
219  if (!about_path.empty() && about_image.isNull()) {
220  QString path = QString::fromStdString(about_path);
221  if (QDir(path).isRelative()) {
222  QString home = QString::fromStdString(App::Application::getHomePath());
223  path = QFileInfo(QDir(home), path).absoluteFilePath();
224  }
225  about_image.load(path);
226 
227  // Now try the icon paths
228  if (about_image.isNull()) {
229  about_image = Gui::BitmapFactory().pixmap(about_path.c_str());
230  }
231  }

254  .arg(fontSize));

258  std::map<std::string, std::string>& config = App::Application::Config();
259  std::map<std::string, std::string>::iterator it;
260  QString banner = QString::fromStdString(config["CopyrightInfo"]);
261  banner = banner.left(banner.indexOf(QLatin1Char('\n')));
262  QString major = QString::fromStdString(config["BuildVersionMajor"]);
263  QString minor = QString::fromStdString(config["BuildVersionMinor"]);
264  QString point = QString::fromStdString(config["BuildVersionPoint"]);
265  QString suffix = QString::fromStdString(config["BuildVersionSuffix"]);
266  QString build = QString::fromStdString(config["BuildRevision"]);
267  QString disda = QString::fromStdString(config["BuildRevisionDate"]);
268  QString mturl = QString::fromStdString(config["MaintainerUrl"]);
269 
270  // we use replace() to keep label formatting, so a label with text "<b>Unknown</b>"
271  // gets replaced to "<b>FreeCAD</b>", for example
272 
273  QString author = ui->labelAuthor->text();
274  author.replace(QString::fromLatin1("Unknown Application"), exeName);
275  author.replace(QString::fromLatin1("(c) Unknown Author"), banner);
276  ui->labelAuthor->setText(author);
277  ui->labelAuthor->setUrl(mturl);
278 
279  if (qApp->styleSheet().isEmpty()) {
280  ui->labelAuthor->setStyleSheet(QString::fromLatin1(
281  "Gui--UrlLabel {color: #0000FF;text-decoration: underline;font-weight: 600;}"));
282  }
283 
284  QString version = ui->labelBuildVersion->text();
285  version.replace(QString::fromLatin1("Unknown"),
286  QString::fromLatin1("%1.%2.%3%4").arg(major, minor, point, suffix));
287  ui->labelBuildVersion->setText(version);
288 
289  QString revision = ui->labelBuildRevision->text();
290  revision.replace(QString::fromLatin1("Unknown"), build);
291  ui->labelBuildRevision->setText(revision);
292 
293  QString date = ui->labelBuildDate->text();
294  date.replace(QString::fromLatin1("Unknown"), disda);
295  ui->labelBuildDate->setText(date);
296 
297  QString os = ui->labelBuildOS->text();
298  os.replace(QString::fromLatin1("Unknown"), prettyProductInfoWrapper());
299  ui->labelBuildOS->setText(os);
300 
301  QString architecture = ui->labelBuildRunArchitecture->text();
302  if (QSysInfo::buildCpuArchitecture() == QSysInfo::currentCpuArchitecture()) {
303  architecture.replace(QString::fromLatin1("Unknown"), QSysInfo::buildCpuArchitecture());
304  }
305  else {
306  architecture.replace(
307  QString::fromLatin1("Unknown"),
308  QString::fromLatin1("%1 (running on: %2)")
309  .arg(QSysInfo::buildCpuArchitecture(), QSysInfo::currentCpuArchitecture()));
310  }
311  ui->labelBuildRunArchitecture->setText(architecture);
312 
313  // branch name
314  it = config.find("BuildRevisionBranch");
315  if (it != config.end()) {
316  QString branch = ui->labelBuildBranch->text();
317  branch.replace(QString::fromLatin1("Unknown"), QString::fromStdString(it->second));
318  ui->labelBuildBranch->setText(branch);
319  }
320  else {
321  ui->labelBranch->hide();
322  ui->labelBuildBranch->hide();
323  }
324 
325  // hash id
326  it = config.find("BuildRevisionHash");
327  if (it != config.end()) {
328  QString hash = ui->labelBuildHash->text();
329  hash.replace(
330  QString::fromLatin1("Unknown"),
331  QString::fromStdString(it->second).left(7)); // Use the 7-char abbreviated hash
332  ui->labelBuildHash->setText(hash);
333  if (auto url_itr = config.find("BuildRepositoryURL"); url_itr != config.end()) {
334  auto url = QString::fromStdString(url_itr->second);
335 
336  if (int space = url.indexOf(QChar::fromLatin1(' ')); space != -1) {
337  url = url.left(space); // Strip off the branch information to get just the repo
338  }
339 
340  if (url == QString::fromUtf8("Unknown")) {
341  url = QString::fromUtf8("https://github.com/FreeCAD/FreeCAD"); // Just take a guess
342  }
343 
344  // This may only create valid URLs for Github, but some other hosts use the same format
345  // so give it a shot...
346  auto https = url.replace(QString::fromUtf8("git://"), QString::fromUtf8("https://"));
347  https.replace(QString::fromUtf8(".git"), QString::fromUtf8(""));
348  ui->labelBuildHash->setUrl(https + QString::fromUtf8("/commit/")
349  + QString::fromStdString(it->second));
350  }
351  }
352  else {
353  ui->labelHash->hide();
354  ui->labelBuildHash->hide();
355  }

537 void AboutDialog::addModuleInfo(QTextStream& str, const QString& modPath, bool& firstMod)
538 {
539  QFileInfo mod(modPath);
540  if (mod.isHidden()) { // Ignore hidden directories
541  return;
542  }
543  if (firstMod) {
544  firstMod = false;
545  str << "Installed mods: \n";
546  }
547  str << " * " << (mod.isDir() ? QDir(modPath).dirName() : mod.fileName());
548  try {
549  auto metadataFile =
550  boost::filesystem::path(mod.absoluteFilePath().toStdString()) / "package.xml";
551  if (boost::filesystem::exists(metadataFile)) {
552  App::Metadata metadata(metadataFile);
553  if (metadata.version() != App::Meta::Version()) {
554  str << QLatin1String(" ") + QString::fromStdString(metadata.version().str());
555  }
556  }
557  }
558  catch (const Base::Exception& e) {
559  auto what = QString::fromUtf8(e.what()).trimmed().replace(QChar::fromLatin1('\n'),
560  QChar::fromLatin1(' '));
561  str << " (Malformed metadata: " << what << ")";
562  }
563  QFileInfo disablingFile(mod.absoluteFilePath(), QString::fromLatin1("ADDON_DISABLED"));
564  if (disablingFile.exists()) {
565  str << " (Disabled)";
566  }
567 
568  str << "\n";
569 }

571 void AboutDialog::copyToClipboard()
572 {
573  QString data;
574  QTextStream str(&data);
575  std::map<std::string, std::string>& config = App::Application::Config();
576  std::map<std::string, std::string>::iterator it;
577  QString exe = QString::fromStdString(App::Application::getExecutableName());
578 
579  QString major = QString::fromStdString(config["BuildVersionMajor"]);
580  QString minor = QString::fromStdString(config["BuildVersionMinor"]);
581  QString point = QString::fromStdString(config["BuildVersionPoint"]);
582  QString suffix = QString::fromStdString(config["BuildVersionSuffix"]);
583  QString build = QString::fromStdString(config["BuildRevision"]);
584 
585  QString deskEnv =
586  QProcessEnvironment::systemEnvironment().value(QStringLiteral("XDG_CURRENT_DESKTOP"),
587  QString());
588  QString deskSess =
589  QProcessEnvironment::systemEnvironment().value(QStringLiteral("DESKTOP_SESSION"),
590  QString());
591  QStringList deskInfoList;
592  QString deskInfo;
593 
594  if (!deskEnv.isEmpty()) {
595  deskInfoList.append(deskEnv);
596  }
597  if (!deskSess.isEmpty()) {
598  deskInfoList.append(deskSess);
599  }
600  if (qGuiApp->platformName() != QLatin1String("windows")
601  && qGuiApp->platformName() != QLatin1String("cocoa")) {
602  deskInfoList.append(qGuiApp->platformName());
603  }
604  if (!deskInfoList.isEmpty()) {
605  deskInfo = QLatin1String(" (") + deskInfoList.join(QLatin1String("/")) + QLatin1String(")");
606  }
607 
608  str << "OS: " << prettyProductInfoWrapper() << deskInfo << '\n';
609  if (QSysInfo::buildCpuArchitecture() == QSysInfo::currentCpuArchitecture()) {
610  str << "Architecture: " << QSysInfo::buildCpuArchitecture() << "\n";
611  }
612  else {
613  str << "Architecture: " << QSysInfo::buildCpuArchitecture()
614  << "(running on: " << QSysInfo::currentCpuArchitecture() << ")\n";
615  }
616  str << "Version: " << major << "." << minor << "." << point << suffix << "." << build;
617 #ifdef FC_CONDA
618  str << " Conda";
619 #endif
620 #ifdef FC_FLATPAK
621  str << " Flatpak";
622 #endif
623  char* appimage = getenv("APPIMAGE");
624  if (appimage) {
625  str << " AppImage";
626  }
627  char* snap = getenv("SNAP_REVISION");
628  if (snap) {
629  str << " Snap " << snap;
630  }
631  str << '\n';
632 
633 #if defined(_DEBUG) || defined(DEBUG)
634  str << "Build type: Debug\n";
635 #elif defined(NDEBUG)
636  str << "Build type: Release\n";
637 #elif defined(CMAKE_BUILD_TYPE)
638  str << "Build type: " << CMAKE_BUILD_TYPE << '\n';
639 #else
640  str << "Build type: Unknown\n";
641 #endif
642  it = config.find("BuildRevisionBranch");
643  if (it != config.end()) {
644  str << "Branch: " << QString::fromStdString(it->second) << '\n';
645  }
