67 void Workbench::activated()
68 {
69  if (!initialized) {
70  QList<QToolBar*> bars =
71  Gui::getMainWindow()->findChildren<QToolBar*>(QString::fromLatin1("Spreadsheet"));
72 
73  if (bars.size() == 1) {
74  QToolBar* bar = bars[0];
75  QtColorPicker* foregroundColor;
76  QtColorPicker* backgroundColor;
77  QPalette palette = Gui::getMainWindow()->palette();
78 
79  QList<QtColorPicker*> fgList = Gui::getMainWindow()->findChildren<QtColorPicker*>(
80  QString::fromLatin1("Spreadsheet_ForegroundColor"));
81  if (!fgList.empty()) {
82  foregroundColor = fgList[0];
83  }
84  else {
85  foregroundColor = new QtColorPicker(bar);
86  foregroundColor->setObjectName(QString::fromLatin1("Spreadsheet_ForegroundColor"));
87  foregroundColor->setStandardColors();
88  foregroundColor->setCurrentColor(palette.color(QPalette::WindowText));
89  QObject::connect(foregroundColor,
90  &QtColorPicker::colorSet,
91  workbenchHelper.get(),
92  &WorkbenchHelper::setForegroundColor);
93  }
94  foregroundColor->setToolTip(QObject::tr("Set cell(s) text color"));
95  foregroundColor->setWhatsThis(QObject::tr("Sets the Spreadsheet cell(s) text color"));
96  foregroundColor->setStatusTip(QObject::tr("Set cell(s) text color"));
97  bar->addWidget(foregroundColor);
98 
99  QList<QtColorPicker*> bgList = Gui::getMainWindow()->findChildren<QtColorPicker*>(
100  QString::fromLatin1("Spreadsheet_BackgroundColor"));
101  if (!bgList.empty()) {
102  backgroundColor = bgList[0];
103  }
104  else {
105  backgroundColor = new QtColorPicker(bar);
106  backgroundColor->setObjectName(QString::fromLatin1("Spreadsheet_BackgroundColor"));
107  backgroundColor->setStandardColors();
108  backgroundColor->setCurrentColor(palette.color(QPalette::Base));
109  QObject::connect(backgroundColor,
110  &QtColorPicker::colorSet,
111  workbenchHelper.get(),
112  &WorkbenchHelper::setBackgroundColor);
113  }
114  backgroundColor->setToolTip(QObject::tr("Set cell(s) background color"));
115  backgroundColor->setWhatsThis(
116  QObject::tr("Sets the Spreadsheet cell(s) background color"));
117  backgroundColor->setStatusTip(QObject::tr("Set cell(s) background color"));
118  bar->addWidget(backgroundColor);
119 
120  initialized = false;
121  }
122  }

125 void WorkbenchHelper::setForegroundColor(const QColor& color)
126 {
127  Gui::Document* doc = Gui::Application::Instance->activeDocument();
128 
129  if (doc) {
130  Gui::MDIView* activeWindow = Gui::getMainWindow()->activeWindow();
131  SpreadsheetGui::SheetView* sheetView =
132  freecad_dynamic_cast<SpreadsheetGui::SheetView>(activeWindow);
133 
134  if (sheetView) {
135  Sheet* sheet = sheetView->getSheet();
136  std::vector<Range> ranges = sheetView->selectedRanges();
137 
138  // Execute mergeCells commands
139  if (!ranges.empty()) {
140  std::vector<Range>::const_iterator i = ranges.begin();
141 
142  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Set text color"));
143  for (; i != ranges.end(); ++i) {
144  Gui::Command::doCommand(Gui::Command::Doc,
145  "App.ActiveDocument.%s.setForeground('%s', (%f,%f,%f))",
146  sheet->getNameInDocument(),
147  i->rangeString().c_str(),
148  color.redF(),
149  color.greenF(),
150  color.blueF());
151  }
152  Gui::Command::commitCommand();
153  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
154  }
155  }
156  }
