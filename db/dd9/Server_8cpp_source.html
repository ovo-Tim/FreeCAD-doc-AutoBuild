198 std::string AppServer::runPython(const QByteArray& msg)
199 {
200  std::string str;
201 
202  try {
203  Firewall* fw = Firewall::getInstance();
204  if (!fw || fw->filter(msg)) {
205  str = Base::Interpreter().runString(msg);
206  }
207  else {
208  str = "Command blocked";
209  }
210  }
211  catch (Base::PyException& e) {
212  str = e.what();
213  str += "\n\n";
214  str += e.getStackTrace();
215  }
216  catch (Base::SystemExitException&) {
217  throw;
218  }
219  catch (Base::Exception& e) {
220  str = e.what();
221  }
222  catch (std::exception& e) {
223  str = e.what();
224  }
225  catch (...) {
226  str = "Unknown exception thrown";
227  }
228 
229  return str;
230 }
