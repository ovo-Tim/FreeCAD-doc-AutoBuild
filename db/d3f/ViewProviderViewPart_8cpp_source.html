170 void ViewProviderViewPart::onChanged(const App::Property* prop)
171 {
172  if (auto part = getViewPart(); part && part->isDerivedFrom(TechDraw::DrawViewDetail::getClassTypeId()) &&
173  prop == &(HighlightAdjust)) {
174  auto detail = static_cast<DrawViewDetail*>(getViewPart());
175  auto baseDvp = dynamic_cast<DrawViewPart*>(detail->BaseView.getValue());
176  if (baseDvp) {
177  baseDvp->requestPaint();
178  }
179  return;
180  }
181 
182  if (prop == &(LineWidth) ||
183  prop == &(HiddenWidth) ||
184  prop == &(IsoWidth) ||
185  prop == &(ExtraWidth) ||
186  prop == &(HighlightAdjust) ||
187  prop == &(ArcCenterMarks) ||
188  prop == &(CenterScale) ||
189  prop == &(ShowSectionLine) ||
190  prop == &(SectionLineStyle) ||
191  prop == &(SectionLineColor) ||
192  prop == &(SectionLineMarks) ||
193  prop == &(IncludeCutLine) ||
194  prop == &(HighlightLineStyle) ||
195  prop == &(HighlightLineColor) ||
196  prop == &(HorizCenterLine) ||
197  prop == &(VertCenterLine) ||
198  prop == &(FaceColor) ||
199  prop == &(FaceTransparency) ||
200  prop == &(BreakLineType) ||
201  prop == &(BreakLineStyle) ) {
202  // redraw QGIVP
203  QGIView* qgiv = getQView();
204  if (qgiv) {
205  qgiv->updateView(true);
206  }
207  }
208 
209  ViewProviderDrawingView::onChanged(prop);

227 std::vector<App::DocumentObject*> ViewProviderViewPart::claimChildren() const
228 {
229  // Collect any child Document Objects and put them in the right place in the Feature tree
230  // valid children of a ViewPart are:
231  // - Dimensions
232  // - Leaders
233  // - Hatches
234  // - GeomHatches
235  // - any drawing views declaring this view as their parent
236  std::vector<App::DocumentObject*> temp;
237  const std::vector<App::DocumentObject *> &views = getViewPart()->getInList();
238  try {
239  for(std::vector<App::DocumentObject *>::const_iterator it = views.begin(); it != views.end(); ++it) {
240  auto view = dynamic_cast<TechDraw::DrawView *>(*it);
241  if (view && view->claimParent() == getViewPart()) {
242  temp.push_back(view);
243  continue;
244  }
245 
246  if((*it)->isDerivedFrom<TechDraw::DrawViewDimension>()) {
247  //TODO: make a list, then prune it. should be faster?
248  bool skip = false;
249  std::string dimName = (*it)->getNameInDocument();
250  for (auto& t: temp) { //only add dim once even if it references 2 geometries
251  std::string tName = t->getNameInDocument();
252  if (dimName == tName) {
253  skip = true;
254  break;
255  }
256  }
257  if (!skip) {
258  temp.push_back(*it);
259  }
260  } else if ((*it)->isDerivedFrom<TechDraw::DrawHatch>()) {
261  temp.push_back((*it));
262  } else if ((*it)->isDerivedFrom<TechDraw::DrawGeomHatch>()) {
263  temp.push_back((*it));
264  } else if ((*it)->isDerivedFrom<TechDraw::DrawViewBalloon>()) {
265  temp.push_back((*it));
266  } else if ((*it)->isDerivedFrom<TechDraw::DrawLeaderLine>()) {
267  temp.push_back((*it));
268  }
269  }
270  return temp;
271  } catch (...) {
272  return std::vector<App::DocumentObject*>();
273  }

325 void ViewProviderViewPart::handleChangedPropertyType(Base::XMLReader &reader, const char *TypeName, App::Property *prop)
326 // transforms properties that had been changed
327 {
328  // property LineWidth had the App::PropertyFloat and was changed to App::PropertyLength
329  if (prop == &LineWidth && strcmp(TypeName, "App::PropertyFloat") == 0) {
330  App::PropertyFloat LineWidthProperty;
331  // restore the PropertyFloat to be able to set its value
332  LineWidthProperty.Restore(reader);
333  LineWidth.setValue(LineWidthProperty.getValue());
334  }
335  // property HiddenWidth had the App::PropertyFloat and was changed to App::PropertyLength
336  else if (prop == &HiddenWidth && strcmp(TypeName, "App::PropertyFloat") == 0) {
337  App::PropertyFloat HiddenWidthProperty;
338  HiddenWidthProperty.Restore(reader);
339  HiddenWidth.setValue(HiddenWidthProperty.getValue());
340  }
341  // property IsoWidth had the App::PropertyFloat and was changed to App::PropertyLength
342  else if (prop == &IsoWidth && strcmp(TypeName, "App::PropertyFloat") == 0) {
343  App::PropertyFloat IsoWidthProperty;
344  IsoWidthProperty.Restore(reader);
345  IsoWidth.setValue(IsoWidthProperty.getValue());
346  }
347  // property ExtraWidth had the App::PropertyFloat and was changed to App::PropertyLength
348  else if (prop == &ExtraWidth && strcmp(TypeName, "App::PropertyFloat") == 0) {
349  App::PropertyFloat ExtraWidthProperty;
350  ExtraWidthProperty.Restore(reader);
351  ExtraWidth.setValue(ExtraWidthProperty.getValue());
352  }
353  else {
354  ViewProviderDrawingView::handleChangedPropertyType(reader, TypeName, prop);

358 bool ViewProviderViewPart::onDelete(const std::vector<std::string> & subNames)
359 {
360  // Base::Console().Message("VPVP::onDelete(%d subNames)\n", subNames.size());
361  // we cannot delete if the view has a section or detail view
362  (void) subNames;
363  QString bodyMessage;
364  QTextStream bodyMessageStream(&bodyMessage);
365 
366  // get child views
367  auto viewSection = getViewObject()->getSectionRefs();
368  auto viewDetail = getViewObject()->getDetailRefs();
369 
370  if (!viewSection.empty() || !viewDetail.empty()) {
371  bodyMessageStream << qApp->translate("Std_Delete",
372  "You cannot delete this view because it has one or more dependent views that would become broken.");
373  QMessageBox::warning(Gui::getMainWindow(),
374  qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
375  QMessageBox::Ok);
376  return false;
377  }
378  return true;
379 }

406 // it can happen that Dimensions/Balloons/etc can lose their parent item if the
407 // the parent is deleted, then undo is invoked. The linkages on the App side are
408 // handled by the undo mechanism, but the QGraphicsScene parentage is not reset.
409 // TODO: does this need to be implemented for Leaderlines and ???? others?
410 void ViewProviderViewPart::fixSceneDependencies()
411 {
412  auto page = getViewProviderPage();
413  if (!page) {
414  return;
415  }
416 
417  auto scene = page->getQGSPage();
418  auto partQView = getQView();
419 
420  auto dimensions = getViewPart()->getDimensions();
421  for (auto& dim : dimensions) {
422  auto dimQView = dynamic_cast<QGIViewDimension *>(scene->findQViewForDocObj(dim));
423  if (dimQView && dimQView->parentItem() != partQView) {
424  // need to add the dim QView to this QGIViewPart
425  scene->addDimToParent(dimQView, partQView);
426  }
427  }
428 
429  auto balloons = getViewPart()->getBalloons();
430  for (auto& bal : balloons) {
431  auto balQView = dynamic_cast<QGIViewBalloon*>(scene->findQViewForDocObj(bal));
432  if (balQView && balQView->parentItem() != partQView) {
433  // need to add the balloon QView to this QGIViewPart
434  scene->addBalloonToParent(balQView, partQView);
435  }
436  }
