104 //static support function: split comma separated string of values into vector of numbers
105 std::vector<double> LineGroup::split(std::string line)
106 {
107  std::vector<double> result;
108  std::stringstream lineStream(line);
109  std::string cell;
110  bool nameCell = true;
111 
112  while(std::getline(lineStream, cell, ','))
113  {
114  if (nameCell) {
115  nameCell = false;
116  continue;
117  }
118  try {
119  result.push_back(std::stod(cell));
120  }
121  catch (const std::invalid_argument& ia) {
122  Base::Console().Warning("Invalid number in cell: %s (%s) \n", cell.c_str(), ia.what());
123  result.push_back(0.0);
124  }
125  }
126  return result;
127 }

129 //static support function: find group definition in file
130 std::string LineGroup::getRecordFromFile(std::string parmFile, int groupNumber)
131 {
132  std::string record;
133  Base::FileInfo fi(parmFile);
134  Base::ifstream inFile(fi, std::ifstream::in);
135  if(!inFile.is_open()) {
136  Base::Console().Message( "Cannot open LineGroup file: %s\n", parmFile.c_str());
137  return record;
138  }
139  // parse file to get the groupNumber'th line
140  int counter = 0; // the combobox enums begin with 0
141  while ( inFile.good() ){
142  std::string line;
143  std::getline(inFile, line);
144  std::string nameTag = line.substr(0, 1);
145  if (nameTag == "*") { // we found a definition line
146  if (counter == groupNumber) {
147  record = line;
148  return record;
149  }
150  ++counter;
151  }
152  } //endwhile
153  // nothing was found
154  Base::Console().Error("LineGroup: the LineGroup file has only %s entries but entry number %s is set\n"
155  , std::to_string(counter).c_str()
156  , std::to_string(groupNumber).c_str());
157  return std::string(); // return an empty string
158 }

160 //static LineGroup maker
161 LineGroup* LineGroup::lineGroupFactory(int groupNumber)
162 {
163  LineGroup* lg = new LineGroup();
164 
165  std::string lgFileName = Preferences::lineGroupFile();
166 
167  std::string lgRecord = LineGroup::getRecordFromFile(lgFileName, groupNumber);
168 
169  std::vector<double> values = LineGroup::split(lgRecord);
170  if (values.size() < 4) {
171  Base::Console().Error( "LineGroup::invalid entry in %s\n", lgFileName.c_str() );
172  } else {
173  lg->setWeight("Thin", values[0]);
174  lg->setWeight("Graphic", values[1]);
175  lg->setWeight("Thick", values[2]);
176  lg->setWeight("Extra", values[3]);
177  }
178  return lg;
179 }

201 std::string LineGroup::getGroupNamesFromFile(std::string FileName)
202 {
203  std::string record;
204  Base::FileInfo fi(FileName);
205  Base::ifstream inFile(fi, std::ifstream::in);
206  if (!inFile.is_open()) {
207  Base::Console().Message("Cannot open LineGroup file: %s\n", FileName.c_str());
208  return record;
209  }
210  // parse the file
211  while (inFile.good()) {
212  std::string line;
213  std::getline(inFile, line);
214  std::string nameTag = line.substr(0, 1);
215  std::string found;
216  std::size_t commaPos;
217  if (nameTag == "*") {
218  commaPos = line.find(',', 1);
219  if (commaPos != std::string::npos) {
220  found = line.substr(1, commaPos - 1);
221  record = record + found + ',';
222  }
223  }
224  } //endwhile
225  if (record.empty()) {
226  Base::Console().Message("LineGroup error: no group found in file %s\n", FileName.c_str());
227  }
228  return record;
