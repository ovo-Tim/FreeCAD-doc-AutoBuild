84 {
85  char* context = nullptr;
86  char* source = nullptr;
87  char* disambiguation = nullptr;
88  int num = -1;
89  if (!PyArg_ParseTuple(args.ptr(), "ss|zi", &context, &source, &disambiguation, &num)) {

91  }
92 
93  QString str = QCoreApplication::translate(context, source, disambiguation, num);
94  return Py::asObject(PyUnicode_FromString(str.toUtf8()));

127 Py::Object Translate::installTranslator(const Py::Tuple& args)
128 {
129  char* Name = nullptr;
130  if (!PyArg_ParseTuple(args.ptr(), "et", "utf-8", &Name)) {
131  throw Py::Exception();
132  }
133  QString filename = QString::fromUtf8(Name);
134  PyMem_Free(Name);
135 
136  bool ok = false;
137  QFileInfo fi(filename);
138  std::shared_ptr<QTranslator> translator(std::make_shared<QTranslator>(nullptr));
139  translator->setObjectName(fi.fileName());
140  if (translator->load(filename)) {
141  qApp->installTranslator(translator.get());
142  translators.push_back(translator);
143  ok = true;
144  }
145 
146  return Py::Boolean(ok); // NOLINT
147 }
