385 PyObject* ComplexGeoDataPy::setElementName(PyObject* args, PyObject* kwds)
386 {
387  const char* element;
388  const char* name = 0;
389  const char* postfix = 0;
390  int tag = 0;
391  PyObject* pySid = Py_None;
392  PyObject* overwrite = Py_False;
393 
394  const std::array<const char*, 7> kwlist =
395  {"element", "name", "postfix", "overwrite", "sid", "tag", nullptr};
396  if (!Wrapped_ParseTupleAndKeywords(args,
397  kwds,
398  "s|sssOOi",
399  kwlist,
400  &element,
401  &name,
402  &postfix,
403  &overwrite,
404  &pySid,
405  &tag)) {
406  return NULL;
407  }
408  ElementIDRefs sids;
409  if (pySid != Py_None) {
410  if (PyObject_TypeCheck(pySid, &App::StringIDPy::Type)) {
411  sids.push_back(static_cast<App::StringIDPy*>(pySid)->getStringIDPtr());
412  }
413  else if (PySequence_Check(pySid)) {
414  Py::Sequence seq(pySid);
415  for (auto it = seq.begin(); it != seq.end(); ++it) {
416  auto ptr = (*it).ptr();
417  if (PyObject_TypeCheck(ptr, &App::StringIDPy::Type)) {
418  sids.push_back(static_cast<App::StringIDPy*>(ptr)->getStringIDPtr());
419  }
420  else {
421  throw Py::TypeError("expect StringID in sid sequence");
422  }
423  }
424  }
425  else {
426  throw Py::TypeError("expect sid to contain either StringID or sequence of StringID");
427  }
428  }
429  PY_TRY
430  {
431  Data::IndexedName index(element, getComplexGeoDataPtr()->getElementTypes());
432  Data::MappedName mapped = Data::MappedName::fromRawData(name);
433  std::ostringstream ss;
434  ElementMapPtr map = getComplexGeoDataPtr()->resetElementMap();
435  map->encodeElementName(getComplexGeoDataPtr()->elementType(index),
436  mapped,
437  ss,
438  &sids,
439  tag,
440  postfix,
441  tag);
442  Data::MappedName res =
443  map->setElementName(index, mapped, tag, &sids, PyObject_IsTrue(overwrite));
444  return Py::new_reference_to(Py::String(res.toString(0)));
445  }
446  PY_CATCH
447 }
