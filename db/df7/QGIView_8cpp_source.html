161 QVariant QGIView::itemChange(GraphicsItemChange change, const QVariant &value)
162 {
163  // Base::Console().Message("QGIV::itemChange(%d)\n", change);
164  if(change == ItemPositionChange && scene()) {
165  QPointF newPos = value.toPointF(); //position within parent!
166 
167  TechDraw::DrawView *viewObj = getViewObject();
168  auto* dpgi = dynamic_cast<TechDraw::DrawProjGroupItem*>(viewObj);
169  if (dpgi && dpgi->getPGroup()) {
170  // restrict movements of secondary views.
171  if(alignHash.size() == 1) { //if aligned.
172  QGraphicsItem* item = alignHash.begin().value();
173  QString alignMode = alignHash.begin().key();
174  if(alignMode == QString::fromLatin1("Vertical")) {
175  newPos.setX(item->pos().x());
176  }
177  else if(alignMode == QString::fromLatin1("Horizontal")) {
178  newPos.setY(item->pos().y());
179  }
180  }
181  }
182  else {
183  // For general views we check if we need to snap to a position
184  if (!(QApplication::keyboardModifiers() & Qt::AltModifier)) {
185  snapPosition(newPos);
186  }
187  }
188 
189  // tell the feature that we have moved
190  Gui::ViewProvider *vp = getViewProvider(viewObj);
191  if (vp && !vp->isRestoring()) {
192  snapping = true; // avoid triggering updateView by the VP updateData
193  viewObj->setPosition(Rez::appX(newPos.x()), Rez::appX(-newPos.y()));
194  snapping = false;
195  }
196 
197  return newPos;
198  }
199 
200  if (change == ItemSelectedHasChanged && scene()) {
201  if(isSelected()) {
202  m_colCurrent = getSelectColor();
203  } else {
204  m_colCurrent = PreferencesGui::getAccessibleQColor(PreferencesGui::normalQColor());
205  }
206  drawBorder();
207  }
208 
209  return QGraphicsItemGroup::itemChange(change, value);
210 }

215 void QGIView::snapPosition(QPointF& newPosition)
216 {
217  if (!Preferences::SnapViews()) {
218  return;
219  }
220 
221  auto feature = getViewObject();
222  if (!feature) {
223  return;
224  }
225 
226  auto dvp = dynamic_cast<DrawViewPart*>(feature);
227  if (dvp &&
228  !dvp->hasGeometry()) {
229  // too early. wait for updates to finish.
230  return;
231  }
232 
233  auto vpPage = getViewProviderPage(feature);
234 
235  QGSPage* scenePage = vpPage->getQGSPage();
236  if (!scenePage) {
237  return;
238  }
239 
240  auto* sectionView = dynamic_cast<TechDraw::DrawViewSection*>(feature);
241  if (sectionView) {
242  snapSectionView(sectionView, newPosition);
243  return;
244  }
245 
246  // For general views we check if the view is close to aligned vertically or horizontally to another view.
247 
248  // if we are not a section view, then we could be in a projection group and
249  // need to get the correct scene position.
250  auto newScenePos = newPosition;
251  if (parentItem()) {
252  newScenePos = parentItem()->mapToScene(newPosition);
253  }
254 
255  // First get a list of the views of the page.
256  qreal snapPercent = Preferences::SnapLimitFactor();
257  std::vector<QGIView*> views = scenePage->getViews();
258  for (auto* view : views) {
259  if (view == this) {
260  continue;
261  }
262  auto viewFeature = view->getViewObject();
263  auto viewDvp = dynamic_cast<DrawViewPart*>(viewFeature);
264 
265  auto viewScenePos = view->scenePos();
266  if (viewDvp &&
267  DrawView::isProjGroupItem(viewDvp)) {
268  viewScenePos = DU::toQPointF(projItemPagePos(viewDvp));
269  viewScenePos = DU::invertY(Rez::guiX(viewScenePos));
270  }
271 
272  auto xwindow = view->boundingRect().width() * snapPercent;
273  auto ywindow = view->boundingRect().height() * snapPercent;
274 
275  auto xerror = fabs(newScenePos.x() - viewScenePos.x());
276  auto yerror = fabs(newScenePos.y() - viewScenePos.y());
277 
278  // if the smaller of vertical and horizontal errors is within the acceptable
279  // window, snap to position.
280  if (xerror <= yerror &&
281  xerror <= xwindow) {
282  newScenePos.setX(viewScenePos.x());
283  if (parentItem()) {
284  newScenePos = parentItem()->mapFromScene(newScenePos);
285  }
286  newPosition = newScenePos;
287  return;
288  }
289 
290  if (yerror < xerror &&
291  yerror <= ywindow) {
292  newScenePos.setY(viewScenePos.y());
293  if (parentItem()) {
294  newScenePos = parentItem()->mapFromScene(newScenePos);
295  }
296  newPosition = newScenePos;
297  return;
298  }
299  }
300 }

306 // Note: positions are in Qt inverted Y coordinates. They need to be converted before
307 // doing math on them, then converted back on return.
308 // Note: section views are never inside a ProjectionGroup, so their position is
309 // always in scene coordinates.
310 void QGIView::snapSectionView(const TechDraw::DrawViewSection* sectionView,
311  QPointF& newPosition)
312 {
313  auto* baseView = sectionView->getBaseDVP();
314  if (!baseView) {
315  return;
316  }
317  auto* vpdv = dynamic_cast<ViewProviderDrawingView*>(getViewProvider(baseView));
318  if (!vpdv) {
319  return;
320  }
321  auto* qgiv(dynamic_cast<QGIView*>(vpdv->getQView()));
322  if (!qgiv) {
323  return;
324  }
325 
326  Base::Vector3d arrowDirection = sectionView->SectionNormal.getValue() * -1;
327  auto arrowDirectionOnBase = baseView->projectPoint(arrowDirection, false);
328  if (arrowDirectionOnBase.Length() < Precision::Confusion()) {
329  return;
330  }
331  arrowDirectionOnBase.Normalize();
332  double baseSize = Rez::guiX(baseView->getSizeAlongVector(arrowDirectionOnBase));
333  double snapDist = baseSize * getScale() * Preferences::SnapLimitFactor();
334 
335  // find the scene position of the SO on the base view
336  auto baseX = baseView->X.getValue();
337  auto baseY = baseView->Y.getValue();
338  Base::Vector3d baseScenePos{baseX, baseY, 0}; // paper space position
339  if (DrawView::isProjGroupItem(baseView)) {
340  baseScenePos = projItemPagePos(baseView);
341  }
342  auto sectionOrg3d = sectionView->SectionOrigin.getValue();
343  auto shapeCenter3d = baseView->getCurrentCentroid();
344  auto baseShapeCenter = baseView->projectPoint(shapeCenter3d, false);
345  auto baseSectionOrg = baseView->projectPoint(sectionOrg3d, false);
346  auto baseSOOffset = (baseSectionOrg - baseShapeCenter) * baseView->getScale();
347  auto baseSOScenePos = baseScenePos + baseSOOffset;
348 
349  // find the SO offset from origin on the rotated & scaled sectionView
350  auto sectionCutCenter = sectionView->projectPoint(sectionView->getCutCentroid(), false);
351  auto sectionSectionOrg = sectionView->projectPoint(sectionOrg3d, false);
352  auto sectionSOOffset = (sectionSectionOrg - sectionCutCenter) * sectionView->getScale();
353  auto sectionRotationDeg = sectionView->Rotation.getValue();
354  sectionSOOffset.RotateZ(Base::toRadians(sectionRotationDeg));
355 
356  // from here on, we work with scene units (1/10 mm)
357  sectionSOOffset = Rez::guiX(sectionSOOffset);
358  baseSOScenePos = Rez::guiX(baseSOScenePos);
359 
360  // check our alignment
361  auto newSOPosition = DU::invertY(DU::toVector3d(newPosition)) + sectionSOOffset;
362 
363  Base::Vector3d actualAlignmentVector = newSOPosition - baseSOScenePos;
364  actualAlignmentVector.Normalize();
365 
366  // if we are not on the correct side of the section line, we should not try to snap
367  auto dot = arrowDirectionOnBase.Dot(actualAlignmentVector);
368  if (dot <= 0) {
369  return;
370  }
371 
372  auto pointOnArrowLine = newSOPosition.Perpendicular(baseSOScenePos, arrowDirectionOnBase);
373  auto errorVector = pointOnArrowLine - newSOPosition;
374  if (errorVector.Length() < snapDist) {
375  // get the position point corresponding to our SO alignment
376  auto netPosition = pointOnArrowLine - sectionSOOffset;
377  netPosition = DU::invertY(netPosition);
378  newPosition = DU::toQPointF(netPosition);
379  }
380 
381  return;
382 }

384 Base::Vector3d QGIView::projItemPagePos(DrawViewPart* item)
385 {
386  if (!DrawView::isProjGroupItem(item)) {
387  return Base::Vector3d(0, 0, 0);
388  }
389  auto dpgi = static_cast<DrawProjGroupItem*>(item);
390  auto group = dpgi->getPGroup();
391 
392  auto itemX = dpgi->X.getValue();
393  auto itemY = dpgi->Y.getValue();
394  Base::Vector3d itemOffsetPos{itemX, itemY, 0}; // relative to group
395  auto groupX = group->X.getValue();
396  auto groupY = group->Y.getValue();
397  Base::Vector3d groupPagePos{groupX, groupY, 0}; // relative to page
398  return groupPagePos + itemOffsetPos;
399 }

528 void QGIView::rotateView()
529 {
530  //NOTE: QPainterPaths have to be rotated individually. This transform handles Rotation for everything else.
531  //Scale is handled in GeometryObject for DVP & descendents
532  //Objects not descended from DVP must setScale for themselves

534  double rot = getViewObject()->Rotation.getValue();
535  QPointF centre = boundingRect().center();
536  setTransform(QTransform().translate(centre.x(), centre.y()).rotate(-rot).translate(-centre.x(), -centre.y()));
