57 def getPickStyleNode(viewprovider, make_if_missing=True):
58  from pivy import coin
59 
60  sa = coin.SoSearchAction()
61  sa.setType(coin.SoPickStyle.getClassTypeId())
62  sa.traverse(viewprovider.RootNode)
63  if sa.isFound() and sa.getPath().getLength() == 1:
64  return sa.getPath().getTail()
65  else:
66  if not make_if_missing:
67  return None
68  pick_style = coin.SoPickStyle()
69  pick_style.style.setValue(coin.SoPickStyle.SHAPE)
70  viewprovider.RootNode.insertChild(pick_style, 0)
