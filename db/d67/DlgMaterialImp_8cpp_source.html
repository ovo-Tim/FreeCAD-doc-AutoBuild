197 void DlgMaterialImp::setMaterial(const std::vector<App::DocumentObject*>& objects)
198 {
199  for (auto it : objects) {
200  if (auto prop = dynamic_cast<Materials::PropertyMaterial*>(it->getPropertyByName("ShapeMaterial"))) {
201  try {
202  const auto& material = prop->getValue();
203  d->ui.widgetMaterial->setMaterial(material.getUUID());
204  return;
205  }
206  catch (const Materials::MaterialNotFound&) {
207  }
208  }
209  }
210  d->ui.widgetMaterial->setMaterial(Materials::MaterialManager::defaultMaterialUUID());
211 }
