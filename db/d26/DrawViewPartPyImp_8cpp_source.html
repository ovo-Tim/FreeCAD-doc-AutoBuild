233 PyObject* DrawViewPartPy::makeCosmeticVertex3d(PyObject *args)
234 {
235  PyObject* pPnt1 = nullptr;
236  if (!PyArg_ParseTuple(args, "O!", &(Base::VectorPy::Type), &pPnt1)) {
237  return nullptr;
238  }
239 
240  DrawViewPart* dvp = getDrawViewPartPtr();
241  Base::Vector3d pnt1 = static_cast<Base::VectorPy*>(pPnt1)->value();
242  Base::Vector3d centroid = dvp->getOriginalCentroid();
243  // center the point
244  pnt1 = pnt1 - centroid;
245  // project but do not invert
246  Base::Vector3d projected = dvp->projectPoint(pnt1);
247  // this is a real world point, it is not scaled or rotated, so so it is in canonical form
248  // add and invert the point.
249  std::string id = dvp->addCosmeticVertex(projected);
250  //int link =
251  dvp->add1CVToGV(id);
252  dvp->refreshCVGeoms();
253  dvp->requestPaint();
254 
255  return PyUnicode_FromString(id.c_str()); //return tag for new CV
256 }

292 PyObject* DrawViewPartPy::removeCosmeticVertex(PyObject *args)
293 {
294  DrawViewPart* dvp = getDrawViewPartPtr();
295  const char* tag;
296  if (PyArg_ParseTuple(args, "s", &tag)) {
297  dvp->removeCosmeticVertex(tag);
298  dvp->refreshCVGeoms();
299  dvp->requestPaint();
300  Py_Return;
301  }
302 
303  PyErr_Clear();
304  PyObject* pCVToDelete = nullptr;
305  if (PyArg_ParseTuple(args, "O!", &(TechDraw::CosmeticVertexPy::Type), &pCVToDelete)) {
306  TechDraw::CosmeticVertexPy* cvPy = static_cast<TechDraw::CosmeticVertexPy*>(pCVToDelete);
307  TechDraw::CosmeticVertex* cv = cvPy->getCosmeticVertexPtr();
308  dvp->removeCosmeticVertex(cv->getTagAsString());
309  dvp->refreshCVGeoms();
310  dvp->requestPaint();
311  Py_Return;
312  }
313 
314  PyErr_Clear();
315  PyObject* pDelList = nullptr;
316  if (!PyArg_ParseTuple(args, "O", &pDelList)) {
317  return nullptr;
318  }
319 
320  if (PySequence_Check(pDelList)) {
321  Py::Sequence sequence(pDelList);
322  for (const auto& item : sequence) {
323  if (!PyObject_TypeCheck(item.ptr(), &(TechDraw::CosmeticVertexPy::Type))) {
324  PyErr_Format(PyExc_TypeError ,"Types in sequence must be 'CosmeticVertex', not %s",
325  Py_TYPE(item.ptr())->tp_name);
326  return nullptr;
327  }
328  TechDraw::CosmeticVertexPy* cvPy = static_cast<TechDraw::CosmeticVertexPy*>(item.ptr());
329  TechDraw::CosmeticVertex* cv = cvPy->getCosmeticVertexPtr();
330  dvp->removeCosmeticVertex(cv->getTagAsString());
331  }
332  dvp->refreshCVGeoms();
333  dvp->requestPaint();
334 
335  Py_Return;
336  }
337  else {
338  PyErr_SetString(PyExc_TypeError, "Expected string, CosmeticVertex or sequence of CosmeticVertex");
339  return nullptr;
340  }
341 }

346 PyObject* DrawViewPartPy::makeCosmeticLine(PyObject *args)
347 {
348  // the input points are expected to use conventional coordinates (Y up) and need to be inverted
349  // before building the line
350  PyObject* pPnt1 = nullptr;
351  PyObject* pPnt2 = nullptr;
352  int style = LineFormat::getDefEdgeStyle();
353  double weight = LineFormat::getDefEdgeWidth();
354  App::Color defCol = LineFormat::getDefEdgeColor();
355  PyObject* pColor = nullptr;
356 
357  if (!PyArg_ParseTuple(args, "O!O!|idO!", &(Base::VectorPy::Type), &pPnt1,
358  &(Base::VectorPy::Type), &pPnt2,
359  &style, &weight,
360  &PyTuple_Type, &pColor)) {
361  return nullptr;
362  }
363 
364  DrawViewPart* dvp = getDrawViewPartPtr();
365  Base::Vector3d pnt1 = static_cast<Base::VectorPy*>(pPnt1)->value();
366  Base::Vector3d pnt2 = static_cast<Base::VectorPy*>(pPnt2)->value();
367  std::string newTag = dvp->addCosmeticEdge(DU::invertY(pnt1), DU::invertY(pnt2));
368  TechDraw::CosmeticEdge* ce = dvp->getCosmeticEdge(newTag);
369  if (ce) {
370  ce->m_format.setStyle(style);
371  ce->m_format.setWidth(weight);
372  ce->m_format.setColor(pColor ? DrawUtil::pyTupleToColor(pColor) : defCol);
373  }
374  else {
375  PyErr_SetString(PyExc_RuntimeError, "DVPPI:makeCosmeticLine - line creation failed");
376  return nullptr;
377  }
378  //int link =
379  dvp->add1CEToGE(newTag);
380  dvp->requestPaint();
381 
382  return PyUnicode_FromString(newTag.c_str()); //return tag for new CE
383 }
