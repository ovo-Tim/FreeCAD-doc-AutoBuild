63  class PiecewiseSparseIntegerSet
64  {
65  public:
66  PiecewiseSparseIntegerSet() = default;
67 
68  private:
69  // Each pair being <lowest, count> represents the span of integers from lowest to
70  // (lowest+count-1) inclusive
71  using SpanType = std::pair<unsigned int, unsigned int>;
72  // This span Comparer class is analogous to std::less and treats overlapping spans as being
73  // neither greater nor less than each other
74  class Comparer
75  {
76  public:
77  bool operator()(SpanType lhs, SpanType rhs) const
78  {
79  // The equality case here is when lhs is below and directly adjacent to rhs.
80  return lhs.second + lhs.first <= rhs.first;
81  }
82  };
83  // spans is the set of spans. Adjacent spans are coalesced so there are always gaps between
84  // the entries.
85  std::set<SpanType, Comparer> spans;
86 
87  public:
88  void add(unsigned int value);
89  void remove(unsigned int value);
90  bool contains(unsigned int value) const;
91  bool empty() const
92  {
93  return spans.empty();
94  }
95  void clear()
96  {
97  spans.clear();
98  }
99  unsigned int next() const
100  {
101  if (spans.empty()) {
102  return 0;
103  }
104  auto last = spans.end();
105  --last;
106  return last->first + last->second;
107  }
108  };
109  // Keyed as uniqueSeeds[baseName][digitCount][digitValue] iff that seed is taken.
110  // We need the double-indexing so that Name01 and Name001 can both be indexed, although we only
111  // ever allocate off the longest for each name i.e. uniqueSeeds[baseName].size()-1 digits.
112  std::map<std::string, std::vector<PiecewiseSparseIntegerSet>> uniqueSeeds;
113  // Counts of inserted strings that have duplicates, i.e. more than one instance in the
114  // collection. This does not contain entries for singleton names.
115  std::map<std::string, unsigned int> duplicateCounts;
