43 def generate_geometry(obj, cached=False):
44  """Sets the geometry of the given object from a corresponding IFC element.
45  This is the main function called by the execute method of FreeCAD document objects
46  It is only meant to be called form there, as it is always the responsibility of the
47  NativeIFC document object to know when it needs to regenerate its geometry.
48 
49  The generate_geometry will call either generate_shape or generate_coin, depending
50  on the type of representation it wants.
51 
52  obj: FreeCAD document object
53  [cached]: If we should try to use the cached version. The document object knows when
54  to use this.
55  """
56 
57  # setup
58  if not obj:
59  return
60  colors = None
61 
62  ifcfile = ifc_tools.get_ifcfile(obj)
63 
64  # annotations
65  if ifc_export.is_annotation(obj):
66  element = ifc_tools.get_ifc_element(obj)
67  if not element:
68  return
69  if obj.ShapeMode == "Shape":
70  shape, placement = get_annotation_shape(element, ifcfile)
71  if shape:
72  obj.Shape = shape
73  if placement:
74  obj.Placement = placement
75  return
76  elif obj.ViewObject and obj.ShapeMode == "Coin":
77  done = False
78  node, placement = get_annotation_shape(element, ifcfile, coin=True)
79  if node:
80  set_representation(obj.ViewObject, node)
81  colors = node[0]
82  done = True
83  else:
84  set_representation(obj.ViewObject, None)
85  print_debug(obj)
86  if placement:
87  obj.Placement = placement
88  if done:
89  return
90 
91  # generate the shape or coin node
92  elements = get_decomposition(obj)
93  if obj.ShapeMode == "Shape":
94  shape, colors = generate_shape(ifcfile, elements, cached)
95  if shape:
96  placement = shape.Placement
97  obj.Shape = shape
98  obj.Placement = placement
99  else:
100  obj.Shape = Part.Shape()
101  print_debug(obj)
102  elif obj.ViewObject and obj.ShapeMode == "Coin":
103  node, placement = generate_coin(ifcfile, elements, cached)
104  if node:
105  # TODO this still needs to be fixed
106  #QtCore.QTimer.singleShot(0, lambda: set_representation(obj.ViewObject, node))
107  set_representation(obj.ViewObject, node)
108  colors = node[0]
109  else:
110  set_representation(obj.ViewObject, None)
111  print_debug(obj)
112  if placement:
113  obj.Placement = placement
114 
115  # set shape and diffuse colors
116  if colors:
117  QtCore.QTimer.singleShot(0, lambda: ifc_tools.set_colors(obj, colors)) # TODO migrate here?

120 def generate_shape(ifcfile, elements, cached=False):
121  """Returns a Part shape and a list of colors for a list of elements"""
122 
123  # setup
124  if not elements:
125  return None, None
126  shapes = []
127  colors = []
128  cache = get_cache(ifcfile)
129 
130  # get cached elements
131  if cached:
132  rest = []
133  for element in elements:
134  if element.id in cache["Shape"]:
135  shape = cache["Shape"][element.id]
136  shapes.append(shape.copy())
137  if element.id in cache["Color"]:
138  color = cache["Color"][element.id]
139  else:
140  color = (0.8, 0.8, 0.8)
141  if len(color) <= 4:
142  for face in shape.Faces:
143  colors.append(color)
144  else:
145  colors = color
146  else:
147  rest.append(element)
148  if not rest:
149  # all elements have been taken from cache, nothing more to do
150  return shapes, colors
151  elements = rest
152 
153  # prepare the iterator
154  iterator = get_geom_iterator(ifcfile, elements, brep_mode=True)
155  if iterator is None:
156  return None, None
157  total = len(elements)
158  progressbar = Base.ProgressIndicator()
159  progressbar.start("Generating " + str(total) + " shapes...", total)
160  done = []
161 
162  # iterate
163  while True:
164  item = iterator.get()
165  if item and item.id not in done:
166  done.append(item.id)
167  # get and transfer brep data
168  brep = item.geometry.brep_data
169  shape = Part.Shape()
170  shape.importBrepFromString(brep, False)
171  if hasattr(item.transformation.matrix, "data"):
172  # IfcOpenShell 0.7
173  mat = ifc_tools.get_freecad_matrix(item.transformation.matrix.data)
174  else:
175  # IfcOpenShell 0.8
176  mat = ifc_tools.get_freecad_matrix(item.transformation.matrix)
177  shape.scale(ifc_tools.SCALE)
178  shape.transformShape(mat)
179  shapes.append(shape)
180 
181  # get colors
182  sstyle = item.geometry.surface_styles
183  scolors = []
184  if (
185  (len(sstyle) > 4)
186  and len(shape.Solids) > 1
187  and len(sstyle) // 4 == len(shape.Solids)
188  ):
189  # multiple colors
190  for i in range(len(shape.Solids)):
191  for j in range(len(shape.Solids[i].Faces)):
192  scolors.append(
193  (
194  sstyle[i * 4],
195  sstyle[i * 4 + 1],
196  sstyle[i * 4 + 2],
197  1.0 - sstyle[i * 4 + 3],
198  )
199  )
200  if len(colors) < len(shape.Faces):
201  for i in range(len(shape.Faces) - len(colors)):
202  scolors.append(
203  (sstyle[0], sstyle[1], sstyle[2], 1.0 - sstyle[3])
204  )
205  else:
206  color = (sstyle[0], sstyle[1], sstyle[2], 1.0 - sstyle[3])
207  for f in shape.Faces:
208  scolors.append(color)
209 
210  # update the cache
211  cache["Shape"][item.id] = shape
212  cache["Color"][item.id] = scolors
213  colors.extend(scolors)
214  progressbar.next(True)
215  if not iterator.next():
216  break
217 
218  # write the cache
219  set_cache(ifcfile, cache)
220 
221  # compound the shape if needed
222  if len(shapes) == 1:
223  shape = shapes[0]
224  else:
225  shape = Part.makeCompound(shapes)
226 
227  progressbar.stop()
228  return shape, colors

231 def generate_coin(ifcfile, elements, cached=False):
232  """Returns coin node data (verts,face and edge index) and a Placement
233  from a list of ifc elements"""
234 
235  # setup
236  # strip out elements without representation, as they can't generate a node anyway
237  elements = [e for e in elements if getattr(e, "Representation", None)]
238  if not elements:
239  return None, None
240  # if we have more than one element, placements will need to be applied on subnodes
241  grouping = bool(len(elements) > 1)
242  nodes = []
243 
244  # process cached elements
245  placement = None
246  cache = get_cache(ifcfile)
247  if cached:
248  rest = []
249  for element in elements:
250  if element.id() in cache["Placement"]:
251  placement = cache["Placement"][element.id()]
252  if element.id() in cache["Coin"]:
253  node = cache["Coin"][element.id()]
254  if grouping:
255  node = apply_placement(node, placement)
256  nodes.append(node)
257  else:
258  rest.append(element)
259  if grouping:
260  placement = None
261  if not rest:
262  # all elements have been taken from cache, nothing more to do
263  return unify(nodes), placement
264  elements = rest
265 
266  # prepare the iterator
267  iterator = get_geom_iterator(ifcfile, elements, brep_mode=False)
268  if iterator is None:
269  return None, None
270  total = len(elements)
271  progressbar = Base.ProgressIndicator()
272  progressbar.start("Generating " + str(total) + " shapes...", total)
273  done = []
274 
275  # iterate
276  while True:
277  item = iterator.get()
278  if item and item.id not in done:
279  done.append(item.id)
280 
281  # colors
282  if item.geometry.materials:
283  color = item.geometry.materials[0].diffuse
284  if hasattr(color, "r") and hasattr(color, "g"):
285  # IfcOpenShell 0.8
286  color = (color.r(), color.g(), color.b())
287  else:
288  # IfcOpenShell 0.7
289  color = (float(color[0]), float(color[1]), float(color[2]))
290  trans = item.geometry.materials[0].transparency
291  if trans >= 0:
292  color += (float(trans),)
293  else:
294  color = (0.85, 0.85, 0.85)
295 
296  # verts
297  if hasattr(item.transformation.matrix, "data"):
298  # IfcOpenShell 0.7
299  matrix = ifc_tools.get_freecad_matrix(item.transformation.matrix.data)
300  else:
301  # IfcOpenShell 0.8
302  matrix = ifc_tools.get_freecad_matrix(item.transformation.matrix)
303  placement = FreeCAD.Placement(matrix)
304  verts = item.geometry.verts
305  verts = [FreeCAD.Vector(verts[i : i + 3]) for i in range(0, len(verts), 3)]
306  verts = [tuple(v.multiply(ifc_tools.SCALE)) for v in verts]
307 
308  # faces
309  faces = list(item.geometry.faces)
310  faces = [
311  f for i in range(0, len(faces), 3) for f in faces[i : i + 3] + [-1]
312  ]
313 
314  # edges
315  edges = list(item.geometry.edges)
316  edges = [
317  e for i in range(0, len(edges), 2) for e in edges[i : i + 2] + [-1]
318  ]
319 
320  # update cache
321  node = [color, verts, faces, edges]
322  cache["Coin"][item.id] = node
323  cache["Placement"][item.id] = placement
324 
325  if grouping:
326  # if we are joining nodes together, their placement
327  # must be baked in
328  node = apply_placement(node, placement)
329  nodes.append(node)
330  progressbar.next(True)
331  if not iterator.next():
332  break
333 
334  # unify nodes
335  nodes = unify(nodes)
336 
337  # nullify placement if already applied
338  if grouping:
339  placement = None
340 
341  # write cache
342  set_cache(ifcfile, cache)
343 
344  progressbar.stop()
345  return nodes, placement

583 def get_annotation_shape(annotation, ifcfile, coin=False):
584  """Returns a shape or a coin node form an IFC annotation.
585  Returns [colors, verts, faces, edges], colors and faces
586  being normally None for 2D shapes."""
587 
588  import Part
589  from importers import importIFCHelper
590 
591  shape = None
592  placement = None
593  ifcscale = importIFCHelper.getScaling(ifcfile)
594  shapes2d = []
595  if hasattr(annotation, "Representation"):
596  for rep in annotation.Representation.Representations:
597  if rep.RepresentationIdentifier in ["Annotation", "FootPrint", "Axis"]:
598  sh = importIFCHelper.get2DShape(rep, ifcscale, notext=True)
599  if sh:
600  shapes2d.extend(sh)
601  elif hasattr(annotation, "AxisCurve"):
602  sh = importIFCHelper.get2DShape(annotation.AxisCurve, ifcscale, notext=True)
603  shapes2d.extend(sh)
604  if shapes2d:
605  shape = Part.makeCompound(shapes2d)
606  if hasattr(annotation, "ObjectPlacement"):
607  placement = importIFCHelper.getPlacement(annotation.ObjectPlacement, ifcscale)
608  else:
609  placement = None
610  if coin:
611  iv = shape.writeInventor()
612  iv = iv.replace("\n", "")
613  segs = re.findall(r"point \[.*?\]",iv)
614  segs = [s.replace("point [","").replace("]","").strip() for s in segs]
615  segs = [s.split(" ") for s in segs]
616  verts = []
617  edges = []
618  for pair in segs:
619  v1 = tuple([float(v) for v in pair[0].split()])
620  v2 = tuple([float(v) for v in pair[1].split()])
621  if not v1 in verts:
622  verts.append(v1)
623  edges.append(verts.index(v1))
624  if not v2 in verts:
625  verts.append(v2)
626  edges.append(verts.index(v2))
627  edges.append(-1)
628  shape = [[None, verts, [], edges]]
629  # unify nodes
630  shape = unify(shape)
631  return shape, placement
