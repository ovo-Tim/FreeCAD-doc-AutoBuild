107 void execInsertPrefixChar(Gui::Command* cmd, std::string prefixFormat, const QAction *action = nullptr) {
108  // insert a prefix character into the format specifier
109  std::vector<Gui::SelectionObject> selection;
110  if (!_checkSelection(cmd, selection, QT_TRANSLATE_NOOP("Command","TechDraw Insert Prefix"))) {
111  return;
112  }
113 
114  std::string prefixText(prefixFormat);
115  if (prefixFormat.find("%s") != std::string::npos) {
116  DlgTemplateField ui;
117  const int MAX_PREFIX_LENGTH = 31;
118 
119  if (action) {
120  if (action->objectName() == QString::fromUtf8("TechDraw_ExtensionInsertRepetition")) {
121  ui.setFieldName(QT_TR_NOOP("Repeat Count"));
122  }
123  }
124 
125  ui.setFieldLength(MAX_PREFIX_LENGTH);
126  ui.setFieldContent("");
127  if (ui.exec() != QDialog::Accepted) {
128  return;
129  }
130 
131  char prefixData[(MAX_PREFIX_LENGTH + 1)*4];
132  snprintf(prefixData, sizeof(prefixData), prefixFormat.c_str(), ui.getFieldContent().toUtf8().constData());
133  prefixText = prefixData;
134  }
135 
136  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Insert Prefix"));
137  for (auto selected : selection) {
138  auto object = selected.getObject();
139  if (object->isDerivedFrom(TechDraw::DrawViewDimension::getClassTypeId())) {
140  auto dim = static_cast<TechDraw::DrawViewDimension*>(selected.getObject());
141  std::string formatSpec = dim->FormatSpec.getStrValue();
142  formatSpec = prefixText + formatSpec;
143  dim->FormatSpec.setValue(formatSpec);
144  }
145  }
146  Gui::Command::commitCommand();
147 }

435 void execIncreaseDecreaseDecimal(Gui::Command* cmd, int delta) {
436  // increase or decrease number of decimal places of a measure
437  std::vector<Gui::SelectionObject> selection;
438  if (!_checkSelection(cmd, selection, QT_TRANSLATE_NOOP("QObject","TechDraw Increase/Decrease Decimal"))) {
439  return;
440  }
441 
442  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Increase/Decrease Decimal"));
443  std::string numStr;
444  for (auto selected : selection) {
445  auto object = selected.getObject();
446  if (object->isDerivedFrom(TechDraw::DrawViewDimension::getClassTypeId())) {
447  auto dim = static_cast<TechDraw::DrawViewDimension*>(selected.getObject());
448  std::string formatSpec = dim->FormatSpec.getStrValue();
449  std::string searchStr("%.");
450  int numFound = formatSpec.find(searchStr) + 2;
451  numStr = formatSpec[numFound];
452  int numInt = std::stoi(numStr, nullptr);
453  numInt = numInt + delta;
454  if (numInt >= 0 && numInt <= 9) {
455  numStr = std::to_string(numInt);
456  formatSpec.replace(numFound, 1, numStr);
457  dim->FormatSpec.setValue(formatSpec);
458  }
459  }
460  }
461  Gui::Command::commitCommand();
462 }
