30 def test_python_syntax(rootdir, whitelist=None):
31  whitelist = whitelist or []
32  log = []
33  for sub_dir, dirs, files in os.walk(rootdir):
34  for fn in files:
35  kargs = {}
36  kargs["encoding"] = "utf-8"
37  if (not fn in whitelist) and os.path.splitext(fn)[1] == ".py":
38  with open(os.path.join(sub_dir, fn), **kargs) as py_file:
39  try:
40  ast.parse(py_file.read())
41  except SyntaxError as err:
42  log.append(str(err).replace("<unknown>", os.path.join(sub_dir, fn)))
43  message = "\n\n" + "#" * 30 + "\n"
44  message += "{} python files are not parseable:\n\n".format(len(log))
45  for i, m in enumerate(log):
46  message += str(i + 1) + " " + m + "\n"
47  if log:
48  raise RuntimeError(
49  "there are some files not parse-able with the used python-interpreter" + message
50  )
51  else:
52  return
