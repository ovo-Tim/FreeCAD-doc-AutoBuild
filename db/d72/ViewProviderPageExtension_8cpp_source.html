107 void ViewProviderPageExtension::extensionDropObject(App::DocumentObject* obj)
108 {
109  bool linkToView = false;
110  if (obj->isDerivedFrom(App::Link::getClassTypeId())) {
111  auto* link = static_cast<App::Link*>(obj);
112  if (link->getLinkedObject()->isDerivedFrom(TechDraw::DrawView::getClassTypeId())) {

116 
117  if (obj->isDerivedFrom(TechDraw::DrawView::getClassTypeId()) || linkToView) {
118  dropObject(obj);

123 //this code used to live in ViewProviderPage
124 void ViewProviderPageExtension::dropObject(App::DocumentObject* obj)
125 {
126  if (obj->isDerivedFrom<TechDraw::DrawProjGroupItem>()) {
127  //DPGI can not be dropped onto the Page if it belongs to DPG
128  auto* dpgi = static_cast<TechDraw::DrawProjGroupItem*>(obj);
129  if (dpgi->getPGroup()) {
130  return;
131  }
132  }
133  if (obj->isDerivedFrom<App::Link>()) {
134  auto* link = static_cast<App::Link*>(obj);
135  obj = link->getLinkedObject();
136  if (!obj->isDerivedFrom<TechDraw::DrawView>()) {
137  return;
138  }
139 
140  std::vector<App::DocumentObject*> deps;
141  for (auto& inObj : obj->getInList()) {
142  if (inObj && inObj->isDerivedFrom<TechDraw::DrawView>()) {
143  deps.push_back(inObj);
144  }
145  }
146 
147  TechDraw::DrawPage* page = nullptr;
148  for (auto& inObj : link->getInList()) {
149  if (inObj->isDerivedFrom<TechDraw::DrawPage>()) {
150  page = static_cast<TechDraw::DrawPage*>(inObj);
151  }
152  }
153  if (page) {
154  for (auto* dep : deps) {
155  page->removeView(dep);
156  }
157  page->removeView(link);
158  }
159 
160  getViewProviderPage()->getDrawPage()->addView(link, false);
161  for (auto* dep : deps) {
162  getViewProviderPage()->getDrawPage()->addView(dep, false);
163  }
164 
165  return;
166  }
167 
168  if (obj->isDerivedFrom<TechDraw::DrawView>()) {
169  auto dv = static_cast<TechDraw::DrawView*>(obj);
170 
171  std::vector<App::DocumentObject*> deps;
172  for (auto* dep : dv->getInList()) {
173  if (dep && dep->isDerivedFrom<TechDraw::DrawView>()) {
174  deps.push_back(dep);
175  }
176  }
177 
178  auto* parentPage = dv->findParentPage();
179  if (parentPage) {
180  for (auto* dep : deps) {
181  parentPage->removeView(dep);
182  }
183  parentPage->removeView(dv);
184  }
185 
186  getViewProviderPage()->getDrawPage()->addView(dv, false);
187  for (auto* dep : deps) {
188  getViewProviderPage()->getDrawPage()->addView(dep, false);
189  }
190 
191  //don't run ancestor's method as addView does everything we need
192  return;
193  }
194  //don't try to drop random objects
195 }
