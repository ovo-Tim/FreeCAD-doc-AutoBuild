103 static std::string getSelectedNodes(Gui::View3DInventorViewer* view)
104 {
105  Gui::SelectionRole role;
106  std::vector<SbVec2f> clPoly = view->getGLPolygon(&role);
107  if (clPoly.size() < 3) {
108  return {};
109  }
110  if (clPoly.front() != clPoly.back()) {
111  clPoly.push_back(clPoly.front());
112  }
113 
114  SoCamera* cam = view->getSoRenderManager()->getCamera();
115  SbViewVolume vv = cam->getViewVolume();
116  Gui::ViewVolumeProjection proj(vv);
117  Base::Polygon2d polygon;
118  for (auto it : clPoly) {
119  polygon.Add(Base::Vector2d(it[0], it[1]));
120  }
121 
122  std::vector<App::DocumentObject*> docObj =
123  Gui::Selection().getObjectsOfType(Fem::FemMeshObject::getClassTypeId());
124  if (docObj.size() != 1) {
125  return {};
126  }
127 
128  const SMESHDS_Mesh* data =
129  static_cast<Fem::FemMeshObject*>(docObj[0])->FemMesh.getValue().getSMesh()->GetMeshDS();
130 
131  SMDS_NodeIteratorPtr aNodeIter = data->nodesIterator();
132  Base::Vector3f pt2d;
133  std::set<int> IntSet;
134 
135  while (aNodeIter->more()) {
136  const SMDS_MeshNode* aNode = aNodeIter->next();
137  Base::Vector3f vec(aNode->X(), aNode->Y(), aNode->Z());
138  pt2d = proj(vec);
139  if (polygon.Contains(Base::Vector2d(pt2d.x, pt2d.y))) {
140  IntSet.insert(aNode->GetID());
141  }
142  }
143 
144  std::stringstream set;
145 
146  set << "[";
147  for (auto it = IntSet.cbegin(); it != IntSet.cend(); ++it) {
148  if (it == IntSet.begin()) {
149  set << *it;
150  }
151  else {
152  set << "," << *it;
153  }
154  }
155  set << "]";
156 
157  return set.str();
158 }

1796  // only active on allowed objects
1797  // For the case the clip filter is set by Python code, we check that the input
1798  // is a post object and issue an error if not.

1800  if (Gui::Selection().getSelection().size() > 1) {
1801  QMessageBox::warning(
1802  Gui::getMainWindow(),
1803  qApp->translate("setupFilter",
1804  "Error: A filter can only be applied to a single object."),

1806  return;
1807  }
1808 
1809  auto selObject = Gui::Selection().getSelection()[0].pObject;
1810 
1811  // issue error if no filter object
1812  if (!(selObject->isDerivedFrom<Fem::FemPostObject>())) {
1813  QMessageBox::warning(
1814  Gui::getMainWindow(),
1815  qApp->translate("setupFilter", "Error: no post processing object selected."),
1816  qApp->translate("setupFilter", "The filter could not be set up."));
1817  return;
1818  }
1819 
1820  std::string FeatName = cmd->getUniqueObjectName(Name.c_str());
1821 
1822  // at first we must determine the pipeline of the selection object
1823  // (which can be a pipeline itself)
1824  bool selectionIsPipeline = false;
1825  Fem::FemPostPipeline* pipeline = nullptr;
1826  if (selObject->isDerivedFrom<Fem::FemPostPipeline>()) {
1827  pipeline = static_cast<Fem::FemPostPipeline*>(selObject);
1828  selectionIsPipeline = true;
1829  }
1830  else {
1831  auto parents = selObject->getInList();
1832  if (!parents.empty()) {
1833  for (auto parentObject : parents) {
1834  if (parentObject->isDerivedFrom<Fem::FemPostPipeline>()) {
1835  pipeline = static_cast<Fem::FemPostPipeline*>(parentObject);
1836  }
1837  }
1838  }
1839  }
1840 
1841  if (!pipeline) {
1842  QMessageBox::warning(
1843  Gui::getMainWindow(),
1844  qApp->translate("setupFilter", "Error: no post processing object selected."),
1845  qApp->translate("setupFilter", "The filter could not be set up."));
1846  return;
1847  }
1848 
1849  // create the object and add it to the pipeline
1850  cmd->openCommand(QT_TRANSLATE_NOOP("Command", "Create filter"));
1851  cmd->doCommand(Gui::Command::Doc,
1852  "App.activeDocument().addObject('Fem::FemPost%sFilter','%s')",
1853  Name.c_str(),
1854  FeatName.c_str());
