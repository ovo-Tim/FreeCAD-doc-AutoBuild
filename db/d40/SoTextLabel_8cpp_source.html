114 void SoTextLabel::GLRender(SoGLRenderAction *action)
115 {
116  if (!this->shouldGLRender(action))
117  return;
118 
119  // only draw text without background
120  if (!this->background.getValue()) {
121  inherited::GLRender(action);
122  return;
123  }
124 
125  SoState * state = action->getState();
126 
127  state->push();
128  SoLazyElement::setLightModel(state, SoLazyElement::BASE_COLOR);
129 
130  SbBox3f box;
131  SbVec3f center;
132  this->computeBBox(action, box, center);
133 
134  if (!SoCullElement::cullTest(state, box, true)) {
135  SoMaterialBundle mb(action);
136  mb.sendFirst();
137  const SbMatrix & mat = SoModelMatrixElement::get(state);
138  const SbMatrix & projmatrix = (mat * SoViewingMatrixElement::get(state) *
139  SoProjectionMatrixElement::get(state));
140  const SbViewportRegion & vp = SoViewportRegionElement::get(state);
141  SbVec2s vpsize = vp.getViewportSizePixels();
142 
143  // font stuff
144  SbName fontname = SoFontNameElement::get(state);
145  int lines = this->string.getNum();
146 
147  // get left bottom corner of the label
148  SbVec3f nilpoint(0.0f, 0.0f, 0.0f);
149  projmatrix.multVecMatrix(nilpoint, nilpoint);
150  nilpoint[0] = (nilpoint[0] + 1.0f) * 0.5f * vpsize[0];
151  nilpoint[1] = (nilpoint[1] + 1.0f) * 0.5f * vpsize[1];
152 
153  // Unfortunately, the size of the label is stored in the pimpl class of
154  // SoText2 which cannot be accessed directly. However, there is a trick
155  // to get the required information: set model, viewing and projection
156  // matrix to the identity matrix and also view volume to some default
157  // values. SoText2::computeBBox() then calls SoText2P::getQuad which
158  // returns the sizes in form of the bounding box. These values can be
159  // reverse-engineered to get width and height.
160  state->push();
161  SoModelMatrixElement::set(state,this,SbMatrix::identity());
162  SoViewingMatrixElement::set(state,this,SbMatrix::identity());
163  SoProjectionMatrixElement::set(state,this,SbMatrix::identity());
164  SbViewVolume vv;
165  vv.ortho(-1,1,-1,1,-1,1);
166  SoViewVolumeElement::set(state,this,vv);
167 
168  SbBox3f box;
169  SbVec3f center;
170  this->computeBBox(action, box, center);
171  state->pop();
172 
173  float xmin,ymin,zmin,xmax,ymax,zmax;
174  box.getBounds(xmin,ymin,zmin,xmax,ymax,zmax);
175  SbVec3f v0(xmin,ymax,zmax);
176  SbVec3f v1(xmax,ymax,zmax);
177  SbVec3f v2(xmax,ymin,zmax);
178  SbVec3f v3(xmin,ymin,zmax);
179  vv.projectToScreen(v0,v0);
180  vv.projectToScreen(v1,v1);
181  vv.projectToScreen(v2,v2);
182  vv.projectToScreen(v3,v3);
183 
184  float width,height;
185  width = (v1[0]-v0[0])*vpsize[0];
186  height = (v1[1]-v3[1])*vpsize[1];
187  switch (this->justification.getValue()) {
188  case SoText2::RIGHT:
189  nilpoint[0] -= width;
190  break;
191  case SoText2::CENTER:
192  nilpoint[0] -= 0.5f*width;
193  break;
194  default:
195  break;
196  }
197 
198  if (lines > 1) {
199  nilpoint[1] -= (float(lines-1)/(float)lines*height);
200  }
201 
202  SbVec3f toppoint = nilpoint;
203  toppoint[0] += width;
204  toppoint[1] += height;
205 
206  // Set new state.
207  glMatrixMode(GL_MODELVIEW);
208  glPushMatrix();
209  glLoadIdentity();
210  glMatrixMode(GL_PROJECTION);
211  glPushMatrix();
212  glLoadIdentity();
213  glOrtho(0, vpsize[0], 0, vpsize[1], -1.0f, 1.0f);
214  glPixelStorei(GL_UNPACK_ALIGNMENT,1);
215 
216  state->push();
217 
218  // disable textures for all units
219  SoGLTextureEnabledElement::set(state, this, false);
220 #if COIN_MAJOR_VERSION > 3
221  SoMultiTextureEnabledElement::set(state, this, false);
222 #else
223  SoGLTexture3EnabledElement::set(state, this, false);
224 #endif
225 
226  glPushAttrib(GL_ENABLE_BIT | GL_PIXEL_MODE_BIT | GL_COLOR_BUFFER_BIT);
227  glPushClientAttrib(GL_CLIENT_PIXEL_STORE_BIT);
228 
229  // color and frame size
230  SbColor color = this->backgroundColor.getValue();
231  float fs = this->frameSize.getValue();
232 
233  // draw background
234  glColor3f(color[0], color[1], color[2]);
235  glBegin(GL_QUADS);
236  glVertex3f(nilpoint[0]-fs,nilpoint[1]-fs,0.0f);
237  glVertex3f(toppoint[0]+fs,nilpoint[1]-fs,0.0f);
238  glVertex3f(toppoint[0]+fs,toppoint[1]+fs,0.0f);
239  glVertex3f(nilpoint[0]-fs,toppoint[1]+fs,0.0f);
240  glEnd();
241 
242  // pop old state
243  glPopClientAttrib();
244  glPopAttrib();
245  state->pop();
246 
247  glPixelStorei(GL_UNPACK_ALIGNMENT,4);
248  // Pop old GL matrix state.
249  glMatrixMode(GL_PROJECTION);
250  glPopMatrix();
251  glMatrixMode(GL_MODELVIEW);
252  glPopMatrix();
253  }
254 
255  state->pop();
256 
257  inherited::GLRender(action);
258 }

308 void SoStringLabel::GLRender(SoGLRenderAction *action)
309 {
310  QtGLWidget* window;
311  SoState * state = action->getState();
312  state->push();
313  SoLazyElement::setLightModel(state, SoLazyElement::BASE_COLOR);
314  SoGLWidgetElement::get(state, window);
315  if (!window) {
316  state->pop();
317  return;
318  }
319 
320  // Enter in 2D screen mode
321  glMatrixMode(GL_PROJECTION);
322  glPushMatrix();
323  glLoadIdentity();
324  glOrtho(-1,1,-1,1,-1,1);
325  glMatrixMode(GL_MODELVIEW);
326  glPushMatrix();
327  glLoadIdentity();
328  glPushAttrib(GL_ENABLE_BIT);
329  glDisable(GL_DEPTH_TEST);
330  glDisable(GL_LIGHTING);
331  glDisable(GL_TEXTURE_2D);
332  glEnable(GL_BLEND);
333 
334  QFont font;
335  font.setStyleStrategy(QFont::NoAntialias);
336  font.setFamily(QLatin1String(this->name.getValue()));
337  font.setPixelSize(this->size.getValue());
338 
339  glBlendFunc(GL_ONE,GL_SRC_ALPHA);
340 
341  // text color
342  SbColor color = this->textColor.getValue();
343  glColor4f(color[0], color[1], color[2], 1);
344  const SbMatrix & mat = SoModelMatrixElement::get(state);
345  const SbMatrix & projmatrix = (mat * SoViewingMatrixElement::get(state) *
346  SoProjectionMatrixElement::get(state));
347  SbVec3f nil(0.0f, 0.0f, 0.0f);
348  projmatrix.multVecMatrix(nil, nil);
349  QStringList list;
350  for (int i=0; i<this->string.getNum(); i++)
351  list << QLatin1String(this->string[i].getString());
352 
353  // Leave 2D screen mode
354  glPopAttrib();
355  glPopMatrix();
356  glMatrixMode(GL_PROJECTION);
357  glPopMatrix();
358  glMatrixMode(GL_MODELVIEW);
359 
360  state->pop();
361 }

410 void SoFrameLabel::drawImage()
411 {
412  const SbString* s = string.getValues(0);
413  int num = string.getNum();
414  if (num == 0) {
415  this->image = SoSFImage();
416  return;
417  }
418 
419  QFont font(QString::fromLatin1(QByteArray(name.getValue())), size.getValue());
420  QFontMetrics fm(font);
421  int w = 0;
422  int h = fm.height() * num;
423  const SbColor& b = backgroundColor.getValue();
424  QColor backgroundBrush;
425  backgroundBrush.setRgbF(b[0],b[1],b[2]);
426  const SbColor& t = textColor.getValue();
427  QColor front;
428  front.setRgbF(t[0],t[1],t[2]);
429  const QPen borderPen(QColor(0,0,127), 2, Qt::SolidLine, Qt::RoundCap, Qt::RoundJoin);
430 
431  QStringList lines;
432  for (int i=0; i<num; i++) {
433  QString line = QString::fromUtf8(s[i].getString());
434  w = std::max<int>(w, QtTools::horizontalAdvance(fm, line));
435  lines << line;
436  }
437 
438  int padding = 5;
439 
440  bool drawIcon = false;
441  QImage iconImg;
442  int widthIcon = 0;
443  int heightIcon = 0;
444  if (!iconPixmap.isNull()) {
445  drawIcon = true;
446  iconImg = iconPixmap.toImage();
447  widthIcon = iconImg.width() + 2*padding;
448  heightIcon = iconImg.height() + 2*padding;
449  }
450 
451  int widthText = w + 2*padding;
452  int heightText = h + 2*padding;
453  int widthTotal = widthText + widthIcon;
454  int heightTotal = heightText > heightIcon ? heightText : heightIcon;
455  int paddingTextV = (heightTotal - h) / 2;
456  int paddingIconV = (heightTotal - iconImg.height()) / 2;
457 
458  QImage image(widthTotal, heightTotal, QImage::Format_ARGB32_Premultiplied);
459  image.fill(0x00000000);
460  QPainter painter(&image);
461  painter.setRenderHint(QPainter::Antialiasing);
462 
463  SbBool drawFrame = frame.getValue();
464  SbBool drawBorder = border.getValue();
465  if (drawFrame || drawBorder) {
466  painter.setPen(drawBorder ? borderPen : QPen(Qt::transparent));
467  painter.setBrush(QBrush(drawFrame ? backgroundBrush : Qt::transparent));
468 
469  QRectF rectangle(0.0, 0.0, widthTotal, heightTotal);
470  painter.drawRoundedRect(rectangle, 5, 5);
471  }
472 
473  if (drawIcon) {
474  painter.drawImage(QPoint(padding, paddingIconV), iconImg);
475  }
476 
477  painter.setPen(front);
478 
479  Qt::Alignment align = Qt::AlignVCenter;
480  if (justification.getValue() == 0)
481  align = Qt::AlignVCenter | Qt::AlignLeft;
482  else if (justification.getValue() == 1)
483  align = Qt::AlignVCenter | Qt::AlignRight;
484  else
485  align = Qt::AlignVCenter | Qt::AlignHCenter;
486  QString text = lines.join(QLatin1String("\n"));
487  painter.setFont(font);
488  painter.drawText(widthIcon + padding, paddingTextV, w, h, align, text);
489  painter.end();
490 
491  SoSFImage sfimage;
492  Gui::BitmapFactory().convert(image, sfimage);
493  this->image = sfimage;
494 }
