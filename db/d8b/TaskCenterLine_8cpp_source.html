344 // check that we are not trying to create an impossible centerline (ex a vertical centerline
345 // between 2 horizontal edges)
346 int TaskCenterLine::checkPathologicalEdges(int inMode)
347 {
348  if (m_type != CenterLine::EDGE) {
349  // not an edge based centerline, this doesn't apply
350  return inMode;
351  }
352 
353  TechDraw::BaseGeomPtr edge1 = m_partFeat->getEdge(m_subNames.front());
354  std::vector<Base::Vector3d> ends1 = edge1->findEndPoints();
355  bool edge1Vertical = DU::fpCompare(ends1.front().x, ends1.back().x, EWTOLERANCE);
356  bool edge1Horizontal = DU::fpCompare(ends1.front().y, ends1.back().y, EWTOLERANCE);
357 
358  TechDraw::BaseGeomPtr edge2 = m_partFeat->getEdge(m_subNames.back());
359  std::vector<Base::Vector3d> ends2 = edge2->findEndPoints();
360  bool edge2Vertical = DU::fpCompare(ends2.front().x, ends2.back().x, EWTOLERANCE);
361  bool edge2Horizontal = DU::fpCompare(ends2.front().y, ends2.back().y, EWTOLERANCE);
362 
363  if (edge1Vertical && edge2Vertical) {
364  return CenterLine::CLMODE::VERTICAL;
365  }
366  if (edge1Horizontal && edge2Horizontal) {
367  return CenterLine::CLMODE::HORIZONTAL;
368  }
369 
370  // not pathological case, just return the input mode
371  return inMode;
372 }
373 
374 // check that we are not trying to create an impossible centerline (ex a vertical centerline
375 // between 2 vertices aligned vertically)
376 int TaskCenterLine::checkPathologicalVertices(int inMode)
377 {
378  if (m_type != CenterLine::VERTEX) {
379  // not a vertex based centerline, this doesn't apply
380  return inMode;
381  }
382 
383  TechDraw::VertexPtr vert1 = m_partFeat->getVertex(m_subNames.front());
384  Base::Vector3d point1 = vert1->point();
385  TechDraw::VertexPtr vert2 = m_partFeat->getVertex(m_subNames.back());
386  Base::Vector3d point2 = vert2->point();
387 
388  if (DU::fpCompare(point1.x, point2.x, EWTOLERANCE)) {
389  // points are aligned vertically, CL must be horizontal
390  return CenterLine::CLMODE::HORIZONTAL;
391  }
392 
393  if (DU::fpCompare(point1.y, point2.y, EWTOLERANCE)) {
394  // points are aligned horizontally, CL must be vertical
395  return CenterLine::CLMODE::VERTICAL;
396  }
397 
398  // not pathological case, just return the input mode
399  return inMode;
400 }

403 void TaskCenterLine::createCenterLine()
404 {
405  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create Centerline"));
406 
407  // check for illogical parameters
408  if (m_type == CenterLine::EDGE) {
409  // between lines
410  m_mode = checkPathologicalEdges(m_mode);
411  } else if (m_type == CenterLine::VERTEX) {
412  // between points
413  m_mode = checkPathologicalVertices(m_mode);
414  }
415 
416  CenterLine* cl = CenterLine::CenterLineBuilder(m_partFeat, m_subNames, m_mode, false);
417 
418  if (!cl) {
419  Gui::Command::abortCommand();
420  return;
421  }
422 
423  double hShift = ui->qsbHorizShift->rawValue();
424  double vShift = ui->qsbVertShift->rawValue();
425  double rotate = ui->qsbRotate->rawValue();
426  double extendBy = ui->qsbExtend->rawValue();
427  cl->setShifts(hShift, vShift);
428  cl->setExtend(extendBy);
429  cl->setRotate(rotate);
430  cl->m_flip2Line = false;
431  App::Color ac;
432  ac.setValue<QColor>(ui->cpLineColor->color());
433  cl->m_format.setColor(ac);
434  cl->m_format.setWidth(ui->dsbWeight->value().getValue());
435  cl->m_format.setLineNumber(ui->cboxStyle->currentIndex() + 1);
436  cl->m_format.setVisible(true);
437  m_partFeat->addCenterLine(cl);
438 
439  m_partFeat->recomputeFeature();
440  Gui::Command::updateActive();
441  Gui::Command::commitCommand();
442 
443  // entering the edit mode
444  m_editMode = true;
445  m_cl = cl;
446 }

448 void TaskCenterLine::updateOrientation()
449 {
450 // Base::Console().Message("TCL::updateOrientation()\n");
451  if (!m_cl) {
452  return;
453  }
454  // When the orientation was changed, it can be that the centerline becomes invalid
455  // this can lead to a crash, see e.g.
456  // https://forum.freecad.org/viewtopic.php?f=35&t=44255&start=20#p503220
457  // The centerline creation can fail if m_type is edge and both selected edges are vertical or horizontal.
458  int orientation = m_cl->m_mode;
459  if (m_type == CenterLine::EDGE) {
460  // between lines
461  if (!m_edgeName.empty() && !m_cl->m_edges.empty()) {
462  // we have an existing centerline, not a freshly created one, and it is a centerline between edges
463  m_subNames = m_cl->m_edges;
464  orientation = checkPathologicalEdges(orientation);
465  }
466  } else if (m_type == CenterLine::VERTEX) {
467  // between points
468  if (!m_edgeName.empty() && !m_cl->m_verts.empty()) {
469  // we have an existing centerline, not a freshly created one, and it is a centerline between points
470  m_subNames = m_cl->m_verts;
471  orientation = checkPathologicalVertices(orientation);
472  }
473  }
474 
475  setUiOrientation(orientation);
476 
477  m_partFeat->recomputeFeature();
478 }

546 bool TaskCenterLine::reject()
547 {
548  Gui::Document* doc = Gui::Application::Instance->getDocument(m_basePage->getDocument());
549  if (!doc)
550  return false;
551 
552  if (getCreateMode() && m_partFeat) {
553  // undo the centerline creation
554  doc->undo(1);
555  }
556  else if (!getCreateMode() && m_partFeat) {
557  // restore the initial centerline
558  m_cl->m_format.setColor((&orig_cl)->m_format.getColor());
559  m_cl->m_format.setWidth((&orig_cl)->m_format.getWidth());
560  m_cl->m_format.setLineNumber((&orig_cl)->m_format.getLineNumber());
561  m_cl->m_format.setVisible((&orig_cl)->m_format.getVisible());
562  m_cl->m_mode = (&orig_cl)->m_mode;
563  m_cl->m_rotate = (&orig_cl)->m_rotate;
564  m_cl->m_vShift = (&orig_cl)->m_vShift;
565  m_cl->m_hShift = (&orig_cl)->m_hShift;
566  m_cl->m_extendBy = (&orig_cl)->m_extendBy;
567  m_cl->m_type = (&orig_cl)->m_type;
568  }
569 
570  if (m_partFeat)
571  m_partFeat->recomputeFeature();
572  Gui::Command::doCommand(Gui::Command::Gui, "App.activeDocument().recompute()");
573  doc->resetEdit();
574 
575  return true;
576 }
