91 App::DocumentObjectExecReturn *DrawViewMulti::execute()
92 {
93  if (!keepUpdated()) {
94  return App::DocumentObject::StdReturn;
95  }
96 
97  const std::vector<App::DocumentObject*>& links = Sources.getValues();
98  if (links.empty()) {
99  Base::Console().Log("INFO - DVM::execute - No Sources - creation?\n");
100  return DrawView::execute();
101  }
102 
103  m_compound = TopoDS::Compound(getSourceShape());
104  if (m_compound.IsNull()) {
105  return new App::DocumentObjectExecReturn("DVP - Linked shape object(s) is invalid");
106  }
107  TopoDS_Compound comp = m_compound;
108 
109  gp_Pnt inputCenter;
110  try {
111  inputCenter = ShapeUtils::findCentroid(comp,
112  Direction.getValue());
113  shapeCentroid = Base::Vector3d(inputCenter.X(), inputCenter.Y(), inputCenter.Z());
114  TopoDS_Shape mirroredShape = ShapeUtils::mirrorShape(comp,
115  inputCenter,
116  getScale());
117  gp_Ax2 viewAxis = getViewAxis(Base::Vector3d(inputCenter.X(), inputCenter.Y(), inputCenter.Z()), Direction.getValue());
118  if (!DrawUtil::fpCompare(Rotation.getValue(), 0.0)) {
119  mirroredShape = ShapeUtils::rotateShape(mirroredShape,
120  viewAxis,
121  Rotation.getValue());
122  }
123  geometryObject = buildGeometryObject(mirroredShape, viewAxis);
124 
125 #if MOD_TECHDRAW_HANDLE_FACES
126  extractFaces();
127 #endif //#if MOD_TECHDRAW_HANDLE_FACES
128  }
129  catch (Standard_Failure& e1) {
130  return new App::DocumentObjectExecReturn(e1.GetMessageString());
131  }
132 
133  requestPaint();
134  return App::DocumentObject::StdReturn;
135 }
