73 void EditModeGeometryCoinManager::processGeometry(const GeoListFacade& geolistfacade)
74 {
75  // enable all layers
76  editModeScenegraphNodes.PointsGroup->enable.setNum(geometryLayerParameters.getCoinLayerCount());
77  editModeScenegraphNodes.CurvesGroup->enable.setNum(
78  geometryLayerParameters.getCoinLayerCount() * geometryLayerParameters.getSubLayerCount());
79  SbBool* swsp = editModeScenegraphNodes.PointsGroup->enable.startEditing();
80  SbBool* swsc = editModeScenegraphNodes.CurvesGroup->enable.startEditing();
81 
82  auto layersconfigurations = viewProvider.VisualLayerList.getValues();
83 
84  for (auto l = 0; l < geometryLayerParameters.getCoinLayerCount(); l++) {
85  auto enabled = layersconfigurations[l].isVisible();
86 
87  swsp[l] = enabled;
88  int slCount = geometryLayerParameters.getSubLayerCount();
89  for (int t = 0; t < slCount; t++) {
90  swsc[l * slCount + t] = enabled;
91  }
92  }
93 
94  editModeScenegraphNodes.PointsGroup->enable.finishEditing();
95  editModeScenegraphNodes.CurvesGroup->enable.finishEditing();
96 
97  // Define the coin nodes that will be filled in with the geometry layers
98  GeometryLayerNodes geometrylayernodes {editModeScenegraphNodes.PointsMaterials,
99  editModeScenegraphNodes.PointsCoordinate,
100  editModeScenegraphNodes.CurvesMaterials,
101  editModeScenegraphNodes.CurvesCoordinate,
102  editModeScenegraphNodes.CurveSet};
103 
104  // process geometry layers
105  EditModeGeometryCoinConverter gcconv(viewProvider,
106  geometrylayernodes,
107  drawingParameters,
108  geometryLayerParameters,
109  coinMapping);
110 
111  gcconv.convert(geolistfacade);
112 
113  // set cross coordinates
114  editModeScenegraphNodes.RootCrossSet->numVertices.set1Value(0, 2);
115  editModeScenegraphNodes.RootCrossSet->numVertices.set1Value(1, 2);
116 
117  analysisResults.combRepresentationScale = gcconv.getCombRepresentationScale();
118  analysisResults.boundingBoxMagnitudeOrder =
119  exp(ceil(log(std::abs(gcconv.getBoundingBoxMaxMagnitude()))));
120  analysisResults.bsplineGeoIds = gcconv.getBSplineGeoIds();
121  analysisResults.arcGeoIds = gcconv.getArcGeoIds();
122 }

124 void EditModeGeometryCoinManager::updateGeometryColor(const GeoListFacade& geolistfacade,
125  bool issketchinvalid)
126 {
127  // Lambdas for convenience retrieval of geometry information
128  auto isDefinedGeomPoint = [&geolistfacade](int GeoId, Sketcher::PointPos PosId) {
129  auto geom = geolistfacade.getGeometryFacadeFromGeoId(GeoId);
130  if (geom) {
131  bool isStartOrEnd =
132  PosId == Sketcher::PointPos::start || PosId == Sketcher::PointPos::end;
133  return isStartOrEnd && !geom->getConstruction();
134  }
135  return false;
136  };
137 
138  auto isCoincident = [&](int GeoId, Sketcher::PointPos PosId) {
139  const std::vector<Sketcher::Constraint*>& constraints =
140  ViewProviderSketchCoinAttorney::getConstraints(viewProvider);
141  for (auto& constr : constraints) {
142  if (constr->Type == Coincident
143  || (constr->Type == Tangent && constr->FirstPos != Sketcher::PointPos::none)
144  || (constr->Type == Perpendicular && constr->FirstPos != Sketcher::PointPos::none
145  && constr->SecondPos != Sketcher::PointPos::none)) {
146  if ((constr->First == GeoId && constr->FirstPos == PosId)
147  || (constr->Second == GeoId && constr->SecondPos == PosId)) {
148  return true;
149  }
150  }
151  }
152  return false;
153  };
154 
155  auto isInternalAlignedGeom = [&geolistfacade](int GeoId) {
156  auto geom = geolistfacade.getGeometryFacadeFromGeoId(GeoId);
157  if (geom) {
158  return geom->isInternalAligned();
159  }
160  return false;
161  };
162 
163  auto isFullyConstraintElement = [&geolistfacade](int GeoId) {
164  auto geom = geolistfacade.getGeometryFacadeFromGeoId(GeoId);
165 
166  if (geom) {
167  if (geom->hasExtension(Sketcher::SolverGeometryExtension::getClassTypeId())) {
168 
169  auto solvext = std::static_pointer_cast<const Sketcher::SolverGeometryExtension>(
170  geom->getExtension(Sketcher::SolverGeometryExtension::getClassTypeId()).lock());
171 
172  return (solvext->getGeometry()
173  == Sketcher::SolverGeometryExtension::FullyConstraint);
174  }
175  }
176  return false;
177  };
178 
179  bool sketchFullyConstrained =
180  ViewProviderSketchCoinAttorney::isSketchFullyConstrained(viewProvider);
181 
182  // Update Colors
183 
184  SbColor* crosscolor = editModeScenegraphNodes.RootCrossMaterials->diffuseColor.startEditing();
185  auto viewOrientationFactor =
186  ViewProviderSketchCoinAttorney::getViewOrientationFactor(viewProvider);
187 
188  for (auto l = 0; l < geometryLayerParameters.getCoinLayerCount(); l++) {
189  float x, y, z;
190  int PtNum = editModeScenegraphNodes.PointsMaterials[l]->diffuseColor.getNum();
191  SbColor* pcolor = editModeScenegraphNodes.PointsMaterials[l]->diffuseColor.startEditing();
192  SbVec3f* pverts = editModeScenegraphNodes.PointsCoordinate[l]->point.startEditing();
193 
194  // colors of the point set
195  for (int i = 0; i < PtNum; i++) {
196  int GeoId = coinMapping.getPointGeoId(i, l);
197  Sketcher::PointPos PosId = coinMapping.getPointPosId(i, l);
198  bool isExternal = GeoId < -1;
199 
200  if (isExternal) {
201  if (isCoincident(GeoId, PosId) && !issketchinvalid) {
202  pcolor[i] = drawingParameters.ConstrIcoColor;
203  }
204  else {
205  pcolor[i] = drawingParameters.CurveExternalColor;
206  }
207  }
208  else if (issketchinvalid) {
209  pcolor[i] = drawingParameters.InvalidSketchColor;
210  }
211  else if (!(i == 0 && l == 0) && sketchFullyConstrained) {
212  // root point is not coloured nor external
213  pcolor[i] = drawingParameters.FullyConstrainedColor;
214  }
215  else {
216  bool constrainedElement = isFullyConstraintElement(GeoId);
217 
218  if (isInternalAlignedGeom(GeoId)) {
219  if (constrainedElement) {
220  pcolor[i] = drawingParameters.FullyConstraintInternalAlignmentColor;
221  }
222  else {
223  if (isCoincident(GeoId, PosId)) {
224  pcolor[i] = drawingParameters.ConstrIcoColor;
225  }
226  else {
227  pcolor[i] = drawingParameters.InternalAlignedGeoColor;
228  }
229  }
230  }
231  else {
232  if (!isDefinedGeomPoint(GeoId, PosId)) {
233  if (constrainedElement) {
234  pcolor[i] = drawingParameters.FullyConstraintConstructionElementColor;
235  }
236  else {
237  if (isCoincident(GeoId, PosId)) {
238  pcolor[i] = drawingParameters.ConstrIcoColor;
239  }
240  else {
241  pcolor[i] = drawingParameters.CurveDraftColor;
242  }
243  }
244  }
245  else { // this is a defined GeomPoint
246  if (constrainedElement) {
247  pcolor[i] = drawingParameters.FullyConstraintElementColor;
248  }
249  else {
250  if (isCoincident(GeoId, PosId)) {
251  pcolor[i] = drawingParameters.ConstrIcoColor;
252  }
253  else {
254  pcolor[i] = drawingParameters.CurveColor;
255  }
256  }
257  }
258  }
259  }
260  }
261 
262  // update rendering height of points
263 
264  auto getRenderHeight = [this](DrawingParameters::GeometryRendering renderingtype,
265  float toprendering,
266  float midrendering,
267  float lowrendering) {
268  if (drawingParameters.topRenderingGeometry == renderingtype) {
269  return toprendering;
270  }
271  else if (drawingParameters.midRenderingGeometry == renderingtype) {
272  return midrendering;
273  }
274  else {
275  return lowrendering;
276  }
277  };
278 
279  float zNormPoint = getRenderHeight(DrawingParameters::GeometryRendering::NormalGeometry,
280  drawingParameters.zHighPoints,
281  drawingParameters.zMidPoints,
282  drawingParameters.zMidPoints);
283 
284  float zConstrPoint = getRenderHeight(DrawingParameters::GeometryRendering::Construction,
285  drawingParameters.zHighPoints,
286  drawingParameters.zMidPoints,
287  drawingParameters.zMidPoints);
288 
289  for (int i = 0; i < PtNum; i++) { // 0 is the origin
290  if (i == 0 && l == 0) { // reset root point to lowest
291  pverts[i].setValue(0, 0, viewOrientationFactor * drawingParameters.zRootPoint);
292  }
293  else {
294  int GeoId = coinMapping.getPointGeoId(i, l);
295  Sketcher::PointPos PosId = coinMapping.getPointPosId(i, l);
296  pverts[i].getValue(x, y, z);
297  auto geom = geolistfacade.getGeometryFacadeFromGeoId(GeoId);
298  bool isExternal = GeoId < -1;
299 
300  if (geom) {
301  z = viewOrientationFactor * zNormPoint;
302 
303  if (isCoincident(GeoId, PosId)) {
304  z = viewOrientationFactor * drawingParameters.zLowPoints;
305  }
306  else {
307  if (isExternal || isInternalAlignedGeom(GeoId)) {
308  z = viewOrientationFactor * drawingParameters.zRootPoint;
309  }
310  else if (geom->getConstruction()) {
311  z = viewOrientationFactor * zConstrPoint;
312  }
313  }
314  pverts[i].setValue(x, y, z);
315  }
316  }
317  }
318 
319  auto preselectpoint = ViewProviderSketchCoinAttorney::getPreselectPoint(viewProvider);
320  auto preselectcross = ViewProviderSketchCoinAttorney::getPreselectCross(viewProvider);
321  auto preselectcurve = ViewProviderSketchCoinAttorney::getPreselectCurve(viewProvider);
322 
323  auto raisePoint = [](SbVec3f& point, float height) {
324  float x, y, z;
325  point.getValue(x, y, z);
326  point.setValue(x, y, height);
327  };
328 
329  MultiFieldId preselectpointmfid;
330 
331  if (preselectcross == 0) {
332  if (l == 0) { // cross only in layer 0
333  pcolor[0] = drawingParameters.PreselectColor;
334  }
335  }
336  else if (preselectpoint != -1) {
337  preselectpointmfid = coinMapping.getIndexLayer(preselectpoint);
338  if (MultiFieldId::Invalid != preselectpointmfid && preselectpointmfid.layerId == l
339  && preselectpointmfid.fieldIndex < PtNum) {
340 
341  pcolor[preselectpointmfid.fieldIndex] = drawingParameters.PreselectColor;
342 
343  raisePoint(pverts[preselectpointmfid.fieldIndex],
344  viewOrientationFactor * drawingParameters.zHighlight);
345  }
346  }
347 
348  ViewProviderSketchCoinAttorney::executeOnSelectionPointSet(
349  viewProvider,
350  [pcolor,
351  pverts,
352  PtNum,
353  preselectpointmfid,
354  layerId = l,
355  &coinMapping = coinMapping,
356  drawingParameters = this->drawingParameters,
357  raisePoint,
358  viewOrientationFactor](const int i) {
359  auto pointindex = coinMapping.getIndexLayer(i);
360  if (layerId == pointindex.layerId && pointindex.fieldIndex >= 0
361  && pointindex.fieldIndex < PtNum) {
362  pcolor[pointindex.fieldIndex] = (preselectpointmfid == pointindex)
363  ? drawingParameters.PreselectSelectedColor
364  : drawingParameters.SelectColor;
365 
366  raisePoint(pverts[pointindex.fieldIndex],
367  viewOrientationFactor * drawingParameters.zHighlight);
368  }
369  });
370 
371  // update colors and rendering height of the curves
372 
373  float zNormLine = getRenderHeight(DrawingParameters::GeometryRendering::NormalGeometry,
374  drawingParameters.zHighLines,
375  drawingParameters.zMidLines,
376  drawingParameters.zLowLines);
377 
378  float zConstrLine = getRenderHeight(DrawingParameters::GeometryRendering::Construction,
379  drawingParameters.zHighLines,
380  drawingParameters.zMidLines,
381  drawingParameters.zLowLines);
382 
383  float zExtLine = getRenderHeight(DrawingParameters::GeometryRendering::ExternalGeometry,
384  drawingParameters.zHighLines,
385  drawingParameters.zMidLines,
386  drawingParameters.zLowLines);
387 
388  for (auto t = 0; t < geometryLayerParameters.getSubLayerCount(); t++) {
389  int CurvNum = editModeScenegraphNodes.CurvesMaterials[l][t]->diffuseColor.getNum();
390  SbColor* color =
391  editModeScenegraphNodes.CurvesMaterials[l][t]->diffuseColor.startEditing();
392  SbVec3f* verts = editModeScenegraphNodes.CurvesCoordinate[l][t]->point.startEditing();
393 
394  int j = 0; // vertexindex
395  for (int i = 0; i < CurvNum; i++) {
396  int GeoId = coinMapping.getCurveGeoId(i, l, t);
397  // CurvId has several vertices associated to 1 material
398  // edit->CurveSet->numVertices => [i] indicates number of vertex for line i.
399  int indexes = (editModeScenegraphNodes.CurveSet[l][t]->numVertices[i]);
400 
401  bool selected =
402  ViewProviderSketchCoinAttorney::isCurveSelected(viewProvider, GeoId);
403  bool preselected = (preselectcurve == GeoId);
404  bool constrainedElement = isFullyConstraintElement(GeoId);
405  bool isExternal = GeoId < -1;
406 
407  if (selected || preselected) {
408  color[i] = selected ? (preselected ? drawingParameters.PreselectSelectedColor
409  : drawingParameters.SelectColor)
410  : drawingParameters.PreselectColor;
411 
412  for (int k = j; j < k + indexes; j++) {
413  verts[j].getValue(x, y, z);
414  verts[j] =
415  SbVec3f(x, y, viewOrientationFactor * drawingParameters.zHighLine);
416  }
417  }
418  else if (isExternal) {
419  auto geom = geolistfacade.getGeometryFacadeFromGeoId(GeoId);
420  auto egf = ExternalGeometryFacade::getFacade(geom->clone());
421  auto ref = egf->getRef();
422  if (egf->testFlag(ExternalGeometryExtension::Missing)) {
423  color[i] = drawingParameters.InvalidSketchColor;
424  }
425  else {
426  color[i] = drawingParameters.CurveExternalColor;
427  }
428  for (int k = j; j < k + indexes; j++) {
429  verts[j].getValue(x, y, z);
430  verts[j] = SbVec3f(x, y, viewOrientationFactor * zExtLine);
431  }
432  }
433  else {
434  if (issketchinvalid) {
435  color[i] = drawingParameters.InvalidSketchColor;
436 
437  for (int k = j; j < k + indexes; j++) {
438  verts[j].getValue(x, y, z);
439  verts[j] = SbVec3f(x, y, viewOrientationFactor * zNormLine);
440  }
441  }
442  else if (geometryLayerParameters.isConstructionSubLayer(t)) {
443  if (constrainedElement) {
444  color[i] = drawingParameters.FullyConstraintConstructionElementColor;
445  }
446  else {
447  color[i] = drawingParameters.CurveDraftColor;
448  }
449 
450  for (int k = j; j < k + indexes; j++) {
451  verts[j].getValue(x, y, z);
452  verts[j] = SbVec3f(x, y, viewOrientationFactor * zConstrLine);
453  }
454  }
455  else if (geometryLayerParameters.isInternalSubLayer(t)) {
456  if (constrainedElement) {
457  color[i] = drawingParameters.FullyConstraintInternalAlignmentColor;
458  }
459  else {
460  color[i] = drawingParameters.InternalAlignedGeoColor;
461  }
462 
463  for (int k = j; j < k + indexes; j++) {
464  verts[j].getValue(x, y, z);
465  verts[j] = SbVec3f(x, y, viewOrientationFactor * zConstrLine);
466  }
467  }
468  else {
469  if (sketchFullyConstrained) {
470  color[i] = drawingParameters.FullyConstrainedColor;
471  }
472  else if (constrainedElement) {
473  color[i] = drawingParameters.FullyConstraintElementColor;
474  }
475  else {
476  color[i] = drawingParameters.CurveColor;
477  }
478 
479  for (int k = j; j < k + indexes; j++) {
480  verts[j].getValue(x, y, z);
481  verts[j] = SbVec3f(x, y, viewOrientationFactor * zNormLine);
482  }
483  }
484  }
485  }
486 
487  editModeScenegraphNodes.CurvesMaterials[l][t]->diffuseColor.finishEditing();
488  editModeScenegraphNodes.CurvesCoordinate[l][t]->point.finishEditing();
489  editModeScenegraphNodes.CurveSet[l][t]->numVertices.finishEditing();
490  }
491 
492  // colors of the cross
493  if (l == 0) { // only in layer 0
494  if (ViewProviderSketchCoinAttorney::isCurveSelected(viewProvider,
495  Sketcher::GeoEnum::HAxis)) {
496  crosscolor[0] = drawingParameters.SelectColor;
497  }
498  else if (preselectcross == 1) { // cross only in layer 0
499  crosscolor[0] = drawingParameters.PreselectColor;
500  }
501  else {
502  crosscolor[0] = drawingParameters.CrossColorH;
503  }
504 
505  if (ViewProviderSketchCoinAttorney::isCurveSelected(viewProvider,
506  Sketcher::GeoEnum::VAxis)) {
507  crosscolor[1] = drawingParameters.SelectColor;
508  }
509  else if (preselectcross == 2) {
510  crosscolor[1] = drawingParameters.PreselectColor;
511  }
512  else {
513  crosscolor[1] = drawingParameters.CrossColorV;
514  }
515  }
516 
517  editModeScenegraphNodes.PointsMaterials[l]->diffuseColor.finishEditing();
518  }
519 
520  editModeScenegraphNodes.RootCrossMaterials->diffuseColor.finishEditing();
521 }
