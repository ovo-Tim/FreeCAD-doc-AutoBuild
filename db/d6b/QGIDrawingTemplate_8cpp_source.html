68 void QGIDrawingTemplate::draw()
69 {
70 
71  TechDraw::DrawParametricTemplate *tmplte = getParametricTemplate();
72  if(!tmplte) {
73  throw Base::RuntimeError("Template Feuature not set for QGIDrawingTemplate");
74  }
75 
76 
77  // Clear the previous geometry stored
78 
79  // Get a list of geometry and iterate
80  const TechDraw::BaseGeomPtrVector &geoms = tmplte->getGeometry();
81 
82  TechDraw::BaseGeomPtrVector::const_iterator it = geoms.begin();
83 
84  QPainterPath path;
85 
86  // Draw Edges
87  // iterate through all the geometries
88  for(; it != geoms.end(); ++it) {
89  switch((*it)->getGeomType()) {
90  case TechDraw::GENERIC: {
91 
92  TechDraw::GenericPtr geom = std::static_pointer_cast<TechDraw::Generic>(*it);
93 
94  path.moveTo(geom->points[0].x, geom->points[0].y);
95  std::vector<Base::Vector3d>::const_iterator it = geom->points.begin();
96 
97  for(++it; it != geom->points.end(); ++it) {
98  path.lineTo((*it).x, (*it).y);
99  }
100  break;
101  }
102  default:
103  break;
104  }
105  }
106 
107  pathItem->setPath(path);
108 }
