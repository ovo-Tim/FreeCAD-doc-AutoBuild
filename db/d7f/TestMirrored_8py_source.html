27 class TestMirrored(unittest.TestCase):
28  def setUp(self):
29  self.Doc = FreeCAD.newDocument("PartDesignTestMirrored")

31  def testMirroredSketchCase(self):
32  """
33  Creates a unit cube cornered at the origin and mirrors it about the Y axis.
34  This operation should create a rectangular prism with volume 2.0.
35  """
36  self.Body = self.Doc.addObject('PartDesign::Body','Body')
37  self.Rect = self.Doc.addObject('Sketcher::SketchObject','Rect')
38  self.Body.addObject(self.Rect)
39  TestSketcherApp.CreateRectangleSketch(self.Rect, (0, 0), (1, 1))
40  self.Doc.recompute()
41  self.Pad = self.Doc.addObject("PartDesign::Pad","Pad")
42  self.Pad.Profile = self.Rect
43  self.Pad.Length = 1
44  self.Body.addObject(self.Pad)
45  self.Doc.recompute()
46  self.Mirrored = self.Doc.addObject("PartDesign::Mirrored","Mirrored")
47  self.Mirrored.Originals = [self.Pad]
48  self.Mirrored.MirrorPlane = (self.Rect, ["V_Axis"])
49  self.Body.addObject(self.Mirrored)
50  self.Doc.recompute()
51  self.assertAlmostEqual(self.Mirrored.Shape.Volume, 2.0)

53  def testMirroredPrimitiveCase(self):
54  """
55  Tests the same mirroring scenario as in the sketch case,
56  but is designed to ensure the same end result occurs with

59  self.Body = self.Doc.addObject('PartDesign::Body','Body')
60  self.Box = self.Doc.addObject('PartDesign::AdditiveBox','Box')

63  self.Box.Height=1

67  self.Mirrored.Originals = [self.Box]

69  self.Body.addObject(self.Mirrored)
