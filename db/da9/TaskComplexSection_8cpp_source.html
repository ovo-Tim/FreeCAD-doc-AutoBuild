135 void TaskComplexSection::setUiPrimary()

138  if (m_baseView) {

145  }
146  ui->cmbStrategy->setCurrentIndex(0);
147 
148  setUiCommon();
149 
150  if (m_baseView) {
151  ui->leBaseView->setText(QString::fromStdString(m_baseView->getNameInDocument()));
152  //if there is a baseView, we don't know the sectionNormal yet and have to wait until
153  //one is picked in the dialog
154  Base::Vector3d defaultNormal(-1.0, 0.0, 0.0);
155  m_saveNormal = defaultNormal;
156  m_saveXDir = Base::Vector3d(0.0, 1.0, 0.0);
157  ui->leBaseView->setText(QString::fromStdString(m_baseView->getNameInDocument()));
158  m_compass->setDialAngle(0.0);
159  m_viewDirectionWidget->setValueNoNotify(Base::Vector3d(1.0, 0.0, 0.0));
160  }
161  else {
162  //if there is no baseView, we use the 3d view to determine the SectionNormal
163  //and XDirection.
164  std::pair<Base::Vector3d, Base::Vector3d> dirs = DrawGuiUtil::get3DDirAndRot();
165  m_saveNormal = dirs.first;
166  m_saveXDir = dirs.second;
167  m_viewDirectionWidget->setValue(m_saveNormal * -1.0);//this will propagate to m_compass
168  }
169 
170  //don't allow updates until a direction is picked
171  ui->pbUpdateNow->setEnabled(false);
172  ui->cbLiveUpdate->setEnabled(false);
173  ui->lPendingUpdates->setText(tr("No direction set"));
174 }

479 bool TaskComplexSection::apply(bool forceUpdate)
480 {
481  // Base::Console().Message("TCS::apply() - liveUpdate: %d force: %d\n",
482  // ui->cbLiveUpdate->isChecked(), forceUpdate);
483  if (!ui->cbLiveUpdate->isChecked() && !forceUpdate) {
484  //nothing to do
485  m_applyDeferred++;
486  QString msgLiteral =
487  QString::fromUtf8(QT_TRANSLATE_NOOP("TaskPojGroup", " updates pending"));
488  QString msgNumber = QString::number(m_applyDeferred);
489  ui->lPendingUpdates->setText(msgNumber + msgLiteral);
490  return false;
491  }
492 
493  Base::Vector3d localUnit = m_viewDirectionWidget->value();
494  if (m_baseView) {
495  if (!DrawComplexSection::canBuild(m_baseView->localVectorToCS(localUnit),
496  m_profileObject)) {
497  Base::Console().Error(
498  "Can not build Complex Section with this profile and direction (1)\n");
499  return false;
500  }
501  }
502  else {
503  gp_Pnt stdOrigin(0.0, 0.0, 0.0);
504  gp_Ax2 sectionCS(stdOrigin, DrawUtil::togp_Dir(m_saveNormal),
505  DrawUtil::togp_Dir(m_saveXDir));
506  if (!DrawComplexSection::canBuild(sectionCS, m_profileObject)) {
507  Base::Console().Error(
508  "Can not build Complex Section with this profile and direction (2)\n");
509  return false;
510  }
511  }
512 
513  Gui::WaitCursor wc;
514  m_modelIsDirty = true;
515 
516  if (!m_section) {
517  createComplexSection();
518  }
519 
520  if (isSectionValid()) {
521  updateComplexSection();
522  }
523  else {
524  failNoObject();
525  }
526 
527  m_section->recomputeFeature();
528  if (isBaseValid()) {
529  m_baseView->requestPaint();
530  }
531 
532  enableAll(true);
533  checkAll(false);
534 
535  wc.restoreCursor();
536  m_applyDeferred = 0;
537  ui->lPendingUpdates->setText(QString());
538  return true;
539 }

554 //pointer to created view is not returned, but stored in m_section
555 void TaskComplexSection::createComplexSection()
556 {
557  // Base::Console().Message("TCS::createComplexSection()\n");
558 
559  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create ComplexSection"));
560  if (!m_section) {
561  const std::string objectName{QT_TR_NOOP("ComplexSection")};
562  m_sectionName = m_page->getDocument()->getUniqueObjectName(objectName.c_str());
563  Command::doCommand(Command::Doc, "App.ActiveDocument.addObject('TechDraw::DrawComplexSection', '%s')",
564  m_sectionName.c_str());
565 
566  // section labels (Section A-A) are not unique, and are not the same as the object name (SectionView)
567  // we pluck the generated suffix from the object name and append it to "Section" to generate
568  // unique Labels
569  QString qTemp = ui->leSymbol->text();
570  std::string temp = qTemp.toStdString();
571  Command::doCommand(Command::Doc, "App.ActiveDocument.%s.SectionSymbol = '%s'",
572  m_sectionName.c_str(), temp.c_str());
573 
574  Command::doCommand(Command::Doc, "App.ActiveDocument.%s.Label = '%s'",
575  m_sectionName.c_str(),
576  makeSectionLabel(qTemp).c_str());
577  Command::doCommand(Command::Doc, "App.ActiveDocument.%s.addView(App.ActiveDocument.%s)",
578  m_page->getNameInDocument(), m_sectionName.c_str());
579 
580  Command::doCommand(Command::Doc, "App.ActiveDocument.%s.Scale = %0.6f",
581  m_sectionName.c_str(), ui->sbScale->value());
582 
583  std::string baseName = m_baseView->getNameInDocument();
584 
585  Command::doCommand(Command::Doc, "App.ActiveDocument.%s.Scale = %0.7f",
586  m_sectionName.c_str(), ui->sbScale->value());
587 
588  int scaleType = ui->cmbScaleType->currentIndex();
589  Command::doCommand(Command::Doc, "App.ActiveDocument.%s.ScaleType = %d",
590  m_sectionName.c_str(), scaleType);
591  int projectionStrategy = ui->cmbStrategy->currentIndex();
592  Command::doCommand(Command::Doc, "App.ActiveDocument.%s.ProjectionStrategy = %d",
593  m_sectionName.c_str(), projectionStrategy);
594 
595  Command::doCommand(Command::Doc,
596  "App.activeDocument().%s.SectionOrigin = FreeCAD.Vector(0.0, 0.0, 0.0)",
597  m_sectionName.c_str());
598  Command::doCommand(Command::Doc, "App.activeDocument().%s.SectionDirection = 'Aligned'",
599  m_sectionName.c_str());
600 
601  App::DocumentObject* newObj = m_page->getDocument()->getObject(m_sectionName.c_str());
602  m_section = dynamic_cast<TechDraw::DrawComplexSection*>(newObj);
603  if (!newObj || !m_section) {
604  throw Base::RuntimeError("TaskComplexSection - new section object not found");
605  }
606  Base::Vector3d localUnit = m_viewDirectionWidget->value();
607  if (m_baseView) {
608  Command::doCommand(Command::Doc,
609  "App.ActiveDocument.%s.BaseView = App.ActiveDocument.%s",
610  m_sectionName.c_str(), m_baseView->getNameInDocument());
611  m_section->setCSFromBase(localUnit * -1.0);
612  m_section->Source.setValues(m_baseView->Source.getValues());
613  m_section->XSource.setValues(m_baseView->XSource.getValues());
614  }
615  else {
616  if (m_directionIsSet) {
