108 const SoEvent *
109 Mouse::translateEvent(QEvent * event)
110 {
111  switch (event->type()) {
112  case QEvent::MouseMove:
113  return PRIVATE(this)->mouseMoveEvent((QMouseEvent *) event);
114  case QEvent::MouseButtonPress:
115  case QEvent::MouseButtonRelease:
116  // a dblclick event comes in a series of press, release, dblclick,
117  // release, so we can simply treat it as an ordinary press event.
118  // -mortene.
119  case QEvent::MouseButtonDblClick:
120  return PRIVATE(this)->mouseButtonEvent((QMouseEvent *) event);
121  case QEvent::Wheel:
122  return PRIVATE(this)->mouseWheelEvent((QWheelEvent *) event);
123  case QEvent::Resize:
124  PRIVATE(this)->resizeEvent((QResizeEvent *) event);
125  return nullptr;
126  default:
127  return nullptr;
128  }
129 }

138 const SoEvent *
139 MouseP::mouseMoveEvent(QMouseEvent * event)
140 {
141  PUBLIC(this)->setModifiers(this->location2, event);
142 
143  assert(this->windowsize[1] != -1);
144  SbVec2s pos(event->pos().x(), this->windowsize[1] - event->pos().y() - 1);
145  // the following corrects for high-dpi displays (e.g. mac retina)
146  pos *= publ->quarter->devicePixelRatio();

152 const SoEvent *
153 MouseP::mouseWheelEvent(QWheelEvent * event)
154 {
155  PUBLIC(this)->setModifiers(this->wheel, event);
156 #if QT_VERSION >= QT_VERSION_CHECK(5,15,0)
157  QPoint pnt = event->position().toPoint();
158  SbVec2s pos(pnt.x(), PUBLIC(this)->windowsize[1] - pnt.y() - 1);
159 #else
160  SbVec2s pos(event->pos().x(), PUBLIC(this)->windowsize[1] - event->pos().y() - 1);
161 #endif
162  // the following corrects for high-dpi displays (e.g. mac retina)
163  pos *= publ->quarter->devicePixelRatio();
164  this->location2->setPosition(pos); //I don't know why location2 is assigned here, I assumed it important --DeepSOIC
165  this->wheel->setPosition(pos);
166 
167  // QWheelEvent::delta() returns the distance that the wheel is
168  // rotated, in eights of a degree. A positive value indicates that
169  // the wheel was rotated forwards away from the user; a negative
170  // value indicates that the wheel was rotated backwards toward the
171  // user. A typical wheel click is 120, but values coming from touchpad
172  // can be a lot lower
173  this->wheel->setDelta(event->angleDelta().y());
174 
175  return this->wheel;
176 }

178 const SoEvent *
179 MouseP::mouseButtonEvent(QMouseEvent * event)
180 {
181  PUBLIC(this)->setModifiers(this->mousebutton, event);
182  SbVec2s pos(event->pos().x(), PUBLIC(this)->windowsize[1] - event->pos().y() - 1);
183  // the following corrects for high-dpi displays (e.g. mac retina)
184  pos *= publ->quarter->devicePixelRatio();
185  this->location2->setPosition(pos);
186  this->mousebutton->setPosition(pos);
187 
188  ((event->type() == QEvent::MouseButtonPress) ||
189  (event->type() == QEvent::MouseButtonDblClick)) ?
190  this->mousebutton->setState(SoButtonEvent::DOWN):
191  this->mousebutton->setState(SoButtonEvent::UP);
192 
193  switch (event->button()) {
194  case Qt::LeftButton:
195  this->mousebutton->setButton(SoMouseButtonEvent::BUTTON1);
196  break;
197  case Qt::RightButton:
198  this->mousebutton->setButton(SoMouseButtonEvent::BUTTON2);
199  break;
200  case Qt::MiddleButton:
201  this->mousebutton->setButton(SoMouseButtonEvent::BUTTON3);
202  break;
203  default:
204  this->mousebutton->setButton(SoMouseButtonEvent::ANY);
205  SoDebugError::postInfo("Mouse::mouseButtonEvent",
206  "Unhandled ButtonState = %x", event->button());
207  break;
208  }
209  return this->mousebutton;
210 }
