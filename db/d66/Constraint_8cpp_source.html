63  // Initialize a random number generator, to avoid Valgrind false positives.

67  static bool seeded = false;
68  static boost::mutex random_number_mutex;

71 
72  if (!seeded) {
73  ran.seed(QDateTime::currentMSecsSinceEpoch() & 0xffffffff);
74  seeded = true;
75  }
76  static boost::uuids::basic_random_generator<boost::mt19937> gen(&ran);

114 Quantity Constraint::getPresentationValue() const
115 {
116  Quantity quantity;
117  switch (Type) {
118  case Distance:
119  case Radius:
120  case Diameter:
121  case DistanceX:
122  case DistanceY:
123  quantity.setValue(Value);
124  quantity.setUnit(Unit::Length);
125  break;
126  case Angle:
127  quantity.setValue(toDegrees<double>(Value));
128  quantity.setUnit(Unit::Angle);
129  break;
130  case SnellsLaw:
131  case Weight:
132  quantity.setValue(Value);
133  break;
134  default:
135  quantity.setValue(Value);
136  break;
137  }
138 
139  QuantityFormat format = quantity.getFormat();
140  format.option = QuantityFormat::None;
141  format.format = QuantityFormat::Default;
142  format.precision = 6; // QString's default
143  quantity.setFormat(format);
144  return quantity;
145 }

178 void Constraint::Restore(XMLReader& reader)
179 {
180  reader.readElement("Constrain");
181  Name = reader.getAttribute("Name");
182  Type = static_cast<ConstraintType>(reader.getAttributeAsInteger("Type"));
183  Value = reader.getAttributeAsFloat("Value");
184  First = reader.getAttributeAsInteger("First");
185  FirstPos = static_cast<PointPos>(reader.getAttributeAsInteger("FirstPos"));
186  Second = reader.getAttributeAsInteger("Second");
187  SecondPos = static_cast<PointPos>(reader.getAttributeAsInteger("SecondPos"));
188 
189  if (this->Type == InternalAlignment) {
190  AlignmentType = static_cast<InternalAlignmentType>(
191  reader.getAttributeAsInteger("InternalAlignmentType"));
192 
193  if (reader.hasAttribute("InternalAlignmentIndex")) {
194  InternalAlignmentIndex = reader.getAttributeAsInteger("InternalAlignmentIndex");
195  }
196  }
197  else {
198  AlignmentType = Undef;
199  }
200 
201  // read the third geo group if present
202  if (reader.hasAttribute("Third")) {
203  Third = reader.getAttributeAsInteger("Third");
204  ThirdPos = static_cast<PointPos>(reader.getAttributeAsInteger("ThirdPos"));
205  }
206 
207  // Read the distance a constraint label has been moved
208  if (reader.hasAttribute("LabelDistance")) {
209  LabelDistance = (float)reader.getAttributeAsFloat("LabelDistance");
210  }
211 
212  if (reader.hasAttribute("LabelPosition")) {
213  LabelPosition = (float)reader.getAttributeAsFloat("LabelPosition");
214  }
215 
216  if (reader.hasAttribute("IsDriving")) {
217  isDriving = reader.getAttributeAsInteger("IsDriving") ? true : false;
218  }
219 
220  if (reader.hasAttribute("IsInVirtualSpace")) {
221  isInVirtualSpace = reader.getAttributeAsInteger("IsInVirtualSpace") ? true : false;
222  }
223 
224  if (reader.hasAttribute("IsActive")) {
225  isActive = reader.getAttributeAsInteger("IsActive") ? true : false;
226  }
