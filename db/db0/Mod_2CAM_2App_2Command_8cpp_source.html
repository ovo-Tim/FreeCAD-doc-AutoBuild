97 std::string Command::toGCode(int precision, bool padzero) const
98 {
99  std::stringstream str;
100  str.fill('0');
101  str << Name;
102  if (precision < 0) {
103  precision = 0;
104  }
105  double scale = std::pow(10.0, precision + 1);
106  std::int64_t iscale = static_cast<std::int64_t>(scale) / 10;
107  for (std::map<std::string, double>::const_iterator i = Parameters.begin();
108  i != Parameters.end();
109  ++i) {
110  if (i->first == "N") {
111  continue;
112  }
113 
114  str << " " << i->first;
115 
116  std::int64_t v = static_cast<std::int64_t>(i->second * scale);
117  if (v < 0) {
118  v = -v;
119  str << '-'; // shall we allow -0 ?
120  }
121  v += 5;
122  v /= 10;
123  str << (v / iscale);
124  if (!precision) {
125  continue;
126  }
127 
128  int width = precision;
129  std::int64_t digits = v % iscale;
130  if (!padzero) {
131  if (!digits) {
132  continue;
133  }
134  while (digits % 10 == 0) {
135  digits /= 10;
136  --width;
137  }
138  }
139  str << '.' << std::setw(width) << std::right << digits;
140  }
141  return str.str();
142 }

144 void Command::setFromGCode(const std::string& str)
145 {
146  Parameters.clear();
147  std::string mode = "none";
148  std::string key;
149  std::string value;
150  for (unsigned int i = 0; i < str.size(); i++) {
151  if ((isdigit(str[i])) || (str[i] == '-') || (str[i] == '.')) {
152  value += str[i];
153  }
154  else if (isalpha(str[i])) {
155  if (mode == "command") {
156  if (!key.empty() && !value.empty()) {
157  std::string cmd = key + value;
158  boost::to_upper(cmd);
159  Name = cmd;
160  key = "";
161  value = "";
162  mode = "argument";
163  }
164  else {
165  throw Base::BadFormatError("Badly formatted GCode command");
166  }
167  mode = "argument";
168  }
169  else if (mode == "none") {
170  mode = "command";
171  }
172  else if (mode == "argument") {
173  if (!key.empty() && !value.empty()) {
174  double val = std::atof(value.c_str());
175  boost::to_upper(key);
176  Parameters[key] = val;
177  key = "";
178  value = "";
179  }
180  else {
181  throw Base::BadFormatError("Badly formatted GCode argument");
182  }
183  }
184  else if (mode == "comment") {
185  value += str[i];
186  }
187  key = str[i];
188  }
189  else if (str[i] == '(') {
190  mode = "comment";
191  }
192  else if (str[i] == ')') {
193  key = "(";
194  value += ")";
195  }
196  else {
197  // add non-ascii characters only if this is a comment
198  if (mode == "comment") {
199  value += str[i];
200  }
201  }
202  }
203  if (!key.empty() && !value.empty()) {
204  if ((mode == "command") || (mode == "comment")) {
205  std::string cmd = key + value;
206  if (mode == "command") {
207  boost::to_upper(cmd);
208  }
209  Name = cmd;
210  }
211  else {
212  double val = std::atof(value.c_str());
213  boost::to_upper(key);
214  Parameters[key] = val;
215  }
216  }
217  else {
218  throw Base::BadFormatError("Badly formatted GCode argument");
219  }
220 }
