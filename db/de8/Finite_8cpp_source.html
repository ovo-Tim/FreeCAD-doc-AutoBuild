23  return 0;
24  }

47 int LineLineIntof(const Span& sp0, const Span& sp1, Point& p, double t[2])
48 {
49  // intersection between 2 Line2d
50  // returns 0 for no intersection in range of either span
51  // returns 1 for intersction in range of both spans
52  // t[0] is parameter on sp0,
53  // t[1] is parameter on sp1
54  Vector2d v0(sp0.p0, sp0.p1);
55  Vector2d v1(sp1.p0, sp1.p1);
56  Vector2d v2(sp0.p0, sp1.p0);
57 
58  double cp = v1 ^ v0;
59 
60  if (fabs(cp) < UNIT_VECTOR_TOLERANCE) {
61  p = INVALID_POINT;
62  return 0; // parallel or degenerate lines
63  }
64 
65  t[0] = (v1 ^ v2) / cp;
66  p = v0 * t[0] + sp0.p0;
67  p.ok = true;
68  double toler = geoff_geometry::TOLERANCE / sp0.length; // calc a parametric tolerance
69 
70  t[1] = (v0 ^ v2) / cp;
71  if (t[0] < -toler || t[0] > 1 + toler) { // intersection on first?
72  return 0;
73  }
74  toler = geoff_geometry::TOLERANCE / sp1.length; // calc a parametric tolerance
75  if (t[1] < -toler || t[1] > 1 + toler) { // intersection on second?
76  return 0;
77  }
78  return 1;
79 }

81 int LineArcIntof(const Span& line, const Span& arc, Point& p0, Point& p1, double t[4])
82 {
83  // inters of line arc
84  // solving x = x0 + dx * t x = y0 + dy * t
85  // x = xc + R * cos(a) y = yc + R * sin(a) for t
86  // gives :- t² (dx² + dy²) + 2t(dx*dx0 + dy*dy0) + (x0-xc)² + (y0-yc)² - R² = 0
87  int nRoots;
88  Vector2d v0(arc.pc, line.p0);
89  Vector2d v1(line.p0, line.p1);
90  double s = v1.magnitudesqd();
91 
92  p0.ok = p1.ok = false;
93  if ((nRoots =
94  quadratic(s, 2 * (v0 * v1), v0.magnitudesqd() - arc.radius * arc.radius, t[0], t[1]))
95  != 0) {
96  double toler = geoff_geometry::TOLERANCE / sqrt(s); // calc a parametric tolerance
97  if (t[0] > -toler && t[0] < 1 + toler) {
98  p0 = v1 * t[0] + line.p0;
99  p0.ok = arc.OnSpan(p0, &t[2]);
100  }
101  if (nRoots == 2) {
102  if (t[1] > -toler && t[1] < 1 + toler) {
103  p1 = v1 * t[1] + line.p0;
104  p1.ok = arc.OnSpan(p1, &t[3]);
105  }
106  }
107  if (!p0.ok && p1.ok) {
108  p0 = p1;
109  p1.ok = false;
110  }
111  nRoots = (int)p0.ok + (int)p1.ok;
112  }
113  return nRoots;
114 }

251 bool Line::Shortest(const Line& l2, Line& lshort, double& t1, double& t2) const
252 {
253  /*
254  Calculate the line segment PaPb that is the shortest route between
255  two lines P1P2 and P3P4. Calculate also the values of mua and mub where
256  Pa = P1 + t1 (P2 - P1)
257  Pb = P3 + t2 (P4 - P3)
258  Return FALSE if no solution exists. P Bourke method.
259  Input this 1st line
260  Input l2 2nd line
261  Output lshort shortest line between lines (if !lshort.ok, the line intersect at a point
262  lshort.p0) Output t1 parameter at intersection on 1st Line Output t2 parameter at intersection
263  on 2nd Line
264 
265  */
266  Vector3d v13(l2.p0, this->p0);
267  if (!this->ok || !l2.ok) {
268  return false;
269  }
270 
271  double d1343 = v13 * l2.v; // dot products
272  double d4321 = l2.v * this->v;
273  double d1321 = v13 * this->v;
274  double d4343 = l2.v * l2.v;
275  double d2121 = this->v * this->v;
276 
277  double denom = d2121 * d4343 - d4321 * d4321;
278  if (fabs(denom) < 1.0e-09) {
279  return false;
280  }
281  double numer = d1343 * d4321 - d1321 * d4343;
282 
283  t1 = numer / denom;
284  t2 = (d1343 + d4321 * t1) / d4343;
285 
286  lshort = Line(t1 * this->v + this->p0, t2 * l2.v + l2.p0);
287  t1 *= this->length;
288  t2 *= l2.length; // parameter in line length for tolerance checking
289  return true;
290 }

292 int Intof(const Line& l0, const Line& l1, Point3d& intof)
293 {
294  /* intersection of 2 vectors
295  returns 0 for intercept but not within either vector
296  returns 1 for intercept on both vectors
297 
298  note that this routine always returns 0 for parallel vectors
299  method:
300  x = x0 + dx0 * t0 for l0
301  ...
302  ...
303  x = x1 + dx1 * t1 for l1
304  ...
305  ...
306 
307  x0 + dx0 * t0 = x1 + dx1 * t1
308  dx0 * t0 - dx1 * t1 + x0 - x1 = 0
309 
310  setup 3 x 3 determinent for
311  a0 t0 + b0 t1 + c0 = 0
312  a1 t0 + b1 t1 + c1 = 0
313  a2 t0 + b2 t1 + c2 = 0
314 
315  from above a = l0.v
316  b = -l1.v
317  c = Vector3d(l1, l0)
318  */
319  // Vector3d a = l0.v;
320  if (l0.box.outside(l1.box)) {
321  return 0;
322  }
323  Vector3d b = -l1.v;
324  Vector3d c = Vector3d(l1.p0, l0.p0);
325  Vector3d det = l0.v ^ b;
326  Vector3d t = b ^ c;
327 
328  // choose largest determinant & corresponding parameter for accuracy
329  double t0 = t.getx();
330  double d = det.getx();
331 
332  if (fabs(det.getz()) > fabs(det.gety())) {
333  if (fabs(det.getz()) > fabs(det.getx())) {
334  t0 = t.getz();
335  d = det.getz();
336  }
337  }
338  else {
339  if (fabs(det.gety()) > fabs(det.getx())) {
340  t0 = t.gety();
341  d = det.gety();
342  }
343  }
344 
345  if (fabs(d) < 1.0e-06) {
346  return 0;
347  }
348 
349  t0 /= d;
350  intof = l0.v * t0 + l0.p0;
351 
352  Point3d other;
353  double t1;
354  if (Dist(l1, intof, other, t1) > geoff_geometry::TOLERANCE) {
355  return 0;
356  }
357 
358  t0 *= l0.length;
359  if (t0 < -geoff_geometry::TOLERANCE || t0 > l0.length + geoff_geometry::TOLERANCE
360  || t1 < -geoff_geometry::TOLERANCE || t1 > l1.length + geoff_geometry::TOLERANCE) {
361  return 0;
362  }
363  return 1;
364 }

411 double Dist(const Span& sp, const Point& p, Point& pnear)
412 {
413  // returns distance of p from span, pnear is the nearpoint on the span (or endpoint)
414  if (!sp.dir) {
415  double d, t;
416  Point3d unused_pnear;
417  d = Dist(Line(sp), Point3d(p), unused_pnear, t);
418  if (t < -geoff_geometry::TOLERANCE) {
419  pnear = sp.p0; // nearpoint
420  d = pnear.Dist(p);
421  }
422  else if (t > sp.length + geoff_geometry::TOLERANCE) {
423  pnear = sp.p1;
424  d = pnear.Dist(p);
425  }
426  return d;
427  }
428  else {
429  // put pnear on the circle
430  double radiusp;
431  Vector2d v(sp.pc, p);
432  if ((radiusp = v.magnitude()) < geoff_geometry::TOLERANCE) {
433  // point specified on circle centre - use first point as near point
434  pnear = sp.p0; // nearpoint
435  return sp.radius;
436  }
437  else {
438  pnear = v * (sp.radius / radiusp) + sp.pc;
439 
440  // check if projected point is on the arc
441  if (sp.OnSpan(pnear)) {
442  return fabs(radiusp - sp.radius);
443  }
444 
445  // point not on arc so calc nearest end-point
446  double ndist = p.Dist(sp.p0);
447  double dist = p.Dist(sp.p1);
448  if (ndist >= dist) {
449  // sp.p1 is near point
450  pnear = sp.p1;
451  return dist;
452  }
453 
454  // sp.p0 is near point
455  pnear = sp.p0; // nearpoint
456  return ndist;
457  }
458  }
459 }

467 bool OnSpan(const Span& sp, const Point& p, bool nearPoints, Point& pNear, Point& pOnSpan)
468 {
469  // function returns true if pNear == pOnSpan
470  // returns pNear & pOnSpan if nearPoints true
471  // pNear (nearest on unbound span)
472  // pOnSpan (nearest on finite span)
473  if (sp.dir) {
474  // arc
475  if (fabs(p.Dist(sp.pc) - sp.radius) > geoff_geometry::TOLERANCE) {
476  if (!nearPoints) {
477  return false;
478  }
479  }
480 
481  pNear = On(Circle(sp.pc, sp.radius), p);
482 
483  if (sp.OnSpan(pNear)) {
484  if (nearPoints) {
485  pOnSpan = pNear;
486  }
487  return true; // near point is on arc - already calculated
488  }
489 
490  // point not on arc return the nearest end-point
491  if (nearPoints) {
492  pOnSpan = (p.Dist(sp.p0) >= p.Dist(sp.p1)) ? sp.p1 : sp.p0;
493  }
494  return false;
495  }
496  else {
497  // straight
498  if (fabs(CLine(sp.p0, sp.vs).Dist(p)) > geoff_geometry::TOLERANCE) {
499  if (!nearPoints) {
500  return false;
501  }
502  }
503  Vector2d v(sp.p0, p);
504  double t = v * sp.vs;
505  if (nearPoints) {
506  pNear = sp.vs * t + sp.p0;
507  }
508  bool onSpan = (t > -geoff_geometry::TOLERANCE && t < sp.length + geoff_geometry::TOLERANCE);
509  if (!onSpan) {
510  if (nearPoints) {
511  pOnSpan = (p.Dist(sp.p0) >= p.Dist(sp.p1)) ? sp.p1 : sp.p0;
512  }
513  }
514  else {
515  if (nearPoints) {
516  pOnSpan = pNear;
517  }
518  }
519  return onSpan;
520  }

544 bool Triangle3d::Intof(const Line& l, Point3d& intof) const
545 {
546  // returns intersection triangle to line in intof
547  // function returns true for intersection, false for no intersection
548  // method based on Möller & Trumbore(1997) (Barycentric coordinates)
549  // based on incorrect Pseudo code from "Geometric Tools for Computer Graphics" p.487
550  if (box.outside(l.box)) {
551  return false;
552  }
553 
554  Vector3d line(l.v);
555  line.normalise();
556 
557  Vector3d p = line ^ v1; // cross product
558  double tmp = p * v0; // dot product
559 
560  if (FEQZ(tmp)) {
561  return false;
562  }
563 
564  tmp = 1 / tmp;
565  Vector3d s(vert1, l.p0);
566 
567  double u = tmp * (s * p); // barycentric coordinate
568  if (u < 0 || u > 1) { // not inside triangle
569  return false;
570  }
571 
572  Vector3d q = s ^ v0;
573  double v = tmp * (line * q); // barycentric coordinate
574  if (v < 0 || v > 1) { // not inside triangle
575  return false;
576  }
577 
578  if (u + v > 1) { // not inside triangle
579  return false;
580  }
581 
582  double t = tmp * (v1 * q);
583  intof = line * t + l.p0;
584  return true;
585 }
