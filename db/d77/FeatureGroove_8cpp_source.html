82 App::DocumentObjectExecReturn *Groove::execute()
83 {
84  if (onlyHasToRefine()){
85  TopoShape result = refineShapeIfActive(rawShape);
86  Shape.setValue(result);
87  return App::DocumentObject::StdReturn;
88  }
89 
90  // Validate parameters
91  double angle = Angle.getValue();
92  if (angle > 360.0)
93  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Angle of groove too large"));
94 
95  angle = Base::toRadians<double>(angle);
96  if (angle < Precision::Angular())
97  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Angle of groove too small"));
98 
99  // Reverse angle if selected
100  if (Reversed.getValue() && !Midplane.getValue())
101  angle *= (-1.0);
102 
103  TopoShape sketchshape;
104  try {
105  sketchshape = getTopoShapeVerifiedFace();
106  } catch (const Base::Exception& e) {
107  return new App::DocumentObjectExecReturn(e.what());
108  }
109 
110  // if the Base property has a valid shape, fuse the prism into it
111  TopoShape base;
112  try {
113  base = getBaseTopoShape();
114  }
115  catch (const Base::Exception&) {
116  std::string text(QT_TRANSLATE_NOOP("Exception", "The requested feature cannot be created. The reason may be that:\n"
117  " - the active Body does not contain a base shape, so there is no\n"
118  " material to be removed;\n"
119  " - the selected sketch does not belong to the active Body."));
120  return new App::DocumentObjectExecReturn(text);
121  }
122 
123  updateAxis();
124 
125  // get revolve axis
126  Base::Vector3d b = Base.getValue();
127  gp_Pnt pnt(b.x,b.y,b.z);
128  Base::Vector3d v = Axis.getValue();
129  gp_Dir dir(v.x,v.y,v.z);
130 
131  try {
132  if (sketchshape.isNull())
133  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Creating a face from sketch failed"));
134 
135  // Rotate the face by half the angle to get Groove symmetric to sketch plane
136  if (Midplane.getValue()) {
137  gp_Trsf mov;
138  mov.SetRotation(gp_Ax1(pnt, dir), Base::toRadians<double>(Angle.getValue()) * (-1.0) / 2.0);
139  TopLoc_Location loc(mov);
140  sketchshape.move(loc);
141  }
142 
143  this->positionByPrevious();
144  auto invObjLoc = getLocation().Inverted();
145  pnt.Transform(invObjLoc.Transformation());
146  dir.Transform(invObjLoc.Transformation());
147  base.move(invObjLoc);
148  sketchshape.move(invObjLoc);
149 
150  // Check distance between sketchshape and axis - to avoid failures and crashes
151  TopExp_Explorer xp;
152  xp.Init(sketchshape.getShape(), TopAbs_FACE);
153  for (;xp.More(); xp.Next()) {
154  if (checkLineCrossesFace(gp_Lin(pnt, dir), TopoDS::Face(xp.Current())))
155  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Revolve axis intersects the sketch"));
156  }
157 
158  // revolve the face to a solid
159  TopoShape result(0);
160  try {
161  result.makeElementRevolve(sketchshape, gp_Ax1(pnt, dir), angle);
162  }catch(Standard_Failure &) {
163  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Could not revolve the sketch!"));
164  }
165  this->AddSubShape.setValue(result);
166 
167  if(base.isNull()) {
168  Shape.setValue(getSolid(result));
169  return App::DocumentObject::StdReturn;
170  }
171 
172  result.Tag = -getID();
173  TopoShape boolOp(0);
174 
175  try {
176  const char *maker;
177  switch (getAddSubType()) {
178  case Additive:
179  maker = Part::OpCodes::Fuse;
180  break;
181 // case Intersecting:
182 // maker = Part::OpCodes::Common;
183 // break;
184  default:
185  maker = Part::OpCodes::Cut;
186  }
187 // this->fixShape(result);
188  boolOp.makeElementBoolean(maker, {base,result});
189  }catch(Standard_Failure &) {
190  return new App::DocumentObjectExecReturn("Failed to cut base feature");
191  }
192  boolOp = this->getSolid(boolOp);
193  if (boolOp.isNull())
194  return new App::DocumentObjectExecReturn("Resulting shape is not a solid");
195 
196  // store shape before refinement
197  this->rawShape = boolOp;
198  boolOp = refineShapeIfActive(boolOp);
199  boolOp = getSolid(boolOp);
200  if (!isSingleSolidRuleSatisfied(boolOp.getShape())) {
201  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Result has multiple solids: that is not currently supported."));
202  }
203  Shape.setValue(boolOp);
204  return App::DocumentObject::StdReturn;
205  }
206  catch (Standard_Failure& e) {
207 
208  if (std::string(e.GetMessageString()) == "TopoDS::Face")
209  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Could not create face from sketch.\n"
210  "Intersecting sketch entities in a sketch are not allowed."));
211  else
212  return new App::DocumentObjectExecReturn(e.GetMessageString());
213  }
214  catch (Base::Exception& e) {
215  return new App::DocumentObjectExecReturn(e.what());
216  }
217 }

239 Groove::RevolMethod Groove::methodFromString(const std::string& methodStr)
240 {
241  if (methodStr == "Angle")
242  return RevolMethod::Dimension;
243  if (methodStr == "UpToLast")
244  return RevolMethod::ToLast;
245  if (methodStr == "ThroughAll")
246  return RevolMethod::ThroughAll;
247  if (methodStr == "UpToFirst")
248  return RevolMethod::ToFirst;
249  if (methodStr == "UpToFace")
250  return RevolMethod::ToFace;

258 void Groove::generateRevolution(TopoDS_Shape& revol,
259  const TopoDS_Shape& sketchshape,
260  const gp_Ax1& axis,
261  const double angle,
262  const double angle2,
263  const bool midplane,
264  const bool reversed,
265  RevolMethod method)
266 {
267  if (method == RevolMethod::Dimension || method == RevolMethod::TwoDimensions || method == RevolMethod::ThroughAll) {
268  double angleTotal = angle;
269  double angleOffset = 0.;
270 
271  if (method == RevolMethod::TwoDimensions) {
272  // Rotate the face by `angle2`/`angle` to get "second" angle
273  angleTotal += angle2;
274  angleOffset = angle2 * -1.0;
275  }
276  else if (method == RevolMethod::ThroughAll) {
277  angleTotal = 2 * M_PI;
278  }
279  else if (midplane) {
280  // Rotate the face by half the angle to get Groove symmetric to sketch plane
281  angleOffset = -angle / 2;
282  }
283 
284  if (fabs(angleTotal) < Precision::Angular())
285  throw Base::ValueError("Cannot create a revolution with zero angle.");
286 
287  TopoDS_Shape from = sketchshape;
288  if (method == RevolMethod::TwoDimensions || midplane) {
289  gp_Trsf mov;
290  mov.SetRotation(axis, angleOffset);
291  TopLoc_Location loc(mov);
292  from.Move(loc);
293  }
294  else if (reversed) {
295  angleTotal *= -1.0;
296  }
297 
298  // revolve the face to a solid
299  // BRepPrimAPI is the only option that allows use of this shape for patterns.
300  // See https://forum.freecadweb.org/viewtopic.php?f=8&t=70185&p=611673#p611673.
301  BRepPrimAPI_MakeRevol RevolMaker(from, axis, angleTotal);
302 
303  if (!RevolMaker.IsDone())
304  throw Base::RuntimeError("ProfileBased: RevolMaker failed! Could not revolve the sketch!");
305  else
306  revol = RevolMaker.Shape();
307  }
308  else {
309  std::stringstream str;
310  str << "ProfileBased: Internal error: Unknown method for generateGroove()";
311  throw Base::RuntimeError(str.str());
312  }
313 }

315 void Groove::generateRevolution(TopoDS_Shape& revol,
316  const TopoDS_Shape& baseshape,
317  const TopoDS_Shape& profileshape,
318  const TopoDS_Face& supportface,
319  const TopoDS_Face& uptoface,
320  const gp_Ax1& axis,
321  RevolMethod method,
322  RevolMode Mode,
323  Standard_Boolean Modify)
324 {
325  if (method == RevolMethod::ToFirst || method == RevolMethod::ToFace || method == RevolMethod::ToLast) {
326  BRepFeat_MakeRevol RevolMaker;
327  TopoDS_Shape base = baseshape;
328  for (TopExp_Explorer xp(profileshape, TopAbs_FACE); xp.More(); xp.Next()) {
329  RevolMaker.Init(base, xp.Current(), supportface, axis, Mode, Modify);
330  RevolMaker.Perform(uptoface);
331  if (!RevolMaker.IsDone())
332  throw Base::RuntimeError("ProfileBased: Up to face: Could not revolve the sketch!");
333 
334  base = RevolMaker.Shape();
335  if (Mode == RevolMode::None)
336  Mode = RevolMode::FuseWithBase;
337  }
338 
339  revol = base;
340  }
341  else {
342  std::stringstream str;
343  str << "ProfileBased: Internal error: Unknown method for generateRevolution()";
344  throw Base::RuntimeError(str.str());
345  }
346 }

348 void Groove::updateProperties(RevolMethod method)
349 {
350  // disable settings that are not valid on the current method
351  // disable everything unless we are sure we need it
352  bool isAngleEnabled = false;
353  bool isAngle2Enabled = false;
354  bool isMidplaneEnabled = false;
355  bool isReversedEnabled = false;
356  bool isUpToFaceEnabled = false;
357  if (method == RevolMethod::Dimension) {
358  isAngleEnabled = true;
359  isMidplaneEnabled = true;
360  isReversedEnabled = !Midplane.getValue();
361  }
362  else if (method == RevolMethod::ToLast) {
363  isReversedEnabled = true;
364  }
365  else if (method == RevolMethod::ThroughAll) {
366  isMidplaneEnabled = true;
367  isReversedEnabled = !Midplane.getValue();
368  }
369  else if (method == RevolMethod::ToFirst) {
370  isReversedEnabled = true;
371  }
372  else if (method == RevolMethod::ToFace) {
373  isReversedEnabled = true;
374  isUpToFaceEnabled = true;
375  }
376  else if (method == RevolMethod::TwoDimensions) {
377  isAngleEnabled = true;
378  isAngle2Enabled = true;
379  isReversedEnabled = true;
380  }
381 
382  Angle.setReadOnly(!isAngleEnabled);
383  Angle2.setReadOnly(!isAngle2Enabled);
384  Midplane.setReadOnly(!isMidplaneEnabled);
385  Reversed.setReadOnly(!isReversedEnabled);
386  UpToFace.setReadOnly(!isUpToFaceEnabled);
387 }
