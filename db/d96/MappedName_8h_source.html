402  void append(const MappedName& other, int startPosition = 0, int size = -1)
403  {
404  // enforce 0 <= startPosition <= other.size
405  if (startPosition < 0) {
406  startPosition = 0;
407  }
408  else if (startPosition > other.size()) {
409  return;
410  }
411 
412  // enforce 0 <= size <= other.size - startPosition
413  if (size < 0 || size > other.size() - startPosition) {
414  size = other.size() - startPosition;
415  }
416 
417 
418  if (startPosition < other.data.size()) // if starting inside data
419  {
420  int count = size;
421  // make sure count doesn't exceed data size and end up in postfix
422  if (count > other.data.size() - startPosition) {
423  count = other.data.size() - startPosition;
424  }
425 
426  // if this is empty append in data else append in postfix
427  if (startPosition == 0 && count == other.data.size() && this->empty()) {
428  this->data = other.data;
429  this->raw = other.raw;
430  }
431  else {
432  append(other.data.constData() + startPosition, count);
433  }
434 
435  // setup startPosition and count to continue appending the remainder to postfix
436  startPosition = 0;
437  size -= count;
438  }
439  else // else starting inside postfix
440  {
441  startPosition -= other.data.size();
442  }
443 
444  // if there is still data to be added to postfix
445  if (size != 0) {
446  if (startPosition == 0 && size == other.postfix.size()) {
447  if (this->empty()) {
448  this->data = other.postfix;
449  }
450  else if (this->postfix.isEmpty()) {
451  this->postfix = other.postfix;
452  }
453  else {
454  this->postfix += other.postfix;
455  }
456  }
457  else {
458  append(other.postfix.constData() + startPosition, size);
459  }
460  }
461  }

488  const char* appendToBuffer(std::string& buffer, int startPosition = 0, int len = -1) const
489  {
490  std::size_t offset = buffer.size();
491  int count = this->size();
492  if (startPosition < 0) {
493  startPosition = 0;
494  }
495  else if (startPosition >= count) {
496  return buffer.c_str() + buffer.size();
497  }
498  if (len < 0 || len > count - startPosition) {
499  len = count - startPosition;
500  }
501  buffer.reserve(buffer.size() + len);
502  if (startPosition < this->data.size()) {
503  count = this->data.size() - startPosition;
504  if (len < count) {
505  count = len;
506  }
507  buffer.append(this->data.constData() + startPosition, count);
508  len -= count;
509  }
510  buffer.append(this->postfix.constData(), len);
511  return buffer.c_str() + offset;
512  }

645  int compare(const MappedName& other) const
646  {
647  int thisSize = this->size();
648  int otherSize = other.size();
649  for (int i = 0, count = std::min(thisSize, otherSize); i < count; ++i) {
650  char thisChar = this->operator[](i);
651  char otherChar = other[i];
652  if (thisChar < otherChar) {
653  return -1;
654  }
655  if (thisChar > otherChar) {
656  return 1;
657  }
658  }
659  if (thisSize < otherSize) {
660  return -1;
661  }
662  if (thisSize > otherSize) {
663  return 1;
664  }
665  return 0;
666  }

983  void append(const MappedName& _name, const ElementIDRefs _sids = ElementIDRefs())
984  {
985  if (!_name) {

987  }

991  compact();

1005  {
1006  if (sids.size() > 1) {
1007  std::sort(sids.begin(), sids.end());
1008  sids.erase(std::unique(sids.begin(), sids.end()), sids.end());

1012  bool erase(const MappedName& _name)
1013  {
1014  if (this->name == _name) {
1015  this->name.clear();
1016  this->sids.clear();
1017  if (this->next) {
1018  this->name = std::move(this->next->name);
1019  this->sids = std::move(this->next->sids);
1020  std::unique_ptr<MappedNameRef> tmp;
1021  tmp.swap(this->next);
1022  this->next = std::move(tmp->next);
1023  }
1024  return true;
1025  }
