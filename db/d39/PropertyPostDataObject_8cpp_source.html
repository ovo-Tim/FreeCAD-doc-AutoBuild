63 void PropertyPostDataObject::scaleDataObject(vtkDataObject* dataObject, double s)
64 {
65  auto scalePoints = [](vtkPoints* points, double s) {
66  for (vtkIdType i = 0; i < points->GetNumberOfPoints(); i++) {
67  double xyz[3];
68  points->GetPoint(i, xyz);
69  for (double& j : xyz) {
70  j *= s;
71  }
72  points->SetPoint(i, xyz);
73  }
74  };
75 
76  if (dataObject->GetDataObjectType() == VTK_POLY_DATA) {
77  vtkPolyData* dataSet = vtkPolyData::SafeDownCast(dataObject);
78  scalePoints(dataSet->GetPoints(), s);
79  }
80  else if (dataObject->GetDataObjectType() == VTK_STRUCTURED_GRID) {
81  vtkStructuredGrid* dataSet = vtkStructuredGrid::SafeDownCast(dataObject);
82  scalePoints(dataSet->GetPoints(), s);
83  }
84  else if (dataObject->GetDataObjectType() == VTK_UNSTRUCTURED_GRID) {
85  vtkUnstructuredGrid* dataSet = vtkUnstructuredGrid::SafeDownCast(dataObject);
86  scalePoints(dataSet->GetPoints(), s);
87  }
88  else if (dataObject->GetDataObjectType() == VTK_MULTIBLOCK_DATA_SET) {
89  vtkMultiBlockDataSet* dataSet = vtkMultiBlockDataSet::SafeDownCast(dataObject);
90  for (unsigned int i = 0; i < dataSet->GetNumberOfBlocks(); i++) {
91  scaleDataObject(dataSet->GetBlock(i), s);
92  }
93  }
94  else if (dataObject->GetDataObjectType() == VTK_MULTIPIECE_DATA_SET) {
95  vtkMultiPieceDataSet* dataSet = vtkMultiPieceDataSet::SafeDownCast(dataObject);
96  for (unsigned int i = 0; i < dataSet->GetNumberOfPieces(); i++) {
97  scaleDataObject(dataSet->GetPiece(i), s);
98  }
99  }
100 }

307 void PropertyPostDataObject::SaveDocFile(Base::Writer& writer) const
308 {
309  // If the shape is empty we simply store nothing. The file size will be 0 which
310  // can be checked when reading in the data.
311  if (!m_dataObject) {
312  return;
313  }
314 
315  // create a temporary file and copy the content to the zip stream
316  // once the tmp. filename is known use always the same because otherwise
317  // we may run into some problems on the Linux platform
318  static Base::FileInfo fi(App::Application::getTempFileName());
319 
320  vtkSmartPointer<vtkXMLDataSetWriter> xmlWriter = vtkSmartPointer<vtkXMLDataSetWriter>::New();
321  xmlWriter->SetInputDataObject(m_dataObject);
322  xmlWriter->SetFileName(fi.filePath().c_str());
323  xmlWriter->SetDataModeToBinary();
324 
325 #ifdef VTK_CELL_ARRAY_V2
326  // Looks like an invalid data object that causes a crash with vtk9
327  vtkUnstructuredGrid* dataGrid = vtkUnstructuredGrid::SafeDownCast(m_dataObject);
328  if (dataGrid && (dataGrid->GetPiece() < 0 || dataGrid->GetNumberOfPoints() <= 0)) {
329  std::cerr << "PropertyPostDataObject::SaveDocFile: ignore empty vtkUnstructuredGrid\n";
330  return;
331  }
332 #endif
333 
334  if (xmlWriter->Write() != 1) {
335  // Note: Do NOT throw an exception here because if the tmp. file could
336  // not be created we should not abort.
337  // We only print an error message but continue writing the next files to the
338  // stream...
339  App::PropertyContainer* father = this->getContainer();
340  if (father && father->isDerivedFrom(App::DocumentObject::getClassTypeId())) {
341  App::DocumentObject* obj = static_cast<App::DocumentObject*>(father);
342  Base::Console().Error("Dataset of '%s' cannot be written to vtk file '%s'\n",
343  obj->Label.getValue(),
344  fi.filePath().c_str());
345  }
346  else {
347  Base::Console().Error("Cannot save vtk file '%s'\n", fi.filePath().c_str());
348  }
349 
350  std::stringstream ss;
351  ss << "Cannot save vtk file '" << fi.filePath() << "'";
352  writer.addError(ss.str());
353  }
354 
355  Base::ifstream file(fi, std::ios::in | std::ios::binary);
356  if (file) {
357  std::streambuf* buf = file.rdbuf();
358  writer.Stream() << buf;
359  }
360 
361  file.close();
362  // remove temp file
363  fi.deleteFile();
364 }

366 void PropertyPostDataObject::RestoreDocFile(Base::Reader& reader)
367 {
368  Base::FileInfo xml(reader.getFileName());
369  // create a temporary file and copy the content from the zip stream
370  Base::FileInfo fi(App::Application::getTempFileName());
371 
372  // read in the ASCII file and write back to the file stream
373  Base::ofstream file(fi, std::ios::out | std::ios::binary);
374  unsigned long ulSize = 0;
375  if (reader) {
376  std::streambuf* buf = file.rdbuf();
377  reader >> buf;
378  file.flush();
379  ulSize = buf->pubseekoff(0, std::ios::cur, std::ios::in);
380  }
381  file.close();
382 
383  // Read the data from the temp file
384  if (ulSize > 0) {
385  std::string extension = xml.extension();
386 
387  // TODO: read in of composite data structures need to be coded,
388  // including replace of "GetOutputAsDataSet()"
389  vtkSmartPointer<vtkXMLReader> xmlReader;
390  if (extension == "vtp") {
391  xmlReader = vtkSmartPointer<vtkXMLPolyDataReader>::New();
392  }
393  else if (extension == "vts") {
394  xmlReader = vtkSmartPointer<vtkXMLStructuredGridReader>::New();
395  }
396  else if (extension == "vtr") {
397  xmlReader = vtkSmartPointer<vtkXMLRectilinearGridReader>::New();
398  }
399  else if (extension == "vtu") {
400  xmlReader = vtkSmartPointer<vtkXMLUnstructuredGridReader>::New();
401  }
402  else if (extension == "vti") {
403  xmlReader = vtkSmartPointer<vtkXMLImageDataReader>::New();
404  }
405 
406  xmlReader->SetFileName(fi.filePath().c_str());
407  xmlReader->Update();
408 
409  if (!xmlReader->GetOutputAsDataSet()) {
410  // Note: Do NOT throw an exception here because if the tmp. created file could
411  // not be read it's NOT an indication for an invalid input stream 'reader'.
412  // We only print an error message but continue reading the next files from the
413  // stream...
414  App::PropertyContainer* father = this->getContainer();
415  if (father && father->isDerivedFrom(App::DocumentObject::getClassTypeId())) {
416  App::DocumentObject* obj = static_cast<App::DocumentObject*>(father);
417  Base::Console().Error("Dataset file '%s' with data of '%s' seems to be empty\n",
418  fi.filePath().c_str(),
419  obj->Label.getValue());
420  }
421  else {
422  Base::Console().Warning("Loaded Dataset file '%s' seems to be empty\n",
423  fi.filePath().c_str());
424  }
425  }
426  else {
427  aboutToSetValue();
428  createDataObjectByExternalType(xmlReader->GetOutputAsDataSet());
429  m_dataObject->DeepCopy(xmlReader->GetOutputAsDataSet());
430  hasSetValue();
431  }
432  }
433 
434  // delete the temp file
435  fi.deleteFile();
436 }
