168 int Array2D::confirmDelete()
169 {

172  box.setWindowTitle(QObject::tr("Confirm Delete"));
173 
174  QString prompt = QObject::tr("Are you sure you want to delete the row?");
175  box.setText(prompt);
176 
177  box.setStandardButtons(QMessageBox::Ok | QMessageBox::Cancel);
178  box.setDefaultButton(QMessageBox::Cancel);
179  box.setEscapeButton(QMessageBox::Cancel);
180 
181  int res = QMessageBox::Cancel;
182  box.adjustSize(); // Silence warnings from Qt on Windows
183  switch (box.exec()) {
184  case QMessageBox::Ok:
185  deleteSelected();
186  res = QMessageBox::Ok;
187  break;
188  }
189 
190  return res;
