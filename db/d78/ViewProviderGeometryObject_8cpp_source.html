132 void ViewProviderGeometryObject::onChanged(const App::Property* prop)
133 {
134  // Actually, the properties 'ShapeColor' and 'Transparency' are part of the property
135  // 'ShapeMaterial'. Both redundant properties are kept due to more convenience for the user. But
136  // we must keep the values consistent of all these properties.
137  std::string propName = prop->getName();
138  if (prop == &Selectable) {
139  bool Sel = Selectable.getValue();
140  setSelectable(Sel);
141  }
142  else if (prop == &Transparency) {
143  long value = toPercent(ShapeAppearance.getTransparency());
144  float trans = fromPercent(Transparency.getValue());
145  if (value != Transparency.getValue()) {
146  ShapeAppearance.setTransparency(trans);
147  }
148 
149  pcShapeMaterial->transparency = trans;
150  }
151  else if (prop == &ShapeAppearance) {
152  if (getObject() && getObject()->testStatus(App::ObjectStatus::TouchOnColorChange)) {
153  getObject()->touch(true);
154  }
155  long value = toPercent(ShapeAppearance.getTransparency());
156  if (value != Transparency.getValue()) {
157  Transparency.setValue(value);
158  }
159  if (ShapeAppearance.getSize() == 1) {
160  const App::Material& Mat = ShapeAppearance[0];
161  setCoinAppearance(Mat);
162  }
163  }
164  else if (prop == &BoundingBox) {
165  showBoundingBox(BoundingBox.getValue());
166  }
167 
168  ViewProviderDragger::onChanged(prop);
169 }

176 void ViewProviderGeometryObject::updateData(const App::Property* prop)
177 {
178  if (prop->isDerivedFrom(App::PropertyComplexGeoData::getClassTypeId())) {
179  Base::BoundBox3d box =
180  static_cast<const App::PropertyComplexGeoData*>(prop)->getBoundingBox();
181  pcBoundingBox->minBounds.setValue(box.MinX, box.MinY, box.MinZ);
182  pcBoundingBox->maxBounds.setValue(box.MaxX, box.MaxY, box.MaxZ);
183  }
184  else if (prop->isDerivedFrom(App::PropertyPlacement::getClassTypeId())) {
185  auto geometry = getObject<App::GeoFeature>();
186  if (geometry && prop == &geometry->Placement) {
187  const App::PropertyComplexGeoData* data = geometry->getPropertyOfGeometry();
188  if (data) {
189  Base::BoundBox3d box = data->getBoundingBox();
190  pcBoundingBox->minBounds.setValue(box.MinX, box.MinY, box.MinZ);
191  pcBoundingBox->maxBounds.setValue(box.MaxX, box.MaxY, box.MaxZ);
192  }
193  }
194  }
195  else if (std::string(prop->getName()) == "ShapeMaterial") {
196  // Set the appearance from the material
197  if (auto geometry = getObject<App::GeoFeature>()) {
198  /*
199  * Change the appearance only if the appearance hasn't been set explicitly. A cached
200  * material appearance is used to see if the current appearance matches the last
201  * material. It is also compared against an empty material to see if the saved
202  * material value has been initialized.
203  */
204  App::Material defaultMaterial;
205  auto material = geometry->getMaterialAppearance();
206  if ((materialAppearance == defaultMaterial)
207  || (ShapeAppearance.getSize() == 1 && ShapeAppearance[0] == materialAppearance)) {
208  ShapeAppearance.setValue(material);
209  }
210  materialAppearance = material;
211  }
212  }
213 
214  ViewProviderDragger::updateData(prop);
215 }

217 SoPickedPointList ViewProviderGeometryObject::getPickedPoints(const SbVec2s& pos,

219  bool pickAll) const
220 {
221  auto root = new SoSeparator;

225  root->addChild(getRoot());
226 
227  SoRayPickAction rp(viewer.getSoRenderManager()->getViewportRegion());
228  rp.setPickAll(pickAll);
229  rp.setRadius(viewer.getPickRadius());
230  rp.setPoint(pos);
231  rp.apply(root);
232  root->unref();
233 
234  // returns a copy of the list
235  return rp.getPickedPointList();

296 void ViewProviderGeometryObject::showBoundingBox(bool show)
297 {
298  if (!pcBoundSwitch && show) {
299  unsigned long bbcol = getBoundColor();
300  float red {};
301  float green {};
302  float blue {};
303  red = ((bbcol >> 24) & 0xff) / 255.0F;
304  green = ((bbcol >> 16) & 0xff) / 255.0F;
305  blue = ((bbcol >> 8) & 0xff) / 255.0F;
306 
307  pcBoundSwitch = new SoSwitch();
308  pcBoundSwitch->setName("BoundSwitch");
309  auto pBoundingSep = new SoSeparator();
310  auto lineStyle = new SoDrawStyle;
311  lineStyle->lineWidth = 2.0F;
312  pBoundingSep->addChild(lineStyle);
313 
314  pcBoundColor->rgb.setValue(red, green, blue);
315  pBoundingSep->addChild(pcBoundColor);
316  auto font = new SoFont();
317  font->size.setValue(getBoundBoxFontSize());
318  pBoundingSep->addChild(font);
319 
320  pBoundingSep->addChild(new SoResetTransform());
321  pBoundingSep->addChild(pcBoundingBox);
322  pcBoundingBox->coordsOn.setValue(false);
323  pcBoundingBox->dimensionsOn.setValue(true);
324 
325  // add to the highlight node
326  pcBoundSwitch->addChild(pBoundingSep);
327  pcRoot->addChild(pcBoundSwitch);
328  }
329 
330  if (pcBoundSwitch) {
331  pcBoundSwitch->whichChild = (show ? 0 : -1);
332  }
333 }

335 void ViewProviderGeometryObject::setSelectable(bool selectable)
336 {
337  SoSearchAction sa;
338  sa.setInterest(SoSearchAction::ALL);
339  sa.setSearchingAll(true);
340  sa.setType(Gui::SoFCSelection::getClassTypeId());
341  sa.apply(pcRoot);
342 
343  SoPathList& pathList = sa.getPaths();
344 
345  for (int i = 0; i < pathList.getLength(); i++) {
346  auto selNode = dynamic_cast<SoFCSelection*>(pathList[i]->getTail());
347  if (selectable) {
348  if (selNode) {
349  selNode->selectionMode = SoFCSelection::SEL_ON;
350  selNode->highlightMode = SoFCSelection::AUTO;
351  }
352  }
353  else {
354  if (selNode) {
355  selNode->selectionMode = SoFCSelection::SEL_OFF;
356  selNode->highlightMode = SoFCSelection::OFF;
357  selNode->selected = SoFCSelection::NOTSELECTED;
358  }
359  }
360  }
361 
362  pickStyle->style.setValue(selectable ? SoPickStyle::SHAPE : SoPickStyle::UNPICKABLE);
363 }
