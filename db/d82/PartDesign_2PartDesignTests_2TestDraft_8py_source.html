32  def testSimpleDraft(self):
33  # fix: create datum plane on YZ. create datum line on Z + 10i
34  # find top face by first making list comprehension of Z-normal faces
35  # and then find which has the higher center of mass Z-value
36  self.Body = self.Doc.addObject('PartDesign::Body','Body')
37  self.Box = self.Doc.addObject('PartDesign::AdditiveBox','Box')
38  self.Body.addObject(self.Box)
39  self.Box.Length=10.00
40  self.Box.Width=10.00
41  self.Box.Height=10.00
42  self.Doc.recompute()
43  self.DatumPlane = self.Doc.addObject('PartDesign::Plane','DatumPlane')
44  self.DatumPlane.AttachmentSupport = [(self.Doc.YZ_Plane,'')]
45  self.DatumPlane.MapMode = 'FlatFace'
46  self.Body.addObject(self.DatumPlane)
47  self.Doc.recompute()
48  self.DatumLine = self.Doc.addObject('PartDesign::Line','DatumLine')
49  self.DatumLine.AttachmentSupport = [(self.Doc.X_Axis,'')]
50  self.DatumLine.MapMode = 'TwoPointLine'
51  self.Body.addObject(self.DatumLine)
52  self.Doc.recompute()
53  self.Draft = self.Doc.addObject("PartDesign::Draft","Draft")
54  # Draft.Base needs to be top face
55  self.Faces = self.Box.Shape.Faces
56  # Grab the two faces with Z-normals and find the higher one
57  self.ZFaceIndexes = [i for i in range(len(self.Faces)) if self.Faces[i].Surface.Axis == App.Vector(0,0,1)]
58  if self.Faces[self.ZFaceIndexes[0]].CenterOfMass.z > self.Faces[self.ZFaceIndexes[1]].CenterOfMass.z:
59  self.TopFaceIndex = self.ZFaceIndexes[0]
60  else:
61  self.TopFaceIndex = self.ZFaceIndexes[1]
62  self.Draft.Base = (self.Box, ["Face"+str(self.TopFaceIndex+1)])
63  self.Draft.NeutralPlane = (self.DatumPlane, [''])
64  self.Draft.PullDirection = (self.DatumLine, [''])
65  self.Draft.Angle = 45.0
66  self.Draft.Reversed = 1
67  self.Body.addObject(self.Draft)
68  self.Doc.recompute()
69  if 'Invalid' in self.Draft.State:
70  self.Draft.Reversed = 0
71  self.Doc.recompute()
72  self.assertAlmostEqual(self.Draft.Shape.Volume, 1500)
