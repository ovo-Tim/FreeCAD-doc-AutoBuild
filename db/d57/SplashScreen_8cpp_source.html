63  explicit SplashObserver(QSplashScreen* splasher = nullptr)
64  : splash(splasher)
65  , alignment(Qt::AlignBottom | Qt::AlignLeft)
66  , textColor(Qt::black)
67  {

69 
70  // allow to customize text position and color
71  const std::map<std::string, std::string>& cfg = App::Application::Config();
72  auto al = cfg.find("SplashAlignment");
73  if (al != cfg.end()) {
74  QString alt = QString::fromLatin1(al->second.c_str());
75  int align = 0;
76  if (alt.startsWith(QLatin1String("VCenter"))) {
77  align = Qt::AlignVCenter;
78  }
79  else if (alt.startsWith(QLatin1String("Top"))) {
80  align = Qt::AlignTop;
81  }
82  else {
83  align = Qt::AlignBottom;
84  }
85 
86  if (alt.endsWith(QLatin1String("HCenter"))) {
87  align += Qt::AlignHCenter;
88  }
89  else if (alt.endsWith(QLatin1String("Right"))) {
90  align += Qt::AlignRight;
91  }
92  else {
93  align += Qt::AlignLeft;
94  }
95 
96  alignment = align;
97  }
98 
99  // choose text color
100  auto tc = cfg.find("SplashTextColor");
101  if (tc != cfg.end()) {
102  QColor col(QString::fromStdString(tc->second));
103  if (col.isValid()) {
104  textColor = col;
105  }
106  }
107  }

135  void Log(const std::string& text)
136  {
137  QString msg(QString::fromStdString(text));
138  QRegularExpression rx;
139  // ignore 'Init:' and 'Mod:' prefixes
140  rx.setPattern(QLatin1String("^\\s*(Init:|Mod:)\\s*"));
141  auto match = rx.match(msg);
142  if (match.hasMatch()) {
143  msg = msg.mid(match.capturedLength());
144  }
145  else {
146  // ignore activation of commands
147  rx.setPattern(QLatin1String(R"(^\s*(\+App::|Create|CmdC:|CmdG:|Act:)\s*)"));
148  match = rx.match(msg);
149  if (match.hasMatch() && match.capturedStart() == 0) {
150  return;
151  }
152  }
153 
154  splash->showMessage(msg.replace(QLatin1String("\n"), QString()), alignment, textColor);
155  QMutex mutex;
156  QMutexLocker ml(&mutex);
157  QWaitCondition().wait(&mutex, 50);
158  }
159 
160 private:
161  QSplashScreen* splash;
162  int alignment;
163  QColor textColor;
164 };

176 static void renderDevBuildWarning(QPainter& painter,
177  const QPoint startPosition,
178  const QSize maxSize,
179  QColor color)
180 {
181  // Create a background box that fades out the artwork for better legibility
182  QColor fader(Qt::white);
183  constexpr float halfDensity(0.65F);
184  fader.setAlphaF(halfDensity);
185  QBrush fillBrush(fader, Qt::BrushStyle::SolidPattern);
186  painter.setBrush(fillBrush);
187 
188  // Construct the lines of text and figure out how much space they need
189  const auto devWarningLine1 = QObject::tr("WARNING: This is a development version.");
190  const auto devWarningLine2 = QObject::tr("Please do not use it in a production environment.");
191  QFontMetrics fontMetrics(painter.font()); // Try to use the existing font
192  int padding = QtTools::horizontalAdvance(fontMetrics, QLatin1String("M")); // Arbitrary
193  int line1Width = QtTools::horizontalAdvance(fontMetrics, devWarningLine1);
194  int line2Width = QtTools::horizontalAdvance(fontMetrics, devWarningLine2);
195  int boxWidth = std::max(line1Width, line2Width) + 2 * padding;
196  int lineHeight = fontMetrics.lineSpacing();
197  if (boxWidth > maxSize.width()) {
198  // Especially if the text was translated, there is a chance that using the existing font
199  // will exceed the width of the Splashscreen graphic. Resize down so that it fits, no matter
200  // how long the text strings are.
201  float reductionFactor = static_cast<float>(maxSize.width()) / static_cast<float>(boxWidth);
202  int newFontSize = static_cast<int>(painter.font().pointSize() * reductionFactor);
203  padding *= reductionFactor;
204  QFont newFont = painter.font();
205  newFont.setPointSize(newFontSize);
206  painter.setFont(newFont);
207  lineHeight = painter.fontMetrics().lineSpacing();
208  boxWidth = maxSize.width();
209  }
210  constexpr float lineExpansionFactor(2.3F);
211  int boxHeight = static_cast<int>(lineHeight * lineExpansionFactor);
212 
213  // Draw the background rectangle and the text
214  painter.setPen(color);
215  painter.drawRect(startPosition.x(), startPosition.y(), boxWidth, boxHeight);
216  painter.drawText(startPosition.x() + padding, startPosition.y() + lineHeight, devWarningLine1);
217  painter.drawText(startPosition.x() + padding,
218  startPosition.y() + 2 * lineHeight,
219  devWarningLine2);
220 }

259 QPixmap SplashScreen::splashImage()
260 {
261  // search in the UserAppData dir as very first
262  QPixmap splash_image;
263  QFileInfo fi(QString::fromLatin1("images:splash_image.png"));
264  if (fi.isFile() && fi.exists()) {
265  splash_image.load(fi.filePath(), "PNG");
266  }
267 
268  // if no image was found try the config
269  std::string splash_path = App::Application::Config()["SplashScreen"];
270  if (splash_image.isNull()) {
271  QString path = QString::fromStdString(splash_path);
272  if (QDir(path).isRelative()) {
273  QString home = QString::fromStdString(App::Application::getHomePath());
274  path = QFileInfo(QDir(home), path).absoluteFilePath();
275  }
276 
277  splash_image.load(path);
278  }
279 
280  // now try the icon paths
281  float pixelRatio(1.0);
282  if (splash_image.isNull()) {
283  // determine the count of splashes
284  QStringList pixmaps =
285  Gui::BitmapFactory().findIconFiles().filter(QString::fromStdString(splash_path));
286  // divide by 2 since there's two sets (normal and 2x)
287  // minus 1 to ignore the default splash that isn't numbered
288  int splash_count = pixmaps.count() / 2 - 1;
289 
290  // set a random splash path
291  if (splash_count > 0) {
292  int random = rand() % splash_count;
293  splash_path += std::to_string(random);
294  }
295 
296  if (qApp->devicePixelRatio() > 1.0) {
297  // For HiDPI screens, we have a double-resolution version of the splash image
298  splash_path += "_2x";
299  splash_image = Gui::BitmapFactory().pixmap(splash_path.c_str());
300  splash_image.setDevicePixelRatio(2.0);
301  pixelRatio = 2.0;
302  }
303  else {
304  splash_image = Gui::BitmapFactory().pixmap(splash_path.c_str());
305  }
306  }
307 
308  // include application name and version number
309  std::map<std::string, std::string>::const_iterator tc =
310  App::Application::Config().find("SplashInfoColor");
311  std::map<std::string, std::string>::const_iterator wc =
312  App::Application::Config().find("SplashWarningColor");
313  if (tc != App::Application::Config().end() && wc != App::Application::Config().end()) {
314  QString title = qApp->applicationName();
315  QString major = QString::fromStdString(App::Application::Config()["BuildVersionMajor"]);
316  QString minor = QString::fromStdString(App::Application::Config()["BuildVersionMinor"]);
317  QString point = QString::fromStdString(App::Application::Config()["BuildVersionPoint"]);
318  QString suffix = QString::fromStdString(App::Application::Config()["BuildVersionSuffix"]);
319  QString version = QString::fromLatin1("%1.%2.%3%4").arg(major, minor, point, suffix);
320  QString position, fontFamily;
321 
322  std::map<std::string, std::string>::const_iterator te =
323  App::Application::Config().find("SplashInfoExeName");
324  std::map<std::string, std::string>::const_iterator tv =
325  App::Application::Config().find("SplashInfoVersion");
326  std::map<std::string, std::string>::const_iterator tp =
327  App::Application::Config().find("SplashInfoPosition");
328  std::map<std::string, std::string>::const_iterator tf =
329  App::Application::Config().find("SplashInfoFont");
330  if (te != App::Application::Config().end()) {
331  title = QString::fromStdString(te->second);
332  }
333  if (tv != App::Application::Config().end()) {
334  version = QString::fromStdString(tv->second);
335  }
336  if (tp != App::Application::Config().end()) {
337  position = QString::fromStdString(tp->second);
338  }
339  if (tf != App::Application::Config().end()) {
340  fontFamily = QString::fromStdString(tf->second);
341  }
342 
343  QPainter painter;
344  painter.begin(&splash_image);
345  if (!fontFamily.isEmpty()) {
346  QFont font = painter.font();
347  if (font.fromString(fontFamily)) {
348  painter.setFont(font);
349  }
350  }
351 
352  QFont fontExe = painter.font();
353  fontExe.setPointSizeF(20.0);
354  QFontMetrics metricExe(fontExe);
355  int l = QtTools::horizontalAdvance(metricExe, title);
356  if (title == QLatin1String("FreeCAD")) {
357  l = 0.0; // "FreeCAD" text is already part of the splashscreen, version goes below it
358  }
359  int w = splash_image.width();
360  int h = splash_image.height();
361 
362  QFont fontVer = painter.font();
363  fontVer.setPointSizeF(11.0);
364  QFontMetrics metricVer(fontVer);
365  int v = QtTools::horizontalAdvance(metricVer, version);
366 
367  int x = -1, y = -1;
368  QRegularExpression rx(QLatin1String("(\\d+).(\\d+)"));
369  auto match = rx.match(position);
370  if (match.hasMatch()) {
371  x = match.captured(1).toInt();
372  y = match.captured(2).toInt();
373  }
374  else {
375  x = w - (l + v + 10);
376  y = h - 20;
377  }
378 
379  QColor color(QString::fromStdString(tc->second));
380  if (color.isValid()) {
381  painter.setPen(color);
382  painter.setFont(fontExe);
383  if (title != QLatin1String("FreeCAD")) {
384  // FreeCAD's Splashscreen already contains the EXE name, no need to draw it
385  painter.drawText(x, y, title);
386  }
387  painter.setFont(fontVer);
388  painter.drawText(x + (l + 235), y - 7, version);
389  QColor warningColor(QString::fromStdString(wc->second));
390  if (suffix == QLatin1String("dev") && warningColor.isValid()) {
391  fontVer.setPointSizeF(14.0);
392  painter.setFont(fontVer);
393  const int lineHeight = metricVer.lineSpacing();
394  const int padding {45}; // Distance from the edge of the graphic's bounding box
395  QPoint startPosition(padding, y + lineHeight * 2);
396  QSize maxSize(w / pixelRatio - 2 * padding, lineHeight * 3);
397  renderDevBuildWarning(painter, startPosition, maxSize, warningColor);
398  }
399  painter.end();
400  }
401  }
402 
403  return splash_image;
404 }
