179 def try_converting_to_float(line: str) -> Union[None, List[float]]:
180  """Does its best to split a line and convert its elements to float
181 
182  Has 3 strategies of splitting:
183  * by comma - mainly in CalculiX .inp files
184  * by space - other solvers
185  * by space and ignoring the 1st word - other solvers
186  If there was only 1 word, the line will fail, no compromises
187 
188  Single characters always pass
189 
190  :param line: line to split and convert to floats
191  :return: None if conversion failed, else list of floats
192  """
193  strategies = [
194  lambda _line: _line[1:].split(","),
195  lambda _line: _line[1:].split(),
196  lambda _line: (lambda split_line: len(split_line) > 1 and split_line or "fail")(
197  _line[1:].split()[1:]
198  ),
199  ]
200  for strategy in strategies:
201  try:
202  return list(map(float, strategy(line)))
203  except ValueError:
204  pass
205  return None

208 def are_floats_equal(
209  orig_floats: Union[None, List[float]], floats_to_compare: Union[None, List[float]]
210 ) -> bool:
211  """Check if floats in lists are equal with some tolerance
212 
213  :param orig_floats: list of floats - left operands of comparison
214  :param floats_to_compare: list of floats - right operands of comparison
215  :return: True if both lists are equal with some tolerance element-wise, else False
216  """
217  if any(floats is None for floats in (orig_floats, floats_to_compare)):
218  return False
219  for orig_float, float_to_compare in zip(orig_floats, floats_to_compare):
220  if not math.isclose(orig_float, float_to_compare, abs_tol=1e-10):
221  return False
222  return True

225 def parse_diff(diff_lines: Iterator[str]) -> List[str]:
226  """Parses lines from `united_diff`
227 
228  Tries to split a line and convert its contents to float, to
229  compare float numbers with some tolerance
230 
231  Recognizes blocks of changes, that start with `@@` and end either at EOF or
232  at the beginning of another block. Only bad lines are added to block.
233  All lines, which didn't pass the element-wise check, are bad lines.
234  :param diff_lines: lines produced by `united_diff`
235  :return: list of bad lines with respect to their block of change
236  """
237  bad_lines = []
238  changes_block = []
239  while True:
240  try:
241  first = next(diff_lines)
242  if first.startswith("@@"):
243  if len(changes_block) > 1:
244  bad_lines.extend(changes_block)
245  changes_block = [first]
246  continue
247  second = next(diff_lines)
248  except StopIteration:
249  break
250  if first.startswith("---"):
251  continue
252  if not are_floats_equal(*map(try_converting_to_float, (first, second))):
253  changes_block.extend((first, second))
254  # check if we have any remaining block
255  if len(changes_block) > 1:
256  bad_lines.extend(changes_block)
257  return bad_lines
