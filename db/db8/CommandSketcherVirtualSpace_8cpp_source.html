53 bool isSketcherVirtualSpaceActive(Gui::Document* doc, bool actsOnSelection)
54 {
55  if (doc) {
56  // checks if a Sketch Viewprovider is in Edit and is in no special mode
57  if (doc->getInEdit()
58  && doc->getInEdit()->isDerivedFrom(SketcherGui::ViewProviderSketch::getClassTypeId())) {
59  if (static_cast<SketcherGui::ViewProviderSketch*>(doc->getInEdit())->getSketchMode()
60  == ViewProviderSketch::STATUS_NONE) {
61  if (!actsOnSelection) {
62  return true;
63  }
64  else if (Gui::Selection().countObjectsOfType<Sketcher::SketchObject>() > 0) {
65  return true;
66  }
67  }
68  }
69  }
70  return false;
71 }

105 void CmdSketcherSwitchVirtualSpace::activated(int iMsg)
106 {
107  Q_UNUSED(iMsg);
108  bool modeChange = true;
109 
110  std::vector<Gui::SelectionObject> selection;
111 
112  if (Gui::Selection().countObjectsOfType<Sketcher::SketchObject>() > 0) {
113  // Now we check whether we have a constraint selected or not.
114  selection = getSelection().getSelectionEx();
115 
116  // only one sketch with its subelements are allowed to be selected
117  if (selection.size() != 1
118  || !selection[0].isObjectTypeOf(Sketcher::SketchObject::getClassTypeId())) {
119  Gui::TranslatedUserWarning(getActiveGuiDocument(),
120  QObject::tr("Wrong selection"),
121  QObject::tr("Select constraint(s) from the sketch."));
122  return;
123  }
124 
125  // get the needed lists and objects
126  const std::vector<std::string>& SubNames = selection[0].getSubNames();
127  if (SubNames.empty()) {
128  Gui::TranslatedUserWarning(getActiveGuiDocument(),
129  QObject::tr("Wrong selection"),
130  QObject::tr("Select constraint(s) from the sketch."));
131  return;
132  }
133 
134  for (std::vector<std::string>::const_iterator it = SubNames.begin(); it != SubNames.end();
135  ++it) {
136  // see if we have constraints, if we do it is not a mode change, but a toggle.
137  if (it->size() > 10 && it->substr(0, 10) == "Constraint") {
138  modeChange = false;
139  }
140  }
141  }
142 
143  if (modeChange) {
144  Gui::Document* doc = getActiveGuiDocument();
145 
146  SketcherGui::ViewProviderSketch* vp =
147  static_cast<SketcherGui::ViewProviderSketch*>(doc->getInEdit());
148  vp->setIsShownVirtualSpace(!vp->getIsShownVirtualSpace());
149  }
150  // toggle the selected constraint(s)
151  else {
152  // get the needed lists and objects
153  const std::vector<std::string>& SubNames = selection[0].getSubNames();
154  if (SubNames.empty()) {
155  Gui::TranslatedUserWarning(getActiveGuiDocument(),
156  QObject::tr("Wrong selection"),
157  QObject::tr("Select constraint(s) from the sketch."));
158 
159  return;
160  }
161 
162  SketcherGui::ViewProviderSketch* sketchgui =
163  static_cast<SketcherGui::ViewProviderSketch*>(getActiveGuiDocument()->getInEdit());
164  Sketcher::SketchObject* Obj = sketchgui->getSketchObject();
165 
166  // undo command open
167  openCommand(QT_TRANSLATE_NOOP("Command", "Toggle constraints to the other virtual space"));
168 
169  int successful = SubNames.size();
170  // go through the selected subelements
171  for (std::vector<std::string>::const_iterator it = SubNames.begin(); it != SubNames.end();
172  ++it) {
173  // only handle constraints
174  if (it->size() > 10 && it->substr(0, 10) == "Constraint") {
175  int ConstrId = Sketcher::PropertyConstraintList::getIndexFromConstraintName(*it);
176  Gui::Command::openCommand(
177  QT_TRANSLATE_NOOP("Command", "Update constraint's virtual space"));
178  try {
179  Gui::cmdAppObjectArgs(Obj, "toggleVirtualSpace(%d)", ConstrId);
180  }
181  catch (const Base::Exception&) {
182  successful--;
183  }
184  }
185  }
186 
187  if (successful > 0) {
188  commitCommand();
189  }
190  else {
191  abortCommand();
192  }
193 
194  // recomputer and clear the selection (convenience)
195  tryAutoRecompute(Obj);
196  getSelection().clearSelection();
197  }
198 }
