49 void ReaderStep::read(Handle(TDocStd_Document) hDoc) // NOLINT
50 {
51  std::string utf8Name = file.filePath();
52  std::string name8bit = Part::encodeFilename(utf8Name);
53  STEPCAFControl_Reader aReader;
54  aReader.SetColorMode(true);
55  aReader.SetNameMode(true);
56  aReader.SetLayerMode(true);
57  aReader.SetSHUOMode(true);
58 #if OCC_VERSION_HEX < 0x070800
59  if (aReader.ReadFile(name8bit.c_str()) != IFSelect_RetDone) {
60 #else
61  Handle(StepData_StepModel) aStepModel = new StepData_StepModel;
62  aStepModel->InternalParameters.InitFromStatic();
63  aStepModel->SetSourceCodePage(codePage);
64  if (aReader.ReadFile(name8bit.c_str(), aStepModel->InternalParameters) != IFSelect_RetDone) {
65 #endif
66  throw Base::FileException("Cannot read STEP file", file);
67  }
68 
69 #if OCC_VERSION_HEX < 0x070500
70  Handle(Message_ProgressIndicator) pi = new Part::ProgressIndicator(100);
71  aReader.Reader().WS()->MapReader()->SetProgress(pi);
72  pi->NewScope(100, "Reading STEP file...");
73  pi->Show();
74 #endif
75  aReader.Transfer(hDoc);
76 #if OCC_VERSION_HEX < 0x070500
77  pi->EndScope();
78 #endif
79 }
