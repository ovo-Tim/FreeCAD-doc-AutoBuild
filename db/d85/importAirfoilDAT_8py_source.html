112 def process(filename):
113  """Process the filename and create a Draft Wire from the data.
114 
115  The common airfoil dat format has many flavors.
116  This code should work with almost every dialect.
117 
118  Parameters
119  ----------
120  filename : str
121  The path to the filename to be opened.
122 
123  Returns
124  -------
125  Part::Part2DObject or None.
126  The created Draft Wire object or None if the file contains less
127  than 3 points.
128  """
129  # Regex to identify data rows and throw away unused metadata
130  xval = r'(?P<xval>\-?\s*\d*\.*\d*([Ee]\-?\d+)?)'
131  yval = r'(?P<yval>\-?\s*\d*\.*\d*([Ee]\-?\d+)?)'
132  _regex = r'^\s*' + xval + r'\,?\s*' + yval + r'\s*$'
133 
134  regex = re.compile(_regex)
135  afile = pyopen(filename, 'r')
136  # read the airfoil name which is always at the first line
137  airfoilname = afile.readline().strip()
138 
139  coords = []
140  # upside = True
141  # last_x = None
142 
143  # Collect the data
144  for lin in afile:
145  curdat = regex.match(lin)
146  if (curdat is not None
147  and curdat.group("xval")
148  and curdat.group("yval")):
149  x = float(curdat.group("xval"))
150  y = float(curdat.group("yval"))
151 
152  # Some files specify the number of upper and lower points on the 2nd line:
153  # " 67. 72."
154  # See: http://airfoiltools.com/airfoil
155  # This line must be skipped:
156  if x < 2 and y < 2:
157  # the normal processing
158  coords.append(Vector(x, y, 0))
159 
160  afile.close()
161 
162  if len(coords) < 3:
163  FCC.PrintError(translate("ImportAirfoilDAT", "Did not find enough coordinates") + "\n")
164  return None
165 
166  # sometimes coords are divided in upper an lower side
167  # so that x-coordinate begin new from leading or trailing edge
168  # check for start coordinates in the middle of list
169  if coords[0:-1].count(coords[0]) > 1:
170  flippoint = coords.index(coords[0], 1)
171  upper = coords[0:flippoint]
172  lower = coords[flippoint+1:]
173  lower.reverse()
174  for i in lower:
175  upper.append(i)
176  coords = upper
177 
178  # do we use the parametric Draft Wire?
179  if useDraftWire:
180  obj = Draft.make_wire(coords, True)
181  # obj.label = airfoilname
182  else:
183  # alternate solution, uses common Part Faces
184  lines = []
185  first_v = None
186  last_v = None
187  for v in coords:
188  if first_v is None:
189  first_v = v
190 
191  # Line between v and last_v if they're not equal
192  if (last_v is not None) and (last_v != v):
193  lines.append(Part.makeLine(last_v, v))
194 
195  # The new last_v
196  last_v = v
197 
198  # close the wire if needed
199  if last_v != first_v:
200  lines.append(Part.makeLine(last_v, first_v))
201 
202  wire = Part.Wire(lines)
203  face = Part.Face(wire)
204  obj = FreeCAD.ActiveDocument.addObject('Part::Feature', airfoilname)
205  obj.Shape = face
