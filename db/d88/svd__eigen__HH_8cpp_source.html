31  int svd_eigen_HH(const MatrixXd& A,MatrixXd& U,VectorXd& S,MatrixXd& V,VectorXd& tmp,int maxiter,double epsilon)
32  {

34  const int rows = A.rows();
35  const int cols = A.cols();
36 
37  U.setZero();
38  U.topLeftCorner(rows,cols)=A;

45  /* Householder reduction to bidiagonal form. */
46  for (i=0;i<cols;i++) {
47  ppi=i+1;
48  tmp(i)=scale*g;
49  g=s=scale=0.0;
50  if (i<rows) {
51  // compute the sum of the i-th column, starting from the i-th row
52  for (k=i;k<rows;k++) scale += fabs(U(k,i));
53  if (fabs(scale)>epsilon) {
54  // multiply the i-th column by 1.0/scale, start from the i-th element
55  // sum of squares of column i, start from the i-th element

57  U(k,i) /= scale;
58  s += U(k,i)*U(k,i);
59  }

64  h=f*g-s;

66  for (j=ppi;j<cols;j++) {

69  if (!(h!=0))

74  }
75  for (k=i;k<rows;k++) U(k,i) *= scale;
76  }
77  }
78  // save singular value
79  S(i)=scale*g;
80  g=s=scale=0.0;
81  if ((i <rows) && (i+1 != cols)) {
82  // sum of row i, start from columns i+1
83  for (k=ppi;k<cols;k++) scale += fabs(U(i,k));
84  if (fabs(scale)>epsilon) {
85  for (k=ppi;k<cols;k++) {
86  U(i,k) /= scale;
87  s += U(i,k)*U(i,k);
88  }
89  f=U(i,ppi);
90  if (!(s>=0))
91  return -5;
92  g = -SIGN(sqrt(s),f);
93  h=f*g-s;
94  U(i,ppi)=f-g;
95  if (!(h!=0))
96  return -6;
97  for (k=ppi;k<cols;k++) tmp(k)=U(i,k)/h;
98  for (j=ppi;j<rows;j++) {
99  for (s=0.0,k=ppi;k<cols;k++) s += U(j,k)*U(i,k);
100  for (k=ppi;k<cols;k++) U(j,k) += s*tmp(k);
101  }
102  for (k=ppi;k<cols;k++) U(i,k) *= scale;
103  }
104  }
105  maxarg1=anorm;
106  maxarg2=(fabs(S(i))+fabs(tmp(i)));

108  }

110  for (i=cols-1;i>=0;i--) {
111  if (i<cols-1) {
112  if (fabs(g)>epsilon) {
113  if (!(U(i,ppi)!=0))
114  return -7;

116  for (j=ppi;j<cols;j++) {
117  for (s=0.0,k=ppi;k<cols;k++) s += U(i,k)*V(k,j);
118  for (k=ppi;k<cols;k++) V(k,j) += s*V(k,i);
119  }
120  }
121  for (j=ppi;j<cols;j++) V(i,j)=V(j,i)=0.0;
122  }
123  V(i,i)=1.0;
124  g=tmp(i);
125  ppi=i;
126  }
127  /* Accumulation of left-hand transformations. */
128  for (i=cols-1<rows-1 ? cols-1:rows-1;i>=0;i--) {
129  ppi=i+1;
130  g=S(i);
131  for (j=ppi;j<cols;j++) U(i,j)=0.0;
132  if (fabs(g)>epsilon) {
133  g=1.0/g;
134  for (j=ppi;j<cols;j++) {

136  if (!(U(i,i)!=0))
137  return -8;
138  f=(s/U(i,i))*g;
139  for (k=i;k<rows;k++) U(k,j) += f*U(k,i);
140  }
141  for (j=i;j<rows;j++) U(j,i) *= g;
142  } else {
143  for (j=i;j<rows;j++) U(j,i)=0.0;
144  }

146  }
