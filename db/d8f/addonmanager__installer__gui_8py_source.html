47 class AddonInstallerGUI(QtCore.QObject):
48  """GUI functions (sequence of dialog boxes) for installing an addon interactively. The actual
49  installation is handled by the AddonInstaller class running in a separate QThread. An instance
50  of this AddonInstallerGUI class should NOT be run in a separate thread, but on the main GUI
51  thread. All dialogs are modal."""
52 
53  # External classes are expected to "set and forget" this class, but in the event that some
54  # action must be taken if the addon is actually installed, this signal is provided. Note that
55  # this class already provides a "Successful installation" dialog, so external code need not
56  # do so.
57  success = QtCore.Signal(object)

59  # Emitted once all work has been completed, regardless of success or failure
60  finished = QtCore.Signal()

89  def run(self):
90  """Instructs this class to begin displaying the necessary dialogs to guide a user through
91  an Addon installation sequence. All dialogs are modal."""
92 
93  # Dependency check
94  deps = MissingDependencies(self.addon_to_install, self.addons)
95 
96  # Python interpreter version check
97  stop_installation = self._check_python_version(deps)
98  if stop_installation:
99  self.finished.emit()
100  return
101 
102  # Required Python
103  if hasattr(deps, "python_requires") and deps.python_requires:
104  # Disallowed packages:
105  stop_installation = self._handle_disallowed_python(deps.python_requires)
106  if stop_installation:
107  self.finished.emit()
108  return
109  # Allowed but uninstalled is handled below
110 
111  # Remove any disallowed packages from the optional list
112  if hasattr(deps, "python_optional") and deps.python_optional:
113  self._clean_up_optional(deps)
114 
115  # Missing FreeCAD workbenches
116  if hasattr(deps, "wbs") and deps.wbs:
117  stop_installation = self._report_missing_workbenches(deps.wbs)
118  if stop_installation:
119  self.finished.emit()
120  return
121 
122  # If we have any missing dependencies, display a dialog to the user asking if they want to
123  # install them.
124  if deps.external_addons or deps.python_requires or deps.python_optional:
125  # Recoverable: ask the user if they want to install the missing deps, do so, then
126  # proceed with the installation
127  self._resolve_dependencies_then_install(deps)
128  else:
129  # No missing deps, just install
130  self.install()

132  def _handle_disallowed_python(self, python_requires: List[str]) -> bool:
133  """Determine if we are missing any required Python packages that are not in the allowed
134  packages list. If so, display a message to the user, and return True. Otherwise return
135  False."""
136 
137  bad_packages = []
138  for dep in python_requires:
139  if dep.lower() not in AddonInstaller.allowed_packages:
140  bad_packages.append(dep)
141 
142  for dep in bad_packages:
143  python_requires.remove(dep)
144 
145  if bad_packages:
146  # pylint: disable=line-too-long
147  message = (
148  "<p>"
149  + translate(
150  "AddonsInstaller",
151  "This addon requires Python packages that are not installed, and cannot be installed automatically. To use this addon you must install the following Python packages manually:",
152  )
153  + "</p><ul>"
154  )
155  if len(bad_packages) < 15:
156  for dep in bad_packages:
157  message += f"<li>{dep}</li>"
158  else:
159  message += "<li>(" + translate("AddonsInstaller", "Too many to list") + ")</li>"
160  message += "</ul>"
161  message += "To ignore this error and install anyway, press OK."
162  r = QtWidgets.QMessageBox.critical(
163  utils.get_main_am_window(),
164  translate("AddonsInstaller", "Missing Requirement"),
165  message,
166  QtWidgets.QMessageBox.Ok | QtWidgets.QMessageBox.Cancel,
167  )
168 
169  if r == QtWidgets.QMessageBox.Ok:
170  # Force the installation to proceed
171  return False
172  return True
173  return False

196  message += "<li>" + wb + "</li>"

232  def _check_python_version(self, missing: MissingDependencies) -> bool:
233  """Make sure we have a compatible Python version. Returns True to stop the installation
234  or False to continue."""
235 
236  # For now only look at the minor version, since major is always Python 3
237  minor_required = missing.python_min_version["minor"]
238  if sys.version_info.minor < minor_required:
239  # pylint: disable=line-too-long
240  QtWidgets.QMessageBox.critical(
241  utils.get_main_am_window(),
242  translate("AddonsInstaller", "Incompatible Python version"),
243  translate(
244  "AddonsInstaller",
245  "This Addon (or one of its dependencies) requires Python {}.{}, and your system is running {}.{}. Installation cancelled.",
246  ).format(
247  missing.python_min_version["major"],
248  missing.python_min_version["minor"],
249  sys.version_info.major,
250  sys.version_info.minor,
251  ),
252  QtWidgets.QMessageBox.Cancel,
253  )
254  return True
255  return False

447  def _cancel_addon_installation(self):
448  dlg = QtWidgets.QMessageBox(
449  QtWidgets.QMessageBox.NoIcon,
450  translate("AddonsInstaller", "Cancelling"),
451  translate("AddonsInstaller", "Cancelling installation of '{}'").format(

454  QtWidgets.QMessageBox.NoButton,
455  parent=utils.get_main_am_window(),
456  )
457  dlg.show()
458  if self.worker_thread.isRunning():
459  # Interruption can take a second or more, depending on what was being done. Make sure
460  # we stay responsive and update the dialog with the text above, etc.
461  self.worker_thread.requestInterruption()
462  self.worker_thread.quit()
463  while self.worker_thread.isRunning():
464  self.worker_thread.wait(50)
465  QtCore.QCoreApplication.processEvents(QtCore.QEventLoop.AllEvents)
466  path = os.path.join(self.installer.installation_path, self.addon_to_install.name)
467  if os.path.exists(path):
468  utils.rmdir(path)
469  dlg.hide()
470  self.finished.emit()
