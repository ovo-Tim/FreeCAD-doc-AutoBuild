39 def getCoincidentVertexes(vtx1, vtx2):
40  pairs = []
41  tol = Part.Precision.confusion()
42  for i in vtx1:
43  for j in vtx2:
44  if i.Point.distanceToPoint(j.Point) < tol:
45  pairs.append((i, j))
46 
47  return pairs
