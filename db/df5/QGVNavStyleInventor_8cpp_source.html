58 void QGVNavStyleInventor::handleMouseMoveEvent(QMouseEvent *event)
59 {
60  if (getViewer()->isBalloonPlacing()) {
61  balloonCursorMovement(event);
62  return;
63  }
64 
65  if ((QGuiApplication::mouseButtons() & Qt::LeftButton) &&
66  (QGuiApplication::mouseButtons() & Qt::MiddleButton)) {
67  //zoom mode 2 - LMB + MMB
68  if (zoomingActive) {
69  zoom(mouseZoomFactor(event->pos()));
70  } else {
71  startZoom(event->pos());
72  }
73  event->accept();
74  } else if (QGuiApplication::mouseButtons() & Qt::MiddleButton) {
75  //pan mode - MMB + move
76  if (panningActive) {
77  pan(event->pos());
78  } else {
79  startPan(event->pos());
80  }
81  event->accept();
82  }
83 }

85 void QGVNavStyleInventor::handleMouseReleaseEvent(QMouseEvent *event)
86 {
87  if (getViewer()->isBalloonPlacing()) {
88  placeBalloon(event->pos());
89  }
90 
91  if (event->button() == Qt::MiddleButton) {
92  //pan mode MMB
93  if (panningActive) {
94  stopPan();
95  event->accept();
96  }
97  }
98 
99  if ((event->button() == Qt::LeftButton) ||
100  (event->button() == Qt::MiddleButton) ){
101  //zoom mode 2 LMB + MMB
102  if (zoomingActive) {
103  zoomingActive = false;
104  event->accept();
105  }
106  }
107 }
