133  QString fileCardStyle() const

433 void StartView::postStart(PostStartBehavior behavior) const
434 {
435  auto hGrp = App::GetApplication().GetParameterGroupByPath(
436  "User parameter:BaseApp/Preferences/Mod/Start");
437 
438  if (behavior == PostStartBehavior::switchWorkbench) {
439  auto wb = hGrp->GetASCII("AutoloadModule", "");
440  if (wb == "$LastModule") {
441  wb = App::GetApplication()
442  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")
443  ->GetASCII("LastModule", "");
444  }
445  if (!wb.empty()) {
446  Gui::Application::Instance->activateWorkbench(wb.c_str());
447  }
448  }
449  auto closeStart = hGrp->GetBool("closeStart", false);
450  if (closeStart) {
451  this->window()->close();
452  }
453 }
