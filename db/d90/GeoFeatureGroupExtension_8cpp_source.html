84 DocumentObject* GeoFeatureGroupExtension::getGroupOfObject(const DocumentObject* obj)
85 {
86  if (!obj) {
87  return nullptr;
88  }
89 
90  // we will find origins, but not origin features
91  if (obj->isDerivedFrom(App::DatumElement::getClassTypeId())) {
92  return OriginGroupExtension::getGroupOfObject(obj);
93  }
94 
95  // compared to GroupExtension we do return here all GeoFeatureGroups including all extensions
96  // derived from it like OriginGroup. That is needed as we use this function to get all local
97  // coordinate systems. Also there is no reason to distinguish between GeoFeatuerGroups, there is
98  // only between group/geofeaturegroup
99  auto list = obj->getInList();
100  for (auto inObj : list) {
101 
102  // There is a chance that a derived geofeaturegroup links with a local link and hence is not
103  // the parent group even though it links to the object. We use hasObject as one and only
104  // truth if it has the object within the group
105  auto group = inObj->getExtensionByType<GeoFeatureGroupExtension>(true);
106  if (group && group->hasObject(obj)) {
107  return inObj;
108  }
109  }
110 
111  return nullptr;
112 }

126 Base::Placement GeoFeatureGroupExtension::recursiveGroupPlacement(
127  GeoFeatureGroupExtension* group,
128  std::unordered_set<GeoFeatureGroupExtension*>& history)
129 {
130  history.insert(this);
131 
132  auto inList = group->getExtendedObject()->getInList();
133  for (auto* link : inList) {
134  auto parent = link->getExtensionByType<GeoFeatureGroupExtension>(true);
135  if (parent && parent->hasObject(group->getExtendedObject())) {
136  // Cyclic dependencies detected
137  if (history.count(parent) > 0) {
138  break;
139  }
140  return recursiveGroupPlacement(parent, history) * group->placement().getValue();
141  }
142  }
143 
144  return group->placement().getValue();
145 }

148 GeoFeatureGroupExtension::addObjects(std::vector<App::DocumentObject*> objects)
149 {
150 
151  std::vector<DocumentObject*> grp = Group.getValues();
152  std::vector<DocumentObject*> ret;
153 
154  for (auto object : objects) {
155 
156  if (!allowObject(object)) {
157  continue;
158  }
159 
160  // cross CoordinateSystem links are not allowed, so we need to move the whole link group
161  std::vector<App::DocumentObject*> links = getCSRelevantLinks(object);
162  links.push_back(object);
163 
164  for (auto obj : links) {
165  // only one geofeaturegroup per object.
166  auto* group = App::GeoFeatureGroupExtension::getGroupOfObject(obj);
167  if (group && group != getExtendedObject()) {
168  group->getExtensionByType<App::GroupExtension>()->removeObject(obj);
169  }
170 
171  if (!hasObject(obj)) {
172  grp.push_back(obj);
173  ret.push_back(obj);
174  }
175  }
176  }
177 
178  Group.setValues(grp);
179  return ret;

183 GeoFeatureGroupExtension::removeObjects(std::vector<App::DocumentObject*> objects)
184 {
185 
186  std::vector<DocumentObject*> removed;
187  std::vector<DocumentObject*> grp = Group.getValues();
188 
189  for (auto object : objects) {
190  // cross CoordinateSystem links are not allowed, so we need to remove the whole link group
191  std::vector<DocumentObject*> links = getCSRelevantLinks(object);
192  links.push_back(object);
193 
194  // remove all links out of group
195  for (auto link : links) {
196  auto end = std::remove(grp.begin(), grp.end(), link);
197  if (end != grp.end()) {
198  grp.erase(end, grp.end());
199  removed.push_back(link);
200  }
201  }
202  }
203 
204  if (!removed.empty()) {
205  Group.setValues(grp);
206  }
207 
208  return removed;
209 }

211 void GeoFeatureGroupExtension::extensionOnChanged(const Property* p)
212 {
213 
214  // objects are only allowed in a single GeoFeatureGroup
215  if (p == &Group && !Group.testStatus(Property::User3)) {
216 
217  if ((!getExtendedObject()->isRestoring()
218  || getExtendedObject()->getDocument()->testStatus(Document::Importing))
219  && !getExtendedObject()->getDocument()->isPerformingTransaction()) {
220 
221  bool error = false;
222  auto corrected = Group.getValues();
223  for (auto obj : Group.getValues()) {
224 
225  // we have already set the obj into the group, so in a case of multiple groups
226  // getGroupOfObject would return anyone of it and hence it is possible that we miss
227  // an error. We need a custom check
228  auto list = obj->getInList();
229  for (auto in : list) {
230  if (in == getExtendedObject()) {
231  continue;
232  }
233  auto parent = in->getExtensionByType<GeoFeatureGroupExtension>(true);
234  if (parent && parent->hasObject(obj)) {
235  error = true;
236  corrected.erase(std::remove(corrected.begin(), corrected.end(), obj),
237  corrected.end());
238  }
239  }
240  }
241 
242  // if an error was found we need to correct the values and inform the user
243  if (error) {
244  Base::ObjectStatusLocker<Property::Status, Property> guard(Property::User3, &Group);
245  Group.setValues(corrected);
246  throw Base::RuntimeError("Object can only be in a single GeoFeatureGroup");
247  }
248  }
249  }
250 
251  App::GroupExtension::extensionOnChanged(p);
252 }

296 void GeoFeatureGroupExtension::getCSOutList(const App::DocumentObject* obj,
297  std::vector<DocumentObject*>& vec)
298 {
299 
300  if (!obj) {
301  return;
302  }
303 
304  // we get all relevant linked objects. We can't use outList() as this includes the links from
305  // expressions, also we only want links with scope Local
306  auto result = getScopedObjectsFromLinks(obj, LinkScope::Local);
307 
308  //we remove all links to origin features and origins, they belong to a CS too and can't be moved
309  result.erase(std::remove_if(result.begin(), result.end(), [](App::DocumentObject* obj)->bool {
310  return (obj->isDerivedFrom(App::DatumElement::getClassTypeId()) ||
311  obj->isDerivedFrom(App::Origin::getClassTypeId()));
312  }), result.end());
313 
314  vec.insert(vec.end(), result.begin(), result.end());
315 
316  // post process the vector
317  std::sort(vec.begin(), vec.end());
318  vec.erase(std::unique(vec.begin(), vec.end()), vec.end());
319 }

321 void GeoFeatureGroupExtension::getCSInList(const DocumentObject* obj,
322  std::vector<DocumentObject*>& vec)
323 {
324 
325  if (!obj) {
326  return;
327  }
328 
329  // search the inlist for objects that have non-expression links to us
330  for (App::DocumentObject* parent : obj->getInList()) {
331 
332  // not interested in other groups (and here we mean all groups, normal ones and
333  // geofeaturegroup)
334  if (parent->hasExtension(App::GroupExtension::getExtensionClassTypeId())) {
335  continue;
336  }
337 
338  // check if the link is real Local scope one or if it is a expression one (could also be
339  // both, so it is not enough to check the expressions)
340  auto res = getScopedObjectsFromLinks(parent, LinkScope::Local);
341  if (std::find(res.begin(), res.end(), obj) != res.end()) {
342  vec.push_back(parent);
343  }
344  }
345 
346  // clear all duplicates
347  std::sort(vec.begin(), vec.end());
348  vec.erase(std::unique(vec.begin(), vec.end()), vec.end());
349 }

371 void GeoFeatureGroupExtension::recursiveCSRelevantLinks(const DocumentObject* obj,
372  std::vector<DocumentObject*>& vec)
373 {
374 
375  if (!obj) {
376  return;
377  }
378 
379  std::vector<DocumentObject*> links;
380  getCSOutList(obj, links);
381  getCSInList(obj, links);
382 
383  // go on traversing the graph in all directions!
384  for (auto o : links) {
385  if (!o || o == obj || std::find(vec.begin(), vec.end(), o) != vec.end()) {
386  continue;
387  }
388 
389  vec.push_back(o);
390  recursiveCSRelevantLinks(o, vec);
391  }

394 bool GeoFeatureGroupExtension::extensionGetSubObject(DocumentObject*& ret,
395  const char* subname,
396  PyObject** pyObj,
397  Base::Matrix4D* mat,
398  bool transform,
399  int depth) const
400 {
401  ret = nullptr;
402  const char* dot;
403  if (!subname || *subname == 0) {
404  auto obj = dynamic_cast<const DocumentObject*>(getExtendedContainer());
405  ret = const_cast<DocumentObject*>(obj);
406  if (mat && transform) {
407  *mat *= const_cast<GeoFeatureGroupExtension*>(this)->placement().getValue().toMatrix();
408  }
409  }
410  else if ((dot = strchr(subname, '.'))) {
411  if (subname[0] != '$') {
412  ret = Group.findUsingMap(std::string(subname, dot));
413  }
414  else {
415  std::string name = std::string(subname + 1, dot);
416  for (auto child : Group.getValues()) {
417  if (name == child->Label.getStrValue()) {
418  ret = child;
419  break;
420  }
421  }
422  }
423  if (ret) {
424  if (dot) {
425  ++dot;
426  }
427  if (dot && *dot && !ret->hasExtension(App::LinkBaseExtension::getExtensionClassTypeId())
428  && !ret->hasExtension(App::GeoFeatureGroupExtension::getExtensionClassTypeId())) {
429  // Consider this
430  // Body
431  // | -- Pad
432  // | -- Sketch
433  //
434  // Suppose we want to getSubObject(Body,"Pad.Sketch.")
435  // Because of the special property of geo feature group, both
436  // Pad and Sketch are children of Body, so we can't call
437  // getSubObject(Pad,"Sketch.") as this will transform Sketch
438  // using Pad's placement.
439  //
440  const char* next = strchr(dot, '.');
441  if (next) {
442  App::DocumentObject* nret = nullptr;
443  extensionGetSubObject(nret, dot, pyObj, mat, transform, depth + 1);
444  if (nret) {
445  ret = nret;
446  return true;
447  }
448  }
449  }
450  if (mat && transform) {
451  *mat *=
452  const_cast<GeoFeatureGroupExtension*>(this)->placement().getValue().toMatrix();
453  }
454  ret = ret->getSubObject(dot ? dot : "", pyObj, mat, true, depth + 1);
455  }
456  }
457  return true;
458 }

478 bool GeoFeatureGroupExtension::isLinkValid(App::Property* prop)
479 {
480 
481  if (!prop) {
482  return true;
483  }
484 
485  // get the object that holds the property
486  if (!prop->getContainer()->isDerivedFrom(App::DocumentObject::getClassTypeId())) {
487  return true; // this link comes not from a document object, scopes are meaningless
488  }
489  auto obj = static_cast<App::DocumentObject*>(prop->getContainer());
490 
491  // no cross CS link for local links.
492  auto result = getScopedObjectsFromLink(prop, LinkScope::Local);
493  auto group = getGroupOfObject(obj);
494  for (auto link : result) {
495  if (getGroupOfObject(link) != group) {
496  return false;
497  }
498  }
499 
500  // for links with scope SubGroup we need to check if all features are part of subgroups
501  if (obj->hasExtension(App::GeoFeatureGroupExtension::getExtensionClassTypeId())) {
502  result = getScopedObjectsFromLink(prop, LinkScope::Child);
503  auto groupExt = obj->getExtensionByType<App::GeoFeatureGroupExtension>();
504  for (auto link : result) {
505  if (!groupExt->hasObject(link, true)) {
506  return false;
507  }
508  }
509  }
510 
511  return true;
512 }
