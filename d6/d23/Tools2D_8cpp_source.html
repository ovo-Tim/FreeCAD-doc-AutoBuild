37 double Vector2d::GetAngle(const Vector2d& vec) const
38 {
39  double fDivid = 0.0;
40  double fNum = 0.0;
41 
42  fDivid = Length() * vec.Length();
43 
44  if ((fDivid < -1e-10) || (fDivid > 1e-10)) {
45  fNum = (*this * vec) / fDivid;
46  if (fNum < -1) {
47  return D_PI;
48  }
49  if (fNum > 1) {
50  return 0.0;
51  }
52 
53  return acos(fNum);
54  }
55 
56  return -FLOAT_MAX; // division by zero
57 }

165 bool Line2d::Intersect(const Line2d& rclLine, Vector2d& rclV) const
166 {
167  double m1 = 0.0;
168  double m2 = 0.0;
169  double b1 = 0.0;
170  double b2 = 0.0;
171 
172  // calc coefficients
173  if (fabs(clV2.x - clV1.x) > 1e-10) {
174  m1 = (clV2.y - clV1.y) / (clV2.x - clV1.x);
175  }
176  else {
177  m1 = DOUBLE_MAX;
178  }
179  if (fabs(rclLine.clV2.x - rclLine.clV1.x) > 1e-10) {
180  m2 = (rclLine.clV2.y - rclLine.clV1.y) / (rclLine.clV2.x - rclLine.clV1.x);
181  }
182  else {
183  m2 = DOUBLE_MAX;
184  }
185  if (m1 == m2) { /****** RETURN ERR (parallel lines) *************/
186  return false;
187  }
188 
189  b1 = clV1.y - m1 * clV1.x;
190  b2 = rclLine.clV1.y - m2 * rclLine.clV1.x;
191 
192  // calc intersection
193  if (m1 == DOUBLE_MAX) {
194  rclV.x = clV1.x;
195  rclV.y = m2 * rclV.x + b2;
196  }
197  else if (m2 == DOUBLE_MAX) {
198  rclV.x = rclLine.clV1.x;
199  rclV.y = m1 * rclV.x + b1;
200  }
201  else {
202  rclV.x = (b2 - b1) / (m1 - m2);
203  rclV.y = m1 * rclV.x + b1;
204  }
205 
206  return true; /*** RETURN true (intersection) **********/
207 }

209 bool Line2d::Intersect(const Vector2d& rclV, double eps) const
210 {
211  double dxc = rclV.x - clV1.x;
212  double dyc = rclV.y - clV1.y;
213 
214  double dxl = clV2.x - clV1.x;
215  double dyl = clV2.y - clV1.y;
216 
217  double cross = dxc * dyl - dyc * dxl;
218 
219  // is point on the infinite line?
220  if (fabs(cross) > eps) {
221  return false;
222  }
223 
224  // point is on line but it is also between V1 and V2?
225  double dot = dxc * dxl + dyc * dyl;
226  double len = dxl * dxl + dyl * dyl;
227  return (dot >= -eps && dot <= len + eps);
228 }

262 static short CalcTorsion(const std::array<double, 4>& pfLine, double fX, double fY)
263 {
264  std::array<int, 2> sQuad {};
265  double fResX = 0.0;
266 
267  // Classification of both polygon points into quadrants
268  for (std::size_t i = 0; i < 2; i++) {
269  if (pfLine.at(i * 2) <= fX) {
270  sQuad[i] = (pfLine.at(i * 2 + 1) > fY) ? 0 : 3;
271  }
272  else {
273  sQuad[i] = (pfLine.at(i * 2 + 1) > fY) ? 1 : 2;
274  }
275  }
276 
277  // Abort at line points within a quadrant
278  // Abort at non-intersecting line points
279  if (abs(sQuad[0] - sQuad[1]) <= 1) {
280  return 0;
281  }
282 
283  // Both points to the left of ulX
284  if (abs(sQuad[0] - sQuad[1]) == 3) {
285  return (sQuad[0] == 0) ? 1 : -1;
286  }
287 
288  // Remaining cases: Quadrant difference from 2
289  // mathematical tests on intersection
290  fResX = pfLine[0] + (fY - pfLine[1]) / ((pfLine[3] - pfLine[1]) / (pfLine[2] - pfLine[0]));
291  if (fResX < fX) {
292 
293  // up/down or down/up
294  return (sQuad[0] <= 1) ? 1 : -1;
295  }
296 
297  // Remaining cases?
298  return 0;
299 }

301 bool Polygon2d::Contains(const Vector2d& rclV) const
302 {
303  // Using the number of turns method, determines
304  // whether a point is contained within a polygon.
305  // The sum of all turns indicates whether yes or no.
306  std::array<double, 4> pfTmp {};
307  unsigned long i = 0;
308  short sTorsion = 0;
309 
310  // Error check
311  if (GetCtVectors() < 3) {
312  return false;
313  }
314 
315  // for all polygon lines
316  for (i = 0; i < GetCtVectors(); i++) {
317  // Evidence of line structure
318  if (i == GetCtVectors() - 1) {
319  // Close polygon automatically
320  pfTmp[0] = _aclVct[i].x;
321  pfTmp[1] = _aclVct[i].y;
322  pfTmp[2] = _aclVct[0].x;
323  pfTmp[3] = _aclVct[0].y;
324  }
325  else {
326  // accept point i and i+1
327  pfTmp[0] = _aclVct[i].x;
328  pfTmp[1] = _aclVct[i].y;
329  pfTmp[2] = _aclVct[i + 1].x;
330  pfTmp[3] = _aclVct[i + 1].y;
331  }
332 
333  // Carry out a cut test and calculate the turn counter
334  sTorsion += CalcTorsion(pfTmp, rclV.x, rclV.y);
335  }
336 
337  // Evaluate turn counter
338  return sTorsion != 0;
339 }

341 void Polygon2d::Intersect(const Polygon2d& rclPolygon,
342  std::list<Polygon2d>& rclResultPolygonList) const
343 {
344  // trim the passed polygon with the current one, the result is a list of polygons (subset of the
345  // passed polygon) your own (trim) polygon is closed
346  //
347  if ((rclPolygon.GetCtVectors() < 2) || (GetCtVectors() < 2)) {
348  return;
349  }
350 
351  // position of first points (in or out of polygon)
352  bool bInner = Contains(rclPolygon[0]);
353 
354  Polygon2d clResultPolygon;
355  if (bInner) { // add first point if inner trim-polygon
356  clResultPolygon.Add(rclPolygon[0]);
357  }
358 
359  // for each polygon segment
360  size_t ulPolyCt = rclPolygon.GetCtVectors();
361  size_t ulTrimCt = GetCtVectors();
362  for (size_t ulVec = 0; ulVec < (ulPolyCt - 1); ulVec++) {
363  Vector2d clPt0 = rclPolygon[ulVec];
364  Vector2d clPt1 = rclPolygon[ulVec + 1];
365  Line2d clLine(clPt0, clPt1);
366 
367  // try to intersect with each line of the trim-polygon
368  std::set<double> afIntersections; // set of intersections (sorted by line parameter)
369  Vector2d clTrimPt2; // second line point
370  for (size_t i = 0; i < ulTrimCt; i++) {
371  clTrimPt2 = At((i + 1) % ulTrimCt);
372  Line2d clToTrimLine(At(i), clTrimPt2);
373 
374  Vector2d clV;
375  if (clLine.IntersectAndContain(clToTrimLine, clV)) {
376  // save line parameter of intersection point
377  double fDist = (clV - clPt0).Length();
378  afIntersections.insert(fDist);
379  }
380  }
381 
382  if (!afIntersections.empty()) // intersections founded
383  {
384  for (double it : afIntersections) {
385  // intersection point
386  Vector2d clPtIS = clLine.FromPos(it);
387  if (bInner) {
388  clResultPolygon.Add(clPtIS);
389  rclResultPolygonList.push_back(clResultPolygon);
390  clResultPolygon.DeleteAll();
391  bInner = false;
392  }
393  else {
394  clResultPolygon.Add(clPtIS);
395  bInner = true;
396  }
397  }
398 
399  if (bInner) { // add line end point if inside
400  clResultPolygon.Add(clPt1);
401  }
402  }
403  else { // no intersections, add line (means second point of it) if inside trim-polygon
404  if (bInner) {
405  clResultPolygon.Add(clPt1);
406  }
407  }
408  }
409 
410  // add last segment
411  if (clResultPolygon.GetCtVectors() > 0) {
412  rclResultPolygonList.push_back(clResultPolygon);
413  }
414 }
