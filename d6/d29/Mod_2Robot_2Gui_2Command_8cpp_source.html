220 void CmdRobotSimulate::activated(int)
221 {
222  const char* SelFilter = "SELECT Robot::RobotObject \n"
223  "SELECT Robot::TrajectoryObject ";
224 
225  Gui::SelectionFilter filter(SelFilter);
226  Robot::RobotObject* pcRobotObject;
227  Robot::TrajectoryObject* pcTrajectoryObject;
228 
229  if (filter.match()) {
230  pcRobotObject = static_cast<Robot::RobotObject*>(filter.Result[0][0].getObject());
231  pcTrajectoryObject = static_cast<Robot::TrajectoryObject*>(filter.Result[1][0].getObject());
232  }
233  else {
234  QMessageBox::warning(Gui::getMainWindow(),
235  QObject::tr("Wrong selection"),
236  QObject::tr("Select one Robot and one Trajectory object."));
237  return;
238  }
239 
240  if (pcTrajectoryObject->Trajectory.getValue().getSize() < 2) {
241  QMessageBox::warning(
242  Gui::getMainWindow(),
243  QObject::tr("Trajectory not valid"),
244  QObject::tr("You need at least two waypoints in a trajectory to simulate."));
245  return;
246  }
247 
248  Gui::TaskView::TaskDialog* dlg = new TaskDlgSimulate(pcRobotObject, pcTrajectoryObject);
249  Gui::Control().showDialog(dlg);
250 }
