146 void PropertyCosmeticVertexList::Restore(Base::XMLReader &reader)
147 {
148  // read my element
149  reader.clearPartialRestoreObject();
150  reader.readElement("CosmeticVertexList");
151  // get the value of my attribute
152  int count = reader.getAttributeAsInteger("count");
153  std::vector<CosmeticVertex*> values;
154  values.reserve(count);
155  for (int i = 0; i < count; i++) {
156  reader.readElement("CosmeticVertex");
157  const char* TypeName = reader.getAttribute("type");
158  CosmeticVertex *newG = static_cast<CosmeticVertex *>(Base::Type::fromName(TypeName).createInstance());
159  newG->Restore(reader);
160 
161  if(reader.testStatus(Base::XMLReader::ReaderStatus::PartialRestoreInObject)) {
162  Base::Console().Error("CosmeticVertex \"%s\" within a PropertyCosmeticVertexList was subject to a partial restore.\n", reader.localName());
163  if(isOrderRelevant()) {
164  // Pushes the best try by the CosmeticVertex class
165  values.push_back(newG);
166  }
167  else {
168  delete newG;
169  }
170  reader.clearPartialRestoreObject();
171  }
172  else {
173  values.push_back(newG);
174  }
175 
176  reader.readEndElement("CosmeticVertex");
177  }
178 
179  reader.readEndElement("CosmeticVertexList");

181  // assignment
