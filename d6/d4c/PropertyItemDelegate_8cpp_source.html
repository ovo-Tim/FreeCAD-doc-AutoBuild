61 void PropertyItemDelegate::paint(QPainter *painter, const QStyleOptionViewItem &opt, const QModelIndex &index) const
62 {
63  QStyleOptionViewItem option = opt;
64 
65  auto property = static_cast<PropertyItem*>(index.internalPointer());
66 
67  if (property && property->isSeparator()) {
68  QColor color = option.palette.color(QPalette::BrightText);
69  QObject* par = parent();
70  if (par) {
71  QVariant value = par->property("groupTextColor");
72  if (value.canConvert<QColor>())
73  color = value.value<QColor>();
74  }
75  option.palette.setColor(QPalette::Text, color);
76  option.font.setBold(true);
77 
78  // Since the group item now parents all the property items and can be
79  // collapsed, it makes sense to have some selection visual clue for it.
80  //
81  // option.state &= ~QStyle::State_Selected;
82  }
83  else if (index.column() == 1) {
84  option.state &= ~QStyle::State_Selected;
85  }
86 
87  option.state &= ~QStyle::State_HasFocus;
88 
89  if (property && property->isSeparator()) {
90  QBrush brush = option.palette.dark();
91  QObject* par = parent();
92  if (par) {
93  QVariant value = par->property("groupBackground");
94  if (value.canConvert<QBrush>())
95  brush = value.value<QBrush>();
96  }
97  painter->fillRect(option.rect, brush);
98  }
99 
100  QPen savedPen = painter->pen();
101 
102  QItemDelegate::paint(painter, option, index);
103 
104  QColor color = static_cast<QRgb>(QApplication::style()->styleHint(QStyle::SH_Table_GridLineColor, &opt, qobject_cast<QWidget*>(parent())));
105  painter->setPen(QPen(color));
106  if (index.column() == 1 || !(property && property->isSeparator())) {
107  int right = (option.direction == Qt::LeftToRight) ? option.rect.right() : option.rect.left();
108  painter->drawLine(right, option.rect.y(), right, option.rect.bottom());
109  }
110  painter->drawLine(option.rect.x(), option.rect.bottom(),
111  option.rect.right(), option.rect.bottom());
112  painter->setPen(savedPen);
113 }

125 bool PropertyItemDelegate::eventFilter(QObject *o, QEvent *ev)
126 {
127  if (ev->type() == QEvent::FocusOut) {
128  auto parentEditor = qobject_cast<PropertyEditor*>(this->parent());
129  auto widget = qobject_cast<QWidget*>(o);
130  if (widget && parentEditor && parentEditor->activeEditor
131  && widget != parentEditor->activeEditor) {
132  // All the attempts to ignore the focus-out event has been approved to not work
133  // reliably because there are still cases that cannot be handled.
134  // So, the best for now is to always ignore this event.
135  // See https://forum.freecad.org/viewtopic.php?p=579530#p579530 why this is not
136  // possible.
137  return false;
138  }
139  }
140  return QItemDelegate::eventFilter(o, ev);
141 }

143 QWidget * PropertyItemDelegate::createEditor (QWidget * parent, const QStyleOptionViewItem & /*option*/,
144  const QModelIndex & index ) const
145 {
146  if (!index.isValid())
147  return nullptr;
148 
149  auto childItem = static_cast<PropertyItem*>(index.internalPointer());
150  if (!childItem)
151  return nullptr;
152 
153  auto parentEditor = qobject_cast<PropertyEditor*>(this->parent());
154  if(parentEditor)
155  parentEditor->closeEditor();
156 
157  if (childItem->isSeparator())
158  return nullptr;
159 
160  FC_LOG("create editor " << index.row() << "," << index.column());
161 
162  QWidget* editor = nullptr;
163  expressionEditor = nullptr;
164  userEditor = nullptr;
165  if (parentEditor && parentEditor->isBinding()) {
166  expressionEditor = editor = childItem->createExpressionEditor(parent, [this]() {
167  const_cast<PropertyItemDelegate*>(this)->valueChanged(); // NOLINT
168  });
169  propertyEditor = editor;
170  }
171  else {
172  const auto &props = childItem->getPropertyData();
173  if (!props.empty() && props[0]->testStatus(App::Property::UserEdit)) {
174  editor = userEditor = childItem->createPropertyEditorWidget(parent);
175  propertyEditor = editor;
176  }
177  else {
178  editor = childItem->createEditor(parent, [this]() {
179  const_cast<PropertyItemDelegate*>(this)->valueChanged(); // NOLINT
180  });
181  propertyEditor = editor;
182  }
183  }
184  if (editor) {
185  // Make sure the editor background is painted so the cell content doesn't show through
186  editor->setAutoFillBackground(true);
187  }
188  if (editor && childItem->isReadOnly()) {
189  editor->setDisabled(true);
190  }
191  else if (editor /*&& this->pressed*/) {
192  // We changed the way editor is activated in PropertyEditor (in response
193  // of signal activated and clicked), so now we should grab focus
194  // regardless of "pressed" or not (e.g. when activated by keyboard
195  // enter)
196  editor->setFocus();
197  }
198  this->pressed = false;
199 
200  if (editor) {
201  const auto widgets = editor->findChildren<QWidget*>();
202  for (auto w : widgets) {
203  if (qobject_cast<QAbstractButton*>(w)
204  || qobject_cast<QLabel*>(w))
205  {
206  w->installEventFilter(const_cast<PropertyItemDelegate*>(this));
207  }
208  }
209  parentEditor->activeEditor = editor;
210  parentEditor->editingIndex = index;
211  }
212 
213  return editor;
214 }

224 void PropertyItemDelegate::setEditorData(QWidget *editor, const QModelIndex &index) const
225 {
226  if (!index.isValid())
227  return;
228  QVariant data = index.data(Qt::EditRole);
229  auto childItem = static_cast<PropertyItem*>(index.internalPointer());
230  editor->blockSignals(true);
231  if (expressionEditor == editor)
232  childItem->setExpressionEditorData(editor, data);
233  else if (userEditor == editor)
234  userEditor->setValue(PropertyItemAttorney::toString(childItem, data));
235  else
236  childItem->setEditorData(editor, data);
237  editor->blockSignals(false);
238  return;
239 }
