52 // returns the relative path length within the segment

54 double Path_Composite::Lookup(double s) const
55 {
56  assert(s>=-1e-12);
57  assert(s<=pathlength+1e-12);
58  if ( (cached_starts <=s) && ( s <= cached_ends) ) {
59  return s - cached_starts;
60  }
61  double previous_s=0;
62  for (unsigned int i=0;i<dv.size();++i) {
63  if ((s <= dv[i])||(i == (dv.size()-1) )) {
64  cached_index = i;
65  cached_starts = previous_s;
66  cached_ends = dv[i];
67  return s - previous_s;
68  }
69  previous_s = dv[i];
70  }
71  return 0;
72 }
