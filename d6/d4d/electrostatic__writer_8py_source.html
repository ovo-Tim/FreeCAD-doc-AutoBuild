42  def getElectrostaticSolver(self, equation):
43  # check if we need to update the equation
44  self._updateElectrostaticSolver(equation)
45  # output the equation parameters
46  s = self.write.createLinearSolver(equation)
47  s["Equation"] = "Stat Elec Solver" # equation.Name
48  s["Procedure"] = sifio.FileAttr("StatElecSolve/StatElecSolver")
49  s["Variable"] = self.write.getUniqueVarName("Potential")
50  s["Variable DOFs"] = 1
51  if equation.CalculateCapacitanceMatrix is True:
52  s["Calculate Capacitance Matrix"] = equation.CalculateCapacitanceMatrix
53  s["Capacitance Matrix Filename"] = equation.CapacitanceMatrixFilename
54  if equation.CalculateElectricEnergy is True:
55  s["Calculate Electric Energy"] = equation.CalculateElectricEnergy
56  if equation.CalculateElectricField is True:
57  s["Calculate Electric Field"] = equation.CalculateElectricField
58  if equation.CalculateElectricFlux is True:
59  s["Calculate Electric Flux"] = equation.CalculateElectricFlux
60  if equation.CalculateSurfaceCharge is True:
61  s["Calculate Surface Charge"] = equation.CalculateSurfaceCharge
62  if equation.ConstantWeights is True:
63  s["Constant Weights"] = equation.ConstantWeights
64  s["Exec Solver"] = "Always"
65  s["Optimize Bandwidth"] = True
66  if equation.CalculateCapacitanceMatrix is False and (equation.PotentialDifference != 0.0):
67  s["Potential Difference"] = equation.PotentialDifference
68  s["Stabilize"] = equation.Stabilize
69  return s
