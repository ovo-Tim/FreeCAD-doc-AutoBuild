53 def toolDepthAndOffset(width, extraDepth, tool, printInfo):
54  """toolDepthAndOffset(width, extraDepth, tool) ... return tuple for given\n
55  parameters."""
56 
57  if not hasattr(tool, "Diameter"):
58  raise ValueError("Deburr requires tool with diameter\n")
59 
60  suppressInfo = False
61  if hasattr(tool, "CuttingEdgeAngle"):
62  angle = float(tool.CuttingEdgeAngle)
63  if Path.Geom.isRoughly(angle, 180) or Path.Geom.isRoughly(angle, 0):
64  angle = 180
65  toolOffset = float(tool.Diameter) / 2
66  else:
67  if hasattr(tool, "TipDiameter"):
68  toolOffset = float(tool.TipDiameter) / 2
69  elif hasattr(tool, "FlatRadius"):
70  toolOffset = float(tool.FlatRadius)
71  else:
72  toolOffset = 0.0
73  if printInfo and not suppressInfo:
74  FreeCAD.Console.PrintMessage(
75  translate(
76  "PathDeburr",
77  "The selected tool has no FlatRadius and no TipDiameter property. Assuming {}\n".format(
78  "Endmill" if angle == 180 else "V-Bit"
79  ),
80  )
81  )
82  suppressInfo = True
83  else:
84  angle = 180
85  toolOffset = float(tool.Diameter) / 2
86  if printInfo:
87  FreeCAD.Console.PrintMessage(
88  translate(
89  "PathDeburr",
90  "The selected tool has no CuttingEdgeAngle property. Assuming Endmill\n",
91  )
92  )
93  suppressInfo = True
94 
95  tan = math.tan(math.radians(angle / 2))
96 
97  toolDepth = 0 if Path.Geom.isRoughly(tan, 0) else width / tan
98  depth = toolDepth + extraDepth
99  extraOffset = -width if angle == 180 else (extraDepth * tan)
100  offset = toolOffset + extraOffset
101 
102  return (depth, offset, extraOffset, suppressInfo)
