105 void QGICenterLine::setTools()
106 {
107  if (m_styleCurrent == Qt::DashDotLine) {
108  QVector<qreal> dashes;
109  qreal space = 4; // in unit width
110  qreal dash = 16;
111  // dot must be really small when using CapStyle RoundCap but > 0
112  // for CapStyle FlatCap you would need to set it to 1
113  qreal dot = 0.000001;
114 
115  dashes << dot << space << dash << space;
116  qreal dashlen = dot + 2 * space + dash;
117  qreal l_len = sqrt(pow(m_start.x() - m_end.x(), 2) + pow(m_start.y() - m_end.y(), 2)) / 2.0;
118  // convert from pixelunits to width units
119  l_len = l_len / m_width;
120  // note that the additional length using RoundCap or SquareCap does not
121  // count here!
122  if (m_isintersection) {
123  m_pen.setDashOffset(dashlen - fmod(l_len, dashlen) + space + dash / 2);
124  } else {
125  m_pen.setDashOffset(dashlen - fmod(l_len, dashlen));
126  }
127 
128  m_pen.setDashPattern(dashes);
129  }
130  else {
131  m_pen.setStyle(m_styleCurrent);
132  }
133  m_pen.setCapStyle(Qt::RoundCap);
134  m_pen.setWidthF(m_width);
135  m_pen.setColor(m_colCurrent);
136  m_line->setPen(m_pen);
137 }
