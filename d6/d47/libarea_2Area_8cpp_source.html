91 void CArea::ChangeStartToNearest(const Point* point, double min_dist)
92 {
93  for (std::list<CCurve>::iterator It = m_curves.begin(), ItNext = It; It != m_curves.end();
94  It = ItNext) {
95  ++ItNext;
96  if (It->m_vertices.size() <= 1) {
97  m_curves.erase(It);
98  }
99  }
100 
101  if (m_curves.empty()) {
102  return;
103  }
104 
105  std::list<CCurve> curves;
106  Point p;
107  if (point) {
108  p = *point;
109  }
110  if (min_dist < Point::tolerance) {
111  min_dist = Point::tolerance;
112  }
113 
114  while (m_curves.size()) {
115  std::list<CCurve>::iterator It = m_curves.begin();
116  std::list<CCurve>::iterator ItBest = It++;
117  Point best_point = ItBest->NearestPoint(p);
118  double best_dist = p.dist(best_point);
119  for (; It != m_curves.end(); ++It) {
120  const CCurve& curve = *It;
121  Point near_point;
122  double dist;
123  if (min_dist > Point::tolerance && !curve.IsClosed()) {
124  double d1 = curve.m_vertices.front().m_p.dist(p);
125  double d2 = curve.m_vertices.back().m_p.dist(p);
126  if (d1 < d2) {
127  dist = d1;
128  near_point = curve.m_vertices.front().m_p;
129  }
130  else {
131  dist = d2;
132  near_point = curve.m_vertices.back().m_p;
133  }
134  }
135  else {
136  near_point = curve.NearestPoint(p);
137  dist = near_point.dist(p);
138  }
139  if (dist < best_dist) {
140  best_dist = dist;
141  best_point = near_point;
142  ItBest = It;
143  }
144  }
145  if (ItBest->IsClosed()) {
146  ItBest->ChangeStart(best_point);
147  }
148  else {
149  double dfront = ItBest->m_vertices.front().m_p.dist(best_point);
150  double dback = ItBest->m_vertices.back().m_p.dist(best_point);
151  if (min_dist > Point::tolerance && dfront > min_dist && dback > min_dist) {
152  ItBest->Break(best_point);
153  m_curves.push_back(*ItBest);
154  m_curves.back().ChangeEnd(best_point);
155  ItBest->ChangeStart(best_point);
156  }
157  else if (dfront > dback) {
158  ItBest->Reverse();
159  }
160  }
161  curves.splice(curves.end(), m_curves, ItBest);
162  p = curves.back().m_vertices.back().m_p;
163  }
164  m_curves.splice(m_curves.end(), curves);
165 }

302 static void make_zig_curve(const CCurve& input_curve, double y0, double y)
303 {
304  CCurve curve(input_curve);
305 
306  if (rightward_for_zigs) {
307  if (curve.IsClockwise()) {
308  curve.Reverse();
309  }
310  }
311  else {
312  if (!curve.IsClockwise()) {
313  curve.Reverse();
314  }
315  }
316 
317  // find a high point to start looking from
318  Point top_left;
319  int top_left_index = 0;
320  bool top_left_found = false;
321  Point top_right;
322  int top_right_index = 0;
323  bool top_right_found = false;
324  Point bottom_left;
325  int bottom_left_index = 0;
326  bool bottom_left_found = false;
327 
328  int i = 0;
329  for (std::list<CVertex>::const_iterator VIt = curve.m_vertices.begin();
330  VIt != curve.m_vertices.end();
331  VIt++, i++) {
332  const CVertex& vertex = *VIt;
333 
334  test_y_point(i,
335  vertex.m_p,
336  top_right,
337  top_right_found,
338  top_right_index,
339  y,
340  !rightward_for_zigs);
341  test_y_point(i,
342  vertex.m_p,
343  top_left,
344  top_left_found,
345  top_left_index,
346  y,
347  rightward_for_zigs);
348  test_y_point(i,
349  vertex.m_p,
350  bottom_left,
351  bottom_left_found,
352  bottom_left_index,
353  y0,
354  rightward_for_zigs);
355  }
356 
357  int start_index = 0;
358  int end_index = 0;
359  int zag_end_index = 0;
360 
361  if (bottom_left_found) {
362  start_index = bottom_left_index;
363  }
364  else if (top_left_found) {
365  start_index = top_left_index;
366  }
367 
368  if (top_right_found) {
369  end_index = top_right_index;
370  zag_end_index = top_left_index;
371  }
372  else {
373  end_index = bottom_left_index;
374  zag_end_index = bottom_left_index;
375  }
376  if (end_index <= start_index) {
377  end_index += (i - 1);
378  }
379  if (zag_end_index <= start_index) {
380  zag_end_index += (i - 1);
381  }
382 
383  CCurve zig, zag;
384 
385  bool zig_started = false;
386  bool zig_finished = false;
387  bool zag_finished = false;
388 
389  int v_index = 0;
390  for (int i = 0; i < 2; i++) {
391  // process the curve twice because we don't know where it will start
392  if (zag_finished) {
393  break;
394  }
395  for (std::list<CVertex>::const_iterator VIt = curve.m_vertices.begin();
396  VIt != curve.m_vertices.end();
397  VIt++) {
398  if (i == 1 && VIt == curve.m_vertices.begin()) {
399  continue;
400  }
401 
402  const CVertex& vertex = *VIt;
403 
404  if (zig_finished) {
405  zag.m_vertices.push_back(unrotated_vertex(vertex));
406  if (v_index == zag_end_index) {
407  zag_finished = true;
408  break;
409  }
410  }
411  else if (zig_started) {
412  zig.m_vertices.push_back(unrotated_vertex(vertex));
413  if (v_index == end_index) {
414  zig_finished = true;
415  if (v_index == zag_end_index) {
416  zag_finished = true;
417  break;
418  }
419  zag.m_vertices.push_back(unrotated_vertex(vertex));
420  }
421  }
422  else {
423  if (v_index == start_index) {
424  zig.m_vertices.push_back(unrotated_vertex(vertex));
425  zig_started = true;
426  }
427  }
428  v_index++;
429  }
430  }
431 
432  if (zig_finished) {
433  zigzag_list_for_zigs.emplace_back(zig, zag);
434  }
435 }

447 void add_reorder_zig(ZigZag& zigzag)
448 {
449  // look in existing lists
450 
451  // see if the zag is part of an existing zig
452  if (zigzag.zag.m_vertices.size() > 1) {
453  const Point& zag_e = zigzag.zag.m_vertices.front().m_p;
454  bool zag_removed = false;
455  for (std::list<std::list<ZigZag>>::iterator It = reorder_zig_list_list.begin();
456  It != reorder_zig_list_list.end() && !zag_removed;
457  It++) {
458  std::list<ZigZag>& zigzag_list = *It;
459  for (std::list<ZigZag>::iterator It2 = zigzag_list.begin();
460  It2 != zigzag_list.end() && !zag_removed;
461  It2++) {
462  const ZigZag& z = *It2;
463  for (std::list<CVertex>::const_iterator It3 = z.zig.m_vertices.begin();
464  It3 != z.zig.m_vertices.end() && !zag_removed;
465  It3++) {
466  const CVertex& v = *It3;
467  if ((fabs(zag_e.x - v.m_p.x) < (0.002 * one_over_units))
468  && (fabs(zag_e.y - v.m_p.y) < (0.002 * one_over_units))) {
469  // remove zag from zigzag
470  zigzag.zag.m_vertices.clear();
471  zag_removed = true;
472  }
473  }
474  }
475  }
476  }
477 
478  // see if the zigzag can join the end of an existing list
479  const Point& zig_s = zigzag.zig.m_vertices.front().m_p;
480  for (std::list<std::list<ZigZag>>::iterator It = reorder_zig_list_list.begin();
481  It != reorder_zig_list_list.end();
482  It++) {
483  std::list<ZigZag>& zigzag_list = *It;
484  const ZigZag& last_zigzag = zigzag_list.back();
485  const Point& e = last_zigzag.zig.m_vertices.back().m_p;
486  if ((fabs(zig_s.x - e.x) < (0.002 * one_over_units))
487  && (fabs(zig_s.y - e.y) < (0.002 * one_over_units))) {
488  zigzag_list.push_back(zigzag);
489  return;
490  }
491  }
492 
493  // else add a new list
494  std::list<ZigZag> zigzag_list;
495  zigzag_list.push_back(zigzag);
496  reorder_zig_list_list.push_back(zigzag_list);
497 }

548 static void zigzag(const CArea& input_a)
549 {
550  if (input_a.m_curves.size() == 0) {
551  CArea::m_processing_done += CArea::m_single_area_processing_length;
552  return;
553  }
554 
555  one_over_units = 1 / CArea::m_units;
556 
557  CArea a(input_a);
558  rotate_area(a);
559 
560  CBox2D b;
561  a.GetBox(b);
562 
563  double x0 = b.MinX() - 1.0;
564  double x1 = b.MaxX() + 1.0;
565 
566  double height = b.MaxY() - b.MinY();
567  int num_steps = int(height / stepover_for_pocket + 1);
568  double y = b.MinY(); // + 0.1 * one_over_units;
569  Point null_point(0, 0);
570  rightward_for_zigs = true;
571 
572  if (CArea::m_please_abort) {
573  return;
574  }
575 
576  double step_percent_increment = 0.8 * CArea::m_single_area_processing_length / num_steps;
577 
578  for (int i = 0; i < num_steps; i++) {
579  double y0 = y;
580  y = y + stepover_for_pocket;
581  Point p0(x0, y0);
582  Point p1(x0, y);
583  Point p2(x1, y);
584  Point p3(x1, y0);
585  CCurve c;
586  c.m_vertices.emplace_back(0, p0, null_point, 0);
587  c.m_vertices.emplace_back(0, p1, null_point, 0);
588  c.m_vertices.emplace_back(0, p2, null_point, 1);
589  c.m_vertices.emplace_back(0, p3, null_point, 0);
590  c.m_vertices.emplace_back(0, p0, null_point, 1);
591  CArea a2;
592  a2.m_curves.push_back(c);
593  a2.Intersect(a);
594  make_zig(a2, y0, y);
595  rightward_for_zigs = !rightward_for_zigs;
596  if (CArea::m_please_abort) {
597  return;
598  }
599  CArea::m_processing_done += step_percent_increment;
600  }
601 
602  reorder_zigs();
603  CArea::m_processing_done += 0.2 * CArea::m_single_area_processing_length;
604 }

634 void CArea::MakePocketToolpath(std::list<CCurve>& curve_list, const CAreaPocketParams& params) const
635 {
636  double radians_angle = params.zig_angle * PI / 180;
637  sin_angle_for_zigs = sin(-radians_angle);
638  cos_angle_for_zigs = cos(-radians_angle);
639  sin_minus_angle_for_zigs = sin(radians_angle);
640  cos_minus_angle_for_zigs = cos(radians_angle);

643  CArea a_offset = *this;
