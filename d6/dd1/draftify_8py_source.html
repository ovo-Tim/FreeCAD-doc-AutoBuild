46 def draftify(objectslist, makeblock=False, delete=True):
47  """draftify(objectslist,[makeblock],[delete])
48 
49  Turn each object of the given list (objectslist can also be a single
50  object) into a Draft parametric wire.
51 
52  TODO: support more objects
53 
54  Parameters
55  ----------
56  objectslist :
57 
58  makeblock : bool
59  If makeblock is True, multiple objects will be grouped in a block.
60 
61  delete : bool
62  If delete = False, old objects are not deleted
63  """
64 
65  if not isinstance(objectslist,list):
66  objectslist = [objectslist]
67  newobjlist = []
68  for obj in objectslist:
69  if hasattr(obj,'Shape'):
70  for cluster in Part.sortEdges(obj.Shape.Edges):
71  w = Part.Wire(cluster)
72  nobj = draftify_shape(w)
73  if nobj is None:
74  nobj = App.ActiveDocument.addObject("Part::Feature", obj.Name)
75  nobj.Shape = w
76  newobjlist.append(nobj)
77  gui_utils.format_object(nobj, obj)
78  # sketches are always in wireframe mode. In Draft we don't like that!
79  if App.GuiUp:
80  nobj.ViewObject.DisplayMode = "Flat Lines"
81  if delete:
82  App.ActiveDocument.removeObject(obj.Name)
83 
84  if makeblock:
85  return make_block.make_block(newobjlist)
86  else:
87  if len(newobjlist) == 1:
88  return newobjlist[0]
89  return newobjlist
