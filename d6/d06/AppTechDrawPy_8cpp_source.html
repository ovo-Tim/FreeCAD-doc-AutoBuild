236  {

238  PyObject *inclBig = Py_True;
239  if (!PyArg_ParseTuple(args.ptr(), "O!|O", &(PyList_Type), &pcObj, &inclBig)) {
240  throw Py::TypeError("expected (listofedges, boolean");
241  }
242 
243  std::vector<TopoDS_Edge> edgeList;
244  try {
245  Py::Sequence list(pcObj);
246  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
247  if (PyObject_TypeCheck((*it).ptr(), &(Part::TopoShapeEdgePy::Type))) {
248  const TopoDS_Shape& shape = static_cast<TopoShapePy*>((*it).ptr())->
249  getTopoShapePtr()->getShape();
250  const TopoDS_Edge edge = TopoDS::Edge(shape);
251  edgeList.push_back(edge);
252  }
253  }
254  }
255  catch (Standard_Failure& e) {
256 
257  throw Py::Exception(Part::PartExceptionOCCError, e.GetMessageString());
258  }
259 
260  if (edgeList.empty()) {
261  return Py::None();
262  }
263 
264  bool biggie = false;
265  if (inclBig == Py_True) {
266  biggie = true;
267  }
268 
269  Py::List result;
270 
271  std::vector<TopoDS_Edge> closedEdges;
272  edgeList = DrawProjectSplit::scrubEdges(edgeList, closedEdges);
273 
274  std::vector<TopoDS_Wire> sortedWires;
275  try {
276  EdgeWalker eWalker;
277  sortedWires = eWalker.execute(edgeList, biggie);
278  }

280  e.setPyException();

282  }
283

285  Base::Console().Warning("ATDP::edgeWalker: Wire detection failed\n");
286  return Py::None();
287  }
288  else {
289  for (auto& w : sortedWires) {
290  PyObject* wire = new TopoShapeWirePy(new Part::TopoShape(w));

292  }
293  }
