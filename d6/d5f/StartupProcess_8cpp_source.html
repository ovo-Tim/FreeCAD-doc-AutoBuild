54 void StartupProcess::setupApplication()
55 {
56  QCoreApplication::setAttribute(Qt::AA_ShareOpenGLContexts);
57 
58 #if (QT_VERSION >= QT_VERSION_CHECK(5, 12, 0))
59  QCoreApplication::setAttribute(Qt::AA_UseDesktopOpenGL);
60 #endif
61 
62  // Automatic scaling for legacy apps (disable once all parts of GUI are aware of HiDpi)
63  ParameterGrp::handle hDPI =
64  App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/HighDPI");
65  bool disableDpiScaling = hDPI->GetBool("DisableDpiScaling", false);
66  if (disableDpiScaling) {
67 #ifdef FC_OS_WIN32
68  SetProcessDPIAware(); // call before the main event loop
69 #endif
70 #if QT_VERSION < QT_VERSION_CHECK(6,0,0)
71  QApplication::setAttribute(Qt::AA_DisableHighDpiScaling);
72 #endif
73  }
74  else {
75  // Enable automatic scaling based on pixel density of display (added in Qt 5.6)
76 #if QT_VERSION < QT_VERSION_CHECK(6,0,0)
77  QCoreApplication::setAttribute(Qt::AA_EnableHighDpiScaling);
78 #endif
79 #if QT_VERSION >= QT_VERSION_CHECK(5,14,0) && defined(Q_OS_WIN)
80  QGuiApplication::setHighDpiScaleFactorRoundingPolicy(Qt::HighDpiScaleFactorRoundingPolicy::PassThrough);
81 #endif
82  }
83 
84 #if QT_VERSION < QT_VERSION_CHECK(6,0,0)
85  //Enable support for highres images (added in Qt 5.1, but off by default)
86  QCoreApplication::setAttribute(Qt::AA_UseHighDpiPixmaps);
87 #endif
88 
89  // Use software rendering for OpenGL
90  ParameterGrp::handle hOpenGL =
91  App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/OpenGL");
92  bool useSoftwareOpenGL = hOpenGL->GetBool("UseSoftwareOpenGL", false);
93  if (useSoftwareOpenGL) {
94  QApplication::setAttribute(Qt::AA_UseSoftwareOpenGL);
95  }
96 
97 #if (QT_VERSION >= QT_VERSION_CHECK(5, 10, 0))
98  // By default (on platforms that support it, see docs for
99  // Qt::AA_CompressHighFrequencyEvents) QT applies compression
100  // for high frequency events (mouse move, touch, window resizes)
101  // to keep things smooth even when handling the event takes a
102  // while (e.g. to calculate snapping).
103  // However, tablet pen move events (and mouse move events
104  // synthesised from those) are not compressed by default (to
105  // allow maximum precision when e.g. hand-drawing curves),
106  // leading to unacceptable slowdowns using a tablet pen. Enable
107  // compression for tablet events here to solve that.
108  QCoreApplication::setAttribute(Qt::AA_CompressTabletEvents);
109 #endif
110 }

307 void StartupPostProcess::checkOpenGL()
308 {
309  QWindow window;
310  window.setSurfaceType(QWindow::OpenGLSurface);
311  window.create();
312 
313  QOpenGLContext context;
314  if (context.create()) {
315  context.makeCurrent(&window);
316  if (!context.functions()->hasOpenGLFeature(QOpenGLFunctions::Framebuffers)) {
317  Base::Console().Log("This system does not support framebuffer objects\n");
318  }
319  if (!context.functions()->hasOpenGLFeature(QOpenGLFunctions::NPOTTextures)) {
320  Base::Console().Log("This system does not support NPOT textures\n");
321  }
322 
323  int major = context.format().majorVersion();
324  int minor = context.format().minorVersion();
325 
326 #ifdef NDEBUG
327  // In release mode, issue a warning to users that their version of OpenGL is
328  // potentially going to cause problems
329  if (major < 2) {
330  auto message =
331  QObject::tr("This system is running OpenGL %1.%2. "
332  "FreeCAD requires OpenGL 2.0 or above. "
333  "Please upgrade your graphics driver and/or card as required.")
334  .arg(major)
335  .arg(minor)
336  + QStringLiteral("\n");
337  Base::Console().Warning(message.toStdString().c_str());
338  Dialog::DlgCheckableMessageBox::showMessage(
339  QCoreApplication::applicationName() + QStringLiteral(" - ")
340  + QObject::tr("Invalid OpenGL Version"),
341  message);
342  }
343 #endif
344  const char* glVersion = reinterpret_cast<const char*>(glGetString(GL_VERSION));
345  Base::Console().Log("OpenGL version is: %d.%d (%s)\n", major, minor, glVersion);
346  }
347 }

362 void StartupPostProcess::setBranding()
363 {
364  QString home = QString::fromStdString(App::Application::getHomePath());
365 
366  const std::map<std::string,std::string>& cfg = App::Application::Config();
367  std::map<std::string,std::string>::const_iterator it;
368  it = cfg.find("WindowTitle");
369  if (it != cfg.end()) {
370  QString title = QString::fromUtf8(it->second.c_str());
371  mainWindow->setWindowTitle(title);
372  }
373  it = cfg.find("WindowIcon");
374  if (it != cfg.end()) {
375  QString path = QString::fromUtf8(it->second.c_str());
376  if (QDir(path).isRelative()) {
377  path = QFileInfo(QDir(home), path).absoluteFilePath();
378  }
379  QApplication::setWindowIcon(QIcon(path));
380  }
381  it = cfg.find("ProgramLogo");
382  if (it != cfg.end()) {
383  QString path = QString::fromUtf8(it->second.c_str());
384  if (QDir(path).isRelative()) {
385  path = QFileInfo(QDir(home), path).absoluteFilePath();
386  }
387  QPixmap px(path);
388  if (!px.isNull()) {
389  auto logo = new QLabel();
390  logo->setPixmap(px.scaledToHeight(32));
391  mainWindow->statusBar()->addPermanentWidget(logo, 0);
392  logo->setFrameShape(QFrame::NoFrame);
393  }

437 void StartupPostProcess::activateWorkbench()
438 {
439  // Activate the correct workbench
440  std::string start = App::Application::Config()["StartWorkbench"];
441  Base::Console().Log("Init: Activating default workbench %s\n", start.c_str());
442  std::string autoload =
443  App::GetApplication()
444  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")
445  ->GetASCII("AutoloadModule", start.c_str());
446  if ("$LastModule" == autoload) {
447  start = App::GetApplication()
448  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")
449  ->GetASCII("LastModule", start.c_str());
450  }
451  else {
452  start = autoload;
453  }
454  // if the auto workbench is not visible then force to use the default workbech
455  // and replace the wrong entry in the parameters
456  QStringList wb = guiApp.workbenches();
457  if (!wb.contains(QString::fromLatin1(start.c_str()))) {
458  start = App::Application::Config()["StartWorkbench"];
459  if ("$LastModule" == autoload) {
460  App::GetApplication()
461  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")
462  ->SetASCII("LastModule", start.c_str());
463  }
464  else {
465  App::GetApplication()
466  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")
467  ->SetASCII("AutoloadModule", start.c_str());
468  }
469  }
470 
471  // Call this before showing the main window because otherwise:
472  // 1. it shows a white window for a few seconds which doesn't look nice
473  // 2. the layout of the toolbars is completely broken
474  guiApp.activateWorkbench(start.c_str());
475 
476  // show the main window
477  if (!Application::hiddenMainWindow()) {
478  Base::Console().Log("Init: Showing main window\n");
479  mainWindow->loadWindowSettings();
480  }
481 
482  //initialize spaceball.
483  if (auto fcApp = qobject_cast<GUIApplicationNativeEventAware*>(qtApp)) {
484  fcApp->initSpaceball(mainWindow);
485  }
486 
487  setStyleSheet();
488 
489  // Now run the background autoload, for workbenches that should be loaded at startup, but not
490  // displayed to the user immediately
491  autoloadModules(wb);
492 
493  // Reactivate the startup workbench
494  guiApp.activateWorkbench(start.c_str());
495 }

514 void StartupPostProcess::autoloadModules(const QStringList& wb)
515 {
516  // Now run the background autoload, for workbenches that should be loaded at startup, but not
517  // displayed to the user immediately
518  std::string autoloadCSV =
519  App::GetApplication()
520  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/General")
521  ->GetASCII("BackgroundAutoloadModules", "");
522 
523  // Tokenize the comma-separated list and load the requested workbenches if they exist in this
524  // installation
525  std::stringstream stream(autoloadCSV);
526  std::string workbench;
527  while (std::getline(stream, workbench, ',')) {
528  if (wb.contains(QString::fromLatin1(workbench.c_str()))) {
529  guiApp.activateWorkbench(workbench.c_str());
530  }
531  }
