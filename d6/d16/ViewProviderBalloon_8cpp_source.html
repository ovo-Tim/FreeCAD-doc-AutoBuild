192 std::vector<App::DocumentObject*> ViewProviderBalloon::claimChildren() const
193 {
194  // What can reasonably have a Balloon as a parent? A leader? a bit unconventional, but not forbidden.
195  // A RichAnno? Maybe? Another Balloon? Maybe?
196 
197  std::vector<App::DocumentObject*> temp;
198  const std::vector<App::DocumentObject*>& candidates = getViewObject()->getInList();
199  for (auto& obj : candidates) {
200  if (obj->isDerivedFrom<TechDraw::DrawViewBalloon>() ||
201  obj->isDerivedFrom<TechDraw::DrawLeaderLine>() ||
202  obj->isDerivedFrom<TechDraw::DrawRichAnno>() ) {
203  temp.push_back(obj);
204  }
205  }
206  return temp;
207 }
