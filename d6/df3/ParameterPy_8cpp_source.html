49 class ParameterGrpObserver: public ParameterGrp::ObserverType // NOLINT
50 {
51 public:
52  explicit ParameterGrpObserver(const Py::Object& obj)
53  : inst {obj}
54  {}
55  ParameterGrpObserver(const Py::Object& obj, const Py::Object& callable, ParameterGrp* target)
56  : callable(callable)
57  , _target(target)
58  , inst(obj)
59  {}
60  ~ParameterGrpObserver() override
61  {
62  Base::PyGILStateLocker lock;
63  inst = Py::None();
64  callable = Py::None();
65  }
66  void OnChange(ParameterGrp::SubjectType& rCaller, ParameterGrp::MessageType Reason) override
67  {
68  Base::PyGILStateLocker lock;
69  try {
70  auto& rGrp = dynamic_cast<ParameterGrp&>(rCaller);
71  ParameterGrp::handle hGrp(&rGrp);
72  Py::Callable method(this->inst.getAttr(std::string("onChange")));
73  Py::Tuple args(2);
74  args.setItem(0, Py::asObject(GetPyObject(hGrp)));
75  // A Reason of null indicates to clear the parameter group
76  if (Reason && Reason[0] != '\0') {
77  args.setItem(1, Py::String(Reason));
78  }
79  method.apply(args);
80  }
81  catch (Py::Exception&) {
82  Base::PyException e; // extract the Python error text
83  e.ReportException();
84  }
85  }
86  bool isEqual(const Py::Object& obj) const
87  {
88  return this->inst.is(obj);
89  }
90 
91 public:
92  // NOLINTBEGIN
93  Py::Object callable;
94  boost::signals2::scoped_connection conn;
95  ParameterGrp* _target = nullptr; // no reference counted, do not access
96  // NOLINTEND
97 
98 private:
99  Py::Object inst;
100 };

338 Py::Object ParameterGrpPy::getGroup(const Py::Tuple& args)

340  char* pstr = nullptr;
341  if (!PyArg_ParseTuple(args.ptr(), "s", &pstr)) {
342  throw Py::Exception();
343  }
344 
345  try {
346  // get the Handle of the wanted group
347  Base::Reference<ParameterGrp> handle = _cParamGrp->GetGroup(pstr);
