290 void DrawViewDimension::onChanged(const App::Property* prop)
291 {
292  if (prop == &References3D) {
293  // have to rebuild the Measurement object
294  clear3DMeasurements(); // Measurement object
295  if (!(References3D.getValues()).empty()) {
296  setAll3DMeasurement();
297  }
298  }
299 
300  if (isRestoring()) {
301  DrawView::onChanged(prop);
302  return;
303  }
304 
305  if (prop == &References2D) {
306  updateSavedGeometry();
307  }
308  else if (prop == &References3D) {
309  // remove the old measurement object
310  clear3DMeasurements();
311  if (!(References3D.getValues()).empty()) {
312  // rebuild the Measurement object
313  setAll3DMeasurement();
314  }
315  else if (MeasureType.isValue("True")) { // empty 3dRefs, but True
316  MeasureType.touch(); // run MeasureType logic for this case
317  }
318  updateSavedGeometry();
319  }
320  else if (prop == &Type) {
321  FormatSpec.setValue(getDefaultFormatSpec().c_str());
322  auto type = static_cast<DimensionType>(Type.getValue());
323  if (type == DimensionType::Angle || type == DimensionType::Angle3Pt) {
324  OverTolerance.setUnit(Base::Unit::Angle);
325  UnderTolerance.setUnit(Base::Unit::Angle);
326  }
327  else {
328  OverTolerance.setUnit(Base::Unit::Length);
329  UnderTolerance.setUnit(Base::Unit::Length);
330  }
331  }
332  else if (prop == &TheoreticalExact) {
333  // if TheoreticalExact disable tolerances and set them to zero
334  if (TheoreticalExact.getValue()) {
335  OverTolerance.setValue(0.0);
336  UnderTolerance.setValue(0.0);
337  OverTolerance.setReadOnly(true);
338  UnderTolerance.setReadOnly(true);
339  FormatSpecOverTolerance.setReadOnly(true);
340  FormatSpecUnderTolerance.setReadOnly(true);
341  ArbitraryTolerances.setValue(false);
342  ArbitraryTolerances.setReadOnly(true);
343  }
344  else {
345  OverTolerance.setReadOnly(false);
346  FormatSpecOverTolerance.setReadOnly(false);
347  ArbitraryTolerances.setReadOnly(false);
348  if (!EqualTolerance.getValue()) {
349  UnderTolerance.setReadOnly(false);
350  FormatSpecUnderTolerance.setReadOnly(false);
351  }
352  }
353  }
354  else if (prop == &EqualTolerance) {
355  // if EqualTolerance set negated overtolerance for untertolerance
356  // then also the OverTolerance must be positive
357  if (EqualTolerance.getValue()) {
358  // if OverTolerance is negative or zero, first set it to zero
359  if (OverTolerance.getValue() < 0) {
360  OverTolerance.setValue(0.0);
361  }
362  OverTolerance.setConstraints(&PositiveConstraint);
363  UnderTolerance.setValue(-1.0 * OverTolerance.getValue());
364  UnderTolerance.setUnit(OverTolerance.getUnit());
365  UnderTolerance.setReadOnly(true);
366  FormatSpecUnderTolerance.setValue(FormatSpecOverTolerance.getValue());
367  FormatSpecUnderTolerance.setReadOnly(true);
368  }
369  else {
370  OverTolerance.setConstraints(&ToleranceConstraint);
371  if (!TheoreticalExact.getValue()) {
372  UnderTolerance.setReadOnly(false);
373  FormatSpecUnderTolerance.setReadOnly(false);
374  }
375  }
376  }
377  else if (prop == &OverTolerance) {
378  // if EqualTolerance set negated overtolerance for untertolerance
379  if (EqualTolerance.getValue()) {
380  UnderTolerance.setValue(-1.0 * OverTolerance.getValue());
381  UnderTolerance.setUnit(OverTolerance.getUnit());
382  }
383  }
384  else if (prop == &FormatSpecOverTolerance) {
385  if (EqualTolerance.getValue() && !ArbitraryTolerances.getValue()) {
386  FormatSpecUnderTolerance.setValue(FormatSpecOverTolerance.getValue());
387  }
388  }
389  else if (prop == &FormatSpecUnderTolerance) {
390  if (EqualTolerance.getValue() && !ArbitraryTolerances.getValue()) {
391  FormatSpecOverTolerance.setValue(FormatSpecUnderTolerance.getValue());
392  }
393  }
394 
395  DrawView::onChanged(prop);
396 }

465 App::DocumentObjectExecReturn* DrawViewDimension::execute()
466 {
467  if (!okToProceed()) {
468  // if we set an error here, it will be triggered many times during
469  // document load.
470  return DrawView::execute();
471  }
472 
473  m_referencesCorrect = true;
474  if (Preferences::autoCorrectDimRefs()) {
475  m_referencesCorrect = autocorrectReferences();
476  }
477  if (!m_referencesCorrect) {
478  // this test needs Phase 2 of auto correct to be useful
479  Base::Console().Log("The references for %s have changed and autocorrect could not match the geometry\n", Label.getValue());
480  }
481 
482  resetLinear();
483  resetAngular();
484  resetArc();
485  resetArea();
486 
487  // we have either or both valid References3D and References2D
488  ReferenceVector references = getEffectiveReferences();
489 
490  if (Type.isValue("Distance") || Type.isValue("DistanceX") || Type.isValue("DistanceY")) {
491  if (getRefType() == oneEdge) {
492  m_linearPoints = getPointsOneEdge(references);
493  }
494  else if (getRefType() == twoEdge) {
495  m_linearPoints = getPointsTwoEdges(references);
496  }
497  else if (getRefType() == twoVertex) {
498  m_linearPoints = getPointsTwoVerts(references);
499  }
500  else if (getRefType() == vertexEdge) {
501  m_linearPoints = getPointsEdgeVert(references);
502  }
503  m_hasGeometry = true;
504  }
505  else if (Type.isValue("Radius") ||
506  Type.isValue("Diameter") ) {
507  m_arcPoints = getArcParameters(references);
508  m_hasGeometry = true;
509  }
510  else if (Type.isValue("Angle")) {
511  if (getRefType() != twoEdge) {
512  throw Base::RuntimeError("Angle dimension has non-edge references");
513  }
514  m_anglePoints = getAnglePointsTwoEdges(references);
515  m_hasGeometry = true;
516  }
517  else if (Type.isValue("Angle3Pt")) {
518  if (getRefType() != threeVertex) {
519  throw Base::RuntimeError("3 point angle dimension has non-vertex references");
520  }
521  m_anglePoints = getAnglePointsThreeVerts(references);
522  m_hasGeometry = true;
523  }
524  else if (Type.isValue("Area")) {
525  if (getRefType() != oneFace) {
526  throw Base::RuntimeError("area dimension has non-face references");
527  }
528  m_areaPoint = getAreaParameters(references);
529  m_hasGeometry = true;
530  }
531 
532  overrideKeepUpdated(false);
533  return DrawView::execute();
534 }

646 double DrawViewDimension::getDimValue()
647 {
648  constexpr double CircleDegrees{360.0};
649  double result = 0.0;
650  if (!has2DReferences() && !has3DReferences()) {
651  // nothing to measure
652  return result;
653  }
654  if (!getViewPart()) {
655  return result;
656  }
657 
658  if (!getViewPart()->hasGeometry()) { // happens when loading saved document
659  return result;
660  }
661 
662  if (MeasureType.isValue("True")) {
663  // True Values
664  if (!measurement->has3DReferences()) {
665  Base::Console().Warning("%s - True dimension has no 3D References\n",
666  getNameInDocument());
667  return result;
668  }
669  result = getTrueDimValue();
670  }
671  else {
672  // Projected Values
673  if (!checkReferences2D()) {
674  Base::Console().Warning("DVD::getDimValue - %s - 2D references are corrupt (5)\n",
675  getNameInDocument());
676  return result;
677  }
678  result = getProjectedDimValue();
679  }
680 
681  result = fabs(result);

721 {

724

726  pointPair pts = getLinearPoints();

728  if (dbv) {
729  // raw pts from view are inverted Y, so we need to un-invert them before mapping
730  // raw pts are scaled, so we need to unscale them for mapPoint2dFromView
731  // then rescale them for the distance calculation below
732  // centers are right side up
733  // if both points are on the expanded side of the last (rightmost/upmost) break
734  // then we should not move the points.
735  //
736  pts.invertY();

738  pts.scale(1 / scale);

741  pts.invertY();
742  pts.scale(scale);
743  }

747  }
