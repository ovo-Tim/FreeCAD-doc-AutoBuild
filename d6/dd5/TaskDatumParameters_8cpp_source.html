91 bool TaskDlgDatumParameters::accept() {
92 
93  Part::Datum* pcDatum = ViewProvider->getObject<Part::Datum>();
94  auto pcActiveBody = PartDesignGui::getBodyFor(pcDatum, false);
95  auto pcActivePart = PartDesignGui::getPartFor(pcActiveBody, false);
96  std::vector<App::DocumentObject*> copies;
97 
98  //see if we are able to assign a mode
99  if (parameter->getActiveMapMode() == mmDeactivated) {
100  QMessageBox msg;
101  msg.setWindowTitle(tr("Incompatible reference set"));
102  msg.setText(tr("There is no attachment mode that fits the current set"
103  " of references. If you choose to continue, the feature will remain where"
104  " it is now, and will not be moved as the references change."
105  " Continue?"));
106  msg.addButton(QMessageBox::Yes);
107  auto btNo = msg.addButton(QMessageBox::No);
108  msg.setDefaultButton(btNo);
109  msg.setIcon(QMessageBox::Warning);
110  msg.exec();
111  if (msg.buttonRole(msg.clickedButton()) == QMessageBox::NoRole)
112  return false;
113  }
114 
115  //see what to do with external references
116  //check the prerequisites for the selected objects
117  //the user has to decide which option we should take if external references are used
118  bool extReference = false;
119  for (App::DocumentObject* obj : pcDatum->AttachmentSupport.getValues()) {
120  if (pcActiveBody && !pcActiveBody->hasObject(obj) && !pcActiveBody->getOrigin()->hasObject(obj))
121  extReference = true;
122  }
123 
124  if(extReference) {
125  // TODO: rewrite this to be shared with CmdPartDesignNewSketch::activated() (2015-10-20, Fat-Zer)
126  QDialog dia(Gui::getMainWindow());
127  PartDesignGui::Ui_DlgReference dlg;
128  dlg.setupUi(&dia);
129  dia.setModal(true);
130  int result = dia.exec();
131  if (result == QDialog::DialogCode::Rejected)
132  return false;
133  else if (!dlg.radioXRef->isChecked()) {
134  std::vector<App::DocumentObject*> copyObjects;
135  std::vector<std::string> copySubValues;
136  std::vector<std::string> subs = pcDatum->AttachmentSupport.getSubValues();
137  int index = 0;
138  for (App::DocumentObject* obj : pcDatum->AttachmentSupport.getValues()) {
139  if (pcActiveBody && !pcActiveBody->hasObject(obj) && !pcActiveBody->getOrigin()->hasObject(obj)) {
140  auto* copy = PartDesignGui::TaskFeaturePick::makeCopy(obj, subs[index], dlg.radioIndependent->isChecked());
141  if (copy) {
142  copyObjects.push_back(copy);
143  copies.push_back(copyObjects.back());
144  copySubValues.emplace_back();
145  }
146  }
147  else {
148  copyObjects.push_back(obj);
149  copySubValues.push_back(subs[index]);
150  }
151 
152  index++;
153  }
154 
155  pcDatum->AttachmentSupport.setValues(copyObjects, copySubValues);
156  }
157  }
158 
159  if (!PartGui::TaskDlgAttacher::accept())
160  return false;
161 
162  //we need to add the copied features to the body after the command action, as otherwise FreeCAD crashes unexplainably
163  for(auto obj : copies) {
164  if (pcActiveBody)
165  pcActiveBody->addObject(obj);
166  else if (pcActivePart)
167  pcActivePart->addObject(obj);
168  }
169 
170  return true;
171 }
