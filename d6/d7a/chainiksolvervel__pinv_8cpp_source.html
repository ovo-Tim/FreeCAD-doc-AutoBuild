47  int ChainIkSolverVel_pinv::CartToJnt(const JntArray& q_in, const Twist& v_in, JntArray& qdot_out)
48  {
49  //Let the ChainJntToJacSolver calculate the jacobian "jac" for
50  //the current joint positions "q_in"
51  jnt2jac.JntToJac(q_in,jac);
52 
53  double sum;
54  unsigned int i,j;
55 
56  // Initialize near zero singular value counter
57  nrZeroSigmas = 0 ;
58 
59  //Do a singular value decomposition of "jac" with maximum
60  //iterations "maxiter", put the results in "U", "S" and "V"
61  //jac = U*S*Vt
62  svdResult = svd.calculate(jac,U,S,V,maxiter);
63  if (0 != svdResult)
64  {
65  qdot_out.data.setZero();
66  return (error = E_SVD_FAILED);
67  }
68 
69  // We have to calculate qdot_out = jac_pinv*v_in
70  // Using the svd decomposition this becomes(jac_pinv=V*S_pinv*Ut):
71  // qdot_out = V*S_pinv*Ut*v_in
72 
73  //first we calculate Ut*v_in
74  for (i=0;i<jac.columns();i++) {
75  sum = 0.0;
76  for (j=0;j<jac.rows();j++) {
77  sum+= U[j](i)*v_in(j);
78  }
79  //If the singular value is too small (<eps), don't invert it but
80  //set the inverted singular value to zero (truncated svd)
81  if ( fabs(S(i))<eps ) {
82  tmp(i) = 0.0 ;
83  // Count number of singular values near zero
84  ++nrZeroSigmas ;
85  }
86  else {
87  tmp(i) = sum/S(i) ;
88  }
89  }
90  //tmp is now: tmp=S_pinv*Ut*v_in, we still have to premultiply
91  //it with V to get qdot_out
92  for (i=0;i<jac.columns();i++) {
93  sum = 0.0;
94  for (j=0;j<jac.columns();j++) {
95  sum+=V[i](j)*tmp(j);
96  }
97  //Put the result in qdot_out
98  qdot_out(i)=sum;
99  }
100 
101  // Note if the solution is singular, i.e. if number of near zero
102  // singular values is greater than the full rank of jac
103  if ( nrZeroSigmas > (jac.columns()-jac.rows()) ) {
104  return (error = E_CONVERGE_PINV_SINGULAR); // converged but pinv singular
105  } else {
106  return (error = E_NOERROR); // have converged
107  }
108  }
