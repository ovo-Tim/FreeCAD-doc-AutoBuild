129 void DlgPrefsTechDrawAdvancedImp::loadBalloonOverride()
130 {
131  uint prefOverride = Preferences::balloonDragModifiers();
132  if (prefOverride == Qt::ControlModifier) {
133  // default case
134  ui->cbBalloonDefault->setChecked(true);
135  clearBalloonOptions();
136  enableBalloonOptions(false);
137  return;
138  }
139 
140  ui->cbBalloonDefault->setChecked(false);
141  enableBalloonOptions(true);
142 
143  if (flagsContainValue(prefOverride, Qt::ShiftModifier)) {
144  ui->cbBalloonShift->setChecked(true);
145  }
146  if (flagsContainValue(prefOverride, Qt::ControlModifier)) {
147  ui->cbBalloonControl->setChecked(true);
148  }
149 
150  if (flagsContainValue(prefOverride, Qt::AltModifier)) {
151  ui->cbBalloonAlt->setChecked(true);
152  }
153 
154  if (flagsContainValue(prefOverride, Qt::MetaModifier)) {
155  ui->cbBalloonMeta->setChecked(true);
156  }
157 }
