111 class QGVPage::Private: public ParameterGrp::ObserverType
112 {
113 public:
115  ParameterGrp::handle hGrp;
116  QGVPage* page;
117  explicit Private(QGVPage* page) : page(page)
118  {
119  // attach parameter Observer
120  hGrp = App::GetApplication().GetParameterGroupByPath(
121  "User parameter:BaseApp/Preferences/View");
122  hGrp->Attach(this);
123  }
124  void init()
125  {
126  page->m_atCursor = hGrp->GetBool("ZoomAtCursor", true);
127  page->m_invertZoom = hGrp->GetBool("InvertZoom", false);
128  page->m_zoomIncrement = hGrp->GetFloat("ZoomStep", 0.02);
129 
130  page->m_reversePan = Preferences::getPreferenceGroup("General")->GetInt("KbPan", 1);
131  page->m_reverseScroll = Preferences::getPreferenceGroup("General")->GetInt("KbScroll", 1);
132  }
134  void OnChange(ParameterGrp::SubjectType& rCaller, ParameterGrp::MessageType Reason) override
135  {
136  const ParameterGrp& rGrp = static_cast<ParameterGrp&>(rCaller);
137  if (strcmp(Reason, "NavigationStyle") == 0) {
138  std::string model =
139  rGrp.GetASCII("NavigationStyle", CADNavigationStyle::getClassTypeId().getName());
140  page->setNavigationStyle(model);
141  }
142  else if (strcmp(Reason, "InvertZoom") == 0) {
143  page->m_invertZoom = rGrp.GetBool("InvertZoom", true);
144  }
145  else if (strcmp(Reason, "ZoomStep") == 0) {
146  page->m_zoomIncrement = rGrp.GetFloat("ZoomStep", 0.0f);
147  }
148  else if (strcmp(Reason, "ZoomAtCursor") == 0) {
149  page->m_atCursor = rGrp.GetBool("ZoomAtCursor", true);
150  if (page->m_atCursor) {
151  page->setResizeAnchor(QGVPage::AnchorUnderMouse);
152  page->setTransformationAnchor(QGVPage::AnchorUnderMouse);
153  }
154  else {
155  page->setResizeAnchor(QGVPage::AnchorViewCenter);
156  page->setTransformationAnchor(QGVPage::AnchorViewCenter);
157  }
158  }
159  }
160  void detach()
161  {
162  hGrp = App::GetApplication().GetParameterGroupByPath(
163  "User parameter:BaseApp/Preferences/View");
164  hGrp->Detach(this);
165  }
166 };

325 void QGVPage::drawBackground(QPainter* painter, const QRectF&)
326 {
327  //Note: Background is not part of scene()
328  if (!drawBkg)
329  return;
330 
331  if (!m_vpPage) {
332  return;
333  }
334 
335  if (!m_vpPage->getDrawPage()) {
336  // Base::Console().Message("QGVP::drawBackground - no Page Feature!\n");
337  return;
338  }
339 
340  painter->save();
341  painter->resetTransform();
342 
343  painter->setBrush(*bkgBrush);
344  painter->drawRect(
345  viewport()->rect().adjusted(-2, -2, 2, 2));//just bigger than viewport to prevent artifacts
346 
347  // Default to A3 landscape, though this is currently relevant
348  // only for opening corrupt docs, etc.
349  float pageWidth = 420, pageHeight = 297;
350 
351  if (m_vpPage->getDrawPage()->hasValidTemplate()) {
352  pageWidth = Rez::guiX(m_vpPage->getDrawPage()->getPageWidth());
353  pageHeight = Rez::guiX(m_vpPage->getDrawPage()->getPageHeight());
354  }
355 
356  // Draw the white page
357  QRectF paperRect(0, -pageHeight, pageWidth, pageHeight);
358  QPolygon poly = mapFromScene(paperRect);
359 
360  QBrush pageBrush(PreferencesGui::pageQColor());
361  painter->setBrush(pageBrush);
362 
363  painter->drawRect(poly.boundingRect());
364 
365  painter->restore();
366 }

572 QPixmap QGVPage::prepareCursorPixmap(const char* iconName, QPoint& hotspot)
573 {
574 
575  QPointF floatHotspot(hotspot);
576  double pixelRatio = getDevicePixelRatio();
577 
578  // Due to impossibility to query cursor size via Qt API, we stick to (32x32)*device_pixel_ratio
579  // as FreeCAD Wiki suggests - see https://wiki.freecad.org/HiDPI_support#Custom_cursor_size
580  double cursorSize = 32.0 * pixelRatio;
581 
582  QPixmap pixmap = Gui::BitmapFactory().pixmapFromSvg(iconName, QSizeF(cursorSize, cursorSize));
583  pixmap.setDevicePixelRatio(pixelRatio);
584 
585  // The default (and here expected) SVG cursor graphics size is 64x64 pixels, thus we must adjust
586  // the 64x64 based hotspot position for our 32x32 based cursor pixmaps accordingly
587  floatHotspot *= 0.5;
588 
589 #if !defined(Q_OS_WIN32) && !defined(Q_OS_MACOS)
590  // On XCB platform, the pixmap device pixel ratio is not taken into account for cursor hot spot,
591  // therefore we must take care of the transformation ourselves...
592  // Refer to QTBUG-68571 - https://bugreports.qt.io/browse/QTBUG-68571
593  if (qGuiApp->platformName() == QLatin1String("xcb")) {
594  floatHotspot *= pixelRatio;
595  }
596 #endif
597 
598  hotspot = floatHotspot.toPoint();
599  return pixmap;
600 }
