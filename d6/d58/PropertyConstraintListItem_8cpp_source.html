65 void PropertyConstraintListItem::initialize()
66 {
67  const Sketcher::PropertyConstraintList* list =
68  static_cast<const Sketcher::PropertyConstraintList*>(getPropertyData()[0]);
69  const std::vector<Sketcher::Constraint*>& vals = list->getValues();
70 
71  int id = 1;
72  int iNamed = 0;
73 
74  std::vector<PropertyUnitItem*> unnamed;
75 
76  for (std::vector<Sketcher::Constraint*>::const_iterator it = vals.begin(); it != vals.end();
77  ++it, ++id) {
78  if ((*it)->Type == Sketcher::Distance || // Datum constraint
79  (*it)->Type == Sketcher::DistanceX || (*it)->Type == Sketcher::DistanceY
80  || (*it)->Type == Sketcher::Radius || (*it)->Type == Sketcher::Diameter
81  || (*it)->Type == Sketcher::Angle) {
82 
83  PropertyUnitItem* item = static_cast<PropertyUnitItem*>(PropertyUnitItem::create());
84 
85  // Get the name
86  QString internalName = QString::fromLatin1("Constraint%1").arg(id);
87  QString name = QString::fromUtf8((*it)->Name.c_str());
88  if (name.isEmpty()) {
89  name = internalName;
90  item->setPropertyName(name);
91  unnamed.push_back(item);
92  }
93  else {
94  iNamed++;
95  item->setParent(this);
96  item->setPropertyName(name);
97  // The call of 'setPropertyName' calls 'setObjectName'. But 'name' can contain
98  // some non-7-bit ASCII characters and thus the delegation of a property to the
99  // parent item may fail because 'qPrintable(objectName())' is used therefore and
100  // may return a different string. See 'PropertyItem::setData()' for more details.
101  // To make the delegation to work properly we set a different object name which is
102  // guaranteed to be 7-bit ASCII.
103  //
104  // See also PropertyConstraintListItem::value()
105  // See also PropertyConstraintListItem::event()
106  item->setObjectName(internalName);
107  this->appendChild(item);
108  }
109 
110  item->bind(list->createPath(id - 1));
111  item->setAutoApply(false);
112  }
113  }
114 
115  // now deal with the unnamed
116  if (iNamed == 0) {
117  onlyUnnamed = true;
118  for (std::vector<PropertyUnitItem*>::const_iterator it = unnamed.begin();
119  it != unnamed.end();
120  ++it) {
121  (*it)->setParent(this);
122  this->appendChild((*it));
123  }
124  }
125  else {
126  onlyUnnamed = false;
127  if (!unnamed.empty()) {
128  PropertyConstraintListItem* item =
129  static_cast<PropertyConstraintListItem*>(PropertyConstraintListItem::create());
130  item->setParent(this);
131  item->setPropertyName(tr("Unnamed"));
132  this->appendChild(item);
133 
134  for (std::vector<PropertyUnitItem*>::const_iterator it = unnamed.begin();
135  it != unnamed.end();
136  ++it) {
137  (*it)->setParent(item);
138  item->appendChild((*it));
139  }
140  }
141  }
142 }

144 void PropertyConstraintListItem::assignProperty(const App::Property* prop)
145 {
146  // Hint: When renaming a constraint that was unnamed before then it can happen that
147  // a constraint appears twice in the property editor, one time in this group and a
148  // second time inside the Unnamed group
149  if (!prop->isDerivedFrom<Sketcher::PropertyConstraintList>()) {
150  return;
151  }
152 
153  const Sketcher::PropertyConstraintList* list =
154  static_cast<const Sketcher::PropertyConstraintList*>(prop);
155  const std::vector<Sketcher::Constraint*>& vals = list->getValues();
156 
157  // search for the group of unnamed items if available and take it out
158  int numUnnamed = 0;
159  PropertyConstraintListItem* unnamed = nullptr;
160  for (int i = this->childCount() - 1; i >= 0; i--) {
161  unnamed = qobject_cast<PropertyConstraintListItem*>(this->child(i));
162  if (unnamed) {
163  numUnnamed = unnamed->childCount();
164  this->takeChild(i);
165  break;
166  }
167  }
168 
169  int id = 1;
170  int namedIndex = 0;
171  int unnamedIndex = 0;
172  int numNamed = this->childCount();
173  this->onlyUnnamed = true;
174 
175  for (std::vector<Sketcher::Constraint*>::const_iterator it = vals.begin(); it != vals.end();
176  ++it, ++id) {
177  if ((*it)->Type == Sketcher::Distance || // Datum constraint
178  (*it)->Type == Sketcher::DistanceX || (*it)->Type == Sketcher::DistanceY
179  || (*it)->Type == Sketcher::Radius || (*it)->Type == Sketcher::Diameter
180  || (*it)->Type == Sketcher::Angle) {
181 
182  PropertyUnitItem* child = nullptr;
183  if ((*it)->Name.empty()) {
184  // search inside the group item for unnamed constraints
185  if (!unnamed) {
186  unnamed = static_cast<PropertyConstraintListItem*>(
187  PropertyConstraintListItem::create());
188  unnamed->setPropertyName(tr("Unnamed"));
189  }
190 
191  if (unnamedIndex < numUnnamed) {
192  child = static_cast<PropertyUnitItem*>(unnamed->child(unnamedIndex));
193  }
194  else {
195  child = static_cast<PropertyUnitItem*>(PropertyUnitItem::create());
196  unnamed->appendChild(child);
197  child->setParent(unnamed);
198  }
199  unnamedIndex++;
200  }
201  else {
202  // search inside this item
203  if (namedIndex < numNamed) {
204  child = dynamic_cast<PropertyUnitItem*>(this->child(namedIndex));
205  }
206 
207  if (!child) {
208  child = static_cast<PropertyUnitItem*>(PropertyUnitItem::create());
209  this->appendChild(child);
210  child->setParent(this);
211  }
212  namedIndex++;
213  this->onlyUnnamed = false;
214  }
215 
216  // Get the name
217  QString internalName = QString::fromLatin1("Constraint%1").arg(id);
218  QString name = QString::fromUtf8((*it)->Name.c_str());
219  if (name.isEmpty()) {
220  name = internalName;
221  }
222 
223  if (child->objectName() != internalName) {
224  child->setPropertyName(name);
225  child->setObjectName(internalName);
226 
227  child->bind(list->createPath(id - 1));
228  child->setAutoApply(false);
229  }
230  }
231  }
232 
233  // at the Unnamed group at very the end
234  if (unnamed) {
235  this->appendChild(unnamed);
236  unnamed->setParent(this);
237  }
238 }

240 QVariant PropertyConstraintListItem::value(const App::Property* prop) const
241 {
242  assert(prop && prop->isDerivedFrom<Sketcher::PropertyConstraintList>());
243 
244  PropertyConstraintListItem* self = const_cast<PropertyConstraintListItem*>(this);
245 
246  int id = 1;
247 
248  QList<Base::Quantity> quantities;
249  QList<Base::Quantity> subquantities;
250  bool onlyNamed = true;
251 
252  const std::vector<Sketcher::Constraint*>& vals =
253  static_cast<const Sketcher::PropertyConstraintList*>(prop)->getValues();
254  for (std::vector<Sketcher::Constraint*>::const_iterator it = vals.begin(); it != vals.end();
255  ++it, ++id) {
256  if ((*it)->Type == Sketcher::Distance || // Datum constraint
257  (*it)->Type == Sketcher::DistanceX || (*it)->Type == Sketcher::DistanceY
258  || (*it)->Type == Sketcher::Radius || (*it)->Type == Sketcher::Diameter
259  || (*it)->Type == Sketcher::Angle) {
260 
261  Base::Quantity quant;
262  if ((*it)->Type == Sketcher::Angle) {
263  double datum = Base::toDegrees<double>((*it)->getValue());
264  quant.setUnit(Base::Unit::Angle);
265  quant.setValue(datum);
266  }
267  else {
268  quant.setUnit(Base::Unit::Length);
269  quant.setValue((*it)->getValue());
270  }
271 
272  quantities.append(quant);
273 
274  // Use a 7-bit ASCII string for the internal name.
275  // See also comment in PropertyConstraintListItem::initialize()
276  QString internalName = QString::fromLatin1("Constraint%1").arg(id);
277 
278  if ((*it)->Name.empty() && !onlyUnnamed) {
279  onlyNamed = false;
280  subquantities.append(quant);
281  PropertyItem* child = self->child(self->childCount() - 1);
282  PropertyConstraintListItem* unnamednode =
283  qobject_cast<PropertyConstraintListItem*>(child);
284  if (unnamednode) {
285  unnamednode->blockEvent = true;
286  unnamednode->setProperty(internalName.toLatin1(),
287  QVariant::fromValue<Base::Quantity>(quant));
288  unnamednode->blockEvent = false;
289  }
290  else {
291  qWarning() << "Item is not of type PropertyConstraintListItem but"
292  << typeid(*child).name();
293  }
294  }
295  else {
296  self->blockEvent = true;
297  self->setProperty(internalName.toLatin1(),
298  QVariant::fromValue<Base::Quantity>(quant));
299  self->blockEvent = false;
300  }
301  }
302  }
303 
304  // The quantities of unnamed constraints are only needed for display purposes inside toString()
305  if (!onlyUnnamed && !onlyNamed) {
306  self->blockEvent = true;
307  self->setProperty("Unnamed", QVariant::fromValue<QList<Base::Quantity>>(subquantities));
308  self->blockEvent = false;
309  }
310 
311  return QVariant::fromValue<QList<Base::Quantity>>(quantities);
312 }

314 bool PropertyConstraintListItem::event(QEvent* ev)
315 {
316  if (ev->type() == QEvent::DynamicPropertyChange) {
317  if (!blockEvent) {
318  QDynamicPropertyChangeEvent* ce = static_cast<QDynamicPropertyChangeEvent*>(ev);
319  // Get property via internal name of a PropertyUnit
320  QVariant prop = property(ce->propertyName());
321  QString propName = QString::fromLatin1(ce->propertyName());
322  Base::Quantity quant = prop.value<Base::Quantity>();
323 
324  Sketcher::PropertyConstraintList* item;
325 
326  int id = 0;
327  if (dynamic_cast<SketcherGui::PropertyConstraintListItem*>(this->parent())) {
328  item = static_cast<Sketcher::PropertyConstraintList*>(
329  this->parent()->getFirstProperty());
330  }
331  else {
332  item = static_cast<Sketcher::PropertyConstraintList*>(getFirstProperty());
333  }
334 
335  const std::vector<Sketcher::Constraint*>& vals = item->getValues();
336  for (std::vector<Sketcher::Constraint*>::const_iterator it = vals.begin();
337  it != vals.end();
338  ++it, ++id) {
339  if ((*it)->Type == Sketcher::Distance || // Datum constraint
340  (*it)->Type == Sketcher::DistanceX || (*it)->Type == Sketcher::DistanceY
341  || (*it)->Type == Sketcher::Radius || (*it)->Type == Sketcher::Diameter
342  || (*it)->Type == Sketcher::Angle) {
343 
344  // Get the internal name
345  QString internalName = QString::fromLatin1("Constraint%1").arg(id + 1);
346  if (internalName == propName) {
347  double datum = quant.getValue();
348  if ((*it)->Type == Sketcher::Angle) {
349  datum = Base::toRadians<double>(datum);
350  }
351  std::unique_ptr<Sketcher::Constraint> copy((*it)->clone());
352  copy->setValue(datum);
353  item->set1Value(id, copy.get());
354  break;
355  }
356  }
357  }
358  }
359  }
360 
361  return PropertyItem::event(ev);
362 }
