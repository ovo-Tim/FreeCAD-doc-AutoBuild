150 {
151  beginInsertRows(parent, row, row + count - 1);
152 
153  int columns = columnCount();
154  for (int i = 0; i < count; i++) {
155  auto rowPtr = std::make_shared<QList<QVariant>>();
156  for (int j = 0; j < columns; j++) {
157  rowPtr->push_back(_property->getColumnNull(j));
158  }

160  _value->insertRow(row, rowPtr);
161  }
162 
163  endInsertRows();
164
