59 def import_z88_disp(filename, analysis=None, result_name_prefix=None):
60  """insert a FreeCAD FEM mechanical result object in the ActiveDocument
61  pure usage:
62  import feminout.importZ88O2Results as importZ88O2Results
63  disp_file = "/pathtofile/z88o2.txt"
64  importZ88O2Results.import_z88_disp(disp_file)
65 
66  the z888i1.txt FEMMesh file needs to be in the same directory as z88o2.txt
67  # ahh, make a new document first ;-)
68  """
69  import ObjectsFem
70  from . import importZ88Mesh
71  from . import importToolsFem
72  from femresult import resulttools
73 
74  if result_name_prefix is None:
75  result_name_prefix = ""
76  disp_read = read_z88_disp(filename)
77  result_mesh_object = None
78  if len(disp_read["Nodes"]) > 0:
79  if analysis:
80  analysis_object = analysis
81 
82  # read result mesh
83  if filename.endswith("z88o2.txt"):
84  mesh_file = filename.replace("o2", "i1")
85  mesh_data = importZ88Mesh.read_z88_mesh(mesh_file)
86  femmesh = importToolsFem.make_femmesh(mesh_data)
87  result_mesh_object = ObjectsFem.makeMeshResult(FreeCAD.ActiveDocument, "Result_mesh")
88  result_mesh_object.FemMesh = femmesh
89  else:
90  Console.PrintError("Z88 mesh file z88i1.txt not found.\n")
91  return None
92 
93  # create result obj
94  for result_set in disp_read["Results"]:
95  results_name = result_name_prefix + "results"
96 
97  res_obj = ObjectsFem.makeResultMechanical(FreeCAD.ActiveDocument, results_name)
98  res_obj.Mesh = result_mesh_object
99  res_obj = importToolsFem.fill_femresult_mechanical(res_obj, result_set)
100  res_obj = resulttools.add_disp_apps(res_obj) # fill DisplacementLengths
101  res_obj = resulttools.fill_femresult_stats(res_obj) # fill Stats
102 
103  if analysis:
104  analysis_object.addObject(res_obj)
105 
106  if FreeCAD.GuiUp:
107  if analysis:
108  import FemGui
109 
110  FemGui.setActiveAnalysis(analysis_object)
111  FreeCAD.ActiveDocument.recompute()
112 
113  else:
114  Console.PrintError(
115  "Problem on Z88 result file import. No nodes found in Z88 result file.\n"
116  )
117  return None
118  return res_obj

121 def read_z88_disp(z88_disp_input):
122  """
123  read a z88 disp file and extract the nodes and elements
124  z88 Displacement output file is z88o2.txt
125  works with Z88OS14
126  """
127  nodes = {}
128  mode_disp = {}
129  mode_results = {}
130  results = []
131 
132  z88_disp_file = pyopen(z88_disp_input, "r")
133 
134  for no, line in enumerate(z88_disp_file):
135  lno = no + 1
136  linelist = line.split()
137 
138  if lno >= 6:
139  # disp line
140  node_no = int(linelist[0])
141  mode_disp_x = float(linelist[1])
142  mode_disp_y = float(linelist[2])
143  if len(linelist) > 3:
144  mode_disp_z = float(linelist[3])
145  else:
146  mode_disp_z = 0.0
147  mode_disp[node_no] = FreeCAD.Vector(mode_disp_x, mode_disp_y, mode_disp_z)
148  nodes[node_no] = node_no
149 
150  mode_results["disp"] = mode_disp
151  results.append(mode_results)
152 
153  for r in results[0]["disp"]:
154  Console.PrintLog("{} --> {}\n".format(r, results[0]["disp"][r]))
155 
156  z88_disp_file.close()
157  return {"Nodes": nodes, "Results": results}
