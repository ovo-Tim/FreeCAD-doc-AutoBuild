52 void GraphicsViewZoom::gentle_zoom(double factor) {
53  _view->scale(factor, factor);
54  _view->centerOn(target_scene_pos);
55  QPointF delta_viewport_pos = target_viewport_pos - QPointF(_view->viewport()->width() / 2.0,
56  _view->viewport()->height() / 2.0);
57  QPointF viewport_center = _view->mapFromScene(target_scene_pos) - delta_viewport_pos;
58  _view->centerOn(_view->mapToScene(viewport_center.toPoint()));
59 }

70 bool GraphicsViewZoom::eventFilter(QObject *object, QEvent *event) {
71  if (event->type() == QEvent::MouseMove) {
72  auto mouse_event = static_cast<QMouseEvent*>(event);
73  QPointF delta = target_viewport_pos - mouse_event->pos();
74  if (qAbs(delta.x()) > 5 || qAbs(delta.y()) > 5) {
75  target_viewport_pos = mouse_event->pos();
76  target_scene_pos = _view->mapToScene(mouse_event->pos());
77  }
78  } else if (event->type() == QEvent::Wheel) {
79  auto wheel_event = static_cast<QWheelEvent*>(event);
80  if (QApplication::keyboardModifiers() == _modifiers) {
81  QPoint delta = wheel_event->angleDelta();
82  if (qAbs(delta.y()) > qAbs(delta.x())) { // vertical
83  double angle = -delta.y();
84  if (m_invert_zoom)
85  angle = -angle;
86  double factor = qPow(_zoom_factor_base, angle);
87  gentle_zoom(factor);
88  return true;
89  }
90  }
91  }
92  Q_UNUSED(object);
93  return false;
94 }
