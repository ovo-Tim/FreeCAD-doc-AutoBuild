387 App::DocumentObjectExecReturn* DrawViewSection::execute()
388 {
389  // Base::Console().Message("DVS::execute() - %s\n", Label.getValue());
390  if (!keepUpdated()) {
391  return App::DocumentObject::StdReturn;
392  }
393 
394  if (!isBaseValid()) {
395  return new App::DocumentObjectExecReturn("BaseView object not found");
396  }
397 
398  if (waitingForCut() || waitingForHlr()) {
399  return DrawView::execute();
400  }
401 
402  TopoDS_Shape baseShape = getShapeToCut();
403 
404  if (baseShape.IsNull()) {
405  return DrawView::execute();
406  }
407 
408  // is SectionOrigin valid?
409  Bnd_Box centerBox;
410  BRepBndLib::AddOptimal(baseShape, centerBox);
411  centerBox.SetGap(0.0);
412  Base::Vector3d orgPnt = SectionOrigin.getValue();
413 
414  if (!isReallyInBox(gp_Pnt(orgPnt.x, orgPnt.y, orgPnt.z), centerBox)) {
415  Base::Console().Warning("DVS: SectionOrigin doesn't intersect part in %s\n",
416  getNameInDocument());
417  }
418 
419  // save important info for later use
420  m_shapeSize = sqrt(centerBox.SquareExtent());
421  m_saveShape = baseShape;
422 
423  bool haveX = checkXDirection();
424  if (!haveX) {
425  // block touch/onChanged stuff
426  Base::Vector3d newX = getXDirection();
427  XDirection.setValue(newX);
428  XDirection.purgeTouched();// don't trigger updates!
429  // unblock
430  }
431 
432  sectionExec(baseShape);
433  addPoints();
434 
435  return DrawView::execute();
436 }

447 void DrawViewSection::sectionExec(TopoDS_Shape& baseShape)
448 {
449  // Base::Console().Message("DVS::sectionExec() - %s baseShape.IsNull: %d\n", Label.getValue(), baseShape.IsNull());
450 
451  if (waitingForHlr() || waitingForCut()) {
452  return;
453  }
454 
455  if (baseShape.IsNull()) {
456  // should be caught before this
457  return;
458  }
459 
460  m_cuttingTool = makeCuttingTool(m_shapeSize);
461 
462  try {
463  // note that &m_cutWatcher in the third parameter is not strictly required,
464  // but using the 4 parameter signature instead of the 3 parameter signature
465  // prevents clazy warning:
466  // https://github.com/KDE/clazy/blob/1.11/docs/checks/README-connect-3arg-lambda.md
467  connectCutWatcher =
468  QObject::connect(&m_cutWatcher, &QFutureWatcherBase::finished, &m_cutWatcher, [this] {
469  this->onSectionCutFinished();
470  });
471 
472  // We create a lambda closure to hold a copy of baseShape.
473  // This is important because this variable might be local to the calling
474  // function and might get destructed before the parallel processing finishes.
475  auto lambda = [this, baseShape]{this->makeSectionCut(baseShape);};
476  m_cutFuture = QtConcurrent::run(std::move(lambda));
477  m_cutWatcher.setFuture(m_cutFuture);
478  waitingForCut(true);
479  }
480  catch (...) {
481  Base::Console().Message("DVS::sectionExec - failed to make section cut");
482  return;
483  }
484 }

486 void DrawViewSection::makeSectionCut(const TopoDS_Shape& baseShape)
487 {
488  // Base::Console().Message("DVS::makeSectionCut() - %s - baseShape.IsNull:%d\n", Label.getValue(), baseShape.IsNull());
489 
490  showProgressMessage(getNameInDocument(), "is making section cut");
491 
492  // We need to copy the shape to not modify the BRepstructure
493  BRepBuilderAPI_Copy BuilderCopy(baseShape);
494  TopoDS_Shape myShape = BuilderCopy.Shape();
495  m_saveShape = myShape;// save shape for 2nd pass
496 
497  if (debugSection()) {
498  BRepTools::Write(myShape, "DVSCopy.brep");// debug
499  }
500 
501  if (debugSection()) {
502  BRepTools::Write(m_cuttingTool, "DVSTool.brep");// debug
503  }
504 
505  // perform the cut. We cut each solid in myShape individually to avoid issues
506  // where a compound BaseShape does not cut correctly.
507  BRep_Builder builder;
508  TopoDS_Compound cutPieces;
509  builder.MakeCompound(cutPieces);
510  TopExp_Explorer expl(myShape, TopAbs_SOLID);
511  for (; expl.More(); expl.Next()) {
512  const TopoDS_Solid& s = TopoDS::Solid(expl.Current());
513  FCBRepAlgoAPI_Cut mkCut(s, m_cuttingTool);
514  if (!mkCut.IsDone()) {
515  Base::Console().Warning("DVS: Section cut has failed in %s\n", getNameInDocument());
516  continue;
517  }
518  builder.Add(cutPieces, mkCut.Shape());
519  }
520 
521  // cutPieces contains result of cutting each subshape in baseShape with tool
522  m_cutPieces = cutPieces;
523  if (debugSection()) {
524  BRepTools::Write(cutPieces, "DVSCutPieces1.brep");// debug
525  }
526 
527  // second cut if requested. Sometimes the first cut includes extra uncut
528  // pieces.
529  if (trimAfterCut()) {
530  FCBRepAlgoAPI_Cut mkCut2(cutPieces, m_cuttingTool);
531  if (mkCut2.IsDone()) {
532  m_cutPieces = mkCut2.Shape();
533  if (debugSection()) {
534  BRepTools::Write(m_cutPieces, "DVSCutPieces2.brep");// debug
535  }
536  }
537  }
538 
539  // check for error in cut
540  Bnd_Box testBox;
541  BRepBndLib::AddOptimal(m_cutPieces, testBox);
542  testBox.SetGap(0.0);
543  if (testBox.IsVoid()) {// prism & input don't intersect. rawShape is
544  // garbage, don't bother.
545  Base::Console().Warning("DVS::makeSectionCut - prism & input don't intersect - %s\n",
546  Label.getValue());
547  return;
548  }
549 
550  waitingForCut(false);
551 }

557  // Base::Console().Message("DVS::prepareShape - %s - rawShape.IsNull: %d

560  // shapeSize);

635 // activities that depend on updated geometry object
636 void DrawViewSection::postHlrTasks(void)
637 {
638  // Base::Console().Message("DVS::postHlrTasks() - %s\n", Label.getValue());
639 
640  DrawViewPart::postHlrTasks();
641 
642  // second pass if required
643  if (ScaleType.isValue("Automatic") && !checkFit()) {
644  double newScale = autoScale();
645  Scale.setValue(newScale);
646  Scale.purgeTouched();
647  sectionExec(m_saveShape);
648  }
649  overrideKeepUpdated(false);
650 
651  // build section face geometry
652  TopoDS_Compound faceIntersections = findSectionPlaneIntersections(getShapeToIntersect());
653  if (faceIntersections.IsNull()) {
654  requestPaint();
655  return;
656  }
657  if (debugSection()) {
658  BRepTools::Write(faceIntersections, "DVSFaceIntersections.brep");// debug
659  }
660

779 TopoDS_Compound DrawViewSection::mapToPage(TopoDS_Shape& shapeToAlign)
780 {
781  // shapeToAlign is compound of TopoDS_Face intersections, but aligned to
782  // pln(origin, sectionNormal) needs to be aligned to paper plane (origin,
783  // stdZ);
784  // project the faces in the shapeToAlign, build new faces from the resulting
785  // wires and combine everything into a compound of faces
786  // Base::Console().Message("DVS::mapToPage() - shapeToAlign.null: %d\n",
787  // shapeToAlign.IsNull());
788  if (debugSection()) {
789  BRepTools::Write(shapeToAlign, "DVSShapeToAlign.brep");// debug
790  }
791 
792  BRep_Builder builder;
793  TopoDS_Compound result;
794  builder.MakeCompound(result);
795 
796  TopExp_Explorer expFace(shapeToAlign, TopAbs_FACE);
797  for (int iFace = 1; expFace.More(); expFace.Next(), iFace++) {
798  const TopoDS_Face& face = TopoDS::Face(expFace.Current());
799  std::vector<TopoDS_Wire> faceWires;
800  TopExp_Explorer expWires(face, TopAbs_WIRE);
801  for (; expWires.More(); expWires.Next()) {
802  const TopoDS_Wire& wire = TopoDS::Wire(expWires.Current());
803  TopoDS_Shape projectedShape =
804  GeometryObject::projectSimpleShape(wire, getProjectionCS());
805  std::vector<TopoDS_Edge> wireEdges;
806  // projectedShape is just a bunch of edges. we have to rebuild the wire.
807  TopExp_Explorer expEdges(projectedShape, TopAbs_EDGE);
808  for (; expEdges.More(); expEdges.Next()) {
809  const TopoDS_Edge& edge = TopoDS::Edge(expEdges.Current());
810  wireEdges.push_back(edge);
811  }
812  TopoDS_Wire cleanWire = EdgeWalker::makeCleanWire(wireEdges, 2.0 * EWTOLERANCE);
813  faceWires.push_back(cleanWire);
814  }
815 
816  // validate section face wires
817  std::vector<TopoDS_Wire> goodWires;
818  constexpr double minWireArea = 0.000001;// arbitrary very small face size
819  for (auto& wire : faceWires) {
820  if (wire.IsNull()) {
821  continue;
822  }
823  if (!BRep_Tool::IsClosed(wire)) {
824  continue;// can not make a face from open wire
825  }
826  double area = ShapeAnalysis::ContourArea(wire);
827  if (area <= minWireArea) {
828  continue;// can not make a face from wire with no area
829  }
830  goodWires.push_back(wire);
831  }
832 
833  if (goodWires.empty()) {
834  // this may or may not be significant. In the offset or noparallel
835  // strategies, a profile segment that is parallel to the SectionNormal
836  // will not generate a face.
837  Base::Console().Log("DVS::mapToPage - %s - section face has no valid wires.\n",
838  getNameInDocument());
839  continue;
840  }
841 
842  TopoDS_Shape holeyShape = makeFaceFromWires(goodWires);
843  if (holeyShape.IsNull()) {
844  continue;
845  }
846 
847  builder.Add(result, TopoDS::Face(holeyShape));
848  if (debugSection()) {
849  std::stringstream ss;
850  ss << "DVSFaceFromWires" << iFace << ".brep";
851  BRepTools::Write(holeyShape, ss.str().c_str());// debug
852  }
853  }
854 
855  return result;
856 }

858 // makes a [perforated] face from an outer wire and wires describing the holes.
859 // Open wires and wires with zero area are assumed to already have been removed.
860 TopoDS_Shape DrawViewSection::makeFaceFromWires(std::vector<TopoDS_Wire>& inWires)
861 {
862  // make sure the largest wire is the first
863  EdgeWalker eWalker;
864  std::vector<TopoDS_Wire> goodWires = eWalker.sortWiresBySize(inWires);
865 
866  // make a face from the good wires
867  // first good wire should be the outer boundary of the face
868  TopoDS_Face faceToFix;
869  TopoDS_Shape orientedShape = goodWires.at(0).Oriented(TopAbs_FORWARD);
870  TopoDS_Wire orientedWire = TopoDS::Wire(orientedShape);
871  orientedWire.Orientation(TopAbs_FORWARD);
872  TopoDS_Face blankFace = BRepBuilderAPI_MakeFace(orientedWire);
873  int wireCount = goodWires.size();
874  if (wireCount < 2) {
875  faceToFix = blankFace;
876  }
877  else {
878  // add the holes
879  BRepBuilderAPI_MakeFace mkFace(blankFace);
880  for (int iWire = 1; iWire < wireCount; iWire++) {
881  // make holes in the face with the rest of the wires
882  orientedShape = goodWires.at(iWire).Oriented(TopAbs_REVERSED);
883  orientedWire = TopoDS::Wire(orientedShape);
884  mkFace.Add(orientedWire);
885  }
886 
887  if (!mkFace.IsDone()) {
888  Base::Console().Warning("DVS::makeFaceFromWires - %s - failed to make section face.\n",
889  getNameInDocument());
890  return TopoDS_Shape();
891  }
892  faceToFix = mkFace.Face();
893  }
894 
895  // setting the wire orientation above should generate a valid face, but
896  // sometimes does not, so we fix the shape to resolve any issues
897  Handle(ShapeFix_Shape) sfs = new ShapeFix_Shape;
898  sfs->Init(faceToFix);
899  sfs->Perform();
900  return sfs->Shape();
901 }
