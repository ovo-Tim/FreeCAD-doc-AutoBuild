165 double QGIViewSymbol::legacyScaler(TechDraw::DrawViewSymbol* feature) const
166 {
167  double rezfactor = Rez::getRezFactor();
168  double scaling = feature->getScale();
169  double pxMm = 3.78;//96px/25.4mm ( CSS/SVG defined value of 96 pixels per inch)
170  // double pxMm = 3.54; //90px/25.4mm ( inkscape value version <= 0.91)
171  //some software uses different px/in, so symbol will need Scale adjusted.
172  //Arch/Draft views are in px and need to be scaled @ rezfactor px/mm to ensure proper representation
173  if (feature->isDerivedFrom(TechDraw::DrawViewArch::getClassTypeId())
174  || feature->isDerivedFrom(TechDraw::DrawViewDraft::getClassTypeId())) {
175  scaling = scaling * rezfactor;
176  }
177  else {
178  scaling = scaling * rezfactor / pxMm;
179  }
180 
181  return scaling;
182 }

187 double QGIViewSymbol::symbolScaler(TechDraw::DrawViewSymbol* feature) const
188 {
189  double scaling = feature->getScale();
190  double rezfactor = Rez::getRezFactor();
191 
192  QByteArray qba(feature->Symbol.getValue(), strlen(feature->Symbol.getValue()));
193  QString qSymbolString = QString::fromUtf8(qba);
194 
195  const QString pxToken{QString::fromUtf8("px")};
196  const QString mmToken{QString::fromUtf8("mm")};
197 
198  // heightRegex finds (height="51.8309mm") in the svg text and returns the mm if present
199  QString heightRegex = QString::fromUtf8(R"(height=\"\d*\.?\d+([a-zA-Z]+)\")");
200  QRegularExpression reHeight(heightRegex);
201  QRegularExpressionMatch matchHeight = reHeight.match(qSymbolString);
202 
203  QString matchUnits;
204  if (matchHeight.hasMatch()) {
205  auto capture0 = matchHeight.captured(0);
206  matchUnits = matchHeight.captured(1);
207  }
208 
209  if (matchUnits.isEmpty() ||
210  matchUnits == pxToken) {
211  scaling *= rezfactor;
212  }
213 
214  if (matchUnits == mmToken) {
215  auto svgSize = m_svgItem->renderer()->defaultSize();
216  auto vpSize = m_svgItem->renderer()->viewBox();
217  // wf: this calculation works, but I don't know why. :(
218  // hints here: https://stackoverflow.com/questions/49866474/get-svg-size-from-qsvgrenderer
219  // and here: https://stackoverflow.com/questions/7544921/qt-qgraphicssvgitem-renders-too-big-0-5-unit-on-each-side
220  scaling *= rezfactor * vpSize.width() / svgSize.width();
221  }
222 
223  return scaling;
224 }
