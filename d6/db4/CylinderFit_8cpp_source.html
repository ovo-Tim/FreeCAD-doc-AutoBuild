23 // Definitions:
24 // Cylinder axis goes through a point (Xc,Yc,Zc) and has direction (L,M,N)
25 // Cylinder radius is R
26 // A point on the axis (X0i,Y0i,Z0i) can be described by:
27 // (X0i,Y0i,Z0i) = (Xc,Yc,Zc) + s(L,M,N)
28 // where s is the distance from (Xc,Yc,Zc) to (X0i,Y0i,Z0i) when (L,M,N) is
29 // of unit length (normalized).
30 // The distance between a cylinder surface point (Xi,Yi,Zi) and its
31 // projection onto the axis (X0i,Y0i,Z0i) is the radius:
32 // (Xi - X0i)^2 + (Yi - Y0i)^2 + (Zi - Z0i)^2 = R^2
33 // Also the vector to a cylinder surface point (Xi,Yi,Zi) from its
34 // projection onto the axis (X0i,Y0i,Z0i) is orthogonal to the axis so we can
35 // write:
36 // (Xi - X0i, Yi - Y0i, Zi - Z0i).(L,M,N) = 0 or
37 // L(Xi - X0i) + M(Yi - Y0i) + N(Zi - Z0i) = 0
38 // If we substitute these various equations into each other and further add
39 // the constraint that L^2 + M^2 + N^2 = 1 then we can arrive at a single
40 // equation for the cylinder surface points:
41 // (Xi - Xc + L*L*(Xc - Xi) + L*M*(Yc - Yi) + L*N*(Zc - Zi))^2 +
42 // (Yi - Yc + M*L*(Xc - Xi) + M*M*(Yc - Yi) + M*N*(Zc - Zi))^2 +
43 // (Zi - Zc + N*L*(Xc - Xi) + N*M*(Yc - Yi) + N*N*(Zc - Zi))^2 - R^2 = 0
44 // This equation is what is used in the least squares solution below. Because
45 // we are constraining the direction vector to a unit length and also because
46 // we need to stop the axis point from moving along the axis we need to fix one
47 // of the ordinates in the solution. So from our initial approximations for the
48 // axis direction (L0,M0,N0):
49 // if (L0 > M0) && (L0 > N0) then fix Xc = Mx and use L = sqrt(1 - M^2 - N^2)
50 // else if (M0 > L0) && (M0 > N0) then fix Yc = My and use M = sqrt(1 - L^2 - N^2)
51 // else if (N0 > L0) && (N0 > M0) then fix Zc = Mz and use N = sqrt(1 - L^2 - M^2)
52 // where (Mx,My,Mz) is the mean of the input points (centre of gravity)
53 // We thus solve for 5 unknown parameters.
54 // Thus for the solution to succeed the initial axis direction should be reasonable.

179 float CylinderFit::GetStdDeviation() const
180 {
181  // Mean: M=(1/N)*SUM Xi
182  // Variance: VAR=(N/N-1)*[(1/N)*SUM(Xi^2)-M^2]
183  // Standard deviation: SD=SQRT(VAR)
184  if (!_bIsFitted) {
185  return FLOAT_MAX;
186  }
187 
188  double sumXi = 0.0;
189  double sumXi2 = 0.0;
190  double dist = 0.0;
191  for (auto it : _vPoints) {
192  dist = GetDistanceToCylinder(it);
193  sumXi += dist;
194  sumXi2 += (dist * dist);
195  }
196 
197  double N = static_cast<double>(CountPoints());
198  double mean = sumXi / N;
199  return static_cast<float>(sqrt((N / (N - 1.0)) * (sumXi2 / N - mean * mean)));
200 }

202 void CylinderFit::ProjectToCylinder()
203 {
204  auto cBase = Base::convertTo<Base::Vector3f>(_vBase);
205  auto cAxis = Base::convertTo<Base::Vector3f>(_vAxis);
206 
207  for (auto& cPnt : _vPoints) {
208  if (cPnt.DistanceToLine(cBase, cAxis) > 0) {
209  Base::Vector3f proj;
210  cBase.ProjectToPlane(cPnt, cAxis, proj);
211  Base::Vector3f diff = cPnt - proj;
212  diff.Normalize();
213  cPnt = proj + diff * _dRadius;
214  }
215  else {
216  // Point is on the cylinder axis, so it can be moved in
217  // any direction perpendicular to the cylinder axis
218  Base::Vector3f cMov(cPnt);
219  do {
220  float x = (float(rand()) / float(RAND_MAX));
221  float y = (float(rand()) / float(RAND_MAX));
222  float z = (float(rand()) / float(RAND_MAX));
223  cMov.Move(x, y, z);
224  } while (cMov.DistanceToLine(cBase, cAxis) == 0);
225 
226  Base::Vector3f proj;
227  cMov.ProjectToPlane(cPnt, cAxis, proj);
228  Base::Vector3f diff = cPnt - proj;
229  diff.Normalize();
230  cPnt = proj + diff * _dRadius;
231  }
232  }
233 }

236 // line through the points. This doesn't work well if the points are only from

239 void CylinderFit::ComputeApproximationsLine()
240 {
241  _bIsFitted = false;
242  _fLastResult = FLOAT_MAX;
243  _numIter = 0;
244  _vBase.Set(0.0, 0.0, 0.0);
245  _vAxis.Set(0.0, 0.0, 0.0);
246  _dRadius = 0.0;
247  if (!_vPoints.empty()) {
248  std::vector<Wm4::Vector3d> input;
249  std::transform(_vPoints.begin(),
250  _vPoints.end(),
251  std::back_inserter(input),
252  [](const Base::Vector3f& v) {
253  return Wm4::Vector3d(v.x, v.y, v.z);
254  });
255  Wm4::Line3<double> kLine = Wm4::OrthogonalLineFit3(input.size(), input.data());
256  _vBase.Set(kLine.Origin.X(), kLine.Origin.Y(), kLine.Origin.Z());
257  _vAxis.Set(kLine.Direction.X(), kLine.Direction.Y(), kLine.Direction.Z());
258 
259  for (const auto& it : _vPoints) {
260  _dRadius += Base::Vector3d(it.x, it.y, it.z).DistanceToLine(_vBase, _vAxis);
261  }
262  _dRadius /= (double)_vPoints.size();
263  }

266 float CylinderFit::Fit()
267 {
268  _bIsFitted = false;
269  _fLastResult = FLOAT_MAX;
270  _numIter = 0;
271 
272  // A minimum of 5 surface points is needed to define a cylinder
273  const int minPts = 5;
274  if (CountPoints() < minPts) {
275  return FLOAT_MAX;
276  }
277 
278  // If approximations have not been set/computed then compute some now using the line fit method
279  if (_dRadius == 0.0) {
280  ComputeApproximationsLine();
281  }
282 
283  // Check parameters to define the best solution direction
284  // There are 7 parameters but 2 are actually dependent on the others
285  // so we are actually solving for 5 parameters.
286  // order of parameters depending on the solution direction:
287  // solL: Yc, Zc, M, N, R
288  // solM: Xc, Zc, L, N, R
289  // solN: Xc, Yc, L, M, R
290  SolutionD solDir {};
291  findBestSolDirection(solDir);
292 
293  // Initialise some matrices and vectors
294  const int dim = 5;
295  std::vector<Base::Vector3d> residuals(CountPoints(), Base::Vector3d(0.0, 0.0, 0.0));
296  Matrix5x5 atpa;
297  Eigen::VectorXd atpl(dim);
298 
299  // Iteration loop...
300  double sigma0 {};
301  bool cont = true;
302  while (cont && (_numIter < _maxIter)) {
303  ++_numIter;
304 
305  // Set up the quasi parametric normal equations
306  setupNormalEquationMatrices(solDir, residuals, atpa, atpl);
307 
308  // Solve the equations for the unknown corrections
309  Eigen::LLT<Matrix5x5> llt(atpa);
310  if (llt.info() != Eigen::Success) {
311  return FLOAT_MAX;
312  }
313  Eigen::VectorXd x = llt.solve(atpl);
314 
315  // Check parameter convergence
316  cont = false;
317  // x(0), x(1): the two position parameter corrections
318  // x(2),x(3): the two direction parameter corrections
319  // x(4): the radius correction
320  if ((fabs(x(0)) > _posConvLimit) || (fabs(x(1)) > _posConvLimit)
321  || (fabs(x(2)) > _dirConvLimit) || (fabs(x(3)) > _dirConvLimit)
322  || (fabs(x(4)) > _posConvLimit)) {
323  cont = true;
324  }
325 
326  // Before updating the unknowns, compute the residuals and sigma0 and check the residual
327  // convergence
328  bool vConverged {};
329  if (!computeResiduals(solDir, x, residuals, sigma0, _vConvLimit, vConverged)) {
330  return FLOAT_MAX;
331  }
332  if (!vConverged) {
333  cont = true;
334  }
335 
336  // Update the parameters
337  if (!updateParameters(solDir, x)) {
338  return FLOAT_MAX;
339  }
340  }
341 
342  // Check for convergence
343  if (cont) {
344  return FLOAT_MAX;
345  }
346 
347  _bIsFitted = true;
348  _fLastResult = sigma0;
349 
350  return _fLastResult;
351 }

353 // Checks initial parameter values and defines the best solution direction to use
354 // Axis direction = (L,M,N)
355 // solution L: L is biggest axis component and L = f(M,N) and X = Mx (we move the base point along
356 // axis to this x-value) solution M: M is biggest axis component and M = f(L,N) and Y = My (we move
357 // the base point along axis to this y-value) solution N: N is biggest axis component and N = f(L,M)
358 // and Z = Mz (we move the base point along axis to this z-value) where (Mx,My,Mz) is the mean of
359 // the input points (centre of gravity)
360 void CylinderFit::findBestSolDirection(SolutionD& solDir)
361 {
362  // Choose the best of the three solution 'directions' to use
363  // This is to avoid a square root of a negative number when computing the dependent parameters
364  Base::Vector3d dir = _vAxis;
365  Base::Vector3d pos = _vBase;
366  dir.Normalize();
367  double biggest = dir.x;
368  solDir = solL;
369  if (fabs(dir.y) > fabs(biggest)) {
370  biggest = dir.y;
371  solDir = solM;
372  }
373  if (fabs(dir.z) > fabs(biggest)) {
374  biggest = dir.z;
375  solDir = solN;
376  }
377  if (biggest < 0.0) {
378  dir.Set(-dir.x, -dir.y, -dir.z); // multiplies by -1
379  }
380 
381  double fixedVal = 0.0;
382  double lambda {};
383  switch (solDir) {
384  case solL:
385  fixedVal = meanXObs();
386  lambda = (fixedVal - pos.x) / dir.x;
387  pos.x = fixedVal;
388  pos.y = pos.y + lambda * dir.y;
389  pos.z = pos.z + lambda * dir.z;
390  break;
391  case solM:
392  fixedVal = meanYObs();
393  lambda = (fixedVal - pos.y) / dir.y;
394  pos.x = pos.x + lambda * dir.x;
395  pos.y = fixedVal;
396  pos.z = pos.z + lambda * dir.z;
397  break;
398  case solN:
399  fixedVal = meanZObs();
400  lambda = (fixedVal - pos.z) / dir.z;
401  pos.x = pos.x + lambda * dir.x;
402  pos.y = pos.y + lambda * dir.y;
403  pos.z = fixedVal;
404  break;
405  }
406  _vAxis = dir;
407  _vBase = pos;
408 }

447 // atpa ... 5x5 normal matrix

449 void CylinderFit::setupNormalEquationMatrices(SolutionD solDir,
450  const std::vector<Base::Vector3d>& residuals,
451  Matrix5x5& atpa,
452  Eigen::VectorXd& atpl) const

455  atpa.setZero();
456  atpl.setZero();
457 
458  // For each point, setup the observation equation coefficients and add their
459  // contribution into the normal equation matrices
460  DoubleArray5 a {};
461  DoubleArray3 b {};
462  double f0 {};
463  double qw {};
464  auto vIt = residuals.begin();
465  for (auto cIt = _vPoints.begin(); cIt != _vPoints.end(); ++cIt, ++vIt) {
466  // if (using this point) { // currently all given points are used (could modify this if
467  // eliminating outliers, etc....

469  addObservationU(a, f0, qw, atpa, atpl);

475 // clang-format off
476 // NOLINTBEGIN
477 // Sets up contributions of given observation to the quasi parametric
478 // normal equation matrices. Assumes uncorrelated coordinates.
479 // point ... point
480 // residual ... residual for this point computed from previous iteration (zero for first iteration)
481 // a[5] ... parameter partials
482 // f0 ... reference to f0 term
483 // qw ... reference to quasi weight (here we are assuming equal unit weights for each observed
484 // point coordinate) b[3] ... observation partials
485 void CylinderFit::setupObservation(SolutionD solDir,
486  const Base::Vector3f& point,
487  const Base::Vector3d& residual,
488  DoubleArray5& a,
489  double& f0,
490  double& qw,
491  DoubleArray3& b) const
492 {
493  // This adjustment requires an update of the observation approximations
494  // because the residuals do not have a linear relationship.
495  // New estimates for the observations:
496  double xEstimate = (double)point.x + residual.x;
497  double yEstimate = (double)point.y + residual.y;
498  double zEstimate = (double)point.z + residual.z;
499 
500  // intermediate parameters
501  double lambda = _vAxis.x * (xEstimate - _vBase.x) + _vAxis.y * (yEstimate - _vBase.y) + _vAxis.z * (zEstimate - _vBase.z);
502  double x0 = _vBase.x + lambda * _vAxis.x;
503  double y0 = _vBase.y + lambda * _vAxis.y;
504  double z0 = _vBase.z + lambda * _vAxis.z;
505  double dx = xEstimate - x0;
506  double dy = yEstimate - y0;
507  double dz = zEstimate - z0;
508  double dx00 = _vBase.x - xEstimate;
509  double dy00 = _vBase.y - yEstimate;
510  double dz00 = _vBase.z - zEstimate;
511 
512  // partials of the observations
513  b[0] = 2.0 * (dx - _vAxis.x * _vAxis.x * dx - _vAxis.x * _vAxis.y * dy - _vAxis.x * _vAxis.z * dz);
514  b[1] = 2.0 * (dy - _vAxis.x * _vAxis.y * dx - _vAxis.y * _vAxis.y * dy - _vAxis.y * _vAxis.z * dz);
515  b[2] = 2.0 * (dz - _vAxis.x * _vAxis.z * dx - _vAxis.y * _vAxis.z * dy - _vAxis.z * _vAxis.z * dz);
516 
517  double ddxdl {}, ddydl {}, ddzdl {};
518  double ddxdm {}, ddydm {}, ddzdm {};
519  double ddxdn {}, ddydn {}, ddzdn {};
520 
521  // partials of the parameters
522  switch (solDir) {
523  case solL:
524  // order of parameters: Yc, Zc, M, N, R
525  ddxdm = -2.0 * _vAxis.y * dx00 + (_vAxis.x - _vAxis.y * _vAxis.y / _vAxis.x) * dy00 - (_vAxis.y * _vAxis.z / _vAxis.x) * dz00;
526  ddydm = (_vAxis.x - _vAxis.y * _vAxis.y / _vAxis.x) * dx00 + 2.0 * _vAxis.y * dy00 + _vAxis.z * dz00;
527  ddzdm = -(_vAxis.y * _vAxis.z / _vAxis.x) * dx00 + _vAxis.z * dy00;
528  ddxdn = -2.0 * _vAxis.z * dx00 - (_vAxis.y * _vAxis.z / _vAxis.x) * dy00 + (_vAxis.x - _vAxis.z * _vAxis.z / _vAxis.x) * dz00;
529  ddydn = -(_vAxis.y * _vAxis.z / _vAxis.x) * dx00 + _vAxis.y * dz00;
530  ddzdn = (_vAxis.x - _vAxis.z * _vAxis.z / _vAxis.x) * dx00 + _vAxis.y * dy00 + 2.0 * _vAxis.z * dz00;
531  a[0] = -b[1];
532  a[1] = -b[2];
533  a[2] = 2.0 * (dx * ddxdm + dy * ddydm + dz * ddzdm);
534  a[3] = 2.0 * (dx * ddxdn + dy * ddydn + dz * ddzdn);
535  a[4] = -2.0 * _dRadius;
536  break;
537  case solM:
538  // order of parameters: Xc, Zc, L, N, R
539  ddxdl = 2.0 * _vAxis.x * dx00 + (_vAxis.y - _vAxis.x * _vAxis.x / _vAxis.y) * dy00 + _vAxis.z * dz00;
540  ddydl = (_vAxis.y - _vAxis.x * _vAxis.x / _vAxis.y) * dx00 - 2.0 * _vAxis.x * dy00 - (_vAxis.x * _vAxis.z / _vAxis.y) * dz00;
541  ddzdl = _vAxis.z * dx00 - (_vAxis.x * _vAxis.z / _vAxis.y) * dy00;
542  ddxdn = -(_vAxis.x * _vAxis.z / _vAxis.y) * dy00 + _vAxis.x * dz00;
543  ddydn = -(_vAxis.x * _vAxis.z / _vAxis.y) * dx00 - 2.0 * _vAxis.z * dy00 + (_vAxis.y - _vAxis.z * _vAxis.z / _vAxis.y) * dz00;
544  ddzdn = _vAxis.x * dx00 + (_vAxis.y - _vAxis.z * _vAxis.z / _vAxis.y) * dy00 + 2.0 * _vAxis.z * dz00;
545  a[0] = -b[0];
546  a[1] = -b[2];
547  a[2] = 2.0 * (dx * ddxdl + dy * ddydl + dz * ddzdl);
548  a[3] = 2.0 * (dx * ddxdn + dy * ddydn + dz * ddzdn);
549  a[4] = -2.0 * _dRadius;
550  break;
551  case solN:
552  // order of parameters: Xc, Yc, L, M, R
553  ddxdl = 2.0 * _vAxis.x * dx00 + _vAxis.y * dy00 + (_vAxis.z - _vAxis.x * _vAxis.x / _vAxis.z) * dz00;
554  ddydl = _vAxis.y * dx00 - (_vAxis.x * _vAxis.y / _vAxis.z) * dz00;
555  ddzdl = (_vAxis.z - _vAxis.x * _vAxis.x / _vAxis.z) * dx00 - (_vAxis.x * _vAxis.y / _vAxis.z) * dy00 - 2.0 * _vAxis.x * dz00;
556  ddxdm = _vAxis.x * dy00 - (_vAxis.x * _vAxis.y / _vAxis.z) * dz00;
557  ddydm = _vAxis.x * dx00 + 2.0 * _vAxis.y * dy00 + (_vAxis.z - _vAxis.y * _vAxis.y / _vAxis.z) * dz00;
558  ddzdm = -(_vAxis.x * _vAxis.y / _vAxis.z) * dx00 + (_vAxis.z - _vAxis.y * _vAxis.y / _vAxis.z) * dy00 - 2.0 * _vAxis.y * dz00;
559  a[0] = -b[0];
560  a[1] = -b[1];
561  a[2] = 2.0 * (dx * ddxdl + dy * ddydl + dz * ddzdl);
562  a[3] = 2.0 * (dx * ddxdm + dy * ddydm + dz * ddzdm);
563  a[4] = -2.0 * _dRadius;
564  break;
565  }
566 
567  // free term
568  f0 = _dRadius * _dRadius - dx * dx - dy * dy - dz * dz + b[0] * residual.x + b[1] * residual.y + b[2] * residual.z;
569 
570  // quasi weight (using equal weights for cylinder point coordinate observations)
571  // w[0] = 1.0;
572  // w[1] = 1.0;
573  // w[2] = 1.0;
574  // qw = 1.0 / (b[0] * b[0] / w[0] + b[1] * b[1] / w[1] + b[2] * b[2] / w[2]);
575  qw = 1.0 / (b[0] * b[0] + b[1] * b[1] + b[2] * b[2]);
576 }
577 // NOLINTEND
578 // clang-format on

580 // Computes contribution of the given observation equation on the normal equation matrices
581 // Call this for each observation (point)
582 // Here we only add the contribution to the upper part of the normal matrix
583 // and then after all observations have been added we need to set the lower part
584 // (which is symmetrical to the upper part)
585 // a[5] ... parameter partials
586 // li ... free term (f0)
587 // pi ... weight of observation (= quasi weight qw for this solution)
588 // atpa ... 5x5 normal equation matrix
589 // atpl ... 5x1 matrix/vector (right-hand side of equations)
590 void CylinderFit::addObservationU(DoubleArray5 a,
591  double li,
592  double pi,
593  Matrix5x5& atpa,
594  Eigen::VectorXd& atpl) const
595 {
596  const int dim = 5;
597  for (int i = 0; i < dim; ++i) {
598  double aipi = a[i] * pi;
599  for (int j = i; j < dim; ++j) {
600  atpa(i, j) += aipi * a[j];
601  // atpa(j, i) = atpa(i, j); // it's a symmetrical matrix, we'll set this later after all
602  // observations processed
603  }
604  atpl(i) += aipi * li;
605  }
606 }
