21 template <class Real>
22 ConvexHull2<Real>::ConvexHull2 (int iVertexQuantity, Vector2<Real>* akVertex,
23  Real fEpsilon, bool bOwner, Query::Type eQueryType)
24  :
25  ConvexHull<Real>(iVertexQuantity,fEpsilon,bOwner,eQueryType),
26  m_kLineOrigin(Vector2<Real>::ZERO),
27  m_kLineDirection(Vector2<Real>::ZERO)
28 {
29  assert(akVertex);
30  m_akVertex = akVertex;
31  m_akSVertex = nullptr;
32  m_pkQuery = nullptr;
33 
34  Mapper2<Real> kMapper(m_iVertexQuantity,m_akVertex,m_fEpsilon);
35  if (kMapper.GetDimension() == 0)
36  {
37  // The values of m_iDimension, m_aiIndex, and m_aiAdjacent were
38  // already initialized by the ConvexHull base class.
39  return;
40  }
41 
42  if (kMapper.GetDimension() == 1)
43  {
44  // The set is (nearly) collinear. The caller is responsible for
45  // creating a ConvexHull1 object.
46  m_iDimension = 1;
47  m_kLineOrigin = kMapper.GetOrigin();
48  m_kLineDirection = kMapper.GetDirection(0);
49  return;
50  }
51 
52  m_iDimension = 2;
53 
54  int i0 = kMapper.GetExtremeIndex(0);
55  int i1 = kMapper.GetExtremeIndex(1);
56  int i2 = kMapper.GetExtremeIndex(2);
57 
58  m_akSVertex = WM4_NEW Vector2<Real>[m_iVertexQuantity];
59  int i;
60 
61  if (eQueryType != Query::QT_RATIONAL && eQueryType != Query::QT_FILTERED)
62  {
63  // Transform the vertices to the square [0,1]^2.
64  Vector2<Real> kMin = kMapper.GetMin();
65  Real fScale = ((Real)1.0)/kMapper.GetMaxRange();
66  for (i = 0; i < m_iVertexQuantity; i++)
67  {
68  m_akSVertex[i] = (m_akVertex[i] - kMin)*fScale;
69  }
70 
71  Real fExpand;
72  if (eQueryType == Query::QT_INT64)
73  {
74  // Scale the vertices to the square [0,2^{20}]^2 to allow use of
75  // 64-bit integers.
76  fExpand = (Real)(1 << 20);
77  m_pkQuery = WM4_NEW Query2Int64<Real>(m_iVertexQuantity,
78  m_akSVertex);
79  }
80  else if (eQueryType == Query::QT_INTEGER)
81  {
82  // Scale the vertices to the square [0,2^{24}]^2 to allow use of
83  // TInteger.
84  fExpand = (Real)(1 << 24);
85  m_pkQuery = WM4_NEW Query2TInteger<Real>(m_iVertexQuantity,
86  m_akSVertex);
87  }
88  else // eQueryType == Query::QT_REAL
89  {
90  // No scaling for floating point.
91  fExpand = (Real)1.0;
92  m_pkQuery = WM4_NEW Query2<Real>(m_iVertexQuantity,m_akSVertex);
93  }
94 
95  for (i = 0; i < m_iVertexQuantity; i++)
96  {
97  m_akSVertex[i] *= fExpand;
98  }
99  }
100  else
101  {
102  // No transformation needed for exact rational arithmetic or filtered
103  // predicates.
104  size_t uiSize = m_iVertexQuantity*sizeof(Vector2<Real>);
105  System::Memcpy(m_akSVertex,uiSize,m_akVertex,uiSize);
106 
107  if (eQueryType == Query::QT_RATIONAL)
108  {
109  m_pkQuery = WM4_NEW Query2TRational<Real>(m_iVertexQuantity,
110  m_akSVertex);
111  }
112  else // eQueryType == Query::QT_FILTERED
113  {
114  m_pkQuery = WM4_NEW Query2Filtered<Real>(m_iVertexQuantity,
115  m_akSVertex,m_fEpsilon);
116  }
117  }
118 
119  Edge* pkE0;
120  Edge* pkE1;
121  Edge* pkE2;
122 
123  if (kMapper.GetExtremeCCW())
124  {
125  pkE0 = WM4_NEW Edge(i0,i1);
126  pkE1 = WM4_NEW Edge(i1,i2);
127  pkE2 = WM4_NEW Edge(i2,i0);
128  }
129  else
130  {
131  pkE0 = WM4_NEW Edge(i0,i2);
132  pkE1 = WM4_NEW Edge(i2,i1);
133  pkE2 = WM4_NEW Edge(i1,i0);
134  }
135 
136  pkE0->Insert(pkE2,pkE1);
137  pkE1->Insert(pkE0,pkE2);
138  pkE2->Insert(pkE1,pkE0);
139 
140  Edge* pkHull = pkE0;
141  for (i = 0; i < m_iVertexQuantity; i++)
142  {
143  if (!Update(pkHull,i))
144  {
145  pkHull->DeleteAll();
146  return;
147  }
148  }
149 
150  pkHull->GetIndices(m_iSimplexQuantity,m_aiIndex);
151  pkHull->DeleteAll();
152 }

182  {

197 template <class Real>
198 bool ConvexHull2<Real>::Update (Edge*& rpkHull, int i)
199 {
200  // Locate an edge visible to the input point (if possible).
201  Edge* pkVisible = nullptr;
202  Edge* pkCurrent = rpkHull;
203  do
204  {
205  if (pkCurrent->GetSign(i,m_pkQuery) > 0)
206  {
207  pkVisible = pkCurrent;
208  break;
209  }
210 
211  pkCurrent = pkCurrent->A[1];
212  }
213  while (pkCurrent != rpkHull);
214 
215  if (!pkVisible)
216  {
217  // The point is inside the current hull; nothing to do.
218  return true;
219  }
220 
221  // Remove the visible edges.
222  Edge* pkAdj0 = pkVisible->A[0];
223  assert(pkAdj0);
224  if (!pkAdj0)
225  {
226  return false;
227  }
228 
229  Edge* pkAdj1 = pkVisible->A[1];
230  assert(pkAdj1);
231  if (!pkAdj1)
232  {
233  return false;
234  }
235 
236  pkVisible->DeleteSelf();
237 
238  while (pkAdj0->GetSign(i,m_pkQuery) > 0)
239  {
240  rpkHull = pkAdj0;
241  pkAdj0 = pkAdj0->A[0];
242  assert(pkAdj0);
243  if (!pkAdj0)
244  {
245  return false;
246  }
247 
248  pkAdj0->A[1]->DeleteSelf();
249  }
250 
251  while (pkAdj1->GetSign(i,m_pkQuery) > 0)
252  {
253  rpkHull = pkAdj1;
254  pkAdj1 = pkAdj1->A[1];
255  assert(pkAdj1);
256  if (!pkAdj1)
257  {
258  return false;
259  }
260 
261  pkAdj1->A[0]->DeleteSelf();
262  }
263 
264  // Insert the new edges formed by the input point and the end points of
265  // the polyline of invisible edges.
266  Edge* pkEdge0 = WM4_NEW Edge(pkAdj0->V[1],i);
267  Edge* pkEdge1 = WM4_NEW Edge(i,pkAdj1->V[0]);
268  pkEdge0->Insert(pkAdj0,pkEdge1);
269  pkEdge1->Insert(pkEdge0,pkAdj1);
270  rpkHull = pkEdge0;
271 
272  return true;
273 }

288 template <class Real>
289 bool ConvexHull2<Real>::Load (const char* acFilename)
290 {
291  FILE* pkIFile = System::Fopen(acFilename,"rb");
292  if (!pkIFile)
293  {
294  return false;
295  }
296 
297  ConvexHull<Real>::Load(pkIFile);
298 
299  WM4_DELETE m_pkQuery;
300  WM4_DELETE[] m_akSVertex;
301  if (m_bOwner)
302  {
303  WM4_DELETE[] m_akVertex;
304  }
305 
306  m_bOwner = true;
307  m_akVertex = WM4_NEW Vector2<Real>[m_iVertexQuantity];
308  m_akSVertex = WM4_NEW Vector2<Real>[m_iVertexQuantity];
309 
310  size_t uiSize = sizeof(Real);
311  int iVQ = 2*m_iVertexQuantity;
312  if (uiSize == 4)
313  {
314  System::Read4le(pkIFile,iVQ,m_akVertex);
315  System::Read4le(pkIFile,iVQ,m_akSVertex);
316  System::Read4le(pkIFile,2,(Real*)m_kLineOrigin);
317  System::Read4le(pkIFile,2,(Real*)m_kLineDirection);
318  }
319  else // iSize == 8
320  {
321  System::Read8le(pkIFile,iVQ,m_akVertex);
322  System::Read8le(pkIFile,iVQ,m_akSVertex);
323  System::Read8le(pkIFile,2,(Real*)m_kLineOrigin);
324  System::Read8le(pkIFile,2,(Real*)m_kLineDirection);
325  }
326 
327  System::Fclose(pkIFile);
328 
329  switch (m_eQueryType)
330  {
331  case Query::QT_INT64:
332  m_pkQuery = WM4_NEW Query2Int64<Real>(m_iVertexQuantity,m_akSVertex);
333  break;
334  case Query::QT_INTEGER:
335  m_pkQuery = WM4_NEW Query2TInteger<Real>(m_iVertexQuantity,
336  m_akSVertex);
337  break;
338  case Query::QT_RATIONAL:
339  m_pkQuery = WM4_NEW Query2TRational<Real>(m_iVertexQuantity,
340  m_akSVertex);
341  break;
342  case Query::QT_REAL:
343  m_pkQuery = WM4_NEW Query2<Real>(m_iVertexQuantity,m_akSVertex);
344  break;
345  case Query::QT_FILTERED:
346  m_pkQuery = WM4_NEW Query2Filtered<Real>(m_iVertexQuantity,
347  m_akSVertex,m_fEpsilon);
348  break;
349  }
350 
351  return true;
352 }
