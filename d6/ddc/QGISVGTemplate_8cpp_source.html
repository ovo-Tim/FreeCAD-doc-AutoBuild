76 void QGISVGTemplate::load(const QByteArray& svgCode)
77 {
78  m_svgRender->load(svgCode);
79 
80  QSize size = m_svgRender->defaultSize();
81  m_svgItem->setSharedRenderer(m_svgRender);
82 
83  if (firstTime) {
84  createClickHandles();
85  firstTime = false;
86  }
87 
88  //convert from pixels or mm or inches in svg file to mm page size
89  TechDraw::DrawSVGTemplate* tmplte = getSVGTemplate();
90  double xaspect, yaspect;
91  xaspect = tmplte->getWidth() / static_cast<double>(size.width());
92  yaspect = tmplte->getHeight() / static_cast<double>(size.height());
93 
94  QTransform qtrans;
95  qtrans.translate(0.0, Rez::guiX(-tmplte->getHeight()));
96  qtrans.scale(Rez::guiX(xaspect), Rez::guiX(yaspect));
97  m_svgItem->setTransform(qtrans);
98 
99  if (Preferences::lightOnDark()) {
100  QColor color = PreferencesGui::getAccessibleQColor(QColor(Qt::black));
101  QGraphicsColorizeEffect* colorizeEffect = new QGraphicsColorizeEffect();
102  colorizeEffect->setColor(color);
103  m_svgItem->setGraphicsEffect(colorizeEffect);
104  }
105  else {
106  //remove and delete any existing graphics effect
107  if (m_svgItem->graphicsEffect()) {
108  m_svgItem->setGraphicsEffect(nullptr);
109  }
110  }
111 }

139 void QGISVGTemplate::createClickHandles()
140 {
141  TechDraw::DrawSVGTemplate* svgTemplate = getSVGTemplate();
142  if (svgTemplate->isRestoring()) {
143  //the embedded file is not available yet, so just return
144  return;
145  }
146 
147  QString templateFilename(QString::fromUtf8(svgTemplate->PageResult.getValue()));
148 
149  if (templateFilename.isEmpty()) {
150  return;
151  }
152 
153  QFile file(templateFilename);
154  if (!file.open(QIODevice::ReadOnly)) {
155  Base::Console().Error(
156  "QGISVGTemplate::createClickHandles - error opening template file %s\n",
157  svgTemplate->PageResult.getValue());
158  return;
159  }
160 
161  QDomDocument templateDocument;
162  if (!templateDocument.setContent(&file)) {
163  Base::Console().Message("QGISVGTemplate::createClickHandles - xml loading error\n");
164  return;
165  }
166  file.close();
167 
168  //TODO: Find location of special fields (first/third angle) and make graphics items for them
169 
170  QColor editClickBoxColor = PreferencesGui::templateClickBoxColor();
171 
172  auto textMap = svgTemplate->EditableTexts.getValues();
173 
174  TechDraw::XMLQuery query(templateDocument);
175 
176  // XPath query to select all <text> nodes with "freecad:editable" attribute
177  // XPath query to select all <tspan> nodes whose <text> parent
178  // has "freecad:editable" attribute
179  query.processItems(QString::fromUtf8("declare default element namespace \"" SVG_NS_URI "\"; "
180  "declare namespace freecad=\"" FREECAD_SVG_NS_URI "\"; "
181  "//text[@" FREECAD_ATTR_EDITABLE "]/tspan"),
182  [&](QDomElement& tspan) -> bool {
183  QString fontSizeString = tspan.attribute(QString::fromUtf8("font-size"));
184  QDomElement textElement = tspan.parentNode().toElement();
185  QString textAnchorString = textElement.attribute(QString::fromUtf8("text-anchor"));
186  QString name = textElement.attribute(QString::fromUtf8(FREECAD_ATTR_EDITABLE));
187 
188  double x = Rez::guiX(
189  textElement.attribute(QString::fromUtf8("x"), QString::fromUtf8("0.0")).toDouble());
190  double y = Rez::guiX(
191  textElement.attribute(QString::fromUtf8("y"), QString::fromUtf8("0.0")).toDouble());
192  if (name.isEmpty()) {
193  Base::Console().Warning(
194  "QGISVGTemplate::createClickHandles - no name for editable text at %f, %f\n", x, y);
195  return true;
196  }
197  std::string editableNameString = textMap[name.toStdString()];
198 
199  // default box size
200  double textHeight{0};
201  QString style = textElement.attribute(QString::fromUtf8("style"));
202  if (!style.isEmpty()) {
203  // get text attributes from style element
204  textHeight = getFontSizeFromStyle(style);
205  }
206 
207  if (textHeight == 0) {
208  textHeight = getFontSizeFromElement(fontSizeString);
209  }
210 
211  if (textHeight == 0.0) {
212  textHeight = Preferences::labelFontSizeMM() * 3.78; // 3.78 = px/mm
213  }
214 
215  QGraphicsTextItem textItemForLength;
216  QFont fontForLength(Preferences::labelFontQString());
217  fontForLength.setPixelSize(textHeight);
218  textItemForLength.setFont(fontForLength);
219  textItemForLength.setPlainText(QString::fromStdString(editableNameString));
220  auto brect = textItemForLength.boundingRect();
221  auto newLength = brect.width();
222 
223  double charWidth = newLength / editableNameString.length();
224  if (textAnchorString == QString::fromUtf8("middle")) {
225  x = x - editableNameString.length() * charWidth / 2;
226  }
227 
228  double textLength = editableNameString.length() * charWidth;
229  textLength = std::max(charWidth, textLength);
230 
231  auto item(new TemplateTextField(this, svgTemplate, name.toStdString()));
232  auto autoValue = svgTemplate->getAutofillByEditableName(name);
233  item->setAutofill(autoValue);
234 
235  double pad = 1.0;
236  double top = Rez::guiX(-svgTemplate->getHeight()) + y - textHeight - pad;
237  double bottom = top + textHeight + 2.0 * pad;
238  double left = x - pad;
239  item->setRectangle(QRectF(left, top,
240  newLength + 2.0 * pad, textHeight + 2.0 * pad));
241  item->setLine(QPointF( left, bottom),
242  QPointF(left + newLength + 2.0 * pad, bottom));
243  item->setLineColor(editClickBoxColor);
244 
245  item->setZValue(ZVALUE::SVGTEMPLATE + 1);
246  addToGroup(item);
247 
248  textFields.push_back(item);
249  return true;
250  });
251 }

254 double QGISVGTemplate::getFontSizeFromStyle(QString style)
255 {
256  if (style.isEmpty()) {
257  return 0.0;
258  }
259 
260  // get text attributes from style element
261  QRegularExpression rxFontSize(QString::fromUtf8("font-size:([0-9]*.?[0-9]*)px;"));
262  QRegularExpressionMatch match;
263 
264  int pos{0};
265  pos = style.indexOf(rxFontSize, 0, &match);
266  if (pos != -1) {
267  return Rez::guiX(match.captured(1).toDouble());
268  }
269 
270  return 0.0;
271 }
