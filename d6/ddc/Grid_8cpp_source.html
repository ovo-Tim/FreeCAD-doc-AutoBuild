94 void MeshGrid::InitGrid()
95 {
96  assert(_pclMesh);
97 
98  // Calculate grid length if not initialised
99  //
100  if ((_ulCtGridsX == 0) || (_ulCtGridsY == 0) || (_ulCtGridsZ == 0)) {
101  CalculateGridLength(MESH_CT_GRID_PER_AXIS);
102  }
103 
104  // Determine grid length and offset
105  //
106  {
107  Base::BoundBox3f clBBMesh = _pclMesh->GetBoundBox();
108 
109  float fLengthX = clBBMesh.LengthX();
110  float fLengthY = clBBMesh.LengthY();
111  float fLengthZ = clBBMesh.LengthZ();
112 
113  {
114  // Offset fGridLen/2
115  //
116  _fGridLenX = (1.0F + fLengthX) / float(_ulCtGridsX);
117  _fMinX = clBBMesh.MinX - 0.5F;
118  }
119 
120  {
121  _fGridLenY = (1.0F + fLengthY) / float(_ulCtGridsY);
122  _fMinY = clBBMesh.MinY - 0.5F;
123  }
124 
125  {
126  _fGridLenZ = (1.0F + fLengthZ) / float(_ulCtGridsZ);
127  _fMinZ = clBBMesh.MinZ - 0.5F;
128  }
129  }
130 
131  // Create data structure
132  _aulGrid.clear();
133  _aulGrid.resize(_ulCtGridsX);
134  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
135  _aulGrid[i].resize(_ulCtGridsY);
136  for (unsigned long j = 0; j < _ulCtGridsY; j++) {
137  _aulGrid[i][j].resize(_ulCtGridsZ);
138  }
139  }
140 }

142 unsigned long MeshGrid::Inside(const Base::BoundBox3f& rclBB,
143  std::vector<ElementIndex>& raulElements,
144  bool bDelDoubles) const
145 {
146  unsigned long ulMinX {};
147  unsigned long ulMinY {};
148  unsigned long ulMinZ {};
149  unsigned long ulMaxX {};
150  unsigned long ulMaxY {};
151  unsigned long ulMaxZ {};
152 
153  raulElements.clear();
154 
155  // Grid boxes for a more detailed selection
156  Position(Base::Vector3f(rclBB.MinX, rclBB.MinY, rclBB.MinZ), ulMinX, ulMinY, ulMinZ);
157  Position(Base::Vector3f(rclBB.MaxX, rclBB.MaxY, rclBB.MaxZ), ulMaxX, ulMaxY, ulMaxZ);
158 
159  for (auto i = ulMinX; i <= ulMaxX; i++) {
160  for (auto j = ulMinY; j <= ulMaxY; j++) {
161  for (auto k = ulMinZ; k <= ulMaxZ; k++) {
162  raulElements.insert(raulElements.end(),
163  _aulGrid[i][j][k].begin(),
164  _aulGrid[i][j][k].end());
165  }
166  }
167  }
168 
169  if (bDelDoubles) {
170  // remove duplicate mentions
171  std::sort(raulElements.begin(), raulElements.end());
172  raulElements.erase(std::unique(raulElements.begin(), raulElements.end()),
173  raulElements.end());
174  }

249 bool MeshGrid::CheckPosition(const Base::Vector3f& rclPoint,
250  unsigned long& rulX,

253 {
254  rulX = static_cast<unsigned long>((rclPoint.x - _fMinX) / _fGridLenX);

294 void MeshGrid::CalculateGridLength(int iCtGridPerAxis)
295 {
296  // Calculate grid lengths or number of grids per dimension
297  // There should be about 10 (?!?!) facets per grid
298  // respectively max grids should not exceed 10000
299  Base::BoundBox3f clBBMesh = _pclMesh->GetBoundBox();
300 
301  float fLenghtX = clBBMesh.LengthX();
302  float fLenghtY = clBBMesh.LengthY();
303  float fLenghtZ = clBBMesh.LengthZ();
304 
305  float fLenghtD = clBBMesh.CalcDiagonalLength();
306 
307  float fLengthTol = 0.05F * fLenghtD;
308 
309  bool bLenghtXisZero = (fLenghtX <= fLengthTol);
310  bool bLenghtYisZero = (fLenghtY <= fLengthTol);
311  bool bLenghtZisZero = (fLenghtZ <= fLengthTol);
312 
313  int iFlag = 0;
314 
315  int iMaxGrids = 1;
316 
317  if (bLenghtXisZero) {
318  iFlag += 1;
319  }
320  else {
321  iMaxGrids *= iCtGridPerAxis;
322  }
323 
324  if (bLenghtYisZero) {
325  iFlag += 2;
326  }
327  else {
328  iMaxGrids *= iCtGridPerAxis;
329  }
330 
331  if (bLenghtZisZero) {
332  iFlag += 4;
333  }
334  else {
335  iMaxGrids *= iCtGridPerAxis;
336  }
337 
338  unsigned long ulGridsFacets = 10;
339 
340  float fFactorVolumen = 40.0;
341  float fFactorArea = 10.0;
342 
343  switch (iFlag) {
344  case 0: {
345  float fVolumen = fLenghtX * fLenghtY * fLenghtZ;
346 
347  float fVolumenGrid = (fVolumen * ulGridsFacets) / (fFactorVolumen * _ulCtElements);
348 
349  if ((fVolumenGrid * iMaxGrids) < fVolumen) {
350  fVolumenGrid = fVolumen / static_cast<float>(iMaxGrids);
351  }
352 
353  float fLengthGrid = std::pow(fVolumenGrid, 1.0F / 3.0F);
354 
355  _ulCtGridsX = std::max(static_cast<unsigned long>(fLenghtX / fLengthGrid), 1UL);
356  _ulCtGridsY = std::max(static_cast<unsigned long>(fLenghtY / fLengthGrid), 1UL);
357  _ulCtGridsZ = std::max(static_cast<unsigned long>(fLenghtZ / fLengthGrid), 1UL);
358 
359  } break;
360  case 1: {
361  _ulCtGridsX = 1; // bLenghtXisZero
362 
363  float fArea = fLenghtY * fLenghtZ;
364 
365  float fAreaGrid = (fArea * ulGridsFacets) / (fFactorArea * _ulCtElements);
366 
367  if ((fAreaGrid * iMaxGrids) < fArea) {
368  fAreaGrid = fArea / static_cast<float>(iMaxGrids);
369  }
370 
371  auto fLengthGrid = float(std::sqrt(fAreaGrid));
372 
373  _ulCtGridsY =
374  std::max<unsigned long>(static_cast<unsigned long>(fLenghtY / fLengthGrid), 1);
375  _ulCtGridsZ =
376  std::max<unsigned long>(static_cast<unsigned long>(fLenghtZ / fLengthGrid), 1);
377  } break;
378  case 2: {
379  _ulCtGridsY = 1; // bLenghtYisZero
380 
381  float fArea = fLenghtX * fLenghtZ;
382 
383  float fAreaGrid = (fArea * ulGridsFacets) / (fFactorArea * _ulCtElements);
384 
385  if ((fAreaGrid * iMaxGrids) < fArea) {
386  fAreaGrid = fArea / static_cast<float>(iMaxGrids);
387  }
388 
389  auto fLengthGrid = float(std::sqrt(fAreaGrid));
390 
391  _ulCtGridsX =
392  std::max<unsigned long>(static_cast<unsigned long>(fLenghtX / fLengthGrid), 1);
393  _ulCtGridsZ =
394  std::max<unsigned long>(static_cast<unsigned long>(fLenghtZ / fLengthGrid), 1);
395  } break;
396  case 3: {
397  _ulCtGridsX = 1; // bLenghtXisZero
398  _ulCtGridsY = 1; // bLenghtYisZero
399  _ulCtGridsZ = static_cast<unsigned long>(iMaxGrids); // bLenghtYisZero
400  } break;
401  case 4: {
402  _ulCtGridsZ = 1; // bLenghtZisZero
403 
404  float fArea = fLenghtX * fLenghtY;
405 
406  float fAreaGrid = (fArea * ulGridsFacets) / (fFactorArea * _ulCtElements);
407 
408  if ((fAreaGrid * iMaxGrids) < fArea) {
409  fAreaGrid = fArea / static_cast<float>(iMaxGrids);
410  }
411 
412  auto fLengthGrid = float(std::sqrt(fAreaGrid));
413 
414  _ulCtGridsX =
415  std::max<unsigned long>(static_cast<unsigned long>(fLenghtX / fLengthGrid), 1);
416  _ulCtGridsY =
417  std::max<unsigned long>(static_cast<unsigned long>(fLenghtY / fLengthGrid), 1);
418  } break;
419  case 5: {
420  _ulCtGridsX = 1; // bLenghtXisZero
421  _ulCtGridsZ = 1; // bLenghtZisZero
422  _ulCtGridsY = static_cast<unsigned long>(iMaxGrids); // bLenghtYisZero
423  } break;
424  case 6: {
425  _ulCtGridsY = 1; // bLenghtYisZero
426  _ulCtGridsZ = 1; // bLenghtZisZero
427  _ulCtGridsX = static_cast<unsigned long>(iMaxGrids); // bLenghtYisZero
428  } break;
429  case 7: {
430  _ulCtGridsX = static_cast<unsigned long>(iMaxGrids); // bLenghtXisZero
431  _ulCtGridsY = static_cast<unsigned long>(iMaxGrids); // bLenghtYisZero
432  _ulCtGridsZ = static_cast<unsigned long>(iMaxGrids); // bLenghtZisZero
433  } break;
434  }
435 }

437 void MeshGrid::SearchNearestFromPoint(const Base::Vector3f& pnt,
438  std::set<ElementIndex>& indices) const
439 {
440  indices.clear();
441  Base::BoundBox3f clBB = GetBoundBox();
442 
443  if (clBB.IsInBox(pnt)) { // Point lies within
444  unsigned long ulX {};
445  unsigned long ulY {};
446  unsigned long ulZ {};
447  Position(pnt, ulX, ulY, ulZ);
448  // int nX = ulX, nY = ulY, nZ = ulZ;
449  unsigned long ulMaxLevel =
450  std::max<unsigned long>(_ulCtGridsX, std::max<unsigned long>(_ulCtGridsY, _ulCtGridsZ));
451  unsigned long ulLevel = 0;
452  while (indices.empty() && ulLevel <= ulMaxLevel) {
453  GetHull(ulX, ulY, ulZ, ulLevel++, indices);
454  }
455  GetHull(ulX, ulY, ulZ, ulLevel, indices);
456  }
457  else { // Point outside
458  Base::BoundBox3f::SIDE tSide = clBB.GetSideFromRay(pnt, clBB.GetCenter() - pnt);
459  switch (tSide) {
460  case Base::BoundBox3f::RIGHT: {
461  unsigned long nX = 0;
462  while (indices.empty() && nX < _ulCtGridsX) {
463  for (unsigned long i = 0; i < _ulCtGridsY; i++) {
464  for (unsigned long j = 0; j < _ulCtGridsZ; j++) {
465  indices.insert(_aulGrid[nX][i][j].begin(), _aulGrid[nX][i][j].end());
466  }
467  }
468  nX++;
469  }
470  break;
471  }
472  case Base::BoundBox3f::LEFT: {
473  unsigned long nX = _ulCtGridsX - 1;
474  while (indices.empty() && nX < _ulCtGridsX) {
475  for (unsigned long i = 0; i < _ulCtGridsY; i++) {
476  for (unsigned long j = 0; j < _ulCtGridsZ; j++) {
477  indices.insert(_aulGrid[nX][i][j].begin(), _aulGrid[nX][i][j].end());
478  }
479  }
480  nX++;
481  }
482  break;
483  }
484  case Base::BoundBox3f::TOP: {
485  unsigned long nY = 0;
486  while (indices.empty() && nY < _ulCtGridsY) {
487  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
488  for (unsigned long j = 0; j < _ulCtGridsZ; j++) {
489  indices.insert(_aulGrid[i][nY][j].begin(), _aulGrid[i][nY][j].end());
490  }
491  }
492  nY++;
493  }
494  break;
495  }
496  case Base::BoundBox3f::BOTTOM: {
497  unsigned long nY = _ulCtGridsY - 1;
498  while (indices.empty() && nY < _ulCtGridsY) {
499  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
500  for (unsigned long j = 0; j < _ulCtGridsZ; j++) {
501  indices.insert(_aulGrid[i][nY][j].begin(), _aulGrid[i][nY][j].end());
502  }
503  }
504  nY--;
505  }
506  break;
507  }
508  case Base::BoundBox3f::BACK: {
509  unsigned long nZ = 0;
510  while (indices.empty() && nZ < _ulCtGridsZ) {
511  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
512  for (unsigned long j = 0; j < _ulCtGridsY; j++) {
513  indices.insert(_aulGrid[i][j][nZ].begin(), _aulGrid[i][j][nZ].end());
514  }
515  }
516  nZ++;
517  }
518  break;
519  }
520  case Base::BoundBox3f::FRONT: {
521  unsigned long nZ = _ulCtGridsZ - 1;
522  while (indices.empty() && nZ < _ulCtGridsZ) {
523  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
524  for (unsigned long j = 0; j < _ulCtGridsY; j++) {
525  indices.insert(_aulGrid[i][j][nZ].begin(), _aulGrid[i][j][nZ].end());
526  }
527  }
528  nZ--;
529  }
530  break;
531  }
532 
533  default:
534  break;
535  }
536  }

539 void MeshGrid::GetHull(unsigned long ulX,
540  unsigned long ulY,
541  unsigned long ulZ,
542  unsigned long ulDistance,
543  std::set<ElementIndex>& raclInd) const
544 {
545  int nX1 = std::max<int>(0, int(ulX) - int(ulDistance));
546  int nY1 = std::max<int>(0, int(ulY) - int(ulDistance));
547  int nZ1 = std::max<int>(0, int(ulZ) - int(ulDistance));
548  int nX2 = std::min<int>(int(_ulCtGridsX) - 1, int(ulX) + int(ulDistance));
549  int nY2 = std::min<int>(int(_ulCtGridsY) - 1, int(ulY) + int(ulDistance));
550  int nZ2 = std::min<int>(int(_ulCtGridsZ) - 1, int(ulZ) + int(ulDistance));
551 
552  // top plane
553  for (int i = nX1; i <= nX2; i++) {
554  for (int j = nY1; j <= nY2; j++) {
555  GetElements(static_cast<unsigned long>(i),
556  static_cast<unsigned long>(j),
557  static_cast<unsigned long>(nZ1),
558  raclInd);
559  }
560  }
561  // bottom plane
562  for (int i = nX1; i <= nX2; i++) {
563  for (int j = nY1; j <= nY2; j++) {
564  GetElements(static_cast<unsigned long>(i),
565  static_cast<unsigned long>(j),
566  static_cast<unsigned long>(nZ2),
567  raclInd);
568  }
569  }
570  // left plane
571  for (int i = nY1; i <= nY2; i++) {
572  for (int j = (nZ1 + 1); j <= (nZ2 - 1); j++) {
573  GetElements(static_cast<unsigned long>(nX1),
574  static_cast<unsigned long>(i),
575  static_cast<unsigned long>(j),
576  raclInd);
577  }
578  }
579  // right plane
580  for (int i = nY1; i <= nY2; i++) {
581  for (int j = (nZ1 + 1); j <= (nZ2 - 1); j++) {
582  GetElements(static_cast<unsigned long>(nX2),
583  static_cast<unsigned long>(i),
584  static_cast<unsigned long>(j),
585  raclInd);
586  }
587  }
588  // front plane
589  for (int i = (nX1 + 1); i <= (nX2 - 1); i++) {
590  for (int j = (nZ1 + 1); j <= (nZ2 - 1); j++) {
591  GetElements(static_cast<unsigned long>(i),
592  static_cast<unsigned long>(nY1),
593  static_cast<unsigned long>(j),
594  raclInd);
595  }
596  }
597  // back plane
598  for (int i = (nX1 + 1); i <= (nX2 - 1); i++) {
599  for (int j = (nZ1 + 1); j <= (nZ2 - 1); j++) {
600  GetElements(static_cast<unsigned long>(i),
601  static_cast<unsigned long>(nY2),
602  static_cast<unsigned long>(j),
603  raclInd);
604  }
605  }
606 }

760 unsigned long MeshFacetGrid::SearchNearestFromPoint(const Base::Vector3f& rclPt) const
761 {
762  ElementIndex ulFacetInd = ELEMENT_INDEX_MAX;
763  float fMinDist = FLOAT_MAX;
764  Base::BoundBox3f clBB = GetBoundBox();
765 
766  if (clBB.IsInBox(rclPt)) { // Point lies within
767  unsigned long ulX {};
768  unsigned long ulY {};
769  unsigned long ulZ {};
770  Position(rclPt, ulX, ulY, ulZ);
771  float fMinGridDist = std::min<float>(std::min<float>(_fGridLenX, _fGridLenY), _fGridLenZ);
772  unsigned long ulDistance = 0;
773  while (fMinDist > (fMinGridDist * float(ulDistance))) {
774  SearchNearestFacetInHull(ulX, ulY, ulZ, ulDistance, rclPt, ulFacetInd, fMinDist);
775  ulDistance++;
776  }
777  SearchNearestFacetInHull(ulX, ulY, ulZ, ulDistance, rclPt, ulFacetInd, fMinDist);
778  }
779  else { // Point outside
780  Base::BoundBox3f::SIDE tSide = clBB.GetSideFromRay(rclPt, clBB.GetCenter() - rclPt);
781  switch (tSide) {
782  case Base::BoundBox3f::RIGHT: {
783  unsigned long nX = 0;
784  while ((fMinDist > ((clBB.MinX - rclPt.x) + (float(nX) * _fGridLenX)))
785  && (nX < _ulCtGridsX)) {
786  for (unsigned long i = 0; i < _ulCtGridsY; i++) {
787  for (unsigned long j = 0; j < _ulCtGridsZ; j++) {
788  SearchNearestFacetInGrid(nX, i, j, rclPt, fMinDist, ulFacetInd);
789  }
790  }
791  nX++;
792  }
793  break;
794  }
795  case Base::BoundBox3f::LEFT: {
796  unsigned long nX = _ulCtGridsX - 1;
797  while ((fMinDist > ((rclPt.x - clBB.MinX) + (float(nX) * _fGridLenX)))
798  && (nX < _ulCtGridsX)) {
799  for (unsigned long i = 0; i < _ulCtGridsY; i++) {
800  for (unsigned long j = 0; j < _ulCtGridsZ; j++) {
801  SearchNearestFacetInGrid(nX, i, j, rclPt, fMinDist, ulFacetInd);
802  }
803  }
804  nX--;
805  }
806  break;
807  }
808  case Base::BoundBox3f::TOP: {
809  unsigned long nY = 0;
810  while ((fMinDist > ((clBB.MinY - rclPt.y) + (float(nY) * _fGridLenY)))
811  && (nY < _ulCtGridsY)) {
812  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
813  for (unsigned long j = 0; j < _ulCtGridsZ; j++) {
814  SearchNearestFacetInGrid(i, nY, j, rclPt, fMinDist, ulFacetInd);
815  }
816  }
817  nY++;
818  }
819  break;
820  }
821  case Base::BoundBox3f::BOTTOM: {
822  unsigned long nY = _ulCtGridsY - 1;
823  while ((fMinDist > ((rclPt.y - clBB.MinY) + (float(nY) * _fGridLenY)))
824  && (nY < _ulCtGridsY)) {
825  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
826  for (unsigned long j = 0; j < _ulCtGridsZ; j++) {
827  SearchNearestFacetInGrid(i, nY, j, rclPt, fMinDist, ulFacetInd);
828  }
829  }
830  nY--;
831  }
832  break;
833  }
834  case Base::BoundBox3f::BACK: {
835  unsigned long nZ = 0;
836  while ((fMinDist > ((clBB.MinZ - rclPt.z) + (float(nZ) * _fGridLenZ)))
837  && (nZ < _ulCtGridsZ)) {
838  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
839  for (unsigned long j = 0; j < _ulCtGridsY; j++) {
840  SearchNearestFacetInGrid(i, j, nZ, rclPt, fMinDist, ulFacetInd);
841  }
842  }
843  nZ++;
844  }
845  break;
846  }
847  case Base::BoundBox3f::FRONT: {
848  unsigned long nZ = _ulCtGridsZ - 1;
849  while ((fMinDist > ((rclPt.z - clBB.MinZ) + (float(nZ) * _fGridLenZ)))
850  && (nZ < _ulCtGridsZ)) {
851  for (unsigned long i = 0; i < _ulCtGridsX; i++) {
852  for (unsigned long j = 0; j < _ulCtGridsY; j++) {
853  SearchNearestFacetInGrid(i, j, nZ, rclPt, fMinDist, ulFacetInd);
854  }
855  }
856  nZ--;
857  }
858  break;
859  }
860  default:
861  break;
862  }
863  }
864 
865  return ulFacetInd;
866 }

898 void MeshFacetGrid::SearchNearestFacetInHull(unsigned long ulX,
899  unsigned long ulY,
900  unsigned long ulZ,
901  unsigned long ulDistance,
902  const Base::Vector3f& rclPt,
903  ElementIndex& rulFacetInd,
904  float& rfMinDist) const
905 {
906  int nX1 = std::max<int>(0, int(ulX) - int(ulDistance));
907  int nY1 = std::max<int>(0, int(ulY) - int(ulDistance));
908  int nZ1 = std::max<int>(0, int(ulZ) - int(ulDistance));
909  int nX2 = std::min<int>(int(_ulCtGridsX) - 1, int(ulX) + int(ulDistance));
910  int nY2 = std::min<int>(int(_ulCtGridsY) - 1, int(ulY) + int(ulDistance));
911  int nZ2 = std::min<int>(int(_ulCtGridsZ) - 1, int(ulZ) + int(ulDistance));
912 
913  // top plane
914  for (int i = nX1; i <= nX2; i++) {
915  for (int j = nY1; j <= nY2; j++) {
916  SearchNearestFacetInGrid(static_cast<unsigned long>(i),
917  static_cast<unsigned long>(j),
918  static_cast<unsigned long>(nZ1),
919  rclPt,
920  rfMinDist,
921  rulFacetInd);
922  }
923  }
924  // bottom plane
925  for (int i = nX1; i <= nX2; i++) {
926  for (int j = nY1; j <= nY2; j++) {
927  SearchNearestFacetInGrid(static_cast<unsigned long>(i),
928  static_cast<unsigned long>(j),
929  static_cast<unsigned long>(nZ2),
930  rclPt,
931  rfMinDist,
932  rulFacetInd);
933  }
934  }
935  // left plane
936  for (int i = nY1; i <= nY2; i++) {
937  for (int j = (nZ1 + 1); j <= (nZ2 - 1); j++) {
938  SearchNearestFacetInGrid(static_cast<unsigned long>(nX1),
939  static_cast<unsigned long>(i),
940  static_cast<unsigned long>(j),
941  rclPt,
942  rfMinDist,
943  rulFacetInd);
944  }
945  }
946  // right plane
947  for (int i = nY1; i <= nY2; i++) {
948  for (int j = (nZ1 + 1); j <= (nZ2 - 1); j++) {
949  SearchNearestFacetInGrid(static_cast<unsigned long>(nX2),
950  static_cast<unsigned long>(i),
951  static_cast<unsigned long>(j),
952  rclPt,
953  rfMinDist,
954  rulFacetInd);
955  }
956  }
957  // front plane
958  for (int i = (nX1 + 1); i <= (nX2 - 1); i++) {
959  for (int j = (nZ1 + 1); j <= (nZ2 - 1); j++) {
960  SearchNearestFacetInGrid(static_cast<unsigned long>(i),
961  static_cast<unsigned long>(nY1),
962  static_cast<unsigned long>(j),
963  rclPt,
964  rfMinDist,
965  rulFacetInd);
966  }
967  }
968  // back plane
969  for (int i = (nX1 + 1); i <= (nX2 - 1); i++) {
970  for (int j = (nZ1 + 1); j <= (nZ2 - 1); j++) {
971  SearchNearestFacetInGrid(static_cast<unsigned long>(i),
972  static_cast<unsigned long>(nY2),
973  static_cast<unsigned long>(j),
974  rclPt,
975  rfMinDist,
976  rulFacetInd);
977  }
978  }
979 }

1195 bool MeshGridIterator::NextOnRay(std::vector<ElementIndex>& raulElements)
1196 {
1197  if (!_bValidRay) {
1198  return false; // not initialized or beam exited
1199  }
1200 
1201  raulElements.clear();
1202 
1203  Base::Vector3f clIntersectPoint;
1204 
1205  // Look for the next adjacent BB on the search beam
1206  Base::BoundBox3f::SIDE tSide =
1207  _rclGrid.GetBoundBox(_ulX, _ulY, _ulZ).GetSideFromRay(_clPt, _clDir, clIntersectPoint);
1208 
1209  // Search area
1210  //
1211  if ((_clPt - clIntersectPoint).Length() > _fMaxSearchArea) {
1212  _bValidRay = false;
1213  }
1214  else {
1215  switch (tSide) {
1216  case Base::BoundBox3f::LEFT:
1217  _ulX--;
1218  break;
1219  case Base::BoundBox3f::RIGHT:
1220  _ulX++;
1221  break;
1222  case Base::BoundBox3f::BOTTOM:
1223  _ulY--;
1224  break;
1225  case Base::BoundBox3f::TOP:
1226  _ulY++;
1227  break;
1228  case Base::BoundBox3f::FRONT:
1229  _ulZ--;
1230  break;
1231  case Base::BoundBox3f::BACK:
1232  _ulZ++;
1233  break;
1234 
1235  default:
1236  case Base::BoundBox3f::INVALID:
1237  _bValidRay = false;
1238  break;
1239  }
1240 
1241  GridElement pos(_ulX, _ulY, _ulZ);
1242  if (_cSearchPositions.find(pos) != _cSearchPositions.end()) {
1243  _bValidRay =
1244  false; // grid element already visited => result from GetSideFromRay invalid
1245  }
1246  }
