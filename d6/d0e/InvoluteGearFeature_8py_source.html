37 def makeInvoluteGear(name):
38  '''makeInvoluteGear(name): makes an InvoluteGear'''
39  obj = FreeCAD.ActiveDocument.addObject("Part::Part2DObjectPython",name)
40  _InvoluteGear(obj)

130  def execute(self,obj):
131  w = fcgear.FCWireBuilder()
132  generator_func = involute.CreateExternalGear if obj.ExternalGear else involute.CreateInternalGear
133  generator_func(w, obj.Modules.Value, obj.NumberOfTeeth, obj.PressureAngle.Value,
134  split=obj.HighPrecision, addCoeff=obj.AddendumCoefficient, dedCoeff=obj.DedendumCoefficient,
135  filletCoeff=obj.RootFilletCoefficient, shiftCoeff=obj.ProfileShiftCoefficient)
136  gearw = Part.Wire([o.toShape() for o in w.wire])
137  obj.Shape = gearw
138  obj.positionBySupport()
139  return
