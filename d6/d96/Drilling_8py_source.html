200  def circularHoleExecute(self, obj, holes):
201  """circularHoleExecute(obj, holes) ... generate drill operation for each hole in holes."""
202  Path.Log.track()
203  machine = PathMachineState.MachineState()
204 
205  self.commandlist.append(Path.Command("(Begin Drilling)"))
206 
207  # rapid to clearance height
208  command = Path.Command("G0", {"Z": obj.ClearanceHeight.Value})
209  machine.addCommand(command)
210  self.commandlist.append(command)
211 
212  self.commandlist.append(Path.Command("G90")) # Absolute distance mode
213 
214  # Calculate offsets to add to target edge
215  endoffset = 0.0
216  if obj.ExtraOffset == "Drill Tip":
217  endoffset = PathUtils.drillTipLength(self.tool)
218  elif obj.ExtraOffset == "2x Drill Tip":
219  endoffset = PathUtils.drillTipLength(self.tool) * 2
220 
221  if not hasattr(obj, "KeepToolDown"):
222  obj.addProperty(
223  "App::PropertyBool",
224  "KeepToolDown",
225  "Drill",
226  QT_TRANSLATE_NOOP(
227  "App::Property",
228  "Apply G99 retraction: only retract to RetractHeight between holes in this operation",
229  ),
230  )
231 
232  if not hasattr(obj, "RetractMode"):
233  obj.addProperty(
234  "App::PropertyEnumeration",
235  "RetractMode",
236  "Drill",
237  QT_TRANSLATE_NOOP(
238  "App::Property",
239  "Controls tool retract height between holes in same op, Default=G98: safety height",
240  ),
241  )
242  # ensure new enums exist in old class
243  for n in self.propertyEnumerations():
244  setattr(obj, n[0], n[1])
245 
246  # http://linuxcnc.org/docs/html/gcode/g-code.html#gcode:g98-g99
247  if obj.KeepToolDown:
248  obj.RetractMode = "G99"
249  else:
250  obj.RetractMode = "G98"
251  self.commandlist.append(Path.Command(obj.RetractMode))
252 
253  holes = PathUtils.sort_locations(holes, ["x", "y"])
254 
255  # This section is technical debt. The computation of the
256  # target shapes should be factored out for reuse.
257  # This will likely mean refactoring upstream CircularHoleBase to pass
258  # spotshapes instead of holes.
259 
260  edgelist = []
261  for hole in holes:
262  v1 = FreeCAD.Vector(hole["x"], hole["y"], obj.StartDepth.Value)
263  v2 = FreeCAD.Vector(hole["x"], hole["y"], obj.FinalDepth.Value - endoffset)
264  edgelist.append(Part.makeLine(v1, v2))
265 
266  # iterate the edgelist and generate gcode
267  for edge in edgelist:
268 
269  Path.Log.debug(edge)
270 
271  # move to hole location
272 
273  startPoint = edge.Vertexes[0].Point
274 
275  # G81,83 will do this move anyway but FreeCAD sets feedrate according to _isVertical,
276  # so do this explicitly before drilling.
277  command = Path.Command("G0", {"X": startPoint.x, "Y": startPoint.y})
278  self.commandlist.append(command)
279  machine.addCommand(command)
280 
281  # Technical Debt: We are assuming the edges are aligned.
282  # This assumption should be corrected and the necessary rotations
283  # performed to align the edge with the Z axis for drilling
284 
285  # Perform drilling
286  dwelltime = obj.DwellTime if obj.DwellEnabled else 0.0
287  peckdepth = obj.PeckDepth.Value if obj.PeckEnabled else 0.0
288  repeat = 1 # technical debt: Add a repeat property for user control
289  chipBreak = obj.chipBreakEnabled and obj.PeckEnabled
290 
291  try:
292  drillcommands = drill.generate(
293  edge,
294  dwelltime,
295  peckdepth,
296  repeat,
297  obj.RetractHeight.Value,
298  chipBreak=chipBreak,
299  feedRetract=obj.feedRetractEnabled,
300  )
301 
302  except ValueError as e: # any targets that fail the generator are ignored
303  Path.Log.info(e)
304  continue
305 
306  for command in drillcommands:
307  self.commandlist.append(command)
308  machine.addCommand(command)
309 
310  # Cancel canned drilling cycle
311  self.commandlist.append(Path.Command("G80"))
312  command = Path.Command("G0", {"Z": obj.SafeHeight.Value})
313  self.commandlist.append(command)
314  machine.addCommand(command)
