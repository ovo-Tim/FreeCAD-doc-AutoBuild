123 void ViewProviderViewClip::dropObject(App::DocumentObject* docObj)
124 {
125  if (docObj->isDerivedFrom(TechDraw::DrawProjGroupItem::getClassTypeId())) {
126  //DPGI can not be dropped onto the Page if it belongs to DPG
127  auto* dpgi = static_cast<TechDraw::DrawProjGroupItem*>(docObj);
128  if (dpgi->getPGroup()) {
129  return;
130  }
131  }
132  if (!docObj->isDerivedFrom(TechDraw::DrawView::getClassTypeId())) {
133  return;
134  }
135 
136  auto dv = static_cast<TechDraw::DrawView*>(docObj);
137  TechDraw::DrawPage* pageClip = getObject()->findParentPage();
138  TechDraw::DrawPage* pageView = dv->findParentPage();
139  if (!pageClip || !pageView) {
140  return;
141  }
142 
143  if (pageClip != pageView) {
144  pageView->removeView(dv);
145  pageClip->addView(dv);
146  }
147 
148  getObject()->addView(dv);
149 }
