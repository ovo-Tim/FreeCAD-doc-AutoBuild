52 Real DistVector3Segment3<Real>::GetSquared ()
53 {
54  Vector3<Real> kDiff = m_rkVector - m_rkSegment.Origin;
55  m_fSegmentParameter = m_rkSegment.Direction.Dot(kDiff);
56 
57  if (-m_rkSegment.Extent < m_fSegmentParameter)
58  {
59  if (m_fSegmentParameter < m_rkSegment.Extent)
60  {
61  m_kClosestPoint1 = m_rkSegment.Origin +
62  m_fSegmentParameter*m_rkSegment.Direction;
63  }
64  else
65  {
66  m_kClosestPoint1 = m_rkSegment.Origin +
67  m_rkSegment.Extent*m_rkSegment.Direction;
68  }
69  }
70  else
71  {
72  m_kClosestPoint1 = m_rkSegment.Origin -
73  m_rkSegment.Extent*m_rkSegment.Direction;
74  }
75 
76  m_kClosestPoint0 = m_rkVector;
77  kDiff = m_kClosestPoint1 - m_kClosestPoint0;
78  return kDiff.SquaredLength();
79 }
