43  """Opens an IFC file"""

58 def insert(
59  filename,
60  docname,
61  strategy=None,
62  shapemode=None,
63  switchwb=None,
64  silent=False,
65  singledoc=False,
66 ):
67  """Inserts an IFC document in a FreeCAD document.
68  Singledoc defines if the produced result is a locked document or not. The
69  strategy is:
70  - When opening IFC files, locked/unlocked depends on the preferences (default locked)
71  - When inserting IFC files, always unlocked (an IFC doc object is created)"""
72 
73  from PySide import QtCore # lazy loading
74 
75  strategy, shapemode, switchwb = get_options(strategy, shapemode, switchwb, silent)
76  if strategy is None:
77  print("Aborted.")
78  return
79  stime = time.time()
80  try:
81  document = FreeCAD.getDocument(docname)
82  except:
83  document = FreeCAD.newDocument()
84  if singledoc is None:
85  singledoc = PARAMS.GetBool("SingleDoc", True)
86  if singledoc:
87  prj_obj = ifc_tools.convert_document(document, filename, shapemode, strategy)
88  QtCore.QTimer.singleShot(100, toggle_lock_on)
89  else:
90  prj_obj = ifc_tools.create_document_object(
91  document, filename, shapemode, strategy
92  )
93  QtCore.QTimer.singleShot(100, toggle_lock_off)
94  if PARAMS.GetBool("LoadOrphans", True):
95  ifc_tools.load_orphans(prj_obj)
96  if not silent and PARAMS.GetBool("LoadMaterials", False):
97  ifc_materials.load_materials(prj_obj)
98  if PARAMS.GetBool("LoadLayers", False):
99  ifc_layers.load_layers(prj_obj)
100  if PARAMS.GetBool("LoadPsets", False):
101  ifc_psets.load_psets(prj_obj)
102  document.recompute()
103  # print a reference to the IFC file on the console
104  if FreeCAD.GuiUp and PARAMS.GetBool("IfcFileToConsole", False):
105  if isinstance(prj_obj, FreeCAD.DocumentObject):
106  pstr = "FreeCAD.getDocument('{}').{}.Proxy.ifcfile"
107  pstr = pstr.format(prj_obj.Document.Name, prj_obj.Name)
108  else:
109  pstr = "FreeCAD.getDocument('{}').Proxy.ifcfile"
110  pstr = pstr.format(prj_obj.Name)
111  pstr = "ifcfile = " + pstr
112  pstr += " # warning: make sure you know what you are doing when using this!"
113  FreeCADGui.doCommand(pstr)
114  endtime = "%02d:%02d" % (divmod(round(time.time() - stime, 1), 60))
115  fsize = round(os.path.getsize(filename) / 1048576, 2)
116  print("Imported", os.path.basename(filename), "(", fsize, "Mb ) in", endtime)
117  if FreeCAD.GuiUp and switchwb:
118  FreeCADGui.activateWorkbench("BIMWorkbench")
119  return document

122 def get_options(strategy=None, shapemode=None, switchwb=None, silent=False):
123  """Shows a dialog to get import options
124
