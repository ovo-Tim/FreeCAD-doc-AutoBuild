85 def setup(doc=None, solvertype="ccxtools"):
86 
87  # init FreeCAD document
88  if doc is None:
89  doc = init_doc()
90 
91  # explanation object
92  # just keep the following line and change text string in get_explanation method
93  manager.add_explanation_obj(doc, get_explanation(manager.get_header(get_information())))
94 
95  diameter = 1000
96  thickness = 100
97  cs_wire_outer = doc.addObject("Part::Circle", "OuterWireOfCrossSection")
98  cs_wire_outer.Radius = diameter / 2
99  cs_wire_outer.Placement = FreeCAD.Placement(
100  FreeCAD.Vector(0, 500, 500),
101  FreeCAD.Rotation(0, 90, 0),
102  FreeCAD.Vector(0, 0, 0),
103  )
104  cs_wire_inner = doc.addObject("Part::Circle", "InnerWireOfCrossSection")
105  cs_wire_inner.Radius = (diameter / 2) - thickness
106  cs_wire_inner.Placement = cs_wire_outer.Placement
107  doc.recompute()
108  cs_face = doc.addObject("Part::Face", "CrossSectionFace")
109  cs_face.Sources = [cs_wire_outer, cs_wire_inner]
110  doc.recompute()
111 
112  # setup CalculiX cantilever
113  doc = setup_cantilever_base_edge(doc, solvertype)
114  beamsection_obj = doc.getObject("BeamCrossSection")
115 
116  # change cross section to a pipe section
117  beamsection_obj.SectionType = "Pipe"
118  beamsection_obj.PipeDiameter = diameter
119  beamsection_obj.PipeThickness = thickness
120 
121  doc.recompute()
122  return doc
