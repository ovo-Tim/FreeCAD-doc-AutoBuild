22 """
23 This module recreates the whole structure of an IFC file in FreeCAD (ie.
24 one FreeCAD object per IFC entity). It serves mostly as a proof of concept
25 and test to see how doable it is with large files.
26 
27 The geometry of objects is not imported, only attributes, which are mapped
28 to FreeCAD properties.
29 
30 The IfcOpenHouse model, included with ifcopenshell, imports in 4 sec on
31 my ryzen9 machine, for 2700 objects. Larger files like the King Arch file
32 (20 Mb / 750 000 objects) would import in 18 minutes...
33 """

51  def claimChildren(self):
52  children = []
53  relprops = ["Item", "ForLayerSet"] # properties that actually store parents
54  for prop in self.Object.PropertiesList:
55  if prop.startswith("Relating") or (prop in relprops):
56  continue
57  else:
58  value = getattr(self.Object, prop)
59  if hasattr(value, "ViewObject"):
60  children.append(value)
61  elif isinstance(value, list):
62  for item in value:
63  if hasattr(item, "ViewObject"):
64  children.append(item)
65  for parent in self.Object.InList:
66  for prop in parent.PropertiesList:
67  if prop.startswith("Relating") or (prop in relprops):
68  value = getattr(parent, prop)
69  if value == self.Object:
70  children.append(parent)
71  return children

74 def create(ifcentity):
75  """The main function that creates objects and fills properties"""
76 
77  name = "Entity" + str(ifcentity.id())
78  obj = FreeCAD.ActiveDocument.getObject(name)
79  if obj:
80  return obj
81  obj = FreeCAD.ActiveDocument.addObject("App::FeaturePython", name)
82  if getattr(ifcentity, "Name", None):
83  obj.Label = ifcentity.Name
84  else:
85  obj.Label = ifcentity.is_a()
86  for attr, value in ifcentity.get_info().items():
87  if attr not in obj.PropertiesList:
88  if attr == "id":
89  attr = "StepId"
90  elif attr == "type":
91  attr = "Type"
92  elif attr == "Name":
93  continue
94  if hasattr(obj, attr):
95  continue
96  elif isinstance(value, int):
97  obj.addProperty("App::PropertyInteger", attr, "IFC")
98  setattr(obj, attr, value)
99  elif isinstance(value, float):
100  obj.addProperty("App::PropertyFloat", attr, "IFC")
101  setattr(obj, attr, value)
102  elif isinstance(value, ifcopenshell.entity_instance):
103  value = create(value)
104  obj.addProperty("App::PropertyLink", attr, "IFC")
105  setattr(obj, attr, value)
106  elif isinstance(value, (list, tuple)) and value:
107  if isinstance(value[0], ifcopenshell.entity_instance):
108  nvalue = []
109  for elt in value:
110  nvalue.append(create(elt))
111  obj.addProperty("App::PropertyLinkList", attr, "IFC")
112  setattr(obj, attr, nvalue)
113  else:
114  obj.addProperty("App::PropertyString", attr, "IFC")
115  if value is not None:
116  setattr(obj, attr, str(value))
117  for parent in ifcfile.get_inverse(ifcentity):
118  create(parent)
119  if FreeCAD.GuiUp:
120  ViewProvider(obj.ViewObject)
121  return obj
