361 void PropertyNormalList::transformGeometry(const Base::Matrix4D& mat)
362 {
363  // A normal vector is only a direction with unit length, so we only need to rotate it
364  // (no translations or scaling)
365 
366  // Extract scale factors (assumes an orthogonal rotation matrix)
367  // Use the fact that the length of the row vectors of R are all equal to 1
368  // And that scaling is applied after rotating
369  double s[3];
370  s[0] = sqrt(mat[0][0] * mat[0][0] + mat[0][1] * mat[0][1] + mat[0][2] * mat[0][2]);
371  s[1] = sqrt(mat[1][0] * mat[1][0] + mat[1][1] * mat[1][1] + mat[1][2] * mat[1][2]);
372  s[2] = sqrt(mat[2][0] * mat[2][0] + mat[2][1] * mat[2][1] + mat[2][2] * mat[2][2]);
373 
374  // Set up the rotation matrix: zero the translations and make the scale factors = 1
375  Base::Matrix4D rot;
376  rot.setToUnity();
377  for (unsigned short i = 0; i < 3; i++) {
378  for (unsigned short j = 0; j < 3; j++) {
379  rot[i][j] = mat[i][j] / s[i];
380  }
381  }
382 
383  aboutToSetValue();
384 
385  // Rotate the normal vectors
386 #ifdef _MSC_VER
387  Concurrency::parallel_for_each(_lValueList.begin(),
388  _lValueList.end(),
389  [rot](Base::Vector3f& value) {
390  value = rot * value;
391  });
392 #else
393  QtConcurrent::blockingMap(_lValueList, [rot](Base::Vector3f& value) {
394  rot.multVec(value, value);
395  });
396 #endif
397 
398  hasSetValue();
399 }

453 std::vector<float> PropertyCurvatureList::getCurvature(int mode) const
454 {
455  const std::vector<Points::CurvatureInfo>& fCurvInfo = getValues();
456  std::vector<float> fValues;
457  fValues.reserve(fCurvInfo.size());
458 
459  // Mean curvature
460  if (mode == MeanCurvature) {
461  for (const auto& it : fCurvInfo) {
462  fValues.push_back(0.5f * (it.fMaxCurvature + it.fMinCurvature));
463  }
464  }
465  // Gaussian curvature
466  else if (mode == GaussCurvature) {
467  for (const auto& it : fCurvInfo) {
468  fValues.push_back(it.fMaxCurvature * it.fMinCurvature);
469  }
470  }
471  // Maximum curvature
472  else if (mode == MaxCurvature) {
473  for (const auto& it : fCurvInfo) {
474  fValues.push_back(it.fMaxCurvature);
475  }
476  }
477  // Minimum curvature
478  else if (mode == MinCurvature) {
479  for (const auto& it : fCurvInfo) {
480  fValues.push_back(it.fMinCurvature);
481  }
482  }
483  // Absolute curvature
484  else if (mode == AbsCurvature) {
485  for (const auto& it : fCurvInfo) {
486  if (fabs(it.fMaxCurvature) > fabs(it.fMinCurvature)) {
487  fValues.push_back(it.fMaxCurvature);
488  }
489  else {
490  fValues.push_back(it.fMinCurvature);
491  }
492  }
493  }
494 
495  return fValues;
496 }

498 void PropertyCurvatureList::transformGeometry(const Base::Matrix4D& mat)
499 {
500  // The principal direction is only a vector with unit length, so we only need to rotate it
501  // (no translations or scaling)
502

504  // Use the fact that the length of the row vectors of R are all equal to 1
505  // And that scaling is applied after rotating
506  double s[3];
507  s[0] = sqrt(mat[0][0] * mat[0][0] + mat[0][1] * mat[0][1] + mat[0][2] * mat[0][2]);
508  s[1] = sqrt(mat[1][0] * mat[1][0] + mat[1][1] * mat[1][1] + mat[1][2] * mat[1][2]);
509  s[2] = sqrt(mat[2][0] * mat[2][0] + mat[2][1] * mat[2][1] + mat[2][2] * mat[2][2]);
510 
511  // Set up the rotation matrix: zero the translations and make the scale factors = 1
512  Base::Matrix4D rot;
513  rot.setToUnity();
514  for (unsigned short i = 0; i < 3; i++) {
