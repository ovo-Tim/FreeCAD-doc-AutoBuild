251 std::vector<LineSet> DrawGeomHatch::getTrimmedLinesSection(DrawViewSection* source,
252  std::vector<LineSet> lineSets,
253  TopoDS_Face f,
254  double scale,
255  double hatchRotation,
256  Base::Vector3d hatchOffset)
257 {
258  std::vector<LineSet> result;
259  gp_Pln p;
260  Base::Vector3d vfc = DrawUtil::getFaceCenter(f);
261  gp_Pnt fc(vfc.x, vfc.y, vfc.z);
262  double dir = -1.0;
263  if (fc.Z() < 0.0) {
264  dir = -dir;
265  }
266  Base::Vector3d stdZ(0.0, 0.0, 1.0);
267  Base::Vector3d offset = stdZ * p.Distance(fc) * dir;
268 
269  //f may be above or below paper plane and must be moved so Common operation in
270  //getTrimmedLines succeeds
271  TopoDS_Shape moved = ShapeUtils::moveShape(f,
272  offset);
273  TopoDS_Face fMoved = TopoDS::Face(ShapeUtils::invertGeometry(moved));
274  return getTrimmedLines(
275  source,
276  lineSets,
277  fMoved,
278  scale,
279  hatchRotation,
280  hatchOffset
281  );
282 }

300 std::vector<LineSet> DrawGeomHatch::getTrimmedLines(DrawViewPart* source,
301  std::vector<LineSet> lineSets,
302  TopoDS_Face f,
303  double scale,
304  double hatchRotation,
305  Base::Vector3d hatchOffset)
306 {
307 // Base::Console().Message("DGH::getTrimmedLines() - rotation: %.3f hatchOffset: %s\n", hatchRotation, DrawUtil::formatVector(hatchOffset).c_str());
308  (void)source;
309  std::vector<LineSet> result;
310 
311  if (lineSets.empty()) {
312  return result;
313  }
314 
315  TopoDS_Face face = f;
316 
317  Bnd_Box bBox;
318  BRepBndLib::AddOptimal(face, bBox);
319  bBox.SetGap(0.0);
320 
321  for (auto& ls: lineSets) {
322  PATLineSpec hl = ls.getPATLineSpec();
323  std::vector<TopoDS_Edge> candidates = DrawGeomHatch::makeEdgeOverlay(hl, bBox, scale); //completely cover face bbox with lines
324 
325  //make Compound for this linespec
326  BRep_Builder builder;
327  TopoDS_Compound gridComp;
328  builder.MakeCompound(gridComp);
329  for (auto& c: candidates) {
330  builder.Add(gridComp, c);
331  }
332 
333  TopoDS_Shape grid = gridComp;
334  if (hatchRotation != 0.0) {
335  double hatchRotationRad = hatchRotation * M_PI / 180.0;
336  gp_Ax1 gridAxis(gp_Pnt(0.0, 0.0, 0.0), gp_Vec(gp::OZ().Direction()));
337  gp_Trsf xGridRotate;
338  xGridRotate.SetRotation(gridAxis, hatchRotationRad);
339  BRepBuilderAPI_Transform mkTransRotate(grid, xGridRotate, true);
340  grid = mkTransRotate.Shape();
341  }
342  gp_Trsf xGridTranslate;
343  xGridTranslate.SetTranslation(DrawUtil::togp_Vec(hatchOffset));
344  BRepBuilderAPI_Transform mkTransTranslate(grid, xGridTranslate, true);
345  grid = mkTransTranslate.Shape();
346 
347  //Common(Compound, Face)
348  FCBRepAlgoAPI_Common mkCommon(face, grid);
349  if (!mkCommon.IsDone() ||
350  mkCommon.Shape().IsNull()) {
351  return result;
352  }
353  TopoDS_Shape common = mkCommon.Shape();
354 
355  //save the boundingBox of hatch pattern
356  Bnd_Box overlayBox;
357  overlayBox.SetGap(0.0);
358  BRepBndLib::AddOptimal(common, overlayBox);
359  ls.setBBox(overlayBox);
360 
361  //get resulting edges
362  std::vector<TopoDS_Edge> resultEdges;
363  TopTools_IndexedMapOfShape mapOfEdges;
364  TopExp::MapShapes(common, TopAbs_EDGE, mapOfEdges);
365  for ( int i = 1 ; i <= mapOfEdges.Extent() ; i++ ) { //remember, TopExp makes no promises about the order it finds edges
366  const TopoDS_Edge& edge = TopoDS::Edge(mapOfEdges(i));
367  if (edge.IsNull()) {
368  continue;
369  }
370  resultEdges.push_back(edge);
371  }
372 
373  std::vector<TechDraw::BaseGeomPtr> resultGeoms;
374  for (auto& e: resultEdges) {
375  TechDraw::BaseGeomPtr base = BaseGeom::baseFactory(e);
376  if (!base) {
377  throw Base::ValueError("DGH::getTrimmedLines - baseFactory failed");
378  }
379  resultGeoms.push_back(base);
380  }
381  ls.setEdges(resultEdges);
382  ls.setGeoms(resultGeoms);
383  result.push_back(ls);
384  }
385  return result;
386 }

388 /* static */
389 std::vector<TopoDS_Edge> DrawGeomHatch::makeEdgeOverlay(PATLineSpec hatchLine, Bnd_Box bBox, double scale)
390 {
391  constexpr double RightAngleDegrees{90.0};
392  constexpr double HalfCircleDegrees{180.0};
393  std::vector<TopoDS_Edge> result;
394 
395  double minX, maxX, minY, maxY, minZ, maxZ;
396  bBox.Get(minX, minY, minZ, maxX, maxY, maxZ);
397  //make the overlay bigger to cover rotations. might need to be bigger than 2x.
398  double widthX = maxX - minX;
399  double widthY = maxY - minY;
400  double width = std::max(widthX, widthY);
401 
402  double centerX = (minX + maxX) / 2;
403  minX = centerX - width;
404  maxX = centerX + width;
405  double centerY = (minY + maxY) / 2;
406  minY = centerY - width;
407  maxY = centerY + width;
408 
409  Base::Vector3d origin = hatchLine.getOrigin();
410  double interval = hatchLine.getIntervalX() * scale;
411  double angle = hatchLine.getAngle();
412 
413  //only dealing with angles -180:180 for now
414  if (angle > RightAngleDegrees) {
415  angle = -(HalfCircleDegrees - angle);
416  } else if (angle < -RightAngleDegrees) {
417  angle = (HalfCircleDegrees + angle);
418  }
419  double slope = hatchLine.getSlope();
420 
421  if (angle == 0.0) { //odd case 1: horizontal lines
422  interval = hatchLine.getInterval() * scale;
423  double atomY = origin.y;
424  int repeatUp = (int) fabs((maxY - atomY)/interval);
425  int repeatDown = (int) fabs(((atomY - minY)/interval));
426  int repeatTotal = repeatUp + repeatDown + 1;
427  double yStart = atomY - repeatDown * interval;
428 
429  // make repeats
430  for (int i = 0; i < repeatTotal; i++) {
431  Base::Vector3d newStart(minX, yStart + float(i)*interval, 0);
432  Base::Vector3d newEnd(maxX, yStart + float(i)*interval, 0);
433  TopoDS_Edge newLine = makeLine(newStart, newEnd);
434  result.push_back(newLine);
435  }
436  } else if (angle == RightAngleDegrees ||
437  angle == -RightAngleDegrees) { //odd case 2: vertical lines
438  interval = hatchLine.getInterval() * scale;
439  double atomX = origin.x;
440  int repeatRight = (int) fabs((maxX - atomX)/interval);
441  int repeatLeft = (int) fabs((atomX - minX)/interval);
442  int repeatTotal = repeatRight + repeatLeft + 1;
443  double xStart = atomX - repeatLeft * interval;
444 
445  // make repeats
446  for (int i = 0; i < repeatTotal; i++) {
447  Base::Vector3d newStart(xStart + float(i)*interval, minY, 0);
448  Base::Vector3d newEnd(xStart + float(i)*interval, maxY, 0);
449  TopoDS_Edge newLine = makeLine(newStart, newEnd);
450  result.push_back(newLine);
451  }
452 //TODO: check if this makes 2-3 extra lines. might be some "left" lines on "right" side of vv
453  } else if (angle > 0) { //oblique (bottom left -> top right)
454  //ex: 60, 0,0, 0,4.0, 25, -25
455 // Base::Console().Message("TRACE - DGH-makeEdgeOverlay - making angle > 0\n");
456  double xLeftAtom = origin.x + (minY - origin.y)/slope; //the "atom" is the fill line that passes through the
457  //pattern-origin (not necc. R2 origin)
458  double xRightAtom = origin.x + (maxY - origin.y)/slope;
459  int repeatRight = (int) fabs((maxX - xLeftAtom)/interval);
460  int repeatLeft = (int) fabs((xRightAtom - minX)/interval);
461 
462  double leftStartX = xLeftAtom - (repeatLeft * interval);
463  double leftEndX = xRightAtom - (repeatLeft * interval);
464  int repeatTotal = repeatRight + repeatLeft + 1;
465 
466  //make repeats
467  for (int i = 0; i < repeatTotal; i++) {
468  Base::Vector3d newStart(leftStartX + (float(i) * interval), minY, 0);
469  Base::Vector3d newEnd (leftEndX + (float(i) * interval), maxY, 0);
470  TopoDS_Edge newLine = makeLine(newStart, newEnd);
471  result.push_back(newLine);
472  }
473  } else { //oblique (bottom right -> top left)
474  // ex: -60, 0,0, 0,4.0, 25.0, -12.5, 12.5, -6
475 // Base::Console().Message("TRACE - DGH-makeEdgeOverlay - making angle < 0\n");
476  double xRightAtom = origin.x + ((minY - origin.y)/slope); //x-coord of left end of Atom line
477  double xLeftAtom = origin.x + ((maxY - origin.y)/slope); //x-coord of right end of Atom line
478  int repeatRight = (int) fabs((maxX - xLeftAtom)/interval); //number of lines to Right of Atom
479  int repeatLeft = (int) fabs((xRightAtom - minX)/interval); //number of lines to Left of Atom
480  double leftEndX = xLeftAtom - (repeatLeft * interval);
481  double leftStartX = xRightAtom - (repeatLeft * interval);
482  int repeatTotal = repeatRight + repeatLeft + 1;
483 
484  // make repeats
485  for (int i = 0; i < repeatTotal; i++) {
486  Base::Vector3d newStart(leftStartX + float(i)*interval, minY, 0);
487  Base::Vector3d newEnd(leftEndX + float(i)*interval, maxY, 0);
488  TopoDS_Edge newLine = makeLine(newStart, newEnd);
489  result.push_back(newLine);
490  }
491  }
492 
493  return result;
494 }

508 std::vector<LineSet> DrawGeomHatch::getFaceOverlay(int iFace)
509 {
510 // Base::Console().Message("TRACE - DGH::getFaceOverlay(%d)\n", iFace);
511  std::vector<LineSet> result;
512  DrawViewPart* source = getSourceView();
513  if (!source ||
514  !source->hasGeometry()) {
515  return result;
516  }
517 
518  TopoDS_Face face = extractFace(source, iFace);
519 
520  Bnd_Box bBox;
521  BRepBndLib::AddOptimal(face, bBox);
522  bBox.SetGap(0.0);
523 
524  if (m_lineSets.empty()) {
525  makeLineSets();
526  }
527 
528  for (auto& ls: m_lineSets) {
529  PATLineSpec hl = ls.getPATLineSpec();
530  std::vector<TopoDS_Edge> candidates = DrawGeomHatch::makeEdgeOverlay(hl, bBox, ScalePattern.getValue());
531  std::vector<TechDraw::BaseGeomPtr> resultGeoms;
532  for (auto& e: candidates) {
533  TechDraw::BaseGeomPtr base = BaseGeom::baseFactory(e);
534  if (!base) {
535  throw Base::ValueError("DGH::getFaceOverlay - baseFactory failed");
536  }
537  resultGeoms.push_back(base);
538  }
539  ls.setEdges(candidates);
540  ls.setGeoms(resultGeoms);
541  result.push_back(ls);
542  }
543 
544  return result;
545 }

547 /* static */
550 TopoDS_Face DrawGeomHatch::extractFace(DrawViewPart* source, int iface )
551 {
552  std::vector<TopoDS_Wire> faceWires = source->getWireForFace(iface);
553 
554  //build face(s) from geometry
555  gp_Pnt gOrg(0.0, 0.0, 0.0);
556  gp_Dir gDir(0.0, 0.0, 1.0);
557  gp_Pln plane(gOrg, gDir);
558 
559  BRepBuilderAPI_MakeFace mkFace(plane, faceWires.front(), true);
560  std::vector<TopoDS_Wire>::iterator itWire = ++faceWires.begin(); //starting with second wire
561  for (; itWire != faceWires.end(); itWire++) {
562  mkFace.Add(*itWire);
563  }
564  if (!mkFace.IsDone()) {
565  return TopoDS_Face();
566  }
567  TopoDS_Face face = mkFace.Face();
568 
569  TopoDS_Shape temp;
570  try {
571  // mirror about the Y axis
572  gp_Trsf mirrorTransform;
573  mirrorTransform.SetMirror( gp_Ax2(gp_Pnt(0.0, 0.0, 0.0), gp_Dir(0, 1, 0)) );
574  BRepBuilderAPI_Transform mkTrf(face, mirrorTransform);
575  temp = mkTrf.Shape();
576  }
577  catch (...) {
578  return TopoDS_Face();
579  }
580  return TopoDS::Face(temp);
581 }
