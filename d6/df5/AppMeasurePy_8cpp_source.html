140  Py::Object getLocatedTopoShape(const Py::Tuple& args)
141  {
142  PyObject* pyRootObject {nullptr};
143  PyObject* pyLeafSubName {nullptr};
144  App::DocumentObject* rootObject {nullptr};
145  std::string leafSub;
146  if (!PyArg_ParseTuple(args.ptr(), "OO", &pyRootObject, &pyLeafSubName)) {
147  throw Py::TypeError("expected (rootObject, subname");
148  }
149 
150  if (PyObject_TypeCheck(pyRootObject, &(App::DocumentObjectPy::Type))) {
151  rootObject = static_cast<App::DocumentObjectPy*>(pyRootObject)->getDocumentObjectPtr();
152  }
153 
154  if (PyUnicode_Check(pyLeafSubName)) {
155  leafSub = PyUnicode_AsUTF8(pyLeafSubName);
156  }
157 
158  if (!rootObject) {
159  return Py::None();
160  }
161 
162  // this is on the stack
163  auto temp = ShapeFinder::getLocatedShape(*rootObject, leafSub);
164  // need new in here to make the twin object on the heap
165  auto topoShapePy = new Part::TopoShapePy(new Part::TopoShape(temp));
