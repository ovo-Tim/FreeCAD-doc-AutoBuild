57 void ViewProviderLine::attach(App::DocumentObject *obj)
58 {
59  ViewProviderDatum::attach(obj);
60 
61  // Setup label text and line colors
62  const char* name = pcObject->getNameInDocument();
63 
64  bool noRole = false;
65  auto axisRoles = App::LocalCoordinateSystem::AxisRoles;
66  if (strncmp(name, axisRoles[0], strlen(axisRoles[0])) == 0) {
67  // X-axis: red
68  ShapeAppearance.setDiffuseColor(ViewParams::instance()->getAxisXColor());
69  pLabel->string.setValue("X");
70  }
71  else if (strncmp(name, axisRoles[1], strlen(axisRoles[1])) == 0) {
72  // Y-axis: green
73  ShapeAppearance.setDiffuseColor(ViewParams::instance()->getAxisYColor());
74  pLabel->string.setValue("Y");
75  }
76  else if (strncmp(name, axisRoles[2], strlen(axisRoles[2])) == 0) {
77  // Z-axis: blue
78  ShapeAppearance.setDiffuseColor(ViewParams::instance()->getAxisZColor());
79  pLabel->string.setValue("Z");
80  }
81  else {
82  noRole = true;
83  }
84 
85  static const float size = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/View")->GetFloat("DatumLineSize", 70.0);
86 
87  auto line = getObject<App::Line>();
88  Base::Vector3d dir = line->getBaseDirection();
89  SbVec3f verts[2];
90  if (noRole) {
91  verts[0] = Base::convertTo<SbVec3f>(dir * 2 * size);
92  verts[1] = SbVec3f(0, 0, 0);
93  }
94  else {
95  verts[0] = Base::convertTo<SbVec3f>(dir * size);
96  verts[1] = Base::convertTo<SbVec3f>(dir * 0.2 * size);
97  }
98 
99  // indexes used to create the edges
100  static const int32_t lines[4] = { 0, 1, -1 };
101 
102  SoSeparator *sep = getDatumRoot();
103 
104  auto pCoords = new SoCoordinate3 ();
105  pCoords->point.setNum (2);
106  pCoords->point.setValues ( 0, 2, verts );
107  sep->addChild ( pCoords );
108 
109  auto pLines = new SoIndexedLineSet ();
110  pLines->coordIndex.setNum(3);
111  pLines->coordIndex.setValues(0, 3, lines);
112  sep->addChild ( pLines );
113 
114  auto textTranslation = new SoTranslation ();
115  textTranslation->translation.setValue(Base::convertTo<SbVec3f>(dir * 1.1 * size));
116  sep->addChild ( textTranslation );
117 
118  auto ps = new SoPickStyle();
119  ps->style.setValue(SoPickStyle::SHAPE_ON_TOP);
120  sep->addChild(ps);
121 
122  sep->addChild (pLabel);
123 }
