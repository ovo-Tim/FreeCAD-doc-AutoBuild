299 bool MeshSelection::deleteSelectionBorder()
300 {
301  // delete all selected faces
302  bool deletion = false;
303  std::list<ViewProviderMesh*> views = getViewProviders();
304  for (auto view : views) {
305  Mesh::Feature* mf = view->getObject<Mesh::Feature>();
306 
307  // mark the selected facet as visited
308  std::vector<Mesh::FacetIndex> selection;
309  std::vector<Mesh::FacetIndex> remove;
310  std::set<Mesh::PointIndex> borderPoints;
311  MeshCore::MeshAlgorithm meshAlg(mf->Mesh.getValue().getKernel());
312  meshAlg.GetFacetsFlag(selection, MeshCore::MeshFacet::SELECTED);
313  meshAlg.GetBorderPoints(selection, borderPoints);
314  std::vector<Mesh::PointIndex> border;
315  border.insert(border.begin(), borderPoints.begin(), borderPoints.end());
316 
317  meshAlg.ResetFacetFlag(MeshCore::MeshFacet::VISIT);
318  meshAlg.SetFacetsFlag(selection, MeshCore::MeshFacet::VISIT);
319  meshAlg.ResetPointFlag(MeshCore::MeshPoint::VISIT);
320  meshAlg.SetPointsFlag(border, MeshCore::MeshPoint::VISIT);
321 
322  // collect neighbour facets that are not selected and that share a border point
323  const MeshCore::MeshPointArray& points = mf->Mesh.getValue().getKernel().GetPoints();
324  const MeshCore::MeshFacetArray& faces = mf->Mesh.getValue().getKernel().GetFacets();
325  unsigned long numFaces = faces.size();
326  for (unsigned long i = 0; i < numFaces; i++) {
327  const MeshCore::MeshFacet& face = faces[i];
328  if (!face.IsFlag(MeshCore::MeshFacet::VISIT)) {
329  for (Mesh::PointIndex ptIndex : face._aulPoints) {
330  if (points[ptIndex].IsFlag(MeshCore::MeshPoint::VISIT)) {
331  remove.push_back(i);
332  break;
333  }
334  }
335  }
336  }
337 
338  if (!remove.empty()) {
339  deletion = true;
340  // remove duplicates
341  std::sort(remove.begin(), remove.end());
342  remove.erase(std::unique(remove.begin(), remove.end()), remove.end());
343 
344  view->setSelection(remove);
345  view->deleteSelection();
346  }
347  }
348 
349  return deletion;
350 }

460 void MeshSelection::selectGLCallback(void* ud, SoEventCallback* n)
461 {
462  // When this callback function is invoked we must leave the edit mode
463  Gui::View3DInventorViewer* view = static_cast<Gui::View3DInventorViewer*>(n->getUserData());
464  MeshSelection* self = static_cast<MeshSelection*>(ud);
465  self->stopInteractiveCallback(view);
466  n->setHandled();
467  std::vector<SbVec2f> polygon = view->getGLPolygon();
468  if (polygon.size() < 3) {
469  return;
470  }
471  if (polygon.front() != polygon.back()) {
472  polygon.push_back(polygon.front());
473  }
474 
475  SbVec3f pnt;
476  SbVec3f dir;
477  view->getNearPlane(pnt, dir);
478  Base::Vector3f normal(dir[0], dir[1], dir[2]);
479 
480  std::list<ViewProviderMesh*> views = self->getViewProviders();
481  for (auto vp : views) {
482  std::vector<Mesh::FacetIndex> faces;
483  const Mesh::MeshObject& mesh = vp->getObject<Mesh::Feature>()->Mesh.getValue();
484  const MeshCore::MeshKernel& kernel = mesh.getKernel();
485 
486  // simply get all triangles under the polygon
487  SoCamera* cam = view->getSoRenderManager()->getCamera();
488  SbViewVolume vv = cam->getViewVolume();
489  Gui::ViewVolumeProjection proj(vv);
490 
491  Base::Placement plm = vp->getObject<Mesh::Feature>()->Placement.getValue();
492  proj.setTransform(plm.toMatrix());
493  vp->getFacetsFromPolygon(polygon, proj, true, faces);
494 
495  if (self->onlyVisibleTriangles) {
496  const SbVec2s& sz = view->getSoRenderManager()->getViewportRegion().getWindowSize();
497  short width {};
498  short height {};
499  sz.getValue(width, height);
500  std::vector<SbVec2s> pixelPoly = view->getPolygon();
501  SbBox2s rect;
502  for (const auto& p : pixelPoly) {
503  rect.extendBy(SbVec2s(p[0], height - p[1]));
504  }
505  std::vector<Mesh::FacetIndex> rf;
506  rf.swap(faces);
507  std::vector<Mesh::FacetIndex> vf =
508  vp->getVisibleFacetsAfterZoom(rect,
509  view->getSoRenderManager()->getViewportRegion(),
510  view->getSoRenderManager()->getCamera());
511 
512  // get common facets of the viewport and the visible one
513  std::sort(vf.begin(), vf.end());
514  std::sort(rf.begin(), rf.end());
515  std::back_insert_iterator<std::vector<Mesh::FacetIndex>> biit(faces);
516  std::set_intersection(vf.begin(), vf.end(), rf.begin(), rf.end(), biit);
517  }
518 
519  // if set filter out all triangles which do not point into user direction
520  if (self->onlyPointToUserTriangles) {
521  std::vector<Mesh::FacetIndex> screen;
522  screen.reserve(faces.size());
523  MeshCore::MeshFacetIterator it_f(kernel);
524  for (Mesh::FacetIndex face : faces) {
525  it_f.Set(face);
526  if (it_f->GetNormal() * normal > 0.0F) {
527  screen.push_back(face);
528  }
529  }
530 
531  faces.swap(screen);
532  }
533 
534  if (self->addToSelection) {
535  vp->addSelection(faces);
536  }
537  else {
538  vp->removeSelection(faces);
539  }
540  }
541 
542  view->redraw();
543 }

545 void MeshSelection::pickFaceCallback(void* ud, SoEventCallback* n)
546 {
547  // handle only mouse button events
548  if (n->getEvent()->isOfType(SoMouseButtonEvent::getClassTypeId())) {
549  const SoMouseButtonEvent* mbe = static_cast<const SoMouseButtonEvent*>(n->getEvent());
550  Gui::View3DInventorViewer* view = static_cast<Gui::View3DInventorViewer*>(n->getUserData());
551 
552  // Mark all incoming mouse button events as handled, especially, to deactivate the selection
553  // node
554  n->getAction()->setHandled();
555  if (mbe->getButton() == SoMouseButtonEvent::BUTTON1
556  && mbe->getState() == SoButtonEvent::DOWN) {
557  const SoPickedPoint* point = n->getPickedPoint();
558  if (!point) {
559  Base::Console().Message("No facet picked.\n");
560  return;
561  }
562 
563  n->setHandled();
564 
565  // By specifying the indexed mesh node 'pcFaceSet' we make sure that the picked point is
566  // really from the mesh we render and not from any other geometry
567  Gui::ViewProvider* vp = view->getViewProviderByPathFromTail(point->getPath());
568  if (!vp || !vp->isDerivedFrom<ViewProviderMesh>()) {
569  return;
570  }
571  ViewProviderMesh* mesh = static_cast<ViewProviderMesh*>(vp);
572  MeshSelection* self = static_cast<MeshSelection*>(ud);
573  std::list<ViewProviderMesh*> views = self->getViewProviders();
574  if (std::find(views.begin(), views.end(), mesh) == views.end()) {
575  return;
576  }
577  const SoDetail* detail = point->getDetail(/*mesh->getShapeNode()*/);
578  if (detail && detail->getTypeId() == SoFaceDetail::getClassTypeId()) {
579  // get the boundary to the picked facet
580  Mesh::FacetIndex uFacet = static_cast<const SoFaceDetail*>(detail)->getFaceIndex();
581  if (self->addToSelection) {
582  if (self->addComponent) {
583  mesh->selectComponent(uFacet);
584  }
585  else {
586  mesh->selectFacet(uFacet);
587  }
588  }
589  else {
590  if (self->removeComponent) {
591  mesh->deselectComponent(uFacet);
592  }
593  else {
594  mesh->deselectFacet(uFacet);
595  }
596  }
597  }
598  }
599  }
