25 template <class Real>
26 bool HeightPlaneFit3 (int iQuantity, const Vector3<Real>* akPoint, Real& rfA,
27  Real& rfB, Real& rfC)
28 {
29  // You need at least three points to determine the plane. Even so, if
30  // the points are on a vertical plane, there is no least-squares fit in
31  // the 'height' sense. This will be trapped by the determinant of the
32  // coefficient matrix.
33 
34  // compute sums for linear system
35  Real fSumX = (Real)0.0, fSumY = (Real)0.0, fSumZ = (Real)0.0;
36  Real fSumXX = (Real)0.0, fSumXY = (Real)0.0, fSumXZ = (Real)0.0;
37  Real fSumYY = (Real)0.0, fSumYZ = (Real)0.0;
38  int i;
39  for (i = 0; i < iQuantity; i++)
40  {
41  fSumX += akPoint[i].X();
42  fSumY += akPoint[i].Y();
43  fSumZ += akPoint[i].Z();
44  fSumXX += akPoint[i].X()*akPoint[i].X();
45  fSumXY += akPoint[i].X()*akPoint[i].Y();
46  fSumXZ += akPoint[i].X()*akPoint[i].Z();
47  fSumYY += akPoint[i].Y()*akPoint[i].Y();
48  fSumYZ += akPoint[i].Y()*akPoint[i].Z();
49  }
50 
51  Real aafA[3][3] =
52  {
53  {fSumXX, fSumXY, fSumX},
54  {fSumXY, fSumYY, fSumY},
55  {fSumX, fSumY, (Real)iQuantity}
56  };
57 
58  Real afB[3] =
59  {
60  fSumXZ,
61  fSumYZ,
62  fSumZ
63  };
64 
65  Real afX[3];
66 
67  bool bNonsingular = LinearSystem<Real>().Solve3(aafA,afB,afX);
68  if (bNonsingular)
69  {
70  rfA = afX[0];
71  rfB = afX[1];
72  rfC = afX[2];
73  }
74  else
75  {
76  rfA = Math<Real>::MAX_REAL;
77  rfB = Math<Real>::MAX_REAL;
78  rfC = Math<Real>::MAX_REAL;
79  }
80 
81  return bNonsingular;
82 }

84 template <class Real>
85 Plane3<Real> OrthogonalPlaneFit3 (int iQuantity, const Vector3<Real>* akPoint)
86 {
87  // compute the mean of the points
88  Vector3<Real> kOrigin = Vector3<Real>::ZERO;
89  int i;
90  for (i = 0; i < iQuantity; i++)
91  {
92  kOrigin += akPoint[i];
93  }
94  Real fInvQuantity = ((Real)1.0)/iQuantity;
95  kOrigin *= fInvQuantity;
96 
97  // compute sums of products
98  Real fSumXX = (Real)0.0, fSumXY = (Real)0.0, fSumXZ = (Real)0.0;
99  Real fSumYY = (Real)0.0, fSumYZ = (Real)0.0, fSumZZ = (Real)0.0;
100  for (i = 0; i < iQuantity; i++)
101  {
102  Vector3<Real> kDiff = akPoint[i] - kOrigin;
103  fSumXX += kDiff.X()*kDiff.X();
104  fSumXY += kDiff.X()*kDiff.Y();
105  fSumXZ += kDiff.X()*kDiff.Z();
106  fSumYY += kDiff.Y()*kDiff.Y();
107  fSumYZ += kDiff.Y()*kDiff.Z();
108  fSumZZ += kDiff.Z()*kDiff.Z();
109  }
110 
111  fSumXX *= fInvQuantity;
112  fSumXY *= fInvQuantity;
113  fSumXZ *= fInvQuantity;
114  fSumYY *= fInvQuantity;
115  fSumYZ *= fInvQuantity;
116  fSumZZ *= fInvQuantity;
117 
118  // setup the eigensolver
119  Eigen<Real> kES(3);
120  kES(0,0) = fSumXX;
121  kES(0,1) = fSumXY;
122  kES(0,2) = fSumXZ;
123  kES(1,0) = fSumXY;
124  kES(1,1) = fSumYY;
125  kES(1,2) = fSumYZ;
126  kES(2,0) = fSumXZ;
127  kES(2,1) = fSumYZ;
128  kES(2,2) = fSumZZ;
129 
130  // compute eigenstuff, smallest eigenvalue is in last position
131  kES.DecrSortEigenStuff3();
132 
133  // get plane normal
134  Vector3<Real> kNormal;
135  kES.GetEigenvector(2,kNormal);
136 
137  // the minimum energy
138  return Plane3<Real>(kNormal,kOrigin);
139 }
