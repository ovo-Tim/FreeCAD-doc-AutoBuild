107 void ViewProviderFilling::highlightReferences(ShapeType type, const References& refs, bool on)
108 {
109  for (const auto& it : refs) {
110  Part::Feature* base = dynamic_cast<Part::Feature*>(it.first);
111  if (base) {
112  PartGui::ViewProviderPartExt* svp = dynamic_cast<PartGui::ViewProviderPartExt*>(
113  Gui::Application::Instance->getViewProvider(base));
114  if (svp) {
115  switch (type) {
116  case ViewProviderFilling::Vertex:
117  if (on) {
118  std::vector<App::Color> colors;
119  TopTools_IndexedMapOfShape vMap;
120  TopExp::MapShapes(base->Shape.getValue(), TopAbs_VERTEX, vMap);
121  colors.resize(vMap.Extent(), svp->PointColor.getValue());
122 
123  for (const auto& jt : it.second) {
124  // check again that the index is in range because it's possible that
125  // the sub-names are invalid
126  std::size_t idx =
127  static_cast<std::size_t>(std::stoi(jt.substr(6)) - 1);
128  if (idx < colors.size()) {
129  colors[idx] = App::Color(1.0, 0.0, 1.0); // magenta
130  }
131  }
132 
133  svp->setHighlightedPoints(colors);
134  }
135  else {
136  svp->unsetHighlightedPoints();
137  }
138  break;
139  case ViewProviderFilling::Edge:
140  if (on) {
141  std::vector<App::Color> colors;
142  TopTools_IndexedMapOfShape eMap;
143  TopExp::MapShapes(base->Shape.getValue(), TopAbs_EDGE, eMap);
144  colors.resize(eMap.Extent(), svp->LineColor.getValue());
145 
146  for (const auto& jt : it.second) {
147  std::size_t idx =
148  static_cast<std::size_t>(std::stoi(jt.substr(4)) - 1);
149  // check again that the index is in range because it's possible that
150  // the sub-names are invalid
151  if (idx < colors.size()) {
152  colors[idx] = App::Color(1.0, 0.0, 1.0); // magenta
153  }
154  }
155 
156  svp->setHighlightedEdges(colors);
157  }
158  else {
159  svp->unsetHighlightedEdges();
160  }
161  break;
162  case ViewProviderFilling::Face:
163  if (on) {
164  std::vector<App::Material> materials;
165  TopTools_IndexedMapOfShape fMap;
166  TopExp::MapShapes(base->Shape.getValue(), TopAbs_FACE, fMap);
167  materials.resize(fMap.Extent(), svp->ShapeAppearance[0]);
168 
169  for (const auto& jt : it.second) {
170  std::size_t idx =
171  static_cast<std::size_t>(std::stoi(jt.substr(4)) - 1);
172  // check again that the index is in range because it's possible that
173  // the sub-names are invalid
174  if (idx < materials.size()) {
175  materials[idx].diffuseColor =
176  App::Color(1.0, 0.0, 1.0); // magenta
177  }
178  }
179 
180  svp->setHighlightedFaces(materials);
181  }
182  else {
183  svp->unsetHighlightedFaces();
184  }
185  break;
186  }
187  }
188  }
189  }
190 }

194 class FillingPanel::ShapeSelection: public Gui::SelectionFilterGate
195 {
196 public:
197  ShapeSelection(FillingPanel::SelectionMode& mode, Surface::Filling* editedObject)
198  : Gui::SelectionFilterGate(nullPointer())
199  , mode(mode)
200  , editedObject(editedObject)
201  {}
202  ~ShapeSelection() override
203  {
204  mode = FillingPanel::None;
205  }
209  bool allow(App::Document*, App::DocumentObject* pObj, const char* sSubName) override
210  {
211  // don't allow references to itself
212  if (pObj == editedObject) {
213  return false;
214  }
215  if (!pObj->isDerivedFrom(Part::Feature::getClassTypeId())) {
216  return false;
217  }
218 
219  if (!sSubName || sSubName[0] == '\0') {
220  return false;
221  }
222 
223  switch (mode) {
224  case FillingPanel::InitFace:
225  return allowFace(pObj, sSubName);
226  case FillingPanel::AppendEdge:
227  return allowEdge(true, pObj, sSubName);
228  case FillingPanel::RemoveEdge:
229  return allowEdge(false, pObj, sSubName);
230  default:
231  return false;
232  }
233  }
234 
235 private:
236  bool allowFace(App::DocumentObject*, const char* sSubName)
237  {
238  std::string element(sSubName);
239  if (element.substr(0, 4) != "Face") {
240  return false;
241  }
242  return true;
243  }
244  bool allowEdge(bool appendEdges, App::DocumentObject* pObj, const char* sSubName)
245  {
246  std::string element(sSubName);
247  if (element.substr(0, 4) != "Edge") {
248  return false;
249  }
250 
251  auto links = editedObject->BoundaryEdges.getSubListValues();
252  for (const auto& it : links) {
253  if (it.first == pObj) {
254  for (const auto& jt : it.second) {
255  if (jt == sSubName) {
256  return !appendEdges;
257  }
258  }
259  }
260  }
261 
262  return appendEdges;
263  }
264 
265 private:
266  FillingPanel::SelectionMode& mode;
267  Surface::Filling* editedObject;
268 };

335 // stores object pointer, its old fill type and adjusts radio buttons according to it.
336 void FillingPanel::setEditedObject(Surface::Filling* fea)
337 {
338  editedObject = fea;
339 
340  // get the link to the initial surface if set
341  App::DocumentObject* initFace = editedObject->InitialFace.getValue();
342  const std::vector<std::string>& subList = editedObject->InitialFace.getSubValues();
343  if (initFace && subList.size() == 1) {
344  QString text =
345  QString::fromLatin1("%1.%2").arg(QString::fromUtf8(initFace->Label.getValue()),
346  QString::fromStdString(subList.front()));
347  ui->lineInitFaceName->setText(text);
348  }
349 
350  // get the boundary edges, if set their adjacent faces and continuities
351  auto objects = editedObject->BoundaryEdges.getValues();
352  auto edges = editedObject->BoundaryEdges.getSubValues();
353  auto count = objects.size();
354 
355  // fill up faces if wrong size
356  auto faces = editedObject->BoundaryFaces.getValues();
357  if (faces.size() != edges.size()) {
358  faces.resize(edges.size());
359  std::fill(faces.begin(), faces.end(), std::string());
360  }
361 
362  // fill up continuities if wrong size
363  auto conts = editedObject->BoundaryOrder.getValues();
364  if (edges.size() != conts.size()) {
365  conts.resize(edges.size());
366  std::fill(conts.begin(), conts.end(), static_cast<long>(GeomAbs_C0));
367  }
368 
369  App::Document* doc = editedObject->getDocument();
370  for (std::size_t i = 0; i < count; i++) {
371  App::DocumentObject* obj = objects[i];
372  std::string edge = edges[i];
373  std::string face = faces[i];
374 
375  QListWidgetItem* item = new QListWidgetItem(ui->listBoundary);
376  ui->listBoundary->addItem(item);
377 
378  QString text = QString::fromLatin1("%1.%2").arg(QString::fromUtf8(obj->Label.getValue()),
379  QString::fromStdString(edge));
380  item->setText(text);
381 
382  // The user data field of a list widget item
383  // is a list of five elementa:
384  // 1. document name
385  // 2. object name
386  // 3. sub-element name of the edge
387  // 4. sub-element of an adjacent face or empty string
388  // 5. the continuity as int
389  QList<QVariant> data;
390  data << QByteArray(doc->getName());
391  data << QByteArray(obj->getNameInDocument());
392  data << QByteArray(edge.c_str());
393  data << QByteArray(face.c_str());
394  data << static_cast<int>(conts[i]);
395  item->setData(Qt::UserRole, data);
396  }
397 
398  // attach this document observer
399  attachDocument(Gui::Application::Instance->getDocument(doc));
400 }

572 void FillingPanel::onListBoundaryItemDoubleClicked(QListWidgetItem* item)
573 {
574  Gui::Selection().clearSelection();
575  Gui::Selection().rmvSelectionGate();
576  selectionMode = None;
577 
578  ui->comboBoxFaces->clear();
579  ui->comboBoxCont->clear();
580 
581  if (item) {
582  QList<QVariant> data;
583  data = item->data(Qt::UserRole).toList();
584 
585  try {
586  App::Document* doc = App::GetApplication().getDocument(data[0].toByteArray());
587  App::DocumentObject* obj = doc ? doc->getObject(data[1].toByteArray()) : nullptr;
588  if (obj && obj->isDerivedFrom<Part::Feature>()) {
589  const Part::TopoShape& shape = static_cast<Part::Feature*>(obj)->Shape.getShape();
590  TopoDS_Shape edge = shape.getSubShape(data[2].toByteArray());
591 
592  // build up map edge->face
593  TopTools_IndexedMapOfShape faces;
594  TopExp::MapShapes(shape.getShape(), TopAbs_FACE, faces);
595  TopTools_IndexedDataMapOfShapeListOfShape edge2Face;
596  TopExp::MapShapesAndAncestors(shape.getShape(),
597  TopAbs_EDGE,
598  TopAbs_FACE,
599  edge2Face);
600  const TopTools_ListOfShape& adj_faces = edge2Face.FindFromKey(edge);
601  if (adj_faces.Extent() > 0) {
602  int n = adj_faces.Extent();
603  ui->statusLabel->setText(tr("Edge has %n adjacent faces", nullptr, n));
604 
605  // fill up the combo boxes
606  modifyBoundary(true);
607  ui->comboBoxFaces->addItem(tr("None"), QByteArray(""));
608  ui->comboBoxCont->addItem(QString::fromLatin1("C0"),
609  static_cast<int>(GeomAbs_C0));
610  ui->comboBoxCont->addItem(QString::fromLatin1("G1"),
611  static_cast<int>(GeomAbs_G1));
612  ui->comboBoxCont->addItem(QString::fromLatin1("G2"),
613  static_cast<int>(GeomAbs_G2));
614  TopTools_ListIteratorOfListOfShape it(adj_faces);
615  for (; it.More(); it.Next()) {
616  const TopoDS_Shape& F = it.Value();
617  int index = faces.FindIndex(F);
618  QString text = QString::fromLatin1("Face%1").arg(index);
619  ui->comboBoxFaces->addItem(text, text.toLatin1());
620  }
621 
622  // activate face and continuity
623  if (data.size() == 5) {
624  int index = ui->comboBoxFaces->findData(data[3]);
625  ui->comboBoxFaces->setCurrentIndex(index);
626  index = ui->comboBoxCont->findData(data[4]);
627  ui->comboBoxCont->setCurrentIndex(index);
628  }
629  }
630  else {
631  ui->statusLabel->setText(tr("Edge has no adjacent faces"));
632  }
633  }
634 
635  Gui::Selection().addSelection(data[0].toByteArray(),
636  data[1].toByteArray(),
637  data[2].toByteArray());
638  }
639  catch (...) {
640  }
641  }
642 }

649

672  QListWidgetItem* item = new QListWidgetItem(ui->listBoundary);
673  ui->listBoundary->addItem(item);
674 
675  Gui::SelectionObject sel(msg);
676  QString text = QString::fromLatin1("%1.%2").arg(
677  QString::fromUtf8(sel.getObject()->Label.getValue()),
678  QString::fromLatin1(msg.pSubName));
679  item->setText(text);
680 
681  QList<QVariant> data;
682  data << QByteArray(msg.pDocName);
683  data << QByteArray(msg.pObjectName);
684  data << QByteArray(msg.pSubName);
685  data << QByteArray("");
686  data << static_cast<int>(GeomAbs_C0);
687  item->setData(Qt::UserRole, data);
688
