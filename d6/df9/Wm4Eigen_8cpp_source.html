187 template <class Real>
188 void Eigen<Real>::Tridiagonal3 ()
189 {
190  Real fM00 = m_kMat[0][0];
191  Real fM01 = m_kMat[0][1];
192  Real fM02 = m_kMat[0][2];
193  Real fM11 = m_kMat[1][1];
194  Real fM12 = m_kMat[1][2];
195  Real fM22 = m_kMat[2][2];
196 
197  m_afDiag[0] = fM00;
198  m_afSubd[2] = (Real)0.0;
199  if (Math<Real>::FAbs(fM02) > Math<Real>::ZERO_TOLERANCE)
200  {
201  Real fLength = Math<Real>::Sqrt(fM01*fM01+fM02*fM02);
202  Real fInvLength = ((Real)1.0)/fLength;
203  fM01 *= fInvLength;
204  fM02 *= fInvLength;
205  Real fQ = ((Real)2.0)*fM01*fM12+fM02*(fM22-fM11);
206  m_afDiag[1] = fM11+fM02*fQ;
207  m_afDiag[2] = fM22-fM02*fQ;
208  m_afSubd[0] = fLength;
209  m_afSubd[1] = fM12-fM01*fQ;
210  m_kMat[0][0] = (Real)1.0;
211  m_kMat[0][1] = (Real)0.0;
212  m_kMat[0][2] = (Real)0.0;
213  m_kMat[1][0] = (Real)0.0;
214  m_kMat[1][1] = fM01;
215  m_kMat[1][2] = fM02;
216  m_kMat[2][0] = (Real)0.0;
217  m_kMat[2][1] = fM02;
218  m_kMat[2][2] = -fM01;
219  m_bIsRotation = false;
220  }
221  else
222  {
223  m_afDiag[1] = fM11;
224  m_afDiag[2] = fM22;
225  m_afSubd[0] = fM01;
226  m_afSubd[1] = fM12;
227  m_kMat[0][0] = (Real)1.0;
228  m_kMat[0][1] = (Real)0.0;
229  m_kMat[0][2] = (Real)0.0;
230  m_kMat[1][0] = (Real)0.0;
231  m_kMat[1][1] = (Real)1.0;
232  m_kMat[1][2] = (Real)0.0;
233  m_kMat[2][0] = (Real)0.0;
234  m_kMat[2][1] = (Real)0.0;
235  m_kMat[2][2] = (Real)1.0;
236  m_bIsRotation = true;
237  }
238 }

240 template <class Real>
241 void Eigen<Real>::TridiagonalN ()
242 {
243  int i0, i1, i2, i3;
244 
245  for (i0 = m_iSize-1, i3 = m_iSize-2; i0 >= 1; i0--, i3--)
246  {
247  Real fH = (Real)0.0, fScale = (Real)0.0;
248 
249  if (i3 > 0)
250  {
251  for (i2 = 0; i2 <= i3; i2++)
252  {
253  fScale += Math<Real>::FAbs(m_kMat[i0][i2]);
254  }
255  if (fScale == (Real)0.0)
256  {
257  m_afSubd[i0] = m_kMat[i0][i3];
258  }
259  else
260  {
261  Real fInvScale = ((Real)1.0)/fScale;
262  for (i2 = 0; i2 <= i3; i2++)
263  {
264  m_kMat[i0][i2] *= fInvScale;
265  fH += m_kMat[i0][i2]*m_kMat[i0][i2];
266  }
267  Real fF = m_kMat[i0][i3];
268  Real fG = Math<Real>::Sqrt(fH);
269  if (fF > (Real)0.0)
270  {
271  fG = -fG;
272  }
273  m_afSubd[i0] = fScale*fG;
274  fH -= fF*fG;
275  m_kMat[i0][i3] = fF-fG;
276  fF = (Real)0.0;
277  Real fInvH = ((Real)1.0)/fH;
278  for (i1 = 0; i1 <= i3; i1++)
279  {
280  m_kMat[i1][i0] = m_kMat[i0][i1]*fInvH;
281  fG = (Real)0.0;
282  for (i2 = 0; i2 <= i1; i2++)
283  {
284  fG += m_kMat[i1][i2]*m_kMat[i0][i2];
285  }
286  for (i2 = i1+1; i2 <= i3; i2++)
287  {
288  fG += m_kMat[i2][i1]*m_kMat[i0][i2];
289  }
290  m_afSubd[i1] = fG*fInvH;
291  fF += m_afSubd[i1]*m_kMat[i0][i1];
292  }
293  Real fHalfFdivH = ((Real)0.5)*fF*fInvH;
294  for (i1 = 0; i1 <= i3; i1++)
295  {
296  fF = m_kMat[i0][i1];
297  fG = m_afSubd[i1] - fHalfFdivH*fF;
298  m_afSubd[i1] = fG;
299  for (i2 = 0; i2 <= i1; i2++)
300  {
301  m_kMat[i1][i2] -= fF*m_afSubd[i2] +
302  fG*m_kMat[i0][i2];
303  }
304  }
305  }
306  }
307  else
308  {
309  m_afSubd[i0] = m_kMat[i0][i3];
310  }
311 
312  m_afDiag[i0] = fH;
313  }
314 
315  m_afDiag[0] = (Real)0.0;
316  m_afSubd[0] = (Real)0.0;
317  for (i0 = 0, i3 = -1; i0 <= m_iSize-1; i0++, i3++)
318  {
319  if (m_afDiag[i0] != (Real)0.0)
320  {
321  for (i1 = 0; i1 <= i3; i1++)
322  {
323  Real fSum = (Real)0.0;
324  for (i2 = 0; i2 <= i3; i2++)
325  {
326  fSum += m_kMat[i0][i2]*m_kMat[i2][i1];
327  }
328  for (i2 = 0; i2 <= i3; i2++)
329  {
330  m_kMat[i2][i1] -= fSum*m_kMat[i2][i0];
331  }
332  }
333  }
334  m_afDiag[i0] = m_kMat[i0][i0];
335  m_kMat[i0][i0] = (Real)1.0;
336  for (i1 = 0; i1 <= i3; i1++)
337  {
338  m_kMat[i1][i0] = (Real)0.0;
339  m_kMat[i0][i1] = (Real)0.0;
340  }
341  }
342 
343  // re-ordering if Eigen::QLAlgorithm is used subsequently
344  for (i0 = 1, i3 = 0; i0 < m_iSize; i0++, i3++)
345  {
346  m_afSubd[i3] = m_afSubd[i0];
347  }
348  m_afSubd[m_iSize-1] = (Real)0.0;
349 
350  m_bIsRotation = ((m_iSize % 2) == 0);
351 }

353 template <class Real>
354 bool Eigen<Real>::QLAlgorithm ()
355 {
356  const int iMaxIter = 32;
357 
358  for (int i0 = 0; i0 < m_iSize; i0++)
359  {
360  int i1;
361  for (i1 = 0; i1 < iMaxIter; i1++)
362  {
363  int i2;
364  for (i2 = i0; i2 <= m_iSize-2; i2++)
365  {
366  Real fTmp = Math<Real>::FAbs(m_afDiag[i2]) +
367  Math<Real>::FAbs(m_afDiag[i2+1]);
368 
369  if (Math<Real>::FAbs(m_afSubd[i2]) + fTmp == fTmp)
370  {
371  break;
372  }
373  }
374  if (i2 == i0)
375  {
376  break;
377  }
378 
379  Real fG = (m_afDiag[i0+1] - m_afDiag[i0])/(((Real)2.0) *
380  m_afSubd[i0]);
381  Real fR = Math<Real>::Sqrt(fG*fG+(Real)1.0);
382  if ( fG < (Real)0.0 )
383  {
384  fG = m_afDiag[i2]-m_afDiag[i0]+m_afSubd[i0]/(fG-fR);
385  }
386  else
387  {
388  fG = m_afDiag[i2]-m_afDiag[i0]+m_afSubd[i0]/(fG+fR);
389  }
390  Real fSin = (Real)1.0, fCos = (Real)1.0, fP = (Real)0.0;
391  for (int i3 = i2-1; i3 >= i0; i3--)
392  {
393  Real fF = fSin*m_afSubd[i3];
394  Real fB = fCos*m_afSubd[i3];
395  if (Math<Real>::FAbs(fF) >= Math<Real>::FAbs(fG))
396  {
397  fCos = fG/fF;
398  fR = Math<Real>::Sqrt(fCos*fCos+(Real)1.0);
399  m_afSubd[i3+1] = fF*fR;
400  fSin = ((Real)1.0)/fR;
401  fCos *= fSin;
402  }
403  else
404  {
405  fSin = fF/fG;
406  fR = Math<Real>::Sqrt(fSin*fSin+(Real)1.0);
407  m_afSubd[i3+1] = fG*fR;
408  fCos = ((Real)1.0)/fR;
409  fSin *= fCos;
410  }
411  fG = m_afDiag[i3+1]-fP;
412  fR = (m_afDiag[i3]-fG)*fSin+((Real)2.0)*fB*fCos;
413  fP = fSin*fR;
414  m_afDiag[i3+1] = fG+fP;
415  fG = fCos*fR-fB;
416 
417  for (int i4 = 0; i4 < m_iSize; i4++)
418  {
419  fF = m_kMat[i4][i3+1];
420  m_kMat[i4][i3+1] = fSin*m_kMat[i4][i3]+fCos*fF;
421  m_kMat[i4][i3] = fCos*m_kMat[i4][i3]-fSin*fF;
422  }
423  }
424  m_afDiag[i0] -= fP;
425  m_afSubd[i0] = fG;
426  m_afSubd[i2] = (Real)0.0;
427  }
428  if (i1 == iMaxIter)
429  {
430  return false;
431  }
432  }
433 
434  return true;
435 }

437 template <class Real>
438 void Eigen<Real>::DecreasingSort ()
439 {
440  // sort eigenvalues in decreasing order, e[0] >= ... >= e[iSize-1]
441  for (int i0 = 0, i1; i0 <= m_iSize-2; i0++)
442  {
443  // locate maximum eigenvalue
444  i1 = i0;
445  Real fMax = m_afDiag[i1];
446  int i2;
447  for (i2 = i0+1; i2 < m_iSize; i2++)
448  {
449  if (m_afDiag[i2] > fMax)
450  {
451  i1 = i2;
452  fMax = m_afDiag[i1];
453  }
454  }
455 
456  if (i1 != i0)
457  {
458  // swap eigenvalues
459  m_afDiag[i1] = m_afDiag[i0];
460  m_afDiag[i0] = fMax;
461 
462  // swap eigenvectors
463  for (i2 = 0; i2 < m_iSize; i2++)
464  {
465  Real fTmp = m_kMat[i2][i0];
466  m_kMat[i2][i0] = m_kMat[i2][i1];
467  m_kMat[i2][i1] = fTmp;
468  m_bIsRotation = !m_bIsRotation;
469  }
470  }
471  }
472 }
