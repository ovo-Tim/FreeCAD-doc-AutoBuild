169  def generate_macro_file(self, replacements: Dict[str, str] = {}) -> os.PathLike:
170  with open(os.path.join(self.test_dir, "macro_template.FCStd")) as f:
171  lines = f.readlines()
172  outfile = tempfile.NamedTemporaryFile(mode="wt", delete=False)
173  for line in lines:
174  for key, value in replacements.items():
175  line = line.replace(key, value)
176 
177  outfile.write(line)
178  outfile.close()
179  return outfile.name
