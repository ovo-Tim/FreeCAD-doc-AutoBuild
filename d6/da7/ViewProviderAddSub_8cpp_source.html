103 void ViewProviderAddSub::updateAddSubShapeIndicator() {
104 
105  TopoDS_Shape cShape(getObject<PartDesign::FeatureAddSub>()->AddSubShape.getValue());
106  if (cShape.IsNull()) {
107  previewCoords ->point .setNum(0);
108  previewNorm ->vector .setNum(0);
109  previewFaceSet ->coordIndex .setNum(0);
110  previewFaceSet ->partIndex .setNum(0);
111  return;
112  }
113 
114  try {
115  // calculating the deflection value
116  Bnd_Box bounds;
117  BRepBndLib::Add(cShape, bounds);
118  bounds.SetGap(0.0);
119  Standard_Real xMin, yMin, zMin, xMax, yMax, zMax;
120  bounds.Get(xMin, yMin, zMin, xMax, yMax, zMax);
121  Standard_Real deflection = ((xMax-xMin)+(yMax-yMin)+(zMax-zMin))/300.0 * Deviation.getValue();
122 
123  // create or use the mesh on the data structure
124  Standard_Real AngDeflectionRads = AngularDeflection.getValue() / 180.0 * M_PI;
125  BRepMesh_IncrementalMesh(cShape, deflection, Standard_False, AngDeflectionRads, Standard_True);
126 
127  // We must reset the location here because the transformation data
128  // are set in the placement property
129  TopLoc_Location aLoc;
130  cShape.Location(aLoc);
131 
132  // count triangles and nodes in the mesh
133  int numTriangles=0,numNodes=0,numNorms=0,numFaces=0;
134  TopExp_Explorer Ex;
135  for (Ex.Init(cShape,TopAbs_FACE);Ex.More();Ex.Next()) {
136  Handle (Poly_Triangulation) mesh = BRep_Tool::Triangulation(TopoDS::Face(Ex.Current()), aLoc);
137  // Note: we must also count empty faces
138  if (!mesh.IsNull()) {
139  numTriangles += mesh->NbTriangles();
140  numNodes += mesh->NbNodes();
141  numNorms += mesh->NbNodes();
142  }
143  numFaces++;
144  }
145 
146  // create memory for the nodes and indexes
147  previewCoords ->point .setNum(numNodes);
148  previewNorm ->vector .setNum(numNorms);
149  previewFaceSet ->coordIndex .setNum(numTriangles*4);
150  previewFaceSet ->partIndex .setNum(numFaces);
151 
152  // get the raw memory for fast fill up
153  SbVec3f* verts = previewCoords ->point .startEditing();
154  SbVec3f* previewNorms = previewNorm ->vector .startEditing();
155  int32_t* index = previewFaceSet ->coordIndex .startEditing();
156  int32_t* parts = previewFaceSet ->partIndex .startEditing();
157 
158  // preset the previewNormal vector with null vector
159  for (int i=0;i < numNorms;i++)
160  previewNorms[i]= SbVec3f(0.0,0.0,0.0);
161 
162  int ii = 0,faceNodeOffset=0,faceTriaOffset=0;
163  for (Ex.Init(cShape, TopAbs_FACE); Ex.More(); Ex.Next(),ii++) {
164  const TopoDS_Face &actFace = TopoDS::Face(Ex.Current());
165 
166  TopLoc_Location loc;
167  Handle(Poly_Triangulation) mesh = BRep_Tool::Triangulation(actFace, loc);
168  if (mesh.IsNull())
169  continue;
170 
171  // get triangulation
172  std::vector<gp_Pnt> points;
173  std::vector<Poly_Triangle> facets;
174  Part::Tools::getTriangulation(actFace, points, facets);
175 
176  // get normal per vertex
177  std::vector<gp_Vec> vertexnormals;
178  Part::Tools::getPointNormals(actFace, mesh, vertexnormals);
179  Part::Tools::applyTransformationOnNormals(loc, vertexnormals);
180 
181  // getting size of node and triangle array of this face
182  std::size_t nbNodesInFace = points.size();
183  std::size_t nbTriInFace = facets.size();
184 
185  for (std::size_t i = 0; i < points.size(); i++) {
186  verts[faceNodeOffset+i] = SbVec3f(points[i].X(), points[i].Y(), points[i].Z());
187  }
188 
189  for (std::size_t i = 0; i < vertexnormals.size(); i++) {
190  previewNorms[faceNodeOffset+i] = SbVec3f(vertexnormals[i].X(), vertexnormals[i].Y(), vertexnormals[i].Z());
191  }
192 
193  // cycling through the poly mesh
194  for (std::size_t g=0; g < nbTriInFace; g++) {
195  // Get the triangle
196  Standard_Integer N1,N2,N3;
197  facets[g].Get(N1,N2,N3);
198 
199  // set the index vector with the 3 point indexes and the end delimiter
200  index[faceTriaOffset*4+4*g] = faceNodeOffset+N1;
201  index[faceTriaOffset*4+4*g+1] = faceNodeOffset+N2;
202  index[faceTriaOffset*4+4*g+2] = faceNodeOffset+N3;
203  index[faceTriaOffset*4+4*g+3] = SO_END_FACE_INDEX;
204  }
205 
206  parts[ii] = nbTriInFace; // new part
207 
208  // counting up the per Face offsets
209  faceNodeOffset += nbNodesInFace;
210  faceTriaOffset += nbTriInFace;
211  }
212 
213  // previewNormalize all previewNormals
214  for (int i = 0; i< numNorms ;i++)
215  previewNorms[i].normalize();
216 
217  // end the editing of the nodes
218  previewCoords ->point .finishEditing();
219  previewNorm ->vector .finishEditing();
220  previewFaceSet ->coordIndex .finishEditing();
221  previewFaceSet ->partIndex .finishEditing();
222  }
223  catch (...) {
224  Base::Console().Error("Cannot compute Inventor representation for the shape of %s.\n",pcObject->getNameInDocument());
225  }
226 }

236 void ViewProviderAddSub::setPreviewDisplayMode(bool onoff) {
237  // A mask mode is always set, also for hidden objects.
238  // Now when changing to another mask mode this automatically
239  // displays an object and when restoring the previous state it's
240  // not sufficient to only revert the mask mode. Also the child
241  // number of the switch node must be reverted.
242  if (onoff) {
243  if(pcModeSwitch->getChild(getDefaultMode()) == previewShape)
244  return;
245  displayMode = getActiveDisplayMode();
246  whichChild = pcModeSwitch->whichChild.getValue();
247  setDisplayMaskMode("Shape preview");
248  }
249 
250  if (!onoff) {
251  if(pcModeSwitch->getChild(getDefaultMode()) != previewShape)
252  return;
253  setDisplayMaskMode(displayMode.c_str());
254  pcModeSwitch->whichChild.setValue(whichChild);
255  }
256 
257  App::DocumentObject* obj = getObject<PartDesign::Feature>()->BaseFeature.getValue();
258  if (obj)
259  static_cast<PartDesignGui::ViewProvider*>(Gui::Application::Instance->getViewProvider(obj))->makeTemporaryVisible(onoff);
260 }
