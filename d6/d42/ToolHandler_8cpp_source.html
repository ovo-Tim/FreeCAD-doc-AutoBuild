111 void ToolHandler::setSvgCursor(const QString& cursorName,
112  int x,
113  int y,
114  const std::map<unsigned long, unsigned long>& colorMapping)
115 {
116  // The TechDraw_Pointer_*.svg icons have a default size of 64x64. When directly creating
117  // them with a size of 32x32 they look very bad.
118  // As a workaround the icons are created with 64x64 and afterwards the pixmap is scaled to
119  // 32x32. This workaround is only needed if pRatio is equal to 1.0
120  //
121  qreal pRatio = devicePixelRatio();
122  bool isRatioOne = (pRatio == 1.0);
123  qreal defaultCursorSize = isRatioOne ? 64 : 32;
124  qreal hotX = x;
125  qreal hotY = y;
126 #if !defined(Q_OS_WIN32) && !defined(Q_OS_MACOS)
127  if (qGuiApp->platformName() == QLatin1String("xcb")) {
128  hotX *= pRatio;
129  hotY *= pRatio;
130  }
131 #endif
132  qreal cursorSize = defaultCursorSize * pRatio;
133 
134  QPixmap pointer = Gui::BitmapFactory().pixmapFromSvg(cursorName.toStdString().c_str(),
135  QSizeF(cursorSize, cursorSize),
136  colorMapping);
137  if (isRatioOne) {
138  pointer = pointer.scaled(32, 32);
139  }
140  pointer.setDevicePixelRatio(pRatio);
141  setCursor(pointer, hotX, hotY, false);
142 }

144 void ToolHandler::setCursor(const QPixmap& p, int x, int y, bool autoScale)
145 {
146 
147  QWidget* cw = getCursorWidget();
148  if (cw) {
149  QCursor cursor;
150  QPixmap p1(p);
151  // TODO remove autoScale after all cursors are SVG-based
152  if (autoScale) {
153  qreal pRatio = devicePixelRatio();
154  int newWidth = p.width() * pRatio;
155  int newHeight = p.height() * pRatio;
156  p1 = p1.scaled(newWidth, newHeight, Qt::KeepAspectRatio, Qt::SmoothTransformation);
157  p1.setDevicePixelRatio(pRatio);
158  qreal hotX = x;
159  qreal hotY = y;
160 #if !defined(Q_OS_WIN32) && !defined(Q_OS_MACOS)
161  if (qGuiApp->platformName() == QLatin1String("xcb")) {
162  hotX *= pRatio;
163  hotY *= pRatio;
164  }
165 #endif
166  cursor = QCursor(p1, hotX, hotY);
167  }
168  else {
169  // already scaled
170  cursor = QCursor(p1, x, y);
171  }
172 
173  actCursor = cursor;
174  actCursorPixmap = p1;
175 
176  setWidgetCursor(cursor);
177  }
178 }

180 void ToolHandler::addCursorTail(std::vector<QPixmap>& pixmaps)
181 {
182  // Create a pixmap that will contain icon and each autoconstraint icon
183  QPixmap baseIcon = QPixmap(actCursorPixmap);
184  baseIcon.setDevicePixelRatio(actCursorPixmap.devicePixelRatio());
185  qreal pixelRatio = baseIcon.devicePixelRatio();
186  // cursor size in device independent pixels
187  qreal baseCursorWidth = baseIcon.width();
188  qreal baseCursorHeight = baseIcon.height();
189 
190  int tailWidth = 0;
191  for (auto const& p : pixmaps) {
192  tailWidth += p.width();
193  }
194 
195  int newIconWidth = baseCursorWidth + tailWidth;
196  int newIconHeight = baseCursorHeight;
197 
198  QPixmap newIcon(newIconWidth, newIconHeight);
199  newIcon.fill(Qt::transparent);
200 
201  QPainter qp;
202  qp.begin(&newIcon);
203 
204  qp.drawPixmap(QPointF(0, 0),
205  baseIcon.scaled(baseCursorWidth * pixelRatio,
206  baseCursorHeight * pixelRatio,
207  Qt::KeepAspectRatio,
208  Qt::SmoothTransformation));
209 
210  // Iterate through pixmaps and them to the cursor pixmap
211  qreal currentIconX = baseCursorWidth;
212  qreal currentIconY;
213 
214  for (auto& icon : pixmaps) {
215  currentIconY = baseCursorHeight - icon.height();
216  qp.drawPixmap(QPointF(currentIconX, currentIconY), icon);
217  currentIconX += icon.width();
218  }
219 
220  qp.end(); // Finish painting
221 
222  // Create the new cursor with the icon.
223  QPoint p = actCursor.hotSpot();
224  newIcon.setDevicePixelRatio(pixelRatio);
225  QCursor newCursor(newIcon, p.x(), p.y());
226  applyCursor(newCursor);
227 }
