52 class UpdateMetadataCacheWorker(QtCore.QThread):
53  """Scan through all available packages and see if our local copy of package.xml needs to be
54  updated"""
55 
56  progress_made = QtCore.Signal(str, int, int)

60  """The type of item being downloaded."""

67  def __init__(self, repos):

69  QtCore.QThread.__init__(self)
70  self.repos = repos
71  self.requests: Dict[int, (Addon, UpdateMetadataCacheWorker.RequestType)] = {}
72  NetworkManager.AM_NETWORK_MANAGER.completed.connect(self.download_completed)
73  self.requests_completed = 0
74  self.total_requests = 0
75  self.store = os.path.join(FreeCAD.getUserCachePath(), "AddonManager", "PackageMetadata")
76  FreeCAD.Console.PrintLog(f"Storing Addon Manager cache data in {self.store}\n")
77  self.updated_repos = set()

80  def run(self):
81  """Not usually called directly: instead, create an instance and call its
82  start() function to spawn a new thread."""
83 
84  self.update_from_remote_cache()
85 
86  current_thread = QtCore.QThread.currentThread()
87 
88  for repo in self.repos:
89  if repo.name in self.remote_cache_data:
90  self.update_addon_from_remote_cache_data(repo)
91  elif not repo.macro and repo.url and utils.recognized_git_location(repo):
92  # package.xml
93  index = NetworkManager.AM_NETWORK_MANAGER.submit_unmonitored_get(
94  utils.construct_git_url(repo, "package.xml")
95  )
96  self.requests[index] = (
97  repo,
98  UpdateMetadataCacheWorker.RequestType.PACKAGE_XML,
99  )
100  self.total_requests += 1
101 
102  # metadata.txt
103  index = NetworkManager.AM_NETWORK_MANAGER.submit_unmonitored_get(
104  utils.construct_git_url(repo, "metadata.txt")
105  )
106  self.requests[index] = (
107  repo,
108  UpdateMetadataCacheWorker.RequestType.METADATA_TXT,
109  )
110  self.total_requests += 1
111 
112  # requirements.txt
113  index = NetworkManager.AM_NETWORK_MANAGER.submit_unmonitored_get(
114  utils.construct_git_url(repo, "requirements.txt")
115  )
116  self.requests[index] = (
117  repo,
118  UpdateMetadataCacheWorker.RequestType.REQUIREMENTS_TXT,
119  )
120  self.total_requests += 1
121 
122  while self.requests:
123  if current_thread.isInterruptionRequested():
124  for request in self.requests:
125  NetworkManager.AM_NETWORK_MANAGER.abort(request)
126  return
127  # 50 ms maximum between checks for interruption
128  QtCore.QCoreApplication.processEvents(QtCore.QEventLoop.AllEvents, 50)
129 
130  # This set contains one copy of each of the repos that got some kind of data in
131  # this process. For those repos, tell the main Addon Manager code that it needs
132  # to update its copy of the repo, and redraw its information.
133  for repo in self.updated_repos:
134  self.package_updated.emit(repo)

136  def update_from_remote_cache(self) -> None:
137  """Pull the data on the official repos from a remote cache site (usually
138  https://freecad.org/addons/addon_cache.json)"""
139  data_source = fci.Preferences().get("AddonsCacheURL")
140  try:
141  fetch_result = NetworkManager.AM_NETWORK_MANAGER.blocking_get(data_source, 5000)
142  if fetch_result:
143  self.remote_cache_data = json.loads(fetch_result.data())
144  else:
145  fci.Console.PrintWarning(
146  f"Failed to read from {data_source}. Continuing without remote cache...\n"
147  )
148  except RuntimeError:
149  # If the remote cache can't be fetched, we continue anyway
150  pass

152  def update_addon_from_remote_cache_data(self, addon: Addon):
153  """Given a repo that exists in the remote cache, load in its metadata."""
154  fci.Console.PrintLog(f"Used remote cache data for {addon.name} metadata\n")
155  if "package.xml" in self.remote_cache_data[addon.name]:
156  self.process_package_xml(addon, self.remote_cache_data[addon.name]["package.xml"])
157  if "requirements.txt" in self.remote_cache_data[addon.name]:
158  self.process_requirements_txt(
159  addon, self.remote_cache_data[addon.name]["requirements.txt"]
160  )
161  if "metadata.txt" in self.remote_cache_data[addon.name]:
162  self.process_metadata_txt(addon, self.remote_cache_data[addon.name]["metadata.txt"])

164  def download_completed(self, index: int, code: int, data: QtCore.QByteArray) -> None:
165  """Callback for handling a completed metadata file download."""
166  if index in self.requests:
167  self.requests_completed += 1
168  request = self.requests.pop(index)
169  if code == 200: # HTTP success
170  self.updated_repos.add(request[0]) # mark this repo as updated
171  file = "unknown"
172  if request[1] == UpdateMetadataCacheWorker.RequestType.PACKAGE_XML:
173  self.process_package_xml(request[0], data)
174  file = "package.xml"
175  elif request[1] == UpdateMetadataCacheWorker.RequestType.METADATA_TXT:
176  self.process_metadata_txt(request[0], data)
177  file = "metadata.txt"
178  elif request[1] == UpdateMetadataCacheWorker.RequestType.REQUIREMENTS_TXT:
179  self.process_requirements_txt(request[0], data)
180  file = "requirements.txt"
181  elif request[1] == UpdateMetadataCacheWorker.RequestType.ICON:
182  self.process_icon(request[0], data)
183  file = "icon"
184  message = translate("AddonsInstaller", "Downloaded {} for {}").format(
185  file, request[0].display_name
186  )
187  self.progress_made.emit(message, self.requests_completed, self.total_requests)

189  def process_package_xml(self, repo: Addon, data: QtCore.QByteArray):
190  """Process the package.xml metadata file"""
191  repo.repo_type = Addon.Kind.PACKAGE # By definition
192  package_cache_directory = os.path.join(self.store, repo.name)
193  if not os.path.exists(package_cache_directory):
194  os.makedirs(package_cache_directory)
195  new_xml_file = os.path.join(package_cache_directory, "package.xml")
196  with open(new_xml_file, "w", encoding="utf-8") as f:
197  string_data = self._ensure_string(data, repo.name, "package.xml")
198  f.write(string_data)
199  try:
200  metadata = MetadataReader.from_file(new_xml_file)
201  except xml.etree.ElementTree.ParseError:
202  fci.Console.PrintWarning("An invalid or corrupted package.xml file was downloaded for")
203  fci.Console.PrintWarning(f" {self.name}... ignoring the bad data.\n")
204  return
205  repo.set_metadata(metadata)
206  FreeCAD.Console.PrintLog(f"Downloaded package.xml for {repo.name}\n")
207 
208  # Grab a new copy of the icon as well: we couldn't enqueue this earlier because
209  # we didn't know the path to it, which is stored in the package.xml file.
210  icon = repo.get_best_icon_relative_path()
211 
212  icon_url = utils.construct_git_url(repo, icon)
213  index = NetworkManager.AM_NETWORK_MANAGER.submit_unmonitored_get(icon_url)
214  self.requests[index] = (repo, UpdateMetadataCacheWorker.RequestType.ICON)
215  self.total_requests += 1

217  def _ensure_string(self, arbitrary_data, addon_name, file_name) -> str:
218  if isinstance(arbitrary_data, str):
219  return arbitrary_data
220  if isinstance(arbitrary_data, QtCore.QByteArray):

224  def _decode_data(self, byte_data, addon_name, file_name) -> str:
225  """UTF-8 decode data, and print an error message if that fails"""
226 
227  # For review and debugging purposes, store the file locally
228  package_cache_directory = os.path.join(self.store, addon_name)
229  if not os.path.exists(package_cache_directory):
230  os.makedirs(package_cache_directory)
231  new_xml_file = os.path.join(package_cache_directory, file_name)
232  with open(new_xml_file, "wb") as f:
233  f.write(byte_data)
234 
235  f = ""
236  try:
237  f = byte_data.decode("utf-8")
238  except UnicodeDecodeError as e:
239  FreeCAD.Console.PrintWarning(
240  translate(
241  "AddonsInstaller",
242  "Failed to decode {} file for Addon '{}'",
243  ).format(file_name, addon_name)
244  + "\n"
245  )
246  FreeCAD.Console.PrintWarning(str(e) + "\n")
247  FreeCAD.Console.PrintWarning(
248  translate(
249  "AddonsInstaller",
250  "Any dependency information in this file will be ignored",
251  )
252  + "\n"
253  )
254  return f

256  def process_metadata_txt(self, repo: Addon, data: QtCore.QByteArray):
257  """Process the metadata.txt metadata file"""
258  f = self._ensure_string(data, repo.name, "metadata.txt")
259  lines = f.splitlines()
260  for line in lines:
261  if line.startswith("workbenches="):
262  depswb = line.split("=")[1].split(",")
263  for wb in depswb:
264  wb_name = wb.strip()
265  if wb_name:
266  repo.requires.add(wb_name)
267  FreeCAD.Console.PrintLog(
268  f"{repo.display_name} requires FreeCAD Addon '{wb_name}'\n"
269  )
270 
271  elif line.startswith("pylibs="):
272  depspy = line.split("=")[1].split(",")
273  for pl in depspy:
274  dep = pl.strip()
275  if dep:
276  repo.python_requires.add(dep)
277  FreeCAD.Console.PrintLog(
278  f"{repo.display_name} requires python package '{dep}'\n"
279  )
280 
281  elif line.startswith("optionalpylibs="):
282  opspy = line.split("=")[1].split(",")
283  for pl in opspy:
284  dep = pl.strip()
285  if dep:
286  repo.python_optional.add(dep)
287  FreeCAD.Console.PrintLog(
288  f"{repo.display_name} optionally imports python package"
289  + f" '{pl.strip()}'\n"
290  )

292  def process_requirements_txt(self, repo: Addon, data: QtCore.QByteArray):
293  """Process the requirements.txt metadata file"""
294 
295  f = self._ensure_string(data, repo.name, "requirements.txt")
296  lines = f.splitlines()
297  for line in lines:
298  break_chars = " <>=~!+#"
299  package = line
300  for n, c in enumerate(line):
301  if c in break_chars:
302  package = line[:n].strip()
303  break
304  if package:
305  repo.python_requires.add(package)
