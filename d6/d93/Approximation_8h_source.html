50 template<class Real>
51 class PolynomialSurface: public ImplicitSurface<Real>
52 {
53 public:
54  explicit PolynomialSurface(const Real afCoeff[6])
55  {
56  for (int i = 0; i < 6; i++) {
57  m_afCoeff[i] = afCoeff[i];
58  }
59  }
60 
61  // the function
62  Real F(const Vector3<Real>& rkP) const override
63  {
64  return (m_afCoeff[0] * rkP.X() * rkP.X() + m_afCoeff[1] * rkP.Y() * rkP.Y()
65  + m_afCoeff[2] * rkP.X() + m_afCoeff[3] * rkP.Y() + m_afCoeff[4] * rkP.X() * rkP.Y()
66  + m_afCoeff[5] - rkP.Z());
67  }
68 
69  // first-order partial derivatives
70  Real FX(const Vector3<Real>& rkP) const override
71  {
72  return (Real)(2.0 * m_afCoeff[0] * rkP.X() + m_afCoeff[2] + m_afCoeff[4] * rkP.Y());
73  }
74  Real FY(const Vector3<Real>& rkP) const override
75  {
76  return (Real)(2.0 * m_afCoeff[1] * rkP.Y() + m_afCoeff[3] + m_afCoeff[4] * rkP.X());
77  }
78  Real FZ(const Vector3<Real>& /*rkP*/) const override
79  {
80  return (Real)-1.0;
81  }
82 
83  // second-order partial derivatives
84  Real FXX(const Vector3<Real>& /*rkP*/) const override
85  {
86  return (Real)(2.0 * m_afCoeff[0]);
87  }
88  Real FXY(const Vector3<Real>& /*rkP*/) const override
89  {
90  return (Real)(m_afCoeff[4]);
91  }
92  Real FXZ(const Vector3<Real>& /*rkP*/) const override
93  {
94  return (Real)0.0;
95  }
96  Real FYY(const Vector3<Real>& /*rkP*/) const override
97  {
98  return (Real)(2.0 * m_afCoeff[1]);
99  }
100  Real FYZ(const Vector3<Real>& /*rkP*/) const override
101  {
102  return (Real)0.0;
103  }
104  Real FZZ(const Vector3<Real>& /*rkP*/) const override
105  {
106  return (Real)0.0;
107  }
108 
109 private:
110  Real m_afCoeff[6];
111 };

602  bool CurvatureInfo(double x, double y, double z, double& rfCurv0, double& rfCurv1)
603  {
604  double dQuot = Fz(x, y, z);
605  double zx = -(Fx(x, y, z) / dQuot);
606  double zy = -(Fy(x, y, z) / dQuot);
607 
608  double zxx = -(2.0 * (dKoeff[5] + dKoeff[6] * zx * zx + dKoeff[8] * zx)) / dQuot;
609  double zyy = -(2.0 * (dKoeff[5] + dKoeff[6] * zy * zy + dKoeff[9] * zy)) / dQuot;
610  double zxy = -(dKoeff[6] * zx * zy + dKoeff[7] + dKoeff[8] * zy + dKoeff[9] * zx) / dQuot;
611 
612  double dNen = 1 + zx * zx + zy * zy;
613  double dNenSqrt = sqrt(dNen);
614  double K = (zxx * zyy - zxy * zxy) / (dNen * dNen);
615  double H = 0.5
616  * ((1.0 + zx * zx - 2 * zx * zy * zxy + (1.0 + zy * zy) * zxx)
617  / (dNenSqrt * dNenSqrt * dNenSqrt));
618 
619  double dDiscr = sqrt(fabs(H * H - K));
620  rfCurv0 = H - dDiscr;
621  rfCurv1 = H + dDiscr;
622 
623  return true;
624  }

627  double F(double x, double y, double z)
628  {
629  return (dKoeff[0] + dKoeff[1] * x + dKoeff[2] * y + dKoeff[3] * z + dKoeff[4] * x * x
630  + dKoeff[5] * y * y + dKoeff[6] * z * z + dKoeff[7] * x * y + dKoeff[8] * x * z
631  + dKoeff[9] * y * z);
632  }
633

637  return (dKoeff[1] + 2.0 * dKoeff[4] * x + dKoeff[7] * y + dKoeff[8] * z);

642  }

655  }

662  }
