154 PartDesign::Body * makeBodyActive(App::DocumentObject *body, App::Document *doc,
155  App::DocumentObject **topParent,
156  std::string *subname)
157 {
158  App::DocumentObject *parent = nullptr;
159  std::string sub;
160 
161  for(auto &v : body->getParents()) {
162  if(v.first->getDocument()!=doc)
163  continue;
164  if(parent) {
165  body = nullptr;
166  break;
167  }
168  parent = v.first;
169  sub = v.second;
170  }
171 
172  if(body) {
173  auto _doc = parent?parent->getDocument():body->getDocument();
174  Gui::cmdGuiDocument(_doc, std::stringstream()
175  << "ActiveView.setActiveObject('" << PDBODYKEY
176  << "'," << Gui::Command::getObjectCmd(parent?parent:body)
177  << ",'" << sub << "')");
178  return Gui::Application::Instance->activeView()->
179  getActiveObject<PartDesign::Body*>(PDBODYKEY,topParent,subname);
180  }
181 
182  return dynamic_cast<PartDesign::Body*>(body);

276 void fixSketchSupport (Sketcher::SketchObject* sketch)
277 {
278  App::DocumentObject* support = sketch->AttachmentSupport.getValue();
279 
280  if (support)
281  return; // Sketch is on a face of a solid, do nothing
282 
283  const App::Document* doc = sketch->getDocument();
284  PartDesign::Body *body = getBodyFor(sketch, /*messageIfNot*/ false);
285  if (!body) {
286  throw Base::RuntimeError ("Couldn't find body for the sketch");
287  }
288 
289  // Get the Origin for the body
290  App::Origin *origin = body->getOrigin (); // May throw by itself
291 
292  Base::Placement plm = sketch->Placement.getValue();
293  Base::Vector3d pnt = plm.getPosition();
294 
295  // Currently we only handle positions that are parallel to the base planes
296  Base::Rotation rot = plm.getRotation();
297  Base::Vector3d sketchVector(0,0,1);
298  rot.multVec(sketchVector, sketchVector);
299  bool reverseSketch = (sketchVector.x + sketchVector.y + sketchVector.z) < 0.0 ;
300  if (reverseSketch) sketchVector *= -1.0;
301 
302  App::Plane *plane =nullptr;
303 
304  if (sketchVector == Base::Vector3d(0,0,1))
305  plane = origin->getXY ();
306  else if (sketchVector == Base::Vector3d(0,1,0))
307  plane = origin->getXZ ();
308  else if (sketchVector == Base::Vector3d(1,0,0))
309  plane = origin->getYZ ();
310  else {
311  throw Base::ValueError("Sketch plane cannot be migrated");
312  }
313  assert (plane);
314 
315  // Find the normal distance from origin to the sketch plane
316  gp_Pln pln(gp_Pnt (pnt.x, pnt.y, pnt.z), gp_Dir(sketchVector.x, sketchVector.y, sketchVector.z));
317  double offset = pln.Distance(gp_Pnt(0,0,0));
318  // TODO Issue a message if sketch have coordinates offset inside the plain (2016-08-15, Fat-Zer)
319 
320  if (fabs(offset) < Precision::Confusion()) {
321  // One of the base planes
322  FCMD_OBJ_CMD(sketch,"AttachmentSupport = (" << Gui::Command::getObjectCmd(plane) << ",[''])");
323  FCMD_OBJ_CMD(sketch,"MapReversed = " << (reverseSketch ? "True" : "False"));
324  FCMD_OBJ_CMD(sketch,"MapMode = '" << Attacher::AttachEngine::getModeName(Attacher::mmFlatFace) << "'");
325 
326  } else {
327  // Offset to base plane
328  // Find out which direction we need to offset
329  double a = sketchVector.GetAngle(pnt);
330  if ((a < -M_PI_2) || (a > M_PI_2))
331  offset *= -1.0;
332 
333  std::string Datum = doc->getUniqueObjectName("DatumPlane");
334  FCMD_DOC_CMD(doc,"addObject('PartDesign::Plane','"<<Datum<<"')");
335  auto obj = doc->getObject(Datum.c_str());
336  FCMD_OBJ_CMD(obj,"AttachmentSupport = [(" << Gui::Command::getObjectCmd(plane) << ",'')]");
337  FCMD_OBJ_CMD(obj,"MapMode = '" << AttachEngine::getModeName(Attacher::mmFlatFace) << "'");
338  FCMD_OBJ_CMD(obj,"AttachmentOffset.Base.z = " << offset);
339  FCMD_OBJ_CMD(body,"insertObject("<<Gui::Command::getObjectCmd(obj)<<','<<
340  Gui::Command::getObjectCmd(sketch)<<")");
341  FCMD_OBJ_CMD(sketch,"AttachmentSupport = (" << Gui::Command::getObjectCmd(obj) << ",[''])");
342  FCMD_OBJ_CMD(sketch,"MapReversed = " << (reverseSketch ? "True" : "False"));
343  FCMD_OBJ_CMD(sketch,"MapMode = '" << Attacher::AttachEngine::getModeName(Attacher::mmFlatFace) << "'");
344  }
345 }

358  App::Document *doc = feature->getDocument();
359 
360  for ( const auto & obj: doc->getObjects () ) {
361  if ( !isPartDesignAwareObjecta ( obj, respectGroups ) ) {
362  std::vector <App::Property *> properties;
