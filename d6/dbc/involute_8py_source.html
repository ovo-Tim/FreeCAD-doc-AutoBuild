107 def _create_involute_profile(
108  wire_builder,
109  module,
110  number_of_teeth,
111  pressure_angle=radians(20.0),
112  split_involute=True,
113  rotation=radians(0),
114  outer_height_coefficient=1.0,
115  inner_height_coefficient=1.0,
116  outer_fillet_coefficient=0.0,
117  inner_fillet_coefficient=0.0,
118  profile_shift_coefficient=0.0):
119  """
120  Create an involute gear profile in the given wire builder
121 
122  This method can be used to create external as well as internal gear and spline profiles.
123  Thus this method does not use the terms "addednum" and "dedendum" or "tip" and "root",
124  but refers to the elements below the reference circle (i.e. towards the center) as "inner",
125  and those above the reference circle (i.e. away from the center) as "outer".
126 
127  For an external gear, outer_height is the addendum, inner_height is the dedendum,
128  and inner_fillet is the root fillet.
129  For an internal gear, inner_height is the addendum, outer_height is the dedendum,
130  and outer_fillet is the root fillet.
131 
132  The "_coefficient" suffix denotes values normalized by the module.
133  """
134 
135  profile_shift = profile_shift_coefficient * module
136  outer_height = outer_height_coefficient * module + profile_shift # ext: addendum, int: dedednum
137  inner_height = inner_height_coefficient * module - profile_shift # ext: dedendum, int: addednum
138 
139  # ****** calculate radii
140  # All distances from the center of the profile start with "R".
141  # As those are mostly used in mathematical formulae, we use "symbols" rather than "speaking
142  # names" to keep them short. Otherwise the math becomes unreadable.
143  Rref = number_of_teeth * module / 2 # reference circle radius
144  Rb = Rref * cos(pressure_angle) # base circle radius
145  Ro = Rref + outer_height # radius of outer circle (tip for external, root for internal gears)
146  Ri = Rref - inner_height # radius of inner circle (root for external, tip for internal gears)
147 
148  fi = inner_fillet_coefficient * module # fillet radius at inner circle (ext: root fillet)
149  Rci = Ri + fi # radius at which the center of the inner fillet circle sits
150  Rfi = Rci # radius at which the inner fillet ends (for now assuming some non-involute flank)
151 
152  fo = outer_fillet_coefficient * module # fillet radius at outer circle (int: root fillet)
153  Rco = Ro - fo # radius at which the center of the outer fillet circle sits
154  Rfo = Ro # radius at which the outer fillet ends (for now assuming no outer fillet)
155 
156  has_non_involute_flank = Rfi < Rb # above the base circle we have the involute,
157  # below we have a straight line towards the center.
158  has_inner_fillet = fi > 0
159  has_outer_fillet = fo > 0
160 
161  if has_inner_fillet and not has_non_involute_flank:
162  # In this case we need tangency of the involute's start and the inner fillet circle.
163  # It has to be somewhere between max(Rb,Ri) and Rci.
164  # So we need the radius R from the origin depending on the tangent angle, for both:
165  # - the involute: Rinv(t) = Rb * sqrt(1 + t**2), rationale: see below.
166  # - the fillet circle: Rcirc(t) = ... a bit more complex, see below.
167  # and then find the R where both tangents are equal.
168  # As the tangent angle of the involute equals the parameter t, we can just take the
169  # parametric polar representation of the involute as our first equation.
170  # For a circle in parameter form, the tangent angle is pi/2 - t (for the first quadrant,
171  # i.e. 0 <= t <= pi/2, and more than one quadrant is not of interest for us). Unfortunately,
172  # the fillet circle is offset from the origin by (Rci,phi), so things becomes more messy:
173  # Rcirc(t) = sqrt((Rci*cos(phi) + fi*cos(t))**2 + (Rci*sin(phi) + fi*sin(t))**2)
174  # To get rid of the sin(t) part we assume phi "very small", i.e. sin(phi) becomes 0.
175  # This is justyfied because Rci is much larger than fi and the parallax error is
176  # neglectable. Next we substitute the parameter t by pi/2-q-pi. For one to account for the
177  # tangent angle definitions (see above), and then to turn our circle by 180Â° as for the
178  # inner fillet we need the third quadrant of the circle (NB: we are looking at the upper
179  # half of the right most tooth, i.e. the involute grows downwards!). This results in
180  # sqrt(2*fi*Rci*cos(-1/2*pi - q) + fi**2 + Rci**2) which simplifies to
181  # Rcirc(q) = sqrt(-2*fi*Rci*sin(q) + fi**2 + Rci**2)
182  # which is the polar r for a given tangent angle q (with the simplification of phi being 0)
183  # This can now be inverted to give the tangent angle at a given r:
184  # Qcirc(r) = asin((-r**2 + fi**2 + Rci**2)/(2*fi*Rci))
185  # For the involute we have Rinv(q) = Rb * sqrt(1 + q**2), with q=t as the circle is already
186  # adapted, thus: Qinv(r) = +- sqrt(r**2 - Rb**2)/Rb
187  # Now to find the r where the tangents match, our Rfi, we set Qcirc=Qinv and solve for r.
188  # This doesn't seem to have an analytical solution, though, so let's apply Newton's
189  # Method to find the root of q := Qinv - Qcirc over Rb...Rci, or for larger number
190  # of teeth, Ri...Rci as in this case the base circle is below the inner circle.
191  # The graph of q is strictly monotnonous and very steep, no surprises to expect.
192  # To simplify the above equations and to find the derivative, SageMath was used.
193  q = lambda r: (sqrt(r**2 - Rb**2) / Rb
194  - asin((-r**2 + fi**2 + Rci**2) / (2 * fi * Rci)))
195  q_prime = lambda r: (r / (sqrt(-Rb**2 + r**2) * Rb)
196  + r / (fi * Rci * sqrt(1 - 1/4 * (r**2 - fi**2 - Rci**2)**2 / (fi**2 * Rci**2))))
197  Rfi = findRootNewton(q, q_prime, x_min=max(Rb, Ri), x_max=Rci)
198 
199  if has_outer_fillet:
200  # In this case we need tangency of the involute's end and the outer fillet circle.
201  # For a detailed explanation refer to the inner fillet above.
202  # Here, however, we substitute t with pi/2-q, i.e. we omit the extra -pi, as we are now
203  # interested in the first quadrant: Again, our involute grows downwards but now the fillet
204  # circle needs to approach the involute "from inside the tooth" and at the far end of the
205  # involute. The fillet is now
206  # Rcirc(q) = sqrt(2*fo*Rco*sin(q) + fo**2 + Rco**2)
207  # which can be inversed to
208  # Qcirc(r) = asin((r**2 - fo**2 - Rco**2)/(2*fo*Rco))
209  # The involute is the very same as for the inner fillet.
210  # However, the simplification of assuming phi=0 is more questionable here as phi doesn't
211  # have an upper bound from the fillet's radius any more but gets larger as the involute
212  # angle grows. Having a non-zero phi in the original Rcirc(q) makes solving impossible,
213  # so lets apply another trick:
214  # If we cannot apply a polar angle to the position of the circle, we could do it for the
215  # involute. Such a rotation doesn't have any influence on Rinv, so not on Qinv, but changes
216  # the interpretation of it: The tangent angle of the involute experiences the same rotation
217  # as the involute itself. So it is just a simple offset:
218  # Our q(r) becomes Qinv(r) - Qcirc(r) - phi_corr, where phi_corr is the amount we
219  # (virtually) rotate our involute around the origin. To bring the fillet circle back on
220  # X-axis, we assume the (real) polar angle of its center position being fo below the
221  # involute's end at Ro.
222  phi_corr = genInvolutePolar(Rb, Ro) + atan(fo / Ro)
223  q = lambda r: (sqrt(r**2 - Rb**2) / Rb
224  - asin((r**2 - fo**2 - Rco**2) / (2 * fo * Rco))
225  - phi_corr)
226  q_prime = lambda r: (r / (sqrt(-Rb**2 + r**2) * Rb)
227  - r / (fo * Rco * sqrt(1 - 1/4 * (r**2 - fo**2 - Rco**2)**2 / (fo**2 * Rco**2))))
228  Rfo = findRootNewton(q, q_prime, x_min=max(Rb, Rco), x_max=Ro)
229 
230  # ****** calculate angles (all in radians)
231  angular_pitch = 2 * pi / number_of_teeth # angle subtended by complete tooth/space pair
232  base_to_ref = genInvolutePolar(Rb, Rref) # angle between base and reference circle
233  ref_to_stop = genInvolutePolar(Rb, Rfo) - base_to_ref # angle between ref and involute end
234  if has_non_involute_flank: # involute starts at base circle
235  start_to_ref = base_to_ref
236  else: # involute starts at top of inner fillet, i.e. somewhat above the base circle
237  start_to_ref = base_to_ref - genInvolutePolar(Rb, Rfi)
238 
239  inner_fillet_width = sqrt(fi**2 - (Rci - Rfi)**2)
240  inner_fillet_angle = atan(inner_fillet_width / Rfi)
241  outer_fillet_width = sqrt(fo**2 - (Rfo - Rco)**2)
242  outer_fillet_angle = atan(outer_fillet_width / Rfo)
243 
244  # ****** generate Higuchi involute approximation
245  fe = 1 # fraction of involute length at end of approx
246  fs = 0.01 # fraction of length offset from base to avoid singularity
247  if (not has_non_involute_flank):
248  fs = (Rfi**2 - Rb**2) / (Rfo**2 - Rb**2) # offset start to top of fillet
249 
250  if split_involute:
251  # approximate in 2 sections, split 25% along the involute
252  fm = fs + (fe - fs) / 4 # fraction of length at junction (25% along involute)
253  part1 = BezCoeffs(Rb, Rfo, 3, fs, fm)
254  part2 = BezCoeffs(Rb, Rfo, 3, fm, fe)
255  inv = part1 + part2[1:] # join the 2 sets of coeffs (skip duplicate mid point)
256  else:
257  inv = BezCoeffs(Rb, Rfo, 4, fs, fe)
258 
259  # ****** calculate angular tooth thickness at reference circle
260  enlargement_by_shift = profile_shift * tan(pressure_angle) / Rref
261  tooth_thickness_half_angle = angular_pitch / 4 + enlargement_by_shift
262  psi = tooth_thickness_half_angle # for the formulae below, a symbol is more readable
263 
264  # rotate all points to make the tooth symmetric to the X axis
265  inv = [rotate(pt, -base_to_ref - psi) for pt in inv]
266 
267  # create the back profile of tooth (mirror image on X axis)
268  invR = [mirror(pt) for pt in inv]
269 
270  # ****** calculate section junction points.
271  # Those are the points where the named element ends (start is the end of the previous element).
272  # Suffix _back is back of this tooth, suffix _next is front of next tooth.
273  inner_fillet = toCartesian(Rfi, -psi - start_to_ref) # top of fillet
274  inner_fillet_back = mirror(inner_fillet) # flip to make same point on back of tooth
275  inner_circle_back = toCartesian(Ri, psi + start_to_ref + inner_fillet_angle)
276  inner_circle_next = toCartesian(Ri, angular_pitch - psi - start_to_ref - inner_fillet_angle)
277  inner_fillet_next = rotate(inner_fillet, angular_pitch) # top of fillet, front of next tooth
278  outer_fillet = toCartesian(Ro, -psi + ref_to_stop + outer_fillet_angle)
279  outer_circle = mirror(outer_fillet)
280 
281  # ****** build the gear profile using the provided wire builder
282  thetas = [x * angular_pitch + rotation for x in range(number_of_teeth)]
283 
284  # Make sure we begin *exactly* where our last curve ends.
285  # In theory start == rotate(end, angle_of_last_tooth), but in practice we have limited
286  # precision. Especially if we don't have an inner fillet, we end at inner_circle_next,
287  # not inner_fillet_next.
288  # And even though these two should also be equal (if no inner fillet), they are calculated
289  # differently, which is enough for the resulting wire not being closed any more.
290  # So to be on the save side, we begin at rotate(end, angle_of_last_tooth), not start.
291  if has_inner_fillet:
292  wire_builder.move(rotate(inner_fillet_next, thetas[-1]))
293  else:
294  wire_builder.move(rotate(inner_circle_next, thetas[-1]))
295 
296  for theta in thetas:
297  wire_builder.theta = theta
298 
299  if (has_non_involute_flank):
300  wire_builder.line(inv[0]) # line from inner fillet up to base circle
301 
302  # front involute
303  if split_involute:
304  wire_builder.curve(inv[1], inv[2], inv[3])
305  wire_builder.curve(inv[4], inv[5], inv[6])
306  else:
307  wire_builder.curve(*inv[1:])
308 
309  # is there a section of outer circle between outer fillets?
310  if (outer_circle[1] > outer_fillet[1]):
311  if has_outer_fillet:
312  wire_builder.arc(outer_fillet, fo, 1) # outer fillet
313  wire_builder.arc(outer_circle, Ro, 1) # arc across the outer circle
314 
315  if has_outer_fillet:
316  wire_builder.arc(invR[-1], fo, 1) # outer fillet on back side
317 
318  # back involute
319  if split_involute:
320  wire_builder.curve(invR[5], invR[4], invR[3])
321  wire_builder.curve(invR[2], invR[1], invR[0])
322  else:
323  wire_builder.curve(*invR[-2::-1])
324 
325  if (has_non_involute_flank):
326  wire_builder.line(inner_fillet_back) # line down to top of inner fillet
327 
328  # is there a section of inner circle between inner fillets?
329  if (inner_circle_next[1] > inner_circle_back[1]):
330  if has_inner_fillet:
331  wire_builder.arc(inner_circle_back, fi, 0) # inner fillet on back side
332  wire_builder.arc(inner_circle_next, Ri, 1) # arc across the inner circle
333 
334  if has_inner_fillet:
335  wire_builder.arc(inner_fillet_next, fi, 0)
336 
337  wire_builder.close()

340 def genInvolutePolar(Rb, R):
341  """return the involute angle as function of radius R.
342  Rb = base circle radius
343  """
344  return (sqrt(R*R - Rb*Rb) / Rb) - acos(Rb / R)

365 def findRootNewton(f, f_prime, x_min, x_max):
366  """Apply Newton's Method to find the root of f within x_min and x_max
367  We assume that there is a root in that range and that f is strictly monotonic,
368  i.e. we don't take precautions for overshooting beyond the input range.
369  """
370  # As initial guess let's take the middle of our input range
371  x = (x_min + x_max) / 2
372 
373  # FreeCAD.Base.Precision.intersection() is 1e-9, but this file doesn't depend on FreeCAD,
374  # a property very handy when testing in isolation, so let's keep it.
375  PRECISION_INTERSECTION = 1e-9
376 
377  # Experience has shown that usually 2-3 iterations are enough, but better save than sorry
378  for i in range(6):
379  f_x = f(x)
380  if abs(f_x) < PRECISION_INTERSECTION:
381  return x
382  x = x - f_x/f_prime(x)
383 
384  raise RuntimeError(f"No convergence after {i+1} iterations.")

387 def chebyExpnCoeffs(j, func):
388  N = 50 # a suitably large number N>>p
389  c = 0
390  for k in range(1, N + 1):
391  c += func(cos(pi * (k - 0.5) / N)) * cos(pi * j * (k - 0.5) / N)
392  return 2 *c / N

395 def chebyPolyCoeffs(p, func):
396  coeffs = [0]*(p+1)
397  fnCoeff = []
398  T = [coeffs[:] for i in range(p+1)]
399  T[0][0] = 1
400  T[1][1] = 1
401  # now generate the Chebyshev polynomial coefficient using
402  # formula T(k+1) = 2xT(k) - T(k-1) which yields
403  # T = [ [ 1, 0, 0, 0, 0, 0], # T0(x) = +1
404  # [ 0, 1, 0, 0, 0, 0], # T1(x) = 0 +x
405  # [-1, 0, 2, 0, 0, 0], # T2(x) = -1 0 +2xx
406  # [ 0, -3, 0, 4, 0, 0], # T3(x) = 0 -3x 0 +4xxx
407  # [ 1, 0, -8, 0, 8, 0], # T4(x) = +1 0 -8xx 0 +8xxxx
408  # [ 0, 5, 0,-20, 0, 16], # T5(x) = 0 5x 0 -20xxx 0 +16xxxxx
409  # ... ]
410 
411  for k in range(1, p):
412  for j in range(len(T[k]) - 1):
413  T[k + 1][j + 1] = 2 * T[k][j]
414  for j in range(len(T[k - 1])):
415  T[k + 1][j] -= T[k - 1][j]
416 
417  # convert the chebyshev function series into a simple polynomial
418  # and collect like terms, out T polynomial coefficients
419  for k in range(p + 1):
420  fnCoeff.append(chebyExpnCoeffs(k, func))
421 
422  for k in range(p + 1):
423  for pwr in range(p + 1):
424  coeffs[pwr] += fnCoeff[k] * T[k][pwr]
425 
426  coeffs[0] -= fnCoeff[0] / 2 # fix the 0th coeff
427  return coeffs

435 def BezCoeffs(baseRadius, limitRadius, order, fstart, fstop):
436  """Approximates an involute using a Bezier-curve
437 
438  Parameters:
439  baseRadius - the radius of base circle of the involute.
440  This is where the involute starts, too.
441  limitRadius - the radius of an outer circle, where the involute ends.
442  order - the order of the Bezier curve to be fitted e.g. 3, 4, 5, ...
443  fstart - fraction of distance along the involute to start the approximation.
444  fstop - fraction of distance along the involute to stop the approximation.
445  """
446  Rb = baseRadius
447  Ra = limitRadius
448  ta = sqrt(Ra * Ra - Rb * Rb) / Rb # involute angle at the limit radius
449  te = sqrt(fstop) * ta # involute angle, theta, at end of approx
450  ts = sqrt(fstart) * ta # involute angle, theta, at start of approx
451  p = order # order of Bezier approximation
452 
453  def involuteXbez(t):
454  "Equation of involute using the Bezier parameter t as variable"
455  # map t (0 <= t <= 1) onto x (where -1 <= x <= 1)
456  x = t * 2 - 1
457  # map theta (where ts <= theta <= te) from x (-1 <=x <= 1)
458  theta = x * (te - ts) / 2 + (ts + te) / 2
459  return Rb * (cos(theta) + theta * sin(theta))
460 
461  def involuteYbez(t):
462  "Equation of involute using the Bezier parameter t as variable"
463  # map t (0 <= t <= 1) onto x (where -1 <= x <= 1)
464  x = t * 2 - 1
465  # map theta (where ts <= theta <= te) from x (-1 <=x <= 1)
466  theta = x * (te - ts) / 2 + (ts + te) / 2
467  return Rb * (sin(theta) - theta * cos(theta))
468 
469  # calc Bezier coeffs
