284 void ViewProviderPartExt::onChanged(const App::Property* prop)
285 {
286  // The lower limit of the deviation has been increased to avoid
287  // to freeze the GUI
288  // https://forum.freecad.org/viewtopic.php?f=3&t=24912&p=195613
289  if (prop == &Deviation) {
290  if(isUpdateForced()||Visibility.getValue())
291  updateVisual();
292  else
293  VisualTouched = true;
294  }
295  if (prop == &AngularDeflection) {
296  if(isUpdateForced()||Visibility.getValue())
297  updateVisual();
298  else
299  VisualTouched = true;
300  }
301  if (prop == &LineWidth) {
302  pcLineStyle->lineWidth = LineWidth.getValue();
303  }
304  else if (prop == &PointSize) {
305  pcPointStyle->pointSize = PointSize.getValue();
306  }
307  else if (prop == &LineColor) {
308  const App::Color& c = LineColor.getValue();
309  pcLineMaterial->diffuseColor.setValue(c.r,c.g,c.b);
310  if (c != LineMaterial.getValue().diffuseColor)
311  LineMaterial.setDiffuseColor(c);
312  LineColorArray.setValue(LineColor.getValue());
313  }
314  else if (prop == &PointColor) {
315  const App::Color& c = PointColor.getValue();
316  pcPointMaterial->diffuseColor.setValue(c.r,c.g,c.b);
317  if (c != PointMaterial.getValue().diffuseColor)
318  PointMaterial.setDiffuseColor(c);
319  PointColorArray.setValue(PointColor.getValue());
320  }
321  else if (prop == &LineMaterial) {
322  const App::Material& Mat = LineMaterial.getValue();
323  if (LineColor.getValue() != Mat.diffuseColor)
324  LineColor.setValue(Mat.diffuseColor);
325  pcLineMaterial->ambientColor.setValue(Mat.ambientColor.r,Mat.ambientColor.g,Mat.ambientColor.b);
326  pcLineMaterial->diffuseColor.setValue(Mat.diffuseColor.r,Mat.diffuseColor.g,Mat.diffuseColor.b);
327  pcLineMaterial->specularColor.setValue(Mat.specularColor.r,Mat.specularColor.g,Mat.specularColor.b);
328  pcLineMaterial->emissiveColor.setValue(Mat.emissiveColor.r,Mat.emissiveColor.g,Mat.emissiveColor.b);
329  pcLineMaterial->shininess.setValue(Mat.shininess);
330  pcLineMaterial->transparency.setValue(Mat.transparency);
331  }
332  else if (prop == &PointMaterial) {
333  const App::Material& Mat = PointMaterial.getValue();
334  if (PointColor.getValue() != Mat.diffuseColor)
335  PointColor.setValue(Mat.diffuseColor);
336  pcPointMaterial->ambientColor.setValue(Mat.ambientColor.r,Mat.ambientColor.g,Mat.ambientColor.b);
337  pcPointMaterial->diffuseColor.setValue(Mat.diffuseColor.r,Mat.diffuseColor.g,Mat.diffuseColor.b);
338  pcPointMaterial->specularColor.setValue(Mat.specularColor.r,Mat.specularColor.g,Mat.specularColor.b);
339  pcPointMaterial->emissiveColor.setValue(Mat.emissiveColor.r,Mat.emissiveColor.g,Mat.emissiveColor.b);
340  pcPointMaterial->shininess.setValue(Mat.shininess);
341  pcPointMaterial->transparency.setValue(Mat.transparency);
342  }
343  else if (prop == &PointColorArray) {
344  setHighlightedPoints(PointColorArray.getValues());
345  }
346  else if (prop == &LineColorArray) {
347  setHighlightedEdges(LineColorArray.getValues());
348  }
349  else if (prop == &_diffuseColor) {
350  // Used to load the old DiffuseColor values asynchronously
351  std::vector<App::Color> colors = _diffuseColor.getValues();
352  std::vector<float> transparencies;
353  transparencies.resize(static_cast<int>(colors.size()));
354  for (int i = 0; i < static_cast<int>(colors.size()); i++) {
355  transparencies[i] = colors[i].transparency();
356  colors[i].a = 1.0F;
357  }
358  ShapeAppearance.setDiffuseColors(colors);
359  ShapeAppearance.setTransparencies(transparencies);
360  }
361  else if (prop == &ShapeAppearance) {
362  setHighlightedFaces(ShapeAppearance);
363  ViewProviderGeometryObject::onChanged(prop);
364  }
365  else if (prop == &Transparency) {
366  const App::Material& Mat = ShapeAppearance[0];
367  long value = toPercent(Mat.transparency);
368  if (value != Transparency.getValue()) {
369  float trans = fromPercent(Transparency.getValue());
370  ShapeAppearance.setTransparency(trans);
371  }
372  }
373  else if (prop == &Lighting) {
374  if (Lighting.getValue() == 0)
375  pShapeHints->vertexOrdering = SoShapeHints::UNKNOWN_ORDERING;
376  else
377  pShapeHints->vertexOrdering = SoShapeHints::COUNTERCLOCKWISE;
378  }
379  else if (prop == &DrawStyle) {
380  if (DrawStyle.getValue() == 0)
381  pcLineStyle->linePattern = 0xffff;
382  else if (DrawStyle.getValue() == 1)
383  pcLineStyle->linePattern = 0xf00f;
384  else if (DrawStyle.getValue() == 2)
385  pcLineStyle->linePattern = 0x0f0f;
386  else
387  pcLineStyle->linePattern = 0xff88;
388  }
389  else {
390  // if the object was invisible and has been changed, recreate the visual
391  if (prop == &Visibility && (isUpdateForced() || Visibility.getValue()) && VisualTouched) {
392  updateVisual();
393  // updateVisual() may not be triggered by any change (e.g.
394  // triggered by an external object through forceUpdate()). And
395  // since ShapeAppearance is not changed here either, do not falsely set
396  // the document modified status
397  Base::ObjectStatusLocker<App::Property::Status,App::Property> guard(
398  App::Property::NoModify, &ShapeAppearance);
399  // The material has to be checked again (#0001736)
400  onChanged(&ShapeAppearance);
401  }
402  }
403 
404  ViewProviderGeometryObject::onChanged(prop);
405 }

527  int face = face_detail->getPartIndex() + 1;
528  str << "Face" << face;
529  }

532  int edge = line_detail->getLineIndex() + 1;
533  str << "Edge" << edge;

536  const SoPointDetail* point_detail = static_cast<const SoPointDetail*>(detail);
537  int vertex = point_detail->getCoordinateIndex() - nodeset->startIndex.getValue() + 1;
538  str << "Vertex" << vertex;
539  }
540  }
541

937 void ViewProviderPartExt::updateVisual()
938 {
939  Gui::SoUpdateVBOAction action;
940  action.apply(this->faceset);
941 
942  // Clear selection
943  Gui::SoSelectionElementAction saction(Gui::SoSelectionElementAction::None);
944  saction.apply(this->faceset);
945  saction.apply(this->lineset);
946  saction.apply(this->nodeset);
947 
948  // Clear highlighting
949  Gui::SoHighlightElementAction haction;
950  haction.apply(this->faceset);
951  haction.apply(this->lineset);
952  haction.apply(this->nodeset);
953 
954  TopoDS_Shape cShape = Part::Feature::getShape(getObject());
955  if (cShape.IsNull()) {
956  coords ->point .setNum(0);
957  norm ->vector .setNum(0);
958  faceset ->coordIndex .setNum(0);
959  faceset ->partIndex .setNum(0);
960  lineset ->coordIndex .setNum(0);
961  nodeset ->startIndex .setValue(0);
962  VisualTouched = false;
963  return;
964  }
965 
966  // time measurement and book keeping
967  Base::TimeElapsed start_time;
968  int numTriangles=0,numNodes=0,numNorms=0,numFaces=0,numEdges=0,numLines=0;
969  std::set<int> faceEdges;
970 
971  try {
972  // calculating the deflection value
973  Bnd_Box bounds;
974  BRepBndLib::Add(cShape, bounds);
975  bounds.SetGap(0.0);
976  Standard_Real xMin, yMin, zMin, xMax, yMax, zMax;
977  bounds.Get(xMin, yMin, zMin, xMax, yMax, zMax);
978  Standard_Real deflection = ((xMax-xMin)+(yMax-yMin)+(zMax-zMin))/300.0 * Deviation.getValue();
979 
980  // Since OCCT 7.6 a value of equal 0 is not allowed any more, this can happen if a single vertex
981  // should be displayed.
982  if (deflection < gp::Resolution()) {
983  deflection = Precision::Confusion();
984  }
985 
986  // For very big objects the computed deflection can become very high and thus leads to a useless
987  // tessellation. To avoid this the upper limit is set to 20.0
988  // See also forum: https://forum.freecad.org/viewtopic.php?t=77521
989  //deflection = std::min(deflection, 20.0);
990 
991  // create or use the mesh on the data structure
992  Standard_Real AngDeflectionRads = AngularDeflection.getValue() / 180.0 * M_PI;
993 
994 #if OCC_VERSION_HEX >= 0x070500
995  IMeshTools_Parameters meshParams;
996  meshParams.Deflection = deflection;
997  meshParams.Relative = Standard_False;
998  meshParams.Angle = AngDeflectionRads;
999  meshParams.InParallel = Standard_True;
1000  meshParams.AllowQualityDecrease = Standard_True;
1001 
1002  BRepMesh_IncrementalMesh(cShape, meshParams);
1003 #else
1004  BRepMesh_IncrementalMesh(cShape, deflection, Standard_False, AngDeflectionRads, Standard_True);
1005 #endif
1006 
1007  // We must reset the location here because the transformation data
1008  // are set in the placement property
1009  TopLoc_Location aLoc;
1010  cShape.Location(aLoc);
1011 
1012  // count triangles and nodes in the mesh
1013  TopTools_IndexedMapOfShape faceMap;
1014  TopExp::MapShapes(cShape, TopAbs_FACE, faceMap);
1015  for (int i=1; i <= faceMap.Extent(); i++) {
1016  Handle (Poly_Triangulation) mesh = BRep_Tool::Triangulation(TopoDS::Face(faceMap(i)), aLoc);
1017  if (mesh.IsNull()) {
1018  mesh = Part::Tools::triangulationOfFace(TopoDS::Face(faceMap(i)));
1019  }
1020  // Note: we must also count empty faces
1021  if (!mesh.IsNull()) {
1022  numTriangles += mesh->NbTriangles();
1023  numNodes += mesh->NbNodes();
1024  numNorms += mesh->NbNodes();
1025  }
1026 
1027  TopExp_Explorer xp;
1028  for (xp.Init(faceMap(i),TopAbs_EDGE);xp.More();xp.Next()) {
1029  faceEdges.insert(Part::ShapeMapHasher{}(xp.Current()));
1030  }
1031  numFaces++;
1032  }
1033 
1034  // get an indexed map of edges
1035  TopTools_IndexedMapOfShape edgeMap;
1036  TopExp::MapShapes(cShape, TopAbs_EDGE, edgeMap);
1037 
1038  // key is the edge number, value the coord indexes. This is needed to keep the same order as the edges.
1039  std::map<int, std::vector<int32_t> > lineSetMap;
1040  std::set<int> edgeIdxSet;
1041  std::vector<int32_t> edgeVector;
1042 
1043  // count and index the edges
1044  for (int i=1; i <= edgeMap.Extent(); i++) {
1045  edgeIdxSet.insert(i);
1046  numEdges++;
1047 
1048  const TopoDS_Edge& aEdge = TopoDS::Edge(edgeMap(i));
1049  TopLoc_Location aLoc;
1050 
1051  // handling of the free edge that are not associated to a face
1052  // Note: The assumption that if for an edge BRep_Tool::Polygon3D
1053  // returns a valid object is wrong. This e.g. happens for ruled
1054  // surfaces which gets created by two edges or wires.
1055  // So, we have to store the hashes of the edges associated to a face.
1056  // If the hash of a given edge is not in this list we know it's really
1057  // a free edge.
1058  int hash = Part::ShapeMapHasher{}(aEdge);
1059  if (faceEdges.find(hash) == faceEdges.end()) {
1060  Handle(Poly_Polygon3D) aPoly = Part::Tools::polygonOfEdge(aEdge, aLoc);
1061  if (!aPoly.IsNull()) {
1062  int nbNodesInEdge = aPoly->NbNodes();
1063  numNodes += nbNodesInEdge;
1064  }
1065  }
1066  }
1067 
1068  // handling of the vertices
1069  TopTools_IndexedMapOfShape vertexMap;
1070  TopExp::MapShapes(cShape, TopAbs_VERTEX, vertexMap);
1071  numNodes += vertexMap.Extent();
1072 
1073  // create memory for the nodes and indexes
1074  coords ->point .setNum(numNodes);
1075  norm ->vector .setNum(numNorms);
1076  faceset ->coordIndex .setNum(numTriangles*4);
1077  faceset ->partIndex .setNum(numFaces);
1078  // get the raw memory for fast fill up
1079  SbVec3f* verts = coords ->point .startEditing();
1080  SbVec3f* norms = norm ->vector .startEditing();
1081  int32_t* index = faceset ->coordIndex .startEditing();
1082  int32_t* parts = faceset ->partIndex .startEditing();
1083 
1084  // preset the normal vector with null vector
1085  for (int i=0;i < numNorms;i++)
1086  norms[i]= SbVec3f(0.0,0.0,0.0);
1087 
1088  int ii = 0,faceNodeOffset=0,faceTriaOffset=0;
1089  for (int i=1; i <= faceMap.Extent(); i++, ii++) {
1090  TopLoc_Location aLoc;
1091  const TopoDS_Face &actFace = TopoDS::Face(faceMap(i));
1092  // get the mesh of the shape
1093  Handle (Poly_Triangulation) mesh = BRep_Tool::Triangulation(actFace,aLoc);
1094  if (mesh.IsNull()) {
1095  mesh = Part::Tools::triangulationOfFace(actFace);
1096  }
1097  if (mesh.IsNull()) {
1098  parts[ii] = 0;
1099  continue;
1100  }
1101 
1102  // getting the transformation of the shape/face
1103  gp_Trsf myTransf;
1104  Standard_Boolean identity = true;
1105  if (!aLoc.IsIdentity()) {
1106  identity = false;
1107  myTransf = aLoc.Transformation();
1108  }
1109 
1110  // getting size of node and triangle array of this face
1111  int nbNodesInFace = mesh->NbNodes();
1112  int nbTriInFace = mesh->NbTriangles();
1113  // check orientation
1114  TopAbs_Orientation orient = actFace.Orientation();
1115 
1116 
1117  // cycling through the poly mesh
1118 #if OCC_VERSION_HEX < 0x070600
1119  const Poly_Array1OfTriangle& Triangles = mesh->Triangles();
1120  const TColgp_Array1OfPnt& Nodes = mesh->Nodes();
1121  TColgp_Array1OfDir Normals (Nodes.Lower(), Nodes.Upper());
1122 #else
1123  int numNodes = mesh->NbNodes();
1124  TColgp_Array1OfDir Normals (1, numNodes);
1125 #endif
1126  if (NormalsFromUV)
1127  Part::Tools::getPointNormals(actFace, mesh, Normals);
1128 
1129  for (int g=1;g<=nbTriInFace;g++) {
1130  // Get the triangle
1131  Standard_Integer N1,N2,N3;
1132 #if OCC_VERSION_HEX < 0x070600
1133  Triangles(g).Get(N1,N2,N3);
1134 #else
1135  mesh->Triangle(g).Get(N1,N2,N3);
1136 #endif
1137 
1138  // change orientation of the triangle if the face is reversed
1139  if ( orient != TopAbs_FORWARD ) {
1140  Standard_Integer tmp = N1;
1141  N1 = N2;
1142  N2 = tmp;
1143  }
1144 
1145  // get the 3 points of this triangle
1146 #if OCC_VERSION_HEX < 0x070600
1147  gp_Pnt V1(Nodes(N1)), V2(Nodes(N2)), V3(Nodes(N3));
1148 #else
1149  gp_Pnt V1(mesh->Node(N1)), V2(mesh->Node(N2)), V3(mesh->Node(N3));
1150 #endif
1151 
1152  // get the 3 normals of this triangle
1153  gp_Vec NV1, NV2, NV3;
1154  if (NormalsFromUV) {
1155  NV1.SetXYZ(Normals(N1).XYZ());
1156  NV2.SetXYZ(Normals(N2).XYZ());
1157  NV3.SetXYZ(Normals(N3).XYZ());
1158  }
1159  else {
1160  gp_Vec v1(V1.X(),V1.Y(),V1.Z()),
1161  v2(V2.X(),V2.Y(),V2.Z()),
1162  v3(V3.X(),V3.Y(),V3.Z());
1163  gp_Vec normal = (v2-v1)^(v3-v1);
1164  NV1 = normal;
1165  NV2 = normal;
1166  NV3 = normal;
1167  }
1168 
1169  // transform the vertices and normals to the place of the face
1170  if (!identity) {
1171  V1.Transform(myTransf);
1172  V2.Transform(myTransf);
1173  V3.Transform(myTransf);
