153 def convert_raw_data_to_mesh_data(raw_mesh_data):
154  """
155  Converts raw dictionary data from JSON or YAML file to proper dict
156  for importToolsFem.make_femmesh(mesh_data). This is necessary since
157  JSON and YAML save dict keys as strings while make_femmesh expects
158  integers.
159  """
160 
161  mesh_data = {}
162  for type_key, type_dict in raw_mesh_data.items():
163  if type_key.lower() != "groups":
164  mesh_data[type_key] = {int(k): v for (k, v) in type_dict.items()}
165  return mesh_data
