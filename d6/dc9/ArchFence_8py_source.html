175  def calculateSections(self, obj, postPlacements, postLength, sectionLength):
176  import Part
177 
178  shapes = []
179 
180  # For the colorization algorithm we have to store the number of faces for each section
181  # It is possible that a section is clipped. Then the number of faces is not equal to the
182  # number of faces in the original section
183  faceNumbers = []
184 
185  for i in range(obj.NumberOfSections):
186  startPlacement = postPlacements[i]
187  endPlacement = postPlacements[i + 1]
188 
189  sectionLine = Part.LineSegment(
190  startPlacement.Base, endPlacement.Base)
191  sectionBase = sectionLine.value(postLength)
192 
193  if startPlacement.Rotation.isSame(endPlacement.Rotation):
194  sectionRotation = endPlacement.Rotation
195  else:
196  direction = endPlacement.Base.sub(startPlacement.Base)
197 
198  sectionRotation = FreeCAD.Rotation(EAST, direction)
199 
200  placement = FreeCAD.Placement()
201  placement.Base = sectionBase
202  placement.Rotation = sectionRotation
203 
204  sectionCopy = obj.Section.Shape.copy()
205 
206  if sectionLength > sectionLine.length() - postLength:
207  # Part.show(Part.Shape([sectionLine]), 'line')
208  sectionCopy = self.clipSection(
209  sectionCopy, sectionLength, sectionLine.length() - postLength)
210 
211  sectionCopy = Part.Compound([sectionCopy]) # nest in compound to ensure correct Placement
212  sectionCopy.Placement = placement

214  shapes.append(sectionCopy)

219  def clipSection(self, shape, length, clipLength):
220  import Part
221 
222  boundBox = shape.BoundBox
223  lengthToCut = length - clipLength
224  halfLengthToCut = lengthToCut / 2
225 
226  leftBox = Part.makeBox(halfLengthToCut, boundBox.YMax + 1, boundBox.ZMax + 1,
227  FreeCAD.Vector(boundBox.XMin, boundBox.YMin, boundBox.ZMin))
228  rightBox = Part.makeBox(halfLengthToCut, boundBox.YMax + 1, boundBox.ZMax + 1,
229  FreeCAD.Vector(boundBox.XMin + halfLengthToCut + clipLength, boundBox.YMin, boundBox.ZMin))
230 
231  newShape = shape.cut([leftBox, rightBox])
232  newBoundBox = newShape.BoundBox
233 
234  newShape.translate(FreeCAD.Vector(-newBoundBox.XMin, 0, 0))
235 
236  return newShape.removeSplitter()

303  def applyColors(self, obj):
304  # Note that the clipSection function changes the face numbering of the
305  # fence section. This happens even if the total number of faces does not
306  # change. If UseOriginalColors is True, the end result of this function
307  # will only be correct if all faces of the section have the same color.
308

311  vobj.ShapeAppearance = [vobj.ShapeAppearance[0]]
312  else:

314  section = obj.Section
315 
316  # If post and/or section are Std_Parts they may not have a Shape attr (yet):
317  if not hasattr(post, "Shape"):
318  return
319  if not hasattr(section, "Shape"):
320  return
321 
322  numberOfPostFaces = len(post.Shape.Faces)
323  numberOfSectionFaces = len(section.Shape.Faces)
324 
325  if hasattr(obj.Proxy, 'sectionFaceNumbers'):
326  sectionFaceNumbers = obj.Proxy.sectionFaceNumbers
327  else:
328  sectionFaceNumbers = [0]
329 
330  if numberOfPostFaces == 0 or sum(sectionFaceNumbers) == 0:
331  return
332

334  defaultSectionColors = self.normalizeColors(
335  section, numberOfSectionFaces)
336 
337  ownColors = []

340  for i in range(obj.NumberOfPosts):

355  def normalizeColors(self, obj, numberOfFaces):
356  if obj.TypeId == "PartDesign::Body":
357  # When colorizing a PartDesign Body we have two options
358  # 1. The whole body got a shape color, that means the tip has only a single diffuse color set
359  # so we use the shape color of the body
360  # 2. "Set colors" was called on the tip and the individual faces where colorized.
361  # We use the diffuseColors of the tip in that case
362  if len(obj.Tip.ViewObject.DiffuseColor) > 1:
363  colors = obj.Tip.ViewObject.DiffuseColor
364  else:
365  colors = obj.ViewObject.DiffuseColor
366  else:
367  import Draft
368  colors = Draft.get_diffuse_color(obj) # To handle Std_Parts for example.
369 
370  numberOfColors = len(colors)
371 
372  if numberOfColors == 1:
373  return colors * numberOfFaces
374 
375  if numberOfColors == numberOfFaces:
376  return colors
377 
378  # It is possible, that we have fewer faces than colors when something
379  # got clipped. Remove the unneeded colors at the beginning and end.
380 
381  # Even if clipSection did not change the face numbering this code would
382  # not work properly.
383  halfNumberOfFacesToRemove = (numberOfColors - numberOfFaces) / 2
384  start = int(math.ceil(halfNumberOfFacesToRemove))
385  end = start + numberOfFaces
386  return colors[start:end]
