126 void ViewProviderSplineExtension::showControlPoints(bool show, const App::Property* prop)
127 {
128  if (!pcControlPoints && show) {
129  pcControlPoints = new SoSwitch();
130  SoSeparator* root = getExtendedViewProvider()->getRoot();
131  root->addChild(pcControlPoints);
132  }
133 
134  if (pcControlPoints) {
135  pcControlPoints->whichChild = (show ? SO_SWITCH_ALL : SO_SWITCH_NONE);
136  }
137 
138  if (!show || !pcControlPoints || pcControlPoints->getNumChildren() > 0)
139  return;
140 
141  // ask for the property we are interested in
142  if (prop && prop->is<Part::PropertyPartShape>()) {
143  const TopoDS_Shape& shape = static_cast<const Part::PropertyPartShape*>(prop)->getValue();
144  if (shape.IsNull())
145  return; // empty shape
146 
147  for (TopExp_Explorer xp(shape, TopAbs_SHELL); xp.More(); xp.Next()) {
148  const TopoDS_Shell& shell = TopoDS::Shell(xp.Current());
149  for (TopExp_Explorer xp2(shell, TopAbs_FACE); xp2.More(); xp2.Next()) {
150  const TopoDS_Face& face = TopoDS::Face(xp2.Current());
151  showControlPointsOfFace(face);
152  }
153  }
154  for (TopExp_Explorer xp(shape, TopAbs_FACE, TopAbs_SHELL); xp.More(); xp.Next()) {
155  const TopoDS_Face& face = TopoDS::Face(xp.Current());
156  showControlPointsOfFace(face);
157  }
158  for (TopExp_Explorer xp(shape, TopAbs_WIRE, TopAbs_FACE); xp.More(); xp.Next()) {
159  const TopoDS_Wire& wire = TopoDS::Wire(xp.Current());
160  for (TopExp_Explorer xp2(wire, TopAbs_EDGE); xp2.More(); xp2.Next()) {
161  const TopoDS_Edge& edge = TopoDS::Edge(xp2.Current());
162  showControlPointsOfEdge(edge);
163  }
164  }
165  for (TopExp_Explorer xp(shape, TopAbs_EDGE, TopAbs_WIRE); xp.More(); xp.Next()) {
166  const TopoDS_Edge& edge = TopoDS::Edge(xp.Current());
167  showControlPointsOfEdge(edge);
168  }
169  }
170 }

172 void ViewProviderSplineExtension::showControlPointsOfEdge(const TopoDS_Edge& edge)
173 {
174  std::list<gp_Pnt> poles, knots;
175  Standard_Integer nCt=0;
176 
177  TopoDS_Edge edge_loc(edge);
178  TopLoc_Location aLoc;
179  edge_loc.Location(aLoc);
180 
181  BRepAdaptor_Curve curve(edge_loc);
182  switch (curve.GetType())
183  {
184  case GeomAbs_BezierCurve:
185  {
186  Handle(Geom_BezierCurve) hBezier = curve.Bezier();
187  nCt = hBezier->NbPoles();
188  for (Standard_Integer i = 1; i <= nCt; i++)
189  poles.push_back(hBezier->Pole(i));
190  if (hBezier->IsClosed()) {
191  nCt++;
192  poles.push_back(hBezier->Pole(1));
193  }
194  } break;
195  case GeomAbs_BSplineCurve:
196  {
197  Handle(Geom_BSplineCurve) hBSpline = curve.BSpline();
198  nCt = hBSpline->NbPoles();
199  for (Standard_Integer i = 1; i <= nCt; i++)
200  poles.push_back(hBSpline->Pole(i));
201  if (hBSpline->IsClosed()) {
202  nCt++;
203  poles.push_back(hBSpline->Pole(1));
204  }
205  for (Standard_Integer i = hBSpline->FirstUKnotIndex()+1; i <= hBSpline->LastUKnotIndex()-1; i++)
206  knots.push_back(hBSpline->Value(hBSpline->Knot(i)));
207  } break;
208  default:
209  break;
210  }
211 
212  if (poles.empty())
213  return; // nothing to do
214 
215  SoCoordinate3 * controlcoords = new SoCoordinate3;
216  controlcoords->point.setNum(nCt + knots.size());
217 
218  int index=0;
219  SbVec3f* verts = controlcoords->point.startEditing();
220  for (const auto & pole : poles) {
221  verts[index++].setValue((float)pole.X(), (float)pole.Y(), (float)pole.Z());
222  }
223  for (const auto & knot : knots) {
224  verts[index++].setValue((float)knot.X(), (float)knot.Y(), (float)knot.Z());
225  }
226  controlcoords->point.finishEditing();
227 
228 
229  SoFCControlPoints* controlpoints = new SoFCControlPoints();
230  controlpoints->numPolesU = nCt;
231  controlpoints->numPolesV = 1;
232 
233  SoSeparator* nodes = new SoSeparator();
234  nodes->addChild(controlcoords);
235  nodes->addChild(controlpoints);
236 
237  pcControlPoints->addChild(nodes);
238 }

240 void ViewProviderSplineExtension::showControlPointsOfFace(const TopoDS_Face& face)
241 {
242  std::list<gp_Pnt> knots;
243  std::vector<std::vector<gp_Pnt> > poles;
244  Standard_Integer nCtU=0, nCtV=0;
245 
246  TopoDS_Face face_loc(face);
247  TopLoc_Location aLoc;
248  face_loc.Location(aLoc);
249 
250  BRepAdaptor_Surface surface(face_loc);
251  switch (surface.GetType())
252  {
253  case GeomAbs_BezierSurface:
254  {
255  Handle(Geom_BezierSurface) hBezier = surface.Bezier();
256  nCtU = hBezier->NbUPoles();
257  nCtV = hBezier->NbVPoles();
258  poles.resize(nCtU);
259  for (Standard_Integer u = 1; u <= nCtU; u++) {
260  poles[u-1].resize(nCtV);
261  for (Standard_Integer v = 1; v <= nCtV; v++)
262  poles[u-1][v-1] = hBezier->Pole(u, v);
263  }
264  } break;
265  case GeomAbs_BSplineSurface:
266  {
267  Handle(Geom_BSplineSurface) hBSpline = surface.BSpline();
268  nCtU = hBSpline->NbUPoles();
269  nCtV = hBSpline->NbVPoles();
270  poles.resize(nCtU);
271  for (Standard_Integer u = 1; u <= nCtU; u++) {
272  poles[u-1].resize(nCtV);
273  for (Standard_Integer v = 1; v <= nCtV; v++)
274  poles[u-1][v-1] = hBSpline->Pole(u, v);
275  }
276 
277  //Standard_Integer nKnU = hBSpline->NbUKnots();
278  //Standard_Integer nKnV = hBSpline->NbVKnots();
279  for (Standard_Integer u = 1; u <= hBSpline->NbUKnots(); u++) {
280  for (Standard_Integer v = 1; v <= hBSpline->NbVKnots(); v++)
281  knots.push_back(hBSpline->Value(hBSpline->UKnot(u), hBSpline->VKnot(v)));
282  }
283  } break;
284  default:
285  break;
286  }
287 
288  if (poles.empty())
289  return; // nothing to do
290 
291  SoCoordinate3 * coords = new SoCoordinate3;
292  coords->point.setNum(nCtU * nCtV + knots.size());
293 
294  int index=0;
295  SbVec3f* verts = coords->point.startEditing();
296  for (const auto & pole : poles) {
297  for (const auto& v : pole) {
298  verts[index++].setValue((float)v.X(), (float)v.Y(), (float)v.Z());
299  }
300  }
301  for (const auto & knot : knots) {
302  verts[index++].setValue((float)knot.X(), (float)knot.Y(), (float)knot.Z());
303  }
304  coords->point.finishEditing();
305 
306 
307  SoFCControlPoints* control = new SoFCControlPoints();
308  control->numPolesU = nCtU;
309  control->numPolesV = nCtV;
310 
311  //if (knots.size() > 0) {
312  // control->numKnotsU = nKnU;
313  // control->numKnotsV = nKnV;
314  //}
315 
316  SoSeparator* nodes = new SoSeparator();
317  nodes->addChild(coords);
318  nodes->addChild(control);
319 
320  pcControlPoints->addChild(nodes);
321 }
