90 def getCutShapes(objs,cutplane,onlySolids,clip,joinArch,showHidden,groupSshapesByObject=False):
91 
92  """
93  returns a list of shapes (visible, hidden, cut lines...)
94  obtained from performing a series of booleans against the given cut plane
95  """
96

99  hshapes = []
100  sshapes = []
101  objectShapes = []
102  objectSshapes = []
103 
104  if joinArch:
105  shtypes = {}
106  for o in objs:
107  if Draft.getType(o) in ["Wall","Structure"]:
108  if o.Shape.isNull():
109  pass
110  elif onlySolids:
111  shtypes.setdefault(o.Material.Name if (hasattr(o,"Material") and o.Material) else "None",[]).extend(o.Shape.Solids)
112  else:
113  shtypes.setdefault(o.Material.Name if (hasattr(o,"Material") and o.Material) else "None",[]).append(o.Shape.copy())
114  elif hasattr(o,'Shape'):
115  if o.Shape.isNull():
116  pass
117  elif onlySolids:
118  shapes.extend(o.Shape.Solids)
119  objectShapes.append((o, o.Shape.Solids))
120  else:
121  shapes.append(o.Shape.copy())
122  objectShapes.append((o,[o.Shape.copy()]))
123  for k,v in shtypes.items():
124  v1 = v.pop()
125  if v:
126  v1 = v1.multiFuse(v)
127  v1 = v1.removeSplitter()
128  if v1.Solids:
129  shapes.extend(v1.Solids)
130  objectShapes.append((k,v1.Solids))
131  else:
132  print("ArchSectionPlane: Fusing Arch objects produced non-solid results")
133  shapes.append(v1)
134  objectShapes.append((k,[v1]))
135  else:
136  for o in objs:
137  if hasattr(o,'Shape'):
138  if o.Shape.isNull():
139  pass
140  elif onlySolids:
141  if o.Shape.isValid():
142  shapes.extend(o.Shape.Solids)
143  objectShapes.append((o,o.Shape.Solids))
144  else:
145  shapes.append(o.Shape)
146  objectShapes.append((o,[o.Shape]))
147 
148  cutface,cutvolume,invcutvolume = ArchCommands.getCutVolume(cutplane,shapes,clip)
149  shapes = []
150  for o, shapeList in objectShapes:
151  tmpSshapes = []
152  for sh in shapeList:
153  for sub in (sh.SubShapes if sh.ShapeType == "Compound" else [sh]):
154  if cutvolume:
155  if sub.Volume < 0:
156  sub = sub.reversed() # Use reversed as sub is immutable.
157  c = sub.cut(cutvolume)
158  s = sub.section(cutface)
159  try:
160  wires = DraftGeomUtils.findWires(s.Edges)
161  for w in wires:
162  f = Part.Face(w)
163  tmpSshapes.append(f)
164  except Part.OCCError:
165  #print "ArchView: unable to get a face"
166  tmpSshapes.append(s)
167  shapes.extend(c.SubShapes if c.ShapeType == "Compound" else [c])
168  if showHidden:
169  c = sub.cut(invcutvolume)
170  hshapes.extend(c.SubShapes if c.ShapeType == "Compound" else [c])
171  else:
172  shapes.append(sub)
173 
174  if len(tmpSshapes) > 0:
175  sshapes.extend(tmpSshapes)
176 
177  if groupSshapesByObject:
178  objectSshapes.append((o, tmpSshapes))
179 
180  if groupSshapesByObject:
181  return shapes,hshapes,sshapes,cutface,cutvolume,invcutvolume,objectSshapes
182  else:
183  return shapes,hshapes,sshapes,cutface,cutvolume,invcutvolume

206 def isOriented(obj,plane):
207 
208  """determines if an annotation is facing the cutplane or not"""
209 
210  norm1 = plane.normalAt(0,0)
211  if hasattr(obj,"Placement"):
212  norm2 = obj.Placement.Rotation.multVec(FreeCAD.Vector(0,0,1))
213  elif hasattr(obj,"Normal"):
214  norm2 = obj.Normal
215  if norm2.Length < 0.01:
216  return True
217  else:
218  return True
219  a = norm1.getAngle(norm2)
220  if (a < 0.01) or (abs(a-math.pi) < 0.01):
221  return True
222  return False
