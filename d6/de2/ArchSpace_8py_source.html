417  def getFootprint(self,obj):
418 
419  "returns a face that represents the footprint of this space at the center of mass"
420 
421  import Part
422  import DraftGeomUtils
423  if not hasattr(obj.Shape,"CenterOfMass"):
424  return None
425  try:
426  pl = Part.makePlane(1,1)
427  pl.translate(obj.Shape.CenterOfMass)
428  sh = obj.Shape.copy()
429  cutplane,v1,v2 = ArchCommands.getCutVolume(pl,sh)
430  e = sh.section(cutplane)
431  e = Part.__sortEdges__(e.Edges)
432  w = Part.Wire(e)
433  dv = FreeCAD.Vector(obj.Shape.CenterOfMass.x,obj.Shape.CenterOfMass.y,obj.Shape.BoundBox.ZMin)
434  dv = dv.sub(obj.Shape.CenterOfMass)
435  w.translate(dv)
436  return Part.Face(w)
437  except Part.OCCError:
438  return None
