147 void CmdApproxPlane::activated(int)
148 {
149  std::vector<App::GeoFeature*> obj = Gui::Selection().getObjectsOfType<App::GeoFeature>();
150  for (const auto& it : obj) {
151  std::vector<Base::Vector3d> aPoints;
152  std::vector<Base::Vector3d> aNormals;
153 
154  std::vector<App::Property*> List;
155  it->getPropertyList(List);
156  for (const auto& jt : List) {
157  if (jt->isDerivedFrom<App::PropertyComplexGeoData>()) {
158  const Data::ComplexGeoData* data =
159  static_cast<App::PropertyComplexGeoData*>(jt)->getComplexData();
160  if (data) {
161  data->getPoints(aPoints, aNormals, 0.01f);
162  if (!aPoints.empty()) {
163  break;
164  }
165  }
166  }
167  }
168 
169  if (!aPoints.empty()) {
170  // get a reference normal for the plane fit
171  Base::Vector3f refNormal(0, 0, 0);
172  if (!aNormals.empty()) {
173  refNormal = Base::convertTo<Base::Vector3f>(aNormals.front());
174  }
175 
176  std::vector<Base::Vector3f> aData;
177  aData.reserve(aPoints.size());
178  for (const auto& jt : aPoints) {
179  aData.push_back(Base::toVector<float>(jt));
180  }
181  MeshCore::PlaneFit fit;
182  fit.AddPoints(aData);
183  float sigma = fit.Fit();
184  Base::Vector3f base = fit.GetBase();
185  Base::Vector3f dirU = fit.GetDirU();
186  Base::Vector3f dirV = fit.GetDirV();
187  Base::Vector3f norm = fit.GetNormal();
188 
189  // if the dot product of the reference with the plane normal is negative
190  // a flip must be done
191  if (refNormal * norm < 0) {
192  norm = -norm;
193  dirU = -dirU;
194  }
195 
196  float length, width;
197  fit.Dimension(length, width);
198 
199  // move to the corner point
200  base = base - (0.5f * length * dirU + 0.5f * width * dirV);
201 
202  Base::CoordinateSystem cs;
203  cs.setPosition(Base::convertTo<Base::Vector3d>(base));
204  cs.setAxes(Base::convertTo<Base::Vector3d>(norm),
205  Base::convertTo<Base::Vector3d>(dirU));
206  Base::Placement pm = Base::CoordinateSystem().displacement(cs);
207  double q0, q1, q2, q3;
208  pm.getRotation().getValue(q0, q1, q2, q3);
209 
210  Base::Console().Log("RMS value for plane fit with %lu points: %.4f\n",
211  aData.size(),
212  sigma);
213  Base::Console().Log(" Plane base(%.4f, %.4f, %.4f)\n", base.x, base.y, base.z);
214  Base::Console().Log(" Plane normal(%.4f, %.4f, %.4f)\n", norm.x, norm.y, norm.z);
215 
216  std::stringstream str;
217  str << "from FreeCAD import Base" << std::endl;
218  str << "App.ActiveDocument.addObject('Part::Plane','Plane_fit')" << std::endl;
219  str << "App.ActiveDocument.ActiveObject.Length = " << length << std::endl;
220  str << "App.ActiveDocument.ActiveObject.Width = " << width << std::endl;
221  str << "App.ActiveDocument.ActiveObject.Placement = Base.Placement("
222  << "Base.Vector(" << base.x << "," << base.y << "," << base.z << "),"
223  << "Base.Rotation(" << q0 << "," << q1 << "," << q2 << "," << q3 << "))"
224  << std::endl;
225 
226  openCommand(QT_TRANSLATE_NOOP("Command", "Fit plane"));
227  runCommand(Gui::Command::Doc, str.str().c_str());
228  commitCommand();
229  updateActive();
230  }
231  }
232 }

255 void CmdApproxCylinder::activated(int)
256 {
257  std::vector<Mesh::Feature*> sel = getSelection().getObjectsOfType<Mesh::Feature>();
258  openCommand(QT_TRANSLATE_NOOP("Command", "Fit cylinder"));
259  for (auto it : sel) {
260  const Mesh::MeshObject& mesh = it->Mesh.getValue();
261  const MeshCore::MeshKernel& kernel = mesh.getKernel();
262  MeshCore::CylinderFit fit;
263  fit.AddPoints(kernel.GetPoints());
264 
265  // get normals
266  {
267  std::vector<MeshCore::FacetIndex> facets(kernel.CountFacets());
268  std::generate(facets.begin(), facets.end(), Base::iotaGen<MeshCore::FacetIndex>(0));
269  std::vector<Base::Vector3f> normals = kernel.GetFacetNormals(facets);
270  Base::Vector3f base = fit.GetGravity();
271  Base::Vector3f axis = fit.GetInitialAxisFromNormals(normals);
272  fit.SetInitialValues(base, axis);
273  }
274 
275  if (fit.Fit() < FLOAT_MAX) {
276  Base::Vector3f base, top;
277  fit.GetBounding(base, top);
278  float height = Base::Distance(base, top);
279 
280  Base::Rotation rot;
281  rot.setValue(Base::Vector3d(0, 0, 1), Base::convertTo<Base::Vector3d>(fit.GetAxis()));
282  double q0, q1, q2, q3;
283  rot.getValue(q0, q1, q2, q3);
284 
285  std::stringstream str;
286  str << "from FreeCAD import Base" << std::endl;
287  str << "App.ActiveDocument.addObject('Part::Cylinder','Cylinder_fit')" << std::endl;
288  str << "App.ActiveDocument.ActiveObject.Radius = " << fit.GetRadius() << std::endl;
289  str << "App.ActiveDocument.ActiveObject.Height = " << height << std::endl;
290  str << "App.ActiveDocument.ActiveObject.Placement = Base.Placement("
291  << "Base.Vector(" << base.x << "," << base.y << "," << base.z << "),"
292  << "Base.Rotation(" << q0 << "," << q1 << "," << q2 << "," << q3 << "))"
293  << std::endl;
294 
295  runCommand(Gui::Command::Doc, str.str().c_str());
296  }
297  }
298  commitCommand();
299  updateActive();
300 }

373  std::vector<Mesh::Feature*> sel = getSelection().getObjectsOfType<Mesh::Feature>();
374  App::Document* doc = App::GetApplication().getActiveDocument();
375  openCommand(QT_TRANSLATE_NOOP("Command", "Fit polynomial surface"));
376  for (auto it : sel) {
377  const Mesh::MeshObject& mesh = it->Mesh.getValue();
378  const MeshCore::MeshKernel& kernel = mesh.getKernel();
379  MeshCore::SurfaceFit fit;
380  fit.AddPoints(kernel.GetPoints());
381  if (fit.Fit() < FLOAT_MAX) {
382  Base::BoundBox3f bbox = fit.GetBoundings();
383  std::vector<Base::Vector3d> poles =
384  fit.toBezier(bbox.MinX, bbox.MaxX, bbox.MinY, bbox.MaxY);
385  fit.Transform(poles);
386 
387  TColgp_Array2OfPnt grid(1, 3, 1, 3);
388  grid.SetValue(1, 1, Base::convertTo<gp_Pnt>(poles.at(0)));
389  grid.SetValue(2, 1, Base::convertTo<gp_Pnt>(poles.at(1)));
390  grid.SetValue(3, 1, Base::convertTo<gp_Pnt>(poles.at(2)));
391  grid.SetValue(1, 2, Base::convertTo<gp_Pnt>(poles.at(3)));
392  grid.SetValue(2, 2, Base::convertTo<gp_Pnt>(poles.at(4)));
393  grid.SetValue(3, 2, Base::convertTo<gp_Pnt>(poles.at(5)));
394  grid.SetValue(1, 3, Base::convertTo<gp_Pnt>(poles.at(6)));
395  grid.SetValue(2, 3, Base::convertTo<gp_Pnt>(poles.at(7)));
396  grid.SetValue(3, 3, Base::convertTo<gp_Pnt>(poles.at(8)));
397 
398  Handle(Geom_BezierSurface) bezier(new Geom_BezierSurface(grid));
399  Part::Feature* part =
400  static_cast<Part::Feature*>(doc->addObject("Part::Spline", "Bezier"));
401  part->Shape.setValue(Part::GeomBezierSurface(bezier).toShape());

543 void CmdMeshBoundary::activated(int)
544 {
545  std::vector<Mesh::Feature*> objs = Gui::Selection().getObjectsOfType<Mesh::Feature>();
546  App::Document* document = App::GetApplication().getActiveDocument();
547  document->openTransaction("Wire from mesh");
548  for (auto it : objs) {
549  const Mesh::MeshObject& mesh = it->Mesh.getValue();
550  std::list<std::vector<Base::Vector3f>> bounds;
551  MeshCore::MeshAlgorithm algo(mesh.getKernel());
552  algo.GetMeshBorders(bounds);
553 
554  BRep_Builder builder;
555  TopoDS_Compound compound;
556  builder.MakeCompound(compound);
557 
558  TopoDS_Shape shape;
559  std::vector<TopoDS_Wire> wires;
560 
561  for (const auto& bt : bounds) {
562  BRepBuilderAPI_MakePolygon mkPoly;
563  for (auto it = bt.rbegin(); it != bt.rend(); ++it) {
564  mkPoly.Add(gp_Pnt(it->x, it->y, it->z));
565  }
566  if (mkPoly.IsDone()) {
567  builder.Add(compound, mkPoly.Wire());
568  wires.push_back(mkPoly.Wire());
569  }
570  }
571 
572  try {
573  shape = Part::FaceMakerCheese::makeFace(wires);
574  }
575  catch (...) {
576  }
577 
578  if (!shape.IsNull()) {
579  Part::Feature* shapeFea =
580  static_cast<Part::Feature*>(document->addObject("Part::Feature", "Face from mesh"));
581  shapeFea->Shape.setValue(shape);
582  }
583  else {
584  Part::Feature* shapeFea =
585  static_cast<Part::Feature*>(document->addObject("Part::Feature", "Wire from mesh"));
586  shapeFea->Shape.setValue(compound);
587  }
588  }
589  document->commitTransaction();
590 }
