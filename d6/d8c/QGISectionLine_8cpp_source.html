206 //arrows go at arrowhead position.
207 void QGISectionLine::makeSymbolsTrad()
208 {
209  prepareGeometryChange();
210  int fontSize = QGIView::exactFontSize(m_symFont.family().toStdString(), m_symSize);
211  m_symFont.setPixelSize(fontSize);
212  m_symbol1->setFont(m_symFont);
213  m_symbol1->setPlainText(QString::fromUtf8(m_symbol));
214  m_symbol2->setFont(m_symFont);
215  m_symbol2->setPlainText(QString::fromUtf8(m_symbol));
216 
217  QRectF symRect = m_symbol1->boundingRect();
218  double symHeight = symRect.height();
219  double gap = 0.5 * symHeight; //symHeight as surrogate for char box
220 
221  QPointF motion1(m_arrowDir1.x, -m_arrowDir1.y); //move in same direction as arrow
222  QPointF motion2(m_arrowDir2.x, -m_arrowDir2.y); //Qt y coords!
223 
224  QPointF symPos1 = m_arrowPos1 + motion1 * gap;
225  QPointF symPos2 = m_arrowPos2 + motion2 * gap;
226 
227  m_symbol1->centerAt(symPos1);
228  m_symbol2->centerAt(symPos2);
229 
230  m_symbol1->setTransformOriginPoint(m_symbol1->mapFromParent(symPos1));
231  m_symbol1->setRotation(360.0 - rotation()); //to Qt angle
232  m_symbol2->setTransformOriginPoint(m_symbol2->mapFromParent(symPos2));
233  m_symbol2->setRotation(360.0 - rotation());
234 }

266 //extension lines are on the stock side of the section line
267 void QGISectionLine::extensionEndsTrad()
268 {
269  if (m_arrowMode == SINGLEDIRECTIONMODE) {
270  QPointF offsetDir(m_arrowDir.x, -m_arrowDir.y); //inverted Y
271  offsetDir = normalizeQPointF(offsetDir);
272 
273  //draw from section line endpoint
274  QPointF offsetEnd = m_extLen * offsetDir;
275  m_beginExt1 = m_start;
276  m_endExt1 = m_start + offsetEnd;
277  m_beginExt2 = m_end;
278  m_endExt2 = m_end + offsetEnd;
279  } else {
280  //extension lines run from point on section line to arrowhead
281  m_beginExt1 = m_start;
282  m_endExt1 = getArrowPosition(m_arrowDir1, m_start);
283  m_beginExt2 = m_end;
284  m_endExt2 = getArrowPosition(m_arrowDir2, m_end);
285  }
286 }

288 //the extension lines are on the waste side of the section line!
289 void QGISectionLine::extensionEndsISO()
290 {
291  if (m_arrowMode == SINGLEDIRECTIONMODE) {
292  QPointF offsetDir(-m_arrowDir.x, m_arrowDir.y); //reversed and inverted y
293  offsetDir = normalizeQPointF(offsetDir);
294 
295  //draw from section line endpoint less arrow length
296  QPointF offsetStart = offsetDir * Rez::guiX(QGIArrow::getPrefArrowSize());
297  QPointF offsetEnd = m_extLen * offsetDir;
298 
299  m_beginExt1 = m_start + offsetStart;
300  m_endExt1 = m_start + offsetStart + offsetEnd;
301  m_beginExt2 = m_end + offsetStart;
302  m_endExt2 = m_end + offsetStart + offsetEnd;
303  } else {
304  //extension lines run in reverse of arrow direction from base of arrowhead for distance m_extLen
305  QPointF offsetDir1(-m_arrowDir1.x, m_arrowDir1.y); //reversed and inverted y
306  offsetDir1 = normalizeQPointF(offsetDir1);
307  QPointF offsetStart1 = offsetDir1 * Rez::guiX(QGIArrow::getPrefArrowSize());
308  QPointF offsetEnd1 = m_extLen * offsetDir1;
309  m_beginExt1 = m_start + offsetStart1;
310  m_endExt1 = m_start + offsetStart1 + offsetEnd1;
311 
312  QPointF offsetDir2(-m_arrowDir2.x, m_arrowDir2.y); //reversed and inverted y
313  offsetDir2 = normalizeQPointF(offsetDir2);
314  QPointF offsetStart2 = offsetDir2 * Rez::guiX(QGIArrow::getPrefArrowSize());
315  QPointF offsetEnd2 = m_extLen * offsetDir2;
316  m_beginExt2 = m_end + offsetStart2;
317  m_endExt2 = m_end + offsetStart2 + offsetEnd2;

321 void QGISectionLine::makeChangePointMarks()
322 {
323 // Base::Console().Message("QGISL::makeChangePointMarks()\n");
324  double segmentLength = 0.50 * QGIArrow::getPrefArrowSize();
325  QPen cPointPen;
326  //TODO: this should really be 2.0 * thickLineWidth, but we only have one
327  //width available (which should be 'thin', for the section line)
328  cPointPen.setWidthF(2.0 * getWidth());
329  cPointPen.setColor(getSectionColor());
330  cPointPen.setStyle(Qt::SolidLine);
331  for (auto& cPoint : m_changePointData) {
332  QGraphicsPathItem* cPointItem = new QGraphicsPathItem();
333  addToGroup(cPointItem);
334 
335  QPainterPath pPath;
336  QPointF location = cPoint.getLocation();
337  QPointF start = location + cPoint.getPreDirection() * segmentLength;
338  QPointF end = location + cPoint.getPostDirection() * segmentLength;
339  pPath.moveTo(Rez::guiPt(start));
340  pPath.lineTo(Rez::guiPt(location));
341  pPath.lineTo(Rez::guiPt(end));
342 
343  cPointItem->setPath(pPath);
344  cPointItem->setPen(cPointPen);
345  cPointItem->setZValue(ZVALUE::SECTIONLINE + 1);
346  cPointItem->setPos(0.0, 0.0);
347 
348  cPointItem->setRotation(rotation());
349 
350  m_changePointMarks.push_back(cPointItem);
351  }
352 }

399 //convert an arrow direction vector into a Qt rotation angle degrees
400 double QGISectionLine::getArrowRotation(Base::Vector3d arrowDir)
401 {
402  arrowDir.Normalize();
403  double angle = atan2f(arrowDir.y, arrowDir.x);
404  if (angle < 0.0) {
405  angle = 2 * M_PI + angle;
406  }
407  double arrowRotation = 360.0 - angle * (180.0/M_PI); //convert to Qt rotation (clockwise degrees)
408  return arrowRotation;
409 }

457 //QPointF does not have length or normalize methods
458 QPointF QGISectionLine::normalizeQPointF(QPointF inPoint)
459 {
460  double x2 = inPoint.x() * inPoint.x();
461  double y2 = inPoint.y() * inPoint.y();
462  double root = sqrt(x2 + y2);
463  return inPoint / root;
464 }
