21 Box3<Real> ContAlignedBox (int iQuantity, const Vector3<Real>* akPoint)
22 {
23  Vector3<Real> kMin, kMax;
24  Vector3<Real>::ComputeExtremes(iQuantity,akPoint,kMin,kMax);
25 
26  Box3<Real> kBox;
27  kBox.Center = ((Real)0.5)*(kMin + kMax);
28  kBox.Axis[0] = Vector3<Real>::UNIT_X;
29  kBox.Axis[1] = Vector3<Real>::UNIT_Y;
30  kBox.Axis[2] = Vector3<Real>::UNIT_Z;
31  Vector3<Real> kHalfDiagonal = ((Real)0.5)*(kMax - kMin);
32  for (int i = 0; i < 3; i++)
33  {
34  kBox.Extent[i] = kHalfDiagonal[i];
35  }
36 
37  return kBox;

40 template <class Real>
41 Box3<Real> ContOrientedBox (int iQuantity, const Vector3<Real>* akPoint)
42 {
43  Box3<Real> kBox = GaussPointsFit3<Real>(iQuantity,akPoint);
44 
45  // Let C be the box center and let U0, U1, and U2 be the box axes. Each
46  // input point is of the form X = C + y0*U0 + y1*U1 + y2*U2. The
47  // following code computes min(y0), max(y0), min(y1), max(y1), min(y2),
48  // and max(y2). The box center is then adjusted to be
49  // C' = C + 0.5*(min(y0)+max(y0))*U0 + 0.5*(min(y1)+max(y1))*U1 +
50  // 0.5*(min(y2)+max(y2))*U2
51 
52  Vector3<Real> kDiff = akPoint[0] - kBox.Center;
53  Vector3<Real> kMin(kDiff.Dot(kBox.Axis[0]),kDiff.Dot(kBox.Axis[1]),
54  kDiff.Dot(kBox.Axis[2]));
55  Vector3<Real> kMax = kMin;
56  for (int i = 1; i < iQuantity; i++)
57  {
58  kDiff = akPoint[i] - kBox.Center;
59  for (int j = 0; j < 3; j++)
60  {
61  Real fDot = kDiff.Dot(kBox.Axis[j]);
62  if (fDot < kMin[j])
63  {
64  kMin[j] = fDot;
65  }
66  else if (fDot > kMax[j])
67  {
68  kMax[j] = fDot;
69  }
70  }
71  }
72 
73  kBox.Center +=
74  (((Real)0.5)*(kMin[0]+kMax[0]))*kBox.Axis[0] +
75  (((Real)0.5)*(kMin[1]+kMax[1]))*kBox.Axis[1] +
76  (((Real)0.5)*(kMin[2]+kMax[2]))*kBox.Axis[2];
77 
78  kBox.Extent[0] = ((Real)0.5)*(kMax[0] - kMin[0]);
79  kBox.Extent[1] = ((Real)0.5)*(kMax[1] - kMin[1]);
80  kBox.Extent[2] = ((Real)0.5)*(kMax[2] - kMin[2]);
81 
82  return kBox;
83 }

85 template <class Real>
86 Box3<Real> ContMinBox (int iQuantity, const Vector3<Real>* akPoint,
87  Real fEpsilon, Query::Type eQueryType)
88 {
89  Box3<Real> kBox;
90 
91  // Get the convex hull of the points.
92  ConvexHull3<Real> kHull(iQuantity,(Vector3<Real>*)akPoint,fEpsilon,false,
93  eQueryType);
94  int iHDim = kHull.GetDimension();
95  int iHQuantity;
96  const int* aiHIndex;
97 
98  if (iHDim == 0)
99  {
100  kBox.Center = akPoint[0];
101  kBox.Axis[0] = Vector3<Real>::UNIT_X;
102  kBox.Axis[1] = Vector3<Real>::UNIT_Y;
103  kBox.Axis[2] = Vector3<Real>::UNIT_Z;
104  kBox.Extent[0] = (Real)0.0;
105  kBox.Extent[1] = (Real)0.0;
106  kBox.Extent[2] = (Real)0.0;
107  return kBox;
108  }
109 
110  if (iHDim == 1)
111  {
112  ConvexHull1<Real>* pkHull1 = kHull.GetConvexHull1();
113  aiHIndex = pkHull1->GetIndices();
114 
115  kBox.Center = ((Real)0.5)*(akPoint[aiHIndex[0]]+akPoint[aiHIndex[1]]);
116  Vector3<Real> kDiff = akPoint[aiHIndex[1]] - akPoint[aiHIndex[0]];
117  kBox.Extent[0] = ((Real)0.5)*kDiff.Normalize();
118  kBox.Extent[1] = (Real)0.0;
119  kBox.Extent[2] = (Real)0.0;
120  kBox.Axis[0] = kDiff;
121  Vector3<Real>::GenerateComplementBasis(kBox.Axis[1],kBox.Axis[2],
122  kBox.Axis[0]);
123 
124  WM4_DELETE pkHull1;
125  return kBox;
126  }
127 
128  int i, j;
129  Vector3<Real> kOrigin, kDiff, kU, kV, kW;
130  Vector2<Real>* akPoint2;
131  Box2<Real> kBox2;
132 
133  if (iHDim == 2)
134  {
135  // When ConvexHull3 reports that the point set is 2-dimensional, the
136  // caller is responsible for projecting the points onto a plane and
137  // calling ConvexHull2. ConvexHull3 does provide information about
138  // the plane of the points. In this application, we need only
139  // project the input points onto that plane and call ContMinBox in
140  // two dimensions.
141 
142  // Get a coordinate system relative to the plane of the points.
143  kOrigin = kHull.GetPlaneOrigin();
144  kW = kHull.GetPlaneDirection(0).Cross(kHull.GetPlaneDirection(1));
145  Vector3<Real>::GenerateComplementBasis(kU,kV,kW);
146 
147  // Project the input points onto the plane.
148  akPoint2 = WM4_NEW Vector2<Real>[iQuantity];
149  for (i = 0; i < iQuantity; i++)
150  {
151  kDiff = akPoint[i] - kOrigin;
152  akPoint2[i].X() = kU.Dot(kDiff);
153  akPoint2[i].Y() = kV.Dot(kDiff);
154  }
155 
156  // Compute the minimum area box in 2D.
157  kBox2 = ContMinBox<Real>(iQuantity,akPoint2,fEpsilon,eQueryType,
158  false);
159  WM4_DELETE[] akPoint2;
160 
161  // Lift the values into 3D.
162  kBox.Center = kOrigin + kBox2.Center.X()*kU + kBox2.Center.Y()*kV;
163  kBox.Axis[0] = kBox2.Axis[0].X()*kU + kBox2.Axis[0].Y()*kV;
164  kBox.Axis[1] = kBox2.Axis[1].X()*kU + kBox2.Axis[1].Y()*kV;
165  kBox.Axis[2] = kW;
166  kBox.Extent[0] = kBox2.Extent[0];
167  kBox.Extent[1] = kBox2.Extent[1];
168  kBox.Extent[2] = (Real)0.0;
169  return kBox;
170  }
171 
172  iHQuantity = kHull.GetSimplexQuantity();
173  aiHIndex = kHull.GetIndices();
174  Real fVolume, fMinVolume = Math<Real>::MAX_REAL;
175 
176  // Create the unique set of hull vertices to minimize the time spent
177  // projecting vertices onto planes of the hull faces.
178  std::set<int> kUniqueIndices;
179  for (i = 0; i < 3*iHQuantity; i++)
180  {
181  kUniqueIndices.insert(aiHIndex[i]);
182  }
183 
184  // Use the rotating calipers method on the projection of the hull onto
185  // the plane of each face. Also project the hull onto the normal line
186  // of each face. The minimum area box in the plane and the height on
187  // the line produce a containing box. If its volume is smaller than the
188  // current volume, this box is the new candidate for the minimum volume
189  // box. The unique edges are accumulated into a set for use by a later
190  // step in the algorithm.
191  const int* piIndex = aiHIndex;
192  Real fHeight, fMinHeight, fMaxHeight;
193  std::set<EdgeKey> kEdges;
194  akPoint2 = WM4_NEW Vector2<Real>[kUniqueIndices.size()];
195  for (i = 0; i < iHQuantity; i++)
196  {
197  // get triangle
198  int iV0 = *piIndex++;
199  int iV1 = *piIndex++;
200  int iV2 = *piIndex++;
201 
202  // save the edges for later use
203  kEdges.insert(EdgeKey(iV0,iV1));
204  kEdges.insert(EdgeKey(iV1,iV2));
205  kEdges.insert(EdgeKey(iV2,iV0));
206 
207  // get 3D coordinate system relative to plane of triangle
208  kOrigin = (akPoint[iV0] + akPoint[iV1] + akPoint[iV2])/(Real)3.0;
209  Vector3<Real> kEdge1 = akPoint[iV1] - akPoint[iV0];
210  Vector3<Real> kEdge2 = akPoint[iV2] - akPoint[iV0];
211  kW = kEdge2.UnitCross(kEdge1); // inner-pointing normal
212  if (kW == Vector3<Real>::ZERO)
213  {
214  // The triangle is needle-like, so skip it.
215  continue;
216  }
217  Vector3<Real>::GenerateComplementBasis(kU,kV,kW);
218 
219  // Project points onto plane of triangle, onto normal line of plane.
220  // TO DO. In theory, minHeight should be zero since W points to the
221  // interior of the hull. However, the snap rounding used in the 3D
222  // convex hull finder involves loss of precision, which in turn can
223  // cause a hull facet to have the wrong ordering (clockwise instead
224  // of counterclockwise when viewed from outside the hull). The
225  // height calculations here trap that problem (the incorrectly ordered
226  // face will not affect the minimum volume box calculations).
227  fMinHeight = (Real)0.0;
228  fMaxHeight = (Real)0.0;
229  j = 0;
230  std::set<int>::const_iterator pkUI = kUniqueIndices.begin();
231  while (pkUI != kUniqueIndices.end())
232  {
233  int index = *pkUI++;
234  kDiff = akPoint[index] - kOrigin;
235  akPoint2[j].X() = kU.Dot(kDiff);
236  akPoint2[j].Y() = kV.Dot(kDiff);
237  fHeight = kW.Dot(kDiff);
238  if (fHeight > fMaxHeight)
239  {
240  fMaxHeight = fHeight;
241  }
242  else if (fHeight < fMinHeight)
243  {
244  fMinHeight = fHeight;
245  }
246 
247  j++;
248  }
249  if (-fMinHeight > fMaxHeight)
250  {
251  fMaxHeight = -fMinHeight;
252  }
253 
254  // compute minimum area box in 2D
255  kBox2 = ContMinBox<Real>((int)kUniqueIndices.size(),akPoint2,fEpsilon,
256  eQueryType,false);
257 
258  // update current minimum-volume box (if necessary)
259  fVolume = fMaxHeight*kBox2.Extent[0]*kBox2.Extent[1];
260  if (fVolume < fMinVolume)
261  {
262  fMinVolume = fVolume;
263 
264  // lift the values into 3D
265  kBox.Extent[0] = kBox2.Extent[0];
266  kBox.Extent[1] = kBox2.Extent[1];
267  kBox.Extent[2] = ((Real)0.5)*fMaxHeight;
268  kBox.Axis[0] = kBox2.Axis[0].X()*kU + kBox2.Axis[0].Y()*kV;
269  kBox.Axis[1] = kBox2.Axis[1].X()*kU + kBox2.Axis[1].Y()*kV;
270  kBox.Axis[2] = kW;
271  kBox.Center = kOrigin + kBox2.Center.X()*kU + kBox2.Center.Y()*kV
272  + kBox.Extent[2]*kW;
273  }
274  }
275 
276  // The minimum-volume box can also be supported by three mutually
277  // orthogonal edges of the convex hull. For each triple of orthogonal
278  // edges, compute the minimum-volume box for that coordinate frame by
279  // projecting the points onto the axes of the frame.
280  std::set<EdgeKey>::const_iterator pkE2;
281  for (pkE2 = kEdges.begin(); pkE2 != kEdges.end(); pkE2++)
282  {
283  kW = akPoint[pkE2->V[1]] - akPoint[pkE2->V[0]];
284  kW.Normalize();
285 
286  std::set<EdgeKey>::const_iterator pkE1 = pkE2;
287  for (++pkE1; pkE1 != kEdges.end(); pkE1++)
288  {
289  kV = akPoint[pkE1->V[1]] - akPoint[pkE1->V[0]];
290  kV.Normalize();
291  Real fDot = kV.Dot(kW);
292  if (Math<Real>::FAbs(fDot) > Math<Real>::ZERO_TOLERANCE)
293  {
294  continue;
295  }
296 
297  std::set<EdgeKey>::const_iterator pkE0 = pkE1;
298  for (++pkE0; pkE0 != kEdges.end(); pkE0++)
299  {
300  kU = akPoint[pkE0->V[1]] - akPoint[pkE0->V[0]];
301  kU.Normalize();
302  fDot = kU.Dot(kV);
303  if (Math<Real>::FAbs(fDot) > Math<Real>::ZERO_TOLERANCE)
304  {
305  continue;
306  }
307  fDot = kU.Dot(kW);
308  if (Math<Real>::FAbs(fDot) > Math<Real>::ZERO_TOLERANCE)
309  {
310  continue;
311  }
312 
313  // The three edges are mutually orthogonal. Project the
314  // hull points onto the lines containing the edges. Use
315  // hull point zero as the origin.
316  Real fUMin = (Real)0.0, fUMax = (Real)0.0;
317  Real fVMin = (Real)0.0, fVMax = (Real)0.0;
318  Real fWMin = (Real)0.0, fWMax = (Real)0.0;
319  kOrigin = akPoint[aiHIndex[0]];
320  std::set<int>::const_iterator pkUI = kUniqueIndices.begin();
321  while (pkUI != kUniqueIndices.end())
322  {
323  int index = *pkUI++;
324  kDiff = akPoint[index] - kOrigin;
325 
326  Real fU = kU.Dot(kDiff);
327  if (fU < fUMin)
328  {
329  fUMin = fU;
330  }
331  else if (fU > fUMax)
332  {
333  fUMax = fU;
334  }
335 
336  Real fV = kV.Dot(kDiff);
337  if (fV < fVMin)
338  {
339  fVMin = fV;
340  }
341  else if (fV > fVMax)
342  {
343  fVMax = fV;
344  }
345 
346  Real fW = kW.Dot(kDiff);
347  if (fW < fWMin)
348  {
349  fWMin = fW;
350  }
351  else if (fW > fWMax)
352  {
353  fWMax = fW;
354  }
355  }
356 
357  Real fUExtent = ((Real)0.5)*(fUMax - fUMin);
358  Real fVExtent = ((Real)0.5)*(fVMax - fVMin);
359  Real fWExtent = ((Real)0.5)*(fWMax - fWMin);
360 
361  // update current minimum-volume box (if necessary)
362  fVolume = fUExtent*fVExtent*fWExtent;
363  if (fVolume < fMinVolume)
364  {
365  fMinVolume = fVolume;
366 
367  kBox.Extent[0] = fUExtent;
368  kBox.Extent[1] = fVExtent;
369  kBox.Extent[2] = fWExtent;
370  kBox.Axis[0] = kU;
371  kBox.Axis[1] = kV;
372  kBox.Axis[2] = kW;
373  kBox.Center = kOrigin +
374  ((Real)0.5)*(fUMin+fUMax)*kU +
375  ((Real)0.5)*(fVMin+fVMax)*kV +
376  ((Real)0.5)*(fWMin+fWMax)*kW;
377  }
378  }
379  }
380  }
381 
382  WM4_DELETE[] akPoint2;
383  return kBox;

388 {

390  for (int i = 0; i < 3; i++)
391  {

401 template <class Real>
402 Box3<Real> MergeBoxes (const Box3<Real>& rkBox0, const Box3<Real>& rkBox1)
403 {
404  // construct a box that contains the input boxes
405  Box3<Real> kBox;
406 
407  // The first guess at the box center. This value will be updated later
408  // after the input box vertices are projected onto axes determined by an
409  // average of box axes.
410  kBox.Center = ((Real)0.5)*(rkBox0.Center + rkBox1.Center);
411 
412  // A box's axes, when viewed as the columns of a matrix, form a rotation
413  // matrix. The input box axes are converted to quaternions. The average
414  // quaternion is computed, then normalized to unit length. The result is
415  // the slerp of the two input quaternions with t-value of 1/2. The result
416  // is converted back to a rotation matrix and its columns are selected as
417  // the merged box axes.
418  Quaternion<Real> kQ0, kQ1;
419  kQ0.FromRotationMatrix(rkBox0.Axis);
420  kQ1.FromRotationMatrix(rkBox1.Axis);
421  if (kQ0.Dot(kQ1) < (Real)0.0)
422  {
423  kQ1 = -kQ1;
424  }
425 
426  Quaternion<Real> kQ = kQ0 + kQ1;
427  Real fInvLength = Math<Real>::InvSqrt(kQ.Dot(kQ));
428  kQ = fInvLength*kQ;
429  kQ.ToRotationMatrix(kBox.Axis);
430 
431  // Project the input box vertices onto the merged-box axes. Each axis
432  // D[i] containing the current center C has a minimum projected value
433  // pmin[i] and a maximum projected value pmax[i]. The corresponding end
434  // points on the axes are C+pmin[i]*D[i] and C+pmax[i]*D[i]. The point C
435  // is not necessarily the midpoint for any of the intervals. The actual
436  // box center will be adjusted from C to a point C' that is the midpoint
437  // of each interval,
438  // C' = C + sum_{i=0}^2 0.5*(pmin[i]+pmax[i])*D[i]
439  // The box extents are
440  // e[i] = 0.5*(pmax[i]-pmin[i])
441 
442  int i, j;
443  Real fDot;
444  Vector3<Real> akVertex[8], kDiff;
445  Vector3<Real> kMin = Vector3<Real>::ZERO;
446  Vector3<Real> kMax = Vector3<Real>::ZERO;
447 
448  rkBox0.ComputeVertices(akVertex);
449  for (i = 0; i < 8; i++)
450  {
451  kDiff = akVertex[i] - kBox.Center;
452  for (j = 0; j < 3; j++)
453  {
454  fDot = kDiff.Dot(kBox.Axis[j]);
455  if (fDot > kMax[j])
456  {
457  kMax[j] = fDot;
458  }
459  else if (fDot < kMin[j])
460  {
461  kMin[j] = fDot;
462  }
463  }
464  }
465 
466  rkBox1.ComputeVertices(akVertex);
467  for (i = 0; i < 8; i++)
468  {
469  kDiff = akVertex[i] - kBox.Center;
470  for (j = 0; j < 3; j++)
471  {
472  fDot = kDiff.Dot(kBox.Axis[j]);
473  if (fDot > kMax[j])
474  {
475  kMax[j] = fDot;
476  }
477  else if (fDot < kMin[j])
478  {
479  kMin[j] = fDot;
480  }
481  }
482  }
483 
484  // [kMin,kMax] is the axis-aligned box in the coordinate system of the
485  // merged box axes. Update the current box center to be the center of
486  // the new box. Compute the extents based on the new center.
487  for (j = 0; j < 3; j++)
488  {
489  kBox.Center += (((Real)0.5)*(kMax[j]+kMin[j]))*kBox.Axis[j];
490  kBox.Extent[j] = ((Real)0.5)*(kMax[j]-kMin[j]);
491  }
492 
493  return kBox;
494 }
