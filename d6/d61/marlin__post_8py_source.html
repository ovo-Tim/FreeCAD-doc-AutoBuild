482 def parse(pathobj):
483  global DRILL_RETRACT_MODE
484  global MOTION_MODE
485  global CURRENT_X
486  global CURRENT_Y
487  global CURRENT_Z
488 
489  out = ""
490  lastcommand = None
491  precision_string = "." + str(PRECISION) + "f"
492 
493  params = [
494  "X",
495  "Y",
496  "Z",
497  "A",
498  "B",
499  "C",
500  "U",
501  "V",
502  "W",
503  "I",
504  "J",
505  "K",
506  "F",
507  "S",
508  "T",
509  "Q",
510  "R",
511  "L",
512  "P",
513  ]
514 
515  if hasattr(pathobj, "Group"): # We have a compound or project.
516  if OUTPUT_COMMENTS:
517  out += linenumber() + "(Compound: " + pathobj.Label + ")\n"
518  for p in pathobj.Group:
519  out += parse(p)
520  return out
521 
522  else: # Parsing simple path
523  # groups might contain non-path things like stock.
524  if not hasattr(pathobj, "Path"):
525  return out
526 
527  if OUTPUT_COMMENTS and OUTPUT_PATH:
528  out += linenumber() + "(Path: " + pathobj.Label + ")\n"
529 
530  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
531  outlist = []
532  command = c.Name
533  outlist.append(command)
534  # Debug:
535  # print('pathobj.Path.Commands:', c)
536 
537  # If modal is True, delete duplicate commands:
538  if MODAL:
539  if command == lastcommand:
540  outlist.pop(0)
541 
542  # Add the remaining parameters in order:
543  for param in params:
544  if param in c.Parameters:
545  if param == "F":
546  if command not in RAPID_MOVES:
547  feedRate = Units.Quantity(c.Parameters["F"], FreeCAD.Units.Velocity)
548  if feedRate.getValueAs(UNIT_FEED_FORMAT) > 0.0:
549  outlist.append(
550  param
551  + format(
552  float(feedRate.getValueAs(UNIT_FEED_FORMAT)),
553  precision_string,
554  )
555  )
556  elif param in ["T", "H", "D", "S", "P", "L"]:
557  outlist.append(param + str(c.Parameters[param]))
558  elif param in ["A", "B", "C"]:
559  outlist.append(param + format(c.Parameters[param], precision_string))
560  # [X, Y, Z, U, V, W, I, J, K, R, Q]
561  else:
562  pos = Units.Quantity(c.Parameters[param], FreeCAD.Units.Length)
563  outlist.append(
564  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
565  )
566 
567  # Store the latest command:
568  lastcommand = command
569 
570  # Capture the current position for subsequent calculations:
571  if command in MOTION_COMMANDS:
572  if "X" in c.Parameters:
573  CURRENT_X = Units.Quantity(c.Parameters["X"], FreeCAD.Units.Length)
574  if "Y" in c.Parameters:
575  CURRENT_Y = Units.Quantity(c.Parameters["Y"], FreeCAD.Units.Length)
576  if "Z" in c.Parameters:
577  CURRENT_Z = Units.Quantity(c.Parameters["Z"], FreeCAD.Units.Length)
578 
579  if command in ("G98", "G99"):
580  DRILL_RETRACT_MODE = command
581 
582  if TRANSLATE_DRILL_CYCLES:
583  if command in ("G81", "G82", "G83"):
584  out += drill_translate(outlist, command, c.Parameters)
585  # Erase the line just translated:
586  outlist = []
587 
588  if SPINDLE_WAIT > 0:
589  if command in ("M3", "M03", "M4", "M04"):
590  out += linenumber() + format_outlist(outlist) + "\n"
591  # Marlin: P for milliseconds, S for seconds, change P to S
592  out += linenumber()
593  out += format_outlist(["G4", "S%s" % SPINDLE_WAIT])
594  out += "\n"
595  outlist = []
596 
597  # Check for Tool Change:
598  if command in ("M6", "M06"):
599  if OUTPUT_COMMENTS:
600  out += linenumber() + "(Begin toolchange)\n"
601  if OUTPUT_TOOL_CHANGE:
602  for line in TOOL_CHANGE.splitlines(True):
603  out += linenumber() + line
604  if not OUTPUT_TOOL_CHANGE and OUTPUT_COMMENTS:
605  outlist[0] = "(" + outlist[0]
606  outlist[-1] = outlist[-1] + ")"
607  if not OUTPUT_TOOL_CHANGE and not OUTPUT_COMMENTS:
608  outlist = []
609 
610  if command == "message":
611  if OUTPUT_COMMENTS:
612  outlist.pop(0) # remove the command
613  else:
614  out = []
615 
616  if command in SUPPRESS_COMMANDS:
617  outlist[0] = "(" + outlist[0]
618  outlist[-1] = outlist[-1] + ")"
619 
620  # Remove embedded comments:
621  if not OUTPUT_COMMENTS:
622  tmplist = []
623  list_index = 0
624  while list_index < len(outlist):
625  left_index = outlist[list_index].find("(")
626  if left_index == -1: # Not a comment
627  tmplist.append(outlist[list_index])
628  else: # This line contains a comment, and possibly more
629  right_index = outlist[list_index].find(")")
630  comment_area = outlist[list_index][left_index : right_index + 1]
631  line_minus_comment = outlist[list_index].replace(comment_area, "").strip()
632  if line_minus_comment:
633  # Line contained more than just a comment
634  tmplist.append(line_minus_comment)
635  list_index += 1
636  # Done removing comments
637  outlist = tmplist
638 
639  # Prepend a line number and append a newline
640  if len(outlist) >= 1:
641  out += linenumber() + format_outlist(outlist) + "\n"
642 
643  return out

718  # * G80 Cancel the drill cycle *

745  # Update next_Stop_Z to next depth:
746  next_Stop_Z -= drill_Step

752  # Rapid down to just above last peck depth:
753  rapid_Z_to(next_Stop_Z + chip_Space)
754  # Dril remainder of the hole depth:
755  feed_Z_to(drill_Z)
756  break # Done
