68 void CmdSpreadsheetMergeCells::activated(int iMsg)
69 {
70  Q_UNUSED(iMsg);
71  if (getActiveGuiDocument()) {
72  Gui::MDIView* activeWindow = Gui::getMainWindow()->activeWindow();
73  SpreadsheetGui::SheetView* sheetView =
74  freecad_dynamic_cast<SpreadsheetGui::SheetView>(activeWindow);
75 
76  if (sheetView) {
77  Sheet* sheet = sheetView->getSheet();
78  std::vector<Range> ranges = sheetView->selectedRanges();
79 
80  // Execute mergeCells commands
81  if (!ranges.empty()) {
82  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Merge cells"));
83  std::vector<Range>::const_iterator i = ranges.begin();
84  for (; i != ranges.end(); ++i) {
85  if (i->size() > 1) {
86  Gui::Command::doCommand(Gui::Command::Doc,
87  "App.ActiveDocument.%s.mergeCells('%s')",
88  sheet->getNameInDocument(),
89  i->rangeString().c_str());
90  }
91  }
92  Gui::Command::commitCommand();
93  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.recompute()");
94  }
95  }
96  }

191 void CmdSpreadsheetImport::activated(int iMsg)
192 {
193  Q_UNUSED(iMsg);
194  QString selectedFilter;
195  QString formatList = QObject::tr("CSV (*.csv *.CSV);;All (*)");
196  QString fileName = Gui::FileDialog::getOpenFileName(Gui::getMainWindow(),
197  QObject::tr("Import file"),
198  QString(),
199  formatList,
200  &selectedFilter);
201  if (!fileName.isEmpty()) {
202  std::string FeatName = getUniqueObjectName("Spreadsheet");
203  Sheet* sheet = freecad_dynamic_cast<Sheet>(
204  App::GetApplication().getActiveDocument()->addObject("Spreadsheet::Sheet",
205  FeatName.c_str()));
206  if (sheet) {
207  char delim, quote, escape;
208  std::string errMsg = "Import";
209  bool isValid = sheet->getCharsFromPrefs(delim, quote, escape, errMsg);
210 
211  if (isValid) {
212  sheet->importFromFile(fileName.toStdString(), delim, quote, escape);
213  sheet->execute();
214  }
215  else {
216  Base::Console().Error(errMsg.c_str());
217  return;
218  }
219  }
220  }
221 }

633 void CmdSpreadsheetStyleBold::activated(int iMsg)
634 {
635  Q_UNUSED(iMsg);
636  if (getActiveGuiDocument()) {
637  Gui::MDIView* activeWindow = Gui::getMainWindow()->activeWindow();
638  SpreadsheetGui::SheetView* sheetView =
639  freecad_dynamic_cast<SpreadsheetGui::SheetView>(activeWindow);
640 
641  if (sheetView) {
642  Sheet* sheet = sheetView->getSheet();
643  QModelIndexList selection = sheetView->selectedIndexes();
644 
645  if (!selection.empty()) {
646  bool allBold = true;
647 
648  for (const auto& it : selection) {
649  const Cell* cell = sheet->getCell(CellAddress(it.row(), it.column()));
650 
651  if (cell) {
652  std::set<std::string> style;
653 
654  cell->getStyle(style);
655  if (style.find("bold") == style.end()) {
656  allBold = false;
657  break;
658  }
659  }
660  }
661 
662  std::vector<Range> ranges = sheetView->selectedRanges();
663  std::vector<Range>::const_iterator i = ranges.begin();
664 
665  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Set bold text"));
666  for (; i != ranges.end(); ++i) {
667  if (!allBold) {
668  Gui::Command::doCommand(
669  Gui::Command::Doc,
