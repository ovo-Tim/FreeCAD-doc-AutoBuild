84 void ViewProviderPart::applyTransparency(float transparency, std::vector<App::Color>& colors)
85 {
86  if (transparency != 0.0) {
87  // transparency has been set object-wide
88  for (auto& j : colors) {
89  // transparency hasn't been set for this face
90  if (j.a == 0.0) {
91  j.setTransparency(transparency/100.0F); // transparency comes in percent
92  }
93  }
94  }
95 }
