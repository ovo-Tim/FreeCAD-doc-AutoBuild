24 template <class Real>
25 Line3<Real> OrthogonalLineFit3 (int iQuantity, const Vector3<Real>* akPoint)
26 {
27  Line3<Real> kLine(Vector3<Real>::ZERO,Vector3<Real>::ZERO);
28 
29  // compute the mean of the points
30  kLine.Origin = akPoint[0];
31  int i;
32  for (i = 1; i < iQuantity; i++)
33  {
34  kLine.Origin += akPoint[i];
35  }
36  Real fInvQuantity = ((Real)1.0)/iQuantity;
37  kLine.Origin *= fInvQuantity;
38 
39  // compute the covariance matrix of the points
40  Real fSumXX = (Real)0.0, fSumXY = (Real)0.0, fSumXZ = (Real)0.0;
41  Real fSumYY = (Real)0.0, fSumYZ = (Real)0.0, fSumZZ = (Real)0.0;
42  for (i = 0; i < iQuantity; i++)
43  {
44  Vector3<Real> kDiff = akPoint[i] - kLine.Origin;
45  fSumXX += kDiff.X()*kDiff.X();
46  fSumXY += kDiff.X()*kDiff.Y();
47  fSumXZ += kDiff.X()*kDiff.Z();
48  fSumYY += kDiff.Y()*kDiff.Y();
49  fSumYZ += kDiff.Y()*kDiff.Z();
50  fSumZZ += kDiff.Z()*kDiff.Z();
51  }
52 
53  fSumXX *= fInvQuantity;
54  fSumXY *= fInvQuantity;
55  fSumXZ *= fInvQuantity;
56  fSumYY *= fInvQuantity;
57  fSumYZ *= fInvQuantity;
58  fSumZZ *= fInvQuantity;
59 
60  // set up the eigensolver
61  Eigen<Real> kES(3);
62  kES(0,0) = fSumYY+fSumZZ;
63  kES(0,1) = -fSumXY;
64  kES(0,2) = -fSumXZ;
65  kES(1,0) = kES(0,1);
66  kES(1,1) = fSumXX+fSumZZ;
67  kES(1,2) = -fSumYZ;
68  kES(2,0) = kES(0,2);
69  kES(2,1) = kES(1,2);
70  kES(2,2) = fSumXX+fSumYY;
71 
72  // compute eigenstuff, smallest eigenvalue is in last position
73  kES.DecrSortEigenStuff3();
74 
75  // unit-length direction for best-fit line
76  kES.GetEigenvector(2,kLine.Direction);
77 
78  return kLine;
79 }
