258  def extrude(self, shift=False, real=False):
259  """Redraw the ghost in extrude mode."""
260  self.newpoint = self.obj.Shape.Faces[0].CenterOfMass
261  dvec = self.point.sub(self.newpoint)
262  if not shift:
263  delta = DraftVecUtils.project(dvec, self.normal)
264  if delta.Length < 1e-7:
265  # Use the normal if self.newpoint is coplanar with the face:
266  delta = self.normal * dvec.Length
267  else:
268  delta = dvec
269  if self.force and delta.Length:
270  ratio = self.force/delta.Length
271  delta.multiply(ratio)
272  if real:
273  return delta
274  self.ghost[0].trans.translation.setValue([delta.x, delta.y, delta.z])
275  for i in range(1, len(self.ghost)):
276  base = self.obj.Shape.Vertexes[i-1].Point
277  self.ghost[i].p1(base)
278  self.ghost[i].p2(base.add(delta))
279  return delta.Length
