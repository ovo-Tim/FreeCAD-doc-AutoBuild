226 void CmdBlendCurve::activated(int)
227 {
228  std::string docName = App::GetApplication().getActiveDocument()->getName();
229  std::string objName[2];
230  std::string edge[2];
231  std::string featName = getUniqueObjectName("BlendCurve");
232  std::vector<Gui::SelectionObject> sel =
233  getSelection().getSelectionEx(nullptr, Part::Feature::getClassTypeId());
234 
235  objName[0] = sel[0].getFeatName();
236  edge[0] = sel[0].getSubNames()[0];
237 
238  if (sel.size() == 1) {
239  objName[1] = sel[0].getFeatName();
240  edge[1] = sel[0].getSubNames()[1];
241  }
242  else {
243  objName[1] = sel[1].getFeatName();
244  edge[1] = sel[1].getSubNames()[0];
245  }
246  openCommand(QT_TRANSLATE_NOOP("Command", "Blend Curve"));
247  doCommand(Doc,
248  "App.ActiveDocument.addObject(\"Surface::FeatureBlendCurve\",\"%s\")",
249  featName.c_str());
250  doCommand(Doc,
251  "App.ActiveDocument.%s.StartEdge = (App.getDocument('%s').getObject('%s'),['%s'])",
252  featName.c_str(),
253  docName.c_str(),
254  objName[0].c_str(),
255  edge[0].c_str());
256  doCommand(Doc,
257  "App.ActiveDocument.%s.EndEdge = (App.getDocument('%s').getObject('%s'),['%s'])",
258  featName.c_str(),
259  docName.c_str(),
260  objName[1].c_str(),
261  edge[1].c_str());
262  updateActive();
263  commitCommand();
264 }

287 void CmdSurfaceExtendFace::activated(int)
288 {
289  Gui::SelectionFilter faceFilter("SELECT Part::Feature SUBELEMENT Face COUNT 1");
290  if (faceFilter.match()) {
291  const std::vector<std::string>& sub = faceFilter.Result[0][0].getSubNames();
292  if (sub.size() == 1) {
293  openCommand(QT_TRANSLATE_NOOP("Command", "Extend surface"));
294  std::string FeatName = getUniqueObjectName("Surface");
295  std::string supportString = faceFilter.Result[0][0].getAsPropertyLinkSubString();
296  doCommand(Doc,
297  "App.ActiveDocument.addObject(\"Surface::Extend\",\"%s\")",
298  FeatName.c_str());
299  doCommand(Doc,
300  "App.ActiveDocument.%s.Face = %s",
301  FeatName.c_str(),
302  supportString.c_str());
303  updateActive();
304  commitCommand();
305  }
306  }
307  else {
308  QMessageBox::warning(Gui::getMainWindow(),
309  qApp->translate("Surface_ExtendFace", "Wrong selection"),
310  qApp->translate("Surface_ExtendFace", "Select a single face"));
311  }
312 }
