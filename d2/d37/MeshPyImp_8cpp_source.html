76  return -1;
77  }
78 
79  try {
80  this->parentProperty = nullptr;
81  // if no mesh is given
82  if (!pcObj) {
83  return 0;
84  }
85  if (PyObject_TypeCheck(pcObj, &(MeshPy::Type))) {
86  getMeshObjectPtr()->operator=(*static_cast<MeshPy*>(pcObj)->getMeshObjectPtr());
87  }
88  else if (PyList_Check(pcObj)) {
89  PyObject* ret = addFacets(args);
90  bool ok = (ret != nullptr);
91  Py_XDECREF(ret);
92  if (!ok) {
93  return -1;
94  }
95  }
96  else if (PyTuple_Check(pcObj)) {
97  PyObject* ret = addFacets(args);
98  bool ok = (ret != nullptr);
99  Py_XDECREF(ret);
100  if (!ok) {
101  return -1;
102  }
103  }
104  else if (PyUnicode_Check(pcObj)) {
105  getMeshObjectPtr()->load(PyUnicode_AsUTF8(pcObj));
106  }
107  else {
108  PyErr_Format(PyExc_TypeError,
109  "Cannot create a mesh out of a '%s'",
110  pcObj->ob_type->tp_name);
111  return -1;
112  }
113  }
114  catch (const Base::Exception& e) {
115  e.setPyException();
116  return -1;
117  }
118  catch (const std::exception& e) {
119  PyErr_SetString(Base::PyExc_FC_GeneralError, e.what());
120  return -1;
121  }
122  catch (const Py::Exception&) {
123  return -1;
124  }

394 PyObject* MeshPy::crossSections(PyObject* args)
395 {
396  PyObject* obj {};
397  PyObject* poly = Py_False;
398  float min_eps = 1.0e-2F;
399  if (!PyArg_ParseTuple(args, "O|fO!", &obj, &min_eps, &PyBool_Type, &poly)) {
400  return nullptr;
401  }
402 
403  Py::Sequence list(obj);
404  Py::Type vType(Base::getTypeAsObject(&Base::VectorPy::Type));
405 
406  std::vector<MeshObject::TPlane> csPlanes;
407  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
408  Py::Tuple pair(*it);
409  Py::Object p1 = pair.getItem(0);
410  Py::Object p2 = pair.getItem(1);
411  if (p1.isType(vType) && p2.isType(vType)) {
412  MeshObject::TPlane plane;
413  Base::Vector3d b = static_cast<Base::VectorPy*>(p1.ptr())->value();
414  Base::Vector3d n = static_cast<Base::VectorPy*>(p2.ptr())->value();
415  plane.first.Set((float)b.x, (float)b.y, (float)b.z);
416  plane.second.Set((float)n.x, (float)n.y, (float)n.z);
417  csPlanes.push_back(plane);
418  }
419  else if (p1.isTuple() && p2.isTuple()) {
420  Py::Tuple b(p1);
421  Py::Tuple n(p2);
422  float bx = Py::Float(b.getItem(0));
423  float by = Py::Float(b.getItem(1));
424  float bz = Py::Float(b.getItem(2));
425  float nx = Py::Float(n.getItem(0));
426  float ny = Py::Float(n.getItem(1));
427  float nz = Py::Float(n.getItem(2));
428 
429  MeshObject::TPlane plane;
430  plane.first.Set(bx, by, bz);
431  plane.second.Set(nx, ny, nz);
432  csPlanes.push_back(plane);
433  }
434  }
435 
436  std::vector<MeshObject::TPolylines> sections;
437  getMeshObjectPtr()->crossSections(csPlanes, sections, min_eps, Base::asBoolean(poly));
438 
439  // convert to Python objects
440  Py::List crossSections;
441  for (const auto& it : sections) {
442  Py::List section;
443  for (const auto& jt : it) {
444  Py::List polyline;
445  for (auto kt : jt) {
446  polyline.append(Py::asObject(new Base::VectorPy(kt)));
447  }
448  section.append(polyline);
449  }
450  crossSections.append(section);
451  }
452 
453  return Py::new_reference_to(crossSections);
454 }
