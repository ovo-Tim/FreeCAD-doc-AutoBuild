46 template <class Real>
47 bool IntrSegment3Box3<Real>::Test ()
48 {
49  Real afAWdU[3], afADdU[3], afAWxDdU[3], fRhs;
50 
51  Vector3<Real> kDiff = m_rkSegment.Origin - m_rkBox.Center;
52 
53  afAWdU[0] = Math<Real>::FAbs(m_rkSegment.Direction.Dot(m_rkBox.Axis[0]));
54  afADdU[0] = Math<Real>::FAbs(kDiff.Dot(m_rkBox.Axis[0]));
55  fRhs = m_rkBox.Extent[0] + m_rkSegment.Extent*afAWdU[0];
56  if (afADdU[0] > fRhs)
57  {
58  return false;
59  }
60 
61  afAWdU[1] = Math<Real>::FAbs(m_rkSegment.Direction.Dot(m_rkBox.Axis[1]));
62  afADdU[1] = Math<Real>::FAbs(kDiff.Dot(m_rkBox.Axis[1]));
63  fRhs = m_rkBox.Extent[1] + m_rkSegment.Extent*afAWdU[1];
64  if (afADdU[1] > fRhs)
65  {
66  return false;
67  }
68 
69  afAWdU[2] = Math<Real>::FAbs(m_rkSegment.Direction.Dot(m_rkBox.Axis[2]));
70  afADdU[2] = Math<Real>::FAbs(kDiff.Dot(m_rkBox.Axis[2]));
71  fRhs = m_rkBox.Extent[2] + m_rkSegment.Extent*afAWdU[2];
72  if (afADdU[2] > fRhs)
73  {
74  return false;
75  }
76 
77  Vector3<Real> kWxD = m_rkSegment.Direction.Cross(kDiff);
78 
79  afAWxDdU[0] = Math<Real>::FAbs(kWxD.Dot(m_rkBox.Axis[0]));
80  fRhs = m_rkBox.Extent[1]*afAWdU[2] + m_rkBox.Extent[2]*afAWdU[1];
81  if (afAWxDdU[0] > fRhs)
82  {
83  return false;
84  }
85 
86  afAWxDdU[1] = Math<Real>::FAbs(kWxD.Dot(m_rkBox.Axis[1]));
87  fRhs = m_rkBox.Extent[0]*afAWdU[2] + m_rkBox.Extent[2]*afAWdU[0];
88  if (afAWxDdU[1] > fRhs)
89  {
90  return false;
91  }
92 
93  afAWxDdU[2] = Math<Real>::FAbs(kWxD.Dot(m_rkBox.Axis[2]));
94  fRhs = m_rkBox.Extent[0]*afAWdU[1] + m_rkBox.Extent[1]*afAWdU[0];
95  if (afAWxDdU[2] > fRhs)
96  {
97  return false;
98  }
99 
100  return true;
101 }
