191 TopoDS_Shape DrawBrokenView::apply1Break(const App::DocumentObject& breakObj, const TopoDS_Shape& inShape) const
192 {
193  // Base::Console().Message("DBV::apply1Break()\n");
194  auto breakPoints = breakPointsFromObj(breakObj);
195  if (breakPoints.first.IsEqual(breakPoints.second, EWTOLERANCE)) {
196  Base::Console().Message("DBV::apply1Break - break points are equal\n");
197  return inShape;
198  }
199 
200  auto breakDirection = DU::closestBasisOriented(directionFromObj(breakObj));
201  breakDirection.Normalize();
202 
203  // make a halfspace that is positioned at the first breakpoint and extends
204  // in the direction of the second point
205  Base::Vector3d moveDir0 = breakPoints.second - breakPoints.first;
206  moveDir0.Normalize();
207  moveDir0 = DU::closestBasisOriented(moveDir0);
208  auto halfSpace0 = makeHalfSpace(breakPoints.first, moveDir0, breakPoints.second);
209  FCBRepAlgoAPI_Cut mkCut0(inShape, halfSpace0);
210  if (!mkCut0.IsDone()) {
211  Base::Console().Message("DBV::apply1Break - cut0 failed\n");
212  }
213  TopoDS_Shape cut0 = mkCut0.Shape();
214 
215  // make a halfspace that is positioned at the second breakpoint and extends
216  // in the direction of the first point
217  Base::Vector3d moveDir1 = breakPoints.first - breakPoints.second;
218  moveDir1.Normalize();
219  moveDir1 = DU::closestBasisOriented(moveDir1);
220  auto halfSpace1 = makeHalfSpace(breakPoints.second, moveDir1, breakPoints.first);
221  FCBRepAlgoAPI_Cut mkCut1(inShape, halfSpace1);
222  if (!mkCut1.IsDone()) {
223  Base::Console().Message("DBV::apply1Break - cut1 failed\n");
224  }
225  TopoDS_Shape cut1 = mkCut1.Shape();
226 
227  BRep_Builder builder;
228  TopoDS_Compound result;
229  builder.MakeCompound(result);
230  builder.Add(result, cut0);
231  builder.Add(result, cut1);
232 
233  // BRepTools::Write(cut0, "DBVcut0.brep"); //debug
234  // BRepTools::Write(cut1, "DBVcut1.brep");
235 
236  return result;
237 }

252 TopoDS_Shape DrawBrokenView::compressHorizontal(const TopoDS_Shape& shapeToCompress)const
253 {
254  // Base::Console().Message("DBV::compressHorizontal()\n");
255  auto pieces = getPieces(shapeToCompress);
256  auto breaksAll = Breaks.getValues();
257  auto moveDirection = DU::closestBasisOriented(DU::toVector3d(getProjectionCS().XDirection()));
258  bool descend = false;
259  auto sortedBreaks = makeSortedBreakList(breaksAll, moveDirection, descend);
260  auto limits = getPieceLimits(pieces, moveDirection);
261  // for each break, move all the pieces left of the break to the right by the removed amount
262  // for the break
263  for (auto& breakItem : sortedBreaks) {
264  // check each break against all the pieces
265  Base::Vector3d netBreakDisplace =
266  moveDirection * (removedLengthFromObj(*breakItem.breakObj) - Gap.getValue());
267  size_t iPiece{0};
268  for (auto& pieceLimit : limits) {
269  // check each piece against the current break
270  if (moveThisPiece(pieceLimit, breakItem, moveDirection)) {
271  // piece is to left of break, so needs to move right
272  TopoDS_Shape temp = ShapeUtils::moveShape(pieces.at(iPiece), netBreakDisplace);
273  pieces.at(iPiece) = temp;
274  }
275  iPiece++;
276  }
277  }
278  // turn updated pieces into a compound
279  BRep_Builder builder;
280  TopoDS_Compound result;
281  builder.MakeCompound(result);
282  for (auto& pieceShape : pieces) {
283  builder.Add(result, pieceShape);
284  }
285 
286  return result;
287 }

291 TopoDS_Shape DrawBrokenView::compressVertical(const TopoDS_Shape& shapeToCompress)const
292 {
293  // Base::Console().Message("DBV::compressVertical()\n");
294  auto pieces = getPieces(shapeToCompress);
295  auto breaksAll = Breaks.getValues();
296  // not sure about using closestBasis here. may prevent oblique breaks later.
297  auto moveDirection = DU::closestBasisOriented(DU::toVector3d(getProjectionCS().YDirection()));
298 
299  bool descend = false;
300  auto sortedBreaks = makeSortedBreakList(breaksAll, moveDirection, descend);
301  auto limits = getPieceLimits(pieces, moveDirection);
302  // for each break, move all the pieces below the break up by the removed amount
303  // for the break
304  for (auto& breakItem : sortedBreaks) {
305  // check each break against all the pieces
306  Base::Vector3d netBreakDisplace =
307  moveDirection * (removedLengthFromObj(*breakItem.breakObj) - Gap.getValue());
308  size_t iPiece{0};
309  for (auto& pieceLimit : limits) {
310  // check each piece against the current break using a fuzzy equality
311  if (moveThisPiece(pieceLimit, breakItem, moveDirection)) {
312  // piece is below the break, move it up
313  TopoDS_Shape temp = ShapeUtils::moveShape(pieces.at(iPiece), netBreakDisplace);
314  pieces.at(iPiece) = temp;
315  }
316  iPiece++;
317  }
318  }
319  // turn updated pieces into a compound
320  BRep_Builder builder;
321  TopoDS_Compound result;
322  builder.MakeCompound(result);
323  for (auto& pieceShape : pieces) {
324  builder.Add(result, pieceShape);
325  }
326 
327  return result;
328 }

436 std::pair<Base::Vector3d, Base::Vector3d> DrawBrokenView::breakPointsFromSketch(const App::DocumentObject& breakObj) const
437 {
438  // Base::Console().Message("DBV::breakPointsFromSketch()\n");
439  TopoDS_Shape locShape = ShapeExtractor::getLocatedShape(&breakObj);
440  if (locShape.IsNull()) {
441  return {Base::Vector3d(), Base::Vector3d()};;
442  }
443 
444  // get the edges from the shape.
445  // vertical or horizontal
446  std::vector<TopoDS_Edge> sketchEdges;
447  TopExp_Explorer expl(locShape, TopAbs_EDGE);
448  for (; expl.More(); expl.Next()) {
449  sketchEdges.push_back(TopoDS::Edge(expl.Current()));
450  }
451  // there should be 2
452  if (sketchEdges.size() != 2) {
453  return {Base::Vector3d(), Base::Vector3d()};
454  }
455 
456  // they should both have the same orientation
457  TopoDS_Edge first = sketchEdges.front();
458  TopoDS_Edge last = sketchEdges.back();
459  if ((isVertical(first) && isVertical(last)) ||
460  (isHorizontal(first) && isHorizontal(last))) {
461  // get points on each edge that are in line with the center of the sketch
462  // along the break direction
463  Bnd_Box edgeBox;
464  edgeBox.SetGap(0.0);
465  BRepBndLib::AddOptimal(first, edgeBox);
466  BRepBndLib::AddOptimal(last, edgeBox);
467  double xMin = 0, xMax = 0, yMin = 0, yMax = 0, zMin = 0, zMax = 0;
468  edgeBox.Get(xMin, yMin, zMin, xMax, yMax, zMax);
469  Base::Vector3d center( (xMin + xMax) / 2.0,
470  (yMin + yMax) / 2.0,
471  (zMin + zMax) / 2.0 );
472  auto ends0 = SU::getEdgeEnds(first);
473  auto dir0 = ends0.second - ends0.first;
474  dir0.Normalize();
475  // get a point on edge first on a perpendicular line through center
476  auto break0 = center.Perpendicular(ends0.first, dir0);
477 
478  auto ends1 = SU::getEdgeEnds(last);
479  auto dir1 = ends1.second - ends1.first;
480  dir1.Normalize();
481  auto break1 = center.Perpendicular(ends1.first, dir1);
482 
483  return { break0, break1 };
484  }
485 
486  return {Base::Vector3d(), Base::Vector3d()};
487 }

526 std::pair<Base::Vector3d, Base::Vector3d> DrawBrokenView::breakBoundsFromSketch(const App::DocumentObject& breakObj) const
527 {
528  //Base::Console().Message("DBV::breakBoundsFromSketch()\n");
529  std::pair<Base::Vector3d, Base::Vector3d> breakPoints = breakPointsFromObj(breakObj);
530  Base::Vector3d center = (breakPoints.first + breakPoints.second) / 2.0;
531  Base::Vector3d breakDir = directionFromObj(breakObj);
532  breakDir.Normalize();
533  Base::Vector3d perpDir = makePerpendicular(breakDir);
534  perpDir.Normalize();
535 
536  // get the midpoint of the zigzags
537  Base::Vector3d ptOnLine0 = center + breakDir * removedLengthFromObj(breakObj) / 2.0;
538  Base::Vector3d ptOnLine1 = center - breakDir * removedLengthFromObj(breakObj) / 2.0;
539  double lineLength = breaklineLength(breakObj);
540 
541  Base::Vector3d corner0 = ptOnLine0 - perpDir * lineLength / 2.0;
542  Base::Vector3d corner1 = ptOnLine1 + perpDir * lineLength / 2.0;
543  corner0 = mapPoint3dToView(corner0);
544  corner1 = mapPoint3dToView(corner1);
545 
546  // these are unscaled, unrotated points
547  return{corner0, corner1};
548 }

552 std::pair<Base::Vector3d, Base::Vector3d> DrawBrokenView::breakBoundsFromEdge(const App::DocumentObject& breakObj) const
553 {
554  // Base::Console().Message("DBV::breakBoundsFromEdge()\n");
555  TopoDS_Shape locShape = ShapeExtractor::getLocatedShape(&breakObj);
556  if (locShape.IsNull() || locShape.ShapeType() != TopAbs_EDGE) {
557  return {Base::Vector3d(), Base::Vector3d()};
558  }
559 
560  auto edge = projectEdge(TopoDS::Edge(locShape));
561  auto start = edge->getStartPoint();
562  auto end = edge->getEndPoint();
563  Base::Vector3d direction = end - start;
564  double length = direction.Length();
565  direction.Normalize();
566  Base::Vector3d stdX{1.0, 0.0, 0.0};
567  Base::Vector3d stdY{0.0, 1.0, 0.0};
568  if (DU::fpCompare(fabs(direction.Dot(stdX)), 1.0, EWTOLERANCE) ) {
569  double left = std::min(start.x, end.x);
570  double right = std::max(start.x, end.x);
571  // not wild about this for top/bottom
572  double top = start.y + length;
573  double bottom = start.y - length;
574  Base::Vector3d topLeft{left, top, 0.0};
575  Base::Vector3d bottomRight{right, bottom, 0.0};
576  return{topLeft, bottomRight};
577  }
578 
579  if (!DU::fpCompare(fabs(direction.Dot(stdY)), 1.0, EWTOLERANCE) ) {
580  Base::Console().Message("DBV::breakBoundsFromEdge - direction is not X or Y\n");
581  // TODO: throw? return nonsense?
582  }
583 
584  double left = start.x - length;
585  double right = start.x + length;
586  double bottom = std::min(start.y, end.y);
587  double top = std::max(start.y, end.y);
588  Base::Vector3d topLeft{left, top, 0.0};
589  Base::Vector3d bottomRight{right, bottom, 0.0};
590  return{topLeft, bottomRight};

609 double DrawBrokenView::breaklineLengthFromSketch(const App::DocumentObject& breakObj) const
610 {
611  // Base::Console().Message("DBV::breaklineLengthFromSketch()\n");
612  TopoDS_Shape locShape = ShapeExtractor::getLocatedShape(&breakObj);
613  if (locShape.IsNull()) {
614  return 0;
615  }
616  // get the edges from the sketch
617  std::vector<TopoDS_Edge> sketchEdges;
618  TopExp_Explorer expl(locShape, TopAbs_EDGE);
619  for (; expl.More(); expl.Next()) {
620  sketchEdges.push_back(TopoDS::Edge(expl.Current()));
621  }
622 
623  if (sketchEdges.size() < 2) {
624  // need 2 edges
625  Base::Console().Message("DBV::breaklineLengthFromSketch - not enough edges\n");
626  }
627 
628  std::pair<Base::Vector3d, Base::Vector3d> ends0 = SU::getEdgeEnds(sketchEdges.front());
629  ends0.first = projectPoint(ends0.first, false);
630  ends0.second = projectPoint(ends0.second, false);
631 
632  std::pair<Base::Vector3d, Base::Vector3d> ends1 = SU::getEdgeEnds(sketchEdges.back());
633  ends1.first = projectPoint(ends1.first, false);
634  ends1.second = projectPoint(ends1.second, false);

852 BreakList DrawBrokenView::makeSortedBreakListCompressed(const std::vector<App::DocumentObject*>& breaks, Base::Vector3d moveDirection, bool descend) const
853 {
854  // Base::Console().Message("DBV::makeSortedBreakListCompressed(%d, %s)\n", breaks.size(),
855  // DU::formatVector(moveDirection).c_str());
856  // sortedBreaks is in lowLimit order
857  auto sortedBreaks = makeSortedBreakList(breaks, moveDirection, descend);
858  BreakList result;
859  size_t iBreak{0};
860  for (auto& breakObj : sortedBreaks) {
861  BreakListEntry newEntry;
862  double breakSum{0};
863  if (isDirectionReversed(moveDirection)) {
864  // reversed X+ high low X-
865  // our list is sorted ascending by low limit - each is further left than the previous
866  // so we must reverse our list to get the correct shift values.
867  std::reverse(sortedBreaks.begin(), sortedBreaks.end());
868  for (size_t iSum = iBreak + 1; iSum < sortedBreaks.size(); iSum++) {
869  // shift right by the removed amount of all the breaks to the right of this break
870  breakSum += sortedBreaks.at(iSum).netRemoved;
871  }
872  newEntry.breakObj = breakObj.breakObj;
873  newEntry.lowLimit = breakObj.lowLimit - breakSum; // move to right
874  newEntry.highLimit = newEntry.lowLimit + Gap.getValue();
875  newEntry.netRemoved = breakObj.netRemoved;
876  result.push_back(newEntry);
877  } else {
878  // forward X- low high X+
879  for (size_t iSum = iBreak + 1; iSum < sortedBreaks.size(); iSum++) {
880  // shift right by the removed amount of all the breaks to the right of this break
881  breakSum += sortedBreaks.at(iSum).netRemoved;
882  }
883  newEntry.breakObj = breakObj.breakObj;
884  newEntry.lowLimit = breakObj.lowLimit + breakObj.netRemoved + breakSum;
885  newEntry.highLimit = newEntry.lowLimit + Gap.getValue();
886  newEntry.netRemoved = breakObj.netRemoved;
887  result.push_back(newEntry);
888  }
889  iBreak++;
890  }
891 
892  if (isDirectionReversed(moveDirection)) {
893  std::reverse(sortedBreaks.begin(), sortedBreaks.end());
894  }
895  return result;
896 }

1025 double DrawBrokenView::shiftAmountShrink(double pointCoord, Base::Vector3d direction, const BreakList& sortedBreaks) const
1026 {
1027  // Base::Console().Message("DBV::shiftAmountShrink(%.3f, %s, %d)\n", pointCoord,
1028  // DU::formatVector(direction).c_str(), sortedBreaks.size());
1029  double shift{0};
1030  for (auto& breakItem : sortedBreaks) {
1031  if (isDirectionReversed(direction)) {
1032  if (pointCoord <= breakItem.lowLimit) {
1033  // h--------l -ve
1034  // p
1035  // point is right/above break
1036  // leave alone, this break doesn't affect us
1037  continue;
1038  }
1039 
1040  if (pointCoord > breakItem.highLimit ||
1041  DU::fpCompare(pointCoord, breakItem.highLimit, Precision::Confusion()) ) {
1042  // h--------l -ve
1043  // p
1044  // point is left/below break
1045  shift += breakItem.netRemoved;
1046  continue;
1047  }
1048 
1049  // h--------l -ve
1050  // p
1051  // g-g
1052  // p'
1053  // break.start < value < break.end - point is in the break area
1054  // we move our point by a fraction of the Gap length
1055  double penetration = fabs(pointCoord - breakItem.highLimit); // (p - h) start(high) to point distance
1056  double removed = removedLengthFromObj(*breakItem.breakObj); // (h-l) full break size
1057  double factor = 1 - (penetration / removed); // fraction of break to right
1058  double toShift = pointCoord - (breakItem.lowLimit - factor * Gap.getValue());
1059  shift += fabs(toShift);
1060 
1061 
1062  } else {
1063  if (pointCoord >= breakItem.highLimit) {
1064  // l--------h +ve
1065  // p
1066  // leave alone, this break doesn't affect us
1067  continue;
1068  }
1069 
1070  if (pointCoord < breakItem.lowLimit ||
1071  DU::fpCompare(pointCoord, breakItem.lowLimit, Precision::Confusion()) ) {
1072  // l--------h +ve
1073  // p
1074  // move right/up by the removed area less the gap
1075  shift += removedLengthFromObj(*breakItem.breakObj) - Gap.getValue();
1076  continue;
1077  }
1078 
1079  // l--------h +ve
1080  // p
1081  // g--g
1082  // p'
1083  // break.start < value < break.end - point is in the break area
1084  // we move our point by a fraction of the Gap length
1085  double penetration = fabs(pointCoord - breakItem.lowLimit);
1086  double removed = removedLengthFromObj(*breakItem.breakObj);
1087  double factor = 1 - (penetration / removed);
1088  double netRemoved = pointCoord - breakItem.highLimit - factor * Gap.getValue();
1089  shift += fabs(netRemoved);
1090  }
1091  }
1092 
1093  return shift;
1094 }
