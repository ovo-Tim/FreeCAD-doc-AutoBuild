143 std::shared_ptr<std::map<QString, std::shared_ptr<ModelTreeNode>>>
144 ModelLibrary::getModelTree(ModelFilter filter) const
145 {
146  std::shared_ptr<std::map<QString, std::shared_ptr<ModelTreeNode>>> modelTree =
147  std::make_shared<std::map<QString, std::shared_ptr<ModelTreeNode>>>();
148 
149  for (auto& it : *_modelPathMap) {
150  auto filename = it.first;
151  auto model = it.second;
152 
153  if (ModelManager::passFilter(filter, model->getType())) {
154  QStringList list = filename.split(QString::fromStdString("/"));
155 
156  // Start at the root
157  std::shared_ptr<std::map<QString, std::shared_ptr<ModelTreeNode>>> node = modelTree;
158  for (auto& itp : list) {
159  if (ModelManager::isModel(itp)) {
160  std::shared_ptr<ModelTreeNode> child = std::make_shared<ModelTreeNode>();
161  child->setData(model);
162  (*node)[itp] = child;
163  }
164  else {
165  // Add the folder only if it's not already there
166  if (node->count(itp) == 0) {
167  auto mapPtr =
168  std::make_shared<std::map<QString, std::shared_ptr<ModelTreeNode>>>();
169  std::shared_ptr<ModelTreeNode> child = std::make_shared<ModelTreeNode>();
170  child->setFolder(mapPtr);
171  (*node)[itp] = child;
172  node = mapPtr;
173  }
174  else {
175  node = (*node)[itp]->getFolder();
176  }
177  }
178  }
179  }
180  }
181 
182  return modelTree;
