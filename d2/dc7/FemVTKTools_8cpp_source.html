105 // Helper function to fill vtkCellArray from SMDS_Mesh using vtk cell order
106 template<typename T, typename E>
107 void fillVtkArray(vtkSmartPointer<vtkCellArray>& elemArray, std::vector<int>& types, const E* elem)
108 {
109  vtkSmartPointer<T> cell = vtkSmartPointer<T>::New();
110  const std::vector<int>& order = SMDS_MeshCell::toVtkOrder(elem->GetEntityType());
111  if (!order.empty()) {
112  for (int i = 0; i < elem->NbNodes(); ++i) {
113  cell->GetPointIds()->SetId(i, elem->GetNode(order[i])->GetID() - 1);
114  }
115  }
116  else {
117  for (int i = 0; i < elem->NbNodes(); ++i) {
118  cell->GetPointIds()->SetId(i, elem->GetNode(i)->GetID() - 1);
119  }
120  }
121  elemArray->InsertNextCell(cell);
122  types.push_back(SMDS_MeshCell::toVtkType(elem->GetEntityType()));
123 }

125 // Helper function to fill SMDS_Mesh elements ID from vtk cell
126 void fillMeshElementIds(vtkCell* cell, std::vector<int>& ids)
127 {
128  VTKCellType cellType = static_cast<VTKCellType>(cell->GetCellType());
129  const std::vector<int>& order = SMDS_MeshCell::fromVtkOrder(cellType);
130  vtkIdType* vtkIds = cell->GetPointIds()->GetPointer(0);
131  ids.clear();
132  int nbPoints = cell->GetNumberOfPoints();
133  ids.resize(nbPoints);
134  if (!order.empty()) {
135  for (int i = 0; i < nbPoints; ++i) {
136  ids[i] = vtkIds[order[i]] + 1;
137  }
138  }
139  else {
140  for (int i = 0; i < nbPoints; ++i) {
141  ids[i] = vtkIds[i] + 1;
142  }
143  }
144 }

149 void FemVTKTools::importVTKMesh(vtkSmartPointer<vtkDataSet> dataset, FemMesh* mesh, float scale)
150 {
151  const vtkIdType nPoints = dataset->GetNumberOfPoints();
152  const vtkIdType nCells = dataset->GetNumberOfCells();
153  Base::Console().Log("%d nodes/points and %d cells/elements found!\n", nPoints, nCells);
154  Base::Console().Log("Build SMESH mesh out of the vtk mesh data.\n", nPoints, nCells);
155 
156  // Now fill the SMESH datastructure
157  SMESH_Mesh* smesh = mesh->getSMesh();
158  SMESHDS_Mesh* meshds = smesh->GetMeshDS();
159  meshds->ClearMesh();
160 
161  for (vtkIdType i = 0; i < nPoints; i++) {
162  double* p = dataset->GetPoint(i);
163  meshds->AddNodeWithID(p[0] * scale, p[1] * scale, p[2] * scale, i + 1);
164  }
165 
166  for (vtkIdType iCell = 0; iCell < nCells; iCell++) {
167  vtkCell* cell = dataset->GetCell(iCell);
168  std::vector<int> ids;
169  fillMeshElementIds(cell, ids);
170  switch (cell->GetCellType()) {
171  // 1D edges
172  case VTK_LINE: // seg2
173  meshds->AddEdgeWithID(ids[0], ids[1], iCell + 1);
174  break;
175  case VTK_QUADRATIC_EDGE: // seg3
176  meshds->AddEdgeWithID(ids[0], ids[1], ids[2], iCell + 1);
177  break;
178  // 2D faces
179  case VTK_TRIANGLE: // tria3
180  meshds->AddFaceWithID(ids[0], ids[1], ids[2], iCell + 1);
181  break;
182  case VTK_QUADRATIC_TRIANGLE: // tria6
183  meshds->AddFaceWithID(ids[0], ids[1], ids[2], ids[3], ids[4], ids[5], iCell + 1);
184  break;
185  case VTK_QUAD: // quad4
186  meshds->AddFaceWithID(ids[0], ids[1], ids[2], ids[3], iCell + 1);
187  break;
188  case VTK_QUADRATIC_QUAD: // quad8
189  meshds->AddFaceWithID(ids[0],
190  ids[1],
191  ids[2],
192  ids[3],
193  ids[4],
194  ids[5],
195  ids[6],
196  ids[7],
197  iCell + 1);
198  break;
199  // 3D volumes
200  case VTK_TETRA: // tetra4
201  meshds->AddVolumeWithID(ids[0], ids[1], ids[2], ids[3], iCell + 1);
202  break;
203  case VTK_QUADRATIC_TETRA: // tetra10
204  meshds->AddVolumeWithID(ids[0],
205  ids[1],
206  ids[2],
207  ids[3],
208  ids[4],
209  ids[5],
210  ids[6],
211  ids[7],
212  ids[8],
213  ids[9],
214  iCell + 1);
215  break;
216  case VTK_HEXAHEDRON: // hexa8
217  meshds->AddVolumeWithID(ids[0],
218  ids[1],
219  ids[2],
220  ids[3],
221  ids[4],
222  ids[5],
223  ids[6],
224  ids[7],
225  iCell + 1);
226  break;
227  case VTK_QUADRATIC_HEXAHEDRON: // hexa20
228  meshds->AddVolumeWithID(ids[0],
229  ids[1],
230  ids[2],
231  ids[3],
232  ids[4],
233  ids[5],
234  ids[6],
235  ids[7],
236  ids[8],
237  ids[9],
238  ids[10],
239  ids[11],
240  ids[12],
241  ids[13],
242  ids[14],
243  ids[15],
244  ids[16],
245  ids[17],
246  ids[18],
247  ids[19],
248  iCell + 1);
249  break;
250  case VTK_WEDGE: // penta6
251  meshds->AddVolumeWithID(ids[0], ids[1], ids[2], ids[3], ids[4], ids[5], iCell + 1);
252  break;
253  case VTK_QUADRATIC_WEDGE: // penta15
254  meshds->AddVolumeWithID(ids[0],
255  ids[1],
256  ids[2],
257  ids[3],
258  ids[4],
259  ids[5],
260  ids[6],
261  ids[7],
262  ids[8],
263  ids[9],
264  ids[10],
265  ids[11],
266  ids[12],
267  ids[13],
268  ids[14],
269  iCell + 1);
270  break;
271  case VTK_PYRAMID: // pyra5
272  meshds->AddVolumeWithID(ids[0], ids[1], ids[2], ids[3], ids[4], iCell + 1);
273  break;
274  case VTK_QUADRATIC_PYRAMID: // pyra13
275  meshds->AddVolumeWithID(ids[0],
276  ids[1],
277  ids[2],
278  ids[3],
279  ids[4],
280  ids[5],
281  ids[6],
282  ids[7],
283  ids[8],
284  ids[9],
285  ids[10],
286  ids[11],
287  ids[12],
288  iCell + 1);
289  break;
290 
291  // not handled cases
292  default: {
293  Base::Console().Error(
294  "Only common 1D, 2D and 3D Cells are supported in VTK mesh import\n");
295  break;
296  }
297  }
298  }

301 FemMesh* FemVTKTools::readVTKMesh(const char* filename, FemMesh* mesh)
302 {
303  Base::TimeElapsed Start;
304  Base::Console().Log("Start: read FemMesh from VTK unstructuredGrid ======================\n");
305  Base::FileInfo f(filename);
306 
307  if (f.hasExtension("vtu")) {
308  vtkSmartPointer<vtkDataSet> dataset = readVTKFile<vtkXMLUnstructuredGridReader>(filename);
309  if (!dataset.Get()) {
310  Base::Console().Error("Failed to load file %s\n", filename);
311  return nullptr;
312  }
313  importVTKMesh(dataset, mesh);
314  }
315  else if (f.hasExtension("pvtu")) {
316  vtkSmartPointer<vtkDataSet> dataset = readVTKFile<vtkXMLPUnstructuredGridReader>(filename);
317  if (!dataset.Get()) {
318  Base::Console().Error("Failed to load file %s\n", filename);
319  return nullptr;
320  }
321  importVTKMesh(dataset, mesh);
322  }
323  else if (f.hasExtension("vtk")) {
324  vtkSmartPointer<vtkDataSet> dataset = readVTKFile<vtkDataSetReader>(filename);
325  if (!dataset.Get()) {
326  Base::Console().Error("Failed to load file %s\n", filename);
327  return nullptr;
328  }
329  importVTKMesh(dataset, mesh);
330  }
331  else {
332  Base::Console().Error("file name extension is not supported\n");
333  return nullptr;
334  }
335  // Mesh should link to the part feature, in order to set up FemConstraint
336 
337  Base::Console().Log(" %f: Done \n",
338  Base::TimeElapsed::diffTimeF(Start, Base::TimeElapsed()));
339  return mesh;

342 void exportFemMeshEdges(vtkSmartPointer<vtkCellArray>& elemArray,
343  std::vector<int>& types,
344  const SMDS_EdgeIteratorPtr& aEdgeIter)
345 {
346  Base::Console().Log(" Start: VTK mesh builder edges.\n");
347 
348  while (aEdgeIter->more()) {
349  const SMDS_MeshEdge* aEdge = aEdgeIter->next();
350  // edge
351  if (aEdge->GetEntityType() == SMDSEntity_Edge) {
352  fillVtkArray<vtkLine>(elemArray, types, aEdge);
353  }
354  // quadratic edge
355  else if (aEdge->GetEntityType() == SMDSEntity_Quad_Edge) {
356  fillVtkArray<vtkQuadraticEdge>(elemArray, types, aEdge);
357  }
358  else {
359  throw Base::TypeError("Edge not yet supported by FreeCAD's VTK mesh builder\n");
360  }
361  }
362 
363  Base::Console().Log(" End: VTK mesh builder edges.\n");
364 }

367  std::vector<int>& types,

416  else if (aVol->GetEntityType() == SMDSEntity_Hexa) { // hexa8
417  fillVtkArray<vtkHexahedron>(elemArray, types, aVol);
418  }
419  else if (aVol->GetEntityType() == SMDSEntity_Quad_Tetra) { // tetra10
420  fillVtkArray<vtkQuadraticTetra>(elemArray, types, aVol);
421  }
422  else if (aVol->GetEntityType() == SMDSEntity_Quad_Pyramid) { // pyra13
423  fillVtkArray<vtkQuadraticPyramid>(elemArray, types, aVol);
424  }
425  else if (aVol->GetEntityType() == SMDSEntity_Quad_Penta) { // penta15
426  fillVtkArray<vtkQuadraticWedge>(elemArray, types, aVol);
427  }
428  else if (aVol->GetEntityType() == SMDSEntity_Quad_Hexa) { // hexa20
429  fillVtkArray<vtkQuadraticHexahedron>(elemArray, types, aVol);
430  }
431  else {
432  throw Base::TypeError("Volume not yet supported by FreeCAD's VTK mesh builder\n");
433  }
434  }
435 
436  Base::Console().Log(" End: VTK mesh builder volumes.\n");

439 void FemVTKTools::exportVTKMesh(const FemMesh* mesh,
440  vtkSmartPointer<vtkUnstructuredGrid> grid,
441  bool highest,
442  float scale)
443 {
444 
445  Base::Console().Log("Start: VTK mesh builder ======================\n");
446  const SMESH_Mesh* smesh = mesh->getSMesh();
447  const SMESHDS_Mesh* meshDS = smesh->GetMeshDS();
448 
449  // nodes
450  Base::Console().Log(" Start: VTK mesh builder nodes.\n");
451 
452  vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New();
453  SMDS_NodeIteratorPtr aNodeIter = meshDS->nodesIterator();
454 
455  while (aNodeIter->more()) {
456  const SMDS_MeshNode* node = aNodeIter->next(); // why float, not double?
457  double coords[3] = {double(node->X() * scale),
458  double(node->Y() * scale),
459  double(node->Z() * scale)};
460  points->InsertPoint(node->GetID() - 1, coords);
461  // memory is allocated by VTK points size for max node id, not for point count
462  // if the SMESH mesh has gaps in node numbering, points without any element
463  // assignment will be inserted in these point gaps too
464  // this needs to be taken into account on node mapping when FreeCAD FEM results
465  // are exported to vtk
466  }
467  grid->SetPoints(points);
468  // nodes debugging
469  const SMDS_MeshInfo& info = meshDS->GetMeshInfo();
470  Base::Console().Log(" Size of nodes in SMESH grid: %i.\n", info.NbNodes());
471  const vtkIdType nNodes = grid->GetNumberOfPoints();
472  Base::Console().Log(" Size of nodes in VTK grid: %i.\n", nNodes);
473  Base::Console().Log(" End: VTK mesh builder nodes.\n");
474 
475  vtkSmartPointer<vtkCellArray> elemArray = vtkSmartPointer<vtkCellArray>::New();
476  std::vector<int> types;
477 
478  if (highest) {
479  // try volumes
480  SMDS_VolumeIteratorPtr aVolIter = meshDS->volumesIterator();
481  exportFemMeshCells(elemArray, types, aVolIter);
482  // try faces
483  if (elemArray->GetNumberOfCells() == 0) {
484  SMDS_FaceIteratorPtr aFaceIter = meshDS->facesIterator();
485  exportFemMeshFaces(elemArray, types, aFaceIter);
486  }
487  // try edges
488  if (elemArray->GetNumberOfCells() == 0) {
489  SMDS_EdgeIteratorPtr aEdgeIter = meshDS->edgesIterator();
490  exportFemMeshEdges(elemArray, types, aEdgeIter);
491  }
492  }
493  else {
494  // export all elements
495  // edges
496  SMDS_EdgeIteratorPtr aEdgeIter = meshDS->edgesIterator();
497  exportFemMeshEdges(elemArray, types, aEdgeIter);
498  // faces
499  SMDS_FaceIteratorPtr aFaceIter = meshDS->facesIterator();
500  exportFemMeshFaces(elemArray, types, aFaceIter);
501  // volumes
502  SMDS_VolumeIteratorPtr aVolIter = meshDS->volumesIterator();
503  exportFemMeshCells(elemArray, types, aVolIter);
504  }
505 
506  if (elemArray->GetNumberOfCells() > 0) {
507  grid->SetCells(types.data(), elemArray);
508  }
509 
510  Base::Console().Log("End: VTK mesh builder ======================\n");
511 }

583 {

591  ds = readVTKFile<vtkXMLUnstructuredGridReader>(filename);

594  ds = readVTKFile<vtkDataSetReader>(filename);
595  }
596  else {
597  Base::Console().Error("file name extension is not supported\n");
598  }
599 
600  App::Document* pcDoc = App::GetApplication().getActiveDocument();
601  if (!pcDoc) {
602  Base::Console().Message("No active document is found thus created\n");
603  pcDoc = App::GetApplication().newDocument();
604  }
605  App::DocumentObject* obj = pcDoc->getActiveObject();
606 
607  vtkSmartPointer<vtkDataSet> dataset = ds;
608  App::DocumentObject* result = nullptr;
609 
610  if (res) {
611  Base::Console().Message(
612  "FemResultObject pointer is NULL, trying to get the active object\n");
613  if (obj->getTypeId() == Base::Type::fromName("Fem::FemResultObjectPython")) {
614  result = obj;
615  }
616  else {
617  Base::Console().Message("the active object is not the correct type, do nothing\n");
618  return nullptr;
619  }
620  }
621 
622  App::DocumentObject* mesh = pcDoc->addObject("Fem::FemMeshObject", "ResultMesh");
623  std::unique_ptr<FemMesh> fmesh(new FemMesh());
624  importVTKMesh(dataset, fmesh.get());
625  static_cast<PropertyFemMesh*>(mesh->getPropertyByName("FemMesh"))->setValuePtr(fmesh.release());
626 
627  if (result) {
628  // PropertyLink is the property type to store DocumentObject pointer
629  App::PropertyLink* link =
630  dynamic_cast<App::PropertyLink*>(result->getPropertyByName("Mesh"));
631  if (link) {
632  link->setValue(mesh);

774 void FemVTKTools::importFreeCADResult(vtkSmartPointer<vtkDataSet> dataset,
775  App::DocumentObject* result)
776 {
777  Base::Console().Log("Start: import vtk result file data into a FreeCAD result object.\n");
778 
779  std::map<std::string, std::string> vectors = _getFreeCADMechResultVectorProperties();
780  std::map<std::string, std::string> scalars = _getFreeCADMechResultScalarProperties();
781 
782  double ts = 0.0; // t=0.0 for static simulation
783  static_cast<App::PropertyFloat*>(result->getPropertyByName("Time"))->setValue(ts);
784 
785  vtkSmartPointer<vtkPointData> pd = dataset->GetPointData();
786  if (pd->GetNumberOfArrays() == 0) {
787  Base::Console().Error("No point data array is found in vtk data set, do nothing\n");
788  // if pointData is empty, data may be in cellDate,
789  // cellData -> pointData interpolation is possible in VTK
790  return;
791  }
792 
793  // NodeNumbers
794  const vtkIdType nPoints = dataset->GetNumberOfPoints();
795  std::vector<long> nodeIds(nPoints);
796  for (vtkIdType i = 0; i < nPoints; ++i) {
797  nodeIds[i] = i + 1;
798  }
799  static_cast<App::PropertyIntegerList*>(result->getPropertyByName("NodeNumbers"))
800  ->setValues(nodeIds);
801  Base::Console().Log(" NodeNumbers have been filled with values.\n");
802 
803  // vectors
804  for (const auto& it : vectors) {
805  int dim = 3; // Fixme: currently 3D only, here we could run into trouble,
806  // FreeCAD only supports dim 3D, I do not know about VTK
807  vtkDataArray* vector_field = vtkDataArray::SafeDownCast(pd->GetArray(it.second.c_str()));
808  if (vector_field && vector_field->GetNumberOfComponents() == dim) {
809  App::PropertyVectorList* vector_list =
810  static_cast<App::PropertyVectorList*>(result->getPropertyByName(it.first.c_str()));
811  if (vector_list) {
812  std::vector<Base::Vector3d> vec(nPoints);
813  for (vtkIdType i = 0; i < nPoints; ++i) {
814  double* p = vector_field->GetTuple(
815  i); // both vtkFloatArray and vtkDoubleArray return double* for GetTuple(i)
816  vec[i] = (Base::Vector3d(p[0], p[1], p[2]));
817  }
818  // PropertyVectorList will not show up in PropertyEditor
819  vector_list->setValues(vec);
820  Base::Console().Log(" A PropertyVectorList has been filled with values: %s\n",
821  it.first.c_str());
822  }
823  else {
824  Base::Console().Error("static_cast<App::PropertyVectorList*>((result->"
825  "getPropertyByName(\"%s\")) failed.\n",
826  it.first.c_str());
827  continue;
828  }
829  }
830  else {
831  Base::Console().Message(" PropertyVectorList NOT found in vkt file data: %s\n",
832  it.first.c_str());
833  }
834  }
835 
836  // scalars
837  for (const auto& scalar : scalars) {
838  vtkDataArray* vec = vtkDataArray::SafeDownCast(pd->GetArray(scalar.second.c_str()));
839  if (nPoints && vec && vec->GetNumberOfComponents() == 1) {
840  App::PropertyFloatList* field = static_cast<App::PropertyFloatList*>(
841  result->getPropertyByName(scalar.first.c_str()));
842  if (!field) {
843  Base::Console().Error("static_cast<App::PropertyFloatList*>((result->"
844  "getPropertyByName(\"%s\")) failed.\n",
845  scalar.first.c_str());
846  continue;
847  }
848 
849  double vmin = 1.0e100, vmax = -1.0e100;
850  std::vector<double> values(nPoints, 0.0);
851  for (vtkIdType i = 0; i < vec->GetNumberOfTuples(); i++) {
852  double v = *(vec->GetTuple(i));
853  values[i] = v;
854  if (v > vmax) {
855  vmax = v;
856  }
857  if (v < vmin) {
858  vmin = v;
859  }
860  }
861  field->setValues(values);
862  Base::Console().Log(" A PropertyFloatList has been filled with vales: %s\n",
863  scalar.first.c_str());
864  }
865  else {
866  Base::Console().Message(" PropertyFloatList NOT found in vkt file data %s\n",
867  scalar.first.c_str());
868  }
869  }
870 
871  // stats
872  // stats are added by importVTKResults
873 
874  Base::Console().Log("End: import vtk result file data into a FreeCAD result object.\n");
875 }

878 void FemVTKTools::exportFreeCADResult(const App::DocumentObject* result,
879  vtkSmartPointer<vtkDataSet> grid)
880 {
881  Base::Console().Log("Start: Create VTK result data from FreeCAD result data.\n");
882 
883  std::map<std::string, std::string> vectors = _getFreeCADMechResultVectorProperties();
884  std::map<std::string, std::string> scalars = _getFreeCADMechResultScalarProperties();
885 
886  const Fem::FemResultObject* res = static_cast<const Fem::FemResultObject*>(result);
887  const vtkIdType nPoints = grid->GetNumberOfPoints();
888 
889  // we need the corresponding mesh to get the correct id for the result data
890  // (when the freecad smesh mesh has gaps in the points
891  // vtk has more points. Vtk does not support point gaps, thus the gaps are
892  // filled with points. Then the mapping must be correct)
893  App::DocumentObject* meshObj = res->Mesh.getValue();
894  if (!meshObj || !meshObj->isDerivedFrom(FemMeshObject::getClassTypeId())) {
895  Base::Console().Error("Result object does not correctly link to mesh");
896  return;
897  }
898  const SMESH_Mesh* smesh = static_cast<FemMeshObject*>(meshObj)->FemMesh.getValue().getSMesh();
899  const SMESHDS_Mesh* meshDS = smesh->GetMeshDS();
900 
901  // all result object meshes are in mm therefore for e.g. length outputs like
902  // displacement we must divide by 1000
903  double factor = 1.0;
904 
905  // vectors
906  for (const auto& it : vectors) {
907  const int dim =
908  3; // Fixme, detect dim, but FreeCAD PropertyVectorList ATM only has DIM of 3
909  App::PropertyVectorList* field = nullptr;
910  if (res->getPropertyByName(it.first.c_str())) {
911  field = static_cast<App::PropertyVectorList*>(res->getPropertyByName(it.first.c_str()));
912  }
913  else {
914  Base::Console().Error(" PropertyVectorList not found: %s\n", it.first.c_str());
915  }
916 
917  if (field && field->getSize() > 0) {
918  const std::vector<Base::Vector3d>& vel = field->getValues();
919  vtkSmartPointer<vtkDoubleArray> data = vtkSmartPointer<vtkDoubleArray>::New();
920  data->SetNumberOfComponents(dim);
921  data->SetNumberOfTuples(nPoints);
922  data->SetName(it.second.c_str());
923 
924  // we need to set values for the unused points.
925  // TODO: ensure that the result bar does not include the used 0 if it is not
926  // part of the result (e.g. does the result bar show 0 as smallest value?)
927  if (nPoints != field->getSize()) {
928  double tuple[] = {0, 0, 0};
929  for (vtkIdType i = 0; i < nPoints; ++i) {
930  data->SetTuple(i, tuple);
931  }
932  }
933 
934  if (it.first.compare("DisplacementVectors") == 0) {
935  factor = 0.001; // to get meter
936  }
937  else {
938  factor = 1.0;
939  }
940 
941  SMDS_NodeIteratorPtr aNodeIter = meshDS->nodesIterator();
942  for (const auto& jt : vel) {
943  const SMDS_MeshNode* node = aNodeIter->next();
944  double tuple[] = {jt.x * factor, jt.y * factor, jt.z * factor};
945  data->SetTuple(node->GetID() - 1, tuple);
946  }
947  grid->GetPointData()->AddArray(data);
948  Base::Console().Log(
949  " The PropertyVectorList %s was exported to VTK vector list: %s\n",
950  it.first.c_str(),
951  it.second.c_str());
952  }
953  else if (field) {
954  Base::Console().Log(" PropertyVectorList NOT exported to vtk: %s size is: %i\n",
955  it.first.c_str(),
956  field->getSize());
957  }
958  }
959 
960  // scalars
961  for (const auto& scalar : scalars) {
962  App::PropertyFloatList* field = nullptr;
963  if (res->getPropertyByName(scalar.first.c_str())) {
964  field =
965  static_cast<App::PropertyFloatList*>(res->getPropertyByName(scalar.first.c_str()));
966  }
967  else {
968  Base::Console().Error("PropertyFloatList %s not found \n", scalar.first.c_str());
969  }
970 
971  if (field && field->getSize() > 0) {
972  const std::vector<double>& vec = field->getValues();
973  vtkSmartPointer<vtkDoubleArray> data = vtkSmartPointer<vtkDoubleArray>::New();
974  data->SetNumberOfValues(nPoints);
975  data->SetName(scalar.second.c_str());
976 
977  // we need to set values for the unused points.
978  // TODO: ensure that the result bar does not include the used 0 if it is not part
979  // of the result (e.g. does the result bar show 0 as smallest value?)
980  if (nPoints != field->getSize()) {
981  for (vtkIdType i = 0; i < nPoints; ++i) {
982  data->SetValue(i, 0);
983  }
984  }
985 
986  if ((scalar.first.compare("MaxShear") == 0)
987  || (scalar.first.compare("NodeStressXX") == 0)
988  || (scalar.first.compare("NodeStressXY") == 0)
989  || (scalar.first.compare("NodeStressXZ") == 0)
990  || (scalar.first.compare("NodeStressYY") == 0)
991  || (scalar.first.compare("NodeStressYZ") == 0)
992  || (scalar.first.compare("NodeStressZZ") == 0)
993  || (scalar.first.compare("PrincipalMax") == 0)
994  || (scalar.first.compare("PrincipalMed") == 0)
995  || (scalar.first.compare("PrincipalMin") == 0)
996  || (scalar.first.compare("vonMises") == 0)
997  || (scalar.first.compare("NetworkPressure") == 0)) {
998  factor = 1e6; // to get Pascal
999  }
1000  else if (scalar.first.compare("DisplacementLengths") == 0) {
1001  factor = 0.001; // to get meter
1002  }
1003  else {
1004  factor = 1.0;
1005  }
1006 
1007  SMDS_NodeIteratorPtr aNodeIter = meshDS->nodesIterator();
1008  for (double i : vec) {
1009  const SMDS_MeshNode* node = aNodeIter->next();
1010  // for the MassFlowRate the last vec entries can be a nullptr, thus check this
1011  if (node) {
1012  data->SetValue(node->GetID() - 1, i * factor);
1013  }
1014  }
1015 
1016  grid->GetPointData()->AddArray(data);
1017  Base::Console().Log(
1018  " The PropertyFloatList %s was exported to VTK scalar list: %s\n",
1019  scalar.first.c_str(),
1020  scalar.second.c_str());
1021  }
1022  else if (field) {
1023  Base::Console().Log(" PropertyFloatList NOT exported to vtk: %s size is: %i\n",
1024  scalar.first.c_str(),
1025  field->getSize());
1026  }
1027  }
1028 
1029  Base::Console().Log("End: Create VTK result data from FreeCAD result data.\n");
1030 }
