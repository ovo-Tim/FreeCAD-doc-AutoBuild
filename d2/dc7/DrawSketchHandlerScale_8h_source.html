87  void updateDataAndDrawToPosition(Base::Vector2d onSketchPos) override
88  {
89  switch (state()) {
90  case SelectMode::SeekFirst: {
91  referencePoint = onSketchPos;
92  } break;
93  case SelectMode::SeekSecond: {
94  refLength = (onSketchPos - referencePoint).Length();
95 
96  startPoint = onSketchPos;

102  endPoint = onSketchPos;
103  scaleFactor = length / refLength;

106  } break;
107  default:
108  break;

226  void createShape(bool onlyeditoutline) override
227  {
228  SketchObject* Obj = sketchgui->getSketchObject();
229 
230  ShapeGeometry.clear();
231 
232  if (state() == SelectMode::SeekSecond) {
233  if (refLength > Precision::Confusion()) {
234  addLineToShapeGeometry(toVector3d(referencePoint), toVector3d(startPoint), true);
235  }
236  return;
237  }
238 
239  if (fabs(scaleFactor) < Precision::Confusion()) {
240  return;
241  }
242 
243  for (auto& geoId : listOfGeoIds) {
244  const Part::Geometry* pGeo = Obj->getGeometry(geoId);
245  auto geoUniquePtr = std::unique_ptr<Part::Geometry>(pGeo->copy());
246  Part::Geometry* geo = geoUniquePtr.get();
247 
248  if (isCircle(*geo)) {
249  auto* circle = static_cast<Part::GeomCircle*>(geo); // NOLINT
250  circle->setRadius(circle->getRadius() * scaleFactor);
251  circle->setCenter(getScaledPoint(circle->getCenter(), referencePoint, scaleFactor));
252  }
253  else if (isArcOfCircle(*geo)) {
254  auto* arcOfCircle = static_cast<Part::GeomArcOfCircle*>(geo); // NOLINT
255  arcOfCircle->setRadius(arcOfCircle->getRadius() * scaleFactor);
256  arcOfCircle->setCenter(
257  getScaledPoint(arcOfCircle->getCenter(), referencePoint, scaleFactor));
258  }
259  else if (isLineSegment(*geo)) {
260  auto* line = static_cast<Part::GeomLineSegment*>(geo); // NOLINT
261  line->setPoints(getScaledPoint(line->getStartPoint(), referencePoint, scaleFactor),
262  getScaledPoint(line->getEndPoint(), referencePoint, scaleFactor));
263  }
264  else if (isEllipse(*geo)) {
265  auto* ellipse = static_cast<Part::GeomEllipse*>(geo); // NOLINT
266  ellipse->setMajorRadius(ellipse->getMajorRadius() * scaleFactor);
267  ellipse->setMinorRadius(ellipse->getMinorRadius() * scaleFactor);
268  ellipse->setCenter(
269  getScaledPoint(ellipse->getCenter(), referencePoint, scaleFactor));
270  }
271  else if (isArcOfEllipse(*geo)) {
272  auto* arcOfEllipse = static_cast<Part::GeomArcOfEllipse*>(geo); // NOLINT
273  arcOfEllipse->setMajorRadius(arcOfEllipse->getMajorRadius() * scaleFactor);
274  arcOfEllipse->setMinorRadius(arcOfEllipse->getMinorRadius() * scaleFactor);
275  arcOfEllipse->setCenter(
276  getScaledPoint(arcOfEllipse->getCenter(), referencePoint, scaleFactor));
277  }
278  else if (isArcOfHyperbola(*geo)) {
279  auto* arcOfHyperbola = static_cast<Part::GeomArcOfHyperbola*>(geo); // NOLINT
280  arcOfHyperbola->setMajorRadius(arcOfHyperbola->getMajorRadius() * scaleFactor);
281  arcOfHyperbola->setMinorRadius(arcOfHyperbola->getMinorRadius() * scaleFactor);
282  arcOfHyperbola->setCenter(
283  getScaledPoint(arcOfHyperbola->getCenter(), referencePoint, scaleFactor));
284  }
285  else if (isArcOfParabola(*geo)) {
286  auto* arcOfParabola = static_cast<Part::GeomArcOfParabola*>(geo); // NOLINT
287  arcOfParabola->setFocal(arcOfParabola->getFocal() * scaleFactor);
288  double start, end;
289  arcOfParabola->getRange(start, end, true);
290  arcOfParabola->setRange(start * scaleFactor, end * scaleFactor, true);
291  arcOfParabola->setCenter(
292  getScaledPoint(arcOfParabola->getCenter(), referencePoint, scaleFactor));
293  }
294  else if (isBSplineCurve(*geo)) {
295  auto* bSpline = static_cast<Part::GeomBSplineCurve*>(geo); // NOLINT
296  std::vector<Base::Vector3d> poles = bSpline->getPoles();
297  for (size_t p = 0; p < poles.size(); p++) {
298  poles[p] = getScaledPoint(std::move(poles[p]), referencePoint, scaleFactor);
299  }
300  bSpline->setPoles(poles);
301  }
302  else if (isPoint(*geo)) {
303  auto* point = static_cast<Part::GeomPoint*>(geo);
304  point->setPoint(getScaledPoint(point->getPoint(), referencePoint, scaleFactor));
305  }
306 
307  ShapeGeometry.emplace_back(std::move(geoUniquePtr));
308  }
309 
310  if (onlyeditoutline) {
311  // Add the lines to show lengths
312  addLineToShapeGeometry(toVector3d(referencePoint), toVector3d(startPoint), true);
313 
314  addLineToShapeGeometry(toVector3d(referencePoint), toVector3d(endPoint), true);
315  }
316  else {
317  int firstCurveCreated = getHighestCurveIndex() + 1;
318 
319  const std::vector<Constraint*>& vals = Obj->Constraints.getValues();
320  // avoid applying equal several times if cloning distanceX and distanceY of the
321  // same part.
322  std::vector<int> geoIdsWhoAlreadyHasEqual = {};
323 
324  for (auto& cstr : vals) {
325  int firstIndex = indexOfGeoId(listOfGeoIds, cstr->First);
326  int secondIndex = indexOfGeoId(listOfGeoIds, cstr->Second);
327  int thirdIndex = indexOfGeoId(listOfGeoIds, cstr->Third);
328 
329  auto newConstr = std::unique_ptr<Constraint>(cstr->copy());
330  newConstr->First = firstCurveCreated + firstIndex;
331 
332  if ((cstr->Type == Symmetric || cstr->Type == Tangent || cstr->Type == Perpendicular
333  || cstr->Type == Angle)
334  && firstIndex >= 0 && secondIndex >= 0 && thirdIndex >= 0) {
335  newConstr->Second = firstCurveCreated + secondIndex;
336  newConstr->Third = firstCurveCreated + thirdIndex;
337  }
338  else if ((cstr->Type == Coincident || cstr->Type == Tangent
339  || cstr->Type == Symmetric || cstr->Type == Perpendicular
340  || cstr->Type == Parallel || cstr->Type == Equal || cstr->Type == Angle
341  || cstr->Type == PointOnObject || cstr->Type == InternalAlignment)
342  && firstIndex >= 0 && secondIndex >= 0
343  && thirdIndex == GeoEnum::GeoUndef) {
344  newConstr->Second = firstCurveCreated + secondIndex;
345  }
346  else if ((cstr->Type == Radius || cstr->Type == Diameter) && firstIndex >= 0) {
347  newConstr->setValue(newConstr->getValue() * scaleFactor);
348  }
349  else if ((cstr->Type == Distance || cstr->Type == DistanceX
350  || cstr->Type == DistanceY)
351  && firstIndex >= 0 && secondIndex >= 0) {
352  newConstr->Second = firstCurveCreated + secondIndex;
353  newConstr->setValue(newConstr->getValue() * scaleFactor);
354  }
355  else if ((cstr->Type == Block || cstr->Type == Weight) && firstIndex >= 0) {
356  newConstr->First = firstCurveCreated + firstIndex;
357  }
358  else {
359  continue;
360  }
361 
362  ShapeConstraints.push_back(std::move(newConstr));
363  }
364  }
365  }

367  Base::Vector3d getScaledPoint(Base::Vector3d&& pointToScale,
368  const Base::Vector2d& referencePoint,
369  double scaleFactor)
370  {
371  Base::Vector2d pointToScale2D;
372  pointToScale2D.x = pointToScale.x;
373  pointToScale2D.y = pointToScale.y;
374  pointToScale2D = (pointToScale2D - referencePoint) * scaleFactor + referencePoint;
375 
376  pointToScale.x = pointToScale2D.x;
377  pointToScale.y = pointToScale2D.y;
378 
379  return pointToScale;
380  }

425 template<>
426 void DSHScaleControllerBase::doEnforceControlParameters(Base::Vector2d& onSketchPos)
427 {
428  switch (handler->state()) {
429  case SelectMode::SeekFirst: {
430  if (onViewParameters[OnViewParameter::First]->isSet) {
431  onSketchPos.x = onViewParameters[OnViewParameter::First]->getValue();
432  }
433 
434  if (onViewParameters[OnViewParameter::Second]->isSet) {
435  onSketchPos.y = onViewParameters[OnViewParameter::Second]->getValue();
436  }
437  } break;
438  case SelectMode::SeekThird: {
439  if (onViewParameters[OnViewParameter::Third]->isSet) {
440  double scaleFactor = onViewParameters[OnViewParameter::Third]->getValue();
441  handler->refLength = 1.0;
442  handler->startPoint = handler->referencePoint + Base::Vector2d(1.0, 0.0);
443  handler->endPoint = handler->referencePoint + Base::Vector2d(scaleFactor, 0.0);
444 
445  onSketchPos = handler->endPoint;
446  }
447  } break;
448  default:
449  break;
450  }
