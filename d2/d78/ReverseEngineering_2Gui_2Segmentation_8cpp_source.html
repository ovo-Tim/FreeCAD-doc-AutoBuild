70 void Segmentation::accept()
71 {
72  if (myMesh.expired()) {
73  return;
74  }
75 
76  Gui::WaitCursor wc;
77  bool createUnused = ui->createUnused->isChecked();
78  bool createCompound = ui->createCompound->isChecked();
79  BRep_Builder builder;
80  TopoDS_Compound compound;
81  builder.MakeCompound(compound);
82 
83  const Mesh::MeshObject* mesh = myMesh.get<Mesh::Feature>()->Mesh.getValuePtr();
84  // make a copy because we might smooth the mesh before
85  MeshCore::MeshKernel kernel = mesh->getKernel();
86  MeshCore::MeshAlgorithm algo(kernel);
87 
88  if (ui->checkBoxSmooth->isChecked()) {
89  MeshCore::LaplaceSmoothing smoother(kernel);
90  smoother.Smooth(ui->smoothSteps->value());
91  }
92 
93  MeshCore::MeshSegmentAlgorithm finder(kernel);
94  MeshCore::MeshCurvature meshCurv(kernel);
95  meshCurv.ComputePerVertex();
96 
97  // First create segments by curavture to get the surface type
98  std::vector<MeshCore::MeshSurfaceSegmentPtr> segm;
99  if (ui->groupBoxPln->isChecked()) {
100  segm.emplace_back(
101  std::make_shared<MeshCore::MeshCurvaturePlanarSegment>(meshCurv.GetCurvature(),
102  ui->numPln->value(),
103  ui->curvTolPln->value()));
104  }
105  finder.FindSegments(segm);
106 
107  std::vector<MeshCore::MeshSurfaceSegmentPtr> segmSurf;
108  for (const auto& it : segm) {
109  const std::vector<MeshCore::MeshSegment>& data = it->GetSegments();
110 
111  // For each planar segment compute a plane and use this then for a more accurate 2nd
112  // segmentation
113  if (strcmp(it->GetType(), "Plane") == 0) {
114  for (const auto& jt : data) {
115  std::vector<MeshCore::PointIndex> indexes = kernel.GetFacetPoints(jt);
116  MeshCore::PlaneFit fit;
117  fit.AddPoints(kernel.GetPoints(indexes));
118  if (fit.Fit() < FLOAT_MAX) {
119  Base::Vector3f base = fit.GetBase();
120  Base::Vector3f axis = fit.GetNormal();
121  MeshCore::AbstractSurfaceFit* fitter =
122  new MeshCore::PlaneSurfaceFit(base, axis);
123  segmSurf.emplace_back(
124  std::make_shared<MeshCore::MeshDistanceGenericSurfaceFitSegment>(
125  fitter,
126  kernel,
127  ui->numPln->value(),
128  ui->distToPln->value()));
129  }
130  }
131  }
132  }
133  finder.FindSegments(segmSurf);
134 
135  App::Document* document = App::GetApplication().getActiveDocument();
136  document->openTransaction("Segmentation");
137 
138  std::string internalname = "Segments_";
139  internalname += myMesh->getNameInDocument();
140 
141  App::DocumentObjectGroup* group = static_cast<App::DocumentObjectGroup*>(
142  document->addObject("App::DocumentObjectGroup", internalname.c_str()));
143  std::string labelname = "Segments ";
144  labelname += myMesh->Label.getValue();
145  group->Label.setValue(labelname);
146 
147  std::vector<App::DocumentObject*> failures;
148  algo.SetFacetFlag(MeshCore::MeshFacet::TMP0);
149 
150  for (const auto& it : segmSurf) {
151  const std::vector<MeshCore::MeshSegment>& data = it->GetSegments();
152  std::shared_ptr<MeshCore::MeshDistanceGenericSurfaceFitSegment> genSegm =
153  std::dynamic_pointer_cast<MeshCore::MeshDistanceGenericSurfaceFitSegment>(it);
154 
155  bool isPlanar = (strcmp(genSegm->GetType(), "Plane") == 0);
156  for (const auto& jt : data) {
157  // reset flag for facets of segment
158  algo.ResetFacetsFlag(jt, MeshCore::MeshFacet::TMP0);
159 
160  Mesh::MeshObject* segment = mesh->meshFromSegment(jt);
161  Mesh::Feature* feaSegm =
162  static_cast<Mesh::Feature*>(group->addObject("Mesh::Feature", "Segment"));
163  Mesh::MeshObject* feaMesh = feaSegm->Mesh.startEditing();
164  feaMesh->swap(*segment);
165  feaSegm->Mesh.finishEditing();
166  delete segment;
167 
168  std::stringstream label;
169  label << feaSegm->Label.getValue() << " (" << it->GetType() << ")";
170  feaSegm->Label.setValue(label.str());
171 
172  if (createCompound) {
173  std::list<std::vector<Base::Vector3f>> bounds;
174  algo.GetFacetBorders(jt, bounds);
175 
176  // Handle planar segments
177  if (isPlanar) {
178  std::vector<float> par = genSegm->Parameters();
179  gp_Pnt loc(par.at(0), par.at(1), par.at(2));
180  gp_Dir dir(par.at(3), par.at(4), par.at(5));
181 
182  Handle(Geom_Plane) hPlane(new Geom_Plane(loc, dir));
183 
184  std::vector<TopoDS_Wire> wires;
185  for (const auto& bound : bounds) {
186  // project the points onto the surface
187  std::vector<gp_Pnt> polygon;
188  std::transform(
189  bound.begin(),
190  bound.end(),
191  std::back_inserter(polygon),
192  [&hPlane](const Base::Vector3f& v) {
193  gp_Pnt p(v.x, v.y, v.z);
194  return GeomAPI_ProjectPointOnSurf(p, hPlane).NearestPoint();
195  });
196 
197  BRepBuilderAPI_MakePolygon mkPoly;
198  for (std::vector<gp_Pnt>::reverse_iterator it = polygon.rbegin();
199  it != polygon.rend();
200  ++it) {
201  mkPoly.Add(*it);
202  }
203  if (mkPoly.IsDone()) {
204  wires.push_back(mkPoly.Wire());
205  }
206  }
207 
208  try {
209  TopoDS_Shape shape = Part::FaceMakerCheese::makeFace(wires);
210  if (!shape.IsNull()) {
211  builder.Add(compound, shape);
212  }
213  else {
214  failures.push_back(feaSegm);
215  Base::Console().Warning("Failed to create face from %s\n",
216  feaSegm->Label.getValue());
217  }
218  }
219  catch (Standard_Failure&) {
220  failures.push_back(feaSegm);
221  Base::Console().Error("Fatal failure to create face from %s\n",
222  feaSegm->Label.getValue());
223  }
224  }
225  }
226  }
227  }
228 
229  if (createUnused) {
230  // collect all facets that don't have set the flag TMP0
231  std::vector<MeshCore::FacetIndex> unusedFacets;
232  algo.GetFacetsFlag(unusedFacets, MeshCore::MeshFacet::TMP0);
233 
234  if (!unusedFacets.empty()) {
235  std::unique_ptr<Mesh::MeshObject> segment(mesh->meshFromSegment(unusedFacets));
236  Mesh::Feature* feaSegm =
237  static_cast<Mesh::Feature*>(group->addObject("Mesh::Feature", "Unused"));
238  Mesh::MeshObject* feaMesh = feaSegm->Mesh.startEditing();
239  feaMesh->swap(*segment);
240  feaSegm->Mesh.finishEditing();
241  }
242  }
243  if (createCompound) {
244  Part::Feature* shapeFea =
245  static_cast<Part::Feature*>(group->addObject("Part::Feature", "Compound"));
246  shapeFea->Shape.setValue(compound);
247 
248  // create a sub-group where to move the problematic segments
249  if (!failures.empty()) {
250  App::DocumentObjectGroup* subgroup = static_cast<App::DocumentObjectGroup*>(
251  group->addObject("App::DocumentObjectGroup", "Failed"));
252  failures = group->removeObjects(failures);
253  subgroup->Group.setValues(failures);
254  }
255  }
256 
257  document->commitTransaction();
258 }
