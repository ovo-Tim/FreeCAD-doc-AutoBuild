68  def val(self, enable, placement=None, offset=0.0):
69  """val(enable, placement = None, offset = 0.0): constructs a value from convenient
70  parameters. Placement is in global CS. The cutting will be by XY plane of Placement;
71  the stuff in negative Z is visible and the stuff in positive Z is invisible."""
72 
73  pldef = None
74  if enable:
75  obj = self.doc.getObject(self.objname)
76  plm_cs = obj.getGlobalPlacement().multiply(
77  obj.Placement.inverse()
78  ) # placement of CS the object is in
79  plm_plane = plm_cs.inverse().multiply(placement)
80  pldef = placement2plane(plm_plane, offset)
81  return (enable, pldef if enable else None)

84 def getClipPlaneNode(viewprovider, make_if_missing=True):
85  from pivy import coin
86 
87  sa = coin.SoSearchAction()
88  sa.setType(coin.SoClipPlane.getClassTypeId())
89  sa.setName("TVClipPlane")
90  sa.traverse(viewprovider.RootNode)
91  if sa.isFound() and sa.getPath().getLength() == 1:
92  return sa.getPath().getTail()
93  elif not sa.isFound():
94  if not make_if_missing:
95  return None
96  clipplane = coin.SoClipPlane()
97  viewprovider.RootNode.insertChild(clipplane, 0)
98  clipplane.setName("TVClipPlane")
99  clipplane.on.setValue(False) # make sure the plane is not activated by default
100  return clipplane

103 def placement2plane(placement, offset):
104  """returns tuple (normal, D) for making coin plane."""
105  normal = placement.Rotation.multVec(App.Vector(0, 0, -1))
106  D = placement.Base * normal - offset
107  return tuple(normal), D
