83 bool ViewProviderPart::isActivePart()
84 {
85  App::DocumentObject* activePart = nullptr;
86  auto activeDoc = Gui::Application::Instance->activeDocument();
87  if(!activeDoc)
88  activeDoc = getDocument();
89  auto activeView = activeDoc->setActiveView(this);
90  if(!activeView)
91  return false;
92 
93  activePart = activeView->getActiveObject<App::DocumentObject*> (PARTKEY);
94 
95  if (activePart == this->getObject()){
96  return true;
97  } else {
98  return false;
99  }
100 }

102 void ViewProviderPart::toggleActivePart()
103 {
104  //make the part the active one
105  if (isActivePart()){
106  //active part double-clicked. Deactivate.
107  Gui::Command::doCommand(Gui::Command::Gui,
108  "Gui.ActiveDocument.ActiveView.setActiveObject('%s', None)",
109  PARTKEY);
110  } else {
111  //set new active part
112  Gui::Command::doCommand(Gui::Command::Gui,
113  "Gui.ActiveDocument.ActiveView.setActiveObject('%s', App.getDocument('%s').getObject('%s'))",
114  PARTKEY,
115  this->getObject()->getDocument()->getName(),
116  this->getObject()->getNameInDocument());
117  }
118 }
