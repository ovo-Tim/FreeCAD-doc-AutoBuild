669  if baseobj:
670  obj.Base = baseobj

673  if FreeCAD.GuiUp:
674  obj.Base.ViewObject.hide()
675  else:

677  baseWire = obj.Base.Shape.Faces[obj.Face-1].Wires[0]

680  elif obj.Base.Shape.Wires:
681  baseWire = obj.Base.Shape.Wires[0]
682  if FreeCAD.GuiUp:
683  obj.Base.ViewObject.hide()

685  if baseWire.isClosed():
686  if FreeCAD.GuiUp:
687  obj.Base.ViewObject.hide()
688  edges = Part.__sortEdges__(baseWire.Edges)

692  obj.IdRel = ArchRoof.adjust_list_len(idrel, ln, idrel[0])
693  obj.Thickness = ArchRoof.adjust_list_len(thickness, ln, thickness[0])
