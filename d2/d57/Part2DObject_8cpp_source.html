101 bool Part2DObject::seekTrimPoints(const std::vector<Geometry *> &geomlist,
102  int geometryIndex, const Base::Vector3d &point,
103  int &geometryIndex1, Base::Vector3d &intersect1,
104  int &geometryIndex2, Base::Vector3d &intersect2)
105 {
106  if ( geometryIndex >= int(geomlist.size()))
107  return false;
108 
109  gp_Pln plane(gp_Pnt(0,0,0),gp_Dir(0,0,1));
110 
111  Standard_Boolean periodic=Standard_False;
112  double period = 0;
113  Handle(Geom2d_Curve) primaryCurve;
114  Handle(Geom_Geometry) geom = (geomlist[geometryIndex])->handle();
115  Handle(Geom_Curve) curve3d = Handle(Geom_Curve)::DownCast(geom);
116 
117  if (curve3d.IsNull())
118  return false;
119  else {
120  primaryCurve = GeomAPI::To2d(curve3d, plane);
121  periodic = primaryCurve->IsPeriodic();
122  if (periodic)
123  period = primaryCurve->Period();
124  }
125 
126  // create the intersector and projector functions
127  Geom2dAPI_InterCurveCurve Intersector;
128  Geom2dAPI_ProjectPointOnCurve Projector;
129 
130  // find the parameter of the picked point on the primary curve
131  Projector.Init(gp_Pnt2d(point.x, point.y), primaryCurve);
132  double pickedParam = Projector.LowerDistanceParameter();
133 
134  // find intersection points
135  geometryIndex1 = -1;
136  geometryIndex2 = -1;
137  double param1=-1e10,param2=1e10;
138  gp_Pnt2d p1,p2;
139  Handle(Geom2d_Curve) secondaryCurve;
140  for (int id=0; id < int(geomlist.size()); id++) {
141  // #0000624: Trim tool doesn't work with construction lines
142  if (id != geometryIndex/* && !geomlist[id]->Construction*/) {
143  geom = (geomlist[id])->handle();
144  curve3d = Handle(Geom_Curve)::DownCast(geom);
145  if (!curve3d.IsNull()) {
146  secondaryCurve = GeomAPI::To2d(curve3d, plane);
147  // perform the curves intersection
148 
149  std::vector<gp_Pnt2d> points;
150 
151  // #2463 Check for endpoints of secondarycurve on primary curve
152  // If the OCCT Intersector should detect endpoint tangency when trimming, then
153  // this is just a work-around until that bug is fixed.
154  // https://www.freecad.org/tracker/view.php?id=2463
155  // https://tracker.dev.opencascade.org/view.php?id=30217
156  if (geomlist[id]->isDerivedFrom<Part::GeomBoundedCurve>()) {
157 
158  Part::GeomBoundedCurve * bcurve = static_cast<Part::GeomBoundedCurve *>(geomlist[id]);
159 
160  points.emplace_back(bcurve->getStartPoint().x,bcurve->getStartPoint().y);
161  points.emplace_back(bcurve->getEndPoint().x,bcurve->getEndPoint().y);
162  }
163 
164  Intersector.Init(primaryCurve, secondaryCurve, 1.0e-12);
165 
166  for (int i=1; i <= Intersector.NbPoints(); i++)
167  points.push_back(Intersector.Point(i));
168 
169  if (Intersector.NbSegments() > 0) {
170  const Geom2dInt_GInter& gInter = Intersector.Intersector();
171  for (int i=1; i <= gInter.NbSegments(); i++) {
172  const IntRes2d_IntersectionSegment& segm = gInter.Segment(i);
173  if (segm.HasFirstPoint()) {
174  const IntRes2d_IntersectionPoint& fp = segm.FirstPoint();
175  points.push_back(fp.Value());
176  }
177  if (segm.HasLastPoint()) {
178  const IntRes2d_IntersectionPoint& fp = segm.LastPoint();
179  points.push_back(fp.Value());
180  }
181  }
182  }
183 
184  for (auto p : points) {
185  // get the parameter of the intersection point on the primary curve
186  Projector.Init(p, primaryCurve);
187 
188  if (Projector.NbPoints()<1 || Projector.LowerDistance() > Precision::Confusion())
189  continue;
190 
191  double param = Projector.LowerDistanceParameter();
192 
193  if (periodic) {
194  // transfer param into the interval (pickedParam-period pickedParam]
195  param = param - period * ceil((param-pickedParam) / period);
196  if (param > param1) {
197  param1 = param;
198  p1 = p;
199  geometryIndex1 = id;
200  }
201  param -= period; // transfer param into the interval (pickedParam pickedParam+period]
202  if (param < param2) {
203  param2 = param;
204  p2 = p;
205  geometryIndex2 = id;
206  }
207  }
208  else if (param < pickedParam && param > param1) {
209  param1 = param;
210  p1 = p;
211  geometryIndex1 = id;
212  }
213  else if (param > pickedParam && param < param2) {
214  param2 = param;
215  p2 = p;
216  geometryIndex2 = id;
217  }
218  }
219  }
220  }
221  }
222 
223  if (periodic) {
224  // in case both points coincide, cancel the selection of one of both
225  if (fabs(param2-param1-period) < 1e-10) {
226  if (param2 - pickedParam >= pickedParam - param1)
227  geometryIndex2 = -1;
228  else
229  geometryIndex1 = -1;
230  }
231  }
232 
233  if ( geometryIndex1 < 0 && geometryIndex2 < 0)
234  return false;
235 
236  if ( geometryIndex1 >= 0)
237  intersect1 = Base::Vector3d(p1.X(),p1.Y(),0.f);
238  if ( geometryIndex2 >= 0)
239  intersect2 = Base::Vector3d(p2.X(),p2.Y(),0.f);
240  return true;
241 }
