389 void TaskExtrudeParameters::selectedShapeFace(const Gui::SelectionChanges& msg)
390 {
391  auto extrude = getObject<PartDesign::FeatureExtrude>();
392  auto document = extrude->getDocument();
393 
394  if (strcmp(msg.pDocName, document->getName()) != 0) {
395  return;
396  }
397 
398  auto base = static_cast<Part::Feature*>(extrude->UpToShape.getValue());
399  if (!base){
400  base = static_cast<Part::Feature*>(extrude);
401  }
402  else if (strcmp(msg.pObjectName, base->getNameInDocument()) != 0) {
403  return;
404  }
405 
406  std::vector<std::string> faces = getShapeFaces();
407  std::string subName(msg.pSubName);
408 
409  if (subName.empty()) {
410  return;
411  }
412 
413  auto positionInList = std::find(faces.begin(), faces.end(), subName);
414 
415  if (positionInList != faces.end()) { // If it's found then it's in the list so we remove it.
416  faces.erase(positionInList);
417  }
418  else { // if it's not found then it's not yet in the list so we add it.
419  faces.push_back(subName);
420  }
421 
422  extrude->UpToShape.setValue(base, faces);
423 
424  updateShapeFaces();
425 
426  tryRecomputeFeature();
427 }

585 void TaskExtrudeParameters::fillDirectionCombo()
586 {
587  Base::StateLocker lock(getUpdateBlockRef(), true);
588 
589  if (axesInList.empty()) {
590  bool hasFace = false;
591  ui->directionCB->clear();
592  // we can have sketches or faces
593  // for sketches just get the sketch normal
594  auto pcFeat = getObject<PartDesign::ProfileBased>();
595  Part::Part2DObject* pcSketch =
596  dynamic_cast<Part::Part2DObject*>(pcFeat->Profile.getValue());
597  // for faces we test if it is verified and if we can get its normal
598  if (!pcSketch) {
599  hasFace = hasProfileFace(pcFeat);
600  }
601 
602  if (pcSketch) {
603  addAxisToCombo(pcSketch, "N_Axis", tr("Sketch normal"));
604  }
605  else if (hasFace) {
606  addAxisToCombo(pcFeat->Profile.getValue(), std::string(), tr("Face normal"), false);
607  }
608 
609  // add the other entries
610  addAxisToCombo(nullptr, std::string(), tr("Select reference..."));
611 
612  // we start with the sketch normal as proposal for the custom direction
613  if (pcSketch) {
614  addAxisToCombo(pcSketch, "N_Axis", tr("Custom direction"));
615  }
616  else if (hasFace) {
617  addAxisToCombo(pcFeat->Profile.getValue(),
618  std::string(),
619  tr("Custom direction"),
620  false);
621  }
622  }
623 
624  // add current link, if not in list
625  // first, figure out the item number for current axis
626  int indexOfCurrent = -1;
627  App::DocumentObject* ax = propReferenceAxis->getValue();
628  const std::vector<std::string>& subList = propReferenceAxis->getSubValues();
629  for (size_t i = 0; i < axesInList.size(); i++) {
630  if (ax == axesInList[i]->getValue() && subList == axesInList[i]->getSubValues()) {
631  indexOfCurrent = i;
632  break;
633  }
634  }
635  // if the axis is not yet listed in the combobox
636  if (indexOfCurrent == -1 && ax) {
637  assert(subList.size() <= 1);
638  std::string sub;
639  if (!subList.empty()) {
640  sub = subList[0];
641  }
642  addAxisToCombo(ax, sub, getRefStr(ax, subList));
643  indexOfCurrent = axesInList.size() - 1;
644  // the axis is not the normal, thus enable along direction
645  ui->checkBoxAlongDirection->setEnabled(true);
646  // we don't have custom direction thus disable its settings
647  ui->XDirectionEdit->setEnabled(false);
648  ui->YDirectionEdit->setEnabled(false);
649  ui->ZDirectionEdit->setEnabled(false);
650  }
651 
652  // highlight either current index or set custom direction
653  auto extrude = getObject<PartDesign::FeatureExtrude>();
654  bool hasCustom = extrude->UseCustomVector.getValue();
655  if (indexOfCurrent != -1 && !hasCustom) {
656  ui->directionCB->setCurrentIndex(indexOfCurrent);
657  }
658  if (hasCustom) {
659  ui->directionCB->setCurrentIndex(DirectionModes::Custom);
660  }
661 }
