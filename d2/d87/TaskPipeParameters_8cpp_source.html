178 void TaskPipeParameters::onSelectionChanged(const Gui::SelectionChanges& msg)
179 {
180  if (stateHandler->getSelectionMode() == StateHandlerTaskPipe::SelectionModes::none) {
181  return;
182  }
183 
184  if (msg.Type == Gui::SelectionChanges::AddSelection) {
185  if (referenceSelected(msg)) {
186  if (stateHandler->getSelectionMode()
187  == StateHandlerTaskPipe::SelectionModes::refProfile) {
188  App::Document* document = App::GetApplication().getDocument(msg.pDocName);
189  App::DocumentObject* object =
190  document ? document->getObject(msg.pObjectName) : nullptr;
191  if (object) {
192  QString label = make2DLabel(object, {msg.pSubName});
193  ui->profileBaseEdit->setText(label);
194  }
195  }
196  else if (stateHandler->getSelectionMode()
197  == StateHandlerTaskPipe::SelectionModes::refSpineEdgeAdd) {
198  QString sub = QString::fromStdString(msg.pSubName);
199  if (!sub.isEmpty()) {
200  QListWidgetItem* item = new QListWidgetItem();
201  item->setText(sub);
202  item->setData(Qt::UserRole, QByteArray(msg.pSubName));
203  ui->listWidgetReferences->addItem(item);
204  }
205 
206  App::Document* document = App::GetApplication().getDocument(msg.pDocName);
207  App::DocumentObject* object =
208  document ? document->getObject(msg.pObjectName) : nullptr;
209  if (object) {
210  QString label = QString::fromUtf8(object->Label.getValue());
211  ui->spineBaseEdit->setText(label);
212  }
213  }
214  else if (stateHandler->getSelectionMode()
215  == StateHandlerTaskPipe::SelectionModes::refSpineEdgeRemove) {
216  QString sub = QString::fromLatin1(msg.pSubName);
217  if (!sub.isEmpty()) {
218  removeFromListWidget(ui->listWidgetReferences, sub);
219  }
220  else {
221  ui->spineBaseEdit->clear();
222  }
223  }
224  else if (stateHandler->getSelectionMode()
225  == StateHandlerTaskPipe::SelectionModes::refSpine) {
226  ui->listWidgetReferences->clear();
227 
228  App::Document* document = App::GetApplication().getDocument(msg.pDocName);
229  App::DocumentObject* object =
230  document ? document->getObject(msg.pObjectName) : nullptr;
231  if (object) {
232  QString label = QString::fromUtf8(object->Label.getValue());
233  ui->spineBaseEdit->setText(label);
234  }
235  }
236 
237  clearButtons();
238  recomputeFeature();
239  }
240 
241  clearButtons();
242  exitSelectionMode();
243  }

287 void TaskPipeParameters::onDeleteEdge()
288 {
289  // Delete the selected path edge
290  int row = ui->listWidgetReferences->currentRow();
291  QListWidgetItem* item = ui->listWidgetReferences->takeItem(row);
292  if (item) {
293  QByteArray data = item->data(Qt::UserRole).toByteArray();
294  delete item;
295 
296  // search inside the list of spines
297  auto pipe = getObject<PartDesign::Pipe>();
298  std::vector<std::string> refs = pipe->Spine.getSubValues();
299  std::string obj = data.constData();
300  std::vector<std::string>::iterator f = std::find(refs.begin(), refs.end(), obj);
301 
302  // if something was found, delete it and update the spine list
303  if (f != refs.end()) {
304  refs.erase(f);
305  pipe->Spine.setValue(pipe->Spine.getValue(), refs);
306  clearButtons();
307  recomputeFeature();
308  }
309  }

312 bool TaskPipeParameters::referenceSelected(const SelectionChanges& msg) const
313 {
314  auto selectionMode = stateHandler->getSelectionMode();
315 
316  if (msg.Type == Gui::SelectionChanges::AddSelection
317  && selectionMode != StateHandlerTaskPipe::SelectionModes::none) {
318  if (strcmp(msg.pDocName, getAppDocument()->getName()) != 0) {
319  return false;
320  }
321 
322  // not allowed to reference ourself
323  const char* fname = getObject()->getNameInDocument();
324  if (strcmp(msg.pObjectName, fname) == 0) {
325  return false;
326  }
327 
328  switch (selectionMode) {
329  case StateHandlerTaskPipe::SelectionModes::refProfile: {
330  auto pipe = getObject<PartDesign::Pipe>();
331  Gui::Document* doc = getGuiDocument();
332 
333  getViewObject<ViewProviderPipe>()->highlightReferences(ViewProviderPipe::Profile,
334  false);
335 
336  bool success = true;
337  App::DocumentObject* profile = pipe->getDocument()->getObject(msg.pObjectName);
338  if (profile) {
339  std::vector<App::DocumentObject*> sections = pipe->Sections.getValues();
340 
341  // cannot use the same object for profile and section
342  if (std::find(sections.begin(), sections.end(), profile) != sections.end()) {
343  success = false;
344  }
345  else {
346  pipe->Profile.setValue(profile, {msg.pSubName});
347  }
348 
349  // hide the old or new profile again
350  auto* pvp = doc->getViewProvider(pipe->Profile.getValue());
351  if (pvp) {
352  pvp->setVisible(false);
353  }
354  }
355  return success;
356  }
357  case StateHandlerTaskPipe::SelectionModes::refSpine:
358  case StateHandlerTaskPipe::SelectionModes::refSpineEdgeAdd:
359  case StateHandlerTaskPipe::SelectionModes::refSpineEdgeRemove: {
360  // change the references
361  std::string subName(msg.pSubName);
362  auto pipe = getObject<PartDesign::Pipe>();
363  std::vector<std::string> refs = pipe->Spine.getSubValues();
364  std::vector<std::string>::iterator f = std::find(refs.begin(), refs.end(), subName);
365 
366  if (selectionMode == StateHandlerTaskPipe::SelectionModes::refSpine) {
367  getViewObject<ViewProviderPipe>()->highlightReferences(ViewProviderPipe::Spine,
368  false);
369  refs.clear();
370  }
371  else if (selectionMode == StateHandlerTaskPipe::SelectionModes::refSpineEdgeAdd) {
372  if (f == refs.end()) {
373  refs.push_back(subName);
374  }
375  else {
376  return false; // duplicate selection
377  }
378  }
379  else if (selectionMode
380  == StateHandlerTaskPipe::SelectionModes::refSpineEdgeRemove) {
381  if (f != refs.end()) {
382  refs.erase(f);
383  }
384  else {
385  return false;
386  }
387  }
388 
389  pipe->Spine.setValue(getAppDocument()->getObject(msg.pObjectName), refs);
390  return true;
391  }
392  default:
393  return false;
394  }
395  }
396 
397  return false;
398 }

443 bool TaskPipeParameters::accept()
444 {
445  // see what to do with external references
446  // check the prerequisites for the selected objects
447  // the user has to decide which option we should take if external references are used
448  auto pipe = getObject<PartDesign::Pipe>();
449  auto pcActiveBody = PartDesignGui::getBodyFor(pipe, false);
450  if (!pcActiveBody) {
451  QMessageBox::warning(this, tr("Input error"), tr("No active body"));
452  return false;
453  }
454  // auto pcActivePart = PartDesignGui::getPartFor (pcActiveBody, false);
455  std::vector<App::DocumentObject*> copies;
456 
457  bool extReference = false;
458  App::DocumentObject* spine = pipe->Spine.getValue();
459  App::DocumentObject* auxSpine = pipe->AuxillerySpine.getValue();
460 
461  // If a spine isn't set but user entered a label then search for the appropriate document object
462  QString label = ui->spineBaseEdit->text();
463  if (!spine && !label.isEmpty()) {
464  QByteArray ba = label.toUtf8();
465  std::vector<App::DocumentObject*> objs =
466  pipe->getDocument()->findObjects(App::DocumentObject::getClassTypeId(),
467  nullptr,
468  ba.constData());
469  if (!objs.empty()) {
470  pipe->Spine.setValue(objs.front());
471  spine = objs.front();
472  }
473  }
474 
475  if (spine && !pcActiveBody->hasObject(spine) && !pcActiveBody->getOrigin()->hasObject(spine)) {
476  extReference = true;
477  }
478  else if (auxSpine && !pcActiveBody->hasObject(auxSpine)
479  && !pcActiveBody->getOrigin()->hasObject(auxSpine)) {
480  extReference = true;
481  }
482  else {
483  for (App::DocumentObject* obj : pipe->Sections.getValues()) {
484  if (!pcActiveBody->hasObject(obj) && !pcActiveBody->getOrigin()->hasObject(obj)) {
485  extReference = true;
486  break;
487  }
488  }
489  }
490 
491  if (extReference) {
492  QDialog dia(Gui::getMainWindow());
493  Ui_DlgReference dlg;
494  dlg.setupUi(&dia);
495  dia.setModal(true);
496  int result = dia.exec();
497  if (result == QDialog::DialogCode::Rejected) {
498  return false;
499  }
500 
501  if (!dlg.radioXRef->isChecked()) {
502  if (!pcActiveBody->hasObject(spine) && !pcActiveBody->getOrigin()->hasObject(spine)) {
503  pipe->Spine.setValue(
504  PartDesignGui::TaskFeaturePick::makeCopy(spine,
505  "",
506  dlg.radioIndependent->isChecked()),
507  pipe->Spine.getSubValues());
508  copies.push_back(pipe->Spine.getValue());
509  }
510  else if (!pcActiveBody->hasObject(auxSpine)
511  && !pcActiveBody->getOrigin()->hasObject(auxSpine)) {
512  pipe->AuxillerySpine.setValue(
513  PartDesignGui::TaskFeaturePick::makeCopy(auxSpine,
514  "",
515  dlg.radioIndependent->isChecked()),
516  pipe->AuxillerySpine.getSubValues());
517  copies.push_back(pipe->AuxillerySpine.getValue());
518  }
519 
520  std::vector<App::PropertyLinkSubList::SubSet> subSets;
521  for (auto& subSet : pipe->Sections.getSubListValues()) {
522  if (!pcActiveBody->hasObject(subSet.first)
523  && !pcActiveBody->getOrigin()->hasObject(subSet.first)) {
524  subSets.emplace_back(
525  PartDesignGui::TaskFeaturePick::makeCopy(subSet.first,
526  "",
527  dlg.radioIndependent->isChecked()),
528  subSet.second);
529  copies.push_back(subSets.back().first);
530  }
531  else {
532  subSets.push_back(subSet);
533  }
534  }
535 
536  pipe->Sections.setSubListValues(subSets);
537  }
538  }
539 
540  try {
541  setVisibilityOfSpineAndProfile();
542 
543  App::DocumentObject* spine = pipe->Spine.getValue();
544  std::vector<std::string> subNames = pipe->Spine.getSubValues();
545  App::PropertyLinkT propT(spine, subNames);
546  Gui::cmdAppObjectArgs(pipe, "Spine = %s", propT.getPropertyPython());
547 
548  Gui::cmdAppDocument(pipe, "recompute()");
549  if (!getObject()->isValid()) {
550  throw Base::RuntimeError(getObject()->getStatusString());
551  }
552  Gui::cmdGuiDocument(pipe, "resetEdit()");
553  Gui::Command::commitCommand();
554 
555  // we need to add the copied features to the body after the command action, as otherwise
556  // FreeCAD crashes unexplainably
557  for (auto obj : copies) {
558  pcActiveBody->addObject(obj);
559  }
560  }
561  catch (const Base::Exception& e) {
562  QMessageBox::warning(this,
563  tr("Input error"),
564  QApplication::translate("Exception", e.what()));
565  return false;
566  }
567 
568  return true;
569 }

777 bool TaskPipeOrientation::referenceSelected(const SelectionChanges& msg) const
778 {

784  || selectionMode == StateHandlerTaskPipe::SelectionModes::refAuxSpineEdgeRemove)) {
785  if (strcmp(msg.pDocName, getObject()->getDocument()->getName()) != 0) {

796  // change the references
797  std::string subName(msg.pSubName);
798  std::vector<std::string> refs = pipe->AuxillerySpine.getSubValues();
799  std::vector<std::string>::iterator f = std::find(refs.begin(), refs.end(), subName);
800 
801  if (selectionMode == StateHandlerTaskPipe::SelectionModes::refAuxSpine) {
802  refs.clear();

805  if (f != refs.end()) {
