110 void TaskDressUpParameters::referenceSelected(const Gui::SelectionChanges& msg, QListWidget* widget)
111 {
112  if (strcmp(msg.pDocName, DressUpView->getObject()->getDocument()->getName()) != 0)
113  return;
114 
115  Gui::Selection().clearSelection();
116 
117  PartDesign::DressUp* pcDressUp = DressUpView->getObject<PartDesign::DressUp>();
118  App::DocumentObject* base = this->getBase();
119 
120  // TODO: Must we make a copy here instead of assigning to const char* ?
121  const char* fname = base->getNameInDocument();
122  if (strcmp(msg.pObjectName, fname) != 0)
123  return;
124 
125  std::string subName(msg.pSubName);
126  std::vector<std::string> refs = pcDressUp->Base.getSubValues();
127  std::vector<std::string>::iterator f = std::find(refs.begin(), refs.end(), subName);
128 
129  if (f != refs.end()) { //If it's found then it's in the list so we remove it.
130  refs.erase(f);
131  removeItemFromListWidget(widget, msg.pSubName);
132  }
133  else { //if it's not found then it's not yet in the list so we add it.
134  refs.push_back(subName);
135  widget->addItem(QString::fromStdString(msg.pSubName));
136  }
137 
138  updateFeature(pcDressUp, refs);
139 }

141 void TaskDressUpParameters::addAllEdges(QListWidget* widget)
142 {
143  Q_UNUSED(widget)
144 
145  if (DressUpView.expired()) {
146  return;
147  }
148 
149  PartDesign::DressUp* pcDressUp = DressUpView->getObject<PartDesign::DressUp>();
150  App::DocumentObject* base = pcDressUp->Base.getValue();
151  if (!base) {
152  return;
153  }
154  int count = Part::Feature::getTopoShape(base).countSubShapes(TopAbs_EDGE);
155  auto subValues = pcDressUp->Base.getSubValues(false);
156  std::size_t len = subValues.size();
157  for (int i = 0; i < count; ++i) {
158  std::string name = "Edge" + std::to_string(i+1);
159  if (std::find(subValues.begin(), subValues.begin() + len, name)
160  == subValues.begin() + len) {
161  subValues.push_back(name);
162  }
163  }
164  if (subValues.size() == len) {
165  return;
166  }
167  try {
168  setupTransaction();
169  pcDressUp->Base.setValue(base, subValues);
170  }
171  catch (Base::Exception& e) {
172  e.ReportException();
173  }
174 }

176 void TaskDressUpParameters::deleteRef(QListWidget* widget)
177 {
178  // delete any selections since the reference(s) being deleted might be highlighted
179  Gui::Selection().clearSelection();
180 
181  // get the list of items to be deleted
182  QList<QListWidgetItem*> selectedList = widget->selectedItems();
183 
184  PartDesign::DressUp* pcDressUp = DressUpView->getObject<PartDesign::DressUp>();
185  std::vector<std::string> refs = pcDressUp->Base.getSubValues();
186 
187  // delete the selection backwards to assure the list index keeps valid for the deletion
188  QSignalBlocker block(widget);
189  for (int i = selectedList.count() - 1; i > -1; i--) {
190  // the ref index is the same as the listWidgetReferences index
191  // so we can erase using the row number of the element to be deleted
192  int rowNumber = widget->row(selectedList.at(i));
193  refs.erase(refs.begin() + rowNumber);
194  widget->model()->removeRow(rowNumber);
195  }
196 
197  updateFeature(pcDressUp, refs);
198 }

255  // Enter selection mode

462 TaskDlgDressUpParameters::TaskDlgDressUpParameters(ViewProviderDressUp *DressUpView)
463  : TaskDlgFeatureParameters(DressUpView)
464  , parameter(nullptr)
465 {
466  assert(DressUpView);
467  auto pcDressUp = DressUpView->getObject<PartDesign::DressUp>();
468  auto base = pcDressUp->Base.getValue();
469  std::vector<std::string> newSubList;
470  bool changed = false;
471  auto& shadowSubs = pcDressUp->Base.getShadowSubs();
472  for ( auto &shadowSub : shadowSubs ) {
473  auto displayName = shadowSub.oldName;
474  // If there is a missing tag on the shadow sub, take a guess at a new name.
475  if ( boost::starts_with(shadowSub.oldName,Data::MISSING_PREFIX)) {
476  Part::Feature::guessNewLink(displayName, base, shadowSub.newName.c_str());
477  newSubList.emplace_back(displayName);
478  changed = true;
479  }
480  }
481  if ( changed )
482  pcDressUp->Base.setValue(base, newSubList);
483 }
