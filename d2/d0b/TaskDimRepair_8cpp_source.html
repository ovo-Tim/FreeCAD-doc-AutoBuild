108 void TaskDimRepair::slotUseSelection()
109 {
110  const std::vector<App::DocumentObject*> dimObjects =
111  Gui::Selection().getObjectsOfType(TechDraw::DrawViewDimension::getClassTypeId());
112  if (dimObjects.empty()) {
113  //selection does not include a dimension, so we need to add our dimension to keep the
114  //validators happy
115  //bool accepted =
116  static_cast<void>(Gui::Selection().addSelection(m_dim->getDocument()->getName(),
117  m_dim->getNameInDocument()));
118  }
119  ReferenceVector references2d;
120  ReferenceVector references3d;
121  TechDraw::DrawViewPart* dvp = TechDraw::getReferencesFromSelection(references2d, references3d);
122  if (dvp != m_saveDvp) {
123  QMessageBox::warning(Gui::getMainWindow(),
124  QObject::tr("Incorrect Selection"),
125  QObject::tr("Can not use references from a different View"));
126  return;
127  }
128 
129  StringVector acceptableGeometry({ "Edge", "Vertex", "Face" });
130  std::vector<int> minimumCounts({1, 1, 1});
131  std::vector<DimensionGeometryType> acceptableDimensionGeometrys;//accept anything
132  DimensionGeometryType geometryRefs2d = validateDimSelection(
133  references2d, acceptableGeometry, minimumCounts, acceptableDimensionGeometrys);
134  if (geometryRefs2d == isInvalid) {
135  QMessageBox::warning(Gui::getMainWindow(),
136  QObject::tr("Incorrect Selection"),
137  QObject::tr("Can not make dimension from selection"));
138  return;
139  }
140  //what 3d geometry configuration did we receive?
141  DimensionGeometryType geometryRefs3d(isInvalid);
142  if (geometryRefs2d == TechDraw::isViewReference && !references3d.empty()) {
143  geometryRefs3d = validateDimSelection3d(
144  dvp, references3d, acceptableGeometry, minimumCounts, acceptableDimensionGeometrys);
145  if (geometryRefs3d == isInvalid) {
146  QMessageBox::warning(Gui::getMainWindow(),
147  QObject::tr("Incorrect Selection"),
148  QObject::tr("Can not make dimension from selection"));
149  return;
150  }
151  }
152 
153  m_toApply2d = references2d;
154  if (references3d.empty()) {
155  m_toApply3d.clear();
156  } else {
157  m_toApply3d = references3d;
158  }
159  updateUi();
160 }

162 void TaskDimRepair::updateUi()
163 {
164  std::string objName = m_dim->getViewPart()->getNameInDocument();
165  std::string objLabel = m_dim->getViewPart()->Label.getValue();
166  ui->leObject2d->setText(QString::fromStdString(objName + " / " + objLabel));

174 
175  loadTableWidget(ui->twReferences3d, m_toApply3d);
176 }

178 void TaskDimRepair::loadTableWidget(QTableWidget* tw, ReferenceVector refs)
179 {
180  tw->clearContents();
181  tw->setRowCount(refs.size() + 1);
182  size_t iRow = 0;
183  for (auto& ref : refs) {
184  QString qName = QString::fromStdString(ref.getObject()->getNameInDocument());
185  QTableWidgetItem* itemName = new QTableWidgetItem(qName);
186  itemName->setTextAlignment(Qt::AlignRight | Qt::AlignVCenter);
187  tw->setItem(iRow, 0, itemName);
188  QString qLabel = QString::fromStdString(std::string(ref.getObject()->Label.getValue()));
189  QTableWidgetItem* itemLabel = new QTableWidgetItem(qLabel);
190  itemLabel->setTextAlignment(Qt::AlignRight | Qt::AlignVCenter);
191  tw->setItem(iRow, 1, itemLabel);
192  QString qSubName = QString::fromStdString(ref.getSubName());
193  QTableWidgetItem* itemSubName = new QTableWidgetItem(qSubName);
194  itemSubName->setTextAlignment(Qt::AlignRight | Qt::AlignVCenter);
195  tw->setItem(iRow, 2, itemSubName);
196  iRow++;
197  }
198 }

200 void TaskDimRepair::fillList(QListWidget* lwItems, std::vector<std::string> labels,
201  std::vector<std::string> names)
202 {
203  QListWidgetItem* item;
204  QString qLabel;
205  QString qName;
206  QString qText;
207  int labelCount = labels.size();
208  int i = 0;
209  lwItems->clear();
210  for (; i < labelCount; i++) {
211  qLabel = QString::fromStdString(labels[i]);
212  qName = QString::fromStdString(names[i]);
213  qText = QString::fromUtf8("%1 %2").arg(qName, qLabel);
214  item = new QListWidgetItem(qText, lwItems);
215  item->setTextAlignment(Qt::AlignRight | Qt::AlignVCenter);
216  item->setData(Qt::UserRole, qName);
217  }
