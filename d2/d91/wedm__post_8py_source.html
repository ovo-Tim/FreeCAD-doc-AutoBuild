247 def export(objectslist, filename, argstring):
248  if not processArguments(argstring):
249  return None
250  global UNITS
251  global UNIT_FORMAT
252  global UNIT_SPEED_FORMAT
253  global MACHINE_NAME

255  for obj in objectslist:
256  if not hasattr(obj, "Path"):
257  print(
258  "the object " + obj.Name + " is not a path. Please select only path and Compounds."
259  )

263  gcode = ""

265  # write header

267  gcode += linenumber() + COMMENT_CHAR + "(Exported by FreeCAD)\n"
268  gcode += linenumber() + COMMENT_CHAR + "(Post Processor: " + __name__ + ")\n"
269  gcode += linenumber() + COMMENT_CHAR + "(Output Time:" + str(now) + ")\n"
270

272  if OUTPUT_COMMENTS:
273  gcode += linenumber() + COMMENT_CHAR + "(begin preamble)\n"
274  for line in PREAMBLE.splitlines(False):

277  gcode += linenumber() + UNITS + ENDLINE + "\n"
278 
279  for obj in objectslist:
280 
281  # Skip inactive operations
282  if hasattr(obj, "Active"):
283  if not obj.Active:
284  continue
285  if hasattr(obj, "Base") and hasattr(obj.Base, "Active"):
286  if not obj.Base.Active:
287  continue
288 
289  # fetch machine details
290  job = PathUtils.findParentJob(obj)
291 
292  # TODO: Not sure of the use
293  if hasattr(job, "MachineName"):
294  MACHINE_NAME = job.MachineName
295 
296  if hasattr(job, "MachineUnits"):
297  print(job.MachineUnits)
298  if job.MachineUnits == "Metric":
299  UNITS = "G21"
300  UNIT_FORMAT = "mm"
301  UNIT_SPEED_FORMAT = "mm/min"
302  else:
303  UNITS = "G20"
304  UNIT_FORMAT = "in"
305  UNIT_SPEED_FORMAT = "in/min"
306 
307  # ignore selected operations
308  if obj.Label in IGNORE_OPERATIONS:
309  continue
310  # do the pre_op
311  if OUTPUT_COMMENTS:
312  gcode += linenumber() + COMMENT_CHAR + "(begin operation: %s)\n" % obj.Label
313  gcode += (
314  linenumber()
315  + COMMENT_CHAR
316  + "(machine: %s, %s)\n" % (MACHINE_NAME, UNIT_SPEED_FORMAT)
317  )
318  for line in PRE_OPERATION.splitlines(True):
319  gcode += linenumber() + line
320 
321  # get coolant mode
322  coolantMode = "None"
323  if hasattr(obj, "CoolantMode") or hasattr(obj, "Base") and hasattr(obj.Base, "CoolantMode"):
324  if hasattr(obj, "CoolantMode"):
325  coolantMode = obj.CoolantMode
326  else:
327  coolantMode = obj.Base.CoolantMode
328 
329  # turn coolant on if required
330  if OUTPUT_COMMENTS:
331  if not coolantMode == "None":
332  gcode += linenumber() + COMMENT_CHAR + "(Coolant On:" + coolantMode + ")\n"
333  if coolantMode == "Flood":
334  gcode += linenumber() + "M8" + "\n"
335  if coolantMode == "Mist":
336  gcode += linenumber() + "M7" + "\n"
337 
338  # process the operation gcode
339  gcode += parse(obj)
340 
341  # do the post_op
342  if OUTPUT_COMMENTS:
343  gcode += linenumber() + COMMENT_CHAR + "(finish operation: %s)\n" % obj.Label
344  for line in POST_OPERATION.splitlines(True):
345  gcode += linenumber() + line
346 
347  # turn coolant off if required
348  if not coolantMode == "None":
349  if OUTPUT_COMMENTS:
350  gcode += linenumber() + COMMENT_CHAR + "(Coolant Off:" + coolantMode + ")\n"
351  gcode += linenumber() + "M9" + "\n"
352 
353  # do the post_amble
354  if OUTPUT_COMMENTS:
355  gcode += linenumber() + COMMENT_CHAR + "(begin postamble)\n"
356  for line in POSTAMBLE.splitlines(True):
357  gcode += linenumber() + line

359  if FreeCAD.GuiUp and SHOW_EDITOR:
360  final = gcode

362  print("Skipping editor since output is greater than 100kb")
363  else:
364  dia = PostUtils.GCodeEditorDialog()

369  else:
370  final = gcode
371 
372  print("done postprocessing.")
373 
374  if not filename == "-":

376  gfile.write(final)

390 def parse(pathobj):
391  global PRECISION
392  global MODAL
393  global OUTPUT_DOUBLES
394  global UNIT_FORMAT
395  global UNIT_SPEED_FORMAT
396  global RELATIVE_GCODE
397  global LOCATION
398 
399  out = ""
400  lastcommand = None
401  precision_string = "." + str(PRECISION) + "f"
402  currLocation = {} # keep track for incremental
403 
404  # the order of parameters
405  # linuxcnc doesn't want K properties on XY plane Arcs need work.
406  params = [
407  "X",
408  "Y",
409  "Z",
410  "A",
411  "B",
412  "C",
413  "I",
414  "J",
415  "F",
416  "S",
417  "T",
418  "Q",
419  "R",
420  "L",
421  "H",
422  "D",
423  "P",
424  ]
425  firstmove = Path.Command("G0", {"X": 0, "Y": 0, "Z": 0, "F": 0.0})
426  currLocation.update(firstmove.Parameters) # set First location Parameters
427 
428  if hasattr(pathobj, "Group"): # We have a compound or project.
429  # if OUTPUT_COMMENTS:
430  # out += linenumber() + COMMENT_CHAR + "(compound: " + pathobj.Label + ")\n"
431  for p in pathobj.Group:
432  out += parse(p)
433  return out
434  else: # parsing simple path
435 
436  # groups might contain non-path things like stock.
437  if not hasattr(pathobj, "Path"):
438  return out
439 
440  # if OUTPUT_COMMENTS:
441  # out += linenumber() + COMMENT_CHAR + "(" + pathobj.Label + ")\n"
442 
443  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
444  # For Debug Only
445  # if OUTPUT_COMMENTS:
446  # out += linenumber() + COMMENT_CHAR + "(" + str(c) + ")\n"
447 
448  outstring = []
449  command = c.Name
450  if not USE_RAPIDS and (command == "G0" or command == "G00"):
451  command = "G1"
452  outstring.append(command)
453 
454  # if modal: suppress the command if it is the same as the last one
455  if MODAL is True:
456  if command == lastcommand:
457  outstring.pop(0)
458 
459  if command[0] == "(": # command is a comment
460  if OUTPUT_COMMENTS: # Edit comment with COMMENT_CHAR
461  outstring.insert(0, COMMENT_CHAR)
462  else:
463  continue
464 
465  move_comands = ["G0", "G00", "G1", "G01"]
466  # Now add the remaining parameters in order
467  for param in params:
468  if param in c.Parameters:
469  if param == "F" and (
470  currLocation[param] != c.Parameters[param] or OUTPUT_DOUBLES
471  ):
472  if command not in [
473  "G0",
474  "G00",
475  ]: # linuxcnc doesn't use rapid speeds
476  speed = Units.Quantity(c.Parameters["F"], FreeCAD.Units.Velocity)
477  if speed.getValueAs(UNIT_SPEED_FORMAT) > 0.0:
478  outstring.append(
479  param
480  + format(
481  float(speed.getValueAs(UNIT_SPEED_FORMAT)),
482  precision_string,
483  )
484  )
485  else:
486  continue
487  elif param == "T":
488  outstring.append(param + str(int(c.Parameters["T"])))
489  elif param == "H":
490  outstring.append(param + str(int(c.Parameters["H"])))
491  elif param == "D":
492  outstring.append(param + str(int(c.Parameters["D"])))
493  elif param == "S":
494  outstring.append(param + str(int(c.Parameters["S"])))
495  # For wire EDM we ignore values for Z
496  elif param == "Z":
497  continue
498  else:
499  if (
500  (not OUTPUT_DOUBLES)
501  and (param in currLocation)
502  and (currLocation[param] == c.Parameters[param])
503  ):
504  continue
505  else:
506  if RELATIVE_GCODE and (param != "I" and param != "J"):
507  pos = Units.Quantity(
508  c.Parameters[param] - currLocation[param], FreeCAD.Units.Length
509  )
510  print(
511  f"currlocation: {currLocation[param]} param: {c.Parameters[param]} pos: {pos}"
512  )
513  if pos == 0:
514  # Remove no movement
515  continue
516  else:
517  pos = Units.Quantity(c.Parameters[param], FreeCAD.Units.Length)
518 
519  pos = pos * SCALE
520  sign = ""
521  if pos >= 0 and FORCE_SIGN:
522  sign = "+"
523 
524  stringout = sign + format(
525  float(pos.getValueAs(UNIT_FORMAT)), precision_string
526  )
527 
528  # Remove unneeded 0s on incremental moves this is needed since some numbers are zero only after applying precision
529  if (
530  RELATIVE_GCODE
531  and command in move_comands
532  and (float(stringout) == 0)
533  ):
534  continue
535  # Force trailing zeros
536  if FIXED_LENGTH > 0:
537  extra = 1 if stringout[0] in ["+", "-"] else 0
538  stringout = stringout.zfill(FIXED_LENGTH + extra)
539  outstring.append(param + stringout)
540 
541  currLocation.update(c.Parameters)
542 
543  # Check for Tool Change:
544  if command == "M6":
545  # stop the spindle
546  out += linenumber() + "M5\n"
547  for line in TOOL_CHANGE.splitlines(True):
548  out += linenumber() + line
549 
550  # add height offset
551  if USE_TLO:
552  tool_height = "\nG43 H" + str(int(c.Parameters["T"]))
553  outstring.append(tool_height)
554 
555  if command == "message":
556  if OUTPUT_COMMENTS is False:
557  out = []
558  else:
559  outstring.pop(0) # remove the command
560 
561  # prepend a line number and append a newline
562  if len(outstring) >= 1:
563  if len(outstring) == 1 and (outstring[0] == "G0" or outstring[0] == "G1"):
564  # Don't write empty moves (Generated when Z moves are ignored for EDM)
565  print("Ignoring: " + command + ":" + outstring[0])
566  continue
567  if TWO_DIGIT_CODES and len(command) == 2 and outstring[0] == command:
568  outstring[0] = command[0] + "0" + command[1]
569  if OUTPUT_LINE_NUMBERS:
570  outstring.insert(0, (linenumber()))
571 
572  # append the line to the final output
573  start = True
574  for w in outstring:
575  if start:
576  start = False
577  else:
578  out += COMMAND_SPACE
579  out += w
580  # Note: Do *not* strip `out`, since that forces the allocation
581  # of a contiguous string & thus quadratic complexity.
582  out += ENDLINE + "\n"
583 
584  # store the latest command after written
585  lastcommand = command
586 
587  return out
