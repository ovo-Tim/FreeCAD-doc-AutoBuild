45 def write_meshdata_constraint(f, femobj, heatflux_obj, ccxwriter):
46 
47  # floats read from ccx should use {:.13G}, see comment in writer module
48 
49  if heatflux_obj.ConstraintType == "Convection":
50  heatflux_key_word = "FILM"
51  heatflux_facetype = "F"
52  heatflux_values = "{:.13G},{:.13G}".format(
53  heatflux_obj.AmbientTemp.getValueAs("K").Value,
54  heatflux_obj.FilmCoef.getValueAs("t/s^3/K").Value,
55  )
56 
57  elif heatflux_obj.ConstraintType == "Radiation":
58  heatflux_key_word = "RADIATE"
59  heatflux_facetype = "R"
60  heatflux_values = "{:.13G},{:.13G}".format(
61  heatflux_obj.AmbientTemp.getValueAs("K").Value, heatflux_obj.Emissivity
62  )
63 
64  elif heatflux_obj.ConstraintType == "DFlux":
65  heatflux_key_word = "DFLUX"
66  heatflux_facetype = "S"
67  heatflux_values = "{:.13G}".format(heatflux_obj.DFlux.getValueAs("t/s^3").Value)
68 
69  f.write(f"*{heatflux_key_word}\n")
70  for ref_shape in femobj["HeatFluxFaceTable"]:
71  elem_string = ref_shape[0]
72  face_table = ref_shape[1]
73  f.write(f"** Heat flux on face {elem_string}\n")
74  for i in face_table:
75  # OvG: Only write out the VolumeIDs linked to a particular face
76  f.write(f"{i[0]},{heatflux_facetype}{i[1]},{heatflux_values}\n")
