202  found_solver_for_use = False
203  for m in self.analysis.Group:

208  # analysis has one solver and no solver was set --> use the one solver

398  error_title = "No or wrong CalculiX binary ccx"

492  def start_ccx(self):
493  import multiprocessing
494 
495  self.ccx_stdout = ""
496  self.ccx_stderr = ""
497  ont_backup = os.environ.get("OMP_NUM_THREADS")
498  self.ccx_prefs = FreeCAD.ParamGet("User parameter:BaseApp/Preferences/Mod/Fem/Ccx")
499  # If number of CPU's specified
500  num_cpu_pref = self.ccx_prefs.GetInt("AnalysisNumCPUs", 1)
501  if not ont_backup:
502  ont_backup = str(num_cpu_pref)
503  if num_cpu_pref > 1:
504  # If user picked a number use that instead
505  _env = os.putenv("OMP_NUM_THREADS", str(num_cpu_pref))
506  else:
507  _env = os.putenv("OMP_NUM_THREADS", str(multiprocessing.cpu_count()))
508  # change cwd because ccx may crash if directory has no write permission
509  # there is also a limit of the length of file names so jump to the document directory
510  cwd = QtCore.QDir.currentPath()
511  f = QtCore.QFileInfo(self.inp_file_name)
512  QtCore.QDir.setCurrent(f.path())
513  p = subprocess.Popen(
514  [self.ccx_binary, "-i ", f.baseName()],
515  stdout=subprocess.PIPE,
516  stderr=subprocess.PIPE,
517  shell=False,
518  env=_env,
519  )
520  self.ccx_stdout, self.ccx_stderr = p.communicate()
521  self.ccx_stdout = self.ccx_stdout.decode()
522  self.ccx_stderr = self.ccx_stderr.decode()
523  os.putenv("OMP_NUM_THREADS", ont_backup)
524  QtCore.QDir.setCurrent(cwd)
525  return p.returncode
