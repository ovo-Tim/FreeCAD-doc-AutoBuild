237 static bool getShapePlane(const TopoDS_Shape& shape, gp_Pln& pln)
238 {

240  return false;

243  BRepAdaptor_Surface adapt(TopoDS::Face(shape));
244  if (adapt.GetType() != GeomAbs_Plane) {
245  return false;

256  // transformation SOMETIME, so we have to transformed the found
257  // plane manually. Need to figure out WHY!

269 bool Area::isCoplanar(const TopoDS_Shape& s1, const TopoDS_Shape& s2)
270 {
271  if (s1.IsNull() || s2.IsNull()) {
272  return false;
273  }
274  if (s1.IsSame(s2)) {
275  return true;
276  }
277  gp_Pln pln1, pln2;
278  if (!getShapePlane(s1, pln1) || !getShapePlane(s2, pln2)) {
279  return false;
280  }
281  return pln1.Position().IsCoplanar(pln2.Position(),
282  Precision::Confusion(),
283  Precision::Confusion());

286 int Area::addShape(CArea& area,
287  const TopoDS_Shape& shape,
288  const gp_Trsf* trsf,
289  double deflection,
290  const TopoDS_Shape* plane,
291  bool force_coplanar,
292  CArea* areaOpen,
293  bool to_edges,
294  bool reorient)
295 {
296  bool haveShape = false;
297  int skipped = 0;
298  for (TopExp_Explorer it(shape, TopAbs_FACE); it.More(); it.Next()) {
299  haveShape = true;
300  const TopoDS_Face& face = TopoDS::Face(it.Current());
301  if (plane && !isCoplanar(face, *plane)) {
302  ++skipped;
303  if (force_coplanar) {
304  continue;
305  }
306  }
307  for (TopExp_Explorer it(face, TopAbs_WIRE); it.More(); it.Next()) {
308  addWire(area, TopoDS::Wire(it.Current()), trsf, deflection);
309  }
310  }
311 
312  if (haveShape) {
313  return skipped;
314  }
315 
316  CArea _area;
317  CArea _areaOpen;
318 
319  for (TopExp_Explorer it(shape, TopAbs_WIRE); it.More(); it.Next()) {
320  haveShape = true;
321  const TopoDS_Wire& wire = TopoDS::Wire(it.Current());
322  if (plane && !isCoplanar(wire, *plane)) {
323  ++skipped;
324  if (force_coplanar) {
325  continue;
326  }
327  }
328  if (BRep_Tool::IsClosed(wire)) {
329  addWire(_area, wire, trsf, deflection);
330  }
331  else if (to_edges) {
332  for (TopExp_Explorer it(wire, TopAbs_EDGE); it.More(); it.Next()) {
333  addWire(_areaOpen,
334  BRepBuilderAPI_MakeWire(TopoDS::Edge(it.Current())).Wire(),
335  trsf,
336  deflection,
337  true);
338  }
339  }
340  else {
341  addWire(_areaOpen, wire, trsf, deflection);
342  }
343  }
344 
345  if (!haveShape) {
346  for (TopExp_Explorer it(shape, TopAbs_EDGE); it.More(); it.Next()) {
347  if (plane && !isCoplanar(it.Current(), *plane)) {
348  ++skipped;
349  if (force_coplanar) {
350  continue;
351  }
352  }
353  TopoDS_Wire wire = BRepBuilderAPI_MakeWire(TopoDS::Edge(it.Current())).Wire();
354  addWire(BRep_Tool::IsClosed(wire) ? _area : _areaOpen, wire, trsf, deflection);
355  }
356  }
357 
358  if (reorient) {
359  _area.Reorder();
360  }
361  area.m_curves.splice(area.m_curves.end(), _area.m_curves);
362  if (areaOpen) {
363  areaOpen->m_curves.splice(areaOpen->m_curves.end(), _areaOpen.m_curves);
364  }
365  else {
366  area.m_curves.splice(area.m_curves.end(), _areaOpen.m_curves);
367  }
368  return skipped;
369 }

371 static std::vector<gp_Pnt> discretize(const TopoDS_Edge& edge, double deflection)
372 {
373  std::vector<gp_Pnt> ret;
374  BRepAdaptor_Curve curve(edge);
375  Standard_Real efirst, elast;
376  efirst = curve.FirstParameter();
377  elast = curve.LastParameter();
378  bool reversed = (edge.Orientation() == TopAbs_REVERSED);
379 
380  // push the first point
381  ret.push_back(curve.Value(reversed ? elast : efirst));
382 
383  // NOTE: QuasiUniformDeflection has trouble with some B-Spline, see
384  // https://forum.freecad.org/viewtopic.php?f=15&t=42628
385  //
386  // GCPnts_QuasiUniformDeflection discretizer(curve, deflection, first, last);
387  //
388  GCPnts_UniformDeflection discretizer(curve, deflection, efirst, elast);
389  if (!discretizer.IsDone()) {
390  Standard_Failure::Raise("Curve discretization failed");
391  }
392  if (discretizer.NbPoints() > 1) {
393  int nbPoints = discretizer.NbPoints();
394  // strangely OCC discretizer points are one-based, not zero-based, why?
395  if (reversed) {
396  for (int i = nbPoints - 1; i >= 1; --i) {
397  ret.push_back(discretizer.Value(i));
398  }
399  }
400  else {
401  for (int i = 2; i <= nbPoints; i++) {
402  ret.push_back(discretizer.Value(i));
403  }
404  }
405  }
406  // push the last point
407  ret.push_back(curve.Value(reversed ? efirst : elast));
408  return ret;
409 }

411 void Area::addWire(CArea& area,
412  const TopoDS_Wire& wire,
413  const gp_Trsf* trsf,
414  double deflection,
415  bool to_edges)
416 {
417  CCurve ccurve;
418  BRepTools_WireExplorer xp(trsf ? TopoDS::Wire(wire.Moved(TopLoc_Location(*trsf))) : wire);
419 
420  if (!xp.More()) {
421  AREA_TRACE("empty wire");
422  return;
423  }
424 
425  gp_Pnt p = BRep_Tool::Pnt(xp.CurrentVertex());
426  ccurve.append(CVertex(Point(p.X(), p.Y())));
427 
428  for (; xp.More(); xp.Next()) {
429  const TopoDS_Edge& edge = TopoDS::Edge(xp.Current());
430  BRepAdaptor_Curve curve(edge);
431  bool reversed = (xp.Current().Orientation() == TopAbs_REVERSED);
432 
433  p = curve.Value(reversed ? curve.FirstParameter() : curve.LastParameter());
434 
435  switch (curve.GetType()) {
436  case GeomAbs_Line: {
437  ccurve.append(CVertex(Point(p.X(), p.Y())));
438  if (to_edges) {
439  area.append(ccurve);
440  ccurve.m_vertices.pop_front();
441  }
442  break;
443  }
444  case GeomAbs_Circle: {
445  double first = curve.FirstParameter();
446  double last = curve.LastParameter();
447  gp_Circ circle = curve.Circle();
448  gp_Dir dir = circle.Axis().Direction();
449  gp_Pnt center = circle.Location();
450  int type = dir.Z() < 0 ? -1 : 1;
451  if (reversed) {
452  type = -type;
453  }
454  if (fabs(first - last) > M_PI) {
455  // Split arc(circle) larger than half circle. Because gcode
456  // can't handle full circle?
457  gp_Pnt mid = curve.Value((last - first) * 0.5 + first);
458  ccurve.append(
459  CVertex(type, Point(mid.X(), mid.Y()), Point(center.X(), center.Y())));
460  }
461  ccurve.append(CVertex(type, Point(p.X(), p.Y()), Point(center.X(), center.Y())));
462  if (to_edges) {
463  ccurve.UnFitArcs();
464  CCurve c;
465  c.append(ccurve.m_vertices.front());
466  auto it = ccurve.m_vertices.begin();
467  for (++it; it != ccurve.m_vertices.end(); ++it) {
468  c.append(*it);
469  area.append(c);
470  c.m_vertices.pop_front();
471  }
472  ccurve.m_vertices.clear();
473  ccurve.append(c.m_vertices.front());
474  }
475  break;
476  }
477  default: {
478  // Discretize all other type of curves
479  const auto& pts = discretize(edge, deflection);
480  for (size_t i = 1; i < pts.size(); ++i) {
481  auto& pt = pts[i];
482  ccurve.append(CVertex(Point(pt.X(), pt.Y())));
483  if (to_edges) {
484  area.append(ccurve);
485  ccurve.m_vertices.pop_front();
486  }
487  }
488  }
489  }
490  }
491  if (!to_edges) {
492  if (BRep_Tool::IsClosed(wire) && !ccurve.IsClosed()) {
493  AREA_WARN("ccurve not closed");
494  ccurve.append(ccurve.m_vertices.front());
495  }
496  area.move(std::move(ccurve));
497  }
498 }

572  void point(const Base::Vector3d& p)
573  {
574  if (p.z <= maxZ) {
575  if (bbox.MinX <= p.x && p.x <= bbox.MaxX && bbox.MinY <= p.y && p.y <= bbox.MaxY) {
576  CCurve curve;
577  curve.append(CVertex {{p.x + radius, p.y}});
578  curve.append(CVertex {1, {p.x - radius, p.y}, {p.x, p.y}});
579  curve.append(CVertex {1, {p.x + radius, p.y}, {p.x, p.y}});
580  holes.append(curve);
581  }
582  }
583  }

585  void line(const Base::Vector3d& last, const Base::Vector3d& next)
586  {
587  if (last.z <= maxZ && next.z <= maxZ) {
588  Base::BoundBox2d segBox = {};
589  segBox.Add({last.x, last.y});
590  segBox.Add({next.x, next.y});
591  if (bbox.Intersect(segBox)) {
592  CCurve curve;
593  curve.append(CVertex {{last.x, last.y}});
594  curve.append(CVertex {{next.x, next.y}});
595  pathSegments.append(curve);
596  }
597  }
598  }

1312  }

1314  vertexStack.erase(vertexStack.begin() + r.iStart, vertexStack.end());
1315  stack.pop_back();
1316  if (stack.empty()) {
1317  break;
1318  }
1319  ++stack.back().iCurrent;
1320  }
1321  if (stack.empty()) {
1322  // If stack is empty, it means this wire is open. Try a new
1323  // starting edge
1324  ++skips;
1325  break;
1326  }
1327  if (pstart.SquareDistance(pend) > tol) {
1328  // if the wire is not closed yet, continue search for the
1329  // next connected edge
1330  continue;
1331  }
1332 
1333  Handle(ShapeExtend_WireData) wireData = new ShapeExtend_WireData();
1334  wireData->Add(e);
1335  for (auto& r : stack) {
1336  const auto& v = vertexStack[r.iCurrent];
1337  auto& info = *v.it;
1338  if (v.start) {
1339  wireData->Add(info.edge);
1340  }
1341  else {
1342  wireData->Add(TopoDS::Edge(info.edge.Reversed()));
1343  }
1344  }
1345  // TechDraw even uses 0.1 as tolerance. Really? Why?
1346  TopoDS_Wire wire = makeCleanWire(wireData, 0.01);
1347  if (!BRep_Tool::IsClosed(wire)) {
1348  FC_WARN("failed to close some projection wire");
1349  Area::showShape(wire, "failed");
1350  ++skips;
1351  }
1352  else {
1353  for (auto& r : stack) {
1354  const auto& v = vertexStack[r.iCurrent];
1355  auto& info = *v.it;
1356  if (!info.used) {
1357  info.used = true;
1358  edgesToVisit.erase(&info);
1359  }
1360  }
1361  Area::showShape(wire, "joined");
1362  builder.Add(comp, wire);
1363  ++count;
1364  }
1365  break;
1366  }
1367  }
1368  FC_TIME_LOG(t, "found " << count << " closed wires, skipped " << skips << "edges. ");
1369  return skips;
1370  }

1486 template<class Func>
1487 static int foreachSubshape(const TopoDS_Shape& shape,
1488  Func func,
1489  int type = TopAbs_FACE,
1490  bool groupOpenEdges = false)
1491 {
1492  int res = -1;
1493  std::vector<TopoDS_Shape> openShapes;
1494  switch (type) {
1495  case TopAbs_SOLID:
1496  for (TopExp_Explorer it(shape, TopAbs_SOLID); it.More(); it.Next()) {
1497  res = TopAbs_SOLID;
1498  func(it.Current(), TopAbs_SOLID);
1499  }
1500  if (res >= 0) {
1501  break;
1502  }
1503  // fall through
1504  case TopAbs_FACE:
1505  for (TopExp_Explorer it(shape, TopAbs_FACE); it.More(); it.Next()) {
1506  res = TopAbs_FACE;
1507  func(it.Current(), TopAbs_FACE);
1508  }
1509  if (res >= 0) {
1510  break;
1511  }
1512  // fall through
1513  case TopAbs_WIRE:
1514  for (TopExp_Explorer it(shape, TopAbs_WIRE); it.More(); it.Next()) {
1515  res = TopAbs_WIRE;
1516  if (groupOpenEdges && !BRep_Tool::IsClosed(TopoDS::Wire(it.Current()))) {
1517  openShapes.push_back(it.Current());
1518  }
1519  else {
1520  func(it.Current(), TopAbs_WIRE);
1521  }
1522  }
1523  if (res >= 0) {
1524  break;
1525  }
1526  // fall through
1527  default:
1528  for (TopExp_Explorer it(shape, TopAbs_EDGE); it.More(); it.Next()) {
1529  res = TopAbs_EDGE;
1530  if (groupOpenEdges) {
1531  TopoDS_Edge e = TopoDS::Edge(it.Current());
1532  gp_Pnt p1, p2;
1533  getEndPoints(e, p1, p2);
1534  if (p1.SquareDistance(p2) > Precision::SquareConfusion()) {
1535  openShapes.push_back(it.Current());
1536  continue;
1537  }
1538  }
1539  func(it.Current(), TopAbs_EDGE);
1540  }
1541  }
1542  if (openShapes.empty()) {
1543  return res;
1544  }
1545 
1546  BRep_Builder builder;
1547  TopoDS_Compound comp;
1548  builder.MakeCompound(comp);
1549  for (auto& s : openShapes) {
1550  builder.Add(comp, s);
1551  }
1552  func(comp, TopAbs_COMPOUND);
1553  return TopAbs_COMPOUND;
1554 }

1612  return;
1613  }

1746  double tolerance = 0.0;
1747  if (_heights.empty()) {
1748  double z;
1749  double d = fabs(myParams.Stepdown);
1750  if (myParams.SectionCount > 1 && d < Precision::Confusion()) {
1751  throw Base::ValueError("invalid stepdown");
1752  }
1753 
1754  if (mode == SectionModeBoundBox) {
1755  if (myParams.Stepdown > 0.0) {
1756  z = zMax - myParams.SectionOffset;

1759  z = zMin + myParams.SectionOffset;
1760  }

1764  // that the work plane is aligned with xy0 plane, the starting Z
1765  // value shall be 0 minus the given section offset. Note the
1766  // section offset is relative to the starting Z
1767  if (myParams.Stepdown > 0.0) {

1769  }

1771  z = myParams.SectionOffset;
1772  }

1780  }
1781  else if (z < zMin) {
1782  z = zMin;
1783  }
1784  double dz;
1785  if (myParams.Stepdown > 0.0) {
1786  dz = z - zMin;
1787  tolerance = myParams.SectionTolerance;
1788  }
1789  else {
1790  dz = zMax - z;
1791  tolerance = -myParams.SectionTolerance;
1792  }

1794  if (count < 0 || count * d > dz) {
1795  count = floor(dz / d) + 1;

1798  for (int i = 0; i < count; ++i, z -= myParams.Stepdown) {
1799  double height = z - tolerance;

1813  AREA_WARN("hit top " << z << ',' << zMax << ',' << height);

1816  if (myParams.Stepdown < 0.0) {
1817  break;
1818  }
1819  }
1820  else {
1821  heights.push_back(height);
1822  }
1823  }

1829  for (double z : _heights) {

1838  break;

1847  continue;

1856  }
1857  else if (zMax - z < myParams.SectionTolerance) {

1859  continue;

1866  }
1867  }

1871  }

1879  if (projectedShapes.empty()) {
1880  AREA_ERR("empty projection");

1883  }
