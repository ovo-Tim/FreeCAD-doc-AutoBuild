55 App::DocumentObjectExecReturn* FeatureShape::execute()
56 {
57  Toolpath path;
58  std::vector<App::DocumentObject*> links = Sources.getValues();
59  if (links.empty()) {
60  Path.setValue(path);
61  return new App::DocumentObjectExecReturn("No shapes linked");
62  }
63 
64  const Base::Vector3d& v = StartPoint.getValue();
65  gp_Pnt pstart(v.x, v.y, v.z);
66 
67  std::list<TopoDS_Shape> shapes;
68  for (std::vector<App::DocumentObject*>::iterator it = links.begin(); it != links.end(); ++it) {
69  if (!(*it && (*it)->isDerivedFrom(Part::Feature::getClassTypeId()))) {
70  continue;
71  }
72  const TopoDS_Shape& shape = static_cast<Part::Feature*>(*it)->Shape.getShape().getShape();
73  if (shape.IsNull()) {
74  continue;
75  }
76  shapes.push_back(shape);
77  }
78 
79  Area::toPath(path,
80  shapes,
81  UseStartPoint.getValue() ? &pstart : nullptr,
82  nullptr,
83  PARAM_PROP_ARGS(AREA_PARAMS_PATH));
84 
85  Path.setValue(path);
86  return App::DocumentObject::StdReturn;
87 }
