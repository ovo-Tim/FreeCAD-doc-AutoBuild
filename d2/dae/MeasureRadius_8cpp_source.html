68 bool MeasureRadius::isValidSelection(const App::MeasureSelection& selection)
69 {
70 
71  if (selection.empty() || selection.size() > 1) {
72  // too few or too many selections
73  return false;
74  }
75 
76  auto element = selection.front();
77  auto type = App::MeasureManager::getMeasureElementType(element);
78 
79  if (type == App::MeasureElementType::INVALID) {
80  return false;
81  }
82 
83  if (type != App::MeasureElementType::CIRCLE && type != App::MeasureElementType::ARC) {
84  return false;
85  }
86 
87  return true;
