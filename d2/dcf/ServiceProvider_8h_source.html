104  template<typename T>
105  T* provide() const
106  {
107  if (auto it = _implementations.find(typeid(T).name()); it != _implementations.end()) {
108  auto descriptors = it->second;
109 
110  if (descriptors.empty()) {
111  return nullptr;
112  }
113 
114  return descriptors.front().get<T>();
115  }
116 
117  return nullptr;
118  }

125  template<typename T>
126  std::list<T*> all() const
127  {
128  if (auto it = _implementations.find(typeid(T).name()); it != _implementations.end()) {
129  auto source = it->second;
130 
131  std::list<T*> result(source.size());
132 
133  std::transform(source.begin(),
134  source.end(),
135  result.begin(),
136  [](const ServiceDescriptor& descriptor) {
137  return descriptor.get<T>();
138  });
139 
140  return result;
141  }
142 
143  return {};
144  }
