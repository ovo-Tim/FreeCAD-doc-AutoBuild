117 QWidget* ArrayDelegate::createWidget(QWidget* parent, const QVariant& item) const
118 {
119  QWidget* widget = nullptr;
120 
121  if (_type == Materials::MaterialValue::String || _type == Materials::MaterialValue::URL
122  || _type == Materials::MaterialValue::List) {
123  widget = new Gui::PrefLineEdit(parent);
124  }
125  else if (_type == Materials::MaterialValue::Integer) {
126  Gui::UIntSpinBox* spinner = new Gui::UIntSpinBox(parent);
127  spinner->setMinimum(0);
128  spinner->setMaximum(UINT_MAX);
129  spinner->setValue(item.toUInt());
130  widget = spinner;
131  }
132  else if (_type == Materials::MaterialValue::Float) {
133  auto spinner = new Gui::DoubleSpinBox(parent);
134 
135  // the magnetic permeability is the parameter for which many decimals matter
136  // the most however, even for this, 6 digits are sufficient
137  spinner->setDecimals(6);
138 
139  // for almost all Float parameters of materials a step of 1 would be too large
140  spinner->setSingleStep(0.1);
141 
142  spinner->setMinimum(std::numeric_limits<double>::min());
143  spinner->setMaximum(std::numeric_limits<double>::max());
144  spinner->setValue(item.toDouble());
145  widget = spinner;
146  }
147  else if (_type == Materials::MaterialValue::Boolean) {
148  auto combo = new Gui::PrefComboBox(parent);
149  combo->insertItem(0, QString::fromStdString(""));
150  combo->insertItem(1, tr("False"));
151  combo->insertItem(2, tr("True"));
152  combo->setCurrentText(item.toString());
153  widget = combo;
154  }
155  else if (_type == Materials::MaterialValue::Quantity) {
156  auto input = new Gui::QuantitySpinBox();
157  input->setMinimum(std::numeric_limits<double>::min());
158  input->setMaximum(std::numeric_limits<double>::max());
159  input->setUnitText(_units);
160  input->setValue(item.value<Base::Quantity>());
161 
162  widget = input;
163  }
164  else if (_type == Materials::MaterialValue::FileList) {
165  auto chooser = new Gui::FileChooser();
166  auto propertyValue = item.toString();
167  if (!propertyValue.isEmpty()) {
168  chooser->setFileName(propertyValue);
169  }
170 
171  widget = chooser;
172  }
173  else {
174  // Default editor
175  widget = new QLineEdit(parent);
176  }
177 
178  widget->setParent(parent);
179 
180  return widget;
181 }
