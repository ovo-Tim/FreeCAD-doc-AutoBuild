34 def booleanFragments(list_of_shapes, mode, tolerance = 0.0):
35  """booleanFragments(list_of_shapes, mode, tolerance = 0.0): functional part of
36  BooleanFragments feature. It's just result of generalFuse plus a bit of
37  post-processing.
38 
39  mode is a string. It can be "Standard", "Split" or "CompSolid".
40  "Standard" - return generalFuse as is.
41  "Split" - wires and shells will be split at intersections.
42  "CompSolid" - solids will be extracted from result of generelFuse, and compsolids will
43  be made from them; all other stuff is discarded."""
44 
45  pieces, map = list_of_shapes[0].generalFuse(list_of_shapes[1:], tolerance)
46  if mode == "Standard":
47  return pieces
48  elif mode == "CompSolid":
49  solids = pieces.Solids
50  if len(solids) < 1:
51  raise ValueError("No solids in the result. Can't make CompSolid.")
52  elif len(solids) == 1:
53  FreeCAD.Console.PrintWarning("Part_BooleanFragments: only one solid in the result, generating trivial compsolid.")
54  return ShapeMerge.mergeSolids(solids, bool_compsolid= True)
55  elif mode == "Split":
56  gr = GeneralFuseResult(list_of_shapes, (pieces,map))
57  gr.splitAggregates()
58  return Part.Compound(gr.pieces)
59  else:
60  raise ValueError("Unknown mode: {mode}".format(mode= mode))

62 def slice(base_shape, tool_shapes, mode, tolerance = 0.0):
63  """slice(base_shape, tool_shapes, mode, tolerance = 0.0): functional part of
64  Slice feature. Splits base_shape into pieces based on intersections with tool_shapes.
65 
66  mode is a string. It can be "Standard", "Split" or "CompSolid".
67  "Standard" - return like generalFuse: edges, faces and solids are split, but wires,
68  shells, compsolids get extra segments but remain in one piece.
69  "Split" - wires and shells will be split at intersections, too.
70  "CompSolid" - slice a solid and glue it back together to make a compsolid"""
71 
72  shapes = [base_shape] + [Part.Compound([tool_shape]) for tool_shape in tool_shapes] # hack: putting tools into compounds will prevent contamination of result with pieces of tools
73  if len(shapes) < 2:
74  raise ValueError("No slicing objects supplied!")
75  pieces, map = shapes[0].generalFuse(shapes[1:], tolerance)
76  gr = GeneralFuseResult(shapes, (pieces,map))
77  if mode == "Standard":
78  result = gr.piecesFromSource(shapes[0])
79  elif mode == "CompSolid":
80  solids = Part.Compound(gr.piecesFromSource(shapes[0])).Solids
81  if len(solids) < 1:
82  raise ValueError("No solids in the result. Can't make compsolid.")
83  elif len(solids) == 1:
84  FreeCAD.Console.PrintWarning("Part_Slice: only one solid in the result, generating trivial compsolid.")
85  result = ShapeMerge.mergeSolids(solids, bool_compsolid= True).childShapes()
86  elif mode == "Split":
87  gr.splitAggregates(gr.piecesFromSource(shapes[0]))
88  result = gr.piecesFromSource(shapes[0])
89  return result[0] if len(result) == 1 else Part.Compound(result)
