133 template<typename float_type>
134 bool Line3<float_type>::Contains(const Vector3<float_type>& pt, float_type eps) const
135 {
136  Vector3<float_type> v1 = p1 - pt;
137  Vector3<float_type> v2 = p2 - pt;
138  Vector3<float_type> v3 = p2 - p1;
139  float_type dot = v1.Dot(v2);
140  if (dot > eps) {
141  return false;
142  }
143  return v3.Cross(v1).Length() < eps;
144 }
