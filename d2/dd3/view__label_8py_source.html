360  def update_label(self, obj, vobj):
361  """Update the label including text size and multiplier."""
362  size = vobj.FontSize.Value * vobj.ScaleMultiplier
363  self.font.size = size
364 
365  if vobj.DisplayMode == "Screen":
366  self.textpos.translation.setValue(obj.Placement.Base)
367  return
368 
369  line_height = size * max(1, vobj.LineSpacing)
370  if vobj.Frame == "None":
371  margin = size * 0.1
372  first_line_height = size
373  # We need to calculate total_height without using get_text_size:
374  # If StraightDirection = "Horizontal" and TextAlignment = "Bottom"
375  # we want the horizontal line segment to be aligned with the
376  # baseline of the bottom text even if there are descenders.
377  total_height = first_line_height + (line_height * (len(obj.Text) - 1))
378  else:
379  margin = line_height * 0.25
380  first_line_height = size + margin
381  box = self.get_text_size(vobj)
382  total_height = box[1] + (2 * margin)
383 
384  # Space between endpoint of line and text:
385  v = App.Vector(margin, 0, 0)
386  if obj.StraightDistance > 0:
387  v = v.negative()
388 
389  if vobj.TextAlignment == "Top":
390  v = v + App.Vector(0, -first_line_height, 0)
391  elif vobj.TextAlignment == "Middle":
392  v = v + App.Vector(0, -first_line_height + (total_height / 2), 0)
393  elif vobj.TextAlignment == "Bottom":
394  v = v + App.Vector(0, -first_line_height + total_height, 0)
395 
396  v = obj.Placement.Rotation.multVec(v)
397  pos = v + obj.Placement.Base
398  self.textpos.translation.setValue(pos)
399  self.textpos.rotation.setValue(obj.Placement.Rotation.Q)
