110 PyObject* HLRBRep_AlgoPy::setProjector(PyObject *args, PyObject *kwds)
111 {
112  PyObject* ps = nullptr;
113  PyObject* zd = nullptr;
114  PyObject* xd = nullptr;
115  double focus = std::numeric_limits<double>::quiet_NaN();
116 
117  static const std::array<const char *, 5> kwlist {"Origin", "ZDir", "XDir", "focus", nullptr};
118  if (Base::Wrapped_ParseTupleAndKeywords(args, kwds, "|O!O!O!d", kwlist,
119  &Base::VectorPy::Type, &ps,
120  &Base::VectorPy::Type, &zd,
121  &Base::VectorPy::Type, &xd,
122  &focus)) {
123  gp_Ax2 ax2;
124  if (ps && zd && xd) {
125  Base::Vector3d p = Py::Vector(ps,false).toVector();
126  Base::Vector3d z = Py::Vector(zd,false).toVector();
127  Base::Vector3d x = Py::Vector(xd,false).toVector();
128  ax2.SetLocation(Base::convertTo<gp_Pnt>(p));
129  ax2.SetDirection(Base::convertTo<gp_Dir>(z));
130  ax2.SetXDirection(Base::convertTo<gp_Dir>(x));
131  }
132  else if (ps && zd) {
133  Base::Vector3d p = Py::Vector(ps,false).toVector();
134  Base::Vector3d z = Py::Vector(zd,false).toVector();
135  ax2.SetLocation(Base::convertTo<gp_Pnt>(p));
136  ax2.SetDirection(Base::convertTo<gp_Dir>(z));
137  }
138 
139  if (boost::math::isnan(focus))
140  getHLRBRep_AlgoPtr()->Projector(HLRAlgo_Projector(ax2));
141  else
142  getHLRBRep_AlgoPtr()->Projector(HLRAlgo_Projector(ax2, focus));
143  Py_Return;
144  }
145 
146  return nullptr;
147 }
