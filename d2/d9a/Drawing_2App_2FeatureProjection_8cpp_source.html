64 App::DocumentObjectExecReturn* FeatureProjection::execute(void)
65 {
66  App::DocumentObject* link = Source.getValue();
67  if (!link) {
68  return new App::DocumentObjectExecReturn("No object linked");
69  }
70  if (!link->isDerivedFrom<Part::Feature>()) {
71  return new App::DocumentObjectExecReturn("Linked object is not a Part object");
72  }
73  const TopoDS_Shape& shape = static_cast<Part::Feature*>(link)->Shape.getShape().getShape();
74  if (shape.IsNull()) {
75  return new App::DocumentObjectExecReturn("Linked shape object is empty");
76  }
77 
78  try {
79  const Base::Vector3d& dir = Direction.getValue();
80  Drawing::ProjectionAlgos alg(shape, dir);
81 
82  TopoDS_Compound comp;
83  BRep_Builder builder;
84  builder.MakeCompound(comp);
85  if (!alg.V.IsNull() && VCompound.getValue()) {
86  builder.Add(comp, alg.V);
87  }
88  if (!alg.V1.IsNull() && Rg1LineVCompound.getValue()) {
89  builder.Add(comp, alg.V1);
90  }
91  if (!alg.VN.IsNull() && RgNLineVCompound.getValue()) {
92  builder.Add(comp, alg.VN);
93  }
94  if (!alg.VO.IsNull() && OutLineVCompound.getValue()) {
95  builder.Add(comp, alg.VO);
96  }
97  if (!alg.VI.IsNull() && IsoLineVCompound.getValue()) {
98  builder.Add(comp, alg.VI);
99  }
100  if (!alg.H.IsNull() && HCompound.getValue()) {
101  builder.Add(comp, alg.H);
102  }
103  if (!alg.H1.IsNull() && Rg1LineHCompound.getValue()) {
104  builder.Add(comp, alg.H1);
105  }
106  if (!alg.HN.IsNull() && RgNLineHCompound.getValue()) {
107  builder.Add(comp, alg.HN);
108  }
109  if (!alg.HO.IsNull() && OutLineHCompound.getValue()) {
110  builder.Add(comp, alg.HO);
111  }
112  if (!alg.HI.IsNull() && IsoLineHCompound.getValue()) {
113  builder.Add(comp, alg.HI);
114  }
115 
116  Shape.setValue(comp);
117  return App::DocumentObject::StdReturn;
118  }
119  catch (Standard_Failure& e) {
120  return new App::DocumentObjectExecReturn(e.GetMessageString());
121  }
122 }
