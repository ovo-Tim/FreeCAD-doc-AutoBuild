39 class GuiCommandSimplest:
40  """Simplest base class for GuiCommands.
41 
42  This class only sets up the command name and the document object
43  to use for the command.
44  When it is executed, it logs the command name to the log file,
45  and prints the command name to the console.
46 
47  It implements the `IsActive` method, which must return `True`
48  when the command should be available.
49  It should return `True` when there is an active document,
50  otherwise the command (button or menu) should be disabled.
51 
52  This class is meant to be inherited by other GuiCommand classes
53  to quickly log the command name, and set the correct document object.
54 
55  Parameter
56  ---------
57  name: str, optional
58  It defaults to `'None'`.
59  The name of the action that is being run,
60  for example, `'Heal'`, `'Flip dimensions'`,
61  `'Line'`, `'Circle'`, etc.
62 
63  doc: App::Document, optional
64  It defaults to the value of `App.activeDocument()`.
65  The document object itself, which indicates where the actions
66  of the command will be executed.
67 
68  Attributes
69  ----------
70  command_name: str
71  This is the command name, which is assigned by `name`.
72 
73  doc: App::Document
74  This is the document object itself, which is assigned by `doc`.
75 
76  This attribute should be used by functions to make sure
77  that the operations are performed in the correct document
78  and not in other documents.
79  To set the active document we can use
80 
81  >>> App.setActiveDocument(self.doc.Name)
82  """
83 
84  def __init__(self, name="None", doc=App.activeDocument()):
85  self.command_name = name
86  self.doc = doc
87 
88  def IsActive(self):
89  """Return True when this command should be available."""
90  return bool(App.activeDocument())
91 
92  def Activated(self):
93  """Execute when the command is called.
94 
95  Log the command name to the log file and console.
96  Also update the `doc` attribute.
97  """
98  self.doc = App.activeDocument()
99  _log("Document: {}".format(self.doc.Label))
100  _log("GuiCommand: {}".format(self.command_name))
101  _toolmsg("{}".format(16*"-"))
102  _toolmsg("GuiCommand: {}".format(self.command_name))

120 class GuiCommandBase:
121  """Generic class that is the basis of all Gui commands.
122 
123  This class should eventually replace `DraftTools.DraftTool`,
124  once all functionality in that class is merged here.
125 
126  Attributes
127  ----------
128  commit_list : list of 2-element tuples
129  Each tuple is made of a string, and a list of strings.
130  ::
131  commit_list = [(string1, list1), (string2, list2), ...]
132 
133  The string is a simple header, for example, a command name,
134  that indicates what is being executed.
135 
136  Each string in the list of strings represents a Python instruction
137  which will be executed in a delayed fashion
138  by `todo.ToDo.delayCommit()`
139  ::
140  list1 = ["a = FreeCAD.Vector()",
141  "pl = FreeCAD.Placement()",
142  "Draft.autogroup(obj)"]
143 
144  commit_list = [("Something", list1)]
145 
146  This is used when the 3D view has event callbacks that crash
147  Coin3D.
148  If this is not needed, those commands could be called in the
149  body of the command without problem.
150  ::
151  >>> a = FreeCAD.Vector()
152  >>> pl = FreeCAD.Placement()
153  >>> Draft.autogroup(obj)
154  """
155 
156  def __init__(self):
157  self.call = None
158  self.commit_list = []
159  self.doc = None
160  App.activeDraftCommand = None
161  self.view = None
162  self.planetrack = None
163 
164  def IsActive(self):
165  """Return True when this command should be available."""
166  return bool(gui_utils.get_3d_view())
167 
168  def finish(self):
169  """Terminate the active command by committing the list of commands.
170 
171  It also perform some other tasks like terminating
172  the plane tracker and the snapper.
173  """
174  App.activeDraftCommand = None
175  if self.planetrack:
176  self.planetrack.finalize()
177  if hasattr(Gui, "Snapper"):
178  Gui.Snapper.off()
179  if self.call:
180  try:
181  self.view.removeEventCallback("SoEvent", self.call)
182  except RuntimeError:
183  # the view has been deleted already
184  pass
185  self.call = None
186  if self.commit_list:
187  todo.ToDo.delayCommit(self.commit_list)
188  self.commit_list = []
189 
190  def commit(self, name, func):
191  """Store actions to be committed to the document.
192 
193  Parameters
194  ----------
195  name : str
196  A string that indicates what is being committed.
197 
198  func : list of strings
199  Each element of the list should be a Python command
200  that will be executed.
201  """
202  self.commit_list.append((name, func))
