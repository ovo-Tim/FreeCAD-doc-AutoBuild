63 void PropertyFileIncluded::aboutToSetValue()
64 {
65  // This is a trick to check in Copy() if it is called
66  // directly from outside or by the Undo/Redo mechanism.
67  // In the latter case it is sufficient to rename the file
68  // because another file will be assigned afterwards.
69  // If Copy() is directly called (e.g. to copy the file to
70  // another document) a copy of the file needs to be created.
71  // This copy will be deleted again in the class destructor.
72  this->StatusBits.set(10);
73  Property::aboutToSetValue();
74  this->StatusBits.reset(10);
75 }

111 void PropertyFileIncluded::setValue(const char* sFile, const char* sName)
112 {
113  if (sFile && sFile[0] != '\0') {
114  if (_cValue == sFile) {
115  throw Base::FileSystemError("Not possible to set the same file!");
116  }
117 
118  // keep the path to the original file
119  _OriginalName = sFile;
120 
121  std::string pathTrans = getDocTransientPath();
122  Base::FileInfo file(sFile);
123  std::string path = file.dirPath();
124  if (!file.exists()) {
125  std::stringstream str;
126  str << "File " << file.filePath() << " does not exist.";
127  throw Base::FileSystemError(str.str());
128  }
129 
130  aboutToSetValue(); // undo/redo by moving the file away with temp name
131 
132  // remove old file (if not moved by undo)
133  Base::FileInfo value(_cValue);
134  std::string pathAct = value.dirPath();
135  if (value.exists()) {
136  value.setPermissions(Base::FileInfo::ReadWrite);
137  value.deleteFile();
138  }
139 
140  // if a special name given, use this instead
141  if (sName) {
142  Base::FileInfo fi(pathTrans + "/" + sName);
143  if (fi.exists()) {
144  // if a file with this name already exists search for a new one
145  std::string dir = pathTrans;
146  std::string fnp = fi.fileNamePure();
147  std::string ext = fi.extension();
148  int i = 0;
149  do {
150  i++;
151  std::stringstream str;
152  str << dir << "/" << fnp << i;
153  if (!ext.empty()) {
154  str << "." << ext;
155  }
156  fi.setFile(str.str());
157  } while (fi.exists());
158 
159  _cValue = fi.filePath();
160  _BaseFileName = fi.fileName();
161  }
162  else {
163  _cValue = pathTrans + "/" + sName;
164  _BaseFileName = sName;
165  }
166  }
167  else if (value.fileName().empty()) {
168  _cValue = pathTrans + "/" + file.fileName();
169  _BaseFileName = file.fileName();
170  }
171 
172  // The following applies only on files that are inside the transient
173  // directory:
174  // When a file is read-only it is supposed to be assigned to a
175  // PropertyFileIncluded instance. In this case we must copy the
176  // file because otherwise the above instance loses its data.
177  // If the file is writable it is supposed to be of free use and
178  // it can be simply renamed.
179 
180  // if the file is already in transient dir of the document, just use it
181  if (path == pathTrans && file.isWritable()) {
182  bool done = file.renameFile(_cValue.c_str());
183  if (!done) {
184  std::stringstream str;
185  str << "Cannot rename file " << file.filePath() << " to " << _cValue;
186  throw Base::FileSystemError(str.str());
187  }
188 
189  // make the file read-only
190  Base::FileInfo dst(_cValue);
191  dst.setPermissions(Base::FileInfo::ReadOnly);
192  }
193  // otherwise copy from origin location
194  else {
195  // if file already exists in transient dir make a new unique name
196  Base::FileInfo fi(_cValue);
197  if (fi.exists()) {
198  // if a file with this name already exists search for a new one
199  std::string dir = fi.dirPath();
200  std::string fnp = fi.fileNamePure();
201  std::string ext = fi.extension();
202  int i = 0;
203  do {
204  i++;
205  std::stringstream str;
206  str << dir << "/" << fnp << i;
207  if (!ext.empty()) {
208  str << "." << ext;
209  }
210  fi.setFile(str.str());
211  } while (fi.exists());
212 
213  _cValue = fi.filePath();
214  _BaseFileName = fi.fileName();
215  }
216 
217  bool done = file.copyTo(_cValue.c_str());
218  if (!done) {
219  std::stringstream str;
220  str << "Cannot copy file from " << file.filePath() << " to " << _cValue;
221  throw Base::FileSystemError(str.str());
222  }
223 
224  // make the file read-only
225  Base::FileInfo dst(_cValue);
226  dst.setPermissions(Base::FileInfo::ReadOnly);
227  }
228 
229  hasSetValue();
230  }

249 const char* getNameFromFile(PyObject* value)
250 {
251  const char* string = nullptr;
252  PyObject* oname = PyObject_GetAttrString(value, "name");
253  if (oname) {
254  if (PyUnicode_Check(oname)) {
255  string = PyUnicode_AsUTF8(oname);
256  }
257  else if (PyBytes_Check(oname)) {
258  string = PyBytes_AsString(oname);
259  }
260  Py_DECREF(oname);
261  }
262 
263  if (!string) {
264  throw Base::TypeError("Unable to get filename");
265  }
266  return string;
267 }

281 void PropertyFileIncluded::setPyObject(PyObject* value)
282 {
283  if (PyUnicode_Check(value)) {
284  std::string string = PyUnicode_AsUTF8(value);
285  setValue(string.c_str());
286  }
287  else if (PyBytes_Check(value)) {
288  std::string string = PyBytes_AsString(value);
289  setValue(string.c_str());
290  }
291  else if (isIOFile(value)) {
292  std::string string = getNameFromFile(value);
293  setValue(string.c_str());
294  }
295  else if (PyTuple_Check(value)) {
296  if (PyTuple_Size(value) != 2) {
297  throw Base::TypeError("Tuple needs size of (filePath,newFileName)");
298  }
299  PyObject* file = PyTuple_GetItem(value, 0);
300  PyObject* name = PyTuple_GetItem(value, 1);
301 
302  // decoding file
303  std::string fileStr;
304  if (PyUnicode_Check(file)) {
305  fileStr = PyUnicode_AsUTF8(file);
306  }
307  else if (PyBytes_Check(file)) {
308  fileStr = PyBytes_AsString(file);
309  }
310  else if (isIOFile(value)) {
311  fileStr = getNameFromFile(file);
312  }
313  else {
314  std::string error = std::string("First item in tuple must be a file or string, not ");
315  error += file->ob_type->tp_name;
316  throw Base::TypeError(error);
317  }
318 
319  // decoding name
320  std::string nameStr;
321  if (PyUnicode_Check(name)) {
322  nameStr = PyUnicode_AsUTF8(name);
323  }
324  else if (PyBytes_Check(name)) {
325  nameStr = PyBytes_AsString(name);
326  }
327  else if (isIOFile(value)) {
328  nameStr = getNameFromFile(name);
329  }
330  else {
331  std::string error = std::string("Second item in tuple must be a string, not ");
332  error += name->ob_type->tp_name;
333  throw Base::TypeError(error);
334  }
335 
336  setValue(fileStr.c_str(), nameStr.c_str());
337  }
338  else if (PyDict_Check(value)) {
339  Py::Dict dict(value);
340  if (dict.hasKey("filter")) {
341  setFilter(Py::String(dict.getItem("filter")));
342  }
343  if (dict.hasKey("filename")) {
344  std::string string = static_cast<std::string>(Py::String(dict.getItem("filename")));
345  setValue(string.c_str());
346  }
347  }
348  else {
349  std::string error = std::string("Type must be string or file, not ");
350  error += value->ob_type->tp_name;
351  throw Base::TypeError(error);
352  }
353 }

355 void PropertyFileIncluded::Save(Base::Writer& writer) const
356 {
357  // when saving a document under a new file name the transient directory
358  // name changes and thus the stored file name doesn't work any more.
359  if (!_cValue.empty() && !Base::FileInfo(_cValue).exists()) {
360  Base::FileInfo fi(getDocTransientPath() + "/" + _BaseFileName);
361  if (fi.exists()) {
362  _cValue = fi.filePath();
363  }
364  }
365 
366  if (writer.isForceXML()) {
367  if (!_cValue.empty()) {
368  Base::FileInfo file(_cValue.c_str());
369  writer.Stream() << writer.ind() << "<FileIncluded data=\"" << file.fileName() << "\">"
370  << std::endl;
371  // write the file in the XML stream
372  writer.incInd();
373  writer.insertBinFile(_cValue.c_str());
374  writer.decInd();
375  writer.Stream() << writer.ind() << "</FileIncluded>" << endl;
376  }
377  else {
378  writer.Stream() << writer.ind() << "<FileIncluded data=\"\"/>" << std::endl;
379  }
380  }
381  else {
382  // instead initiate an extra file
383  if (!_cValue.empty()) {
384  Base::FileInfo file(_cValue.c_str());
385  std::string filename = writer.addFile(file.fileName().c_str(), this);
386  filename = encodeAttribute(filename);
387  writer.Stream() << writer.ind() << "<FileIncluded file=\"" << filename << "\"/>"
388  << std::endl;
389  }
390  else {
391  writer.Stream() << writer.ind() << "<FileIncluded file=\"\"/>" << std::endl;
392  }
393  }

396 void PropertyFileIncluded::Restore(Base::XMLReader& reader)
397 {
398  reader.readElement("FileIncluded");
399  if (reader.hasAttribute("file")) {
400  string file(reader.getAttribute("file"));
401  if (!file.empty()) {
402  // initiate a file read
403  reader.addFile(file.c_str(), this);
404  // is in the document transient path
405  aboutToSetValue();
406  _cValue = getDocTransientPath() + "/" + file;
407  _BaseFileName = file;
408  hasSetValue();
409  }
410  }
411  // section is XML stream
412  else if (reader.hasAttribute("data")) {
413  string file(reader.getAttribute("data"));
414  if (!file.empty()) {
415  // is in the document transient path
416  aboutToSetValue();
417  _cValue = getDocTransientPath() + "/" + file;
418  reader.readBinFile(_cValue.c_str());
419  reader.readEndElement("FileIncluded");
420  _BaseFileName = file;
421  // set read-only after restoring the file
422  Base::FileInfo fi(_cValue.c_str());
423  fi.setPermissions(Base::FileInfo::ReadOnly);
424  hasSetValue();
425  }

448 {

470

476 Property* PropertyFileIncluded::Copy() const
477 {
478  std::unique_ptr<PropertyFileIncluded> prop(new PropertyFileIncluded());
479 
480  // remember the base name
481  prop->_BaseFileName = _BaseFileName;
482 
483  Base::FileInfo file(_cValue);
484  if (file.exists()) {
485  // create a new name in the document transient directory
486  Base::FileInfo newName(getUniqueFileName(file.dirPath(), file.fileName()));
487  if (this->StatusBits.test(10)) {
488  // rename the file
489  bool done = file.renameFile(newName.filePath().c_str());
490  if (!done) {
491  std::stringstream str;
492  str << "PropertyFileIncluded::Copy(): "
493  << "Renaming the file '" << file.filePath() << "' to '" << newName.filePath()
494  << "' failed.";
495  throw Base::FileSystemError(str.str());
496  }
497  }
498  else {
499  // copy the file
500  bool done = file.copyTo(newName.filePath().c_str());
501  if (!done) {
502  std::stringstream str;
503  str << "PropertyFileIncluded::Copy(): "
504  << "Copying the file '" << file.filePath() << "' to '" << newName.filePath()
505  << "' failed.";
506  throw Base::FileSystemError(str.str());
507  }
508  }
509 
510  // remember the new name for the Undo
511  Base::Console().Log("Copy '%s' to '%s'\n", _cValue.c_str(), newName.filePath().c_str());
512  prop->_cValue = newName.filePath().c_str();
513 
514  // make backup files writable to avoid copying them again on undo/redo
515  newName.setPermissions(Base::FileInfo::ReadWrite);
516  }
517

521 void PropertyFileIncluded::Paste(const Property& from)
522 {
523  aboutToSetValue();
524  const PropertyFileIncluded& prop = dynamic_cast<const PropertyFileIncluded&>(from);
525  // make sure that source and destination file are different
526  if (_cValue != prop._cValue) {
527  // delete old file (if still there)
528  Base::FileInfo fi(_cValue);
529  fi.setPermissions(Base::FileInfo::ReadWrite);
530  fi.deleteFile();
531 
532  // get path to destination which can be the transient directory
533  // of another document
534  std::string pathTrans = getDocTransientPath();
535  Base::FileInfo fiSrc(prop._cValue);
536  Base::FileInfo fiDst(pathTrans + "/" + prop._BaseFileName);
537  std::string path = fiSrc.dirPath();
538 
539  if (fiSrc.exists()) {
540  fiDst.setFile(getUniqueFileName(fiDst.dirPath(), fiDst.fileName()));
541 
542  // if the file is already in transient dir of the document, just use it
543  if (path == pathTrans) {
544  if (!fiSrc.renameFile(fiDst.filePath().c_str())) {
545  std::stringstream str;
546  str << "PropertyFileIncluded::Paste(): "
547  << "Renaming the file '" << fiSrc.filePath() << "' to '" << fiDst.filePath()
548  << "' failed.";
549  throw Base::FileSystemError(str.str());
550  }
551  }
552  else {
553  if (!fiSrc.copyTo(fiDst.filePath().c_str())) {
554  std::stringstream str;
