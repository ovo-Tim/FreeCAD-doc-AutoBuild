645  def recomputePointsBezier(self, obj, pts, idx, v,
646  degree, moveTrackers=False):
647  """
648  (object, Points as list, nodeIndex as Int, App.Vector of new point, moveTrackers as Bool)
649  return the new point list, applying the App.Vector to the given index point
650  """
651  # DNC: allows one to close the curve by placing ends close to each other
652  tol = 0.001
653  if ( ( idx == 0 ) and ( (v - pts[-1]).Length < tol) ) or (
654  idx == len(pts) - 1 ) and ( (v - pts[0]).Length < tol):
655  obj.Closed = True
656  # DNC: fix error message if edited point coincides with one of the existing points
657  #if ( v in pts ) == False:
658  knot = None
659  ispole = idx % degree
660 
661  if ispole == 0: #knot
662  if degree >= 3:
663  if idx >= 1: #move left pole
664  knotidx = idx if idx < len(pts) else 0
665  pts[idx-1] = pts[idx-1] + v - pts[knotidx]
666  #if moveTrackers: # trackers are reset after editing
667  # self.trackers[obj.Name][idx-1].set(pts[idx-1])
668  if idx < len(pts)-1: #move right pole
669  pts[idx+1] = pts[idx+1] + v - pts[idx]
670  #if moveTrackers:
671  # self.trackers[obj.Name][idx+1].set(pts[idx+1])
672  if idx == 0 and obj.Closed: # move last pole
673  pts[-1] = pts [-1] + v -pts[idx]
674  #if moveTrackers:
675  # self.trackers[obj.Name][-1].set(pts[-1])
676 
677  elif ispole == 1 and (idx >=2 or obj.Closed): #right pole
678  knot = idx -1
679  changep = idx - 2 # -1 in case of closed curve
680 
681  elif ispole == degree-1 and idx <= len(pts)-3: # left pole
682  knot = idx + 1
683  changep = idx + 2
684 
685  elif ispole == degree-1 and obj.Closed and idx == len(pts)-1: #last pole
686  knot = 0
687  changep = 1
688 
689  if knot is not None: # we need to modify the opposite pole
690  segment = int(knot / degree) - 1
691  cont = obj.Continuity[segment] if len(obj.Continuity) > segment else 0
692  if cont == 1: #tangent
693  pts[changep] = obj.Proxy.modifytangentpole(pts[knot],
694  v,pts[changep])
695  #if moveTrackers:
696  # self.trackers[obj.Name][changep].set(pts[changep])
697  elif cont == 2: #symmetric
698  pts[changep] = obj.Proxy.modifysymmetricpole(pts[knot],v)
699  #if moveTrackers:

701  pts[idx] = v

706  def smoothBezPoint(self, obj, point, style='Symmetric'):
707  """called when changing the continuity of a knot
708  """
709  style2cont = {'Sharp':0,'Tangent':1,'Symmetric':2}
710  if point is None:
711  return
712  if not (utils.get_type(obj) == "BezCurve"):
713  return
714  pts = obj.Points
715  deg = obj.Degree
716  if deg < 2:
717  return
718  if point % deg != 0: # point is a pole
719  if deg >=3: # allow one to select poles
720  if (point % deg == 1) and (point > 2 or obj.Closed): #right pole
721  knot = point -1
722  keepp = point
723  changep = point -2
724  elif point < len(pts) -3 and point % deg == deg -1: #left pole
725  knot = point +1
726  keepp = point
727  changep = point +2
728  elif point == len(pts)-1 and obj.Closed: #last pole
729  # if the curve is closed the last pole has the last
730  # index in the points lists
731  knot = 0
732  keepp = point
733  changep = 1
734  else:
735  App.Console.PrintWarning(translate("draft",
736  "Can't change Knot belonging to pole %d"%point)
737  + "\n")
738  return
739  if knot:
740  if style == 'Tangent':
741  pts[changep] = obj.Proxy.modifytangentpole(pts[knot],
742  pts[keepp],pts[changep])
743  elif style == 'Symmetric':
744  pts[changep] = obj.Proxy.modifysymmetricpole(pts[knot],
745  pts[keepp])
746  else: #sharp
747  pass #
748  else:
749  App.Console.PrintWarning(translate("draft",
750  "Selection is not a Knot")
751  + "\n")
752  return
753  else: #point is a knot
754  if style == 'Sharp':
755  if obj.Closed and point == len(pts)-1:
756  knot = 0
757  else:
758  knot = point
759  elif style == 'Tangent' and point > 0 and point < len(pts)-1:
760  prev, next = obj.Proxy.tangentpoles(pts[point], pts[point-1], pts[point+1])
761  pts[point-1] = prev
762  pts[point+1] = next
763  knot = point # index for continuity
764  elif style == 'Symmetric' and point > 0 and point < len(pts)-1:
765  prev, next = obj.Proxy.symmetricpoles(pts[point], pts[point-1], pts[point+1])
766  pts[point-1] = prev
767  pts[point+1] = next
768  knot = point # index for continuity
769  elif obj.Closed and (style == 'Symmetric' or style == 'Tangent'):
770  if style == 'Tangent':
771  pts[1], pts[-1] = obj.Proxy.tangentpoles(pts[0], pts[1], pts[-1])
772  elif style == 'Symmetric':
773  pts[1], pts[-1] = obj.Proxy.symmetricpoles(pts[0], pts[1], pts[-1])
774  knot = 0
775  else:
776  App.Console.PrintWarning(translate("draft",
777  "Endpoint of Bézier curve can't be smoothed")
778  + "\n")
779  return
780  segment = knot // deg # segment index
781  newcont = obj.Continuity[:] # don't edit a property inplace !!!
782  if not obj.Closed and (len(obj.Continuity) == segment -1 or
783  segment == 0) :
784  pass # open curve
785  elif (len(obj.Continuity) >= segment or obj.Closed and segment == 0 and
786  len(obj.Continuity) >1):
787  newcont[segment-1] = style2cont.get(style)
788  else: #should not happen
789  App.Console.PrintWarning('Continuity indexing error:'
790  + 'point:%d deg:%d len(cont):%d' % (knot,deg,
791  len(obj.Continuity)))
792  obj.Points = pts
793  obj.Continuity = newcont

795  def add_point(self, edit_command, obj, pos):
796  """Add point to obj and reset trackers.
797  """
798  info, pt = edit_command.get_specific_object_info(obj,pos)
799  if not info or (pt is None):
800  return
801 
802  import Part
803 
804  pts = obj.Points
805  if not info['Component'].startswith('Edge'):
806  return # clicked control point

811  seg1 = wire.Edges[edgeindex].copy().Curve
812  seg2 = wire.Edges[edgeindex].copy().Curve

819  seg2.increase(degree)

821  + wire.Edges[edgeindex + 1:]
822  pts = edges[0].Curve.getPoles()[0:1]

824  pts.extend(edge.Curve.getPoles()[1:])

827  c = obj.Continuity
828  # assume we have a tangent continuity for an arbitrarily split
829  # segment, unless it's linear
