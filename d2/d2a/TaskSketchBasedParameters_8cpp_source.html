61 const QString TaskSketchBasedParameters::onAddSelection(const Gui::SelectionChanges& msg)
62 {
63  // Note: The validity checking has already been done in ReferenceSelection.cpp
64  auto sketchBased = getObject<PartDesign::ProfileBased>();
65  App::DocumentObject* selObj = sketchBased->getDocument()->getObject(msg.pObjectName);
66  if (selObj == sketchBased) {
67  return QString();
68  }
69 
70  std::string subname = msg.pSubName;
71  QString refStr;
72 
73  // Remove subname for planes and datum features
74  if (PartDesign::Feature::isDatum(selObj)) {
75  subname = "";
76  refStr = QString::fromLatin1(selObj->getNameInDocument());
77  }
78  else if (subname.size() > 4) {
79  int faceId = std::atoi(&subname[4]);
80  refStr = QString::fromLatin1(selObj->getNameInDocument()) + QString::fromLatin1(":") + QObject::tr("Face") + QString::number(faceId);
81  }
82 
83  std::vector<std::string> upToFaces(1,subname);
84  sketchBased->UpToFace.setValue(selObj, upToFaces);
85  recomputeFeature();
86 
87  return refStr;
88 }

112 void TaskSketchBasedParameters::onSelectReference(AllowSelectionFlags allow)
113 {
114  // Note: Even if there is no solid, App::Plane and Part::Datum can still be selected
115  if (auto sketchBased = getObject<PartDesign::ProfileBased>()) {
116  // The solid this feature will be fused to
117  App::DocumentObject* prevSolid = sketchBased->getBaseObject( /* silent =*/ true );
118 
119  if (AllowSelectionFlags::Int(allow) != int(AllowSelection::NONE)) {
120  startReferenceSelection(sketchBased, prevSolid);
121  this->blockSelection(false);
122  Gui::Selection().clearSelection();
123  Gui::Selection().addSelectionGate(new ReferenceSelection(prevSolid, allow));
124  }
125  else {
126  Gui::Selection().rmvSelectionGate();
127  finishReferenceSelection(sketchBased, prevSolid);
128  this->blockSelection(true);
129  }
130  }
131 }

139 QVariant TaskSketchBasedParameters::setUpToFace(const QString& text)
140 {
141  if (text.isEmpty())
142  return {};
143 
144  QStringList parts = text.split(QChar::fromLatin1(':'));
145  if (parts.length() < 2) {
146  parts.push_back(QString());
147  }
148 
149  // Check whether this is the name of an App::Plane or Part::Datum feature
150  App::Document* doc = getAppDocument();
151  if (!doc) {
152  return {};
153  }
154 
155  App::DocumentObject* obj = doc->getObject(parts[0].toLatin1());
156  if (!obj) {
157  return {};
158  }
159 
160  if (obj->isDerivedFrom<App::Plane>()) {
161  // everything is OK (we assume a Part can only have exactly 3 App::Plane objects
162  // located at the base of the feature tree)
163  return {};
164  }
165 
166  if (obj->isDerivedFrom<Part::Datum>()) {
167  // it's up to the document to check that the datum plane is in the same body
168  return {};
169  }
170 
171  // We must expect that "parts[1]" is the translation of "Face" followed by an ID.
172  QString name;
173  QTextStream str(&name);
174  str << "^" << tr("Face") << "(\\d+)$";
175  QRegularExpression rx(name);
176  QRegularExpressionMatch match;
177  if (parts[1].indexOf(rx, 0, &match) < 0) {
178  return {};
179  }
180 
181  int faceId = match.captured(1).toInt();
182  std::stringstream ss;
183  ss << "Face" << faceId;
184 
185  std::vector<std::string> upToFaces(1, ss.str());
186  auto sketchBased = getObject<PartDesign::ProfileBased>();
187  sketchBased->UpToFace.setValue(obj, upToFaces);
188  recomputeFeature();
189 
190  return QByteArray(ss.str().c_str());
191 }

193 QVariant TaskSketchBasedParameters::objectNameByLabel(const QString& label,
194  const QVariant& suggest) const
195 {
196  // search for an object with the given label
197  App::Document* doc = getAppDocument();
198  if (!doc) {
199  return {};
200  }
201 
202  // for faster access try the suggestion
203  if (suggest.isValid()) {
204  App::DocumentObject* obj = doc->getObject(suggest.toByteArray());
205  if (obj && QString::fromUtf8(obj->Label.getValue()) == label) {
206  return QVariant(QByteArray(obj->getNameInDocument()));
207  }
208  }
209 
210  // go through all objects and check the labels
211  std::string name = label.toUtf8().data();
212  std::vector<App::DocumentObject*> objs = doc->getObjects();
213  for (auto obj : objs) {
214  if (name == obj->Label.getValue()) {
215  return QVariant(QByteArray(obj->getNameInDocument()));
216  }
217  }
218 
219  return {}; // no such feature found
220 }

222 QString TaskSketchBasedParameters::getFaceReference(const QString& obj, const QString& sub) const
223 {
224  App::Document* doc = getAppDocument();
225  if (!doc) {
226  return {};
227  }
228 
229  QString o = obj.left(obj.indexOf(QString::fromLatin1(":")));
230  if (o.isEmpty()) {
231  return {};
232  }
233 
234  return QString::fromLatin1(R"((App.getDocument("%1").%2, ["%3"]))")
235  .arg(QString::fromLatin1(doc->getName()), o, sub);
236 }

238 QString TaskSketchBasedParameters::make2DLabel(const App::DocumentObject* section,
239  const std::vector<std::string>& subValues)
240 {

245  Base::Console().Error("No valid subelement linked in %s\n", section->Label.getValue());

248  else {
249  return QString::fromStdString((std::string(section->getNameInDocument()) + ":" + subValues[0]));

304

307

309
