182 void Cell::setExpression(App::ExpressionPtr&& expr)
183 {
184  PropertySheet::AtomicPropertyChange signaller(*owner);
185 
186  owner->setDirty(address);
187 
188  /* Remove dependencies */
189  owner->removeDependencies(address);
190 
191  if (expr && !expr->comment.empty()) {
192  if (!boost::starts_with(expr->comment, "<Cell ")) {
193  FC_WARN("Unknown style of cell " << owner->sheet()->getFullName() << '.'
194  << address.toString());
195  }
196  else {
197  try {
198  std::istringstream in(expr->comment);
199  ReaderPrivate reader("<memory>", in);
200  reader.read();
201  restore(reader, true);
202  }
203  catch (Base::Exception& e) {
204  e.ReportException();
205  FC_ERR("Failed to restore style of cell " << owner->sheet()->getFullName() << '.'
206  << address.toString() << ": "
207  << e.what());
208  }
209  }
210  expr->comment.clear();
211  }
212 
213  expression = std::move(expr);
214  setUsed(EXPRESSION_SET, !!expression);
215 
216  /* Update dependencies */
217  owner->addDependencies(address);
218 
219  signaller.tryInvoke();
220 }

283 void Cell::setContent(const char* value)
284 {
285  PropertySheet::AtomicPropertyChange signaller(*owner);
286  ExpressionPtr newExpr;
287 
288  clearException();
289  if (value) {
290  if (owner->sheet()->isRestoring()) {
291  if (value[0] == '\0' || (value[0] == '\'' && value[1] == '\0')) {
292  return;
293  }
294  expression = std::make_unique<App::StringExpression>(owner->sheet(), value);
295  setUsed(EXPRESSION_SET, true);
296  return;
297  }
298  if (*value == '=') {
299  try {
300  newExpr = ExpressionPtr(App::ExpressionParser::parse(owner->sheet(), value + 1));
301  }
302  catch (Base::Exception& e) {
303  newExpr = std::make_unique<App::StringExpression>(owner->sheet(), value);
304  setParseException(e.what());
305  }
306  }
307  else if (*value == '\'') {
308  if (value[1] == '\0') {
309  value = nullptr;
310  }
311  else {
312  newExpr = std::make_unique<App::StringExpression>(owner->sheet(), value + 1);
313  }
314  }
315  else if (*value != '\0') {
316  // check if value is just a number
317  char* end;
318  errno = 0;
319  const double float_value = strtod(value, &end);
320  if (errno == 0) {
321  const bool isEndEmpty = *end == '\0' || strspn(end, " \t\n\r") == strlen(end);
322  if (isEndEmpty) {
323  newExpr = std::make_unique<App::NumberExpression>(owner->sheet(),
324  Quantity(float_value));
325  }
326  }
327 
328  // if not a float, check if it is a quantity or compatible fraction
329  const bool isStartingWithNumber = value != end;
330  if (!newExpr && isStartingWithNumber) {
331  try {
332  ExpressionPtr parsedExpr(App::ExpressionParser::parse(owner->sheet(), value));
333 
334  if (const auto fraction =
335  freecad_dynamic_cast<OperatorExpression>(parsedExpr.get())) {
336  if (fraction->getOperator() == OperatorExpression::UNIT) {
337  const auto left =
338  freecad_dynamic_cast<NumberExpression>(fraction->getLeft());
339  const auto right =
340  freecad_dynamic_cast<UnitExpression>(fraction->getRight());
341  if (left && right) {
342  newExpr = std::move(parsedExpr);
343  }
344  }
345  else if (fraction->getOperator() == OperatorExpression::DIV) {
346  // only the following types of fractions are ok:
347  // 1/2, 1m/2, 1/2s, 1m/2s, 1/m
348 
349  // check for numbers in (de)nominator
350  const bool isNumberNom =
351  freecad_dynamic_cast<NumberExpression>(fraction->getLeft());
352  const bool isNumberDenom =
353  freecad_dynamic_cast<NumberExpression>(fraction->getRight());
354 
355  // check for numbers with units in (de)nominator
356  const auto opNom =
357  freecad_dynamic_cast<OperatorExpression>(fraction->getLeft());
358  const auto opDenom =
359  freecad_dynamic_cast<OperatorExpression>(fraction->getRight());
360  const bool isQuantityNom =
361  opNom && opNom->getOperator() == OperatorExpression::UNIT;
362  const bool isQuantityDenom =
363  opDenom && opDenom->getOperator() == OperatorExpression::UNIT;
364 
365  // check for units in denomainator
366  const auto uDenom =
367  freecad_dynamic_cast<UnitExpression>(fraction->getRight());
368  const bool isUnitDenom = uDenom && uDenom->is<UnitExpression>();
369 
370  const bool isNomValid = isNumberNom || isQuantityNom;
371  const bool isDenomValid =
372  isNumberDenom || isQuantityDenom || isUnitDenom;
373  if (isNomValid && isDenomValid) {
374  newExpr = std::move(parsedExpr);
375  }
376  }
377  }
378  else if (const auto number =
379  freecad_dynamic_cast<NumberExpression>(parsedExpr.get())) {
380  // NumbersExpressions can accept more than can be parsed with strtod.
381  // Example: 12.34 and 12,34 are both valid NumberExpressions
382  newExpr = std::move(parsedExpr);
383  }
384  }
385  catch (...) {
386  }
387  }
388  }
389 
390  if (!newExpr && value && *value != '\0') {
391  newExpr = std::make_unique<App::StringExpression>(owner->sheet(), value);
392  }
393 
394  // trying to add an empty string will make newExpr = nullptr
395  }
396 
397  // set expression, or delete the current expression by setting nullptr if empty string was
398  // entered
399  setExpression(std::move(newExpr));
400  signaller.tryInvoke();
401 }

733 void Cell::restore(Base::XMLReader& reader, bool checkAlias)
734 {
735  const char* style = reader.hasAttribute("style") ? reader.getAttribute("style") : nullptr;
736  const char* alignment =
737  reader.hasAttribute("alignment") ? reader.getAttribute("alignment") : nullptr;
738  const char* content = reader.hasAttribute("content") ? reader.getAttribute("content") : "";
739  const char* foregroundColor =
740  reader.hasAttribute("foregroundColor") ? reader.getAttribute("foregroundColor") : nullptr;
741  const char* backgroundColor =
742  reader.hasAttribute("backgroundColor") ? reader.getAttribute("backgroundColor") : nullptr;
743  const char* displayUnit =
744  reader.hasAttribute("displayUnit") ? reader.getAttribute("displayUnit") : nullptr;
745  const char* alias = reader.hasAttribute("alias") ? reader.getAttribute("alias") : nullptr;
746  const char* rowSpan = reader.hasAttribute("rowSpan") ? reader.getAttribute("rowSpan") : nullptr;
747  const char* colSpan = reader.hasAttribute("colSpan") ? reader.getAttribute("colSpan") : nullptr;
748 
749  // Don't trigger multiple updates below; wait until everything is loaded by calling unfreeze()
750  // below.
751  PropertySheet::AtomicPropertyChange signaller(*owner);
752 
753  if (content) {
754  setContent(content);
755  }
756  if (style) {
757  using namespace boost;
758  std::set<std::string> styleSet;
759 
760  escaped_list_separator<char> e('\0', '|', '\0');
761  std::string line = std::string(style);
762  tokenizer<escaped_list_separator<char>> tok(line, e);
763 
764  for (tokenizer<escaped_list_separator<char>>::iterator i = tok.begin(); i != tok.end();
765  ++i) {
766  styleSet.insert(*i);
767  }
768  setStyle(styleSet);
769  }
770  if (alignment) {
771  int alignmentCode = 0;
772  using namespace boost;
773 
774  escaped_list_separator<char> e('\0', '|', '\0');
775  std::string line = std::string(alignment);
776  tokenizer<escaped_list_separator<char>> tok(line, e);
777 
778  for (tokenizer<escaped_list_separator<char>>::iterator i = tok.begin(); i != tok.end();
779  ++i) {
780  alignmentCode = decodeAlignment(*i, alignmentCode);
781  }
782 
783  setAlignment(alignmentCode);
784  }
785  if (foregroundColor) {
786  App::Color color = decodeColor(foregroundColor, App::Color(0, 0, 0, 1));
787 
788  setForeground(color);
789  }
790  if (backgroundColor) {
791  App::Color color = decodeColor(backgroundColor, App::Color(1, 1, 1, 1));
792 
793  setBackground(color);
794  }
795  if (displayUnit) {
796  setDisplayUnit(displayUnit);
797  }
798  if (alias && (!checkAlias || !owner->revAliasProp.count(alias))) {
799  setAlias(alias);
800  }
801 
802  if (rowSpan || colSpan) {
803  int rs = rowSpan ? atoi(rowSpan) : 1;
804  int cs = colSpan ? atoi(colSpan) : 1;
805 
806  setSpans(rs, cs);
807  }
808 }
