85 void SolverGeometryExtension::notifyAttachment(Part::Geometry* geo)
86 {
87  // maps type to number of solver parameters taken by the edge
88  static std::map<Base::Type, int> edgeParamMap = {
89  {Part::GeomPoint::getClassTypeId(), 0},
90  {Part::GeomLineSegment::getClassTypeId(), 0},
91  {Part::GeomArcOfCircle::getClassTypeId(), 3},
92  {Part::GeomCircle::getClassTypeId(), 1},
93  {Part::GeomArcOfEllipse::getClassTypeId(), 5},
94  {Part::GeomEllipse::getClassTypeId(), 3},
95  {Part::GeomArcOfHyperbola::getClassTypeId(), 5},
96  {Part::GeomArcOfParabola::getClassTypeId(), 4},
97  {Part::GeomBSplineCurve::getClassTypeId(), 0} // is dynamic
98  };
99 
100  GeometryType = geo->getTypeId();
101 
102  auto result = edgeParamMap.find(GeometryType);
103 
104  if (result == edgeParamMap.end()) {
105  THROWM(Base::TypeError,
106  "SolverGeometryExtension - notifyAttachment - Geometry not supported!!");
107  }
108 
109  auto nedgeparams = (*result).second;
110 
111  if (nedgeparams > 0) {
112  Edge.init(nedgeparams);
113  }
114 }
