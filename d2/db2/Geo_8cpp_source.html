82 DeriVector2 DeriVector2::getNormalized() const
83 {
84  double l = length();
85  if (l == 0.0) {
86  return DeriVector2(0, 0, dx, dy);
87  }
88  else {
89  DeriVector2 rtn;
90  rtn.x = x / l;
91  rtn.y = y / l;
92  // first, simply scale the derivative accordingly.
93  rtn.dx = dx / l;
94  rtn.dy = dy / l;
95  // next, remove the collinear part of dx,dy (make a projection onto a normal)
96  double dsc = rtn.dx * rtn.x + rtn.dy * rtn.y; // scalar product d*v
97  rtn.dx -= dsc * rtn.x; // subtract the projection
98  rtn.dy -= dsc * rtn.y;
99  return rtn;
100  }
101 }

192 DeriVector2 Circle::Value(double u, double du, const double* derivparam) const
193 {
194  DeriVector2 cv(center, derivparam);
195  double r, dr;
196  r = *(this->rad);
197  dr = (derivparam == this->rad) ? 1.0 : 0.0;
198  DeriVector2 ex(r, 0.0, dr, 0.0);
199  DeriVector2 ey = ex.rotate90ccw();
200  double si, dsi, co, dco;
201  si = std::sin(u);
202  dsi = du * std::cos(u);
203  co = std::cos(u);
204  dco = du * (-std::sin(u));
205  return cv.sum(ex.multD(co, dco).sum(ey.multD(si, dsi)));
206 }

331 DeriVector2 Ellipse::Value(double u, double du, const double* derivparam) const
332 {
333  // In local coordinate system, value() of ellipse is:
334  //(a*cos(u), b*sin(u))
335  // In global, it is (vector formula):
336  // center + a_vec*cos(u) + b_vec*sin(u).
337  // That's what is being computed here.
338 
339  // <construct a_vec, b_vec>
340  DeriVector2 c(this->center, derivparam);
341  DeriVector2 f1(this->focus1, derivparam);
342 
343  DeriVector2 emaj = f1.subtr(c).getNormalized();
344  DeriVector2 emin = emaj.rotate90ccw();
345  double b, db;
346  b = *(this->radmin);
347  db = this->radmin == derivparam ? 1.0 : 0.0;
348  double a, da;
349  a = this->getRadMaj(c, f1, b, db, da);
350  DeriVector2 a_vec = emaj.multD(a, da);
351  DeriVector2 b_vec = emin.multD(b, db);
352  // </construct a_vec, b_vec>
353 
354  // sin, cos with derivatives:
355  double co, dco, si, dsi;
356  co = std::cos(u);
357  dco = -std::sin(u) * du;
358  si = std::sin(u);
359  dsi = std::cos(u) * du;
360 
361  DeriVector2 ret; // point of ellipse at parameter value of u, in global coordinates
362  ret = a_vec.multD(co, dco).sum(b_vec.multD(si, dsi)).sum(c);
363  return ret;
364 }

500  //(a*cosh(u), b*sinh(u))
501  // In global, it is (vector formula):
502  // center + a_vec*cosh(u) + b_vec*sinh(u).

609 DeriVector2 Parabola::CalculateNormal(const Point& p, const double* derivparam) const
610 {
611  // fill some vectors in
612  DeriVector2 cv(vertex, derivparam);
613  DeriVector2 f1v(focus1, derivparam);
614  DeriVector2 pv(p, derivparam);
615 
616  // the normal is the vector from the focus to the intersection of ano thru the point p and
617  // direction of the symmetry axis of the parabola with the directrix. As both point to directrix
618  // and point to focus are of equal magnitude, we can work with unitary vectors to calculate the
619  // normal, substraction of those vectors.
620 
621  DeriVector2 ret = cv.subtr(f1v).getNormalized().subtr(f1v.subtr(pv).getNormalized());
622 
623  return ret;
624 }

626 DeriVector2 Parabola::Value(double u, double du, const double* derivparam) const
627 {
628 
629  // In local coordinate system, value() of parabola is:
630  // P(U) = O + U*U/(4.*F)*XDir + U*YDir
631 
632  DeriVector2 c(this->vertex, derivparam);
633  DeriVector2 f1(this->focus1, derivparam);
634 
635  DeriVector2 fv = f1.subtr(c);
636 
637  double f, df;
638 
639  f = fv.length(df);
640 
641  DeriVector2 xdir = fv.getNormalized();
642  DeriVector2 ydir = xdir.rotate90ccw();
643 
644  DeriVector2 dirx = xdir.multD(u, du).multD(u, du).divD(4 * f, 4 * df);
645  DeriVector2 diry = ydir.multD(u, du);
646 
647  DeriVector2 dir = dirx.sum(diry);
648 
649  DeriVector2 ret; // point of parabola at parameter value of u, in global coordinates
650 
651  ret = c.sum(dir);
652 
653  return ret;
654 }

733  DeriVector2 ret;

736  // sensus but a normal vector, where the vector should point to the left when one walks along
737  // the curve from start to end.
738  //
739  // https://forum.freecad.org/viewtopic.php?f=10&t=26312#p209486
740 
741  if (mult[0] > degree && mult[mult.size() - 1] > degree) {
742  // if endpoints through end poles
743  if (*p.x == *start.x && *p.y == *start.y) {
744  // and you are asking about the normal at start point
745  // then tangency is defined by first to second poles
746  DeriVector2 endpt(this->poles[1], derivparam);
747  DeriVector2 spt(this->poles[0], derivparam);
748 
749  DeriVector2 tg = endpt.subtr(spt);
750  ret = tg.rotate90ccw();
751  }

753  // and you are asking about the normal at end point
754  // then tangency is defined by last to last but one poles
755  DeriVector2 endpt(this->poles[poles.size() - 1], derivparam);
756  DeriVector2 spt(this->poles[poles.size() - 2], derivparam);
757 
758  DeriVector2 tg = endpt.subtr(spt);

763  ret = DeriVector2();
764  }
765  }
766  else {
767  // either periodic or abnormal endpoint multiplicity, we have no clue so currently
768  // unsupported
769  ret = DeriVector2();

776 DeriVector2 BSpline::CalculateNormal(const double* param, const double* derivparam) const
777 {
778  // TODO: is there any advantage in making this a `static`?
779  size_t startpole = 0;
780  for (size_t j = 1; j < mult.size() && *(knots[j]) <= *param; ++j) {
781  startpole += mult[j];
782  }
783  if (!periodic && startpole >= poles.size()) {
784  startpole = poles.size() - degree - 1;
785  }
786 
787  auto polexat = [&](size_t i) {
788  return poles[(startpole + i) % poles.size()].x;
789  };
790  auto poleyat = [&](size_t i) {
791  return poles[(startpole + i) % poles.size()].y;
792  };
793  auto weightat = [&](size_t i) {
794  return weights[(startpole + i) % weights.size()];
795  };
796 
797  double xsum, xslopesum;
798  double ysum, yslopesum;
799  double wsum, wslopesum;
800 
801  valueHomogenous(*param, &xsum, &ysum, &wsum, &xslopesum, &yslopesum, &wslopesum);
802 
803  // Tangent vector
804  // This should in principle be identical to error gradient wrt curve parameter in
805  // point-on-object
806  DeriVector2 result(wsum * xslopesum - wslopesum * xsum, wsum * yslopesum - wslopesum * ysum);
807 
808  size_t numpoints = degree + 1;
809 
810  // FIXME: Find an appropriate name for this method
811  auto doSomething = [&](size_t i, double& factor, double& slopefactor) {
812  VEC_D d(numpoints);
813  d[i] = 1;
814  factor = BSpline::splineValue(*param, startpole + degree, degree, d, flattenedknots);
815  VEC_D sd(numpoints - 1);
816  if (i > 0) {
817  sd[i - 1] =
818  1.0 / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
819  }
820  if (i < numpoints - 1) {
821  sd[i] = -1.0
822  / (flattenedknots[startpole + i + 1 + degree] - flattenedknots[startpole + i + 1]);
823  }
824  slopefactor =
825  BSpline::splineValue(*param, startpole + degree, degree - 1, sd, flattenedknots);
826  };
827 
828  // get dx, dy of the normal as well
829  for (size_t i = 0; i < numpoints; ++i) {
830  if (derivparam == polexat(i)) {
831  double factor, slopefactor;
832  doSomething(i, factor, slopefactor);
833  result.dx = *weightat(i) * (wsum * slopefactor - wslopesum * factor);
834  break;
835  }
836  if (derivparam == poleyat(i)) {
837  double factor, slopefactor;
838  doSomething(i, factor, slopefactor);
839  result.dy = *weightat(i) * (wsum * slopefactor - wslopesum * factor);
840  break;
841  }
842  if (derivparam == weightat(i)) {
843  double factor, slopefactor;
844  doSomething(i, factor, slopefactor);
845  result.dx = degree
846  * (factor * (xslopesum - wslopesum * (*polexat(i)))
847  - slopefactor * (xsum - wsum * (*polexat(i))));
848  result.dy = degree
849  * (factor * (yslopesum - wslopesum * (*poleyat(i)))
850  - slopefactor * (ysum - wsum * (*poleyat(i))));
851  break;
852  }
853  }
854 
855  // the curve parameter being used by the constraint is not known to the geometry (there can be
856  // many tangent constraints on the same curve after all). Assume that this is the param
857  // provided.
858  if (derivparam != param) {
859  return result.rotate90ccw();
860  }
861 
862  // derivparam == param now. Done this way just to reduce "cognitive complexity".
863  VEC_D sd(numpoints - 1), ssd(numpoints - 2);
864  for (size_t i = 1; i < numpoints; ++i) {
865  sd[i - 1] = (*weightat(i) - *weightat(i - 1))
866  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
867  }
868  for (size_t i = 1; i < numpoints - 1; ++i) {
869  ssd[i - 1] = (sd[i] - sd[i - 1])
870  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
871  }
872  double wslopeslopesum = degree * (degree - 1)
873  * BSpline::splineValue(*param, startpole + degree, degree - 2, ssd, flattenedknots);
874 
875  for (size_t i = 1; i < numpoints; ++i) {
876  sd[i - 1] = (*polexat(i) * *weightat(i) - *polexat(i - 1) * *weightat(i - 1))
877  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
878  }
879  for (size_t i = 1; i < numpoints - 1; ++i) {
880  ssd[i - 1] = (sd[i] - sd[i - 1])
881  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
882  }
883  double xslopeslopesum = degree * (degree - 1)
884  * BSpline::splineValue(*param, startpole + degree, degree - 2, ssd, flattenedknots);
885 
886  for (size_t i = 1; i < numpoints; ++i) {
887  sd[i - 1] = (*poleyat(i) * *weightat(i) - *poleyat(i - 1) * *weightat(i - 1))
888  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
889  }
890  for (size_t i = 1; i < numpoints - 1; ++i) {
891  ssd[i - 1] = (sd[i] - sd[i - 1])
892  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
893  }
894  double yslopeslopesum = degree * (degree - 1)
895  * BSpline::splineValue(*param, startpole + degree, degree - 2, ssd, flattenedknots);
896 
897  result.dx = wsum * xslopeslopesum - wslopeslopesum * xsum;
898  result.dy = wsum * yslopeslopesum - wslopeslopesum * ysum;
899 
900  return result.rotate90ccw();
901 }

903 DeriVector2 BSpline::Value(double u, double /*du*/, const double* /*derivparam*/) const
904 {
905  // TODO: is there any advantage in making this a `static`?
906  size_t startpole = 0;
907  for (size_t j = 1; j < mult.size() && *(knots[j]) <= u; ++j) {
908  startpole += mult[j];
909  }
910  if (!periodic && startpole >= poles.size()) {
911  startpole = poles.size() - degree - 1;
912  }
913 
914  // double xsum = 0., xslopesum = 0.;
915  // double ysum = 0., yslopesum = 0.;
916  // double wsum = 0., wslopesum = 0.;
917 
918  auto polexat = [&](size_t i) {
919  return poles[(startpole + i) % poles.size()].x;
920  };
921  auto poleyat = [&](size_t i) {
922  return poles[(startpole + i) % poles.size()].y;
923  };
924  auto weightat = [&](size_t i) {
925  return weights[(startpole + i) % weights.size()];
926  };
927 
928  size_t numpoints = degree + 1;
929  // Tangent vector
930  // This should in principle be identical to error gradient wrt curve parameter in
931  // point-on-object
932  VEC_D d(numpoints);
933  for (size_t i = 0; i < numpoints; ++i) {
934  d[i] = *polexat(i) * *weightat(i);
935  }
936  double xsum = BSpline::splineValue(u, startpole + degree, degree, d, flattenedknots);
937  for (size_t i = 0; i < numpoints; ++i) {
938  d[i] = *poleyat(i) * *weightat(i);
939  }
940  double ysum = BSpline::splineValue(u, startpole + degree, degree, d, flattenedknots);
941  for (size_t i = 0; i < numpoints; ++i) {
942  d[i] = *weightat(i);
943  }
944  double wsum = BSpline::splineValue(u, startpole + degree, degree, d, flattenedknots);
945 
946  d.resize(numpoints - 1);
947  for (size_t i = 1; i < numpoints; ++i) {
948  d[i - 1] = (*polexat(i) * *weightat(i) - *polexat(i - 1) * *weightat(i - 1))
949  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
950  }
951  double xslopesum =
952  degree * BSpline::splineValue(u, startpole + degree, degree - 1, d, flattenedknots);
953  for (size_t i = 1; i < numpoints; ++i) {
954  d[i - 1] = (*poleyat(i) * *weightat(i) - *poleyat(i - 1) * *weightat(i - 1))
955  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
956  }
957  double yslopesum =
958  degree * BSpline::splineValue(u, startpole + degree, degree - 1, d, flattenedknots);
959  for (size_t i = 1; i < numpoints; ++i) {
960  d[i - 1] = (*weightat(i) - *weightat(i - 1))
961  / (flattenedknots[startpole + i + degree] - flattenedknots[startpole + i]);
962  }
963  double wslopesum =
964  degree * BSpline::splineValue(u, startpole + degree, degree - 1, d, flattenedknots);
965 
966  // place holder
967  DeriVector2 ret = DeriVector2(xsum / wsum,
968  ysum / wsum,
969  (wsum * xslopesum - wslopesum * xsum) / wsum / wsum,
970  (wsum * yslopesum - wslopesum * ysum) / wsum / wsum);
971 
972  return ret;
973 }

1099 double BSpline::getLinCombFactor(double x, size_t k, size_t i, unsigned int p)
1100 {
1101  // Adapted to C++ from the python implementation in the Wikipedia page for de Boor algorithm
1102  // https://en.wikipedia.org/wiki/De_Boor%27s_algorithm.
1103 
1104  // FIXME: This should probably be guaranteed by now, and done somewhere else
1105  // To elaborate: `flattenedknots` should be set up as soon as `knots`
1106  // and `mult` have been defined after creating the B-spline.
1107  // However, in the future the values of `knots` could go into the solver
1108  // as well, when alternatives may be needed to keep `flattenedknots` updated.
1109  // Slightly more detailed discussion here:
1110  // https://github.com/FreeCAD/FreeCAD/pull/7484#discussion_r1020858392
1111  if (flattenedknots.empty()) {
1112  setupFlattenedKnots();
1113  }
1114 
1115  std::vector d(p + 1, 0.0);
1116  // Ensure this is within range
1117  int idxOfPole = static_cast<int>(i) + p - static_cast<int>(k);
1118  if (idxOfPole < 0 || idxOfPole > static_cast<int>(p)) {
1119  return 0.0;
1120  }
1121  d[idxOfPole] = 1.0;
1122 
1123  for (size_t r = 1; r < p + 1; ++r) {
1124  for (size_t j = p; j > r - 1; --j) {
1125  double alpha = (x - flattenedknots[j + k - p])
1126  / (flattenedknots[j + 1 + k - r] - flattenedknots[j + k - p]);
1127  d[j] = (1.0 - alpha) * d[j - 1] + alpha * d[j];
1128  }
1129  }
1130 
1131  return d[p];
1132 }

1147 void BSpline::setupFlattenedKnots()
1148 {
1149  flattenedknots.clear();
1150 
1151  for (size_t i = 0; i < knots.size(); ++i) {
1152  flattenedknots.insert(flattenedknots.end(), mult[i], *knots[i]);
1153  }
1154 
1155  // Adjust for periodic: see OCC documentation for explanation
1156  if (periodic) {
1157  double period = *knots.back() - *knots.front();
1158  int c = degree + 1 - mult[0]; // number of knots to pad
1159 
1160  // Add capacity so that iterators remain valid
1161  flattenedknots.reserve(flattenedknots.size() + 2 * c);
1162 
1163  // get iterators first for convenience
1164  auto frontStart = flattenedknots.end() - mult.back() - c;
1165  auto frontEnd = flattenedknots.end() - mult.back();
1166  auto backStart = flattenedknots.begin() + mult.front();
1167  auto backEnd = flattenedknots.begin() + mult.front() + c;
1168 
1169  // creating new vectors because above iterators can become invalidated upon insert
1170  std::vector<double> frontNew(frontStart, frontEnd);
1171  std::vector<double> backNew(backStart, backEnd);
1172 
1173  flattenedknots.insert(flattenedknots.end(), backNew.begin(), backNew.end());
1174  flattenedknots.insert(flattenedknots.begin(), frontNew.begin(), frontNew.end());
1175 
1176  for (int i = 0; i < c; ++i) {
1177  *(flattenedknots.begin() + i) -= period;
1178  *(flattenedknots.end() - 1 - i) += period;
1179  }
1180  }
