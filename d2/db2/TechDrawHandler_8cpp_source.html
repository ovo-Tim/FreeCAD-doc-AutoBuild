59 void TechDrawHandler::activate(QGVPage* vp)
60 {
61  auto* mdi = dynamic_cast<MDIViewPage*>(Gui::getMainWindow()->activeWindow());
62  if (!mdi) {
63  return;
64  }
65  mdi->enableContextualMenu(false);
66

68 
69  if (!Gui::ToolHandler::activate()) {

72 }

74 void TechDrawHandler::deactivate()
75 {
76  Gui::ToolHandler::deactivate();
77 
78  // The context menu event of MDIViewPage comes after the tool is deactivated.
79  // So to prevent the menu from appearing when the tool is cleared by right mouse click
80  // we set a small timer.
81  QTimer::singleShot(100, []() { // 100 milliseconds delay
82  auto* mdi = dynamic_cast<MDIViewPage*>(Gui::getMainWindow()->activeWindow());
83  if (!mdi) {
84  return;
85  }
86  mdi->enableContextualMenu(true);
87  });
88 }
