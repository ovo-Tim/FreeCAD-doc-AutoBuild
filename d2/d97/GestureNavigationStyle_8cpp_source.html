23 /*
24  *A few notes on this style. (by DeepSOIC)
25  *
26  * In this style, LMB serves dual purpose. It is selecting objects, as well as
27  * spinning the view. The trick that enables it is to consume mouse events
28  * before move threshold is detected, and refire the events if the mouse was
29  * released but not moved.
30  *
31  * This navigation style does not exactly follow the structure of other
32  * navigation styles, it does not fill many of the global variables defined in
33  * NavigationStyle.
34  *
35  * It uses a statemachine based on boost::statechart to simplify differences in
36  * event handling depending on mode.
37  *
38  * Dealing with touchscreen gestures with Qt5 on Windows is a pain in the arse.
39  *
40  * For pinch gesture, Qt5 starts generating mouse input as soon as the first
41  * finger lands on the screen. As the second finger touches, gesture events
42  * begin to arrive. But in the process, more synthetic mouse input keeps
43  * coming, sometimes including RMB press. This mouse input is usually properly
44  * terminated by LMB and RMB release events, but they don't always come;
45  * proofing the logic against this inconsistency was quite a challenge.
46  *
47  * Tap-and-hold was yet another terrible beast. Again, as soon as the finger
48  * touches the screen, LMB press comes in. Then, after the finger is released,
49  * RMB press comes. This one is usually complemented by all release events.
50  * However, with tap-hold-move-release sequence, RMB release event does not
51  * arrive.
52  *
53  * So, to avoid entering Tilt mode, the style implements its own tap-and-hold
54  * detection, and a special Pan state for the state machine - StickyPanState.
55  *
56  * This style wasn't tested with spacemouse during development (I don't have one).
57  *
58  * See also GestureNavigationStyle-state-machine-diagram.docx for a crude
59  * diagram of the state machine.
60  *
61  */

236 class NS::IdleState : public sc::state<NS::IdleState, NS::NaviMachine>
237 {
238 public:
239  using reactions = sc::custom_reaction<NS::Event>;
240 
241  explicit IdleState(my_context ctx):my_base(ctx)
242  {
243  auto &ns = this->outermost_context().ns;
244  ns.setViewingMode(NavigationStyle::IDLE);
245  if (ns.logging)
246  Base::Console().Log(" -> IdleState\n");
247  }
248  virtual ~IdleState() = default;
249 
250  sc::result react(const NS::Event& ev){
251  auto &ns = this->outermost_context().ns;
252 
253  auto posn = ns.normalizePixelPos(ev.inventor_event->getPosition());
254 
255  //special handling for some special states of viewer
256  switch (ns.getViewingMode()) {
257  case NavigationStyle::SEEK_WAIT_MODE:{
258  if (ev.isPress(1)) {
259  ns.seekToPoint(ev.inventor_event->getPosition()); // implicitly calls interactiveCountInc()
260  ns.setViewingMode(NavigationStyle::SEEK_MODE);
261  ev.flags->processed = true;
262  return transit<NS::AwaitingReleaseState>();
263  }
264  } ; //not end of SEEK_WAIT_MODE. Fall through by design!!!
265  /* FALLTHRU */
266  case NavigationStyle::SPINNING:
267  case NavigationStyle::SEEK_MODE: {
268  //animation modes
269  if (!ev.flags->processed) {
270  if (ev.isMouseButtonEvent()){
271  ev.flags->processed = true;
272  return transit<NS::AwaitingReleaseState>();
273  } else if (ev.isGestureEvent()
274  || ev.isKeyboardEvent()
275  || ev.isMotion3Event())
276  ns.setViewingMode(NavigationStyle::IDLE);
277  }
278  } break; //end of animation modes
279  case BOXZOOM:
280  return forward_event();
281  }
282 
283  //testing for draggers
284  if(ev.isPress(1) && ev.mbstate() == 0x100){
285  if (ns.isDraggerUnderCursor(ev.inventor_event->getPosition()))
286  return transit<NS::InteractState>();
287  }
288 
289  //left and right clicks - special handling, postpone the events
290  if ( (ev.isPress(1) && ev.mbstate() == 0x100)
291  || (ev.isPress(2) && ev.mbstate() == 0x001)){
292  ns.postponedEvents.post(ev);
293  ev.flags->processed = true;
294  return transit<NS::AwaitingMoveState>();
295  }
296 
297  //MMB click
298  if(ev.isPress(3) && ev.mbstate() == 0x010){
299  ev.flags->processed = true;
300  ns.setupPanningPlane(ns.viewer->getCamera());
301  ns.lookAtPoint(ev.inventor_event->getPosition());
302  return transit<NS::AwaitingReleaseState>();
303  }
304 
305  //touchscreen gestures
306  if(ev.isGestureActive()){
307  ev.flags->processed = true;
308  return transit<NS::GestureState>();
309  }
310 
311  //keyboard
312  if(ev.isKeyboardEvent()){
313  auto const &kbev = ev.asKeyboardEvent();
314  ev.flags->processed = true;
315  bool press = (kbev->getState() == SoKeyboardEvent::DOWN);
316  switch (kbev->getKey()) {
317  case SoKeyboardEvent::H:
318  if (!press) {
319  ns.setupPanningPlane(ns.viewer->getCamera());
320  ns.lookAtPoint(kbev->getPosition());
321  }
322  break;
323  case SoKeyboardEvent::PAGE_UP:
324  if(!press){
325  ns.doZoom(ns.viewer->getSoRenderManager()->getCamera(), ns.getDelta(), posn);
326  }
327  break;
328  case SoKeyboardEvent::PAGE_DOWN:
329  if(!press){
330  ns.doZoom(ns.viewer->getSoRenderManager()->getCamera(), -ns.getDelta(), posn);
331  }
332  break;
333  default:
334  ev.flags->processed = false;
335  }
336  }
337 
338  return forward_event();
339  }

342 class NS::AwaitingMoveState : public sc::state<NS::AwaitingMoveState, NS::NaviMachine>
343 {
344 public:
345  using reactions = sc::custom_reaction<NS::Event>;
346 
347 private:
348  SbVec2s base_pos;
349  SbTime since; //the time of mouse-down event
350  int hold_timeout; //in milliseconds
351 
352 public:
353  explicit AwaitingMoveState(my_context ctx):my_base(ctx)
354  {
355  auto &ns = this->outermost_context().ns;
356  if (ns.logging)
357  Base::Console().Log(" -> AwaitingMoveState\n");
358  ns.setViewingMode(NavigationStyle::IDLE);
359  this->base_pos = static_cast<const NS::Event*>(this->triggering_event())->inventor_event->getPosition();
360  this->since = static_cast<const NS::Event*>(this->triggering_event())->inventor_event->getTime();
361 
362  ns.mouseMoveThreshold = App::GetApplication().GetParameterGroupByPath
363  ("User parameter:BaseApp/Preferences/View")->GetInt("GestureMoveThreshold", ns.mouseMoveThreshold);
364 
365  this->hold_timeout = int(double(QTapAndHoldGesture::timeout()) *0.9);
366  this->hold_timeout = App::GetApplication().GetParameterGroupByPath
367  ("User parameter:BaseApp/Preferences/View")->GetInt("GestureTapHoldTimeout", this->hold_timeout);
368  if (this->hold_timeout == 0)
369  this->hold_timeout = 650; //a fail-safe
370  QTapAndHoldGesture::setTimeout(int(double(this->hold_timeout)/0.9));
371  // Why *0.9? We need tap-and-hold detection to be slightly faster than
372  // Qt's one, to filter out spurious events. It'd be better to disable
373  // Tap-and-hold altogether, but my attempts to use ungrabGesture and
374  // unregisterRecognizer routines failed to affect anything.
375 
376  }
377  virtual ~AwaitingMoveState(){
378  //always clear postponed events when leaving this state.
379  this->outermost_context().ns.postponedEvents.discardAll();
380  }
381 
382  sc::result react(const NS::Event& ev){
383  auto &ns = this->outermost_context().ns;
384 
386  auto refire = [&]{
387  ns.postponedEvents.forwardAll();
388  ev.flags->processed = ns.processSoEvent_bypass(ev.inventor_event);
389  ev.flags->propagated = true;
390  };
391 
392  bool long_click = (ev.inventor_event->getTime() - this->since).getValue()*1000.0 >= this->hold_timeout;
393 
394  //this state consumes all mouse events.
395  ev.flags->processed = ev.isMouseButtonEvent() || ev.isLocation2Event();
396 
397  //right-click
398  if (ev.isRelease(2)
399  && ev.mbstate() == 0
400  && !ns.viewer->isEditing()
401  && ns.isPopupMenuEnabled()){
402  ns.openPopupMenu(ev.inventor_event->getPosition());
403  return transit<NS::IdleState>();
404  }
405 
406  //roll gestures
407  //direction is determined at the moment the second button is pressed.
408  if (ev.mbstate() == 0x101){
409  if (ev.isPress(1))
410  ns.rollDir = -1;
411  if (ev.isPress(2))
412  ns.rollDir = +1;
413  }
414  //The roll gesture is fired when one of the two buttons in then released.
415  if ( (ev.isRelease(1) && ev.mbstate() == 0x001)
416  || (ev.isRelease(2) && ev.mbstate() == 0x100) ){
417  ns.onRollGesture(ns.rollDir);
418  return transit<NS::AwaitingReleaseState>();
419  }
420 
421  if(ev.isMouseButtonEvent() && ev.mbstate() == 0){
422  //all buttons released
423  if (long_click){
424  //emulate RMB-click
425  ns.openPopupMenu(ev.inventor_event->getPosition());
426  return transit<NS::IdleState>();
427  } else {
428  //refire all events && return to idle state
429  ns.setViewingMode(NavigationStyle::SELECTION);
430  refire();
431  return transit<NS::IdleState>();
432  }
433  }
434  if (ev.isPress(3)){
435  //mmb pressed, exit navigation
436  refire();
437  return transit<NS::IdleState>();
438  }
439  if (ev.isMouseButtonEvent() /* and still not processed*/){
440  ns.postponedEvents.post(ev);
441  }
442  if(ev.isLocation2Event()){
443  auto mv = ev.inventor_event->getPosition() - this->base_pos;
444  if(SbVec2f(mv).length() > ns.mouseMoveThreshold)
445  {
446  //mouse moved while buttons are held. decide how to navigate...
447  switch(ev.mbstate()){
448  case 0x100:{
449  if (!long_click) {
450  bool alt = ev.modifiers & NS::Event::ALTDOWN;
451  bool allowSpin = alt == ns.is2DViewing();
452  if(allowSpin)
453  return transit<NS::RotateState>();
454  else {
455  refire();
456  return transit<NS::IdleState>();
457  }
458  } else {
459  return transit<NS::StickyPanState>();
460  }
461  }break;
462  case 0x001:
463  return transit<NS::PanState>();
464  break;
465  case (0x101):
466  return transit<NS::TiltState>();
467  break;
468  default:
469  //MMB was held? refire all events.
470  refire();
471  return transit<NS::IdleState>();
472  }
473  }
474  }
475  if(ev.isGestureActive()){
476  ev.flags->processed = true;
477  return transit<NS::GestureState>();
478  }
479  return forward_event();
480  }
481 };

870 SbBool GestureNavigationStyle::processSoEvent(const SoEvent* const ev)
871 {
872  // Events when in "ready-to-seek" mode are ignored, except those
873  // which influence the seek mode itself -- these are handled further
874  // up the inheritance hierarchy.
875  if (this->isSeekMode()) {
876  return superclass::processSoEvent(ev);
877  }
878  // Switch off viewing mode (Bug #0000911)
879  if (!this->isSeekMode()&& !this->isAnimating() && this->isViewing() )
880  this->setViewing(false); // by default disable viewing mode to render the scene
881 
882  NS::Event smev;
883  smev.inventor_event = ev;
884 
885  //mode-independent spaceball/joystick handling
886  if (ev->isOfType(SoMotion3Event::getClassTypeId())){
887  smev.flags->processed = true;
888  this->processMotionEvent(static_cast<const SoMotion3Event*>(ev));
889  return true;
890  }
891 
892  // give the nodes in the foreground root the chance to handle events (e.g color bar)
893  if (!viewer->isEditing()) {
894  bool processed = handleEventInForeground(ev);
895  if (processed)
896  return true;
897  }
898 
899  if ( (smev.isRelease(1) && !this->button1down)
900  || (smev.isRelease(2) && !this->button2down)
901  || (smev.isRelease(3) && !this->button3down)) {
902  //a button release event cane, but we didn't see the corresponding down
903  //event. Discard it. This discarding is relied upon in some hacks to
904  //overcome buggy synthetic mouse input coming from Qt when doing
905  //touchecteen gestures.
906  return true;
907  }
908 
909 
910  if (smev.isMouseButtonEvent()) {
911  const int button = smev.asMouseButtonEvent()->getButton();
912  const SbBool press //the button was pressed (if false -> released)
913  = smev.asMouseButtonEvent()->getState() == SoButtonEvent::DOWN ? true : false;
914  switch (button) {
915  case SoMouseButtonEvent::BUTTON1:
916  this->button1down = press;
917  break;
918  case SoMouseButtonEvent::BUTTON2:
919  this->button2down = press;
920  break;
921  case SoMouseButtonEvent::BUTTON3:
922  this->button3down = press;
923  break;
924  //whatever else, we don't track
925  }
926  }
927 
928  syncModifierKeys(ev);
929 
930  smev.modifiers =
931  (this->button1down ? NS::Event::BUTTON1DOWN : 0) |
932  (this->button2down ? NS::Event::BUTTON2DOWN : 0) |
933  (this->button3down ? NS::Event::BUTTON3DOWN : 0) |
934  (this->ctrldown ? NS::Event::CTRLDOWN : 0) |
935  (this->shiftdown ? NS::Event::SHIFTDOWN : 0) |
936  (this->altdown ? NS::Event::ALTDOWN : 0);
937 
938 #ifdef FC_OS_MACOSX
939  // On Mac, Qt gesture events seem to be broken. At least that's what event
940  // logs from @chrisb tell me. So, for until a developer on a mac gets here to
941  // make gestures work, I disable them. --DeepSOIC
942 
943  if (smev.isGestureEvent())
944  return superclass::processSoEvent(ev);
945 #endif
946 
947 
948  if (! smev.flags->processed)
949  this->naviMachine->processEvent(smev);
950  if(! smev.flags->propagated && ! smev.flags->processed)
951  return superclass::processSoEvent(ev);
952  else
953  return smev.flags->processed;
954 }
