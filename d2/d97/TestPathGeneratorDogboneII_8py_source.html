38 def createKinks(maneuver):
39  k = []
40  moves = maneuver.getMoves()
41  if moves:
42  move0 = moves[0]
43  prev = move0
44  for m in moves[1:]:
45  k.append(dogboneII.Kink(prev, m))
46  prev = m
47  if Path.Geom.pointsCoincide(move0.positionBegin(), prev.positionEnd()):
48  k.append(dogboneII.Kink(prev, move0))
49  return k

52 def findDogboneKinks(maneuver, threshold):
53  if threshold > 0:
54  return [k for k in createKinks(maneuver) if k.deflection() > threshold]
55  if threshold < 0:
56  return [k for k in createKinks(maneuver) if k.deflection() < threshold]
57  return createKinks(maneuver)
