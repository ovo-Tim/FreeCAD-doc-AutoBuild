297 LINMATH_H_FUNC void mat4x4_invert(mat4x4 T, mat4x4 const M)
298 {
299  float s[6];
300  float c[6];
301  s[0] = M[0][0] * M[1][1] - M[1][0] * M[0][1];
302  s[1] = M[0][0] * M[1][2] - M[1][0] * M[0][2];
303  s[2] = M[0][0] * M[1][3] - M[1][0] * M[0][3];
304  s[3] = M[0][1] * M[1][2] - M[1][1] * M[0][2];
305  s[4] = M[0][1] * M[1][3] - M[1][1] * M[0][3];
306  s[5] = M[0][2] * M[1][3] - M[1][2] * M[0][3];
307 
308  c[0] = M[2][0] * M[3][1] - M[3][0] * M[2][1];
309  c[1] = M[2][0] * M[3][2] - M[3][0] * M[2][2];
310  c[2] = M[2][0] * M[3][3] - M[3][0] * M[2][3];
311  c[3] = M[2][1] * M[3][2] - M[3][1] * M[2][2];
312  c[4] = M[2][1] * M[3][3] - M[3][1] * M[2][3];
313  c[5] = M[2][2] * M[3][3] - M[3][2] * M[2][3];
314 
315  /* Assumes it is invertible */
316  float idet =
317  1.0f / (s[0] * c[5] - s[1] * c[4] + s[2] * c[3] + s[3] * c[2] - s[4] * c[1] + s[5] * c[0]);
318 
319  T[0][0] = (M[1][1] * c[5] - M[1][2] * c[4] + M[1][3] * c[3]) * idet;
320  T[0][1] = (-M[0][1] * c[5] + M[0][2] * c[4] - M[0][3] * c[3]) * idet;
321  T[0][2] = (M[3][1] * s[5] - M[3][2] * s[4] + M[3][3] * s[3]) * idet;
322  T[0][3] = (-M[2][1] * s[5] + M[2][2] * s[4] - M[2][3] * s[3]) * idet;
323 
324  T[1][0] = (-M[1][0] * c[5] + M[1][2] * c[2] - M[1][3] * c[1]) * idet;
325  T[1][1] = (M[0][0] * c[5] - M[0][2] * c[2] + M[0][3] * c[1]) * idet;
326  T[1][2] = (-M[3][0] * s[5] + M[3][2] * s[2] - M[3][3] * s[1]) * idet;
327  T[1][3] = (M[2][0] * s[5] - M[2][2] * s[2] + M[2][3] * s[1]) * idet;
328 
329  T[2][0] = (M[1][0] * c[4] - M[1][1] * c[2] + M[1][3] * c[0]) * idet;
330  T[2][1] = (-M[0][0] * c[4] + M[0][1] * c[2] - M[0][3] * c[0]) * idet;
331  T[2][2] = (M[3][0] * s[4] - M[3][1] * s[2] + M[3][3] * s[0]) * idet;
332  T[2][3] = (-M[2][0] * s[4] + M[2][1] * s[2] - M[2][3] * s[0]) * idet;
333 
334  T[3][0] = (-M[1][0] * c[3] + M[1][1] * c[1] - M[1][2] * c[0]) * idet;
335  T[3][1] = (M[0][0] * c[3] - M[0][1] * c[1] + M[0][2] * c[0]) * idet;
336  T[3][2] = (-M[3][0] * s[3] + M[3][1] * s[1] - M[3][2] * s[0]) * idet;
337  T[3][3] = (M[2][0] * s[3] - M[2][1] * s[1] + M[2][2] * s[0]) * idet;
338 }

362 LINMATH_H_FUNC void mat4x4_frustum(mat4x4 M, float l, float r, float b, float t, float n, float f)
363 {
364  M[0][0] = 2.f * n / (r - l);
365  M[0][1] = M[0][2] = M[0][3] = 0.f;
366 
367  M[1][1] = 2.f * n / (t - b);
368  M[1][0] = M[1][2] = M[1][3] = 0.f;
369 
370  M[2][0] = (r + l) / (r - l);
371  M[2][1] = (t + b) / (t - b);
372  M[2][2] = -(f + n) / (f - n);
373  M[2][3] = -1.f;
374 
375  M[3][2] = -2.f * (f * n) / (f - n);
376  M[3][0] = M[3][1] = M[3][3] = 0.f;
377 }

504 LINMATH_H_FUNC void quat_mul_vec3(vec3 r, quat const q, vec3 const v)
505 {
506  /*
507  * Method by Fabian 'ryg' Giessen (of Farbrausch)
508  t = 2 * cross(q.xyz, v)
509  v' = v + q.w * t + cross(q.xyz, t)
510  */
511  vec3 t;
512  vec3 q_xyz = {q[0], q[1], q[2]};
513  vec3 u = {q[0], q[1], q[2]};
514 
515  vec3_mul_cross(t, q_xyz, v);
516  vec3_scale(t, t, 2);
517 
518  vec3_mul_cross(u, q_xyz, t);
519  vec3_scale(t, t, q[3]);
520 
521  vec3_add(r, v, t);
522  vec3_add(r, r, u);
523 }
524 LINMATH_H_FUNC void mat4x4_from_quat(mat4x4 M, quat const q)
525 {
526  float a = q[3];
527  float b = q[0];
528  float c = q[1];
529  float d = q[2];
530  float a2 = a * a;
531  float b2 = b * b;
532  float c2 = c * c;
533  float d2 = d * d;
534 
535  M[0][0] = a2 + b2 - c2 - d2;
536  M[0][1] = 2.f * (b * c + a * d);
537  M[0][2] = 2.f * (b * d - a * c);
538  M[0][3] = 0.f;
539 
540  M[1][0] = 2 * (b * c - a * d);
541  M[1][1] = a2 - b2 + c2 - d2;
542  M[1][2] = 2.f * (c * d + a * b);
543  M[1][3] = 0.f;
544 
545  M[2][0] = 2.f * (b * d + a * c);
546  M[2][1] = 2.f * (c * d - a * b);
547  M[2][2] = a2 - b2 - c2 + d2;
548  M[2][3] = 0.f;
549 
550  M[3][0] = M[3][1] = M[3][2] = 0.f;
551  M[3][3] = 1.f;
552 }

568 LINMATH_H_FUNC void quat_from_mat4x4(quat q, mat4x4 const M)
569 {
570  float r = 0.f;
571  int i;
572 
573  int perm[] = {0, 1, 2, 0, 1};
574  int* p = perm;
575 
576  for (i = 0; i < 3; i++) {
577  float m = M[i][i];
578  if (m < r) {
579  continue;
580  }
581  m = r;
582  p = &perm[i];
583  }
584 
585  r = sqrtf(1.f + M[p[0]][p[0]] - M[p[1]][p[1]] - M[p[2]][p[2]]);
586 
587  if (r < 1e-6) {
588  q[0] = 1.f;
589  q[1] = q[2] = q[3] = 0.f;
590  return;
591  }
592 
593  q[0] = r / 2.f;
594  q[1] = (M[p[0]][p[1]] - M[p[1]][p[0]]) / (2.f * r);
595  q[2] = (M[p[2]][p[0]] - M[p[0]][p[2]]) / (2.f * r);
596  q[3] = (M[p[2]][p[1]] - M[p[1]][p[2]]) / (2.f * r);
597 }
