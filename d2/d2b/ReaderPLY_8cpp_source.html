46 bool ReaderPLY::CheckHeader(std::istream& input) const
47 {
48  if (!input || input.bad()) {
49  return false;
50  }
51 
52  std::streambuf* buf = input.rdbuf();
53  if (!buf) {
54  return false;
55  }
56 
57  // read in the first three characters
58  std::array<char, 3> ply {};
59  input.read(ply.data(), ply.size());
60  input.ignore(1);
61  if (!input) {
62  return false;
63  }
64 
65  return ((ply[0] == 'p') && (ply[1] == 'l') && (ply[2] == 'y'));
66 }

68 bool ReaderPLY::ReadFormat(std::istream& str)
69 {
70  std::string format_string;
71  std::string version;
72  char space_format_string {};
73  char space_format_version {};
74  str >> space_format_string >> std::ws >> format_string >> space_format_version >> std::ws
75  >> version;
76 
77  // clang-format off
78  if (std::isspace(static_cast<unsigned char>(space_format_string)) == 0 ||
79  std::isspace(static_cast<unsigned char>(space_format_version)) == 0) {
80  return false;
81  }
82  // clang-format on
83 
84  if (format_string == "ascii") {
85  format = ascii;
86  }
87  else if (format_string == "binary_big_endian") {
88  format = binary_big_endian;
89  }
90  else if (format_string == "binary_little_endian") {
91  format = binary_little_endian;
92  }
93  else {
94  // wrong format version
95  return false;
96  }
97 
98  return (version == "1.0");
99 }

271 bool ReaderPLY::ReadHeader(std::istream& input)
272 {
273  std::string line;
274  std::string element;
275  while (std::getline(input, line)) {
276  std::istringstream str(line);
277  str.unsetf(std::ios_base::skipws);
278  str >> std::ws;
279  if (str.eof()) {
280  continue; // empty line
281  }
282  std::string kw;
283  str >> kw;
284  if (kw == "format") {
285  if (!ReadFormat(str)) {
286  return false;
287  }
288  }
289  else if (kw == "element") {
290  if (!ReadElement(str, element)) {
291  return false;
292  }
293  }
294  else if (kw == "property") {
295  if (!ReadProperty(str, element)) {
296  return false;
297  }
298  }
299  else if (kw == "end_header") {
300  break; // end of the header, now read the data
301  }
302  }
303 
304  return true;
305 }

408 bool ReaderPLY::ReadVertexes(std::istream& input)
409 {
410  std::string line;
411  for (std::size_t i = 0; i < v_count && std::getline(input, line); i++) {
412  std::istringstream str(line);
413  str.unsetf(std::ios_base::skipws);
414  str >> std::ws;
415 
416  // go through the vertex properties
417  PropertyArray prop_values {};
418  std::size_t count_props = vertex_props.size();
419  for (const auto& it : vertex_props) {
420  switch (it.second) {
421  case int8:
422  case int16:
423  case int32: {
424  int vt {};
425  str >> vt >> std::ws;
426  prop_values[it.first] = static_cast<float>(vt);
427  } break;
428  case uint8:
429  case uint16:
430  case uint32: {
431  unsigned int vt {};
432  str >> vt >> std::ws;
433  prop_values[it.first] = static_cast<float>(vt);
434  } break;
435  case float32: {
436  float vt {};
437  str >> vt >> std::ws;
438  prop_values[it.first] = vt;
439  } break;
440  case float64: {
441  double vt {};
442  str >> vt >> std::ws;
443  prop_values[it.first] = static_cast<float>(vt);
444  } break;
445  default:
446  return false;
447  }
448 
449  // does line contain all properties
450  if (--count_props > 0 && str.eof()) {
451  return false;
452  }
453  }
454 
455  addVertexProperty(prop_values);
456  }
457 
458  return true;
459 }
