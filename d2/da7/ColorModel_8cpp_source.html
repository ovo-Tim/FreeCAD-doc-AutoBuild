118 void ColorField::rebuild()
119 {
120  colorField.resize(ctColors);
121 
122  std::size_t usStep =
123  std::min<std::size_t>(ctColors / (colorModel.getCountColors() - 1), ctColors - 1);
124  std::size_t usInd1 = 0;
125  std::size_t usInd2 = usStep;
126  for (std::size_t i = 0; i < (colorModel.getCountColors() - 1); i++) {
127  interpolate(colorModel.colors[i], usInd1, colorModel.colors[i + 1], usInd2);
128  usInd1 = usInd2;
129  if ((i + 1) == (colorModel.getCountColors() - 2)) {
130  usInd2 = ctColors - 1;
131  }
132  else {
133  usInd2 += usStep;
134  }
135  }
136 
137  fAscent = float(ctColors) / (fMax - fMin);
138  fConstant = -fAscent * fMin;
139 }

141 // fuellt das Array von Farbe 1, Index 1 bis Farbe 2, Index 2
142 void ColorField::interpolate(Color clCol1, std::size_t usInd1, Color clCol2, std::size_t usInd2)
143 {
144  float fStep = 1.0f, fLen = float(usInd2 - usInd1);
145 
146  colorField[usInd1] = clCol1;
147  colorField[usInd2] = clCol2;
148 
149  float fR = (float(clCol2.r) - float(clCol1.r)) / fLen;
150  float fG = (float(clCol2.g) - float(clCol1.g)) / fLen;
151  float fB = (float(clCol2.b) - float(clCol1.b)) / fLen;
152 
153  for (std::size_t i = (usInd1 + 1); i < usInd2; i++) {
154  float ucR = clCol1.r + fR * fStep;
155  float ucG = clCol1.g + fG * fStep;
156  float ucB = clCol1.b + fB * fStep;
157  colorField[i] = Color(ucR, ucG, ucB);
158  fStep += 1.0f;
159  }
160 }

249 void ColorGradient::rebuild()
250 {
251  switch (profile.tStyle) {
252  case ColorBarStyle::FLOW: {
253  colorField1.set(currentModelPack.totalModel,
254  profile.fMin,
255  profile.fMax,
256  profile.ctColors);
257  break;
258  }
259  case ColorBarStyle::ZERO_BASED: {
260  if ((profile.fMin < 0.0f) && (profile.fMax > 0.0f)) {
261  colorField1.set(currentModelPack.bottomModel,
262  profile.fMin,
263  0.0f,
264  profile.ctColors / 2);
265  colorField2.set(currentModelPack.topModel,
266  0.0f,
267  profile.fMax,
268  profile.ctColors / 2);
269  }
270  else if (profile.fMin >= 0.0f) {
271  colorField1.set(currentModelPack.topModel, 0.0f, profile.fMax, profile.ctColors);
272  }
273  else {
274  colorField1.set(currentModelPack.bottomModel, profile.fMin, 0.0f, profile.ctColors);
275  }
276  break;
277  }
278  }
279 }
