270 Handle(Geom_BSplineCurve) getSplineFromPolesAndKnots(struct SplineData& sd)
271 {
272  std::size_t numPoles = sd.control_points;
273  if (sd.controlx.size() > numPoles || sd.controly.size() > numPoles
274  || sd.controlz.size() > numPoles || sd.weight.size() > numPoles) {
275  return nullptr;
276  }
277 
278  // handle the poles
279  TColgp_Array1OfPnt occpoles(1, sd.control_points);
280  int index = 1;
281  for (auto coordinate : sd.controlx) {
282  occpoles(index++).SetX(coordinate);
283  }
284 
285  index = 1;
286  for (auto coordinate : sd.controly) {
287  occpoles(index++).SetY(coordinate);
288  }
289 
290  index = 1;
291  for (auto coordinate : sd.controlz) {
292  occpoles(index++).SetZ(coordinate);
293  }
294 
295  // handle knots and mults
296  std::set<double> unique;
297  unique.insert(sd.knot.begin(), sd.knot.end());
298 
299  int numKnots = int(unique.size());
300  TColStd_Array1OfInteger occmults(1, numKnots);
301  TColStd_Array1OfReal occknots(1, numKnots);
302  index = 1;
303  for (auto knot : unique) {
304  occknots(index) = knot;
305  occmults(index) = (int)std::count(sd.knot.begin(), sd.knot.end(), knot);
306  index++;
307  }
308 
309  // handle weights
310  TColStd_Array1OfReal occweights(1, sd.control_points);
311  if (sd.weight.size() == std::size_t(sd.control_points)) {
312  index = 1;
313  for (auto weight : sd.weight) {
314  occweights(index++) = weight;
315  }
316  }
317  else {
318  // non-rational
319  for (int i = occweights.Lower(); i <= occweights.Upper(); i++) {
320  occweights(i) = 1.0;
321  }
322  }
323 
324  Standard_Boolean periodic = sd.flag == 2;
325  Handle(Geom_BSplineCurve) geom =
326  new Geom_BSplineCurve(occpoles, occweights, occknots, occmults, sd.degree, periodic);
327  return geom;
328 }

330 Handle(Geom_BSplineCurve) getInterpolationSpline(struct SplineData& sd)
331 {
332  std::size_t numPoints = sd.fit_points;
333  if (sd.fitx.size() > numPoints || sd.fity.size() > numPoints || sd.fitz.size() > numPoints) {
334  return nullptr;
335  }
336 
337  // handle the poles
338  Handle(TColgp_HArray1OfPnt) fitpoints = new TColgp_HArray1OfPnt(1, sd.fit_points);
339  int index = 1;
340  for (auto coordinate : sd.fitx) {
341  fitpoints->ChangeValue(index++).SetX(coordinate);
342  }
343 
344  index = 1;
345  for (auto coordinate : sd.fity) {
346  fitpoints->ChangeValue(index++).SetY(coordinate);
347  }
348 
349  index = 1;
350  for (auto coordinate : sd.fitz) {
351  fitpoints->ChangeValue(index++).SetZ(coordinate);
352  }
353 
354  Standard_Boolean periodic = sd.flag == 2;
355  GeomAPI_Interpolate interp(fitpoints, periodic, Precision::Confusion());
356  interp.Perform();
357  return interp.Curve();
358 }

360 void ImpExpDxfRead::OnReadSpline(struct SplineData& sd)
361 {
362  // https://documentation.help/AutoCAD-DXF/WS1a9193826455f5ff18cb41610ec0a2e719-79e1.htm
363  // Flags:
364  // 1: Closed, 2: Periodic, 4: Rational, 8: Planar, 16: Linear
365 
366  try {
367  Handle(Geom_BSplineCurve) geom;
368  if (sd.control_points > 0) {
369  geom = getSplineFromPolesAndKnots(sd);
370  }
371  else if (sd.fit_points > 0) {
372  geom = getInterpolationSpline(sd);
373  }
374 
375  if (geom.IsNull()) {
376  throw Standard_Failure();
377  }
378 
379  Collector->AddObject(BRepBuilderAPI_MakeEdge(geom).Edge(), "Spline");
380  }
381  catch (const Standard_Failure&) {
382  Base::Console().Warning("ImpExpDxf - failed to create bspline\n");

459 void ImpExpDxfRead::ExpandInsert(const std::string& name,
460  const Base::Matrix4D& transform,
461  const Base::Vector3d& point,
462  double rotation,
463  const Base::Vector3d& scale)
464 {
465  if (Blocks.count(name) == 0) {
466  ImportError("Reference to undefined or external block '%s'\n", name);
467  return;
468  }
469  Block& block = Blocks.at(name);
470  // Apply the scaling, rotation, and move before the OCSEnttityTransform and place the result io
471  // BaseEntityTransform,
472  Base::Matrix4D localTransform;
473  localTransform.scale(scale.x, scale.y, scale.z);
474  localTransform.rotZ(rotation);
475  localTransform.move(point[0], point[1], point[2]);
476  localTransform = transform * localTransform;
477  CommonEntityAttributes mainAttributes = m_entityAttributes;
478  for (const auto& [attributes, shapes] : block.Shapes) {
479  // Put attributes into m_entityAttributes after using the latter to set byblock values in
480  // the former.
481  m_entityAttributes = attributes;
482  m_entityAttributes.ResolveByBlockAttributes(mainAttributes);
483 
484  for (const TopoDS_Shape& shape : shapes) {
485  // TODO???: See the comment in TopoShape::makeTransform regarding calling
486  // Moved(identityTransform) on the new shape
487  Collector->AddObject(
488  BRepBuilderAPI_Transform(shape,
489  Part::TopoShape::convert(localTransform),
490  Standard_True)
491  .Shape(),
492  "InsertPart"); // TODO: The collection should contain the nameBase to use
493  }
494  }
495  for (const auto& [attributes, featureBuilders] : block.FeatureBuildersList) {
496  // Put attributes into m_entityAttributes after using the latter to set byblock values in
497  // the former.
498  m_entityAttributes = attributes;
499  m_entityAttributes.ResolveByBlockAttributes(mainAttributes);
500 
501  for (const FeaturePythonBuilder& featureBuilder : featureBuilders) {
502  // TODO: Any non-identity transform from the original entity record needs to be applied
503  // before OCSEntityTransform (which includes this INSERT's transform followed by the
504  // transform for the INSERT's context (i.e. from an outeer INSERT)
505  // TODO: Perhaps pass a prefix ("Insert") to the builder to make the object name so
506  // Draft objects in a block get named similarly to Shapes.
507  App::FeaturePython* feature = featureBuilder(localTransform);
508  if (feature != nullptr) {
509  // Note that the featureBuilder has already placed this object in the drawing as a
510  // top-level object, so we don't have to add them but we must place it in its layer
511  // and set its gui styles
512  MoveToLayer(feature);
513  ApplyGuiStyles(feature);
514  }
515  }
516  }
517  for (const auto& [attributes, inserts] : block.Inserts) {
518  // Put attributes into m_entityAttributes after using the latter to set byblock values in
519  // the former.
520  m_entityAttributes = attributes;
521  m_entityAttributes.ResolveByBlockAttributes(mainAttributes);
522 
523  for (const Block::Insert& insert : inserts) {
524  // TODO: Apply the OCSOrientationTransform saved with the Insert statement to
525  // localTransform. (pass localTransform*insert.OCSDirectionTransform)
526  ExpandInsert(insert.Name, localTransform, insert.Point, insert.Rotation, insert.Scale);
527  }
528  }
529 }
