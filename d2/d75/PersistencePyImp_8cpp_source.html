60 PyObject* PersistencePy::dumpContent(PyObject* args, PyObject* kwds)
61 {
62  int compression = 3;
63  static const std::array<const char*, 2> kwds_def {"Compression", nullptr};
64  PyErr_Clear();
65  if (!Base::Wrapped_ParseTupleAndKeywords(args, kwds, "|i", kwds_def, &compression)) {
66  return nullptr;
67  }
68 
69  // setup the stream. the in flag is needed to make "read" work
70  std::stringstream stream(std::stringstream::out | std::stringstream::in
71  | std::stringstream::binary);
72  try {
73  getPersistencePtr()->dumpToStream(stream, compression);
74  }
75  catch (Base::NotImplementedError&) {
76  PyErr_SetString(PyExc_NotImplementedError,
77  "Dumping content of this object type is not implemented");
78  return nullptr;
79  }
80  catch (...) {
81  PyErr_SetString(PyExc_IOError, "Unable to parse content into binary representation");
82  return nullptr;
83  }
84 
85  // build the byte array with correct size
86  if (!stream.seekp(0, std::stringstream::end)) {
87  PyErr_SetString(PyExc_IOError, "Unable to find end of stream");
88  return nullptr;
89  }
90 
91  std::stringstream::pos_type offset = stream.tellp();
92  if (!stream.seekg(0, std::stringstream::beg)) {
93  PyErr_SetString(PyExc_IOError, "Unable to find begin of stream");
94  return nullptr;
95  }
96 
97  PyObject* ba = PyByteArray_FromStringAndSize(nullptr, offset);
98 
99  // use the buffer protocol to access the underlying array and write into it
100  Py_buffer buf = Py_buffer();
101  PyObject_GetBuffer(ba, &buf, PyBUF_WRITABLE);
102  try {
103  if (!stream.read((char*)buf.buf, offset)) {
104  PyErr_SetString(PyExc_IOError, "Error copying data into byte array");
105  return nullptr;
106  }
107  PyBuffer_Release(&buf);
108  }
109  catch (...) {
110  PyBuffer_Release(&buf);
111  PyErr_SetString(PyExc_IOError, "Error copying data into byte array");
112  return nullptr;
113  }
114 
115  return ba;
116 }

118 PyObject* PersistencePy::restoreContent(PyObject* args)
119 {
120  PyObject* buffer = nullptr;
121  if (!PyArg_ParseTuple(args, "O", &buffer)) {
122  return nullptr;
123  }
124 
125  // check if it really is a buffer
126  if (!PyObject_CheckBuffer(buffer)) {
127  PyErr_SetString(PyExc_TypeError, "Must be a buffer object");
128  return nullptr;
129  }
130 
131  Py_buffer buf;
132  if (PyObject_GetBuffer(buffer, &buf, PyBUF_SIMPLE) < 0) {
133  return nullptr;
134  }
135 
136  if (!PyBuffer_IsContiguous(&buf, 'C')) {
137  PyErr_SetString(PyExc_TypeError, "Buffer must be contiguous");
138  return nullptr;
139  }
140 
141  // check if it really is a buffer
142  try {
143  using Device = boost::iostreams::basic_array_source<char>;
144  boost::iostreams::stream<Device> stream((char*)buf.buf, buf.len);
145  getPersistencePtr()->restoreFromStream(stream);
146  }
147  catch (...) {
148  PyErr_SetString(PyExc_IOError, "Unable to restore content");
149  return nullptr;
150  }
151 
152  Py_Return;
153 }
