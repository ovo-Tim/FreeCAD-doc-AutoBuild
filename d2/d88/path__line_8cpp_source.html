48 Path_Line::Path_Line(const Frame& startpos,
49  const Frame& endpos,
50  RotationalInterpolation* _orient,
51  double _eqradius,
52  bool _aggregate ):
53  orient(_orient),
54  V_base_start(startpos.p),
55  V_base_end(endpos.p),
56  eqradius(_eqradius),
57  aggregate(_aggregate)
58  {
59  V_start_end = V_base_end - V_base_start;
60  double dist = V_start_end.Normalize();
61  orient->SetStartEnd(startpos.M,endpos.M);
62  double alpha = orient->Angle();
63 
64  // See what has the slowest eq. motion, and adapt
65  // the other to this slower motion
66  // use eqradius to transform between rot and transl.
67 
68  // Only modify if non zero (prevent division by zero)
69  if ( alpha != 0 && alpha*eqradius > dist) {
70  // rotational_interpolation is the limitation
71  pathlength = alpha*eqradius;
72  scalerot = 1/eqradius;
73  scalelin = dist/pathlength;
74  } else if ( dist != 0 ) {
75  // translation is the limitation
76  pathlength = dist;
77  scalerot = alpha/pathlength;
78  scalelin = 1;
79  } else {
80  // both were zero
81  pathlength = 0;
82  scalerot = 1;
83  scalelin = 1;
84  }
85  }

87 Path_Line::Path_Line(const Frame& startpos,
88  const Twist& starttwist,
89  RotationalInterpolation* _orient,
90  double _eqradius,
91  bool _aggregate ):
92  orient(_orient),
93  V_base_start(startpos.p),
94  V_base_end(startpos.p + starttwist.vel),
95  eqradius(_eqradius),
96  aggregate(_aggregate)
97  {
98  // startframe and starttwist are expressed in Wo.
99  // after 1 time unit, startframe has translated over starttwist.vel
100  // and rotated over starttwist.rot.Norm() (both vectors can be zero)
101  // Thus the frame on the path after 1 time unit is defined by
102  // startframe.Integrate(starttwist, 1);
103  V_start_end = V_base_end - V_base_start;
104  double dist = V_start_end.Normalize(); // distance traveled during 1 time unit
105  orient->SetStartEnd(startpos.M, (startpos*Frame( Rotation::Rot(starttwist.rot, starttwist.rot.Norm() ), starttwist.vel )).M);
106  double alpha = orient->Angle(); // rotation during 1 time unit
107 
108  // See what has the slowest eq. motion, and adapt
109  // the other to this slower motion
110  // use eqradius to transform between rot and transl.
111  // Only modify if non zero (prevent division by zero)
112  if ( alpha != 0 && alpha*eqradius > dist) {
113  // rotational_interpolation is the limitation
114  pathlength = alpha*eqradius;
115  scalerot = 1/eqradius;
116  scalelin = dist/pathlength;
117  } else if ( dist != 0 ) {
118  // translation is the limitation
119  pathlength = dist;
120  scalerot = alpha/pathlength;
121  scalelin = 1;
122  } else {
123  // both were zero
124  pathlength = 0;
125  scalerot = 1;
126  scalelin = 1;
127  }
128  }
