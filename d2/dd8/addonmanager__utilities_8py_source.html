80 def symlink(source, link_name):
81  """Creates a symlink of a file, if possible. Note that it fails on most modern Windows
82  installations"""
83 
84  if os.path.exists(link_name) or os.path.lexists(link_name):
85  pass
86  else:
87  os_symlink = getattr(os, "symlink", None)
88  if callable(os_symlink):
89  os_symlink(source, link_name)
90  else:
91  # NOTE: This does not work on most normal Windows 10 and later installations, unless
92  # developer mode is turned on. Make sure to catch any exception thrown and have a
93  # fallback plan.
94  csl = ctypes.windll.kernel32.CreateSymbolicLinkW
95  csl.argtypes = (ctypes.c_wchar_p, ctypes.c_wchar_p, ctypes.c_uint32)
96  csl.restype = ctypes.c_ubyte
97  flags = 1 if os.path.isdir(source) else 0
98  # set the SYMBOLIC_LINK_FLAG_ALLOW_UNPRIVILEGED_CREATE flag
99  # (see https://blogs.windows.com/buildingapps/2016/12/02/symlinks-windows-10)
100  flags += 2
101  if csl(link_name, source, flags) == 0:
102  raise ctypes.WinError()

282 def get_assigned_string_literal(line: str) -> Optional[str]:
283  """Look for a line of the form my_var = "A string literal" and return the string literal.
284  If the assignment is of a floating point value, that value is converted to a string
285  and returned. If neither is true, returns None."""
286 
287  string_search_regex = re.compile(r"\s*(['\"])(.*)\1")
288  _, _, after_equals = line.partition("=")
289  match = re.match(string_search_regex, after_equals)
290  if match:
291  return str(match.group(2))
292  if is_float(after_equals):
293  return str(after_equals).strip()
294  return None

297 def get_macro_version_from_file(filename: str) -> str:
298  """Get the version of the macro from a local macro file. Supports strings, ints, and floats,
299  as well as a reference to __date__"""
300 
301  date = ""
302  with open(filename, errors="ignore", encoding="utf-8") as f:
303  line_counter = 0
304  max_lines_to_scan = 200
305  while line_counter < max_lines_to_scan:
306  line_counter += 1
307  line = f.readline()
308  if not line: # EOF
309  break
310  if line.lower().startswith("__version__"):
311  match = get_assigned_string_literal(line)
312  if match:
313  return match
314  if "__date__" in line.lower():
315  # Don't do any real syntax checking, just assume the line is something
316  # like __version__ = __date__
317  if date:
318  return date
319  # pylint: disable=line-too-long,consider-using-f-string
320  fci.Console.PrintWarning(
321  translate(
322  "AddonsInstaller",
323  "Macro {} specified '__version__ = __date__' prior to setting a value for __date__".format(
324  filename
325  ),
326  )
327  )
328  elif line.lower().startswith("__date__"):
329  match = get_assigned_string_literal(line)
330  if match:
331  date = match
332  return ""

382  """Wrapper around three possible ways of accessing data, depending on the current run mode and

384  succeeded, or an empty string if it failed, or returned no data. The method argument is

406 def run_interruptable_subprocess(args, timeout_secs: int = 10) -> subprocess.CompletedProcess:
407  """Wrap subprocess call so it can be interrupted gracefully."""
408  creation_flags = 0
409  if hasattr(subprocess, "CREATE_NO_WINDOW"):
410  # Added in Python 3.7 -- only used on Windows
411  creation_flags = subprocess.CREATE_NO_WINDOW
412  try:
413  p = subprocess.Popen(
414  args,
415  stdout=subprocess.PIPE,
416  stderr=subprocess.PIPE,
417  creationflags=creation_flags,
418  text=True,
419  encoding="utf-8",
420  )
421  except OSError as e:
422  raise subprocess.CalledProcessError(-1, args, "", e.strerror)
423  stdout = ""
424  stderr = ""
425  return_code = None
426  start_time = time.time()
427  while return_code is None:
428  try:
429  # one second timeout allows interrupting the run once per second
430  stdout, stderr = p.communicate(timeout=1)
431  return_code = p.returncode
432  except subprocess.TimeoutExpired:
433  if (
434  hasattr(QtCore, "QThread")
435  and QtCore.QThread.currentThread().isInterruptionRequested()
436  ):
437  p.kill()
438  raise ProcessInterrupted()
439  if time.time() - start_time >= timeout_secs: # The real timeout
440  p.kill()
441  stdout, stderr = p.communicate()
442  return_code = -1
443  if return_code is None or return_code != 0:
444  raise subprocess.CalledProcessError(
445  return_code if return_code is not None else -1, args, stdout, stderr
446  )
447  return subprocess.CompletedProcess(args, return_code, stdout, stderr)

450 def process_date_string_to_python_datetime(date_string: str) -> datetime:
451  """For modern macros the expected date format is ISO 8601, YYYY-MM-DD. For older macros this standard was not always
452  used, and various orderings and separators were used. This function tries to match the majority of those older
453  macros. Commonly-used separators are periods, slashes, and dashes."""
454 
455  def raise_error(bad_string: str, root_cause: Exception = None):
456  raise ValueError(
457  f"Unrecognized date string '{bad_string}' (expected YYYY-MM-DD)"
458  ) from root_cause
459 
460  split_result = re.split(r"[ ./-]+", date_string.strip())
461  if len(split_result) != 3:
462  raise_error(date_string)
463 
464  try:
465  split_result = [int(x) for x in split_result]
466  # The earliest possible year an addon can be created or edited is 2001:
467  if split_result[0] > 2000:
468  return datetime(split_result[0], split_result[1], split_result[2])
469  elif split_result[2] > 2000:
470  # Generally speaking it's not possible to distinguish between DD-MM and MM-DD, so try the first, and
471  # only if that fails try the second
472  if split_result[1] <= 12:
473  return datetime(split_result[2], split_result[1], split_result[0])
474  return datetime(split_result[2], split_result[0], split_result[1])
475  else:
476  raise ValueError(f"Invalid year in date string '{date_string}'")
477  except ValueError as exception:
478  raise_error(date_string, exception)
