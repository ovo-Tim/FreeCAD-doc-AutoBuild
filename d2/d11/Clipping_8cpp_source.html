89  static void moveCallback(void* data, SoSensor* sensor)
90  {
91  Q_UNUSED(sensor);
92  auto self = static_cast<Private*>(data);
93  if (self->view) {
94  Gui::View3DInventorViewer* view = self->view->getViewer();
95  SoClipPlane* clip = self->clipView;
96  SbPlane pln = clip->plane.getValue();
97  clip->plane.setValue(SbPlane(view->getViewDirection(), pln.getDistanceFromOrigin()));
98  }
99  }
100 };

104 Clipping::Clipping(Gui::View3DInventor* view, QWidget* parent)
105  : QDialog(parent)
106  , d(new Private)
107 {
108  // create widgets
109  d->ui.setupUi(this);
110  setupConnections();
111 
112  d->ui.clipView->setRange(-INT_MAX, INT_MAX);
113  d->ui.clipView->setSingleStep(0.1f);
114  d->ui.clipX->setRange(-INT_MAX, INT_MAX);
115  d->ui.clipX->setSingleStep(0.1f);
116  d->ui.clipY->setRange(-INT_MAX, INT_MAX);
117  d->ui.clipY->setSingleStep(0.1f);
118  d->ui.clipZ->setRange(-INT_MAX, INT_MAX);
119  d->ui.clipZ->setSingleStep(0.1f);
120 
121  d->ui.dirX->setRange(-INT_MAX, INT_MAX);
122  d->ui.dirX->setSingleStep(0.1f);
123  d->ui.dirY->setRange(-INT_MAX, INT_MAX);
124  d->ui.dirY->setSingleStep(0.1f);
125  d->ui.dirZ->setRange(-INT_MAX, INT_MAX);
126  d->ui.dirZ->setSingleStep(0.1f);
127  d->ui.dirZ->setValue(1.0f);
128 
129  d->view = view;
130  View3DInventorViewer* viewer = view->getViewer();
131  d->node = static_cast<SoGroup*>(viewer->getSceneGraph());
132  d->node->ref();
133  d->node->insertChild(d->clipX, 0);
134  d->node->insertChild(d->clipY, 0);
135  d->node->insertChild(d->clipZ, 0);
136  d->node->insertChild(d->clipView, 0);
137 
138  SoGetBoundingBoxAction action(viewer->getSoRenderManager()->getViewportRegion());
139  action.apply(viewer->getSceneGraph());
140  SbBox3f box = action.getBoundingBox();
141 
142  if (!box.isEmpty()) {
143  SbVec3f cnt = box.getCenter();
144  d->ui.clipView->setValue(cnt[2]);
145  d->ui.clipX->setValue(cnt[0]);
146  d->ui.clipY->setValue(cnt[1]);
147  d->ui.clipZ->setValue(cnt[2]);
148 
149  int minDecimals = 2;
150  float lenx, leny, lenz;
151  box.getSize(lenx, leny, lenz);
152  int steps = 100;
153  float minlen = std::min<float>(lenx, std::min<float>(leny, lenz));
154 
155  // determine the single step values
156  {
157  minlen = minlen / steps;
158  int dim = static_cast<int>(log10(minlen));
159  double singleStep = pow(10.0, dim);
160  d->ui.clipView->setSingleStep(singleStep);
161  minDecimals = std::max(minDecimals, -dim);
162  }
163  {
164  lenx = lenx / steps;
165  int dim = static_cast<int>(log10(lenx));
166  double singleStep = pow(10.0, dim);
167  d->ui.clipX->setSingleStep(singleStep);
168  }
169  {
170  leny = leny / steps;
171  int dim = static_cast<int>(log10(leny));
172  double singleStep = pow(10.0, dim);
173  d->ui.clipY->setSingleStep(singleStep);
174  }
175  {
176  lenz = lenz / steps;
177  int dim = static_cast<int>(log10(lenz));
178  double singleStep = pow(10.0, dim);
179  d->ui.clipZ->setSingleStep(singleStep);
180  }
181 
182  // set decimals
183  d->ui.clipView->setDecimals(minDecimals);
184  d->ui.clipX->setDecimals(minDecimals);
185  d->ui.clipY->setDecimals(minDecimals);
186  d->ui.clipZ->setDecimals(minDecimals);
187  }
188 }
