66 bool ViewProviderBoolean::setEdit(int ModNum)
67 {
68  if (ModNum == ViewProvider::Default ) {
69  // When double-clicking on the item for this fillet the
70  // object unsets and sets its edit mode without closing
71  // the task panel
72  Gui::TaskView::TaskDialog *dlg = Gui::Control().activeDialog();
73  TaskDlgBooleanParameters *booleanDlg = qobject_cast<TaskDlgBooleanParameters *>(dlg);
74  if (booleanDlg && booleanDlg->getBooleanView() != this)
75  booleanDlg = nullptr; // another pad left open its task panel
76  if (dlg && !booleanDlg) {
77  QMessageBox msgBox;
78  msgBox.setText(QObject::tr("A dialog is already open in the task panel"));
79  msgBox.setInformativeText(QObject::tr("Do you want to close this dialog?"));
80  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
81  msgBox.setDefaultButton(QMessageBox::Yes);
82  int ret = msgBox.exec();
83  if (ret == QMessageBox::Yes)
84  Gui::Control().closeDialog();
85  else
86  return false;
87  }
88 
89  // clear the selection (convenience)
90  Gui::Selection().clearSelection();
91 
92  // always change to PartDesign WB, remember where we come from
93  oldWb = Gui::Command::assureWorkbench("PartDesignWorkbench");
94 
95  // start the edit dialog
96  if (booleanDlg)
97  Gui::Control().showDialog(booleanDlg);
98  else
99  Gui::Control().showDialog(new TaskDlgBooleanParameters(this));
100 
101  return true;
102  }
103  else {
104  return PartGui::ViewProviderPart::setEdit(ModNum); // clazy:exclude=skipped-base-method
105  }
106 }
