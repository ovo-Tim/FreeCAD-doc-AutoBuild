56 void SolidObject::GenerateSolid(std::vector<Vertex>& verts, std::vector<GLushort>& indices)
57 {
58  shape.SetModelData(verts, indices);
59 
60  // calculate object's bounding box:
61  float x = 999999.0f, y = 999999.0f, z = 999999.0f;
62  float l = -999999.0f, w = -999999.0f, h = -999999.0f;
63  for (auto& vert : verts) {
64  x = std::fminf(x, vert.x);
65  y = std::fminf(y, vert.y);
66  z = std::fminf(z, vert.z);
67  l = std::fmaxf(l, vert.x);
68  w = std::fmaxf(w, vert.y);
69  h = std::fmaxf(h, vert.z);
70  }
71  l -= x;
72  w -= y;
73  h -= z;
74  vec3_set(position, x, y, z);
75  vec3_set(center, x + l / 2, y + w / 2, z + h / 2);
76  vec3_set(size, l, w, h);
77  isValid = true;
78 }
