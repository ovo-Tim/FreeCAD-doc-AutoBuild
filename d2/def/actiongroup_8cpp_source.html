186 void ActionGroup::paintEvent ( QPaintEvent * event )
187 {
188  Q_UNUSED(event);
189  QPainter p(this);
190 
191  if (myDummy->isVisible()) {
192  if (myScheme->groupFoldThaw) {
193  if (m_foldDirection < 0)
194  p.setOpacity((double)m_foldStep / myScheme->groupFoldSteps);
195  else
196  p.setOpacity((double)(myScheme->groupFoldSteps-m_foldStep) / myScheme->groupFoldSteps);
197  }
198 
199  switch (myScheme->groupFoldEffect)
200  {
201  case ActionPanelScheme::ShrunkFolding:
202  p.drawPixmap(myDummy->pos(), m_foldPixmap.scaled(myDummy->size()) );
203  break;
204 
205  case ActionPanelScheme::SlideFolding:
206  p.drawPixmap(myDummy->pos(), m_foldPixmap,
207  QRect(0, m_foldPixmap.height()-myDummy->height(),
208  m_foldPixmap.width(), myDummy->width()
209  ) );
210  break;
211 
212  default:
