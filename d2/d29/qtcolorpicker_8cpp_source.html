314 void QtColorPicker::buttonPressed(bool toggled)
315 {
316  if (!toggled)
317  return;
318 
319  const QRect desktop = QApplication::activeWindow()->geometry();
320 
321  // Make sure the popup is inside the desktop.
322  QPoint pos = mapToGlobal(rect().bottomLeft());
323  if (pos.x() < desktop.left())
324  pos.setX(desktop.left());
325  if (pos.y() < desktop.top())
326  pos.setY(desktop.top());
327 
328  if ((pos.x() + popup->sizeHint().width()) > desktop.right())
329  pos.setX(desktop.right() - popup->sizeHint().width());
330  if ((pos.y() + popup->sizeHint().height()) > desktop.bottom())
331  pos.setY(desktop.bottom() - popup->sizeHint().height());
332  popup->move(pos);
333 
334  if (ColorPickerItem *item = popup->find(col))
335  item->setSelected(true);
336 
337  // Remove focus from this widget, preventing the focus rect
338  // from showing when the popup is shown. Order an update to
339  // make sure the focus rect is cleared.
340  clearFocus();
341  update();
342 
343  // Allow keyboard navigation as soon as the popup shows.
344  popup->setFocus();
345 
346  // Execute the popup. The popup will enter the event loop.
347  popup->show();

607 void ColorPickerPopup::insertColor(const QColor &col, const QString &text, int index)
608 {
609  // Don't add colors that we have already.
610  ColorPickerItem *existingItem = find(col);
611  ColorPickerItem *lastSelectedItem = find(lastSelected());
612 
613  if (existingItem) {
614  if (lastSelectedItem && existingItem != lastSelectedItem)
615  lastSelectedItem->setSelected(false);
616  existingItem->setFocus();
617  existingItem->setSelected(true);
618  return;
619  }
620 
621  ColorPickerItem *item = new ColorPickerItem(col, text, this);
622 
623  if (lastSelectedItem) {
624  lastSelectedItem->setSelected(false);
625  }
626  else {
627  item->setSelected(true);
628  lastSel = col;
629  }
630  item->setFocus();
631 
632  connect(item, &ColorPickerItem::selected, this, &ColorPickerPopup::updateSelected);
633 
634  if (index == -1)
635  index = items.count();
636 
637  items.insert((unsigned int)index, item);
638  regenerateGrid();
639 
640  update();
641 }

707 void ColorPickerPopup::keyPressEvent(QKeyEvent *e)
708 {
709  int curRow = 0;
710  int curCol = 0;
711 
712  bool foundFocus = false;
713  for (int j = 0; !foundFocus && j < grid->rowCount(); ++j) {
714  for (int i = 0; !foundFocus && i < grid->columnCount(); ++i) {
715  if (widgetAt[j][i] && widgetAt[j][i]->hasFocus()) {
716  curRow = j;
717  curCol = i;
718  foundFocus = true;
719  break;
720  }
721  }
722  }
723 
724  switch (e->key()) {
725  case Qt::Key_Left:
726  if (curCol > 0) --curCol;
727  else if (curRow > 0) { --curRow; curCol = grid->columnCount() - 1; }
728  break;
729  case Qt::Key_Right:
730  if (curCol < grid->columnCount() - 1 && widgetAt[curRow][curCol + 1]) ++curCol;
731  else if (curRow < grid->rowCount() - 1) { ++curRow; curCol = 0; }
732  break;
733  case Qt::Key_Up:
734  if (curRow > 0) --curRow;
735  else curCol = 0;
736  break;
737  case Qt::Key_Down:
738  if (curRow < grid->rowCount() - 1) {
739  QWidget *w = widgetAt[curRow + 1][curCol];
740  if (w) {
741  ++curRow;
742  } else for (int i = 1; i < grid->columnCount(); ++i) {
743  if (!widgetAt[curRow + 1][i]) {
744  curCol = i - 1;
745  ++curRow;
746  break;
747  }
748  }
749  }
750  break;
751  case Qt::Key_Space:
752  case Qt::Key_Return:
753  case Qt::Key_Enter: {
754  QWidget *w = widgetAt[curRow][curCol];
755  if (w && w->inherits("ColorPickerItem")) {
756  ColorPickerItem *wi = reinterpret_cast<ColorPickerItem *>(w);
757  wi->setSelected(true);
758 
759  QLayoutItem *layoutItem;
760  int i = 0;
761  while ((layoutItem = grid->itemAt(i)) != nullptr) {
762  QWidget *w = layoutItem->widget();
763  if (w && w->inherits("ColorPickerItem")) {
764  ColorPickerItem *litem
765  = reinterpret_cast<ColorPickerItem *>(layoutItem->widget());
766  if (litem != wi)
767  litem->setSelected(false);
768  }
769  ++i;
770  }
771 
772  lastSel = wi->color();
773  Q_EMIT selected(wi->color());
774  hide();
775  } else if (w && w->inherits("QPushButton")) {
776  ColorPickerItem *wi = reinterpret_cast<ColorPickerItem *>(w);
777  wi->setSelected(true);
778 
779  QLayoutItem *layoutItem;
780  int i = 0;
781  while ((layoutItem = grid->itemAt(i)) != nullptr) {
782  QWidget *w = layoutItem->widget();
783  if (w && w->inherits("ColorPickerItem")) {
784  ColorPickerItem *litem
785  = reinterpret_cast<ColorPickerItem *>(layoutItem->widget());
786  if (litem != wi)
787  litem->setSelected(false);
788  }
789  ++i;
790  }
791 
792  lastSel = wi->color();
793  Q_EMIT selected(wi->color());
794  hide();
795  }
796  }
797  break;
798  case Qt::Key_Escape:
799  hide();
800  break;
801  default:
802  e->ignore();
803  break;
804  }
805 
806  widgetAt[curRow][curCol]->setFocus();
807 }

864 void ColorPickerPopup::regenerateGrid()
865 {
866  widgetAt.clear();
867 
868  int columns = cols;
869  if (columns == -1)
870  columns = (int) ceil(sqrt((float) items.count()));
871 
872  // When the number of columns grows, the number of rows will
873  // fall. There's no way to shrink a grid, so we create a new
874  // one.
875  if (grid) delete grid;
876  grid = new QGridLayout(this);
877  grid->setContentsMargins(1, 1, 1, 1);
878  grid->setSpacing(0);
879 
880  int ccol = 0, crow = 0;
881  for (int i = 0; i < items.size(); ++i) {
882  if (items.at(i)) {
883  widgetAt[crow][ccol] = items.at(i);
884  grid->addWidget(items.at(i), crow, ccol++);
885  if (ccol == columns) {
886  ++crow;
887  ccol = 0;
888  }
889  }
890  }
891 
892  if (moreButton) {
893  grid->addWidget(moreButton, crow, ccol);
894  widgetAt[crow][ccol] = moreButton;
895  }
896  updateGeometry();
897 }
