192 {
193  QMenu *editMenu = createStandardContextMenu();
194  editMenu->setTitle(tr("Edit"));
195  auto menu = new QMenu(QString::fromLatin1("InputFieldContextmenu"));
196 
197  menu->addMenu(editMenu);
198  menu->addSeparator();
199 
200  // datastructure to remember actions for values
201  std::vector<QString> values;
202  std::vector<QAction *> actions;
203 
204  // add the history menu part...
205  std::vector<QString> history = getHistory();
206 
207  for(const auto & it : history){
208  actions.push_back(menu->addAction(it));
209  values.push_back(it);
210  }
211 
212  // add the save value portion of the menu
213  menu->addSeparator();
214  QAction *SaveValueAction = menu->addAction(tr("Save value"));
215  std::vector<QString> savedValues = getSavedValues();
216 
217  for(const auto & savedValue : savedValues){
218  actions.push_back(menu->addAction(savedValue));
219  values.push_back(savedValue);
220  }
221 
222  // call the menu and wait until its back
223  QAction *saveAction = menu->exec(event->globalPos());
224 
225  // look what the user has chosen
226  if(saveAction == SaveValueAction)
227  pushToSavedValues();
228  else{
229  int i=0;
230  for(auto it = actions.begin();it!=actions.end();++it,i++)
231  if(*it == saveAction)
232  this->setText(values[i]);
233  }
234 
235  delete menu;
236 }

238 void InputField::newInput(const QString & text)
239 {
240  Quantity res;
241  try {
242  QString input = text;
243  fixup(input);
244 
245  if (isBound()) {
246  std::shared_ptr<Expression> e(ExpressionParser::parse(getPath().getDocumentObject(), input.toUtf8()));
247 
248  setExpression(e);
249 
250  std::unique_ptr<Expression> evalRes(getExpression()->eval());
251 
252  auto * value = freecad_dynamic_cast<NumberExpression>(evalRes.get());
253  if (value) {
254  res.setValue(value->getValue());
255  res.setUnit(value->getUnit());
256  }
257  }
258  else
259  res = Quantity::parse(input.toStdString());
260  }
261  catch(Base::Exception &e){
262  QString errorText = QString::fromLatin1(e.what());
263  QPixmap pixmap = getValidationIcon(":/icons/button_invalid.svg", QSize(sizeHint().height(),sizeHint().height()));
264  iconLabel->setPixmap(pixmap);
265  Q_EMIT parseError(errorText);
266  validInput = false;
267  return;
268  }
269 
270  if (res.getUnit().isEmpty())
271  res.setUnit(this->actUnit);
272 
273  // check if unit fits!
274  if(!actUnit.isEmpty() && !res.getUnit().isEmpty() && actUnit != res.getUnit()){
275  QPixmap pixmap = getValidationIcon(":/icons/button_invalid.svg", QSize(sizeHint().height(),sizeHint().height()));
276  iconLabel->setPixmap(pixmap);
277  Q_EMIT parseError(QString::fromLatin1("Wrong unit"));
278  validInput = false;
279  return;
280  }
281 
282 
283  QPixmap pixmap = getValidationIcon(":/icons/button_valid.svg", QSize(sizeHint().height(),sizeHint().height()));
284  iconLabel->setPixmap(pixmap);
285  validInput = true;
286 
287  if (res.getValue() > Maximum){
288  res.setValue(Maximum);
289  }
290  if (res.getValue() < Minimum){
291  res.setValue(Minimum);
292  }
293 
294  double dFactor;
295  std::string unitStr;
296  res.getUserString(dFactor, unitStr);
297  actUnitValue = res.getValue()/dFactor;
298  // Preserve previous format
299  res.setFormat(this->actQuantity.getFormat());
300  actQuantity = res;
301 
302  // signaling
303  Q_EMIT valueChanged(res);
304  Q_EMIT valueChanged(res.getValue());
305 }

651 void InputField::keyPressEvent(QKeyEvent *event)
652 {
653  if (isReadOnly()) {
654  QLineEdit::keyPressEvent(event);
655  return;
656  }
657 
658  double val = actUnitValue;
659 
660  switch (event->key()) {
661  case Qt::Key_Up:
662  val += StepSize;
663  if (val > Maximum)
664  val = Maximum;
665  break;
666  case Qt::Key_Down:
667  val -= StepSize;
668  if (val < Minimum)
669  val = Minimum;
670  break;
671  default:
672  QLineEdit::keyPressEvent(event);
673  return;
674  }
675 
676  double dFactor;
677  std::string unitStr;
678  actQuantity.getUserString(dFactor, unitStr);
679  this->setText(QString::fromUtf8("%L1 %2").arg(val).arg(QString::fromStdString(unitStr)));
680  event->accept();
681 }

683 void InputField::wheelEvent (QWheelEvent * event)
684 {
685  if (!hasFocus())
686  return;
687 
688  if (isReadOnly()) {
689  QLineEdit::wheelEvent(event);
690  return;
691  }
692 
693  double factor = event->modifiers() & Qt::ControlModifier ? 10 : 1;
694  double step = event->angleDelta().y() > 0 ? StepSize : -StepSize;
695  double val = actUnitValue + factor * step;
696  if (val > Maximum)
697  val = Maximum;
698  else if (val < Minimum)
699  val = Minimum;
700 
701  double dFactor;
702  std::string unitStr;
703  actQuantity.getUserString(dFactor, unitStr);
704 
705  this->setText(QString::fromUtf8("%L1 %2").arg(val).arg(QString::fromStdString(unitStr)));
706  selectNumber();
707  event->accept();
708 }

727 QValidator::State InputField::validate(QString& input, int& pos) const
728 {
729  Q_UNUSED(pos);
730  try {
731  Quantity res;
732  QString text = input;
733  fixup(text);
734  res = Quantity::parse(text.toStdString());
735 
736  double factor;
737  std::string unitStr;
738  res.getUserString(factor, unitStr);
739  double value = res.getValue()/factor;
740  // disallow one to enter numbers out of range
741  if (value > this->Maximum || value < this->Minimum)
742  return QValidator::Invalid;
743  }
744  catch(Base::Exception&) {
745  // Actually invalid input but the newInput slot gives
746  // some feedback
747  return QValidator::Intermediate;
748  }
749 
750  return QValidator::Acceptable;
751 }
