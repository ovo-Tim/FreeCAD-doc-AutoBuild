64 App::DocumentObjectExecReturn* Edge2TracObject::execute()
65 {
66  App::DocumentObject* link = Source.getValue();
67  if (!link) {
68  return new App::DocumentObjectExecReturn("No object linked");
69  }
70  if (!link->isDerivedFrom<Part::Feature>()) {
71  return new App::DocumentObjectExecReturn("Linked object is not a Part object");
72  }
73  Part::Feature* base = static_cast<Part::Feature*>(Source.getValue());
74  const Part::TopoShape& TopShape = base->Shape.getShape();
75 
76  const std::vector<std::string>& SubVals = Source.getSubValuesStartsWith("Edge");
77  if (SubVals.empty()) {
78  return new App::DocumentObjectExecReturn("No Edges specified");
79  }
80 
81  // container for all the edges
82  std::vector<TopoDS_Edge> edges;
83 
84  // run through the edge name and get the real objects from the TopoShape
85  for (const auto& SubVal : SubVals) {
86  TopoDS_Edge edge = TopoDS::Edge(TopShape.getSubShape(SubVal.c_str()));
87  edges.push_back(edge);
88  }
89 
90  // instantiate an edge cluster sorter and get the result
91  Part::Edgecluster acluster(edges);
92  Part::tEdgeClusterVector aclusteroutput = acluster.GetClusters();
93 
94  if (aclusteroutput.empty()) {
95  return new App::DocumentObjectExecReturn("No Edges specified");
96  }
97 
98  // set the number of cluster and edges
99  NbrOfCluster = aclusteroutput.size();
100  NbrOfEdges = 0;
101  for (const auto& it : aclusteroutput) {
102  NbrOfEdges += it.size();
103  }
104 
105  // trajectory to fill
106  Robot::Trajectory trac;
107  bool first = true;
108 
109  // cycle through the cluster
110  for (const auto& it : aclusteroutput) {
111  // cycle through the edges of the cluster
112  for (const auto& it2 : it) {
113  BRepAdaptor_Curve adapt(it2);
114 
115  switch (adapt.GetType()) {
116  case GeomAbs_Line: {
117  // get start and end point
118  gp_Pnt P1 = adapt.Value(adapt.FirstParameter());
119  gp_Pnt P2 = adapt.Value(adapt.LastParameter());
120 
121  Base::Rotation R1;
122  Base::Rotation R2;
123 
124  // if orientation is used
125  if (UseRotation.getValue()) {
126  // here get the orientation of the start and end point...
127  // R1 = ;
128  // R2 = ;
129  }
130 
131  // if reverse orintation, switch the points
132  if (it2.Orientation() == TopAbs_REVERSED) {
133  // switch the points and orientation
134  gp_Pnt tmpP = P1;
135  Base::Rotation tmpR = R1;
136  P1 = P2;
137  R1 = R2;
138  R2 = tmpR;
139  P2 = tmpP;
140  }
141  if (first) {
142  Waypoint wp("Pt",
143  Base::Placement(Base::Vector3d(P1.X(), P1.Y(), P1.Z()), R1));
144  trac.addWaypoint(wp);
145  first = false;
146  }
147  Waypoint wp("Pt", Base::Placement(Base::Vector3d(P2.X(), P2.Y(), P2.Z()), R2));
148  trac.addWaypoint(wp);
149  break;
150  }
151  case GeomAbs_BSplineCurve: {
152  Standard_Real Length = CPnts_AbscissaPoint::Length(adapt);
153  Standard_Real ParLength = adapt.LastParameter() - adapt.FirstParameter();
154  Standard_Real NbrSegments = Round(Length / SegValue.getValue());
155 
156  Standard_Real beg = adapt.FirstParameter();
157  Standard_Real end = adapt.LastParameter();
158  Standard_Real stp = ParLength / NbrSegments;
159  bool reversed = false;
160  if (it2.Orientation() == TopAbs_REVERSED) {
161  std::swap(beg, end);
162  stp = -stp;
163  reversed = true;
164  }
165 
166  if (first) {
167  first = false;
168  }
169  else {
170  beg += stp;
171  }
172  Base::SequencerLauncher seq("Create waypoints",
173  static_cast<size_t>((end - beg) / stp));
174  if (reversed) {
175  for (; beg > end; beg += stp) {
176  gp_Pnt P = adapt.Value(beg);
177  Base::Rotation R1;
178  // if orientation is used
179  if (UseRotation.getValue()) {
180  // here get the orientation of the start and end point...
181  // R1 = ;
182  }
183 
184  Waypoint wp("Pt",
185  Base::Placement(Base::Vector3d(P.X(), P.Y(), P.Z()), R1));
186  trac.addWaypoint(wp);
187  seq.next();
188  }
189  }
190  else {
191  for (; beg < end; beg += stp) {
192  gp_Pnt P = adapt.Value(beg);
193  Base::Rotation R1;
194  // if orientation is used
195  if (UseRotation.getValue()) {
196  // here get the orientation of the start and end point...
197  // R1 = ;
198  }
199  Waypoint wp("Pt",
200  Base::Placement(Base::Vector3d(P.X(), P.Y(), P.Z()), R1));
201  trac.addWaypoint(wp);
202  seq.next();
203  }
204  }
205 
206  } break;
207  case GeomAbs_Circle: {
208  Standard_Real Length = CPnts_AbscissaPoint::Length(adapt);
209  Standard_Real ParLength = adapt.LastParameter() - adapt.FirstParameter();
210  Standard_Real NbrSegments = Round(Length / SegValue.getValue());
211  Standard_Real SegLength = ParLength / NbrSegments;
212 
213  if (it2.Orientation() == TopAbs_REVERSED) {
214  // Beginning and End switch
215  double i = adapt.LastParameter();
216  if (first) {
217  first = false;
218  }
219  else {
220  i -= SegLength;
221  }
222  for (; i > adapt.FirstParameter(); i -= SegLength) {
223  gp_Pnt P = adapt.Value(i);
224  Base::Rotation R1;
225  // if orientation is used
226  if (UseRotation.getValue()) {
227  // here get the orientation of the start and end point...
228  // R1 = ;
229  }
230  Waypoint wp("Pt",
231  Base::Placement(Base::Vector3d(P.X(), P.Y(), P.Z()), R1));
232  trac.addWaypoint(wp);
233  }
234  }
235  else {
236  double i = adapt.FirstParameter();
237  if (first) {
238  first = false;
239  }
240  else {
241  i += SegLength;
242  }
243  for (; i < adapt.LastParameter(); i += SegLength) {
244  gp_Pnt P = adapt.Value(i);
245  Base::Rotation R1;
246  // if orientation is used
247  if (UseRotation.getValue()) {
248  // here get the orientation of the start and end point...
249  // R1 = ;
250  }
251  Waypoint wp("Pt",
252  Base::Placement(Base::Vector3d(P.X(), P.Y(), P.Z()), R1));
253  trac.addWaypoint(wp);
254  }
255  }
256  break;
257  }
258 
259  default:
260  throw Base::TypeError("Unknown Edge type in Robot::Edge2TracObject::execute()");
261  }
262  }
263  }
264 
265  Trajectory.setValue(trac);
266 
267  return App::DocumentObject::StdReturn;
268 }
