75 bool ViewProviderGroupExtension::extensionCanDropObject(App::DocumentObject* obj) const
76 {
77 #ifdef FC_DEBUG
78  Base::Console().Log("Check ViewProviderGroupExtension\n");
79 #endif
80 
81  auto extobj = getExtendedViewProvider()->getObject();
82  auto group = extobj->getExtensionByType<App::GroupExtension>();
83 
84  //we cannot drop thing of this group into it again if it does not allow reorder
85  if (group->hasObject(obj) && !getExtendedViewProvider()->acceptReorderingObjects()) {
86  return false;
87  }
88 
89  // Check for possible cyclic dependencies if we allowed to drop the object
90  const auto& list = obj->getOutList();
91  if (std::find(list.begin(), list.end(), extobj) != list.end()) {
92  Base::Console().Warning("Do not add cyclic dependency to %s\n", extobj->Label.getValue());
93  return false;
94  }
95 
96  return group->allowObject(obj);
97 }

126 void ViewProviderGroupExtension::extensionShow() {
127 
128  // avoid possible infinite recursion
129  if (guard)
130  return;
131  Base::StateLocker lock(guard);
132 
133  // when reading the Visibility property from file then do not hide the
134  // objects of this group because they have stored their visibility status, too
135  //
136  // Property::User1 is used by ViewProviderDocumentObject to mark for
137  // temporary visibility changes. Do not propagate the change to children.
138  if (!getExtendedViewProvider()->isRestoring()
139  && !getExtendedViewProvider()->Visibility.testStatus(App::Property::User1)) {
140  auto* group = getExtendedViewProvider()->getObject()->getExtensionByType<App::GroupExtension>();
141  for(auto obj : group->Group.getValues()) {
142  if(obj && !obj->Visibility.getValue())
143  obj->Visibility.setValue(true);
144  }
145  }
146 
147  ViewProviderExtension::extensionShow();
148 }

174 bool ViewProviderGroupExtension::extensionOnDelete(const std::vector< std::string >& ) {
175 
176  auto* group = getExtendedViewProvider()->getObject()->getExtensionByType<App::GroupExtension>();
177  // If the group is nonempty ask the user if they want to delete its content
178  if (group->Group.getSize() > 0) {
179  QMessageBox::StandardButton choice =
180  QMessageBox::question(getMainWindow(), QObject::tr ( "Delete group content?" ),
181  QObject::tr ( "The %1 is not empty, delete its content as well?")
182  .arg ( QString::fromUtf8 ( getExtendedViewProvider()->getObject()->Label.getValue () ) ),
183  QMessageBox::Yes | QMessageBox::No, QMessageBox::Yes );
184 
185  if (choice == QMessageBox::Yes) {
186  Gui::Command::doCommand(Gui::Command::Doc,
187  "App.getDocument(\"%s\").getObject(\"%s\").removeObjectsFromDocument()"
188  , getExtendedViewProvider()->getObject()->getDocument()->getName()
189  , getExtendedViewProvider()->getObject()->getNameInDocument());
190  }
191  }
192  return true;
193 }
