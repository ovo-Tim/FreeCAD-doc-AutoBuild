225 void CmdSketcherCompBSplineShowHideGeometryInformation::activated(int iMsg)
226 {
227  Gui::CommandManager& rcCmdMgr = Gui::Application::Instance->commandManager();
228  Gui::Command* cmd;
229 
230  if (iMsg == 0) {
231  cmd = rcCmdMgr.getCommandByName("Sketcher_BSplineDegree");
232  }
233  else if (iMsg == 1) {
234  cmd = rcCmdMgr.getCommandByName("Sketcher_BSplinePolygon");
235  }
236  else if (iMsg == 2) {
237  cmd = rcCmdMgr.getCommandByName("Sketcher_BSplineComb");
238  }
239  else if (iMsg == 3) {
240  cmd = rcCmdMgr.getCommandByName("Sketcher_BSplineKnotMultiplicity");
241  }
242  else if (iMsg == 4) {
243  cmd = rcCmdMgr.getCommandByName("Sketcher_BSplinePoleWeight");
244  }
245  else {
246  return;
247  }
248 
249  cmd->invoke(0);
250 
251  // Since the default icon is reset when enabling/disabling the command we have
252  // to explicitly set the icon of the used command.
253  Gui::ActionGroup* pcAction = qobject_cast<Gui::ActionGroup*>(_pcAction);
254  QList<QAction*> a = pcAction->actions();
255 
256  assert(iMsg < a.size());
257  pcAction->setIcon(a[iMsg]->icon());
258  // we must also set the tooltip of the used command
259  pcAction->setToolTip(a[iMsg]->toolTip());
260 }
