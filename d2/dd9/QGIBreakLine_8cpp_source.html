99 void QGIBreakLine::drawLargeZigZag()
100 {
101  Base::Vector3d horizontal{1.0, 0.0, 0.0};
102  prepareGeometryChange();
103  double offset = zigzagWidth / 2.0;
104  if (DU::fpCompare(fabs(m_direction.Dot(horizontal)), 1.0, EWTOLERANCE)) {
105  // m_direction connects the two cut points. The zigzags have
106  // to be perpendicular to m_direction
107  // 2x vertical zigzag
108  Base::Vector3d start = Base::Vector3d(m_left - offset, m_bottom, 0.0);
109  m_line0->setPath(makeVerticalZigZag(start));
110 
111  start = Base::Vector3d(m_right - offset, m_bottom, 0.0);
112  m_line1->setPath(makeVerticalZigZag(start));
113  } else {
114  // m_top is lower than m_bottom due to Qt Y+ down coords
115  // the higher break line
116  // 2x horizontal zigszags
117  Base::Vector3d start = Base::Vector3d(m_left, m_bottom - offset, 0.0);
118  m_line0->setPath(makeHorizontalZigZag(start));
119 
120  // the lower break line
121  start = Base::Vector3d(m_left, m_top - offset, 0.0);
122  m_line1->setPath(makeHorizontalZigZag(start));
123  }
124 
125  QRectF backgroundRect(m_left - offset, m_bottom - offset,
126  std::fabs(m_right - m_left + zigzagWidth),
127  std::fabs(m_top - m_bottom + zigzagWidth));
128  m_background->setRect(backgroundRect);
129 }

131 // start needs to be Rez'd and +Y up
132 QPainterPath QGIBreakLine::makeHorizontalZigZag(Base::Vector3d start) const
133 {
134  // Base::Console().Message("QGIBL::makeHorizontalZigZag(%s)\n", DU::formatVector(start).c_str());
135  QPainterPath pPath;
136  double step = (m_right - m_left) / segments;
137  Base::Vector3d xOffset = Base::Vector3d(step, 0.0, 0.0); // 1/2 wave length
138  Base::Vector3d yOffset = Base::Vector3d(0.0, zigzagWidth, 0.0); // amplitude
139 
140  pPath.moveTo(DU::toQPointF(start));
141  Base::Vector3d current = start;
142  int iSegment = 0;
143  double flipflop = 1.0;
144  for (; iSegment < segments; iSegment++) {
145  current = current + xOffset;
146  current = current + yOffset * flipflop;
147  pPath.lineTo(DU::toQPointF(current));
148  flipflop *= -1.0;
149  }
150  return pPath;
151 }

153 QPainterPath QGIBreakLine::makeVerticalZigZag(Base::Vector3d start) const
154 {
155  // Base::Console().Message("QGIBL::makeVerticalZigZag(%s)\n", DU::formatVector(start).c_str());
156  QPainterPath pPath;
157  double step = (m_top - m_bottom) / segments;
158  Base::Vector3d xOffset = Base::Vector3d(zigzagWidth, 0.0, 0.0); // amplitude
159  Base::Vector3d yOffset = Base::Vector3d(0.0, step, 0.0); // 1/2 wave length
160 
161  pPath.moveTo(DU::toQPointF(start));
162  Base::Vector3d current = start;
163  int iSegment = 0;
164  double flipflop = 1.0;
165  for (; iSegment < segments; iSegment++) {
166  current = current + xOffset * flipflop;
167  current = current + yOffset;
168  pPath.lineTo(DU::toQPointF(current));
169  flipflop *= -1.0;
170  }
171  return pPath;
172 }

175 void QGIBreakLine::drawSimpleLines()
176 {
177  Base::Vector3d horizontal{1.0, 0.0, 0.0};
178  prepareGeometryChange();
179  if (DU::fpCompare(fabs(m_direction.Dot(horizontal)), 1.0, EWTOLERANCE)) {
180  // m_direction connects the two cut points. The break lines have
181  // to be perpendicular to m_direction
182  Base::Vector3d start = Base::Vector3d(m_left, m_bottom, 0.0);
183  Base::Vector3d end = Base::Vector3d(m_left, m_top, 0.0);
184  m_line0->setPath(pathFromPoints(start, end));
185 
186  start = Base::Vector3d(m_right, m_bottom, 0.0);
187  end = Base::Vector3d(m_right, m_top, 0.0);
188  m_line1->setPath(pathFromPoints(start, end));
189  } else {
190  // m_top is lower than m_bottom due to Qt Y+ down coords
191  // the higher break line
192  // 2x horizontal zigszags
193  Base::Vector3d start = Base::Vector3d(m_left, m_bottom, 0.0);
194  Base::Vector3d end = Base::Vector3d(m_right, m_bottom, 0.0);
195  m_line0->setPath(pathFromPoints(start, end));
196 
197  // the lower break line
198  start = Base::Vector3d(m_left, m_top, 0.0);
199  end = Base::Vector3d(m_right, m_top, 0.0);
200  m_line1->setPath(pathFromPoints(start, end));
201  }
202 }
