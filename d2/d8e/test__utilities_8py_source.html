82  def test_get_readme_url(self):
83  github_urls = [
84  "https://github.com/FreeCAD/FreeCAD",
85  ]
86  gitlab_urls = [
87  "https://gitlab.com/freecad/FreeCAD",
88  "https://framagit.org/freecad/FreeCAD",
89  "https://salsa.debian.org/science-team/freecad",
90  "https://unknown.location/and/path",
91  ]
92 
93  # GitHub and Gitlab have two different schemes for file URLs: unrecognized URLs are
94  # presumed to be local instances of a GitLab server. Note that in neither case does this
95  # take into account the redirects that are used to actually fetch the data.
96 
97  for url in github_urls:
98  branch = "branchname"
99  expected_result = f"{url}/raw/{branch}/README.md"
100  repo = Addon("Test Repo", url, "Addon.Status.NOT_INSTALLED", branch)
101  actual_result = get_readme_url(repo)
102  self.assertEqual(actual_result, expected_result)
103 
104  for url in gitlab_urls:
105  branch = "branchname"
106  expected_result = f"{url}/-/raw/{branch}/README.md"
107  repo = Addon("Test Repo", url, "Addon.Status.NOT_INSTALLED", branch)
108  actual_result = get_readme_url(repo)
109  self.assertEqual(actual_result, expected_result)
