85 void ViewProviderRobotObject::setDragger()
86 {
87  assert(!pcDragger);
88  pcDragger = new SoJackDragger();
89  pcDragger->addMotionCallback(sDraggerMotionCallback, this);
90  pcTcpRoot->addChild(pcDragger);
91 
92  // set the actual TCP position
93  Robot::RobotObject* robObj = static_cast<Robot::RobotObject*>(pcObject);
94  Base::Placement loc = robObj->Tcp.getValue();
95  SbMatrix M;
96  M.setTransform(SbVec3f(loc.getPosition().x, loc.getPosition().y, loc.getPosition().z),
97  SbRotation(loc.getRotation()[0],
98  loc.getRotation()[1],
99  loc.getRotation()[2],
100  loc.getRotation()[3]),
101  SbVec3f(150, 150, 150));
102  pcDragger->setMotionMatrix(M);
103 }
