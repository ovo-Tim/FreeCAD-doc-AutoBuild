52 def relAngle(vec, ref):
53  """
54  Takes a vector and a reference axis (refAxis) vector. Calculates the
55  relative angle. The result is returned in degrees (plus or minus)
56  """
57 
58  Path.Log.debug("vec: {} ref: {}".format(vec, ref))
59  norm = vec * 1 # copy vec so we don't alter original
60 
61  if ref == refAxis.x:
62  plane = refAxis.y.value
63  elif ref == refAxis.y:
64  plane = refAxis.z.value
65  else:
66  plane = refAxis.x.value
67 
68  norm.projectToPlane(FreeCAD.Vector(0, 0, 0), plane)
69 
70  ref = ref.value
71  rot = FreeCAD.Rotation(norm, ref)

79 def __getCRotation(normalVector, cMin=-360, cMax=360):
80  """
81  Calculate the valid C axis rotations component to align the normalVector
82  with either the +y or -y axis.
83  multiple poses may be possible. Returns a list of all valid poses
84  """
85  Path.Log.debug("normalVector: {} cMin: {} cMax: {}".format(normalVector, cMin, cMax))
86 
87  angle = relAngle(normalVector, refAxis.y)
88 
89  # Given an angle, there are four possibilities; rotating +- to each of the
90  # two axes +y and -y
91  candidates = [angle]
92  if angle == 0:
93  candidates.append(180)
94  elif angle == 180:
95  candidates.append(0)
96  elif angle >= 0:
97  candidates.append(angle - 180)
98  candidates.append(180 + angle)
99  candidates.append(angle - 360)
100  else:
101  candidates.append(angle + 180)
102  candidates.append(-180 + angle)
103  candidates.append(angle + 360)
104 
105  # final results are candidates that don't violate rotation limits
106  results = [c for c in candidates if c >= cMin and c <= cMax]
107 
108  return results
