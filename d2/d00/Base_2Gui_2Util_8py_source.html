62 def updateInputField(obj, prop, widget, onBeforeChange=None):
63  """updateInputField(obj, prop, widget) ... update obj's property prop with the value of widget.
64  The property's value is only assigned if the new value differs from the current value.
65  This prevents onChanged notifications where the value didn't actually change.
66  Gui::InputField and Gui::QuantitySpinBox widgets are supported - and the property can
67  be of type Quantity or Float.
68  If onBeforeChange is specified it is called before a new value is assigned to the property.
69  Returns True if a new value was assigned, False otherwise (new value is the same as the current).
70  """
71  value = widget.property("rawValue")
72  Path.Log.track("value: {}".format(value))
73  attr = PathUtil.getProperty(obj, prop)
74  attrValue = attr.Value if hasattr(attr, "Value") else attr
75 
76  isDiff = False
77  if not Path.Geom.isRoughly(attrValue, value):
78  isDiff = True
79  else:
80  if hasattr(obj, "ExpressionEngine"):
81  exprSet = False
82  for prp, expr in obj.ExpressionEngine:
83  if prp == prop:
84  exprSet = True
85  Path.Log.debug('prop = "expression": {} = "{}"'.format(prp, expr))
86  value = FreeCAD.Units.Quantity(obj.evalExpression(expr)).Value
87  if not Path.Geom.isRoughly(attrValue, value):
88  isDiff = True
89  break
90  if exprSet:
91  widget.setReadOnly(True)
92  widget.setProperty("exprSet", "true")
93  widget.style().unpolish(widget)
94  widget.ensurePolished()
95  else:
96  widget.setReadOnly(False)
97  widget.setProperty("exprSet", "false")
98  widget.style().unpolish(widget)
99  widget.ensurePolished()
100  widget.update()
101 
102  if isDiff:
103  Path.Log.debug("updateInputField(%s, %s): %.2f -> %.2f" % (obj.Label, prop, attr, value))
104  if onBeforeChange:
105  onBeforeChange(obj)
106  PathUtil.setProperty(obj, prop, value)
107  return True
108 
109  return False
