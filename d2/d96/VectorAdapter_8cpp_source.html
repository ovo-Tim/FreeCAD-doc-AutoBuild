52 VectorAdapter::VectorAdapter(const TopoDS_Face &faceIn, const gp_Vec &pickedPointIn) :
53  status(false), vector(), origin(pickedPointIn)
54 {
55  Handle(Geom_Surface) surface = BRep_Tool::Surface(faceIn);
56  if (surface->IsKind(STANDARD_TYPE(Geom_ElementarySurface)))
57  {
58  Handle(Geom_ElementarySurface) eSurface = Handle(Geom_ElementarySurface)::DownCast(surface);
59  gp_Dir direction = eSurface->Axis().Direction();
60  vector = direction;
61  vector.Normalize();
62  if (faceIn.Orientation() == TopAbs_REVERSED) {
63  vector.Reverse();
64  }
65  if (surface->IsKind(STANDARD_TYPE(Geom_CylindricalSurface)) ||
66  surface->IsKind(STANDARD_TYPE(Geom_SphericalSurface))
67  )
68  {
69  origin = eSurface->Axis().Location().XYZ();
70  projectOriginOntoVector(pickedPointIn);
71  }
72  else {
73  origin = pickedPointIn + vector;
74  }
75  status = true;
76  }
77 }

79 VectorAdapter::VectorAdapter(const TopoDS_Edge &edgeIn, const gp_Vec &pickedPointIn) :
80  status(false), vector(), origin(pickedPointIn)
81 {
82  TopoDS_Vertex firstVertex = TopExp::FirstVertex(edgeIn, Standard_True);
83  TopoDS_Vertex lastVertex = TopExp::LastVertex(edgeIn, Standard_True);
84  vector = convert(lastVertex) - convert(firstVertex);
85  if (vector.Magnitude() < Precision::Confusion()) {
86  return;
87  }
88  vector.Normalize();
89 
90  status = true;
91  projectOriginOntoVector(pickedPointIn);
92 }

94 VectorAdapter::VectorAdapter(const TopoDS_Vertex &vertex1In, const TopoDS_Vertex &vertex2In) :
95  status(false), vector(), origin()
96 {
97  vector = convert(vertex2In) - convert(vertex1In);
98  vector.Normalize();
99 
100  //build origin half way.
101  gp_Vec tempVector = (convert(vertex2In) - convert(vertex1In));
102  double mag = tempVector.Magnitude();
103  tempVector.Normalize();
104  tempVector *= (mag / 2.0);
105  origin = tempVector + convert(vertex1In);
106 
107  status = true;
108 }

110 VectorAdapter::VectorAdapter(const gp_Vec &vector1, const gp_Vec &vector2) :
111  status(false), vector(), origin()
112 {
113  vector = vector2- vector1;
114  vector.Normalize();
115 
116  //build origin half way.
117  gp_Vec tempVector = vector2 - vector1;
118  double mag = tempVector.Magnitude();
119  tempVector.Normalize();
120  tempVector *= (mag / 2.0);
121  origin = tempVector + vector1;
122 
123  status = true;
124 }

126 void VectorAdapter::projectOriginOntoVector(const gp_Vec &pickedPointIn)
127 {
128  Handle(Geom_Curve) heapLine = new Geom_Line(origin.XYZ(), vector.XYZ());
129  gp_Pnt tempPoint(pickedPointIn.XYZ());
130  GeomAPI_ProjectPointOnCurve projection(tempPoint, heapLine);
131  if (projection.NbPoints() < 1) {
132  return;
133  }
134  origin.SetXYZ(projection.Point(1).XYZ());
135 }
