116 App::DocumentObject* Body::getPrevSolidFeature(App::DocumentObject *start)
117 {
118  if (!start) { // default to tip
119  start = Tip.getValue();
120  }
121 
122  if (!start) { // No Tip
123  return nullptr;
124  }
125 
126  if (!hasObject(start))
127  return nullptr;
128 
129  const std::vector<App::DocumentObject*> & features = Group.getValues();
130 
131  auto startIt = std::find (features.rbegin(), features.rend(), start);
132  if (startIt == features.rend()) { // object not found
133  return nullptr;
134  }
135 
136  auto rvIt = std::find_if (startIt + 1, features.rend(), isSolidFeature);
137  if (rvIt != features.rend()) { // the solid found in model list
138  return *rvIt;
139  }
140  return nullptr;

144 {

150  return nullptr;
151  }

154  std::vector<App::DocumentObject*>::const_iterator startIt;

157  if (startIt == features.end()) { // object not found
158  return nullptr;

160 
161  startIt++;

165

167  if (rvIt != features.end()) { // the solid found in model list

282 void Body::insertObject(App::DocumentObject* feature, App::DocumentObject* target, bool after)
283 {
284  if (target && !hasObject (target)) {
285  throw Base::ValueError("Body: the feature we should insert relative to is not part of that body");
286  }
287 
288  //ensure that all origin links are ok
289  relinkToOrigin(feature);
290 
291  std::vector<App::DocumentObject*> model = Group.getValues();
292  std::vector<App::DocumentObject*>::iterator insertInto;
293 
294  // Find out the position there to insert the feature
295  if (!target) {
296  if (after) {
297  insertInto = model.begin();
298  } else {
299  insertInto = model.end();
300  }
301  } else {
302  std::vector<App::DocumentObject*>::iterator targetIt = std::find (model.begin(), model.end(), target);
303  assert (targetIt != model.end());
304  if (after) {
305  insertInto = targetIt + 1;
306  } else {
307  insertInto = targetIt;
308  }
309  }
310 
311  // Insert the new feature after the given
312  model.insert (insertInto, feature);
313 
314  Group.setValues (model);
315 
316  if(feature->isDerivedFrom(PartDesign::Feature::getClassTypeId()))
317  static_cast<PartDesign::Feature*>(feature)->_Body.setValue(this);
318 
319  // Set the BaseFeature property
320  setBaseProperty(feature);
321 }

323 void Body::setBaseProperty(App::DocumentObject* feature)
324 {

327  App::DocumentObject* prevSolidFeature = getPrevSolidFeature(feature);
328  // NULL is ok here, it just means we made the current one fiature the base solid
329  static_cast<PartDesign::Feature*>(feature)->BaseFeature.setValue(prevSolidFeature);

340 std::vector<App::DocumentObject*> Body::removeObject(App::DocumentObject* feature)
341 {
342  App::DocumentObject* nextSolidFeature = getNextSolidFeature(feature);
343  App::DocumentObject* prevSolidFeature = getPrevSolidFeature(feature);
344  // This method must be called BEFORE the feature is removed from the Document!
345  if (isSolidFeature(feature)) {
346  // This is a solid feature
347  // If the next feature is solid, reroute its BaseFeature property to the previous solid feature
348  if (nextSolidFeature) {
349  assert ( nextSolidFeature->isDerivedFrom ( PartDesign::Feature::getClassTypeId () ) );
350  // Note: It's ok to remove the first solid feature, that just mean the next feature become the base one
351  static_cast<PartDesign::Feature*>(nextSolidFeature)->BaseFeature.setValue(prevSolidFeature);
352  }
353  }
354 
355  std::vector<App::DocumentObject*> model = Group.getValues();
356  std::vector<App::DocumentObject*>::iterator it = std::find(model.begin(), model.end(), feature);
357 
358  // Adjust Tip feature if it is pointing to the deleted object
359  if (Tip.getValue()== feature) {
360  if (prevSolidFeature) {
361  Tip.setValue(prevSolidFeature);
362  } else {
363  Tip.setValue(nextSolidFeature);
364  }
365  }
366 
367  // Erase feature from Group
368  if (it != model.end()) {
369  model.erase(it);
370  Group.setValues(model);
371  }
372  std::vector<App::DocumentObject*> result = {feature};
373  return result;
374 }

433 void Body::onChanged(const App::Property* prop) {
434  // we neither load a project nor perform undo/redo
435  if (!this->isRestoring()
436  && this->getDocument()
437  && !this->getDocument()->isPerformingTransaction()) {
438  if (prop == &BaseFeature) {
439  FeatureBase* bf = nullptr;
440  auto first = Group.getValues().empty() ? nullptr : Group.getValues().front();
441 
442  if (BaseFeature.getValue()) {
443  //setup the FeatureBase if needed
444  if (!first || !first->isDerivedFrom(FeatureBase::getClassTypeId())) {
445  bf = static_cast<FeatureBase*>(getDocument()->addObject("PartDesign::FeatureBase", "BaseFeature"));
446  insertObject(bf, first, false);
447 
448  if (!Tip.getValue())
449  Tip.setValue(bf);
450  }
451  else {
452  bf = static_cast<FeatureBase*>(first);
453  }
454  }
455 
456  if (bf && (bf->BaseFeature.getValue() != BaseFeature.getValue())) {
457  bf->BaseFeature.setValue(BaseFeature.getValue());
458  }
459  }
460  else if (prop == &Group) {
461  //if the FeatureBase was deleted we set the BaseFeature link to nullptr
462  if (BaseFeature.getValue() &&
463  (Group.getValues().empty() || !Group.getValues().front()->isDerivedFrom(FeatureBase::getClassTypeId()))) {
464  BaseFeature.setValue(nullptr);
465  }
466  }
467  else if (prop == &AllowCompound) {
468  // As disallowing compounds can break the model we need to recompute the whole tree.
469  // This will inform user about first place where there is more than one solid.
470  // On allowing compounds we must also recompute the entire feature tree
471  for (auto feature : getFullModel()) {
472  feature->enforceRecompute();
473  }
474 
475  }
476  else if (prop == &ShapeMaterial) {
477  std::vector<App::DocumentObject*> features = Group.getValues();
478  if (!features.empty()) {
479  for (auto it : features) {
480  auto feature = dynamic_cast<Part::Feature*>(it);
481  if (feature) {
482  if (feature->ShapeMaterial.getValue().getUUID()
483  != ShapeMaterial.getValue().getUUID()) {
484  feature->ShapeMaterial.setValue(ShapeMaterial.getValue());
485  }
486  }
487  }
488  }
489  }
490  }
491 
492  Part::BodyBase::onChanged(prop);
493 }

518 App::DocumentObject *Body::getSubObject(const char *subname,
519  PyObject **pyObj, Base::Matrix4D *pmat, bool transform, int depth) const
520 {
521  while (subname && *subname == '.') {
522  ++subname; // skip leading .
523  }
524 
525  // PartDesign::Feature now support grouping sibling features, and the user
526  // is free to expand/collapse at any time. To not disrupt subname path
527  // because of this, the body will peek the next two sub-objects reference,
528  // and skip the first sub-object if possible.
529  if (subname) {
530  const char* firstDot = strchr(subname, '.');
531  if (firstDot) {
532  const char* secondDot = strchr(firstDot + 1, '.');
533  if (secondDot) {
534  auto firstObj = Group.find(std::string(subname, firstDot).c_str());
535  if (!firstObj || firstObj->isDerivedFrom(PartDesign::Feature::getClassTypeId())) {
536  auto secondObj = Group.find(std::string(firstDot + 1, secondDot).c_str());
537  if (secondObj) {
538  // we support only one level of sibling grouping, so no
539  // recursive call to our own getSubObject()
540  return Part::BodyBase::getSubObject(firstDot + 1,
541  pyObj,
542  pmat,
543  transform,
544  depth + 1);
545  }
546  }
547  }
548  }
549  }
550 #if 1
551  return Part::BodyBase::getSubObject(subname,pyObj,pmat,transform,depth);
552 #else
553  // The following code returns Body shape only if there is at least one
554  // child visible in the body (when show through, not show tip). The
555  // original intention is to sync visual to shape returned by
556  // Part.getShape() when the body is included in some other group. But this
557  // interfere with direct modeling using body shape. Therefore it is
558  // disabled here.
559 
560  if(!pyObj || showTip ||
561  (subname && !Data::ComplexGeoData::isMappedElement(subname) && strchr(subname,'.')))
562  return Part::BodyBase::getSubObject(subname,pyObj,pmat,transform,depth);
563 
564  // We return the shape only if there are feature visible inside
565  for(auto obj : Group.getValues()) {
566  if(obj->Visibility.getValue() &&
567  obj->isDerivedFrom(PartDesign::Feature::getClassTypeId()))
568  {
569  return Part::BodyBase::getSubObject(subname,pyObj,pmat,transform,depth);
570  }
571  }
572  if(pmat && transform)
573  *pmat *= Placement.getValue().toMatrix();
574  return const_cast<Body*>(this);
575 #endif
