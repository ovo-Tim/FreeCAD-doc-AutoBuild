121 double Toolpath::getLength()
122 {
123  if (vpcCommands.empty()) {
124  return 0;
125  }
126  double l = 0;
127  Vector3d last(0, 0, 0);
128  Vector3d next;
129  for (std::vector<Command*>::const_iterator it = vpcCommands.begin(); it != vpcCommands.end();
130  ++it) {
131  std::string name = (*it)->Name;
132  next = (*it)->getPlacement(last).getPosition();
133  if ((name == "G0") || (name == "G00") || (name == "G1") || (name == "G01")) {
134  // straight line
135  l += (next - last).Length();
136  last = next;
137  }
138  else if ((name == "G2") || (name == "G02") || (name == "G3") || (name == "G03")) {
139  // arc
140  Vector3d center = (*it)->getCenter();
141  double radius = (last - center).Length();
142  double angle = (next - center).GetAngle(last - center);
143  l += angle * radius;
144  last = next;
145  }
146  }
147  return l;
148 }

150 double Toolpath::getCycleTime(double hFeed, double vFeed, double hRapid, double vRapid)
151 {
152  // check the feedrates are set
153  if ((hFeed == 0) || (vFeed == 0)) {
154  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
155  "User parameter:BaseApp/Preferences/Mod/CAM");
156  if (!hGrp->GetBool("WarningsSuppressAllSpeeds", true)) {
157  Base::Console().Warning("Feed Rate Error: Check Tool Controllers have Feed Rates");
158  }
159  return 0;
160  }
161 
162  if (hRapid == 0) {
163  hRapid = hFeed;
164  }
165 
166  if (vRapid == 0) {
167  vRapid = vFeed;
168  }
169 
170  if (vpcCommands.empty()) {
171  return 0;
172  }
173  double l = 0;
174  double time = 0;
175  bool verticalMove = false;
176  Vector3d last(0, 0, 0);
177  Vector3d next;
178  for (std::vector<Command*>::const_iterator it = vpcCommands.begin(); it != vpcCommands.end();
179  ++it) {
180  std::string name = (*it)->Name;
181  float feedrate = (*it)->getParam("F");
182 
183  l = 0;
184  verticalMove = false;
185  feedrate = hFeed;
186  next = (*it)->getPlacement(last).getPosition();
187 
188  if (last.z != next.z) {
189  verticalMove = true;
190  feedrate = vFeed;
191  }
192 
193  if ((name == "G0") || (name == "G00")) {
194  // Rapid Move
195  l += (next - last).Length();
196  feedrate = hRapid;
197  if (verticalMove) {
198  feedrate = vRapid;
199  }
200  }
201  else if ((name == "G1") || (name == "G01")) {
202  // Feed Move
203  l += (next - last).Length();
204  }
205  else if ((name == "G2") || (name == "G02") || (name == "G3") || (name == "G03")) {
206  // Arc Move
207  Vector3d center = (*it)->getCenter();
208  double radius = (last - center).Length();
209  double angle = (next - center).GetAngle(last - center);
210  l += angle * radius;
211  }
212 
213  time += l / feedrate;
214  last = next;
215  }
216  return time;
217 }

328 void Toolpath::setFromGCode(const std::string instr)
329 {
330  clear();
331 
332  // remove comments
333  // boost::regex e("\\(.*?\\)");
334  // std::string str = boost::regex_replace(instr, e, "");
335  std::string str(instr);
336 
337  // split input string by () or G or M commands
338  std::string mode = "command";
339  std::size_t found = str.find_first_of("(gGmM");
340  int last = -1;
341  bool inches = false;
342  while (found != std::string::npos) {
343  if (str[found] == '(') {
344  // start of comment
345  if ((last > -1) && (mode == "command")) {
346  // before opening a comment, add the last found command
347  std::string gcodestr = str.substr(last, found - last);
348  bulkAddCommand(gcodestr, vpcCommands, inches);
349  }
350  mode = "comment";
351  last = found;
352  found = str.find_first_of(')', found + 1);
353  }
354  else if (str[found] == ')') {
355  // end of comment
356  std::string gcodestr = str.substr(last, found - last + 1);
357  bulkAddCommand(gcodestr, vpcCommands, inches);
358  last = -1;
359  found = str.find_first_of("(gGmM", found + 1);
360  mode = "command";
361  }
362  else if (mode == "command") {
363  // command
364  if (last > -1) {
365  std::string gcodestr = str.substr(last, found - last);
366  bulkAddCommand(gcodestr, vpcCommands, inches);
367  }
368  last = found;
369  found = str.find_first_of("(gGmM", found + 1);
370  }
371  }
372  // add the last command found, if any
373  if (last > -1) {
374  if (mode == "command") {
375  std::string gcodestr = str.substr(last, std::string::npos);
376  bulkAddCommand(gcodestr, vpcCommands, inches);
377  }
378  }
379  recalculate();
380 }
