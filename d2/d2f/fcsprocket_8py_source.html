78 def midpoints(p1, p2, r):
79  """A very ugly function that returns the midpoint of a p1 and p2
80  on the circle which radius is r and which pass through p1 and
81  p2
82 
83  Return the 2 possible solutions
84  """
85  vx, vy = p2[0]-p1[0], p2[1]-p1[1]
86  b = (vx**2 + vy**2)**.5
87  v = (vx/b, vy/b)
88  cosA = b**2 / (2*b*r)
89  A = acos(cosA)
90 
91  vx, vy = rotate(v, A)
92  c1 = (p1[0]+r*vx, p1[1]+r*vy)
93  m1x, m1y = ((p1[0]+p2[0])/2 - c1[0], (p1[1]+p2[1])/2 - c1[1])
94  dm1 = (m1x**2+m1y**2)**.5
95  m1x, m1y = (c1[0] + r*m1x/dm1, c1[1] + r*m1y/dm1)
96  m1 = (m1x, m1y)
97 
98  vx, vy = rotate(v, -A)
99  c2 = (p1[0]+r*vx, p1[1]+r*vy)
100  m2x, m2y = ((p1[0]+p2[0])/2 - c2[0], (p1[1]+p2[1])/2 - c2[1])
101  dm2 = (m2x**2+m2y**2)**.5
102  m2x, m2y = (c2[0] + r*m2x/dm2, c2[1] + r*m2y/dm2)
103  m2 = (m2x, m2y)
104 
105  return m1, m2
