114  SbRotation getTurntable(SbRotation rot, const SbVec3f &point1, const SbVec3f &point2) const
115  {
116  // 0000333: Turntable camera rotation
117  SbVec3f axis;
118  float angle{};
119  rot.getValue(axis, angle);
120  SbVec3f dif = point1 - point2;
121  if (fabs(dif[1]) > fabs(dif[0])) {
122  SbVec3f xaxis(1,0,0);
123  if (dif[1] < 0) {
124  angle = -angle;
125  }
126  rot.setValue(xaxis, angle);
127  }
128  else {
129  SbVec3f zaxis(0,0,1);
130  this->worldToScreen.multDirMatrix(zaxis, zaxis);
131  if (zaxis[1] < 0) {
132  if (dif[0] < 0) {
133  angle = -angle;
134  }
135  }
136  else {
137  if (dif[0] > 0) {
138  angle = -angle;
139  }
140  }

142  }

344  if (!camera) {
345  return;
346  }

390 {
391  SoCamera* camera = getCamera();
392  if (!camera)
393  return;
394 
395  animator->stop();
396 
397  SbVec3f focalPoint = getFocalPoint();
398  SbVec3f translation(0, 0, 0);
399 
400  if (moveToCenter) {
401  SoGetBoundingBoxAction action(viewer->getSoRenderManager()->getViewportRegion());
402  action.apply(viewer->getSceneGraph());
403  SbBox3f box = action.getBoundingBox();
404  if (!box.isEmpty()) {
405  translation = box.getCenter() - focalPoint;
406  }
407  }
408 
409  // Start an animation or set the pose directly
410  if (isAnimationEnabled()) {
411  viewer->startAnimation(orientation, focalPoint, translation);
412  }
413  else {
414  // Distance from rotation center to camera position in camera coordinate system
415  SbVec3f rotationCenterDistanceCam = camera->focalDistance.getValue() * SbVec3f(0, 0, 1);
416 
417  // Set to the given orientation
418  camera->orientation = orientation;
419 
420  // Distance from rotation center to new camera position in global coordinate system
421  SbVec3f newRotationCenterDistance;
422  camera->orientation.getValue().multVec(rotationCenterDistanceCam, newRotationCenterDistance);
423

425  // Optionally add translation to move to center

447 void NavigationStyle::boxZoom(const SbBox2s& box)
448 {
449  SoCamera* cam = viewer->getSoRenderManager()->getCamera();
450  if (!cam) // no camera
451  return;
452  const SbViewportRegion & vp = viewer->getSoRenderManager()->getViewportRegion();
453  SbViewVolume vv = cam->getViewVolume(vp.getViewportAspectRatio());
454 
455  short sizeX{},sizeY{};
456  box.getSize(sizeX, sizeY);
457  SbVec2s size = vp.getViewportSizePixels();
458 
459  // The bbox must not be empty i.e. width and length is zero, but it is possible that
460  // either width or length is zero
461  if (sizeX == 0 && sizeY == 0)
462  return;
463 
464  // Get the new center in normalized pixel coordinates
465  short xmin{},xmax{},ymin{},ymax{};
466  box.getBounds(xmin,ymin,xmax,ymax);
467  const SbVec2f center((float) ((xmin+xmax)/2) / (float) std::max((int)(size[0] - 1), 1),
468  (float) (size[1]-(ymin+ymax)/2) / (float) std::max((int)(size[1] - 1), 1));
469 
470  SbPlane plane = vv.getPlane(cam->focalDistance.getValue());
471  panCamera(cam,vp.getViewportAspectRatio(),plane, SbVec2f(0.5,0.5), center);
472 
473  // Set height or height angle of the camera
474  float scaleX = (float)sizeX/(float)size[0];
475  float scaleY = (float)sizeY/(float)size[1];
476  float scale = std::max<float>(scaleX, scaleY);
477  if (cam->getTypeId() == SoOrthographicCamera::getClassTypeId()) {
478  float height = static_cast<SoOrthographicCamera*>(cam)->height.getValue() * scale;
479  static_cast<SoOrthographicCamera*>(cam)->height = height;
480  }
481  else if (cam->getTypeId() == SoPerspectiveCamera::getClassTypeId()) {
482  float height = static_cast<SoPerspectiveCamera*>(cam)->heightAngle.getValue() / 2.0f;
483  height = 2.0f * atan(tan(height) * scale);
484  static_cast<SoPerspectiveCamera*>(cam)->heightAngle = height;
485  }
486 }

488 void NavigationStyle::viewAll()
489 {
490  // Get the bounding box of the scene
491  SoGetBoundingBoxAction action(viewer->getSoRenderManager()->getViewportRegion());
492  action.apply(viewer->getSceneGraph());
493  SbBox3f box = action.getBoundingBox();
494  if (box.isEmpty())
495  return;
496 
497 
498  SoCamera* cam = viewer->getSoRenderManager()->getCamera();
499  if (!cam)
500  return;
501 
502  SbViewVolume vol = cam->getViewVolume();
503  if (vol.ulf == vol.llf)
504  return; // empty frustum (no view up vector defined)
505  SbVec2f s = vol.projectBox(box);
506  SbVec2s size = viewer->getSoRenderManager()->getSize();
507 
508  SbVec3f pt1, pt2, pt3, tmp;
509  vol.projectPointToLine( SbVec2f(0.0f,0.0f), pt1, tmp );
510  vol.projectPointToLine( SbVec2f(s[0],0.0f), pt2, tmp );
511  vol.projectPointToLine( SbVec2f(0.0f,s[1]), pt3, tmp );
512 
513  float cam_width = (pt2-pt1).length();
514  float cam_height = (pt3-pt1).length();
515 
516  // add a small border
517  cam_height = 1.08f * std::max<float>((cam_width*(float)size[1])/(float)size[0],cam_height);
518 
519  float aspect = cam->aspectRatio.getValue();
520 
521  if (cam->getTypeId() == SoOrthographicCamera::getClassTypeId()) {
522  auto ocam = static_cast<SoOrthographicCamera *>(cam);
523  if (aspect < 1.0f)
524  ocam->height = cam_height / aspect;
525  else
526  ocam->height = cam_height;
527  }

550 void NavigationStyle::reorientCamera(SoCamera* camera, const SbRotation& rotation, const SbVec3f& rotationCenter)
551 {
552  if (!camera) {
553  return;
554  }
555 
556  // Distance from rotation center to camera position in camera coordinate system
557  SbVec3f rotationCenterDistanceCam;
558  camera->orientation.getValue().inverse().multVec(camera->position.getValue() - rotationCenter, rotationCenterDistanceCam);
559 
560  // Set new orientation value by accumulating the new rotation
561  camera->orientation = rotation * camera->orientation.getValue();
562 
563  // Distance from rotation center to new camera position in global coordinate system
564  SbVec3f newRotationCenterDistance;
565  camera->orientation.getValue().multVec(rotationCenterDistanceCam, newRotationCenterDistance);
566 
567  // Reposition camera so the rotation center stays in the same place
568  camera->position = rotationCenter + newRotationCenterDistance;
569 
570 #if (COIN_MAJOR_VERSION * 100 + COIN_MINOR_VERSION * 10 + COIN_MICRO_VERSION < 403)
571  // Fix issue with near clipping in orthogonal view
572  if (camera->getTypeId().isDerivedFrom(SoOrthographicCamera::getClassTypeId())) {
573 
574  // The center of the bounding sphere in camera coordinate system
575  SbVec3f center;
576  camera->orientation.getValue().inverse().multVec(boundingSphere.getCenter() - camera->position.getValue(), center);
577 
578  SbVec3f dir;
579  camera->orientation.getValue().multVec(SbVec3f(0, 0, -1), dir);
580 
581  // Reposition the camera but keep the focal point the same
582  // nearDistance is 0 and farDistance is the diameter of the bounding sphere
583  float repositionDistance = -center.getValue()[2] - boundingSphere.getRadius();
584  camera->position = camera->position.getValue() + repositionDistance * dir;
585  camera->nearDistance = 0;
586  camera->farDistance = 2 * boundingSphere.getRadius() + 1;
587  camera->focalDistance = camera->focalDistance.getValue() - repositionDistance;
588  }

592 void NavigationStyle::panCamera(SoCamera * cam, float aspectratio, const SbPlane & panplane,
593  const SbVec2f & currpos, const SbVec2f & prevpos)
594 {
595  if (!cam) // can happen for empty scenegraph
596  return;
597  if (currpos == prevpos) // useless invocation
598  return;
599 
600 
601  // Find projection points for the last and current mouse coordinates.
602  SbViewVolume vv = cam->getViewVolume(aspectratio);
603 
604  // See note in Coin docs for SoCamera::getViewVolume re:viewport mapping
605  if(aspectratio < 1.0)
606  vv.scale(1.0 / aspectratio);
607 
608  SbLine line;
609  vv.projectPointToLine(currpos, line);
610  SbVec3f current_planept;
611  panplane.intersect(line, current_planept);
612  vv.projectPointToLine(prevpos, line);
613  SbVec3f old_planept;
614  panplane.intersect(line, old_planept);
615 
616  // Reposition camera according to the vector difference between the
617  // projected points.
618  cam->position = cam->position.getValue() - (current_planept - old_planept);
619 
620  if (this->currentmode != NavigationStyle::IDLE) {
621  hasPanned = true;
622  }

686  first = false;

700  // user zooms the camera out to a distance from origo larger than
701  // what we still can safely do floating point calculations on
702  // (i.e. without getting NaN or Inf values), the faulty floating

704  // messages and eventually an assert failure from core Coin code.

707  //
708  // (But note that we depend on the input argument ''diffvalue'' to
709  // be small enough that zooming happens gradually. Ideally, we

711  // inversely; isinfite()), but those only became standardized with
712  // C99.)
713  const float distorigo = newpos.length();
714  // sqrt(FLT_MAX) == ~ 1e+19, which should be both safe for further
715  // calculations and ok for the end-user and app-programmer.
716  if (distorigo > float(sqrt(FLT_MAX))) {
717  // do nothing here
718  }
719  else {
720  cam->position = newpos;

866 void NavigationStyle::spin(const SbVec2f & pointerpos)
867 {

869  return;
870  assert(this->spinprojector);
871 
872  const SbViewportRegion & vp = viewer->getSoRenderManager()->getViewportRegion();
873  SbVec2s glsize(vp.getViewportSizePixels());
874  SbVec2f lastpos;
875  lastpos[0] = float(this->log.position[1][0]) / float(std::max((int)(glsize[0]-1), 1));
876  lastpos[1] = float(this->log.position[1][1]) / float(std::max((int)(glsize[1]-1), 1));
877 
878  float sensitivity = getSensitivity();
879 
880  // Adjust the spin projector sphere to the screen position of the rotation center when the mouse intersects an object
881  if (getOrbitStyle() == Trackball && rotationCenterMode & RotationCenterMode::ScenePointAtCursor && rotationCenterFound && rotationCenterIsScenePointAtCursor) {
882  const auto pointOnScreen = viewer->getPointOnViewport(rotationCenter);
883  const auto sphereCenter = 2 * normalizePixelPos(pointOnScreen) - SbVec2f {1, 1};
884 
885  float x, y;
886  sphereCenter.getValue(x, y);
887 
888  const float sphereScale = 1 + sphereCenter.length();
889  const float radius = FCSphereSheetProjector::defaultSphereRadius * sphereScale;
890  sensitivity *= sphereScale;
891 
892  spinprojector->setSphere(SbSphere {SbVec3f {x, y, 0}, radius});
893  }
894  else {
895  spinprojector->setSphere(SbSphere {SbVec3f {0, 0, 0}, FCSphereSheetProjector::defaultSphereRadius});
896  }
897 
898  if (this->rotationCenterMode && this->rotationCenterFound) {
899  SbVec3f hitpoint = this->rotationCenter;
900 
901  // set to the given position
902  SbVec3f direction;
903  viewer->getSoRenderManager()->getCamera()->orientation.getValue().multVec(SbVec3f(0, 0, -1), direction);
904  viewer->getSoRenderManager()->getCamera()->position = hitpoint - viewer->getSoRenderManager()->getCamera()->focalDistance.getValue() * direction;
905  }
906 
907  // 0000333: Turntable camera rotation
908  SbMatrix mat;
909  viewer->getSoRenderManager()->getCamera()->orientation.getValue().getValue(mat);
910  this->spinprojector->setWorkingSpace(mat);
911 
912  this->spinprojector->project(lastpos);
913  SbRotation r;
914  this->spinprojector->projectAndGetRotation(pointerpos, r);
915  if (sensitivity > 1.0f) {
916  SbVec3f axis;
917  float radians{};
918  r.getValue(axis, radians);
919  radians = sensitivity * radians;
920  r.setValue(axis, radians);
921  }
922  r.invert();
923  this->reorientCamera(viewer->getSoRenderManager()->getCamera(), r);
924 
925  if (this->rotationCenterMode && this->rotationCenterFound) {
926  float ratio = vp.getViewportAspectRatio();
927  SbViewVolume vv = viewer->getSoRenderManager()->getCamera()->getViewVolume(vp.getViewportAspectRatio());
928  SbPlane panplane = vv.getPlane(viewer->getSoRenderManager()->getCamera()->focalDistance.getValue());
929  SbVec2f posn;
930  posn[0] = float(this->localPos[0]) / float(std::max((int)(glsize[0]-1), 1));
931  posn[1] = float(this->localPos[1]) / float(std::max((int)(glsize[1]-1), 1));
932  panCamera(viewer->getSoRenderManager()->getCamera(), ratio, panplane, posn, SbVec2f(0.5,0.5));
933  }
934 
935  // Calculate an average angle magnitude value to make the transition
936  // to a possible spin animation mode appear smooth.
937 
938  SbVec3f dummy_axis, newaxis;
939  float acc_angle{}, newangle{};
940  this->spinincrement.getValue(dummy_axis, acc_angle);
941  acc_angle *= this->spinsamplecounter; // weight
942  r.getValue(newaxis, newangle);
943  acc_angle += newangle;
944 
945  this->spinsamplecounter++;
946  acc_angle /= this->spinsamplecounter;
947  // FIXME: accumulate and average axis vectors as well? 19990501 mortene.
948  this->spinincrement.setValue(newaxis, acc_angle);
949 
950  // Don't carry too much baggage, as that'll give unwanted results
951  // when the user quickly trigger (as in "click-drag-release") a spin
952  // animation.
953  if (this->spinsamplecounter > 3) this->spinsamplecounter = 3;
954 
955  if (this->currentmode != NavigationStyle::IDLE) {
956  hasDragged = true;
957  }
958 }
