89 App::DocumentObjectExecReturn *MultiFuse::execute()
90 {
91  std::vector<TopoShape> shapes;
92  std::vector<App::DocumentObject*> obj = Shapes.getValues();
93 
94  std::vector<App::DocumentObject*>::iterator it;
95  for (it = obj.begin(); it != obj.end(); ++it) {
96  shapes.push_back(Feature::getTopoShape(*it));
97  }
98 
99  bool argumentsAreInCompound = false;
100  TopoShape compoundOfArguments;
101 
102  // if only one source shape, and it is a compound - fuse children of the compound
103  const int maxIterations = 1'000'000; // will trigger "not enough shape objects linked" error below if ever reached
104  for (int i = 0; shapes.size() == 1 && i < maxIterations; ++i) {
105  compoundOfArguments = shapes[0];
106  if (compoundOfArguments.getShape().ShapeType() == TopAbs_COMPOUND) {
107  shapes.clear();
108  shapes = compoundOfArguments.getSubTopoShapes();
109  argumentsAreInCompound = true;
110  } else {
111  break;
112  }
113  }
114 
115  if (shapes.size() >= 2) {
116  try {
117  std::vector<ShapeHistory> history;
118  FCBRepAlgoAPI_Fuse mkFuse;
119  TopTools_ListOfShape shapeArguments, shapeTools;
120  const TopoShape& shape = shapes.front();
121  if (shape.isNull()) {
122  throw Base::RuntimeError("Input shape is null");
123  }
124  shapeArguments.Append(shape.getShape());
125 
126  for (auto it2 = shapes.begin() + 1; it2 != shapes.end(); ++it2) {
127  if (it2->isNull()) {
128  throw Base::RuntimeError("Input shape is null");
129  }
130  shapeTools.Append(it2->getShape());
131  }
132 
133  mkFuse.SetArguments(shapeArguments);
134  mkFuse.SetTools(shapeTools);
135  mkFuse.setAutoFuzzy();
136  mkFuse.Build();
137 
138  if (!mkFuse.IsDone()) {
139  throw Base::RuntimeError("MultiFusion failed");
140  }
141 
142  TopoShape res(0);
143  res = res.makeShapeWithElementMap(mkFuse.Shape(), MapperMaker(mkFuse), shapes, OpCodes::Fuse);
144  for (const auto& it2 : shapes) {
145  history.push_back(
146  buildHistory(mkFuse, TopAbs_FACE, res.getShape(), it2.getShape()));
147  }
148  if (res.isNull()) {
149  throw Base::RuntimeError("Resulting shape is null");
150  }
151 
152  throwIfInvalidIfCheckModel(res.getShape());
153 
154  if (this->Refine.getValue()) {
155  try {
156  TopoDS_Shape oldShape = res.getShape();
157  BRepBuilderAPI_RefineModel mkRefine(oldShape);
158  // We just built an element map above for the fuse, don't erase it for a refine.
159  res.setShape(mkRefine.Shape(), false);
160  ShapeHistory hist =
161  buildHistory(mkRefine, TopAbs_FACE, res.getShape(), oldShape);
162  for (auto& jt : history) {
163  jt = joinHistory(jt, hist);
164  }
165  }
166  catch (Standard_Failure&) {
167  // do nothing
168  }
169  }
170 
171  if (argumentsAreInCompound) {
172  // combine histories of every child of source compound into one
173  ShapeHistory overallHist;
174  TopTools_IndexedMapOfShape facesOfCompound;
175  TopAbs_ShapeEnum type = TopAbs_FACE;
176  TopExp::MapShapes(compoundOfArguments.getShape(), type, facesOfCompound);
177  for (std::size_t iChild = 0; iChild < history.size();
178  iChild++) { // loop over children of source compound
179  // for each face of a child, find the inex of the face in compound, and assign
180  // the corresponding right-hand-size of the history
181  TopTools_IndexedMapOfShape facesOfChild;
182  TopExp::MapShapes(shapes[iChild].getShape(), type, facesOfChild);
183  for (std::pair<const int, ShapeHistory::List>& histitem :
184  history[iChild].shapeMap) { // loop over elements of history - that is -
185  // over faces of the child of source compound
186  int iFaceInChild = histitem.first;
187  ShapeHistory::List& iFacesInResult = histitem.second;
188  const TopoDS_Shape& srcFace = facesOfChild(
189  iFaceInChild
190  + 1); //+1 to convert our 0-based to OCC 1-bsed conventions
191  int iFaceInCompound = facesOfCompound.FindIndex(srcFace) - 1;
192  overallHist.shapeMap[iFaceInCompound] =
193  iFacesInResult; // this may overwrite existing info if the same face is
194  // used in several children of compound. This shouldn't
195  // be a problem, because the histories should match
196  // anyway...
197  }
198  }
199  history.clear();
200  history.push_back(overallHist);
201  }
202  this->Shape.setValue(res);
203  this->History.setValues(history);
204 
205  App::DocumentObject* link = Shapes.getValues()[0];
206  copyMaterial(link);
207  return Part::Feature::execute();
208  }
209  catch (Standard_Failure& e) {
210  return new App::DocumentObjectExecReturn(e.GetMessageString());
211  }
212  }
213  else {
214  throw Base::CADKernelError("Not enough shape objects linked");
215  }
216 }
