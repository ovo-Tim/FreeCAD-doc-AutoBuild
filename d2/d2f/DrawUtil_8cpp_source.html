276 bool DrawUtil::isWithinRange(double actualAngleIn, double targetAngleIn, double allowableError)
277 {
278  constexpr double DegreesPerRevolution{360};
279  constexpr double DegreesPerHalfRevolution{180};
280  // map both angles from [0, 360] to [-180, 180]. This solves the problem of
281  // comparing angles near 0, such as 5deg & 355deg where the desired answer is
282  // 10, not 350;
283  double actualAngleDeg = actualAngleIn;
284  if (actualAngleDeg < DegreesPerRevolution &&
285  actualAngleDeg > DegreesPerHalfRevolution) {
286  actualAngleDeg = actualAngleDeg - DegreesPerRevolution;
287  }
288 
289  double targetAngleDeg = targetAngleIn;
290  if (targetAngleDeg < DegreesPerRevolution &&
291  targetAngleDeg > DegreesPerHalfRevolution) {
292  targetAngleDeg = targetAngleDeg - DegreesPerRevolution;
293  }
294 
295  double actualError = fabs(targetAngleDeg - actualAngleDeg);
296  if (actualError <= allowableError) {
297  return true;
298  }
299 
300  return false;
301 }

322 std::pair<Base::Vector3d, Base::Vector3d>
323 DrawUtil::boxIntersect2d(Base::Vector3d point, Base::Vector3d dirIn, double xRange, double yRange)
324 {
325  std::pair<Base::Vector3d, Base::Vector3d> result;
326  Base::Vector3d p1, p2;
327  Base::Vector3d dir = dirIn;
328  dir.Normalize();
329  // y = mx + b
330  // m = (y1 - y0) / (x1 - x0)
331  if (DrawUtil::fpCompare(dir.x, 0.0)) {//vertical case
332  p1 = Base::Vector3d(point.x, point.y - (yRange / 2.0), 0.0);
333  p2 = Base::Vector3d(point.x, point.y + (yRange / 2.0), 0.0);
334  } else {
335  double slope = dir.y / dir.x;
336  double left = -xRange / 2.0;
337  double right = xRange / 2.0;
338  if (DrawUtil::fpCompare(slope, 0.0)) {//horizontal case
339  p1 = Base::Vector3d(point.x - (xRange / 2.0), point.y);
340  p2 = Base::Vector3d(point.x + (xRange / 2.0), point.y);
341  } else {//normal case
342  double top = yRange / 2.0;
343  double bottom = -yRange / 2.0;
344  double yLeft = point.y - slope * (point.x - left);
345  double yRight = point.y - slope * (point.x - right);
346  double xTop = point.x - ((point.y - top) / slope);
347  double xBottom = point.x - ((point.y - bottom) / slope);
348 
349  if ((bottom < yLeft) && (top > yLeft)) {
350  p1 = Base::Vector3d(left, yLeft);
351  } else if (yLeft <= bottom) {
352  p1 = Base::Vector3d(xBottom, bottom);
353  } else if (yLeft >= top) {
354  p1 = Base::Vector3d(xTop, top);
355  }
356 
357  if ((bottom < yRight) && (top > yRight)) {
358  p2 = Base::Vector3d(right, yRight);
359  } else if (yRight <= bottom) {
360  p2 = Base::Vector3d(xBottom, bottom);
361  } else if (yRight >= top) {
362  p2 = Base::Vector3d(xTop, top);
363  }
364  }
365  }
366  result.first = p1;
367  result.second = p2;
368  Base::Vector3d dirCheck = p2 - p1;
369  dirCheck.Normalize();
370  if (!dir.IsEqual(dirCheck, 0.00001)) {
371  result.first = p2;
372  result.second = p1;
373  }
374 
375  return result;
376 }

397 bool DrawUtil::apparentIntersection(TopoDS_Edge& edge0, TopoDS_Edge& edge1, gp_Pnt& intersect)
398 {
399  gp_Pnt gStart0 = BRep_Tool::Pnt(TopExp::FirstVertex(edge0));
400  gp_Pnt gEnd0 = BRep_Tool::Pnt(TopExp::LastVertex(edge0));
401  gp_Pnt gStart1 = BRep_Tool::Pnt(TopExp::FirstVertex(edge1));
402  gp_Pnt gEnd1 = BRep_Tool::Pnt(TopExp::LastVertex(edge1));
403 
404  //intersection of 2 3d lines in point&direction form
405  //https://math.stackexchange.com/questions/270767/find-intersection-of-two-3d-lines
406  gp_Vec C(gStart0.XYZ());
407  gp_Vec D(gStart1.XYZ());
408  gp_Vec e(gEnd0.XYZ() - gStart0.XYZ());//direction of line0
409  gp_Vec f(gEnd1.XYZ() - gStart1.XYZ());//direction of line1
410  Base::Console().Message(
411  "DU::apparentInter - e: %s f: %s\n", formatVector(e).c_str(), formatVector(f).c_str());
412 
413  //check for cases the algorithm doesn't handle well
414  gp_Vec C1(gEnd0.XYZ());
415  gp_Vec D1(gEnd1.XYZ());
416  if (C.IsEqual(D, EWTOLERANCE, EWTOLERANCE) || C.IsEqual(D1, EWTOLERANCE, EWTOLERANCE)) {
417  intersect = gp_Pnt(C.XYZ());
418  return true;
419  }
420  if (C1.IsEqual(D, EWTOLERANCE, EWTOLERANCE) || C1.IsEqual(D1, EWTOLERANCE, EWTOLERANCE)) {
421  intersect = gp_Pnt(C1.XYZ());
422  return true;
423  }
424 
425  gp_Vec g(D - C);//between a point on each line
426  Base::Console().Message("DU::apparentInter - C: %s D: %s g: %s\n",
427  formatVector(C).c_str(),
428  formatVector(D).c_str(),
429  formatVector(g).c_str());
430 
431  gp_Vec fxg = f.Crossed(g);
432  double h = fxg.Magnitude();
433  gp_Vec fxe = f.Crossed(e);
434  double k = fxe.Magnitude();
435  Base::Console().Message("DU::apparentInter - h: %.3f k: %.3f\n", h, k);
436  if (fpCompare(k, 0.0)) {
437  //no intersection
438  return false;
439  }
440  gp_Vec el = e * (h / k);
441  double pm = 1.0;
442  if (fpCompare(fxg.Dot(fxe), -1.0)) {
443  //opposite directions
444  pm = -1.0;
445  }
446  intersect = gp_Pnt(C.XYZ() + el.XYZ() * pm);
447  return true;
448 }

452 bool DrawUtil::intersect2Lines3d(Base::Vector3d point0, Base::Vector3d dir0, Base::Vector3d point1,
453  Base::Vector3d dir1, Base::Vector3d& intersect)
454 {

456  Base::Vector3d fxg = dir1.Cross(g);
457  Base::Vector3d fxgn = fxg;
458  fxgn.Normalize();
459  Base::Vector3d fxe = dir1.Cross(dir0);
460  Base::Vector3d fxen = fxe;
461  fxen.Normalize();
462  Base::Vector3d dir0n = dir0;
463  dir0n.Normalize();
464  Base::Vector3d dir1n = dir1;
465  dir1n.Normalize();
466  if (fabs(dir0n.Dot(dir1n)) == 1.0) {
467  //parallel lines, no intersection

470  }

473  double direction = -1.0;
474  if (fxgn == fxen) {
475  direction = 1.0;
476  }
477

1440 std::pair<int, int> DrawUtil::nearestFraction(double val, int maxDenom)
1441 {
1442 // Find rational approximation to given real number
1443 // David Eppstein / UC Irvine / 8 Aug 1993
1444 //
1445 // With corrections from Arno Formella, May 2008
1446 // and additional fiddles by WF 2017
1447 // usage: a.out r d
1448 // r is real number to approx
1449 // d is the maximum denominator allowed
1450 //
1451 // Based on the theory of continued fractions
1452 // if x = a1 + 1/(a2 + 1/(a3 + 1/(a4 + ...)))
1453 // then best approximation is found by truncating this series
1454 // (with some adjustments in the last term).
1455 //
1456 // Note the fraction can be recovered as the first column of the matrix
1457 // ( a1 1 ) ( a2 1 ) ( a3 1 ) ...
1458 // ( 1 0 ) ( 1 0 ) ( 1 0 )
1459 // Instead of keeping the sequence of continued fraction terms,
1460 // we just keep the last partial product of these matrices.
1461  std::pair<int, int> result;
1462  long m[2][2];
1463  long maxden = maxDenom;
1464  long ai;
1465  double x = val;
1466  double startx = x;
1467 
1468  /* initialize matrix */
1469  m[0][0] = m[1][1] = 1;
1470  m[0][1] = m[1][0] = 0;
1471 
1472  /* loop finding terms until denom gets too big */
1473  while (m[1][0] * ( ai = (long)x ) + m[1][1] <= maxden) {
1474  long t;
1475  t = m[0][0] * ai + m[0][1];
1476  m[0][1] = m[0][0];
1477  m[0][0] = t;
1478  t = m[1][0] * ai + m[1][1];
1479  m[1][1] = m[1][0];
1480  m[1][0] = t;
1481  if(x == (double) ai)
1482  break; // AF: division by zero
1483  x = 1/(x - (double) ai);
1484  if(x > (double) std::numeric_limits<int>::max())
1485  break; // AF: representation failure
1486  }
1487 
1488  /* now remaining x is between 0 and 1/ai */
1489  /* approx as either 0 or 1/m where m is max that will fit in maxden */
1490  /* first try zero */
1491  double error1 = startx - ((double) m[0][0] / (double) m[1][0]);
1492  int n1 = m[0][0];
1493  int d1 = m[1][0];
1494 
1495  /* now try other possibility */
1496  ai = (maxden - m[1][1]) / m[1][0];
1497  m[0][0] = m[0][0] * ai + m[0][1];
1498  m[1][0] = m[1][0] * ai + m[1][1];
1499  double error2 = startx - ((double) m[0][0] / (double) m[1][0]);
1500  int n2 = m[0][0];
1501  int d2 = m[1][0];
1502 
1503  if (std::fabs(error1) <= std::fabs(error2)) {
1504  result.first = n1;
1505  result.second = d1;
1506  } else {
1507  result.first = n2;
1508  result.second = d2;
1509  }
1510 
1511  return result;
1512 }
