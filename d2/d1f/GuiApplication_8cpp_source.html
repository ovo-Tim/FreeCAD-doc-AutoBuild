71 bool GUIApplication::notify (QObject * receiver, QEvent * event)
72 {
73  if (!receiver) {
74  Base::Console().Log("GUIApplication::notify: Unexpected null receiver, event type: %d\n",
75  (int)event->type());
76  return false;
77  }
78 
79  // https://github.com/FreeCAD/FreeCAD/issues/16905
80  std::string exceptionWarning =
81 #if FC_DEBUG
82  "Exceptions must be caught before they go through Qt."
83  " Ignoring this will cause crashes on some systems.\n";
84 #else
85  "";
86 #endif
87 
88  try {
89  if (event->type() == Spaceball::ButtonEvent::ButtonEventType ||
90  event->type() == Spaceball::MotionEvent::MotionEventType)
91  return processSpaceballEvent(receiver, event);
92  else
93  return QApplication::notify(receiver, event);
94  }
95  catch (const Base::SystemExitException &e) {
96  caughtException.reset(new Base::SystemExitException(e));
97  qApp->exit(e.getExitCode());
98  return true;
99  }
100  catch (const Base::Exception& e) {
101  Base::Console().Error("Unhandled Base::Exception caught in GUIApplication::notify.\n"
102  "The error message is: %s\n%s", e.what(), exceptionWarning);
103  }
104  catch (const std::exception& e) {
105  Base::Console().Error("Unhandled std::exception caught in GUIApplication::notify.\n"
106  "The error message is: %s\n%s", e.what(), exceptionWarning);
107  }
108  catch (...) {
109  Base::Console().Error("Unhandled unknown exception caught in GUIApplication::notify.\n%s",
110  exceptionWarning);
111  }
112 
113  // Print some more information to the log file (if active) to ease bug fixing
114  try {
115  std::stringstream dump;
116  dump << "The event type " << (int)event->type() << " was sent to "
117  << receiver->metaObject()->className() << "\n";
118  dump << "Object tree:\n";
119  if (receiver->isWidgetType()) {
120  QWidget* w = qobject_cast<QWidget*>(receiver);
121  while (w) {
122  dump << "\t";
123  dump << w->metaObject()->className();
124  QString name = w->objectName();
125  if (!name.isEmpty())
126  dump << " (" << (const char*)name.toUtf8() << ")";
127  w = w->parentWidget();
128  if (w)
129  dump << " is child of\n";
130  }
131  std::string str = dump.str();
132  Base::Console().Log("%s",str.c_str());
133  }
134  }
135  catch (...) {
136  Base::Console().Log("Invalid recipient and/or event in GUIApplication::notify\n");
137  }
138 
139  return true;
140 }

159 bool GUIApplication::event(QEvent * ev)
160 {
161  if (ev->type() == QEvent::FileOpen) {
162  // (macOS workaround when opening FreeCAD by opening a .FCStd file in 1.0)
163  // With the current implementation of the splash screen boot procedure, Qt will
164  // start an event loop before FreeCAD is fully initialized. This event loop will
165  // process the QFileOpenEvent that is sent by macOS before the main window is ready.
166  if (!Gui::getMainWindow()->property("eventLoop").toBool()) {
167  // If we never reach this point when opening FreeCAD by double clicking an
168  // .FCStd file, then the workaround isn't needed anymore and can be removed
169  QEvent* eventCopy = new QFileOpenEvent(static_cast<QFileOpenEvent*>(ev)->file());
170  QTimer::singleShot(0, [eventCopy, this]() {
171  QCoreApplication::postEvent(this, eventCopy);
172  });
173  return true;
174  }
175 
176  QString file = static_cast<QFileOpenEvent*>(ev)->file();
177  QFileInfo fi(file);
178  if (fi.suffix().toLower() == QLatin1String("fcstd")) {
179  QByteArray fn = file.toUtf8();
180  Application::Instance->open(fn, "FreeCAD");
181  return true;
182  }
183  }
184 
185  return GUIApplicationNativeEventAware::event(ev);
186 }

220  void startServer()
221  {
222  // Start a QLocalServer to listen for connections
223  server = new QLocalServer();
224  QObject::connect(server, &QLocalServer::newConnection,
225  q_ptr, &GUISingleApplication::receiveConnection);
226  // first attempt
227  if (!server->listen(serverName)) {
228  if (server->serverError() == QAbstractSocket::AddressInUseError) {
229  // second attempt
230  server->removeServer(serverName);
231  server->listen(serverName);
232  }
233  }
234  if (server->isListening()) {
235  Base::Console().Log("Local server '%s' started\n", qPrintable(serverName));
236  }
237  else {
238  Base::Console().Log("Local server '%s' failed to start\n", qPrintable(serverName));
239  }
240  }

265 bool GUISingleApplication::sendMessage(const QString &message, int timeout)
266 {
267  QLocalSocket socket;
268  bool connected = false;
269  for(int i = 0; i < 2; i++) {
270  socket.connectToServer(d_ptr->serverName);
271  connected = socket.waitForConnected(timeout/2);
272  if (connected || i > 0) {
273  break;
274  }
275  int ms = 250;
276 #if defined(Q_OS_WIN)
277  Sleep(DWORD(ms));
278 #else
279  usleep(ms*1000);
280 #endif
281  }
282  if (!connected) {
283  return false;
284  }
285 
286  QTextStream ts(&socket);
287 #if QT_VERSION <= QT_VERSION_CHECK(6, 0, 0)
288  ts.setCodec("UTF-8");
289 #else
290  ts.setEncoding(QStringConverter::Utf8);
291 #endif
292 #if QT_VERSION <= QT_VERSION_CHECK(5, 15, 0)
293  ts << message << endl;
294 #else
295  ts << message << Qt::endl;
296 #endif
297 
298  return socket.waitForBytesWritten(timeout);
299 }

301 void GUISingleApplication::readFromSocket()
302 {
303  auto socket = qobject_cast<QLocalSocket*>(sender());
304  if (socket) {
305  QTextStream in(socket);
306 #if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
307  in.setCodec("UTF-8");
308 #else
309  in.setEncoding(QStringConverter::Utf8);
310 #endif
311  while (socket->canReadLine()) {
312  d_ptr->timer->stop();
313  QString message = in.readLine();
314  Base::Console().Log("Received message: %s\n", message.toStdString());
315  d_ptr->messages.push_back(message);
316  d_ptr->timer->start(1000);
317  }
318  }
319 }

347 bool WheelEventFilter::eventFilter(QObject* obj, QEvent* ev)
348 {
349  if (qobject_cast<QComboBox*>(obj) && ev->type() == QEvent::Wheel)
350  return true;
351  auto sb = qobject_cast<QAbstractSpinBox*>(obj);
352  if (sb) {
353  if (ev->type() == QEvent::Show) {
354  sb->setFocusPolicy(Qt::StrongFocus);
355  }
356  else if (ev->type() == QEvent::Wheel) {
357  return !sb->hasFocus();
358  }
359  }
360  return false;
361 }
