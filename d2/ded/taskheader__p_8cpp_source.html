142 void TaskHeader::animate()
143 {
144  if (!myScheme->headerAnimation)
145  return;
146 
147  if (!isEnabled()) {
148  m_opacity = 0.1;
149  update();
150  return;
151  }
152 
153  if (m_over) {
154  if (m_opacity >= 0.3) {
155  m_opacity = 0.3;
156  return;

158  m_opacity += 0.05;
159  } else {
160  if (m_opacity <= 0.1) {

162  return;

164  m_opacity = qMax(0.1, m_opacity-0.05);
165  }
166 
167  QTimer::singleShot(100, this, &TaskHeader::animate);
