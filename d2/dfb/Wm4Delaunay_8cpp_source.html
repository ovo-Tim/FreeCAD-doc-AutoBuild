94 template <class Real>
95 bool Delaunay<Real>::Load (FILE* pkIFile)
96 {
97  WM4_DELETE[] m_aiIndex;
98  WM4_DELETE[] m_aiAdjacent;
99 
100  // fixed-size members
101  int iQueryType;
102  System::Read4le(pkIFile,1,&iQueryType);
103  m_eQueryType = (Query::Type)iQueryType;
104  System::Read4le(pkIFile,1,&m_iVertexQuantity);
105  System::Read4le(pkIFile,1,&m_iDimension);
106  System::Read4le(pkIFile,1,&m_iSimplexQuantity);
107  System::Read4le(pkIFile,1,&m_fEpsilon);
108 
109  // variable-size members
110  int iIQuantity;
111  System::Read4le(pkIFile,1,&iIQuantity);
112  if (1 <= m_iDimension && m_iDimension <= 3)
113  {
114  assert(iIQuantity == (m_iDimension+1)*m_iSimplexQuantity);
115  m_aiIndex = WM4_NEW int[iIQuantity];
116  m_aiAdjacent = WM4_NEW int[iIQuantity];
117  System::Read4le(pkIFile,iIQuantity,m_aiIndex);
118  System::Read4le(pkIFile,iIQuantity,m_aiAdjacent);
119  return true;
120  }
121 
122  m_aiIndex = nullptr;
123  m_aiAdjacent = nullptr;
124  return m_iDimension == 0;
125 }

135  System::Write4le(pkOFile,1,&m_iSimplexQuantity);
136  System::Write4le(pkOFile,1,&m_fEpsilon);
137 
138  // The member m_bOwner is not streamed because on a Load call, this
139  // object will allocate the vertices and own this memory.
140 
141  // variable-size members
142  int iIQuantity;
143  if (1 <= m_iDimension && m_iDimension <= 3)
144  {
145  iIQuantity = (m_iDimension+1)*m_iSimplexQuantity;
146  System::Write4le(pkOFile,1,&iIQuantity);
147  System::Write4le(pkOFile,iIQuantity,m_aiIndex);
148  System::Write4le(pkOFile,iIQuantity,m_aiAdjacent);
149  return true;
150  }
151 
152  iIQuantity = 0;
153  System::Write4le(pkOFile,1,&iIQuantity);
154  return m_iDimension == 0;
155 }
