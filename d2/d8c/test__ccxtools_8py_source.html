340  def input_file_writing_test(
341  self,
342  base_name,
343  analysis_dir=None,
344  test_end=False,
345  ):
346  fcc_print(f"\n--------------- Start of FEM ccxtools {base_name} test---------------")
347 
348  if analysis_dir is None:
349  analysis_dir = testtools.get_fem_test_tmp_dir(self.pre_dir_name + base_name)
350  analysis = self.document.Analysis
351  solver_object = self.document.CalculiXCcxTools
352  fea = ccxtools.FemToolsCcx(analysis, solver_object, test_mode=True)
353  fea.update_objects()
354 
355  fcc_print(f"Setting up working directory {analysis_dir}")
356  fea.setup_working_dir(analysis_dir)
357  self.assertTrue(
358  True if fea.working_dir == analysis_dir else False,
359  f"Setting working directory {analysis_dir} failed",
360  )
361 
362  fcc_print(f"Checking FEM inp file prerequisites for {base_name} ...")
363  error = fea.check_prerequisites()
364  self.assertFalse(
365  error,
366  f"ccxtools check_prerequisites returned error message: {error}",
367  )
368 
369  inpfile_given = join(self.test_file_dir, (base_name + ".inp"))
370  inpfile_totest = join(analysis_dir, (self.mesh_name + ".inp"))
371  fcc_print("Checking FEM inp file write...")
372  fcc_print(f"Writing {inpfile_totest} for {base_name}")
373  error = fea.write_inp_file()
374  self.assertFalse(error, "Writing failed")
375 
376  fcc_print(f"Comparing {inpfile_given} to {inpfile_totest}")
377  ret = testtools.compare_inp_files(inpfile_given, inpfile_totest)
378  self.assertFalse(ret, f"ccxtools write_inp_file test failed.\n{ret}")
379 
380  if test_end is True:
381  # do not save and print End of tests
382  return fea
383 
384  save_fc_file = join(analysis_dir, base_name + ".FCStd")
385  fcc_print(f"Save FreeCAD file for {base_name} to {save_fc_file}...")
386  self.document.saveAs(save_fc_file)
387 
388  fcc_print(f"\n--------------- End of FEM ccxtools {base_name}---------------")
