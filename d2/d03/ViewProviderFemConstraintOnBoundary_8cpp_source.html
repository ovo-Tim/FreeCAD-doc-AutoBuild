44 void ViewProviderFemConstraintOnBoundary::highlightReferences(const bool on)
45 {
46  Fem::Constraint* pcConstraint = this->getObject<Fem::Constraint>();
47  const auto& subSets = pcConstraint->References.getSubListValues();
48 
49  for (auto& subSet : subSets) {
50  Part::Feature* base = dynamic_cast<Part::Feature*>(subSet.first);
51  if (!base) {
52  continue;
53  }
54  PartGui::ViewProviderPart* vp = dynamic_cast<PartGui::ViewProviderPart*>(
55  Gui::Application::Instance->getViewProvider(base));
56  if (!vp) {
57  continue;
58  }
59 
60  // if somehow the subnames are empty, clear any existing colors
61  if (on && !subSet.second.empty()) {
62  // identify the type of subelements
63  // TODO: Assumed here the subelements are of the same type.
64  // It is a requirement but we should keep safeguards.
65  if (subSet.second[0].find("Vertex") != std::string::npos) {
66  // make sure original colors are remembered
67  if (originalPointColors[base].empty()) {
68  originalPointColors[base] = vp->PointColorArray.getValues();
69  }
70  std::vector<App::Color> colors = originalPointColors[base];
71 
72  // go through the subelements with constraint and recolor them
73  // TODO: Replace `ShapeAppearance` with anything more appropriate
74  PartGui::ReferenceHighlighter highlighter(
75  base->Shape.getValue(),
76  colors.empty() ? ShapeAppearance.getDiffuseColor() : colors[0]);
77  highlighter.getVertexColors(subSet.second, colors);
78  vp->PointColorArray.setValues(colors);
79  }
80  else if (subSet.second[0].find("Edge") != std::string::npos) {
81  // make sure original colors are remembered
82  if (originalLineColors[base].empty()) {
83  originalLineColors[base] = vp->LineColorArray.getValues();
84  }
85  std::vector<App::Color> colors = originalLineColors[base];
86 
87  // go through the subelements with constraint and recolor them
88  // TODO: Replace `ShapeAppearance` with anything more appropriate
89  PartGui::ReferenceHighlighter highlighter(
90  base->Shape.getValue(),
91  colors.empty() ? ShapeAppearance.getDiffuseColor() : colors[0]);
92  highlighter.getEdgeColors(subSet.second, colors);
93  vp->LineColorArray.setValues(colors);
94  }
95  else if (subSet.second[0].find("Face") != std::string::npos) {
96  // make sure original colors are remembered
97  if (originalFaceColors[base].empty()) {
98  originalFaceColors[base] = vp->ShapeAppearance.getDiffuseColors();
99  }
100  std::vector<App::Color> colors = originalFaceColors[base];
101 
102  // go through the subelements with constraint and recolor them
103  // TODO: Replace shape DiffuseColor with anything more appropriate
104  PartGui::ReferenceHighlighter highlighter(
105  base->Shape.getValue(),
106  colors.empty() ? ShapeAppearance.getDiffuseColor() : colors[0]);
107  highlighter.getFaceColors(subSet.second, colors);
108  vp->ShapeAppearance.setDiffuseColors(colors);
109  }
110  }
111  else {
112  if (!originalPointColors[base].empty()) {
113  vp->PointColorArray.setValues(originalPointColors[base]);
114  originalPointColors[base].clear();
115  }
116  else if (!originalLineColors[base].empty()) {
117  vp->LineColorArray.setValues(originalLineColors[base]);
118  originalLineColors[base].clear();
119  }
120  else if (!originalFaceColors[base].empty()) {
121  vp->ShapeAppearance.setDiffuseColors(originalFaceColors[base]);
122  originalFaceColors[base].clear();
123  }
124  }
125  }
126 
127  if (subSets.empty()) {
128  // there is nothing selected but previous selection may have highlighting
129  // reset that highlighting here
130  for (auto& ogPair : originalPointColors) {
131  if (ogPair.second.empty()) {
132  continue;
133  }
134  PartGui::ViewProviderPart* vp = dynamic_cast<PartGui::ViewProviderPart*>(
135  Gui::Application::Instance->getViewProvider(ogPair.first));
136  if (!vp) {
137  continue;
138  }
139 
140  vp->PointColorArray.setValues(ogPair.second);
141  ogPair.second.clear();
142  }
143 
144  for (auto& ogPair : originalLineColors) {
145  if (ogPair.second.empty()) {
146  continue;
147  }
148  PartGui::ViewProviderPart* vp = dynamic_cast<PartGui::ViewProviderPart*>(
149  Gui::Application::Instance->getViewProvider(ogPair.first));
150  if (!vp) {
151  continue;
152  }
153 
154  vp->LineColorArray.setValues(ogPair.second);
155  ogPair.second.clear();
156  }
157 
158  for (auto& ogPair : originalFaceColors) {
159  if (ogPair.second.empty()) {
160  continue;
161  }
162  PartGui::ViewProviderPart* vp = dynamic_cast<PartGui::ViewProviderPart*>(
163  Gui::Application::Instance->getViewProvider(ogPair.first));
164  if (!vp) {
165  continue;
166  }
167 
168  vp->ShapeAppearance.setDiffuseColors(ogPair.second);
169  ogPair.second.clear();
170  }
171  }
172 }
