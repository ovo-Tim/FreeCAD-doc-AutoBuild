60 int BlendPointPy::PyInit(PyObject* args, PyObject*)
61 {
62  PyObject* plist;
63  std::vector<Base::Vector3d> vecs;
64  if (PyArg_ParseTuple(args, "O", &plist)) {
65  Py::Sequence list(plist);
66  if (list.size() == 0) {
67  vecs.emplace_back(Base::Vector3d(0, 0, 0));
68  }
69  else {
70  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
71  Py::Vector v(*it);
72  Base::Vector3d vec = v.toVector();
73  vecs.emplace_back(vec);
74  }
75  }
76  this->getBlendPointPtr()->vectors = vecs;
77  return 0;
78  }
79 
80  PyErr_Clear();
81  if (PyArg_ParseTuple(args, "")) {
82  vecs.emplace_back(Base::Vector3d(0, 0, 0));
83  this->getBlendPointPtr()->vectors = vecs;
84  return 0;
85  }
86 
87  double param;
88  int cont;
89  PyObject* pcObj;
90  PyErr_Clear();
91  // Create a curve with an edge, parameter and continiuity.
92  if (PyArg_ParseTuple(args, "O!di", &(Part::TopoShapePy::Type), &pcObj, &param, &cont)) {
93  try {
94  gp_Pnt Pt;
95  TopoDS_Shape shape =
96  static_cast<Part::TopoShapePy*>(pcObj)->getTopoShapePtr()->getShape();
97  const TopoDS_Edge& e = TopoDS::Edge(shape);
98  BRepAdaptor_Curve adapt(e);
99  if (param < adapt.FirstParameter() || param > adapt.LastParameter()) {
100  PyErr_Warn(PyExc_UserWarning, "BlendPoint: edge is not a closed curve");
101  Base::Console().Message("fp=%f\n", adapt.FirstParameter());
102  Base::Console().Message("lp=%f\n", adapt.LastParameter());
103  }
104 
105  adapt.D0(param, Pt);
106  Base::Vector3d bv(Pt.X(), Pt.Y(), Pt.Z());
107  vecs.emplace_back(bv);
108 
109  for (int i = 1; i <= cont; i++) {
110  gp_Vec v1 = adapt.DN(param, i);
111  Base::Vector3d bbv1(v1.X(), v1.Y(), v1.Z());
112  vecs.emplace_back(bbv1);
113  }
114  this->getBlendPointPtr()->vectors = vecs;
115  return 0;
116  }
117  catch (const std::exception& e) {
118  PyErr_SetString(PyExc_RuntimeError, e.what());
119  return -1;
120  }
121  }
122 
123  PyErr_SetString(PyExc_TypeError,
124  "supported signatures:\n"
125  "BlendPoint()\n"
126  "BlendPoint(list of Vector)\n"
127  "BlendPoint(edge, parameter and continiuity)\n");
128  return -1;
129 }
