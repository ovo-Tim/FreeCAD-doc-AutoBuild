74 def _traverseTemplateAttributes(attrs, codec):
75  Path.Log.debug(attrs)
76  coded = {}
77  for key, value in attrs.items():
78  if type(value) == dict:
79  Path.Log.debug("%s is a dict" % key)
80  coded[key] = _traverseTemplateAttributes(value, codec)
81  elif type(value) == list:
82  Path.Log.debug("%s is a list" % key)
83  coded[key] = [_traverseTemplateAttributes(attr, codec) for attr in value]
84  elif isinstance(value, str):
85  Path.Log.debug("%s is a string" % key)
86  coded[key] = codec(value)
87  else:
88  Path.Log.debug("%s is %s" % (key, type(value)))
89  coded[key] = value
90  return coded

319  def expressionReference(self):
320  """expressionReference() ... returns the string to be used in expressions"""
321  # Using the Name here and not the Label (both would be valid) because the Name 'fails early'.
322  #
323  # If there is a Name/Label conflict and an expression is bound to the Name we'll get an error
324  # on creation (Property not found). Not good, but at least there's some indication that
325  # something's afoul.
326  #
327  # If the expression is based on the Label everything works out nicely - until the document is
328  # saved and loaded from disk. The Labels change in order to avoid the Name/Label conflict
329  # but the expression stays the same. If the user's lucky the expression is broken because the
330  # conflicting object doesn't have the properties reference by the expressions. If the user is
331  # not so lucky those properties also exist in the other object, there is no indication that
332  # anything is wrong but the expressions will substitute the values from the wrong object.
333  #
334  # I prefer the question: "why do I get this error when I create ..." over "my cnc machine just
335  # rammed it's tool head into the table ..." or even "I saved my file and now it's corrupt..."
336  #
337  # https://forum.freecad.org/viewtopic.php?f=10&t=24839
338  # https://forum.freecad.org/viewtopic.php?f=10&t=24845
339  return self.obj.Name
