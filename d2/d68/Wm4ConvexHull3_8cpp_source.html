21 template <class Real>
22 ConvexHull3<Real>::ConvexHull3 (int iVertexQuantity, Vector3<Real>* akVertex,
23  Real fEpsilon, bool bOwner, Query::Type eQueryType)
24  :
25  ConvexHull<Real>(iVertexQuantity,fEpsilon,bOwner,eQueryType),
26  m_kLineOrigin(Vector3<Real>::ZERO),
27  m_kLineDirection(Vector3<Real>::ZERO),
28  m_kPlaneOrigin(Vector3<Real>::ZERO)
29 {
30  assert(akVertex);
31  m_akVertex = akVertex;
32  m_akPlaneDirection[0] = Vector3<Real>::ZERO;
33  m_akPlaneDirection[1] = Vector3<Real>::ZERO;
34  m_akSVertex = nullptr;
35  m_pkQuery = nullptr;
36 
37  Mapper3<Real> kMapper(m_iVertexQuantity,m_akVertex,m_fEpsilon);
38  if (kMapper.GetDimension() == 0)
39  {
40  // The values of m_iDimension, m_aiIndex, and m_aiAdjacent were
41  // already initialized by the ConvexHull base class.
42  return;
43  }
44 
45  if (kMapper.GetDimension() == 1)
46  {
47  // The set is (nearly) collinear. The caller is responsible for
48  // creating a ConvexHull1 object.
49  m_iDimension = 1;
50  m_kLineOrigin = kMapper.GetOrigin();
51  m_kLineDirection = kMapper.GetDirection(0);
52  return;
53  }
54 
55  if (kMapper.GetDimension() == 2)
56  {
57  // The set is (nearly) coplanar. The caller is responsible for
58  // creating a ConvexHull2 object.
59  m_iDimension = 2;
60  m_kPlaneOrigin = kMapper.GetOrigin();
61  m_akPlaneDirection[0] = kMapper.GetDirection(0);
62  m_akPlaneDirection[1] = kMapper.GetDirection(1);
63  return;
64  }
65 
66  m_iDimension = 3;
67 
68  int i0 = kMapper.GetExtremeIndex(0);
69  int i1 = kMapper.GetExtremeIndex(1);
70  int i2 = kMapper.GetExtremeIndex(2);
71  int i3 = kMapper.GetExtremeIndex(3);
72 
73  m_akSVertex = WM4_NEW Vector3<Real>[m_iVertexQuantity];
74  int i;
75 
76  if (eQueryType != Query::QT_RATIONAL && eQueryType != Query::QT_FILTERED)
77  {
78  // Transform the vertices to the cube [0,1]^3.
79  Vector3<Real> kMin = kMapper.GetMin();
80  Real fScale = ((Real)1.0)/kMapper.GetMaxRange();
81  for (i = 0; i < m_iVertexQuantity; i++)
82  {
83  m_akSVertex[i] = (m_akVertex[i] - kMin)*fScale;
84  }
85 
86  Real fExpand;
87  if (eQueryType == Query::QT_INT64)
88  {
89  // Scale the vertices to the square [0,2^{20}]^2 to allow use of
90  // 64-bit integers.
91  fExpand = (Real)(1 << 20);
92  m_pkQuery = WM4_NEW Query3Int64<Real>(m_iVertexQuantity,
93  m_akSVertex);
94  }
95  else if (eQueryType == Query::QT_INTEGER)
96  {
97  // Scale the vertices to the square [0,2^{24}]^2 to allow use of
98  // TInteger.
99  fExpand = (Real)(1 << 24);
100  m_pkQuery = WM4_NEW Query3TInteger<Real>(m_iVertexQuantity,
101  m_akSVertex);
102  }
103  else // eQueryType == Query::QT_REAL
104  {
105  // No scaling for floating point.
106  fExpand = (Real)1.0;
107  m_pkQuery = WM4_NEW Query3<Real>(m_iVertexQuantity,m_akSVertex);
108  }
109 
110  for (i = 0; i < m_iVertexQuantity; i++)
111  {
112  m_akSVertex[i] *= fExpand;
113  }
114  }
115  else
116  {
117  // No transformation needed for exact rational arithmetic or filtered
118  // predicates.
119  size_t uiSize = m_iVertexQuantity*sizeof(Vector3<Real>);
120  System::Memcpy(m_akSVertex,uiSize,m_akVertex,uiSize);
121 
122  if (eQueryType == Query::QT_RATIONAL)
123  {
124  m_pkQuery = WM4_NEW Query3TRational<Real>(m_iVertexQuantity,
125  m_akSVertex);
126  }
127  else // eQueryType == Query::QT_FILTERED
128  {
129  m_pkQuery = WM4_NEW Query3Filtered<Real>(m_iVertexQuantity,
130  m_akSVertex,m_fEpsilon);
131  }
132  }
133 
134  Triangle* pkT0;
135  Triangle* pkT1;
136  Triangle* pkT2;
137  Triangle* pkT3;
138 
139  if (kMapper.GetExtremeCCW())
140  {
141  pkT0 = WM4_NEW Triangle(i0,i1,i3);
142  pkT1 = WM4_NEW Triangle(i0,i2,i1);
143  pkT2 = WM4_NEW Triangle(i0,i3,i2);
144  pkT3 = WM4_NEW Triangle(i1,i2,i3);
145  pkT0->AttachTo(pkT1,pkT3,pkT2);
146  pkT1->AttachTo(pkT2,pkT3,pkT0);
147  pkT2->AttachTo(pkT0,pkT3,pkT1);
148  pkT3->AttachTo(pkT1,pkT2,pkT0);
149  }
150  else
151  {
152  pkT0 = WM4_NEW Triangle(i0,i3,i1);
153  pkT1 = WM4_NEW Triangle(i0,i1,i2);
154  pkT2 = WM4_NEW Triangle(i0,i2,i3);
155  pkT3 = WM4_NEW Triangle(i1,i3,i2);
156  pkT0->AttachTo(pkT2,pkT3,pkT1);
157  pkT1->AttachTo(pkT0,pkT3,pkT2);
158  pkT2->AttachTo(pkT1,pkT3,pkT0);
159  pkT3->AttachTo(pkT0,pkT2,pkT1);
160  }
161 
162  m_kHull.clear();
163  m_kHull.insert(pkT0);
164  m_kHull.insert(pkT1);
165  m_kHull.insert(pkT2);
166  m_kHull.insert(pkT3);
167 
168  for (i = 0; i < m_iVertexQuantity; i++)
169  {
170  if (!Update(i))
171  {
172  DeleteHull();
173  return;
174  }
175  }
176 
177  ExtractIndices();
178 }

257 template <class Real>
258 bool ConvexHull3<Real>::Update (int i)
259 {
260  // Locate a triangle visible to the input point (if possible).
261  Triangle* pkVisible = nullptr;
262  Triangle* pkTri;
263  typename std::set<Triangle*>::iterator pkIter;
264  for (pkIter = m_kHull.begin(); pkIter != m_kHull.end(); pkIter++)
265  {
266  pkTri = *pkIter;
267  if (pkTri->GetSign(i,m_pkQuery) > 0)
268  {
269  pkVisible = pkTri;
270  break;
271  }
272  }
273 
274  if (!pkVisible)
275  {
276  // The point is inside the current hull; nothing to do.
277  return true;
278  }
279 
280  // Locate and remove the visible triangles.
281  std::stack<Triangle*> kVisible;
282  std::map<int,TerminatorData> kTerminator;
283  kVisible.push(pkVisible);
284  pkVisible->OnStack = true;
285  int j, iV0, iV1;
286  while (!kVisible.empty())
287  {
288  pkTri = kVisible.top();
289  kVisible.pop();
290  pkTri->OnStack = false;
291  for (j = 0; j < 3; j++)
292  {
293  Triangle* pkAdj = pkTri->A[j];
294  if (pkAdj)
295  {
296  // Detach triangle and adjacent triangle from each other.
297  int iNullIndex = pkTri->DetachFrom(j,pkAdj);
298 
299  if (pkAdj->GetSign(i,m_pkQuery) > 0)
300  {
301  if (!pkAdj->OnStack)
302  {
303  // Adjacent triangle is visible.
304  kVisible.push(pkAdj);
305  pkAdj->OnStack = true;
306  }
307  }
308  else
309  {
310  // Adjacent triangle is invisible.
311  iV0 = pkTri->V[j];
312  iV1 = pkTri->V[(j+1)%3];
313  kTerminator[iV0] = TerminatorData(iV0,iV1,iNullIndex,
314  pkAdj);
315  }
316  }
317  }
318  m_kHull.erase(pkTri);
319  WM4_DELETE pkTri;
320  }
321 
322  // Insert the new edges formed by the input point and the terminator
323  // between visible and invisible triangles.
324  int iSize = (int)kTerminator.size();
325  assert(iSize >= 3);
326  typename std::map<int,TerminatorData>::iterator pkEdge =
327  kTerminator.begin();
328  iV0 = pkEdge->second.V[0];
329  iV1 = pkEdge->second.V[1];
330  pkTri = WM4_NEW Triangle(i,iV0,iV1);
331  m_kHull.insert(pkTri);
332 
333  // save information for linking first/last inserted new triangles
334  int iSaveV0 = pkEdge->second.V[0];
335  Triangle* pkSaveTri = pkTri;
336 
337  // establish adjacency links across terminator edge
338  pkTri->A[1] = pkEdge->second.Tri;
339  pkEdge->second.Tri->A[pkEdge->second.NullIndex] = pkTri;
340  for (j = 1; j < iSize; j++)
341  {
342  pkEdge = kTerminator.find(iV1);
343  assert(pkEdge != kTerminator.end());
344  iV0 = iV1;
345  iV1 = pkEdge->second.V[1];
346  Triangle* pkNext = WM4_NEW Triangle(i,iV0,iV1);
347  m_kHull.insert(pkNext);
348 
349  // establish adjacency links across terminator edge
350  pkNext->A[1] = pkEdge->second.Tri;
351  pkEdge->second.Tri->A[pkEdge->second.NullIndex] = pkNext;
352 
353  // establish adjacency links with previously inserted triangle
354  pkNext->A[0] = pkTri;
355  pkTri->A[2] = pkNext;
356 
357  pkTri = pkNext;
358  }
359  assert(iV1 == iSaveV0);
360  (void)iSaveV0; // avoid warning in Release build
361 
362  // establish adjacency links between first/last triangles
363  pkSaveTri->A[0] = pkTri;
364  pkTri->A[2] = pkSaveTri;
365 
366  return true;
367 }

369 template <class Real>
370 void ConvexHull3<Real>::ExtractIndices ()
371 {
372  int iTQuantity = (int)m_kHull.size();
373  m_iSimplexQuantity = iTQuantity;
374  m_aiIndex = WM4_NEW int[3*m_iSimplexQuantity];
375 
376  typename std::set<Triangle*>::iterator pkIter;
377  int i = 0;
378  for (pkIter = m_kHull.begin(); pkIter != m_kHull.end(); pkIter++)
379  {
380  Triangle* pkTri = *pkIter;
381  for (int j = 0; j < 3; j++)
382  {
383  m_aiIndex[i++] = pkTri->V[j];
384  }
385  WM4_DELETE pkTri;
386  }
387  m_kHull.clear();
388 }

417 {

419  if (!pkIFile)
420  {
421  return false;
422  }
423 
424  ConvexHull<Real>::Load(pkIFile);
425 
426  WM4_DELETE m_pkQuery;

450  {
451  System::Read8le(pkIFile,iVQ,m_akVertex);
452  System::Read8le(pkIFile,iVQ,m_akSVertex);

479  m_pkQuery = WM4_NEW Query3Filtered<Real>(m_iVertexQuantity,
480  m_akSVertex,m_fEpsilon);

482  }
483 
484  return true;
485 }
