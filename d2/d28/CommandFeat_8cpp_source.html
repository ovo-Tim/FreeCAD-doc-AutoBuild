87 void StdCmdRandomColor::activated(int iMsg)
88 {
89  Q_UNUSED(iMsg);
90 
91  auto setRandomColor = [](ViewProvider* view) {
92  // NOLINTBEGIN
93  auto fMax = (float)RAND_MAX;
94  auto fRed = (float)rand()/fMax;
95  auto fGrn = (float)rand()/fMax;
96  auto fBlu = (float)rand()/fMax;
97  // NOLINTEND
98  auto objColor = App::Color(fRed, fGrn, fBlu);
99 
100  auto vpLink = dynamic_cast<ViewProviderLink*>(view);
101  if (vpLink) {
102  if (!vpLink->OverrideMaterial.getValue()) {
103  vpLink->OverrideMaterial.setValue(true);
104  }
105  vpLink->ShapeMaterial.setDiffuseColor(objColor);
106  }
107  else if (view) {
108  // clang-format off
109  // get the view provider of the selected object and set the shape color
110  if (auto prop = dynamic_cast<App::PropertyMaterialList*>(view->getPropertyByName("ShapeAppearance"))) {
111  prop->setDiffuseColor(objColor);
112  }
113  else if (auto prop = dynamic_cast<App::PropertyMaterial*>(view->getPropertyByName("ShapeAppearance"))) {
114  prop->setDiffuseColor(objColor);
115  }
116  // clang-format on
117  }
118  };
119 
120  auto allowToChangeColor = [](const App::DocumentObject* obj) {
121  return (obj->isDerivedFrom<App::Part>() || obj->isDerivedFrom<App::DocumentObjectGroup>());
122  };
123 
124  // get the complete selection
125  std::vector<SelectionSingleton::SelObj> sel = Selection().getCompleteSelection();
126 
127  Command::openCommand(QT_TRANSLATE_NOOP("Command", "Set Random Color"));
128  for (const auto & it : sel) {
129  ViewProvider* view = Application::Instance->getViewProvider(it.pObject);
130  setRandomColor(view);
131 
132  if (auto grp = it.pObject->getExtension<App::GroupExtension>()) {
133  if (allowToChangeColor(it.pObject)) {
134  std::vector<App::DocumentObject*> objs = grp->getObjects();
135  for (auto obj : objs) {
136  ViewProvider* view = Application::Instance->getViewProvider(obj);
137  setRandomColor(view);
138  }
139  }
140  }
141  }
142 
143  Command::commitCommand();
144 }

172 void StdCmdToggleFreeze::activated(int iMsg)
173 {
174  Q_UNUSED(iMsg);
175 
176  std::vector<Gui::SelectionSingleton::SelObj> sels = Gui::Selection().getCompleteSelection();
177 
178  Command::openCommand(QT_TRANSLATE_NOOP("Command", "Toggle freeze"));
179  for (Gui::SelectionSingleton::SelObj& sel : sels) {
180  App::DocumentObject* obj = sel.pObject;
181  if (!obj)
182  continue;
183 
184  if (obj->isFreezed()){
185  obj->unfreeze();
186  for (auto child : obj->getInListRecursive())
187  child->unfreeze();
188  } else {
189  obj->freeze();
190  for (auto parent : obj->getOutListRecursive())
191  parent->freeze();
192  }
193 
194  }
195  Command::commitCommand();
196 }

231 void StdCmdSendToPythonConsole::activated(int iMsg)
232 {
233  Q_UNUSED(iMsg);
234  const std::vector<Gui::SelectionObject> &sels = Gui::Selection().getSelectionEx("*", App::DocumentObject::getClassTypeId(),
235  ResolveMode::OldStyleElement, false);
236  if (sels.empty())
237  return;
238  const App::DocumentObject *obj = sels[0].getObject();
239  if (!obj)
240  return;
241  QString docname = QString::fromLatin1(obj->getDocument()->getName());
242  QString objname = QString::fromLatin1(obj->getNameInDocument());
243  try {
244  // clear variables from previous run, if any
245  QString cmd = QLatin1String("try:\n del(doc,lnk,obj,shp,sub,subs)\nexcept Exception:\n pass\n");
246  Gui::Command::runCommand(Gui::Command::Gui,cmd.toLatin1());
247  cmd = QString::fromLatin1("doc = App.getDocument(\"%1\")").arg(docname);
248  Gui::Command::runCommand(Gui::Command::Gui,cmd.toLatin1());
249  //support links
250  if (obj->isDerivedFrom<App::Link>()) {
251  cmd = QString::fromLatin1("lnk = doc.getObject(\"%1\")").arg(objname);
252  Gui::Command::runCommand(Gui::Command::Gui,cmd.toLatin1());
253  cmd = QString::fromLatin1("obj = lnk.getLinkedObject()");
254  Gui::Command::runCommand(Gui::Command::Gui,cmd.toLatin1());
255  const auto link = static_cast<const App::Link*>(obj);
256  obj = link->getLinkedObject();
257  } else {
258  cmd = QString::fromLatin1("obj = doc.getObject(\"%1\")").arg(objname);
259  Gui::Command::runCommand(Gui::Command::Gui,cmd.toLatin1());
260  }
261  if (obj->isDerivedFrom<App::GeoFeature>()) {
262  const auto geoObj = static_cast<const App::GeoFeature*>(obj);
263  const App::PropertyGeometry* geo = geoObj->getPropertyOfGeometry();
264  if (geo){
265  cmd = QString::fromLatin1("shp = obj.") + QLatin1String(geo->getName()); //"Shape", "Mesh", "Points", etc.
266  Gui::Command::runCommand(Gui::Command::Gui, cmd.toLatin1());
267  if (sels[0].hasSubNames()) {
268  std::vector<std::string> subnames = sels[0].getSubNames();
269  QString subname = QString::fromLatin1(subnames[0].c_str());
270  cmd = QString::fromLatin1("sub = obj.getSubObject(\"%1\")").arg(subname);
271  Gui::Command::runCommand(Gui::Command::Gui,cmd.toLatin1());
272  if (subnames.size() > 1) {
273  std::ostringstream strm;
274  strm << "subs = [";
275  for (const auto & subname : subnames) {
276  strm << "obj.getSubObject(\"" << subname << "\"),";
277  }
278  strm << "]";
279  Gui::Command::runCommand(Gui::Command::Gui, strm.str().c_str());
280  }
281  }
282  }
283  }
284  //show the python console if it's not already visible, and set the keyboard focus to it
285  QWidget* pc = DockWindowManager::instance()->getDockWindow("Python console");
286  auto pcPython = qobject_cast<PythonConsole*>(pc);
287  if (pcPython) {
288  DockWindowManager::instance()->activate(pcPython);
289  pcPython->setFocus();
290  }
291  }
292  catch (const Base::Exception& e) {
293  e.ReportException();
294  }
295 
296 }
