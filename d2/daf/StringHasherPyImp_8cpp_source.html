71 PyObject* StringHasherPy::getID(PyObject* args)
72 {
73  long id;
74  int index = 0;
75  if (PyArg_ParseTuple(args, "l|i", &id, &index)) {
76  if (id > 0) {
77  PY_TRY
78  {
79  auto sid = getStringHasherPtr()->getID(id, index);
80  if (!sid) {
81  Py_Return;
82  }
83 
84  return sid.getPyObject();
85  }
86  PY_CATCH;
87  }
88  else {
89  PyErr_SetString(PyExc_ValueError, "Id must be positive integer");
90  return nullptr;
91  }
92  }
93 
94  PyErr_Clear();
95  PyObject* value = nullptr;
96  PyObject* base64 = Py_False;
97  if (PyArg_ParseTuple(args, "O!|O!", &PyUnicode_Type, &value, &PyBool_Type, &base64)) {
98  PY_TRY
99  {
100  std::string txt = PyUnicode_AsUTF8(value);
101  QByteArray data;
102  StringIDRef sid;
103  if (PyObject_IsTrue(base64)) {
104  data = QByteArray::fromBase64(QByteArray::fromRawData(txt.c_str(), txt.size()));
105  sid = getStringHasherPtr()->getID(data, true);
106  }
107  else {
108  sid = getStringHasherPtr()->getID(txt.c_str(), txt.size());
109  }
110 
111  return sid.getPyObject();
112  }
113  PY_CATCH;
114  }
115 
116  PyErr_SetString(PyExc_TypeError,
117  "Positive integer and optional integer or "
118  "string and optional boolean is required");
119  return nullptr;
120 }
