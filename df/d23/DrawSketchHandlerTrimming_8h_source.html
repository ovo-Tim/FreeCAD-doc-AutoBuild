47 class TrimmingSelection: public Gui::SelectionFilterGate
48 {
49  App::DocumentObject* object;
50 
51 public:
52  explicit TrimmingSelection(App::DocumentObject* obj)
53  : Gui::SelectionFilterGate(nullPointer())
54  , object(obj)
55  {}
56 
57  bool allow(App::Document* /*pDoc*/, App::DocumentObject* pObj, const char* sSubName) override
58  {
59  if (pObj != this->object) {
60  return false;
61  }
62  if (!sSubName || sSubName[0] == '\0') {
63  return false;
64  }
65  std::string element(sSubName);
66  if (element.substr(0, 4) == "Edge") {
67  int GeoId = std::atoi(element.substr(4, 4000).c_str()) - 1;
68  Sketcher::SketchObject* Sketch = static_cast<Sketcher::SketchObject*>(object);
69  const Part::Geometry* geom = Sketch->getGeometry(GeoId);
70  if (geom->isDerivedFrom<Part::GeomTrimmedCurve>() || geom->is<Part::GeomCircle>()
71  || geom->is<Part::GeomEllipse>() || geom->is<Part::GeomBSplineCurve>()) {
72  // We do not trim internal geometry of complex geometries
73  if (Sketcher::GeometryFacade::isInternalType(geom, Sketcher::InternalType::None)) {
74  return true;
75  }
76  }
77  }
78  return false;
79  }
80 };

91  void mouseMove(Base::Vector2d onSketchPos) override
92  {
93  Q_UNUSED(onSketchPos);
94  if (mousePressed) {
95  executeCommands(onSketchPos);
96  }
97  else {
98  int GeoId = getPreselectCurve();
99 
100  if (GeoId > -1) {
101  auto sk = sketchgui->getObject<Sketcher::SketchObject>();
102  int GeoId1, GeoId2;
103  Base::Vector3d intersect1, intersect2;
104  if (sk->seekTrimPoints(GeoId,
105  Base::Vector3d(onSketchPos.x, onSketchPos.y, 0),
106  GeoId1,
107  intersect1,
108  GeoId2,
109  intersect2)) {
110 
111  EditMarkers.resize(0);
112 
113  if (GeoId1 != Sketcher::GeoEnum::GeoUndef) {
114  EditMarkers.emplace_back(intersect1.x, intersect1.y);
115  }
116  else {
117  auto start = sk->getPoint(GeoId, Sketcher::PointPos::start);
118  EditMarkers.emplace_back(start.x, start.y);
119  }
120 
121  if (GeoId2 != Sketcher::GeoEnum::GeoUndef) {
122  EditMarkers.emplace_back(intersect2.x, intersect2.y);
123  }
124  else {
125  auto end = sk->getPoint(GeoId, Sketcher::PointPos::end);
126  EditMarkers.emplace_back(end.x, end.y);
127  }
128 
129  // maker augmented by two sizes (see supported marker sizes)
130  drawEditMarkers(EditMarkers, 2);
131  }
132  }
133  else {
134  EditMarkers.resize(0);
135  drawEditMarkers(EditMarkers, 2);
136  }
137  }
138  }
139 
140  bool pressButton(Base::Vector2d onSketchPos) override

145  EditMarkers.resize(0);

160  void executeCommands(Base::Vector2d onSketchPos)
161  {
162  int GeoId = getPreselectCurve();
163  if (GeoId < 0) {
164  return;
165  }
166 
167  const Part::Geometry* geo = sketchgui->getSketchObject()->getGeometry(GeoId);
168  if (geo->isDerivedFrom<Part::GeomTrimmedCurve>() || geo->is<Part::GeomCircle>()
169  || geo->is<Part::GeomEllipse>() || geo->is<Part::GeomBSplineCurve>()) {
170  try {
171  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Trim edge"));
172  Gui::cmdAppObjectArgs(sketchgui->getObject(),
173  "trim(%d,App.Vector(%f,%f,0))",
174  GeoId,
175  onSketchPos.x,
176  onSketchPos.y);
177  Gui::Command::commitCommand();
178  tryAutoRecompute(sketchgui->getObject<Sketcher::SketchObject>());
179  }
180  catch (const Base::Exception&) {
181  Gui::NotifyError(sketchgui,
182  QT_TRANSLATE_NOOP("Notifications", "Error"),
183  QT_TRANSLATE_NOOP("Notifications", "Failed to trim edge"));
184 
185  Gui::Command::abortCommand();
186  }
187  }
188  }
