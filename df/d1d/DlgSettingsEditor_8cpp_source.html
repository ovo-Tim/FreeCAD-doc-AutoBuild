245 void DlgSettingsEditor::loadSettings()
246 {
247  ui->EnableLineNumber->onRestore();
248  ui->EnableBlockCursor->onRestore();
249  ui->EnableFolding->onRestore();
250  ui->tabSize->onRestore();
251  ui->indentSize->onRestore();

253  ui->radioSpaces->onRestore();
254 
255  setEditorTabWidth(ui->tabSize->value());
256  ui->textEdit1->setPlainText(QString::fromLatin1("# Short Python sample\n"
257  "import sys\n"
258  "\n"
259  "def foo(begin, end):\n"
260  " i = begin\n"
261  " while i < end:\n"
262  " print(i)\n"
263  " i = i + 1\n"
264  " print(\"Text\")\n"
265  " return None\n"
266  "\n"
267  "foo(0, 20)\n"));
268 
269  // Restores the color map
270  ParameterGrp::handle hGrp = WindowParameter::getDefaultParameter()->GetGroup("Editor");
271  for (auto& [textType, textColor] : d->colormap) {
272  auto col = static_cast<unsigned long>(textColor);
273  col = hGrp->GetUnsigned(textType.toLatin1(), col);
274  textColor = static_cast<unsigned int>(col);
275  QColor color = App::Color::fromPackedRGB<QColor>(col);
276  pythonSyntax->setColor(textType, color);
277  }
278 
279  // fill up font styles
280  //
281  ui->fontSize->setValue(10);
282  ui->fontSize->setValue(hGrp->GetInt("FontSize", ui->fontSize->value()));
283 
284  QByteArray defaultMonospaceFont = getMonospaceFont().family().toLatin1();
285 
286 #if QT_VERSION < QT_VERSION_CHECK(6, 0, 0)
287  QStringList familyNames = QFontDatabase().families(QFontDatabase::Any);
288  QStringList fixedFamilyNames;
289  for (const auto& name : familyNames) {
290  if (QFontDatabase().isFixedPitch(name)) {
291  if (name.compare(QLatin1String("8514oem"), Qt::CaseInsensitive) != 0) {
292  fixedFamilyNames.append(name);
293  }
294  }
295  }
296 #else
297  QStringList familyNames = QFontDatabase::families(QFontDatabase::Any);
298  QStringList fixedFamilyNames;
299  for (const auto& name : familyNames) {
300  if (QFontDatabase::isFixedPitch(name)) {
301  if (name.compare(QLatin1String("8514oem"), Qt::CaseInsensitive) != 0) {
302  fixedFamilyNames.append(name);
303  }
304  }
305  }
306 #endif
307  ui->fontFamily->addItems(fixedFamilyNames);
308  int index = fixedFamilyNames.indexOf(
309  QString::fromLatin1(hGrp->GetASCII("Font", defaultMonospaceFont).c_str()));
310  if (index < 0) {
311  index = 0;
312  }
313  ui->fontFamily->setCurrentIndex(index);
314  onFontFamilyActivated(ui->fontFamily->currentText());
315  ui->displayItems->setCurrentItem(ui->displayItems->topLevelItem(0));
316 }

318 void DlgSettingsEditor::resetSettingsToDefaults()
319 {
320  ParameterGrp::handle hGrp;
321  hGrp = WindowParameter::getDefaultParameter()->GetGroup("Editor");
322  // reset the parameters in the "Editor" group
323  for (const auto& [textType, textColor] : d->colormap) {
324  hGrp->RemoveUnsigned(textType.toLatin1());
325  }
326  // reset "FontSize" parameter
327  hGrp->RemoveInt("FontSize");
328  // reset "Font" parameter
329  hGrp->RemoveASCII("Font");
330 
331  // finally reset all the parameters associated to Gui::Pref* widgets
332  PreferencePage::resetSettingsToDefaults();
333 }
