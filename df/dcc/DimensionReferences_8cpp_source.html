99 TopoDS_Shape ReferenceEntry::getGeometry() const
100 {
101  // first, make sure the object has not been deleted!
102  App::DocumentObject* obj = getDocument()->getObject(getObjectName().c_str());
103  if (!obj) {
104  return {};
105  }
106 
107  if (getSubName().empty()) {
108  return {};
109  }
110 
111  if ( getObject()->isDerivedFrom(TechDraw::DrawViewPart::getClassTypeId()) ) {
112  // 2d geometry from DrawViewPart will be rotated and scaled
113  return getGeometry2d();
114  }
115 
116  // 3d geometry
117  return ShapeFinder::getLocatedShape(*getObject(), getSubName(true));
118 }

122 TopoDS_Shape ReferenceEntry::getGeometry2d() const
123 {
124  std::string gType;
125  try {
126  auto dvp = getObject<TechDraw::DrawViewPart>(); //NOLINT cppcoreguidelines-pro-type-static-cast-downcast
127  gType = geomType();
128  if (gType == "Vertex") {
129  // getVertex throws on not found, but we want to return null
130  // shape
131  auto vgeom = dvp->getVertex(getSubName());
132  if (!vgeom) {
133  return {};
134  }
135  return vgeom->getOCCVertex();
136  }
137  if (gType == "Edge") {
138  auto egeom = dvp->getEdge(getSubName());
139  if (!egeom) {
140  return {};
141  }
142  return egeom->getOCCEdge();
143  }
144  if (gType == "Face") {
145  auto fgeom = dvp->getFace(getSubName());
146  if (!fgeom) {
147  return {};
148  }
149  return fgeom->toOccFace();
150  }
151  }
152  catch (...) {
153  Base::Console().Message("RE::getGeometry2d - no shape for dimension 2d reference - gType: **%s**\n", gType.c_str());
154  }
155 
156  return {};
157 }

185 Part::TopoShape ReferenceEntry::asTopoShape() const
186 {
187  TopoDS_Shape geom = getGeometry();
188  if (geom.IsNull()) {
189  return {};
190  }
191  if (geom.ShapeType() == TopAbs_VERTEX) {
192  TopoDS_Vertex vert = TopoDS::Vertex(geom);
193  return asTopoShapeVertex(vert);
194  }
195  if (geom.ShapeType() == TopAbs_EDGE) {
196  TopoDS_Edge edge = TopoDS::Edge(geom);
197  return asTopoShapeEdge(edge);
198  }
199  if (geom.ShapeType() == TopAbs_FACE) {
200  TopoDS_Face face = TopoDS::Face(geom);
201  return asTopoShapeFace(face);
202  }
203  throw Base::RuntimeError("Dimension Reference has unsupported geometry");
204 }

224 Part::TopoShape ReferenceEntry::asCanonicalTopoShape(const Part::TopoShape& inShape, const DrawViewPart& dvp)
225 {
226  gp_Ax2 OXYZ;
227  auto unscaledShape = SU::scaleShape(inShape.getShape(), 1.0 / dvp.getScale());
228  if (dvp.Rotation.getValue() != 0.0) {
229  auto rotationDeg = dvp.Rotation.getValue();
230  unscaledShape = SU::invertGeometry(unscaledShape);
231  unscaledShape = SU::rotateShape(unscaledShape, OXYZ, -rotationDeg);
232  unscaledShape = SU::invertGeometry(unscaledShape);
233  }
234  return {unscaledShape};
235 }

277 bool ReferenceEntry::is3d() const
278 {
279  if (getObject() &&
280  getObject()->isDerivedFrom(TechDraw::DrawViewPart::getClassTypeId()) &&
281  !getSubName().empty()) {
282  // this is a well formed 2d reference
283  return false;
284  }
285 
286  if (getObject() &&
287  getObject()->isDerivedFrom(TechDraw::DrawViewPart::getClassTypeId()) &&
288  getSubName().empty()) {
289  // this is a broken 3d reference, so it should be treated as 3d
290  return true;

292 
293  // either we have no object or we have an object and it is a 3d object
294  return true;

320 bool ReferenceEntry::hasGeometry2d() const
321 {
322  auto dvp = getObject<TechDraw::DrawViewPart>(); //NOLINT cppcoreguidelines-pro-type-static-cast-downcast
323  if (getSubName().empty()) {
324  return false;
325  }
326  int geomNumber = DU::getIndexFromName(getSubName());
327  std::string gType = geomType();
328  if (gType == "Vertex") {
329  auto vert = dvp->getProjVertexByIndex(geomNumber);
330  if (vert) {
331  return true;
332  }
333  }
334  else if (gType == "Edge") {
335  auto edge = dvp->getGeomByIndex(geomNumber);
336  if (edge) {
337  return true;
338  }
339  }
340  else if (gType == "Face") {
341  auto face = dvp->getFace(getSubName());
342  if (face) {
343  return true;
344  }
345  }
346  return false;
347 }
