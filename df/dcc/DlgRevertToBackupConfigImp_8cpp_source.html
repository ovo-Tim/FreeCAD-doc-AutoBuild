74 void DlgRevertToBackupConfigImp::showEvent(QShowEvent* event)
75 {
76  ui->listWidget->clear();
77  const auto& backups = Application::Instance->prefPackManager()->configBackups();
78  for (const auto& backup : backups) {
79  auto modification_date = QDateTime::fromSecsSinceEpoch(fs::last_write_time(backup));
80  auto item = new QListWidgetItem(QLocale().toString(modification_date));
81  item->setData(Qt::UserRole, QString::fromStdString(backup.string()));
82  ui->listWidget->addItem(item);
83  }
84  ui->buttonBox->button(QDialogButtonBox::StandardButton::Ok)->setEnabled(false);

88 void DlgRevertToBackupConfigImp::accept()
89 {
90  auto items = ui->listWidget->selectedItems();
91  if (items.count() != 1) {
92  Base::Console().Error(tr("No selection in dialog, cannot load backup file").toStdString().c_str());
93  return;
94  }
95  auto item = items[0];
96  auto path = item->data(Qt::UserRole).toString().toStdString();
97  if (fs::exists(path)) {
98  auto newParameters = ParameterManager::Create();
99  newParameters->LoadDocument(path.c_str());
100  auto baseAppGroup = App::GetApplication().GetUserParameter().GetGroup("BaseApp");
101  newParameters->GetGroup("BaseApp")->copyTo(baseAppGroup);
102  }
103  else {
104  Base::Console().Error("Preference Pack Internal Error: Invalid backup file location");
105  }
106 
107  QDialog::accept();
108 }
