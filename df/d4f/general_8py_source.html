46 def precision():
47  """Return the Draft precision setting."""
48  # Set precision level with a cap to avoid overspecification that:
49  # 1 - whilst it is precise enough (e.g. that OCC would consider
50  # 2 points are coincident)
51  # (not sure what it should be 10 or otherwise);
52  # 2 - but FreeCAD/OCC can handle 'internally'
53  # (e.g. otherwise user may set something like
54  # 15 that the code would never consider 2 points are coincident
55  # as internal float is not that precise)
56  precisionMax = 10
57  precisionInt = params.get_param("precision")
58  precisionInt = precisionInt if precisionInt <= 10 else precisionMax
59  return precisionInt

62 def vec(edge, use_orientation = False):
63  """Return a vector from an edge or a Part.LineSegment.
64 
65  If use_orientation is True, it takes into account the edges orientation.
66  If edge is not straight, you'll get strange results!
67  """
68  if isinstance(edge, Part.Shape):
69  if use_orientation and isinstance(edge, Part.Edge) and edge.Orientation == "Reversed":
70  return edge.Vertexes[0].Point.sub(edge.Vertexes[-1].Point)
71  else:
72  return edge.Vertexes[-1].Point.sub(edge.Vertexes[0].Point)
73  elif isinstance(edge, Part.LineSegment):
74  return edge.EndPoint.sub(edge.StartPoint)
75  else:
76  return None

146 def isAligned(edge, axis="x"):
147  """Check if the given edge or line is aligned to the given axis.
148 
149  The axis can be 'x', 'y' or 'z'.
150  """
151  def is_same(a, b):
152  return round(a, precision()) == round(b, precision())
153 
154  if axis == "x":
155  if isinstance(edge, Part.Edge):
156  if len(edge.Vertexes) == 2:
157  return is_same(edge.Vertexes[0].X, edge.Vertexes[-1].X)
158  elif isinstance(edge, Part.LineSegment):
159  return is_same(edge.StartPoint.x, edge.EndPoint.x)
160 
161  elif axis == "y":
162  if isinstance(edge, Part.Edge):
163  if len(edge.Vertexes) == 2:
164  return is_same(edge.Vertexes[0].Y, edge.Vertexes[-1].Y)
165  elif isinstance(edge, Part.LineSegment):
166  return is_same(edge.StartPoint.y, edge.EndPoint.y)
167 
168  elif axis == "z":
169  if isinstance(edge, Part.Edge):
170  if len(edge.Vertexes) == 2:
171  return is_same(edge.Vertexes[0].Z, edge.Vertexes[-1].Z)
172  elif isinstance(edge, Part.LineSegment):
173  return is_same(edge.StartPoint.z, edge.EndPoint.z)
174 
175  return False

178 def getQuad(face):
179  """Return a list of 3 vectors if the face is a quad, otherwise None.
180 
181  Returns
182  -------
183  basepoint, Xdir, Ydir
184  If the face is a quad.
185 
186  None
187  If the face is not a quad.
188  """
189  if len(face.Edges) != 4:
190  return None
191 
192  v1 = vec(face.Edges[0]) # Warning redefinition of function v1
193  v2 = vec(face.Edges[1])
194  v3 = vec(face.Edges[2])
195  v4 = vec(face.Edges[3])
196  angles90 = [round(math.pi*0.5, precision()),
197  round(math.pi*1.5, precision())]
198 
199  angles180 = [0,
200  round(math.pi, precision()),
201  round(math.pi*2, precision())]
202  for ov in [v2, v3, v4]:
203  if not (round(v1.getAngle(ov), precision()) in angles90 + angles180):
204  return None
205 
206  for ov in [v2, v3, v4]:
207  if round(v1.getAngle(ov), precision()) in angles90:
208  v1.normalize()
209  ov.normalize()
210  return [face.Edges[0].Vertexes[0].Point, v1, ov]

213 def areColinear(e1, e2):
214  """Return True if both edges are colinear."""
215  if not isinstance(e1.Curve, (Part.LineSegment, Part.Line)):
216  return False
217  if not isinstance(e2.Curve, (Part.LineSegment, Part.Line)):
218  return False
219 
220  v1 = vec(e1)
221  v2 = vec(e2)
222  a = round(v1.getAngle(v2), precision())
223  if (a == 0) or (a == round(math.pi, precision())):
224  v3 = e2.Vertexes[0].Point.sub(e1.Vertexes[0].Point)
225  if DraftVecUtils.isNull(v3):
226  return True
227  else:
228  a2 = round(v1.getAngle(v3), precision())
229  if (a2 == 0) or (a2 == round(math.pi, precision())):
230  return True
231  return False

315 def getBoundaryAngles(angle, alist):
316  """Return the 2 closest angles that encompass the given angle."""
317  negs = True
318  while negs:
319  negs = False
320  for i in range(len(alist)):
321  if alist[i] < 0:
322  alist[i] = 2*math.pi + alist[i]
323  negs = True
324  if angle < 0:
325  angle = 2*math.pi + angle
326  negs = True
327 
328  lower = None
329  for a in alist:
330  if a < angle:
331  if lower is None:
332  lower = a
333  else:
334  if a > lower:
335  lower = a
336 
337  if lower is None:
338  lower = 0
339  for a in alist:
340  if a > lower:
341  lower = a
342 
343  higher = None
344  for a in alist:
345  if a > angle:
346  if higher is None:
347  higher = a
348  else:
349  if a < higher:
350  higher = a
351 
352  if higher is None:
353  higher = 2*math.pi
354  for a in alist:
355  if a < higher:
356  higher = a
357 
358  return lower, higher
