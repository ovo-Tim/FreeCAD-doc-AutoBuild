43 TopoShape TopoShapeCache::Ancestry::_getTopoShape(const TopoShape& parent, int index)
44 {
45  auto& ts = topoShapes[index - 1];
46  if (ts.isNull()) {
47  ts.setShape(shapes.FindKey(index), true);
48  ts.initCache();
49  ts._cache->subLocation = ts._Shape.Location();
50  }
51 
52  if (ts._Shape.IsEqual(parent._cache->shape)) {
53  return parent;
54  }
55 
56  TopoShape res(ts);
57  res.Tag = parent.Tag;
58  res.Hasher = parent.Hasher;
59 
60  if (!parent.getShape().Location().IsIdentity()) {
61  res.setShape(TopoShape::moved(res._Shape, parent.getShape().Location()), false);
62  }
63 
64  if (ts._cache->cachedElementMap) {
65  res.resetElementMap(ts._cache->cachedElementMap);
66  }
67  else if (parent._parentCache) {
68  // If no cachedElementMap exists, we use _parentCache for
69  // delayed generation of sub element map so that we don't need
70  // to always generate a full map whenever we return a sub
71  // shape. To simplify the mapping and avoid circular
72  // dependency, we do not chain parent and grandparent.
73  // Instead, we always use the cache from the top parent. And to
74  // make it work, we must accumulate the TopLoc_Location along
75  // the lineage, which is required for OCCT shape mapping to
76  // work.
77  //
78  // Cache::subLocation is shared and only contains the location
79  // in the direct parent shape, while TopoShape::_subLocation is
80  // used to accumulate locations in higher ancestors. We
81  // separate these two to avoid invalidating cache.
82 
83  res._subLocation = parent._subLocation * parent._cache->subLocation;
84  res._parentCache = parent._parentCache;
85  }
86  else {
87  res._parentCache = owner->shared_from_this();
88  }
89  return res;
90 }

219 TopoDS_Shape TopoShapeCache::findAncestor(const TopoDS_Shape& parent,
220  const TopoDS_Shape& subShape,
221  TopAbs_ShapeEnum type,
222  std::vector<TopoDS_Shape>* ancestors)
223 {
224  TopoDS_Shape nullShape;
225  if (shape.IsNull() || subShape.IsNull() || type == TopAbs_SHAPE) {
226  return nullShape;
227  }
228 
229  auto& info = getAncestry(type);
230 
231  auto& ancestorInfo = info.ancestors.at(subShape.ShapeType());
232  if (!ancestorInfo.initialized) {
233  ancestorInfo.initialized = true;
234  // ancestorInfo.shapes is the output variable here, storing (and caching) the actual map
235  TopExp::MapShapesAndAncestors(shape, subShape.ShapeType(), type, ancestorInfo.shapes);
236  }
237  int index = parent.Location().IsIdentity()
238  ? ancestorInfo.shapes.FindIndex(subShape)
239  : ancestorInfo.shapes.FindIndex(info.stripLocation(parent, subShape));
240  if (index == 0) {
241  return nullShape;
242  }
243  const auto& shapes = ancestorInfo.shapes.FindFromIndex(index);
244  if (shapes.Extent() == 0) {
245  return nullShape;
246  }
247 
248  if (ancestors) {
249  ancestors->reserve(ancestors->size() + shapes.Extent());
250  for (TopTools_ListIteratorOfListOfShape it(shapes); it.More(); it.Next()) {
251  ancestors->push_back(TopoShape::moved(it.Value(), parent.Location()));
252  }
253  }
254  return TopoShape::moved(shapes.First(), parent.Location());
255 }
