80 bool PropertyModel::setData(const QModelIndex& index, const QVariant& value, int role)
81 {
82  if (!index.isValid()) {
83  return false;
84  }
85 
86  // we check whether the data has really changed, otherwise we ignore it
87  if (role == Qt::EditRole) {
88  auto item = static_cast<PropertyItem*>(index.internalPointer());
89  QVariant data = item->data(index.column(), role);
90  if (data.userType() == QMetaType::Double && value.userType() == QMetaType::Double) {
91  // since we store some properties as floats we get some round-off
92  // errors here. Thus, we use an epsilon here.
93  // NOTE: Since 0.14 PropertyFloat uses double precision, so this is maybe unnecessary
94  // now?
95  double d = data.toDouble();
96  double v = value.toDouble();
97  if (fabs(d - v) > DBL_EPSILON) {
98  return item->setData(value);
99  }
100  }
101  // Special case handling for quantities
102  else if (data.canConvert<Base::Quantity>() && value.canConvert<Base::Quantity>()) {
103  const Base::Quantity& val1 = data.value<Base::Quantity>();
104  const Base::Quantity& val2 = value.value<Base::Quantity>();
105  if (!(val1 == val2)) {
106  return item->setData(value);
107  }
108  }
109  else if (data != value) {
110  return item->setData(value);
111  }
112  }
113 
114  return true;
115 }

209 QModelIndex PropertyModel::propertyIndexFromPath(const QStringList& path) const
210 {
211  if (path.size() < 2) {
212  return {};
213  }
214 
215  auto it = groupItems.find(path.front());
216  if (it == groupItems.end()) {
217  return {};
218  }
219 
220  PropertyItem* item = it->second.groupItem;
221  QModelIndex index = this->index(item->row(), 0, QModelIndex());
222 
223  for (int j = 1; j < path.size(); ++j) {
224  bool found = false;
225  for (int i = 0, c = item->childCount(); i < c; ++i) {
226  auto child = item->child(i);
227  if (child->propertyName() == path[j]) {
228  index = this->index(i, 1, index);
229  item = child;
230  found = true;
231  break;
232  }
233  }
234  if (!found) {
235  return j == 1 ? QModelIndex() : index;
236  }
237  }
238  return index;
239 }

271 PropertyModel::GroupInfo& PropertyModel::getGroupInfo(App::Property* prop)
272 {
273  const char* group = prop->getGroup();
274  bool isEmpty = (!group || group[0] == '\0');
275  QString groupName =
276  QString::fromLatin1(isEmpty ? QT_TRANSLATE_NOOP("App::Property", "Base") : group);
277 
278  auto res = groupItems.insert(std::make_pair(groupName, GroupInfo()));
279  if (res.second) {
280  auto& groupInfo = res.first->second;
281  groupInfo.groupItem = static_cast<PropertySeparatorItem*>(PropertySeparatorItem::create());
282  groupInfo.groupItem->setReadOnly(true);
283  groupInfo.groupItem->setExpanded(true);
284  groupInfo.groupItem->setParent(rootItem);
285  groupInfo.groupItem->setPropertyName(groupName);
286 
287  auto it = res.first;
288  int row = 0;
289  if (it != groupItems.begin()) {
290  --it;
291  row = it->second.groupItem->_row + 1;
292  ++it;
293  }
294  groupInfo.groupItem->_row = row;
295  beginInsertRows(QModelIndex(), row, row);
296  rootItem->insertChild(row, groupInfo.groupItem);
297  // update row index for all group items behind
298  for (++it; it != groupItems.end(); ++it) {
299  ++it->second.groupItem->_row;
300  }
301  endInsertRows();
302  }
303 
304  return res.first->second;
305 }

307 void PropertyModel::buildUp(const PropertyModel::PropertyList& props)
308 {
309  // If props empty, then simply reset all property items, but keep the group
310  // items.
311  if (props.empty()) {
312  resetGroups();
313  return;
314  }
315 
316  // First step, init group info
317  initGroups();
318 
319  // Second step, either find existing items or create new items for the given
320  // properties. There is no signaling of model change at this stage. The
321  // change information is kept pending in GroupInfo::children
322  findOrCreateChildren(props);
323 
324  // Third step, signal item insertion and movement.
325  insertOrMoveChildren();
326 
327 
328  // Final step, signal item removal. This is separated from the above because
329  // of the possibility of moving items between groups.
330  removeChildren();
331 }

353 void PropertyModel::findOrCreateChildren(const PropertyModel::PropertyList& props)
354 {
355  for (const auto& jt : props) {
356  App::Property* prop = jt.second.front();
357 
358  PropertyItem* item = nullptr;
359  for (auto prop : jt.second) {
360  auto it = itemMap.find(prop);
361  if (it == itemMap.end() || !it->second) {
362  continue;
363  }
364  item = it->second;
365  break;
366  }
367 
368  if (!item) {
369  item = createPropertyItem(prop);
370  if (!item) {
371  continue;
372  }
373  }
374 
375  GroupInfo& groupInfo = getGroupInfo(prop);
376  groupInfo.children.push_back(item);
377 
378  item->setLinked(boost::ends_with(jt.first, "*"));
379  setPropertyItemName(item, prop->getName(), groupInfo.groupItem->propertyName());
380 
381  if (jt.second != item->getPropertyData()) {
382  for (auto prop : item->getPropertyData()) {
383  itemMap.erase(prop);
384  }
385  for (auto prop : jt.second) {
386  itemMap[prop] = item;
387  }
388  // TODO: is it necessary to make sure the item has no pending commit?
389  item->setPropertyData(jt.second);
390  }
391  else {
392  item->updateData();
393  }
394  }
395 }

397 void PropertyModel::insertOrMoveChildren()
398 {
399  for (auto& v : groupItems) {
400  auto& groupInfo = v.second;
401  int beginChange = -1;
402  int endChange = 0;
403  int beginInsert = -1;
404  int endInsert = 0;
405  int row = -1;
406  QModelIndex midx = this->index(groupInfo.groupItem->_row, 0, QModelIndex());
407 
408  auto flushInserts = [&]() {
409  if (beginInsert < 0) {
410  return;
411  }
412  beginInsertRows(midx, beginInsert, endInsert);
413  for (int i = beginInsert; i <= endInsert; ++i) {
414  groupInfo.groupItem->insertChild(i, groupInfo.children[i]);
415  }
416  endInsertRows();
417  beginInsert = -1;
418  };
419 
420  auto flushChanges = [&]() {
421  if (beginChange < 0) {
422  return;
423  }
424  (void)endChange;
425  // There is no need to signal dataChange(), because PropertyEditor
426  // will call PropertyModel::updateProperty() on any property
427  // changes.
428  //
429  // dataChanged(this->index(beginChange,0,midx), this->index(endChange,1,midx));
430  beginChange = -1;
431  };
432 
433  for (auto item : groupInfo.children) {
434  ++row;
435  if (!item->parent()) {
436  flushChanges();
437  item->setParent(groupInfo.groupItem);
438  if (beginInsert < 0) {
439  beginInsert = row;
440  }
441  endInsert = row;
442  }
443  else {
444  flushInserts();
445  int oldRow = item->row();
446  // Dynamic property can rename group, so must check
447  auto groupItem = item->parent();
448  assert(groupItem);
449  if (oldRow == row && groupItem == groupInfo.groupItem) {
450  if (beginChange < 0) {
451  beginChange = row;
452  }
453  endChange = row;
454  }
455  else {
456  flushChanges();
457  beginMoveRows(createIndex(groupItem->row(), 0, groupItem),
458  oldRow,
459  oldRow,
460  midx,
461  row);
462  if (groupItem == groupInfo.groupItem) {
463  groupInfo.groupItem->moveChild(oldRow, row);
464  }
465  else {
466  groupItem->takeChild(oldRow);
467  item->setParent(groupInfo.groupItem);
468  groupInfo.groupItem->insertChild(row, item);
469  }
470  endMoveRows();
471  }
472  }
473  }
474 
475  flushChanges();
476  flushInserts();
477  }
478 }

481 {
482  for (auto& v : groupItems) {
483  auto& groupInfo = v.second;
484  int first, last;
485  getRange(groupInfo, first, last);
486  if (last > first) {
487  QModelIndex midx = this->index(groupInfo.groupItem->_row, 0, QModelIndex());
488  // This can trigger a recursive call of PropertyView::onTimer()
489  beginRemoveRows(midx, first, last - 1);

492  }
493  else {
494  assert(last == first);
495  }

522 void PropertyModel::appendProperty(const App::Property& _prop)
523 {
524  auto prop = const_cast<App::Property*>(&_prop);
525  if (!prop->getName()) {
526  return;
527  }
528  auto it = itemMap.find(prop);
529  if (it == itemMap.end() || !it->second) {
530  return;
531  }
532 
533  PropertyItem* item = createPropertyItem(prop);
534  GroupInfo& groupInfo = getGroupInfo(prop);
535 
536  int row = 0;
537  for (int c = groupInfo.groupItem->childCount(); row < c; ++row) {
538  PropertyItem* child = groupInfo.groupItem->child(row);
539  App::Property* firstProp = item->getFirstProperty();
540  if (firstProp && firstProp->testStatus(App::Property::PropDynamic)
541  && item->propertyName() >= child->propertyName()) {
542  break;
543  }
544  }
545 
546  QModelIndex midx = this->index(groupInfo.groupItem->_row, 0, QModelIndex());
547  beginInsertRows(midx, row, row);
548  groupInfo.groupItem->insertChild(row, item);
549  setPropertyItemName(item, prop->getName(), groupInfo.groupItem->propertyName());
550  item->setPropertyData({prop});
551  endInsertRows();
552 }
