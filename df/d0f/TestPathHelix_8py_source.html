226  def assertPathDirection(self, path, expected_direction, msg=None):
227  """Asserts that the given path goes into the expected direction.
228 
229  For the general case we'd need to check the sign of the second derivative,
230  but as we know we work on a helix here, we can take a short cut and just
231  look at the G2/G3 arc commands.
232  """
233  has_g2 = any(filter(lambda cmd: cmd.Name == "G2", path.Commands))
234  has_g3 = any(filter(lambda cmd: cmd.Name == "G3", path.Commands))
235  if has_g2 and not has_g3:
236  self.assertEqual("CW", expected_direction, msg)
237  elif has_g3 and not has_g2:
238  self.assertEqual("CCW", expected_direction, msg)
239  else:
240  raise NotImplementedError("Cannot determine direction for arbitrary paths")
