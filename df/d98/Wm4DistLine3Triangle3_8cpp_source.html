52 template <class Real>
53 Real DistLine3Triangle3<Real>::GetSquared ()
54 {
55  // Test if line intersects triangle. If so, the squared distance is zero.
56  Vector3<Real> kEdge0 = m_rkTriangle.V[1] - m_rkTriangle.V[0];
57  Vector3<Real> kEdge1 = m_rkTriangle.V[2] - m_rkTriangle.V[0];
58  Vector3<Real> kNormal = kEdge0.UnitCross(kEdge1);
59  Real fNdD = kNormal.Dot(m_rkLine.Direction);
60  if (Math<Real>::FAbs(fNdD) > Math<Real>::ZERO_TOLERANCE)
61  {
62  // The line and triangle are not parallel, so the line intersects
63  // the plane of the triangle.
64  Vector3<Real> kDiff = m_rkLine.Origin - m_rkTriangle.V[0];
65  Vector3<Real>& rkD = (Vector3<Real>&) m_rkLine.Direction;
66  Vector3<Real> kU, kV;
67  Vector3<Real>::GenerateComplementBasis(kU,kV,rkD);
68  Real fUdE0 = kU.Dot(kEdge0);
69  Real fUdE1 = kU.Dot(kEdge0);
70  Real fUdDiff = kU.Dot(kDiff);
71  Real fVdE0 = kV.Dot(kEdge0);
72  Real fVdE1 = kV.Dot(kEdge0);
73  Real fVdDiff = kV.Dot(kDiff);
74  Real fInvDet = ((Real)1.0)/(fUdE0*fVdE1 - fUdE1*fVdE0);
75 
76  // Barycentric coordinates for the point of intersection.
77  Real fB1 = (fVdE1*fUdDiff - fUdE1*fVdDiff)*fInvDet;
78  Real fB2 = (fUdE0*fVdDiff - fVdE0*fUdDiff)*fInvDet;
79  Real fB0 = (Real)1.0 - fB1 - fB2;
80 
81  if (fB0 >= (Real)0.0 && fB1 >= (Real)0.0 && fB2 >= (Real)0.0)
82  {
83  // Line parameter for the point of intersection.
84  Real fDdE0 = rkD.Dot(kEdge0);
85  Real fDdE1 = rkD.Dot(kEdge1);
86  Real fDdDiff = m_rkLine.Direction.Dot(kDiff);
87  m_fLineParameter = fB1*fDdE0 + fB2*fDdE1 - fDdDiff;
88 
89  // Barycentric coordinates for the point of intersection.
90  m_afTriangleBary[0] = fB0;
91  m_afTriangleBary[1] = fB1;
92  m_afTriangleBary[2] = fB2;
93 
94  // The intersection point is inside or on the triangle.
95  m_kClosestPoint0 = m_rkLine.Origin +
96  m_fLineParameter*m_rkLine.Direction;
97 
98  m_kClosestPoint1 = m_rkTriangle.V[0] + fB1*kEdge0 + fB2*kEdge1;
99  return (Real)0.0;
100  }
101  }
102 
103  // Either (1) the line is not parallel to the triangle and the point of
104  // intersection of the line and the plane of the triangle is outside the
105  // triangle or (2) the line and triangle are parallel. Regardless, the
106  // closest point on the triangle is on an edge of the triangle. Compare
107  // the line to all three edges of the triangle.
108  Real fSqrDist = Math<Real>::MAX_REAL;
109  for (int i0 = 2, i1 = 0; i1 < 3; i0 = i1++)
110  {
111  Segment3<Real> kSeg;
112  kSeg.Origin = ((Real)0.5)*(m_rkTriangle.V[i0] + m_rkTriangle.V[i1]);
113  kSeg.Direction = m_rkTriangle.V[i1] - m_rkTriangle.V[i0];
114  kSeg.Extent = ((Real)0.5)*kSeg.Direction.Normalize();
115  DistLine3Segment3<Real> kLSDist(m_rkLine,kSeg);
116  Real fSqrDistTmp = kLSDist.GetSquared();
117  if (fSqrDistTmp < fSqrDist)
118  {
119  m_kClosestPoint0 = kLSDist.GetClosestPoint0();
120  m_kClosestPoint1 = kLSDist.GetClosestPoint1();
121  fSqrDist = fSqrDistTmp;
122 
123  m_fLineParameter = kLSDist.GetLineParameter();
124  Real fRatio = kLSDist.GetSegmentParameter()/kSeg.Extent;
125  m_afTriangleBary[i0] = ((Real)0.5)*((Real)1.0 - fRatio);
126  m_afTriangleBary[i1] = (Real)1.0 - m_afTriangleBary[i0];
127  m_afTriangleBary[3-i0-i1] = (Real)0.0;
128  }
129  }
130  return fSqrDist;
131 }
