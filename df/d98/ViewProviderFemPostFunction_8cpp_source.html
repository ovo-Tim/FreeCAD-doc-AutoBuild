197 void ViewProviderFemPostFunction::attach(App::DocumentObject* pcObj)
198 {
199  ViewProviderDocumentObject::attach(pcObj);
200 
201  // setup the graph for editing the function unit geometry
202  SoMaterial* color = new SoMaterial();
203  color->diffuseColor.setValue(0, 0, 1);
204  color->transparency.setValue(0.5);
205 
206  SoTransform* transform = new SoTransform();
207 
208  m_manip = setupManipulator();
209  m_manip->ref();
210 
211  SoSeparator* pcEditNode = new SoSeparator();
212  pcEditNode->ref();
213 
214  pcEditNode->addChild(color);
215  pcEditNode->addChild(transform);
216  pcEditNode->addChild(m_geometrySeperator);
217 
218  m_geometrySeperator->insertChild(m_scale, 0);
219 
220  // Now we replace the SoTransform node by a manipulator
221  // Note: Even SoCenterballManip inherits from SoTransform
222  // we cannot use it directly (in above code) because the
223  // translation and center fields are overridden.
224  SoSearchAction sa;
225  sa.setInterest(SoSearchAction::FIRST);
226  sa.setSearchingAll(FALSE);
227  sa.setNode(transform);
228  sa.apply(pcEditNode);
229  SoPath* path = sa.getPath();
230  if (path) {
231  m_manip->replaceNode(path);
232 
233  SoDragger* dragger = m_manip->getDragger();
234  dragger->addStartCallback(dragStartCallback, this);
235  dragger->addFinishCallback(dragFinishCallback, this);
236  dragger->addMotionCallback(dragMotionCallback, this);
237  }
238 
239  addDisplayMaskMode(pcEditNode, "Default");
240  setDisplayMaskMode("Default");
241  pcEditNode->unref();
242 }

258 bool ViewProviderFemPostFunction::findScaleFactor(double& scale) const
259 {
260  SbBox3f bbox = getBoundingsOfView();
261  if (bbox.hasVolume()) {
262  float dx, dy, dz;
263  bbox.getSize(dx, dy, dz);
264  // we want the manipulator to have 20 % of the max size of the object
265  scale = 0.2 * std::max(std::max(dx, dy), dz);
266  return true;
267  }

269  return false;
