57 Base::Vector3d Fem::Tools::getDirectionFromShape(const TopoDS_Shape& shape)
58 {
59  gp_XYZ dir(0, 0, 0);
60 
61  // "Direction must be a planar face or linear edge"
62  //
63  if (shape.ShapeType() == TopAbs_FACE) {
64  if (isPlanar(TopoDS::Face(shape))) {
65  dir = getDirection(TopoDS::Face(shape));
66  }
67  }
68  else if (shape.ShapeType() == TopAbs_EDGE) {
69  if (isLinear(TopoDS::Edge(shape))) {
70  dir = getDirection(TopoDS::Edge(shape));
71  }
72  }
73 
74  Base::Vector3d the_direction(dir.X(), dir.Y(), dir.Z());
75  return the_direction;

78 bool Fem::Tools::isPlanar(const TopoDS_Face& face)
79 {
80  BRepAdaptor_Surface surface(face);
81  if (surface.GetType() == GeomAbs_Plane) {
82  return true;
83  }
84  else if (surface.GetType() == GeomAbs_BSplineSurface) {
85  Handle(Geom_BSplineSurface) spline = surface.BSpline();
86  try {
87  TColgp_Array2OfPnt poles(1, spline->NbUPoles(), 1, spline->NbVPoles());
88  spline->Poles(poles);
89 
90  // get the plane from three control points
91  gp_Pnt p1 = poles(poles.LowerRow(), poles.LowerCol());
92  gp_Pnt p2 = poles(poles.UpperRow(), poles.LowerCol());
93  gp_Pnt p3 = poles(poles.LowerRow(), poles.UpperCol());
94  gp_Vec vec1(p1, p2);
95  gp_Vec vec2(p1, p3);
96  gp_Vec vec3 = vec1.Crossed(vec2);
97  gp_Pln plane(p1, gp_Dir(vec3));
98 
99  for (int i = poles.LowerRow(); i <= poles.UpperRow(); i++) {
100  for (int j = poles.LowerCol(); j < poles.UpperCol(); j++) {
101  // are control points coplanar?
102  const gp_Pnt& pole = poles(i, j);
103  Standard_Real dist = plane.Distance(pole);
104  if (dist > Precision::Confusion()) {
105  return false;
106  }
107  }
108  }
109 
110  return true;
111  }
112  catch (Standard_Failure&) {
113  return false;
114  }
115  }
116  else if (surface.GetType() == GeomAbs_BezierSurface) {
117  Handle(Geom_BezierSurface) bezier = surface.Bezier();
118  try {
119  TColgp_Array2OfPnt poles(1, bezier->NbUPoles(), 1, bezier->NbVPoles());
120  bezier->Poles(poles);
121 
122  // get the plane from three control points
123  gp_Pnt p1 = poles(poles.LowerRow(), poles.LowerCol());
124  gp_Pnt p2 = poles(poles.UpperRow(), poles.LowerCol());
125  gp_Pnt p3 = poles(poles.LowerRow(), poles.UpperCol());
126  gp_Vec vec1(p1, p2);
127  gp_Vec vec2(p1, p3);
128  gp_Vec vec3 = vec1.Crossed(vec2);
129  gp_Pln plane(p1, gp_Dir(vec3));
130 
131  for (int i = poles.LowerRow(); i <= poles.UpperRow(); i++) {
132  for (int j = poles.LowerCol(); j < poles.UpperCol(); j++) {
133  // are control points coplanar?
134  const gp_Pnt& pole = poles(i, j);
135  Standard_Real dist = plane.Distance(pole);
136  if (dist > Precision::Confusion()) {
137  return false;
138  }
139  }
140  }
141 
142  return true;
143  }
144  catch (Standard_Failure&) {
145  return false;
146  }
147  }
148 
149  return false;

202 bool Fem::Tools::isLinear(const TopoDS_Edge& edge)
203 {
204  BRepAdaptor_Curve curve(edge);
205  if (curve.GetType() == GeomAbs_Line) {
206  return true;
207  }
208  else if (curve.GetType() == GeomAbs_BSplineCurve) {
209  Handle(Geom_BSplineCurve) spline = curve.BSpline();
210  try {
211  gp_Pnt s1 = spline->Pole(1);
212  gp_Pnt sn = spline->Pole(spline->NbPoles());
213  gp_Vec vec(s1, sn);
214  gp_Lin line(s1, gp_Dir(vec));
215 
216  for (int i = 2; i < spline->NbPoles(); i++) {
217  // are control points collinear?
218  Standard_Real dist = line.Distance(spline->Pole(i));
219  if (dist > Precision::Confusion()) {
220  return false;
221  }
222  }
223 
224  return true;
225  }
226  catch (Standard_Failure&) {
227  return false;
228  }
229  }
230  else if (curve.GetType() == GeomAbs_BezierCurve) {
231  Handle(Geom_BezierCurve) bezier = curve.Bezier();
232  try {
233  gp_Pnt s1 = bezier->Pole(1);
234  gp_Pnt sn = bezier->Pole(bezier->NbPoles());
235  gp_Vec vec(s1, sn);
236  gp_Lin line(s1, gp_Dir(vec));
237 
238  for (int i = 2; i < bezier->NbPoles(); i++) {

292 // function to determine 3rd-party binaries used by the FEM WB
293 std::string Fem::Tools::checkIfBinaryExists(std::string prefSection,
294  std::string prefBinaryName,
295  std::string binaryName)
296 {
297  // if "Search in known binary directories" is set in the preferences, we ignore custom path
298  auto paramPath = "User parameter:BaseApp/Preferences/Mod/Fem/" + prefSection;
299  auto knownDirectoriesString = "UseStandard" + prefSection + "Location";
300  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(paramPath.c_str());
301  bool knownDirectories = hGrp->GetBool(knownDirectoriesString.c_str(), true);
302 
303  if (knownDirectories) {
304  // first check the environment paths, normally determined by the PATH environment variable
305  // On Windows, the executable extensions(".exe" etc.) should be automatically appended
306  QString executablePath =
307  QStandardPaths::findExecutable(QString::fromLatin1(binaryName.c_str()));
308  if (!executablePath.isEmpty()) {
309  return executablePath.toStdString();
310  }
311  // check the folder of the FreeCAD binary
312  else {
313  auto appBinaryPath = App::Application::getHomePath() + "bin/";
314  QStringList pathCandidates = {QString::fromLatin1(appBinaryPath.c_str())};
315  QString executablePath =
316  QStandardPaths::findExecutable(QString::fromLatin1(binaryName.c_str()),
317  pathCandidates);
318  if (!executablePath.isEmpty()) {
319  return executablePath.toStdString();
320  }
321  }
322  }
323  else {
324  auto binaryPathString = prefBinaryName + "BinaryPath";
325  // use binary path from settings, fall back to system path if not defined
326  auto binaryPath = hGrp->GetASCII(binaryPathString.c_str(), binaryName.c_str());
327  QString executablePath =
328  QStandardPaths::findExecutable(QString::fromLatin1(binaryPath.c_str()));
329  if (!executablePath.isEmpty()) {
330  return executablePath.toStdString();
331  }
332  }
333  return "";
334 }

373 bool Fem::Tools::getCylinderParams(const TopoDS_Shape& sh,
374  Base::Vector3d& base,
375  Base::Vector3d& axis,
376  double& height,
377  double& radius)
378 {
379  TopoDS_Face face = TopoDS::Face(sh);
380  BRepAdaptor_Surface surface(face);
381  if (!(surface.GetType() == GeomAbs_Cylinder)) {
382  return false;
383  }
384 
385  gp_Cylinder cyl = surface.Cylinder();
386  gp_Pnt start = surface.Value(surface.FirstUParameter(), surface.FirstVParameter());
387  gp_Pnt end = surface.Value(surface.FirstUParameter(), surface.LastVParameter());
388 
389  Handle(Geom_Curve) handle = new Geom_Line(cyl.Axis());
390  GeomAPI_ProjectPointOnCurve proj(start, handle);
391  gp_XYZ startProj = proj.NearestPoint().XYZ();
392  proj.Perform(end);
393  gp_XYZ endProj = proj.NearestPoint().XYZ();
394 
395  gp_XYZ ax(endProj - startProj);
396  gp_XYZ center = (startProj + endProj) / 2.0;
397  gp_Dir dir(ax);
398 
399  height = ax.Modulus();
400  radius = cyl.Radius();
401  base = Base::Vector3d(center.X(), center.Y(), center.Z());
402  axis = Base::Vector3d(dir.X(), dir.Y(), dir.Z());
403 
404  return true;
405 }
