49 TextureMapping::TextureMapping(QWidget* parent, Qt::WindowFlags fl)
50  : QDialog(parent, fl), grp(nullptr), tex(nullptr), env(nullptr)
51 {
52  ui = new Ui_TextureMapping();
53  ui->setupUi(this);
54  connect(ui->fileChooser, &FileChooser::fileNameSelected,
55  this, &TextureMapping::onFileChooserFileNameSelected);
56  connect(ui->checkEnv, &QCheckBox::toggled,
57  this, &TextureMapping::onCheckEnvToggled);
58 
59  ui->checkGlobal->hide();
60 
61  // set a dummy string which is not a valid file name
62  fileName = QLatin1String("<>");
63 
64  // add all supported QImage formats
65  QStringList formats;
66  QList<QByteArray> qtformats = QImageReader::supportedImageFormats();
67  for (const auto & it : qtformats) {
68  formats << QString::fromLatin1("*.%1").arg(QLatin1String(it));
69  }
70 
71  ui->fileChooser->setFilter(tr("Image files (%1)").arg(formats.join(QLatin1String(" "))));
72 
73  this->tex = new SoTexture2();
74  this->tex->ref();
75  this->env = new SoTextureCoordinateEnvironment();
76  this->env->ref();
77 
78  std::string path = App::GetApplication().Config()["TextureImage"];
79  if (!path.empty()) {
80  QString file = QString::fromUtf8(path.c_str());
81  ui->fileChooser->setFileName(file);
82  onFileChooserFileNameSelected(file);
83  }

127 void TextureMapping::onFileChooserFileNameSelected(const QString& s)
128 {
129  QImage image;
130  if (!image.load(s)) {
131  // This slot is always invoked when leaving the focus of the line edit
132  // and to avoid to run into an infinite loop of popping up the dialog
133  // there is a check to report the last selected file name (which might
134  // be an empty string) only once.
135  if (fileName != s) {
136  QMessageBox::warning(this, tr("No image"), tr("The specified file is not a valid image file."));
137  fileName = s;
138  }
139  return;
140  }
141 
142  if (!this->grp) {
143  Gui::Document* doc = Gui::Application::Instance->activeDocument();
144  if (doc) {
145  Gui::MDIView* mdi = doc->getActiveView();
146  if (mdi && mdi->isDerivedFrom(View3DInventor::getClassTypeId())) {
147  Gui::View3DInventorViewer* view = static_cast<View3DInventor*>(mdi)->getViewer();
148  this->grp = static_cast<SoGroup *>(view->getSceneGraph());
149  this->grp->ref();
150  this->grp->insertChild(this->tex,1);
151  if (ui->checkEnv->isChecked())
152  this->grp->insertChild(this->env,2);
153  }
154  }
155  }
156 
157  if (!this->grp) {
158  QMessageBox::warning(this, tr("No 3D view"), tr("No active 3D view found."));
159  return;
160  }
161 
162  SoSFImage texture;
163  Gui::BitmapFactory().convert(image, texture);
164  this->tex->image = texture;
165  //this->tex->filename = (const char*)s.toUtf8();
166  App::GetApplication().Config()["TextureImage"] = (const char*)s.toUtf8();
167 }
