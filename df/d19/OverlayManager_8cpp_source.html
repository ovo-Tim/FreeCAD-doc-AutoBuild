434  bool toggleOverlay(QDockWidget *dock, ToggleMode toggle, int dockPos=Qt::NoDockWidgetArea)
435  {
436  if(!dock)
437  return false;
438 
439  auto it = _overlayMap.find(dock);
440  if(it != _overlayMap.end()) {
441  auto o = it->second;
442  switch(toggle) {
443  case ToggleMode::Transparent:
444  o->tabWidget->setTransparent(!o->tabWidget->isTransparent());
445  break;
446  case ToggleMode::Unset:
447  case ToggleMode::Toggle:
448  _overlayMap.erase(it);
449  o->tabWidget->removeWidget(dock);
450  return false;
451  default:
452  break;
453  }
454  return true;
455  }
456 
457  if(toggle == ToggleMode::Unset)
458  return false;
459 
460  if(dockPos == Qt::NoDockWidgetArea)
461  dockPos = getMainWindow()->dockWidgetArea(dock);
462  OverlayInfo *o;
463  switch(dockPos) {
464  case Qt::LeftDockWidgetArea:
465  o = &_left;
466  break;
467  case Qt::RightDockWidgetArea:
468  o = &_right;
469  break;
470  case Qt::TopDockWidgetArea:
471  o = &_top;
472  break;
473  case Qt::BottomDockWidgetArea:
474  o = &_bottom;
475  break;
476  default:
477  return false;
478  }
479  if(toggle == ToggleMode::Check && !o->tabWidget->count())
480  return false;
481  if(o->addWidget(dock)) {
482  if(toggle == ToggleMode::Transparent)
483  o->tabWidget->setTransparent(true);
484  }
485  refresh();
486  return true;
487  }

700  void setOverlayMode(OverlayMode mode)
701  {
702  switch(mode) {
703  case OverlayManager::OverlayMode::DisableAll:
704  case OverlayManager::OverlayMode::EnableAll: {
705  auto docks = getMainWindow()->findChildren<QDockWidget*>();
706  // put visible dock widget first
707  std::sort(docks.begin(),docks.end(),
708  [](const QDockWidget *a, const QDockWidget *b) {
709  return !a->visibleRegion().isEmpty() && b->visibleRegion().isEmpty();
710  });
711  for(auto dock : docks) {
712  if(mode == OverlayManager::OverlayMode::DisableAll)
713  toggleOverlay(dock, ToggleMode::Unset);
714  else
715  toggleOverlay(dock, ToggleMode::Set);
716  }
717  return;
718  }
719  case OverlayManager::OverlayMode::ToggleAll:
720  for(auto o : _overlayInfos) {
721  if(o->tabWidget->count()) {
722  setOverlayMode(OverlayManager::OverlayMode::DisableAll);

820  void onToggleDockWidget(QDockWidget *dock, int checked)
821  {
822  if(!dock)
823  return;
824 
825  auto it = _overlayMap.find(dock);
826  if(it == _overlayMap.end())
827  return;
828 
829  OverlayTabWidget *tabWidget = it->second->tabWidget;
830  int index = tabWidget->dockWidgetIndex(dock);
831  if(index < 0)
832  return;
833  auto sizes = tabWidget->getSplitter()->sizes();
834  while(index >= sizes.size())
835  sizes.append(0);
836 
837  if (checked < -1)
838  checked = 0;
839  else if (checked == 3) {
840  checked = 1;
841  sizes[index] = 0; // force expand the tab in full
842  } else if (checked <= 1) {
843  if (sizes[index] != 0 && tabWidget->isHidden())
844  checked = 1;
845  else {
846  // child widget inside splitter by right shouldn't been hidden, so
847  // we ignore the given toggle bit, but rely on its splitter size to
848  // decide.
849  checked = sizes[index] == 0 ? 1 : 0;
850  }
851  }
852  if(sizes[index]==0) {
853  if (!checked)
854  return;
855  tabWidget->setCurrent(dock);
856  tabWidget->onCurrentChanged(tabWidget->dockWidgetIndex(dock));
857  } else if (!checked) {
858  if (sizes[index] > 0 && sizes.size() > 1) {
859  int newtotal = 0;
860  auto newsizes = sizes;
861  newsizes[index] = 0;
862  for (int i=0; i<sizes.size(); ++i) {
863  if (i != index) {
864  auto d = tabWidget->dockWidget(i);
865  auto it = tabWidget->_sizemap.find(d);
866  if (it == tabWidget->_sizemap.end())
867  newsizes[i] = 0;
868  else {
869  if (newtotal == 0)
870  tabWidget->setCurrent(d);
871  newsizes[i] = it->second;
872  newtotal += it->second;
873  }
874  }
875  }
876  if (!newtotal) {
877  int expand = 0;
878  for (int i=0; i<sizes.size(); ++i) {
879  if (i != index && sizes[i] > 0) {
880  ++expand;
881  break;
882  }
883  }
884  if (expand) {
885  int expansion = sizes[index];
886  int step = expansion / expand;
887  for (int i=0; i<sizes.size(); ++i) {
888  if (i == index)
889  newsizes[i] = 0;
890  else if (--expand) {
891  expansion -= step;
892  newsizes[i] += step;
893  } else {
894  newsizes[i] += expansion;
895  break;
896  }
897  }
898  }
899  newsizes[index] = 0;
900  }
901  tabWidget->splitter->setSizes(newsizes);
902  tabWidget->saveTabs();
903  }
904  }
905  for (int i=0; i<sizes.size(); ++i) {
906  if (sizes[i])
907  tabWidget->_sizemap[tabWidget->dockWidget(i)] = sizes[i];
908  }
909  if (checked)
910  tabWidget->setRevealTime(QTime::currentTime().addMSecs(
911  OverlayParams::getDockOverlayRevealDelay()));
912  refresh();
913  }

1617 bool OverlayManager::eventFilter(QObject *o, QEvent *ev)
1618 {
1619  if (d->intercepting || !getMainWindow() || !o->isWidgetType())
1620  return false;
1621  auto mdi = getMainWindow()->getMdiArea();
1622  if (!mdi)
1623  return false;
1624 
1625  switch(ev->type()) {
1626  case QEvent::Enter:
1627  if (Selection().hasPreselection()
1628  && !qobject_cast<View3DInventorViewer*>(o)
1629  && !isUnderOverlay())
1630  {
1631  Selection().rmvPreselect();
1632  }
1633  break;
1634  case QEvent::ZOrderChange: {
1635  if(!d->raising && getMainWindow() && o == mdi) {
1636  // On Windows, for some reason, it will raise mdi window on tab
1637  // change in any docked widget, which will then obscure any overlay
1638  // docked widget here.
1639  for (auto child : getMainWindow()->children()) {
1640  if (child == mdi || qobject_cast<QDockWidget*>(child)) {
1641  QMetaObject::invokeMethod(this, "raiseAll", Qt::QueuedConnection);
1642  break;
1643  }
1644  if (qobject_cast<OverlayTabWidget*>(child))
1645  break;
1646  }
1647  }
1648  break;
1649  }
1650  case QEvent::Resize: {
1651  if(getMainWindow() && o == mdi)
1652  refresh();
1653  return false;
1654  }
1655  case QEvent::KeyPress: {
1656  QKeyEvent *ke = static_cast<QKeyEvent*>(ev);
1657  bool accepted = false;
1658  if (ke->modifiers() == Qt::NoModifier && ke->key() == Qt::Key_Escape) {
1659  if (d->mouseTransparent) {
1660  d->setMouseTransparent(false);
1661  accepted = true;
1662  } else if (OverlayTabWidget::_Dragging && OverlayTabWidget::_Dragging != o) {
1663  if (auto titleBar = qobject_cast<OverlayTitleBar*>(OverlayTabWidget::_Dragging))
1664  titleBar->endDrag();
1665  else if (auto splitHandle = qobject_cast<OverlaySplitterHandle*>(OverlayTabWidget::_Dragging))
1666  splitHandle->endDrag();
1667  }
1668  else if (!OverlayTabWidget::_Dragging) {
1669  for (OverlayTabWidget *tabWidget : _Overlays) {
1670  if (tabWidget->onEscape())
1671  accepted = true;
1672  }
1673  }
1674  }
1675  if (accepted) {
1676  ke->accept();
1677  return true;
1678  }
1679  break;
1680  }
1681  case QEvent::Paint:
1682  if (auto widget = qobject_cast<QWidget*>(o)) {
1683  // QAbstractItemView optimize redraw using its item delegate's
1684  // visualRect(). However, if we are using QGraphicsEffects, the
1685  // effect may touch areas outside of visualRect(), so
1686  // OverlayTabWidget offers a timer for a delayed redraw.
1687  widget = qobject_cast<QAbstractItemView*>(widget->parentWidget());
1688  if(widget) {
1689  auto tabWidget = findTabWidget(widget, true);
1690  if (tabWidget)
1691  tabWidget->scheduleRepaint();
1692  }
1693  }
1694  break;
1695  // case QEvent::NativeGesture:
1696  case QEvent::Wheel:
1697  if (!OverlayParams::getDockOverlayWheelPassThrough())
1698  return false;
1699  // fall through
1700  case QEvent::ContextMenu: {
1701  auto cev = static_cast<QContextMenuEvent*>(ev);
1702  if (cev->reason() != QContextMenuEvent::Mouse)
1703  return false;
1704  } // fall through
1705  case QEvent::MouseButtonDblClick:
1706  case QEvent::MouseButtonRelease:
1707  case QEvent::MouseButtonPress:
1708  case QEvent::MouseMove: {
1709  if (OverlayTabWidget::_Dragging && OverlayTabWidget::_Dragging != o) {
1710  if (auto titleBar = qobject_cast<OverlayTitleBar*>(OverlayTabWidget::_Dragging))
1711  titleBar->endDrag();
1712  else if (auto splitHandle = qobject_cast<OverlaySplitterHandle*>(OverlayTabWidget::_Dragging))
1713  splitHandle->endDrag();
1714  }
1715  QWidget *grabber = QWidget::mouseGrabber();
1716  d->lastIntercept = nullptr;
1717  if (d->mouseTransparent || (grabber && grabber != d->_trackingOverlay))
1718  return false;
1719  if (qobject_cast<QAbstractButton*>(o))
1720  return false;
1721  if (ev->type() != QEvent::Wheel) {
1722  if (qobject_cast<OverlayTitleBar*>(o))
1723  return false;
1724  } else if (qobject_cast<QScrollBar*>(o))
1725  return false;
1726 
1727  if (d->_trackingWidget) {
1728  if(!isTreeViewDragging())
1729  d->interceptEvent(d->_trackingWidget, ev);
1730  if(isTreeViewDragging()
1731  || (ev->type() == QEvent::MouseButtonRelease
1732  && QApplication::mouseButtons() == Qt::NoButton))
1733  {
1734  d->_trackingWidget = nullptr;
1735  if (d->_trackingOverlay == grabber
1736  && ev->type() == QEvent::MouseButtonRelease)
1737  {
1738  d->_trackingOverlay = nullptr;
1739  // Must not release mouse here, because otherwise the event
1740  // will find its way to the actual widget under cursor.
1741  // Instead, return false here to let OverlayTabWidget::event()
1742  // release the mouse.
1743  return false;
1744  }
1745  if(d->_trackingOverlay && grabber == d->_trackingOverlay)
1746  d->_trackingOverlay->releaseMouse();
1747  d->_trackingOverlay = nullptr;
1748  }
1749  // Must return true here to filter the event, otherwise ContextMenu
1750  // event may be routed to the actual widget. Other types of event
1751  // probably do not matter.
1752  return true;
1753  } else if (ev->type() != QEvent::MouseButtonPress
1754  && ev->type() != QEvent::MouseButtonDblClick
1755  && QApplication::mouseButtons() != Qt::NoButton)
1756  return false;
1757 
1758  if(isTreeViewDragging())
1759  return false;
1760 
1761  OverlayTabWidget *activeTabWidget = nullptr;
1762  int hit = 0;
1763  QPoint pos = QCursor::pos();
1764  if (OverlayParams::getDockOverlayAutoMouseThrough()
1765  && ev->type() != QEvent::Wheel
1766  && pos == d->_lastPos)
1767  {
1768  hit = 1;
1769  } else if (ev->type() == QEvent::Wheel
1770  && !d->wheelDelay.isNull()
1771  && (isNear(pos, d->wheelPos) || d->wheelDelay > QTime::currentTime()))
1772  {
1773  d->wheelDelay = QTime::currentTime().addMSecs(
1774  OverlayParams::getDockOverlayWheelDelay());
1775  d->wheelPos = pos;
1776  return false;
1777  } else {
1778  for(auto widget=qApp->widgetAt(pos); widget ; widget=widget->parentWidget()) {
1779  int type = widget->windowType();
1780  if (type != Qt::Widget && type != Qt::Window) {
1781  if (type != Qt::SubWindow)
1782  hit = -1;
1783  break;
1784  }
1785  if (ev->type() == QEvent::Wheel) {
1786  if (qobject_cast<OverlayTitleBar*>(widget))
1787  activeTabWidget = qobject_cast<OverlayTabWidget*>(widget->parentWidget());
1788  else if (qobject_cast<OverlaySplitterHandle*>(widget)) {
1789  auto parent = widget->parentWidget();
1790  if (parent)
1791  activeTabWidget = qobject_cast<OverlayTabWidget*>(parent->parentWidget());
1792  }
1793  if (activeTabWidget)
1794  break;
1795  }
1796  if (auto tabWidget = qobject_cast<OverlayTabWidget*>(widget)) {
1797  if (tabWidget->testAlpha(pos, ev->type() == QEvent::Wheel ? 4 : 1) == 0)
1798  activeTabWidget = tabWidget;
1799  break;
1800  }
1801  }
1802  if (activeTabWidget) {
1803  hit = OverlayParams::getDockOverlayAutoMouseThrough();
1804  d->_lastPos = pos;
1805  }
1806  }
1807 
1808  for (OverlayTabWidget *tabWidget : _Overlays) {
1809  if (tabWidget->getProxyWidget()->hitTest(pos) == OverlayProxyWidget::HitTest::HitInner) {
1810  if ((ev->type() == QEvent::MouseButtonRelease
1811  || ev->type() == QEvent::MouseButtonPress)
1812  && static_cast<QMouseEvent*>(ev)->button() == Qt::LeftButton)
1813  {
1814  if (ev->type() == QEvent::MouseButtonRelease)
1815  tabWidget->getProxyWidget()->onMousePress();
1816  return true;
1817  }
1818  }
1819  }
1820 
1821  if (hit <= 0) {
1822  d->_lastPos.setX(INT_MAX);
1823  if (ev->type() == QEvent::Wheel) {
1824  d->wheelDelay = QTime::currentTime().addMSecs(OverlayParams::getDockOverlayWheelDelay());
1825  d->wheelPos = pos;
1826  }
1827  return false;
1828  }
1829 
1830  auto hitWidget = mdi->childAt(mdi->mapFromGlobal(pos));
1831  if (!hitWidget)
1832  return false;
1833 
1834  if (!activeTabWidget)
1835  activeTabWidget = findTabWidget(qApp->widgetAt(QCursor::pos()));
1836  if(!activeTabWidget || !activeTabWidget->isTransparent())
1837  return false;
1838 
1839  ev->accept();
1840  d->interceptEvent(hitWidget, ev);
1841  if (ev->isAccepted() && ev->type() == QEvent::MouseButtonPress) {
1842  hitWidget->setFocus();
1843  d->_trackingWidget = hitWidget;
1844  d->_trackingOverlay = activeTabWidget;
1845  d->_trackingOverlay->grabMouse();
1846  }
1847  return true;
1848  }
1849  default:
1850  break;
1851  }
1852  return false;

1875 void OverlayManager::Private::interceptEvent(QWidget *widget, QEvent *ev)
1876 {
1877  Base::StateLocker guard(this->intercepting);
1878  MouseGrabberGuard grabberGuard(_trackingOverlay);
1879 
1880  lastIntercept = nullptr;
1881  auto getChildAt = [](QWidget *w, const QPoint &pos) {
1882  QWidget *res = w;
1883  for (; w; w = w->childAt(w->mapFromGlobal(pos))) {
1884  if (auto scrollArea = qobject_cast<QAbstractScrollArea*>(w)) {
1885  return scrollArea->viewport();
1886  }
1887  res = w;
1888  }
1889  return res;
1890  };
1891 
1892  switch(ev->type()) {
1893  case QEvent::MouseButtonRelease:
1894  case QEvent::MouseButtonPress:
1895  case QEvent::MouseMove:
1896  case QEvent::MouseButtonDblClick: {
1897  auto me = static_cast<QMouseEvent*>(ev);
1898 #if QT_VERSION < QT_VERSION_CHECK(6,0,0)
1899  QPointF screenPos = me->screenPos();
1900  QPoint point = me->globalPos();
1901 #else
1902  QPointF screenPos = me->globalPosition();
1903  QPoint point = screenPos.toPoint();
1904 #endif
1905  lastIntercept = getChildAt(widget, point);
1906  QMouseEvent mouseEvent(ev->type(),
1907  lastIntercept->mapFromGlobal(point),
1908  screenPos,
1909  me->button(),
1910  me->buttons(),
1911  me->modifiers());
1912  QApplication::sendEvent(lastIntercept, &mouseEvent);
1913  break;
1914  }
1915  case QEvent::Wheel: {
1916  auto we = static_cast<QWheelEvent*>(ev);
1917 #if QT_VERSION < QT_VERSION_CHECK(5,15,0)
1918  QPoint globalPos = we->globalPos();
1919 #else
1920  QPoint globalPos = we->globalPosition().toPoint();
1921 #endif
1922  lastIntercept = getChildAt(widget, globalPos);
1923 
1924  // For some reason in case of 3D View we have to target it directly instead of targeting
1925  // the viewport of QAbstractScrollArea like it works for all other widgets of that kind.
1926  // That's why for this event we have to traverse up the widget tree to find if it is part
1927  // of the 3D view.
1928  for (auto parent = lastIntercept; parent; parent = parent->parentWidget()) {
1929  if (qobject_cast<View3DInventorViewer*>(parent)) {
1930  lastIntercept = parent;
1931  break;
1932  }
1933  }
1934 
1935 #if QT_VERSION >= QT_VERSION_CHECK(5,12,0)
1936  QWheelEvent wheelEvent(lastIntercept->mapFromGlobal(globalPos),
1937  globalPos,
1938  we->pixelDelta(),
1939  we->angleDelta(),
1940  we->buttons(),
1941  we->modifiers(),
1942  we->phase(),
1943  we->inverted(),
1944  we->source());
1945 #else
1946  QWheelEvent wheelEvent(lastIntercept->mapFromGlobal(globalPos),
1947  globalPos,
1948  we->pixelDelta(),
1949  we->angleDelta(),
1950  0,
1951  Qt::Vertical,
1952  we->buttons(),
1953  we->modifiers(),
1954  we->phase(),
1955  we->source(),
1956  we->inverted());
1957 #endif
1958  QApplication::sendEvent(lastIntercept, &wheelEvent);
1959  break;
1960  }
1961  case QEvent::ContextMenu: {
1962  auto ce = static_cast<QContextMenuEvent*>(ev);
1963  lastIntercept = getChildAt(widget, ce->globalPos());
1964  QContextMenuEvent contextMenuEvent(ce->reason(),
1965  lastIntercept->mapFromGlobal(ce->globalPos()),
1966  ce->globalPos());
1967  QApplication::sendEvent(lastIntercept, &contextMenuEvent);
1968  break;
1969  }
1970  default:
1971  break;
1972  }
