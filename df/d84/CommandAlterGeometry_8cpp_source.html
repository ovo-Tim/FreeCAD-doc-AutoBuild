140 void CmdSketcherToggleConstruction::activated(int iMsg)
141 {
142  Q_UNUSED(iMsg);
143  // Option A: nothing is selected change creation mode from/to construction
144  if (Gui::Selection().countObjectsOfType<Sketcher::SketchObject>() == 0) {
145 
146  Gui::CommandManager& rcCmdMgr = Gui::Application::Instance->commandManager();
147 
148  if (geometryCreationMode == GeometryCreationMode::Construction) {
149  geometryCreationMode = GeometryCreationMode::Normal;
150  }
151  else {
152  geometryCreationMode = GeometryCreationMode::Construction;
153  }
154 
155  rcCmdMgr.updateCommands("ToggleConstruction", static_cast<int>(geometryCreationMode));
156  }
157  else // there was a selection, so operate in toggle mode.
158  {
159  // get the selection
160  std::vector<Gui::SelectionObject> selection;
161  selection =
162  getSelection().getSelectionEx(nullptr, Sketcher::SketchObject::getClassTypeId());
163 
164  auto* Obj = static_cast<Sketcher::SketchObject*>(selection[0].getObject());
165 
166  // only one sketch with its subelements are allowed to be selected
167  if (selection.size() != 1) {
168  Gui::TranslatedUserWarning(Obj,
169  QObject::tr("Wrong selection"),
170  QObject::tr("Select edge(s) from the sketch."));
171  return;
172  }
173 
174  // get the needed lists and objects
175  const std::vector<std::string>& SubNames = selection[0].getSubNames();
176  if (SubNames.empty()) {
177  Gui::TranslatedUserWarning(Obj,
178  QObject::tr("Wrong selection"),
179  QObject::tr("Select edge(s) from the sketch."));
180  return;
181  }
182 
183  // undo command open
184  openCommand(QT_TRANSLATE_NOOP("Command", "Toggle draft from/to draft"));
185 
186  // go through the selected subelements
187  bool verticesonly = true;
188 
189  for (const auto& subname : SubNames) {
190  if ((subname.size() > 4 && subname.substr(0, 4) == "Edge")
191  || (subname.size() > 12 && subname.substr(0, 12) == "ExternalEdge")) {
192  verticesonly = false;
193  }
194  }
195 
196  for (const auto& subname : SubNames) {
197  // It was decided to provide a special behaviour:
198  // Vertices will only be toggled to/from construction IF ONLY
199  // vertices are within the group.
200  // If there are a mixture of edges and vertices, vertices will be ignored.
201  //
202  // Why?
203  // Because it is quite common to box select geometry for toggling (specially in
204  // connection with carbon copy operations). In 99% of the cases the user does not
205  // want to toggle individual points during such operations. For the remaining 1%,
206  // in 90% of the cases the uses will select just the points only naturally.
207 
208 
209  // only handle edges
210  if (subname.size() > 4 && subname.substr(0, 4) == "Edge") {
211  int geoId = std::atoi(subname.substr(4, 4000).c_str()) - 1;
212  // issue the actual commands to toggle
213  Gui::cmdAppObjectArgs(Obj, "toggleConstruction(%d) ", geoId);
214  }
215  else if (subname.size() > 12 && subname.substr(0, 12) == "ExternalEdge") {
216  int geoId = GeoEnum::RefExt - std::atoi(subname.substr(12, 4000).c_str()) + 1;
217  Gui::cmdAppObjectArgs(Obj, "toggleConstruction(%d) ", geoId);
218  }
219  else if (verticesonly && subname.size() > 6 && subname.substr(0, 6) == "Vertex") {
220  int vertexId = std::atoi(subname.substr(6, 4000).c_str()) - 1;
221 
222  int geoId;
223  PointPos pos;
224  Obj->getGeoVertexIndex(vertexId, geoId, pos);
225 
226  auto geo = Obj->getGeometry(geoId);
227 
228  if (geo && geo->is<Part::GeomPoint>()) {
229  // issue the actual commands to toggle
230  Gui::cmdAppObjectArgs(Obj, "toggleConstruction(%d) ", geoId);
231  }
232  }
233  }
234  // finish the transaction and update
235  commitCommand();
236 
237  tryAutoRecompute(Obj);
238 
239  // clear the selection (convenience)
240  getSelection().clearSelection();
241  }
