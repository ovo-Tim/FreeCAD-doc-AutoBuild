33 class Kink(object):
34  """A Kink represents the angle at which two moves connect.
35  A positive kink angle represents a move to the left, and a negative angle represents a move to the right.
36  """
37 
38  def __init__(self, m0, m1):
39  if m1 is None:
40  m1 = m0[1]
41  m0 = m0[0]
42  self.m0 = m0
43  self.m1 = m1
44  self.t0 = m0.anglesOfTangents()[1]
45  self.t1 = m1.anglesOfTangents()[0]
46  if Path.Geom.isRoughly(self.t0, self.t1):
47  self.defl = 0
48  else:
49  self.defl = Path.Geom.normalizeAngle(self.t1 - self.t0)
50 
51  def isKink(self):
52  return self.defl != 0
53 
54  def goesLeft(self):
55  return self.defl > 0
56 
57  def goesRight(self):
58  return self.defl < 0
59 
60  def deflection(self):
61  """deflection() ... returns the tangential difference of the two edges at their intersection"""
62  return self.defl
63 
64  def normAngle(self):
65  """normAngle() ... returns the angle opposite between the two tangents"""
66 
67  # The normal angle is perpendicular to the "average tangent" of the kink. The question
68  # is into which direction to turn. One lies in the center between the two edges and the
69  # other is opposite to that. As it turns out, the magnitude of the tangents tell it all.
70  if self.t0 > self.t1:
71  return Path.Geom.normalizeAngle((self.t0 + self.t1 + math.pi) / 2)
72  return Path.Geom.normalizeAngle((self.t0 + self.t1 - math.pi) / 2)
73 
74  def position(self):
75  """position() ... position of the edge's intersection"""
76  return self.m0.positionEnd()
77 
78  def x(self):
79  return self.position().x
80 
81  def y(self):
82  return self.position().y
83 
84  def __repr__(self):
85  return f"({self.x():.4f}, {self.y():.4f})[t0={180*self.t0/math.pi:.2f}, t1={180*self.t1/math.pi:.2f}, deflection={180*self.defl/math.pi:.2f}, normAngle={180*self.normAngle()/math.pi:.2f}]"

131 def generate_bone(kink, length, angle):
132  dx = length * math.cos(angle)
133  dy = length * math.sin(angle)
134  p0 = kink.position()
135 
136  if Path.Geom.isRoughly(0, dx):
137  # vertical bone
138  moveIn = PathLanguage.MoveStraight(kink.position(), "G1", {"Y": p0.y + dy})
139  moveOut = PathLanguage.MoveStraight(moveIn.positionEnd(), "G1", {"Y": p0.y})
140  elif Path.Geom.isRoughly(0, dy):
141  # horizontal bone
142  moveIn = PathLanguage.MoveStraight(kink.position(), "G1", {"X": p0.x + dx})
143  moveOut = PathLanguage.MoveStraight(moveIn.positionEnd(), "G1", {"X": p0.x})
144  else:
145  moveIn = PathLanguage.MoveStraight(kink.position(), "G1", {"X": p0.x + dx, "Y": p0.y + dy})
146  moveOut = PathLanguage.MoveStraight(moveIn.positionEnd(), "G1", {"X": p0.x, "Y": p0.y})
147 
148  return Bone(kink, angle, length, [moveIn, moveOut])
