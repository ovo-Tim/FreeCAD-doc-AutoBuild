77  def parameterization (self, pts, a, closed):
78  """Computes a knot Sequence for a set of points.
79  fac (0-1) : parameterization factor
80  fac = 0 -> Uniform / fac=0.5 -> Centripetal / fac=1.0 -> Chord-Length
81  """
82  if closed: # we need to add the first point as the end point
83  pts.append(pts[0])
84  params = [0]
85  for i in range(1,len(pts)):
86  p = pts[i].sub(pts[i-1])
87  pl = pow(p.Length,a)
88  params.append(params[-1] + pl)
89  return params
