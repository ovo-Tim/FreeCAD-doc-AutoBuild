482  def orderFacesAroundCenterOfMass(self, subObjTups):
483  """orderFacesAroundCenterOfMass(subObjTups)
484  Order list of faces by center of mass in angular order around
485  average center of mass for all faces. Positive X-axis is zero degrees.
486  return: subObjTups [ordered/sorted]"""
487  import math
488 
489  newList = []
490  vectList = []
491  comList = []
492  sortList = []
493  subCnt = 0
494  sumCom = FreeCAD.Vector(0.0, 0.0, 0.0)
495  avgCom = FreeCAD.Vector(0.0, 0.0, 0.0)
496 
497  def getDrctn(vectItem):
498  return vectItem[3]
499 
500  def getFaceIdx(sub):
501  return int(sub.replace("Face", "")) - 1
502 
503  # get CenterOfMass for each face and add to sumCenterOfMass for average calculation
504  for sub, face in subObjTups:
505  # for (bsNm, fIdx, eIdx, vIdx) in bfevList:
506  # face = FreeCAD.ActiveDocument.getObject(bsNm).Shape.Faces[fIdx]
507  subCnt += 1
508  com = face.CenterOfMass
509  comList.append((sub, face, com))
510  sumCom = sumCom.add(com) # add sub COM to sum
511 
512  # Calculate average CenterOfMass for all faces combined
513  avgCom.x = sumCom.x / subCnt
514  avgCom.y = sumCom.y / subCnt
515  avgCom.z = sumCom.z / subCnt
516 
517  # calculate vector (mag, direct) for each face from avgCom
518  for sub, face, com in comList:
519  adjCom = com.sub(avgCom) # effectively treats avgCom as origin for each face.
520  mag = math.sqrt(adjCom.x**2 + adjCom.y**2) # adjCom.Length without Z values
521  drctn = 0.0
522  # Determine direction of vector
523  if adjCom.x > 0.0:
524  if adjCom.y > 0.0: # Q1
525  drctn = math.degrees(math.atan(adjCom.y / adjCom.x))
526  elif adjCom.y < 0.0:
527  drctn = -math.degrees(math.atan(adjCom.x / adjCom.y)) + 270.0
528  elif adjCom.y == 0.0:
529  drctn = 0.0
530  elif adjCom.x < 0.0:
531  if adjCom.y < 0.0:
532  drctn = math.degrees(math.atan(adjCom.y / adjCom.x)) + 180.0
533  elif adjCom.y > 0.0:
534  drctn = -math.degrees(math.atan(adjCom.x / adjCom.y)) + 90.0
535  elif adjCom.y == 0.0:
536  drctn = 180.0
537  elif adjCom.x == 0.0:
538  if adjCom.y < 0.0:
539  drctn = 270.0
540  elif adjCom.y > 0.0:
541  drctn = 90.0
542  vectList.append((sub, face, mag, drctn))
543 
544  # Sort faces by directional component of vector
545  sortList = sorted(vectList, key=getDrctn)
546 
547  # remove magnitute and direction values
548  for sub, face, mag, drctn in sortList:
549  newList.append((sub, face))
550 
551  # Rotate list items so highest face is first
552  zmax = newList[0][1].BoundBox.ZMax
553  idx = 0
554  for i in range(0, len(newList)):
555  (sub, face) = newList[i]
556  fIdx = getFaceIdx(sub)
557  # face = FreeCAD.ActiveDocument.getObject(bsNm).Shape.Faces[fIdx]
558  if face.BoundBox.ZMax > zmax:
559  zmax = face.BoundBox.ZMax
560  idx = i
561  if face.BoundBox.ZMax == zmax:
562  if fIdx < getFaceIdx(newList[idx][0]):
563  idx = i
564  if idx > 0:
565  for z in range(0, idx):
566  newList.append(newList.pop(0))
567 
568  return newList
