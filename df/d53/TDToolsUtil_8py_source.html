128 def getCoordinateVectors(view):
129  '''
130  (px,py,pz) = getCoordinateVectors(view)
131  view ... selected view
132  (px,py,pz) ... returned tuple of vectors (App.Vector)
133  calculate projected vectors of x-, y- and z-axis
134  '''
135  diagonal = view.Direction
136  xDirection = view.XDirection
137  origin = App.Vector()
138  xAxisProj = App.Vector(1,0,0).projectToPlane(origin,diagonal)
139  yAxisProj = App.Vector(0,1,0).projectToPlane(origin,diagonal)
140  zAxisProj = App.Vector(0,0,1).projectToPlane(origin,diagonal)
141  # create wire polygon and roll back in 3D
142  wire3D = Part.makePolygon([xAxisProj,yAxisProj,zAxisProj,xDirection,xAxisProj])
143  rotation3D = App.Rotation(diagonal,App.Vector(0,0,1))
144  wire3D.Placement.Rotation = rotation3D
145  px = wire3D.Edges[0].Vertexes[0].Point
146  py = wire3D.Edges[1].Vertexes[0].Point
147  pz = wire3D.Edges[2].Vertexes[0].Point
148  pDir = wire3D.Edges[3].Vertexes[0].Point
149  pDir.z = 0.0
150  # rotate inside drawing plane
151  wire2D = Part.makePolygon([px,py,pz,px])
152  rotation2D = App.Rotation(pDir,App.Vector(1,0,0))
153  wire2D.Placement.Rotation = rotation2D
154  px = wire2D.Edges[0].Vertexes[0].Point
155  py = wire2D.Edges[1].Vertexes[0].Point
156  pz = wire2D.Edges[2].Vertexes[0].Point
157  return (px,py,pz)
