85  Py::Object startServer(const Py::Tuple& args)
86  {
87  const char* addr = "127.0.0.1";
88  int port = 0;
89  if (!PyArg_ParseTuple(args.ptr(), "|si", &addr, &port)) {
90  throw Py::Exception();
91  }
92  if (port > USHRT_MAX) {
93  throw Py::OverflowError("port number is greater than maximum");
94  }
95  else if (port < 0) {
96  throw Py::OverflowError("port number is lower than 0");
97  }
98 
99  AppServer* server = new AppServer();
100  if (server->listen(QHostAddress(QString::fromLatin1(addr)), port)) {
101  QString a = server->serverAddress().toString();
102  quint16 p = server->serverPort();
103  Py::Tuple t(2);
104  t.setItem(0, Py::String((const char*)a.toLatin1()));
105  t.setItem(1, Py::Long(p));
106  return t;

116  Py::Object waitForConnection(const Py::Tuple& args)
117  {
118  const char* addr = "127.0.0.1";
119  int port = 0;
120  int timeout = 0;
121  if (!PyArg_ParseTuple(args.ptr(), "|sii", &addr, &port, &timeout)) {
122  throw Py::Exception();
123  }
124  if (port > USHRT_MAX) {
125  throw Py::OverflowError("port number is greater than maximum");
126  }
127  else if (port < 0) {
128  throw Py::OverflowError("port number is lower than 0");
129  }
130 
131  try {
132  AppServer server(true);
133  if (server.listen(QHostAddress(QString::fromLatin1(addr)), port)) {
134  bool ok = server.waitForNewConnection(timeout);
135  QTcpSocket* socket = server.nextPendingConnection();
136  if (socket) {
137  socket->waitForReadyRead();
138  }
139 
140  server.close();
141  return Py::Boolean(ok);
142  }
143  else {
144  std::stringstream out;
145  out << "Server failed to listen at address " << addr << " and port " << port;
146  throw Py::RuntimeError(out.str());
147  }
148  }
149  catch (const Base::SystemExitException& e) {
150  throw Py::RuntimeError(e.what());
151  }
152  }
