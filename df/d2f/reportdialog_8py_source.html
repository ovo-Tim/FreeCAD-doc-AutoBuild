62  def _getText(self, report):
63  text = ""
64  for i in report.infos:
65  line = "<b>Info:</b> %s" % i
66  text += "%s<br>" % self._getColoredLine(line, INFO_COLOR)
67  for w in report.warnings:
68  line = "<b>Warning:</b> %s" % w
69  text += "%s<br>" % self._getColoredLine(line, WARNING_COLOR)
70  for e in report.errors:
71  line = "<b>Error:</b> %s" % e
72  text += "%s<br>" % self._getColoredLine(line, ERROR_COLOR)
73  return text
