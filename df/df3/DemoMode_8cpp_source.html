163 SbVec3f DemoMode::getDirection(Gui::View3DInventor* view) const
164 {
165  SoCamera* cam = view->getViewer()->getSoRenderManager()->getCamera();
166  if (!cam) {
167  return this->viewAxis;
168  }
169  SbRotation rot = cam->orientation.getValue();
170  SbRotation inv = rot.inverse();
171  SbVec3f vec(this->viewAxis);
172  inv.multVec(vec, vec);
173  if (vec.length() < FLT_EPSILON) {
174  vec = this->viewAxis;
175  }
176  vec.normalize();
177  return vec;
178 }

180 void DemoMode::onAngleSliderValueChanged(int v)
181 {
182  Gui::View3DInventor* view = activeView();
183  if (view) {
184  SoCamera* cam = view->getViewer()->getSoRenderManager()->getCamera();
185  if (!cam) {
186  return;
187  }
188  auto angle = Base::toRadians<float>(/*90-v*/ v - this->oldvalue);
189  SbRotation rot(SbVec3f(-1, 0, 0), angle);
190  reorientCamera(cam, rot);
191  this->oldvalue = v;
192  if (view->getViewer()->isSpinning()) {
193  startAnimation(view);
194  }
195  }
196 }

198 void DemoMode::reorientCamera(SoCamera* cam, const SbRotation& rot)
199 {
200  // Find global coordinates of focal point.
201  SbVec3f direction;
202  cam->orientation.getValue().multVec(SbVec3f(0, 0, -1), direction);
203  SbVec3f focalpoint = cam->position.getValue() + cam->focalDistance.getValue() * direction;
204 
205  // Set new orientation value by accumulating the new rotation.
206  cam->orientation = rot * cam->orientation.getValue();
207 
208  // Reposition camera so we are still pointing at the same old focal point.
209  cam->orientation.getValue().multVec(SbVec3f(0, 0, -1), direction);
210  cam->position = focalpoint - cam->focalDistance.getValue() * direction;
211 }

222 void DemoMode::onPlayButtonToggled(bool pressed)
223 {
224  Gui::View3DInventor* view = activeView();
225  if (view) {
226  if (pressed) {
227  if (!view->getViewer()->isSpinning()) {
228  SoCamera* cam = view->getViewer()->getSoRenderManager()->getCamera();
229  if (cam) {
230  SbRotation rot = cam->orientation.getValue();
231  SbVec3f vec(0, -1, 0);
232  rot.multVec(vec, this->viewAxis);
233  }
234  }
235 
236  startAnimation(view);
237  ui->playButton->setText(tr("Stop"));
238  }
239  else {
240  view->getViewer()->stopAnimating();
241  ui->playButton->setText(tr("Play"));
242  }
243  }
244 }
