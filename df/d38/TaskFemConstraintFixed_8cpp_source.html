105 void TaskFemConstraintFixed::addToSelection()
106 {
107  // gets vector of selected objects of active document
108  std::vector<Gui::SelectionObject> selection = Gui::Selection().getSelectionEx();
109  if (selection.empty()) {
110  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
111  return;
112  }
113  Fem::ConstraintFixed* pcConstraint = ConstraintView->getObject<Fem::ConstraintFixed>();
114  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
115  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
116 
117  for (auto& it : selection) { // for every selected object
118  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
119  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
120  return;
121  }
122  std::vector<std::string> subNames = it.getSubNames();
123  App::DocumentObject* obj =
124  ConstraintView->getObject()->getDocument()->getObject(it.getFeatName());
125  for (const auto& subName : subNames) { // for every selected sub element
126  bool addMe = true;
127  for (std::vector<std::string>::iterator itr =
128  std::find(SubElements.begin(), SubElements.end(), subName);
129  itr != SubElements.end();
130  itr = std::find(++itr,
131  SubElements.end(),
132  subName)) { // for every sub element in selection that
133  // matches one in old list
134  if (obj
135  == Objects[std::distance(
136  SubElements.begin(),
137  itr)]) { // if selected sub element's object equals the one in old list
138  // then it was added before so don't add
139  addMe = false;
140  }
141  }
142  // limit constraint such that only vertexes or faces or edges can be used depending
143  // on what was selected first
144  std::string searchStr;
145  if (subName.find("Vertex") != std::string::npos) {
146  searchStr = "Vertex";
147  }
148  else if (subName.find("Edge") != std::string::npos) {
149  searchStr = "Edge";
150  }
151  else {
152  searchStr = "Face";
153  }
154  for (const auto& SubElement : SubElements) {
155  if (SubElement.find(searchStr) == std::string::npos) {
156  QString msg = tr("Only one type of selection (vertex, face or edge) per "
157  "analysis feature allowed!");
158  QMessageBox::warning(this, tr("Selection error"), msg);
159  addMe = false;
160  break;
161  }
162  }
163  if (addMe) {
164  QSignalBlocker block(ui->lw_references);
165  Objects.push_back(obj);
166  SubElements.push_back(subName);
167  ui->lw_references->addItem(makeRefText(obj, subName));
168  }
169  }
170  }
171  // Update UI
172  pcConstraint->References.setValues(Objects, SubElements);
173  updateUI();
174 }

176 void TaskFemConstraintFixed::removeFromSelection()
177 {
178  // gets vector of selected objects of active document
179  std::vector<Gui::SelectionObject> selection = Gui::Selection().getSelectionEx();
180  if (selection.empty()) {
181  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
182  return;
183  }
184  Fem::ConstraintFixed* pcConstraint = ConstraintView->getObject<Fem::ConstraintFixed>();
185  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
186  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
187  std::vector<size_t> itemsToDel;
188  for (const auto& it : selection) { // for every selected object
189  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
190  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
191  return;
192  }
193  const std::vector<std::string>& subNames = it.getSubNames();
194  const App::DocumentObject* obj = it.getObject();
195 
196  for (const auto& subName : subNames) { // for every selected sub element
197  for (std::vector<std::string>::iterator itr =
198  std::find(SubElements.begin(), SubElements.end(), subName);
199  itr != SubElements.end();
200  itr = std::find(++itr,
201  SubElements.end(),
202  subName)) { // for every sub element in selection that
203  // matches one in old list
204  if (obj
205  == Objects[std::distance(
206  SubElements.begin(),
207  itr)]) { // if selected sub element's object equals the one in old list
208  // then it was added before so mark for deletion
209  itemsToDel.push_back(std::distance(SubElements.begin(), itr));
210  }
211  }
212  }
213  }
214  std::sort(itemsToDel.begin(), itemsToDel.end());
215  while (!itemsToDel.empty()) {
216  Objects.erase(Objects.begin() + itemsToDel.back());
217  SubElements.erase(SubElements.begin() + itemsToDel.back());
218  itemsToDel.pop_back();
219  }
220  // Update UI
221  {
222  QSignalBlocker block(ui->lw_references);
223  ui->lw_references->clear();
224  for (unsigned int j = 0; j < Objects.size(); j++) {
225  ui->lw_references->addItem(makeRefText(Objects[j], SubElements[j]));
226  }
227  }
228  pcConstraint->References.setValues(Objects, SubElements);
229  updateUI();
230 }
