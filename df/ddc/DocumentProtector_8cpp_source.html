210 // Posts an event and waits for it to finish processing
211 void DocumentReceiver::postEventAndWait(QEvent* e)
212 {
213  QThread *currentThread = QThread::currentThread();
214  QThread *thr = this->thread(); // this is the main thread
215 
216  if (currentThread == thr) {
217  // we're in the main thread
218  QCoreApplication::sendEvent(this, e);
219  delete e;
220  }
221  else {
222  // NOTE: We send an event to this instance that's why it is important
223  // that this object is part of the main thread
224  QSemaphore semaphore;
225  static_cast<AbstractCustomProtectorEvent*>(e)->semaphore = &semaphore;
226  QCoreApplication::postEvent(this, e);
227  // wait until the event has been processed
228  semaphore.acquire();
229  }
230 }
