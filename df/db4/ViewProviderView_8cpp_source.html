64 void ViewProviderDrawingView::show(void)
65 {
66  ViewProviderDocumentObject::show();
67 
68  App::DocumentObject* obj = getObject();
69  if (!obj || obj->isRestoring()) {
70  return;
71  }
72  if (obj->isDerivedFrom<Drawing::FeatureView>()) {
73  // The 'Visible' property is marked as 'Output'. To update the drawing on recompute
74  // the parent page object is touched.
75  static_cast<Drawing::FeatureView*>(obj)->Visible.setValue(true);
76  std::vector<App::DocumentObject*> inp = obj->getInList();
77  for (std::vector<App::DocumentObject*>::iterator it = inp.begin(); it != inp.end(); ++it) {
78  (*it)->touch();
79  }
80  }
81 }
