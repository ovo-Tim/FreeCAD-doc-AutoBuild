72  """of(ptRef, pt)

164 def isVertical(obj):
165  """isVertical(obj) ... answer True if obj points into Z"""
166  if type(obj) == FreeCAD.Vector:
167  return isRoughly(obj.x, 0) and isRoughly(obj.y, 0)
168 
169  if obj.ShapeType == "Face":
170  if type(obj.Surface) == Part.Plane:
171  return isHorizontal(obj.Surface.Axis)
172  if type(obj.Surface) == Part.Cylinder or type(obj.Surface) == Part.Cone:
173  return isVertical(obj.Surface.Axis)
174  if type(obj.Surface) == Part.Sphere:
175  return True
176  if type(obj.Surface) == Part.SurfaceOfExtrusion:
177  return isVertical(obj.Surface.Direction)
178  if type(obj.Surface) == Part.SurfaceOfRevolution:
179  return isHorizontal(obj.Surface.Direction)

181  Path.Log.info(

185  return None
186 
187  if obj.ShapeType == "Edge":
188  if type(obj.Curve) == Part.Line or type(obj.Curve) == Part.LineSegment:
189  return isVertical(obj.Vertexes[1].Point - obj.Vertexes[0].Point)
190  if (
191  type(obj.Curve) == Part.Circle or type(obj.Curve) == Part.Ellipse
192  ): # or type(obj.Curve) == Part.BSplineCurve:
193  return isHorizontal(obj.Curve.Axis)
194  if type(obj.Curve) == Part.BezierCurve:
195  # the current assumption is that a bezier curve is vertical if its end points are vertical
196  return isVertical(obj.Curve.EndPoint - obj.Curve.StartPoint)

254 def speedBetweenPoints(p0, p1, hSpeed, vSpeed):
255  if isRoughly(hSpeed, vSpeed):
256  return hSpeed
257 
258  d = p1 - p0
259  if isRoughly(0.0, d.z):
260  return hSpeed
261  if isRoughly(0.0, d.x) and isRoughly(0.0, d.y):
262  return vSpeed
263  # need to interpolate between hSpeed and vSpeed depending on the pitch
264  pitch = 2 * math.atan2(xy(d).Length, math.fabs(d.z)) / math.pi
265  while pitch < 0:
266  pitch = pitch + 1
267  while pitch > 1:
268  pitch = pitch - 1
269  Path.Log.debug(
270  " pitch = %g %g (%.2f, %.2f, %.2f) -> %.2f"
271  % (pitch, math.atan2(xy(d).Length, d.z), d.x, d.y, d.z, xy(d).Length)
272  )
273  speed = vSpeed + pitch * (hSpeed - vSpeed)
274  if speed > hSpeed and speed > vSpeed:
275  return max(hSpeed, vSpeed)
276  if speed < hSpeed and speed < vSpeed:
277  return min(hSpeed, vSpeed)
278  return speed

281 def cmdsForEdge(edge, flip=False, useHelixForBSpline=True, segm=50, hSpeed=0, vSpeed=0):
282  """cmdsForEdge(edge, flip=False, useHelixForBSpline=True, segm=50) -> List(Path.Command)
283  Returns a list of Path.Command representing the given edge.
284  If flip is True the edge is considered to be backwards.
285  If useHelixForBSpline is True an Edge based on a BSplineCurve is considered
286  to represent a helix and results in G2 or G3 command. Otherwise edge has
287  no direct Path.Command mapping and will be approximated by straight segments.
288  segm is a factor for the segmentation of arbitrary curves not mapped to G1/2/3
289  commands. The higher the value the more segments will be used."""
290  pt = edge.valueAt(edge.LastParameter) if not flip else edge.valueAt(edge.FirstParameter)
291  params = {"X": pt.x, "Y": pt.y, "Z": pt.z}
292  if type(edge.Curve) == Part.Line or type(edge.Curve) == Part.LineSegment:
293  if hSpeed > 0 and vSpeed > 0:
294  pt2 = (
295  edge.valueAt(edge.FirstParameter) if not flip else edge.valueAt(edge.LastParameter)
296  )
297  params.update({"F": speedBetweenPoints(pt, pt2, hSpeed, vSpeed)})
298  commands = [Path.Command("G1", params)]
299  else:
300  p1 = edge.valueAt(edge.FirstParameter) if not flip else edge.valueAt(edge.LastParameter)
301  p2 = edge.valueAt((edge.FirstParameter + edge.LastParameter) / 2)
302  p3 = pt
303 
304  if hasattr(edge.Curve, "Axis") and (
305  (
306  type(edge.Curve) == Part.Circle
307  and isRoughly(edge.Curve.Axis.x, 0)
308  and isRoughly(edge.Curve.Axis.y, 0)
309  )
310  or (useHelixForBSpline and type(edge.Curve) == Part.BSplineCurve)
311  ):
312  # This is an arc or a helix and it should be represented by a simple G2/G3 command
313  if edge.Curve.Axis.z < 0:
314  cmd = "G2" if not flip else "G3"
315  else:
316  cmd = "G3" if not flip else "G2"
317 
318  if pointsCoincide(p1, p3):
319  # A full circle
320  offset = edge.Curve.Center - pt
321  else:
322  pd = Part.Circle(xy(p1), xy(p2), xy(p3)).Center
323  Path.Log.debug(
324  "**** %s.%d: (%.2f, %.2f, %.2f) - (%.2f, %.2f, %.2f) - (%.2f, %.2f, %.2f) -> center=(%.2f, %.2f)"
325  % (
326  cmd,
327  flip,
328  p1.x,
329  p1.y,
330  p1.z,
331  p2.x,
332  p2.y,
333  p2.z,
334  p3.x,
335  p3.y,
336  p3.z,
337  pd.x,
338  pd.y,
339  )
340  )
341 
342  # Have to calculate the center in the XY plane, using pd leads to an error if this is a helix
343  pa = xy(p1)
344  pb = xy(p2)
345  pc = xy(p3)
346  offset = Part.Circle(pa, pb, pc).Center - pa
347 
348  Path.Log.debug(
349  "**** (%.2f, %.2f, %.2f) - (%.2f, %.2f, %.2f)"
350  % (pa.x, pa.y, pa.z, pc.x, pc.y, pc.z)
351  )
352  Path.Log.debug(
353  "**** (%.2f, %.2f, %.2f) - (%.2f, %.2f, %.2f)"
354  % (pb.x, pb.y, pb.z, pd.x, pd.y, pd.z)
355  )
356  Path.Log.debug("**** (%.2f, %.2f, %.2f)" % (offset.x, offset.y, offset.z))
357 
358  params.update({"I": offset.x, "J": offset.y, "K": (p3.z - p1.z) / 2})
359  # G2/G3 commands are always performed at hSpeed
360  if hSpeed > 0:
361  params.update({"F": hSpeed})
362  commands = [Path.Command(cmd, params)]
363 
364  else:
365  # We're dealing with a helix or a more complex shape and it has to get approximated
366  # by a number of straight segments
367  points = edge.discretize(Deflection=0.01)
368  if flip:
369  points = points[::-1]
370 
371  commands = []
372  if points:
373  p0 = points[0]
374  for p in points[1:]:
375  params = {"X": p.x, "Y": p.y, "Z": p.z}
376  if hSpeed > 0 and vSpeed > 0:
377  params["F"] = speedBetweenPoints(p0, p, hSpeed, vSpeed)
378  cmd = Path.Command("G1", params)
379  # print("***** {}".format(cmd))
380  commands.append(cmd)
381  p0 = p
382  # print commands
383  return commands

386 def edgeForCmd(cmd, startPoint):
387  """edgeForCmd(cmd, startPoint).
388  Returns an Edge representing the given command, assuming a given startPoint."""
389 
390  Path.Log.debug("cmd: {}".format(cmd))
391  Path.Log.debug("startpoint {}".format(startPoint))
392 
393  endPoint = commandEndPoint(cmd, startPoint)
394  if (cmd.Name in CmdMoveStraight) or (cmd.Name in CmdMoveRapid):
395  if pointsCoincide(startPoint, endPoint):
396  return None
397  return Part.Edge(Part.LineSegment(startPoint, endPoint))
398 
399  if cmd.Name in CmdMoveArc:
400  center = startPoint + commandEndPoint(cmd, Vector(0, 0, 0), "I", "J", "K")
401  A = xy(startPoint - center)
402  B = xy(endPoint - center)
403  d = -B.x * A.y + B.y * A.x
404 
405  if isRoughly(d, 0, 0.005):
406  Path.Log.debug(
407  "Half circle arc at: (%.2f, %.2f, %.2f)" % (center.x, center.y, center.z)
408  )
409  # we're dealing with half a circle here
410  angle = getAngle(A) + math.pi / 2
411  if cmd.Name in CmdMoveCW:
412  angle -= math.pi
413  else:
414  C = A + B
415  angle = getAngle(C)
416  Path.Log.debug(
417  "Arc (%8f) at: (%.2f, %.2f, %.2f) -> angle=%f"
418  % (d, center.x, center.y, center.z, angle / math.pi)
419  )
420 
421  R = A.Length
422  Path.Log.debug(
423  "arc: p1=(%.2f, %.2f) p2=(%.2f, %.2f) -> center=(%.2f, %.2f)"
424  % (startPoint.x, startPoint.y, endPoint.x, endPoint.y, center.x, center.y)
425  )
426  Path.Log.debug("arc: A=(%.2f, %.2f) B=(%.2f, %.2f) -> d=%.2f" % (A.x, A.y, B.x, B.y, d))
427  Path.Log.debug("arc: R=%.2f angle=%.2f" % (R, angle / math.pi))
428  if isRoughly(startPoint.z, endPoint.z):
429  midPoint = center + Vector(math.cos(angle), math.sin(angle), 0) * R
430  Path.Log.debug(
431  "arc: (%.2f, %.2f) -> (%.2f, %.2f) -> (%.2f, %.2f)"
432  % (
433  startPoint.x,
434  startPoint.y,
435  midPoint.x,
436  midPoint.y,
437  endPoint.x,
438  endPoint.y,
439  )
440  )
441  Path.Log.debug("StartPoint:{}".format(startPoint))
442  Path.Log.debug("MidPoint:{}".format(midPoint))
443  Path.Log.debug("EndPoint:{}".format(endPoint))
444 
445  if pointsCoincide(startPoint, endPoint, 0.001):
446  return Part.makeCircle(R, center, FreeCAD.Vector(0, 0, 1))
447  else:
448  return Part.Edge(Part.Arc(startPoint, midPoint, endPoint))
449 
450  # It's a Helix
451  # print('angle: A=%.2f B=%.2f' % (getAngle(A)/math.pi, getAngle(B)/math.pi))
452  if cmd.Name in CmdMoveCW:
453  cw = True
454  else:
455  cw = False
456  angle = diffAngle(getAngle(A), getAngle(B), "CW" if cw else "CCW")
457  height = endPoint.z - startPoint.z
458  pitch = height * math.fabs(2 * math.pi / angle)
459  if angle > 0:
460  cw = not cw
461  # print("Helix: R=%.2f h=%.2f angle=%.2f pitch=%.2f" % (R, height, angle/math.pi, pitch))
462  helix = Part.makeHelix(pitch, height, R, 0, not cw)
463  helix.rotate(Vector(), Vector(0, 0, 1), 180 * getAngle(A) / math.pi)
464  e = helix.Edges[0]
465  helix.translate(startPoint - e.valueAt(e.FirstParameter))
466  return helix.Edges[0]
467  return None

610 def flipEdge(edge):
611  """flipEdge(edge)
612  Flips given edge around so the new Vertexes[0] was the old Vertexes[-1] and vice versa, without changing the shape.
613  Currently only lines, line segments, circles and arcs are supported."""
614 
615  if Part.Line == type(edge.Curve) and not edge.Vertexes:
616  return Part.Edge(
617  Part.Line(edge.valueAt(edge.LastParameter), edge.valueAt(edge.FirstParameter))
618  )
619  elif Part.Line == type(edge.Curve) or Part.LineSegment == type(edge.Curve):
620  return Part.Edge(Part.LineSegment(edge.Vertexes[-1].Point, edge.Vertexes[0].Point))
621  elif Part.Circle == type(edge.Curve):
622  # Create an inverted circle
623  circle = Part.Circle(edge.Curve.Center, -edge.Curve.Axis, edge.Curve.Radius)
624  # Rotate the circle appropriately so it starts at edge.valueAt(edge.LastParameter)
625  circle.rotate(
626  FreeCAD.Placement(
627  circle.Center,
628  circle.Axis,
629  180 - math.degrees(edge.LastParameter + edge.Curve.AngleXU),
630  )
631  )
632  # Now the edge always starts at 0 and LastParameter is the value range
633  arc = Part.Edge(circle, 0, edge.LastParameter - edge.FirstParameter)
634  return arc
635  elif type(edge.Curve) in [Part.BSplineCurve, Part.BezierCurve]:
636  if type(edge.Curve) == Part.BSplineCurve:
637  spline = edge.Curve
638  else:
639  spline = edge.Curve.toBSpline()
640 
641  mults = spline.getMultiplicities()
642  weights = spline.getWeights()
643  knots = spline.getKnots()
644  poles = spline.getPoles()
645  perio = spline.isPeriodic()
646  ratio = spline.isRational()
647  degree = spline.Degree
648 
649  ma = max(knots)
650  mi = min(knots)
651  knots = [ma + mi - k for k in knots]
652 
653  mults.reverse()
654  weights.reverse()
655  poles.reverse()
656  knots.reverse()
657 
658  flipped = Part.BSplineCurve()
659  flipped.buildFromPolesMultsKnots(poles, mults, knots, perio, degree, weights, ratio)
660 
661  return Part.Edge(flipped, ma + mi - edge.LastParameter, ma + mi - edge.FirstParameter)
662  elif type(edge.Curve) == Part.OffsetCurve:
663  return edge.reversed()
664 
665  Path.Log.warning(translate("PathGeom", "%s not supported for flipping") % type(edge.Curve))
