279 void TaskRichAnno::createAnnoFeature()
280 {
281 // Base::Console().Message("TRA::createAnnoFeature()");
282  const std::string objectName{QT_TR_NOOP("RichTextAnnotation")};
283  std::string annoName = m_basePage->getDocument()->getUniqueObjectName(objectName.c_str());
284  std::string generatedSuffix {annoName.substr(objectName.length())};
285  std::string annoType = "TechDraw::DrawRichAnno";
286 
287  std::string PageName = m_basePage->getNameInDocument();
288 
289  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create Anno"));
290  Command::doCommand(Command::Doc, "App.activeDocument().addObject('%s', '%s')",
291  annoType.c_str(), annoName.c_str());
292  Command::doCommand(Command::Doc, "App.activeDocument().%s.addView(App.activeDocument().%s)",
293  PageName.c_str(), annoName.c_str());
294 
295  if (m_baseFeat) {
296  Command::doCommand(Command::Doc, "App.activeDocument().%s.AnnoParent = App.activeDocument().%s",
297  annoName.c_str(), m_baseFeat->getNameInDocument());
298  }
299  App::DocumentObject* obj = m_basePage->getDocument()->getObject(annoName.c_str());
300  if (!obj) {
301  throw Base::RuntimeError("TaskRichAnno - new RichAnno object not found");
302  }
303  if (obj->isDerivedFrom(TechDraw::DrawRichAnno::getClassTypeId())) {
304  m_annoFeat = static_cast<TechDraw::DrawRichAnno*>(obj);
305  commonFeatureUpdate();
306  if (m_baseFeat) {
307  QPointF qTemp = calcTextStartPos(m_annoFeat->getScale());
308  Base::Vector3d vTemp(qTemp.x(), qTemp.y());
309  m_annoFeat->X.setValue(Rez::appX(vTemp.x));
310  m_annoFeat->Y.setValue(Rez::appX(vTemp.y));
311  } else {
312  //if we don't have a base feature, we can't calculate start position, so just put it mid-page
313  m_annoFeat->X.setValue(m_basePage->getPageWidth()/2.0);
314  m_annoFeat->Y.setValue(m_basePage->getPageHeight()/2.0);
315  }
316  }
317 
318  if (m_annoFeat) {
319  Gui::ViewProvider* vp = QGIView::getViewProvider(m_annoFeat);
320  auto annoVP = dynamic_cast<ViewProviderRichAnno*>(vp);
321  if (annoVP) {
322  App::Color ac;
323  ac.setValue<QColor>(ui->cpFrameColor->color());
324  annoVP->LineColor.setValue(ac);
325  annoVP->LineWidth.setValue(ui->dsbWidth->rawValue());
326  annoVP->LineStyle.setValue(ui->cFrameStyle->currentIndex());
327  }
328  }
329 
330  std::string translatedObjectName{tr(objectName.c_str()).toStdString()};
331  obj->Label.setValue(translatedObjectName + generatedSuffix);
332 
333  Gui::Command::commitCommand();
334  Gui::Command::updateActive();
335 
336  //trigger claimChildren in tree
337  if (m_baseFeat) {
338  m_baseFeat->touch();
339  }
340 
341  m_basePage->touch();
342 
343  if (m_annoFeat) {
344  m_annoFeat->requestPaint();
345  }
346 }

399 //we don't know the bounding rect of the text, so we have to calculate a reasonable
400 //guess at the size of the text block.
401 QPointF TaskRichAnno::calcTextStartPos(double scale)
402 {
403  Q_UNUSED(scale)
404 // Base::Console().Message("TRA::calcTextStartPos(%.3f)\n", scale);
405  double textWidth = 100.0;
406  double textHeight = 20.0;
407  double horizGap(20.0);
408  double tPosX(0.0);
409  double tPosY(0.0);
410 
411  double width = m_annoFeat->MaxWidth.getValue();
412  if (width > 0 ) {
413  textWidth = width;
414  }
415 
416  std::vector<Base::Vector3d> points;
417  if (m_baseFeat) {
418  if (m_baseFeat->isDerivedFrom(TechDraw::DrawLeaderLine::getClassTypeId())) {
419  TechDraw::DrawLeaderLine* dll = static_cast<TechDraw::DrawLeaderLine*>(m_baseFeat);
420  points = dll->WayPoints.getValues();
421  } else {
422 // Base::Console().Message("TRA::calcTextPos - m_baseFeat is not Leader\n");
423  return QPointF(0.0, 0.0);
424  }
425  } else {
426 // Base::Console().Message("TRA::calcStartPos - no m_baseFeat\n");
427  if (m_basePage) {
428  double w = Rez::guiX(m_basePage->getPageWidth() / 2.0);
429  double h = Rez::guiX(m_basePage->getPageHeight() / 2.0);
430  return QPointF(w, h);
431  } else {
432  Base::Console().Message("TRA::calcStartPos - no m_basePage\n"); //shouldn't happen. caught elsewhere
433  }
434  }
435 
436  if (!points.empty()) {
437  QPointF lastPoint(points.back().x, points.back().y);
438  QPointF firstPoint(points.front().x, points.front().y);
439  QPointF lastOffset = lastPoint - firstPoint;
440 
441  if (lastPoint.x() < firstPoint.x()) { //last is left of first
442  tPosX = lastOffset.x() - horizGap - textWidth; //left of last
443  tPosY = lastOffset.y() - textHeight;
444  } else { //last is right of first
445  tPosX = lastOffset.x() + horizGap; //right of last
446  tPosY = lastOffset.y() - textHeight;
447  }
448  }
449  return QPointF(tPosX, -tPosY);
450 }

467 bool TaskRichAnno::accept()
468 {
469 // Base::Console().Message("TRA::accept()\n");
470  if (m_inProgressLock) {
471 // Base::Console().Message("TRA::accept - edit in progress!!\n");
472  //TODO: kill MRTE dialog?
473  return true;
474  }
475 
476  Gui::Document* doc = Gui::Application::Instance->getDocument(m_basePage->getDocument());
477  if (!doc)
478  return false;
479 
480  if (getCreateMode()) {
481  createAnnoFeature();
482  } else {
483  updateAnnoFeature();
484  }
485 
486  m_annoFeat->requestPaint();
487 
488  Gui::Command::doCommand(Gui::Command::Gui, "Gui.ActiveDocument.resetEdit()");
489 
490  return true;
491 }

493 bool TaskRichAnno::reject()
494 {
495 // Base::Console().Message("TRA::reject()\n");
496  if (m_inProgressLock) {
497 // Base::Console().Message("TRA::reject - edit in progress!!\n");
498  return false;
499  }
500 
501  if (m_basePage) {
502  Gui::Document* doc = Gui::Application::Instance->getDocument(m_basePage->getDocument());
503  if (!doc) {
504  return false;
505  }
506  if (getCreateMode() && m_annoFeat) {
507  removeFeature();
508  }
509  }
510 
511  //make sure any dangling objects are cleaned up
512  Gui::Command::doCommand(Gui::Command::Gui, "App.activeDocument().recompute()");
513  Gui::Command::doCommand(Gui::Command::Gui, "Gui.ActiveDocument.resetEdit()");
514 
515  return false;
