166 void DrawView::onChanged(const App::Property* prop)
167 {
168 //Coding note: calling execute, recompute or recomputeFeature inside an onChanged
169 //method can create infinite loops if the called method changes a property. In general
170 //don't do this! There are situations where it is OK, but careful analysis is a must.
171 
172  if (prop == &Scale &&
173  Scale.getValue() < Precision::Confusion()) {
174  //this is not supposed to happen since Scale has constraints, but it may
175  //happen during changes made in PropertyEditor?
176  Scale.setValue(1.0);
177  return;
178  }
179 
180  if (isRestoring()) {
181  App::DocumentObject::onChanged(prop);
182  return;
183  }
184 
185  if (prop == &ScaleType) {
186  auto page = findParentPage();
187  if (!page) {
188  //we don't belong to a page yet, so don't bother
189  return;
190  }
191  if (ScaleType.isValue("Page")) {
192  Scale.setStatus(App::Property::ReadOnly, true);
193  if(std::abs(page->Scale.getValue() - getScale()) > FLT_EPSILON) {
194  Scale.setValue(page->Scale.getValue());
195  }
196  } else if ( ScaleType.isValue("Custom") ) {
197  //don't change Scale
198  Scale.setStatus(App::Property::ReadOnly, false);
199  } else if ( ScaleType.isValue("Automatic") ) {
200  Scale.setStatus(App::Property::ReadOnly, true);
201  if (!checkFit(page)) {
202  double newScale = autoScale(page->getPageWidth(), page->getPageHeight());
203  if(std::abs(newScale - getScale()) > FLT_EPSILON) {
204  Scale.setValue(newScale);
205  }
206  }
207  }
208  } else if (prop == &LockPosition) {
209  handleXYLock();
210  requestPaint(); //change lock icon
211  LockPosition.purgeTouched();
212  } else if ((prop == &Caption) ||
213  (prop == &Label)) {
214  requestPaint();
215  } else if ( prop == &X ||
216  prop == &Y ) {
217  //X,Y changes are only interesting to DPGI and Gui side
218  X.purgeTouched();
219  Y.purgeTouched();
220  }
221 
222  App::PropertyLink *ownerProp = getOwnerProperty();
223  if (ownerProp && prop == ownerProp) {
224  touchTreeOwner(ownerProp->getValue());
225  }
226 
227  App::DocumentObject::onChanged(prop);
228 }

256  }
257  if (Y.testStatus(App::Property::ReadOnly)) {
258  Y.setStatus(App::Property::ReadOnly, false);
259  Y.purgeTouched();

325 //note this won't find parent pages for DrawProjItem since their parent is DrawProjGroup!
326 int DrawView::countParentPages() const
327 {
328  int count = 0;
329  std::vector<App::DocumentObject*> parentRaw = getInList();
330  std::vector<App::DocumentObject*> parentAll;
331 
332  // Some parents are Links, we need the pages.
333  for (auto& parent : parentRaw) {
334  if (parent->isDerivedFrom<App::Link>()) {
335  for (auto& linkParent : parent->getInList()) {
336  if (linkParent->isDerivedFrom<DrawPage>()) {
337  parentAll.push_back(linkParent);
338  }
339  }
340  }
341  else {
342  parentAll.push_back(parent);
343  }
344  }
345 
346  //it can happen that a page is repeated in the InList, so we need to
347  //prune the duplicates
348  std::sort(parentAll.begin(), parentAll.end());
349  auto last = std::unique(parentAll.begin(), parentAll.end());
350  parentAll.erase(last, parentAll.end());
351 
352  for (auto& parent : parentAll) {
353  if (parent->isDerivedFrom<DrawPage>()) {
354  count++;
355  }
356  }
357  return count;
358 }

360 //finds the first DrawPage in this Document that claims to own this DrawView
361 //note that it is possible to manipulate the Views property of DrawPage so that
362 //more than 1 DrawPage claims a DrawView.
363 DrawPage* DrawView::findParentPage() const
364 {
365  // Get Feature Page
366  DrawPage *page = nullptr;
367  DrawViewCollection *collection = nullptr;
368  std::vector<App::DocumentObject*> parentsAll = getInList();
369  for (auto& parent : parentsAll) {
370  if (parent->isDerivedFrom<DrawPage>()) {
371  page = static_cast<TechDraw::DrawPage *>(parent);
372  } else if (parent->isDerivedFrom<DrawViewCollection>()) {
373  collection = static_cast<TechDraw::DrawViewCollection *>(parent);
374  page = collection->findParentPage();
375  }
376 
377  if(page)
378  break; // Found a page so leave
379  }
380 
381  return page;
382 }

385 std::vector<DrawPage*> DrawView::findAllParentPages() const
386 {
387  std::vector<DrawPage*> pages;
388 
389  for (auto parent : getInList()) {
390  if (parent->isDerivedFrom<App::Link>()) {
391  for (auto& linkParent : parent->getInList()) {
392  if (linkParent->isDerivedFrom<DrawPage>()
393  || linkParent->isDerivedFrom<DrawViewCollection>()) {
394  parent = linkParent;
395  break;
396  }
397  }
398  }
399 
400  if (parent->isDerivedFrom<DrawPage>()) {
401  pages.emplace_back(static_cast<TechDraw::DrawPage*>(parent));
402  }
403  else if (parent->isDerivedFrom<DrawViewCollection>()) {
404  auto* collection = static_cast<TechDraw::DrawViewCollection*>(parent);
405  pages.emplace_back(collection->findParentPage());
406  }
407  }
408 
409  //prune the duplicates
410  std::sort(pages.begin(), pages.end());
411  auto last = std::unique(pages.begin(), pages.end());
412  pages.erase(last, pages.end());
413 
414  return pages;
415 }

469 //compare 1:1 rect of view to pagesize(pw, h)
470 double DrawView::autoScale(double pw, double ph) const
471 {
472 // Base::Console().Message("DV::autoScale(Page: %.3f, %.3f) - %s\n", pw, ph, getNameInDocument());
473  QRectF viewBox = getRect(); //getRect is scaled (ie current actual size)
474  if (!viewBox.isValid()) {
475  return 1.0;
476  }
477  //have to unscale rect to determine new scale
478  double vbw = viewBox.width()/getScale();
479  double vbh = viewBox.height()/getScale();
480  double xScale = pw/vbw; // > 1 page bigger than figure
481  double yScale = ph/vbh; // < 1 page is smaller than figure
482  double newScale = std::min(xScale, yScale);
483  double sensibleScale = DrawUtil::sensibleScale(newScale);
484  return sensibleScale;
485 }

495 bool DrawView::checkFit(TechDraw::DrawPage* p) const
496 {
497 // Base::Console().Message("DV::checkFit(page) - %s\n", getNameInDocument());
498  bool result = true;
499 
500  double width = 0.0;
501  double height = 0.0;
502  QRectF viewBox = getRect(); //rect is scaled
503  if (!viewBox.isValid()) {
504  result = true;
505  } else {
506  width = viewBox.width(); //scaled rect w x h
507  height = viewBox.height();
508  if ( (width > p->getPageWidth()) ||
509  (height > p->getPageHeight()) ) {
510  result = false;
511  }
512  }
513  return result;
514 }

561 void DrawView::handleChangedPropertyType(Base::XMLReader &reader, const char * TypeName, App::Property * prop)
562 {
563  if (prop == &Scale) {
564  App::PropertyFloat tmp;
565  if (strcmp(tmp.getTypeId().getName(), TypeName)==0) { //property in file is Float
566  tmp.setContainer(this);
567  tmp.Restore(reader);
568  double tmpValue = tmp.getValue();
569  if (tmpValue > 0.0) {
570  Scale.setValue(tmpValue);
571  } else {
572  Scale.setValue(1.0);
573  }
574  }
575  }
576  else if (prop->isDerivedFrom(App::PropertyLinkList::getClassTypeId())
577  && strcmp(prop->getName(), "Source") == 0) {
578  App::PropertyLinkGlobal glink;
579  App::PropertyLink link;
580  if (strcmp(glink.getTypeId().getName(), TypeName) == 0) { //property in file is plg
581  glink.setContainer(this);
582  glink.Restore(reader);
583  if (glink.getValue()) {
584  static_cast<App::PropertyLinkList*>(prop)->setScope(App::LinkScope::Global);
585  static_cast<App::PropertyLinkList*>(prop)->setValue(glink.getValue());
586  }
587  }
588  else if (strcmp(link.getTypeId().getName(), TypeName) == 0) { //property in file is pl
589  link.setContainer(this);
590  link.Restore(reader);
591  if (link.getValue()) {
592  static_cast<App::PropertyLinkList*>(prop)->setScope(App::LinkScope::Global);
593  static_cast<App::PropertyLinkList*>(prop)->setValue(link.getValue());
594  }
595  }
596  }
597 
598  // property X had App::PropertyFloat and was changed to App::PropertyLength
599  // and later to PropertyDistance because some X, Y are relative to existing points on page
600  else if (prop == &X && strcmp(TypeName, "App::PropertyFloat") == 0) {
601  App::PropertyFloat XProperty;
602  XProperty.setContainer(this);
603  // restore the PropertyFloat to be able to set its value
604  XProperty.Restore(reader);
605  X.setValue(XProperty.getValue());
606  }
607  else if (prop == &X && strcmp(TypeName, "App::PropertyLength") == 0) {
608  App::PropertyLength X2Property;
609  X2Property.Restore(reader);
610  X.setValue(X2Property.getValue());
611  }
612  else if (prop == &Y && strcmp(TypeName, "App::PropertyFloat") == 0) {
613  App::PropertyFloat YProperty;
614  YProperty.setContainer(this);
615  YProperty.Restore(reader);
616  Y.setValue(YProperty.getValue());
617  }
618  else if (prop == &Y && strcmp(TypeName, "App::PropertyLength") == 0) {
619  App::PropertyLength Y2Property;
620  Y2Property.Restore(reader);
621  Y.setValue(Y2Property.getValue());
622  }
623 
624 // property Rotation had App::PropertyFloat and was changed to App::PropertyAngle
625  else if (prop == &Rotation && strcmp(TypeName, "App::PropertyFloat") == 0) {
626  App::PropertyFloat RotationProperty;
627  RotationProperty.setContainer(this);
628  RotationProperty.Restore(reader);
629  Rotation.setValue(RotationProperty.getValue());
630  }
631 }
