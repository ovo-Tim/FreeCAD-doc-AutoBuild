34 def local_cache_needs_update() -> bool:
35  """Determine whether we need to update the cache, based on user preference, and previous
36  cache update status. Returns either True or False."""
37 
38  if not _cache_exists():
39  return True
40 
41  if _last_update_was_interrupted(reset_status=True):
42  return True
43 
44  if _custom_repo_list_changed():
45  return True
46 
47  # Figure out our cache update frequency: there is a combo box in the preferences dialog
48  # with three options: never, daily, and weekly.
49  days_between_updates = _days_between_updates()
50  pref = fci.ParamGet("User parameter:BaseApp/Preferences/Addons")
51  last_cache_update_string = pref.GetString("LastCacheUpdate", "never")
52 
53  if last_cache_update_string == "never":
54  return True
55  elif days_between_updates > 0:
56  last_cache_update = date.fromisoformat(last_cache_update_string)
57  delta_update = timedelta(days=days_between_updates)
58  if date.today() >= last_cache_update + delta_update:
59  return True
60  elif days_between_updates == 0:
61  return True
62 
63  return False

66 def _days_between_updates() -> int:
67  pref = fci.ParamGet("User parameter:BaseApp/Preferences/Addons")
68  update_frequency = pref.GetInt("UpdateFrequencyComboEntry", 0)
69  if update_frequency == 0:
70  return -1
71  elif update_frequency == 1:
72  return 1
73  elif update_frequency == 2:
74  return 7
75  else:
76  return 0

86  flag_file = utils.get_cache_file_name("CACHE_UPDATE_INTERRUPTED")
87  if os.path.exists(flag_file):

89  os.remove(flag_file)

99 def _custom_repo_list_changed() -> bool:
100  pref = fci.ParamGet("User parameter:BaseApp/Preferences/Addons")
101  stored_hash = pref.GetString("CustomRepoHash", "")
102  custom_repos = pref.GetString("CustomRepositories", "")
103  if custom_repos:
104  hasher = hashlib.sha1()
105  hasher.update(custom_repos.encode("utf-8"))
106  new_hash = hasher.hexdigest()
107  else:
108  new_hash = ""
109  if new_hash != stored_hash:
110  pref.SetString("CustomRepoHash", new_hash)
111  fci.Console.PrintMessage(
112  translate(
113  "AddonsInstaller",
114  "Custom repo list changed, forcing recache...\n",
115  )
116  )
117  return True
118  return False
