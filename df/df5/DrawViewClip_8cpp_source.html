70 void DrawViewClip::addView(App::DocumentObject* docObj)
71 {
72  if (!docObj->isDerivedFrom<DrawView>() && !docObj->isDerivedFrom<App::Link>()) {
73  return;
74  }
75 
76  auto* view = dynamic_cast<DrawView*>(docObj);
77 
78  if (!view) {
79  auto* link = dynamic_cast<App::Link*>(docObj);
80  if (!link) {
81  return;
82  }
83 
84  if (link) {
85  view = dynamic_cast<DrawView*>(link->getLinkedObject());
86  if (!view) {
87  return;
88  }
89  }
90  }
91 
92  std::vector<App::DocumentObject*> newViews(Views.getValues());
93  newViews.push_back(docObj);
94  Views.setValues(newViews);
95  QRectF viewRect = view->getRectAligned();
96  QPointF clipPoint(X.getValue(), Y.getValue());
97  if (viewRect.contains(clipPoint)) {
98  //position so the part of view that is overlapped by clip frame
99  //stays in the clip frame
100  double deltaX = view->X.getValue() - X.getValue();
101  double deltaY = view->Y.getValue() - Y.getValue();
102  view->X.setValue(deltaX);
103  view->Y.setValue(deltaY);
104  } else {
105  //position in centre of clip group frame
106  view->X.setValue(0.0);
107  view->Y.setValue(0.0);
108  }
109 
110  //reparent view to clip in tree
111  auto page = findParentPage();
112  page->Views.touch();
113 }
