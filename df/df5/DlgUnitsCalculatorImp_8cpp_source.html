135 void DlgUnitsCalculator::valueChanged(const Base::Quantity& quant)
136 {
137  // first check the unit, if it is invalid, getTypeString() outputs an empty string
138  // explicitly check for "ee" like in "eeV" because this would trigger an exception in Base::Unit
139  // since it expects then a scientific notation number like "1e3"
140  if ((ui->UnitInput->text().mid(0, 2) == QString::fromLatin1("ee"))
141  || Base::Unit(ui->UnitInput->text().toStdString()).getTypeString().empty()) {
142  ui->ValueOutput->setText(
143  QString::fromLatin1("%1 %2").arg(tr("unknown unit:"), ui->UnitInput->text()));
144  ui->pushButton_Copy->setEnabled(false);
145  }
146  else { // the unit is valid
147  // we can only convert units of the same type, thus check
148  if (Base::Unit(ui->UnitInput->text().toStdString()).getTypeString()
149  != quant.getUnit().getTypeString()) {
150  ui->ValueOutput->setText(tr("unit mismatch"));
151  ui->pushButton_Copy->setEnabled(false);
152  }
153  else { // the unit is valid and has the same type
154  double convertValue =
155  Base::Quantity::parse("1" + ui->UnitInput->text().toStdString()).getValue();
156  // we got now e.g. for "1 in" the value '25.4' because 1 in = 25.4 mm
157  // the result is now just quant / convertValue because the input is always in a base
158  // unit (an input of "1 cm" will immediately be converted to "10 mm" by Gui::InputField
159  // of the dialog)
160  double value = quant.getValue() / convertValue;
161  // determine how many decimals we will need to avoid an output like "0.00"
162  // at first use scientific notation, if there is no "e", we can round it to the
163  // user-defined decimals, but the user-defined decimals might be too low for cases like
164  // "10 um" in "in", thus only if value > 0.005 because FC's default are 2 decimals
165  QString val = QLocale().toString(value, 'g');
166  if (!val.contains(QChar::fromLatin1('e')) && (value > 0.005)) {
167  val = QLocale().toString(value, 'f', Base::UnitsApi::getDecimals());
168  }
169  // create the output string
170  QString out = QString::fromLatin1("%1 %2").arg(val, ui->UnitInput->text());
171  ui->ValueOutput->setText(out);
172  ui->pushButton_Copy->setEnabled(true);
173  }
174  }
175  // store the input value
176  actValue = quant;
177 }

200 void DlgUnitsCalculator::onUnitsBoxActivated(int index)
201 {
202  // SI units use [m], not [mm] for lengths
203  //
204  Base::Quantity q = ui->quantitySpinBox->value();
205  int32_t old = q.getUnit().length();
206  double value = q.getValue();
207 
208  Base::Unit unit = units[index];
209  int32_t len = unit.length();
210  ui->quantitySpinBox->setValue(Base::Quantity(value * std::pow(10.0, 3 * (len - old)), unit));
211 }
