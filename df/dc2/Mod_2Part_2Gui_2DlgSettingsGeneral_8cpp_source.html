132 void DlgImportExportIges::loadSettings()
133 {
134  Part::IGES::ImportExportSettings settings;
135 
136  ui->comboBoxUnits->setCurrentIndex(static_cast<int>(settings.getUnit()));
137 
138  bool brep = settings.getBRepMode();
139  if (brep)
140  ui->radioButtonBRepOn->setChecked(true);
141  else
142  ui->radioButtonBRepOff->setChecked(true);
143 
144  // Import
145  ui->checkSkipBlank->setChecked(settings.getSkipBlankEntities());
146 
147  // header info
148  ui->lineEditCompany->setText(QString::fromStdString(settings.getCompany()));
149  ui->lineEditAuthor->setText(QString::fromStdString(settings.getAuthor()));
150  ui->lineEditProduct->setText(QString::fromStdString(settings.getProductName()));
