78 MDIView::~MDIView()
79 {
80  //This view might be the focus widget of the main window. In this case we must
81  //clear the focus and e.g. set the focus directly to the main window, otherwise
82  //the application crashes when accessing this deleted view.
83  //This effect only occurs if this widget is not in Child mode, because otherwise
84  //the focus stuff is done correctly.
85  if (getMainWindow()) {
86  QWidget* foc = getMainWindow()->focusWidget();
87  if (foc) {
88  QWidget* par = foc;
89  while (par) {
90  if (par == this) {
91  getMainWindow()->setFocus();
92  break;
93  }
94  par = par->parentWidget();
95  }
96  }
97  }
98  if (connectDelObject.connected())
99  connectDelObject.disconnect();
100 
101  if (pythonObject) {
102  Base::PyGILStateLocker lock;
103  Py_DECREF(pythonObject);
104  pythonObject = nullptr;
105  }
106 }

108 void MDIView::deleteSelf()
109 {
110  // When using QMdiArea make sure to remove the QMdiSubWindow

114  // Use deleteLater() instead of delete operator.
115  QWidget* parent = this->parentWidget();
116  if (qobject_cast<QMdiSubWindow*>(parent)) {
117  // https://forum.freecad.org/viewtopic.php?f=22&t=23070
118  parent->close();
119  }
120  else {
121  this->close();
122  }
123 
124  // detach from document
125  if (_pcDocument)
126  onClose();
127  _pcDocument = nullptr;

148 void MDIView::onRelabel(Gui::Document *pDoc)
149 {
150  if (!bIsPassive) {
151  // Try to separate document name and view number if there is one
152  QString cap = windowTitle();
153  // Either with dirty flag ...
154  QRegularExpression rx(QLatin1String(R"((\s\:\s\d+\[\*\])$)"));
155  QRegularExpressionMatch match;
156  //int pos =
157  boost::ignore_unused(cap.lastIndexOf(rx, -1, &match));
158  if (!match.hasMatch()) {
159  // ... or not
160  rx.setPattern(QLatin1String(R"((\s\:\s\d+)$)"));
161  //pos =
162  boost::ignore_unused(cap.lastIndexOf(rx, -1, &match));
163  }
164  if (match.hasMatch()) {
165  cap = QString::fromUtf8(pDoc->getDocument()->Label.getValue());
166  cap += match.captured();
167  setWindowTitle(cap);
168  }
169  else {
170  cap = QString::fromUtf8(pDoc->getDocument()->Label.getValue());
171  cap = QString::fromLatin1("%1[*]").arg(cap);
172  setWindowTitle(cap);
173  }
174  }
175 }

195 bool MDIView::canClose()
196 {
197  if (getAppDocument() && getAppDocument()->testStatus(App::Document::TempDoc))
198  return true;
199 
200  if (!bIsPassive && getGuiDocument() && getGuiDocument()->isLastView()) {
201  this->setFocus(); // raises the view to front
202  return (getGuiDocument()->canClose(true,true));
203  }
204 
205  return true;

208 void MDIView::closeEvent(QCloseEvent *e)
209 {
210  if (canClose()) {
211  e->accept();
212  Application::Instance->viewClosed(this);
213 
214  if (!bIsPassive) {
215  // must be detached so that the last view can get asked
216  Document* doc = this->getGuiDocument();
217  if (doc && !doc->isLastView())
218  doc->detachView(this);
219  }
220 
221  // Note: When using QMdiArea we must not use removeWindow()
222  // because otherwise the QMdiSubWindow will lose its parent
223  // and thus the notification in QMdiSubWindow::closeEvent of
224  // other mdi windows to get maximized if this window
225  // is maximized will fail.
226  // This odd behaviour is caused by the invocation of
227  // d->mdiArea->removeSubWindow(parent) which we must let there
228  // because otherwise other parts don't work as they should.
229  QMainWindow::closeEvent(e);
230  }
231  else {
232  e->ignore();
233  }
234 }

296 void MDIView::restorePrinterSettings(QPrinter* printer)
297 {
298  auto hGrp = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/Printer");
299  QString printerName = printer->printerName();
300  if (printerName.isEmpty()) {

305  hGrp = hGrp->GetGroup(printerName.toUtf8());
306 
307  QPrinterInfo info = QPrinterInfo::defaultPrinter();
308  int initialDefaultPageSize = info.isNull() ? QPageSize::A4 : info.defaultPageSize().id();
309  int defaultPageSize = hGrp->GetInt("DefaultPageSize", initialDefaultPageSize);
310  int defaultPageOrientation = hGrp->GetInt("DefaultPageOrientation", QPageLayout::Portrait);

314  printer->setPageOrientation(static_cast<QPageLayout::Orientation>(defaultPageOrientation));
315  printer->setColorMode(static_cast<QPrinter::ColorMode>(defaultColorMode));

380 void MDIView::setCurrentViewMode(ViewMode mode)
381 {
382  switch (mode) {
383  // go to normal mode
384  case Child:
385  {
386  if (this->currentMode == FullScreen) {
387  showNormal();
388  setWindowFlags(windowFlags() & ~Qt::Window);
389  }
390  else if (this->currentMode == TopLevel) {
391  this->wstate = windowState();
392  setWindowFlags( windowFlags() & ~Qt::Window );
393  }
394 
395  if (this->currentMode != Child) {
396  this->currentMode = Child;
397  getMainWindow()->addWindow(this);
398  getMainWindow()->activateWindow();
399  update();
400  }
401  } break;
402  // go to top-level mode
403  case TopLevel:
404  {
405  if (this->currentMode == Child) {
406  if (qobject_cast<QMdiSubWindow*>(this->parentWidget()))
407  getMainWindow()->removeWindow(this,false);
408  setWindowFlags(windowFlags() | Qt::Window);
409  setParent(nullptr, Qt::Window | Qt::WindowTitleHint | Qt::WindowSystemMenuHint |
410  Qt::WindowMinMaxButtonsHint);
411  if (this->wstate & Qt::WindowMaximized)
412  showMaximized();
413  else
414  showNormal();
415 
416 #if defined(Q_WS_X11)
417  //extern void qt_x11_wait_for_window_manager( QWidget* w ); // defined in qwidget_x11.cpp
418  qt_x11_wait_for_window_manager(this);
419 #endif
420  activateWindow();
421  }
422  else if (this->currentMode == FullScreen) {
423  if (this->wstate & Qt::WindowMaximized)
424  showMaximized();
425  else
426  showNormal();
427  }
428 
429  this->currentMode = TopLevel;
430  update();
431  } break;
432  // go to fullscreen mode
433  case FullScreen:
434  {
435  if (this->currentMode == Child) {
436  if (qobject_cast<QMdiSubWindow*>(this->parentWidget()))
437  getMainWindow()->removeWindow(this,false);
438  setWindowFlags(windowFlags() | Qt::Window);
439  setParent(nullptr, Qt::Window);
440  showFullScreen();
441  }
442  else if (this->currentMode == TopLevel) {
443  this->wstate = windowState();
444  showFullScreen();
445  }
446 
447  this->currentMode = FullScreen;
448  update();
449  } break;
450  }
451 }
