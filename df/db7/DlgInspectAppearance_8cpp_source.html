48 void ColorWidget::paintEvent(QPaintEvent* event)
49 {
50  QPainter painter(this);
51 
52  // Draw a white background
53  auto color = QColor(255, 255, 255);
54  auto left = event->rect().left();
55  auto width = event->rect().width();
56  painter.fillRect(left, event->rect().top(), width, event->rect().height(), QBrush(color));
57 
58  // Draw a black border
59  color = QColor(0, 0, 0);
60  left = event->rect().left() + 2;
61  width = event->rect().width() - 4;
62  if (event->rect().width() > 75) {
63  left += (event->rect().width() - 75) / 2;
64  width = 71;
65  }
66  painter.fillRect(left, event->rect().top() + 2, width, event->rect().height() - 4, QBrush(color));
67 
68  // Draw the color
69  left = event->rect().left() + 5;
70  width = event->rect().width() - 10;
71  if (event->rect().width() > 75) {
72  left += (event->rect().width() - 75) / 2;
73  width = 65;
74  }
75  painter.fillRect(left, event->rect().top() + 5, width, event->rect().height() - 10, QBrush(_color));
76 }

134 void DlgInspectAppearance::update(std::vector<Gui::ViewProvider*>& views)
135 {
136  App::Document* doc = App::GetApplication().getActiveDocument();
137  if (doc) {
138  ui->editDocument->setText(QString::fromUtf8(doc->Label.getValue()));
139 
140  if (views.size() == 1) {
141  auto view = dynamic_cast<Gui::ViewProviderDocumentObject*>(views[0]);
142  if (!view) {
143  return;
144  }
145  auto* obj = view->getObject();
146  if (!obj) {
147  return;
148  }
149  auto* labelProp = dynamic_cast<App::PropertyString*>(obj->getPropertyByName("Label"));
150  if (labelProp) {
151  ui->editObjectLabel->setText(QString::fromUtf8(labelProp->getValue()));
152  }
153  else {
154  ui->editObjectLabel->setText(QLatin1String(""));
155  }
156  ui->editObjectName->setText(QLatin1String(obj->getNameInDocument()));
157 
158  auto subElement = Gui::Selection().getSelectionEx();
159  if (subElement.size() > 0) {
160  auto& subObject = subElement[0];
161  if (subObject.getSubNames().size() > 0) {
162  ui->editSubShape->setText(QString::fromStdString(subObject.getSubNames()[0]));
163  }
164  else {
165  ui->editSubShape->setText(QLatin1String(""));
166  }
167  }
168  else {
169  ui->editSubShape->setText(QLatin1String(""));
170  }
171 
172  auto subShapeType = QString::fromUtf8(obj->getTypeId().getName());
173  subShapeType.remove(subShapeType.indexOf(QLatin1String("::")), subShapeType.size());
174  ui->editSubShapeType->setText(subShapeType);
175  ui->editShapeType->setText(QString::fromUtf8(obj->getTypeId().getName()));
176 
177  ui->tabAppearance->clear();
178  if (labelProp && QString::fromUtf8(labelProp->getValue()).size() > 0) {
179  auto* prop =
180  dynamic_cast<App::PropertyMaterialList*>(view->getPropertyByName("ShapeAppearance"));
181  if (prop) {
182  for (int index = 0; index < prop->getSize(); index++) {
183  auto& material = (prop->getValues())[index];
184  auto* tab = makeAppearanceTab(material);
185  ui->tabAppearance->addTab(tab, QString::number(index));
186  }
187  }
188  }
189  }
190  }
191 }
