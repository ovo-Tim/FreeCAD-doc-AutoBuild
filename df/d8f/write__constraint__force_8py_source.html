45 def write_meshdata_constraint(f, femobj, force_obj, ccxwriter):
46 
47  # floats read from ccx should use {:.13G}, see comment in writer module
48 
49  direction_vec = femobj["Object"].DirectionVector
50  dir_zero_tol = 1e-15 # TODO: should this be more generally for more values?
51  # be careful with raising the tolerance, a big load would have an impact
52  # but compared to the real direction the impact would be small again
53  for ref_shape in femobj["NodeLoadTable"]:
54  f.write(f"** {ref_shape[0]}\n")
55  for n in sorted(ref_shape[1]):
56  node_load = ref_shape[1][n]
57  # the loads in ref_shape[1][n] are without unit
58  if abs(direction_vec.x) > dir_zero_tol:
59  v1 = f"{(direction_vec.x * node_load).Value:.13G}"
60  f.write(f"{n},1,{v1}\n")
61  if abs(direction_vec.y) > dir_zero_tol:
62  v2 = f"{(direction_vec.y * node_load).Value:.13G}"
63  f.write(f"{n},2,{v2}\n")
64  if abs(direction_vec.z) > dir_zero_tol:
65  v3 = f"{(direction_vec.z * node_load).Value:.13G}"
66  f.write(f"{n},3,{v3}\n")
67  f.write("\n")
68  f.write("\n")
