233 QPixmap SystemPanelScheme::drawFoldIcon(const QPalette& p, bool fold) const
234 {
235  QImage img(17,17,QImage::Format_ARGB32_Premultiplied);
236  img.fill(0x00000000);
237  QPainter painter;
238  painter.begin(&img);
239  painter.setBrush(p.window());
240  painter.drawEllipse(2,2,13,13);
241  painter.setPen(p.color(QPalette::Base));
242  painter.drawEllipse(2,2,13,13);
243  painter.setPen(p.color(QPalette::Highlight));
244  painter.drawLine(QLine(5,7,8,4));
245  painter.drawLine(QLine(6,7,8,5));
246  painter.drawLine(QLine(8,4,11,7));
247  painter.drawLine(QLine(8,5,10,7));
248  painter.drawLine(QLine(5,11,8,8));
249  painter.drawLine(QLine(6,11,8,9));
250  painter.drawLine(QLine(8,8,11,11));
251  painter.drawLine(QLine(9,8,10,11));
252  painter.end();
253 
254  if (!fold) {
255  QTransform mat;
256  mat.rotate(180.0);
257  img = img.transformed(mat);
258  }
259  return QPixmap::fromImage(img);
260 }
