183 PyObject* UnitPy::richCompare(PyObject* v, PyObject* w, int op)
184 {
185  if (PyObject_TypeCheck(v, &(UnitPy::Type)) && PyObject_TypeCheck(w, &(UnitPy::Type))) {
186  const Unit* u1 = static_cast<UnitPy*>(v)->getUnitPtr();
187  const Unit* u2 = static_cast<UnitPy*>(w)->getUnitPtr();
188 
189  PyObject* res = nullptr;
190  if (op != Py_EQ && op != Py_NE) {
191  PyErr_SetString(PyExc_TypeError, "no ordering relation is defined for Units");
192  return nullptr;
193  }
194  if (op == Py_EQ) {
195  res = (*u1 == *u2) ? Py_True : Py_False; // NOLINT
196  Py_INCREF(res);
197  return res;
198  }
199  res = (*u1 != *u2) ? Py_True : Py_False; // NOLINT
200  Py_INCREF(res);
201  return res;
202  }
203  // This always returns False
204  Py_INCREF(Py_NotImplemented);
205  return Py_NotImplemented;
206 }
