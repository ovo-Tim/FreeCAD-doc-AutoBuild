507 inline MeshGeomFacet MeshKernel::GetFacet(FacetIndex ulIndex) const
508 {
509  assert(ulIndex < _aclFacetArray.size());
510 
511  const MeshFacet* pclF = &_aclFacetArray[ulIndex];
512  MeshGeomFacet clFacet;
513 
514  clFacet._aclPoints[0] = _aclPointArray[pclF->_aulPoints[0]];
515  clFacet._aclPoints[1] = _aclPointArray[pclF->_aulPoints[1]];
516  clFacet._aclPoints[2] = _aclPointArray[pclF->_aulPoints[2]];
517  clFacet._ulProp = pclF->_ulProp;
518  clFacet._ucFlag = pclF->_ucFlag;
519  clFacet.CalcNormal();
520  return clFacet;

566 inline void MeshKernel::AdjustNormal(MeshFacet& rclFacet, const Base::Vector3f& rclNormal)
567 {
568  Base::Vector3f clN =
569  (_aclPointArray[rclFacet._aulPoints[1]] - _aclPointArray[rclFacet._aulPoints[0]])
570  % (_aclPointArray[rclFacet._aulPoints[2]] - _aclPointArray[rclFacet._aulPoints[0]]);
571  if ((clN * rclNormal) < 0.0F) {
572  rclFacet.FlipNormal();
573  }
574 }

585 inline Base::Vector3f MeshKernel::GetGravityPoint(const MeshFacet& rclFacet) const
586 {
587  const Base::Vector3f& p0 = _aclPointArray[rclFacet._aulPoints[0]];
588  const Base::Vector3f& p1 = _aclPointArray[rclFacet._aulPoints[1]];
589  const Base::Vector3f& p2 = _aclPointArray[rclFacet._aulPoints[2]];
