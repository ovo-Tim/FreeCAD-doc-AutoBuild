53 template <class Real>
54 Real DistSegment3Triangle3<Real>::GetSquared ()
55 {
56  DistLine3Triangle3<Real> kLTDist(Line3<Real>(m_rkSegment.Origin,
57  m_rkSegment.Direction),m_rkTriangle);
58 
59  Real fSqrDist;
60 
61  m_fSegmentParameter = kLTDist.GetLineParameter();
62  if (m_fSegmentParameter >= -m_rkSegment.Extent)
63  {
64  if (m_fSegmentParameter <= m_rkSegment.Extent)
65  {
66  fSqrDist = kLTDist.GetSquared();
67  m_kClosestPoint0 = kLTDist.GetClosestPoint0();
68  m_kClosestPoint1 = kLTDist.GetClosestPoint1();
69  m_afTriangleBary[0] = kLTDist.GetTriangleBary(0);
70  m_afTriangleBary[1] = kLTDist.GetTriangleBary(1);
71  m_afTriangleBary[2] = kLTDist.GetTriangleBary(2);
72  }
73  else
74  {
75  m_kClosestPoint0 = m_rkSegment.GetPosEnd();
76  DistVector3Triangle3<Real> kVTDist(m_kClosestPoint0,m_rkTriangle);
77  fSqrDist = kVTDist.GetSquared();
78  m_kClosestPoint1 = kVTDist.GetClosestPoint1();
79  m_fSegmentParameter = m_rkSegment.Extent;
80  m_afTriangleBary[0] = kVTDist.GetTriangleBary(0);
81  m_afTriangleBary[1] = kVTDist.GetTriangleBary(1);
82  m_afTriangleBary[2] = kVTDist.GetTriangleBary(2);
83  }
84  }
85  else
86  {
87  m_kClosestPoint0 = m_rkSegment.GetNegEnd();
88  DistVector3Triangle3<Real> kVTDist(m_kClosestPoint0,m_rkTriangle);
89  fSqrDist = kVTDist.GetSquared();
90  m_kClosestPoint1 = kVTDist.GetClosestPoint1();
91  m_fSegmentParameter = -m_rkSegment.Extent;
92  m_afTriangleBary[0] = kVTDist.GetTriangleBary(0);
93  m_afTriangleBary[1] = kVTDist.GetTriangleBary(1);
94  m_afTriangleBary[2] = kVTDist.GetTriangleBary(2);
95  }
96 
97  return fSqrDist;
98 }
