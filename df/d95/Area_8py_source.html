317  def _buildProfileOpenEdges(self, obj, edgeList, isHole, start, getsim):
318  """_buildPathArea(obj, edgeList, isHole, start, getsim) ... internal function."""
319  Path.Log.track()
320 
321  paths = []
322  heights = [i for i in self.depthparams]
323  Path.Log.debug("depths: {}".format(heights))
324  for i in range(0, len(heights)):
325  for baseShape in edgeList:
326  hWire = Part.Wire(Part.__sortEdges__(baseShape.Edges))
327  hWire.translate(FreeCAD.Vector(0, 0, heights[i] - hWire.BoundBox.ZMin))
328 
329  pathParams = {}
330  pathParams["shapes"] = [hWire]
331  pathParams["feedrate"] = self.horizFeed
332  pathParams["feedrate_v"] = self.vertFeed
333  pathParams["verbose"] = True
334  pathParams["resume_height"] = obj.SafeHeight.Value
335  pathParams["retraction"] = obj.ClearanceHeight.Value
336  pathParams["return_end"] = True
337  # Note that emitting preambles between moves breaks some dressups and prevents path optimization on some controllers
338  pathParams["preamble"] = False
339 
340  # Always manually setting pathParams["start"] to the first or
341  # last vertex of the wire (depending on obj.Direction) ensures
342  # the edge is always milled in the correct direction. Using
343  # self.endVector would allow Path.fromShapes to reverse the
344  # direction if that would shorten the travel move and thus cause
345  # the edges being milled in seemingly random directions.
346 
347  verts = hWire.Wires[0].Vertexes
348  idx = 0
349  if obj.Direction == "CCW":
350  idx = len(verts) - 1
351  x = verts[idx].X
352  y = verts[idx].Y
353  # Zero start value adjustments for Path.fromShapes() bug
354  if Path.Geom.isRoughly(x, 0.0):
355  x = 0.00001
356  if Path.Geom.isRoughly(y, 0.0):
357  y = 0.00001
358  pathParams["start"] = FreeCAD.Vector(x, y, verts[0].Z)
359 
360  obj.PathParams = str(
361  {key: value for key, value in pathParams.items() if key != "shapes"}
362  )
363  Path.Log.debug("Path with params: {}".format(obj.PathParams))
364 
365  (pp, end_vector) = Path.fromShapes(**pathParams)
366  paths.extend(pp.Commands)
367  Path.Log.debug("pp: {}, end vector: {}".format(pp, end_vector))
368 
369  self.endVector = end_vector
370  simobj = None
371 
372  return paths, simobj
