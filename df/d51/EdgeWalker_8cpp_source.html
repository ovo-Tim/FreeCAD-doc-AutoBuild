146 bool EdgeWalker::prepare()
147 {
148  //Base::Console().Message("TRACE - EW::prepare()\n");
149  // Initialize the interior edge index
150  property_map<TechDraw::graph, edge_index_t>::type e_index = get(edge_index, m_g);
151  graph_traits<TechDraw::graph>::edges_size_type edge_count = 0;
152  graph_traits<TechDraw::graph>::edge_iterator ei, ei_end;
153  for(boost::tie(ei, ei_end) = edges(m_g); ei != ei_end; ++ei)
154  put(e_index, *ei, edge_count++);
155 
156  //make planar embedding
157  planar_embedding_storage_t planar_embedding_storage(num_vertices(m_g));
158  planar_embedding_t planar_embedding(planar_embedding_storage.begin(),
159  get(vertex_index, m_g) );
160  for (auto& e : m_embedding) {
161  for (auto& i: e.incidenceList) {
162  planar_embedding[e.iVertex].push_back(i.eDesc);
163  }
164  }
165 
166  m_eV.setGraph(m_g);
167  planar_face_traversal(m_g, &planar_embedding[0], m_eV);
168 
169  return true;
170 }

269 std::vector<TopoDS_Vertex> EdgeWalker:: makeUniqueVList(std::vector<TopoDS_Edge> edges)
270 {
271 // Base::Console().Message("TRACE - EW::makeUniqueVList() - edgesIn: %d\n", edges.size());
272  std::vector<TopoDS_Vertex> uniqueVert;
273  for(auto& e:edges) {
274  Base::Vector3d v1 = DrawUtil::vertex2Vector(TopExp::FirstVertex(e));
275  Base::Vector3d v2 = DrawUtil::vertex2Vector(TopExp::LastVertex(e));
276  bool addv1 = true;
277  bool addv2 = true;
278  //check if we've already added this vertex
279  for (const auto& v: uniqueVert) {
280  Base::Vector3d v3d = DrawUtil::vertex2Vector(v);
281  if (v3d.IsEqual(v1, EWTOLERANCE)) {
282  addv1 = false;
283  }
284  if (v3d.IsEqual(v2, EWTOLERANCE)) {
285  addv2 = false;
286  }
287  }
288  if (addv1) {
289  uniqueVert.push_back(TopExp::FirstVertex(e));
290  }
291  if (addv2) {
292  uniqueVert.push_back(TopExp::LastVertex(e));
293  }
294  }
295 // Base::Console().Message("EW::makeUniqueVList - verts out: %d\n", uniqueVert.size());
296  return uniqueVert;
297 }

300 std::vector<WalkerEdge> EdgeWalker::makeWalkerEdges(std::vector<TopoDS_Edge> edges,
301  std::vector<TopoDS_Vertex> verts)
302 {
303 // Base::Console().Message("TRACE - EW::makeWalkerEdges() - edges: %d verts: %d\n", edges.size(), verts.size());
304  m_saveInEdges = edges;
305  std::vector<WalkerEdge> walkerEdges;
306  for (const auto& e:edges) {
307  TopoDS_Vertex edgeVertex1 = TopExp::FirstVertex(e);
308  TopoDS_Vertex edgeVertex2 = TopExp::LastVertex(e);
309  std::size_t vertex1Index = findUniqueVert(edgeVertex1, verts);
310  if (vertex1Index == SIZE_MAX) {
311  continue;
312  }
313  std::size_t vertex2Index = findUniqueVert(edgeVertex2, verts);
314  if (vertex2Index == SIZE_MAX) {
315  continue;
316  }
317 
318  WalkerEdge we;
319  we.v1 = vertex1Index;
320  we.v2 = vertex2Index;
321  we.idx = 0;
322  walkerEdges.push_back(we);
323  }
324 
325  //Base::Console().Message("TRACE - EW::makeWalkerEdges - returns we: %d\n", walkerEdges.size());
326  return walkerEdges;

385 std::vector<embedItem> EdgeWalker::makeEmbedding(const std::vector<TopoDS_Edge> edges,
386  const std::vector<TopoDS_Vertex> uniqueVList)
387 {
388 // Base::Console().Message("TRACE - EW::makeEmbedding(edges: %d, verts: %d)\n",
389 // edges.size(), uniqueVList.size());
390  std::vector<embedItem> result;
391 
392  std::size_t iVert = 0;
393  //make an embedItem for each vertex in uniqueVList
394  //for each vertex v
395  // find all the edges that have v as first or last vertex
396  for (auto& v: uniqueVList) {
397  TopoDS_Vertex cv = v; //v is const but we need non-const for vertexEqual
398  std::size_t iEdge = 0;
399  std::vector<incidenceItem> iiList;
400  for (auto& e: edges) {
401  double angle = 0;
402  TopoDS_Vertex edgeVertex1 = TopExp::FirstVertex(e);
403  TopoDS_Vertex edgeVertex2 = TopExp::LastVertex(e);
404  if (DrawUtil::vertexEqual(cv, edgeVertex1)) {
405  angle = DrawUtil::incidenceAngleAtVertex(e,v,EWTOLERANCE);
406  incidenceItem ii(iEdge, angle, m_saveWalkerEdges[iEdge].ed);
407  iiList.push_back(ii);
408  } else if (DrawUtil::vertexEqual(cv, edgeVertex2)) {
409  angle = DrawUtil::incidenceAngleAtVertex(e,v,EWTOLERANCE);
410  incidenceItem ii(iEdge, angle, m_saveWalkerEdges[iEdge].ed);
411  iiList.push_back(ii);
412  }
413  iEdge++;
414  }
415  //sort incidenceList by angle
416  iiList = embedItem::sortIncidenceList(iiList, false);
417  embedItem embed(iVert, iiList);
418  result.push_back(embed);
419  iVert++;
420  }
421  return result;
422 }
