38 class FormatString
39 {
40 public:
41  static std::string str(const std::string& s) {
42  return s;
43  }
44  static std::string str(const char* s) {
45  return s;
46  }
47  static std::string str(const QString& s) {
48  return s.toStdString();
49  }
50  static std::string str(const std::stringstream& s) {
51  return s.str();
52  }
53  static std::string str(const std::ostringstream& s) {
54  return s.str();
55  }
56  static std::string str(const std::ostream& s) {
57  if (typeid(s) == typeid(std::ostringstream))
58  return dynamic_cast<const std::ostringstream&>(s).str();
59  else if (typeid(s) == typeid(std::stringstream))
60  return dynamic_cast<const std::stringstream&>(s).str();
61  throw Base::TypeError("Not a std::stringstream or std::ostringstream");
62  }
63  static std::string toStr(boost::format& f) {
64  return f.str();
65  }
66 
67  template<typename T, typename... Args>
68  static std::string toStr(boost::format& f, T&& t, Args&&... args) {
69  return toStr(f % std::forward<T>(t), std::forward<Args>(args)...);
70  }
