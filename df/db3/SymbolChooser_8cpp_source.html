127 void SymbolChooser::loadSymbolNames(QString pathToSymbols)
128 {
129  // Fill selection list with names and icons
130  QDir symbolDir(pathToSymbols);
131  symbolDir.setFilter(QDir::Files);
132  QStringList fileNames = symbolDir.entryList();
133 
134  for (auto& fn: fileNames) {
135  QString text = (new QFileInfo(fn))->baseName();
136  QIcon icon(pathToSymbols + fn);
137 
138  // Create a symbol in the QListWidget lwSymbols
139  new QListWidgetItem(icon, text, ui->lwSymbols);
140  }
