61 def setup(doc=None, solvertype="ccxtools"):
62 
63  if solvertype == "z88":
64  # constraint displacement is not supported for Z88
65  # pass a not valid solver name for z88, thus no solver is created
66  solvertype = "z88_not_valid"
67 
68  # init FreeCAD document
69  if doc is None:
70  doc = init_doc()
71 
72  # explanation object
73  # just keep the following line and change text string in get_explanation method
74  manager.add_explanation_obj(doc, get_explanation(manager.get_header(get_information())))
75 
76  # setup CalculiX cantilever
77  # apply a prescribed displacement of 250 mm in -z on the front end face
78  doc = setup_cantilever_base_solid(doc, solvertype)
79  analysis = doc.Analysis
80  geom_obj = doc.Box
81 
82  # constraint displacement
83  con_disp = ObjectsFem.makeConstraintDisplacement(doc, name="ConstraintDisplacmentPrescribed")
84  con_disp.References = [(geom_obj, "Face2")]
85  con_disp.zFree = False
86  con_disp.zDisplacement = -250.0
87  analysis.addObject(con_disp)
88 
89  doc.recompute()
90  return doc
