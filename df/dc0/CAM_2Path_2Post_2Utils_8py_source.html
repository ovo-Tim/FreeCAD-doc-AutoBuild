153  def generate_filenames(self):
154  """Yield filenames indefinitely with proper substitutions."""
155  while True:
156  temp_filename = self.qualified_filename
157  Path.Log.debug(f"temp_filename: {temp_filename}")
158  explicit_sequence = False
159  matches = re.findall(r"%S", temp_filename)
160  if matches:
161  Path.Log.debug(f"matches: {matches}")
162  temp_filename = re.sub(r"%S", str(self.sequencenumber), temp_filename)
163  explicit_sequence = True
164 
165  subpart = f"-{self.subpartname}" if self.subpartname else ""
166  sequence = (
167  f"-{self.sequencenumber}" if not explicit_sequence and self.sequencenumber else ""
168  )
169  filename = f"{temp_filename}{subpart}{sequence}{self.extension}"
170  full_path = os.path.join(self.qualified_path, filename)
171 
172  self.sequencenumber += 1
173  Path.Log.debug(f"yielding filename: {full_path}")
174  yield os.path.normpath(full_path)

195  def highlightBlock(self, text):
196  for pattern, hlFormat in self.highlightingRules:
197  expression = QtCore.QRegularExpression(pattern)
198  index = expression.match(text)

200  length = index.capturedLength()
201  self.setFormat(index.capturedStart(), length, hlFormat)
202  index = expression.match(text, index.capturedStart() + length)

333 def splitArcs(path):
334  """Filter a path object and replace all G2/G3 moves with discrete G1 moves.
335 
336  Returns a Path object.
337  """
338  prefGrp = FreeCAD.ParamGet("User parameter:BaseApp/Preferences/Mod/CAM")
339  deflection = prefGrp.GetFloat("LibAreaCurveAccuarcy", 0.01)
340 
341  results = []
342  if not isinstance(path, Path.Path):
343  raise TypeError("path must be a Path object")
344 
345  machine = MachineState()
346  for command in path.Commands:
347 
348  if command.Name not in Path.Geom.CmdMoveArc:
349  machine.addCommand(command)
350  results.append(command)
351  continue
352 
353  edge = Path.Geom.edgeForCmd(command, machine.getPosition())
354  pts = edge.discretize(Deflection=deflection)
355  edges = [Part.makeLine(v1, v2) for v1, v2 in zip(pts, pts[1:])]
356  for edge in edges:
357  results.extend(Path.Geom.cmdsForEdge(edge))
358 
359  machine.addCommand(command)
360 
361  return Path.Path(results)
