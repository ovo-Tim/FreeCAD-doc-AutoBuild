119 #define FC_PY_ELEMENT_DEFINE(_name) Py::Object py_##_name;
120 
121 #define FC_PY_ELEMENT_INIT(_name) \
122  FC_PY_GetCallable(pyobj, #_name, py_##_name); \
123  if (!py_##_name.isNone()) { \
124  PyObject* pyRecursive = PyObject_GetAttrString(pyobj, "__allow_recursive_" #_name); \
125  if (!pyRecursive) { \
126  PyErr_Clear(); \
127  _Flags.set(FlagAllowRecursive_##_name, false); \
128  } \
129  else { \
130  _Flags.set(FlagAllowRecursive_##_name, PyObject_IsTrue(pyRecursive)); \
131  Py_DECREF(pyRecursive); \
132  } \
133  }
134 
135 #define FC_PY_ELEMENT_FLAG(_name) FlagCalling_##_name, FlagAllowRecursive_##_name,
136 
137 #define _FC_PY_CALL_CHECK(_name, _ret) \
138  if ((!_Flags.test(FlagAllowRecursive_##_name) && _Flags.test(FlagCalling_##_name)) \
139  || py_##_name.isNone()) { \
140  _ret; \
141  } \
142  Base::BitsetLocker<Flags> guard(_Flags, FlagCalling_##_name);
143 
144 #undef FC_PY_ELEMENT
