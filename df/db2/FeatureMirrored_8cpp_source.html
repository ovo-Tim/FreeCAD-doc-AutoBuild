61 const std::list<gp_Trsf> Mirrored::getTransformations(const std::vector<App::DocumentObject*>)
62 {
63  using getMirrorAxis = std::function<bool(gp_Pnt&, gp_Dir&)>;
64 
65  // 2D part object
66  getMirrorAxis axisOfSketch = [this](gp_Pnt& axbase, gp_Dir& axdir) {
67  App::DocumentObject* refObject = MirrorPlane.getValue();
68  std::vector<std::string> subStrings = MirrorPlane.getSubValues();
69 
70  if (auto refSketch = dynamic_cast<Part::Part2DObject*>(refObject)) {
71  Base::Axis axis;
72  if (subStrings.empty() || subStrings[0].empty()) {
73  axis = refSketch->getAxis(Part::Part2DObject::N_Axis);
74  }
75  else if (subStrings[0] == "H_Axis") {
76  axis = refSketch->getAxis(Part::Part2DObject::V_Axis);
77  }
78  else if (subStrings[0] == "V_Axis") {
79  axis = refSketch->getAxis(Part::Part2DObject::H_Axis);
80  }
81  else if (subStrings[0].compare(0, 4, "Axis") == 0) {
82  int AxId = std::atoi(subStrings[0].substr(4,4000).c_str());
83  if (AxId >= 0 && AxId < refSketch->getAxisCount()) {
84  axis = refSketch->getAxis(AxId);
85  axis.setBase(axis.getBase() + 0.5 * axis.getDirection());
86  axis.setDirection(Base::Vector3d(-axis.getDirection().y, axis.getDirection().x, axis.getDirection().z));
87  }
88  else {
89  throw Base::ValueError("No valid axis specified");
90  }
91  }
92  axis *= refSketch->Placement.getValue();
93  axbase = gp_Pnt(axis.getBase().x, axis.getBase().y, axis.getBase().z);
94  axdir = gp_Dir(axis.getDirection().x, axis.getDirection().y, axis.getDirection().z);
95  return true;
96  }
97 
98  return false;
99  };
100 
101  // Datum plane
102  getMirrorAxis axisOfDatumPlane = [this](gp_Pnt& axbase, gp_Dir& axdir) {
103  App::DocumentObject* refObject = MirrorPlane.getValue();
104  if (auto plane = dynamic_cast<PartDesign::Plane*>(refObject)) {
105  Base::Vector3d base = plane->getBasePoint();
106  axbase = gp_Pnt(base.x, base.y, base.z);
107  Base::Vector3d dir = plane->getNormal();
108  axdir = gp_Dir(dir.x, dir.y, dir.z);
109  return true;
110  }
111 
112  return false;
113  };
114 
115  // Normal plane
116  getMirrorAxis axisOfPlane = [this](gp_Pnt& axbase, gp_Dir& axdir) {
117  App::DocumentObject* refObject = MirrorPlane.getValue();
118  if (auto plane = dynamic_cast<App::Plane*>(refObject)) {
119  Base::Vector3d base = plane->getBasePoint();
120  axbase = gp_Pnt(base.x, base.y, base.z);
121  Base::Vector3d dir = plane->getDirection();
122  axdir = gp_Dir(dir.x, dir.y, dir.z);
123  return true;
124  }
125 
126  return false;
127  };
128 
129  // Planar shape
130  getMirrorAxis axisOfPlanarShape = [this](gp_Pnt& axbase, gp_Dir& axdir) {
131  App::DocumentObject* refObject = MirrorPlane.getValue();
132  std::vector<std::string> subStrings = MirrorPlane.getSubValues();
133 
134  if (auto feature = dynamic_cast<Part::Feature*>(refObject)) {
135  if (subStrings.empty()) {
136  throw Base::ValueError("No mirror plane reference specified");
137  }
138  if (subStrings[0].empty()) {
139  throw Base::ValueError("No direction reference specified");
140  }
141 
142  Part::TopoShape baseShape = feature->Shape.getShape();
143  // TODO: Check for multiple mirror planes?
144  TopoDS_Shape shape = baseShape.getSubShape(subStrings[0].c_str());
145  TopoDS_Face face = TopoDS::Face(shape);
146  if (face.IsNull()) {
147  throw Base::ValueError("Failed to extract mirror plane");
148  }
149 
150  BRepAdaptor_Surface adapt(face);
151  if (adapt.GetType() != GeomAbs_Plane) {
152  throw Base::TypeError("Mirror face must be planar");
153  }
154 
155  axbase = getPointFromFace(face);
156  axdir = adapt.Plane().Axis().Direction();
157  return true;
158  }
159 
160  return false;
161  };
162 
163  // ------------------------------------------------------------------------
164 
165  App::DocumentObject* refObject = MirrorPlane.getValue();
166  if (!refObject) {
167  throw Base::ValueError("No mirror plane reference specified");
168  }
169 
170  gp_Pnt axbase;
171  gp_Dir axdir;
172  std::vector<getMirrorAxis> axisCheckers;
173  axisCheckers.push_back(axisOfSketch);
174  axisCheckers.push_back(axisOfDatumPlane);
175  axisCheckers.push_back(axisOfPlane);
176  axisCheckers.push_back(axisOfPlanarShape);
177 
178  for (const auto& getAxis : axisCheckers) {
179  if (getAxis(axbase, axdir)) {
180  return createTransformations(axbase, axdir);
181  }
182  }
183 
184  throw Base::ValueError("Mirror plane reference must be a sketch axis, a face of a feature or a datum plane");
185 }

187 std::list<gp_Trsf> Mirrored::createTransformations(gp_Pnt& axbase, gp_Dir& axdir) const
188 {
189  TopLoc_Location invObjLoc = this->getLocation().Inverted();
190  axbase.Transform(invObjLoc.Transformation());
191  axdir.Transform(invObjLoc.Transformation());
192 
193  gp_Ax2 mirrorAxis(axbase, axdir);
194 
195  std::list<gp_Trsf> transformations;
196  gp_Trsf trans;
197  transformations.push_back(trans); // identity transformation
198  trans.SetMirror(mirrorAxis);
199  transformations.push_back(trans); // mirrored transformation
200  return transformations;
201 }
