50 SoGroup *AxisOrigin::getNode() {

54  node.reset(new SoGroup);

59  SbVec3f(0,0,0), SbVec3f(size,0,0),
60  SbVec3f(0,size,0), SbVec3f(0,0,size),

63  SbVec3f(0,dist,dist), SbVec3f(0,pSize,dist), SbVec3f(0,dist,pSize) // XY Plane
64  };
65 
66  // indexes used to create the edges
67  const int32_t lines[21] =
68  {
69  0,1,-1,
70  0,2,-1,
71  0,3,-1,
72  5,4,6,-1,
73  8,7,9,-1,
74  11,10,12,-1
75  };
76 
77  pMat->diffuseColor.setNum(3);
78  pMat->diffuseColor.set1Value(0, SbColor(1.0f, 0.2f, 0.2f));
79  pMat->diffuseColor.set1Value(1, SbColor(0.2f, 0.6f, 0.2f));
80  pMat->diffuseColor.set1Value(2, SbColor(0.2f, 0.2f, 1.0f));
81  pMat->diffuseColor.set1Value(4, SbColor(0.8f, 0.8f, 0.8f));

103 #define CREATE_AXIS(_type,_key,_count,_offset,_mat) do{\
104  const char *label=_key;\
105  if(labels.size()){\
106  auto iter = labels.find(_key);\
107  if(iter == labels.end())\
108  break;\
109  else if(iter->second.size())\
110  label = iter->second.c_str();\
111  }\
112  auto pAxis = new SoFCSelection;\
113  pAxis->applySettings();\
114  pAxis->style = SoFCSelection::EMISSIVE_DIFFUSE;\
115  pAxis->subElementName = label;\
116  nodeMap[label].reset(pAxis);\
117  node->addChild(pAxis);\
118  auto _type = new SoIndexed##_type##Set;\
119  pAxis->addChild(_type);\
120  _type->coordIndex.setNum(_count);\
121  _type->coordIndex.setValues(0,_count,lines+_offset);\
122  _type->materialIndex.setValue(_mat);\
123  }while(0)
124 
125  CREATE_AXIS(Point,"O",1,0,4);
126  CREATE_AXIS(Line,"X",3,0,0);
127  CREATE_AXIS(Line,"Y",3,3,1);
128  CREATE_AXIS(Line,"Z",3,6,2);
129  CREATE_AXIS(Line,"XY",4,9,2);
130  CREATE_AXIS(Line,"XZ",4,13,1);
131  CREATE_AXIS(Line,"YZ",4,17,0);
132  return node;
133 }

135 bool AxisOrigin::getElementPicked(const SoPickedPoint *pp, std::string &subname) const {
136  SoPath *path = pp->getPath();
137  int length = path->getLength();
138  for(int i=0;i<length;++i) {
139  auto node = path->getNodeFromTail(i);
140  if(node->isOfType(SoFCSelection::getClassTypeId())) {
141  subname = static_cast<SoFCSelection*>(node)->subElementName.getValue().getString();
142  return true;
143  } else if(node->isOfType(SoFCSelectionRoot::getClassTypeId()))
144  break;
145  }
146  return false;
147 }
