118 PartExport void getPyShapes(PyObject* obj, std::vector<TopoShape>& shapes)
119 {
120  if (!obj) {
121  return;
122  }
123  if (PyObject_TypeCheck(obj, &Part::TopoShapePy::Type)) {
124  shapes.push_back(*static_cast<TopoShapePy*>(obj)->getTopoShapePtr());
125  }
126  else if (PyObject_TypeCheck(obj, &GeometryPy::Type)) {
127  shapes.emplace_back(static_cast<GeometryPy*>(obj)->getGeometryPtr()->toShape());
128  }
129  else if (PySequence_Check(obj)) {
130  Py::Sequence list(obj);
131  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
132  if (PyObject_TypeCheck((*it).ptr(), &(Part::TopoShapePy::Type))) {
133  shapes.push_back(*static_cast<TopoShapePy*>((*it).ptr())->getTopoShapePtr());
134  }
135  else if (PyObject_TypeCheck((*it).ptr(), &GeometryPy::Type)) {
136  shapes.emplace_back(
137  static_cast<GeometryPy*>((*it).ptr())->getGeometryPtr()->toShape());
138  }
139  else {
140  throw Py::TypeError("expect shape in sequence");
141  }
142  }
143  }
144  else {
145  throw Py::TypeError("expect shape or sequence of shapes");
146  }
147 }

166 PartExport std::list<TopoDS_Edge> sort_Edges(double tol3d, std::list<TopoDS_Edge>& edges)
167 {
168  tol3d = tol3d * tol3d;
169  std::list<EdgePoints> edge_points;
170  TopExp_Explorer xp;
171  for (std::list<TopoDS_Edge>::iterator it = edges.begin(); it != edges.end(); ++it) {
172  EdgePoints ep;
173  xp.Init(*it,TopAbs_VERTEX);
174  ep.v1 = BRep_Tool::Pnt(TopoDS::Vertex(xp.Current()));
175  xp.Next();
176  ep.v2 = BRep_Tool::Pnt(TopoDS::Vertex(xp.Current()));
177  ep.it = it;
178  ep.edge = *it;
179  edge_points.push_back(ep);
180  }
181 
182  if (edge_points.empty())
183  return {};
184 
185  std::list<TopoDS_Edge> sorted;
186  gp_Pnt first, last;
187  first = edge_points.front().v1;
188  last = edge_points.front().v2;
189 
190  sorted.push_back(edge_points.front().edge);
191  edges.erase(edge_points.front().it);
192  edge_points.erase(edge_points.begin());
193 
194  while (!edge_points.empty()) {
195  // search for adjacent edge
196  std::list<EdgePoints>::iterator pEI;
197  for (pEI = edge_points.begin(); pEI != edge_points.end(); ++pEI) {
198  if (pEI->v1.SquareDistance(last) <= tol3d) {
199  last = pEI->v2;
200  sorted.push_back(pEI->edge);
201  edges.erase(pEI->it);
202  edge_points.erase(pEI);
203  pEI = edge_points.begin();
204  break;
205  }
206  else if (pEI->v2.SquareDistance(first) <= tol3d) {
207  first = pEI->v1;
208  sorted.push_front(pEI->edge);
209  edges.erase(pEI->it);
210  edge_points.erase(pEI);
211  pEI = edge_points.begin();
212  break;
213  }
214  else if (pEI->v2.SquareDistance(last) <= tol3d) {
215  last = pEI->v1;
216  Standard_Real first, last;
217  const Handle(Geom_Curve) & curve = BRep_Tool::Curve(pEI->edge, first, last);
218  first = curve->ReversedParameter(first);
219  last = curve->ReversedParameter(last);
220  TopoDS_Edge edgeReversed = BRepBuilderAPI_MakeEdge(curve->Reversed(), last, first);
221  sorted.push_back(edgeReversed);
222  edges.erase(pEI->it);
223  edge_points.erase(pEI);
224  pEI = edge_points.begin();
225  break;
226  }
227  else if (pEI->v1.SquareDistance(first) <= tol3d) {
228  first = pEI->v2;
229  Standard_Real first, last;
230  const Handle(Geom_Curve) & curve = BRep_Tool::Curve(pEI->edge, first, last);
231  first = curve->ReversedParameter(first);
232  last = curve->ReversedParameter(last);
233  TopoDS_Edge edgeReversed = BRepBuilderAPI_MakeEdge(curve->Reversed(), last, first);
234  sorted.push_front(edgeReversed);
235  edges.erase(pEI->it);
236  edge_points.erase(pEI);
237  pEI = edge_points.begin();
238  break;
239  }
240  }
241 
242  if ((pEI == edge_points.end()) || (last.SquareDistance(first) <= tol3d)) {
243  // no adjacent edge found or polyline is closed
244  return sorted;
245  }
246  }
247 
248  return sorted;
249 }
