26 // Quadratic fit is
27 //
28 // 0 = C[0] + C[1]*X + C[2]*Y + C[3]*Z + C[4]*X^2 + C[5]*Y^2
29 // + C[6]*Z^2 + C[7]*X*Y + C[8]*X*Z + C[9]*Y*Z
30 //
31 // subject to Length(C) = 1. Minimize E(C) = C^t M C with Length(C) = 1
32 // and M = (sum_i V_i)(sum_i V_i)^t where
33 //
34 // V = (1, X, Y, Z, X^2, Y^2, Z^2, X*Y, X*Z, Y*Z)
35 //
36 // Minimum value is the smallest eigenvalue of M and C is a corresponding
37 // unit length eigenvector.
38 //
39 // Input:
40 // n = number of points to fit
41 // p[0..n-1] = array of points to fit
42 //
43 // Output:
44 // c[0..9] = coefficients of quadratic fit (the eigenvector)
45 // return value of function is nonnegative and a measure of the fit
46 // (the minimum eigenvalue; 0 = exact fit, positive otherwise)
47 
48 // Canonical forms. The quadratic equation can be factored into
49 // P^T A P + B^T P + K = 0 where P = (X,Y,Z), K = C[0], B = (C[1],C[2],C[3]),
50 // and A is a 3x3 symmetric matrix with A00 = C[4], A11 = C[5], A22 = C[6],
51 // A01 = C[7]/2, A02 = C[8]/2, and A12 = C[9]/2. Matrix A = R^T D R where
52 // R is orthogonal and D is diagonal (using an eigendecomposition). Define
53 // V = R P = (v0,v1,v2), E = R B = (e0,e1,e2), D = diag(d0,d1,d2), and f = K
54 // to obtain
55 //
56 // d0 v0^2 + d1 v1^2 + d2 v^2 + e0 v0 + e1 v1 + e2 v2 + f = 0
57 //
58 // Characterization depends on the signs of the d_i.
59 template <class Real> WM4_FOUNDATION_ITEM
60 Real QuadraticFit3 (int iQuantity, const Vector3<Real>* akPoint,
61  Real afCoeff[10]);
62 
63 // If you think your points are nearly spherical, use this. Sphere is of form
64 // C'[0]+C'[1]*X+C'[2]*Y+C'[3]*Z+C'[4]*(X^2+Y^2+Z^2) where Length(C') = 1.
65 // Function returns C = (C'[0]/C'[4],C'[1]/C'[4],C'[2]/C'[4],C'[3]/C'[4]), so
66 // fitted sphere is C[0]+C[1]*X+C[2]*Y+C[3]*Z+X^2+Y^2+Z^2. Center is
67 // (xc,yc,zc) = -0.5*(C[1],C[2],C[3]) and radius is rad =
68 // sqrt(xc*xc+yc*yc+zc*zc-C[0]).
69 template <class Real> WM4_FOUNDATION_ITEM
70 Real QuadraticSphereFit3 (int iQuantity, const Vector3<Real>* akPoint,
71  Vector3<Real>& rkCenter, Real& rfRadius);
