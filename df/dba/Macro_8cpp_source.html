74 bool MacroFile::commit()
75 {
76  QFile file(this->macroName);
77  if (!file.open(QFile::WriteOnly)) {
78  return false;
79  }
80 
81  // sort import lines and avoid duplicates
82  QTextStream str(&file);
83 #if QT_VERSION < QT_VERSION_CHECK(6,0,0)
84  str.setCodec("UTF-8");
85 #endif
86  QStringList import;
87  import << QString::fromLatin1("import FreeCAD");
88  QStringList body;
89 
90  for (const auto& it : std::as_const(this->macroInProgress)) {
91  if (it.startsWith(QLatin1String("import ")) ||
92  it.startsWith(QLatin1String("#import "))) {
93  if (import.indexOf(it) == -1)
94  import.push_back(it);
95  }
96  else {
97  body.push_back(it);
98  }
99  }
100 
101  QString header;
102  header += QString::fromLatin1("# -*- coding: utf-8 -*-\n\n");
103  header += QString::fromLatin1("# Macro Begin: ");
104  header += this->macroName;
105  header += QString::fromLatin1(" +++++++++++++++++++++++++++++++++++++++++++++++++\n");
106 
107  QString footer = QString::fromLatin1("# Macro End: ");
108  footer += this->macroName;
109  footer += QString::fromLatin1(" +++++++++++++++++++++++++++++++++++++++++++++++++\n");
110 
111  // write the data to the text file
112  str << header;
113  for (const auto& it : std::as_const(import)) {
114  str << it << QLatin1Char('\n');
115  }
116  str << QLatin1Char('\n');
117  for (const auto& it : body) {
118  str << it << QLatin1Char('\n');
119  }
120  str << footer;
121 
122  this->macroInProgress.clear();
123  this->macroName.clear();
124  this->openMacro = false;
125  file.close();
126  return true;
127 }

269 void MacroManager::addLine(LineType Type, const char* sLine)
270 {

305 void MacroManager::addToOutput(LineType type, const char* line)
306 {
307  auto [comment, record] = option.values(type);
308 
309  QStringList lines = QString::fromUtf8(line).split(QLatin1String("\n"));
310  if (comment) {
311  makeComment(lines);
312  }
313 
314  if (record && macroFile.isOpen()) {
315  macroFile.append(lines);
316  }
317 
318  if (option.scriptToPyConsole) {
319  // search for the Python console
320  auto console = getPythonConsole();
321  if (console) {
322  for(auto &line : lines) {
323  console->printStatement(line);
324  }
325  }
