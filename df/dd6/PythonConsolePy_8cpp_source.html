66 Py::Object PythonStdout::write(const Py::Tuple& args)
67 {
68  PyObject* output;
69  if (!PyArg_ParseTuple(args.ptr(), "O!",&PyUnicode_Type, &output))
70  throw Py::TypeError("PythonStdout.write() takes exactly one argument of type str");
71 
72  PyObject* unicode = PyUnicode_AsEncodedString(output, "utf-8", nullptr);
73  if (unicode) {
74  const char* string = PyBytes_AsString(unicode);
75  int maxlen = qstrlen(string) > 10000 ? 10000 : -1;
76  pyConsole->insertPythonOutput(QString::fromUtf8(string, maxlen));
77  Py_DECREF(unicode);
78  }
79 
80  return Py::None();
81 }
