386 // not currently used
387 App::Color Preferences::lightenColor(App::Color orig)
388 {
389  // get component colours on [0, 255]
390  uchar red = orig.r * 255;
391  uchar blue = orig.b * 255;
392  uchar green = orig.g * 255;
393  // uchar alpha = orig.a * 255;
394 
395  // shift color values
396  uchar m = std::min({red, blue, green});
397  red -= m;
398  blue -= m;
399  green -= m;
400 
401  // calculate chroma (colour range)
402  uchar chroma = std::max({red, blue, green});
403 
404  // calculate lightened colour value
405  uchar newm = 255 - chroma - m;
406  red += newm;
407  green += newm;
408  blue += newm;
409 
410  double redF = (double)red / 255.0;
411  double greenF = (double)green / 255.0;
412  double blueF = (double)blue / 255.0;
413 
414  return App::Color(redF, greenF, blueF, orig.a);
415 }

456 int Preferences::lineStandard()
457 {
458  // there is a condition where the LineStandard parameter exists, but is -1 (the
459  // qt value for no current index in a combobox). This is likely caused by an old
460  // development version writing an unvalidated value. In this case, the
461  // existing but invalid value will be returned. This is a temporary fix and
462  // can be removed for production.
463  // this message will appear many times if the parameter is invalid.
464  int parameterValue = getPreferenceGroup("Standards")->GetInt("LineStandard", 1);
465  if (parameterValue < 0) {
466  Base::Console().Warning(qPrintable(QApplication::translate(
467  "Preferences", "The LineStandard parameter is invalid. Using zero instead.", nullptr)));
468  return 0;
469  }
470  return getPreferenceGroup("Standards")->GetInt("LineStandard", 1);
471 }
