62 bool ViewProviderMultiTransform::onDelete(const std::vector<std::string> &svec) {
63  // Delete the transformation features
64  PartDesign::MultiTransform* pcMultiTransform = getObject<PartDesign::MultiTransform>();
65  std::vector<App::DocumentObject*> transformFeatures = pcMultiTransform->Transformations.getValues();
66 
67  // if the multitransform object was deleted the transformed features must be deleted, too

70  Gui::Command::doCommand(

74  }
75 
76  // Handle Originals
77  return ViewProviderTransformed::onDelete(svec);
