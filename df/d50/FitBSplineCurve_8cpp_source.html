98 void FitBSplineCurveWidget::tryAccept()
99 {
100  QString document = QString::fromStdString(d->obj.getDocumentPython());
101  QString object = QString::fromStdString(d->obj.getObjectPython());
102 
103  QStringList arguments;
104  arguments.append(
105  QString::fromLatin1("Points=getattr(%1, %1.getPropertyNameOfGeometry())").arg(object));
106  if (!d->ui.groupBoxSmooth->isChecked()) {
107  arguments.append(QString::fromLatin1("MinDegree = %1").arg(d->ui.degreeMin->value()));
108  }
109  arguments.append(QString::fromLatin1("MaxDegree = %1").arg(d->ui.degreeMax->value()));
110  arguments.append(QString::fromLatin1("Continuity = %1").arg(d->ui.continuity->currentIndex()));
111  if (d->ui.checkBoxClosed->isChecked()) {
112  arguments.append(QString::fromLatin1("Closed = True"));
113  }
114  else {
115  arguments.append(QString::fromLatin1("Closed = False"));
116  }
117  if (d->ui.checkBox->isChecked()) {
118  int index = d->ui.paramType->currentIndex();
119  arguments.append(QString::fromLatin1("ParametrizationType = %1").arg(index));
120  }
121  if (d->ui.groupBoxSmooth->isChecked()) {
122  arguments.append(QString::fromLatin1("Weight1 = %1").arg(d->ui.curveLength->value()));
123  arguments.append(QString::fromLatin1("Weight2 = %1").arg(d->ui.curvature->value()));
124  arguments.append(QString::fromLatin1("Weight3 = %1").arg(d->ui.torsion->value()));
125  }
126 
127  QString argument = arguments.join(QLatin1String(", "));
128  QString command = QString::fromLatin1("%1.addObject(\"Part::Spline\", \"Spline\").Shape = "
129  "ReverseEngineering.approxCurve(%2).toShape()")
130  .arg(document, argument);
131 
132  tryCommand(command);
133 }
