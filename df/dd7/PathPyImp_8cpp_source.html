56 int PathPy::PyInit(PyObject* args, PyObject* /*kwd*/)
57 {
58  PyObject* pcObj = nullptr;
59  char* gcode;
60  if (PyArg_ParseTuple(args, "|O!", &(PyList_Type), &pcObj)) {
61  if (pcObj) {
62  Py::List list(pcObj);
63  for (Py::List::iterator it = list.begin(); it != list.end(); ++it) {
64  if (PyObject_TypeCheck((*it).ptr(), &(Path::CommandPy::Type))) {
65  Path::Command& cmd =
66  *static_cast<Path::CommandPy*>((*it).ptr())->getCommandPtr();
67  getToolpathPtr()->addCommand(cmd);
68  }
69  else {
70  PyErr_SetString(PyExc_TypeError, "The list must contain only Path Commands");
71  return -1;
72  }
73  }
74  }
75  return 0;
76  }
77  PyErr_Clear(); // set by PyArg_ParseTuple()
78  if (PyArg_ParseTuple(args, "|s", &gcode)) {
79  getToolpathPtr()->setFromGCode(gcode);
80  return 0;
81  }
82  PyErr_SetString(PyExc_TypeError, "Argument must be a list of commands or a gcode string");
83  return -1;
