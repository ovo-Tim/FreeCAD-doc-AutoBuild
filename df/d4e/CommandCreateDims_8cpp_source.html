294  void mouseMoveEvent(QMouseEvent* event) override
295  {
296  mousePos = event->pos();
297 
298  if (dims.empty()){
299  return;
300  }
301 
302  bool textToMiddle = false;
303  Base::Vector3d dirMaster, delta;
304  //Change distance dimension based on position of mouse.
305  if (specialDimension == SpecialDimension::LineOr2PointsDistance
306  || specialDimension == SpecialDimension::LineOr2PointsChamfer){
307  updateDistanceType();
308  }
309  else if (specialDimension == SpecialDimension::ExtendDistance){
310  updateExtentDistanceType();
311  }
312  else if (specialDimension == SpecialDimension::ChainDistance || specialDimension == SpecialDimension::CoordDistance){
313  updateChainDistanceType();
314  textToMiddle = true;
315  pointPair pp = dims[0]->getLinearPoints();
316  dirMaster = pp.second() - pp.first();
317  //dirMaster.y = -dirMaster.y; // not needed because y is reversed between property X/Y and scenePositions
318 
319  QPointF firstPos = getDimLabel(dims[0])->pos();
320  Base::Vector3d pMaster(firstPos.x(), firstPos.y(), 0.0);
321  Base::Vector3d ipDelta = DrawUtil::getTrianglePoint(pMaster, dirMaster, Base::Vector3d());
322  delta = ipDelta.Normalize() * Rez::guiX(activeDimAttributes.getCascadeSpacing());
323  }
324 
325  int i = 0;
326  for (auto* dim : dims) {
327  auto dimType = static_cast<DimensionType>(dim->Type.getValue());
328  moveDimension(mousePos, dim, textToMiddle, dirMaster, delta, dimType, i);
329 
330  if (specialDimension == SpecialDimension::CoordDistance) {
331  i++;
332  }
333  }

348  void moveDimension(QPoint pos, DrawViewDimension* dim, bool textToMiddle = false, Base::Vector3d dir = Base::Vector3d(),
349  Base::Vector3d delta = Base::Vector3d(), DimensionType type = DimensionType::Distance, int i = 0)
350  {
351  if (!dim) { return; }
352  auto label = getDimLabel(dim);
353  if (!label) { return; }
354 
355  label->setPos(getDimPositionToBe(pos, label->pos(), textToMiddle, dir, delta, type, i));

357  QPointF getDimPositionToBe(QPoint pos, QPointF curPos = QPointF(), bool textToMiddle = false, Base::Vector3d dir = Base::Vector3d(),
358  Base::Vector3d delta = Base::Vector3d(), DimensionType type = DimensionType::Distance, int i = 0)
359  {
360  auto* vpp = dynamic_cast<ViewProviderDrawingView*>(Gui::Application::Instance->getViewProvider(partFeat));
361  if (!vpp) { return QPointF(); }
362 
363 
364  QPointF scenePos = viewPage->mapToScene(pos) - vpp->getQView()->scenePos();
365 
366  if (textToMiddle) {
367  // delta is for coord distances. i = 0 when it's a chain so delta is ignored.
368  float dimDistance = Rez::guiX(activeDimAttributes.getCascadeSpacing());
369  if (type == DimensionType::Distance) {
370  Base::Vector3d pos3d(scenePos.x(), scenePos.y(), 0.0);
371  float xDim = curPos.x();
372  float yDim = curPos.y();
373  Base::Vector3d pDim(xDim, yDim, 0.0);
374  Base::Vector3d p3 = DrawUtil::getTrianglePoint(pos3d, dir, pDim);
375  p3 = p3 + delta * i;
376  scenePos = QPointF(p3.x, p3.y);
377  }
378  else if(type == DimensionType::DistanceX) {
379  pointPair pp = dims[0]->getLinearPoints();
380  if (Rez::guiX(pp.first().y) > scenePos.y())
381  dimDistance = -dimDistance;
382 
383  double y = scenePos.y() + i * dimDistance;
384  scenePos = QPointF(curPos.x(), y);
385  }
386  else if(type == DimensionType::DistanceY) {
387  pointPair pp = dims[0]->getLinearPoints();
388  if (Rez::guiX(pp.first().x) > scenePos.x())
389  dimDistance = -dimDistance;
390 
391  double x = scenePos.x() + i * dimDistance;
392  scenePos = QPointF(x, curPos.y());
393  }
394  }
395 
396  return scenePos;
397  }
