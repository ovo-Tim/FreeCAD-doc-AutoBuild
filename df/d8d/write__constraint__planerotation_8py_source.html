60 def write_meshdata_constraint(f, femobj, fric_obj, ccxwriter):
61  # write nodes to file
62  if not ccxwriter.femnodes_mesh:
63  ccxwriter.femnodes_mesh = ccxwriter.femmesh.Nodes
64  # info about ccxwriter.constraint_conflict_nodes:
65  # is used to check if MPC and constraint fixed and
66  # constraint displacement share same nodes
67  # because MPC"s and constraints fixed and
68  # constraints displacement can't share same nodes.
69  # Thus call write_node_sets_constraints_planerotation has to be
70  # after constraint fixed and constraint displacement
71  l_nodes = femobj["Nodes"]
72  f.write(f"*NSET,NSET={fric_obj.Name}\n")
73  # Code to extract nodes and coordinates on the PlaneRotation support face
74  nodes_coords = []
75  for node in l_nodes:
76  nodes_coords.append(
77  (
78  node,
79  ccxwriter.femnodes_mesh[node].x,
80  ccxwriter.femnodes_mesh[node].y,
81  ccxwriter.femnodes_mesh[node].z,
82  )
83  )
84  node_planerotation = meshtools.get_three_non_colinear_nodes(nodes_coords)
85  for i in range(len(l_nodes)):
86  if l_nodes[i] not in node_planerotation:
87  node_planerotation.append(l_nodes[i])
88  MPC_nodes = []
89  for i in range(len(node_planerotation)):
90  cnt = 0
91  for j in range(len(ccxwriter.constraint_conflict_nodes)):
92  if node_planerotation[i] == ccxwriter.constraint_conflict_nodes[j]:
93  cnt = cnt + 1
94  if cnt == 0:
95  MPC = node_planerotation[i]
96  MPC_nodes.append(MPC)
97  for i in range(len(MPC_nodes)):
98  f.write(f"{MPC_nodes[i]},\n")
