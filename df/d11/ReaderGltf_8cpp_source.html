58 void ReaderGltf::read(Handle(TDocStd_Document) hDoc)
59 {

63  aReader.SetSystemLengthUnit(unit);
64  aReader.SetSystemCoordinateSystem(RWMesh_CoordinateSystem_Zup);
65  aReader.SetDocument(hDoc);
66  aReader.SetParallel(true);
67 
68  TCollection_AsciiString filename(file.filePath().c_str());
69  Standard_Boolean ret = aReader.Perform(filename, Message_ProgressRange());

71  throw Base::FileException("Cannot read from file: ", file);
72  }
73 
74  processDocument(hDoc);
75 
76 #else

82 // NOLINTNEXTLINE
83 void ReaderGltf::processDocument(Handle(TDocStd_Document) hDoc)
84 {
85 #if OCC_VERSION_HEX >= 0x070500
86  Handle(XCAFDoc_ShapeTool) aShapeTool = XCAFDoc_DocumentTool::ShapeTool(hDoc->Main());
87 
88  TDF_LabelSequence shapeLabels;
89  aShapeTool->GetShapes(shapeLabels);
90  for (Standard_Integer i = 1; i <= shapeLabels.Length(); i++) {
91  auto topLevelshape = shapeLabels.Value(i);
92  TopoDS_Shape shape = aShapeTool->GetShape(topLevelshape);
93  if (!shape.IsNull()) {
94  TDF_LabelSequence subShapeLabels;
95  if (XCAFDoc_ShapeTool::GetSubShapes(topLevelshape, subShapeLabels)) {
96  TopoDS_Shape compound = processSubShapes(hDoc, subShapeLabels);
97  aShapeTool->SetShape(topLevelshape, compound);
98  }
99  else {
100  aShapeTool->SetShape(topLevelshape, fixShape(shape));
101  }
102  }
103  }
104 #else
105  boost::ignore_unused(hDoc);
106 #endif
107 }

110 TopoDS_Shape ReaderGltf::processSubShapes(Handle(TDocStd_Document) hDoc,
111  const TDF_LabelSequence& subShapeLabels)
112 {
113  TopoDS_Compound compound;
114 #if OCC_VERSION_HEX >= 0x070500
115  Handle(XCAFDoc_ShapeTool) aShapeTool = XCAFDoc_DocumentTool::ShapeTool(hDoc->Main());
116  Handle(XCAFDoc_ColorTool) aColorTool = XCAFDoc_DocumentTool::ColorTool(hDoc->Main());
117  Handle(XCAFDoc_VisMaterialTool) aVisTool = XCAFDoc_DocumentTool::VisMaterialTool(hDoc->Main());
118 
119  BRep_Builder builder;
120  builder.MakeCompound(compound);
121  for (Standard_Integer i = 1; i <= subShapeLabels.Length(); i++) {
122  auto faceLabel = subShapeLabels.Value(i);
123 
124  // OCCT handles colors of a glTF with material labels but the ImportOCAF(2) class
125  // expects color labels. Thus, the material labels are converted into color labels.
126  Handle(XCAFDoc_VisMaterial) aVisMat = aVisTool->GetShapeMaterial(faceLabel);
127 
128  Quantity_ColorRGBA rgba;
129  bool hasVisMat {false};
130  if (!aVisMat.IsNull()) {
131  rgba = aVisMat->BaseColor();
132  hasVisMat = true;
133  }
134 
135  TopoDS_Shape face = aShapeTool->GetShape(faceLabel);
136  TopoDS_Shape fixed = fixShape(face);
137  builder.Add(compound, fixed);
138  aShapeTool->SetShape(faceLabel, fixed);
139 
140  if (hasVisMat) {
141  aColorTool->SetColor(faceLabel, rgba, XCAFDoc_ColorSurf);
142  }
143  }
144 #else
145  boost::ignore_unused(hDoc);
146  boost::ignore_unused(subShapeLabels);
147 #endif
148 
149  return {std::move(compound)};
150 }

162 TopoDS_Shape ReaderGltf::fixShape(TopoDS_Shape shape) // NOLINT
163 {
164  // The glTF reader creates a compound of faces that only contains the triangulation
165  // but not the underlying surfaces. This leads to faces without boundaries.
166  // The triangulation is used to create a valid shape.
167  const double tolerance = 0.5;
168  std::vector<Base::Vector3d> points;
169  std::vector<Data::ComplexGeoData::Facet> facets;
170  Part::TopoShape sh(shape);
171  sh.getFaces(points, facets, tolerance);
172  sh.setFaces(points, facets, tolerance);
173 
174  if (cleanup()) {
175  sh.sewShape();
176  return sh.removeSplitter();
177  }
178 
179  return sh.getShape();
180 }
