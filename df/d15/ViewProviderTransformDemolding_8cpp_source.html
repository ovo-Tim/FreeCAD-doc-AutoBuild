99 
100  auto pcMatBinding = new SoMaterialBinding;

131 void ViewProviderMeshTransformDemolding::calcNormalVector()
132 {
133  const Mesh::MeshObject& mesh = getMeshObject();
134  const MeshCore::MeshKernel& kernel = mesh.getKernel();
135 
136  MeshFacetIterator cFIt(kernel);

139 
140  Base::Vector3f norm(rFace.GetNormal());
