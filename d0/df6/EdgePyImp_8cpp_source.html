96 PyObject* EdgePy::intersectWithEdge(PyObject* args)
97 {
98  PyObject* object {};
99  if (!PyArg_ParseTuple(args, "O!", &EdgePy::Type, &object)) {
100  return nullptr;
101  }
102  EdgePy* edge = static_cast<EdgePy*>(object);
103  EdgePy::PointerType edge_ptr = edge->getEdgePtr();
104  EdgePy::PointerType this_ptr = this->getEdgePtr();
105  Base::Vector3f p;
106  bool ok = this_ptr->IntersectWithEdge(*edge_ptr, p);
107 
108  try {
109  Py::List sct;
110  if (ok) {
111  Py::Tuple pt(3);
112  pt.setItem(0, Py::Float(p.x));
113  pt.setItem(1, Py::Float(p.y));
114  pt.setItem(2, Py::Float(p.z));
115  sct.append(pt);
116  }
117  return Py::new_reference_to(sct);
118  }
119  catch (const Py::Exception&) {
120  return nullptr;
