99 void SoFCOffscreenRenderer::writeToImageFile(const char* filename, const char* comment, const SbMatrix& mat, const QImage& image)
100 {
101  if (image.isNull()) {
102  std::stringstream str;
103  str << "Cannot save null image.";
104  throw Base::ValueError(str.str());
105  }
106 
107  Base::FileInfo file(filename);
108  if (file.hasExtension({"JPG", "JPEG"})) {
109  // writing comment in case of jpeg (Qt ignores setText() in case of jpeg)
110  std::string com;
111  if (strcmp(comment,"")==0)
112  com = "Screenshot created by FreeCAD";
113  else if (strcmp(comment,"$MIBA")==0)
114  com = createMIBA(mat);
115  else
116  com = comment;
117 
118  // write into memory
119  QByteArray ba;
120  QBuffer buffer(&ba);
121  buffer.open(QIODevice::WriteOnly);
122  image.save(&buffer, "JPG", 90);
123  writeJPEGComment(com, ba);
124 
125  QFile file(QString::fromUtf8(filename));
126  if (file.open(QFile::WriteOnly)) {
127  file.write(ba);
128  file.close();
129  }
130  else {
131  std::stringstream str;
132  str << "Cannot open file '" << filename << "' for writing.";
133  throw Base::FileSystemError(str.str());
134  }
135  }
136  else {
137  // check for all QImage formats
138  bool supported = false;
139  QByteArray format;
140  QList<QByteArray> qtformats = QImageWriter::supportedImageFormats();
141  for (const auto & it : qtformats) {
142  if (file.hasExtension(it.data())) {
143  format = it;
144  supported = true;
145  break;
146  }
147  }
148 
149  // Supported by Qt
150  if (supported) {
151  QImage img = image;
152  // set keywords for PNG format
153  if (file.hasExtension("PNG")) {
154  img.setText(QLatin1String("Title"), QString::fromUtf8(filename));
155  img.setText(QLatin1String("Author"), QLatin1String("FreeCAD (https://www.freecad.org)"));
156  if (strcmp(comment,"")==0)
157  img.setText(QLatin1String("Description"), QLatin1String("Screenshot created by FreeCAD"));
158  else if (strcmp(comment,"$MIBA")==0)
159  img.setText(QLatin1String("Description"), QLatin1String(createMIBA(mat).c_str()));
160  else
161  img.setText(QLatin1String("Description"), QString::fromUtf8(comment));
162  img.setText(QLatin1String("Creation Time"), QDateTime::currentDateTime().toString());
163  img.setText(QLatin1String("Software"),
164  QString::fromStdString(App::Application::getExecutableName()));
165  }
166 
167  QFile f(QString::fromUtf8(filename));
168  if (f.open(QFile::WriteOnly)) {
169  if (img.save(&f, format.data())) {
170  f.close();
171  }
172  else {
173  f.close();
174  std::stringstream str;
175  str << "Cannot save image to file '" << filename << "'.";
176  throw Base::ValueError(str.str());
177  }
178  }
179  else {
180  std::stringstream str;
181  str << "Cannot open file '" << filename << "' for writing.";
182  Base::FileException e;
183  e.setMessage(str.str());
184  throw e;
185  }
186  }
187  //
188  // Use internal buffer instead of QImage
189  //
190  else if (isWriteSupported(file.extension().c_str())) {
191  // Any format which is supported by Coin only
192  if (!writeToFile(filename, file.extension().c_str()))
193  throw Base::FileException("Error writing image file", filename);
194  }
195  else if (file.hasExtension({"EPS", "PS"})) {
196  // Any format which is supported by Coin only
197 #ifdef FC_OS_WIN32
198  FILE* fd = _wfopen(file.toStdWString().c_str(), L"w");
199 #else
200  FILE* fd = fopen(filename, "w");
201 #endif
202  bool ok = writeToPostScript(fd);
203  fclose(fd);
204  if (!ok)
205  throw Base::FileException("Error writing image file", filename);
206  }
207  else if (file.hasExtension({"RGB", "SGI"})) {
208  // Any format which is supported by Coin only
209 #ifdef FC_OS_WIN32
210  FILE* fd = _wfopen(file.toStdWString().c_str(), L"w");
211 #else
212  FILE* fd = fopen(filename, "w");
213 #endif
214  bool ok = writeToRGB(fd);
215  fclose(fd);
216  if (!ok)
217  throw Base::FileException("Error writing image file", filename);
218  }
219  }

222 QStringList SoFCOffscreenRenderer::getWriteImageFiletypeInfo()
223 {
224  QStringList formats;
225 
226  // get all supported formats by Coin3D
227  int num = getNumWriteFiletypes();
228  for (int i=0; i < num; i++) {
229 
230  SbPList extlist;
231 
232  SbString fullname, description;
233  getWriteFiletypeInfo(i, extlist, fullname, description);
234 
235  for (int j=0; j < extlist.getLength(); j++) {
236  QString ext = QLatin1String((const char*) extlist[j]);
237  if (formats.indexOf(ext.toUpper()) == -1)
238  formats << ext.toUpper();
239  }
240  }
241 
242  // add now all further QImage formats
243  QList<QByteArray> qtformats = QImageWriter::supportedImageFormats();
244  for (const auto & it : qtformats) {
245  // not supported? then append
246  if (!isWriteSupported(it.data()) && formats.indexOf(QLatin1String(it)) == -1)
247  formats << QLatin1String(it);
248  }
249 
250  // now add PostScript and SGI RGB
251  if (formats.indexOf(QLatin1String("EPS")) == -1)
252  formats << QLatin1String("EPS");
253  else if (formats.indexOf(QLatin1String("SGI")) == -1)
254  formats << QLatin1String("SGI");
255 
256  formats.sort();
257 
258  return formats;
259 }

294 void writeJPEGComment(const std::string& comment, QByteArray& ba)
295 {
296  const unsigned char M_SOF0 = 0xc0;
297  const unsigned char M_SOF1 = 0xc1;
298  const unsigned char M_SOF2 = 0xc2;
299  const unsigned char M_SOF3 = 0xc3;
300  const unsigned char M_SOF5 = 0xc5;
301  const unsigned char M_SOF6 = 0xc6;
302  const unsigned char M_SOF7 = 0xc7;
303  const unsigned char M_SOF9 = 0xc9;
304  const unsigned char M_SOF10 = 0xcA;
305  const unsigned char M_SOF11 = 0xcb;
306  const unsigned char M_SOF13 = 0xcd;
307  const unsigned char M_SOF14 = 0xce;
308  const unsigned char M_SOF15 = 0xcf;
309  const unsigned char M_SOI = 0xd8;
310  const unsigned char M_EOI = 0xd9;
311  const unsigned char M_COM = 0xfe;
312 
313  class Byte {
314  char c{};
315  unsigned char u{};
316  public:
317  void setc(char v) {
318  c = v;
319  std::memcpy(&u, &c, sizeof(u));
320  }
321  void setu(unsigned char v) {
322  u = v;
323  std::memcpy(&c, &u, sizeof(c));
324  }
325  char getc() const {
326  return c;
327  }
328  unsigned char getu() const {
329  return u;
330  }
331  };
332 
333  if (comment.empty() || ba.length() < 2)
334  return;
335 
336  // first marker
337  Byte a,b;
338  a.setc(ba[0]);
339  b.setc(ba[1]);
340  if (a.getu() == 0xff && b.getu() == M_SOI) {
341  int index = 2;
342  int len = ba.length();
343  while (index < len) {
344  // next marker
345  a.setc(ba[index++]);
346  while (a.getu() != 0xff && index < len) {
347  a.setc(ba[index++]);
348  }
349  do {
350  b.setc(ba[index++]);
351  } while (b.getu() == 0xff && index < len);
352  switch (b.getu()) {
353  case M_SOF0:
354  case M_SOF1:
355  case M_SOF2:
356  case M_SOF3:
357  case M_SOF5:
358  case M_SOF6:
359  case M_SOF7:
360  case M_SOF9:
361  case M_SOF10:
362  case M_SOF11:
363  case M_SOF13:
364  case M_SOF14:
365  case M_SOF15:
366  case M_EOI:
367  {
368  Byte a, b;
369  a.setu(0xff);
370  b.setu(M_COM);
371  index -= 2; // insert comment before marker
372  ba.insert(index++, a.getc());
373  ba.insert(index++, b.getc());
374  int val = comment.size() + 2;
375  ba.insert(index++,(val >> 8) & 0xff);
376  ba.insert(index++,val & 0xff);
377  ba.insert(index, comment.c_str());
378  index = len; // finished
379  } break;
380  case M_COM:
381  default:
382  {
383  Byte a, b;
384  a.setc(ba[index++]);
385  b.setc(ba[index++]);
386  int off = ((unsigned int)a.getu() << 8) + (unsigned int)b.getu();
387  index += off;
388  index -= 2; // next marker
389  } break;
390  }
391  }
392  }
393 }

557 SoQtOffscreenRenderer::makeFrameBuffer(int width, int height, int samples)
558 {
559  if (framebuffer) {
560  delete framebuffer;
561  framebuffer = nullptr;
562  }
563 
564  viewport.setWindowSize(width, height);
565 
566  QtGLFramebufferObjectFormat fmt;
567  fmt.setSamples(samples);
568  fmt.setAttachment(QtGLFramebufferObject::Depth);
569  // With enabled alpha a transparent background is supported but
570  // at the same time breaks semi-transparent models. A workaround
571  // is to use a certain background color using GL_RGB as texture
572  // format and in the output image search for the above color and

574  fmt.setInternalTextureFormat(this->texFormat);

577  cache_context = SoGLCacheContextElement::getUniqueCacheContext(); // unique per pixel buffer object, just to be sure
578 }

580 SbBool
581 SoQtOffscreenRenderer::renderFromBase(SoBase * base)
582 {
583  const SbVec2s fullsize = this->viewport.getViewportSizePixels();
584 
585  QSurfaceFormat format;
586  format.setSamples(PRIVATE(this)->numSamples);
587  QOpenGLContext context;
588  context.setFormat(format);
589  if (!context.create())
590  return false;
591  QOffscreenSurface offscreen;
592  offscreen.setFormat(format);
593  offscreen.create();
594  context.makeCurrent(&offscreen);
595 
596  if (!framebuffer) {
597  makeFrameBuffer(fullsize[0], fullsize[1], PRIVATE(this)->numSamples);
598  }
599  else if (framebuffer->width() != fullsize[0] || framebuffer->height() != fullsize[1]) {
600  // get the size right!
601  makeFrameBuffer(fullsize[0], fullsize[1], PRIVATE(this)->numSamples);
602  }
603 
604  framebuffer->bind(); // activate us!
605 
606  // oldcontext is used to restore the previous context id, in case
607  // the render action is not allocated by us.
608  const uint32_t oldcontext = this->renderaction->getCacheContext();
609  this->renderaction->setCacheContext(cache_context);
610 
611  glEnable(GL_DEPTH_TEST);
612  glClearColor(this->backgroundopaque[0],
613  this->backgroundopaque[1],
614  this->backgroundopaque[2],
615  this->backgroundopaque[3]);
616 
617  // needed to clear viewport after glViewport() is called from
618  // SoGLRenderAction
619  this->renderaction->addPreRenderCallback(pre_render_cb, nullptr);
620  this->renderaction->setViewportRegion(this->viewport);
621 
622  if (base->isOfType(SoNode::getClassTypeId()))
623  this->renderaction->apply((SoNode *)base);
624  else if (base->isOfType(SoPath::getClassTypeId()))
625  this->renderaction->apply((SoPath *)base);
626  else {
627  assert(false && "Cannot apply to anything else than an SoNode or an SoPath");
628  }
629 
630  this->renderaction->removePreRenderCallback(pre_render_cb, nullptr);
631  framebuffer->release();
632 
633  this->renderaction->setCacheContext(oldcontext); // restore old
634 
635  glImage = framebuffer->toImage();
636  context.doneCurrent();
637 
638  return true;
639 }
