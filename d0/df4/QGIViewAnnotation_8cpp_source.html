107 void QGIViewAnnotation::drawAnnotation()
108 {
109  // Base::Console().Message("QGIVA::drawAnnotation()\n");
110  auto viewAnno(dynamic_cast<TechDraw::DrawViewAnnotation*>(getViewObject()));
111  if (!viewAnno) {
112  return;
113  }
114 
115  const std::vector<std::string>& annoRawText = viewAnno->Text.getValues();
116  std::vector<std::string> annoText;
117  // v0.19- stored text as escapedUnicode
118  // v0.20+ stores text as utf8
119  for (auto& line : annoRawText) {
120  if (line.find("\\x") == std::string::npos) {
121  // not escaped
122  annoText.push_back(line);
123  } else {
124  // is escaped
125  std::string newLine = Base::Tools::escapedUnicodeToUtf8(line);
126  annoText.push_back(newLine);
127  }
128  }
129 
130  int scaledSize = exactFontSize(viewAnno->Font.getValue(), viewAnno->TextSize.getValue());
131 
132  //build HTML/CSS formatting around Text lines
133  std::stringstream ss;
134  ss << "<html>\n<head>\n<style>\n";
135  ss << "p {";
136  ss << "font-family:" << viewAnno->Font.getValue() << "; ";
137  ss << "font-size:" << scaledSize << "px; ";
138  if (viewAnno->TextStyle.isValue("Normal")) {
139  ss << "font-weight:normal; font-style:normal; ";
140  } else if (viewAnno->TextStyle.isValue("Bold")) {
141  ss << "font-weight:bold; font-style:normal; ";
142  } else if (viewAnno->TextStyle.isValue("Italic")) {
143  ss << "font-weight:normal; font-style:italic; ";
144  } else if (viewAnno->TextStyle.isValue("Bold-Italic")) {
145  ss << "font-weight:bold; font-style:italic; ";
146  } else {
147  Base::Console().Warning("%s has invalid TextStyle\n", viewAnno->getNameInDocument());
148  ss << "font-weight:normal; font-style:normal; ";
149  }
150  ss << "line-height:" << viewAnno->LineSpace.getValue() << "%; ";
151  App::Color c = viewAnno->TextColor.getValue();
152  c = TechDraw::Preferences::getAccessibleColor(c);
153  ss << "color:" << c.asHexString() << "; ";
154  ss << "}\n</style>\n</head>\n<body>\n<p>";
155  for (std::vector<std::string>::const_iterator it = annoText.begin(); it != annoText.end();
156  it++) {
157  if (it != annoText.begin()) {
158  ss << "<br>";
159  }
160 
161  //"less than" symbol chops off line. need to use html sub.
162  std::string lt = std::regex_replace((*it), std::regex("<"), "&lt;");
163  ss << lt;
164  }
165  ss << "</p>\n</body>\n</html> ";
166 
167  prepareGeometryChange();
168  m_textItem->setTextWidth(Rez::guiX(viewAnno->MaxWidth.getValue()));
169  QString qs = QString::fromUtf8(ss.str().c_str());
170  m_textItem->setHtml(qs);
171  m_textItem->centerAt(0., 0.);
172 }
