498 def CalculatePlacement(baramount, barnumber, bardiameter, size, axis, rotation, offsetstart, offsetend, RebarShape = ""):
499 
500  """ CalculatePlacement([baramount, barnumber, bardiameter, size, axis, rotation, offsetstart, offsetend, RebarShape]):
501  Calculate the placement of the bar from given values."""
502  if baramount == 1:
503  interval = offsetstart
504  else:
505  if RebarShape == "Stirrup":
506  interval = size - (offsetstart + offsetend + bardiameter)
507  else:
508  interval = size - (offsetstart + offsetend)

516

530 def strprocessOfCustomSpacing(span_string):
531 
532  """ strprocessOfCustomSpacing(span_string): This function take input
533  in specific syntax and return output in the form of list. For eg.
534  Input: "3@100+2@200+3@100"
535  Output: [100, 100, 100, 200, 200, 100, 100, 100]"""
536  # import string
537  span_st = span_string.strip()
538  span_sp = span_st.split('+')
539  index = 0
540  spacinglist = []
541  while index < len(span_sp):
542  # Find "@" recursively in span_sp array.
543  # If not found, append the index value to "spacinglist" array.
544  if span_sp[index].find('@') == -1:
545  spacinglist.append(float(span_sp[index]))
546  else:
547  in_sp = span_sp[index].split('@')
548  count = 0
549  while count < int(in_sp[0]):
550  spacinglist.append(float(in_sp[1]))
551  count += 1
552  index += 1
553  return spacinglist
