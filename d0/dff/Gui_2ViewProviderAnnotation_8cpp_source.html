116 void ViewProviderAnnotation::onChanged(const App::Property* prop)
117 {
118  if (prop == &TextColor) {
119  const App::Color& c = TextColor.getValue();
120  pColor->rgb.setValue(c.r,c.g,c.b);
121  }
122  else if (prop == &Justification) {
123  if (Justification.getValue() == 0) {
124  pLabel->justification = SoText2::LEFT;
125  pLabel3d->justification = SoAsciiText::LEFT;
126  }
127  else if (Justification.getValue() == 1) {
128  pLabel->justification = SoText2::RIGHT;
129  pLabel3d->justification = SoAsciiText::RIGHT;
130  }
131  else if (Justification.getValue() == 2) {
132  pLabel->justification = SoText2::CENTER;
133  pLabel3d->justification = SoAsciiText::CENTER;
134  }
135  }
136  else if (prop == &FontSize) {
137  pFont->size = FontSize.getValue();
138  }
139  else if (prop == &FontName) {
140  pFont->name = FontName.getValue();
141  }
142  else if (prop == &LineSpacing) {
143  pLabel->spacing = LineSpacing.getValue();
144  pLabel3d->spacing = LineSpacing.getValue();
145  }
146  else if (prop == &RotationAxis) {
147  if (RotationAxis.getValue() == 0) {
148  pRotationXYZ->axis = SoRotationXYZ::X;
149  }
150  else if (RotationAxis.getValue() == 1) {
151  pRotationXYZ->axis = SoRotationXYZ::Y;
152  }
153  else if (RotationAxis.getValue() == 2) {
154  pRotationXYZ->axis = SoRotationXYZ::Z;
155  }
156  }
157  else if (prop == &Rotation) {
158  pRotationXYZ->angle = (Rotation.getValue()/360)*(2*M_PI);
159  }
160  else {
161  ViewProviderDocumentObject::onChanged(prop);
162  }
163 }

239 void ViewProviderAnnotation::updateData(const App::Property* prop)
240 {
241  if (prop->is<App::PropertyStringList>() &&
242  strcmp(prop->getName(),"LabelText") == 0) {
243  const std::vector<std::string> lines = static_cast<const App::PropertyStringList*>(prop)->getValues();
244  int index=0;
245  pLabel->string.setNum((int)lines.size());
246  pLabel3d->string.setNum((int)lines.size());
247  for (const auto & line : lines) {
248  const char* cs = line.c_str();
249  if (line.empty())
250  cs = " "; // empty lines make coin crash, we use a space instead
251 #if (COIN_MAJOR_VERSION <= 3)
252  QByteArray latin1str;
253  latin1str = (QString::fromUtf8(cs)).toLatin1();
254  pLabel->string.set1Value(index, SbString(latin1str.constData()));
255  pLabel3d->string.set1Value(index, SbString(latin1str.constData()));
256 #else
257  pLabel->string.set1Value(index, SbString(cs));
258  pLabel3d->string.set1Value(index, SbString(cs));
259 #endif
260  index++;
261  }
262  }
263  else if (prop->is<App::PropertyVector>() &&
264  strcmp(prop->getName(),"Position") == 0) {
265  Base::Vector3d v = static_cast<const App::PropertyVector*>(prop)->getValue();
266  pTranslation->translation.setValue(v.x,v.y,v.z);
267  }
268 
269  ViewProviderDocumentObject::updateData(prop);
270 }

445 void ViewProviderAnnotationLabel::drawImage(const std::vector<std::string>& s)
446 {
447  if (s.empty()) {
448  pImage->image = SoSFImage();
449  this->hide();
450  return;
451  }
452 
453  QFont font(QString::fromLatin1(this->FontName.getValue()), (int)this->FontSize.getValue());
454  QFontMetrics fm(font);
455  int w = 0;
456  int h = fm.height() * s.size();
457  const App::Color& b = this->BackgroundColor.getValue();
458  QColor brush;
459  brush.setRgbF(b.r,b.g,b.b);
460  const App::Color& t = this->TextColor.getValue();
461  QColor front;
462  front.setRgbF(t.r,t.g,t.b);
463 
464  QStringList lines;
465  for (const auto & it : s) {
466  QString line = QString::fromUtf8(it.c_str());
467  w = std::max<int>(w, QtTools::horizontalAdvance(fm, line));
468  lines << line;
469  }
470 
471  QImage image(w+10,h+10,QImage::Format_ARGB32_Premultiplied);
472  image.fill(0x00000000);
473  QPainter painter(&image);
474  painter.setRenderHint(QPainter::Antialiasing);
475 
476  bool drawFrame = this->Frame.getValue();
477  if (drawFrame) {
478  painter.setPen(QPen(QColor(0,0,127), 2, Qt::SolidLine, Qt::RoundCap,
479  Qt::RoundJoin));
480  painter.setBrush(QBrush(brush, Qt::SolidPattern));
481  QRectF rectangle(0.0, 0.0, w+10, h+10);
482  painter.drawRoundedRect(rectangle, 5, 5);
483  }
484 
485  painter.setPen(front);
486 
487  Qt::Alignment align = Qt::AlignVCenter;
488  if (Justification.getValue() == 0)
489  align = Qt::AlignVCenter | Qt::AlignLeft;
490  else if (Justification.getValue() == 1)
491  align = Qt::AlignVCenter | Qt::AlignRight;
492  else
493  align = Qt::AlignVCenter | Qt::AlignHCenter;
494  QString text = lines.join(QLatin1String("\n"));
495  painter.setFont(font);
496  painter.drawText(5,5,w,h,align,text);
497  painter.end();
498 
499  SoSFImage sfimage;
500  Gui::BitmapFactory().convert(image, sfimage);
501  pImage->image = sfimage;
502 }
