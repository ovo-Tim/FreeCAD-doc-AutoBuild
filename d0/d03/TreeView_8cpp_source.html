55 void TreeView::mouseDoubleClickEvent (QMouseEvent * event)
56 {
57  QModelIndex index = indexAt(event->pos());
58  if (!index.isValid() || index.internalPointer() == Application::Instance)
59  return;
60  Base::BaseClass* item = nullptr;
61  item = static_cast<Base::BaseClass*>(index.internalPointer());
62  if (item->is<Document>()) {
63  QTreeView::mouseDoubleClickEvent(event);
64  const Gui::Document* doc = static_cast<Gui::Document*>(item);
65  MDIView *view = doc->getActiveView();
66  if (!view)
67  return;
68  getMainWindow()->setActiveWindow(view);
69  }
70  else if (item->isDerivedFrom<ViewProvider>()) {
71  if (!static_cast<ViewProvider*>(item)->doubleClicked())
72  QTreeView::mouseDoubleClickEvent(event);
73  }
74 }
