81  void readXML()
82  {
83  readProgramVersion();
84 
85  std::map<std::string, std::string> propMap = initMap();
86 
87  DOMNodeList* nodes = xmlDocument->getElementsByTagName(XStr("Properties").unicodeForm());
88  for (XMLSize_t i = 0; i < nodes->getLength(); i++) {
89  DOMNode* node = nodes->item(i);
90  if (node->getNodeType() == DOMNode::ELEMENT_NODE) {
91  auto elem = static_cast<DOMElement*>(node); // NOLINT
92  DOMNodeList* propList = elem->getElementsByTagName(XStr("Property").unicodeForm());
93  for (XMLSize_t j = 0; j < propList->getLength(); j++) {
94  DOMNode* propNode = propList->item(j);
95  readProperty(propNode, propMap);
96  }
97  }
98  break;
99  }
100 
101  setMetadata(propMap);
102  }
103 
104 private:
105  void readProgramVersion()
106  {
107  if (DOMNodeList* nodes =
108  xmlDocument->getElementsByTagName(XStr("Document").unicodeForm())) {
109  for (XMLSize_t i = 0; i < nodes->getLength(); i++) {
110  DOMNode* node = nodes->item(i);
111  if (node->getNodeType() == DOMNode::ELEMENT_NODE) {
112  DOMNode* nameAttr =
113  node->getAttributes()->getNamedItem(XStr("ProgramVersion").unicodeForm());
114  if (nameAttr) {
115  std::string value = StrX(nameAttr->getNodeValue()).c_str();
116  metadata.programVersion = value;
117  break;
118  }
119  }
120  }
121  }
122  }

154  static void readProperty(DOMNode* propNode, std::map<std::string, std::string>& propMap)
155  {
156  DOMNode* nameAttr = propNode->getAttributes()->getNamedItem(XStr("name").unicodeForm());
157  if (nameAttr) {
158  std::string name = StrX(nameAttr->getNodeValue()).c_str();
159  auto it = propMap.find(name);
160  if (it != propMap.end()) {
161  it->second = readValue(propNode);
162  }
163  }
164  }
165 
166  static std::string readValue(DOMNode* node)
167  {
168  if (node->getNodeType() == DOMNode::ELEMENT_NODE) {
169  if (DOMElement* child =
170  static_cast<DOMElement*>(node)->getFirstElementChild()) { // NOLINT
171  if (DOMNode* nameAttr =
172  child->getAttributes()->getNamedItem(XStr("value").unicodeForm())) {
173  std::string value = StrX(nameAttr->getNodeValue()).c_str();
174  return value;
175  }
176  }
177  }
178 
179  return {};
180  }

209 bool ProjectFile::loadDocument()
210 {
211  if (xmlDocument) {
212  return true; // already loaded
213  }
214 
215  zipios::ZipFile project(stdFile);
216  if (!project.isValid()) {
217  return false;
218  }
219  std::unique_ptr<std::istream> str(project.getInputStream("Document.xml"));
220  if (str) {
221  std::unique_ptr<XercesDOMParser> parser(new XercesDOMParser);
222  parser->setValidationScheme(XercesDOMParser::Val_Auto);
223  parser->setDoNamespaces(false);
224  parser->setDoSchema(false);
225  parser->setValidationSchemaFullChecking(false);
226  parser->setCreateEntityReferenceNodes(false);
227 
228  try {
229  Base::StdInputSource inputSource(*str, stdFile.c_str());
230  parser->parse(inputSource);
231  xmlDocument = parser->adoptDocument();
232  return true;
233  }
234  catch (const XMLException&) {
235  return false;
236  }
237  catch (const DOMException&) {
238  return false;
239  }
240  }
241 
242  return false;
243 }

319 bool ProjectFile::restoreObject(const std::string& name, App::PropertyContainer* obj, bool verbose)
320 {
321  Base::FileInfo fi(stdFile);
322  Base::ifstream file(fi, std::ios::in | std::ios::binary);
323 
324  zipios::ZipInputStream zipstream(file);
325  Base::XMLReader reader(stdFile.c_str(), zipstream);
326  reader.setVerbose(verbose);
327 
328  if (!reader.isValid()) {
329  return false;
330  }
331 
332  // skip document properties
333  reader.readElement("Properties");
334  reader.readEndElement("Properties");
335 
336  // skip objects
337  reader.readElement("Objects");
338  reader.readEndElement("Objects");
339 
340  reader.readElement("ObjectData");
341  long Cnt = reader.getAttributeAsInteger("Count");
342  for (long i = 0; i < Cnt; i++) {
343  reader.readElement("Object");
344  std::string nameAttr = reader.getAttribute("name");
345 
346  if (nameAttr == name) {
347  // obj->StatusBits.set(4);
348  obj->Restore(reader);
349  // obj->StatusBits.reset(4);
350  }
351  reader.readEndElement("Object");
352  }
353  reader.readEndElement("ObjectData");
354 
355  reader.readFiles(zipstream);
356 
357  return true;
358 }

432 void ProjectFile::findFiles(XERCES_CPP_NAMESPACE_QUALIFIER DOMNode* node,
433  std::list<ProjectFile::PropertyFile>& files) const
434 {
435  if (node->hasAttributes()) {
436  ProjectFile::PropertyFile prop;
437  DOMNode* fileAttr = node->getAttributes()->getNamedItem(XStr("file").unicodeForm());
438  if (fileAttr) {
439  DOMNode* parentNode = node->getParentNode();
440  if (parentNode) {
441  DOMNode* nameAttr =
442  parentNode->getAttributes()->getNamedItem(XStr("name").unicodeForm());
443  if (nameAttr) {
444  prop.name = StrX(nameAttr->getNodeValue()).c_str();
445  }
446 
447  DOMNode* typeAttr =
448  parentNode->getAttributes()->getNamedItem(XStr("type").unicodeForm());
449  if (typeAttr) {
450  prop.type = Base::Type::fromName(StrX(typeAttr->getNodeValue()).c_str());
451  }
452  }
453 
454  prop.file = StrX(fileAttr->getNodeValue()).c_str();
455  files.push_back(prop);
456  }
457  }
458 
459  DOMNodeList* subNodes = node->getChildNodes();
460  for (XMLSize_t i = 0; i < subNodes->getLength(); i++) {
461  DOMNode* child = subNodes->item(i);
462  findFiles(child, files);
463  }
464 }

569 std::string ProjectFile::replaceInputFile(const std::string& name, std::istream& inp)
570 {
571  // create a new zip file with the name '<zipfile>.<uuid>'
572  std::string uuid = Base::Uuid::createUuid();
573  std::string fn = stdFile;
574  fn += ".";
575  fn += uuid;
576  Base::FileInfo tmp(fn);
577  Base::ofstream newZip(tmp, std::ios::out | std::ios::binary);
578 
579  // standard compression
580  const int compressionLevel = 6;
581  zipios::ZipOutputStream outZip(newZip);
582  outZip.setComment("FreeCAD Document");
583  outZip.setLevel(compressionLevel);
584 
585  // open the original zip file
586  zipios::ZipFile project(stdFile);
587  zipios::ConstEntries files = project.entries();
588  for (const auto& it : files) {
589  std::string file = it->getFileName();
590  outZip.putNextEntry(file);
591  if (file == name) {
592  inp >> outZip.rdbuf();
593  }
594  else {
595  std::unique_ptr<std::istream> str(project.getInputStream(file));
596  if (str) {
597  *str >> outZip.rdbuf();
598  }
599  }
600  }
601 
602  project.close();
603  outZip.close();
604  newZip.close();
605 
606  return fn;
607 }
