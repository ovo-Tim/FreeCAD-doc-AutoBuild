110 static inline
111 SoMaterial *createMaterial(void)
112 {
113  const QColor ambientColor {0xff333333},
114  diffuseColor {0xffd2d2ff},
115  emissiveColor {0xff000000},
116  specularColor {0xffcccccc};
117

121  material->emissiveColor.setValue(emissiveColor.redF(), emissiveColor.greenF(), emissiveColor.blueF());

138 void DlgSettingsLightSources::createViewer()
139 {
140  const QColor default_bg_color {180, 180, 180};
141  const SbVec3f default_view_direction {1.0f, 1.0f, -5.0f};
142 
143  // NOLINTBEGIN
144  view->setRedirectToSceneGraph(true);
145  view->setViewing(true);
146  view->setPopupMenuEnabled(false);
147 
148  view->setBackgroundColor(default_bg_color);
149  view->setGradientBackground(Gui::View3DInventorViewer::NoGradient);
150  view->setEnabledNaviCube(false);
151 
152  auto root = static_cast<SoSeparator*>(view->getSceneGraph());
153  root->addChild(createDragger());
154  root->addChild(createMaterial());
155  root->addChild(createSphere());
156 
157  auto callback = new SoEventCallback();
158  root->addChild(callback);
159  callback->addEventCallback(SoEvent::getClassTypeId(),
160  [] (void* ud, SoEventCallback* cb) {
161  Q_UNUSED(ud)
162  cb->setHandled();
163  });
164 
165  view->setCameraType(SoOrthographicCamera::getClassTypeId());
166  view->setViewDirection(default_view_direction);
167  view->viewAll();
168 
169  camera = dynamic_cast <SoOrthographicCamera *> (view->getCamera());
170  const float camera_height = camera->height.getValue() * 2.0f;
171  camera->height = camera_height;
172  cam_step = camera_height / 14.0f;
173  // NOLINTEND
174 }

176 SoDirectionalLightDragger* DlgSettingsLightSources::createDragger()
177 {
178  // NOLINTBEGIN
179  lightDragger = new SoDirectionalLightDragger();
180  if (SoDragger* translator = dynamic_cast<SoDragger *>(lightDragger->getPart("translator", false))) {
181  translator->setPartAsDefault("xTranslator.translatorActive", nullptr);
182  translator->setPartAsDefault("yTranslator.translatorActive", nullptr);
183  translator->setPartAsDefault("zTranslator.translatorActive", nullptr);
184  translator->setPartAsDefault("xTranslator.translator", nullptr);
185  translator->setPartAsDefault("yTranslator.translator", nullptr);
186  translator->setPartAsDefault("zTranslator.translator", nullptr);
187  SoNode* node = translator->getPart("yzTranslator.translator", false);
188  if (node && node->isOfType(SoGroup::getClassTypeId())) {
189  auto ps = new SoPickStyle();
190  ps->style = SoPickStyle::UNPICKABLE;
191  static_cast<SoGroup*>(node)->insertChild(ps, 0);
192  }
193  }
194 
195  lightDragger->addMotionCallback(dragMotionCallback, this);
196  return lightDragger;
197  // NOLINTEND
198 }

249 void DlgSettingsLightSources::loadDirection()
250 {
251  ParameterGrp::handle grp = ui->sliderIntensity1->getWindowParameter();
252  SbRotation rotation = lightDragger->rotation.getValue();
253 
254  auto get_q = [&grp](const char *name, const float def){return static_cast <float> (grp->GetFloat(name, def));};
255 
256  const float q0 = get_q("HeadlightRotationX", rotation[0]),
257  q1 = get_q("HeadlightRotationY", rotation[1]),
258  q2 = get_q("HeadlightRotationZ", rotation[2]),
259  q3 = get_q("HeadlightRotationW", rotation[3]);
260 
261  rotation.setValue(q0, q1, q2, q3);
262 
263  setLightDirection(rotation, ui->viewer);
264  setLightDraggerDirection(rotation, lightDragger);
265 
266  setValueSilently(ui->q0_spnBox, rotation[0]);
267  setValueSilently(ui->q1_spnBox, rotation[1]);
