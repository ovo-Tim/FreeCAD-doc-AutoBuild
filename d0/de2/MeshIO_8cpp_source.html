226  Base::FileInfo fi(FileName);
227  if (!fi.exists() || !fi.isFile()) {
228  throw Base::FileException("File does not exist", FileName);
229  }
230  if (!fi.isReadable()) {
231  throw Base::FileException("No permission on the file", FileName);
232  }
233 
234  Base::ifstream str(fi, std::ios::in | std::ios::binary);
235 
236  if (fi.hasExtension("bms")) {
237  _rclMesh.Read(str);
238  return true;
239  }
240 
241  // read file
242  bool ok = false;
243  if (fi.hasExtension({"stl", "ast"})) {
244  ok = LoadSTL(str);
245  }
246  else if (fi.hasExtension("iv")) {
247  ok = LoadInventor(str);
248  if (ok && _rclMesh.CountFacets() == 0) {
249  Base::Console().Warning("No usable mesh found in file '%s'", FileName);
250  }
251  }
252  else if (fi.hasExtension({"nas", "bdf"})) {
253  ok = LoadNastran(str);
254  }
255  else if (fi.hasExtension("obj")) {
256  ok = LoadOBJ(str, FileName);
257  }
258  else if (fi.hasExtension("smf")) {
259  ok = LoadSMF(str);
260  }
261  else if (fi.hasExtension("3mf")) {
262  try {
263  ok = Load3MF(str);
264  }
265  catch (const zipios::FCollException&) {
266  ZipFixer zip(FileName);
267  ok = Load3MF(zip.getStream());
268  }
269  }
270  else if (fi.hasExtension("off")) {
271  ok = LoadOFF(str);
272  }
273  else if (fi.hasExtension("ply")) {
274  ok = LoadPLY(str);
275  }
276  else {
277  throw Base::FileException("File extension not supported", FileName);
278  }
279 
280  return ok;

318 bool MeshInput::LoadSTL(std::istream& input)
319 {
320  char szBuf[200];
321 
322  if (!input || input.bad()) {
323  return false;
324  }
325 
326  // Read in 50 characters from position 80 on and check for keywords like 'SOLID', 'FACET',
327  // 'NORMAL', 'VERTEX', 'ENDFACET' or 'ENDLOOP'. As the file can be binary with one triangle only
328  // we must not read in more than (max.) 54 bytes because the file size has only 134 bytes in
329  // this case. On the other hand we must overread the first 80 bytes because it can happen that
330  // the file is binary but contains one of these keywords.
331  std::streambuf* buf = input.rdbuf();
332  if (!buf) {
333  return false;
334  }
335  buf->pubseekoff(80, std::ios::beg, std::ios::in);
336  uint32_t ulCt {}, ulBytes = 50;
337  input.read((char*)&ulCt, sizeof(ulCt));
338  // if we have a binary STL with a single triangle we can only read-in 50 bytes
339  if (ulCt > 1) {
340  ulBytes = 100;
341  }
342  // Either it's really an invalid STL file or it's just empty. In this case the number of facets
343  // must be 0.
344  if (!input.read(szBuf, ulBytes)) {
345  return (ulCt == 0);
346  }
347  szBuf[ulBytes] = 0;
348  boost::algorithm::to_upper(szBuf);
349 
350  try {
351  if (!strstr(szBuf, "SOLID") && !strstr(szBuf, "FACET") && !strstr(szBuf, "NORMAL")
352  && !strstr(szBuf, "VERTEX") && !strstr(szBuf, "ENDFACET")
353  && !strstr(szBuf, "ENDLOOP")) {
354  // probably binary STL
355  buf->pubseekoff(0, std::ios::beg, std::ios::in);
356  return LoadBinarySTL(input);
357  }
358 
359  // Ascii STL
360  buf->pubseekoff(0, std::ios::beg, std::ios::in);
361  return LoadAsciiSTL(input);
362  }
363  catch (const Base::MemoryException&) {
364  _rclMesh.Clear();
365  throw; // Throw the same instance of Base::MemoryException
366  }
367  catch (const Base::AbortException&) {
368  _rclMesh.Clear();
369  return false;
370  }
371  catch (const Base::Exception&) {
372  _rclMesh.Clear();
373  throw; // Throw the same instance of Base::Exception
374  }
375  catch (...) {
376  _rclMesh.Clear();
377  throw;
378  }
379 
380  return true;
381 }

416 bool MeshInput::LoadSMF(std::istream& input)
417 {
418  boost::regex rx_p("^v\\s+([-+]?[0-9]*)\\.?([0-9]+([eE][-+]?[0-9]+)?)"
419  "\\s+([-+]?[0-9]*)\\.?([0-9]+([eE][-+]?[0-9]+)?)"
420  "\\s+([-+]?[0-9]*)\\.?([0-9]+([eE][-+]?[0-9]+)?)\\s*$");
421  boost::regex rx_f3("^f\\s+([-+]?[0-9]+)"
422  "\\s+([-+]?[0-9]+)"
423  "\\s+([-+]?[0-9]+)\\s*$");
424  boost::cmatch what;
425 
426  unsigned long segment = 0;
427  MeshPointArray meshPoints;
428  MeshFacetArray meshFacets;
429 
430  std::string line;
431  float fX {}, fY {}, fZ {};
432  int i1 = 1, i2 = 1, i3 = 1;
433  MeshFacet item;
434 
435  if (!input || input.bad()) {
436  return false;
437  }
438 
439  std::streambuf* buf = input.rdbuf();
440  if (!buf) {
441  return false;
442  }
443 
444  while (std::getline(input, line)) {
445  if (boost::regex_match(line.c_str(), what, rx_p)) {
446  fX = (float)std::atof(what[1].first);
447  fY = (float)std::atof(what[4].first);
448  fZ = (float)std::atof(what[7].first);
449  meshPoints.push_back(MeshPoint(Base::Vector3f(fX, fY, fZ)));
450  }
451  else if (boost::regex_match(line.c_str(), what, rx_f3)) {
452  // 3-vertex face
453  i1 = std::atoi(what[1].first);
454  i1 = i1 > 0 ? i1 - 1 : i1 + static_cast<int>(meshPoints.size());
455  i2 = std::atoi(what[2].first);
456  i2 = i2 > 0 ? i2 - 1 : i2 + static_cast<int>(meshPoints.size());
457  i3 = std::atoi(what[3].first);
458  i3 = i3 > 0 ? i3 - 1 : i3 + static_cast<int>(meshPoints.size());
459  item.SetVertices(i1, i2, i3);
460  item.SetProperty(segment);
461  meshFacets.push_back(item);
462  }
463  }
464 
465  this->_rclMesh.Clear(); // remove all data before
466 
467  MeshCleanup meshCleanup(meshPoints, meshFacets);
468  meshCleanup.RemoveInvalids();
469  MeshPointFacetAdjacency meshAdj(meshPoints.size(), meshFacets);
470  meshAdj.SetFacetNeighbourhood();
471  this->_rclMesh.Adopt(meshPoints, meshFacets);
472 
473  return true;
474 }

477 bool MeshInput::LoadOFF(std::istream& input)
478 {
479  // http://edutechwiki.unige.ch/en/3D_file_format
480  boost::regex rx_n(R"(^\s*([0-9]+)\s+([0-9]+)\s+([0-9]+)\s*$)");
481  boost::cmatch what;
482 
483  bool colorPerVertex = false;
484  std::vector<App::Color> diffuseColor;
485  MeshPointArray meshPoints;
486  MeshFacetArray meshFacets;
487 
488  std::string line;
489  MeshFacet item;
490 
491  if (!input || input.bad()) {
492  return false;
493  }
494 
495  std::streambuf* buf = input.rdbuf();
496  if (!buf) {
497  return false;
498  }
499 
500  std::getline(input, line);
501  boost::algorithm::to_lower(line);
502  if (line.find("coff") != std::string::npos) {
503  // we expect colors to be there per vertex: x y z r g b a
504  colorPerVertex = true;
505  }
506  else if (line.find("off") == std::string::npos) {
507  return false; // not an OFF file
508  }
509 
510  // get number of vertices and faces
511  int numPoints = 0, numFaces = 0;
512 
513  while (true) {
514  std::getline(input, line);
515  boost::algorithm::to_lower(line);
516  if (boost::regex_match(line.c_str(), what, rx_n)) {
517  numPoints = std::atoi(what[1].first);
518  numFaces = std::atoi(what[2].first);
519  break;
520  }
521  }
522 
523  if (numPoints == 0 || numFaces == 0) {
524  return false;
525  }
526 
527  meshPoints.reserve(numPoints);
528  meshFacets.reserve(numFaces);
529  if (colorPerVertex) {
530  diffuseColor.reserve(numPoints);
531  }
532  else {
533  diffuseColor.reserve(numFaces);
534  }
535 
536  int cntPoints = 0;
537  while (cntPoints < numPoints) {
538  if (!std::getline(input, line)) {
539  break;
540  }
541  std::istringstream str(line);
542  str.unsetf(std::ios_base::skipws);
543  str >> std::ws;
544  if (str.eof()) {
545  continue; // empty line
546  }
547 
548  float fX {}, fY {}, fZ {};
549  str >> fX >> std::ws >> fY >> std::ws >> fZ;
550  if (str) {
551  meshPoints.push_back(MeshPoint(Base::Vector3f(fX, fY, fZ)));
552  cntPoints++;
553 
554  if (colorPerVertex) {
555  std::size_t pos = std::size_t(str.tellg());
556  if (line.size() > pos) {
557  float r {}, g {}, b {}, a {};
558  str >> std::ws >> r >> std::ws >> g >> std::ws >> b;
559  if (str) {
560  str >> std::ws >> a;
561  // no transparency
562  if (!str) {
563  a = 1.0F;
564  }
565 
566  if (r > 1.0F || g > 1.0F || b > 1.0F || a > 1.0F) {
567  r = static_cast<float>(r) / 255.0F;
568  g = static_cast<float>(g) / 255.0F;
569  b = static_cast<float>(b) / 255.0F;
570  a = static_cast<float>(a) / 255.0F;
571  }
572  diffuseColor.emplace_back(r, g, b, a);
573  }
574  }
575  }
576  }
577  }
578 
579  int cntFaces = 0;
580  while (cntFaces < numFaces) {
581  if (!std::getline(input, line)) {
582  break;
583  }
584  std::istringstream str(line);
585  str.unsetf(std::ios_base::skipws);
586  str >> std::ws;
587  if (str.eof()) {
588  continue; // empty line
589  }
590  int count {}, index {};
591  str >> count;
592  if (count >= 3) {
593  std::vector<int> faces;
594  faces.reserve(count);
595 
596  for (int i = 0; i < count; i++) {
597  str >> std::ws;
598  str >> index;
599  faces.push_back(index);
600  }
601 
602  for (int i = 0; i < count - 2; i++) {
603  item.SetVertices(faces[0], faces[i + 1], faces[i + 2]);
604  meshFacets.push_back(item);
605  }
606  cntFaces++;
607 
608  std::size_t pos = std::size_t(str.tellg());
609  if (line.size() > pos) {
610  float r {}, g {}, b {}, a {};
611  str >> std::ws >> r >> std::ws >> g >> std::ws >> b;
612  if (str) {
613  str >> std::ws >> a;
614  // no transparency
615  if (!str) {
616  a = 1.0F;
617  }
618 
619  if (r > 1.0F || g > 1.0F || b > 1.0F || a > 1.0F) {
620  r = static_cast<float>(r) / 255.0F;
621  g = static_cast<float>(g) / 255.0F;
622  b = static_cast<float>(b) / 255.0F;
623  a = static_cast<float>(a) / 255.0F;
624  }
625  for (int i = 0; i < count - 2; i++) {
626  diffuseColor.emplace_back(r, g, b, a);
627  }
628  }
629  }
630  }
631  }
632 
633  if (_material) {
634  if (colorPerVertex) {
635  if (meshPoints.size() == diffuseColor.size()) {
636  _material->binding = MeshIO::PER_VERTEX;
637  _material->diffuseColor.swap(diffuseColor);
638  }
639  }
640  else {
641  if (meshFacets.size() == diffuseColor.size()) {
642  _material->binding = MeshIO::PER_FACE;
643  _material->diffuseColor.swap(diffuseColor);
644  }
645  }
646  }
647  this->_rclMesh.Clear(); // remove all data before
648 
649  MeshCleanup meshCleanup(meshPoints, meshFacets);
650  if (_material) {
651  meshCleanup.SetMaterial(_material);
652  }
653  meshCleanup.RemoveInvalids();
654  MeshPointFacetAdjacency meshAdj(meshPoints.size(), meshFacets);
655  meshAdj.SetFacetNeighbourhood();
656  this->_rclMesh.Adopt(meshPoints, meshFacets);
657 
658  return true;
659 }

691  }
