91 void LoftWidget::findShapes()
92 {
93  App::Document* activeDoc = App::GetApplication().getActiveDocument();
94  Gui::Document* activeGui = Gui::Application::Instance->getDocument(activeDoc);
95  if (!activeGui)
96  return;
97  d->document = activeDoc->getName();
98 
99  std::vector<App::DocumentObject*> objs = activeDoc->getObjectsOfType<App::DocumentObject>();
100 
101  for (auto obj : objs) {
102  Part::TopoShape topoShape = Part::Feature::getTopoShape(obj);
103  if (topoShape.isNull()) {
104  continue;
105  }
106  TopoDS_Shape shape = topoShape.getShape();
107  if (shape.IsNull()) continue;
108 
109  // also allow compounds with a single face, wire or vertex or
110  // if there are only edges building one wire
111  if (shape.ShapeType() == TopAbs_COMPOUND) {
112  Handle(TopTools_HSequenceOfShape) hEdges = new TopTools_HSequenceOfShape();
113  Handle(TopTools_HSequenceOfShape) hWires = new TopTools_HSequenceOfShape();
114 
115  TopoDS_Iterator it(shape);
116  int numChilds=0;
117  TopoDS_Shape child;
118  for (; it.More(); it.Next(), numChilds++) {
119  if (!it.Value().IsNull()) {
120  child = it.Value();
121  if (child.ShapeType() == TopAbs_EDGE) {
122  hEdges->Append(child);
123  }
124  }
125  }
126 
127  // a single child
128  if (numChilds == 1) {
129  shape = child;
130  }
131  // or all children are edges
132  else if (hEdges->Length() == numChilds) {
133  ShapeAnalysis_FreeBounds::ConnectEdgesToWires(hEdges,
134  Precision::Confusion(), Standard_False, hWires);
135  if (hWires->Length() == 1)
136  shape = hWires->Value(1);
137  }
138  }
139 
140  if (!shape.Infinite() &&
141  (shape.ShapeType() == TopAbs_FACE ||
142  shape.ShapeType() == TopAbs_WIRE ||
143  shape.ShapeType() == TopAbs_EDGE ||
144  shape.ShapeType() == TopAbs_VERTEX)) {
145  QString label = QString::fromUtf8(obj->Label.getValue());
146  QString name = QString::fromLatin1(obj->getNameInDocument());
147  QTreeWidgetItem* child = new QTreeWidgetItem();
148  child->setText(0, label);
149  child->setToolTip(0, label);
150  child->setData(0, Qt::UserRole, name);
151  Gui::ViewProvider* vp = activeGui->getViewProvider(obj);
152  if (vp) child->setIcon(0, vp->getIcon());
153  d->ui.selector->availableTreeWidget()->addTopLevelItem(child);
154  }
155  }
156 }

158 bool LoftWidget::accept()
159 {
160  QString list, solid, ruled, closed;
161  if (d->ui.checkSolid->isChecked())
162  solid = QString::fromLatin1("True");
163  else
164  solid = QString::fromLatin1("False");
165 
166  if (d->ui.checkRuledSurface->isChecked())
167  ruled = QString::fromLatin1("True");
168  else
169  ruled = QString::fromLatin1("False");
170 
171  if (d->ui.checkClosed->isChecked())
172  closed = QString::fromLatin1("True");
173  else
174  closed = QString::fromLatin1("False");
175 
176  QTextStream str(&list);
177 
178  int count = d->ui.selector->selectedTreeWidget()->topLevelItemCount();
179  if (count < 2) {
180  QMessageBox::critical(this, tr("Too few elements"), tr("At least two vertices, edges, wires or faces are required."));
181  return false;
182  }
183  for (int i=0; i<count; i++) {
184  QTreeWidgetItem* child = d->ui.selector->selectedTreeWidget()->topLevelItem(i);
185  QString name = child->data(0, Qt::UserRole).toString();
186  str << "App.getDocument('" << d->document.c_str() << "')." << name << ", ";
187  }
188 
189  try {
190  QString cmd;
191  cmd = QString::fromLatin1(
192  "App.getDocument('%5').addObject('Part::Loft','Loft')\n"
193  "App.getDocument('%5').ActiveObject.Sections=[%1]\n"
194  "App.getDocument('%5').ActiveObject.Solid=%2\n"
195  "App.getDocument('%5').ActiveObject.Ruled=%3\n"
196  "App.getDocument('%5').ActiveObject.Closed=%4\n"
197  ).arg(list, solid, ruled, closed, QString::fromLatin1(d->document.c_str()));
198 
199  Gui::Document* doc = Gui::Application::Instance->getDocument(d->document.c_str());
200  if (!doc)
201  throw Base::RuntimeError("Document doesn't exist anymore");
202  doc->openCommand(QT_TRANSLATE_NOOP("Command", "Loft"));
203  Gui::Command::runCommand(Gui::Command::App, cmd.toLatin1());
204  doc->getDocument()->recompute();
205  App::DocumentObject* obj = doc->getDocument()->getActiveObject();
206  if (obj && !obj->isValid()) {
207  std::string msg = obj->getStatusString();
208  doc->abortCommand();
209  throw Base::RuntimeError(msg);
210  }
211  doc->commitCommand();
212  }
213  catch (const Base::Exception& e) {
214  QMessageBox::warning(this, tr("Input error"), QCoreApplication::translate("Exception", e.what()));
215  return false;
216  }
217 
218  return true;
219 }
