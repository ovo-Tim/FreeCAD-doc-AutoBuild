39 def calculate_normals_no_smoothing(pointarray,facearray,smarray=None):
40  """Calculate normals all perpendicular to the faces.
41 
42  points,norms = calculate_normals_no_smoothing(
43  pointarray,facearray,smarray=None)
44 
45  Takes an array of points and faces in exactly the same form in
46  which they appear in the 3DS DOM. It accepts a smoothing array,
47  but ignores it.
48 
49  Returns a numpy.array of points, one per row, and a
50  numpy.array of the corresponding normals. The points are
51  returned as a list of consecutive triangles; the first three rows
52  make up the first triangle, the second three rows make up the
53  second triangle, and so on.
54 
55  The normal vectors are determined by calculating the normal to
56  each face. There is no smoothing.
57 
58  """
59 
60  # prepare to calculate normals. define some arrays
61 
62  m = len(facearray)
63  fnorms = numpy.empty((m*3,3),_calc_precision_type)
64  points = pointarray[facearray.ravel()]
65 
66  # calculate normals for each face
67 
68  A = numpy.asarray(pointarray[facearray[:,0]],_calc_precision_type)
69  B = numpy.asarray(pointarray[facearray[:,1]],_calc_precision_type)
70  C = numpy.asarray(pointarray[facearray[:,2]],_calc_precision_type)
71  b = A - C
72  c = B - A
73  fnorms[2::3,0] = c[:,2]*b[:,1]-c[:,1]*b[:,2]
74  fnorms[2::3,1] = c[:,0]*b[:,2]-c[:,2]*b[:,0]
75  fnorms[2::3,2] = c[:,1]*b[:,0]-c[:,0]*b[:,1]
76  a = fnorms[2::3]
77  q = numpy.maximum(numpy.sqrt(numpy.sum(a*a,axis=1)),1e-10)
78  q = q[:,numpy.newaxis]
79  a /= q
80  fnorms[0::3] = fnorms[1::3] = fnorms[2::3]
81 
82  # we're done
83 
84  return points, numpy.asarray(fnorms,numpy.float32)

171 def calculate_normals_by_angle_subtended(pointarray,facearray,smarray):
172  """Calculate normals by smoothing, weighting by angle subtended.
173 
174  points,norms = calculate_normals_by_angle_subtended(
175  pointarray,facearray,smarray)
176 
177  Takes an array of points, faces, and a smoothing group in exactly
178  the same form in which they appear in the 3DS DOM.
179 
180  Returns a numpy.array of points, one per row, and a numpy.array of
181  the corresponding normals. The points are returned as a list of
182  consecutive triangles; the first three rows make up the first
183  triangle, the second three rows make up the second triangle, and
184  so on.
185 
186  To calculate the normal of a given vertex on a given face, this
187  function averages the normal vector for all faces which have share
188  that vertex, and a smoothing group.
189 
190  The normals being averaged are weighted by the angle subtended.
191 
192  """
193 
194  # prepare to calculate normals. define some arrays
195 
196  m = len(facearray)
197  rnorms = numpy.zeros((m*3,3),_calc_precision_type)
198  fnorms = numpy.zeros((m*3,3),_calc_precision_type)
199  points = pointarray[facearray.ravel()]
200  exarray = numpy.zeros(3*m,numpy.uint32)
201  if smarray is not None:
202  exarray[0::3] = exarray[1::3] = exarray[2::3] = smarray
203 
204  # weed out degenerate triangles first
205  # unlike cross-product, angle subtended blows up on degeneracy
206 
207  A = numpy.asarray(pointarray[facearray[:,0]],_calc_precision_type)
208  B = numpy.asarray(pointarray[facearray[:,1]],_calc_precision_type)
209  C = numpy.asarray(pointarray[facearray[:,2]],_calc_precision_type)
210  a = C - B
211  b = A - C
212  c = B - A
213  p = numpy.zeros((len(facearray),3),_calc_precision_type)
214  p[:,0] = c[:,2]*b[:,1]-c[:,1]*b[:,2]
215  p[:,1] = c[:,0]*b[:,2]-c[:,2]*b[:,0]
216  p[:,2] = c[:,1]*b[:,0]-c[:,0]*b[:,1]
217  aa = numpy.sum(a*a,axis=1)
218  bb = numpy.sum(b*b,axis=1)
219  cc = numpy.sum(c*c,axis=1)
220  pp = numpy.sum(p*p,axis=1)
221  ndg = numpy.nonzero(numpy.logical_and.reduce((aa,bb,cc,pp)))[0]
222 
223  # calculate scaled normals (according to angle subtended)
224 
225  p = p[ndg]
226  la = numpy.sqrt(aa[ndg])
227  lb = numpy.sqrt(bb[ndg])
228  lc = numpy.sqrt(cc[ndg])
229  lp = numpy.sqrt(pp[ndg])
230  sinA = numpy.clip(lp/lb/lc,-1.0,1.0)
231  sinB = numpy.clip(lp/la/lc,-1.0,1.0)
232  sinC = numpy.clip(lp/la/lb,-1.0,1.0)
233  sinA2 = sinA*sinA
234  sinB2 = sinB*sinB
235  sinC2 = sinC*sinC
236  angA = numpy.arcsin(sinA)
237  angB = numpy.arcsin(sinB)
238  angC = numpy.arcsin(sinC)
239  angA = numpy.where(sinA2 > sinB2 + sinC2, numpy.pi - angA, angA)
240  angB = numpy.where(sinB2 > sinA2 + sinC2, numpy.pi - angB, angB)
241  angC = numpy.where(sinC2 > sinA2 + sinB2, numpy.pi - angC, angC)
242  rnorms[0::3][ndg] = p*(angA/lp)[:,numpy.newaxis]
243  rnorms[1::3][ndg] = p*(angB/lp)[:,numpy.newaxis]
244  rnorms[2::3][ndg] = p*(angC/lp)[:,numpy.newaxis]
