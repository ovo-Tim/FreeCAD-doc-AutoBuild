48 /* ----------------------------------------------------------------------------
49  Handler for driver events. This function is able to handle the events in
50  different ways: (1) re-package the events as Carbon events, (2) compute
51  them directly, (3) write the event info in a shared memory location for
52  usage by reader threads.
53 */
54 void
55 Gui::GuiNativeEvent::tdx_drv_handler(io_connect_t connection,
56  natural_t messageType,
57  void *messageArgument)
58 {
59  static bool firstTime = true;
60  if (firstTime) {
61  firstTime = false;
62  Base::Console().Log("Call connexion handler\n");
63  }
64  Q_UNUSED(connection)
65  //printf("tdx_drv_handler\n");
66  //printf("connection: %X\n", connection);
67  //printf("messageType %c%c%c%c\n", messageType/0x1000000, messageType/0x10000, messageType/0x100, messageType);
68  auto msg = (ConnexionDeviceStatePtr)messageArgument;
69 
70  switch(messageType) {
71  case kConnexionMsgDeviceState:
72  /* Device state messages are broadcast to all clients. It is up to
73  * the client to figure out if the message is meant for them. This
74  * is done by comparing the "client" id sent in the message to our
75  * assigned id when the connection to the driver was established.
76  */
77  //printf("msg->client: %d, tdxClientID: %d\n", msg->client, tdxClientID);
78  Base::Console().Log("msg->client: %d, msg->command: %d\n", msg->client, msg->command);
79 
80  if (msg->client == tdxClientID) {
81  switch (msg->command) {
82  case kConnexionCmdHandleAxis:
83  {
84  motionDataArray[0] = -msg->axis[0];
85  motionDataArray[1] = msg->axis[1];
86  motionDataArray[2] = msg->axis[2];
87  motionDataArray[3] = -msg->axis[3];
88  motionDataArray[4] = msg->axis[4];
89  motionDataArray[5] = msg->axis[5];
90  mainApp->postMotionEvent(motionDataArray);
91  break;
92  }
93 
94  case kConnexionCmdHandleButtons:
95  {
96  //printf("value: %d\n", msg->value);
97  //printf("buttons: %u\n", msg->buttons);
98  uint32_t changedButtons = msg->buttons ^ lastButtons;
99  uint32_t pressedButtons = msg->buttons & changedButtons;
100  uint32_t releasedButtons = lastButtons & changedButtons;
101  for (uint8_t bt = 0; bt < 32; bt++) {
102  if (pressedButtons & 1)
103  mainApp->postButtonEvent(bt, 1);
104  pressedButtons = pressedButtons >> 1;
105  }
106  for (uint8_t bt = 0; bt < 32; bt++) {
107  if (releasedButtons & 1)
108  mainApp->postButtonEvent(bt, 0);
109  releasedButtons = releasedButtons >> 1;
110  }
111 
112  lastButtons = msg->buttons;
113  break;
114  }
115 
116  default:
117  break;
118 
119  } /* switch */
120  }
121  break;
122 
123  default:
124  /* other messageTypes can happen and should be ignored */
125  break;
126  }
127 }

146 void Gui::GuiNativeEvent::initSpaceball(QMainWindow *window)
147 {
148  Q_UNUSED(window)
149  OSStatus err;
150  /* make sure the framework is installed */
151  if (SetConnexionHandlers == NULL)
152  {
153  Base::Console().Log("3Dconnexion framework not found!\n");
154  return;
155  }
156  /* install 3dx message handler in order to receive driver events */
157  err = SetConnexionHandlers(tdx_drv_handler, 0L, 0L, false);
158  assert(err == 0);
159  if (err)
160  {
161  Base::Console().Log("Error installing 3Dconnexion handler\n");
162  return;
163  }
164  /* register our app with the driver */
165  // Pascal string Application name required to register driver for application
166  UInt8 tdxAppName[] = {7,'F','r','e','e','C','A','D'};
167  tdxClientID = RegisterConnexionClient( kConnexionClientWildcard, tdxAppName,
168  kConnexionClientModeTakeOver,
169  kConnexionMaskAll );
170  if (tdxClientID == 0)
171  {
172  Base::Console().Log("Couldn't connect to 3Dconnexion driver\n");
173  return;
174  }
175 
176  // Turn on all features and buttons
177  SetConnexionClientMask(tdxClientID, kConnexionMaskAll);
178  SetConnexionClientButtonMask(tdxClientID, kConnexionMaskAllButtons);
179 
180  Base::Console().Log("3Dconnexion driver initialized. Client ID: %d\n", tdxClientID);
181  mainApp->setSpaceballPresent(true);
182 }
