217 TStringMap Translator::supportedLocales() const
218 {
219  if (!d->mapSupportedLocales.empty())
220  return d->mapSupportedLocales;
221 
222  // List all .qm files
223  for (const auto& domainMap : d->mapLanguageTopLevelDomain) {
224  for (const auto& directoryName : std::as_const(d->paths)) {
225  QDir dir(directoryName);
226  QString filter = QString::fromLatin1("*_%1.qm").arg(QString::fromStdString(domainMap.second));
227  QStringList fileNames = dir.entryList(QStringList(filter), QDir::Files, QDir::Name);
228  if (!fileNames.isEmpty()) {
229  d->mapSupportedLocales[domainMap.first] = domainMap.second;
230  break;
231  }
232  }
233  }
234 
235  return d->mapSupportedLocales;

310 void Translator::installQMFiles(const QDir& dir, const char* locale)
311 {
312  QString filter = QString::fromLatin1("*_%1.qm").arg(QLatin1String(locale));
313  QStringList fileNames = dir.entryList(QStringList(filter), QDir::Files, QDir::Name);
314  for (const auto &it : fileNames){
315  bool ok=false;
316  for (std::list<QTranslator*>::const_iterator tt = d->translators.begin();
317  tt != d->translators.end(); ++tt) {
318  if ((*tt)->objectName() == it) {
319  ok = true; // this file is already installed
320  break;
321  }
322  }
323 
324  // okay, we need to install this file
325  if (!ok) {
326  auto translator = new QTranslator;
327  translator->setObjectName(it);
328  if (translator->load(dir.filePath(it))) {
329  qApp->installTranslator(translator);
330  d->translators.push_back(translator);
331  }
332  else {
333  delete translator;
334  }
335  }
336  }
337 }

368 bool Translator::eventFilter(QObject* obj, QEvent* ev)
369 {
370  if (ev->type() == QEvent::KeyPress || ev->type() == QEvent::KeyRelease) {
371  QKeyEvent *kev = static_cast<QKeyEvent *>(ev);
372  Qt::KeyboardModifiers mod = kev->modifiers();
373  int key = kev->key();
374  if ((mod & Qt::KeypadModifier) && (key == Qt::Key_Period || key == Qt::Key_Comma)) {
375  if (ev->spontaneous()) {
376  auto dp = QString(QLocale().decimalPoint());
377 #if QT_VERSION < QT_VERSION_CHECK(6,0,0)
378  int dpcode = QKeySequence(dp)[0];
379 #else
380  int dpcode = QKeySequence(dp)[0].key();
381 #endif
382  if (kev->text() != dp) {
383  QKeyEvent modifiedKeyEvent(kev->type(), dpcode, mod, dp, kev->isAutoRepeat(), kev->count());
384  qApp->sendEvent(obj, &modifiedKeyEvent);
385  return true;
386  }
387  }
388  if (dynamic_cast<Gui::TextEdit*>(obj) && key != Qt::Key_Period) {
389  QKeyEvent modifiedKeyEvent(kev->type(), Qt::Key_Period, mod, QChar::fromLatin1('.'), kev->isAutoRepeat(), kev->count());
390  qApp->sendEvent(obj, &modifiedKeyEvent);
391  return true;
392  }
393  }
394  }
395  return false;
396 }

398 void Translator::enableDecimalPointConversion(bool on)
399 {
400  if (!qApp) {
401  return;

403 
404  if (!on) {
