64 class FeatureBooleanFragments:
65  """The BooleanFragments feature object."""
66 
67  def __init__(self,obj):
68  obj.addProperty("App::PropertyLinkList","Objects","BooleanFragments","Object to compute intersections between.")
69  obj.addProperty("App::PropertyEnumeration","Mode","BooleanFragments",
70  "- Standard: wires, shells, compsolids remain in one piece.\n"
71  "- Split: wires, shells, compsolids are split.\n"
72  "- CompSolid: make compsolid from solid fragments.")
73  obj.Mode = ["Standard", "Split", "CompSolid"]
74  obj.addProperty("App::PropertyLength","Tolerance","BooleanFragments",
75  "Tolerance when intersecting (fuzzy value). "
76  "In addition to tolerances of the shapes.")
77 
78  obj.Proxy = self
79  self.Type = "FeatureBooleanFragments"
80 
81  def execute(self,selfobj):
82  shapes = [obj.Shape for obj in selfobj.Objects]
83  if len(shapes) == 1 and shapes[0].ShapeType == "Compound":
84  shapes = shapes[0].childShapes()
85  if len(shapes) < 2:
86  raise ValueError("At least two shapes are needed for computing boolean fragments. Got only {num}.".format(num=len(shapes)))
87  selfobj.Shape = SplitAPI.booleanFragments(shapes, selfobj.Mode, selfobj.Tolerance)
