48 void QGIViewSection::drawSectionFace()
49 {
50  // Base::Console().Message("QGIVS::drawSectionFace()\n");
51  auto section( dynamic_cast<TechDraw::DrawViewSection *>(getViewObject()) );
52  if (!section) {
53  return;
54  }
55 
56  if (!section->hasGeometry()) {
57  return;
58  }
59 
60  ViewProviderViewSection* sectionVp = dynamic_cast<ViewProviderViewSection*>(QGIView::getViewProvider(section));
61  if (!sectionVp) {
62  return;
63  }
64 
65  auto sectionFaces( section->getTDFaceGeometry() );
66  if (sectionFaces.empty()) {
67  return;
68  }
69 
70  float lineWidth = sectionVp->LineWidth.getValue();
71 
72  std::vector<TechDraw::FacePtr>::iterator fit = sectionFaces.begin();
73  int i = 0;
74  for(; fit != sectionFaces.end(); fit++, i++) {
75  QGIFace* newFace = drawFace(*fit, -1);
76  newFace->setZValue(ZVALUE::SECTIONFACE);
77  if (section->showSectionEdges()) {
78  newFace->setDrawEdges(true);
79  newFace->setStyle(Qt::SolidLine);
80  newFace->setWidth(lineWidth);
81  } else {
82  newFace->setDrawEdges(false);
83  }
84 
85  if (section->CutSurfaceDisplay.isValue("Hide")) {
86  return;
87  }
88 
89  if (section->CutSurfaceDisplay.isValue("Color")) {
90  newFace->isHatched(true);
91  QColor faceColor = (sectionVp->CutSurfaceColor.getValue()).asValue<QColor>();
92  faceColor.setAlpha((100 - sectionVp->CutSurfaceTransparency.getValue())*255/100);
93  newFace->setFillColor(faceColor);
94  newFace->setFillMode(faceColor.alpha() ? QGIFace::PlainFill : QGIFace::NoFill);
95  } else if (section->CutSurfaceDisplay.isValue("SvgHatch")) {
96  newFace->isHatched(true);
97  newFace->setFillMode(QGIFace::SvgFill);
98  newFace->setHatchColor(sectionVp->HatchColor.getValue());
99  newFace->setHatchScale(section->HatchScale.getValue());
100  newFace->setHatchRotation(section->HatchRotation.getValue());
101  newFace->setHatchOffset(section->HatchOffset.getValue());
102  std::string hatchSpec = section->SvgIncluded.getValue();
103  newFace->setHatchFile(hatchSpec);
104  } else if (section->CutSurfaceDisplay.isValue("PatHatch")) {
105  newFace->isHatched(true);
106  newFace->setFillMode(QGIFace::GeomHatchFill);
107  newFace->setHatchColor(sectionVp->GeomHatchColor.getValue());
108  newFace->setHatchScale(section->HatchScale.getValue());
109  newFace->setHatchRotation(section->HatchRotation.getValue());
110  newFace->setHatchOffset(section->HatchOffset.getValue());
111  newFace->setLineWeight(sectionVp->WeightPattern.getValue());
112  std::vector<TechDraw::LineSet> lineSets = section->getDrawableLines(i);
113  if (!lineSets.empty()) {
114  newFace->clearLineSets();
115  for (auto& ls: lineSets) {
116  newFace->addLineSet(ls);
117  }
118  }
119  } else {
120  Base::Console().Warning("QGIVS::draw - unknown CutSurfaceDisplay: %d\n",
121  section->CutSurfaceDisplay.getValue());
122  }
123 
124  newFace->draw();
125  newFace->setPrettyNormal();
126  newFace->setAcceptHoverEvents(false);
127  newFace->setFlag(QGraphicsItem::ItemIsSelectable, false);
128  }
129 }
