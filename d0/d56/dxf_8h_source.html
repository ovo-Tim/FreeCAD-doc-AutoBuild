757  // There is some uncertainty in the documentation. If the line type is omitted it should
758  // act as by-layer.
759  // The question is whether an explicit line type "BYLAYER" should also mean by-layer. If
760  // this is the case we should set m_LineType to "BYLAYER" and only check for "BYLAYER"
761  // in ResolveByLayerAttributes. If this is not the case and "BYLAYER" can actually refer
762  // to a line type of that name, then we should init m_LineType to empty and check only
763  // for a zero-length line type to trigger by-layer line type. An experiment in R13
764  // reveals that it is not possible to make a line type BYLAYER so the former is what we
765  // do.
766  m_Color = ColorBylayer;
767  m_LineType = LineTypeByLayer;
768  m_paperSpace = false;

771  {
772  // This should be called after the entire entity is read in case the layer reference
773  // name comes after a BYLAYER color. Also we can't be sure of BYLAYER line type until
774  // the entity ends
775  if (m_Color == ColorBylayer) {
776  m_Color = m_Layer != nullptr ? m_Layer->Color : DefaultColor;
777  }
778  if (m_LineType == LineTypeByLayer) {
779  m_LineType = m_Layer != nullptr ? m_Layer->LineType : DefaultLineType;
780  }
781  // We this point we not longer have any use for m_Layer except as a destination for the
782  // entity. If the import is not preserving layers, we clear out that field here so all
783  // entities get placed at the drawing root and are subject to compounding despite being
784  // originally in different layers.
785  if (!reader.m_preserveColors) {

787  }
