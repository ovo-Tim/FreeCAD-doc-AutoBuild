69  void mouseMove(Base::Vector2d onSketchPos) override
70  {
71  if (Mode == STATUS_SEEK_First) {
72  setPositionText(onSketchPos);
73  seekAndRenderAutoConstraint(sugConstr1, onSketchPos, Base::Vector2d(0.f, 0.f));
74  }
75  else if (Mode == STATUS_SEEK_Second) {
76  EditCurve[1] = onSketchPos;
77 
78  // Display radius for user
79  float radius = (onSketchPos - focusPoint).Length();
80  if (showCursorCoords()) {
81  SbString text;
82  std::string radiusString = lengthToDisplayFormat(radius, 1);
83  text.sprintf(" (F%s)", radiusString.c_str());
84  setPositionText(onSketchPos, text);
85  }
86 
87  drawEdit(EditCurve);
88  seekAndRenderAutoConstraint(sugConstr2, onSketchPos, Base::Vector2d(0.f, 0.f));
89  }
90  else if (Mode == STATUS_SEEK_Third) {
91  double focal = (axisPoint - focusPoint).Length();
92  double phi = atan2(focusPoint.y - axisPoint.y, focusPoint.x - axisPoint.x);
93 
94  // P(U) = O + U*U/(4.*F)*XDir + U*YDir
95  //
96  // pnt = Base::Vector3d(pnt0.x + angle * angle / 4 / focal * cos(phi) - angle *
97  // sin(phi),
98  // pnt0.y + angle * angle / 4 / focal * sin(phi) + angle *
99  // cos(phi), 0.f);
100 
101  // This is the angle at cursor point
102  double u = (cos(phi) * (onSketchPos.y - axisPoint.y)
103  - (onSketchPos.x - axisPoint.x) * sin(phi));
104 
105  for (int i = 15; i >= -15; i--) {
106  double angle = i * u / 15;
107  double rx = angle * angle / 4 / focal * cos(phi) - angle * sin(phi);
108  double ry = angle * angle / 4 / focal * sin(phi) + angle * cos(phi);
109  EditCurve[15 + i] = Base::Vector2d(axisPoint.x + rx, axisPoint.y + ry);
110  }
111 
112  // Display radius for user
113  if (showCursorCoords()) {
114  SbString text;
115  std::string focalString = lengthToDisplayFormat(focal, 1);
116  text.sprintf(" (F%s)", focalString.c_str());
117  setPositionText(onSketchPos, text);
118  }
119 
120  drawEdit(EditCurve);
121 
122  seekAndRenderAutoConstraint(sugConstr3, onSketchPos, Base::Vector2d(0.f, 0.f));
123  }
124  else if (Mode == STATUS_SEEK_Fourth) {
125  double focal = (axisPoint - focusPoint).Length();
126  double phi = atan2(focusPoint.y - axisPoint.y, focusPoint.x - axisPoint.x);
127 
128  // P(U) = O + U*U/(4.*F)*XDir + U*YDir
129  //
130  // pnt = Base::Vector3d(pnt0.x + angle * angle / 4 / focal * cos(phi) - angle *
131  // sin(phi),
132  // pnt0.y + angle * angle / 4 / focal * sin(phi) + angle *
133  // cos(phi), 0.f);
134 
135  // This is the angle at starting point
136  double ustartpoint = (cos(phi) * (startingPoint.y - axisPoint.y)
137  - (startingPoint.x - axisPoint.x) * sin(phi));
138 
139  double startValue = ustartpoint;
140 
141  double u = (cos(phi) * (onSketchPos.y - axisPoint.y)
142  - (onSketchPos.x - axisPoint.x) * sin(phi));
143 
144 
145  arcAngle = u - startValue;
146 
147  if (!boost::math::isnan(arcAngle)) {
148  EditCurve.resize(33);
149  for (std::size_t i = 0; i < 33; i++) {
150  double angle = startValue + i * arcAngle / 32.0;
151  double rx = angle * angle / 4 / focal * cos(phi) - angle * sin(phi);
152  double ry = angle * angle / 4 / focal * sin(phi) + angle * cos(phi);
153  EditCurve[i] = Base::Vector2d(axisPoint.x + rx, axisPoint.y + ry);
154  }
155 
156  if (showCursorCoords()) {
157  SbString text;
158  std::string focalString = lengthToDisplayFormat(focal, 1);
159  text.sprintf(" (F%s)", focalString.c_str());
160  setPositionText(onSketchPos, text);
161  }
162  }
163  else {
164  arcAngle = 0.;
165  }
166 
167  drawEdit(EditCurve);
168  seekAndRenderAutoConstraint(sugConstr4, onSketchPos, Base::Vector2d(0.f, 0.f));
169  }
170  }
171 
172  bool pressButton(Base::Vector2d onSketchPos) override
173  {
174  if (Mode == STATUS_SEEK_First) {
175  EditCurve[0] = onSketchPos;
176  focusPoint = onSketchPos;
177  EditCurve.resize(2);
178  Mode = STATUS_SEEK_Second;
179  }
180  else if (Mode == STATUS_SEEK_Second) {
181  EditCurve[1] = onSketchPos;
182  axisPoint = onSketchPos;
183  EditCurve.resize(31);
184  Mode = STATUS_SEEK_Third;
185  }
186  else if (Mode == STATUS_SEEK_Third) {
187  startingPoint = onSketchPos;
188  arcAngle = 0.;
189  arcAngle_t = 0.;
190  Mode = STATUS_SEEK_Fourth;
191  }
192  else { // Fourth
193  endPoint = onSketchPos;
194  Mode = STATUS_Close;
195  }
196  return true;
197  }
198 
199  bool releaseButton(Base::Vector2d /*onSketchPos*/) override
200  {
201  if (Mode == STATUS_Close) {
202  unsetCursor();
203  resetPositionText();
204 
205  double phi = atan2(focusPoint.y - axisPoint.y, focusPoint.x - axisPoint.x);
206 
207  double ustartpoint = (cos(phi) * (startingPoint.y - axisPoint.y)
208  - (startingPoint.x - axisPoint.x) * sin(phi));
209 
210  double startAngle = ustartpoint;
211 
212  double endAngle;
213 
214  bool isOriginalArcCCW = true;
215 
216  if (arcAngle > 0) {
217  endAngle = startAngle + arcAngle;
218  }
219  else {
220  endAngle = startAngle;
221  startAngle += arcAngle;
222  isOriginalArcCCW = false;
223  }
224 
225  int currentgeoid = getHighestCurveIndex();
226 
227  try {
228  Gui::Command::openCommand(
229  QT_TRANSLATE_NOOP("Command", "Add sketch arc of Parabola"));
230 
231  // Add arc of parabola
232  Gui::cmdAppObjectArgs(sketchgui->getObject(),
233  "addGeometry(Part.ArcOfParabola"
234  "(Part.Parabola(App.Vector(%f,%f,0),App.Vector(%f,%f,0),App."
235  "Vector(0,0,1)),%f,%f),%s)",
236  focusPoint.x,
237  focusPoint.y,
238  axisPoint.x,
239  axisPoint.y,
240  startAngle,
241  endAngle,
242  constructionModeAsBooleanText());
243 
244  currentgeoid++;
245 
246  Gui::cmdAppObjectArgs(sketchgui->getObject(),
247  "exposeInternalGeometry(%d)",
248  currentgeoid);
249  }
250  catch (const Base::Exception&) {
251  Gui::NotifyError(
252  sketchgui,
253  QT_TRANSLATE_NOOP("Notifications", "Error"),
254  QT_TRANSLATE_NOOP("Notifications", "Cannot create arc of parabola"));
255  Gui::Command::abortCommand();
256 
257  tryAutoRecomputeIfNotSolve(sketchgui->getObject<Sketcher::SketchObject>());
258 
259  return false;
260  }
261 
262  Gui::Command::commitCommand();
263 
264  // add auto constraints for the focus point
265  if (!sugConstr1.empty()) {
266  createAutoConstraints(sugConstr1, currentgeoid + 1, Sketcher::PointPos::start);
267  sugConstr1.clear();
268  }
269 
270  // add suggested constraints for vertex point
271  if (!sugConstr2.empty()) {
272  createAutoConstraints(sugConstr2, currentgeoid, Sketcher::PointPos::mid);
273  sugConstr2.clear();
274  }
275 
276  // add suggested constraints for start of arc
277  if (!sugConstr3.empty()) {
278  createAutoConstraints(sugConstr3,
279  currentgeoid,
280  isOriginalArcCCW ? Sketcher::PointPos::start
281  : Sketcher::PointPos::end);
282  sugConstr3.clear();
283  }
284 
285  // add suggested constraints for start of arc
286  if (!sugConstr4.empty()) {
287  createAutoConstraints(sugConstr4,
288  currentgeoid,
289  isOriginalArcCCW ? Sketcher::PointPos::end
290  : Sketcher::PointPos::start);
291  sugConstr4.clear();
292  }
293 
294  tryAutoRecomputeIfNotSolve(sketchgui->getObject<Sketcher::SketchObject>());
295 
296  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
297  "User parameter:BaseApp/Preferences/Mod/Sketcher");
298  bool continuousMode = hGrp->GetBool("ContinuousCreationMode", true);
299  if (continuousMode) {
300  // This code enables the continuous creation mode.
301  Mode = STATUS_SEEK_First;
302  EditCurve.clear();
303  drawEdit(EditCurve);
304  EditCurve.resize(34);
305  applyCursor();
306  /* It is ok not to call to purgeHandler
307  * in continuous creation mode because the
308  * handler is destroyed by the quit() method on pressing the
309  * right button of the mouse */
310  }
311  else {
312  sketchgui->purgeHandler(); // no code after this line, Handler get deleted in
313  // ViewProvider
314  }
315  }
316  return true;
317  }
