68  Py::Object open(const Py::Tuple& args)
69  {

71  if (!PyArg_ParseTuple(args.ptr(), "et", "utf-8", &Name)) {
72  throw Py::Exception();
73  }
74  std::string EncodedName = std::string(Name);
75  PyMem_Free(Name);
76  Base::FileInfo fi(EncodedName);
77  if (!fi.exists()) {
78  throw Py::RuntimeError("File not found");
79  }
80 
81  Gui::WaitCursor wc;
82  wc.restoreCursor();
83 
84  try {
85  std::string path = App::Application::getHomePath();
86  path += "Mod/CAM/Path/Post/scripts/";
87  QDir dir1(QString::fromUtf8(path.c_str()), QString::fromLatin1("*_pre.py"));
88  std::string cMacroPath =
89  App::GetApplication()
90  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/Macro")
91  ->GetASCII("MacroPath", App::Application::getUserMacroDir().c_str());
92  QDir dir2(QString::fromUtf8(cMacroPath.c_str()), QString::fromLatin1("*_pre.py"));
93  QFileInfoList list = dir1.entryInfoList();
94  list << dir2.entryInfoList();
95  std::vector<std::string> scripts;
96  for (int i = 0; i < list.size(); ++i) {
97  QFileInfo fileInfo = list.at(i);
98  scripts.push_back(fileInfo.baseName().toStdString());
99  }
100  std::string processor;
101  PathGui::DlgProcessorChooser Dlg(scripts);
102  if (Dlg.exec() != QDialog::Accepted) {
103  return Py::None();
104  }
105  processor = Dlg.getProcessor();
106 
107  std::ostringstream pre;
108  std::ostringstream cmd;
109  if (processor.empty()) {
110  App::Document* pcDoc = App::GetApplication().newDocument();
111  Gui::Command::runCommand(Gui::Command::Gui, "import Path");
112  cmd << "Path.read(\"" << EncodedName << "\",\"" << pcDoc->getName() << "\")";
113  Gui::Command::runCommand(Gui::Command::Gui, cmd.str().c_str());
114  }
115  else {
116  for (int i = 0; i < list.size(); ++i) {
117  QFileInfo fileInfo = list.at(i);
118  if (fileInfo.baseName().toStdString() == processor) {
119  if (fileInfo.absoluteFilePath().contains(QString::fromLatin1("scripts"))) {
120  pre << "from Path.Post.scripts import " << processor;
121  }
122  else {
123  pre << "import " << processor;
124  }
125  Gui::Command::runCommand(Gui::Command::Gui, pre.str().c_str());
126  cmd << processor << ".open(\"" << EncodedName << "\")";
127  Gui::Command::runCommand(Gui::Command::Gui, cmd.str().c_str());
128  }
129  }
130  }
131  }
132  catch (const Base::Exception& e) {
133  throw Py::RuntimeError(e.what());
