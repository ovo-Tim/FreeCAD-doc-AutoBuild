314 void TaskDetail::onScaleTypeEdit()
315 {
316  TechDraw::DrawViewDetail* detailFeat = getDetailFeat();
317 
318  if (ui->cbScaleType->currentIndex() == 0) {
319  // page scale
320  ui->qsbScale->setEnabled(false);
321  detailFeat->ScaleType.setValue(0.0);
322  // set the page scale if there is a valid page
323  if (m_basePage) {
324  // set the page scale
325  detailFeat->Scale.setValue(m_basePage->Scale.getValue());
326  ui->qsbScale->setValue(m_basePage->Scale.getValue());
327  }
328  // finally update the view
329  updateDetail();
330  }
331  else if (ui->cbScaleType->currentIndex() == 1) {
332  // automatic scale (if view is too large to fit into page, it will be scaled down)
333  ui->qsbScale->setEnabled(false);
334  detailFeat->ScaleType.setValue(1.0);
335  // updating the feature will trigger the rescaling
336  updateDetail();
337  }
338  else if (ui->cbScaleType->currentIndex() == 2) {
339  // custom scale
340  ui->qsbScale->setEnabled(true);
341  detailFeat->ScaleType.setValue(2.0);
342  // no updateDetail() necessary since nothing visibly was changed
343  }
344 }

382 //dragEnd is in scene coords.
383 void TaskDetail::onHighlightMoved(QPointF dragEnd)
384 {
385 // Base::Console().Message("TD::onHighlightMoved(%s) - highlight: %X\n",
386 // DrawUtil::formatVector(dragEnd).c_str(), m_ghost);
387  ui->pbDragger->setEnabled(true);
388 
389  double scale = getBaseFeat()->getScale();
390  double x = Rez::guiX(getBaseFeat()->X.getValue());
391  double y = Rez::guiX(getBaseFeat()->Y.getValue());
392 
393  DrawViewPart* dvp = getBaseFeat();
394  DrawProjGroupItem* dpgi = dynamic_cast<DrawProjGroupItem*>(dvp);
395  if (dpgi) {
396  DrawProjGroup* dpg = dpgi->getPGroup();
397  if (!dpg) {
398  Base::Console().Message("TD::getAnchorScene - projection group is confused\n");
399  //TODO::throw something.
400  return;
401  }
402  x += Rez::guiX(dpg->X.getValue());
403  y += Rez::guiX(dpg->Y.getValue());
404  }
405 
406  QPointF basePosScene(x, -y); //base position in scene coords
407  QPointF anchorDisplace = dragEnd - basePosScene;
408  QPointF newAnchorPos = Rez::appX(anchorDisplace / scale);
409 
410  updateUi(newAnchorPos);
411  updateDetail();
412  enableInputFields(true);
413  m_ghost->setSelected(false);
414  m_ghost->hide();
415 }

503 //get the current Anchor highlight position in scene coords
504 QPointF TaskDetail::getAnchorScene()
505 {
506  DrawViewPart* dvp = getBaseFeat();
507  DrawProjGroupItem* dpgi = dynamic_cast<DrawProjGroupItem*>(dvp);
508  DrawViewDetail* dvd = getDetailFeat();
509  Base::Vector3d anchorPos = dvd->AnchorPoint.getValue();
510  anchorPos.y = -anchorPos.y;
511  Base::Vector3d basePos;
512  double scale = 1;
513 
514  if (!dpgi) { //base is normal view
515  double x = dvp->X.getValue();
516  double y = dvp->Y.getValue();
517  basePos = Base::Vector3d (x, -y, 0.0);
518  scale = dvp->getScale();
519  } else { //part of projection group
520 
521  DrawProjGroup* dpg = dpgi->getPGroup();
522  if (!dpg) {
523  Base::Console().Message("TD::getAnchorScene - projection group is confused\n");
524  //TODO::throw something.
525  return QPointF(0.0, 0.0);
526  }
527  double x = dpg->X.getValue();
528  x += dpgi->X.getValue();
529  double y = dpg->Y.getValue();
530  y += dpgi->Y.getValue();
531  basePos = Base::Vector3d(x, -y, 0.0);
532  scale = dpgi->getScale();
533  }
534 
535  Base::Vector3d xyScene = Rez::guiX(basePos);
536  Base::Vector3d anchorOffsetScene = Rez::guiX(anchorPos) * scale;
537  Base::Vector3d netPos = xyScene + anchorOffsetScene;
538  return QPointF(netPos.x, netPos.y);
539 }
