143 template <Base::LogStyle type, Base::IntendedRecipient recipient, Base::ContentType content, typename TNotifier, typename TCaption, typename TMessage>
144 inline void Gui::Notify(TNotifier && notifier, TCaption && caption, TMessage && message)
145 {
146  static_assert(content != Base::ContentType::Untranslatable,
147  "A Gui notification cannot be provided with an untranslatable message.");
148  static_assert(recipient != Base::IntendedRecipient::Developer,
149  "A Gui notification cannot be intended only for the developer, use Console instead.");
150  static_assert(!(recipient == Base::IntendedRecipient::All && content == Base::ContentType::Translated),
151  "Information intended for Developers must not be translated. Provide an untranslated message instead.");
152 
153  Base::Reference<ParameterGrp> hGrp = App::GetApplication().GetUserParameter().
154  GetGroup("BaseApp")->GetGroup("Preferences")->
155  GetGroup("NotificationArea");
156 
157  bool nonIntrusive = hGrp->GetBool("NonIntrusiveNotificationsEnabled", true);
158 
159  if(!nonIntrusive) {
160 
161  // If Developers are also an intended recipient, notify before creating the blocking pop-up
162  if constexpr(recipient == Base::IntendedRecipient::All) {
163  // Send also to log for developer only
164  auto msg = std::string(message).append("\n"); // use untranslated message
165 
166  if constexpr( std::is_base_of_v<App::DocumentObject, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
167  Base::Console().Send<type,
168  Base::IntendedRecipient::Developer,
169  Base::ContentType::Untranslated>(notifier->getFullLabel(), msg.c_str());
170  }
171  else if constexpr( std::is_base_of_v<Gui::ViewProviderDocumentObject, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
172  Base::Console().Send<type,
173  Base::IntendedRecipient::Developer,
174  Base::ContentType::Untranslated>(notifier->getObject()->getFullLabel(), msg.c_str());
175  }
176  else if constexpr( std::is_base_of_v<Gui::Document, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
177  Base::Console().Send<type,
178  Base::IntendedRecipient::Developer,
179  Base::ContentType::Untranslated>(notifier->getDocument()->Label.getStrValue(), msg.c_str());
180  }
181  else if constexpr( std::is_base_of_v<App::Document, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
182  Base::Console().Send<type,
183  Base::IntendedRecipient::Developer,
184  Base::ContentType::Untranslated>(notifier->Label.getStrValue(), msg.c_str());
185  }
186  else {
187  Base::Console().Send<type,
188  Base::IntendedRecipient::Developer,
189  Base::ContentType::Untranslated>(notifier, msg.c_str());
190  }
191  }
192 
193  if constexpr(content == Base::ContentType::Untranslated) {
194  if constexpr( type == Base::LogStyle::Warning) {
195  QMessageBox::warning(Gui::getMainWindow(),
196  QCoreApplication::translate("Notifications", caption),
197  QCoreApplication::translate("Notifications", message));
198  }
199  else
200  if constexpr( type == Base::LogStyle::Error) {
201  QMessageBox::critical(Gui::getMainWindow(),
202  QCoreApplication::translate("Notifications", caption),
203  QCoreApplication::translate("Notifications", message));
204  }
205  else {
206  QMessageBox::information(Gui::getMainWindow(),
207  QCoreApplication::translate("Notifications", caption),
208  QCoreApplication::translate("Notifications", message));
209  }
210  }
211  else {
212  if constexpr( type == Base::LogStyle::Warning) {
213  QMessageBox::warning(Gui::getMainWindow(),
214  caption,
215  message);
216  }
217  else
218  if constexpr( type == Base::LogStyle::Error) {
219  QMessageBox::critical(Gui::getMainWindow(),
220  caption,
221  message);
222  }
223  else {
224  QMessageBox::information(Gui::getMainWindow(),
225  caption,
226  message);
227  }
228  }
229  }
230  else {
231 
232  if constexpr( content == Base::ContentType::Translated) {
233  // trailing newline is not necessary as translated messages are not shown in logs
234  auto msg = QStringLiteral("%1. %2").arg(caption).arg(message); // QString
235 
236  if constexpr( std::is_base_of_v<App::DocumentObject, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
237  Base::Console().Send<type, recipient, content>(notifier->getFullLabel(), msg.toUtf8());
238  }
239  else if constexpr( std::is_base_of_v<Gui::ViewProviderDocumentObject, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
240  Base::Console().Send<type, recipient, content>(notifier->getObject()->getFullLabel(), msg.toUtf8());
241  }
242  else if constexpr( std::is_base_of_v<Gui::Document, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
243  Base::Console().Send<type, recipient, content>(notifier->getDocument()->Label.getStrValue(), msg.toUtf8());
244  }
245  else if constexpr( std::is_base_of_v<App::Document, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
246  Base::Console().Send<type, recipient, content>(notifier->Label.getStrValue(), msg.toUtf8());
247  }
248  else {
249  Base::Console().Send<type, recipient, content>(notifier, msg.toUtf8());
250  }
251  }
252  else {
253  // trailing newline is necessary as this may be shown too in a console requiring them (depending on the configuration).
254  auto msg = std::string(message).append("\n");
255 
256  if constexpr( std::is_base_of_v<App::DocumentObject, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
257  Base::Console().Send<type, recipient, content>(notifier->getFullLabel(), msg.c_str());
258  }
259  else if constexpr( std::is_base_of_v<Gui::ViewProviderDocumentObject, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
260  Base::Console().Send<type, recipient, content>(notifier->getObject()->getFullLabel(), msg.c_str());
261  }
262  else if constexpr( std::is_base_of_v<Gui::Document, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
263  Base::Console().Send<type, recipient, content>(notifier->getDocument()->Label.getStrValue(), msg.c_str());
264  }
265  else if constexpr( std::is_base_of_v<App::Document, std::remove_pointer_t<typename std::decay<TNotifier>::type>> ) {
266  Base::Console().Send<type, recipient, content>(notifier->Label.getStrValue(), msg.c_str());
267  }
268  else {
269  Base::Console().Send<type, recipient, content>(notifier, msg.c_str());
270  }
271  }
272  }
273 }
