234 def getcolor(color):
235  """Check if the given string is an RGB value, or if it is a named color.
236 
237  Parameters
238  ----------
239  color : str
240  Color in hexadecimal format, long '#12ab9f' or short '#1af'
241 
242  Returns
243  -------
244  tuple
245  (r, g, b, a)
246  RGBA float tuple, where each value is between 0.0 and 1.0.
247  """
248  if color == "none":
249  FreeCAD.Console.PrintMessage("Color defined as 'none', defaulting to black\n")
250  return (0.0, 0.0, 0.0, 0.0)
251  if color[0] == "#":
252  if len(color) == 7 or len(color) == 9: # Color string '#RRGGBB' or '#RRGGBBAA'
253  r = float(int(color[1:3], 16) / 255.0)
254  g = float(int(color[3:5], 16) / 255.0)
255  b = float(int(color[5:7], 16) / 255.0)
256  a = 1.0
257  if len(color) == 9:
258  a = float(int(color[7:9], 16) / 255.0)
259  FreeCAD.Console.PrintMessage(f"Non standard color format #RRGGBBAA : {color}\n")
260  return (r, g, b, 1-a)
261  if len(color) == 4: # Color string '#RGB'
262  # Expand the hex digits
263  r = float(int(color[1], 16) * 17 / 255.0)
264  g = float(int(color[2], 16) * 17 / 255.0)
265  b = float(int(color[3], 16) * 17 / 255.0)
266  return (r, g, b, 0.0)
267  if color.lower().startswith('rgb(') or color.lower().startswith('rgba('): # Color string 'rgb[a](0.12,0.23,0.3,0.0)'
268  cvalues = color.lstrip('rgba(').rstrip(')').replace('%', '').split(',')
269  if len(cvalues) == 3:
270  a = 1.0
271  if '%' in color:
272  r, g, b = [int(float(cv)) / 100.0 for cv in cvalues]
273  else:
274  r, g, b = [int(float(cv)) / 255.0 for cv in cvalues]
275  if len(cvalues) == 4:
276  if '%' in color:
277  r, g, b, a = [int(float(cv)) / 100.0 for cv in cvalues]
278  else:
279  r, g, b, a = [int(float(cv)) / 255.0 for cv in cvalues]
280  return (r, g, b, 1-a)
281  # Trying named color like 'MediumAquamarine'
282  v = svgcolorslower.get(color.lower())
283  if v:
284  r, g, b = [float(vf) / 255.0 for vf in v]
285  return (r, g, b, 0.0)
286  FreeCAD.Console.PrintWarning(f"Unknown color format : {color} : defaulting to black\n")
287  return (0.0, 0.0, 0.0, 0.0)

290 def transformCopyShape(shape, m):
291  """Apply transformation matrix m on given shape.
292 
293  Since OCCT 6.8.0 transformShape can be used to apply certain
294  non-orthogonal transformations on shapes. This way a conversion
295  to BSplines in transformGeometry can be avoided.
296 
297  @sa: Part::TopoShape::transformGeometry(), TopoShapePy::transformGeometry()
298  @sa: Part::TopoShape::transformShape(), TopoShapePy::transformShape()
299 
300  Parameters
301  ----------
302  shape : Part::TopoShape
303  A given shape
304  m : Base::Matrix4D
305  A transformation matrix
306 
307  Returns
308  -------
309  shape : Part::TopoShape
310  The shape transformed by the matrix
311  """
312  # If there is no shear, these matrix operations will be very small
313  _s1 = abs(m.A11**2 + m.A12**2 - m.A21**2 - m.A22**2)
314  _s2 = abs(m.A11 * m.A21 + m.A12 * m.A22)
315  if _s1 < 1e-8 and _s2 < 1e-8:
316  try:
317  newshape = shape.copy()
318  newshape.transformShape(m)
319  return newshape
320  # Older versions of OCCT will refuse to work on
321  # non-orthogonal matrices
322  except Part.OCCError:
323  pass
324  return shape.transformGeometry(m)

328  """Parse the length string containing number and unit.
329 
330  Parameters
331  ----------
332  length : str
333  The length is a string, including sign, exponential notation,
334  and unit: '+56215.14565E+6mm', '-23.156e-2px'.
335  mode : str, optional
336  One of 'discard', 'tuple', 'css90.0', 'css96.0', 'mm90.0', 'mm96.0'.
337  'discard' (default), it discards the unit suffix, and extracts
338  a number from the given string.
339  'tuple', return number and unit as a tuple
340  'css90.0', convert the unit to pixels assuming 90 dpi
341  'css96.0', convert the unit to pixels assuming 96 dpi
342  'mm90.0', convert the unit to millimeters assuming 90 dpi
343  'mm96.0', convert the unit to millimeters assuming 96 dpi
344  base : float, optional
345  A base to scale the length.
346 
347  Returns
348  -------
349  float
350  The numeric value of the length, as is, or transformed to
351  millimeters or pixels.
352  float, string
353  A tuple with the numeric value, and the unit if `mode='tuple'`.
354  """

356  #
357  # The `em` and `ex` units are typographical units used in systems
358  # like LaTeX. Here the conversion factors are arbitrarily chosen,
359  # as they should depend on a specific font size used.
360  #

362  # on the viewport size or for filling patterns on the bounding box.
363  if mode == 'mm90.0':
364  tomm = {
365  '': 25.4/90, # default
366  'px': 25.4/90,
367  'pt': 4.0/3 * 25.4/90,
368  'pc': 15 * 25.4/90,
369  'mm': 1.0,
370  'cm': 10.0,
371  'in': 25.4,
372  'em': 15 * 2.54/90,
373  'ex': 10 * 2.54/90,

375  }
376  elif mode == 'mm96.0':

378  '': 25.4/96, # default
379  'px': 25.4/96,
380  'pt': 4.0/3 * 25.4/96,
381  'pc': 15 * 25.4/96,
382  'mm': 1.0,
383  'cm': 10.0,
384  'in': 25.4,
385  'em': 15 * 2.54/96,
386  'ex': 10 * 2.54/96,
387  '%': 100
388  }
389  elif mode == 'css90.0':
390  topx = {
391  '': 1.0, # default
392  'px': 1.0,
393  'pt': 4.0/3,
394  'pc': 15,
395  'mm': 90.0/25.4,
396  'cm': 90.0/254.0,
397  'in': 90,
398  'em': 15,
399  'ex': 10,
400  '%': 100
401  }
402  elif mode == 'css96.0':
403  topx = {
404  '': 1.0, # default
405  'px': 1.0,
406  'pt': 4.0/3,
407  'pc': 15,
408  'mm': 96.0/25.4,

410  'in': 96,
411  'em': 15,
412  'ex': 10,
413  '%': 100

418  _unit = '(px|pt|pc|mm|cm|in|em|ex|%)?'

429  elif mode == 'css96.0' or mode == 'css90.0':

431  return float(number) * topx[unit]

490 
491  Parameters
492  ----------

494  Coordinates of the center of the ellipse.
495  rx : float

504  Default 0. Rotation around the Z axis

515  ry * math.sin(angle1 + angledelta),

521  fa = ((abs(angledelta) / math.pi) % 2) > 1 # < 180 deg
522  fs = angledelta < 0
523  return v1, v2, fa, fs

526 def arcend2center(lastvec, currentvec, rx, ry,
527  xrotation=0.0, correction=False):
528  '''Calculate the possible centers for an arc in endpoint parameterization.
529 
530  Calculate (positive and negative) possible centers for an arc given in
531  ``endpoint parametrization``.
532  See http://www.w3.org/TR/SVG/implnote.html#ArcImplementationNotes
533 
534  the sweepflag is interpreted as: sweepflag <==> arc is travelled clockwise
535 
536  Parameters
537  ----------
538  lastvec : Base::Vector3
539  First point of the arc.
540  currentvec : Base::Vector3
541  End point (current) of the arc.
542  rx : float
543  Radius of the ellipse, semi-major axis in the X direction.
544  ry : float
545  Radius of the ellipse, semi-minor axis in the Y direction.
546  xrotation : float, optional
547  Default is 0. Rotation around the Z axis, in radians (CCW).
548  correction : bool, optional
549  Default is `False`. If it is `True`, the radii will be scaled
550  by a factor.
551 
552  Returns
553  -------
554  list, (float, float)
555  A tuple that consists of one list, and a tuple of radii.
556  [(positive), (negative)], (rx, ry)
557  The first element of the list is the positive tuple,
558  the second is the negative tuple.
559  [(Base::Vector3, float, float),
560  (Base::Vector3, float, float)], (float, float)
561  Types
562  [(vcenter+, angle1+, angledelta+),
563  (vcenter-, angle1-, angledelta-)], (rx, ry)
564  The first element of the list is the positive tuple,
565  consisting of center, angle, and angle increment;
566  the second element is the negative tuple.
567  '''
568  # scalefacsign = 1 if (largeflag != sweepflag) else -1
569  rx = float(rx)
570  ry = float(ry)
571  v0 = lastvec.sub(currentvec)
572  v0.multiply(0.5)
573  m1 = FreeCAD.Matrix()
574  m1.rotateZ(-xrotation) # eq. 5.1
575  v1 = m1.multiply(v0)
576  if correction:
577  eparam = v1.x**2 / rx**2 + v1.y**2 / ry**2
578  if eparam > 1:
579  eproot = math.sqrt(eparam)
580  rx = eproot * rx
581  ry = eproot * ry
582  denom = rx**2 * v1.y**2 + ry**2 * v1.x**2
583  numer = rx**2 * ry**2 - denom
584  results = []
585 
586  # If the division is very small, set the scaling factor to zero,
587  # otherwise try to calculate it by taking the square root
588  if abs(numer/denom) < 10**(-1 * (Draft.precision())):
589  scalefacpos = 0
590  else:
591  try:
592  scalefacpos = math.sqrt(numer/denom)
593  except ValueError:
594  _msg("sqrt({0}/{1})".format(numer, denom))
595  scalefacpos = 0
596 
597  # Calculate two values because the square root may be positive or negative
598  for scalefacsign in (1, -1):
599  scalefac = scalefacpos * scalefacsign
600  # Step2 eq. 5.2
601  vcx1 = Vector(v1.y * rx/ry, -v1.x * ry/rx, 0).multiply(scalefac)
602  m2 = FreeCAD.Matrix()
603  m2.rotateZ(xrotation)
604  centeroff = currentvec.add(lastvec)

1600  def getMatrix(self, tr):
1601  """Return a FreeCAD matrix from an SVG transform attribute.
1602 
1603  Parameters
1604  ----------
1605  tr : str
1606  The type of transform: 'matrix', 'translate', 'scale',
1607  'rotate', 'skewX', 'skewY' and its value
1608 
1609  Returns
1610  -------
1611  Base::Matrix4D
1612  The translated matrix.
1613  """
1614  _op = '(matrix|translate|scale|rotate|skewX|skewY)'
1615  _val = '\\((.*?)\\)'
1616  _transf = _op + '\\s*?' + _val
1617  transformre = re.compile(_transf, re.DOTALL)
1618  m = FreeCAD.Matrix()
1619  for transformation, arguments in transformre.findall(tr):
1620  _args_rep = arguments.replace(',', ' ').split()
1621  argsplit = [float(arg) for arg in _args_rep]
1622  # m.multiply(FreeCAD.Matrix(1, 0, 0, 0, 0, -1))
1623  # print('%s:%s %s %d' % (transformation, arguments,
1624  # argsplit, len(argsplit)))

1626  tx = argsplit[0]
1627  ty = argsplit[1] if len(argsplit) > 1 else 0.0
1628  m.move(Vector(tx, -ty, 0))
1629  elif transformation == 'scale':
1630  sx = argsplit[0]
1631  sy = argsplit[1] if len(argsplit) > 1 else sx
1632  m.scale(Vector(sx, sy, 1))
1633  elif transformation == 'rotate':
1634  cx = 0
1635  cy = 0

1637  if len(argsplit) >= 3:
1638  # Rotate around a non-origin centerpoint (note: SVG y axis is opposite FreeCAD y axis)
1639  cx = argsplit[1]
1640  cy = argsplit[2]
1641  m.move(Vector(-cx, cy, 0)) # Reposition for rotation
1642  # Mirroring one axis is equal to changing the direction
1643  # of rotation
1644  m.rotateZ(math.radians(-angle))
1645  if len(argsplit) >= 3:
1646  m.move(Vector(cx, -cy, 0)) # Reverse repositioning

1649  -math.tan(math.radians(argsplit[0])))
1650  m = m.multiply(_m)
1651  elif transformation == 'skewY':
1652  _m = FreeCAD.Matrix(1, 0, 0, 0,
1653  -math.tan(math.radians(argsplit[0])))
1654  m = m.multiply(_m)

1657  # FreeCAD SVG
1658  # (+A -C +0 +E) (A C 0 E)

1660  # (+0 -0 +1 +0) (0 0 1 0)
1661  # (+0 -0 +0 +1) (0 0 0 1)

1665  0, argsplit[4],

1667  0, -argsplit[5])
1668  m = m.multiply(_m)
1669  # else:
1670  # print('SKIPPED %s' % transformation)
1671  # print("m = ", m)
1672  # print("generating transformation: ", m)
