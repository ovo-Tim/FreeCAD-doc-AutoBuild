248  std::string fn = doc->TransientDir.getValue();
249  fn += "/";
250  fn += uuid;

254  // open extra scope to close ZipWriter properly
255  {
256  Base::ofstream file(tmp, std::ios::out | std::ios::binary);
257  Base::ZipWriter writer(file);
258 
259  writer.setComment("FreeCAD Document");
260  writer.setLevel(0);
261  writer.putNextEntry("Document.xml");
262 
263  doc->Save(writer);
264 
265  // Special handling for Gui document.
266  doc->signalSaveDocument(writer);
267 
268  // write additional files
269  writer.writeFiles();
270  }
