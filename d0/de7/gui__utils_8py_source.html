85 def autogroup(obj):
86  """Add a given object to the defined Draft autogroup, if applicable.
87 
88  This function only works if the graphical interface is available.
89  It checks that the `App.draftToolBar` class is available,
90  which contains the group to use to automatically store
91  new created objects.
92 
93  Originally, it worked with standard groups (`App::DocumentObjectGroup`),
94  and Arch Workbench containers like `'Site'`, `'Building'`, `'Floor'`,
95  and `'BuildingPart'`.
96 
97  Now it works with Draft Layers.
98 
99  Parameters
100  ----------
101  obj: App::DocumentObject
102  Any type of object that will be stored in the group.
103  """
104 
105  # check for required conditions for autogroup to work
106  if not App.GuiUp:
107  return
108  if not hasattr(Gui,"draftToolBar"):
109  return
110  if not hasattr(Gui.draftToolBar,"autogroup"):
111  return
112  if Gui.draftToolBar.isConstructionMode():
113  return
114 
115  # check first for objects that do autogroup themselves
116  # at the moment only Arch_BuildingPart, which is an App::GeometryPython
117  for par in App.ActiveDocument.findObjects(Type="App::GeometryPython"):
118  if hasattr(par.Proxy,"autogroup"):
119  if par.Proxy.autogroup(par,obj):
120  return
121 
122  # autogroup code
123  active_group = None
124  if Gui.draftToolBar.autogroup is not None:
125  active_group = App.ActiveDocument.getObject(Gui.draftToolBar.autogroup)
126  if active_group:
127  gr = active_group.Group
128  if not obj in gr:
129  gr.append(obj)
130  active_group.Group = gr
131 
132  if Gui.ActiveDocument.ActiveView.getActiveObject("NativeIFC") is not None:
133  # NativeIFC handling
134  try:
135  from nativeifc import ifc_tools
136  parent = Gui.ActiveDocument.ActiveView.getActiveObject("NativeIFC")
137  if parent != active_group:
138  ifc_tools.aggregate(obj, parent)
139  except:
140  pass
141 
142  elif Gui.ActiveDocument.ActiveView.getActiveObject("Arch") is not None:
143  # add object to active Arch Container
144  active_arch_obj = Gui.ActiveDocument.ActiveView.getActiveObject("Arch")
145  if active_arch_obj != active_group:
146  if obj in active_arch_obj.InListRecursive:
147  # do not autogroup if obj points to active_arch_obj to prevent cyclic references
148  return
149  active_arch_obj.addObject(obj)
150 
151  elif Gui.ActiveDocument.ActiveView.getActiveObject("part") is not None:
152  # add object to active part and change it's placement accordingly
153  # so object does not jump to different position, works with App::Link
154  # if not scaled. Modified accordingly to realthunder suggestions
155  active_part, parent, sub = Gui.ActiveDocument.ActiveView.getActiveObject("part", False)
156  if active_part != active_group:
157  if obj in active_part.InListRecursive:
158  # do not autogroup if obj points to active_part to prevent cyclic references
159  return
160  matrix = parent.getSubObject(sub, retType=4)
161  if matrix.hasScale() == App.ScaleType.Uniform:
162  err = translate("draft",
163  "Unable to insert new object into "
164  "a scaled part")
165  App.Console.PrintMessage(err)
166  return
167  inverse_placement = App.Placement(matrix.inverse())
168  if utils.get_type(obj) == 'Point':
169  point_vector = App.Vector(obj.X, obj.Y, obj.Z)
170  real_point = inverse_placement.multVec(point_vector)
171  obj.X = real_point.x
172  obj.Y = real_point.y
173  obj.Z = real_point.z
174  elif utils.get_type(obj) in ["Dimension", "LinearDimension"]:
175  obj.Start = inverse_placement.multVec(obj.Start)
176  obj.End = inverse_placement.multVec(obj.End)
177  obj.Dimline = inverse_placement.multVec(obj.Dimline)
178  obj.Normal = inverse_placement.Rotation.multVec(obj.Normal)
179  obj.Direction = inverse_placement.Rotation.multVec(obj.Direction)
180  elif utils.get_type(obj) in ["Label"]:
181  obj.Placement = App.Placement(inverse_placement.multiply(obj.Placement))
182  obj.TargetPoint = inverse_placement.multVec(obj.TargetPoint)
183  elif hasattr(obj,"Placement"):
184  # every object that have a placement is processed here
185  obj.Placement = App.Placement(inverse_placement.multiply(obj.Placement))
186 
187  active_part.addObject(obj)

190 def dim_symbol(symbol=None, invert=False):
191  """Return the specified dimension symbol.
192 
193  Parameters
194  ----------
195  symbol: int, optional
196  It defaults to `None`, in which it gets the value from the parameter
197  database, `get_param("dimsymbol")`.
198 
199  A numerical value defines different markers
200  * 0, `SoSphere`
201  * 1, `SoSeparator` with a `SoLineSet`, a circle (in fact a 24 sided polygon)
202  * 2, `SoSeparator` with a `soCone`
203  * 3, `SoSeparator` with a `SoFaceSet`
204  * 4, `SoSeparator` with a `SoLineSet`, calling `dim_dash`
205  * Otherwise, `SoSphere`
206 
207  invert: bool, optional
208  It defaults to `False`.
209  If it is `True` and `symbol=2`, the cone will be rotated
210  -90 degrees around the Z axis, otherwise the rotation is positive,
211  +90 degrees.
212 
213  Returns
214  -------
215  Coin.SoNode
216  A `Coin.SoSphere`, or `Coin.SoSeparator` (circle, cone, face, line)
217  that will be used as a dimension symbol.
218  """
219  if symbol is None:
220  symbol = params.get_param("dimsymbol")
221 
222  if symbol == 0:
223  # marker = coin.SoMarkerSet()
224  # marker.markerIndex = 80
225 
226  # Returning a sphere means that the bounding box will
227  # be 3-dimensional; a marker will always be planar seen from any
228  # orientation but it currently doesn't work correctly
229  marker = coin.SoSphere()
230  return marker
231  elif symbol == 1:
232  marker = coin.SoSeparator()
233  v = coin.SoVertexProperty()
234  for i in range(25):
235  ang = math.radians(i * 15)
236  v.vertex.set1Value(i, (math.sin(ang), math.cos(ang), 0))
237  p = coin.SoLineSet()
238  p.vertexProperty = v
239  marker.addChild(p)
240  return marker
241  elif symbol == 2:
242  marker = coin.SoSeparator()
243  t = coin.SoTransform()
244  t.translation.setValue((0, -2, 0))
245  t.center.setValue((0, 2, 0))
246  if invert:
247  t.rotation.setValue(coin.SbVec3f((0, 0, 1)), -math.pi/2)
248  else:
249  t.rotation.setValue(coin.SbVec3f((0, 0, 1)), math.pi/2)
250  c = coin.SoCone()
251  c.height.setValue(4)
252  marker.addChild(t)
253  marker.addChild(c)
254  return marker
255  elif symbol == 3:
256  marker = coin.SoSeparator()
257  # hints are required otherwise only the bottom of the face is colored
258  h = coin.SoShapeHints()
259  h.vertexOrdering = h.COUNTERCLOCKWISE
260  c = coin.SoCoordinate3()
261  c.point.setValues([(-1, -2, 0), (0, 2, 0),
262  (1, 2, 0), (0, -2, 0)])
263  f = coin.SoFaceSet()
264  marker.addChild(h)
265  marker.addChild(c)
266  marker.addChild(f)
267  return marker
268  elif symbol == 4:
269  return dim_dash((-1.5, -1.5, 0), (1.5, 1.5, 0))
270  else:
271  _wrn(translate("draft", "Symbol not implemented. Using a default symbol."))
272  return coin.SoSphere()

278 def dim_dash(p1, p2):
279  """Return a SoSeparator with a line used to make dimension dashes.
280 
281  It is used by `dim_symbol` to create line end symbols
282  like `'Tick-2'`, `'DimOvershoot'`, and `'ExtOvershoot'` dashes.
283 
284  Parameters
285  ----------
286  p1: tuple of three floats or Base::Vector3
287  A point to define a line vertex.
288 
289  p2: tuple of three floats or Base::Vector3
290  A point to define a line vertex.
291 
292  Returns
293  -------
294  Coin.SoSeparator
295  A Coin object with a `SoLineSet` created from `p1` and `p2`
296  as vertices.
297  """
298  dash = coin.SoSeparator()
299  v = coin.SoVertexProperty()
300  v.vertex.set1Value(0, p1)
301  v.vertex.set1Value(1, p2)
302  line = coin.SoLineSet()
303  line.vertexProperty = v
304  dash.addChild(line)
305  return dash

343 def get_diffuse_color(objs):
344  """Get a (cumulative) diffuse color from one or more objects.
345 
346  If all tuples in the result are identical a list with a single tuple is
347  returned. In theory all faces of an object can be set to the same diffuse
348  color that is different from its shape color, but that seems rare. The
349  function does not take that into account.
350 
351  Parameters
352  ----------
353  objs: a single object or an iterable with objects.
354 
355  Returns
356  -------
357  list of tuples
358  The list will be empty if no valid object is found.
359  """
360  def _get_color(obj):
361  if hasattr(obj, "ColoredElements"):
362  if hasattr(obj, "Count") or hasattr(obj, "ElementCount"):
363  # Link and Link array.
364  if hasattr(obj, "Count"):
365  count = obj.Count
366  base = obj.Base
367  else:
368  count = obj.ElementCount if obj.ElementCount > 0 else 1
369  base = obj.LinkedObject
370  if base is None:
371  return []
372  cols = _get_color(base) * count
373  if obj.ColoredElements is None:
374  return cols
375  face_num = len(base.Shape.Faces)
376  for elm, overide in zip(obj.ColoredElements[1], obj.ViewObject.OverrideColorList):
377  if "Face" in elm: # Examples: "Face3" and "1.Face6". Int before "." is zero-based, other int is 1-based.
378  if "." in elm:
379  elm0, elm1 = elm.split(".")
380  i = (int(elm0) * face_num) + int(elm1[4:]) - 1
381  cols[i] = overide
382  else:
383  i = int(elm[4:]) - 1
384  for j in range(count):
385  cols[(j * face_num) + i] = overide
386  return cols
387  elif hasattr(obj, "ElementList"):
388  # LinkGroup
389  cols = []
390  for sub in obj.ElementList:
391  sub_cols = _get_color(sub)
392  if obj.ColoredElements is None:
393  cols += sub_cols
394  else:
395  for elm, overide in zip(obj.ColoredElements[1], obj.ViewObject.OverrideColorList):
396  if sub.Name + ".Face" in elm:
397  i = int(elm[(len(sub.Name) + 5):]) - 1
398  sub_cols[i] = overide
399  cols += sub_cols
400  return cols
401  else:
402  return []
403  elif hasattr(obj.ViewObject, "DiffuseColor"):
404  if len(obj.ViewObject.DiffuseColor) == len(obj.Shape.Faces):
405  return obj.ViewObject.DiffuseColor
406  else:
407  col = obj.ViewObject.ShapeColor
408  col = (col[0], col[1], col[2], 1.0 - obj.ViewObject.Transparency / 100.0)
409  return [col] * len(obj.Shape.Faces)
410  elif obj.hasExtension("App::GeoFeatureGroupExtension"):
411  cols = []
412  for sub in obj.Group:
413  cols += _get_color(sub)
414  return cols
415  else:
416  return []
417 
418  if not isinstance(objs, list):
419  # Quick check to avoid processing a single object:
420  obj = objs
421  if not hasattr(obj, "ColoredElements") \
422  and hasattr(obj.ViewObject, "DiffuseColor") \
423  and (len(obj.ViewObject.DiffuseColor) == 1 \
424  or len(obj.ViewObject.DiffuseColor) == len(obj.Shape.Faces)):
425  return obj.ViewObject.DiffuseColor
426  # Create a list for further processing:
427  objs = [objs]
428 
429  colors = []
430  for obj in objs:
431  colors += _get_color(obj)
432 
433  if len(colors) > 1:
434  first = colors[0]
435  for next in colors[1:]:
436  if next != first:
437  break
438  else:
439  colors = [first]
440  return colors

504 def format_object(target, origin=None):
505  """Apply visual properties to an object.
506 
507  This function only works if the graphical interface is available.
508 
509  If origin is `None` and target is not an annotation, the DefaultDrawStyle
510  and DefaultDisplayMode preferences are applied. Else, the properties of
511  origin are applied to target.
512 
513  If construction mode is active target is then placed in the construction
514  group and the `constr` color is applied to its applicable color properties:
515  TextColor, PointColor, LineColor, and ShapeColor.
516 
517  Parameters
518  ----------
519  target: App::DocumentObject
520 
521  origin: App::DocumentObject, optional
522  Defaults to `None`.
523  If construction mode is not active, its visual properties are assigned
524  to `target`, with the exception of `BoundingBox`, `Proxy`, `RootNode`
525  and `Visibility`.
526  """
527  if not target:
528  return
529  if not App.GuiUp:
530  return
531  if not hasattr(Gui, "draftToolBar"):
532  return
533  if not hasattr(target, 'ViewObject'):
534  return
535  obrep = target.ViewObject
536  obprops = obrep.PropertiesList
537  if origin and hasattr(origin, 'ViewObject'):
538  matchrep = origin.ViewObject
539  for p in matchrep.PropertiesList:
540  if p not in ("DisplayMode", "BoundingBox",
541  "Proxy", "RootNode", "Visibility"):
542  if p in obprops:
543  if not obrep.getEditorMode(p):
544  if hasattr(getattr(matchrep, p), "Value"):
545  val = getattr(matchrep, p).Value
546  else:
547  val = getattr(matchrep, p)
548  try:
549  setattr(obrep, p, val)
550  except Exception:
551  pass
552  if matchrep.DisplayMode in obrep.listDisplayModes():
553  obrep.DisplayMode = matchrep.DisplayMode
554  if hasattr(obrep, "DiffuseColor"):
555  difcol = get_diffuse_color(origin)
556  if difcol:
557  obrep.DiffuseColor = difcol
558  elif "FontName" not in obprops:
559  # Apply 2 Draft style preferences, other style preferences are applied by Core.
560  if "DrawStyle" in obprops:
561  obrep.DrawStyle = utils.DRAW_STYLES[params.get_param("DefaultDrawStyle")]
562  if "DisplayMode" in obprops:
563  dm = utils.DISPLAY_MODES[params.get_param("DefaultDisplayMode")]
564  if dm in obrep.listDisplayModes():
565  obrep.DisplayMode = dm
566  if Gui.draftToolBar.isConstructionMode():
567  doc = App.ActiveDocument
568  col = params.get_param("constructioncolor") & 0xFFFFFF00
569  grp = doc.getObject("Draft_Construction")
570  if not grp:
571  grp = doc.addObject("App::DocumentObjectGroup", "Draft_Construction")
572  grp.Label = params.get_param("constructiongroupname")
573  grp.addObject(target)
574  if "TextColor" in obprops:
575  obrep.TextColor = col
576  if "PointColor" in obprops:
577  obrep.PointColor = col
578  if "LineColor" in obprops:
579  obrep.LineColor = col
580  if "ShapeColor" in obprops:
581  obrep.ShapeColor = col
582  if hasattr(obrep, "Transparency"):
583  obrep.Transparency = 80
