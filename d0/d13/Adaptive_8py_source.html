111 def CalcHelixConePoint(height, cur_z, radius, angle):
112  x = ((height - cur_z) / height) * radius * math.cos(math.radians(angle) * cur_z)
113  y = ((height - cur_z) / height) * radius * math.sin(math.radians(angle) * cur_z)
114  z = cur_z
115 
116  return {"X": x, "Y": y, "Z": z}

777 def _get_working_edges(op, obj):
778  """_get_working_edges(op, obj)...
779  Compile all working edges from the Base Geometry selection (obj.Base)
780  for the current operation.
781  Additional modifications to selected region(face), such as extensions,
782  should be placed within this function.
783  """
784  all_regions = list()
785  edge_list = list()
786  avoidFeatures = list()
787  rawEdges = list()

790  extensions = FeatureExtensions.getExtensions(obj)
791  for e in extensions:
792  if e.avoid:
793  avoidFeatures.append(e.feature)
794 
795  # Get faces selected by user
796  for base, subs in obj.Base:
797  for sub in subs:
798  if sub.startswith("Face"):
799  if sub not in avoidFeatures:
800  if obj.UseOutline:
801  face = base.Shape.getElement(sub)
802  # get outline with wire_A method used in PocketShape, but it does not play nicely later
803  # wire_A = TechDraw.findShapeOutline(face, 1, FreeCAD.Vector(0.0, 0.0, 1.0))
804  wire_B = face.OuterWire
805  shape = Part.Face(wire_B)
806  else:
807  shape = base.Shape.getElement(sub)
808  all_regions.append(shape)
809  elif sub.startswith("Edge"):
810  # Save edges for later processing
811  rawEdges.append(base.Shape.getElement(sub))
812  # Efor
813 
814  # Process selected edges
815  if rawEdges:
816  edgeWires = DraftGeomUtils.findWires(rawEdges)
817  if edgeWires:
818  for w in edgeWires:
819  for e in w.Edges:
820  edge_list.append([discretize(e)])
821 
822  # Apply regular Extensions
823  op.exts = []
824  for ext in extensions:
825  if not ext.avoid:
826  wire = ext.getWire()
827  if wire:
828  for f in ext.getExtensionFaces(wire):
829  op.exts.append(f)
830  all_regions.append(f)
831 
832  # Second face-combining method attempted
833  horizontal = Path.Geom.combineHorizontalFaces(all_regions)
834  if horizontal:
835  obj.removalshape = Part.makeCompound(horizontal)
836  for f in horizontal:
837  for w in f.Wires:
838  for e in w.Edges:
839  edge_list.append([discretize(e)])
