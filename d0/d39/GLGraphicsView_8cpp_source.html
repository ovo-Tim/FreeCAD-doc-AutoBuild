79 bool GraphicsView::viewportEvent(QEvent* event)
80 {
81  if (event->type() == QEvent::Paint || event->type() == QEvent::Resize) {
82  return QGraphicsView::viewportEvent(event);
83  }
84  else if (event->type() == QEvent::MouseMove ||
85  event->type() == QEvent::Wheel ||
86  event->type() == QEvent::MouseButtonDblClick ||
87  event->type() == QEvent::MouseButtonRelease ||
88  event->type() == QEvent::MouseButtonPress) {
89  QMouseEvent* mouse = static_cast<QMouseEvent*>(event);
90  QGraphicsItem *item = itemAt(mouse->pos());
91  if (!item) {
92  QGraphicsView::viewportEvent(event);
93  return false;
94  }
95 
96  return QGraphicsView::viewportEvent(event);
97  }
98 
99  return QGraphicsView::viewportEvent(event);
100 }

197 bool
198 SceneEventFilter::eventFilter(QObject *, QEvent * qevent)
199 {
200  // Convert the scene event back to a standard event
201  std::unique_ptr<QEvent> sceneev;
202  switch (qevent->type()) {
203 
204  case QEvent::GraphicsSceneMouseMove:
205  {
206  QGraphicsSceneMouseEvent* ev = static_cast<QGraphicsSceneMouseEvent*>(qevent);
207  sceneev.reset(new QMouseEvent(QEvent::MouseMove, ev->pos().toPoint(),
208  ev->button(), ev->buttons(), ev->modifiers()));
209  qevent = sceneev.get();
210  break;
211  }
212  case QEvent::GraphicsSceneMousePress:
213  {
214  QGraphicsSceneMouseEvent* ev = static_cast<QGraphicsSceneMouseEvent*>(qevent);
215  sceneev.reset(new QMouseEvent(QEvent::MouseButtonPress, ev->pos().toPoint(),
216  ev->button(), ev->buttons(), ev->modifiers()));
217  qevent = sceneev.get();
218  break;
219  }
220  case QEvent::GraphicsSceneMouseRelease:
221  {
222  QGraphicsSceneMouseEvent* ev = static_cast<QGraphicsSceneMouseEvent*>(qevent);
223  sceneev.reset(new QMouseEvent(QEvent::MouseButtonRelease, ev->pos().toPoint(),
224  ev->button(), ev->buttons(), ev->modifiers()));
225  qevent = sceneev.get();
226  break;
227  }
228  case QEvent::GraphicsSceneMouseDoubleClick:
229  {
230  QGraphicsSceneMouseEvent* ev = static_cast<QGraphicsSceneMouseEvent*>(qevent);
231  sceneev.reset(new QMouseEvent(QEvent::MouseButtonDblClick, ev->pos().toPoint(),
232  ev->button(), ev->buttons(), ev->modifiers()));
233  qevent = sceneev.get();
234  break;
235  }
236  case QEvent::GraphicsSceneWheel:
237  {
238 #if QT_VERSION < QT_VERSION_CHECK(6,0,0)
239  QGraphicsSceneWheelEvent* ev = static_cast<QGraphicsSceneWheelEvent*>(qevent);
240  sceneev.reset(new QWheelEvent(ev->pos().toPoint(), ev->delta(), ev->buttons(),
241  ev->modifiers(), ev->orientation()));
242  qevent = sceneev.get();
243 #endif
244  break;
245  }
246  case QEvent::GraphicsSceneResize:
247  {
248  QGraphicsSceneResizeEvent* ev = static_cast<QGraphicsSceneResizeEvent*>(qevent);
249  sceneev.reset(new QResizeEvent(ev->newSize().toSize(), ev->oldSize().toSize()));
250  qevent = sceneev.get();
251  break;
252  }
253  default:
254  break;
255  }
256 
257  // make sure every device has updated screen size and mouse position
258  // before translating events
259  switch (qevent->type()) {
260  case QEvent::MouseMove:
261  case QEvent::MouseButtonPress:
262  case QEvent::MouseButtonRelease:
263  case QEvent::MouseButtonDblClick:
264  PRIVATE(this)->trackPointerPosition(dynamic_cast<QMouseEvent *>(qevent));
265  break;
266  case QEvent::Resize:
267  PRIVATE(this)->trackWindowSize(dynamic_cast<QResizeEvent *>(qevent));
268  break;
269  default:
270  break;
271  }
272 
273  // translate QEvent into SoEvent and see if it is handled by scene
274  // graph
275  Q_FOREACH(SIM::Coin3D::Quarter::InputDevice * device, PRIVATE(this)->devices) {
276  const SoEvent * soevent = device->translateEvent(qevent);
277  if (soevent && PRIVATE(this)->scene->processSoEvent(soevent)) {
278  //return true;
279  }
280  }
281  return false;
282 }

420 void
421 GraphicsScene::setNavigationModeFile(const QUrl & url)
422 {
423  QString filename;
424 
425  if (url.scheme()== QLatin1String("coin")) {
426  filename = url.path();
427  //FIXME: This conditional needs to be implemented when the
428  //CoinResources systems if working
429  //Workaround for differences between url scheme, and Coin internal
430  //scheme in Coin 3.0.
431  if (filename[0] == QLatin1Char('/')) {
432  filename.remove(0,1);
433  }
434  filename = url.scheme() + QLatin1Char(':') + filename;
435  }
436  else if (url.scheme() == QLatin1String("file"))
437  filename = url.toLocalFile();
438  else if (url.isEmpty()) {
439 
440  return;
441  }
442  else {
443  return;
444  }
445 
446  QByteArray filenametmp = filename.toLocal8Bit();
447  ScXMLStateMachine * stateMachine = NULL;
448 
449  if (filenametmp.startsWith("coin:")){
450  stateMachine = ScXML::readFile(filenametmp.data());
451  }
452  else {
453  //Use Qt to read the file in case it is a Qt resource
454  QFile file(QString::fromLatin1(filenametmp));
455  if (file.open(QIODevice::ReadOnly)) {
456  QByteArray fileContents = file.readAll();
457 #if COIN_MAJOR_VERSION >= 4
458  stateMachine = ScXML::readBuffer(SbByteBuffer(fileContents.size(), fileContents.constData()));
459 #else
460  stateMachine = ScXML::readBuffer(fileContents.constData());
461 #endif
462  file.close();
463  }
464  }
465 
466  if (stateMachine && stateMachine->isOfType(SoScXMLStateMachine::getClassTypeId())) {
467  SoScXMLStateMachine * newsm = static_cast<SoScXMLStateMachine *>(stateMachine);
468 
469  this->addStateMachine(newsm);
470  newsm->initialize();
471  }
472  else {
473  if (stateMachine)
474  delete stateMachine;
475 
476  return;
477  }
478 }

503 {
