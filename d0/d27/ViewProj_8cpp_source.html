80 ViewProjMatrix::ViewProjMatrix(const Matrix4D& rclMtx)
81  : _clMtx(rclMtx)
82 {
83  double m30 = _clMtx[3][0];
84  double m31 = _clMtx[3][1];
85  double m32 = _clMtx[3][2];
86  double m33 = _clMtx[3][3];
87  isOrthographic = (m30 == 0.0 && m31 == 0.0 && m32 == 0.0 && m33 == 1.0);
88 
89  // Only for orthographic projection mode we can compute a single
90  // matrix performing all steps.
91  // For perspective projection the scaling and translation must
92  // be done afterwards because it depends on the input point.
93  if (isOrthographic) {
94  // Scale from [-1,1] to [0,1]
95  // As done in OpenInventor sources (see SbDPViewVolume::projectToScreen)
96  _clMtx.scale(0.5, 0.5, 0.5);
97  _clMtx.move(0.5, 0.5, 0.5);
98  }
99 
100  _clMtxInv = _clMtx;
101  _clMtxInv.inverseGauss();
102 }

116 template<typename Vec>
117 void perspectiveTransform(const Base::Matrix4D& mat, Vec& pnt)
118 {
119  double m30 = mat[3][0];
120  double m31 = mat[3][1];
121  double m32 = mat[3][2];
122  double m33 = mat[3][3];
123  double ww = (static_cast<double>(pnt.x) * m30 + static_cast<double>(pnt.y) * m31
124  + static_cast<double>(pnt.z) * m32 + m33);
125 
126  mat.multVec(pnt, pnt);
127  pnt /= static_cast<typename Vec::num_type>(ww);
128 }

131 {
