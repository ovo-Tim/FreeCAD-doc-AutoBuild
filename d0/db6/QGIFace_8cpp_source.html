107 void QGIFace::draw()
108 {
109 // Base::Console().Message("QGIF::draw - pen style: %d\n", m_pen.style());
110  setPath(m_outline); //Face boundary
111 
112  m_svgHatchArea->hide();
113  m_imageSvgHatchArea->hide();
114 
115  if (isHatched()) {
116  if (m_mode == GeomHatchFill) {
117  //GeomHatch does not appear in pdf if clipping is set to true
118  setFlag(QGraphicsItem::ItemClipsChildrenToShape, false);
119  if (!m_lineSets.empty()) {
120  m_brush.setTexture(QPixmap());
121  m_fillStyleCurrent = m_fillDef;
122  m_fillNormal = m_fillStyleCurrent;
123  for (auto& ls: m_lineSets) {
124  lineSetToFillItems(ls);
125  }
126  }
127  } else if (m_mode == SvgFill) {
128  m_brush.setTexture(QPixmap());
129  m_fillNormal = m_fillDef;
130  m_fillStyleCurrent = m_fillNormal;
131  setFlag(QGraphicsItem::ItemClipsChildrenToShape,true);
132  loadSvgHatch(m_fileSpec);
133  if (exporting()) {
134  buildPixHatch();
135  m_imageSvgHatchArea->show();
136  } else {
137  buildSvgHatch();
138  m_svgHatchArea->show();
139  }
140  } else if (m_mode == BitmapFill) {
141  m_fillStyleCurrent = Qt::TexturePattern;
142  m_texture = textureFromBitmap(m_fileSpec);
143  m_brush.setTexture(m_texture);
144  } else if (m_mode == PlainFill) {
145  setFill(m_colNormalFill, m_fillNormal);
146  }
147  }
148 
149  show();
150 }

197 void QGIFace::loadSvgHatch(std::string fileSpec)
198 {
199  QString qfs(QString::fromUtf8(fileSpec.data(), fileSpec.size()));
200  QFile file(qfs);
201  if (!file.open(QFile::ReadOnly | QFile::Text)) {
202  Base::Console().Error("QGIFace could not read %s\n", fileSpec.c_str());
203  return;
204  }
205  m_svgXML = file.readAll();
206 
207  // search in the file for the "stroke" specification in order to find out what declaration style is used
208  // this is necessary to apply a color set by the user to the SVG
209  QByteArray pattern("stroke:");
210  QByteArrayMatcher matcher(pattern);
211  int pos = 0;
212  if (matcher.indexIn(m_svgXML, pos) != -1) {
213  SVGCOLPREFIX = "stroke:"; // declaration part of a style="" statement
214  } else {
215  SVGCOLPREFIX = "stroke=\""; // declaration of its own
216  }
217 }

307 void QGIFace::buildSvgHatch()
308 {
309 // Base::Console().Message("QGIF::buildSvgHatch() - offset: %s\n", DrawUtil::formatVector(getHatchOffset()).c_str());
310  double wTile = SVGSIZEW * m_fillScale;
311  double hTile = SVGSIZEH * m_fillScale;
312  double faceWidth = m_outline.boundingRect().width();
313  double faceHeight = m_outline.boundingRect().height();
314  double faceOverlaySize = Preferences::svgHatchFactor() * std::max(faceWidth, faceHeight);
315  QPointF faceCenter = m_outline.boundingRect().center();
316  double tilesWide = ceil(faceOverlaySize / wTile);
317  double tilesHigh = ceil(faceOverlaySize / hTile);
318 
319  double overlayWidth = tilesWide * wTile;
320  double overlayHeight = tilesHigh * hTile;
321  m_svgHatchArea->setRect(0., 0., overlayWidth,-overlayHeight);
322  m_svgHatchArea->centerAt(faceCenter);
323 
324  QByteArray before = QString::fromStdString(SVGCOLPREFIX + SVGCOLDEFAULT).toUtf8();
325  QByteArray after = QString::fromStdString(SVGCOLPREFIX + m_svgCol).toUtf8();
326  QByteArray colorXML = m_svgXML.replace(before, after);
327  if (!m_sharedRender->load(colorXML)) {
328  Base::Console().Message("QGIF::buildSvgHatch - failed to load svg string\n");
329  return;
330  }
331  long int tileCount = 0;
332  for (int iw = 0; iw < int(tilesWide); iw++) {
333  for (int ih = 0; ih < int(tilesHigh); ih++) {
334  QGCustomSvg* tile = new QGCustomSvg();
335  tile->setScale(m_fillScale);
336  tile->setSharedRenderer(m_sharedRender);
337  tile->setParentItem(m_svgHatchArea);
338  tile->setPos(iw*wTile + getHatchOffset().x,
339  -overlayWidth + ih*hTile + getHatchOffset().y);
340  tileCount++;
341  if (tileCount > m_maxTile) {
342  Base::Console().Warning("SVG tile count exceeded: %ld. Change hatch scale or raise limit.\n", tileCount);
343  break;
344  }
345  }
346  if (tileCount > m_maxTile) {
347  break;
348  }
349  }
350  QPointF faceCenterToMRect = mapToItem(m_svgHatchArea, faceCenter);
351  m_svgHatchArea->setTransformOriginPoint(faceCenterToMRect);
352  m_svgHatchArea->setRotation(m_hatchRotation);
353 }

362 void QGIFace::buildPixHatch()
363 {
364  double wTile = SVGSIZEW * m_fillScale;
365  double hTile = SVGSIZEH * m_fillScale;
366  double faceWidth = m_outline.boundingRect().width();
367  double faceHeight = m_outline.boundingRect().height();
368  double faceOverlaySize = Preferences::svgHatchFactor() * std::max(faceWidth, faceHeight);
369  QPointF faceCenter = m_outline.boundingRect().center();
370  double tilesWide = ceil(faceOverlaySize / wTile);
371  double tilesHigh = ceil(faceOverlaySize / hTile);
372 
373  double overlayWidth = tilesWide * wTile;
374  double overlayHeight = tilesHigh * hTile;
375 
376  // handle color by brute force find & replace
377  QByteArray before = QString::fromStdString(SVGCOLPREFIX + SVGCOLDEFAULT).toUtf8();
378  QByteArray after = QString::fromStdString(SVGCOLPREFIX + m_svgCol).toUtf8();
379  QByteArray colorXML = m_svgXML.replace(before,after);
380 
381 
382  // TODO: there is a lot of switching back and forth between svg, QPixmap and QImage here that I
383  // don't really understand.
384  // render svg tile onto a QImage
385  if (!m_sharedRender->load(colorXML)) {
386  Base::Console().Message("QGIF::buildSvgHatch - failed to load svg string\n");
387  return;
388  }
389 
390  QImage svgImage(round(wTile), round(hTile), QImage::Format_ARGB32);
391  svgImage.fill(Qt::transparent);
392  QPainter painter(&svgImage);
393  if (svgImage.isNull()) {
394  Base::Console().Error("QGIF::buildPixHatch - svgImage is null\n");
395  return;
396  }
397 
398  m_sharedRender->render(&painter);
399 
400  // convert the QImage into a QPixmap
401  QPixmap tilePixmap(round(wTile), round(hTile));
402  tilePixmap = QPixmap::fromImage(svgImage);
403  if (tilePixmap.isNull()) {
404  Base::Console().Error("QGIF::buildPixHatch - tilePixmap is null\n");
405  return;
406  }
407 
408  // layout a field of bitmap tiles big enough to cover this face onto a Qimage
409  QImage tileField(overlayWidth, overlayHeight, QImage::Format_ARGB32);
410  QPointF fieldCenter(overlayWidth / 2.0, overlayHeight / 2.0);
411 
412  tileField.fill(Qt::transparent);
413  QPainter painter2(&tileField);
414  QPainter::RenderHints hints = painter2.renderHints();
415  hints = hints & QPainter::Antialiasing;
416  painter2.setRenderHints(hints);
417  QPainterPath clipper = path();
418  QPointF offset = (fieldCenter - faceCenter);
419  clipper.translate(offset);
420  painter2.setClipPath(clipper);
421 
422  long int tileCount = 0;
423  for (int iw = 0; iw < int(tilesWide); iw++) {
424  for (int ih = 0; ih < int(tilesHigh); ih++) {
425  painter2.drawPixmap(QRectF(iw * wTile, ih * hTile, wTile, hTile), //target rect
426  tilePixmap,
427  QRectF(0, 0, wTile, hTile)); //source rect
428  tileCount++;
429  if (tileCount > m_maxTile) {
430  Base::Console().Warning("Pixmap tile count exceeded: %ld\n",tileCount);
431  break;
432  }
433  }
434  if (tileCount > m_maxTile) {
435  break;
436  }
437  }
438 
439  // turn the QImage field into a pixmap
440  QPixmap fieldPixmap(overlayWidth, overlayHeight);
441  fieldPixmap = QPixmap::fromImage(tileField);
442 
443  // TODO: figure out how to rotate the pixmap without it looking terrible - far worse than the unrotated pixmap. svg hatch exported to svg will not be shown rotated
444  // QTransform xFormPixmap;
445  // xFormPixmap.rotate(m_hatchRotation);
446  // xFormPixmap.translate(getHatchOffset().x, getHatchOffset().y);
447  // m_imageSvgHatchArea->load(fieldPixmap.transformed(xFormPixmap));
448 
449  QPixmap nothing;
450  m_imageSvgHatchArea->setPixmap(nothing);
451  m_imageSvgHatchArea->load(fieldPixmap);
452  m_imageSvgHatchArea->centerAt(faceCenter);
453 }

479 QPixmap QGIFace::textureFromBitmap(std::string fileSpec) const
480 {
481  QPixmap pix;
482 
483  QString qfs(QString::fromUtf8(fileSpec.data(), fileSpec.size()));
484  QFile file(qfs);
485  if (!file.open(QFile::ReadOnly)) {
486  Base::Console().Error("QGIFace could not read %s\n", fileSpec.c_str());
487  return pix;
488  }
489  QByteArray bytes = file.readAll();
490  pix.loadFromData(bytes);
491  if (m_hatchRotation != 0.0) {
492  QTransform rotator;
493  rotator.rotate(m_hatchRotation);
494  pix = pix.transformed(rotator);
495  }
496  return pix;
497 }
