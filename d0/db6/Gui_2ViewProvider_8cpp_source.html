70 void coinRemoveAllChildren(SoGroup *group) {
71  if(!group)
72  return;
73  int count = group->getNumChildren();
74  if(!count)
75  return;
76  FC_TRACE("coin remove all children " << count);
77  SbBool autonotify = group->enableNotify(FALSE);

196 void ViewProvider::eventCallback(void * ud, SoEventCallback * node)
197 {
198  const SoEvent * ev = node->getEvent();
199  auto viewer = static_cast<Gui::View3DInventorViewer*>(node->getUserData());
200  auto self = static_cast<ViewProvider*>(ud);
201  assert(self);
202 
203  try {
204  // Keyboard events
205  if (ev->getTypeId().isDerivedFrom(SoKeyboardEvent::getClassTypeId())) {
206  auto ke = static_cast<const SoKeyboardEvent *>(ev);
207  const SbBool press = ke->getState() == SoButtonEvent::DOWN ? true : false;
208  switch (ke->getKey()) {
209  case SoKeyboardEvent::ESCAPE:
210  if (self->keyPressed (press, ke->getKey())) {
211  node->setHandled();
212  }
213  else if(QApplication::mouseButtons()==Qt::NoButton) {
214  // Because of a Coin bug (https://bitbucket.org/Coin3D/coin/pull-requests/119),
215  // FC may crash if user hits ESC to cancel while still
216  // holding the mouse button while using some SoDragger.
217  // Therefore, we shall ignore ESC while any mouse button is
218  // pressed, until this Coin bug is fixed.
219  if (!press) {
220  // react only on key release
221  // Let first selection mode terminate
222  Gui::Document* doc = Gui::Application::Instance->activeDocument();
223  auto view = static_cast<Gui::View3DInventor*>(doc->getActiveView());
224  if (view)
225  {
226  Gui::View3DInventorViewer* viewer = view->getViewer();
227  if (viewer->isSelecting())
228  {
229  return;
230  }
231  }
232 
233  auto func = new Gui::TimerFunction();
234  func->setAutoDelete(true);
235  func->setFunction([doc]() {
236  doc->resetEdit();
237  });
238  func->singleShot(0);
239  }
240  }
241  else if (press) {
242  FC_WARN("Please release all mouse buttons before exiting editing");
243  }
244  break;
245  default:
246  // call the virtual method
247  if (self->keyPressed (press, ke->getKey()))
248  node->setHandled();
249  break;
250  }
251  }
252  // switching the mouse buttons
253  else if (ev->getTypeId().isDerivedFrom(SoMouseButtonEvent::getClassTypeId())) {
254 
255  const auto event = (const SoMouseButtonEvent *) ev;
256  const int button = event->getButton();
257  const SbBool press = event->getState() == SoButtonEvent::DOWN ? true : false;
258 
259  // call the virtual method
260  if (self->mouseButtonPressed(button,press,ev->getPosition(),viewer))
261  node->setHandled();
262  }
263  else if (ev->getTypeId().isDerivedFrom(SoMouseWheelEvent::getClassTypeId())) {
264  const auto event = (const SoMouseWheelEvent *) ev;
265 
266  if (self->mouseWheelEvent(event->getDelta(), event->getPosition(), viewer))
267  node->setHandled();
268  }
269  // Mouse Movement handling
270  else if (ev->getTypeId().isDerivedFrom(SoLocation2Event::getClassTypeId())) {
271  if (self->mouseMove(ev->getPosition(),viewer))
272  node->setHandled();
273  }
274  }
275  catch (const Base::Exception& e) {
276  Base::Console().Error("Unhandled exception in ViewProvider::eventCallback: %s\n"
277  "(Event type: %s, object type: %s)\n"
278  , e.what(), ev->getTypeId().getName().getString()
279  , self->getTypeId().getName());
280  }
281  catch (const std::exception& e) {
282  Base::Console().Error("Unhandled std exception in ViewProvider::eventCallback: %s\n"
283  "(Event type: %s, object type: %s)\n"
284  , e.what(), ev->getTypeId().getName().getString()
285  , self->getTypeId().getName());
286  }
287  catch (...) {
288  Base::Console().Error("Unhandled unknown C++ exception in ViewProvider::eventCallback"
289  " (Event type: %s, object type: %s)\n"
290  , ev->getTypeId().getName().getString()
291  , self->getTypeId().getName());
292  }
293 }

559 using Graph = boost::adjacency_list <
560  boost::vecS, // class OutEdgeListS : a Sequence or an AssociativeContainer
561  boost::vecS, // class VertexListS : a Sequence or a RandomAccessContainer
562  boost::directedS, // class DirectedS : This is a directed graph
563  boost::no_property, // class VertexProperty:
564  boost::no_property, // class EdgeProperty:
565  boost::no_property, // class GraphProperty:
566  boost::listS // class EdgeListS:
567 >;
568 using Vertex = boost::graph_traits<Graph>::vertex_descriptor;
569 using Edge = boost::graph_traits<Graph>::edge_descriptor;
570 
571 void addNodes(Graph& graph, std::map<SoNode*, Vertex>& vertexNodeMap, SoNode* node)
572 {
573  if (node->getTypeId().isDerivedFrom(SoGroup::getClassTypeId())) {
574  auto group = static_cast<SoGroup*>(node);
575  Vertex groupV = vertexNodeMap[group];
576 
577  for (int i=0; i<group->getNumChildren(); i++) {
578  SoNode* child = group->getChild(i);
579  auto it = vertexNodeMap.find(child);
580 
581  // the child node is not yet added to the map
582  if (it == vertexNodeMap.end()) {
583  Vertex childV = add_vertex(graph);
584  vertexNodeMap[child] = childV;
585  add_edge(groupV, childV, graph);
586  addNodes(graph, vertexNodeMap, child);
587  }
588  // the child is already there, only add the edge then
589  else {
590  add_edge(groupV, it->second, graph);
591  }
592  }
593  }
594 }
595 }

597 bool ViewProvider::checkRecursion(SoNode* node)
598 {
599  if (node->getTypeId().isDerivedFrom(SoGroup::getClassTypeId())) {
600  std::list<Vertex> make_order;
601  Graph graph;
602  std::map<SoNode*, Vertex> vertexNodeMap;
603  Vertex groupV = add_vertex(graph);
604  vertexNodeMap[node] = groupV;
605  addNodes(graph, vertexNodeMap, node);
606 
607  try {
608  boost::topological_sort(graph, std::front_inserter(make_order));
609  }
610  catch (const std::exception&) {
611  return false;
612  }
613  }
614 
615  return true;
616 }

950 int ViewProvider::partialRender(const std::vector<std::string> &elements, bool clear) {
951  if(elements.empty()) {
952  auto node = pcModeSwitch->getChild(_iActualMode);
953  if(node) {
954  FC_LOG("partial render clear");
955  SoSelectionElementAction action(SoSelectionElementAction::None,true);
956  action.apply(node);
957  }
958  }
959  int count = 0;
960  auto path = static_cast<SoFullPath*>(new SoPath);
961  path->ref();
962  SoSelectionElementAction action;
963  action.setSecondary(true);
964  for(auto element : elements) {
965  bool hidden = hasHiddenMarker(element.c_str());
966  if(hidden)
967  element.resize(element.size()-hiddenMarker().size());
968  path->truncate(0);
969  SoDetail *det = nullptr;
970  if(getDetailPath(element.c_str(),path,false,det)) {
971  if(!hidden && !det) {
972  FC_LOG("partial render element not found: " << element);
973  continue;
974  }
975  FC_LOG("partial render (" << path->getLength() << "): " << element);
976  if(!hidden)
977  action.setType(clear?SoSelectionElementAction::Remove:SoSelectionElementAction::Append);
978  else
979  action.setType(clear?SoSelectionElementAction::Show:SoSelectionElementAction::Hide);
980  action.setElement(det);
981  action.apply(path);
982  ++count;
983  }
984  delete det;
985  }
986  path->unref();
987  return count;
988 }

1005 Base::BoundBox3d ViewProvider::getBoundingBox(const char *subname, bool transform, MDIView *view) const {
1006  if(!pcRoot || !pcModeSwitch || pcRoot->findChild(pcModeSwitch)<0)
1007  return Base::BoundBox3d();
1008 
1009  if(!view)
1010  view = Application::Instance->activeView();
1011  auto iview = dynamic_cast<View3DInventor*>(view);
1012  if(!iview) {
1013  auto doc = Application::Instance->activeDocument();
1014  if(doc) {

1023  }

1027 
1028  auto mode = pcModeSwitch->whichChild.getValue();

1037  if(mode < 0)

1050  }
1051  if(path.getLength())
1052  bboxAction.apply(&path);
1053  else
1054  bboxAction.apply(pcRoot);
1055  if(mode < 0)
1056  pcModeSwitch->whichChild = mode;
1057  resetPath.unrefNoDelete();
1058  path.unrefNoDelete();
1059  auto bbox = bboxAction.getBoundingBox();
1060  float minX,minY,minZ,maxX,maxY,maxZ;
1061  bbox.getMax().getValue(maxX,maxY,maxZ);
1062  bbox.getMin().getValue(minX,minY,minZ);
