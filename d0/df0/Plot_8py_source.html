165 def legend(status=True, pos=None, fontsize=None):
166  """Show/Hide the legend from the active plot.
167 
168  Keyword arguments:
169  status -- True if legend must be shown, False otherwise.
170  pos -- Legend position.
171  fontsize -- Font size
172  """
173  plt = getPlot()
174  if not plt:
175  return
176  plt.legend = status
177  if fontsize:
178  plt.legSiz = fontsize
179  # Hide all legends
180  for axes in plt.axesList:
181  axes.legend_ = None
182  # Legend must be activated on last axes
183  axes = plt.axesList[-1]
184  if status:
185  # Setup legend handles and names
186  lines = series()
187  handles = []
188  names = []
189  for l in lines:
190  if l.name is not None:
191  handles.append(l.line)
192  names.append(l.name)
193  # Show the legend (at selected position or at best)
194  if pos:
195  l = axes.legend(handles, names, bbox_to_anchor=pos)
196  plt.legPos = pos
197  else:
198  l = axes.legend(handles, names, loc="best")
199  # Update canvas in order to compute legend data
200  plt.canvas.draw()
201  # Get resultant position
202  try:
203  fax = axes.get_frame().get_extents()
204  except Exception:
205  fax = axes.patch.get_extents()
206  fl = l.get_frame()
207  plt.legPos = (
208  (fl._x + fl._width - fax.x0) / fax.width,
209  (fl._y + fl._height - fax.y0) / fax.height,
210  )
211  # Set fontsize
212  for t in l.get_texts():
213  t.set_fontsize(plt.legSiz)
214  plt.update()
