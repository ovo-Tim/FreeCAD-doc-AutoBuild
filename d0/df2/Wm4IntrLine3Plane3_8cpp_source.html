44 template <class Real>
45 bool IntrLine3Plane3<Real>::Test ()
46 {
47  Real fDdN = m_rkLine.Direction.Dot(m_rkPlane.Normal);
48  if (Math<Real>::FAbs(fDdN) > Math<Real>::ZERO_TOLERANCE)
49  {
50  // The line is not parallel to the plane, so they must intersect.
51  // The line parameter is *not* set, since this is a test-intersection
52  // query.
53  m_iIntersectionType = IT_POINT;
54  return true;
55  }
56 
57  // The line and plane are parallel. Determine if they are numerically
58  // close enough to be coincident.
59  Real fSDistance = m_rkPlane.DistanceTo(m_rkLine.Origin);
60  if (Math<Real>::FAbs(fSDistance) <= Math<Real>::ZERO_TOLERANCE)
61  {
62  m_iIntersectionType = IT_LINE;
63  return true;
64  }
65 
66  m_iIntersectionType = IT_EMPTY;
67  return false;
68 }

70 template <class Real>
71 bool IntrLine3Plane3<Real>::Find ()
72 {
73  Real fDdN = m_rkLine.Direction.Dot(m_rkPlane.Normal);
74  Real fSDistance = m_rkPlane.DistanceTo(m_rkLine.Origin);
75  if (Math<Real>::FAbs(fDdN) > Math<Real>::ZERO_TOLERANCE)
76  {
77  // The line is not parallel to the plane, so they must intersect.
78  m_fLineT = -fSDistance/fDdN;
79  m_iIntersectionType = IT_POINT;
80  return true;
81  }
82 
83  // The Line and plane are parallel. Determine if they are numerically
84  // close enough to be coincident.
85  if (Math<Real>::FAbs(fSDistance) <= Math<Real>::ZERO_TOLERANCE)
86  {
87  // The line is coincident with the plane, so choose t = 0 for the
88  // parameter.
89  m_fLineT = (Real)0.0;
90  m_iIntersectionType = IT_LINE;
91  return true;
92  }
93 
94  m_iIntersectionType = IT_EMPTY;
95  return false;
96 }
