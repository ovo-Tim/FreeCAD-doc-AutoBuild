48 TaskFemConstraintGear::TaskFemConstraintGear(ViewProviderFemConstraint* ConstraintView,

50  const char* pixmapname)
51  : TaskFemConstraintBearing(ConstraintView, parent, pixmapname)
52 {
53  connect(ui->spinDiameter,
54  qOverload<double>(&QDoubleSpinBox::valueChanged),
55  this,
56  &TaskFemConstraintGear::onDiameterChanged);
57  connect(ui->spinForce,
58  qOverload<double>(&QDoubleSpinBox::valueChanged),

62  qOverload<double>(&QDoubleSpinBox::valueChanged),
63  this,
64  &TaskFemConstraintGear::onForceAngleChanged);

66  onButtonDirection(true);
67  });
68  connect(ui->checkReversed, &QCheckBox::toggled, this, &TaskFemConstraintGear::onCheckReversed);
69 
70  // Temporarily prevent unnecessary feature recomputes
71  ui->spinDiameter->blockSignals(true);
72  ui->spinForce->blockSignals(true);
73  ui->spinForceAngle->blockSignals(true);
74  ui->checkReversed->blockSignals(true);
75 
76  // Get the feature data
77  Fem::ConstraintGear* pcConstraint = ConstraintView->getObject<Fem::ConstraintGear>();
78  double dia = pcConstraint->Diameter.getValue();
79  double force = pcConstraint->Force.getValue();
80  double angle = pcConstraint->ForceAngle.getValue();
81  std::vector<std::string> dirStrings = pcConstraint->Direction.getSubValues();
82  QString dir;
83  if (!dirStrings.empty()) {
84  dir = makeRefText(pcConstraint->Direction.getValue(), dirStrings.front());
85  }
86  bool reversed = pcConstraint->Reversed.getValue();
87 
88  // Fill data into dialog elements
89  ui->spinDiameter->setMinimum(0);
90  ui->spinDiameter->setMaximum(FLOAT_MAX);
91  ui->spinDiameter->setValue(dia);
92  ui->spinForce->setMinimum(0);
93  ui->spinForce->setMaximum(FLOAT_MAX);
94  ui->spinForce->setValue(force);
95  ui->spinForceAngle->setMinimum(-360);
96  ui->spinForceAngle->setMaximum(360);
97  ui->spinForceAngle->setValue(angle);
98  ui->lineDirection->setText(dir);
99  ui->checkReversed->setChecked(reversed);
100 
101  // Adjust ui
102  ui->labelDiameter->setVisible(true);
103  ui->spinDiameter->setVisible(true);
104  ui->labelForce->setVisible(true);
105  ui->spinForce->setVisible(true);
106  ui->labelForceAngle->setVisible(true);
107  ui->spinForceAngle->setVisible(true);
108  ui->buttonDirection->setVisible(true);
109  ui->lineDirection->setVisible(true);
110  ui->checkReversed->setVisible(true);
111  ui->checkAxial->setVisible(false);
112 
113  ui->spinDiameter->blockSignals(false);
114  ui->spinForce->blockSignals(false);
115  ui->spinForceAngle->blockSignals(false);
116  ui->checkReversed->blockSignals(false);
117 }

119 void TaskFemConstraintGear::onSelectionChanged(const Gui::SelectionChanges& msg)
120 {
121  TaskFemConstraintBearing::onSelectionChanged(msg);
122 
123  if (msg.Type == Gui::SelectionChanges::AddSelection) {
124  // Don't allow selection in other document
125  if (strcmp(msg.pDocName, ConstraintView->getObject()->getDocument()->getName()) != 0) {
126  return;
127  }
128 
129  if (!msg.pSubName || msg.pSubName[0] == '\0') {
130  return;
131  }
132  std::string subName(msg.pSubName);
133 
134  if (selectionMode == selnone) {
135  return;
136  }
137 
138  std::vector<std::string> references(1, subName);
139  Fem::ConstraintGear* pcConstraint = ConstraintView->getObject<Fem::ConstraintGear>();
140  App::DocumentObject* obj =
141  ConstraintView->getObject()->getDocument()->getObject(msg.pObjectName);
142  Part::Feature* feat = static_cast<Part::Feature*>(obj);
143  TopoDS_Shape ref = feat->Shape.getShape().getSubShape(subName.c_str());
144 
145  if (selectionMode == seldir) {
146  if (subName.substr(0, 4) == "Face") {
147  if (!Fem::Tools::isPlanar(TopoDS::Face(ref))) {
148  QMessageBox::warning(this,
149  tr("Selection error"),
150  tr("Only planar faces can be picked"));
151  return;
152  }
153  }
154  else if (subName.substr(0, 4) == "Edge") {
155  if (!Fem::Tools::isLinear(TopoDS::Edge(ref))) {
156  QMessageBox::warning(this,
157  tr("Selection error"),
158  tr("Only linear edges can be picked"));
159  return;
160  }
161  }
162  else {
163  QMessageBox::warning(this,
164  tr("Selection error"),
165  tr("Only faces and edges can be picked"));
166  return;
167  }
168  pcConstraint->Direction.setValue(obj, references);
169  ui->lineDirection->setText(makeRefText(obj, subName));
170 
171  // Turn off direction selection mode
172  onButtonDirection(false);
173  }
174 
175  Gui::Selection().clearSelection();
176  }
177 }
