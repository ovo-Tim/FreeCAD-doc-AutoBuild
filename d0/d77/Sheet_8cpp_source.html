143 {
144  bool isValid = true;
145  ParameterGrp::handle group = App::GetApplication().GetParameterGroupByPath(
146  "User parameter:BaseApp/Preferences/Mod/Spreadsheet");
147  QString delimiter = QString::fromStdString(group->GetASCII("ImportExportDelimiter", "tab"));
148  QString quoteChar = QString::fromStdString(group->GetASCII("ImportExportQuoteCharacter", "\""));
149  QString escapeChar =
150  QString::fromStdString(group->GetASCII("ImportExportEscapeCharacter", "\\"));
151 
152  delim = delimiter.size() == 1 ? delimiter[0].toLatin1() : '\0';
153  if (delimiter.compare(QLatin1String("tab"), Qt::CaseInsensitive) == 0
154  || delimiter.compare(QLatin1String("\\t"), Qt::CaseInsensitive) == 0) {
155  delim = '\t';
156  }
157  else if (delimiter.compare(QLatin1String("comma"), Qt::CaseInsensitive) == 0) {
158  delim = ',';
159  }
160  else if (delimiter.compare(QLatin1String("semicolon"), Qt::CaseInsensitive) == 0) {
161  delim = ';';
162  }
163  if (delim != '\0' && quoteChar.size() == 1 && escapeChar.size() == 1) {
164  quote = quoteChar[0].toLatin1();
165  escape = escapeChar[0].toLatin1();
166  }
167  else {
168  isValid = false;
169  std::string importExport = errMsg;
170  std::stringstream errStream;
171  errStream << "Invalid spreadsheet Import/Export parameter.\n";
172  if (delim == '\0') {
173  errStream
174  << "Unrecognized delimiter: " << delimiter.toStdString()
175  << " (recognized tokens: \\t, tab, semicolon, comma, or any single character)\n";
176  }
177  if (quoteChar.size() != 1) {
178  errStream << "Invalid quote character: " << quoteChar.toStdString()
179  << " (quote character must be one single character)\n";
180  }
181  if (escapeChar.size() != 1) {
182  errStream << "Invalid escape character: " << escapeChar.toStdString()
183  << " (escape character must be one single character)\n";
184  }
185  errStream << importExport << " not done.\n";
186  errMsg = errStream.str();
187  }
188  return isValid;

204  char delimiter,
205  char quoteChar,
206  char escapeChar)
207 {
208  Base::FileInfo fi(filename);
209  Base::ifstream file(fi, std::ios::in);
210  int row = 0;
211 
212  PropertySheet::AtomicPropertyChange signaller(cells);
213 
214  clearAll();
215 
216  if (file.is_open()) {
217  std::string line;
218 
219  while (std::getline(file, line)) {
220  using namespace boost;
221 
222  try {
223  escaped_list_separator<char> e;
224  int col = 0;
225 
226  if (quoteChar) {
227  e = escaped_list_separator<char>(escapeChar, delimiter, quoteChar);
228  }
229  else {
230  e = escaped_list_separator<char>('\0', delimiter, '\0');
231  }
232

234

241  col++;
242  }
243  }
244  catch (...) {
245  signaller.tryInvoke();
246  return false;
247  }
248 
249  ++row;
250  }
251  file.close();
252  signaller.tryInvoke();
253  return true;
254  }
255  else {

297 bool Sheet::exportToFile(const std::string& filename,
298  char delimiter,
299  char quoteChar,
300  char escapeChar) const
301 {
302  Base::ofstream file;
303  int prevRow = -1, prevCol = -1;
304 
305  Base::FileInfo fi(filename);
306  file.open(fi, std::ios::out | std::ios::ate | std::ios::binary);
307 
308  if (!file.is_open()) {
309  return false;
310  }
311 
312  auto usedCells = cells.getNonEmptyCells();
313  auto i = usedCells.begin();
314 
315  while (i != usedCells.end()) {
316  Property* prop = getProperty(*i);
317 
318  if (prevRow != -1 && prevRow != i->row()) {
319  for (int j = prevRow; j < i->row(); ++j) {
320  file << std::endl;
321  }
322  prevCol = usedCells.begin()->col();
323  }
324  if (prevCol != -1 && i->col() != prevCol) {
325  for (int j = prevCol; j < i->col(); ++j) {
326  file << delimiter;
327  }
328  }
329 
330  std::stringstream field;
331 
332  if (prop->isDerivedFrom((PropertyQuantity::getClassTypeId()))) {
333  field << static_cast<PropertyQuantity*>(prop)->getValue();
334  }
335  else if (prop->isDerivedFrom((PropertyFloat::getClassTypeId()))) {
336  field << static_cast<PropertyFloat*>(prop)->getValue();
337  }
338  else if (prop->isDerivedFrom((PropertyInteger::getClassTypeId()))) {
339  field << static_cast<PropertyInteger*>(prop)->getValue();
340  }
341  else if (prop->isDerivedFrom((PropertyString::getClassTypeId()))) {
342  field << static_cast<PropertyString*>(prop)->getValue();
343  }
344  else {
345  assert(0);
346  }
347 
348  std::string str = field.str();
349 
350  if (quoteChar
351  && str.find_first_of(std::string(quoteChar, delimiter)) != std::string::npos) {
352  writeEscaped(str, quoteChar, escapeChar, file);
353  }
354  else {
355  file << str;
356  }
357 
358  prevRow = i->row();
359  prevCol = i->col();
360  ++i;
361  }
362  file << std::endl;
363  file.close();
364 
365  return true;
366 }

756 void Sheet::updateProperty(CellAddress key)
757 {
758  Cell* cell = getCell(key);
759 
760  if (cell) {
761  std::unique_ptr<Expression> output;
762  const Expression* input = cell->getExpression();
763 
764  if (input) {
765  CurrentAddressLock lock(currentRow, currentCol, key);
766  output.reset(input->eval());
767  }
768  else {
769  std::string s;
770 
771  if (cell->getStringContent(s) && !s.empty()) {
772  output = std::make_unique<StringExpression>(this, s);
773  }
774  else {
775  this->removeDynamicProperty(key.toString().c_str());
776  return;
777  }
778  }
779 
780  /* Eval returns either NumberExpression or StringExpression, or
781  * PyObjectExpression objects */
782  auto number = freecad_dynamic_cast<NumberExpression>(output.get());
783  if (number) {
784  long l;
785  auto constant = freecad_dynamic_cast<ConstantExpression>(output.get());
786  if (constant && !constant->isNumber()) {
787  Base::PyGILStateLocker lock;
788  setObjectProperty(key, constant->getPyValue());
789  }
790  else if (!number->getUnit().isEmpty()) {
791  setQuantityProperty(key, number->getValue(), number->getUnit());
792  }
793  else if (number->isInteger(&l)) {
794  setIntegerProperty(key, l);
795  }
796  else {
797  setFloatProperty(key, number->getValue());
798  }
799  }
800  else {

810  }
811  else {
812  setObjectProperty(key, Py::Object());
813  }

815  }

1036  cells.recomputeDependencies(cellError);

1044  while (!workQueue.empty()) {

1052  VertexIndexList[res.first->second] = currPos;
1053  }
1054 
1055  // Process cells that depend on the current cell
1056  for (auto& dep : providesTo(currPos)) {
1057  auto resDep = VertexList.emplace(dep, Vertex());
1058  if (resDep.second) {
1059  resDep.first->second = add_vertex(graph);
1060  VertexIndexList[resDep.first->second] = dep;
1061  if (dirtyCells.insert(dep).second) {
1062  workQueue.push_back(dep);
1063  }
1064  }
1065  // Add edge to graph to signal dependency
1066  add_edge(res.first->second, resDep.first->second, graph);
1067  }
1068  }
1069  // Compute cells
1070  std::list<Vertex> make_order;
1071  // Sort graph topologically to find evaluation order
1072  try {
1073  boost::topological_sort(graph, std::front_inserter(make_order));
1074  // Recompute cells
1075  FC_LOG("recomputing " << getFullName());
1076  for (auto& pos : make_order) {
1077  const auto& addr = VertexIndexList[pos];
1078  FC_TRACE(addr.toString());

1080  }
1081  }
1082  catch (std::exception&) {

1084  Cell* cell = cells.getValue(v.first);
1085  // Mark as erroneous
1086  if (cell) {
1087  cellErrors.insert(v.first);
1088  cell->setException("Pending computation due to cyclic dependency", true);
1089  cellUpdated(v.first);
1090  }
1091  }
1092 
1093  // Try to be more user friendly by finding individual loops
1094  while (!dirtyCells.empty()) {
1095 
1096  std::deque<CellAddress> workQueue;
1097  DependencyList graph;
1098  std::map<CellAddress, Vertex> VertexList;
1099  std::map<Vertex, CellAddress> VertexIndexList;
1100

1102  workQueue.push_back(currentAddr);
1103  dirtyCells.erase(dirtyCells.begin());

1119  if (resDep.second) {

1126  add_edge(res.first->second, resDep.first->second, graph);
1127  }
1128  }

1131  try {

1134  catch (std::exception&) { // TODO: evaluate using a more specific exception (not_a_dag)

1153  if (cell) {
1154  cell->setException(msg.c_str(), true);
1155  cellUpdated(v.first);
1156  }
1157  }
1158  }
1159  }
