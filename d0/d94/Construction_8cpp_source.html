499 CLine Tanto(int AT, const Circle& c, const Point& p)
500 {
501  // CLine tangent to a circle through a point
502  Vector2d v(p, c.pc);
503  double d = v.magnitude();
504  CLine s(p, ~v, false); // initialise cline
505 
506  if (d < TOLERANCE || d < fabs(c.radius) - TOLERANCE) { // point inside circle ?
507  return INVALID_CLINE;
508  }
509  else {
510  if (d > fabs(c.radius) + TOLERANCE) { // point outside circle
511  v.Rotate(sqrt((d - c.radius) * (d + c.radius)), -AT * c.radius);
512  s.v = v;
513  }
514  }
515  s.Normalise();
516  return s;
517 }

573 Point Intof(int NF, const CLine& s, const Circle& c, Point& otherInters)
574 {
575  // inters of cline & circle eg. p1 = Intof(NEARINT, s1, c1);
576  // otherInters returns the other intersection
577 #if 1
578  // solving x = x0 + dx * t x = y0 + dy * t
579  // x = xc + R * cos(a) y = yc + R * sin(a) for t
580  // gives :- t (dx + dy) + 2t(dx*dx0 + dy*dy0) + (x0-xc) + (y0-yc) - R = 0
581  int nRoots;
582  double t, tFar, tNear, tOther;
583  Vector2d v0(c.pc, s.p);
584  if ((nRoots =
585  quadratic(1, 2 * (v0 * s.v), v0.magnitudesqd() - c.radius * c.radius, tFar, tNear))
586  != 0) {
587  if (nRoots == 2 && NF == NEARINT) {
588  t = tNear;
589  tOther = tFar;
590  }
591  else {
592  t = tFar;
593  tOther = (nRoots == 2) ? tNear : tFar;
594  }
595  otherInters = s.v * tOther + s.p;
596  return s.v * t + s.p;
597  }
598  return INVALID_POINT;
599 }
600 #else
601  // geometric solution - this is similar to the peps method, and it may offer better tolerancing
602  // than above??
603  Point intof;
604  CLine normal = Normal(s, c.pc);
605  intof = s.Intof(normal);
606  double d = intof.Dist(c.pc);
607 
608  if (fabs(d - c.radius) < TOLERANCE) { // tangent (near enough for non-large radius I suppose?)
609  return intof;
610  }
611

613  return INVALID_POINT;
614  }
615

617  if (q < 0) { // line inside tolerance

619  }
620 
621  return Along(s, -(double)NF * sqrt(q), intof); // 2 intersections (return near/far case)
622 }

653 int Intof(const Circle& c0, const Circle& c1, Point& pLeft, Point& pRight)
654 {
655  // inters of 2 circles
656  // returns the number of intersctions
657  Vector2d v(c0.pc, c1.pc);
658  double d = v.normalise();
659  if (d < TOLERANCE) { // co-incident circles
660  return 0;
661  }
662 
663  double sum = fabs(c0.radius) + fabs(c1.radius);
664  double diff = fabs(fabs(c0.radius) - fabs(c1.radius));
665  if (d > sum + TOLERANCE || d < diff - TOLERANCE) {
666  return 0;
667  }
668 
669  // dist from centre of this circle to mid intersection
670  double d0 = 0.5 * (d + (c0.radius + c1.radius) * (c0.radius - c1.radius) / d);
671  if (d0 - c0.radius > TOLERANCE) { // circles don't intersect
672  return 0;
673  }
674 
675  double h = (c0.radius - d0) * (c0.radius + d0); // half distance between intersects squared
676  if (h < 0) {
677  d0 = c0.radius; // tangent
678  }
679  pLeft = v * d0 + c0.pc; // mid-point of intersects

864 {
865  // returns the absolute included angle between 2 vectors in the direction of dir ( 1=acw -1=cw)
866  double inc_ang = v0 * v1;
867  if (inc_ang > 1. - UNIT_VECTOR_TOLERANCE) {
868  return 0;
869  }
870  if (inc_ang < -1. + UNIT_VECTOR_TOLERANCE) {

920 int quadratic(double a, double b, double c, double& x0, double& x1)
921 {
922  // solves quadratic equation ax + bx + c = 0
923  // returns number of real roots
924  // double epsilon = 1.0e-6;
