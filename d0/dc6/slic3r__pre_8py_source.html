53 def parse(inputstring):
54  "parse(inputstring): returns a parsed output string"
55  print("preprocessing...")
56 
57  # split the input by line
58  lines = inputstring.split("\n")
59  output = ""
60  lastcommand = None
61 
62  for l in lines:
63  # remove any leftover trailing and preceding spaces
64  l = l.strip()
65  if not l:
66  # discard empty lines
67  continue
68  if l[0].upper() in ["N"]:
69  # remove line numbers
70  l = l.split(" ", 1)[1]
71  if ";" in l:
72  # replace ; comments with ()
73  l = l.replace(";", "(")
74  l = l + ")"
75  if l[0].upper() in ["G", "M", "("]:
76  # found a G or M command: we store it
77  output += l + "\n"
78  last = l[0].upper()
79  for c in l[1:]:
80  if not c.isdigit():
81  break
82  else:
83  last += c
84  lastcommand = last
85  elif lastcommand:
86  # no G or M command: we repeat the last one
87  output += lastcommand + " " + l + "\n"
88
