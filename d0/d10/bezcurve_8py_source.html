170  def symmetricpoles(cls,knot, p1, p2):
171  """Make two poles symmetric respective to the knot."""
172  p1h = App.Vector(p1)
173  p2h = App.Vector(p2)
174  p1h.multiply(0.5)

176  return ( knot+p1h-p2h , knot+p2h-p1h )

178  @classmethod
179  def tangentpoles(cls,knot, p1, p2,allowsameside=False):
180  """Make two poles have the same tangent at knot."""
181  p12n = p2.sub(p1)
182  p12n.normalize()
183  p1k = knot-p1
184  p2k = knot-p2
185  p1k_= App.Vector(p12n)
186  kon12=(p1k * p12n)
187  if allowsameside or not (kon12 < 0 or p2k * p12n > 0):# instead of moving
188  p1k_.multiply(kon12)
189  pk_k = knot - p1 - p1k_
190  return (p1 + pk_k, p2 + pk_k)
191  else:
192  return cls.symmetricpoles(knot, p1, p2)
