309  subresult = []
310  # do not create if a child with same stepid already exists
311  if not element.id() in [
312  getattr(c, "StepId", 0) for c in get_parent_objects(parent)
313  ]:

322  buildings = [

324  ]

326  subresult.extend(create_child(child, building))

328  # force-create contained windows too if we just created an opening
329  windows = [

333  ]
334  for window in windows:

583  else:

586  obj.addProperty("App::PropertyLinkList", "Group", "Base")
587  if "ShapeMode" not in obj.PropertiesList:
588  obj.addProperty("App::PropertyEnumeration", "ShapeMode", "Base")
589  shapemodes = [
590  "Shape",
591  "Coin",
592  "None",

596  obj.ShapeMode = shapemodes
597  obj.ShapeMode = shapemode
598  if not obj.isDerivedFrom("Part::Feature"):
599  obj.setPropertyStatus("ShapeMode", "Hidden")
600  if ifcentity.is_a("IfcProduct"):
601  obj.addProperty("App::PropertyLink", "Type", "IFC")
602  attr_defs = ifcentity.wrapped_data.declaration().as_entity().all_attributes()
603  try:
604  info_ifcentity = ifcentity.get_info()
605  except:
606  # slower but no errors
607  info_ifcentity = get_elem_attribs(ifcentity)
608  for attr, value in info_ifcentity.items():
609  if attr == "type":
610  attr = "Class"
611  elif attr == "id":
612  attr = "StepId"
613  elif attr == "Name":
614  continue

616  continue

618  data_type = (
619  ifcopenshell.util.attribute.get_primitive_type(attr_def)
620  if attr_def
621  else None
622  )
623  if attr == "Class":
624  # main enum property, not saved to file
625  if attr not in obj.PropertiesList:
626  obj.addProperty("App::PropertyEnumeration", attr, "IFC")
627  obj.setPropertyStatus(attr, "Transient")
628  # to avoid bug/crash: we populate first the property with only the
629  # class, then we add the sibling classes
