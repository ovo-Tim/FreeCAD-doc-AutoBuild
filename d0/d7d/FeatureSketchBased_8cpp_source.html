169 TopoShape ProfileBased::getTopoShapeVerifiedFace(bool silent,
170  [[maybe_unused]]bool doFit, // TODO: Remove parameter
171  bool allowOpen,
172  const App::DocumentObject* profile,
173  const std::vector<std::string>& _subs) const
174 {
175  auto obj = profile ? profile : Profile.getValue();
176  if (!obj || !obj->getNameInDocument()) {
177  if (silent) {
178  return TopoShape();
179  }
180  throw Base::ValueError("No profile linked");
181  }
182  const auto& subs = profile ? _subs : Profile.getSubValues();
183  try {
184  TopoShape shape;
185  if (AllowMultiFace.getValue()) {
186  if (subs.empty()) {
187  shape = Part::Feature::getTopoShape(obj);
188  }
189  else {
190  std::vector<TopoShape> shapes;
191  for (auto& sub : subs) {
192  auto subshape =
193  Part::Feature::getTopoShape(obj, sub.c_str(), /*needSubElement*/ true);
194  if (subshape.isNull()) {
195  FC_THROWM(Base::CADKernelError,
196  "Sub shape not found: " << obj->getFullName() << "." << sub);
197  }
198  shapes.push_back(subshape);
199  }
200  shape.makeElementCompound(shapes);
201  }
202  }
203  else {
204  std::string sub;
205  if (!obj->getTypeId().isDerivedFrom(Part::Part2DObject::getClassTypeId())) {
206  if (!subs.empty()) {
207  sub = subs[0];
208  }
209  }
210  shape = Part::Feature::getTopoShape(obj, sub.c_str(), !sub.empty());
211  }
212  if (shape.isNull()) {
213  if (silent) {
214  return shape;
215  }
216  throw Base::CADKernelError("Linked shape object is empty");
217  }
218  TopoShape openshape;
219  if (!shape.hasSubShape(TopAbs_FACE)) {
220  try {
221  if (!shape.hasSubShape(TopAbs_WIRE)) {
222  shape = shape.makeElementWires();
223  }
224  if (shape.hasSubShape(TopAbs_WIRE)) {
225  shape.Hasher = getDocument()->getStringHasher();
226  if (allowOpen) {
227  std::vector<TopoShape> openwires;
228  std::vector<TopoShape> wires;
229  for (auto& wire : shape.getSubTopoShapes(TopAbs_WIRE)) {
230  if (!wire.isClosed()) {
231  openwires.push_back(wire);
232  }
233  else {
234  wires.push_back(wire);
235  }
236  }
237  if (openwires.size()) {
238  openshape.makeElementCompound(
239  openwires,
240  nullptr,
241  TopoShape ::SingleShapeCompoundCreationPolicy::returnShape);
242  if (wires.empty()) {
243  shape = TopoShape();
244  }
245  else {
246  shape.makeElementCompound(
247  wires,
248  nullptr,
249  TopoShape ::SingleShapeCompoundCreationPolicy::returnShape);
250  }
251  }
252  }
253  if (!shape.isNull()) {
254  if (AllowMultiFace.getValue()) {
255  shape = shape.makeElementFace(); // default to use FaceMakerBullseye
256  }
257  else {
258  shape = shape.makeElementFace(nullptr, "Part::FaceMakerCheese");
259  }
260  }
261  }
262  }
263  catch (const Base::Exception&) {
264  if (silent) {
265  return TopoShape();
266  }
267  throw;
268  }
269  catch (const Standard_Failure&) {
270  if (silent) {
271  return TopoShape();
272  }
273  throw;
274  }
275  }
276  int count = shape.countSubShapes(TopAbs_FACE);
277  if (!count && !allowOpen) {
278  if (silent) {
279  return TopoShape();
280  }
281  throw Base::CADKernelError("Cannot make face from profile");
282  }
283 
284  // Toponaming April 2024: This appears to be new feature, not TNP:
285 // if (doFit && (std::abs(Fit.getValue()) > Precision::Confusion()
286 // || std::abs(InnerFit.getValue()) > Precision::Confusion())) {
287 //
288 // if (!shape.isNull())
289 // shape = shape.makeElementOffsetFace(Fit.getValue(),
290 // InnerFit.getValue(),
291 // static_cast<Part::TopoShape::JoinType>(FitJoin.getValue()),
292 // static_cast<Part::TopoShape::JoinType>(InnerFitJoin.getValue()));
293 // if (!openshape.isNull())
294 // openshape.makeElementOffset2D(Fit.getValue());
295 // }
296 
297  if (!openshape.isNull()) {
298  if (shape.isNull()) {
299  shape = openshape;
300  }
301  else {
302  shape.makeElementCompound({shape, openshape});
303  }
304  }
305  if (count > 1) {
306  if (AllowMultiFace.getValue()
307  || obj->isDerivedFrom(Part::Part2DObject::getClassTypeId())) {
308  return shape;
309  }
310  FC_WARN("Found more than one face from profile");
311  }
312  if (!openshape.isNull()) {
313  return shape;
314  }
315  if (count) {
316  return shape.getSubTopoShape(TopAbs_FACE, 1);
317  }
318  return shape;
319  }
320  catch (Standard_Failure&) {
321  if (silent) {
322  return TopoShape();
323  }
324  throw;
325  }

329 TopoDS_Shape ProfileBased::getVerifiedFace(bool silent) const {
330 
331  App::DocumentObject* result = Profile.getValue();
332  const char* err = nullptr;
333  std::string _err;
334 
335  if (!result) {
336  err = "No profile linked";
337  }
338  else if (AllowMultiFace.getValue()) {
339  try {
340  auto shape = getProfileShape();
341  if (shape.isNull())
342  err = "Linked shape object is empty";
343  else {
344  auto faces = shape.getSubTopoShapes(TopAbs_FACE);
345  if (faces.empty()) {
346  if (!shape.hasSubShape(TopAbs_WIRE))
347  shape = shape.makeWires();
348  if (shape.hasSubShape(TopAbs_WIRE))
349  shape = shape.makeFace(nullptr, "Part::FaceMakerBullseye");
350  else
351  err = "Cannot make face from profile";
352  }
353  else if (faces.size() == 1)
354  shape = faces.front();
355  else
356  shape = TopoShape().makeCompound(faces);
357  }
358  if (!err)
359  return shape.getShape();
360  }
361  catch (Standard_Failure& e) {
362  _err = e.GetMessageString();
363  err = _err.c_str();
364  }
365  }
366  else {
367  if (result->isDerivedFrom<Part::Part2DObject>()) {
368 
369  auto wires = getProfileWires();
370  return Part::FaceMakerCheese::makeFace(wires);
371  }
372  else if (result->isDerivedFrom<Part::Feature>()) {
373  if (Profile.getSubValues().empty())
374  err = "Linked object has no subshape specified";
375  else {
376 
377  const Part::TopoShape& shape = Profile.getValue<Part::Feature*>()->Shape.getShape();
378  TopoDS_Shape sub = shape.getSubShape(Profile.getSubValues()[0].c_str());
379  if (sub.ShapeType() == TopAbs_FACE)
380  return TopoDS::Face(sub);
381  else if (sub.ShapeType() == TopAbs_WIRE) {
382 
383  auto wire = TopoDS::Wire(sub);
384  if (!wire.Closed())
385  err = "Linked wire is not closed";
386  else {
387  BRepBuilderAPI_MakeFace mk(wire);
388  mk.Build();
389  return TopoDS::Face(mk.Shape());
390  }
391  }
392  else
393  err = "Linked Subshape cannot be used";
394  }
395  }
396  else
397  err = "Linked object is neither Sketch, Part2DObject or Part::Feature";
398  }
399 
400  if (!silent && err) {
401  throw Base::RuntimeError(err);
402  }
403 
404  return TopoDS_Face();
405 }

431 std::vector<TopoDS_Wire> ProfileBased::getProfileWires() const {
432  std::vector<TopoDS_Wire> result;
433 
434  if (!Profile.getValue() || !Profile.getValue()->isDerivedFrom(Part::Feature::getClassTypeId()))
435  throw Base::TypeError("No valid profile linked");
436 
437  TopoDS_Shape shape;
438  if (Profile.getValue()->isDerivedFrom(Part::Part2DObject::getClassTypeId()))
439  shape = Profile.getValue<Part::Part2DObject*>()->Shape.getValue();
440  else {
441  if (Profile.getSubValues().empty())
442  throw Base::ValueError("No valid subelement linked in Part::Feature");
443 
444  shape = Profile.getValue<Part::Feature*>()->Shape.getShape().getSubShape(Profile.getSubValues().front().c_str());
445  }
446 
447  if (shape.IsNull())
448  throw Base::ValueError("Linked shape object is empty");
449 
450  // this is a workaround for an obscure OCC bug which leads to empty tessellations
451  // for some faces. Making an explicit copy of the linked shape seems to fix it.
452  // The error almost happens when re-computing the shape but sometimes also for the
453  // first time
454  BRepBuilderAPI_Copy copy(shape);
455  shape = copy.Shape();
456  if (shape.IsNull())
457  throw Base::ValueError("Linked shape object is empty");
458 
459  TopExp_Explorer ex;
460  for (ex.Init(shape, TopAbs_WIRE); ex.More(); ex.Next()) {
461  result.push_back(TopoDS::Wire(ex.Current()));
462  }
463  if (result.empty()) // there can be several wires
464  throw Base::ValueError("Linked shape object is not a wire");
465 
466  return result;
467 }

502 TopoDS_Face ProfileBased::getSupportFace(const Part::Part2DObject* sketch) const
503 {
504  if (sketch && sketch->MapMode.getValue() == Attacher::mmFlatFace
505  && sketch->AttachmentSupport.getValue()) {
506  const auto& AttachmentSupport = sketch->AttachmentSupport;
507  App::DocumentObject* ref = AttachmentSupport.getValue();
508 
509  Part::Feature* part = dynamic_cast<Part::Feature*>(ref);
510  if (part) {
511  const std::vector<std::string>& sub = AttachmentSupport.getSubValues();
512  assert(sub.size() == 1);
513 
514  if (sub.at(0).empty()) {
515  // This seems to happen when sketch is on a datum plane
516  return TopoDS::Face(Feature::makeShapeFromPlane(sketch));
517  }
518 
519  // get the selected sub shape (a Face)
520  const Part::TopoShape& shape = part->Shape.getShape();
521  if (shape.getShape().IsNull()) {
522  throw Base::ValueError("Sketch support shape is empty!");
523  }
524 
525  TopoDS_Shape sh = shape.getSubShape(sub[0].c_str());
526  if (sh.IsNull()) {
527  throw Base::ValueError("Null shape in SketchBased::getSupportFace()!");
528  }
529 
530  const TopoDS_Face face = TopoDS::Face(sh);

534

537  throw Base::TypeError("No planar face in SketchBased::getSupportFace()!");

1393

1395  auto obj = getVerifiedObject(true);
1396  if (!obj)
1397  return SketchVector;

1400  if (obj->isDerivedFrom(Part::Part2DObject::getClassTypeId())) {
1401  Base::Placement SketchPos = obj->Placement.getValue();
1402  Base::Rotation SketchOrientation = SketchPos.getRotation();

1405  }
1406 
1407  // For newer version, do not do fitting, as it may flip the face normal for
1408  // some reason.
1409  TopoShape shape = getTopoShapeVerifiedFace(true); //, _ProfileBasedVersion.getValue() <= 0);
1410 
1411  gp_Pln pln;
1412  if (shape.findPlane(pln)) {
1413  gp_Dir dir = pln.Axis().Direction();

1415  }
1416 
1417  if (shape.hasSubShape(TopAbs_EDGE)) {
1418  // Find the first planar face that contains the edge, and return the plane normal
1419  TopoShape objShape = Part::Feature::getTopoShape(obj);
1420  for (int idx : objShape.findAncestors(shape.getSubShape(TopAbs_EDGE, 1), TopAbs_FACE)) {
1421  if (objShape.getSubTopoShape(TopAbs_FACE, idx).findPlane(pln)) {
1422  gp_Dir dir = pln.Axis().Direction();
1423  return Base::Vector3d(dir.X(), dir.Y(), dir.Z());
1424  }
1425  }
1426  }
1427 
1428  // If no planar face, try to use the normal of the center of the first face.
1429  if (shape.hasSubShape(TopAbs_FACE)) {
1430  TopoDS_Face face = TopoDS::Face(shape.getSubShape(TopAbs_FACE, 1));
1431  BRepAdaptor_Surface adapt(face);
1432  double u =
1433  adapt.FirstUParameter() + (adapt.LastUParameter() - adapt.FirstUParameter()) / 2.;
1434  double v =
1435  adapt.FirstVParameter() + (adapt.LastVParameter() - adapt.FirstVParameter()) / 2.;
1436  BRepLProp_SLProps prop(adapt, u, v, 2, Precision::Confusion());
1437  if (prop.IsNormalDefined()) {
1438  gp_Pnt pnt;
1439  gp_Vec vec;
1440  // handles the orientation state of the shape
1441  BRepGProp_Face(face).Normal(u, v, pnt, vec);
1442  return Base::Vector3d(vec.X(), vec.Y(), vec.Z());
1443  }
1444  }
1445 
1446  if (!shape.hasSubShape(TopAbs_EDGE)) {
1447  return SketchVector;
1448  }
1449 
1450  // If the shape is a line, then return an arbitrary direction that is perpendicular to the line
1451  auto geom = Part::Geometry::fromShape(shape.getSubShape(TopAbs_EDGE, 1), true);
1452  auto geomLine = Base::freecad_dynamic_cast<Part::GeomLine>(geom.get());
1453  if (geomLine) {
1454  Base::Vector3d dir = geomLine->getDir();
1455  double x = std::fabs(dir.x);
1456  double y = std::fabs(dir.y);
1457  double z = std::fabs(dir.z);
1458  if (x > y && x > z && x > 1e-7) {
1459  if (y + z < 1e-7) {
1460  return Base::Vector3d(0, 0, 1);
1461  }
1462  dir.x = -(dir.z + dir.y) / dir.x;
1463  }
1464  else if (y > x && y > z && y > 1e-7) {
1465  if (x + z < 1e-7) {
1466  return Base::Vector3d(0, 0, 1);
1467  }
1468  dir.y = -(dir.z + dir.x) / dir.y;
1469  }
1470  else if (z > 1e-7) {
1471  if (x + y < 1e-7) {
1472  return Base::Vector3d(1, 0, 0);
1473  }
1474  dir.z = -(dir.x + dir.y) / dir.z;
1475  }
1476  else {
1477  return SketchVector;
1478  }
1479  return dir.Normalize();
1480  }
1481  return SketchVector;
