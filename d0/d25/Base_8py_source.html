79 class ObjectOp(object):
80  """
81  Base class for proxy objects of all Path operations.
82 
83  Use this class as a base class for new operations. It provides properties
84  and some functionality for the standard properties each operation supports.
85  By OR'ing features from the feature list an operation can select which ones
86  of the standard features it requires and/or supports.
87 
88  The currently supported features are:
89  FeatureTool ... Use of a ToolController
90  FeatureDepths ... Depths, for start, final
91  FeatureHeights ... Heights, safe and clearance
92  FeatureStartPoint ... Supports setting a start point
93  FeatureFinishDepth ... Operation supports a finish depth
94  FeatureStepDown ... Support for step down
95  FeatureNoFinalDepth ... Disable support for final depth modifications
96  FeatureBaseVertexes ... Base geometry support for vertexes
97  FeatureBaseEdges ... Base geometry support for edges
98  FeatureBaseFaces ... Base geometry support for faces
99  FeatureLocations ... Base location support
100  FeatureCoolant ... Support for operation coolant
101  FeatureDiameters ... Support for turning operation diameters
102 
103  The base class handles all base API and forwards calls to subclasses with
104  an op prefix. For instance, an op is not expected to overwrite onChanged(),
105  but implement the function opOnChanged().
106  If a base class overwrites a base API function it should call the super's
107  implementation - otherwise the base functionality might be broken.
108  """

632  """updateDepths(obj) ... base implementation calculating depths depending on base geometry.
633  Should not be overwritten."""

635  def faceZmin(bb, fbb):
636  if fbb.ZMax == fbb.ZMin and fbb.ZMax == bb.ZMax: # top face
637  return fbb.ZMin
638  elif fbb.ZMax > fbb.ZMin and fbb.ZMax == bb.ZMax: # vertical face, full cut
639  return fbb.ZMin
640  elif fbb.ZMax > fbb.ZMin and fbb.ZMin > bb.ZMin: # internal vertical wall
641  return fbb.ZMin
642  elif fbb.ZMax == fbb.ZMin and fbb.ZMax > bb.ZMin: # face/shelf
643  return fbb.ZMin
644  return bb.ZMin
