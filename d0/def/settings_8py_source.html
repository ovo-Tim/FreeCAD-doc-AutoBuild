24 """ Query FEM specific settings including solver settings.
25 
26 Query settings from the hierarchically organized settings/parameter system of
27 FreeCAD related to the FEM module. The collection of independent functions use
28 the settings system as a backend and expose a easy to use interface for other

53 class DirSetting:
54  """Enum of possible directory setting values.
55 
56  Strings used to indicate the solver directory setting set in FreeCADs
57  setting system. Returned by :func:`get_dir_setting` for that purpose. There
58  are three different possible values:
59 
60  :cvar TEMPORARY:
61  Let FreeCAD manage (create, delete) the working directories for all
62  solver. Use temporary directories.
63 
64  :cvar BESIDE:
65  Create a directory in the same folder in which the FCStd file of the
66  document is located. Use Subfolder for each solver (e.g. for a file
67  ./mydoc.FCStd and a solver with the label Elmer002 use
68  ./mydoc/Elmer002).
69 
70  :cvar CUSTOM:
71  Use directory set below. Create own subdirectory for every solver. Name
72  directory after the solver label prefixed with the document name.
73  """
74 
75  TEMPORARY = "temporary"
76  BESIDE = "beside"
77  CUSTOM = "custom"

85 def get_binary(name, silent=False):
86  """Find binary of solver *name* honoring user settings.
87 
88  Return the specific path set by the user in FreeCADs settings/parameter
89  system if set or the default binary name if no specific path is set. If no
90  path was found because the solver *name* is not supported ``None`` is
91  returned.
92  This method does not check whether the binary actually exists and is callable.
93  That check is done in DlgSettingsFem_Solver_Imp.cpp
94 
95  :param name: solver id as a ``str`` (see :mod:`femsolver.settings`)
96  :param silent: whether to output error if binary not found
97  """
98  if name in _SOLVER_PARAM:
99  binary = _SOLVER_PARAM[name].get_binary(silent)
100  return binary
101  else:
102  if not silent:
103  FreeCAD.Console.PrintError(
104  "Settings solver name: {} not found in "
105  "solver settings modules _SOLVER_PARAM dirctionary.\n".format(name)
106  )
107  return None

153 def get_dir_setting():
154  """Return directory setting set by the user.
155 
156  Return one of the three possible values of the :class:`DirSetting` enum
157  depending on the setting set in FreeCAD parameter system. Result dependes
158  on the values of :term:`General/UseTempDirectory`,
159  :term:`General/UseBesideDirectory` and :term:`General/UseCustomDirectory`.
160  """
161  param_group = FreeCAD.ParamGet(_GENERAL_PARAM)
162  if param_group.GetBool("UseBesideDirectory"):
163  return DirSetting.BESIDE
164  elif param_group.GetBool("UseCustomDirectory"):
165  return DirSetting.CUSTOM
166  return DirSetting.TEMPORARY

169 def get_default_solver():
170  """Return default solver name."""
171  solver_map = {0: "None"}
172  if get_binary("Calculix", True):
173  solver_map[1] = "CalculiXCcxTools"
174  if get_binary("ElmerSolver", True):
175  solver_map[len(solver_map)] = "Elmer"
176  if get_binary("Mystran", True):
177  solver_map[len(solver_map)] = "Mystran"
178  if get_binary("Z88", True):
179  solver_map[len(solver_map)] = "Z88"
180  param_group = FreeCAD.ParamGet(_GENERAL_PARAM)
181  return solver_map[param_group.GetInt("DefaultSolver", 0)]

184 class _SolverDlg:
185  """Internal query logic for solver specific settings.
186 
187  Each instance queries settings for one specific solver (e.g. Elmer) common
188  among all solvers. To clarify: There are a few settings that are useful
189  for every solver (e.g. where to find the solver binary) but the value and
190  the FreeCAD parameter path is different for each one. A instance of this
191  class contains all the solver specific paths needed. The settings can be
192  queried via the methods which use those path members to query the value for
193  the specific solver.
194 
195  :ivar default:
196  Default binary name as a string preferably without a prefix path to
197  make it more generic (e.g. "ccx"). This only works if the binary can be
198  found via the PATH environment variable on linux or similar mechanisms
199  on other operating systems. Used if nothing else is specified by the
200  user.
201 
202  :ivar param_path:
203  Parent param path (FreeCADs settings/parameter system) that contains
204  all settings for the specific solver.
205 
206  :ivar use_default:
207  Param path identifying the "use_default" setting. Only specifie the
208  last part as the *param_path* is prepended to this value.
209 
210  :ivar custom_path:
211  Param path identifying the "custom_path" setting. Only specifie the
212  last part as the *param_path* is prepended to this value.
213  """

225  def get_binary(self, silent=False):
226 
227  # set the binary path to the FreeCAD defaults
228  # ATM pure unix shell commands without path names are used as standard
229  # TODO the binaries provided with the FreeCAD distribution should be found
230  # without any additional user input
231  # see ccxttols, it works for Windows and Linux there
232  binary = self.default
233  FreeCAD.Console.PrintLog(f"Solver binary path default: {binary} \n")
234 
235  # check if use_default is set to True
236  # if True the standard binary path will be overwritten with a user binary path
237  if self.param_group.GetBool(self.use_default, True) is False:
238  binary = self.param_group.GetString(self.custom_path)
239  FreeCAD.Console.PrintLog(f"Solver binary path user setting: {binary} \n")
240 
241  # get the whole binary path name for the given command or binary path and return it
242  # None is returned if the binary has not been found
243  # The user does not know what exactly has going wrong.
244  from shutil import which as find_bin
245 
246  the_found_binary = find_bin(binary)
247  if (the_found_binary is None) and (not silent):
248  FreeCAD.Console.PrintError(
249  f"The binary has not been found. Full binary search path: {binary}\n"
250  )
251  else:
252  FreeCAD.Console.PrintLog(f"Found solver binary path: {the_found_binary}\n")
253  return the_found_binary
