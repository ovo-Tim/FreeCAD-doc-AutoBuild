71 
72  if (prop == &pcConstraint->References) {

76  if (prop == &pcConstraint->BasePoint) {

85 
86  SbVec3f b(base.x, base.y, base.z);

89

91  pShapeSep->addChild(GuiTools::createFixed(radius / 2,

94  }
95  else if (prop == &pcConstraint->AxialFree) {
