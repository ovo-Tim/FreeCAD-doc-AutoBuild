90  def __init__(self):
91  super().__init__()
92  self.menutext = Qt.QT_TRANSLATE_NOOP("FEM_ClippingPlaneAdd", "Clipping plane on face")

95  )
96  self.is_active = "with_document"

98  def Activated(self):
99  from pivy import coin
100  from femtools.femutils import getBoundBoxOfAllDocumentShapes
101  from femtools.femutils import getSelectedFace
102 
103  overallboundbox = getBoundBoxOfAllDocumentShapes(FreeCAD.ActiveDocument)
104  # print(overallboundbox)
105  if overallboundbox:
106  min_bb_length = min(
107  {
108  overallboundbox.XLength,
109  overallboundbox.YLength,
110  overallboundbox.ZLength,
111  }
112  )
113  else:
114  min_bb_length = 10.0 # default
115 
116  dbox = min_bb_length * 0.2
117 
118  aFace = getSelectedFace(FreeCADGui.Selection.getSelectionEx())
119  if aFace:
120  f_CoM = aFace.CenterOfMass
121  f_uvCoM = aFace.Surface.parameter(f_CoM) # u,v at CoM for normalAt calculation
122  f_normal = aFace.normalAt(f_uvCoM[0], f_uvCoM[1])
123  else:
124  f_CoM = FreeCAD.Vector(0, 0, 0)
125  f_normal = FreeCAD.Vector(0, 0, 1)
126 
127  coin_normal_vector = coin.SbVec3f(-f_normal.x, -f_normal.y, -f_normal.z)
128  coin_bound_box = coin.SbBox3f(
129  f_CoM.x - dbox,
130  f_CoM.y - dbox,
131  f_CoM.z - dbox * 0.15,
132  f_CoM.x + dbox,
133  f_CoM.y + dbox,
134  f_CoM.z + dbox * 0.15,
135  )
136  clip_plane = coin.SoClipPlaneManip()
137  clip_plane.setValue(coin_bound_box, coin_normal_vector, 1)
138  FreeCADGui.ActiveDocument.ActiveView.getSceneGraph().insertChild(clip_plane, 1)
