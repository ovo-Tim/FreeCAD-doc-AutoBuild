119 void
120 SmSwitchboard::doAction(SoAction * action)
121 {
122  // FIXME: take PathCode and stuff into consideration...
123  if (action->isOfType(SoGetBoundingBoxAction::getClassTypeId())) {
124  // calculate center of bbox if bboxaction. This makes the
125  // switchboard node behave exactly like a group node
126  auto bbaction = (SoGetBoundingBoxAction*) action;
127  // Initialize accumulation variables.
128  SbVec3f acccenter(0.0f, 0.0f, 0.0f);
129  int numcenters = 0;
130  for (int idx = 0; idx < this->enable.getNum(); idx++) {
131  const int numchildren = this->children->getLength();
132  if ( numchildren > 0 )
133  this->children->traverse(action, (idx % numchildren));
134  // If center point is set, accumulate.
135  if (bbaction->isCenterSet()) {
136  acccenter += bbaction->getCenter();
137  numcenters++;
138  bbaction->resetCenter();
139  }
140  }
141  if (numcenters != 0) {
142  bbaction->setCenter(acccenter / float(numcenters), FALSE);
143  }
144  } else { // not a GetBoundingBoxAction
145  for ( int idx = 0; idx < this->enable.getNum(); idx++ ) {
146  if ( this->enable[idx] ) {
147  const int numchildren = this->children->getLength();
148  if ( numchildren > 0 )
149  this->children->traverse(action, (idx % numchildren));
150  }
151  }
152  }
153 }
