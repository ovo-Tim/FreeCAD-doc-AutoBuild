44 bool MeshProjection::bboxInsideRectangle(const Base::BoundBox3f& bbox,
45  const Base::Vector3f& p1,
46  const Base::Vector3f& p2,
47  const Base::Vector3f& view) const
48 {
49  Base::Vector3f dir(p2 - p1);
50  Base::Vector3f base(p1), normal(view % dir);
51  normal.Normalize();
52 
53  if (bbox.IsCutPlane(base, normal)) {
54  dir.Normalize();
55  Base::Vector3f cnt(bbox.GetCenter());
56 
57  return (fabs(cnt.DistanceToPlane(p1, dir)) + fabs(cnt.DistanceToPlane(p2, dir)))
58  <= (bbox.CalcDiagonalLength() + (p2 - p1).Length());
59  }
60 
61  return false;
62 }

64 bool MeshProjection::isPointInsideDistance(const Base::Vector3f& p1,
65  const Base::Vector3f& p2,
66  const Base::Vector3f& pt) const
67 {
68  // project point on line
69  Base::Vector3f proj, dir(p2 - p1);
70  Base::Vector3f move(pt - p1);
71  proj.ProjectToLine(move, dir);
72  proj = pt + proj;
73  return (((p1 - proj) * (p2 - proj)) < 0.0F);
74 }

76 bool MeshProjection::connectLines(std::list<std::pair<Base::Vector3f, Base::Vector3f>>& cutLines,
77  const Base::Vector3f& startPoint,
78  const Base::Vector3f& endPoint,
79  std::vector<Base::Vector3f>& polyline) const
80 {
81  const float fMaxDist = float(std::sqrt(FLOAT_MAX)); // max. length of a gap
82  const float fMinEps = 1.0e-4F;
83 
84  polyline.clear();
85  polyline.push_back(startPoint);
86 
87  Base::Vector3f curr(startPoint);
88  while ((curr != endPoint) && (!cutLines.empty())) {
89  std::list<std::pair<Base::Vector3f, Base::Vector3f>>::iterator it, pCurr = cutLines.end();
90 
91  // get nearest line
92  float fMin = fMaxDist * fMaxDist;
93 
94  bool bPos = false;
95  for (it = cutLines.begin(); it != cutLines.end(); ++it) {
96  float fD1 = Base::DistanceP2(curr, it->first);
97  float fD2 = Base::DistanceP2(curr, it->second);
98  if (std::min<float>(fD1, fD2) < fMin) {
99  pCurr = it;
100  bPos = fD1 < fD2;
101  fMin = std::min<float>(fD1, fD2);
102  if (fMin < fMinEps) { // abort because next line already found
103  break;
104  }
105  }
106  }
107 
108  if (pCurr != cutLines.end()) {
109  if (bPos) {
110  if (fMin > fMinEps) { // gap, insert point
111  polyline.push_back(pCurr->first);
112  }
113  polyline.push_back(pCurr->second);
114  curr = pCurr->second;
115  }
116  else {
117  if (fMin > fMinEps) { // gap, insert point
118  polyline.push_back(pCurr->second);
119  }
120  polyline.push_back(pCurr->first);
121  curr = pCurr->first;
122  }
123  }
124  else {
125  return false; // abort because no line was found
126  }
127 
128  cutLines.erase(pCurr);
129  }
130 
131  return true;
132 }

134 bool MeshProjection::projectLineOnMesh(const MeshFacetGrid& grid,
135  const Base::Vector3f& v1,
136  FacetIndex f1,
137  const Base::Vector3f& v2,
138  FacetIndex f2,
139  const Base::Vector3f& vd,
140  std::vector<Base::Vector3f>& polyline)
141 {
142  Base::Vector3f dir(v2 - v1);
143  Base::Vector3f base(v1), normal(vd % dir);
144  normal.Normalize();
145  dir.Normalize();
146 
147 
148  std::vector<FacetIndex> facets;
149 
150  // special case: start and endpoint inside same facet
151  if (f1 == f2) {
152  polyline.push_back(v1);
153  polyline.push_back(v2);
154  return true;
155  }
156 
157  // cut all facets between the two endpoints
158  MeshGridIterator gridIter(grid);
159  for (gridIter.Init(); gridIter.More(); gridIter.Next()) {
160  // bbox cuts plane
161  if (bboxInsideRectangle(gridIter.GetBoundBox(), v1, v2, vd)) {
162  gridIter.GetElements(facets);
163  }
164  }
165 
166  std::sort(facets.begin(), facets.end());
167  facets.erase(std::unique(facets.begin(), facets.end()), facets.end());
168 
169  // cut all facets with plane
170  std::list<std::pair<Base::Vector3f, Base::Vector3f>> cutLine;
171  for (FacetIndex facet : facets) {
172  Base::Vector3f e1, e2;
173  MeshGeomFacet tria = kernel.GetFacet(facet);
174  if (bboxInsideRectangle(tria.GetBoundBox(), v1, v2, vd)) {
175  if (tria.IntersectWithPlane(base, normal, e1, e2)) {
176  if ((facet != f1) && (facet != f2)) {
177  // inside cut line
178  if (!isPointInsideDistance(v1, v2, e1) || !isPointInsideDistance(v1, v2, e2)) {
179  continue;
180  }
181 
182  cutLine.emplace_back(e1, e2);
183  }
184  else {
185  if (facet == f1) { // start facet
186  if (((e2 - v1) * dir) > 0.0F) {
187  cutLine.emplace_back(v1, e2);
188  }
189  else {
190  cutLine.emplace_back(v1, e1);
191  }
192 
193  // start = it - facets.begin();
194  }
195 
196  if (facet == f2) { // end facet
197  if (((e2 - v2) * -dir) > 0.0F) {
198  cutLine.emplace_back(v2, e2);
199  }
200  else {
201  cutLine.emplace_back(v2, e1);
202  }
203 
204  // end = it - facets.begin();
205  }
206  }
207  }
208  }
209  }
210 
211  return connectLines(cutLine, v1, v2, polyline);
212 }
