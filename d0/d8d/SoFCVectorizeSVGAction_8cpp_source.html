207 void SoFCVectorizeSVGActionP::printTriangle(const SoVectorizeTriangle * item) const

211 
212  const SbBSPTree & bsp = publ->getBSPTree();
213 
214  SbVec3f v[3];
215  SbColor c[3];
216  float t[3];
217 
218  for (int i = 0; i < 3; i++) {
219  v[i] = bsp.getPoint(item->vidx[i]);
220  v[i][0] = (v[i][0] * mul[0]) + add[0];
221  v[i][1] = ((1.0f-v[i][1]) * mul[1]) + add[1];

334  const SbBSPTree & bsp = publ->getBSPTree();
335 
336  SbVec3f v[2];
337  SbColor c[2];

339 
340  for (int i = 0; i < 2; i++) {

342  v[i][0] = (v[i][0] * mul[0]) + add[0];
