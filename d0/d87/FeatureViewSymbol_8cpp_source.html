61 void FeatureViewSymbol::onChanged(const App::Property* prop)

84 App::DocumentObjectExecReturn* FeatureViewSymbol::execute(void)
85 {
86  std::string svg = Symbol.getValue();
87  const std::vector<std::string>& editText = EditableTexts.getValues();
88 
89  if (!editText.empty()) {
90  boost::regex e1("<text.*?freecad:editable=\"(.*?)\".*?<tspan.*?>(.*?)</tspan>");
91  string::const_iterator begin, end;
92  begin = svg.begin();
93  end = svg.end();
94  boost::match_results<std::string::const_iterator> what;
95  std::size_t count = 0;
96  std::string newsvg;
97  newsvg.reserve(svg.size());
98 
99  while (boost::regex_search(begin, end, what, e1)) {
100  if (count < editText.size()) {
101  // change values of editable texts. Also strip the "freecad:editable"
102  // attribute so it isn't detected by the page
103  boost::regex e2("(<text.*?)(freecad:editable=\"" + what[1].str()
104  + "\")(.*?<tspan.*?)>(.*?)(</tspan>)");
105  std::back_insert_iterator<std::string> out(newsvg);
106  boost::regex_replace(out,
107  begin,
108  what[0].second,
109  e2,
110  "$1$3>" + editText[count] + "$5");
111  }
112  count++;
113  begin = what[0].second;
114  }
115 
116  // now copy the rest
117  newsvg.insert(newsvg.end(), begin, end);
118  svg = newsvg;
119  }
120 
121  std::stringstream result;
122  result << "<g transform=\"translate(" << X.getValue() << "," << Y.getValue() << ")"
123  << " rotate(" << Rotation.getValue() << ")"
124  << " scale(" << Scale.getValue() << ")\">" << endl
125  << svg << endl
126  << "</g>" << endl;
127 
128  // Apply the resulting fragment
129  ViewResult.setValue(result.str().c_str());
130 
131  return App::DocumentObject::StdReturn;
132 }
