75  void setArrow(SoSwitch* pcSwitch = nullptr)
76  {
77  if (pcSwitch == pcLastArrowSwitch) {
78  return;
79  }
80  if (pcLastArrowSwitch) {
81  pcLastArrowSwitch->whichChild = -1;
82  pcLastArrowSwitch->unref();
83  pcLastArrowSwitch = nullptr;

85  if (pcSwitch) {

87  pcSwitch->whichChild = 0;

92  void onSelectionChanged(const Gui::SelectionChanges& msg) override
93  {
94  if (msg.Type == Gui::SelectionChanges::RmvPreselect) {
95  setArrow();
96  return;
97  }
98  if (msg.Type != Gui::SelectionChanges::SetPreselect
99  && msg.Type != Gui::SelectionChanges::MovePreselect) {
100  return;
101  }
102  auto obj = msg.Object.getObject();
103  if (!obj) {
104  return;
105  }
106  Base::Matrix4D mat;
107  auto sobj = obj->getSubObject(msg.pSubName, nullptr, &mat);
108  if (!sobj) {
109  return;
110  }
111  Base::Matrix4D linkMat;
112  auto linked = sobj->getLinkedObject(true, &linkMat, false);
113  auto vp = Base::freecad_dynamic_cast<ViewProviderPath>(
114  Application::Instance->getViewProvider(linked));
115  if (!vp) {
116  setArrow();
117  return;
118  }
119 
120  if (vp->pt0Index >= 0) {
121  mat *= linkMat;
122  mat.inverse();
123  Base::Vector3d pt = mat * Base::Vector3d(msg.x, msg.y, msg.z);
124  if (vp->pcLineCoords->point.getNum() > 0) {
125  auto ptTo = vp->pcLineCoords->point.getValues(vp->pt0Index);
126  SbVec3f ptFrom(pt.x, pt.y, pt.z);
127  if (ptTo && ptFrom != *ptTo) {
128  vp->pcArrowTransform->pointAt(ptFrom, *ptTo);
129  setArrow(vp->pcArrowSwitch);
130  return;
131  }
132  }
133  }
134  setArrow();
135  }
136 
137  SoSwitch* pcLastArrowSwitch = nullptr;
138 };

340 std::string ViewProviderPath::getElement(const SoDetail* detail) const
341 {
342  if (edgeStart >= 0 && detail && detail->getTypeId() == SoLineDetail::getClassTypeId()) {
343  const SoLineDetail* line_detail = static_cast<const SoLineDetail*>(detail);
344  int index = line_detail->getLineIndex() + edgeStart;
345  if (index >= 0 && index < (int)edge2Command.size()) {
346  index = edge2Command[index];
347  Path::Feature* pcPathObj = static_cast<Path::Feature*>(pcObject);
348  const Toolpath& tp = pcPathObj->Path.getValue();
349  if (index < (int)tp.getSize()) {
350  std::stringstream str;
351  str << index + 1 << " " << tp.getCommand(index).toGCode(6, false);
352  pt0Index = line_detail->getPoint0()->getCoordinateIndex();
353  if (pt0Index < 0 || pt0Index >= pcLineCoords->point.getNum()) {
354  pt0Index = -1;
355  }
356  return boost::replace_all_copy(str.str(), ".", ",");

358  }
359  }
360  pt0Index = -1;
361  pcArrowSwitch->whichChild = -1;
362  return std::string();

379 void ViewProviderPath::onChanged(const App::Property* prop)
380 {
381  if (blockPropertyChange) {
382  return;
383  }
384 
385  if (prop == &LineWidth) {
386  pcDrawStyle->lineWidth = LineWidth.getValue();
387  }
388  else if (prop == &NormalColor) {
389  if (!colorindex.empty() && coordStart >= 0 && coordStart < (int)colorindex.size()) {
390  const App::Color& c = NormalColor.getValue();
391  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
392  "User parameter:BaseApp/Preferences/Mod/CAM");
393  unsigned long rcol =
394  hGrp->GetUnsigned("DefaultRapidPathColor", 2852126975UL); // dark red (170,0,0)
395  float rr, rg, rb;
396  rr = ((rcol >> 24) & 0xff) / 255.0;
397  rg = ((rcol >> 16) & 0xff) / 255.0;
398  rb = ((rcol >> 8) & 0xff) / 255.0;
399 
400  unsigned long pcol =
401  hGrp->GetUnsigned("DefaultProbePathColor", 4293591295UL); // yellow (255,255,5)
402  float pr, pg, pb;
403  pr = ((pcol >> 24) & 0xff) / 255.0;
404  pg = ((pcol >> 16) & 0xff) / 255.0;
405  pb = ((pcol >> 8) & 0xff) / 255.0;
406 
407  pcMatBind->value = SoMaterialBinding::PER_PART;
408  // resizing and writing the color vector:
409 
410  int count = coordEnd - coordStart;
411  if (count > (int)colorindex.size() - coordStart) {
412  count = colorindex.size() - coordStart;
413  }
414  pcLineColor->diffuseColor.setNum(count);
415  SbColor* colors = pcLineColor->diffuseColor.startEditing();
416  for (int i = 0; i < count; i++) {
417  switch (colorindex[i + coordStart]) {
418  case 0:
419  colors[i] = SbColor(rr, rg, rb);
420  break;
421  case 1:
422  colors[i] = SbColor(c.r, c.g, c.b);
423  break;
424  default:
425  colors[i] = SbColor(pr, pg, pb);
426  }
427  }
428  pcLineColor->diffuseColor.finishEditing();
429  }
430  }
431  else if (prop == &MarkerColor) {
432  const App::Color& c = MarkerColor.getValue();
433  pcMarkerColor->rgb.setValue(c.r, c.g, c.b);
434  }
435  else if (prop == &ShowNodes) {
436  pcMarkerSwitch->whichChild = ShowNodes.getValue() ? 0 : -1;
437  }
438  else if (prop == &ShowCount || prop == &StartIndex) {
439  bool vis = isShow();
440  if (vis) {
441  hide();
442  }
443  updateVisual();
444  if (vis) {
445  show();
446  }
447  }
448  else if (prop == &StartPosition) {
449  if (pcLineCoords->point.getNum()) {
450  const Base::Vector3d& pt = StartPosition.getValue();
451  pcLineCoords->point.set1Value(0, pt.x, pt.y, pt.z);
452  pcMarkerCoords->point.set1Value(0, pt.x, pt.y, pt.z);
453  }
454  }
455  else {
456  inherited::onChanged(prop);
457  if (prop == &SelectionStyle && SelectionStyle.getValue() == 2) {
458  hideSelection();
459  }
460  }
461 }
