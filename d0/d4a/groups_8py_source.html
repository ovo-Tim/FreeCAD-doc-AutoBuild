44 def is_group(obj):
45  """Return True if the given object is considered a group.
46 
47  Parameters
48  ----------
49  obj : App::DocumentObject
50  The object to check.
51 
52  Returns
53  -------
54  bool
55  Returns `True` if `obj` is considered a group:
56 
57  The object is derived from `App::DocumentObjectGroup` but not
58  a `'LayerContainer'`.
59 
60  Or the object is of the type `'Project'`, `'Site'`, `'Building'`,
61  `'Floor'`, `'BuildingPart'` or `'Space'` from the Arch Workbench.
62  Note that `'Floor'` and `'Building'` are obsolete types.
63 
64  Otherwise returns `False`.
65  """
66  typ = utils.get_type(obj)
67  return ((obj.isDerivedFrom("App::DocumentObjectGroup")
68  and typ != "LayerContainer")
69  or typ in ("Project", "Site", "Building",
70  "Floor", "BuildingPart", "Space"))

113 def ungroup(obj):
114  """Remove the object from any group to which it belongs.
115 
116  A "group" is any object returned by `get_group_names`.
117 
118  Parameters
119  ----------
120  obj: App::DocumentObject or str
121  Any type of object.
122  If it is a string, it must be the `Label` of that object.
123  Since a label is not guaranteed to be unique in a document,
124  it will use the first object found with this label.
125  """
126  if isinstance(obj, str):
127  obj_str = obj
128 
129  found, obj = utils.find_object(obj, doc=App.activeDocument())
130  if not found:
131  _err(translate("draft", "Wrong input: object {} not in document.").format(obj_str))
132  return None
133 
134  doc = obj.Document
135 
136  for name in get_group_names():
137  group = doc.getObject(name)
138  if obj in group.Group:
139  # The list of objects cannot be modified directly,
140  # so a new list is created, this new list is modified,
141  # and then it is assigned over the older list.
142  objects = group.Group
143  objects.remove(obj)
144  group.Group = objects

147 def get_windows(obj):
148  """Return the windows and rebars inside a host.
149 
150  Parameters
151  ----------
152  obj: App::DocumentObject
153  A scripted object of type `'Wall'` or `'Structure'`
154  (Arch Workbench).
155  This will be searched for objects of type `'Window'` and `'Rebar'`,
156  and clones of them, and the found elements will be added
157  to the output list.
158 
159  The function will search recursively all elements under `obj.OutList`,
160  in case the windows and rebars are nested under other walls
161  and structures.
162 
163  Returns
164  -------
165  list
166  A list of all found windows and rebars in `obj`.
167  If `obj` is itself a `'Window'` or a `'Rebar'`, or a clone of them,
168  it will return the same `obj` element.
169  """
170  out = []
171  if utils.get_type(obj) in ("Wall", "Structure"):
172  for o in obj.OutList:
173  out.extend(get_windows(o))
174  for i in obj.InList:
175  if (utils.get_type(i.getLinkedObject()) == "Window"
176  or utils.is_clone(obj, "Window")):
177  if hasattr(i, "Hosts"):
178  if obj in i.Hosts:
179  out.append(i)
180  elif (utils.get_type(i) == "Rebar"
181  or utils.is_clone(obj, "Rebar")):
182  if hasattr(i, "Host"):
183  if obj == i.Host:
184  out.append(i)
185  elif (utils.get_type(obj.getLinkedObject()) in ("Window", "Rebar")
186  or utils.is_clone(obj, ["Window", "Rebar"])):
187  out.append(obj)
188 
189  return out

192 def get_group_contents(objectslist,
193  walls=False, addgroups=False,
194  spaces=False, noarchchild=False):
195  """Return a list of objects from expanding the input groups.
196 
197  The function accepts any type of object, although it is most useful
198  with "groups", as it is meant to unpack the objects inside these groups.
199 
200  Parameters
201  ----------
202  objectslist: list
203  If any object in the list is considered a group, see the `is_group`
204  function, its contents (`obj.Group`) are extracted and added to the
205  output list.
206 
207  Single items that aren't groups are added to the output list
208  as is.
209 
210  walls: bool, optional
211  It defaults to `False`.
212  If it is `True`, Wall and Structure objects (Arch Workbench)
213  are treated as groups; they are scanned for Window, Door,
214  and Rebar objects, and these are added to the output list.
215 
216  addgroups: bool, optional
217  It defaults to `False`.
218  If it is `True`, the group itself is kept as part of the output list.
219 
220  spaces: bool, optional
221  It defaults to `False`.
222  If it is `True`, Arch Spaces are added to the output list even
223  when addgroups is False (their contents are always added).
224 
225  noarchchild: bool, optional
226  It defaults to `False`.
227  If it is `True`, the objects inside Building and BuildingParts
228  (Arch Workbench) aren't added to the output list.
229 
230  Returns
231  -------
232  list
233  The list of objects from each group present in `objectslist`,
234  plus any other individual object given in `objectslist`.
235  """
236  newlist = []
237  if not isinstance(objectslist, list):
238  objectslist = [objectslist]
239 
240  for obj in objectslist:
241  if obj:
242  if is_group(obj):
243  if addgroups or (spaces
244  and utils.get_type(obj) == "Space"):
245  newlist.append(obj)
246  if not (noarchchild
247  and utils.get_type(obj) in ("Building",
248  "BuildingPart")):
249  newlist.extend(get_group_contents(obj.Group,
250  walls, addgroups,
251  spaces, noarchchild))
252  else:
253  # print("adding ", obj.Name)
254  newlist.append(obj)
255  if walls:
256  newlist.extend(get_windows(obj))
257 
258  # Clean possible duplicates
259  cleanlist = []
260  for obj in newlist:
261  if obj not in cleanlist:
262  cleanlist.append(obj)
263 
264  return cleanlist

278 def get_movable_children(objectslist, recursive=True, _donelist=[]):
279  """Return a list of objects with child objects that move with a host.
280 
281  Builds a list of objects with all child objects (`obj.OutList`)
282  that have their `MoveWithHost` attribute set to `True`.
283  This function is mostly useful for Arch Workbench objects.
284 
285  Parameters
286  ----------
287  objectslist: list of App::DocumentObject
288  A single scripted object or list of objects.
289 
290  recursive: bool, optional
291  It defaults to `True`, in which case the function
292  is called recursively to also extract the children of children
293  objects.
294  Otherwise, only direct children of the input objects
295  are added to the output list.
296 
297  _donelist: list
298  List of object names. Used internally to prevent an endless loop.
299 
300  Returns
301  -------
302  list
303  List of children objects that have their `MoveWithHost` attribute
304  set to `True`.
305  """
306  added = []
307  if not isinstance(objectslist, list):
308  objectslist = [objectslist]
309 
310  for obj in objectslist:
311  if obj.Name in _donelist:
312  continue
313 
314  _donelist.append(obj.Name)
315 
316  # Skips some objects that should never move their children
317  if utils.get_type(obj) not in ("App::Part", "PartDesign::Body",
318  "Clone", "SectionPlane",
319  "Facebinder", "BuildingPart", "App::Link"):
320  children = obj.OutList
321  if (hasattr(obj, "Proxy") and obj.Proxy
322  and hasattr(obj.Proxy, "getSiblings")
323  and utils.get_type(obj) != "Window"):
324  # children.extend(obj.Proxy.getSiblings(obj))
325  pass
326 
327  for child in children:
328  if hasattr(child, "MoveWithHost") and child.MoveWithHost:
329  if hasattr(obj, "CloneOf") and obj.CloneOf:
330  if obj.CloneOf.Name != child.Name:
331  added.append(child)
332  else:
333  added.append(child)
334 
335  if recursive:
336  added.extend(get_movable_children(children, recursive, _donelist))
337 
338  return added
