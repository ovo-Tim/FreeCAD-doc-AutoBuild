65 class AddonInstaller(QtCore.QObject):
66  """The core, non-GUI installer class. Usually instantiated and moved to its own thread,
67  otherwise it will block the GUI (if the GUI is running). In all cases in this class, the
68  generic Python 'object' is intended to be an Addon-like object that provides, at a minimum,
69  a 'name', 'url', and 'branch' attribute. The Addon manager uses the Addon class for this
70  purpose, but external code may use any other class that meets those criteria.
71 
72  Recommended Usage (when running with the GUI up, so you don't block the GUI thread):
73 
74  import functools # With the rest of your imports, for functools.partial
75 
76  ...
77

79 
80  self.worker_thread = QtCore.QThread()
81  self.installer = AddonInstaller(addon_to_install)
82  self.installer.moveToThread(self.worker_thread)
83  self.installer.success.connect(self.installation_succeeded)
84  self.installer.failure.connect(self.installation_failed)
85  self.installer.finished.connect(self.worker_thread.quit)
86  self.worker_thread.started.connect(self.installer.run)
87  self.worker_thread.start() # Returns immediately

90  # on failure, self.installation_failed is called.

142  def run(self, install_method: InstallationMethod = InstallationMethod.ANY) -> bool:
143  """Install an addon. Returns True if the addon was installed, or False if not. Emits
144  either success or failure prior to returning."""
145  try:
146  addon_url = self.addon_to_install.url.replace(os.path.sep, "/")
147  method_to_use = self._determine_install_method(addon_url, install_method)
148  success = False
149  if method_to_use == InstallationMethod.ZIP:
150  success = self._install_by_zip()
151  elif method_to_use == InstallationMethod.GIT:
152  success = self._install_by_git()
153  elif method_to_use == InstallationMethod.COPY:
154  success = self._install_by_copy()
155  if (
156  hasattr(self.addon_to_install, "contains_workbench")
157  and self.addon_to_install.contains_workbench()
158  ):
159  self.addon_to_install.enable_workbench()
160  except utils.ProcessInterrupted:
161  pass
162  except Exception as e:
163  FreeCAD.Console.PrintLog(e + "\n")
164  success = False
165  if success:
166  if (
167  hasattr(self.addon_to_install, "contains_workbench")
168  and self.addon_to_install.contains_workbench()
169  ):
170  self.addon_to_install.set_status(Addon.Status.PENDING_RESTART)
171  else:
172  self.addon_to_install.set_status(Addon.Status.NO_UPDATE_AVAILABLE)
173  self.finished.emit()
174  return success

213  def _determine_install_method(
214  self, addon_url: str, install_method: InstallationMethod
215  ) -> Optional[InstallationMethod]:
216  """Given a URL and preferred installation method, determine the actual installation method
217  to use. Will return either None, if installation is not possible for the given url and
218  method, or a specific concrete method (GIT, ZIP, or COPY) based on the inputs."""
219 
220  # If we don't have access to git, and that is the method selected, return early
221  if not self.git_manager and install_method == InstallationMethod.GIT:
222  return None
223 
224  parse_result = urlparse(addon_url)
225  is_git_only = parse_result.scheme in ["git", "ssh", "rsync"]
226  is_remote = parse_result.scheme in ["http", "https", "git", "ssh", "rsync"]
227  is_zipfile = parse_result.path.lower().endswith(".zip")
228 
229  # Can't use "copy" for a remote URL
230  if is_remote and install_method == InstallationMethod.COPY:
231  return None
232 
233  if is_git_only:
234  if (
235  install_method in (InstallationMethod.GIT, InstallationMethod.ANY)
236  ) and self.git_manager:
237  # If it's a git-only URL, only git can be used for the installation
238  return InstallationMethod.GIT
239  # So if it's not a git installation, return None
240  return None
241 
242  if is_zipfile:
243  if install_method == InstallationMethod.GIT:
244  # Can't use git on zip files
245  return None
246  return InstallationMethod.ZIP # Copy just becomes zip
247  if not is_remote and install_method == InstallationMethod.ZIP:
248  return None # Can't use zip on local paths that aren't zip files
249 
250  # Whatever scheme was passed in appears to be reasonable, return it
251  if install_method != InstallationMethod.ANY:
252  return install_method
253 
254  # Prefer to copy, if it's local:
255  if not is_remote:
256  return InstallationMethod.COPY
257 
258  # Prefer git if we have git
259  if self.git_manager:
260  return InstallationMethod.GIT
261 
262  # Fall back to ZIP in other cases, though this relies on remote hosts falling
263  # into one of a few particular patterns
264  return InstallationMethod.ZIP

266  def _install_by_copy(self) -> bool:
267  """Installs the specified url by copying directly from it into the installation
268  location. addon_url must be copyable using filesystem operations. Any existing files at
269  that location are overwritten."""
270  addon_url = self.addon_to_install.url
271  if addon_url.startswith("file://"):
272  addon_url = addon_url[len("file://") :] # Strip off the file:// part
273  name = self.addon_to_install.name
274  shutil.copytree(addon_url, os.path.join(self.installation_path, name), dirs_exist_ok=True)
275  self._finalize_successful_installation()
276  return True

278  def _can_use_update(self) -> bool:
279  addon = self.addon_to_install
280  install_path = os.path.join(self.installation_path, self.addon_to_install.name)
281  if not os.path.isdir(install_path):
282  return False
283  if addon.metadata is None or addon.installed_metadata is None:
284  return True # We can't check if the branch name changed, but the install path exists
285  old_branch = get_branch_from_metadata(self.addon_to_install.installed_metadata)
286  new_branch = get_branch_from_metadata(self.addon_to_install.metadata)
287  if old_branch != new_branch:
288  return False # Branch name changed, we have to re-clone
289  return True # Checkout exists, same branch as last time, update OK

291  def _install_by_git(self) -> bool:
292  """Installs the specified url by using git to clone from it. The URL can be local or remote,
293  but must represent a git repository, and the url must be in a format that git can handle
294  (git, ssh, rsync, file, or a bare filesystem path)."""
295  install_path = os.path.join(self.installation_path, self.addon_to_install.name)
296  try:
297  if self._can_use_update():
298  self.git_manager.update(install_path)
299  else:
300  if os.path.isdir(install_path):
301  utils.rmdir(install_path)
302  self.git_manager.clone(self.addon_to_install.url, install_path)
303  self.git_manager.checkout(install_path, self.addon_to_install.branch)
304  except GitFailed as e:
305  self.failure.emit(self.addon_to_install, str(e))
306  return False
307  self._finalize_successful_installation()
308  return True

310  def _install_by_zip(self) -> bool:
311  """Installs the specified url by downloading the file (if it is remote) and unzipping it
312  into the appropriate installation location. If the GUI is running the download is
313  asynchronous, and issues periodic updates about how much data has been downloaded."""
314  if self.addon_to_install.url.endswith(".zip"):
315  zip_url = self.addon_to_install.url
316  else:
317  zip_url = utils.get_zip_url(self.addon_to_install)
318 
319  FreeCAD.Console.PrintLog(f"Downloading ZIP file from {zip_url}...\n")
320  parse_result = urlparse(zip_url)
321  is_remote = parse_result.scheme in ["http", "https"]
322 
323  if is_remote:
324  if FreeCAD.GuiUp:
325  self._run_zip_downloader_in_event_loop(zip_url)
326  else:
327  zip_data = utils.blocking_get(zip_url)
328  with tempfile.NamedTemporaryFile(delete=False) as f:
329  tempfile_name = f.name
330  f.write(zip_data)
331  self._finalize_zip_installation(tempfile_name)
332  else:
333  self._finalize_zip_installation(zip_url)
334  return True

336  def _run_zip_downloader_in_event_loop(self, zip_url: str):
337  """Runs the zip downloader in a private event loop. This function does not exit until the
338  ZIP download is complete. It requires the GUI to be up, and should not be run on the main
339  GUI thread."""
340  NetworkManager.AM_NETWORK_MANAGER.progress_made.connect(self._update_zip_status)
341  NetworkManager.AM_NETWORK_MANAGER.progress_complete.connect(self._finish_zip)
342  self.zip_download_index = NetworkManager.AM_NETWORK_MANAGER.submit_monitored_get(zip_url)
343  while self.zip_download_index is not None:
344  if QtCore.QThread.currentThread().isInterruptionRequested():
345  break
346  QtCore.QCoreApplication.processEvents(QtCore.QEventLoop.AllEvents, 50)

348  def _update_zip_status(self, index: int, bytes_read: int, data_size: int):
349  """Called periodically when downloading a zip file, emits a signal to display the
350  download progress."""
351  if index == self.zip_download_index:
352  self.progress_update.emit(bytes_read, data_size)

374  def _finalize_zip_installation(self, filename: os.PathLike):
375  """Given a path to a zipfile, extract that file and put its contents in the correct
376  location. Has special handling for GitHub's zip structure, which places the data in a
377  subdirectory of the main directory."""
378 
379  destination = os.path.join(self.installation_path, self.addon_to_install.name)
380  if os.path.exists(destination):
381  remove_succeeded = utils.rmdir(destination)
382  if not remove_succeeded:
383  FreeCAD.Console.PrintError(f"Failed to remove {destination}, aborting update")
384  raise RuntimeError(f"Failed to remove outdated Addon from {destination}")
385 
386  with zipfile.ZipFile(filename, "r") as zfile:
387  zfile.extractall(destination)
388 
389  # GitHub (and possibly other hosts) put all files in the zip into a subdirectory named
390  # after the branch. If that is the setup that we just extracted, move all files out of
391  # that subdirectory.
392  if self._code_in_branch_subdirectory(destination):
393  actual_path = os.path.join(
394  destination, f"{self.addon_to_install.name}-{self.addon_to_install.branch}"
395  )
396  FreeCAD.Console.PrintLog(
397  f"ZIP installation moving code from {actual_path} to {destination}"
398  )
399  self._move_code_out_of_subdirectory(destination)
400 
401  FreeCAD.Console.PrintLog("ZIP installation complete.\n")
402  self._finalize_successful_installation()

447  def _install_macros(self):

499 class MacroInstaller(QtCore.QObject):
500  """Install a macro."""
501 
502  # Signals: success and failure
503  # Emitted when the installation process is complete. The object emitted is the object that the
504  # installation was requested for (usually of class Addon, but any class that provides a macro
505  # can be used).
506  success = QtCore.Signal(object)
507  failure = QtCore.Signal(object)
508 
509  # Finished: regardless of the outcome, this is emitted when all work that is going to be done
510  # is done (i.e. whatever thread this is running in can quit).
511  finished = QtCore.Signal()
512 
513  def __init__(self, addon: object):
514  """The provided addon object must have an attribute called "macro", and that attribute must
515  itself provide a callable "install" method that takes a single string, the path to the
516  installation location."""
517  super().__init__()
518  self._validate_object(addon)
519  self.addon_to_install = addon
520  self.installation_path = FreeCAD.getUserMacroDir(True)
521 
522  def run(self) -> bool:
523  """Install a macro. Returns True if the macro was installed, or False if not. Emits
524  either success or failure prior to returning."""
525 
526  # To try to ensure atomicity, perform the installation into a temp directory
527  macro = self.addon_to_install.macro
528  with tempfile.TemporaryDirectory() as temp_dir:
529  temp_install_succeeded, error_list = macro.install(temp_dir)
530  if not temp_install_succeeded:
531  FreeCAD.Console.PrintError(
532  translate("AddonsInstaller", "Failed to install macro {}").format(macro.name)
533  + "\n"
534  )
535  for e in error_list:
536  FreeCAD.Console.PrintError(e + "\n")
537  self.failure.emit(self.addon_to_install, "\n".join(error_list))
538  self.finished.emit()
539  return False
540 
541  # If it succeeded, move all the files to the macro install location,
542  # keeping a list of all the files we installed, so they can be removed later
543  # if this macro is uninstalled.
544  manifest = []
545  for item in os.listdir(temp_dir):
546  src = os.path.join(temp_dir, item)
547  dst = os.path.join(self.installation_path, item)
548  shutil.move(src, dst)
549  manifest.append(dst)
550  self._write_installation_manifest(manifest)
551  self.success.emit(self.addon_to_install)
552  self.addon_to_install.set_status(Addon.Status.NO_UPDATE_AVAILABLE)
553  self.finished.emit()
554  return True
