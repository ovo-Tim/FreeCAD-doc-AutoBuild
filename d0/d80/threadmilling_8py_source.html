100 def generate(center, cmd, zStart, zFinal, pitch, radius, leadInOut, elevator, start):
101  """generate(center, cmd, zStart, zFinal, pitch, radius, leadInOut, elevator, start) ... returns the g-code to mill the given internal thread"""
102  thread = _Thread(cmd, zStart, zFinal, pitch, radius > elevator)
103 
104  yMin = center.y - radius
105  yMax = center.y + radius
106 
107  path = []
108  # at this point the tool is at a safe heiht (depending on the previous thread), so we can move
109  # into position first, and then drop to the start height. If there is any material in the way this
110  # op hasn't been setup properly.
111  if leadInOut:
112  _comment(path, "lead-in")
113  if start is None:
114  path.append(Path.Command("G0", {"X": center.x, "Y": center.y + elevator}))
115  path.append(Path.Command("G0", {"Z": thread.zStart}))
116  else:
117  path.append(
118  Path.Command(
119  thread.g4Start2Elevator(),
120  {
121  "X": center.x,
122  "Y": center.y + elevator,
123  "Z": thread.zStart,
124  "I": (center.x - start.x) / 2,
125  "J": (center.y + elevator - start.y) / 2,
126  "K": (thread.zStart - start.z) / 2,
127  },
128  )
129  )
130  path.append(
131  Path.Command(
132  thread.g4LeadInOut(),
133  {"Y": yMax, "J": (yMax - (center.y + elevator)) / 2},
134  )
135  )
136  _comment(path, "lead-in")
137  else:
138  if start is None:
139  path.append(Path.Command("G0", {"X": center.x, "Y": center.y + elevator}))
140  path.append(Path.Command("G0", {"Z": thread.zStart}))
141  else:
142  path.append(
143  Path.Command("G0", {"X": center.x, "Y": center.y + elevator, "Z": thread.zStart})
144  )
145  path.append(Path.Command("G1", {"Y": yMax}))
146 
147  z = thread.zStart
148  r = -radius
149  i = 0
150  while not Path.Geom.isRoughly(z, thread.zFinal):
151  if thread.overshoots(z):
152  break
153  if 0 == (i & 0x01):
154  y = yMin
155  else:
156  y = yMax
157  path.append(Path.Command(thread.cmd, {"Y": y, "Z": z + thread.hPitch, "J": r}))
158  r = -r
159  i = i + 1
160  z = z + thread.hPitch
161 
162  if Path.Geom.isRoughly(z, thread.zFinal):
163  x = center.x
164  y = yMin if (i & 0x01) else yMax
165  else:
166  n = math.fabs(thread.zFinal - thread.zStart) / thread.hPitch
167  k = n - int(n)
168  dy = math.cos(k * math.pi)
169  dx = math.sin(k * math.pi)
170  y = thread.adjustY(center.y, r * dy)
171  x = thread.adjustX(center.x, r * dx)
172  _comment(path, "finish-thread")
173  path.append(Path.Command(thread.cmd, {"X": x, "Y": y, "Z": thread.zFinal, "J": r}))
174  _comment(path, "finish-thread")
175 
176  a = math.atan2(y - center.y, x - center.x)
177  dx = math.cos(a) * (radius - elevator)
178  dy = math.sin(a) * (radius - elevator)
179  Path.Log.debug("")
180  Path.Log.debug("a={}: dx={:.2f}, dy={:.2f}".format(a / math.pi * 180, dx, dy))
181 
182  elevatorX = x - dx
183  elevatorY = y - dy
184  Path.Log.debug("({:.2f}, {:.2f}) -> ({:.2f}, {:.2f})".format(x, y, elevatorX, elevatorY))
185

187  _comment(path, "lead-out")
188  path.append(
189  Path.Command(
190  thread.g4LeadInOut(),
191  {"X": elevatorX, "Y": elevatorY, "I": -dx / 2, "J": -dy / 2},
192  )
193  )
194  _comment(path, "lead-out")
195  else:
196  path.append(Path.Command("G1", {"X": elevatorX, "Y": elevatorY}))
