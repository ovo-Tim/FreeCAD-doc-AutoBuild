168 void QGIDatumLabel::snapPosition(QPointF& pos)
169 {
170  qreal snapPercent = 0.4;
171  double dimSpacing = Rez::guiX(activeDimAttributes.getCascadeSpacing());
172 
173  auto* qgivd = dynamic_cast<QGIViewDimension*>(parentItem());
174  if (!qgivd) {
175  return;
176  }
177  auto* dim(dynamic_cast<TechDraw::DrawViewDimension*>(qgivd->getViewObject()));
178  if (!dim) {
179  return;
180  }
181 
182  // We only have snap for distances constraints
183  std::string type = dim->Type.getValueAsString();
184  if(type != "Distance" && type != "DistanceX" && type != "DistanceY") {
185  return;
186  }
187 
188  // 1 - We try to snap the label to its center position.
189  pointPair pp = dim->getLinearPoints();
190  Base::Vector3d p1_3d = Rez::guiX(pp.first());
191  Base::Vector3d p2_3d = Rez::guiX(pp.second());
192  Base::Vector2d p1 = Base::Vector2d(p1_3d.x, p1_3d.y);
193  Base::Vector2d p2 = Base::Vector2d(p2_3d.x, p2_3d.y);
194  if (type == "DistanceX") {
195  p2 = Base::Vector2d(p2.x, p1.y);
196  }
197  else if (type == "DistanceY") {
198  p2 = Base::Vector2d(p1.x, p2.y);
199  }
200  Base::Vector2d mid = (p1 + p2) * 0.5;
201  Base::Vector2d dir = p2 - p1;
202  Base::Vector2d normal = Base::Vector2d(-dir.y, dir.x);
203 
204  Base::Vector2d toCenter = getPosToCenterVec();
205 
206  Base::Vector2d posV = Base::Vector2d(pos.x(), pos.y()) + toCenter;
207 
208  Base::Vector2d projPnt;
209  projPnt.ProjectToLine(posV - mid, normal);
210  projPnt = projPnt + mid;
211 
212  if ((projPnt - posV).Length() < dimSpacing * snapPercent) {
213  posV = projPnt;
214  pos.setX(posV.x - toCenter.x);
215  pos.setY(posV.y - toCenter.y);
216  }
217 
218  // 2 - We check for coord/chain dimensions to offer proper snapping
219  auto* qgiv = dynamic_cast<QGIView*>(qgivd->parentItem());
220  if (qgiv) {
221  auto* dvp = dynamic_cast<TechDraw::DrawViewPart*>(qgiv->getViewObject());
222  if (dvp) {
223  snapPercent = 0.2;
224  std::vector<TechDraw::DrawViewDimension*> dims = dvp->getDimensions();
225  for (auto& d : dims) {
226  if (d == dim) { continue; }
227 
228  std::string typei = d->Type.getValueAsString();
229  if (typei != "Distance" && typei != "DistanceX" && typei != "DistanceY") {
230  continue;
231  }
232 
233  pp = d->getLinearPoints();
234  Base::Vector3d ip1_3d = Rez::guiX(pp.first());
235  Base::Vector3d ip2_3d = Rez::guiX(pp.second());
236 
237  Base::Vector2d ip1 = Base::Vector2d(ip1_3d.x, ip1_3d.y);
238  Base::Vector2d ip2 = Base::Vector2d(ip2_3d.x, ip2_3d.y);
239  if (typei == "DistanceX") {
240  ip2 = Base::Vector2d(ip2.x, ip1.y);
241  }
242  else if (typei == "DistanceY") {
243  ip2 = Base::Vector2d(ip1.x, ip2.y);
244  }
245 
246  Base::Vector2d idir = ip2 - ip1;
247 
248  if (fabs(dir.x * idir.y - dir.y * idir.x) > Precision::Confusion()) {
249  //dimensions not parallel
250  continue;
251  }
252 
253  auto* vp = dynamic_cast<ViewProviderDimension*>(Gui::Application::Instance->getViewProvider(d));
254  if (!vp) { continue; }
255  auto* qgivDi(dynamic_cast<QGIViewDimension*>(vp->getQView()));
256  if (!qgivDi) { continue; }
257  auto labeli = qgivDi->getDatumLabel();
258  if (!labeli) { continue; }
259  QPointF posi = labeli->pos();
260  Base::Vector2d toCenteri = labeli->getPosToCenterVec();
261  Base::Vector2d posVi = Base::Vector2d(posi.x(), posi.y()) + toCenteri;
262 
263  Base::Vector2d projPnt2;
264  projPnt2.ProjectToLine(posV - posVi, idir);
265  projPnt2 = projPnt2 + posVi;
266 
267  if ((projPnt2 - posV).Length() < dimSpacing * snapPercent) {
268  posV = projPnt2;
269  pos.setX(posV.x - toCenter.x);
270  pos.setY(posV.y - toCenter.y);
271  break;
272  }
273  else if (fabs((projPnt2 - posV).Length() - fabs(dimSpacing)) < dimSpacing * snapPercent) {
274  posV = projPnt2 + (posV - projPnt2).Normalize() * dimSpacing;
275  pos.setX(posV.x - toCenter.x);
276  pos.setY(posV.y - toCenter.y);
277  break;
278  }
279  }
280  }
281  }
282 
283 
284  setPos(pos); // no infinite loop because if pos doesn't change then itemChanged is not triggered.
285 }

397 void QGIDatumLabel::setPosFromCenter(const double& xCenter, const double& yCenter)
398 {
399  prepareGeometryChange();
400  auto* qgivd = dynamic_cast<QGIViewDimension*>(parentItem());
401  if (!qgivd) {
402  return;
403  }
404  const auto dim(dynamic_cast<TechDraw::DrawViewDimension*>(qgivd->getViewObject()));
405  if (!dim) {
406  return;
407  }
408 
409  //set label's Qt position(top, left) given boundingRect center point
410  Base::Vector2d vec = getPosToCenterVec();
411  setPos(xCenter - vec.x, yCenter - vec.y);
412 
413  QString uText = m_unitText->toPlainText();
414  if ((uText.size() > 0) && (uText.at(0) != QChar::fromLatin1(' '))) {
415  QString vText = m_dimText->toPlainText();
416  vText = vText + uText;
417  m_dimText->setPlainText(vText);
418  m_unitText->setPlainText(QString());
419  }
420 
421  QRectF labelBox = m_dimText->alignmentRect();
422  double right = labelBox.right();
423  double top = labelBox.top();
424  double bottom = labelBox.bottom();
425  double middle = (top + bottom) / 2.0;
426 
427  //set unit position
428  QRectF unitBox = m_unitText->alignmentRect();
429  double unitWidth = unitBox.width();
430  double unitRight = right + unitWidth;
431  // Set the m_unitText font *baseline* at same height as the m_dimText font baseline
432  m_unitText->setPos(right, 0.0);
433 
434  //set tolerance position
435  QRectF overBox = m_tolTextOver->alignmentRect();
436  double tolLeft = unitRight;
437 
438  // Adjust for difference in tight and original bounding box sizes, note the y-coord down system
439  QPointF tol_adj = m_tolTextOver->tightBoundingAdjust();
440  m_tolTextOver->justifyLeftAt(tolLeft + tol_adj.x(), middle - tol_adj.y(), false);
441  tol_adj = m_tolTextUnder->tightBoundingAdjust();
442  m_tolTextUnder->justifyLeftAt(tolLeft + tol_adj.x(), middle + overBox.height() - tol_adj.y(),
443  false);
444 }

490 void QGIDatumLabel::setToleranceString()
491 {
492  prepareGeometryChange();
493  QGIViewDimension* qgivd = dynamic_cast<QGIViewDimension*>(parentItem());
494  if (!qgivd) {
495  return;
496  }
497  const auto dim(dynamic_cast<TechDraw::DrawViewDimension*>(qgivd->getViewObject()));
498  if (!dim) {
499  return;
500  // don't show if both are zero or if EqualTolerance is true
501  }
502  else if (!dim->hasOverUnderTolerance() || dim->EqualTolerance.getValue()
503  || dim->TheoreticalExact.getValue()) {
504  m_tolTextOver->hide();
505  m_tolTextUnder->hide();
506  // we must explicitly empty the text otherwise the frame drawn for
507  // TheoreticalExact would be as wide as necessary for the text
508  m_tolTextOver->setPlainText(QString());
509  m_tolTextUnder->setPlainText(QString());
510  updateFrameRect();
511  return;
512  }
513 
514  std::pair<std::string, std::string> labelTexts, unitTexts;
515 
516  if (dim->ArbitraryTolerances.getValue()) {
517  labelTexts = dim->getFormattedToleranceValues(1);//copy tolerance spec
518  unitTexts.first = "";
519  unitTexts.second = "";
520  }
521  else {
522  if (dim->isMultiValueSchema()) {
523  labelTexts = dim->getFormattedToleranceValues(0);//don't format multis
524  unitTexts.first = "";
525  unitTexts.second = "";
526  }
527  else {
528  labelTexts = dim->getFormattedToleranceValues(1);// prefix value [unit] postfix
529  unitTexts = dim->getFormattedToleranceValues(2); //just the unit
530  }
531  }
532 
533  if (labelTexts.first.empty()) {
534  m_tolTextUnder->hide();
535  }
536  else {
537  m_tolTextUnder->setPlainText(QString::fromUtf8(labelTexts.first.c_str()));
538  m_tolTextUnder->show();
539  }
540  if (labelTexts.second.empty()) {
541  m_tolTextOver->hide();
542  }
543  else {
544  m_tolTextOver->setPlainText(QString::fromUtf8(labelTexts.second.c_str()));
545  m_tolTextOver->show();
546  }
547 
548  updateFrameRect();
549 }

1254 bool QGIViewDimension::constructDimensionLine(

1256  const Base::BoundBox2d& labelRectangle, int arrowCount, int standardStyle, bool flipArrows,
1257  std::vector<std::pair<double, bool>>& outputMarking) const
1258 {

1260  if (startPosition > 0.0) {
1261  Base::Console().Error(
1262  "QGIVD::constructDimLine - Start Position must not be positive! Received: %f\n",
1263  startPosition);
1264  return false;
1265  }
1266 
1267  double labelBorder = 0.0;
1268  if (standardStyle == ViewProviderDimension::STD_STYLE_ISO_ORIENTED) {
1269  labelBorder = labelRectangle.Width() * 0.5 + getDefaultIsoReferenceLineOverhang();
1270  }
1271  else if (standardStyle == ViewProviderDimension::STD_STYLE_ASME_INLINED) {
1272  std::vector<Base::Vector2d> intersectionPoints;
1273  DrawUtil::findLineRectangleIntersections(targetPoint, lineAngle, labelRectangle,

1277  labelBorder = (intersectionPoints[0] - labelRectangle.GetCenter()).Length();
1278  }
1279  }

1282  if (jointPosition + labelBorder > 0.0) {
1283  // If label sticks out, extend the dimension line beyond the end point (0.0)
1284  DrawUtil::intervalMarkLinear(outputMarking, 0.0, jointPosition + labelBorder, true);
1285  autoFlipArrows = true;
1286  }
1287 
1288  if (jointPosition - labelBorder < startPosition) {
1289  DrawUtil::intervalMarkLinear(outputMarking, startPosition,
1290  jointPosition - labelBorder - startPosition, true);
1291 
1292  // For only one arrow and zero width line skip flipping, it already points correctly
1293  if (arrowCount > 1 || startPosition < 0.0) {
1294  autoFlipArrows = true;
1295  }
1296  }
1297 
1298  flipArrows ^= autoFlipArrows;
1299  if (!flipArrows
1300  || (standardStyle != ViewProviderDimension::STD_STYLE_ASME_INLINED
1301  && standardStyle != ViewProviderDimension::STD_STYLE_ASME_REFERENCING)) {
1302  // If arrows point outside, or ASME standard is not followed,
1303  // add the line part between start and end
1304  DrawUtil::intervalMarkLinear(outputMarking, 0.0, startPosition, true);
1305  }
1306 
1307  // For ASME Inlined, cut out the part of line occupied by the value
1308  if (standardStyle == ViewProviderDimension::STD_STYLE_ASME_INLINED) {
1309  DrawUtil::intervalMarkLinear(outputMarking, jointPosition - labelBorder, labelBorder * 2.0,
1310  false);
1311  }
1312 
1313  // Add the arrow tails - these are drawn always
1314  double placementFactor = flipArrows ? +1.0 : -1.0;
1315  DrawUtil::intervalMarkLinear(outputMarking, 0.0, placementFactor * getDefaultArrowTailLength(),
1316  true);
1317  if (arrowCount > 1) {
