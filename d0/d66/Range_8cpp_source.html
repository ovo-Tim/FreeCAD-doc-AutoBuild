241 App::CellAddress App::stringToAddress(const char* strAddress, bool silent)
242 {
243  assert(strAddress);
244 
245  static boost::regex e("(\\$?[A-Z]{1,2})(\\$?[0-9]{1,5})");
246  boost::cmatch cm;
247 
248  if (boost::regex_match(strAddress, cm, e)) {
249  bool absCol = (cm[1].first[0] == '$');
250  std::string r, c;
251  if (absCol) {
252  c = std::string(cm[1].first + 1, cm[1].second);
253  }
254  else {
255  c = std::string(cm[1].first, cm[1].second);
256  }
257 
258  bool absRow = (cm[2].first[0] == '$');
259  if (absRow) {
260  r = std::string(cm[2].first + 1, cm[2].second);
261  }
262  else {
263  r = std::string(cm[2].first, cm[2].second);
264  }
265 
266  return CellAddress(decodeRow(r, silent), decodeColumn(c, silent), absRow, absCol);
267  }
268  else if (silent) {
269  return CellAddress();
270  }
271 
272  throw Base::RuntimeError("Invalid cell specifier.");
273 }
