99 void PropertyCosmeticEdgeList::setPyObject(PyObject *value)
100 {
101  if (PySequence_Check(value)) {
102  Py::Sequence sequence(value);
103  Py_ssize_t nSize = sequence.size();
104  std::vector<CosmeticEdge*> values;
105  values.resize(nSize);
106 
107  for (Py_ssize_t i=0; i < nSize; ++i) {
108  Py::Object item = sequence.getItem(i);
109  if (!PyObject_TypeCheck(item.ptr(), &(CosmeticEdgePy::Type))) {
110  std::string error = std::string("types in list must be 'CosmeticEdge', not ");
111  error += item.ptr()->ob_type->tp_name;
112  throw Base::TypeError(error);
113  }
114 
115  values[i] = static_cast<CosmeticEdgePy*>(item.ptr())->getCosmeticEdgePtr();
116  }
117 
118  setValues(values);
119  }
120  else if (PyObject_TypeCheck(value, &(CosmeticEdgePy::Type))) {
121  CosmeticEdgePy *pcObject = static_cast<CosmeticEdgePy*>(value);
122  setValue(pcObject->getCosmeticEdgePtr());

124  else {

148 void PropertyCosmeticEdgeList::Restore(Base::XMLReader &reader)
149 {
150  // read my element
151  reader.clearPartialRestoreObject();
152  reader.readElement("CosmeticEdgeList");
153  // get the value of my attribute
154  int count = reader.getAttributeAsInteger("count");
155  std::vector<CosmeticEdge*> values;
156  values.reserve(count);
157  for (int i = 0; i < count; i++) {
158  reader.readElement("CosmeticEdge");
159  const char* TypeName = reader.getAttribute("type");
160  CosmeticEdge *newG = static_cast<CosmeticEdge *>(Base::Type::fromName(TypeName).createInstance());
161  newG->Restore(reader);
162 
163  if(reader.testStatus(Base::XMLReader::ReaderStatus::PartialRestoreInObject)) {
164  Base::Console().Error("CosmeticEdge \"%s\" within a PropertyCosmeticEdgeList was subject to a partial restore.\n", reader.localName());
165  if(isOrderRelevant()) {
166  // Pushes the best try by the CosmeticEdge class
167  values.push_back(newG);
168  }
169  else {
170  delete newG;
171  }
172  reader.clearPartialRestoreObject();
173  }
174  else {
175  values.push_back(newG);
176  }
177 
178  reader.readEndElement("CosmeticEdge");
179  }
180 
181  reader.readEndElement("CosmeticEdgeList");
182 
183  // assignment
184  setValues(values);
185 }
