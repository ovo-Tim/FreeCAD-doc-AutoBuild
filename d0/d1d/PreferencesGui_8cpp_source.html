175 QString PreferencesGui::weldingDirectory()
176 {
177  std::string defaultDir = App::Application::getResourceDir() + "Mod/TechDraw/Symbols/Welding/AWS/";
178 
179  std::string symbolDir = Preferences::getPreferenceGroup("Files")->GetASCII("WeldingDir", defaultDir.c_str());
180  if (symbolDir.empty()) {
181  symbolDir = defaultDir;
182  }
183  QString qSymbolDir = QString::fromUtf8(symbolDir.c_str());
184  Base::FileInfo fi(symbolDir);
185  if (!fi.isReadable()) {
186  Base::Console().Warning("Welding Directory: %s is not readable\n", symbolDir.c_str());
187  qSymbolDir = QString::fromUtf8(defaultDir.c_str());
188  }
189  return qSymbolDir;
190 }

258 // largely based on code from https://invent.kde.org/graphics/okular and
259 // https://en.wikipedia.org/wiki/HSL_and_HSV#HSL_to_RGB
260 QColor PreferencesGui::lightenColor(QColor orig)
261 {
262  // get component colours on [0, 255]
263  uchar red = orig.red();
264  uchar blue = orig.blue();
265  uchar green = orig.green();
266  uchar alpha = orig.alpha();
267 
268  // shift color values
269  uchar m = std::min( {red, blue, green} );
270  red -= m;
271  blue -= m;
272  green -= m;
273 
274  // calculate chroma (colour range)
275  uchar chroma = std::max( {red, blue, green} );
276 
277  // calculate lightened colour value
278  uchar newm = 255 - chroma - m;
279  red += newm;
280  green += newm;
281  blue += newm;
282 
283  return QColor(red, green, blue, alpha);
284 }
