41 def time_string_to_datetime(inp: str) -> Optional[datetime]:
42  try:
43  return datetime.fromisoformat(inp)
44  except ValueError:
45  try:
46  # Support for the trailing "Z" was added in Python 3.11 -- strip it and see if it works now
47  return datetime.fromisoformat(inp[:-1])
48  except ValueError:
49  fci.Console.PrintWarning(f"Unable to parse '{str}' as a Python datetime")
50  return None
