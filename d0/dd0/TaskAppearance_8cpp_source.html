120 void TaskAppearance::slotChangedObject(const Gui::ViewProvider& obj, const App::Property& prop)
121 {
122  // This method gets called if a property of any view provider is changed.
123  // We pick out all the properties for which we need to update this dialog.
124  const std::vector<Gui::ViewProvider*> Provider = getSelection();
125  auto vp = std::find_if(Provider.begin(), Provider.end(), [&obj](Gui::ViewProvider* v) {
126  return v == &obj;
127  });
128 
129  if (vp != Provider.end()) {
130  std::string prop_name = obj.getPropertyName(&prop);
131  if (prop.isDerivedFrom<App::PropertyInteger>()) {
132  long value = static_cast<const App::PropertyInteger&>(prop).getValue();
133  if (prop_name == "Transparency") {
134  bool blocked = ui->spinTransparency->blockSignals(true);
135  ui->spinTransparency->setValue(value);
136  ui->spinTransparency->blockSignals(blocked);
137  blocked = ui->horizontalSlider->blockSignals(true);
138  ui->horizontalSlider->setValue(value);
139  ui->horizontalSlider->blockSignals(blocked);
140  }
141  }
142  else if (prop.isDerivedFrom<App::PropertyFloat>()) {
143  float value = static_cast<const App::PropertyFloat&>(prop).getValue();
144  if (prop_name == "PointSize") {
145  bool blocked = ui->spinPointSize->blockSignals(true);
146  ui->spinPointSize->setValue((int)value);
147  ui->spinPointSize->blockSignals(blocked);
148  }
149  else if (prop_name == "LineWidth") {
150  bool blocked = ui->spinLineWidth->blockSignals(true);
151  ui->spinLineWidth->setValue((int)value);
152  ui->spinLineWidth->blockSignals(blocked);
153  }
154  }
155  }
156 }

224 void TaskAppearance::setDisplayModes(const std::vector<Gui::ViewProvider*>& views)
225 {
226  QStringList commonModes, modes;
227  for (auto it = views.begin(); it != views.end(); ++it) {
228  App::Property* prop = (*it)->getPropertyByName("DisplayMode");
229  if (prop && prop->is<App::PropertyEnumeration>()) {
230  auto display = static_cast<App::PropertyEnumeration*>(prop);
231  if (!display->hasEnums()) {
232  return;
233  }
234  std::vector<std::string> value = display->getEnumVector();
235  if (it == views.begin()) {
236  for (const auto& jt : value) {
237  commonModes << QLatin1String(jt.c_str());
238  }
239  }
240  else {
241  for (const auto& jt : value) {
242  if (commonModes.contains(QLatin1String(jt.c_str()))) {
243  modes << QLatin1String(jt.c_str());
244  }
245  }
246 
247  commonModes = modes;
248  modes.clear();
249  }
250  }
251  }
252 
253  ui->changeMode->clear();
254  ui->changeMode->addItems(commonModes);
255  ui->changeMode->setDisabled(commonModes.isEmpty());
256 
257  // find the display mode to activate
258  for (const auto view : views) {
259  App::Property* prop = view->getPropertyByName("DisplayMode");
260  if (prop && prop->is<App::PropertyEnumeration>()) {
261  auto display = static_cast<App::PropertyEnumeration*>(prop);
262  QString activeMode = QString::fromLatin1(display->getValueAsString());
263  int index = ui->changeMode->findText(activeMode);
264  if (index != -1) {
265  ui->changeMode->setCurrentIndex(index);
266  break;
267  }
268  }
269  }
270 }
