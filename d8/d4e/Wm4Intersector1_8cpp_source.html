75 template <class Real>
76 bool Intersector1<Real>::Find ()
77 {
78  if (m_afU[1] < m_afV[0] || m_afU[0] > m_afV[1])
79  {
80  m_iQuantity = 0;
81  }
82  else if (m_afU[1] > m_afV[0])
83  {
84  if (m_afU[0] < m_afV[1])
85  {
86  m_iQuantity = 2;
87  m_afOverlap[0] = (m_afU[0] < m_afV[0] ? m_afV[0] : m_afU[0]);
88  m_afOverlap[1] = (m_afU[1] > m_afV[1] ? m_afV[1] : m_afU[1]);
89  if (m_afOverlap[0] == m_afOverlap[1])
90  {
91  m_iQuantity = 1;
92  }
93  }
94  else // m_afU[0] == m_afV[1]
95  {
96  m_iQuantity = 1;
97  m_afOverlap[0] = m_afU[0];
98  }
99  }
100  else // m_afU[1] == m_afV[0]
101  {
102  m_iQuantity = 1;
103  m_afOverlap[0] = m_afU[1];
104  }
105 
106  return m_iQuantity > 0;
107 }

109 template <class Real>
110 bool Intersector1<Real>::Test (Real fTMax, Real fSpeedU, Real fSpeedV)
111 {
112  Real fDiffSpeed, fInvDiffSpeed, fDiffPos;
113 
114  if (m_afU[1] < m_afV[0])
115  {
116  // [u0,u1] initially to the left of [v0,v1]
117  fDiffSpeed = fSpeedU - fSpeedV;
118  if (fDiffSpeed > (Real)0.0)
119  {
120  // the intervals must move towards each other
121  fDiffPos = m_afV[0] - m_afU[1];
122  if (fDiffPos <= fTMax*fDiffSpeed)
123  {
124  // the intervals intersect within the specified time
125  fInvDiffSpeed = ((Real)1.0)/fDiffSpeed;
126  m_fFirstTime = fDiffPos*fInvDiffSpeed;
127  m_fLastTime = (m_afV[1] - m_afU[0])*fInvDiffSpeed;
128  return true;
129  }
130  }
131  }
132  else if (m_afU[0] > m_afV[1])

134  // [u0,u1] initially to the right of [v0,v1]
