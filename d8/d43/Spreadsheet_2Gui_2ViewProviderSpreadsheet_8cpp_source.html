112 void ViewProviderSheet::exportAsFile()
113 {
114  auto* sheet = getObject<Spreadsheet::Sheet>();
115  QString selectedFilter;
116  QString formatList = QObject::tr("CSV (*.csv *.CSV);;All (*)");
117  QString fileName = Gui::FileDialog::getSaveFileName(Gui::getMainWindow(),
118  QObject::tr("Export file"),
119  QString(),
120  formatList,
121  &selectedFilter);
122  if (!fileName.isEmpty()) {
123  if (sheet) {
124  char delim = '\0';
125  char quote = '\0';
126  char escape = '\0';
127  std::string errMsg = "Export";
128  bool isValid = sheet->getCharsFromPrefs(delim, quote, escape, errMsg);
129 
130  if (isValid) {
131  sheet->exportToFile(fileName.toStdString(), delim, quote, escape);
132  }
133  else {
134  Base::Console().Error(errMsg.c_str());
135  }
136  }
137  }
138 }
