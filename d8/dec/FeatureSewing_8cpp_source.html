68 App::DocumentObjectExecReturn* Sewing::execute()
69 {
70  // Assign Variables
71  double atol = Tolerance.getValue();
72  bool opt1 = SewingOption.getValue();
73  bool opt2 = DegenerateShape.getValue();
74  bool opt3 = CutFreeEdges.getValue();
75  bool opt4 = Nonmanifold.getValue();
76 
77  try {
78  BRepBuilderAPI_Sewing builder(atol, opt1, opt2, opt3, opt4);
79 
80  std::vector<App::PropertyLinkSubList::SubSet> subset = ShapeList.getSubListValues();
81  for (const auto& it : subset) {
82  // the subset has the documentobject and the element name which belongs to it,
83  // in our case for example the cube object and the "Edge1" string
84  if (it.first->isDerivedFrom<Part::Feature>()) {
85  // we get the shape of the document object which resemble the whole box
86  Part::TopoShape ts = static_cast<Part::Feature*>(it.first)->Shape.getShape();
87 
88  // we want only the subshape which is linked
89  for (const auto& jt : it.second) {
90  TopoDS_Shape sub = ts.getSubShape(jt.c_str());
91  builder.Add(sub);
92  }
93  }
94  else {
95  Standard_Failure::Raise("Shape item not from Part::Feature");
96  }
97  }
98 
99  builder.Perform(); // Perform Sewing
100 
101  TopoDS_Shape aShape = builder.SewedShape(); // Get Shape
102  if (aShape.IsNull()) {
103  return new App::DocumentObjectExecReturn("Resulting shape is null");
104  }
105  this->Shape.setValue(aShape);
106  return StdReturn;
107  }
108  catch (Standard_Failure& e) {
109 
110  return new App::DocumentObjectExecReturn(e.GetMessageString());
111  }
112 }
