87  * The class FemPostObjectSelectionObserver notifies a ViewProviderFemPostObject
88  * only if its selection status has changed
89  */
90 class FemPostObjectSelectionObserver
91 {
92 public:
93  static FemPostObjectSelectionObserver& instance()
94  {
95  static FemPostObjectSelectionObserver inst;
96  return inst;
97  }
98  void registerFemPostObject(ViewProviderFemPostObject* vp)
99  {
100  views.insert(vp);
101  }
102  void unregisterFemPostObject(ViewProviderFemPostObject* vp)
103  {
104  auto it = views.find(vp);
105  if (it != views.end()) {
106  views.erase(it);
107  }
108  }
109 
110  void selectionChanged(const Gui::SelectionChanges& msg)
111  {
112  Gui::SelectionObject obj(msg);
113  auto findVP = std::find_if(views.begin(), views.end(), [&obj](const auto& vp) {
114  return obj.getObject() == vp->getObject();
115  });
116 
117  if (findVP != views.end()) {
118  (*findVP)->onSelectionChanged(msg);
119  }
120  }
121 
122 private:
123  FemPostObjectSelectionObserver()
124  {
125  // NOLINTBEGIN
126  this->connectSelection = Gui::Selection().signalSelectionChanged.connect(
127  std::bind(&FemPostObjectSelectionObserver::selectionChanged, this, sp::_1));
128  // NOLINTEND
129  }
130 
131  ~FemPostObjectSelectionObserver() = default;
132 
133 public:
134  FemPostObjectSelectionObserver(const FemPostObjectSelectionObserver&) = delete;
135  FemPostObjectSelectionObserver& operator=(const FemPostObjectSelectionObserver&) = delete;
136 
137 private:
138  std::set<ViewProviderFemPostObject*> views;
139  using Connection = boost::signals2::scoped_connection;
140  Connection connectSelection;

398 void ViewProviderFemPostObject::updateProperties()
399 {
400 
401  m_blockPropertyChanges = true;
402  vtkPolyData* poly = m_currentAlgorithm->GetOutput();
403 
404  // coloring
405  std::string val;
406  if (Field.hasEnums() && Field.getValue() >= 0) {
407  val = Field.getValueAsString();
408  }
409 
410  std::vector<std::string> colorArrays;
411  colorArrays.emplace_back("None");
412 
413  vtkPointData* point = poly->GetPointData();
414  for (int i = 0; i < point->GetNumberOfArrays(); ++i) {
415  std::string FieldName = point->GetArrayName(i);
416  if (FieldName != "Texture Coordinates") {
417  colorArrays.push_back(FieldName);
418  }
419  }
420 
421  vtkCellData* cell = poly->GetCellData();
422  for (int i = 0; i < cell->GetNumberOfArrays(); ++i) {
423  colorArrays.emplace_back(cell->GetArrayName(i));
424  }
425 
426  App::Enumeration empty;
427  Field.setValue(empty);
428  m_coloringEnum.setEnums(colorArrays);
429  Field.setValue(m_coloringEnum);
430 
431  std::vector<std::string>::iterator it = std::find(colorArrays.begin(), colorArrays.end(), val);
432  if (!val.empty() && it != colorArrays.end()) {
433  Field.setValue(val.c_str());
434  }
435 
436  Field.purgeTouched();
437 
438  // Vector mode
439  if (VectorMode.hasEnums() && VectorMode.getValue() >= 0) {
440  val = VectorMode.getValueAsString();
441  }
442 
443  colorArrays.clear();
444  if (Field.getValue() == 0) {
445  colorArrays.emplace_back("Not a vector");
446  }
447  else {
448  int array = Field.getValue() - 1; // 0 is none
449  vtkDataArray* data = point->GetArray(array);
450  if (!data) {
451  return;
452  }
453 
454  if (data->GetNumberOfComponents() == 1) {
455  colorArrays.emplace_back("Not a vector");
456  }
457  else {
458  colorArrays.emplace_back("Magnitude");
459  if (data->GetNumberOfComponents() >= 2) {
460  colorArrays.emplace_back("X");
461  colorArrays.emplace_back("Y");
462  }
463  if (data->GetNumberOfComponents() >= 3) {
464  colorArrays.emplace_back("Z");
465  }
466  }
467  }
468 
469  VectorMode.setValue(empty);
470  m_vectorEnum.setEnums(colorArrays);
471  VectorMode.setValue(m_vectorEnum);
472 
473  it = std::find(colorArrays.begin(), colorArrays.end(), val);
474  if (!val.empty() && it != colorArrays.end()) {
475  VectorMode.setValue(val.c_str());
476  }
477 
478  m_blockPropertyChanges = false;
479 }

494  points = pd->GetPoints();

506

509  cells = pd->GetPolys();

587 void ViewProviderFemPostObject::WritePointData(vtkPoints* points,
588  vtkDataArray* normals,
589  vtkDataArray* tcoords)
590 {
591  Q_UNUSED(tcoords);
592 
593  if (!points) {
594  return;
595  }
596 
597  m_coordinates->point.setNum(points->GetNumberOfPoints());
598  SbVec3f* pnts = m_coordinates->point.startEditing();
599  for (int i = 0; i < points->GetNumberOfPoints(); i++) {
600  double* p = points->GetPoint(i);
601  pnts[i].setValue(p[0], p[1], p[2]);
602  }
603  m_coordinates->point.finishEditing();
604 
605  // write out the point normal data
606  if (normals) {
607  m_normals->vector.setNum(normals->GetNumberOfTuples());
608  SbVec3f* dirs = m_normals->vector.startEditing();
609  for (int i = 0; i < normals->GetNumberOfTuples(); i++) {
610  double* p = normals->GetTuple(i);
611  dirs[i].setValue(p[0], p[1], p[2]);
612  }
613  m_normals->vector.finishEditing();
614 
615  m_normalBinding->value = SoNormalBinding::PER_VERTEX_INDEXED;
616  m_normalBinding->value.touch();
617  }

620 void ViewProviderFemPostObject::setRangeOfColorBar(float min, float max)
621 {
622  try {
623  // setRange expects max value greater than min value.
624  // A typical case is max equal to min, so machine epsilon
625  // is used to overwrite and differentiate both values
626  if (min >= max) {
627  static constexpr float eps = std::numeric_limits<float>::epsilon();
628  if (max > 0) {
629  min = max * (1 - eps);
630  max = max * (1 + eps);
631  }
632  else if (max < 0) {
633  min = max * (1 + eps);
634  max = max * (1 - eps);
635  }
636  else {
637  static constexpr float minF = std::numeric_limits<float>::min();
638  min = -1 * minF;
639  max = minF;
640  }
641  }
642  m_colorBar->setRange(min, max);
643  }
644  catch (const Base::ValueError& e) {
645  e.ReportException();
646  }
647 }

654 void ViewProviderFemPostObject::WriteColorData(bool ResetColorBarRange)
655 {
656  if (!setupPipeline()) {
657  return;
658  }
659 
660  if (Field.getEnumVector().empty() || Field.getValue() == 0) {
661  m_material->diffuseColor.setValue(SbColor(0.8, 0.8, 0.8));
662  float trans = float(Transparency.getValue()) / 100.0;
663  m_material->transparency.setValue(trans);
664  m_materialBinding->value = SoMaterialBinding::OVERALL;
665  m_materialBinding->touch();
666  // since there is no field, set the range to the default
667  // range as if a new object is created
668  setRangeOfColorBar(-0.5, 0.5);
669  return;
670  };
671 
672  int array = Field.getValue() - 1; // 0 is none
673  vtkPolyData* pd = m_currentAlgorithm->GetOutput();

770 void ViewProviderFemPostObject::filterArtifacts(vtkDataSet* dset)
771 {
772  // The problem is that in the surface view the boundary regions of the volumes
773  // calculated by the different CPU cores is always visible, independent of the
774  // transparency setting. Elmer is not to blame because this is a property of the
775  // partial VTK file reader. So this can happen with various inputs
776  // since FreeCAD can also be used to view VTK files without the need to perform
777  // an analysis. Therefore it is impossible to know in advance when a filter
778  // is necessary or not.
779  // Only for pure CCX analyses we know that no filtering is necessary. However,
780  // the effort to catch this case is not worth it since the filtering is
781  // only as time-consuming as enabling the surface filter. In fact, it is like
782  // performing the surface filter twice.
783 
784  // We need to set the filter clipping plane below the z-minimum of the data.
785  // We can either do this by checking the VTK data or by getting the info from
786  // the 3D view. We use here the latter because this is much faster.
787 
788  // since we will set the filter according to the visible bounding box
789  // assure the object is visible
790  bool visibility = this->Visibility.getValue();
791  if (!visibility) {
792  this->Visibility.setValue(true);
793  }
794  m_blockPropertyChanges = true;
795 
796  Gui::Document* doc = this->getDocument();
797  Gui::View3DInventor* view =
798  qobject_cast<Gui::View3DInventor*>(doc->getViewOfViewProvider(this));
799 
800  if (view) {
801  Gui::View3DInventorViewer* viewer = view->getViewer();
802  SbBox3f boundingBox;
803  boundingBox = viewer->getBoundingBox();
804  if (boundingBox.hasVolume()) {
805  // setup
806  vtkSmartPointer<vtkImplicitFunction> m_implicit;
807  auto m_plane = vtkSmartPointer<vtkPlane>::New();
808  m_implicit = m_plane;
809  m_plane->SetNormal(0., 0., 1.);
810  auto extractor = vtkSmartPointer<vtkTableBasedClipDataSet>::New();
811  float dx, dy, dz;
812  boundingBox.getSize(dx, dy, dz);
813  // Set plane below the minimum to assure there are
814  // no boundary cells (touching the function) and for Warp filters
815  // the user might change the warp factor a lot. Thus set
816  // 10 times dz to be safe even for unrealistic warp deformations
817  m_plane->SetOrigin(0., 0., -10 * dz);
818  extractor->SetClipFunction(m_implicit);
819  extractor->SetInputData(dset);
820  extractor->Update();
821  auto extractorResult = extractor->GetOutputDataObject(0);
822  if (extractorResult) {
823  m_surface->SetInputData(extractorResult);
824  }
825  else {
826  m_surface->SetInputData(dset);
827  }
828  }
829  else {
830  // for the case that there are only 2D objects
831  m_surface->SetInputData(dset);
832  }
833  }
834 
835  m_blockPropertyChanges = false;
836 
837  // restore initial vsibility
838  if (!visibility) {
839  this->Visibility.setValue(visibility);
840  }
841 }

843 bool ViewProviderFemPostObject::setupPipeline()
844 {
845  if (m_blockPropertyChanges) {
846  return false;
847  }
848 
849  auto postObject = getObject<Fem::FemPostObject>();
850 
851  vtkDataObject* data = postObject->Data.getValue();
852  if (!data) {
853  return false;
854  }
855 
856  // check all fields if there is a real/imaginary one and if so
857  // add a field with an absolute value
858  vtkSmartPointer<vtkDataObject> SPdata = data;
859  vtkDataSet* dset = vtkDataSet::SafeDownCast(SPdata);
860  if (!dset) {
861  return false;
862  }
863  std::string FieldName;
864  auto numFields = dset->GetPointData()->GetNumberOfArrays();
865  for (int i = 0; i < numFields; ++i) {
866  FieldName = std::string(dset->GetPointData()->GetArrayName(i));
867  addAbsoluteField(dset, FieldName);
868  }
869 
870  m_outline->SetInputData(dset);
871  m_points->SetInputData(dset);
872  m_wireframe->SetInputData(dset);
873 
874  // Filtering artifacts is necessary for partial VTU files (*.pvtu) independent of the
875  // current Elmer CPU core settings because the user might load an external file.
876  // It is only necessary for the surface filter.
877  // The problem is that when opening an existing FreeCAD file, we get no information how the
878  // Data of the postObject was once created. The vtkDataObject type does not provide this info.
879  // Therefore the only way is the hack to filter only if the used Elmer CPU cores are > 1.
880  auto hGrp = App::GetApplication().GetParameterGroupByPath(
881  "User parameter:BaseApp/Preferences/Mod/Fem/Elmer");
882  bool FilterMultiCPUResults = hGrp->GetBool("FilterMultiCPUResults", true);
883  int UseNumberOfCores = hGrp->GetInt("UseNumberOfCores", 1);
884  // filtering is only necessary for pipelines and warp filters
885  if (FilterMultiCPUResults && (UseNumberOfCores > 1)
886  && ((postObject->getTypeId() == Base::Type::fromName("Fem::FemPostPipeline"))
887  || (postObject->getTypeId() == Base::Type::fromName("Fem::FemPostWarpVectorFilter")))) {
888  filterArtifacts(dset);
889  }
890  else {
891  m_surface->SetInputData(dset);
892  }
893 
894  return true;
895 }

1099 // if there is a real and an imaginary field, an absolute field is added
1100 void ViewProviderFemPostObject::addAbsoluteField(vtkDataSet* dset, std::string FieldName)
1101 {
1102  // real field names have the suffix " re", given by Elmer
1103  // if the field does not have this suffix, we can return
1104  auto suffix = FieldName.substr(FieldName.size() - 3, FieldName.size() - 1);
1105  if (strcmp(suffix.c_str(), " re") != 0) {
1106  return;
1107  }
1108 
1109  // absolute fields might have already been created, then do nothing
1110  auto strAbsoluteFieldName = FieldName.substr(0, FieldName.size() - 2) + "abs";
1111  vtkDataArray* testArray = dset->GetPointData()->GetArray(strAbsoluteFieldName.c_str());
1112  if (testArray) {
1113  return;
1114  }
1115 
1116  // safety check
1117  vtkDataArray* realDdata = dset->GetPointData()->GetArray(FieldName.c_str());
1118  if (!realDdata) {
1119  return;
1120  }
1121 
1122  // now check if the imaginary counterpart exists
1123  auto strImaginaryFieldName = FieldName.substr(0, FieldName.size() - 2) + "im";
1124  vtkDataArray* imagDdata = dset->GetPointData()->GetArray(strImaginaryFieldName.c_str());
1125  if (!imagDdata) {
1126  return;
1127  }
1128 
1129  // create a new array and copy over the real data
1130  // since one cannot directly access the values of a vtkDataSet
1131  // we need to copy them over in a loop
1132  vtkSmartPointer<vtkDoubleArray> absoluteData = vtkSmartPointer<vtkDoubleArray>::New();
1133  absoluteData->SetNumberOfComponents(realDdata->GetNumberOfComponents());
1134  auto numTuples = realDdata->GetNumberOfTuples();
1135  absoluteData->SetNumberOfTuples(numTuples);
1136  double tuple[] = {0, 0, 0};
1137  for (vtkIdType i = 0; i < numTuples; ++i) {
1138  absoluteData->SetTuple(i, tuple);
1139  }
1140  // name the array
1141  auto strAbsFieldName = FieldName.substr(0, FieldName.size() - 2) + "abs";
1142  absoluteData->SetName(strAbsFieldName.c_str());
1143 
1144  // add array to data set
1145  dset->GetPointData()->AddArray(absoluteData);
1146 
1147  // step through all mesh points and calculate them
1148  double realValue = 0;
1149  double imaginaryValue = 0;
1150  double absoluteValue = 0;
1151  for (int i = 0; i < dset->GetNumberOfPoints(); ++i) {
1152  if (absoluteData->GetNumberOfComponents() == 1) {
1153  realValue = realDdata->GetComponent(i, 0);
1154  imaginaryValue = imagDdata->GetComponent(i, 0);
1155  absoluteValue = sqrt(pow(realValue, 2) + pow(imaginaryValue, 2));
1156  absoluteData->SetComponent(i, 0, absoluteValue);
1157  }
1158  // if field is a vector
1159  else {
1160  for (int j = 0; j < absoluteData->GetNumberOfComponents(); ++j) {
1161  realValue = realDdata->GetComponent(i, j);
1162  imaginaryValue = imagDdata->GetComponent(i, j);
1163  absoluteValue = sqrt(pow(realValue, 2) + pow(imaginaryValue, 2));
1164  absoluteData->SetComponent(i, j, absoluteValue);
1165  }
1166  }
1167  }
