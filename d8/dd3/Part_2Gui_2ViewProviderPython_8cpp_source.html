41 void ViewProviderCustom::onChanged(const App::Property* prop)
42 {
43  std::map<const App::Property*, Gui::ViewProvider*>::iterator it;
44  for (it = propView.begin(); it != propView.end(); ++it) {
45  App::Property* view = it->second->getPropertyByName(prop->getName());
46  if (view) {
47  App::Property* copy = prop->Copy();
48  if (copy) {
49  view->Paste(*copy);
50  delete copy;
51  }
52  }
53  }
54  PartGui::ViewProviderPart::onChanged(prop);
55 }

57 void ViewProviderCustom::updateData(const App::Property* prop)
58 {
59  if (prop->isDerivedFrom<App::PropertyComplexGeoData>()) {
60  std::map<const App::Property*, Gui::ViewProvider*>::iterator it = propView.find(prop);
61  if (it == propView.end()) {
62  Gui::ViewProvider* view = Gui::ViewProviderBuilder::create(prop->getTypeId());
63  if (view) {
64  if (view->isDerivedFrom<Gui::ViewProviderDocumentObject>()) {
65  static_cast<Gui::ViewProviderDocumentObject*>(view)->attach(this->getObject());
66  static_cast<Gui::ViewProviderDocumentObject*>(view)->setDisplayMode(this->getActiveDisplayMode().c_str());
67  }
68  propView[prop] = view;
69  view->updateData(prop);
70  this->getRoot()->addChild(view->getRoot());
71  }
72  }
73  else {
74  it->second->updateData(prop);
75  }
76  }
77 }
