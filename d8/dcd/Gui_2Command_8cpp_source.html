358  auto pcActionGroup = qobject_cast<Gui::ActionGroup*>(_pcAction);
359  if(pcActionGroup) {
360  QList<QAction*> a = pcActionGroup->actions();
361  assert(iMsg < a.size());

364  action = _pcAction->action();
365 
366  if(!action)
367  return;
368 
369  bool checkable = action->isCheckable();

371  if(checkable) {

373  switch(triggerSource()) {
374  case TriggerNone:
375  checked = !action->isChecked();
376  break;
377  case TriggerAction:

385  action->setChecked(checked);
386  action->blockSignals(blocked);

411 void Command::_invoke(int id, bool disablelog)
412 {
413  try {
414  // Because Transaction now captures ViewObject changes, auto named
415  // transaction is disabled here to avoid too many unnecessary transactions.
416  //
417  App::AutoTransaction committer(nullptr, true);
418 
419  // set the application module type for the macro
420  getGuiApplication()->macroManager()->setModule(sAppModule);
421 
422  std::unique_ptr<LogDisabler> logdisabler;
423  if (disablelog) {
424  logdisabler = std::make_unique<LogDisabler>();
425  }
426 
427  // check if it really works NOW (could be a delay between click deactivation of the button)
428  if (isActive()) {
429  auto manager = getGuiApplication()->macroManager();
430  auto editDoc = getGuiApplication()->editDocument();
431 
432  if (!logdisabler) {
433  activated(id);
434  }
435  else {
436  Gui::SelectionLogDisabler seldisabler;
437  auto lines = manager->getLines();
438  std::ostringstream ss;
439  ss << "### Begin command " << sName;
440  // Add a pending line to mark the start of a command
441  PendingLine pending(MacroManager::Cmt, ss.str().c_str());
442  ss.str("");
443 
444  activated(id);
445 
446  if (manager->getLines() == lines) {
447  // This command does not record any lines, lets do it for
448  // it. The above LogDisabler is to prevent nested command
449  // logging, i.e. we only auto log the first invoking
450  // command.
451 
452  // Cancel the above pending line first
453  pending.cancel();
454  ss << "Gui.runCommand('" << sName << "'," << id << ')';
455  manager->addLine(MacroManager::Gui, ss.str().c_str());
456  }
457  else {
458  // In case the command has any output to the console, lets
459  // mark the end of the command here
460  ss << "### End command " << sName;
461  manager->addLine(MacroManager::Cmt, ss.str().c_str());
462  }
463  }
464 
465  getMainWindow()->updateActions();
466 
467  // If this command starts an editing, let the transaction persist
468  if (!editDoc && getGuiApplication()->editDocument())
469  committer.setEnable(false);
470  }
471  }
472  catch (const Base::SystemExitException&) {
473  throw;
474  }
475  catch (Base::PyException &e) {
476  e.ReportException();
477  }
478  catch (Py::Exception&) {
479  Base::PyGILStateLocker lock;
480  Base::PyException e;
481  e.ReportException();
482  }
483  catch (Base::AbortException&) {
484  }
485  catch (Base::Exception &e) {
486  e.ReportException();
487  // Pop-up a dialog for FreeCAD-specific exceptions
488  QMessageBox::critical(Gui::getMainWindow(), QObject::tr("Exception"), QLatin1String(e.what()));
489  }
490  catch (std::exception &e) {
491  Base::Console().Error("C++ exception thrown (%s)\n", e.what());
492  }
493  catch (const char* e) {
494  Base::Console().Error("%s\n", e);
495  }
496 #ifndef FC_DEBUG
497  catch (...) {
498  Base::Console().Error("Gui::Command::activated(%d): Unknown C++ exception thrown\n", id);
499  }
500 #endif
501 }

503 void Command::testActive()
504 {
505  if (!_pcAction)
506  return;
507 
508  if (_blockCmd || !bEnabled) {
509  _pcAction->setEnabled(false);
510  return;
511  }
512 
513  if (!(eType & ForEdit)) { // special case for commands which are only in some edit modes active
514 
515  if ((!Gui::Control().isAllowedAlterDocument() && eType & AlterDoc) ||
516  (!Gui::Control().isAllowedAlterView() && eType & Alter3DView) ||
517  (!Gui::Control().isAllowedAlterSelection() && eType & AlterSelection)) {
518  _pcAction->setEnabled(false);
519  return;
520  }
521  }
522 
523  auto pcAction = qobject_cast<Gui::ActionGroup*>(_pcAction);
524  if(pcAction) {
525  Gui::CommandManager &rcCmdMgr = Gui::Application::Instance->commandManager();
526  const auto actions = pcAction->actions();
527  for(auto action : actions) {
528  auto name = action->property("CommandName").toByteArray();
529  if(!name.size())
530  continue;
531  Command* cmd = rcCmdMgr.getCommandByName(name);
532  if(cmd)
533  action->setEnabled(cmd->isActive());
534  }
535  }
536 
537  bool bActive = isActive();
538  _pcAction->setEnabled(bActive);
539 }

783 void Command::_copyVisual(const char *file, int line, const App::DocumentObject *to, const char* attr_to, const App::DocumentObject *from, const char *attr_from)
784 {
785  if(!from || !from->isAttachedToDocument() || !to || !to->isAttachedToDocument())
786  return;
787  static std::map<std::string,std::string> attrMap = {
788  // {"ShapeColor","ShapeMaterial.DiffuseColor"},
789  {"ShapeAppearance", "ShapeMaterial"},
790  // {"LineColor","ShapeMaterial.DiffuseColor"},
791  // {"PointColor","ShapeMaterial.DiffuseColor"},
792  {"Transparency","Transparency"},
793  };
794  auto it = attrMap.find(attr_to);
795  auto objCmd = getObjectCmd(to);
796  if(it!=attrMap.end()) {
797  auto obj = from;
798  for(int depth=0;;++depth) {
799  auto vp = dynamic_cast<Gui::ViewProviderLink*>(
800  Gui::Application::Instance->getViewProvider(obj));
801  if(vp && vp->OverrideMaterial.getValue()) {
802  _doCommand(file,line,Gui,"%s.ViewObject.%s=%s.ViewObject.%s",
803  objCmd.c_str(),attr_to,getObjectCmd(obj).c_str(),it->second.c_str());
804  return;
805  }
806  auto linked = obj->getLinkedObject(false,nullptr,false,depth);
807  if(!linked || linked==obj)
808  break;
809  obj = linked;
810  }
811  }
812 
813  try {

933 void Command::adjustCameraPosition()
934 {
935  Gui::Document* doc = Gui::Application::Instance->activeDocument();
936  if (doc) {
937  auto view = static_cast<Gui::View3DInventor*>(doc->getActiveView());
938  Gui::View3DInventorViewer* viewer = view->getViewer();
939  SoCamera* camera = viewer->getSoRenderManager()->getCamera();
940  if (!camera || !camera->isOfType(SoOrthographicCamera::getClassTypeId()))
941  return;
942

944  SoGetBoundingBoxAction action(viewer->getSoRenderManager()->getViewportRegion());
945  action.apply(viewer->getSceneGraph());
946  SbBox3f box = action.getBoundingBox();
947  if (box.isEmpty())
948  return;
949 
950  // get cirumscribing sphere and check if camera is inside
951  SbVec3f cam_pos = camera->position.getValue();
952  SbVec3f box_cnt = box.getCenter();
953  SbSphere bs;
954  bs.circumscribe(box);
955  float radius = bs.getRadius();
956  float distance_to_midpoint = (box_cnt-cam_pos).length();
957  if (radius >= distance_to_midpoint) {
958  // Move the camera to the edge of the bounding sphere, while still
959  // pointing at the scene.
960  SbVec3f direction = cam_pos - box_cnt;
961  (void) direction.normalize(); // we know this is not a null vector
962  camera->position.setValue(box_cnt + direction * radius);
963 
964  // New distance to mid point
965  distance_to_midpoint =
966  (camera->position.getValue() - box.getCenter()).length();
967  camera->nearDistance = distance_to_midpoint - radius;
968  camera->farDistance = distance_to_midpoint + radius;
969  camera->focalDistance = distance_to_midpoint;
970  }
