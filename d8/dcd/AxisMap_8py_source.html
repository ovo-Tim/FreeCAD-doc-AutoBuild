103  def execute(self, obj):
104 
105  inAxis = obj.AxisMap[0]
106  outAxis = obj.AxisMap[3]
107  d = 0.1
108 
109  if obj.Base:
110  if obj.Base.isDerivedFrom("Path::Feature"):
111  if obj.Base.Path:
112  if obj.Base.Path.Commands:
113  pp = PathUtils.getPathWithPlacement(obj.Base).Commands
114  if len([i for i in pp if i.Name in Path.Geom.CmdMoveArc]) == 0:
115  pathlist = pp
116  else:
117  pathlist = self._stripArcs(pp, d)
118 
119  newcommandlist = []
120  currLocation = {"X": 0, "Y": 0, "Z": 0, "F": 0}
121 
122  for c in pathlist:
123  newparams = dict(c.Parameters)
124  remapvar = newparams.pop(inAxis, None)
125  if remapvar is not None:
126  newparams[outAxis] = self._linear2angular(obj.Radius, remapvar)
127  locdiff = dict(set(newparams.items()) - set(currLocation.items()))
128  if (
129  len(locdiff) == 1 and outAxis in locdiff
130  ): # pure rotation. Calculate rotational feed rate
131  if "F" in c.Parameters:
132  feed = c.Parameters["F"]
133  else:
134  feed = currLocation["F"]
135  newparams.update({"F": self._linear2angular(obj.Radius, feed)})
136  newcommand = Path.Command(c.Name, newparams)
137  newcommandlist.append(newcommand)
138  currLocation.update(newparams)
139  else:
140  newcommandlist.append(c)
141  currLocation.update(c.Parameters)
