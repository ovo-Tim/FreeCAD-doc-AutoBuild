75 QRgb BaseDelegate::parseColor(const QString& color) const
76 {
77  QString trimmed = color;
78  trimmed.replace(QRegularExpression(QString::fromStdString("\\(([^<]*)\\)")),
79  QString::fromStdString("\\1"));
80  QStringList parts = trimmed.split(QString::fromStdString(","));
81  if (parts.length() < 3) {
82  return qRgba(0, 0, 0, 255);
83  }
84  int red = parts.at(0).toDouble() * 255;
85  int green = parts.at(1).toDouble() * 255;
86  int blue = parts.at(2).toDouble() * 255;
87  int alpha = 255;
88  if (parts.length() > 3) {
89  alpha = parts.at(3).toDouble() * 255;
90  }
91 
92  return qRgba(red, green, blue, alpha);
93 }

160 void BaseDelegate::paintColor(QPainter* painter,
161  const QStyleOptionViewItem& option,
162  const QModelIndex& index) const
163 {
164  auto propertyValue = getStringValue(index);
165  painter->save();
166 
167  QColor color;
168  color.setRgba(qRgba(0, 0, 0, 255)); // Black border
169  int left = option.rect.left() + 2;
170  int width = option.rect.width() - 4;
171  if (option.rect.width() > 75) {
172  left += (option.rect.width() - 75) / 2;
173  width = 71;
174  }
175  painter->fillRect(left, option.rect.top() + 2, width, option.rect.height() - 4, QBrush(color));
176 
177  color.setRgba(parseColor(propertyValue));
178  left = option.rect.left() + 5;
179  width = option.rect.width() - 10;
180  if (option.rect.width() > 75) {
181  left += (option.rect.width() - 75) / 2;
182  width = 65;
183  }
184  painter->fillRect(left, option.rect.top() + 5, width, option.rect.height() - 10, QBrush(color));
185 
186  painter->restore();
187 }

388 QWidget* BaseDelegate::createEditor(QWidget* parent,
389  const QStyleOptionViewItem& styleOption,
390  const QModelIndex& index) const
391 {
392  Q_UNUSED(styleOption)
393 
394  auto model = index.model();
395 
396  // If this is a new row then it has to be added before the editor is created. Otherwise
397  // Adding the row while the editor is active can change where the editor is placed.
398  if (newRow(model, index)) {

402 
403  QWidget* editor = createWidget(parent, item, index);

408 QWidget*
409 BaseDelegate::createWidget(QWidget* parent, const QVariant& item, const QModelIndex& index) const
410 {
411  QWidget* widget = nullptr;
412 
413  auto type = getType(index);
414  if (type == Materials::MaterialValue::Integer) {
415  auto spinner = new Gui::UIntSpinBox(parent);
416  spinner->setMinimum(0);
417  spinner->setMaximum(UINT_MAX);
418  spinner->setValue(item.toUInt());
419  widget = spinner;
420  }
421  else if (type == Materials::MaterialValue::Float) {
422  auto spinner = new Gui::DoubleSpinBox(parent);
423 
424  // the magnetic permeability is the parameter for which many decimals matter
425  // the most however, even for this, 6 digits are sufficient
426  spinner->setDecimals(6);
427 
428  // for almost all Float parameters of materials a step of 1 would be too large
429  spinner->setSingleStep(0.1);
430 
431  spinner->setMinimum(std::numeric_limits<double>::min());
432  spinner->setMaximum(std::numeric_limits<double>::max());
433  spinner->setValue(item.toDouble());
434  widget = spinner;
435  }
436  else if (type == Materials::MaterialValue::Boolean) {
437  auto combo = new Gui::PrefComboBox(parent);
438  combo->insertItem(0, QString::fromStdString(""));
439  combo->insertItem(1, tr("False"));
440  combo->insertItem(2, tr("True"));
441  combo->setCurrentText(item.toString());
442  widget = combo;
443  }
444  else if (type == Materials::MaterialValue::Quantity) {
445  auto input = new Gui::QuantitySpinBox(parent);
446  input->setMinimum(std::numeric_limits<double>::min());
447  input->setMaximum(std::numeric_limits<double>::max());
448  input->setUnitText(getUnits(index));
449  input->setValue(item.value<Base::Quantity>());
450 
451  widget = input;
452  }
453  else if (type == Materials::MaterialValue::File) {
454  auto chooser = new Gui::FileChooser(parent);
455  if (!item.toString().isEmpty()) {
456  chooser->setFileName(item.toString());
457  }
458 
459  widget = chooser;
460  }
461  else if (type == Materials::MaterialValue::FileList) {
462  auto chooser = new Gui::FileChooser(parent);
463  auto propertyValue = item.toString();
464 
465  connect(chooser,
466  &Gui::FileChooser::fileNameChanged,
467  [this, chooser, index](const QString&) {
468  setModelData(chooser, const_cast<QAbstractItemModel*>(index.model()), index);
469  });
470 
471  connect(chooser,
472  &Gui::FileChooser::fileNameSelected,
473  [this, chooser, index](const QString&) {
474  setModelData(chooser, const_cast<QAbstractItemModel*>(index.model()), index);
475  });
476  widget = chooser;
477  }
478  else {
479  // Default editor
480  auto lineEdit = new Gui::PrefLineEdit(parent);
481  lineEdit->setText(item.toString());
482  widget = lineEdit;
483  }
484 
485  return widget;
486 }
