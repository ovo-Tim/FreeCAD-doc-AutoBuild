44 {

199 // The first template parameter represents the basic geometric class e.g. Rotation,
200 // the second parameter is reserved for its Python binding class, i.e. RotationPy.
201 // The third template parameter is the definition of a pointer to the method
202 // of the Python binding class to return the managed geometric instance. In our
203 // example this is the method RotationPy::getRotationPtr.
204 template<class T, class PyT, T* (PyT::*valuePtr)() const>
205 class GeometryT: public Object // NOLINT
206 {
207 public:
208  explicit GeometryT(PyObject* pyob, bool owned)
209  : Object(pyob, owned)
210  {
211  validate();
212  }
213  GeometryT(const GeometryT& ob)
214  : Object(*ob)
215  {
216  validate();
217  }
218  explicit GeometryT()
219  {
220  set(new PyT(new T()), true);
221  validate();
222  }
223  explicit GeometryT(const T& val)
224  {
225  set(new PyT(new T(val)), true);
226  validate();
227  }
228  GeometryT(const Object& other)
229  : Object(other.ptr())
230  {
231  validate();
232  }
233  bool accepts(PyObject* pyob) const override
234  {
235  return pyob && Geometry_TypeCheck(pyob);
236  }
237  GeometryT& operator=(const Object& rhs)
238  {
239  *this = *rhs;
240  return *this;
241  }
242  GeometryT& operator=(PyObject* rhsp)
243  {
244  if (ptr() == rhsp) {
245  return *this;
246  }
247  set(rhsp, false);
248  return *this;
249  }
250  GeometryT& operator=(const T& val)
251  {
252  set(new PyT(val), true);
253  return *this;
254  }
255  const T& getValue() const
256  {
257  // cast the PyObject pointer to the matching sub-class
258  // and call then the defined member function
259  PyT* py = static_cast<PyT*>(ptr());
260  T* val = (py->*valuePtr)();
261  return *val;
262  }
263  operator T() const
264  {
265  return getValue();
266  }
267  PyT* getPy() const
268  {
269  PyT* py = static_cast<PyT*>(ptr());
270  return py;
271  }
272 
273 private:
274  static int Geometry_TypeCheck(PyObject* obj)
275  {
276  return PyObject_TypeCheck(obj, &(PyT::Type));
277  }
278 };
