61 App::DocumentObjectExecReturn* SegmentByMesh::execute()
62 {
63  Mesh::PropertyMeshKernel* kernel = nullptr;
64  App::DocumentObject* mesh = Source.getValue();
65  if (mesh) {
66  App::Property* prop = mesh->getPropertyByName("Mesh");
67  if (prop && prop->is<Mesh::PropertyMeshKernel>()) {
68  kernel = static_cast<Mesh::PropertyMeshKernel*>(prop);
69  }
70  }
71  if (!kernel) {
72  return new App::DocumentObjectExecReturn("No mesh specified.\n");
73  }
74  if (mesh->isError()) {
75  return new App::DocumentObjectExecReturn("No valid mesh.\n");
76  }
77 
78  Mesh::PropertyMeshKernel* toolmesh = nullptr;
79  App::DocumentObject* tool = Tool.getValue();
80  if (tool) {
81  App::Property* prop = tool->getPropertyByName("Mesh");
82  if (prop && prop->is<Mesh::PropertyMeshKernel>()) {
83  toolmesh = static_cast<Mesh::PropertyMeshKernel*>(prop);
84  }
85  }
86  if (!toolmesh) {
87  return new App::DocumentObjectExecReturn("No toolmesh specified.\n");
88  }
89  if (tool->isError()) {
90  return new App::DocumentObjectExecReturn("No valid toolmesh.\n");
91  }
92 
93  // the clipping plane
94  Base::Vector3f cBase, cNormal;
95  cBase = Base::convertTo<Base::Vector3f>(Base.getValue());
96  cNormal = Base::convertTo<Base::Vector3f>(Normal.getValue());
97 
98 
99  const MeshKernel& rMeshKernel = kernel->getValue().getKernel();
100  const MeshKernel& rToolMesh = toolmesh->getValue().getKernel();
101 
102  // check if the toolmesh is a solid
103  if (!MeshEvalSolid(rToolMesh).Evaluate()) {
104  return new App::DocumentObjectExecReturn("Toolmesh is not solid.\n");
105  }
106 
107  std::vector<MeshCore::FacetIndex> faces;
108 
109  MeshAlgorithm cAlg(rMeshKernel);
110  if (cNormal.Length() > 0.1F) { // not a null vector
111  cAlg.GetFacetsFromToolMesh(rToolMesh, cNormal, faces);
112  }
113  else {
114  cAlg.GetFacetsFromToolMesh(rToolMesh, Base::Vector3f(0.0, 1.0F, 0.0F), faces);
115  }
116 
117  // if the clipping plane was set then we want only the visible facets
118  if (cNormal.Length() > 0.1F) { // not a null vector
119  // now we have too many facets since we have (invisible) facets near to the back clipping
120  // plane, so we need the nearest facet to the front clipping plane
121  //
122  float fDist = FLOAT_MAX;
123  MeshCore::FacetIndex uIdx = MeshCore::FACET_INDEX_MAX;
124  MeshFacetIterator cFIt(rMeshKernel);
125 
126  // get the nearest facet to the user (front clipping plane)
127  for (MeshCore::FacetIndex it : faces) {
128  cFIt.Set(it);
129  float dist = (float)fabs(cFIt->GetGravityPoint().DistanceToPlane(cBase, cNormal));
130  if (dist < fDist) {
131  fDist = dist;
132  uIdx = it;
133  }
134  }
135 
136  // succeeded
137  if (uIdx != MeshCore::FACET_INDEX_MAX) {
138  // set VISIT-Flag to all outer facets
139  cAlg.SetFacetFlag(MeshFacet::VISIT);
140  cAlg.ResetFacetsFlag(faces, MeshFacet::VISIT);
141 
142  faces.clear();
143  MeshTopFacetVisitor clVisitor(faces);
144  rMeshKernel.VisitNeighbourFacets(clVisitor, uIdx);
145 
146  // append also the start facet
147  faces.push_back(uIdx);
148  }
149  }
150 
151  std::vector<MeshGeomFacet> aFaces;
152  aFaces.reserve(faces.size());
153  for (MeshCore::FacetIndex it : faces) {
154  aFaces.push_back(rMeshKernel.GetFacet(it));
155  }
156 
157  std::unique_ptr<MeshObject> pcKernel(new MeshObject);
158  pcKernel->addFacets(aFaces);
159  Mesh.setValuePtr(pcKernel.release());
160 
161  return App::DocumentObject::StdReturn;
162 }
