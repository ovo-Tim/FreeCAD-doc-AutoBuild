66 Base::Quantity PropertyQuantity::createQuantityFromPy(PyObject* value)
67 {
68  Base::Quantity quant;
69 
70  if (PyUnicode_Check(value)) {
71  quant = Quantity::parse(PyUnicode_AsUTF8(value));
72  }
73  else if (PyFloat_Check(value)) {
74  quant = Quantity(PyFloat_AsDouble(value), _Unit);
75  }
76  else if (PyLong_Check(value)) {
77  quant = Quantity(double(PyLong_AsLong(value)), _Unit);
78  }
79  else if (PyObject_TypeCheck(value, &(QuantityPy::Type))) {
80  Base::QuantityPy* pcObject = static_cast<Base::QuantityPy*>(value);
81  quant = *(pcObject->getQuantityPtr());
82  }
83  else {
84  std::string error = std::string("wrong type as quantity: ");
85  error += value->ob_type->tp_name;
86  throw Base::TypeError(error);
87  }
88 
89  return quant;
90 }

92 void PropertyQuantity::setPyObject(PyObject* value)
93 {
94  // Set the unit if Unit object supplied, else check the unit
95  // and set the value
96 
97  if (PyObject_TypeCheck(value, &(UnitPy::Type))) {
98  Base::UnitPy* pcObject = static_cast<Base::UnitPy*>(value);
99  Base::Unit unit = *(pcObject->getUnitPtr());
100  aboutToSetValue();
101  _Unit = unit;
102  hasSetValue();
103  }
104  else {
105  Base::Quantity quant = createQuantityFromPy(value);
106 
107  Unit unit = quant.getUnit();
108  if (unit.isEmpty()) {
109  PropertyFloat::setValue(quant.getValue());
110  return;
111  }
112 
113  if (unit != _Unit) {
114  throw Base::UnitsMismatchError("Not matching Unit!");
115  }
116 
117  PropertyFloat::setValue(quant.getValue());
118  }
119 }

185 void PropertyQuantityConstraint::setPyObject(PyObject* value)
186 {
187  Base::Quantity quant = createQuantityFromPy(value);
188 
189  Unit unit = quant.getUnit();
190  double temp = quant.getValue();
191  if (_ConstStruct) {
192  if (temp > _ConstStruct->UpperBound) {
193  temp = _ConstStruct->UpperBound;
194  }
195  else if (temp < _ConstStruct->LowerBound) {
196  temp = _ConstStruct->LowerBound;
197  }
198  }
199  quant.setValue(temp);
200 
201  if (unit.isEmpty()) {
202  PropertyFloat::setValue(quant.getValue()); // clazy:exclude=skipped-base-method
203  return;
204  }
205 
206  if (unit != _Unit) {
207  throw Base::UnitsMismatchError("Not matching Unit!");
208  }
209 
210  PropertyFloat::setValue(quant.getValue()); // clazy:exclude=skipped-base-method
211 }
