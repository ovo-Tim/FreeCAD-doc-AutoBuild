136 void ReportHighlighter::highlightBlock (const QString & text)
137 {
138  if (text.isEmpty())
139  return;
140  auto ud = static_cast<TextBlockData*>(this->currentBlockUserData());
141  if (!ud) {
142  ud = new TextBlockData;
143  this->setCurrentBlockUserData(ud);
144  }
145 
146  TextBlockData::State b;
147  b.length = text.length();
148  b.type = this->type;
149  ud->block.append(b);
150 
151  QVector<TextBlockData::State> block = ud->block;
152  int start = 0;
153  for (const auto & it : block) {
154  switch (it.type)
155  {
156  case Message:
157  setFormat(start, it.length-start, txtCol);
158  break;
159  case Warning:
160  setFormat(start, it.length-start, warnCol);
161  break;
162  case Error:
163  setFormat(start, it.length-start, errCol);
164  break;
165  case LogText:
166  setFormat(start, it.length-start, logCol);
167  break;
168  case Critical:
169  setFormat(start, it.length-start, criticalCol);
170  break;
171  default:
172  break;
173  }
174 
175  start = it.length;
176  }
177 }

317 bool ReportOutputObserver::eventFilter(QObject *obj, QEvent *event)
318 {
319  if (event->type() == QEvent::User && obj == reportView.data()) {
320  auto cr = dynamic_cast<CustomReportEvent*>(event);
321  if (cr) {
322  ReportHighlighter::Paragraph msgType = cr->messageType();
323  if (msgType == ReportHighlighter::Warning) {
324  if (ReportOutputParameter::showOnWarning()) {
325  showReportView();
326  }
327  }
328  else if (msgType == ReportHighlighter::Error) {
329  if (ReportOutputParameter::showOnError()) {
330  showReportView();
331  }
332  }
333  else if (msgType == ReportHighlighter::Message) {
334  if (ReportOutputParameter::showOnMessage()) {
335  showReportView();
336  }
337  }
338  else if (msgType == ReportHighlighter::LogText) {
339  if (ReportOutputParameter::showOnLogMessage()) {
340  showReportView();
341  }
342  }
343  else if (msgType == ReportHighlighter::Critical) {
344  if (ReportOutputParameter::showOnCritical()) {
345  showReportView();
346  }
347  }
348  }
349  return false; //true would prevent the messages reaching the report view
350  }
351 
352  // standard event processing
353  return QObject::eventFilter(obj, event);
354 }

471 void ReportOutput::SendLog(const std::string& notifiername, const std::string& msg, Base::LogStyle level,
472  Base::IntendedRecipient recipient, Base::ContentType content)
473 {
474  // Do not log translated messages, or messages intended only to the user to the Report View
475  if( recipient == Base::IntendedRecipient::User ||
476  content == Base::ContentType::Translated)
477  return;
478 
479  ReportHighlighter::Paragraph style = ReportHighlighter::LogText;
480  switch (level) {
481  case Base::LogStyle::Warning:
482  style = ReportHighlighter::Warning;
483  break;
484  case Base::LogStyle::Message:
485  style = ReportHighlighter::Message;
486  break;
487  case Base::LogStyle::Error:
488  style = ReportHighlighter::Error;
489  break;
490  case Base::LogStyle::Log:
491  style = ReportHighlighter::LogText;
492  break;
493  case Base::LogStyle::Critical:
494  style = ReportHighlighter::Critical;
495  break;
496  default:
497  break;
498  }
499 
500  QString qMsg;
501 
502  if(!notifiername.empty()) {
503  qMsg = QStringLiteral("%1: %2").arg(QString::fromUtf8(notifiername.c_str()),
504  QString::fromUtf8(msg.c_str()));
505  }
506  else {
507  qMsg = QString::fromUtf8(msg.c_str());
508  }
509 
510  // This truncates log messages that are too long
511  if (style == ReportHighlighter::LogText) {
512  if (messageSize > 0 && qMsg.size()>messageSize) {
513  qMsg.truncate(messageSize);
514  qMsg += QString::fromLatin1("...\n");
515  }
516  }
517 
518  // Send the event to itself to allow thread-safety. Qt will delete it when done.
519  auto ev = new CustomReportEvent(style, qMsg);
520  QApplication::postEvent(this, ev);
521 }

523 void ReportOutput::customEvent ( QEvent* ev )
524 {
525  // Appends the text stored in the event to the text view
526  if ( ev->type() == QEvent::User ) {
527  auto ce = static_cast<CustomReportEvent*>(ev);
528  reportHl->setParagraphType(ce->messageType());
529 
530  bool showTimecode = getWindowParameter()->GetBool("checkShowReportTimecode", true);
531  QString text = ce->message();
532 
533  // The time code can only be set when the cursor is at the block start
534  if (showTimecode && blockStart) {
535  QTime time = QTime::currentTime();
536  text.prepend(time.toString(QLatin1String("hh:mm:ss ")));
537  }
538 
539  QTextCursor cursor(this->document());
540  cursor.beginEditBlock();
541  cursor.movePosition(QTextCursor::End);
542  cursor.insertText(text);
543  cursor.endEditBlock();
544 
545  blockStart = cursor.atBlockStart();
546  if (gotoEnd) {
547  setTextCursor(cursor);
548  }
549  ensureCursorVisible();
550  }
551 }
