32 def add_femelement_geometry(f, model, mystran_writer):
33 
34  # generate pyNastran code
35  # HACK, the if statemant needs improvement, see calculix solver
36  if mystran_writer.member.geos_beamsection:
37  beamsec_obj = mystran_writer.member.geos_beamsection[0]["Object"]
38  if beamsec_obj.SectionType == "Rectangular":
39  height = beamsec_obj.RectHeight.getValueAs("mm").Value
40  width = beamsec_obj.RectWidth.getValueAs("mm").Value
41  pynas_code = "# pbarl card, properties of a simple beam element (CBAR entry)\n"
42  pynas_code += "# defined by cross-sectional dimensions\n"
43  pynas_code += f"dim = [{width}, {height}]\n"
44  pynas_code += "model.add_pbarl(pid=1, mid=1, Type={}, dim=dim, nsm=0.0)\n".format(
45  '"BAR"'
46  )
47  pynas_code += "# pbarl.validate()\n\n\n"
48  else:
49  return
50  elif mystran_writer.member.geos_shellthickness:
51  # only use the first shellthickness object
52  shellth_obj = mystran_writer.member.geos_shellthickness[0]["Object"]
53  thickness = shellth_obj.Thickness.getValueAs("mm").Value
54  pynas_code = "# pshell card, thin shell element properties\n"
55  pynas_code += "model.add_pshell(pid=1, mid1=1, t={}, mid2=1, mid3=1)\n\n\n".format(
56  thickness
57  )
58  else:
59  pynas_code = "# psolid card, defines solid element\n"
60  pynas_code += "model.add_psolid(pid=1, mid=1)\n\n\n"
61 
62  # write the pyNastran code
63  f.write(pynas_code)
64 
65  # execute pyNastran code to add data to the model
66  # print(model.get_bdf_stats())
67  exec(pynas_code)
68  # print(model.get_bdf_stats())
69 
70  return model
