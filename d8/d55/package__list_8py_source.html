197  def data(self, index: QtCore.QModelIndex, role: int = QtCore.Qt.DisplayRole):
198  """Get the data for a given index and role."""
199  if not index.isValid():
200  return None
201  row = index.row()
202  if role == QtCore.Qt.ToolTipRole:
203  tooltip = ""
204  if self.repos[row].repo_type == Addon.Kind.PACKAGE:
205  tooltip = translate("AddonsInstaller", "Click for details about package {}").format(
206  self.repos[row].display_name
207  )
208  elif self.repos[row].repo_type == Addon.Kind.WORKBENCH:
209  tooltip = translate(
210  "AddonsInstaller", "Click for details about workbench {}"
211  ).format(self.repos[row].display_name)
212  elif self.repos[row].repo_type == Addon.Kind.MACRO:
213  tooltip = translate("AddonsInstaller", "Click for details about macro {}").format(
214  self.repos[row].display_name
215  )
216  return tooltip
217  if role == PackageListItemModel.DataAccessRole:
218  return self.repos[row]
219 
220  # Sorting
221  if role == SortOptions.Alphabetical:
222  return self.repos[row].display_name
223  if role == SortOptions.LastUpdated:
224  update_date = self.repos[row].update_date
225  if update_date and hasattr(update_date, "timestamp"):
226  return update_date.timestamp()
227  return 0
228  if role == SortOptions.DateAdded:
229  if self.repos[row].stats and self.repos[row].stats.date_created:
230  return self.repos[row].stats.date_created.timestamp()
231  return 0
232  if role == SortOptions.Stars:
233  if self.repos[row].stats and self.repos[row].stats.stars:
234  return self.repos[row].stats.stars
235  return 0
236  if role == SortOptions.Score:
237  return self.repos[row].score

239  def headerData(self, _unused1, _unused2, _role=QtCore.Qt.DisplayRole):
240  """No header in this implementation: always returns None."""
241  return None
242 
243  def append_item(self, repo: Addon) -> None:
244  """Adds this addon to the end of the model. Thread safe."""
245  if repo in self.repos:
246  # Cowardly refuse to insert the same repo a second time
247  return
248  with self.write_lock:
249  self.beginInsertRows(QtCore.QModelIndex(), self.rowCount(), self.rowCount())
250  self.repos.append(repo)
251  self.endInsertRows()
252 
253  def clear(self) -> None:
254  """Clear the model, removing all rows. Thread safe."""
255  if self.rowCount() > 0:
256  with self.write_lock:
257  self.beginRemoveRows(QtCore.QModelIndex(), 0, self.rowCount() - 1)
258  self.repos = []
259  self.endRemoveRows()
260 
261  def reload_item(self, repo: Addon) -> None:
262  """Sets the addon data for the given addon (based on its name)"""
263  for index, item in enumerate(self.repos):
264  if item.name == repo.name:
265  with self.write_lock:
266  self.repos[index] = repo
267  return

423  def _get_sort_label_text(self, addon: Addon) -> str:
424  if self.sort_order == SortOptions.Alphabetical:
425  return ""
426  elif self.sort_order == SortOptions.Stars:
427  if addon.stats and addon.stats.stars and addon.stats.stars > 0:
428  return translate("AddonsInstaller", "{} ★ on GitHub").format(addon.stats.stars)
429  return translate("AddonsInstaller", "No ★, or not on GitHub")
430  elif self.sort_order == SortOptions.DateAdded:
431  if addon.stats and addon.stats.date_created:
432  epoch_seconds = addon.stats.date_created.timestamp()
433  qdt = QtCore.QDateTime.fromSecsSinceEpoch(int(epoch_seconds)).date()
434  time_string = QtCore.QLocale().toString(qdt, QtCore.QLocale.ShortFormat)
435  return translate("AddonsInstaller", "Created ") + time_string
436  return ""
437  elif self.sort_order == SortOptions.LastUpdated:
438  update_date = addon.update_date
439  if update_date:
440  epoch_seconds = update_date.timestamp()
441  qdt = QtCore.QDateTime.fromSecsSinceEpoch(int(epoch_seconds)).date()
442  time_string = QtCore.QLocale().toString(qdt, QtCore.QLocale.ShortFormat)
443  return translate("AddonsInstaller", "Updated ") + time_string
444  return ""
445  elif self.sort_order == SortOptions.Score:
446  return translate("AddonsInstaller", "Score: ") + str(addon.score)
447  return ""

504  QtCore.QLocale.ShortFormat,
505  )
506 
507  available_version_string = ""
508  if repo.metadata:
509  available_version_string = (
510  "<br/>" + translate("AddonsInstaller", "Available version") + ": "
511  )
512  available_version_string += str(repo.metadata.version)
513 
514  if repo.status() == Addon.Status.UNCHECKED:
515  result = translate("AddonsInstaller", "Installed")
516  result += installed_version_string
517  result += installed_date_string
518  elif repo.status() == Addon.Status.NO_UPDATE_AVAILABLE:
519  result = translate("AddonsInstaller", "Up-to-date")
520  result += installed_version_string
521  result += installed_date_string
522  elif repo.status() == Addon.Status.UPDATE_AVAILABLE:
523  result = translate("AddonsInstaller", "Update available")
524  result += installed_version_string
525  result += installed_date_string
526  result += available_version_string
527  elif repo.status() == Addon.Status.PENDING_RESTART:
528  result = translate("AddonsInstaller", "Pending restart")
529 
530  if repo.is_disabled():
531  style = "style='color:" + utils.warning_color_string() + "; font-weight:bold;'"
532  result += (
533  f"<br/><span {style}>[" + translate("AddonsInstaller", "DISABLED") + "]</span>"
534  )
535 
536  return result
