259 void TaskProjGroup::viewToggled(bool toggle)
260 {
261  Gui::WaitCursor wc;
262  bool changed = false;
263  // Obtain name of checkbox
264  int index = sender()->objectName().mid(7).toInt();
265  const char *viewNameCStr = viewChkIndexToCStr(index);
266 
267  if (!blockCheckboxes) {
268  if (multiView) {
269  // Check if only front is left. If so switch to normal view.
270  if (multiView->Views.getValues().size() == 2 && !toggle) {
271  turnProjGroupToView();
272  wc.restoreCursor();
273  return;
274  }
275  }
276  else {
277  // If toggle then we remove the view object and create a proj group instead.
278  turnViewToProjGroup();
279  }
280  }
281 
282  if (toggle && !multiView->hasProjection(viewNameCStr)) {
283  Gui::Command::doCommand(Gui::Command::Doc,
284  "App.activeDocument().%s.addProjection('%s')",
285  multiView->getNameInDocument(), viewNameCStr);
286  changed = true;
287  }
288  else if (!toggle && multiView->hasProjection(viewNameCStr)) {
289  if (multiView->canDelete(viewNameCStr)) {
290  multiView->removeProjection( viewNameCStr );
291  changed = true;
292  }
293  }
294 
295  if (changed) {
296  if (view->ScaleType.isValue("Automatic")) {
297  double scale = view->getScale();
298  setFractionalScale(scale);
299  }
300  view->recomputeFeature();
301  }
302  wc.restoreCursor();
303 }

306 void TaskProjGroup::turnViewToProjGroup()
307 {
308  App::Document* doc = view->getDocument();
309 
310  std::string multiViewName = doc->getUniqueObjectName("ProjGroup");
311  Gui::Command::doCommand(Gui::Command::Gui, "App.activeDocument().addObject('TechDraw::DrawProjGroup', '%s')", multiViewName.c_str());
312  Gui::Command::doCommand(Gui::Command::Gui, "App.activeDocument().%s.addView(App.activeDocument().%s)", view->findParentPage()->getNameInDocument(), multiViewName.c_str());
313 
314  auto* viewPart = static_cast<TechDraw::DrawViewPart*>(view);
315  m_page->removeView(viewPart);
316 
317  multiView = static_cast<TechDraw::DrawProjGroup*>(doc->getObject(multiViewName.c_str()));
318  multiView->Source.setValues(viewPart->Source.getValues());
319  multiView->XSource.setValues(viewPart->XSource.getValues());
320  multiView->X.setValue(viewPart->X.getValue());
321  multiView->Y.setValue(viewPart->Y.getValue());
322  multiView->Scale.setValue(viewPart->Scale.getValue());
323  multiView->ScaleType.setValue(viewPart->ScaleType.getValue());
324  multiView->ProjectionType.setValue(Preferences::projectionAngle());
325  viewPart->X.setValue(0.0);
326  viewPart->Y.setValue(0.0);
327  viewPart->ScaleType.setValue("Custom");
328  viewPart->ScaleType.setStatus(App::Property::Hidden, true);
329  viewPart->Scale.setStatus(App::Property::Hidden, true);
330  viewPart->Label.setValue("Front");
331 
332  multiView->addView(viewPart);
333  multiView->Anchor.setValue(view);
334  multiView->Anchor.purgeTouched();
335 
336  viewPart->LockPosition.setValue(true);
337  viewPart->LockPosition.setStatus(App::Property::ReadOnly, true); //Front should stay locked.
338  viewPart->LockPosition.purgeTouched();
339 
340  m_page->requestPaint();
341  view = multiView;
342 
343  updateUi();
344 }

410 void TaskProjGroup::rotateButtonClicked()
411 {
412  if ( view && ui ) {
413  const QObject *clicked = sender();
414 
415  auto handleCameraButton = [&]() {
416  std::string faceName;
417  App::DocumentObject* obj = nullptr;
418  auto selection = Gui::Command::getSelection().getSelectionEx();
419  for (auto& sel : selection) {
420  for (auto& sub : sel.getSubNames()) {
421  if (TechDraw::DrawUtil::getGeomTypeFromName(sub) == "Face") {
422  obj = sel.getObject();
423  faceName = sub;
424  break;
425  }
426  }
427  if (!faceName.empty()) {
428  break;
429  }
430  }
431 
432  std::pair<Base::Vector3d, Base::Vector3d> dirs = !faceName.empty() ?
433  DrawGuiUtil::getProjDirFromFace(obj, faceName)
434  : DrawGuiUtil::get3DDirAndRot();
435  return dirs;
436  };
437 
438  if (multiView) {
439  //change Front View Dir by 90
440  if (clicked == ui->butTopRotate) multiView->rotate("Up");
441  else if (clicked == ui->butDownRotate) multiView->rotate("Down");
442  else if (clicked == ui->butRightRotate) multiView->rotate("Right");
443  else if (clicked == ui->butLeftRotate) multiView->rotate("Left");
444  else if (clicked == ui->butCWRotate) multiView->spin("CW");
445  else if (clicked == ui->butCCWRotate) multiView->spin("CCW");
446  else if (clicked == ui->butFront) {
447  multiView->getAnchor()->Direction.setValue(Base::Vector3d(0.0, -1.0, 0.0));
448  multiView->getAnchor()->RotationVector.setValue(Base::Vector3d(1.0, 0.0, 0.0));
449  multiView->getAnchor()->XDirection.setValue(Base::Vector3d(1.0, 0.0, 0.0));
450  multiView->updateSecondaryDirs();
451  }
452  else if (clicked == ui->butCam) {
453  std::pair<Base::Vector3d, Base::Vector3d> dirs = handleCameraButton();
454  multiView->getAnchor()->Direction.setValue(dirs.first);
455  multiView->getAnchor()->RotationVector.setValue(dirs.second);
456  multiView->getAnchor()->XDirection.setValue(dirs.second);
457  multiView->updateSecondaryDirs();
458  }
459  }
460  else {
461  auto* viewPart = static_cast<TechDraw::DrawViewPart*>(view);
462  if (clicked == ui->butTopRotate) viewPart->rotate("Up");
463  else if (clicked == ui->butDownRotate) viewPart->rotate("Down");
464  else if (clicked == ui->butRightRotate) viewPart->rotate("Right");
465  else if (clicked == ui->butLeftRotate) viewPart->rotate("Left");
466  else if (clicked == ui->butCWRotate) viewPart->spin("CW");
467  else if (clicked == ui->butCCWRotate) viewPart->spin("CCW");
468  else if (clicked == ui->butFront) {
469  viewPart->Direction.setValue(Base::Vector3d(0.0,-1.0,0.0));
470  viewPart->XDirection.setValue(Base::Vector3d(1.0, 0.0, 0.0));
471  viewPart->recomputeFeature();
472  }
473  else if (clicked == ui->butCam) {
474  std::pair<Base::Vector3d, Base::Vector3d> dirs = handleCameraButton();
475 
476  viewPart->Direction.setValue(dirs.first);
477  viewPart->XDirection.setValue(dirs.second);
478  viewPart->recomputeFeature();
479  }
480  }
481 
482  setUiPrimary();
483  }

556  //unblock recompute
557  }

630 const char * TaskProjGroup::viewChkIndexToCStr(int index)
631 {

634  // FBL B FBRight
