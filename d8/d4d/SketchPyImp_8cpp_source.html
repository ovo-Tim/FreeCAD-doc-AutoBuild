68 PyObject* SketchPy::addGeometry(PyObject* args)
69 {
70  PyObject* pcObj;
71  if (!PyArg_ParseTuple(args, "O", &pcObj)) {
72  return nullptr;
73  }
74 
75  if (PyObject_TypeCheck(pcObj, &(Part::GeometryPy::Type))) {
76  Part::Geometry* geo = static_cast<Part::GeometryPy*>(pcObj)->getGeometryPtr();
77  return Py::new_reference_to(Py::Long(this->getSketchPtr()->addGeometry(geo)));
78  }
79  else if (PyObject_TypeCheck(pcObj, &(PyList_Type))
80  || PyObject_TypeCheck(pcObj, &(PyTuple_Type))) {
81  std::vector<Part::Geometry*> geoList;
82  Py::Sequence list(pcObj);
83  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
84  if (PyObject_TypeCheck((*it).ptr(), &(Part::GeometryPy::Type))) {
85  Part::Geometry* geo = static_cast<Part::GeometryPy*>((*it).ptr())->getGeometryPtr();
86  geoList.push_back(geo);
87  }
88  }
89 
90  int ret = this->getSketchPtr()->addGeometry(geoList) + 1;
91  std::size_t numGeo = geoList.size();
92  Py::Tuple tuple(numGeo);
93  for (std::size_t i = 0; i < numGeo; ++i) {
94  int geoId = ret - int(numGeo - i);
95  tuple.setItem(i, Py::Long(geoId));
96  }
97  return Py::new_reference_to(tuple);
98  }
99 
100  std::string error = std::string("type must be 'Geometry' or list of 'Geometry', not ");
101  error += pcObj->ob_type->tp_name;
102  throw Py::TypeError(error);
103 }

105 PyObject* SketchPy::addConstraint(PyObject* args)
106 {

111
