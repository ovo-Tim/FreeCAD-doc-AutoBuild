76 App::DocumentObjectExecReturn* TrajectoryDressUpObject::execute()
77 {
78  Robot::Trajectory result;
79 
80  App::DocumentObject* link = Source.getValue();
81  if (!link) {
82  return new App::DocumentObjectExecReturn("No object linked");
83  }
84  if (!link->isDerivedFrom<Robot::TrajectoryObject>()) {
85  return new App::DocumentObjectExecReturn("Linked object is not a Trajectory object");
86  }
87 
88  const std::vector<Waypoint*>& wps =
89  static_cast<Robot::TrajectoryObject*>(link)->Trajectory.getValue().getWaypoints();
90  for (auto wp : wps) {
91  Waypoint wpt = *wp;
92  if (UseSpeed.getValue()) {
93  wpt.Velocity = Speed.getValue();
94  }
95  if (UseAcceleration.getValue()) {
96  wpt.Acceleration = Acceleration.getValue();
97  }
98  switch (ContType.getValue()) {
99  case 0:
100  break;
101  case 1:
102  wpt.Cont = true;
103  break;
104  case 2:
105  wpt.Cont = false;
106  break;
107  default:
108  assert(0); // must not happen!
109  }
110  switch (AddType.getValue()) {
111  // do nothing
112  case 0:
113  break;
114  // use orientation
115  case 1:
116  wpt.EndPos.setRotation(PosAdd.getValue().getRotation());
117  break;
118  // add position
119  case 2:
120  wpt.EndPos.setPosition(wpt.EndPos.getPosition() + PosAdd.getValue().getPosition());
121  break;
122  // add orientation
123  case 3:
124  wpt.EndPos.setRotation(wpt.EndPos.getRotation() * PosAdd.getValue().getRotation());
125  break;
126  // add orientation & position
127  case 4:
128  wpt.EndPos = wpt.EndPos * PosAdd.getValue();
129  break;
130  default:
131  assert(0); // must not happen!
132  }
133 
134  result.addWaypoint(wpt);
135  }
136 
137  // set the resulting Trajectory to the object
138  Trajectory.setValue(result);
139 
140  return App::DocumentObject::StdReturn;
141 }
