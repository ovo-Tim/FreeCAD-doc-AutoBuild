182 void AttachEngine::setReferences(const App::PropertyLinkSubList& references)
183 {
184  std::string docname;
185  std::vector<std::string> names;
186  for (auto obj : references.getValues()) {
187  if (!obj->getNameInDocument()) {
188  throw AttachEngineException("AttachEngine::invalid object");
189  }
190  if (docname.empty()) {
191  docname = obj->getDocument()->getName();
192  }
193  else if (docname != obj->getDocument()->getName()) {
194  throw AttachEngineException("AttachEngine::object from multiple document");
195  }
196  names.emplace_back(obj->getNameInDocument());
197  }
198  this->docName = docname;
199  this->objNames = std::move(names);
200  this->subnames.clear();
201  this->subnames.reserve(this->objNames.size());
202  this->shadowSubs.clear();
203  this->shadowSubs.reserve(this->objNames.size());
204  for (auto& shadow : references.getShadowSubs()) {
205  this->shadowSubs.push_back(shadow.newName);
206  this->subnames.push_back(shadow.oldName);
207  }
208  assert(this->objNames.size() == this->subnames.size());
209 }

272 Base::Placement AttachEngine::placementFactory(const gp_Dir &ZAxis,
273  gp_Vec XAxis,
274  gp_Pnt Origin,
275  gp_Pnt refOrg,
276  bool useRefOrg_Line,
277  bool useRefOrg_Plane,
278  bool makeYVertical,
279  bool makeLegacyFlatFaceOrientation,
280  Base::Placement* placeOfRef) const
281 {
282  if(useRefOrg_Line){
283  //move Origin to projection of refOrg onto ZAxis
284  gp_Vec refOrgV = gp_Vec(refOrg.XYZ());
285  gp_Vec OriginV = gp_Vec(Origin.XYZ());
286  gp_Vec ZAxisV = gp_Vec(ZAxis);
287  Origin = gp_Pnt((
288  OriginV + ZAxisV*ZAxisV.Dot(refOrgV-OriginV)
289  ).XYZ());
290  }
291  if(useRefOrg_Plane){
292  //move Origin to projection of refOrg onto plane (ZAxis, Origin)
293  gp_Vec refOrgV = gp_Vec(refOrg.XYZ());
294  gp_Vec OriginV = gp_Vec(Origin.XYZ());
295  gp_Vec ZAxisV = gp_Vec(ZAxis);
296  Origin = gp_Pnt((
297  refOrgV + ZAxisV*ZAxisV.Dot(OriginV-refOrgV)
298  ).XYZ());
299  }
300 
301  if (XAxis.Magnitude() < Precision::Confusion())
302  makeYVertical = true;
303 
304  gp_Ax3 ax3;//OCC representation of the final placement
305  if (!makeYVertical) {
306  ax3 = gp_Ax3(Origin, ZAxis, XAxis);
307  }
308  else if (!makeLegacyFlatFaceOrientation) {
309  //align Y along Z, if possible
310  gp_Vec YAxis(0.0,0.0,1.0);
311  XAxis = YAxis.Crossed(gp_Vec(ZAxis));
312  if (XAxis.Magnitude() < Precision::Confusion()){
313  //ZAxis is along true ZAxis
314  XAxis = (gp_Vec(1,0,0)*ZAxis.Z()).Normalized();
315  }
316  ax3 = gp_Ax3(Origin, ZAxis, XAxis);
317  }
318  else if (makeLegacyFlatFaceOrientation) {
319  //find out, to which axis of support Normal is closest to.
320  //The result will be written into pos variable (0..2 = X..Z)
321  if (!placeOfRef)
322  throw AttachEngineException("AttachEngine::placementFactory: for Legacy mode, placement of the reference must be supplied. Got null instead!");
323  Base::Placement &Place = *placeOfRef;
324  Base::Vector3d dX,dY,dZ;//internal axes of support object, as they are in global space
325  Place.getRotation().multVec(Base::Vector3d(1,0,0),dX);
326  Place.getRotation().multVec(Base::Vector3d(0,1,0),dY);
327  Place.getRotation().multVec(Base::Vector3d(0,0,1),dZ);
328  gp_Dir dirX(dX.x, dX.y, dX.z);
329  gp_Dir dirY(dY.x, dY.y, dY.z);
330  gp_Dir dirZ(dZ.x, dZ.y, dZ.z);
331  double cosNX = ZAxis.Dot(dirX);
332  double cosNY = ZAxis.Dot(dirY);
333  double cosNZ = ZAxis.Dot(dirZ);
334  std::vector<double> cosXYZ;
335  cosXYZ.push_back(fabs(cosNX));
336  cosXYZ.push_back(fabs(cosNY));
337  cosXYZ.push_back(fabs(cosNZ));
338 
339  int pos = std::max_element(cosXYZ.begin(), cosXYZ.end()) - cosXYZ.begin();
340 
341  // +X/-X
342  if (pos == 0) {
343  if (cosNX > 0)
344  ax3 = gp_Ax3(Origin, ZAxis, dirY);
345  else
346  ax3 = gp_Ax3(Origin, ZAxis, -dirY);
347  }
348  // +Y/-Y
349  else if (pos == 1) {
350  if (cosNY > 0)
351  ax3 = gp_Ax3(Origin, ZAxis, -dirX);
352  else
353  ax3 = gp_Ax3(Origin, ZAxis, dirX);
354  }
355  // +Z/-Z
356  else {
357  ax3 = gp_Ax3(Origin, ZAxis, dirX);
358  }
359  }
360 
361  if(this->mapReverse){
362  ax3.ZReverse();
363  ax3.XReverse();
364  }
365 
366  //convert ax3 into Base::Placement
367  gp_Trsf Trf;
368  Trf.SetTransformation(ax3);
369  Trf.Invert();
370  Trf.SetScaleFactor(Standard_Real(1.0));
371 
372  Base::Matrix4D mtrx;
373  TopoShape::convertToMatrix(Trf,mtrx);
374 
375  return Base::Placement(mtrx);
376

379 void AttachEngine::suggestMapModes(SuggestResult &result) const
380 {
381  std::vector<eMapMode> &mlist = result.allApplicableModes;
382  mlist.clear();
383  mlist.reserve(mmDummy_NumberOfModes);
384 
385  std::set<eRefType> &hints = result.nextRefTypeHint;
386  hints.clear();
387 
388  std::map<eMapMode,refTypeStringList> &mlist_reachable = result.reachableModes;
389  mlist_reachable.clear();
390 
391  result.message = SuggestResult::srLinkBroken;
392  result.bestFitMode = mmDeactivated;
393 
394  std::vector<const TopoShape*> shapes;
395  std::vector<TopoShape> shapeStorage;
396  std::vector<eRefType> typeStr;
397  try{
398  readLinks(getRefObjects(),subnames, shapes, shapeStorage, typeStr);
399  } catch (Base::Exception &err) {
400  result.references_Types = typeStr;
401  result.message = SuggestResult::srLinkBroken;
402  result.error.Exception::operator = (err);
403  return;
404  }
405 
406  result.references_Types = typeStr;
407 
408  //search valid modes.
409  int bestMatchScore = -1;
410  result.message = SuggestResult::srNoModesFit;
411  for (std::size_t iMode = 0; iMode < this->modeRefTypes.size(); ++iMode) {
412  if (! this->modeEnabled[iMode])
413  continue;
414  const refTypeStringList &listStrings = modeRefTypes[iMode];
415  for (const auto & str : listStrings) {
416  int score = 1; //-1 = topo incompatible, 0 = topo compatible, geom incompatible; 1+ = compatible (the higher - the more specific is the mode for the support)
417  for (std::size_t iChr = 0; iChr < str.size() && iChr < typeStr.size(); ++iChr) {
418  int match = AttachEngine::isShapeOfType(typeStr[iChr], str[iChr]);
419  switch(match){
420  case -1:
421  score = -1;
422  break;
423  case 0:
424  score = 0;
425  break;
426  case 1:
427  //keep score
428  break;
429  default: //2 and above
430  if (score > 0)
431  score += match;
432  break;
433  }
434  }
435 
436  if (score > 0 && str.size() > typeStr.size()){
437  //mode does not fit, but adding more references will make this mode fit.
438  hints.insert(str[typeStr.size()]);
439 
440  //build string of references to be added to fit this mode
441  refTypeString extraRefs;
442  extraRefs.resize(str.size() - typeStr.size());
443  for (std::size_t iChr = typeStr.size(); iChr < str.size(); iChr++) {
444  extraRefs[iChr - typeStr.size()] = str[iChr];
445  }
446 
447  //add reachable mode
448  auto it_r = mlist_reachable.find(eMapMode(iMode));
449  if (it_r == mlist_reachable.end()){
450  it_r = mlist_reachable.insert(std::pair<eMapMode,refTypeStringList>(eMapMode(iMode),refTypeStringList())).first;
451  }
452  refTypeStringList &list = it_r->second;
453  list.push_back(extraRefs);
454  }
455 
456  //size check is last, because we needed to collect hints
457  if (str.size() != typeStr.size())
458  score = -1;
459 
460  if (score > -1){//still output a best match, even if it is not completely compatible
461  if (score > bestMatchScore){
462  bestMatchScore = score;
463  result.bestFitMode = eMapMode(iMode);
464  result.message = score > 0 ? SuggestResult::srOK : SuggestResult::srIncompatibleGeometry;
465  }
466  }
467  if (score > 0){
468  if(mlist.empty())
469  mlist.push_back(eMapMode(iMode));
470  else if (mlist.back() != eMapMode(iMode))
471  mlist.push_back(eMapMode(iMode));
472  }
473  }
474  }

487 eRefType AttachEngine::getShapeType(const TopoDS_Shape& sh)
488 {
489  if(sh.IsNull())
490  return rtAnything;
491 
492  switch (sh.ShapeType()){
493  case TopAbs_SHAPE:
494  return rtAnything; //note: there's no rtPart detection here - not enough data!
495  break;
496  case TopAbs_SOLID:
497  return rtSolid;
498  break;
499  case TopAbs_COMPOUND:{
500  const TopoDS_Compound &cmpd = TopoDS::Compound(sh);
501  TopoDS_Iterator it (cmpd, Standard_False, Standard_False);//don't mess with placements, to hopefully increase speed
502  if (! it.More())//empty compound
503  return rtAnything;
504  const TopoDS_Shape &sh1 = it.Value();
505  it.Next();
506  if (it.More()){
507  //more than one object, a true compound
508  return rtAnything;
509  } else {
510  //just one object, let's take a look inside
511  return getShapeType(sh1);
512  }
513  }break;
514  case TopAbs_COMPSOLID:
515  case TopAbs_SHELL:
516  return rtAnything;
517  break;
518  case TopAbs_FACE:{
519  const TopoDS_Face &f = TopoDS::Face(sh);
520  BRepAdaptor_Surface surf(f, /*restriction=*/Standard_False);
521  switch(surf.GetType()) {
522  case GeomAbs_Plane:
523  return rtFlatFace;
524  case GeomAbs_Cylinder:
525  return rtCylindricalFace;
526  case GeomAbs_Cone:
527  return rtConicalFace;
528  case GeomAbs_Sphere:
529  return rtSphericalFace;
530  case GeomAbs_Torus:
531  return rtToroidalFace;
532  case GeomAbs_BezierSurface:
533  break;
534  case GeomAbs_BSplineSurface:
535  break;
536  case GeomAbs_SurfaceOfRevolution:
537  return rtSurfaceRev;
538  case GeomAbs_SurfaceOfExtrusion:
539  break;
540  case GeomAbs_OffsetSurface:
541  break;
542  case GeomAbs_OtherSurface:
543  break;
544  }
545  return rtFace;
546  }break;
547  case TopAbs_EDGE:{
548  const TopoDS_Edge &e = TopoDS::Edge(sh);
549  BRepAdaptor_Curve crv(e);
550  switch (crv.GetType()){
551  case GeomAbs_Line:
552  return rtLine;
553  case GeomAbs_Circle:
554  return rtCircle;
555  case GeomAbs_Ellipse:
556  return rtEllipse;
557  case GeomAbs_Hyperbola:
558  return rtHyperbola;
559  case GeomAbs_Parabola:
560  return rtParabola;
561  case GeomAbs_BezierCurve:
562  case GeomAbs_BSplineCurve:
563  case GeomAbs_OtherCurve:
564  case GeomAbs_OffsetCurve:
565  return rtCurve;
566  }
567  }break;
568  case TopAbs_WIRE:
569  return rtWire;
570  case TopAbs_VERTEX:
571  return rtVertex;
572  default:
573  throw AttachEngineException("AttachEngine::getShapeType: unexpected TopoDS_Shape::ShapeType");
574  }//switch shapetype
575  return rtAnything;//shouldn't happen, it's here to shut up compiler warning
576 }

648 int AttachEngine::isShapeOfType(eRefType shapeType, eRefType requirement)
649 {
650  //first up, check for hasplacement flag
651  if (requirement & rtFlagHasPlacement) {
652  if(! (shapeType & rtFlagHasPlacement))
653  return -1;
654  }
655 
656  //get rid of hasplacement flags, to simplify the rest
657  shapeType = eRefType(shapeType & (rtFlagHasPlacement - 1));
658  requirement = eRefType(requirement & (rtFlagHasPlacement - 1));
659 
660  if (requirement == rtAnything)
661  return 1;
662 
663  int reqRank = getTypeRank(requirement);
664 
665  //test for valid match
666  eRefType shDeg = shapeType;
667  while(shDeg != rtAnything){
668  if (shDeg == requirement)
669  return reqRank;
670  shDeg = downgradeType(shDeg);
671  }
672 
673  //test for slightly invalid match (e.g. requirement==line, shapeType == curve)
674  requirement = downgradeType(requirement);
675  if (requirement != rtAnything) {
676  eRefType shDeg = shapeType;
677  while(shDeg != rtAnything){
678  if (shDeg == requirement)
679  return 0;

970 Base::Placement AttachEngine::calculateAttachedPlacement(const Base::Placement& origPlacement,
971  bool* subChanged)
972 {
973  std::map<int, std::pair<std::string, std::string>> subChanges;
974  int i = -1;
975  auto objs = getRefObjects();
976  for (auto obj : objs) {
977  ++i;
978  auto& sub = subnames[i];
979  obj = obj->getSubObject(sub.c_str());
980  auto& shadow = shadowSubs[i];
981  if (shadow.empty() || !Data::hasMissingElement(sub.c_str())) {
982  continue;
983  }
984  auto related = Part::Feature::getRelatedElements(obj,
985  shadow.c_str(),
986  Part::HistoryTraceType::followTypeChange,
987  false);
988  if (!related.empty()) {
989  auto& res = subChanges[i];
990  res.first = Data::ComplexGeoData::elementMapPrefix();
991  related.front().name.appendToBuffer(res.first);
992  res.second.clear();
993  related.front().index.appendToStringBuffer(res.second);
994  }
995  else {
996  std::string name = Data::oldElementName(shadow.c_str());
997  if (!name.empty()) {
998  auto& res = subChanges[i];
999  res.first.clear();
1000  res.second = name;
1001  }
1002  else {
1003  subnames[i] = shadow;
1004  }
1005  }
1006  }
1007  if (!subChanges.empty()) {
1008  // In case there is topological name changes, we only auto change the
1009  // subname if the calculated placement stays the same. If not, just
1010  // proceed as normal, which will throw exception and catch user's
1011  // attention.
1012  auto subs = subnames;
1013  for (auto& change : subChanges) {
1014  auto [subkey, namechange] =change;
1015  auto [_oldname, newname] = namechange;
1016  subs[subkey] = newname;
1017  }
1018  auto pla = _calculateAttachedPlacement(objs, subs, origPlacement);
1019  // check equal placement with some tolerance
1020  if (pla.getPosition().IsEqual(origPlacement.getPosition(), Precision::Confusion())
1021  && pla.getRotation().isSame(origPlacement.getRotation(), Precision::Angular())) {
1022  // Only make changes if the caller supplies 'subChanged', because
1023  // otherwise it means the caller just want to do an immutable test.
1024  // See AttachExtension::isAttacherActive().
1025  if (subChanged) {
1026  *subChanged = true;
1027  subnames = std::move(subs);
1028  for (auto& v : subChanges) {
1029  shadowSubs[v.first] = v.second.first;
1030  }
1031  }
1032  return pla;
1033  }
1034  }
1035  return _calculateAttachedPlacement(objs, subnames, origPlacement);
1036 }

1159 AttachEngine3D::_calculateAttachedPlacement(const std::vector<App::DocumentObject*>& objs,
1160  const std::vector<std::string>& subs,
1161  const Base::Placement& origPlacement) const
1162 {

1164  if (mmode == mmDeactivated) {
1165  throw ExceptionCancel(); // to be handled in positionBySupport, to not do anything if
1166  // disabled
1167  }

1169  std::vector<TopoShape> copiedShapeStorage;
1170  std::vector<eRefType> types;
1171  readLinks(objs, subs, shapes, copiedShapeStorage, types);
1172 
1173  if (shapes.empty()) {
1174  throw ExceptionCancel();
1175  }
1176 
1177  // common stuff for all map modes
1178  App::DocumentObject* subObj = objs[0]->getSubObject(subs[0].c_str());
1179  Base::Placement Place = App::GeoFeature::getGlobalPlacement(subObj, objs[0], subs[0]);

1181  gp_Pnt refOrg = gp_Pnt(vec.x, vec.y, vec.z); // origin of linked object
1182 
1183  // variables to derive the actual placement.
1184  // They are to be set, depending on the mode:
1185  // to the sketch
1186  gp_Dir SketchNormal; // points at the user
1187  gp_Vec SketchXAxis; // if left zero, a guess will be made
1188  gp_Pnt SketchBasePoint; // where to put the origin of the sketch
1189 
1190 
1191  switch (mmode) {
1192  case mmDeactivated:
1193  // should have been filtered out already!
1194  break;
1195  case mmTranslate: {
1196  if (shapes.empty()) {
1197  throw Base::ValueError("AttachEngine3D::calculateAttachedPlacement: no subobjects "
1198  "specified (need one vertex).");
1199  }
1200  const TopoDS_Shape& sh = shapes[0]->getShape();
1201  if (sh.IsNull()) {
1202  throw Base::ValueError(
1203  "Null shape in AttachEngine3D::calculateAttachedPlacement()!");
1204  }
1205  if (sh.ShapeType() != TopAbs_VERTEX) {
1206  throw Base::ValueError("AttachEngine3D::calculateAttachedPlacement: no subobjects "
1207  "specified (need one vertex).");
1208  }
1209  gp_Pnt p = BRep_Tool::Pnt(TopoDS::Vertex(sh));
1210  Base::Placement plm = Base::Placement();
1211  plm.setPosition(Base::Vector3d(p.X(), p.Y(), p.Z()));
1212  plm.setPosition(plm.getPosition() + this->attachmentOffset.getPosition());
1213  plm.setRotation(origPlacement.getRotation());
1214  return plm;
1215  } break;
1216  case mmObjectXY:
1217  case mmObjectXZ:
1218  case mmObjectYZ:
1219  case mmParallelPlane: {
1220  // DeepSOIC: could have been done much more efficiently, but I'm lazy...
1221  gp_Dir dirX, dirY, dirZ;
1222  if (types[0] & rtFlagHasPlacement) {
1223  Base::Vector3d dX, dY,
1224  dZ; // internal axes of support object, as they are in global space
1225  Place.getRotation().multVec(Base::Vector3d(1, 0, 0), dX);
1226  Place.getRotation().multVec(Base::Vector3d(0, 1, 0), dY);
1227  Place.getRotation().multVec(Base::Vector3d(0, 0, 1), dZ);
1228  dirX = gp_Dir(dX.x, dX.y, dX.z);
1229  dirY = gp_Dir(dY.x, dY.y, dY.z);
1230  dirZ = gp_Dir(dZ.x, dZ.y, dZ.z);
1231  SketchBasePoint =
1232  gp_Pnt(Place.getPosition().x, Place.getPosition().y, Place.getPosition().z);
1233  }
1234  else if (isShapeOfType(types[0], rtConic) > 0) {
1235  const TopoDS_Edge& e = TopoDS::Edge(shapes[0]->getShape());
1236  BRepAdaptor_Curve adapt(e);
1237  gp_Ax3 pos;
1238  switch (adapt.GetType()) {
1239  case GeomAbs_Ellipse: {
1240  gp_Elips cc = adapt.Ellipse();
1241  pos = gp_Ax3(cc.Position());
1242  } break;
1243  case GeomAbs_Hyperbola: {
1244  gp_Hypr cc = adapt.Hyperbola();
1245  pos = gp_Ax3(cc.Position());
1246  } break;
1247  case GeomAbs_Parabola: {
1248  gp_Parab cc = adapt.Parabola();
1249  pos = gp_Ax3(cc.Position());
1250  } break;
1251  default:
1252  assert(0); // conics should have been filtered out by testing shape type in
1253  // the above if.
1254  }

2023  &Place);

2028 double AttachEngine3D::calculateFoldAngle(gp_Vec axA, gp_Vec axB, gp_Vec edA, gp_Vec edB) const
2029 {
2030  //DeepSOIC: this hardcore math can probably be replaced with a couple of
2031  //clever OCC calls... See forum thread "Sketch mapping enhancement" for a
2032  //picture on how this math was derived.
2033  //https://forum.freecad.org/viewtopic.php?f=8&t=10511&sid=007946a934530ff2a6c9259fb32624ec&start=40#p87584
2034  axA.Normalize();
2035  axB.Normalize();
2036  edA.Normalize();
2037  edB.Normalize();
2038  gp_Vec norm = axA.Crossed(axB);
2039  if (norm.Magnitude() < Precision::Confusion())
2040  throw AttachEngineException("calculateFoldAngle: Folding axes are parallel, folding angle cannot be computed.");
2041  norm.Normalize();
2042  double a = edA.Dot(axA);
2043  double ra = edA.Crossed(axA).Magnitude();
2044  if (fabs(ra) < Precision::Confusion())
2045  throw AttachEngineException("calculateFoldAngle: axisA and edgeA are parallel, folding can't be computed.");
2046  double b = edB.Dot(axB);
2047  double costheta = axB.Dot(axA);
2048  double sintheta = axA.Crossed(axB).Dot(norm);
2049  double singama = -costheta;
2050  double cosgama = sintheta;
2051  double k = b*cosgama;
2052  double l = a + b*singama;
2053  double xa = k + l*singama/cosgama;
2054  double cos_unfold = -xa/ra;
2055  if (fabs(cos_unfold)>0.999)
2056  throw AttachEngineException("calculateFoldAngle: cosine of folding angle is too close to or above 1.");
2057  return acos(cos_unfold);
2058 }
