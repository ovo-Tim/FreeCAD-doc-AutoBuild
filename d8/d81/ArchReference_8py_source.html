238  def cleanShape(self, shapedata, obj, materials=None):
239 
240  """cleans the imported shape"""
241 
242  import Part
243  shape = Part.Shape()
244  shape.importBrepFromString(shapedata)
245  if obj.FuseArch and materials:
246  # separate lone edges
247  shapes = []
248  for edge in shape.Edges:
249  found = False
250  for solid in shape.Solids:
251  for soledge in solid.Edges:
252  if edge.hashCode() == soledge.hashCode():
253  found = True
254  break
255  if found:
256  break
257  if found:
258  break
259  else:
260  shapes.append(edge)
261  #print("solids:",len(shape.Solids),"mattable:",materials)
262  for key,solindexes in materials.items():
263  if key == "Undefined":
264  # do not join objects with no defined material
265  for solindex in [int(i) for i in solindexes.split(",")]:
266  shapes.append(shape.Solids[solindex])
267  else:
268  fusion = None
269  for solindex in [int(i) for i in solindexes.split(",")]:
270  if not fusion:
271  fusion = shape.Solids[solindex]
272  else:
273  fusion = fusion.fuse(shape.Solids[solindex])
274  if fusion:
275  shapes.append(fusion)
276  shape = Part.makeCompound(shapes)
277  try:
278  shape = shape.removeSplitter()
279  except Exception:
280  t = translate("Arch","Error removing splitter")
281  FreeCAD.Console.PrintError(obj.Label+": "+t+"\n")
282  return shape

303

305  filename = obj.File
306  if not filename:
307  return None

325  for i in range(len(subdirs)):
326  subpath = [currentdir]+subdirs[-i:]+[basename]
327  altfile = os.path.join(*subpath)
328  if self.exists(altfile):
329  return self.exists(altfile)
330  return None

408  part = n[0]

410  elif "<Property name=\"MaterialsTable\" type=\"App::PropertyMap\"" in line:
411  writemode = True

414  v = re.findall(r'value=\"(.*?)\"',line)
