174 void ExpressionSpinBox::openFormulaDialog()
175 {
176  Q_ASSERT(isBound());
177 
178  auto * qprop = freecad_dynamic_cast<PropertyQuantity>(getPath().getProperty());
179  Unit unit;
180 
181  if (qprop)
182  unit = qprop->getUnit();
183 
184  auto box = new Gui::Dialog::DlgExpressionInput(getPath(), getExpression(), unit, spinbox);
185  QObject::connect(box, &Gui::Dialog::DlgExpressionInput::finished, [this, box]() {
186  if (box->result() == QDialog::Accepted)
187  setExpression(box->getExpression());
188  else if (box->discardedFormula())
189  setExpression(std::shared_ptr<Expression>());
190 
191  box->deleteLater();
192  });
193  box->show();
194 
195  QPoint pos = spinbox->mapToGlobal(QPoint(0,0));
196  box->move(pos-box->expressionPosition());
197  box->setExpressionInputSize(spinbox->width(), spinbox->height());
198 }

242 {

282  UIntSpinBoxPrivate() = default;
283  uint mapToUInt( int v ) const
284  {
285  uint ui;
286  if ( v == INT_MIN ) {
287  ui = 0;
288  } else if ( v == INT_MAX ) {
289  ui = UINT_MAX;
290  } else if ( v < 0 ) {
291  v -= INT_MIN; ui = (uint)v;
292  } else {
293  ui = (uint)v; ui -= INT_MIN;
294  } return ui;
295  }
296  int mapToInt( uint v ) const
297  {
298  int in;
299  if ( v == UINT_MAX ) {
300  in = INT_MAX;
301  } else if ( v == 0 ) {
302  in = INT_MIN;
303  } else if ( v > INT_MAX ) {
304  v += INT_MIN; in = (int)v;
305  } else {
306  in = v; in += INT_MIN;
307  } return in;
308  }
309 };
