198 std::vector<Base::Vector3d> DrawLeaderLine::horizLastSegment(const std::vector<Base::Vector3d>& inDeltas, double rotationDeg)
199 {
200 
201  Base::Vector3d stdX{1, 0, 0};
202 
203  std::vector<Base::Vector3d> wp = inDeltas;
204  if (wp.size() > 1) {
205  size_t iLast = wp.size() - 1;
206  size_t iPen = wp.size() - 2;
207  Base::Vector3d last = wp.at(iLast);
208  Base::Vector3d penUlt = wp.at(iPen);
209 
210  auto lastSeg = DU::invertY(last - penUlt);
211  auto lastSegLong = lastSeg.Length();
212  auto lastSegRotated = lastSeg;
213  lastSegRotated.RotateZ(Base::toRadians(rotationDeg));
214  auto lastSegRotatedUnit = lastSegRotated;
215  lastSegRotatedUnit.Normalize();
216  auto dot = lastSegRotatedUnit.Dot(stdX);
217 
218  auto newLast = penUlt + stdX * lastSegLong;
219  if (dot < 0) {
220  newLast = penUlt - stdX * lastSegLong;
221  }
222 
223  wp.at(iLast) = newLast;
224  }
225  return wp;
226 }

279 Base::Vector3d DrawLeaderLine::lastSegmentDirection() const
280 {
281  std::vector<Base::Vector3d> wp = getTransformedWayPoints();
282  if (wp.empty()) {
283  return Base::Vector3d(1,0,0);
284  }
285  // this direction is in conventional coords? Y+ up?
286  // vertical segment will be small negative Y - large negative Y -> a positive Y but we want a negative Y
287  auto tailPoint = DU::invertY(wp.rbegin()[0]);
288  auto kinkPoint = DU::invertY(wp.rbegin()[1]);
289  auto direction = kinkPoint - tailPoint; // from kink to tail
290  direction = DU::invertY(direction);
291  direction.Normalize();
292  return direction;
293 }

297 DrawLeaderLine* DrawLeaderLine::makeLeader(DrawViewPart* parent, std::vector<Base::Vector3d> pagePoints, int iStartSymbol, int iEndSymbol)
298 {
299  Base::Console().Message("DLL::makeLeader(%s, %d, %d, %d)\n", parent->getNameInDocument(), pagePoints.size(), iStartSymbol, iEndSymbol);
300  if (pagePoints.size() < 2) {
301  Base::Console().Message("DLL::makeLeader - not enough pagePoints\n");
302  return {};
303  }
304 
305  // this is +/- the same code as in TaskLeaderLine::createLeaderFeature()
306  const std::string objectName{"LeaderLine"};
307  const std::string leaderType = "TechDraw::DrawLeaderLine";
308  std::string leaderName = parent->getDocument()->getUniqueObjectName(objectName.c_str());
309  std::string pageName = parent->findParentPage()->getNameInDocument();
310  std::string parentName = parent->getNameInDocument();
311 
312  Base::Interpreter().runStringArg("App.activeDocument().addObject('%s', '%s')",
313  leaderType.c_str(), leaderName.c_str());
314  Base::Interpreter().runStringArg("App.activeDocument().%s.translateLabel('DrawLeaderLine', 'LeaderLine', '%s')",
315  leaderName.c_str(), leaderName.c_str());
316  Base::Interpreter().runStringArg("App.activeDocument().%s.addView(App.activeDocument().%s)",
317  pageName.c_str(), leaderName.c_str());
318  Base::Interpreter().runStringArg("App.activeDocument().%s.LeaderParent = App.activeDocument().%s",
319  leaderName.c_str(), parentName.c_str());
320  // we assume here that the caller will handle AutoHorizontal, Scalable and Rotatable as required
321 
322 
323  App::DocumentObject* obj = parent->getDocument()->getObject(leaderName.c_str());
324  if (!obj || !obj->isDerivedFrom(TechDraw::DrawLeaderLine::getClassTypeId())) {
325  throw Base::RuntimeError("DrawLeaderLine::makeLeader - new object not found");
326  }
327 
328  // set leader x,y position
329  auto leaderFeature = static_cast<TechDraw::DrawLeaderLine*>(obj);
330  Base::Vector3d parentPagePos{ parent->X.getValue(), parent->Y.getValue(), 0.0};
331  Base::Vector3d leaderPagePos = pagePoints.front() - parentPagePos;
332  bool force = true; // update position even though leaders default to locked.
333  leaderFeature->setPosition(leaderPagePos.x, leaderPagePos.y, force);
334 
335  // page positions to deltas
336  std::vector<Base::Vector3d> pageDeltas;
337  for (auto& point : pagePoints) {
338  auto temp = point - pagePoints.front();
339  pageDeltas.emplace_back(temp);
340  }
341 
342  // deltas to unscaled, unrotated form
343  auto leaderPoints = leaderFeature->makeCanonicalPoints(pageDeltas);
344  // invert the canonical points
345  std::vector<Base::Vector3d> inverted;
346  inverted.reserve(leaderPoints.size());
347  for (auto& point : leaderPoints) {
348  inverted.push_back(DU::invertY(point));
349  }
350  leaderFeature->WayPoints.setValues(inverted);
351 
352  leaderFeature->StartSymbol.setValue(iStartSymbol);
353  leaderFeature->EndSymbol.setValue(iEndSymbol);
354 
355  parent->touch();
356 
357  return leaderFeature;
358 }

362 std::vector<Base::Vector3d> DrawLeaderLine::getScaledAndRotatedPoints(bool doScale, bool doRotate) const
363 {
364  auto dvp = getBaseView();
365  if (!dvp) {
366  // document is restoring?
367  // Base::Console().Message("DLL::getScaledAndRotatedPoints - no DV\n");
368  return {};
369  }
370 
371  double scale{1.0};
372  if (Scalable.getValue() && doScale) {
373  scale = dvp->getScale();
374  }
375 
376  double rotationRad{0.0};
377  if (doRotate) {
378  rotationRad = dvp->Rotation.getValue() * M_PI / DegreesHalfCircle;
379  }
380 
381  std::vector<Base::Vector3d> pointsAll = WayPoints.getValues();
382  std::vector<Base::Vector3d> result;
383  for (auto& point : pointsAll) {
384  Base::Vector3d newPoint = DU::invertY(point * scale);
385  if (rotationRad != 0.0) {
386  // the waypoints use conventional coords
387  newPoint.RotateZ(rotationRad);
388  }
389  result.push_back(DU::invertY(newPoint));
390  }
391 
392  return result;
393 }

398 std::vector<Base::Vector3d>
399 DrawLeaderLine::makeCanonicalPoints(const std::vector<Base::Vector3d>& inPoints,
400  bool doScale,
401  bool doRotate) const
402 {
403  auto dvp = getBaseView();
404 
405  double scale{1.0};
406  if (Scalable.getValue() && doScale) {
407  scale = dvp->getScale();
408  }
409 
410  double rotationRad{0.0};
411  if (doRotate) {
412  rotationRad = - dvp->Rotation.getValue() * M_PI / DegreesHalfCircle;
413  }
414 
415  std::vector<Base::Vector3d> result;
416  for (auto& point : inPoints) {
417  Base::Vector3d newPoint = point / scale;
418  if (rotationRad != 0.0) {
419  newPoint.RotateZ(rotationRad);
420  }
421  result.push_back(newPoint);
422  }
423 
424  return result;
425 }
