130 def run_analysis(doc, base_name, filepath="", run_solver=False):
131 
132  from os.path import join, exists
133  from os import makedirs
134  from tempfile import gettempdir as gettmp
135 
136  # recompute
137  doc.recompute()
138 
139  # print(doc.Objects)
140  # print([obj.Name for obj in doc.Objects])
141 
142  # filepath
143  if filepath == "":
144  filepath = join(gettmp(), "FEM_examples")
145  if not exists(filepath):
146  makedirs(filepath)
147 
148  # find the first solver
149  # thus ATM only one solver per analysis is supported
150  from femtools.femutils import is_derived_from
151 
152  for m in doc.Analysis.Group:
153  if is_derived_from(m, "Fem::FemSolverObjectPython"):
154  solver = m
155  break
156 
157  # a file name is needed for the besides dir to work
158  save_fc_file = join(filepath, (base_name + ".FCStd"))
159  FreeCAD.Console.PrintMessage(f"Save FreeCAD file for {base_name} analysis to {save_fc_file}\n.")
160  doc.saveAs(save_fc_file)
161 
162  # get analysis workig dir
163  from femtools.femutils import get_beside_dir
164 
165  working_dir = get_beside_dir(solver)
166 
167  # run analysis
168  from femsolver.run import run_fem_solver
169 
170  if run_solver is True:
171  run_fem_solver(solver, working_dir)
172 
173  # save doc once again with results
174  doc.save()

177 def run_example(example, solver=None, base_name=None, run_solver=False):
178 
179  from importlib import import_module
180 
181  module = import_module("femexamples." + example)
182  if not hasattr(module, "setup"):
183  FreeCAD.Console.PrintError(f"Setup method not found in {example}\n")
184  return None
185 
186  if solver is None:
187  doc = getattr(module, "setup")()
188  else:
189  doc = getattr(module, "setup")(solvertype=solver)
190 
191  if base_name is None:
192  base_name = example
193  if solver is not None:
194  base_name += "_" + solver
195  run_analysis(doc, base_name, run_solver=run_solver)
196  doc.recompute()
197 
198  return doc
