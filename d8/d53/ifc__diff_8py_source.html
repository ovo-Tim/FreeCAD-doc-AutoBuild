35 def get_diff(proj):
36  """Obtains a diff between the current version and the saved version of a project"""
37 
38  if not getattr(proj, "IfcFilePath", None):
39  old = []
40  return 1
41  else:
42  # cannot use open() here as it gives different encoding
43  # than ifcopenshell and diff does not work
44  f = ifcopenshell.open(proj.IfcFilePath)
45  old = f.wrapped_data.to_string().split("\n")
46  if not old:
47  return ""
48  ifcfile = ifc_tools.get_ifcfile(proj)
49  if not ifcfile:
50  return ""
51  new = ifcfile.wrapped_data.to_string().split("\n")
52  # diff = difflib.HtmlDiff().make_file(old,new) # UGLY
53  res = [l for l in difflib.unified_diff(old, new, lineterm="")]
54  res = [l for l in res if l.startswith("+") or l.startswith("-")]
55  res = [l for l in res if not l.startswith("+++") and not l.startswith("---")]
56  return "\n".join(res)

59 def htmlize(diff):
60  """Returns an HTML version of a diff list"""
61 
62  html = "<html><body>\n"
63  if diff == 1:
64  html += translate("BIM", "The IFC file is not saved. Please save once"
65  " to have an existing IFC file to compare with."
66  " Then, run this command again.") + "<br/>\n"
67  elif diff:
68  diff = diff.split("\n")
69  for l in diff:
70  if l.startswith("+"):
71  html += "<span style='color:green;'>" + l[:100] + "</span><br/>\n"
72  elif l.startswith("-"):
73  html += "<span style='color:red;'>" + l[:100] + "</span><br/>\n"
74  else:
75  html += l + "<br/>\n"
76  else:
77  html += translate("BIM", "No changes to display.") + "<br/>\n"
78  html += "</body></html>"
