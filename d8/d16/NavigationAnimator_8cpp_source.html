65 bool NavigationAnimator::startAndWait(const std::shared_ptr<NavigationAnimation>& animation)
66 {
67  stop();
68  bool finished = true;
69  QEventLoop loop;
70  connect(animation.get(), &NavigationAnimation::finished, &loop, &QEventLoop::quit);
71  connect(animation.get(), &NavigationAnimation::interrupted, &loop, [&finished, &loop]() {
72  finished = false;
73  loop.quit();
74  });
75  start(animation);
76  loop.exec();

83 void NavigationAnimator::stop()
