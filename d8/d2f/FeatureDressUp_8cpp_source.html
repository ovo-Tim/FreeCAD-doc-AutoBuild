112 void DressUp::getContinuousEdges(Part::TopoShape TopShape, std::vector< std::string >& SubNames, std::vector< std::string >& FaceNames) {
113 
114  TopTools_IndexedMapOfShape mapOfEdges;
115  TopTools_IndexedDataMapOfShapeListOfShape mapEdgeFace;
116  TopExp::MapShapesAndAncestors(TopShape.getShape(), TopAbs_EDGE, TopAbs_FACE, mapEdgeFace);
117  TopExp::MapShapes(TopShape.getShape(), TopAbs_EDGE, mapOfEdges);
118 
119  unsigned int i = 0;
120  while(i < SubNames.size())
121  {
122  std::string aSubName = static_cast<std::string>(SubNames.at(i));
123 
124  if (aSubName.compare(0, 4, "Edge") == 0) {
125  TopoDS_Edge edge = TopoDS::Edge(TopShape.getSubShape(aSubName.c_str()));
126  const TopTools_ListOfShape& los = mapEdgeFace.FindFromKey(edge);
127 
128  if(los.Extent() != 2)
129  {
130  SubNames.erase(SubNames.begin()+i);
131  continue;
132  }
133 
134  const TopoDS_Shape& face1 = los.First();
135  const TopoDS_Shape& face2 = los.Last();
136  GeomAbs_Shape cont = BRep_Tool::Continuity(TopoDS::Edge(edge),
137  TopoDS::Face(face1),
138  TopoDS::Face(face2));
139  if (cont != GeomAbs_C0) {
140  SubNames.erase(SubNames.begin()+i);
141  continue;
142  }
143 
144  i++;
145  }
146  else if(aSubName.compare(0, 4, "Face") == 0) {
147  TopoDS_Face face = TopoDS::Face(TopShape.getSubShape(aSubName.c_str()));
148 
149  TopTools_IndexedMapOfShape mapOfFaces;
150  TopExp::MapShapes(face, TopAbs_EDGE, mapOfFaces);
151 
152  for(int j = 1; j <= mapOfFaces.Extent(); ++j) {
153  TopoDS_Edge edge = TopoDS::Edge(mapOfFaces.FindKey(j));
154 
155  int id = mapOfEdges.FindIndex(edge);
156 
157  std::stringstream buf;
158  buf << "Edge";
159  buf << id;
160 
161  if(std::find(SubNames.begin(),SubNames.end(),buf.str()) == SubNames.end())
162  {
163  SubNames.push_back(buf.str());
164  }
165 
166  }
167 
168  FaceNames.emplace_back(aSubName.c_str());
169  SubNames.erase(SubNames.begin()+i);
170  }
171  // empty name or any other sub-element
172  else {
173  SubNames.erase(SubNames.begin()+i);
174  }
175  }
176 }

178 std::vector<TopoShape> DressUp::getContinuousEdges(const TopoShape& shape)
179 {
180  std::vector<TopoShape> ret;
181  std::unordered_set<TopoDS_Shape, Part::ShapeHasher, Part::ShapeHasher> shapeSet;
182 
183  auto addEdge = [&](const TopoDS_Shape& subshape, const std::string& ref) {
184  if (!shapeSet.insert(subshape).second) {
185  return;
186  }
187 
188  auto faces = shape.findAncestorsShapes(subshape, TopAbs_FACE);
189  if (faces.size() != 2) {
190  FC_WARN(getFullName() << ": skip edge " << ref << " with less two attaching faces");
191  return;
192  }
193  const TopoDS_Shape& face1 = faces.front();
194  const TopoDS_Shape& face2 = faces.back();
195  GeomAbs_Shape cont =
196  BRep_Tool::Continuity(TopoDS::Edge(subshape), TopoDS::Face(face1), TopoDS::Face(face2));
197  if (cont != GeomAbs_C0) {
198  FC_WARN(getFullName() << ": skip edge " << ref << " that is not C0 continuous");
199  return;
200  }
201  ret.push_back(subshape);
202  };
203 
204  for (const auto& v : Base.getShadowSubs()) {
205  TopoDS_Shape subshape;
206  const auto& ref = v.newName.size() ? v.newName : v.oldName;
207  subshape = shape.getSubShape(ref.c_str(), true);
208  if (subshape.IsNull()) {
209  FC_THROWM(Base::CADKernelError, "Invalid edge link: " << ref);
210  }
211 
212  if (subshape.ShapeType() == TopAbs_EDGE) {
213  addEdge(subshape, ref);
214  }
215  else if (subshape.ShapeType() == TopAbs_FACE || subshape.ShapeType() == TopAbs_WIRE) {
216  for (TopExp_Explorer exp(subshape, TopAbs_EDGE); exp.More(); exp.Next()) {
217  addEdge(exp.Current(), std::string());
218  }
219  }
220  else {
221  FC_WARN(getFullName() << ": skip invalid shape '" << ref << "' with type "
222  << TopoShape::shapeName(subshape.ShapeType()));
223  }
224  }
225  return ret;
226 }

228 std::vector<TopoShape> DressUp::getFaces(const TopoShape& shape)
229 {
230  std::vector<TopoShape> ret;
231  const auto& vals = Base.getSubValues();
232  const auto& subs = Base.getShadowSubs();
233  size_t i = 0;
234  for (auto& val : vals) {
235  if (!boost::starts_with(val, "Face")) {
236  continue;
237  }
238  auto& sub = subs[i++];
239  auto& ref = sub.newName.size() ? sub.newName : val;
240  TopoShape subshape;
241  try {
242  subshape = shape.getSubTopoShape(ref.c_str());
243  }
244  catch (...) {
245  }
246 
247  if (subshape.isNull()) {
248  FC_ERR(getFullName() << ": invalid face reference '" << ref << "'");
249  throw Part::NullShapeException("Invalid Invalid face link");
250  }
251 
252  if (subshape.shapeType() != TopAbs_FACE) {
253  FC_WARN(getFullName() << ": skip invalid shape '" << ref << "' with type "
254  << subshape.shapeName());
255  continue;
256  }
257  ret.push_back(subshape);
258  }
259  return ret;
260 }

262 void DressUp::onChanged(const App::Property* prop)
263 {
264  // the BaseFeature property should track the Base and vice-versa as long as
265  // the feature is inside a body (aka BaseFeature is nonzero)
266  if (prop == &BaseFeature) {
267  if (BaseFeature.getValue()
268  && Base.getValue()
269  && Base.getValue() != BaseFeature.getValue()) {
270 
271  auto subs = Base.getSubValues(false);
272  auto shadows = Base.getShadowSubs();
273  Base.setValue (BaseFeature.getValue(),std::move(subs),std::move(shadows));
274  }
275  } else if (prop == &Base) {
276  // track the vice-versa changes
277  if (BaseFeature.getValue() && Base.getValue() != BaseFeature.getValue()) {
278  BaseFeature.setValue (Base.getValue());
279  }
280  } else if (prop == &Shape || prop == &SupportTransform) {
281  if (!getDocument()->testStatus(App::Document::Restoring) &&
282  !getDocument()->isPerformingTransaction())
283  {
284  // AddSubShape in DressUp acts as a shape cache. And here we shall
285  // invalidate the cache upon changes in Shape. Other features
286  // (currently only feature Transformed) shall call getAddSubShape()
287  // to rebuild the cache. This allow us to perform expensive
288  // calculation of AddSubShape only when necessary.
289  AddSubShape.setValue(Part::TopoShape());
290  }
291  }
292 
293  Feature::onChanged(prop);
294 }

296 void DressUp::getAddSubShape(Part::TopoShape &addShape, Part::TopoShape &subShape)
297 {
298  Part::TopoShape res = AddSubShape.getShape();
299 
300  if(res.isNull()) {
301  try {
302  std::vector<Part::TopoShape> shapes;
303  Part::TopoShape shape = Shape.getShape();
304  shape.setPlacement(Base::Placement());
305 
306  FeatureAddSub *base = nullptr;
307  if(SupportTransform.getValue()) {
308  // SupportTransform means transform the support together with
309  // the dressing. So we need to find the previous support
310  // feature (which must be of type FeatureAddSub), and skipping
311  // any consecutive DressUp in-between.
312  for(Feature *current=this; ;current=static_cast<DressUp*>(base)) {
313  base = Base::freecad_dynamic_cast<FeatureAddSub>(current->getBaseObject(true));
314  if(!base)
315  FC_THROWM(Base::CADKernelError,
316  "Cannot find additive or subtractive support for " << getFullName());
317  if(!base->isDerivedFrom(DressUp::getClassTypeId()))
318  break;
319  }
320  }
321 
322  Part::TopoShape baseShape;
323  if(base) {
324  baseShape = base->getBaseTopoShape(true);
325  baseShape.move(base->getLocation().Inverted());
326  if (base->getAddSubType() == Additive) {
327  if(!baseShape.isNull() && baseShape.hasSubShape(TopAbs_SOLID))
328  shapes.emplace_back(shape.makeElementCut(baseShape.getShape()));
329  else
330  shapes.push_back(shape);
331  } else {
332  BRep_Builder builder;
333  TopoDS_Compound comp;
334  builder.MakeCompound(comp);
335  // push an empty compound to indicate null additive shape
336  shapes.emplace_back(comp);
337  if(!baseShape.isNull() && baseShape.hasSubShape(TopAbs_SOLID))
338  shapes.emplace_back(baseShape.makeElementCut(shape.getShape()));
339  else
340  shapes.push_back(shape);
341  }
342  } else {
343  baseShape = getBaseTopoShape();
344  baseShape.move(getLocation().Inverted());
345  shapes.emplace_back(shape.makeElementCut(baseShape.getShape()));
346  shapes.emplace_back(baseShape.makeElementCut(shape.getShape()));
347  }
348 
349  // Make a compound to contain both additive and subtractive shape,
350  // bceause a dressing (e.g. a fillet) can either be additive or
351  // subtractive. And the dressup feature can contain mixture of both.
352  AddSubShape.setValue(Part::TopoShape().makeElementCompound(shapes));
353 
354  } catch (Standard_Failure &e) {
355  FC_THROWM(Base::CADKernelError, "Failed to calculate AddSub shape: "
356  << e.GetMessageString());
357  }
358  res = AddSubShape.getShape();
359  }
360 
361  if(res.isNull())
362  throw Part::NullShapeException("Null AddSub shape");
363 
364  if(res.getShape().ShapeType() != TopAbs_COMPOUND) {
365  addShape = res;
366  } else {
367  int count = res.countSubShapes(TopAbs_SHAPE);
368  if(!count)
369  throw Part::NullShapeException("Null AddSub shape");
370  if(count) {
371  Part::TopoShape s = res.getSubTopoShape(TopAbs_SHAPE, 1);
372  if(!s.isNull() && s.hasSubShape(TopAbs_SOLID))
373  addShape = s;
374  }
375  if(count > 1) {
376  Part::TopoShape s = res.getSubTopoShape(TopAbs_SHAPE, 2);
377  if(!s.isNull() && s.hasSubShape(TopAbs_SOLID))
378  subShape = s;
379  }
380  }
381 }
