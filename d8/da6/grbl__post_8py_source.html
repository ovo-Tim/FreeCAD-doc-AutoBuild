418 def parse(pathobj):
419 
420  global DRILL_RETRACT_MODE
421  global MOTION_MODE
422  global CURRENT_X
423  global CURRENT_Y
424  global CURRENT_Z
425 
426  out = ""
427  lastcommand = None
428  precision_string = "." + str(PRECISION) + "f"
429 
430  params = [
431  "X",
432  "Y",
433  "Z",
434  "A",
435  "B",
436  "C",
437  "U",
438  "V",
439  "W",
440  "I",
441  "J",
442  "K",
443  "F",
444  "S",
445  "T",
446  "Q",
447  "R",
448  "L",
449  "P",
450  ]
451 
452  if hasattr(pathobj, "Group"): # We have a compound or project.
453  if OUTPUT_COMMENTS:
454  out += linenumber() + "(Compound: " + pathobj.Label + ")\n"
455  for p in pathobj.Group:
456  out += parse(p)
457  return out
458 
459  else: # parsing simple path
460  if not hasattr(pathobj, "Path"): # groups might contain non-path things like stock.
461  return out
462 
463  if OUTPUT_COMMENTS:
464  out += linenumber() + "(Path: " + pathobj.Label + ")\n"
465 
466  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
467  outstring = []
468  command = c.Name
469 
470  outstring.append(command)
471 
472  # if modal: only print the command if it is not the same as the last one
473  if MODAL:
474  if command == lastcommand:
475  outstring.pop(0)
476 
477  # Now add the remaining parameters in order
478  for param in params:
479  if param in c.Parameters:
480  if param == "F":
481  if command not in RAPID_MOVES:
482  speed = Units.Quantity(c.Parameters["F"], FreeCAD.Units.Velocity)
483  if speed.getValueAs(UNIT_SPEED_FORMAT) > 0.0:
484  outstring.append(
485  param
486  + format(
487  float(speed.getValueAs(UNIT_SPEED_FORMAT)),
488  precision_string,
489  )
490  )
491  elif param in ["T", "H", "S"]:
492  outstring.append(param + str(int(c.Parameters[param])))
493  elif param in ["D", "P", "L"]:
494  outstring.append(param + str(c.Parameters[param]))
495  elif param in ["A", "B", "C"]:
496  outstring.append(param + format(c.Parameters[param], precision_string))
497  else: # [X, Y, Z, U, V, W, I, J, K, R, Q] (Conversion eventuelle mm/inches)
498  pos = Units.Quantity(c.Parameters[param], FreeCAD.Units.Length)
499  outstring.append(
500  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
501  )
502 
503  # store the latest command
504  lastcommand = command
505 
506  # Memorizes the current position for calculating the related movements and the withdrawal plan
507  if command in MOTION_COMMANDS:
508  if "X" in c.Parameters:
509  CURRENT_X = Units.Quantity(c.Parameters["X"], FreeCAD.Units.Length)
510  if "Y" in c.Parameters:
511  CURRENT_Y = Units.Quantity(c.Parameters["Y"], FreeCAD.Units.Length)
512  if "Z" in c.Parameters:
513  CURRENT_Z = Units.Quantity(c.Parameters["Z"], FreeCAD.Units.Length)
514 
515  if command in ("G98", "G99"):
516  DRILL_RETRACT_MODE = command
517 
518  if command in ("G90", "G91"):
519  MOTION_MODE = command
520 
521  if TRANSLATE_DRILL_CYCLES:
522  if command in ("G81", "G82", "G83"):
523  out += drill_translate(outstring, command, c.Parameters)
524  # Erase the line we just translated
525  outstring = []
526 
527  if SPINDLE_WAIT > 0:
528  if command in ("M3", "M03", "M4", "M04"):
529  out += linenumber() + format_outstring(outstring) + "\n"
530  out += linenumber() + format_outstring(["G4", "P%s" % SPINDLE_WAIT]) + "\n"
531  outstring = []
532 
533  # Check for Tool Change:
534  if command in ("M6", "M06"):
535  if OUTPUT_COMMENTS:
536  out += linenumber() + "(Begin toolchange)\n"
537  if not OUTPUT_TOOL_CHANGE:
538  outstring.insert(0, "(")
539  outstring.append(")")
540  else:
541  for line in TOOL_CHANGE.splitlines(True):
542  out += linenumber() + line
543 
544  if command == "message":
545  if OUTPUT_COMMENTS is False:
546  out = []
547  else:
548  outstring.pop(0) # remove the command
549 
550  if command in SUPPRESS_COMMANDS:
551  outstring.insert(0, "(")
552  outstring.append(")")
553 
554  # prepend a line number and append a newline
555  if len(outstring) >= 1:
556  out += linenumber() + format_outstring(outstring) + "\n"
557 
558  # Check for comments containing machine-specific commands to pass literally to the controller
559  m = re.match(r"^\(MC_RUN_COMMAND: ([^)]+)\)$", command)
560  if m:
561  raw_command = m.group(1)
562  out += linenumber() + raw_command + "\n"
563 
564  return out

567 def drill_translate(outstring, cmd, params):
568  global DRILL_RETRACT_MODE
569  global MOTION_MODE
570  global CURRENT_X
571  global CURRENT_Y
572  global CURRENT_Z
573  global UNITS
574  global UNIT_FORMAT
575  global UNIT_SPEED_FORMAT
576 
577  strFormat = "." + str(PRECISION) + "f"
578  strG0_Initial_Z = "G0 Z" + format(float(CURRENT_Z.getValueAs(UNIT_FORMAT)), strFormat) + "\n"
579 
580  trBuff = ""
581 
582  if OUTPUT_COMMENTS: # Comment the original command
583  outstring[0] = "(" + outstring[0]
584  outstring[-1] = outstring[-1] + ")"
585  trBuff += linenumber() + format_outstring(outstring) + "\n"
586 
587  # cycle conversion
588  # currently only cycles in XY are provided (G17)
589  # other plains ZX (G18) and YZ (G19) are not dealt with : Z drilling only.
590  drill_X = Units.Quantity(params["X"], FreeCAD.Units.Length)
591  drill_Y = Units.Quantity(params["Y"], FreeCAD.Units.Length)
592  drill_Z = Units.Quantity(params["Z"], FreeCAD.Units.Length)
593  RETRACT_Z = Units.Quantity(params["R"], FreeCAD.Units.Length)
594  # R less than Z is error
595  if RETRACT_Z < drill_Z:
596  trBuff += linenumber() + "(drill cycle error: R less than Z )\n"
597  return trBuff
598 
599  if MOTION_MODE == "G91": # G91 relative movements
600  drill_X += CURRENT_X
601  drill_Y += CURRENT_Y
602  drill_Z += CURRENT_Z
603  RETRACT_Z += CURRENT_Z
604 
605  # if DRILL_RETRACT_MODE == "G98" and CURRENT_Z >= RETRACT_Z:
606  # RETRACT_Z = CURRENT_Z
607 
608  # get the other parameters
609  drill_feedrate = Units.Quantity(params["F"], FreeCAD.Units.Velocity)
610  if cmd == "G83":
611  drill_Step = Units.Quantity(params["Q"], FreeCAD.Units.Length)
612  a_bit = (
613  drill_Step * 0.05
614  ) # NIST 3.5.16.4 G83 Cycle: "current hole bottom, backed off a bit."
615  elif cmd == "G82":
616  drill_DwellTime = params["P"]
617 
618  # wrap this block to ensure machine MOTION_MODE is restored in case of error
619  try:
620  if MOTION_MODE == "G91":
621  trBuff += linenumber() + "G90\n" # force absolute coordinates during cycles
622 
623  strG0_RETRACT_Z = (
624  "G0 Z" + format(float(RETRACT_Z.getValueAs(UNIT_FORMAT)), strFormat) + "\n"
625  )
626  strF_Feedrate = (
627  " F" + format(float(drill_feedrate.getValueAs(UNIT_SPEED_FORMAT)), ".2f") + "\n"
628  )
629  print(strF_Feedrate)
630 
631  # preliminary movement(s)
632  if CURRENT_Z < RETRACT_Z:
633  trBuff += linenumber() + strG0_RETRACT_Z
634  trBuff += (
635  linenumber()
636  + "G0 X"
637  + format(float(drill_X.getValueAs(UNIT_FORMAT)), strFormat)
638  + " Y"
639  + format(float(drill_Y.getValueAs(UNIT_FORMAT)), strFormat)
640  + "\n"
641  )
642  if CURRENT_Z > RETRACT_Z:
643  # NIST GCODE 3.5.16.1 Preliminary and In-Between Motion says G0 to RETRACT_Z.
644  trBuff += (
645  linenumber()
646  + "G0 Z"
647  + format(float(RETRACT_Z.getValueAs(UNIT_FORMAT)), strFormat)
648  + strF_Feedrate
649  )
650  last_Stop_Z = RETRACT_Z
651 
652  # drill moves
653  if cmd in ("G81", "G82"):
654  trBuff += (
655  linenumber()
656  + "G1 Z"
657  + format(float(drill_Z.getValueAs(UNIT_FORMAT)), strFormat)
658  + strF_Feedrate
659  )
