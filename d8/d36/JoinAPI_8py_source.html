33 def shapeOfMaxSize(list_of_shapes):
34  """shapeOfMaxSize(list_of_shapes): finds the shape that has the largest "
35  "mass in the list and returns it. The shapes in the list must be of same dimension."""
36  #first, check if shapes can be compared by size
37  ShapeMerge.dimensionOfShapes(list_of_shapes)
38 
39  rel_precision = 1e-8
40 
41  #find it!
42  max_size = -1e100 # max size encountered so far
43  count_max = 0 # number of shapes with size equal to max_size
44  shape_max = None # shape of max_size
45  for sh in list_of_shapes:
46  v = abs(Part.cast_to_shape(sh).Mass)
47  if v > max_size*(1 + rel_precision) :
48  max_size = v
49  shape_max = sh
50  count_max = 1
51  elif (1-rel_precision) * max_size <= v and v <= (1+rel_precision) * max_size :
52  count_max = count_max + 1
53  if count_max > 1 :
54  raise ValueError("There is more than one largest piece!")
55  return shape_max

57 def connect(list_of_shapes, tolerance = 0.0):
58  """connect(list_of_shapes, tolerance = 0.0): connects solids (walled objects), shells and
59  wires by throwing off small parts that result when splitting them at intersections.
60 
61  Compounds in list_of_shapes are automatically exploded, so self-intersecting compounds
62  are valid for connect."""
63 
64  # explode all compounds before GFA.
65  new_list_of_shapes = []
66  for sh in list_of_shapes:
67  new_list_of_shapes.extend( compoundLeaves(sh) )
68  list_of_shapes = new_list_of_shapes
69 
70  #test if shapes are compatible for connecting
71  dim = ShapeMerge.dimensionOfShapes(list_of_shapes)
72  if dim == 0:
73  raise TypeError("Cannot connect vertices!")
74 
75  if len(list_of_shapes) < 2:
76  return Part.makeCompound(list_of_shapes)
77 
78  pieces, map = list_of_shapes[0].generalFuse(list_of_shapes[1:], tolerance)
79  ao = GeneralFuseResult(list_of_shapes, (pieces, map))
80  ao.splitAggregates()
81  #print len(ao.pieces)," pieces total"
82 
83  keepers = []
84  all_danglers = [] # debug
85 
86  #add all biggest dangling pieces
87  for src in ao.source_shapes:
88  danglers = [piece for piece in ao.piecesFromSource(src) if len(ao.sourcesOfPiece(piece)) == 1]
89  all_danglers.extend(danglers)
90  largest = shapeOfMaxSize(danglers)
91  if largest is not None:
92  keepers.append(largest)
93 
94  touch_test_list = Part.Compound(keepers)
95  #add all intersection pieces that touch danglers, triple intersection pieces that touch duals, and so on
96  for ii in range(2, ao.largestOverlapCount()+1):
97  list_ii_pieces = [piece for piece in ao.pieces if len(ao.sourcesOfPiece(piece)) == ii]
98  keepers_2_add = []
99  for piece in list_ii_pieces:
100  if ShapeMerge.isConnected(piece, touch_test_list):
101  keepers_2_add.append(piece)
102  if len(keepers_2_add) == 0:
103  break
104  keepers.extend(keepers_2_add)
105  touch_test_list = Part.Compound(keepers_2_add)
106 
107 
108  #merge, and we are done!
109  #print len(keepers)," pieces to keep"
110  return ShapeMerge.mergeShapes(keepers)
