260  template<class T>
261  static std::shared_ptr<T> getActionContext(
262  SoAction *action, SoNode *node, std::shared_ptr<T> def=std::shared_ptr<T>(), bool create=true)
263  {
264  auto res = findActionContext(action,node,create,false);
265  if(!res.second) {
266  if(res.first)
267  return std::shared_ptr<T>();
268  // default context is only applicable for non-secondary context query
269  return def;
270  }
271  // make a new context if there is none
272  auto &ctx = *res.second;
273  if(ctx) {
274  auto ret = std::dynamic_pointer_cast<T>(ctx);
275  if(!ret)
276  ctx.reset();
277  }
278  if(!ctx && create)
279  ctx = std::make_shared<T>();
280  return std::static_pointer_cast<T>(ctx);
281  }
