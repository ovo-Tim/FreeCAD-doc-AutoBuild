128 bool ViewProviderProjGroupItem::onDelete(const std::vector<std::string> &)
129 {
130  // we cannot delete the anchor view, thus check if the item is the front item
131  // we also cannot delete if the item has a section or detail view
132 
133  QString bodyMessage;
134  QTextStream bodyMessageStream(&bodyMessage);
135  bool isAnchor = false;
136 
137  // get the item and group
138  TechDraw::DrawProjGroupItem* dpgi = getObject();
139  TechDraw::DrawProjGroup* dpg = dpgi->getPGroup();
140  // check if it is the anchor projection
141  if (dpg && (dpg->getAnchor() == dpgi))
142  isAnchor = true;
143 
144  // get child views
145  auto viewSection = dpgi->getSectionRefs();
146  auto viewDetail = dpgi->getDetailRefs();
147  auto viewLeader = dpgi->getLeaders();
148 
149  if (isAnchor)
150  {
151  // generate dialog
152  bodyMessageStream << qApp->translate("Std_Delete",
153  "You cannot delete the anchor view of a projection group.");
154  QMessageBox::warning(Gui::getMainWindow(),
155  qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
156  QMessageBox::Ok);
157  // don't allow one to delete
158  return false;
159  }
160  else if (!viewSection.empty()) {
161  bodyMessageStream << qApp->translate("Std_Delete",
162  "You cannot delete this view because it has a section view that would become broken.");
163  QMessageBox::warning(Gui::getMainWindow(),
164  qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
165  QMessageBox::Ok);
166  return false;
167  }
168  else if (!viewDetail.empty()) {
169  bodyMessageStream << qApp->translate("Std_Delete",
170  "You cannot delete this view because it has a detail view that would become broken.");
171  QMessageBox::warning(Gui::getMainWindow(),
172  qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
173  QMessageBox::Ok);
174  return false;
175  }
176  else if (!viewLeader.empty()) {
177  bodyMessageStream << qApp->translate("Std_Delete",
178  "You cannot delete this view because it has a leader line that would become broken.");
179  QMessageBox::warning(Gui::getMainWindow(),
180  qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
181  QMessageBox::Ok);
182  return false;
183  }
184  else {
185  return true;
186  }
187 }
