112 void UnitsApi::setSchema(UnitSystem system)
113 {
114  if (UserPrefSystem) {
115  UserPrefSystem->resetSchemaUnits(); // for schemas changed the Quantity constants
116  }
117 
118  UserPrefSystem = createSchema(system);
119  currentSystem = system;
120 
121  // for wrong value fall back to standard schema
122  if (!UserPrefSystem) {
123  UserPrefSystem = std::make_unique<UnitsSchemaInternal>();
124  currentSystem = UnitSystem::SI1;
125  }
126 
127  UserPrefSystem->setSchemaUnits(); // if necessary a unit schema can change the constants in
128  // Quantity (e.g. mi=1.8km rather then 1.6km).
129 }

185 double UnitsApi::toDouble(PyObject* args, const Base::Unit& u)
186 {
187  if (PyUnicode_Check(args)) {
188  std::string str(PyUnicode_AsUTF8(args));
189  // Parse the string
190  Quantity q = Quantity::parse(str);
191  if (q.getUnit() == u) {
192  return q.getValue();
193  }
194  throw Base::UnitsMismatchError("Wrong unit type!");
195  }
196 
197  if (PyFloat_Check(args)) {
198  return PyFloat_AsDouble(args);
199  }
200  if (PyLong_Check(args)) {
201  return static_cast<double>(PyLong_AsLong(args));
202  }
203 
204  throw Base::UnitsMismatchError("Wrong parameter type!");
205 }

207 Quantity UnitsApi::toQuantity(PyObject* args, const Base::Unit& u)

222  else {

224  }
225 
226  return Quantity(d, u);
