54  for (auto f : transFeatures) {

78 const std::list<gp_Trsf>
79 MultiTransform::getTransformations(const std::vector<App::DocumentObject*> originals)
80 {
81  std::vector<App::DocumentObject*> transFeatures = Transformations.getValues();
82 
83  gp_Pnt cog;
84  if (!originals.empty()) {
85  // Find centre of gravity of first original
86  // FIXME: This method will NOT give the expected result for more than one original!
87  if (auto addFeature =
88  Base::freecad_dynamic_cast<PartDesign::FeatureAddSub>(originals.front())) {
89  TopoDS_Shape original = addFeature->AddSubShape.getShape().getShape();
90 
91  GProp_GProps props;
92  BRepGProp::VolumeProperties(original, props);
93  cog = props.CentreOfMass();
94  }
95  }
96 
97  std::list<gp_Trsf> result;
98  std::list<gp_Pnt> cogs;
99 
100  for (auto const& f : transFeatures) {
101  auto transFeature = Base::freecad_dynamic_cast<PartDesign::Transformed>(f);
102  if (!transFeature) {
103  throw Base::TypeError("Transformation features must be subclasses of Transformed");
104  }
105 
106  std::list<gp_Trsf> newTransformations = transFeature->getTransformations(originals);
107  if (result.empty()) {
108  // First transformation Feature
109  result = newTransformations;
110  for (auto nt : newTransformations) {
111  cogs.push_back(cog.Transformed(nt));
112  }
113  }
114  else {
115  // Retain a copy of the first set of transformations for iterator ot
116  // We can't iterate through result if we are also adding elements with push_back()!
117  std::list<gp_Trsf> oldTransformations;
118  result.swap(oldTransformations); // empty result to receive new transformations
119  std::list<gp_Pnt> oldCogs;
120  cogs.swap(oldCogs); // empty cogs to receive new cogs
121 
122  if (transFeature->is<PartDesign::Scaled>()) {
123  // Diagonal method
124  // Multiply every element in the old transformations' slices with the corresponding
125  // element in the newTransformations. Example:
126  // a11 a12 a13 a14 b1 a11*b1 a12*b1 a13*b1 a14*b1
127  // a21 a22 a23 a24 diag b2 = a21*b2 a22*b2 a23*b2 a24*b1
128  // a31 a23 a33 a34 b3 a31*b3 a23*b3 a33*b3 a34*b1
129  // In other words, the length of the result vector is equal to the length of the
130  // oldTransformations vector
131 
132  if (newTransformations.empty()) {
133  throw Base::ValueError("Number of occurrences must be a divisor of previous "
134  "number of occurrences");
135  }
136  if (oldTransformations.size() % newTransformations.size() != 0) {
137  throw Base::ValueError("Number of occurrences must be a divisor of previous "
138  "number of occurrences");
139  }
140 
141  unsigned sliceLength = oldTransformations.size() / newTransformations.size();
142  auto ot = oldTransformations.begin();
143  auto oc = oldCogs.begin();
144 
145  for (auto const& nt : newTransformations) {
146  for (unsigned s = 0; s < sliceLength; s++) {
147  gp_Trsf trans;
148  double factor = nt.ScaleFactor(); // extract scale factor
149 
150  if (factor > Precision::Confusion()) {
151  trans.SetScale(*oc, factor); // recreate the scaled transformation to
152  // use the correct COG
153  trans = trans * (*ot);
154  cogs.push_back(*oc); // Scaling does not affect the COG
155  }
156  else {
157  trans = nt * (*ot);
158  cogs.push_back(oc->Transformed(nt));
159  }
160  result.push_back(trans);
161  ++ot;
162  ++oc;
163  }
164  }
165  }
166  else {
167  // Multiplication method: Combine the new transformations with the old ones.
168  // All old transformations are multiplied with all new ones, so that the length of
169  // the result vector is the length of the old and new transformations multiplied.
170  // a11 a12 b1 a11*b1 a12*b1 a11*b2 a12*b2 a11*b3 a12*b3
171  // a21 a22 mul b2 = a21*b1 a22*b1 a21*b2 a22*b2 a21*b3 a22*b3
172  // b3
173  for (auto const& nt : newTransformations) {
174  auto oc = oldCogs.begin();
175 
176  for (auto const& ot : oldTransformations) {
177  result.push_back(nt * ot);
178  cogs.push_back(oc->Transformed(nt));
179  ++oc;
180  }
181  }
182  }
183  // What about the Additive method: Take the last (set of) transformations and use them
184  // as "originals" for the next transformationFeature, so that something similar to a
185  // sweep for transformations could be put together?
186  }
187  }
188 
189  return result;
190 }
