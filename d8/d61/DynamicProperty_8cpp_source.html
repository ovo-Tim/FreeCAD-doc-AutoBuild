163 Property* DynamicProperty::addDynamicProperty(PropertyContainer& pc,
164  const char* type,
165  const char* name,
166  const char* group,
167  const char* doc,
168  short attr,
169  bool ro,
170  bool hidden)
171 {
172  if (!type) {
173  type = "<null>";
174  }
175 
176  std::string _name;
177 
178  static ParameterGrp::handle hGrp =
179  GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/Document");
180  if (hGrp->GetBool("AutoNameDynamicProperty", false)) {
181  if (!name || !name[0]) {
182  name = type;
183  }
184  _name = getUniquePropertyName(pc, name);
185  if (_name != name) {
186  FC_WARN(pc.getFullName()
187  << " rename dynamic property from '" << name << "' to '" << _name << "'");
188  }
189  name = _name.c_str();
190  }
191  else if (!name) {
192  name = "<null>"; // setting a bad name to trigger exception
193  }
194 
195  auto prop = pc.getPropertyByName(name);
196  if (prop && prop->getContainer() == &pc) {
197  FC_THROWM(Base::NameError,
198  "Property " << pc.getFullName() << '.' << name << " already exists");
199  }
200 
201  if (Base::Tools::getIdentifier(name) != name) {
202  FC_THROWM(Base::NameError, "Invalid property name '" << name << "'");
203  }
204 
205  Base::Type propType =
206  Base::Type::getTypeIfDerivedFrom(type, App::Property::getClassTypeId(), true);
207  if (propType.isBad()) {
208  FC_THROWM(Base::TypeError,
209  "Invalid type " << type << " for property " << pc.getFullName() << '.' << name);
210  }
211 
212  void* propInstance = propType.createInstance();
213  if (!propInstance) {
214  FC_THROWM(Base::RuntimeError,
215  "Failed to create property " << pc.getFullName() << '.' << name << " of type "
216  << type);
217  }
218 
219  Property* pcProperty = static_cast<Property*>(propInstance);
220 
221  auto res = props.get<0>().emplace(pcProperty, name, nullptr, group, doc, attr, ro, hidden);
222 
223  pcProperty->setContainer(&pc);
224  pcProperty->myName = res.first->name.c_str();
225 
226  if (ro) {
227  attr |= Prop_ReadOnly;
228  }
229  if (hidden) {
230  attr |= Prop_Hidden;
231  }
232 
233  pcProperty->syncType(attr);
234  pcProperty->StatusBits.set((size_t)Property::PropDynamic);
235 
236  GetApplication().signalAppendDynamicProperty(*pcProperty);
237 
238  return pcProperty;
239 }

272 bool DynamicProperty::removeDynamicProperty(const char* name)
273 {
274  auto& index = props.get<0>();
275  auto it = index.find(name);
276  if (it != index.end()) {
277  if (it->property->testStatus(Property::LockDynamic)) {
278  throw Base::RuntimeError("property is locked");
279  }
280  else if (!it->property->testStatus(Property::PropDynamic)) {
281  throw Base::RuntimeError("property is not dynamic");
282  }
283  Property* prop = it->property;
284  GetApplication().signalRemoveDynamicProperty(*prop);
285 
286  // Handle possible recursive calls of removeDynamicProperty
287  if (prop->myName) {
288  Property::destroy(prop);
289  index.erase(it);
290  // memory of myName has been freed
291  prop->myName = nullptr;
292  }
293  return true;
294  }
295 
296  return false;
297 }

299 std::string DynamicProperty::getUniquePropertyName(PropertyContainer& pc, const char* Name) const
300 {
301  std::string CleanName = Base::Tools::getIdentifier(Name);
302 
303  // name in use?
304  std::map<std::string, Property*> objectProps;
305  pc.getPropertyMap(objectProps);
306  auto pos = objectProps.find(CleanName);
307 
308  if (pos == objectProps.end()) {
309  // if not, name is OK
310  return CleanName;
311  }
312  else {
313  std::vector<std::string> names;
314  names.reserve(objectProps.size());
315  for (pos = objectProps.begin(); pos != objectProps.end(); ++pos) {
316  names.push_back(pos->first);
317  }
318  return Base::Tools::getUniqueName(CleanName, names);

335 Property* DynamicProperty::restore(PropertyContainer& pc,
336  const char* PropName,
337  const char* TypeName,
338  Base::XMLReader& reader)
339 {
340  if (!reader.hasAttribute("group")) {
341  return nullptr;
342  }
343 
344  short attribute = 0;
345  bool readonly = false, hidden = false;
346  const char *group = nullptr, *doc = nullptr, *attr = nullptr, *ro = nullptr, *hide = nullptr;
347  group = reader.getAttribute("group");
348  if (reader.hasAttribute("doc")) {
349  doc = reader.getAttribute("doc");
350  }
351  if (reader.hasAttribute("attr")) {
352  attr = reader.getAttribute("attr");
353  if (attr) {
354  std::istringstream str(attr);
355  str >> attribute;
356  }
357  }
358  if (reader.hasAttribute("ro")) {
359  ro = reader.getAttribute("ro");
360  if (ro) {
361  readonly = (ro[0] - 48) != 0;
362  }
363  }
364  if (reader.hasAttribute("hide")) {
365  hide = reader.getAttribute("hide");
366  if (hide) {
367  hidden = (hide[0] - 48) != 0;
368  }
369  }
370 
371  return addDynamicProperty(pc, TypeName, PropName, group, doc, attribute, readonly, hidden);
372 }
