75 void Part::closestPointsOnLines(const gp_Lin& lin1, const gp_Lin& lin2, gp_Pnt& p1, gp_Pnt& p2)
76 {
77  // they might be the same point
78  gp_Vec v1(lin1.Direction());
79  gp_Vec v2(lin2.Direction());
80  gp_Vec v3(lin2.Location(), lin1.Location());
81 
82  double a = v1*v1;
83  double b = v1*v2;
84  double c = v2*v2;
85  double d = v1*v3;
86  double e = v2*v3;
87  double D = a*c - b*b;
88  double s, t;
89 
90  // D = (v1 x v2) * (v1 x v2)
91  if (D < Precision::Angular()){
92  // the lines are considered parallel
93  s = 0.0;
94  t = (b>c ? d/b : e/c);
95  }
96  else {
97  s = (b*e - c*d) / D;
98  t = (a*e - b*d) / D;
99  }
100 
101  p1 = lin1.Location().XYZ() + s * v1.XYZ();
102  p2 = lin2.Location().XYZ() + t * v2.XYZ();
103 }

133 Handle(Geom_Surface)
134 Part::Tools::makeSurface(const TColStd_ListOfTransient &theBoundaries,
135  const Standard_Real theTol,
136  const Standard_Integer theNbPnts,
137  const Standard_Integer theNbIter,
138  const Standard_Integer theMaxDeg)
139 {
140  (void)theTol;
141  //constants for algorithm
142  const Standard_Integer aNbIter = theNbIter; //number of algorithm iterations
143  const Standard_Integer aNbPnts = theNbPnts; //sample points per each constraint
144  const Standard_Integer aDeg = 3; //requested surface degree ?
145  const Standard_Integer aMaxDeg = theMaxDeg;
146  const Standard_Integer aMaxSeg = 10000;
147  const Standard_Real aTol3d = 1.e-04;
148  const Standard_Real aTol2d = 1.e-05;
149  const Standard_Real anAngTol = 1.e-02; //angular
150  const Standard_Real aCurvTol = 1.e-01; //curvature
151 
152  Handle(Geom_Surface) aRes;
153  GeomPlate_BuildPlateSurface aPlateBuilder (aDeg, aNbPnts, aNbIter, aTol2d, aTol3d, anAngTol, aCurvTol);
154 
155  TColStd_ListIteratorOfListOfTransient anIt (theBoundaries);
156  if (anIt.More()) {
157  int i = 1;
158  for (; anIt.More(); anIt.Next(), i++) {
159  const Handle(Standard_Transient)& aCur = anIt.Value();
160  if (aCur.IsNull()) {
161  assert (0);
162  Standard_ConstructionError::Raise ("Tools::makeSurface()");
163  }
164 #if OCC_VERSION_HEX >= 0x070600
165  else if (aCur->IsKind (STANDARD_TYPE (Adaptor3d_CurveOnSurface))) {
166  //G1 constraint
167  Handle(Adaptor3d_CurveOnSurface) aHCOS (Handle(Adaptor3d_CurveOnSurface)::DownCast (aCur));
168  Handle (GeomPlate_CurveConstraint) aConst = new GeomPlate_CurveConstraint (aHCOS, 1 /*GeomAbs_G1*/,aNbPnts, aTol3d, anAngTol, aCurvTol);
169  aPlateBuilder.Add (aConst);
170  }
171  else if (aCur->IsKind (STANDARD_TYPE (GeomAdaptor_Curve))) {
172  //G0 constraint
173  Handle(GeomAdaptor_Curve) aHC (Handle(GeomAdaptor_Curve)::DownCast (aCur));
174  Handle (GeomPlate_CurveConstraint) aConst = new GeomPlate_CurveConstraint (aHC, 0 /*GeomAbs_G0*/, aNbPnts, aTol3d);
175  aPlateBuilder.Add (aConst);
176  }
177 #else
178  else if (aCur->IsKind (STANDARD_TYPE (Adaptor3d_HCurveOnSurface))) {
179  //G1 constraint
180  Handle(Adaptor3d_HCurveOnSurface) aHCOS (Handle(Adaptor3d_HCurveOnSurface)::DownCast (aCur));
181  Handle (GeomPlate_CurveConstraint) aConst = new GeomPlate_CurveConstraint (aHCOS, 1 /*GeomAbs_G1*/,aNbPnts, aTol3d, anAngTol, aCurvTol);
182  aPlateBuilder.Add (aConst);
183  }
184  else if (aCur->IsKind (STANDARD_TYPE (GeomAdaptor_HCurve))) {
185  //G0 constraint
186  Handle(GeomAdaptor_HCurve) aHC (Handle(GeomAdaptor_HCurve)::DownCast (aCur));
187  Handle (GeomPlate_CurveConstraint) aConst = new GeomPlate_CurveConstraint (aHC, 0 /*GeomAbs_G0*/, aNbPnts, aTol3d);
188  aPlateBuilder.Add (aConst);
189  }
190 #endif
191  else if (aCur->IsKind (STANDARD_TYPE (Geom_Point))) {
192  //Point constraint
193  Handle(Geom_Point) aGP (Handle(Geom_Point)::DownCast (aCur));
194  Handle(GeomPlate_PointConstraint) aConst = new GeomPlate_PointConstraint(aGP->Pnt(),0);
195  aPlateBuilder.Add(aConst);
196  }
197  else {
198  Standard_TypeMismatch::Raise ("Tools::makeSurface()");
199  }
200  }
201  }
202  else {
203  Standard_ConstructionError::Raise ("Tools::makeSurface()");
204  }
205 
206  //construct
207  aPlateBuilder.Perform();
208 
209  if (!aPlateBuilder.IsDone()) {
210  return aRes;
211  }
212 
213  const Handle(GeomPlate_Surface)& aPlate = aPlateBuilder.Surface();
214  //approximation (see BRepFill_Filling - when no initial surface was given)
215  Standard_Real aDMax = aPlateBuilder.G0Error();
216  TColgp_SequenceOfXY aS2d;
217  TColgp_SequenceOfXYZ aS3d;
218  aPlateBuilder.Disc2dContour (4, aS2d);
219  aPlateBuilder.Disc3dContour (4, 0, aS3d);
220  Standard_Real aMax = Max (aTol3d, 10. * aDMax);
221  GeomPlate_PlateG0Criterion aCriterion (aS2d, aS3d, aMax);
222  {
223  //data races in AdvApp2Var used by GeomApprox_Surface, use global mutex
224  //Standard_Mutex::Sentry aSentry (theBSMutex);
225  GeomPlate_MakeApprox aMakeApprox (aPlate, aCriterion, aTol3d, aMaxSeg, aMaxDeg);
226  aRes = aMakeApprox.Surface();
227  }
228 
229  return aRes;
230 }

232 bool Part::Tools::getTriangulation(const TopoDS_Face& face, std::vector<gp_Pnt>& points, std::vector<Poly_Triangle>& facets)
233 {
234  TopLoc_Location loc;
235  Handle(Poly_Triangulation) hTria = BRep_Tool::Triangulation(face, loc);
236  if (hTria.IsNull())
237  return false;
238 
239  // getting the transformation of the face
240  gp_Trsf transf;
241  bool identity = true;
242  if (!loc.IsIdentity()) {
243  identity = false;
244  transf = loc.Transformation();
245  }
246 
247  // check orientation
248  TopAbs_Orientation orient = face.Orientation();
249 
250  Standard_Integer nbNodes = hTria->NbNodes();
251  Standard_Integer nbTriangles = hTria->NbTriangles();
252 #if OCC_VERSION_HEX < 0x070600
253  const TColgp_Array1OfPnt& nodes = hTria->Nodes();
254  const Poly_Array1OfTriangle& triangles = hTria->Triangles();
255 #endif
256 
257  points.reserve(nbNodes);
258  facets.reserve(nbTriangles);
259 
260  // cycling through the poly mesh
261  //
262  for (int i = 1; i <= nbNodes; i++) {
263 #if OCC_VERSION_HEX < 0x070600
264  gp_Pnt p = nodes(i);
265 #else
266  gp_Pnt p = hTria->Node(i);
267 #endif
268 
269  // transform the vertices to the location of the face
270  if (!identity) {
271  p.Transform(transf);
272  }
273 
274  points.push_back(p);
275  }
276 
277  for (int i = 1; i <= nbTriangles; i++) {
278  // Get the triangle
279  Standard_Integer n1,n2,n3;
280 #if OCC_VERSION_HEX < 0x070600
281  triangles(i).Get(n1, n2, n3);
282 #else
283  hTria->Triangle(i).Get(n1, n2, n3);
284 #endif
285  --n1; --n2; --n3;
286 
287  // change orientation of the triangles
288  if (orient != TopAbs_FORWARD) {
289  std::swap(n1, n2);
290  }
291 
292  facets.emplace_back(n1, n2, n3);
293  }
294 
295  return true;
296 }

377 void Part::Tools::getPointNormals(const std::vector<gp_Pnt>& points, const std::vector<Poly_Triangle>& facets, std::vector<gp_Vec>& vertexnormals)
378 {
379  vertexnormals.resize(points.size());
380 
381  for (const auto& it : facets) {
382  // Get the triangle
383  Standard_Integer n1,n2,n3;
384  it.Get(n1,n2,n3);
385 
386  // Calculate triangle normal
387  gp_Vec v1(points[n1].XYZ());
388  gp_Vec v2(points[n2].XYZ());
389  gp_Vec v3(points[n3].XYZ());
390  gp_Vec n = (v2 - v1) ^ (v3 - v1);
391 
392  // add the triangle normal to the vertex normal for all points of this triangle
393  vertexnormals[n1] += n;
394  vertexnormals[n2] += n;
395  vertexnormals[n3] += n;
396  }
397 
398  for (auto& it : vertexnormals)
399  it.Normalize();
400 }

402 void Part::Tools::getPointNormals(const std::vector<gp_Pnt>& points, const TopoDS_Face& face, std::vector<gp_Vec>& vertexnormals)
403 {
404  if (points.size() != vertexnormals.size())
405  return;
406 
407  Handle(Geom_Surface) hSurface = BRep_Tool::Surface(face);
408  if (hSurface.IsNull())
409  return;
410 
411  // normalize all vertex normals
412  for (std::size_t i = 0; i < points.size(); i++) {
413  try {
414  GeomAPI_ProjectPointOnSurf ProPntSrf(points[i], hSurface);
415  Standard_Real u, v;
416  ProPntSrf.Parameters(1, u, v);
417 
418  GeomLProp_SLProps propOfFace(hSurface, u, v, 2, gp::Resolution());
419 
420  gp_Dir normal = propOfFace.Normal();
421  gp_Vec temp = normal;
422  if (temp * vertexnormals[i] < 0.0)
423  temp = -temp;
424  vertexnormals[i] = temp;
425 
426  }
427  catch (...) {
428  }
429 
430  vertexnormals[i].Normalize();
431  }
432 }

505  }

514  }
515  }
516  }

523  {
524  theNormals(aNodeIter) = aPolyTri->Normal(aNodeIter);
525  }
526

528  {
529  for(Standard_Integer aNodeIter = 1; aNodeIter <= numNodes; ++aNodeIter)
530  {
531  theNormals.ChangeValue(aNodeIter).Reverse();
532  }
533  }
534  }
535  else {
536  // take in face the surface location
537  Poly_Connect thePolyConnect(aPolyTri);
538  const TopoDS_Face aZeroFace = TopoDS::Face(theFace.Located(TopLoc_Location()));
539  Handle(Geom_Surface) aSurf = BRep_Tool::Surface(aZeroFace);
540  const Standard_Real aTol = Precision::Confusion();
541  Standard_Boolean hasNodesUV = aPolyTri->HasUVNodes() && !aSurf.IsNull();
542  Standard_Integer aTri[3];
543 
544  aPolyTri->AddNormals();
545  for(Standard_Integer aNodeIter = 1; aNodeIter <= numNodes; ++aNodeIter)
546  {
547  // try to retrieve normal from real surface first, when UV coordinates are available
548  if (!hasNodesUV || GeomLib::NormEstim(aSurf, aPolyTri->UVNode(aNodeIter), aTol, theNormals(aNodeIter)) > 1)
549  {

552 
553  for(thePolyConnect.Initialize(aNodeIter); thePolyConnect.More(); thePolyConnect.Next())
554  {

556  const gp_XYZ v1(aPolyTri->Node(aTri[1]).Coord() - aPolyTri->Node(aTri[0]).Coord());
557  const gp_XYZ v2(aPolyTri->Node(aTri[2]).Coord() - aPolyTri->Node(aTri[1]).Coord());

685 template <typename T>
686 void getNormalBySLProp(T& prop, double u, double v, Standard_Real lastU, Standard_Real lastV,
687  const Standard_Real tol, gp_Dir& dir, Standard_Boolean& done)
688 {
689  if (prop.D1U().Magnitude() > tol &&
690  prop.D1V().Magnitude() > tol &&
691  prop.IsNormalDefined()) {
692  dir = prop.Normal();
693  done = Standard_True;
694  }
695  // use an alternative method in case of a null normal
696  else {
697  CSLib_NormalStatus stat;
698  CSLib::Normal(prop.D1U(), prop.D1V(), prop.D2U(), prop.D2V(), prop.DUV(),
699  tol, done, stat, dir);
700  // at the right boundary, the normal is flipped with respect to the
701  // normal on surrounding points.
702  if (stat == CSLib_D1NuIsNull) {
703  if (Abs(lastV - v) < tol)
704  dir.Reverse();
705  }
706  else if (stat == CSLib_D1NvIsNull || stat == CSLib_D1NuIsParallelD1Nv) {
707  if (Abs(lastU - u) < tol)
708  dir.Reverse();
709  }
710  }
