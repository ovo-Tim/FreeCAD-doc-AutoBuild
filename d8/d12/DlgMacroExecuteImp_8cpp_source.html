183 QStringList DlgMacroExecuteImp::filterFiles(const QString& folder)
184 {
185  QDir dir(folder, QLatin1String("*.FCMacro *.py"));
186  QStringList unfiltered = dir.entryList(); // all .fcmacro and .py files
187  QString fileFilter = ui->LineEditFind->text(); // used to search by filename
188  QString searchText = ui->LineEditFindInFiles->text(); // used to search in file content
189 
190  if (fileFilter.isEmpty() && searchText.isEmpty()) { // skip filtering if no filters
191  return unfiltered;
192  }
193  QStringList filteredByFileName;
194  if (fileFilter.isEmpty()) {
195  filteredByFileName = unfiltered; // skip the loop if no file filter
196  }
197  else {
198  QRegularExpression regexFileName(fileFilter, QRegularExpression::CaseInsensitiveOption);
199  bool isValidFileFilter = regexFileName.isValid(); // check here instead of inside the loop
200  for (auto uf : unfiltered) {
201  if (isValidFileFilter) {
202  if (regexFileName.match(uf).hasMatch()) {
203  filteredByFileName.append(uf);
204  }
205  }
206  else { // not valid, so do a simple text search
207  if (uf.contains(fileFilter, Qt::CaseInsensitive)) {
208  filteredByFileName.append(uf);
209  }
210  }
211  }
212  }
213 
214  if (searchText.isEmpty()) { // skip reading file contents if no find in file filter
215  return filteredByFileName;
216  }
217 
218  QRegularExpression regexContent(searchText, QRegularExpression::CaseInsensitiveOption);
219  bool isValidContentFilter = regexContent.isValid();
220  QStringList filteredByContent;
221  for (auto fn : filteredByFileName) {
222  const QString& fileName = fn;
223  QString filePath = dir.filePath(fileName);
224  QFile file(filePath);
225  if (file.open(QIODevice::ReadOnly | QIODevice::Text)) {
226  QTextStream in(&file);
227  QString fileContent = in.readAll();
228  if (isValidContentFilter) {
229  if (regexContent.match(fileContent).hasMatch()) {
230  filteredByContent.append(fileName);
231  }
232  }
233  else {
234  if (fileContent.contains(searchText, Qt::CaseInsensitive)) {
235  filteredByContent.append(fileName);
236  }
237  }
238  file.close();
239  }
240  }
241  return filteredByContent;
242 }

455 void DlgMacroExecuteImp::onCreateButtonClicked()
456 {

461  App::GetApplication()
462  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/Macro")
463  ->SetBool("ReplaceSpaces", replaceSpaces); // create parameter
464 
465  QString fn = QInputDialog::getText(this,
466  tr("Macro file"),
467  tr("Enter a file name, please:"),
468  QLineEdit::Normal,
469  QString(),
470  nullptr,
471  Qt::MSWindowsFixedSizeDialogHint);
472 
473  if (replaceSpaces) {
474  fn = fn.replace(QString::fromStdString(" "), QString::fromStdString("_"));
475  }
476 
477  if (!fn.isEmpty()) {
478  QString suffix = QFileInfo(fn).suffix().toLower();
479  if (suffix != QLatin1String("fcmacro") && suffix != QLatin1String("py")) {
480  fn += QLatin1String(".FCMacro");
481  }
482  QDir dir(this->macroPath);
483  // create the macroPath if nonexistent
484  if (!dir.exists()) {
485  dir.mkpath(this->macroPath);
486  }
487  QFileInfo fi(dir, fn);
488  if (fi.exists() && fi.isFile()) {
489  QMessageBox::warning(this,
490  tr("Existing file"),
491  tr("'%1'.\nThis file already exists.").arg(fi.fileName()));
492  }
493  else {
494  QFile file(fi.absoluteFilePath());
495  if (!file.open(QFile::WriteOnly)) {
496  QMessageBox::warning(
497  this,
498  tr("Cannot create file"),
499  tr("Creation of file '%1' failed.").arg(fi.absoluteFilePath()));
500  return;
501  }
502  file.close();
503  auto editor = new PythonEditor();
504  editor->setWindowIcon(Gui::BitmapFactory().iconFromTheme("applications-python"));
505  auto edit = new PythonEditorView(editor, getMainWindow());
506  edit->open(fi.absoluteFilePath());
507  getMainWindow()->appendRecentMacro(fi.absoluteFilePath());
508  edit->setWindowTitle(QString::fromLatin1("%1[*]").arg(fn));
509  edit->resize(400, 300);
510  getMainWindow()->addWindow(edit);
511  close();
512  }
513  }

905 void DlgMacroExecuteImp::onDuplicateButtonClicked()
906 {
907  QDir dir;
908  QTreeWidgetItem* item = nullptr;
909 
910  // When duplicating a macro we can either begin trying to find a unique name with @001 or begin
911  // with the current @NNN if applicable
912 
913  bool from001 = App::GetApplication()
914  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/Macro")
915  ->GetBool("DuplicateFrom001", false);
916  App::GetApplication()
917  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/Macro")
918  ->SetBool("DuplicateFrom001", from001); // create parameter
919 
920  // A user may wish to add a note to end of the filename when duplicating
921  // example: mymacro@005.fix_bug_in_dialog.FCMacro
922  // and then when duplicating to have the extra note removed so the suggested new name is:
923  // mymacro@006.FCMacro instead of mymacro@006.fix_bug_in_dialog.FCMacro since the new duplicate
924  // will be given a new note
925 
926  bool ignoreExtra = App::GetApplication()
927  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/Macro")
928  ->GetBool("DuplicateIgnoreExtraNote", false);
929  App::GetApplication()
930  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/Macro")
931  ->SetBool("DuplicateIgnoreExtraNote", ignoreExtra); // create parameter
932 
933  // when creating a note it will be convenient to convert spaces to underscores if the user
934  // desires this behavior
935 
936  bool replaceSpaces = App::GetApplication()
937  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/Macro")
938  ->GetBool("ReplaceSpaces", true);
939  App::GetApplication()
940  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/Macro")
941  ->SetBool("ReplaceSpaces", replaceSpaces); // create parameter
942 
943  int index = ui->tabMacroWidget->currentIndex();
944  if (index == 0) { // user-specific
945  item = ui->userMacroListBox->currentItem();
946  dir.setPath(this->macroPath);
947  }
948 
949  if (!item) {
950  return;
951  }
952 
953  QString oldName = item->text(0);
954  QFileInfo oldfi(dir, oldName);
955  QFile oldfile(oldfi.absoluteFilePath());
956  QString completeSuffix = oldfi.completeSuffix(); // everything after the first "."
957  QString extraNote = completeSuffix.left(completeSuffix.size() - oldfi.suffix().size());
958  QString baseName = oldfi.baseName(); // everything before first "."
959  QString neutralSymbol = QString::fromStdString("@");
960  QString last3 = baseName.right(3);
961  bool ok = true; // was conversion to int successful?
962  int nLast3 = last3.toInt(&ok);
963  last3 = QString::fromStdString("001"); // increment beginning with 001 unless from001 = false
964  if (ok) {
965  // last3 were all digits, so we strip them from the base name
966  if (baseName.size() > 3) { // if <= 3 leave be (e.g. 2.py becomes 2@001.py)
967  if (!from001) {
968  last3 = baseName.right(3); // use these instead of 001
969  }
970  baseName = baseName.left(baseName.size() - 3); // strip digits
971  if (baseName.endsWith(neutralSymbol)) {
972  baseName =
973  baseName.left(baseName.size() - 1); // trim the "@", will be added back later
974  }
975  }
976  }
977  // at this point baseName = the base name without any digits, e.g. "MyMacro"
978  // neutralSymbol = "@"
979  // last3 is a string representing 3 digits, always "001"
980  // unless from001 = false, in which case we begin with previous numbers
981  // completeSuffix = FCMacro or py or FCMacro.py or else suffix will become FCMacro below
982  // if ignoreExtra any extra notes added between @NN. and .FCMacro will be ignored
983  // when suggesting a new filename
984 
985  if (ignoreExtra && !extraNote.isEmpty()) {
986  nLast3++;
987  last3 = QString::number(nLast3);
988  while (last3.size() < 3) {
989  last3.prepend(QString::fromStdString("0")); // pad 0's if needed
990  }
991  }
992 
993

995  baseName + neutralSymbol + last3 + QString::fromStdString(".") + completeSuffix;
996  QFileInfo fi(dir, oldNameDigitized);

1000  // test from "001" through "999", then give up and let user enter name of choice
1001 
1002  while (fi.exists()) {
1003  nLast3 = last3.toInt() + 1;
1004  if (nLast3 >= 1000) { // avoid infinite loop, 999 files will have to be enough
1005  break;
1006  }
1007  last3 = QString::number(nLast3);
1008  while (last3.size() < 3) {
1009  last3.prepend(QString::fromStdString("0")); // pad 0's if needed
1010  }
1011  oldNameDigitized =
1012  baseName + neutralSymbol + last3 + QString::fromStdString(".") + completeSuffix;
1013  fi = QFileInfo(dir, oldNameDigitized);
1014  }
1015 
1016  if (ignoreExtra && !extraNote.isEmpty()) {
1017  oldNameDigitized = oldNameDigitized.remove(extraNote);
1018  }
1019 
1020  // give user a chance to pick a different name from digitized name suggested
1021  QString fn = QInputDialog::getText(this,
1022  tr("Duplicate Macro"),
1023  tr("Enter new name:"),
1024  QLineEdit::Normal,
1025  oldNameDigitized,
1026  nullptr,
1027  Qt::MSWindowsFixedSizeDialogHint);

1029  fn = fn.replace(QString::fromStdString(" "), QString::fromStdString("_"));
1030  }

1034  fn += QLatin1String(".FCMacro");
