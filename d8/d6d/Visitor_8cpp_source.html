37 unsigned long MeshKernel::VisitNeighbourFacets(MeshFacetVisitor& rclFVisitor,
38  FacetIndex ulStartFacet) const
39 {
40  unsigned long ulVisited = 0, ulLevel = 0;
41  unsigned long ulCount = _aclFacetArray.size();
42  std::vector<FacetIndex> clCurrentLevel, clNextLevel;
43  std::vector<FacetIndex>::iterator clCurrIter;
44  MeshFacetArray::_TConstIterator clCurrFacet, clNBFacet;
45 
46  if (ulStartFacet >= _aclFacetArray.size()) {
47  return 0;
48  }
49 
50  // pick up start point
51  clCurrentLevel.push_back(ulStartFacet);
52  _aclFacetArray[ulStartFacet].SetFlag(MeshFacet::VISIT);
53 
54  // as long as free neighbours
55  while (!clCurrentLevel.empty()) {
56  // visit all neighbours of the current level
57  for (clCurrIter = clCurrentLevel.begin(); clCurrIter < clCurrentLevel.end(); ++clCurrIter) {
58  clCurrFacet = _aclFacetArray.begin() + *clCurrIter;
59 
60  // visit all neighbours of the current level if not yet done
61  for (unsigned short i = 0; i < 3; i++) {
62  auto j = clCurrFacet->_aulNeighbours[i]; // index to neighbour facet
63  if (j == FACET_INDEX_MAX) {
64  continue; // no neighbour facet
65  }
66 
67  if (j >= ulCount) {
68  continue; // error in data structure
69  }
70 
71  clNBFacet = _aclFacetArray.begin() + j;
72 
73  if (!rclFVisitor.AllowVisit(*clNBFacet, *clCurrFacet, j, ulLevel, i)) {
74  continue;
75  }
76  if (clNBFacet->IsFlag(MeshFacet::VISIT)) {
77  continue; // neighbour facet already visited
78  }
79 
80  // visit and mark
81  ulVisited++;
82  clNextLevel.push_back(j);
83  clNBFacet->SetFlag(MeshFacet::VISIT);
84  if (!rclFVisitor.Visit(*clNBFacet, *clCurrFacet, j, ulLevel)) {
85  return ulVisited;
86  }
87  }
88  }
89 
90  clCurrentLevel = clNextLevel;
91  clNextLevel.clear();
92  ulLevel++;
93  }
94 
95  return ulVisited;
96 }

98 unsigned long MeshKernel::VisitNeighbourFacetsOverCorners(MeshFacetVisitor& rclFVisitor,
99  FacetIndex ulStartFacet) const
100 {
101  unsigned long ulVisited = 0, ulLevel = 0;
102  MeshRefPointToFacets clRPF(*this);
103  const MeshFacetArray& raclFAry = _aclFacetArray;
104  MeshFacetArray::_TConstIterator pFBegin = raclFAry.begin();
105  std::vector<FacetIndex> aclCurrentLevel, aclNextLevel;
106 
107  if (ulStartFacet >= _aclFacetArray.size()) {
108  return 0;
109  }
110 
111  aclCurrentLevel.push_back(ulStartFacet);
112  raclFAry[ulStartFacet].SetFlag(MeshFacet::VISIT);
113 
114  while (!aclCurrentLevel.empty()) {
115  // visit all neighbours of the current level
116  for (auto pCurrFacet = aclCurrentLevel.begin(); pCurrFacet < aclCurrentLevel.end();
117  ++pCurrFacet) {
118  for (int i = 0; i < 3; i++) {
119  const MeshFacet& rclFacet = raclFAry[*pCurrFacet];
120  const std::set<FacetIndex>& raclNB = clRPF[rclFacet._aulPoints[i]];
121  for (FacetIndex pINb : raclNB) {
122  if (!pFBegin[pINb].IsFlag(MeshFacet::VISIT)) {
123  // only visit if VISIT Flag not set
124  ulVisited++;
125  FacetIndex ulFInd = pINb;
126  aclNextLevel.push_back(ulFInd);
127  pFBegin[pINb].SetFlag(MeshFacet::VISIT);
128  if (!rclFVisitor.Visit(pFBegin[pINb],
129  raclFAry[*pCurrFacet],
130  ulFInd,
131  ulLevel)) {
132  return ulVisited;
133  }
134  }
135  }
136  }
137  }
138  aclCurrentLevel = aclNextLevel;
139  aclNextLevel.clear();
140  ulLevel++;
141  }

143  return ulVisited;

146 unsigned long MeshKernel::VisitNeighbourPoints(MeshPointVisitor& rclPVisitor,
147  PointIndex ulStartPoint) const
148 {
149  unsigned long ulVisited = 0, ulLevel = 0;
150  std::vector<PointIndex> aclCurrentLevel, aclNextLevel;
151  std::vector<PointIndex>::iterator clCurrIter;
152  MeshPointArray::_TConstIterator pPBegin = _aclPointArray.begin();
153  MeshRefPointToPoints clNPs(*this);
154 
155  aclCurrentLevel.push_back(ulStartPoint);
156  (pPBegin + ulStartPoint)->SetFlag(MeshPoint::VISIT);
157 
158  while (!aclCurrentLevel.empty()) {
159  // visit all neighbours of the current level
160  for (clCurrIter = aclCurrentLevel.begin(); clCurrIter < aclCurrentLevel.end();
161  ++clCurrIter) {
162  const std::set<PointIndex>& raclNB = clNPs[*clCurrIter];
163  for (PointIndex pINb : raclNB) {
164  if (!pPBegin[pINb].IsFlag(MeshPoint::VISIT)) {
165  // only visit if VISIT Flag not set
166  ulVisited++;
167  PointIndex ulPInd = pINb;
168  aclNextLevel.push_back(ulPInd);
169  pPBegin[pINb].SetFlag(MeshPoint::VISIT);
170  if (!rclPVisitor.Visit(pPBegin[pINb],
171  *(pPBegin + (*clCurrIter)),
172  ulPInd,
173  ulLevel)) {
174  return ulVisited;
175  }
176  }
177  }
178  }
179  aclCurrentLevel = aclNextLevel;
180  aclNextLevel.clear();
181  ulLevel++;
182  }

227 bool MeshPlaneVisitor::AllowVisit(const MeshFacet& face,
228  const MeshFacet&,
229  FacetIndex,
230  unsigned long,
231  unsigned short)
232 {
233  if (!fitter->Done()) {
234  fitter->Fit();
235  }
236  MeshGeomFacet triangle = mesh.GetFacet(face);
237  for (const auto& pnt : triangle._aclPoints) {
238  if (std::fabs(fitter->GetDistanceToPlane(pnt)) > max_deviation) {
239  return false;
240  }
241  }
242  return true;
243 }
