179  void Notify(MsgType rcReason)
180  {

182  Iter != _ObserverSet.end();
183  ++Iter) {
184  try {
185  (*Iter)->OnChange(*this, rcReason); // send OnChange-signal
186  }
187  catch (Base::Exception& e) {

189  "The error message is: %s\n",

191  }
192  catch (std::exception& e) {

196  }
