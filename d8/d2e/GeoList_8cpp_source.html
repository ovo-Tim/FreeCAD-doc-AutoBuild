169 template<typename T>
170 Base::Vector3d GeoListModel<T>::getPoint(const Part::Geometry* geo, Sketcher::PointPos pos) const
171 {
172  using namespace Sketcher;
173 
174  if (geo->is<Part::GeomPoint>()) {
175  const Part::GeomPoint* p = static_cast<const Part::GeomPoint*>(geo);
176  if (pos == PointPos::start || pos == PointPos::mid || pos == PointPos::end) {
177  return p->getPoint();
178  }
179  }
180  else if (geo->is<Part::GeomLineSegment>()) {
181  const Part::GeomLineSegment* lineSeg = static_cast<const Part::GeomLineSegment*>(geo);
182  if (pos == PointPos::start) {
183  return lineSeg->getStartPoint();
184  }
185  else if (pos == PointPos::end) {
186  return lineSeg->getEndPoint();
187  }
188  }
189  else if (geo->is<Part::GeomCircle>()) {
190  const Part::GeomCircle* circle = static_cast<const Part::GeomCircle*>(geo);
191  if (pos == PointPos::mid) {
192  return circle->getCenter();
193  }
194  }
195  else if (geo->is<Part::GeomEllipse>()) {
196  const Part::GeomEllipse* ellipse = static_cast<const Part::GeomEllipse*>(geo);
197  if (pos == PointPos::mid) {
198  return ellipse->getCenter();
199  }
200  }
201  else if (geo->is<Part::GeomArcOfCircle>()) {
202  const Part::GeomArcOfCircle* aoc = static_cast<const Part::GeomArcOfCircle*>(geo);
203  if (pos == PointPos::start) {
204  return aoc->getStartPoint(/*emulateCCW=*/true);
205  }
206  else if (pos == PointPos::end) {
207  return aoc->getEndPoint(/*emulateCCW=*/true);
208  }
209  else if (pos == PointPos::mid) {
210  return aoc->getCenter();
211  }
212  }
213  else if (geo->is<Part::GeomArcOfEllipse>()) {
214  const Part::GeomArcOfEllipse* aoc = static_cast<const Part::GeomArcOfEllipse*>(geo);
215  if (pos == PointPos::start) {
216  return aoc->getStartPoint(/*emulateCCW=*/true);
217  }
218  else if (pos == PointPos::end) {
219  return aoc->getEndPoint(/*emulateCCW=*/true);
220  }
221  else if (pos == PointPos::mid) {
222  return aoc->getCenter();
223  }
224  }
225  else if (geo->is<Part::GeomArcOfHyperbola>()) {
226  const Part::GeomArcOfHyperbola* aoh = static_cast<const Part::GeomArcOfHyperbola*>(geo);
227  if (pos == PointPos::start) {
228  return aoh->getStartPoint();
229  }
230  else if (pos == PointPos::end) {
231  return aoh->getEndPoint();
232  }
233  else if (pos == PointPos::mid) {
234  return aoh->getCenter();
235  }
236  }
237  else if (geo->is<Part::GeomArcOfParabola>()) {
238  const Part::GeomArcOfParabola* aop = static_cast<const Part::GeomArcOfParabola*>(geo);
239  if (pos == PointPos::start) {
240  return aop->getStartPoint();
241  }
242  else if (pos == PointPos::end) {
243  return aop->getEndPoint();
244  }
245  else if (pos == PointPos::mid) {
246  return aop->getCenter();
247  }
248  }
249  else if (geo->is<Part::GeomBSplineCurve>()) {
250  const Part::GeomBSplineCurve* bsp = static_cast<const Part::GeomBSplineCurve*>(geo);
251  if (pos == PointPos::start) {
252  return bsp->getStartPoint();
253  }
254  else if (pos == PointPos::end) {
255  return bsp->getEndPoint();
256  }
257  }
258 
259  return Base::Vector3d();
260 }

262 template<typename T>
263 void GeoListModel<T>::rebuildVertexIndex() const
264 {
265  VertexId2GeoElementId.clear();
266  GeoElementId2VertexId.clear();
267 
268  int geoId = 0;
269  int pointId = 0;
270 
271  auto addGeoElement = [this, &pointId](int geoId, PointPos pos) {
272  VertexId2GeoElementId.emplace_back(geoId, pos);
273  GeoElementId2VertexId.emplace(std::piecewise_construct,
274  std::forward_as_tuple(geoId, pos),
275  std::forward_as_tuple(pointId++));
276  };
277 
278  if (geomlist.size() <= 2) {
279  return;
280  }
281  for (auto it = geomlist.begin(); it != geomlist.end(); ++it, geoId++) {
282 
283  Base::Type type;
284 
285  if constexpr (std::is_same<T, Part::Geometry*>::value) {
286  type = (*it)->getTypeId();
287  }
288  else if constexpr (std::is_same<T,
289  std::unique_ptr<const Sketcher::GeometryFacade>>::value) {
290  type = (*it)->getGeometry()->getTypeId();
291  }
292 
293  if (geoId > getInternalCount()) {
294  geoId = -getExternalCount();
295  }
296 
297  if (type == Part::GeomPoint::getClassTypeId()) {
298  addGeoElement(geoId, PointPos::start);
299  }
300  else if (type == Part::GeomLineSegment::getClassTypeId()
301  || type == Part::GeomBSplineCurve::getClassTypeId()) {
302  addGeoElement(geoId, PointPos::start);
303  addGeoElement(geoId, PointPos::end);
304  }
305  else if (type == Part::GeomCircle::getClassTypeId()
306  || type == Part::GeomEllipse::getClassTypeId()) {
307  addGeoElement(geoId, PointPos::mid);
308  }
309  else if (type == Part::GeomArcOfCircle::getClassTypeId()
310  || type == Part::GeomArcOfEllipse::getClassTypeId()
311  || type == Part::GeomArcOfHyperbola::getClassTypeId()
312  || type == Part::GeomArcOfParabola::getClassTypeId()) {
313  addGeoElement(geoId, PointPos::start);
314  addGeoElement(geoId, PointPos::end);
315  addGeoElement(geoId, PointPos::mid);
316  }
317  }
318 
319  indexInit = true;
320 }

332 template<typename T>
333 int GeoListModel<T>::getVertexIdFromGeoElementId(const Sketcher::GeoElementId& geoelementId) const
334 {
335  if (!indexInit) { // lazy initialised
336  rebuildVertexIndex();
337  }
338 
339  auto found =
340  std::find(VertexId2GeoElementId.begin(), VertexId2GeoElementId.end(), geoelementId);
341 
342  if (found != VertexId2GeoElementId.end()) {
343  return std::distance(found, VertexId2GeoElementId.begin());
344  }
345 
346  THROWM(Base::IndexError, "GeoElementId not indexed");
347 }

354 template<>
355 GeoListModel<GeometryFacadeUniquePtr>::GeoListModel(
356  std::vector<GeometryFacadeUniquePtr>&& geometrylist,
357  int intgeocount,
358  bool ownerT)
359  : geomlist(std::move(geometrylist))
360  , intGeoCount(intgeocount)
361  , OwnerT(false)
362  , indexInit(false)
363 {
364  // GeometryFacades hold the responsibility for releasing the resources.
365  //
366  // This means that each GeometryFacade that is passed to the GeoListModel,
367  // must set the ownership (GF->setOwner(true)), if it should be the owner.
368  // Otherwise, it follows the default behaviour that some other class, which
369  // created the pointer, is responsible for freeing it.
370  //
371  // Under the Single Responsibility Principle GeoListModel cannot be made
372  // responsible for releasing those pointers.
373  assert(ownerT == false);
374  boost::ignore_unused(ownerT);
375 }

377 template<>
378 GeoListModel<GeometryFacadeUniquePtr>::GeoListModel(
379  const std::vector<GeometryFacadeUniquePtr>& geometrylist,
380  int intgeocount)
381  : intGeoCount(intgeocount)
382  , OwnerT(false)
383  , indexInit(false)
384 {
385  // GeometryFacades are movable, but not copiable, so they need to be reconstructed (shallow copy
386  // of vector) Under the Single Responsibility Principle, these will not take over a
387  // responsibility that shall be enforced on the original GeometryFacade. Use the move version of
388  // getGeoListModel if moving the responsibility is intended.
389 
390  geomlist.reserve(geometrylist.size());
391 
392  for (auto& v : geometrylist) {
393  geomlist.push_back(GeometryFacade::getFacade(v->getGeometry()));
394  }
