102 ElementNamePair GeoFeature::_getElementName(const char* name,
103  const Data::MappedElement& mapped) const
104 {
105  ElementNamePair ret;
106  if (mapped.index && mapped.name) {
107  std::ostringstream ss;
108  ss << Data::ComplexGeoData::elementMapPrefix() << mapped.name << '.' << mapped.index;
109  std::string result;
110  mapped.index.appendToStringBuffer(result);
111  return ElementNamePair(ss.str().c_str(), result.c_str());
112  }
113  else if (mapped.name) {
114  // FC_TRACE("element mapped name " << name << " not found in " << getFullName());
115  const char* dot = strrchr(name, '.');
116  if (dot) {
117  // deliberately mangle the old style element name to signal a
118  // missing reference
119  std::ostringstream ss;
120  ss << Data::MISSING_PREFIX << dot + 1;
121  return ElementNamePair(name, ss.str().c_str());
122  }
123  return ElementNamePair(name, "");
124  }
125  else {
126  std::string oldName;
127  mapped.index.appendToStringBuffer(oldName);
128  return ElementNamePair("", oldName.c_str());
129  }
130 }

132 DocumentObject* GeoFeature::resolveElement(const DocumentObject* obj,
133  const char* subname,
134  ElementNamePair& elementName,
135  bool append,
136  ElementNameType type,
137  const DocumentObject* filter,
138  const char** _element,
139  GeoFeature** geoFeature)
140 {
141  elementName.newName.clear();
142  elementName.oldName.clear();
143  if (!obj || !obj->isAttachedToDocument()) {
144  return nullptr;
145  }
146  if (!subname) {
147  subname = "";
148  }
149  const char* element = Data::findElementName(subname);
150  if (_element) {
151  *_element = element;
152  }
153  auto sobj = obj->getSubObject(std::string(subname, element).c_str());
154  if (!sobj) {
155  return nullptr;
156  }
157  auto linked = sobj->getLinkedObject(true);
158  auto geo = Base::freecad_dynamic_cast<GeoFeature>(linked);
159  if (!geo && linked) {
160  auto ext = linked->getExtensionByType<LinkBaseExtension>(true);
161  if (ext) {
162  geo = Base::freecad_dynamic_cast<GeoFeature>(ext->getTrueLinkedObject(true));
163  }
164  }
165  if (geoFeature) {
166  *geoFeature = geo;
167  }
168  if (filter && geo != filter) {
169  return nullptr;
170  }
171  if (!element || !element[0]) {
172  if (append) {
173  elementName.oldName = Data::oldElementName(subname);
174  }
175  return sobj;
176  }
177 
178  if (!geo || hasHiddenMarker(element)) {
179  if (!append) {
180  elementName.oldName = element;
181  }
182  else {
183  elementName.oldName = Data::oldElementName(subname);
184  }
185  return sobj;
186  }
187  if (!append) {
188  elementName = geo->getElementName(element, type);
189  }
190  else {
191  const auto& names = geo->getElementName(element, type);
192  std::string prefix(subname, element - subname);
193  if (!names.newName.empty()) {
194  elementName.newName = prefix + names.newName;
195  }
196  elementName.oldName = prefix + names.oldName;
197  }
198  return sobj;
199 }

298 Base::Placement GeoFeature::getGlobalPlacement(App::DocumentObject* targetObj,
299  App::DocumentObject* rootObj,
300  const std::string& sub)
301 {
302  if (!targetObj || !rootObj) {
303  return Base::Placement();
304  }
305  std::vector<std::string> names = Base::Tools::splitSubName(sub);
306 
307  App::Document* doc = rootObj->getDocument();
308  Base::Placement plc = getPlacementFromProp(rootObj, "Placement");
309 
310  if (targetObj == rootObj) {
311  return plc;
312  }
313 
314  if (rootObj->isLink()) {
315  // Update doc in case its an external link.
316  doc = rootObj->getLinkedObject()->getDocument();
317  }
318 
319  for (auto& name : names) {
320  App::DocumentObject* obj = doc->getObject(name.c_str());
321  if (!obj) {
322  return Base::Placement();
323  }
324 
325  plc = plc * getPlacementFromProp(obj, "Placement");
326 
327  if (obj == targetObj) {
328  return plc;
329  }
330  if (obj->isLink()) {
331  // Update doc in case its an external link.
332  doc = obj->getLinkedObject()->getDocument();
333  }
334  }
335 
336  // If targetObj has not been found there's a problem
337  return Base::Placement();
338 }
