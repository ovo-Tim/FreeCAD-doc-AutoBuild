546 def parse(pathobj):
547  out = ""
548  lastcommand = None
549  precision_string = "." + str(PRECISION) + "f"
550  currLocation = {} # keep track for no doubles
551 
552  # The params list control the order of parameters
553  params = [
554  "X",
555  "Y",
556  "Z",
557  "A",
558  "B",
559  "C",
560  "I",
561  "J",
562  "K",
563  "R",
564  "F",
565  "S",
566  "T",
567  "H",
568  "L",
569  "Q",
570  ]
571  firstmove = Path.Command("G0", {"X": -1, "Y": -1, "Z": -1, "F": 0.0})
572  currLocation.update(firstmove.Parameters) # set First location Parameters
573 
574  if hasattr(pathobj, "Group"):
575  # We have a compound or project.
576 
577  # if OUTPUT_COMMENTS:
578  # out += linenumber() + "(compound: " + pathobj.Label + ")\n"
579  for p in pathobj.Group:
580  out += parse(p)
581  return out
582  else:
583  # parsing simple path
584 
585  # groups might contain non-path things like stock.
586  if not hasattr(pathobj, "Path"):
587  return out
588 
589  # if OUTPUT_COMMENTS:
590  # out += linenumber() + "(" + pathobj.Label + ")\n"
591 
592  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
593  commandlist = [] # list of elements in the command, code and params.
594  command = c.Name.strip() # command M or G code or comment string
595  commandlist.append(command)
596 
597  # if modal: only print the command if it is not the same as the last one
598  if MODAL is True:
599  if command == lastcommand:
600  commandlist.pop(0)
601 
602  if c.Name[0] == "(" and not OUTPUT_COMMENTS: # command is a comment
603  continue
604 
605  # Now add the remaining parameters in order
606  for param in params:
607  if param in c.Parameters:
608  if param == "F" and (
609  currLocation[param] != c.Parameters[param] or REPEAT_ARGUMENTS
610  ):
611  if c.Name not in ["G0", "G00"]: # No F in G0
612  speed = Units.Quantity(c.Parameters["F"], FreeCAD.Units.Velocity)
613  if speed.getValueAs(UNIT_SPEED_FORMAT) > 0.0:
614  commandlist.append(
615  param
616  + format(
617  float(speed.getValueAs(UNIT_SPEED_FORMAT)),
618  precision_string,
619  )
620  )
621  else:
622  continue
623  elif param == "T":
624  commandlist.append(param + str(int(c.Parameters["T"])))
625  elif param == "H":
626  commandlist.append(param + str(int(c.Parameters["H"])))
627  elif param == "D":
628  commandlist.append(param + str(int(c.Parameters["D"])))
629  elif param == "S":
630  commandlist.append(param + str(int(c.Parameters["S"])))
631  else:
632  if (
633  (not REPEAT_ARGUMENTS and c.Name not in ["G81", "G82", "G83"])
634  and (param in currLocation)
635  and (currLocation[param] == c.Parameters[param])
636  ):
637  continue
638  else:
639  pos = Units.Quantity(c.Parameters[param], FreeCAD.Units.Length)
640  commandlist.append(
641  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
642  )
643 
644  # store the latest command
645  lastcommand = command
646  currLocation.update(c.Parameters)
647 
648  # Check for Tool Change:
649  if command == "M6":
650  for line in TOOL_CHANGE.splitlines(True):
651  out += linenumber() + line
652 
653  # add height offset
654  if USE_TLO:
655  tool_height = "\nG43 H" + str(int(c.Parameters["T"]))
656  commandlist.append(tool_height)
657 
658  if command == "message":
659  if OUTPUT_COMMENTS is False:
660  out = []
661  else:
662  commandlist.pop(0) # remove the command
663 
664  # prepend a line number and append a newline
665  if len(commandlist) >= 1:
666  if OUTPUT_LINE_NUMBERS:
667  commandlist.insert(0, (linenumber()))
668 
669  # append the line to the final output
670  for w in commandlist:
671  out += w.strip() + COMMAND_SPACE
672  if trace_gcode:
673  print("parse : >>{}".format(out))
674  out = out.strip() + "\n"
675 
676  return out
