26 template <class Real>
27 class Plane3
28 {
29 public:
30  // The plane is represented as Dot(N,X) = c where N is a unit-length
31  // normal vector, c is the plane constant, and X is any point on the
32  // plane. The user must ensure that the normal vector satisfies this
33  // condition.
34 
35  Plane3 (); // uninitialized
36  Plane3 (const Plane3& rkPlane);
37 
38  // specify N and c directly
39  Plane3 (const Vector3<Real>& rkNormal, Real fConstant);
40 
41  // N is specified, c = Dot(N,P) where P is on the plane
42  Plane3 (const Vector3<Real>& rkNormal, const Vector3<Real>& rkP);
43 
44  // N = Cross(P1-P0,P2-P0)/Length(Cross(P1-P0,P2-P0)), c = Dot(N,P0) where
45  // P0, P1, P2 are points on the plane.
46  Plane3 (const Vector3<Real>& rkP0, const Vector3<Real>& rkP1,
47  const Vector3<Real>& rkP2);
48 
49  // assignment
50  Plane3& operator= (const Plane3& rkPlane);
51 
52  // The "positive side" of the plane is the half space to which the plane
53  // normal points. The "negative side" is the other half space. The
54  // function returns +1 for the positive side, -1 for the negative side,
55  // and 0 for the point being on the plane.
56  int WhichSide (const Vector3<Real>& rkP) const;
57 
58  // Compute d = Dot(N,Q)-c where N is the plane normal and c is the plane
59  // constant. This is a signed distance. The sign of the return value is
60  // positive if the point is on the positive side of the plane, negative if
61  // the point is on the negative side, and zero if the point is on the
62  // plane.
63  Real DistanceTo (const Vector3<Real>& rkQ) const;
64 
65  Vector3<Real> Normal;
66  Real Constant;
67 };
