73 bool ViewProviderGeomHatch::setEdit(int ModNum)

76  Gui::TaskView::TaskDialog *dlg = Gui::Control().activeDialog();
77  TaskDlgGeomHatch *projDlg = qobject_cast<TaskDlgGeomHatch *>(dlg);
