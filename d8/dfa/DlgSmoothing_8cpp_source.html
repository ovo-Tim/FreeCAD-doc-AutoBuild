167 bool TaskSmoothing::accept()
168 {
169  std::vector<App::DocumentObject*> meshes = selection->getObjects();
170  if (meshes.empty()) {
171  return true;
172  }
173 
174  Gui::WaitCursor wc;
175  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Mesh Smoothing"));
176 
177  bool hasSelection = false;
178  for (auto it : meshes) {
179  Mesh::Feature* mesh = static_cast<Mesh::Feature*>(it);
180  std::vector<Mesh::FacetIndex> selection;
181  if (widget->smoothSelection()) {
182  // clear the selection before editing the mesh to avoid
183  // to have coloured triangles when doing an 'undo'
184  const Mesh::MeshObject* mm = mesh->Mesh.getValuePtr();
185  mm->getFacetsFromSelection(selection);
186  selection = mm->getPointsFromFacets(selection);
187  mm->clearFacetSelection();
188  if (!selection.empty()) {
189  hasSelection = true;
190  }
191  }
192  Mesh::MeshObject* mm = mesh->Mesh.startEditing();
193  switch (widget->method()) {
194  case MeshGui::DlgSmoothing::Taubin: {
195  MeshCore::TaubinSmoothing s(mm->getKernel());
196  s.SetLambda(widget->lambdaStep());
197  s.SetMicro(widget->microStep());
198  if (widget->smoothSelection()) {
199  s.SmoothPoints(widget->iterations(), selection);
200  }
201  else {
202  s.Smooth(widget->iterations());
203  }
204  } break;
205  case MeshGui::DlgSmoothing::Laplace: {
206  MeshCore::LaplaceSmoothing s(mm->getKernel());
207  s.SetLambda(widget->lambdaStep());
208  if (widget->smoothSelection()) {
209  s.SmoothPoints(widget->iterations(), selection);
210  }
211  else {
212  s.Smooth(widget->iterations());
213  }
214  } break;
215  case MeshGui::DlgSmoothing::MedianFilter: {
216  MeshCore::MedianFilterSmoothing s(mm->getKernel());
217  if (widget->smoothSelection()) {
218  s.SmoothPoints(widget->iterations(), selection);
219  }
220  else {
221  s.Smooth(widget->iterations());
222  }
223  } break;
224  default:
225  break;
226  }
227  mesh->Mesh.finishEditing();
228  }
229 
230  if (widget->smoothSelection() && !hasSelection) {
231  Gui::Command::abortCommand();
232  return false;
233  }
234 
235  Gui::Command::commitCommand();
236  return true;
237 }
