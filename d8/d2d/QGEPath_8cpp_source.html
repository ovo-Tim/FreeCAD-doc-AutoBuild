59 void QGMarker::mousePressEvent(QGraphicsSceneMouseEvent * event)
60 {
61 // Base::Console().Message("QGMarker::mousePressEvent() - focustype: %d\n",
62 // scene()->focusItem()->type() - QGraphicsItem::UserType);
63 
64  if (event->button() == Qt::RightButton) { //we're done
65  Q_EMIT endEdit();
66  event->accept();
67  return;
68  }
69 
70  if(scene() && this == scene()->mouseGrabberItem()) {
71  //start dragging
72  m_dragging = true;
73  Q_EMIT dragging(pos(), getProjIndex()); //pass center of marker[i] to epath
74  }
75  QGIVertex::mousePressEvent(event);
76 }

78 void QGMarker::mouseReleaseEvent(QGraphicsSceneMouseEvent * event)
79 {
80  if (event->button() == Qt::RightButton) {
81  // we are finished our edit session
82  Q_EMIT endEdit();
83  m_dragging = false;
84  return;
85  }
86 
87  if(this->scene() && this == this->scene()->mouseGrabberItem()) {
88  if (m_dragging) {
89  m_dragging = false;
90  setSelected(false);
91  // send this marker's new position to QGEPath
92  Q_EMIT dragFinished(pos(), getProjIndex()); //pass center of marker[i] to epath
93  }
94  }
95  QGIVertex::mouseReleaseEvent(event);
96 }

214 void QGEPath::showMarkers(const std::vector<QPointF>& points)
215 {
216 // Base::Console().Message("QGEPath::showMarkers()\n");
217  if (!inEdit()) {
218  return;
219  }
220 
221  if (points.empty()) {
222  Base::Console().Message("QGEP::showMarkers - no deltas\n");
223  return;
224  }
225 
226  clearMarkers();
227 // dumpGhostPoints("QGEPath::showMarkers");
228 
229  int pointDx = 0;
230  for (auto& p: points) {
231  QGMarker* v = new QGMarker(pointDx);
232  v->setFlag(QGraphicsItem::ItemIsMovable, true);
233  v->setFlag(QGraphicsItem::ItemIsFocusable, true);
234  v->setParentItem(this);
235  QObject::connect(
236  v, &QGMarker::dragFinished,
237  this, &QGEPath::onDragFinished
238  );
239  QObject::connect(
240  v, &QGMarker::dragging,
241  this, &QGEPath::onDragging
242  );
243  QObject::connect(
244  v, &QGMarker::doubleClick,
245  this, &QGEPath::onDoubleClick
246  );
247  QObject::connect(
248  v, &QGMarker::endEdit,
249  this, &QGEPath::onEndEdit
250  );
251  v->setRadius(PreferencesGui::get3dMarkerSize());
252  // v->setRadius(50.0); // this make a huge marker
253  v->setNormalColor(PreferencesGui::getAccessibleQColor(QColor(Qt::black)));
254  v->setZValue(ZVALUE::VERTEX);
255  v->setPos(p);
256  v->show();
257 
258  m_markers.push_back(v);
259  pointDx++;
260  }
261 }

336 void QGEPath::drawGhost()
337 {
338 // Base::Console().Message("QGEPath::drawGhost()\n");
