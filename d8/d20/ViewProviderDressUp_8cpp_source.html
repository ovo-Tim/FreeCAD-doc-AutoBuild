65 bool ViewProviderDressUp::setEdit(int ModNum) {
66  if (ModNum == ViewProvider::Default) {
67  // Here we should prevent edit of a Feature with missing base
68  // Otherwise it could call unhandled exception.
69  PartDesign::DressUp* dressUp = getObject<PartDesign::DressUp>();
70  assert (dressUp);
71  if (dressUp->getBaseObject (/*silent =*/ true)) {
72  return ViewProvider::setEdit(ModNum);
73  } else {
74  QMessageBox::warning ( nullptr, QObject::tr("Feature error"),
75  QObject::tr("%1 misses a base feature.\n"
76  "This feature is broken and can't be edited.")
77  .arg( QString::fromLatin1(dressUp->getNameInDocument()) )
78  );
79  return false;
80  }
81 
82  } else {
83  return ViewProvider::setEdit(ModNum);
84  }
85 }

87 void ViewProviderDressUp::highlightReferences(const bool on)
88 {
89  PartDesign::DressUp* pcDressUp = getObject<PartDesign::DressUp>();
90  Part::Feature* base = pcDressUp->getBaseObject (/*silent =*/ true);
91  if (!base)
92  return;
93  PartGui::ViewProviderPart* vp = dynamic_cast<PartGui::ViewProviderPart*>(
94  Gui::Application::Instance->getViewProvider(base));
95  if (!vp)
96  return;
97 
98  std::vector<std::string> faces = pcDressUp->Base.getSubValuesStartsWith("Face");
99  std::vector<std::string> edges = pcDressUp->Base.getSubValuesStartsWith("Edge");
100 
101  if (on) {
102  if (!faces.empty()) {
103  std::vector<App::Material> materials = vp->ShapeAppearance.getValues();
104 
105  PartGui::ReferenceHighlighter highlighter(base->Shape.getValue(), ShapeAppearance.getDiffuseColor());
106  highlighter.getFaceMaterials(faces, materials);
107 
108  vp->setHighlightedFaces(materials);
109  }
110  if (!edges.empty()) {
111  std::vector<App::Color> colors = vp->LineColorArray.getValues();
112 
113  PartGui::ReferenceHighlighter highlighter(base->Shape.getValue(), LineColor.getValue());
114  highlighter.getEdgeColors(edges, colors);
115 
116  vp->setHighlightedEdges(colors);
117  }
118  } else {
119  vp->unsetHighlightedFaces();
120  vp->unsetHighlightedEdges();
121  }
122 }
