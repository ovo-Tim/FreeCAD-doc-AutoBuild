148 void TaskLoftParameters::onSelectionChanged(const Gui::SelectionChanges& msg)
149 {
150  if (selectionMode == none) {
151  return;
152  }
153 
154  if (msg.Type == Gui::SelectionChanges::AddSelection) {
155  if (referenceSelected(msg)) {
156  App::Document* document = App::GetApplication().getDocument(msg.pDocName);
157  App::DocumentObject* object = document ? document->getObject(msg.pObjectName) : nullptr;
158  if (object) {
159  // TODO: if it is a single vertex of a sketch, use that subshape's name
160  QString label = make2DLabel(object, {msg.pSubName});
161  if (selectionMode == refProfile) {
162  ui->profileBaseEdit->setText(label);
163  }
164  else if (selectionMode == refAdd) {
165  QListWidgetItem* item = new QListWidgetItem();
166  item->setText(label);
167  item->setData(
168  Qt::UserRole,
169  QVariant::fromValue(
170  std::make_pair(object, std::vector<std::string>(1, msg.pSubName))));
171  ui->listWidgetReferences->addItem(item);
172  }
173  else if (selectionMode == refRemove) {
174  removeFromListWidget(ui->listWidgetReferences, label);
175  }
176  }
177 
178  clearButtons();
179  recomputeFeature();
180  }
181 
182  clearButtons();
183  exitSelectionMode();
184  updateUI();
185  }
186 }

188 bool TaskLoftParameters::referenceSelected(const Gui::SelectionChanges& msg) const
189 {
190 
191  if (msg.Type == Gui::SelectionChanges::AddSelection && selectionMode != none) {
192 
193  if (strcmp(msg.pDocName, getObject()->getDocument()->getName()) != 0) {
194  return false;
195  }
196 
197  // not allowed to reference ourself
198  const char* fname = getObject()->getNameInDocument();
199  if (strcmp(msg.pObjectName, fname) == 0) {
200  return false;
201  }
202 
203  // every selection needs to be a profile in itself, hence currently only full objects are
204  // supported, not individual edges of a part
205 
206  // change the references
207  auto loft = getObject<PartDesign::Loft>();
208  App::Document* doc = loft->getDocument();
209  App::DocumentObject* obj = doc->getObject(msg.pObjectName);
210 
211  if (selectionMode == refProfile) {
212  auto view = getViewObject<ViewProviderLoft>();
213  view->highlightReferences(ViewProviderLoft::Profile, false);
214  loft->Profile.setValue(obj, {msg.pSubName});
215  return true;
216  }
217  else if (selectionMode == refAdd || selectionMode == refRemove) {
218  // now check the sections
219  std::vector<App::DocumentObject*> refs = loft->Sections.getValues();
220  std::vector<App::DocumentObject*>::iterator f =
221  std::find(refs.begin(), refs.end(), obj);
222 
223  if (selectionMode == refAdd) {
224  if (f != refs.end()) {
225  return false; // duplicate selection
226  }
227 
228  loft->Sections.addValue(obj, {msg.pSubName});
229  }
230  else if (selectionMode == refRemove) {
231  if (f == refs.end()) {
232  return false;
233  }
234 
235  loft->Sections.removeValue(obj);
236  }
237 
238  return true;
239  }
240  }
241 
242  return false;
243 }

257 void TaskLoftParameters::onDeleteSection()
258 {
259  // Delete the selected profile
260  int row = ui->listWidgetReferences->currentRow();
261  QListWidgetItem* item = ui->listWidgetReferences->takeItem(row);
262  if (item) {
263  QByteArray data(item->data(Qt::UserRole)
264  .value<App::PropertyLinkSubList::SubSet>()
265  .first->getNameInDocument());
266  delete item;
267 
268  // search inside the list of sections
269  if (auto loft = getObject<PartDesign::Loft>()) {
270  std::vector<App::DocumentObject*> refs = loft->Sections.getValues();
271  App::DocumentObject* obj = loft->getDocument()->getObject(data.constData());
272  std::vector<App::DocumentObject*>::iterator f =
273  std::find(refs.begin(), refs.end(), obj);
274  if (f != refs.end()) {
275  loft->Sections.removeValue(obj);
276 
277  recomputeFeature();
278  updateUI();
279  }
280  }
281  }

284 void TaskLoftParameters::indexesMoved()
285 {
286  QAbstractItemModel* model = qobject_cast<QAbstractItemModel*>(sender());
287  if (!model) {
288  return;
289  }
290 
291  if (auto loft = getObject<PartDesign::Loft>()) {
292  auto originals = loft->Sections.getSubListValues();
293 
294  int rows = model->rowCount();
295  for (int i = 0; i < rows; i++) {
296  QModelIndex index = model->index(i, 0);
297  originals[i] = index.data(Qt::UserRole).value<App::PropertyLinkSubList::SubSet>();
298  }
299 
300  loft->Sections.setSubListValues(originals);
301  recomputeFeature();
302  updateUI();
303  }
