93 Part::TopoShape ShapeBinder::updatedShape() const
94 {
95  Part::TopoShape shape;
96  App::GeoFeature* obj = nullptr;
97  std::vector<std::string> subs;
98 
99  ShapeBinder::getFilteredReferences(&Support, obj, subs);
100 
101  //if we have a link we rebuild the shape, but we change nothing if we are a simple copy
102  if (obj) {
103  shape = ShapeBinder::buildShapeFromReferences(obj, subs);
104  //now, shape is in object's CS, and includes local Placement of obj but nothing else.
105 
106  if (TraceSupport.getValue()) {
107  //compute the transform, and apply it to the shape.
108  Base::Placement sourceCS = //full placement of container of obj
109  obj->globalPlacement() * obj->Placement.getValue().inverse();
110  Base::Placement targetCS = //full placement of container of this shapebinder
111  this->globalPlacement() * this->Placement.getValue().inverse();
112  Base::Placement transform = targetCS.inverse() * sourceCS;
113  shape.setPlacement(transform * shape.getPlacement());
114  }
115  }
116 
117  return shape;
118 }

140 void ShapeBinder::getFilteredReferences(const App::PropertyLinkSubList* prop,
141  App::GeoFeature*& obj,
142  std::vector< std::string >& subobjects)
143 {
144  obj = nullptr;
145  subobjects.clear();
146 
147  auto objs = prop->getValues();
148  auto subs = prop->getSubValues();
149 
150  if (objs.empty()) {
151  return;
152  }
153 
154  //we only allow one part feature, so get the first one we find
155  size_t index = 0;
156  for (auto* it : objs) {
157  if (auto part = dynamic_cast<Part::Feature*>(it)) {
158  obj = part;
159  break;
160  }
161  index++;
162  }
163 
164  //do we have any part feature?
165  if (obj) {
166  //if we have no subshpape we use the whole shape
167  if (subs[index].empty()) {
168  return;
169  }
170 
171  //collect all subshapes for the object
172  for (index = 0; index < objs.size(); index++) {
173  //we only allow subshapes from a single Part::Feature
174  if (objs[index] != obj) {
175  continue;
176  }
177 
178  //in this mode the full shape is not allowed, as we already started the subshape
179  //processing
180  if (subs[index].empty()) {
181  continue;
182  }
183 
184  subobjects.push_back(subs[index]);
185  }
186  }
187  else {
188  // search for Origin features
189  for (auto* it : objs) {
190  if (auto line = dynamic_cast<App::Line*>(it)) {
191  obj = line;
192  break;
193  }
194  if (auto plane = dynamic_cast<App::Plane*>(it)) {
195  obj = plane;
196  break;
197  }
198  if (auto point = dynamic_cast<App::Point*>(it)) {
199  obj = point;
200  break;
201  }
202  }
203  }
204 }

206 Part::TopoShape ShapeBinder::buildShapeFromReferences(App::GeoFeature* obj, std::vector< std::string > subs) {
207 
208  if (!obj) {
209  return TopoDS_Shape();
210  }
211 
212  if (obj->isDerivedFrom<Part::Feature>()) {
213  auto part = static_cast<Part::Feature*>(obj);
214  if (subs.empty()) {
215  return part->Shape.getValue();
216  }
217 
218  std::vector<TopoDS_Shape> shapes;
219  shapes.reserve(subs.size());
220  for (const std::string& sub : subs) {
221  shapes.push_back(part->Shape.getShape().getSubShape(sub.c_str()));
222  }
223 
224  if (shapes.size() == 1) {
225  //single subshape. Return directly.
226  return shapes[0];
227  }
228 
229  //multiple subshapes. Make a compound.
230  BRep_Builder builder;
231  TopoDS_Compound cmp;
232  builder.MakeCompound(cmp);
233  for (const TopoDS_Shape& sh : shapes) {
234  builder.Add(cmp, sh);
235  }
236  return cmp;
237  }
238  else if (obj->isDerivedFrom<App::Line>()) {
239  gp_Lin line;
240  BRepBuilderAPI_MakeEdge mkEdge(line);
241  Part::TopoShape shape(mkEdge.Shape());
242  shape.setPlacement(obj->Placement.getValue());
243  return shape;
244  }
245  else if (obj->isDerivedFrom<App::Plane>()) {
246  gp_Pln plane;
247  BRepBuilderAPI_MakeFace mkFace(plane);
248  Part::TopoShape shape(mkFace.Shape());
249  shape.setPlacement(obj->Placement.getValue());
250  return shape;
251  }
252  else if (obj->isDerivedFrom<App::Point>()) {
253  gp_Pnt point;
254  BRepBuilderAPI_MakeVertex mkPoint(point);
255  Part::TopoShape shape(mkPoint.Shape());
256  shape.setPlacement(obj->Placement.getValue());
257  return shape;
258  }
259 
260  return TopoDS_Shape();
261 }

285 void ShapeBinder::slotChangedObject(const App::DocumentObject& Obj, const App::Property& Prop)
286 {
287  App::Document* doc = getDocument();
288  if (!doc || doc->testStatus(App::Document::Restoring)) {
289  return;
290  }
291  if (this == &Obj) {
292  return;
293  }
294  if (!TraceSupport.getValue()) {
295  return;
296  }
297  if (!Prop.isDerivedFrom<App::PropertyPlacement>()) {
298  return;
299  }
300 
301  App::GeoFeature* obj = nullptr;
302  std::vector<std::string> subs;
303  ShapeBinder::getFilteredReferences(&Support, obj, subs);
304  if (obj) {
305  if (obj == &Obj) {
306  // the directly referenced object has changed
307  if (hasPlacementChanged()) {
308  enforceRecompute();
309  }
310  }
311  else if (Obj.hasExtension(App::GroupExtension::getExtensionClassTypeId())) {
312  // check if the changed property belongs to a group-like object
313  // like Body or Part
314  std::vector<App::DocumentObject*> chain;
315  std::vector<App::DocumentObject*> list = getInListRecursive();
316  chain.insert(chain.end(), list.begin(), list.end());
317  list = obj->getInListRecursive();
318  chain.insert(chain.end(), list.begin(), list.end());
319 
320  auto it = std::find(chain.begin(), chain.end(), &Obj);
321  if (it != chain.end()) {
322  if (hasPlacementChanged()) {
323  enforceRecompute();
324  }
325  }
326  }
327  }
328 }

404 App::DocumentObject* SubShapeBinder::getSubObject(const char* subname, PyObject** pyObj,
405  Base::Matrix4D* mat, bool transform, int depth) const
406 {
407  auto sobj = Part::Feature::getSubObject(subname, pyObj, mat, transform, depth);
408  if (sobj)
409  return sobj;
410  if (Data::findElementName(subname) == subname)
411  return nullptr;
412 
413  const char* dot = strchr(subname, '.');
414  if (!dot)
415  return nullptr;
416 
417  App::GetApplication().checkLinkDepth(depth);
418  std::string name(subname, dot - subname);
419  for (auto& l : Support.getSubListValues()) {
420  auto obj = l.getValue();
421  if (!obj || !obj->isAttachedToDocument())
422  continue;
423  for (auto& sub : l.getSubValues()) {
424  auto sobj = obj->getSubObject(sub.c_str());
425  if (!sobj || !sobj->isAttachedToDocument())
426  continue;
427  if (subname[0] == '$') {
428  if (sobj->Label.getStrValue() != name.c_str() + 1)
429  continue;
430  }
431  else if (!boost::equals(sobj->getNameInDocument(), name))
432  continue;
433  name = Data::noElementName(sub.c_str());
434  name += dot + 1;
435  if (mat && transform)
436  *mat *= Placement.getValue().toMatrix();
437  return obj->getSubObject(name.c_str(), pyObj, mat, true, depth + 1);
438  }
439  }
440  return nullptr;
441 }

499 void SubShapeBinder::update(SubShapeBinder::UpdateOption options) {
500  Part::TopoShape result;
501  std::vector<Part::TopoShape> shapes;
502  std::vector<std::pair<int,int> > shapeOwners;
503  std::vector<const Base::Matrix4D*> shapeMats;
504 
505  bool forced = (Shape.getValue().IsNull() || (options & UpdateForced)) ? true : false;
506  bool init = (!forced && (options & UpdateForced)) ? true : false;
507 
508  std::string errMsg;
509  auto parent = Context.getValue();
510  std::string parentSub = Context.getSubName(false);
511  if (!Relative.getValue())
512  parent = nullptr;
513  else {
514  if (parent && parent->getSubObject(parentSub.c_str()) == this) {
515  auto parents = parent->getParents();
516  if (!parents.empty()) {
517  parent = parents.begin()->first;
518  parentSub = parents.begin()->second + parentSub;
519  }
520  }
521  else
522  parent = nullptr;
523  if (!parent && parentSub.empty()) {
524  auto parents = getParents();
525  if (!parents.empty()) {
526  parent = parents.begin()->first;
527  parentSub = parents.begin()->second;
528  }
529  }
530  if (parent && (parent != Context.getValue() || parentSub != Context.getSubName(false)))
531  Context.setValue(parent, parentSub.c_str());
532  }
533 
534  bool first = false;
535  std::unordered_map<const App::DocumentObject*, Base::Matrix4D> mats;
536  int idx = -1;
537  for (auto& l : Support.getSubListValues()) {
538  ++idx;
539  auto obj = l.getValue();
540  if (!obj || !obj->isAttachedToDocument())
541  continue;
542  auto res = mats.emplace(obj, Base::Matrix4D());
543  if (parent && res.second) {
544  std::string resolvedSub = parentSub;
545  std::string linkSub;
546  auto link = obj;
547  auto resolved = parent->resolveRelativeLink(resolvedSub, link, linkSub);
548  if (!resolved) {
549  if (!link) {
550  FC_WARN(getFullName() << " cannot resolve relative link of "
551  << parent->getFullName() << '.' << parentSub
552  << " -> " << obj->getFullName());
553  }
554  }
555  else {
556  Base::Matrix4D mat;
557  auto sobj = resolved->getSubObject(resolvedSub.c_str(), nullptr, &mat);
558  if (sobj != this) {
559  FC_LOG(getFullName() << " skip invalid parent " << resolved->getFullName()
560  << '.' << resolvedSub);
561  }
562  else if (_Version.getValue() == 0) {
563  // For existing legacy SubShapeBinder, we use its Placement
564  // to store the position adjustment of the first Support
565  if (first) {
566  auto pla = Placement.getValue() * Base::Placement(mat).inverse();
567  Placement.setValue(pla);
568  first = false;
569  }
570  else {
571  // The remaining support will cancel the Placement
572  mat.inverseGauss();
573  res.first->second = mat;
574  }
575  }
576  else {
577  // For newer SubShapeBinder, the Placement property is free
578  // to use by the user to add additional offset to the
579  // binding object
580  mat.inverseGauss();
581  res.first->second = Placement.getValue().toMatrix() * mat;
582  }
583  }
584  }
585  if (init)
586  continue;
587 
588  App::DocumentObject* copied = nullptr;
589 
590  if (BindCopyOnChange.getValue() == 2 && Support.getSubListValues().size() == 1) {
591  if (!_CopiedObjs.empty())
592  copied = _CopiedObjs.front().getObject();
593 
594  bool recomputeCopy = false;
595  int copyerror = 0;
596  if (!copied || !copied->isValid()) {
597  recomputeCopy = true;
598  clearCopiedObjects();
599 
600  auto tmpDoc = App::GetApplication().newDocument(
601  "_tmp_binder", nullptr, false, true);
602  tmpDoc->setUndoMode(0);
603  auto objs = tmpDoc->copyObject({ obj }, true, true);
604  if (!objs.empty()) {
605  for (auto it = objs.rbegin(); it != objs.rend(); ++it)
606  _CopiedObjs.emplace_back(*it);
607  copied = objs.back();
608  // IMPORTANT! must make a recomputation first before any
609  // further change so that we can generate the correct
610  // geometry element map.
611  if (!copied->recomputeFeature(true))
612  copyerror = 1;
613  }
614  }
615 
616  if (copied) {
617  if (!copyerror) {
618  std::vector<App::Property*> props;
619  getPropertyList(props);
620  // lambda for copying values of copy-on-change properties
621  const auto copyPropertyValues = [this, &recomputeCopy, &props, copied](const bool to_support) {
622  for (auto prop : props) {
623  if (!App::LinkBaseExtension::isCopyOnChangeProperty(this, *prop))
624  continue;
625  // we only copy read-only and output properties from support to binder
626  if (!to_support && !(prop->testStatus(App::Property::Output) && prop->testStatus(App::Property::ReadOnly)))
627  continue;
628  auto p = copied->getPropertyByName(prop->getName());
629  if (p && p->getContainer() == copied
630  && p->getTypeId() == prop->getTypeId()
631  && !p->isSame(*prop))
632  {
633  recomputeCopy = true;
634  auto* const from = to_support ? prop : p;
635  auto* const to = to_support ? p : prop;
636 
637  std::unique_ptr<App::Property> pcopy(from->Copy());
638  to->Paste(*pcopy);
639  }
640  }
641  };
642 
643  copyPropertyValues(true);
644  if (recomputeCopy && !copied->recomputeFeature(true))
645  copyerror = 2;
646  if (!copyerror)
647  copyPropertyValues(false);
648  }
649  obj = copied;
650  _CopiedLink.setValue(copied, l.getSubValues(false));
651  if (copyerror) {
652  FC_THROWM(Base::RuntimeError,
653  (copyerror == 1 ? "Initial copy failed." : "Copy on change failed.")
654  << " Please check report view for more details.\n"
655  "You can show temporary document to reveal the failed objects using\n"
656  "tree view context menu.");
657  }
658  }
659  }
660 
661  const auto& subvals = copied ? _CopiedLink.getSubValues() : l.getSubValues();
662  std::set<std::string> subs(subvals.begin(), subvals.end());
663  int sidx = copied?-1:idx;
664  int subidx = -1;
665  static std::string none;
666  if (subs.empty())
667  subs.insert(none);

888 void SubShapeBinder::collapseGeoChildren()
889 {
890  // Geo children, i.e. children of GeoFeatureGroup may group some tool
891  // features under itself but does not function as a container. In addition,
892  // its parent group can directly reference the tool feature grouped without
893  // referencing the child. The purpose of this function is to remove any
894  // intermediate Non group features in the object path to avoid unnecessary
895  // dependencies.
896  if (Support.testStatus(App::Property::User3))
897  return;
898 
899  Base::ObjectStatusLocker<App::Property::Status, App::Property>
900  guard(App::Property::User3, &Support);
901  App::PropertyXLinkSubList::atomic_change guard2(Support, false);
902 
903  std::vector<App::DocumentObject*> removes;
904  std::map<App::DocumentObject*, std::vector<std::string> > newVals;
905  std::ostringstream ss;
906  for(auto &l : Support.getSubListValues()) {
907  auto obj = l.getValue();
908  if(!obj || !obj->getNameInDocument())
909  continue;
910  auto subvals = l.getSubValues();
911  if (subvals.empty())
912  continue;
913  bool touched = false;
914  for (auto itSub=subvals.begin(); itSub!=subvals.end();) {
915  auto &sub = *itSub;
916  App::SubObjectT sobjT(obj, sub.c_str());
917  if (sobjT.normalize(App::SubObjectT::NormalizeOption::KeepSubName)) {
918  touched = true;
919  auto newobj = sobjT.getObject();
920  sub = sobjT.getSubName();
921  if (newobj != obj) {
922  newVals[newobj].push_back(std::move(sub));
923  itSub = subvals.erase(itSub);
924  continue;
925  }
926  }
927  ++itSub;
928  }
929  if (touched)
930  removes.push_back(obj);
931  if (!subvals.empty() && touched) {
932  auto &newSubs = newVals[obj];
933  if (newSubs.empty())
934  newSubs = std::move(subvals);
935  else
936  newSubs.insert(newSubs.end(),
937  std::make_move_iterator(subvals.begin()),
938  std::make_move_iterator(subvals.end()));
939  }
940  }
941 
942  if (removes.size() || newVals.size())
943  guard2.aboutToChange();
944  for (auto obj : removes)
945  Support.removeValue(obj);
946  if (newVals.size())
947  setLinks(std::move(newVals));
948 }

1020 void SubShapeBinder::setLinks(std::map<App::DocumentObject*, std::vector<std::string> >&& values, bool reset)
1021 {

1023  if (reset) {
1024  Support.setValue(nullptr);

1027  return;

1029  auto inSet = getInListEx(true);
1030  inSet.insert(this);
1031 
1032  for (auto& v : values) {
1033  if (!v.first || !v.first->isAttachedToDocument())
1034  FC_THROWM(Base::ValueError, "Invalid document object");
1035  if (inSet.find(v.first) != inSet.end())
1036  FC_THROWM(Base::ValueError, "Cyclic reference to " << v.first->getFullName());
1037 
1038  if (v.second.empty()) {
1039  v.second.emplace_back("");

1046  wholeSubs.clear();

1048  v.second[0].clear();

1050  }
1051  else if (sub[sub.size() - 1] == '.')
1052  wholeSubs.push_back(sub);
1053  }

1055  for (auto it = v.second.begin(); it != v.second.end();) {
1056  auto& sub = *it;
1057  if (!boost::starts_with(sub, whole) || sub.size() == whole.size())
1058  ++it;
1059  else {
1060  FC_LOG("Remove subname " << sub << " because of whole selection " << whole);
1061  it = v.second.erase(it);
1062  }
1063  }
1064  }
1065  }
1066 
1067  if (!reset) {
1068  for (auto& link : Support.getSubListValues()) {
1069  auto subs = link.getSubValues();
1070  auto& s = values[link.getValue()];
1071  if (s.empty()) {
1072  s = std::move(subs);
1073  continue;
1074  }
1075  else if (subs.empty() || s[0].empty())
1076  continue;
1077 
1078  for (auto& sub : s) {
1079  for (auto it = subs.begin(); it != subs.end();) {
1080  if (sub[sub.size() - 1] == '.') {
1081  if (boost::starts_with(*it, sub)) {
1082  FC_LOG("Remove subname " << *it << " because of whole selection " << sub);
1083  it = subs.erase(it);
1084  }
1085  else
1086  ++it;
1087  }
1088  else if (it->empty()
1089  || (it->back() == '.' && boost::starts_with(sub, *it)))
1090  {
1091  FC_LOG("Remove whole subname " << *it << " because of " << sub);
1092  it = subs.erase(it);
1093  }
1094  else
1095  ++it;
1096  }
1097  }
1098  subs.insert(subs.end(), s.begin(), s.end());
1099  s = std::move(subs);
1100  }
1101  }
1102  Support.setValues(std::move(values));
