94

237 void ViewProviderBody::onChanged(const App::Property* prop) {
238 
239  if(prop == &DisplayModeBody) {
240  auto body = getObject<PartDesign::Body>();
241 
242  if ( DisplayModeBody.getValue() == 0 ) {
243  //if we are in an override mode we need to make sure to come out, because
244  //otherwise the maskmode is blocked and won't go into "through"
245  if(getOverrideMode() != "As Is") {
246  auto mode = getOverrideMode();
247  ViewProvider::setOverrideMode("As Is");
248  overrideMode = mode;
249  }
250  setDisplayMaskMode("Group");
251  if(body)
252  body->setShowTip(false);
253  }
254  else {
255  if(body)
256  body->setShowTip(true);
257  if(getOverrideMode() == "As Is")
258  setDisplayMaskMode(DisplayMode.getValueAsString());
259  else {
260  Base::Console().Message("Set override mode: %s\n", getOverrideMode().c_str());
261  setDisplayMaskMode(getOverrideMode().c_str());
262  }
263  }
264 
265  // #0002559: Body becomes visible upon changing DisplayModeBody
266  Visibility.touch();
267  }
268  else
269  unifyVisualProperty(prop);
270 
271  PartGui::ViewProviderPartExt::onChanged(prop);
272 }

275 void ViewProviderBody::unifyVisualProperty(const App::Property* prop) {
276 
277  if (!pcObject || isRestoring()) {
278  return;
279  }
280 
281  if (prop == &Visibility ||
282  prop == &Selectable ||
283  prop == &DisplayModeBody ||
284  prop == &PointColorArray ||
285  prop == &LineColorArray) {
286  return;
287  }
288 
289  // Fixes issue 11197. In case of affected projects where the bounding box of a sub-feature
290  // is shown allow it to hide it
291  if (prop == &BoundingBox) {
292  if (BoundingBox.getValue()) {
293  return;
294  }
295  }
296 
297  Gui::Document *gdoc = Gui::Application::Instance->getDocument ( pcObject->getDocument() ) ;
298 
299  PartDesign::Body *body = static_cast<PartDesign::Body *> ( getObject() );
300  auto features = body->Group.getValues();
301  for (auto feature : features) {
302 
303  if (!feature->isDerivedFrom(PartDesign::Feature::getClassTypeId())) {
304  continue;
305  }
306 
307  //copy over the properties data
308  if (Gui::ViewProvider* vp = gdoc->getViewProvider(feature)) {
309  if (auto fprop = vp->getPropertyByName(prop->getName())) {
310  fprop->Paste(*prop);
311  }
312  }
313  }

353 bool ViewProviderBody::canDropObject(App::DocumentObject* obj) const
354 {
355  if (obj->isDerivedFrom<App::VarSet>()) {
356  return true;
357  }
358  else if (obj->isDerivedFrom(App::DatumElement::getClassTypeId())) {
359  // accept only datums that are not part of a LCS.
360  auto* lcs = static_cast<App::DatumElement*>(obj)->getLCS();
361  return !lcs;
362  }
363  else if (obj->isDerivedFrom(App::LocalCoordinateSystem::getClassTypeId())) {
364  return !obj->isDerivedFrom(App::Origin::getClassTypeId());
365  }
366  else if (!obj->isDerivedFrom(Part::Feature::getClassTypeId())) {
367  return false;
368  }
369  else if (PartDesign::Body::findBodyOf(obj)) {
370  return false;
371  }
372  else if (obj->isDerivedFrom (Part::BodyBase::getClassTypeId())) {
373  return false;
374  }
375 
376  App::Part *actPart = PartDesignGui::getActivePart();
377  App::Part* partOfBaseFeature = App::Part::getPartOfObject(obj);
378  if (partOfBaseFeature && partOfBaseFeature != actPart)
379  return false;
380 
381  return true;
382 }

384 void ViewProviderBody::dropObject(App::DocumentObject* obj)
385 {
386  auto* body = getObject<PartDesign::Body>();
387  if (obj->isDerivedFrom<Part::Part2DObject>()
388  || obj->isDerivedFrom<App::DatumElement>()
389  || obj->isDerivedFrom<App::LocalCoordinateSystem>()) {
390  body->addObject(obj);
391  }
392  else if (PartDesign::Body::isAllowed(obj) && PartDesignGui::isFeatureMovable(obj)) {
393  std::vector<App::DocumentObject*> move;
394  move.push_back(obj);
395  std::vector<App::DocumentObject*> deps = PartDesignGui::collectMovableDependencies(move);
396  move.insert(std::end(move), std::begin(deps), std::end(deps));
397 
398  PartDesign::Body* source = PartDesign::Body::findBodyOf(obj);
399  if (source)
400  source->removeObjects(move);
401  try {
402  body->addObjects(move);
403  }
404  catch (const Base::Exception& e) {
405  e.ReportException();
406  }
407  }
408  else if (!body->BaseFeature.getValue()) {
409  body->BaseFeature.setValue(obj);
410  }
411 
412  App::Document* doc = body->getDocument();
413  doc->recompute();
414 
415  // check if a proxy object has been created for the base feature
416  std::vector<App::DocumentObject*> links = body->Group.getValues();
417  for (auto it : links) {
418  if (it->isDerivedFrom<PartDesign::FeatureBase>()) {
419  PartDesign::FeatureBase* base = static_cast<PartDesign::FeatureBase*>(it);
420  if (base && base->BaseFeature.getValue() == obj) {
421  Gui::Application::Instance->hideViewProvider(obj);
422  break;
423  }
424  }
425  }
426 }
