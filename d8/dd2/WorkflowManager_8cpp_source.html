126 Workflow WorkflowManager::determineWorkflow(App::Document *doc) {
127  Workflow rv = getWorkflowForDocument (doc);
128 
129  if (rv != Workflow::Undetermined) {
130  // Return if workflow is known
131  return rv;
132  }
133 
134  // Guess the workflow again
135  rv = guessWorkflow (doc);
136  if (rv != Workflow::Modern) {
137  QMessageBox msgBox(Gui::getMainWindow());
138 
139  if ( rv == Workflow::Legacy ) { // legacy messages
140  msgBox.setText( QObject::tr( "The document \"%1\" you are editing was designed with an old version of "
141  "PartDesign workbench." ).arg( QString::fromStdString ( doc->getName()) ) );
142  msgBox.setInformativeText (
143  QObject::tr( "Do you want to migrate in order to use modern PartDesign features?" ) );
144  }
145  else { // The document is already in the middle of migration
146  msgBox.setText( QObject::tr( "The document \"%1\" seems to be either in the middle of"
147  " the migration process from legacy PartDesign or have a slightly broken structure."
148  ).arg( QString::fromStdString ( doc->getName()) ) );
149  msgBox.setInformativeText (
150  QObject::tr( "Do you want to make the migration automatically?" ) );
151  }
152  msgBox.setDetailedText( QObject::tr( "Note: If you choose to migrate you won't be able to edit"
153  " the file with an older FreeCAD version.\n"
154  "If you refuse to migrate you won't be able to use new PartDesign features"
155  " like Bodies and Parts. As a result you also won't be able to use your parts"
156  " in the assembly workbench.\n"
157  "Although you will be able to migrate any moment later with 'Part Design -> Migrate'." ) );
158  msgBox.setIcon( QMessageBox::Question );
159  QPushButton * yesBtn = msgBox.addButton ( QMessageBox::Yes );
160  QPushButton * manuallyBtn = msgBox.addButton (
161  QObject::tr ( "Migrate manually" ), QMessageBox::YesRole );
162 
163  // If it is already a document in the middle of the migration the user shouldn't refuse to migrate
164  if ( rv != Workflow::Undetermined ) {
165  msgBox.addButton ( QMessageBox::No );
166  }
167  msgBox.setDefaultButton ( yesBtn );
168  // TODO: Add some description of manual migration mode (2015-08-09, Fat-Zer)
169 
170  msgBox.exec();
171 
172  if ( msgBox.clickedButton() == yesBtn ) {
173  Gui::Application::Instance->commandManager().runCommandByName("PartDesign_Migrate");
174  rv = Workflow::Modern;
175  } else if ( msgBox.clickedButton() == manuallyBtn ) {
176  rv = Workflow::Undetermined;
177  } else {
178  rv = Workflow::Legacy;
179  }
180  }
181 
182  // Actually set the result in our map
183  dwMap[ doc ] = rv;
184 
185  return rv;
186 }

192 Workflow WorkflowManager::guessWorkflow(const App::Document *doc) {
193  // Retrieve bodies of the document
194  auto features = doc->getObjectsOfType( PartDesign::Feature::getClassTypeId() );
195 
196  if( features.empty() ) {
197  // a new file should be done in the new workflow
198  return Workflow::Modern;
199  } else {
200  auto bodies = doc->getObjectsOfType( PartDesign::Body::getClassTypeId() );
201  if (bodies.empty()) {
202  // If there are no bodies workflow is legacy
203  return Workflow::Legacy;
204  } else {
205  bool features_without_bodies = false;
206 
207  for( auto feat: features ) {
208  if( !PartDesign::Body::findBodyOf( feat ) ) {
209  features_without_bodies = true;
210  break;
211  }
212  }
213  // if there are features not belonging to any body it means that migration was incomplete, otherwise it's Modern
214  return features_without_bodies ? Workflow::Undetermined : Workflow::Modern;
215  }
216  }
217 }
