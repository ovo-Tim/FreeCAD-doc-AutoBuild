35 // FilterValue and SpecialFilterValue are the filters used for Constraint Filtering in the
36 // Constraint's Widget. FilterValue includes the filters used for the Multi-Filter dialog
37 // (SpecialFilterValue is not part of the multi-filter).
38 //
39 // The values are hardcoded to be the same as the indices in the combobox and viewlist. Addition of
40 // an element here requires the addition of the corresponding entry there and vice versa.
41

95 inline bool checkFilterBitset(FilterValueBitset set, FilterValue filter)
96 {
97  auto underlyingFilterValue = static_cast<std::underlying_type_t<FilterValue>>(filter);
98 
99  return set[underlyingFilterValue];

104 template<typename... Args>
105 constexpr decltype(auto) buildBitset(Args... args)
106 {
107  return (... | (1 << static_cast<std::underlying_type_t<Args>>(args)));
108 }
