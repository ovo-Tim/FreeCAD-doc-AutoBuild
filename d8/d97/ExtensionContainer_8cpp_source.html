52 void ExtensionContainer::registerExtension(Base::Type extension, Extension* ext)
53 {
54  if (ext->getExtendedContainer() != this) {
55  throw Base::ValueError(
56  "ExtensionContainer::registerExtension: Extension has not this as base object");
57  }
58 
59  // no duplicate extensions (including base classes)
60  if (hasExtension(extension)) {
61  for (const auto& entry : _extensions) {
62  if (entry.first == extension || entry.first.isDerivedFrom(extension)) {
63  _extensions.erase(entry.first);
64  break;
65  }
66  }
67  }
68 
69  _extensions[extension] = ext;
70 }

102 Extension* ExtensionContainer::getExtension(Base::Type t, bool derived, bool no_except) const
103 {
104 
105  auto result = _extensions.find(t);
106  if ((result == _extensions.end()) && derived) {
107  // we need to check for derived types
108  for (const auto& entry : _extensions) {
109  if (entry.first.isDerivedFrom(t)) {
110  return entry.second;
111  }
112  }
113  if (no_except) {
114  return nullptr;
115  }
116  // if we arrive here we don't have anything matching
117  throw Base::TypeError(
118  "ExtensionContainer::getExtension: No extension of given type available");
119  }
120  else if (result != _extensions.end()) {
121  return result->second;
122  }
123  else {
124  if (no_except) {
125  return nullptr;
126  }
127  // if we arrive here we don't have anything matching
128  throw Base::TypeError(
129  "ExtensionContainer::getExtension: No extension of given type available");
130  }
131 }

361 void ExtensionContainer::saveExtensions(Base::Writer& writer) const
362 {
363 
364  // we don't save anything if there are no dynamic extensions
365  if (!hasExtensions()) {
366  return;
367  }
368 
369  // save dynamic extensions
370  writer.incInd(); // indentation for 'Extensions'
371  writer.Stream() << writer.ind() << "<Extensions Count=\"" << _extensions.size() << "\">"
372  << std::endl;
373  for (const auto& entry : _extensions) {
374 
375  auto ext = entry.second;
376  writer.incInd(); // indentation for 'Extension name'
377  writer.Stream() << writer.ind() << "<Extension"
378  << " type=\"" << ext->getExtensionTypeId().getName() << "\""
379  << " name=\"" << ext->name() << "\">" << std::endl;
380  writer.incInd(); // indentation for the actual Extension
381  try {
382  // We must make sure to handle all exceptions accordingly so that
383  // the project file doesn't get invalidated. In the error case this
384  // means to proceed instead of aborting the write operation.
385  ext->extensionSave(writer);
386  }
387  catch (const Base::Exception& e) {
388  Base::Console().Error("%s\n", e.what());
389  }
390  catch (const std::exception& e) {
391  Base::Console().Error("%s\n", e.what());
392  }
393  catch (const char* e) {
394  Base::Console().Error("%s\n", e);
395  }
396 #ifndef FC_DEBUG
397  catch (...) {
398  Base::Console().Error(
399  "ExtensionContainer::Save: Unknown C++ exception thrown. Try to continue...\n");
400  }
401 #endif
402  writer.decInd(); // indentation for the actual extension
403  writer.Stream() << writer.ind() << "</Extension>" << std::endl;
404  writer.decInd(); // indentation for 'Extension name'
405  }
406  writer.Stream() << writer.ind() << "</Extensions>" << std::endl;
407  writer.decInd();
408 }

410 void ExtensionContainer::restoreExtensions(Base::XMLReader& reader)
411 {
412 
413  // Dynamic extensions are optional (also because they are introduced late into the document
414  // format) and hence it is possible that the element does not exist. As we cannot check for the
415  // existence of an element a object attribute is set if extensions are available. Here we check
416  // that attribute, and only if it exists the extensions element will be available.
417  if (!reader.hasAttribute("Extensions")) {
418  return;
419  }
420 
421  reader.readElement("Extensions");
422  int Cnt = reader.getAttributeAsInteger("Count");
423 
424  for (int i = 0; i < Cnt; i++) {
425  reader.readElement("Extension");
426  const char* Type = reader.getAttribute("type");
427  const char* Name = reader.getAttribute("name");
428  try {
429  App::Extension* ext = getExtension(Name);
430  if (!ext) {
431  // get the extension type asked for
432  Base::Type extension = Base::Type::fromName(Type);
433  if (extension.isBad()
434  || !extension.isDerivedFrom(App::Extension::getExtensionClassTypeId())) {
435  std::stringstream str;
436  str << "No extension found of type '" << Type << "'" << std::ends;
437  throw Base::TypeError(str.str());
438  }
439 
440  // register the extension
441  ext = static_cast<App::Extension*>(extension.createInstance());
442  // check if this really is a python extension!
443  if (!ext->isPythonExtension()) {
444  delete ext;
445  std::stringstream str;
446  str << "Extension is not a python addable version: '" << Type << "'";
447  throw Base::TypeError(str.str());
448  }
449 
450  ext->initExtension(this);
451  }
452  if (ext && strcmp(ext->getExtensionTypeId().getName(), Type) == 0) {
453  ext->extensionRestore(reader);
454  }
455  }
456  catch (const Base::XMLParseException&) {
457  throw; // re-throw
458  }
459  catch (const Base::Exception& e) {
460  Base::Console().Error("%s\n", e.what());
461  }
462  catch (const std::exception& e) {
463  Base::Console().Error("%s\n", e.what());
464  }
465  catch (const char* e) {
466  Base::Console().Error("%s\n", e);
467  }
468 #ifndef FC_DEBUG
469  catch (...) {
470  Base::Console().Error("ExtensionContainer::Restore: Unknown C++ exception thrown\n");
471  }
472 #endif
473 
474  reader.readEndElement("Extension");
475  }
476  reader.readEndElement("Extensions");
477 }
