194 class BaseExport Exception: public BaseClass
195 {
196  TYPESYSTEM_HEADER_WITH_OVERRIDE();
197 
198 public:
199  ~Exception() noexcept override = default;
200 
201  Exception& operator=(const Exception& inst);
202  Exception& operator=(Exception&& inst) noexcept;
203 
204  virtual const char* what() const noexcept;
205 
207  virtual void ReportException() const;
208 
209  inline void setMessage(const char* sMessage);
210  inline void setMessage(const std::string& sMessage);
211  // what may differ from the message given by the user in
212  // derived classes
213  inline std::string getMessage() const;
214  inline std::string getFile() const;
215  inline int getLine() const;
216  inline std::string getFunction() const;
217  inline bool getTranslatable() const;
218  inline bool getReported() const
219  {
220  return _isReported;
221  }
222 
225  inline void setDebugInformation(const std::string& file, int line, const std::string& function);
226 
227  inline void setTranslatable(bool translatable);
228 
229  inline void setReported(bool reported)
230  {
231  _isReported = reported;
232  }
233 
235  PyObject* getPyObject() override;
237  void setPyObject(PyObject* pydict) override;
238 
240  virtual PyObject* getPyExceptionType() const;
242  virtual void setPyException() const;
243 
244 protected:
245  /* sMessage may be:
246  * - a UI compliant string susceptible to being translated and shown to the user in the UI
247  * - a very technical message not intended to be translated or shown to the user in the UI
248  * The preferred way of throwing an exception is using the macros above.
249  * This way, the file, line, and function are automatically inserted. */
250  explicit Exception(const char* sMessage);
251  explicit Exception(std::string sMessage);
252  Exception();
253  Exception(const Exception& inst);
254  Exception(Exception&& inst) noexcept;
255 
256 protected:
257  std::string _sErrMsg;
258  std::string _file;
259  int _line;

261  bool _isTranslatable;
262  mutable bool _isReported;
263 };
