184  for (auto gy1 = 0UL; gy1 < ctGy1; gy1++) {

186  if (grid1.GetCtElements(gx1, gy1, gz1) > 0) {
187  std::vector<FacetIndex> vecFacets2;
188  grid2.Inside(grid1.GetBoundBox(gx1, gy1, gz1), vecFacets2);

190  if (!vecFacets2.empty()) {
191  std::set<FacetIndex> vecFacets1;
192  grid1.GetElements(gx1, gy1, gz1, vecFacets1);
193 
194  std::set<FacetIndex>::iterator it1;
195  for (it1 = vecFacets1.begin(); it1 != vecFacets1.end(); ++it1) {
196  FacetIndex fidx1 = *it1;
197  MeshGeomFacet f1 = _cutMesh0.GetFacet(*it1);
198 
199  std::vector<FacetIndex>::iterator it2;
200  for (it2 = vecFacets2.begin(); it2 != vecFacets2.end(); ++it2) {
201  FacetIndex fidx2 = *it2;
202  MeshGeomFacet f2 = _cutMesh1.GetFacet(fidx2);
203 
204  MeshPoint p0, p1;
205 
206  int isect = f1.IntersectWithFacet(f2, p0, p1);
207  if (isect > 0) {
208  // optimize cut line if distance to nearest point is too small
209  float minDist1 = _minDistanceToPoint,
210  minDist2 = _minDistanceToPoint;
211  MeshPoint np0 = p0, np1 = p1;
212  for (int i = 0; i < 3; i++) // NOLINT
213  {
214  float d1 = (f1._aclPoints[i] - p0).Length();
215  float d2 = (f1._aclPoints[i] - p1).Length();
216  if (d1 < minDist1) {
217  minDist1 = d1;
218  np0 = f1._aclPoints[i];
219  }
220  if (d2 < minDist2) {
221  minDist2 = d2;
222  p1 = f1._aclPoints[i];
223  }
224  } // for (int i = 0; i < 3; i++)
225 
226  // optimize cut line if distance to nearest point is too small
227  for (int i = 0; i < 3; i++) // NOLINT
228  {
229  float d1 = (f2._aclPoints[i] - p0).Length();
230  float d2 = (f2._aclPoints[i] - p1).Length();
231  if (d1 < minDist1) {
232  minDist1 = d1;
233  np0 = f2._aclPoints[i];
234  }
235  if (d2 < minDist2) {
236  minDist2 = d2;
237  np1 = f2._aclPoints[i];
238  }
239  } // for (int i = 0; i < 3; i++)
240 
241  MeshPoint mp0 = np0;
242  MeshPoint mp1 = np1;
243 
244  if (mp0 != mp1) {
245  facetsCuttingEdge0.insert(fidx1);
246  facetsCuttingEdge1.insert(fidx2);
247 
248  _cutPoints.insert(mp0);
249  _cutPoints.insert(mp1);
250 
251  std::pair<std::set<MeshPoint>::iterator, bool> pit0 =
252  _cutPoints.insert(mp0);
253  std::pair<std::set<MeshPoint>::iterator, bool> pit1 =
254  _cutPoints.insert(mp1);
255 
256  _edges[Edge(mp0, mp1)] = EdgeInfo();
257 
258  _facet2points[0][fidx1].push_back(pit0.first);
259  _facet2points[0][fidx1].push_back(pit1.first);
260  _facet2points[1][fidx2].push_back(pit0.first);

262  }
263  else {

265  _cutPoints.insert(mp0);
266 
267  // do not insert a facet when only one corner point cuts the
268  // edge if (!((mp0 == f1._aclPoints[0]) || (mp0 ==

274

292 void SetOperations::TriangulateMesh(const MeshKernel& cutMesh, int side)
293 {
294  // Triangulate Mesh
295  std::map<FacetIndex, std::list<std::set<MeshPoint>::iterator>>::iterator it1;
296  for (it1 = _facet2points[side].begin(); it1 != _facet2points[side].end(); ++it1) {
297  std::vector<Vector3f> points;
298  std::set<MeshPoint> pointsSet;
299 
300  FacetIndex fidx = it1->first;
301  MeshGeomFacet f = cutMesh.GetFacet(fidx);
302 
303  // if (side == 1)
304  // _builder.addSingleTriangle(f._aclPoints[0], f._aclPoints[1], f._aclPoints[2], 3, 0,
305  // 1, 1);
306 
307  // facet corner points
308  // const MeshFacet& mf = cutMesh._aclFacetArray[fidx];
309  for (int i = 0; i < 3; i++) // NOLINT
310  {
311  pointsSet.insert(f._aclPoints[i]);
312  points.push_back(f._aclPoints[i]);
313  }
314 
315  // triangulated facets
316  std::list<std::set<MeshPoint>::iterator>::iterator it2;
317  for (it2 = it1->second.begin(); it2 != it1->second.end(); ++it2) {
318  if (pointsSet.find(*(*it2)) == pointsSet.end()) {
319  pointsSet.insert(*(*it2));
320  points.push_back(*(*it2));
321  }
322  }
323 
324  Vector3f normal = f.GetNormal();
325  Vector3f base = points[0];
326  Vector3f dirX = points[1] - points[0];
327  dirX.Normalize();
328  Vector3f dirY = dirX % normal;
329 
330  // project points to 2D plane
331  std::vector<Vector3f>::iterator it;
332  std::vector<Vector3f> vertices;
333  for (it = points.begin(); it != points.end(); ++it) {
334  Vector3f pv = *it;
335  pv.TransformToCoordinateSystem(base, dirX, dirY);
336  vertices.push_back(pv);
337  }
338 
339  DelaunayTriangulator tria;
340  tria.SetPolygon(vertices);
341  tria.TriangulatePolygon();
342 
343  std::vector<MeshFacet> facets = tria.GetFacets();
344  for (auto& it : facets) {
345  if ((it._aulPoints[0] == it._aulPoints[1]) || (it._aulPoints[1] == it._aulPoints[2])
346  || (it._aulPoints[2] == it._aulPoints[0])) { // two same triangle corner points
347  continue;
348  }
349 
350  MeshGeomFacet facet(points[it._aulPoints[0]],
351  points[it._aulPoints[1]],
352  points[it._aulPoints[2]]);
353 
354  // if (side == 1)
355  // _builder.addSingleTriangle(facet._aclPoints[0], facet._aclPoints[1],
356  // facet._aclPoints[2], true, 3, 0, 1, 1);
357 
358  // if (facet.Area() < 0.0001f)
359  //{ // too small facet
360  // continue;
361  // }
362 
363  float dist0 =
364  facet._aclPoints[0].DistanceToLine(facet._aclPoints[1],
365  facet._aclPoints[1] - facet._aclPoints[2]);
366  float dist1 =
367  facet._aclPoints[1].DistanceToLine(facet._aclPoints[0],
368  facet._aclPoints[0] - facet._aclPoints[2]);
369  float dist2 =
370  facet._aclPoints[2].DistanceToLine(facet._aclPoints[0],
371  facet._aclPoints[0] - facet._aclPoints[1]);
372 
373  if ((dist0 < _minDistanceToPoint) || (dist1 < _minDistanceToPoint)
374  || (dist2 < _minDistanceToPoint)) {
375  continue;
376  }
377 
378  // dist0 = (facet._aclPoints[0] - facet._aclPoints[1]).Length();
379  // dist1 = (facet._aclPoints[1] - facet._aclPoints[2]).Length();
380  // dist2 = (facet._aclPoints[2] - facet._aclPoints[3]).Length();
381 
382  // if ((dist0 < _minDistanceToPoint) || (dist1 < _minDistanceToPoint) || (dist2 <
383  // _minDistanceToPoint))
384  //{
385  // continue;
386  // }
387 
388  facet.CalcNormal();
389  if ((facet.GetNormal() * f.GetNormal()) < 0.0F) { // adjust normal
390  std::swap(facet._aclPoints[0], facet._aclPoints[1]);
391  facet.CalcNormal();
392  }
393 
394 
395  for (int j = 0; j < 3; j++) {
396  auto eit = _edges.find(Edge(facet._aclPoints[j], facet._aclPoints[(j + 1) % 3]));
397 
398  if (eit != _edges.end()) {
399 
400  if (eit->second.fcounter[side] < 2) {
401  // if (side == 0)
402  // _builder.addSingleTriangle(facet._aclPoints[0], facet._aclPoints[1],
403  // facet._aclPoints[2], true, 3, 0, 1, 1);
404 
405  eit->second.facet[side] = fidx;
406  eit->second.facets[side][eit->second.fcounter[side]] = facet;
407  eit->second.fcounter[side]++;
408  facet.SetFlag(
409  MeshFacet::MARKED); // set all facets connected to an edge: MARKED
410  }
411  }
412  }
413 
414  _newMeshFacets[side].push_back(facet);
415  }
416  }

670  if (ret == 2) {

733 void MeshIntersection::connectLines(bool onlyclosed,
734  const std::list<MeshIntersection::Tuple>& rdata,
735  std::list<std::list<MeshIntersection::Triple>>& lines)
736 {
737  float fMinEps = minDistance * minDistance;
738 
739  std::list<Tuple> data = rdata;
740  while (!data.empty()) {
741  std::list<Tuple>::iterator pF;
742  std::list<Triple> newPoly;
743 
744  // add first line and delete from the list
745  Triple front, back;
746  front.f1 = data.begin()->f1;
747  front.f2 = data.begin()->f2;
748  front.p = data.begin()->p1; // current start point of the polyline
749  back.f1 = data.begin()->f1;
750  back.f2 = data.begin()->f2;
751  back.p = data.begin()->p2; // current end point of the polyline
752  newPoly.push_back(front);
753  newPoly.push_back(back);
754  data.erase(data.begin());
755 
756  // search for the next line on the begin/end of the polyline and add it
757  std::list<Tuple>::iterator pFront, pEnd;
758  bool bFoundLine {};
759  do {
760  float fFrontMin = fMinEps, fEndMin = fMinEps;
761  bool bFrontFirst = false, bEndFirst = false;
762 
763  pFront = data.end();
764  pEnd = data.end();
765  bFoundLine = false;
766 
767  for (pF = data.begin(); pF != data.end(); ++pF) {
768  if (Base::DistanceP2(front.p, pF->p1) < fFrontMin) {
769  fFrontMin = Base::DistanceP2(front.p, pF->p1);
770  pFront = pF;
771  bFrontFirst = true;
772  }
773  else if (Base::DistanceP2(back.p, pF->p1) < fEndMin) {
774  fEndMin = Base::DistanceP2(back.p, pF->p1);
775  pEnd = pF;
776  bEndFirst = true;
777  }
778  else if (Base::DistanceP2(front.p, pF->p2) < fFrontMin) {
779  fFrontMin = Base::DistanceP2(front.p, pF->p2);
780  pFront = pF;
781  bFrontFirst = false;
782  }
783  else if (Base::DistanceP2(back.p, pF->p2) < fEndMin) {
784  fEndMin = Base::DistanceP2(back.p, pF->p2);
785  pEnd = pF;
786  bEndFirst = false;
787  }
788 
789  if (fFrontMin == 0.0F || fEndMin == 0.0F) {
790  break;
791  }
792  }
793 
794  if (pFront != data.end()) {
795  bFoundLine = true;
796  if (bFrontFirst) {
797  front.f1 = pFront->f1;
798  front.f2 = pFront->f2;
799  front.p = pFront->p2;
800  newPoly.push_front(front);
801  }
802  else {
803  front.f1 = pFront->f1;
804  front.f2 = pFront->f2;
805  front.p = pFront->p1;
806  newPoly.push_front(front);
807  }
808 
809  data.erase(pFront);
810  }
811 
812  if (pEnd != data.end()) {
813  bFoundLine = true;
814  if (bEndFirst) {
815  back.f1 = pEnd->f1;
816  back.f2 = pEnd->f2;
817  back.p = pEnd->p2;
818  newPoly.push_back(back);
819  }
820  else {
821  back.f1 = pEnd->f1;
822  back.f2 = pEnd->f2;
823  back.p = pEnd->p1;
824  newPoly.push_back(back);
825  }
826 
827  data.erase(pEnd);
828  }
829  } while (bFoundLine);
830 
831  if (onlyclosed) {
832  if (newPoly.size() > 2
833  && Base::DistanceP2(newPoly.front().p, newPoly.back().p) < fMinEps) {
834  lines.push_back(newPoly);
835  }
836  }
837  else {

839  }
