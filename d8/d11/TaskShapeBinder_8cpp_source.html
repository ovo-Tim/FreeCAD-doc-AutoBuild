188 void TaskShapeBinder::deleteItem()
189 {
190  if (vp.expired())
191  return;
192 
193  // Delete the selected spine
194  int row = ui->listWidgetReferences->currentRow();
195  QListWidgetItem* item = ui->listWidgetReferences->takeItem(row);
196  if (item) {
197  QByteArray data = item->text().toLatin1();
198  delete item;
199 
200  // search inside the list of sub-elements
201  App::GeoFeature* obj = nullptr;
202  std::vector<std::string> subs;
203 
204  PartDesign::ShapeBinder* binder = vp->getObject<PartDesign::ShapeBinder>();
205  PartDesign::ShapeBinder::getFilteredReferences(&binder->Support, obj, subs);
206 
207  std::string subname = data.constData();
208  std::vector<std::string>::iterator it = std::find(subs.begin(), subs.end(), subname);
209 
210  // if something was found, delete it and update the support
211  if (it != subs.end()) {
212  subs.erase(it);
213  binder->Support.setValue(obj, subs);
214 
215  vp->highlightReferences(false);
216  vp->getObject()->getDocument()->recomputeFeature(vp->getObject());
217 
218  clearButtons();
219  }
220  }

234 void TaskShapeBinder::onSelectionChanged(const Gui::SelectionChanges& msg)
235 {
236  auto setObjectLabel = [this](const Gui::SelectionChanges& msg) {
237  App::DocumentObject* obj = msg.Object.getObject();
238  if (obj) {
239  ui->baseEdit->setText(QString::fromStdString(obj->Label.getStrValue()));
240  }
241  };
242 
243  if (selectionMode == none)
244  return;
245 
246  if (msg.Type == Gui::SelectionChanges::AddSelection) {
247  if (referenceSelected(msg)) {
248  if (selectionMode == refAdd) {
249  QString sub = QString::fromUtf8(msg.pSubName);
250  if (!sub.isEmpty())
251  ui->listWidgetReferences->addItem(sub);
252 
253  setObjectLabel(msg);
254  }
255  else if (selectionMode == refRemove) {
256  QString sub = QString::fromUtf8(msg.pSubName);
257  if (!sub.isEmpty())
258  removeFromListWidget(ui->listWidgetReferences, sub);
259  }
260  else if (selectionMode == refObjAdd) {
261  ui->listWidgetReferences->clear();
262  setObjectLabel(msg);
263  }
264 
265  clearButtons();
266 
267  if (!vp.expired()) {
268  vp->highlightReferences(false);
269  vp->getObject()->getDocument()->recomputeFeature(vp->getObject());
270  }
271  }
272 
273  clearButtons();
274  exitSelectionMode();
275  }

278 bool TaskShapeBinder::referenceSelected(const SelectionChanges& msg) const
279 {
280  if (vp.expired())
281  return false;
282 
283  if ((msg.Type == Gui::SelectionChanges::AddSelection) && (
284  (selectionMode == refAdd) || (selectionMode == refRemove) || (selectionMode == refObjAdd))) {
285 
286  if (strcmp(msg.pDocName, vp->getObject()->getDocument()->getName()) != 0)
287  return false;
288 
289  // not allowed to reference ourself
290  const char* fname = vp->getObject()->getNameInDocument();
291  if (strcmp(msg.pObjectName, fname) == 0)
292  return false;
293 
294  //change the references
295  std::string subName(msg.pSubName);
296 
297  Part::Feature* selectedObj = nullptr;
298  App::GeoFeature* obj = nullptr;
299  std::vector<std::string> refs;
300 
301  PartDesign::ShapeBinder::getFilteredReferences(&vp->getObject<PartDesign::ShapeBinder>()->Support, obj, refs);
302 
303  // get selected object
304  auto docObj = vp->getObject()->getDocument()->getObject(msg.pObjectName);
305  if (docObj && docObj->isDerivedFrom(Part::Feature::getClassTypeId())) {
306  selectedObj = static_cast<Part::Feature*>(docObj);
307  }
308 
309  // ensure we have a valid object
310  if (!selectedObj) {
311  return false;
312  }
313  if (!obj) {
314  // Support has not been set before
315  obj = selectedObj;
316  }
317 
318  if (selectionMode != refObjAdd) {
319  // ensure the new selected subref belongs to the same object
320  if (strcmp(msg.pObjectName, obj->getNameInDocument()) != 0)
321  return false;
322 
323  std::vector<std::string>::iterator f = std::find(refs.begin(), refs.end(), subName);
324 
325  if (selectionMode == refAdd) {
326  if (f == refs.end())
327  refs.push_back(subName);
328  else
329  return false; // duplicate selection
330  }
331  else {
332  if (f != refs.end())
333  refs.erase(f);
334  else
335  return false;
336  }
337  }
338  else {
339  // change object
340  refs.clear();
341  obj = selectedObj;
342  }
343 
344  vp->getObject<PartDesign::ShapeBinder>()->Support.setValue(obj, refs);
345 
346  return true;
347  }
348 
349  return false;
350 }

365 void TaskShapeBinder::accept()
366 {
367  if (vp.expired())
368  return;
369 
370  std::string label = ui->baseEdit->text().toStdString();
371  PartDesign::ShapeBinder* binder = vp->getObject<PartDesign::ShapeBinder>();
372  if (!binder->Support.getValue() && !label.empty()) {
373  auto mode = selectionMode;
374  selectionMode = refObjAdd;
375  SelectionChanges msg(SelectionChanges::AddSelection, binder->getDocument()->getName(), label.c_str());
376  referenceSelected(msg);
377  selectionMode = mode;
378  }
