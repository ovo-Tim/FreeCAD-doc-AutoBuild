91 QIcon ViewProviderFeaturePythonImp::getIcon() const
92 {
93  _FC_PY_CALL_CHECK(getIcon,return(QIcon()));
94 
95  // Run the getIcon method of the proxy object.
96  Base::PyGILStateLocker lock;
97  try {
98  Py::Object ret(Base::pyCall(py_getIcon.ptr()));
99  if(ret.isNone())
100  return {};
101 
102  if(ret.isString()) {
103  std::string content = Py::String(ret).as_std_string("utf-8");
104  QPixmap icon;
105  if (BitmapFactory().findPixmapInCache(content.c_str(), icon))
106  return icon;
107 
108  // Check if the passed string is a filename, otherwise treat as xpm data
109  QFileInfo fi(QString::fromUtf8(content.c_str()));
110  if (fi.isFile() && fi.exists()) {
111  icon.load(fi.absoluteFilePath());
112  } else {
113  QByteArray ary;
114  int strlen = (int)content.size();
115  ary.resize(strlen);
116  for (int j=0; j<strlen; j++)
117  ary[j]=content[j];
118  // Make sure to remove crap around the XPM data
119  QList<QByteArray> lines = ary.split('\n');
120  QByteArray buffer;
121  buffer.reserve(ary.size()+lines.size());
122  for (const auto & line : lines) {
123  QByteArray trim = line.trimmed();
124  if (!trim.isEmpty()) {
125  buffer.append(trim);
126  buffer.append('\n');
127  }
128  }
129  icon.loadFromData(buffer, "XPM");
130  }
131  if (!icon.isNull()) {
132  return icon;
133  }
134  } else {
135  PythonWrapper wrap;
136  wrap.loadGuiModule();
137  wrap.loadWidgetsModule();
138  QIcon *picon = wrap.toQIcon(ret.ptr());
139  if(picon)
140  return *picon;
141  }
142  }
143  catch (Py::Exception&) {
144  if (PyErr_ExceptionMatches(PyExc_NotImplementedError))
145  PyErr_Clear();
146  else {
147  Base::PyException e; // extract the Python error text
148  e.ReportException();
149  }
150  }
151 
152  return {};
153 }
