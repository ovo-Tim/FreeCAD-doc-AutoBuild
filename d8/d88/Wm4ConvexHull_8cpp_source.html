80 {
81  WM4_DELETE[] m_aiIndex;
82 
83  // fixed-size members
84  int iQueryType;

90  System::Read4le(pkIFile,1,&m_fEpsilon);
91 
92  // variable-size members
93  int iIQuantity;
94  System::Read4le(pkIFile,1,&iIQuantity);
95  if (1 <= m_iDimension && m_iDimension <= 3)
96  {
97  assert(iIQuantity == (m_iDimension+1)*m_iSimplexQuantity);
98  m_aiIndex = WM4_NEW int[iIQuantity];
99  System::Read4le(pkIFile,iIQuantity,m_aiIndex);
100  return true;
101  }
102 
103  m_aiIndex = nullptr;
104  return m_iDimension == 0;
105 }

122  int iIQuantity;

124  {
125  iIQuantity = (m_iDimension+1)*m_iSimplexQuantity;
126  System::Write4le(pkOFile,1,&iIQuantity);
127  System::Write4le(pkOFile,iIQuantity,m_aiIndex);
128  return true;
