381 void TaskFemConstraintFluidBoundary::updateBoundaryTypeUI()
382 {
383  Fem::ConstraintFluidBoundary* pcConstraint =
384  ConstraintView->getObject<Fem::ConstraintFluidBoundary>();
385  std::string boundaryType = ui->comboBoundaryType->currentText().toStdString();
386  // std::string boundaryType = pcConstraint->BoundaryType.getValueAsString();
387 
388  // Update subtypes, any change here should be written back to FemConstraintFluidBoundary.cpp
389  if (boundaryType == "wall") {
390  ui->labelBoundaryValue->setText(QString::fromUtf8("velocity (m/s)"));
391  ui->tabBasicBoundary->setEnabled(false);
392  pcConstraint->Subtype.setEnums(WallSubtypes);
393  }
394  else if (boundaryType == "interface") {
395  ui->labelBoundaryValue->setText(QString::fromUtf8("value not needed"));
396  ui->tabBasicBoundary->setEnabled(false);
397  pcConstraint->Subtype.setEnums(InterfaceSubtypes);
398  }
399  else if (boundaryType == "freestream") {
400  ui->tabBasicBoundary->setEnabled(false);
401  ui->labelBoundaryValue->setText(QString::fromUtf8("value not needed"));
402  ui->tabBasicBoundary->setEnabled(false);
403  pcConstraint->Subtype.setEnums(FreestreamSubtypes);
404  }
405  else if (boundaryType == "inlet") {
406  ui->tabBasicBoundary->setEnabled(true);
407  pcConstraint->Subtype.setEnums(InletSubtypes);
408  ui->labelBoundaryValue->setText(QString::fromUtf8("Pressure [Pa]")); // default to pressure
409  pcConstraint->Reversed.setValue(true); // inlet must point into volume
410  }
411  else if (boundaryType == "outlet") {
412  ui->tabBasicBoundary->setEnabled(true);
413  pcConstraint->Subtype.setEnums(OutletSubtypes);
414  ui->labelBoundaryValue->setText(QString::fromUtf8("Pressure [Pa]"));
415  pcConstraint->Reversed.setValue(false); // outlet must point outward
416  }
417  else {
418  Base::Console().Error("Error: Fluid boundary type `%s` is not defined\n",
419  boundaryType.c_str());
420  }
421  // std::string subtypeLabel = boundaryType + std::string(" type");
422  // ui->labelSubtype->setText(QString::fromUtf8(subtypeLabel)); // too long to show in UI
423  ui->tabWidget->setCurrentIndex(0); // activate the basic pressure-momentum setting tab
424 
425  std::vector<std::string> subtypes = pcConstraint->Subtype.getEnumVector();
426  initComboBox(ui->comboSubtype, subtypes, "default to the second subtype");
427  updateSubtypeUI();
428 }

814 void TaskFemConstraintFluidBoundary::addToSelection()
815 {
816  std::vector<Gui::SelectionObject> selection =
817  Gui::Selection().getSelectionEx(); // gets vector of selected objects of active document
818  if (selection.empty()) {
819  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
820  return;
821  }
822  Fem::ConstraintFluidBoundary* pcConstraint =
823  ConstraintView->getObject<Fem::ConstraintFluidBoundary>();
824  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
825  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
826 
827  for (auto& it : selection) { // for every selected object
828  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
829  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
830  return;
831  }
832  const std::vector<std::string>& subNames = it.getSubNames();
833  App::DocumentObject* obj = it.getObject();
834  for (const auto& subName : subNames) { // for every selected sub element
835  bool addMe = true;
836  for (std::vector<std::string>::iterator itr =
837  std::find(SubElements.begin(), SubElements.end(), subName);
838  itr != SubElements.end();
839  itr = std::find(++itr,
840  SubElements.end(),
841  subName)) { // for every sub element in selection that
842  // matches one in old list
843  if (obj
844  == Objects[std::distance(
845  SubElements.begin(),
846  itr)]) { // if selected sub element's object equals the one in old list
847  // then it was added before so don't add
848  addMe = false;
849  }
850  }
851  // limit constraint such that only vertexes or faces or edges can be used depending on
852  // what was selected first
853  std::string searchStr;
854  if (subName.find("Vertex") != std::string::npos) {
855  searchStr = "Vertex";
856  }
857  else if (subName.find("Edge") != std::string::npos) {
858  searchStr = "Edge";
859  }
860  else {
861  searchStr = "Face";
862  }
863 
864  for (const auto& SubElement : SubElements) {
865  if (SubElement.find(searchStr) == std::string::npos) {
866  QString msg = tr("Only one type of selection (vertex, face or edge) per "
867  "analysis feature allowed!");
868  QMessageBox::warning(this, tr("Selection error"), msg);
869  addMe = false;
870  break;
871  }
872  }
873  if (addMe) {
874  QSignalBlocker block(ui->listReferences);
875  Objects.push_back(obj);
876  SubElements.push_back(subName);
877  ui->listReferences->addItem(makeRefText(obj, subName));
878  }
879  }
880  }
881  // Update UI
882  pcConstraint->References.setValues(Objects, SubElements);
883  updateUI();
884 }

886 void TaskFemConstraintFluidBoundary::removeFromSelection()
887 {
888  std::vector<Gui::SelectionObject> selection =
889  Gui::Selection().getSelectionEx(); // gets vector of selected objects of active document
890  if (selection.empty()) {
891  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
892  return;
893  }
894  Fem::ConstraintFluidBoundary* pcConstraint =
895  ConstraintView->getObject<Fem::ConstraintFluidBoundary>();
896  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
897  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
898  std::vector<size_t> itemsToDel;
899  for (const auto& it : selection) { // for every selected object
900  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
901  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
902  return;
903  }
904  const std::vector<std::string>& subNames = it.getSubNames();
905  const App::DocumentObject* obj = it.getObject();
906 
907  for (const auto& subName : subNames) { // for every selected sub element
908  for (std::vector<std::string>::iterator itr =
909  std::find(SubElements.begin(), SubElements.end(), subName);
910  itr != SubElements.end();
911  itr = std::find(++itr,
912  SubElements.end(),
913  subName)) { // for every sub element in selection that
914  // matches one in old list
915  if (obj
916  == Objects[std::distance(
917  SubElements.begin(),
918  itr)]) { // if selected sub element's object equals the one in old list
919  // then it was added before so mark for deletion
920  itemsToDel.push_back(std::distance(SubElements.begin(), itr));
921  }
922  }
923  }
924  }
925  std::sort(itemsToDel.begin(), itemsToDel.end());
926  while (!itemsToDel.empty()) {
927  Objects.erase(Objects.begin() + itemsToDel.back());
928  SubElements.erase(SubElements.begin() + itemsToDel.back());
929  itemsToDel.pop_back();
930  }
931  // Update UI
932  {
933  QSignalBlocker block(ui->listReferences);
934  ui->listReferences->clear();
935  for (size_t j = 0; j < Objects.size(); j++) {
936  ui->listReferences->addItem(makeRefText(Objects[j], SubElements[j]));
937  }
938  }
939  pcConstraint->References.setValues(Objects, SubElements);
940  updateUI();
941 }
