221 TaskWatcherPython::TaskWatcherPython(const Py::Object& o)
222  : TaskWatcher(nullptr), watcher(o)
223 {
224  QString title;
225  if (watcher.hasAttr(std::string("title"))) {
226  Py::String name(watcher.getAttr(std::string("title")));
227  std::string s = static_cast<std::string>(name);
228  title = QString::fromUtf8(s.c_str());
229  }
230 
231  QPixmap icon;
232  if (watcher.hasAttr(std::string("icon"))) {
233  Py::String name(watcher.getAttr(std::string("icon")));
234  std::string s = static_cast<std::string>(name);
235  icon = BitmapFactory().pixmap(s.c_str());
236  }
237 
238  Gui::TaskView::TaskBox *tb = nullptr;
239  if (watcher.hasAttr(std::string("commands"))) {
240  tb = new Gui::TaskView::TaskBox(icon, title, true, nullptr);
241  Py::Sequence cmds(watcher.getAttr(std::string("commands")));
242  CommandManager &mgr = Gui::Application::Instance->commandManager();
243  for (Py::Sequence::iterator it = cmds.begin(); it != cmds.end(); ++it) {
244  Py::String name(*it);
245  std::string s = static_cast<std::string>(name);
246  Command *c = mgr.getCommandByName(s.c_str());
247  if (c)
248  c->addTo(tb);
249  }
250  }
251 
252  if (watcher.hasAttr(std::string("widgets"))) {
253  if (!tb && !title.isEmpty())
254  tb = new Gui::TaskView::TaskBox(icon, title, true, nullptr);
255  Py::Sequence list(watcher.getAttr(std::string("widgets")));
256 
257  Gui::PythonWrapper wrap;
258  if (wrap.loadCoreModule()) {
259  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
260  QObject* object = wrap.toQObject(*it);
261  if (object) {
262  QWidget* w = qobject_cast<QWidget*>(object);
263  if (w) {
264  if (tb)
265  tb->groupLayout()->addWidget(w);
266  else
267  Content.push_back(w);
268  }
269  }
270  }
271  }
272  }
273 
274  if (tb) Content.push_back(tb);
275 
276  if (watcher.hasAttr(std::string("filter"))) {
277  Py::String name(watcher.getAttr(std::string("filter")));
278  std::string s = static_cast<std::string>(name);
279  this->setFilter(s.c_str());
280  }
281 }

567 bool TaskDialogPython::tryLoadUiFile()
568 {
569  if (dlg.hasAttr(std::string("ui"))) {
570  auto loader = UiLoader::newInstance();
571  QString fn, icon;
572  Py::String ui(dlg.getAttr(std::string("ui")));
573  std::string path = static_cast<std::string>(ui);
574  fn = QString::fromUtf8(path.c_str());
575 
576  QFile file(fn);
577  QWidget* form = nullptr;
578  if (file.open(QFile::ReadOnly))
579  form = loader->load(&file, nullptr);
580  file.close();
581  if (form) {
582  appendForm(form, QPixmap(icon));
583  }
584  else {
585  Base::Console().Error("Failed to load UI file from '%s'\n",
586  (const char*)fn.toUtf8());
587  }
588 
589  return true;
590  }
591 
592  return false;
593 }

595 bool TaskDialogPython::tryLoadForm()
596 {
597  if (dlg.hasAttr(std::string("form"))) {
598  Py::Object f(dlg.getAttr(std::string("form")));
599  Py::List widgets;
600  if (f.isList()) {
601  widgets = f;
602  }
603  else {
604  widgets.append(f);
605  }
606 
607  Gui::PythonWrapper wrap;
608  if (wrap.loadCoreModule()) {
609  for (Py::List::iterator it = widgets.begin(); it != widgets.end(); ++it) {
610  QObject* object = wrap.toQObject(*it);
611  if (object) {
612  QWidget* form = qobject_cast<QWidget*>(object);
613  if (form) {
614  appendForm(form, form->windowIcon().pixmap(32));
615  }
616  }
617  }
618  }
619 
620  return true;
621  }
622 
623  return false;
624 }
