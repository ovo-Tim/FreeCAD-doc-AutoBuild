95 void FeaturePickDialog::updateList()
96 {
97  int index = 0;
98 
99  for (std::vector<featureStatus>::const_iterator st = statuses.begin(); st != statuses.end(); st++) {
100  QListWidgetItem* item = ui->listWidget->item(index);
101 
102  switch (*st) {
103  case validFeature: item->setFlags(Qt::ItemIsSelectable | Qt::ItemIsEnabled); break;
104  case invalidShape: item->setFlags(Qt::NoItemFlags); break;
105  case noWire: item->setFlags(Qt::NoItemFlags); break;
106  case isUsed: item->setFlags(ui->checkOtherFeature->isChecked() ? Qt::ItemIsSelectable | Qt::ItemIsEnabled : Qt::NoItemFlags); break;
107  case otherBody: item->setFlags(ui->checkOtherBody->isChecked() ? Qt::ItemIsSelectable | Qt::ItemIsEnabled : Qt::NoItemFlags); break;
108  case basePlane: item->setFlags(Qt::ItemIsSelectable | Qt::ItemIsEnabled); break;
