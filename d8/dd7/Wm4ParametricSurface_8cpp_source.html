90 template <class Real>
91 void ParametricSurface<Real>::ComputePrincipalCurvatureInfo (Real fU, Real fV,
92  Real& rfCurv0, Real& rfCurv1, Vector3<Real>& rkDir0,
93  Vector3<Real>& rkDir1)
94 {
95  // Tangents: T0 = (x_u,y_u,z_u), T1 = (x_v,y_v,z_v)
96  // Normal: N = Cross(T0,T1)/Length(Cross(T0,T1))
97  // Metric Tensor: G = +- -+
98  // | Dot(T0,T0) Dot(T0,T1) |
99  // | Dot(T1,T0) Dot(T1,T1) |
100  // +- -+
101  //
102  // Curvature Tensor: B = +- -+
103  // | -Dot(N,T0_u) -Dot(N,T0_v) |
104  // | -Dot(N,T1_u) -Dot(N,T1_v) |
105  // +- -+
106  //
107  // Principal curvatures k are the generalized eigenvalues of
108  //
109  // Bw = kGw
110  //
111  // If k is a curvature and w=(a,b) is the corresponding solution to
112  // Bw = kGw, then the principal direction as a 3D vector is d = a*U+b*V.
113  //
114  // Let k1 and k2 be the principal curvatures. The mean curvature
115  // is (k1+k2)/2 and the Gaussian curvature is k1*k2.
116 
117  // derivatives
118  Vector3<Real> kDerU = PU(fU,fV);
119  Vector3<Real> kDerV = PV(fU,fV);
120  Vector3<Real> kDerUU = PUU(fU,fV);
121  Vector3<Real> kDerUV = PUV(fU,fV);
122  Vector3<Real> kDerVV = PVV(fU,fV);
123 
124  // metric tensor
125  Matrix2<Real> kMetricTensor;
126  kMetricTensor[0][0] = kDerU.Dot(kDerU);
127  kMetricTensor[0][1] = kDerU.Dot(kDerV);
128  kMetricTensor[1][0] = kMetricTensor[0][1];
129  kMetricTensor[1][1] = kDerV.Dot(kDerV);
130 
131  // curvature tensor
132  Vector3<Real> kNormal = kDerU.UnitCross(kDerV);
133  Matrix2<Real> kCurvatureTensor;
134  kCurvatureTensor[0][0] = -kNormal.Dot(kDerUU);
135  kCurvatureTensor[0][1] = -kNormal.Dot(kDerUV);
136  kCurvatureTensor[1][0] = kCurvatureTensor[0][1];
137  kCurvatureTensor[1][1] = -kNormal.Dot(kDerVV);
138 
139  // characteristic polynomial is 0 = det(B-kG) = c2*k^2+c1*k+c0
140  Real fC0 = kCurvatureTensor.Determinant();
141  Real fC1 = ((Real)2.0)*kCurvatureTensor[0][1]* kMetricTensor[0][1] -
142  kCurvatureTensor[0][0]*kMetricTensor[1][1] -
143  kCurvatureTensor[1][1]*kMetricTensor[0][0];
144  Real fC2 = kMetricTensor.Determinant();
145 
146  // principal curvatures are roots of characteristic polynomial
147  Real fTemp = Math<Real>::Sqrt(Math<Real>::FAbs(fC1*fC1 -
148  ((Real)4.0)*fC0*fC2));
149  Real fMult = ((Real)0.5)/fC2;
150  rfCurv0 = -fMult*(fC1+fTemp);
151  rfCurv1 = fMult*(-fC1+fTemp);
152 
153  // principal directions are solutions to (B-kG)w = 0
154  // w1 = (b12-k1*g12,-(b11-k1*g11)) OR (b22-k1*g22,-(b12-k1*g12))
155  Real fA0 = kCurvatureTensor[0][1] - rfCurv0*kMetricTensor[0][1];
156  Real fA1 = rfCurv0*kMetricTensor[0][0] - kCurvatureTensor[0][0];
157  Real fLength = Math<Real>::Sqrt(fA0*fA0+fA1*fA1);
158  if (fLength >= Math<Real>::ZERO_TOLERANCE)
159  {
160  rkDir0 = fA0*kDerU + fA1*kDerV;
161  }
162  else
163  {
164  fA0 = kCurvatureTensor[1][1] - rfCurv0*kMetricTensor[1][1];
165  fA1 = rfCurv0*kMetricTensor[0][1] - kCurvatureTensor[0][1];
166  fLength = Math<Real>::Sqrt(fA0*fA0+fA1*fA1);
167  if (fLength >= Math<Real>::ZERO_TOLERANCE)
168  {
169  rkDir0 = fA0*kDerU + fA1*kDerV;
170  }
171  else
172  {
173  // umbilic (surface is locally sphere, any direction principal)
174  rkDir0 = kDerU;
175  }
176  }
177  rkDir0.Normalize();
178 
179  // second tangent is cross product of first tangent and normal
180  rkDir1 = rkDir0.Cross(kNormal);
181 }
