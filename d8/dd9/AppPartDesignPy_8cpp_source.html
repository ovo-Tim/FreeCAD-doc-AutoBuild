44  Py::Object makeFilletArc(const Py::Tuple& args)
45  {
46  PyObject *pM1;
47  PyObject *pP;
48  PyObject *pQ;
49  PyObject *pN;
50  double r2;
51  int ccw;
52  if (!PyArg_ParseTuple(args.ptr(), "O!O!O!O!di",
53  &(Base::VectorPy::Type), &pM1,
54  &(Base::VectorPy::Type), &pP,
55  &(Base::VectorPy::Type), &pQ,
56  &(Base::VectorPy::Type), &pN,
57  &r2, &ccw))
58  throw Py::Exception();
59 
60  Base::Vector3d M1 = Py::Vector(pM1, false).toVector();
61  Base::Vector3d P = Py::Vector(pP, false).toVector();
62  Base::Vector3d Q = Py::Vector(pQ, false).toVector();
63  Base::Vector3d N = Py::Vector(pN, false).toVector();
64 
65  Base::Vector3d u = Q - P;
66  Base::Vector3d v = P - M1;
67  Base::Vector3d b;
68  if (ccw)
69  b = u % N;
70  else
71  b = N % u;
72  b.Normalize();
73 
74  double uu = u * u;
75  double uv = u * v;
76  double r1 = v.Length();
77 
78  // distinguish between internal and external fillets
79  r2 *= Base::sgn(uv);
80 
81  double cc = 2.0 * r2 * (b * v - r1);
82  double d = uv * uv - uu * cc;
83  if (d < 0) {
84  throw Py::RuntimeError("Unable to calculate intersection points");
85  }
86 
87  double t;
88  double t1 = (-uv + sqrt(d)) / uu;
89  double t2 = (-uv - sqrt(d)) / uu;
90 
91  if (fabs(t1) < fabs(t2))
92  t = t1;
93  else
94  t = t2;
95 
96  Base::Vector3d M2 = P + (u*t) + (b*r2);
97  Base::Vector3d S1 = (r2 * M1 + r1 * M2)/(r1+r2);
98  Base::Vector3d S2 = M2 - (b*r2);
99 
100  Py::Tuple tuple(3);
101  tuple.setItem(0, Py::Vector(S1));
102  tuple.setItem(1, Py::Vector(S2));
103  tuple.setItem(2, Py::Vector(M2));
104 
105  return tuple;
106  }
