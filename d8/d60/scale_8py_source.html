40 def scale(objectslist, scale=App.Vector(1,1,1),
41  center=App.Vector(0,0,0), copy=False):
42  """scale(objects, scale, [center], copy)
43 
44  Scales the objects contained in objects (that can be a list of objects or
45  an object) of the given around given center.
46 
47  Parameters
48  ----------
49  objectlist : list
50 
51  scale : Base.Vector
52  Scale factors defined by a given vector (in X, Y, Z directions).
53 
54  objectlist : Base.Vector
55  Center of the scale operation.
56 
57  copy : bool
58  If copy is True, the actual objects are not scaled, but copies
59  are created instead.
60 
61  Return
62  ----------
63  The objects (or their copies) are returned.
64  """
65  if not isinstance(objectslist, list):
66  objectslist = [objectslist]
67  newobjlist = []
68  for obj in objectslist:
69  if copy:
70  newobj = make_copy.make_copy(obj)
71  else:
72  newobj = obj
73  if hasattr(obj,'Shape'):
74  scaled_shape = obj.Shape.copy()
75  m = App.Matrix()
76  m.move(center.negative())
77  m.scale(scale.x,scale.y,scale.z)
78  m.move(center)
79  scaled_shape = scaled_shape.transformGeometry(m)
80  if utils.get_type(obj) == "Rectangle":
81  p = []
82  for v in scaled_shape.Vertexes:
83  p.append(v.Point)
84  pl = obj.Placement.copy()
85  pl.Base = p[0]
86  diag = p[2].sub(p[0])
87  bb = p[1].sub(p[0])
88  bh = p[3].sub(p[0])
89  nb = DraftVecUtils.project(diag,bb)
90  nh = DraftVecUtils.project(diag,bh)
91  if obj.Length < 0: l = -nb.Length
92  else: l = nb.Length
93  if obj.Height < 0: h = -nh.Length
94  else: h = nh.Length
95  newobj.Length = l
96  newobj.Height = h
97  tr = p[0].sub(obj.Shape.Vertexes[0].Point) # unused?
98  newobj.Placement = pl
99  elif utils.get_type(obj) == "Wire" or utils.get_type(obj) == "BSpline":
100  for index, point in enumerate(newobj.Points):
101  scale_vertex(newobj, index, scale, center)
102  elif hasattr(obj,'Shape'):
103  newobj.Shape = scaled_shape
104  elif hasattr(obj,"Position"):
105  d = obj.Position.sub(center)
106  newobj.Position = center.add(App.Vector(d.x * scale.x,
107  d.y * scale.y,
108  d.z * scale.z))
109  elif hasattr(obj,"Placement"):
110  d = obj.Placement.Base.sub(center)
111  newobj.Placement.Base = center.add(App.Vector(d.x * scale.x,
112  d.y * scale.y,
113  d.z * scale.z))
114  if hasattr(obj,"Height"):
115  obj.setExpression('Height', None)
116  obj.Height = obj.Height * scale.y
117  if hasattr(obj,"Width"):
118  obj.setExpression('Width', None)
119  obj.Width = obj.Width * scale.x
120  if hasattr(obj,"XSize"):
121  obj.setExpression('XSize', None)
122  obj.XSize = obj.XSize * scale.x
123  if hasattr(obj,"YSize"):
124  obj.setExpression('YSize', None)
125  obj.YSize = obj.YSize * scale.y
126  if obj.ViewObject and hasattr(obj.ViewObject,"FontSize"):
127  obj.ViewObject.FontSize = obj.ViewObject.FontSize * scale.y
128 
129 
130  if copy:
131  gui_utils.format_object(newobj,obj)
132  newobjlist.append(newobj)
133  if copy and params.get_param("selectBaseObjects"):
134  gui_utils.select(objectslist)
135  else:
136  gui_utils.select(newobjlist)
137  if len(newobjlist) == 1: return newobjlist[0]
138  return newobjlist
