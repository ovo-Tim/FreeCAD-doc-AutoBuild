295 void DlgDisplayPropertiesImp::slotChangedObject(const Gui::ViewProvider& obj,
296  const App::Property& prop)
297 {
298  // This method gets called if a property of any view provider is changed.
299  // We pick out all the properties for which we need to update this dialog.
300  std::vector<Gui::ViewProvider*> Provider = getSelection();
301  auto vp = std::find_if(Provider.begin(), Provider.end(), [&obj](Gui::ViewProvider* v) {
302  return v == &obj;
303  });
304 
305  if (vp != Provider.end()) {
306  const char* name = obj.getPropertyName(&prop);
307  // this is not a property of the view provider but of the document object
308  if (!name) {
309  return;
310  }
311  std::string prop_name = name;
312  if (prop.is<App::PropertyColor>()) {
313  App::Color value = static_cast<const App::PropertyColor&>(prop).getValue();
314  if (prop_name == "LineColor") {
315  bool blocked = d->ui.buttonLineColor->blockSignals(true);
316  d->ui.buttonLineColor->setColor(value.asValue<QColor>());
317  d->ui.buttonLineColor->blockSignals(blocked);
318  }
319  else if (prop_name == "PointColor") {
320  bool blocked = d->ui.buttonPointColor->blockSignals(true);
321  d->ui.buttonPointColor->setColor(value.asValue<QColor>());
322  d->ui.buttonPointColor->blockSignals(blocked);
323  }
324  }
325  else if (prop.isDerivedFrom<App::PropertyMaterialList>()) {
326  if (prop_name == "ShapeAppearance") {
327  auto& values = static_cast<const App::PropertyMaterialList&>(prop).getValues();
328  auto& material = values[0];
329  d->ui.widgetMaterial->setMaterial(QString::fromStdString(material.uuid));
330  }
331  }
332  else if (prop.isDerivedFrom<App::PropertyInteger>()) {
333  long value = static_cast<const App::PropertyInteger&>(prop).getValue();
334  if (prop_name == "Transparency") {
335  bool blocked = d->ui.spinTransparency->blockSignals(true);
336  d->ui.spinTransparency->setValue(value);
337  d->ui.spinTransparency->blockSignals(blocked);
338  blocked = d->ui.horizontalSlider->blockSignals(true);
339  d->ui.horizontalSlider->setValue(value);
340  d->ui.horizontalSlider->blockSignals(blocked);
341  }
342  else if (prop_name == "LineTransparency") {
343  bool blocked = d->ui.spinLineTransparency->blockSignals(true);
344  d->ui.spinLineTransparency->setValue(value);
345  d->ui.spinLineTransparency->blockSignals(blocked);
346  blocked = d->ui.sliderLineTransparency->blockSignals(true);
347  d->ui.sliderLineTransparency->setValue(value);
348  d->ui.sliderLineTransparency->blockSignals(blocked);
349  }
350  }
351  else if (prop.isDerivedFrom<App::PropertyFloat>()) {
352  double value = static_cast<const App::PropertyFloat&>(prop).getValue();
353  if (prop_name == "PointSize") {
354  bool blocked = d->ui.spinPointSize->blockSignals(true);
355  d->ui.spinPointSize->setValue((int)value);
356  d->ui.spinPointSize->blockSignals(blocked);
357  }
358  else if (prop_name == "LineWidth") {
359  bool blocked = d->ui.spinLineWidth->blockSignals(true);
360  d->ui.spinLineWidth->setValue((int)value);
361  d->ui.spinLineWidth->blockSignals(blocked);
362  }
363  }
364  }

515 void DlgDisplayPropertiesImp::setDisplayModes(const std::vector<Gui::ViewProvider*>& views)
516 {
517  QStringList commonModes;
518  QStringList modes;
519  for (auto it = views.begin(); it != views.end(); ++it) {
520  if (auto* prop =
521  dynamic_cast<App::PropertyEnumeration*>((*it)->getPropertyByName("DisplayMode"))) {
522  if (!prop->hasEnums()) {
523  return;
524  }
525  std::vector<std::string> value = prop->getEnumVector();
526  if (it == views.begin()) {
527  for (const auto& jt : value) {
528  commonModes << QLatin1String(jt.c_str());
529  }
530  }
531  else {
532  for (const auto& jt : value) {
533  if (commonModes.contains(QLatin1String(jt.c_str()))) {
534  modes << QLatin1String(jt.c_str());
535  }
536  }
537 
538  commonModes = modes;
539  modes.clear();
540  }
541  }
542  }
543 
544  d->ui.changeMode->clear();
545  d->ui.changeMode->addItems(commonModes);
546  d->ui.changeMode->setDisabled(commonModes.isEmpty());
547 
548  // find the display mode to activate
549  for (const auto& view : views) {
550  if (auto* prop =
551  dynamic_cast<App::PropertyEnumeration*>(view->getPropertyByName("DisplayMode"))) {
552  QString activeMode = QString::fromLatin1(prop->getValueAsString());
553  int index = d->ui.changeMode->findText(activeMode);
554  if (index != -1) {
555  d->ui.changeMode->setCurrentIndex(index);
556  break;
557  }
558  }
559  }
560 }
