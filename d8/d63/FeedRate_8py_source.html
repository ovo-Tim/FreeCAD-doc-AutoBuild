44 def setFeedRate(commandlist, ToolController):
45  """Set the appropriate feed rate for a list of Path commands using the information from a Tool Controller
46 
47  Every motion command in the list will have a feed rate parameter added or overwritten based
48  on the information stored in the tool controller. If a motion is a plunge (vertical) motion, the
49  VertFeed value will be used, otherwise the HorizFeed value will be used instead."""
50 
51  def _isVertical(currentposition, command):
52  x = command.Parameters["X"] if "X" in command.Parameters else currentposition.x
53  y = command.Parameters["Y"] if "Y" in command.Parameters else currentposition.y
54  z = command.Parameters["Z"] if "Z" in command.Parameters else currentposition.z
55  endpoint = FreeCAD.Vector(x, y, z)
56  if Path.Geom.pointsCoincide(currentposition, endpoint):
57  return True
58  return Path.Geom.isVertical(Part.makeLine(currentposition, endpoint))
59 
60  machine = PathMachineState.MachineState()
61 
62  for command in commandlist:
63  if command.Name not in Path.Geom.CmdMoveAll:
64  continue
65 
66  if _isVertical(machine.getPosition(), command):
67  rate = (
68  ToolController.VertRapid.Value
69  if command.Name in Path.Geom.CmdMoveRapid
70  else ToolController.VertFeed.Value
71  )
72  else:
73  rate = (
74  ToolController.HorizRapid.Value
75  if command.Name in Path.Geom.CmdMoveRapid
76  else ToolController.HorizFeed.Value
77  )
78 
79  params = command.Parameters
80  params["F"] = rate
81  command.Parameters = params
82 
83  machine.addCommand(command)
84 
85  return commandlist
