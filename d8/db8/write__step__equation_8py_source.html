32 def write_step_equation(f, ccxwriter):
33 
34  f.write("\n{}\n".format(59 * "*"))
35  f.write("** At least one step is needed to run an CalculiX analysis of FreeCAD\n")
36 
37  # build STEP line
38  step = "*STEP"
39  if ccxwriter.solver_obj.GeometricalNonlinearity == "nonlinear":
40  if ccxwriter.analysis_type == "static" or ccxwriter.analysis_type == "thermomech":
41  # https://www.comsol.com/blogs/what-is-geometric-nonlinearity
42  step += ", NLGEOM"
43  elif ccxwriter.analysis_type == "frequency":
44  FreeCAD.Console.PrintMessage(
45  "Analysis type frequency and geometrical nonlinear "
46  "analysis are not allowed together, linear is used instead!\n"
47  )
48  if ccxwriter.solver_obj.IterationsMaximum:
49  if ccxwriter.analysis_type == "thermomech" or ccxwriter.analysis_type == "static":
50  step += f", INC={ccxwriter.solver_obj.IterationsMaximum}"
51  elif ccxwriter.analysis_type == "frequency" or ccxwriter.analysis_type == "buckling":
52  # parameter is for thermomechanical analysis only, see ccx manual *STEP
53  pass
54  # write STEP line
55  f.write(step + "\n")
56 
57  # CONTROLS line
58  # all analysis types, ... really in frequency too?!?
59  if ccxwriter.solver_obj.IterationsControlParameterTimeUse:
60  f.write("*CONTROLS, PARAMETERS=TIME INCREMENTATION\n")
61  f.write(ccxwriter.solver_obj.IterationsControlParameterIter + "\n")
62  f.write(ccxwriter.solver_obj.IterationsControlParameterCutb + "\n")
63 
64  # ANALYSIS type line
65  # analysis line --> analysis type
66  if ccxwriter.analysis_type == "static":
67  analysis_type = "*STATIC"
68  elif ccxwriter.analysis_type == "frequency":
69  analysis_type = "*FREQUENCY"
70  elif ccxwriter.analysis_type == "thermomech":
71  if ccxwriter.solver_obj.ThermoMechType == "coupled":
72  analysis_type = "*COUPLED TEMPERATURE-DISPLACEMENT"
73  elif ccxwriter.solver_obj.ThermoMechType == "uncoupled":
74  analysis_type = "*UNCOUPLED TEMPERATURE-DISPLACEMENT"
75  elif ccxwriter.solver_obj.ThermoMechType == "pure heat transfer":
76  analysis_type = "*HEAT TRANSFER"
77  elif ccxwriter.analysis_type == "check":
78  analysis_type = "*NO ANALYSIS"
79  elif ccxwriter.analysis_type == "buckling":
80  analysis_type = "*BUCKLE"
81  # analysis line --> solver type
82  # https://forum.freecad.org/viewtopic.php?f=18&t=43178
83  if ccxwriter.solver_obj.MatrixSolverType == "default":
84  pass
85  elif ccxwriter.solver_obj.MatrixSolverType == "pastix":
86  analysis_type += ", SOLVER=PASTIX"
87  elif ccxwriter.solver_obj.MatrixSolverType == "pardiso":
88  analysis_type += ", SOLVER=PARDISO"
89  elif ccxwriter.solver_obj.MatrixSolverType == "spooles":
90  analysis_type += ", SOLVER=SPOOLES"
91  elif ccxwriter.solver_obj.MatrixSolverType == "iterativescaling":
92  analysis_type += ", SOLVER=ITERATIVE SCALING"
93  elif ccxwriter.solver_obj.MatrixSolverType == "iterativecholesky":
94  analysis_type += ", SOLVER=ITERATIVE CHOLESKY"
95  # analysis line --> user defined incrementations --> parameter DIRECT
96  # --> completely switch off ccx automatic incrementation
97  if ccxwriter.solver_obj.IterationsUserDefinedIncrementations:
98  if ccxwriter.analysis_type == "static":
99  analysis_type += ", DIRECT"
100  elif ccxwriter.analysis_type == "thermomech":
101  analysis_type += ", DIRECT"
102  elif ccxwriter.analysis_type == "frequency":
103  FreeCAD.Console.PrintMessage(
104  "Analysis type frequency and IterationsUserDefinedIncrementations "
105  "are not allowed together, it is ignored\n"
106  )
107  # analysis line --> steadystate --> thermomech only
108  if ccxwriter.solver_obj.ThermoMechSteadyState:
109  # bernd: I do not know if STEADY STATE is allowed with DIRECT
110  # but since time steps are 1.0 it makes no sense IMHO
111  if ccxwriter.analysis_type == "thermomech":
112  analysis_type += ", STEADY STATE"
113  # Set time to 1 and ignore user inputs for steady state
114  ccxwriter.solver_obj.TimeInitialStep = 1.0
115  ccxwriter.solver_obj.TimeEnd = 1.0
116  elif (
117  ccxwriter.analysis_type == "static"
118  or ccxwriter.analysis_type == "frequency"
119  or ccxwriter.analysis_type == "buckling"
120  ):
121  pass # not supported for static and frequency!
122 
123  # ANALYSIS parameter line
124  analysis_parameter = ""
125  if ccxwriter.analysis_type == "static" or ccxwriter.analysis_type == "check":
126  if (
127  ccxwriter.solver_obj.IterationsUserDefinedIncrementations is True
128  or ccxwriter.solver_obj.IterationsUserDefinedTimeStepLength is True
129  ):
130  analysis_parameter = "{},{},{},{}".format(
131  ccxwriter.solver_obj.TimeInitialStep,
132  ccxwriter.solver_obj.TimeEnd,
133  ccxwriter.solver_obj.TimeMinimumStep,
134  ccxwriter.solver_obj.TimeMaximumStep,
135  )
136  elif ccxwriter.analysis_type == "frequency":
137  if (
138  ccxwriter.solver_obj.EigenmodeLowLimit == 0.0
139  and ccxwriter.solver_obj.EigenmodeHighLimit == 0.0
140  ):
141  analysis_parameter = f"{ccxwriter.solver_obj.EigenmodesCount}\n"
142  else:
143  analysis_parameter = "{},{},{}\n".format(
144  ccxwriter.solver_obj.EigenmodesCount,
145  ccxwriter.solver_obj.EigenmodeLowLimit,
146  ccxwriter.solver_obj.EigenmodeHighLimit,
147  )
148  elif ccxwriter.analysis_type == "thermomech":
149  # OvG: 1.0 increment, total time 1 for steady state will cut back automatically
150  analysis_parameter = "{},{},{},{}".format(
151  ccxwriter.solver_obj.TimeInitialStep,
152  ccxwriter.solver_obj.TimeEnd,
153  ccxwriter.solver_obj.TimeMinimumStep,
154  ccxwriter.solver_obj.TimeMaximumStep,
155  )
156  elif ccxwriter.analysis_type == "buckling":
157  analysis_parameter = "{},{}".format(
158  ccxwriter.solver_obj.BucklingFactors,
159  ccxwriter.solver_obj.BucklingAccuracy,
160  )
161 
162  # write analysis type line, analysis parameter line
163  f.write(analysis_type + "\n")
164  f.write(analysis_parameter + "\n")
