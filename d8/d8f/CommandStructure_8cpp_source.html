147 void StdCmdVarSet::activated(int iMsg)
148 {
149  Q_UNUSED(iMsg);
150 
151  openCommand(QT_TRANSLATE_NOOP("Command", "Add a variable set"));
152 
153  std::string VarSetName;
154  VarSetName = getUniqueObjectName("VarSet");
155  doCommand(Doc,"App.activeDocument().addObject('App::VarSet','%s')",VarSetName.c_str());
156 
157  // add the varset to a group if it is selected
158  auto sels = Selection().getSelectionEx(nullptr, App::DocumentObject::getClassTypeId(),
159  ResolveMode::OldStyleElement, true);
160  if (sels.size() == 1) {
161  App::DocumentObject* obj = sels[0].getObject();
162  auto group = obj->getExtension<App::GroupExtension>();
163  if (group) {
164  Gui::Document* docGui = Application::Instance->activeDocument();
165  App::Document* doc = docGui->getDocument();
166  group->addObject(doc->getObject(VarSetName.c_str()));
167  }
168  }
169  commitCommand();
170 
171  doCommand(Doc, "App.ActiveDocument.getObject('%s').ViewObject.doubleClicked()", VarSetName.c_str());
172 }
