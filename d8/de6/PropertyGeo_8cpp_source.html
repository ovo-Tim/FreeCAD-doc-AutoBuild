93 void PropertyVector::setPyObject(PyObject* value)
94 {
95  if (PyObject_TypeCheck(value, &(Base::VectorPy::Type))) {
96  Base::VectorPy* pcObject = static_cast<Base::VectorPy*>(value);
97  Base::Vector3d* val = pcObject->getVectorPtr();
98  setValue(*val);
99  }
100  else if (PyTuple_Check(value) && PyTuple_Size(value) == 3) {
101  PyObject* item {};
102  Base::Vector3d cVec;
103  // x
104  item = PyTuple_GetItem(value, 0);
105  if (PyFloat_Check(item)) {
106  cVec.x = PyFloat_AsDouble(item);
107  }
108  else if (PyLong_Check(item)) {
109  cVec.x = (double)PyLong_AsLong(item);
110  }
111  else {
112  throw Base::TypeError("Not allowed type used in tuple (float expected)...");
113  }
114  // y
115  item = PyTuple_GetItem(value, 1);
116  if (PyFloat_Check(item)) {
117  cVec.y = PyFloat_AsDouble(item);
118  }
119  else if (PyLong_Check(item)) {
120  cVec.y = (double)PyLong_AsLong(item);
121  }
122  else {
123  throw Base::TypeError("Not allowed type used in tuple (float expected)...");
124  }
125  // z
126  item = PyTuple_GetItem(value, 2);
127  if (PyFloat_Check(item)) {
128  cVec.z = PyFloat_AsDouble(item);
129  }
130  else if (PyLong_Check(item)) {
131  cVec.z = (double)PyLong_AsLong(item);
132  }
133  else {
134  throw Base::TypeError("Not allowed type used in tuple (float expected)...");
135  }
136  setValue(cVec);
137  }
138  else {
139  std::string error = std::string("type must be 'Vector' or tuple of three floats, not ");
140  error += value->ob_type->tp_name;
141  throw Base::TypeError(error);
142  }
143 }

427 void PropertyMatrix::setPyObject(PyObject* value)
428 {
429  if (PyObject_TypeCheck(value, &(Base::MatrixPy::Type))) {
430  Base::MatrixPy* pcObject = static_cast<Base::MatrixPy*>(value);
431  setValue(pcObject->value());
432  }
433  else if (PyTuple_Check(value) && PyTuple_Size(value) == 16) {
434  PyObject* item;
435  Base::Matrix4D cMatrix;
436 
437  const int dim = 4;
438  for (int x = 0; x < dim; x++) {
439  for (int y = 0; y < dim; y++) {
440  item = PyTuple_GetItem(value, x + y * dim);
441  if (PyFloat_Check(item)) {
442  cMatrix[x][y] = PyFloat_AsDouble(item);
443  }
444  else if (PyLong_Check(item)) {
445  cMatrix[x][y] = (double)PyLong_AsLong(item);
446  }
447  else {
448  throw Base::TypeError(
449  "Not allowed type used in matrix tuple (a number expected)...");
450  }
451  }
452  }
453 
454  setValue(cMatrix);
455  }
456  else {
457  std::string error = std::string("type must be 'Matrix' or tuple of 16 float or int, not ");
458  error += value->ob_type->tp_name;
459  throw Base::TypeError(error);
460  }

591 double toDouble(const boost::any& value)
592 {
593  double avalue {};
594 
595  if (value.type() == typeid(Base::Quantity)) {
596  avalue = boost::any_cast<Base::Quantity>(value).getValue();
597  }
598  else if (value.type() == typeid(double)) {
599  avalue = boost::any_cast<double>(value);
600  }
601  else if (value.type() == typeid(int)) {
602  avalue = boost::any_cast<int>(value);
603  }
604  else if (value.type() == typeid(unsigned int)) {
605  avalue = boost::any_cast<unsigned int>(value);
606  }
607  else if (value.type() == typeid(short)) {
608  avalue = boost::any_cast<short>(value);
609  }
610  else if (value.type() == typeid(unsigned short)) {
611  avalue = boost::any_cast<unsigned short>(value);
612  }
613  else if (value.type() == typeid(long)) {
614  avalue = boost::any_cast<long>(value);
615  }
616  else if (value.type() == typeid(unsigned long)) {
617  avalue = boost::any_cast<unsigned long>(value);
618  }
619  else {
620  throw std::bad_cast();
621  }
622  return avalue;
623 }

626 void PropertyPlacement::setPathValue(const ObjectIdentifier& path, const boost::any& value)
627 {
628  auto updateAxis = [this](int index, double coord) {
629  Base::Vector3d axis;
630  double angle;
631  Base::Vector3d base = _cPos.getPosition();
632  Base::Rotation rot = _cPos.getRotation();
633  rot.getRawValue(axis, angle);
634  axis[index] = coord;
635  rot.setValue(axis, angle);
636  Base::Placement plm(base, rot);
637  setValue(plm);
638  };
639 
640  auto updateYawPitchRoll = [this](int index, double angle) {
641  Base::Vector3d base = _cPos.getPosition();
642  Base::Rotation rot = _cPos.getRotation();
643  double yaw, pitch, roll;
644  rot.getYawPitchRoll(yaw, pitch, roll);
645  if (index == 0) {
646  if (angle < -180.0 || angle > 180.0) {
647  throw Base::ValueError("Yaw angle is out of range [-180, +180]");
648  }
649  yaw = angle;
650  }
651  else if (index == 1) {
652  if (angle < -90.0 || angle > 90.0) {
653  throw Base::ValueError("Pitch angle is out of range [-90, +90]");
654  }
655  pitch = angle;
656  }
657  else if (index == 2) {
658  if (angle < -180.0 || angle > 180.0) {
659  throw Base::ValueError("Roll angle is out of range [-180, +180]");
660  }
661  roll = angle;
662  }
663  rot.setYawPitchRoll(yaw, pitch, roll);
664  Base::Placement plm(base, rot);
665  setValue(plm);
666  };
667 
668  std::string subpath = path.getSubPathStr();
669  if (subpath == ".Rotation.Angle") {
670  double avalue = toDouble(value);
671  Property::setPathValue(path, Base::toRadians(avalue));
672  }
673  else if (subpath == ".Rotation.Axis.x") {
674  updateAxis(0, toDouble(value));
675  }
676  else if (subpath == ".Rotation.Axis.y") {
677  updateAxis(1, toDouble(value));
678  }
679  else if (subpath == ".Rotation.Axis.z") {
680  updateAxis(2, toDouble(value));
681  }
682  else if (subpath == ".Rotation.Yaw") {
683  updateYawPitchRoll(0, toDouble(value));
684  }
685  else if (subpath == ".Rotation.Pitch") {
686  updateYawPitchRoll(1, toDouble(value));
687  }
688  else if (subpath == ".Rotation.Roll") {
689  updateYawPitchRoll(2, toDouble(value));
690  }
691  else {
692  Property::setPathValue(path, value);
693  }
694 }

697 {
