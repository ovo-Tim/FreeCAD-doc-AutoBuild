61 bool Multisample::testSamples(int num) const
62 {
63  // This is always true
64  if (num == 0 || num == 1) {
65  return true;
66  }
67 
68  QOpenGLFramebufferObjectFormat fboFormat;
69  fboFormat.setAttachment(QOpenGLFramebufferObject::Depth);
70  fboFormat.setSamples(num);
71  QOpenGLFramebufferObject fbo(100, 100, fboFormat); // NOLINT
72  return fbo.format().samples() == num;
73 }

75 std::vector<std::pair<QString, AntiAliasing>> Multisample::supported() const
76 {
77  std::vector<std::pair<QString, AntiAliasing>> modes;
78  std::for_each(textMSAA.begin(), textMSAA.end(), [&modes, this](const auto& mode) {
79  if (testSamples(std::get<idMSAA>(mode))) {
80  const char* context = "Gui::Dialog::DlgSettings3DView";
81  QString str = QCoreApplication::translate(context, std::get<idStr>(mode).data());
82  modes.emplace_back(str, std::get<idEnum>(mode));
83  }
84  });
85  return modes;
86 }
