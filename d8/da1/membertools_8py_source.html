39 def get_member(analysis, t):
40  """Return list of all members of *analysis* of type *t*.
41 
42  Search *analysis* for members of type *t*. This method checks the custom
43  python typesystem (BaseType class property) used by the Fem module if
44  possible. If the object does not use the python typesystem the usual
45  isDerivedFrom from the C++ dynamic type system is used.
46 
47  :param analysis: only objects part of this analysis are considered
48  :param t: only objects of this type are returned
49 
50  :note:
51  Inheritance of Fem types is not checked. If *obj* uses Fems typesystem the
52  type is just checked for equality. If the type doesn't match
53  ``obj.isDerivedFrom`` is called as usual. See
54  https://forum.freecad.org/viewtopic.php?f=10&t=32625
55  """
56  if analysis is None:
57  raise ValueError("Analysis must not be None")
58  matching = []
59  for m in analysis.Group:
60  # since is _derived_from is used the father could be used
61  # to test too (ex. "Fem::FemMeshObject")
62  if femutils.is_derived_from(m, t) and not (
63  m.hasExtension("App::SuppressibleExtension") and m.Suppressed
64  ):
65  matching.append(m)
66  return matching

76  is used.
77 
78  :param analysis: only objects part of this analysis are considered
79  :param t: only a object of this type is returned
80 
81  :note:
82  Inheritance of Fem types is not checked. If *obj* uses Fems typesystem the
83  type is just checked for equality. If the type doesn't match
84  ``obj.isDerivedFrom`` is called as usual. See
85  https://forum.freecad.org/viewtopic.php?f=10&t=32625

87  objs = get_member(analysis, t)

91 def get_several_member(analysis, t):
92  """Get members and pack them for Calculix/Z88.
93 
94  Collect members by calling :py:func:`get_member` and pack them into a
95  data structure that can be consumed by calculix and Z88 solver modules.
96 
97  :param analysis: see :py:func:`get_member`
98  :param t: see :py:func:`get_member`
99 
100  :returns:
101  A list containing one dict per member. Each dict has two entries:
102  ``"Object"`` and ``"RefShapeType"``. ``dict["Object"]`` contains the
103  member document object. ``dict["RefShapeType"]`` contains the shape type
104  of the *References* property of the member (used by constraints) as a
105  string ("Vertex", "Edge", "Face" or "Solid"). If the member doesn't have a
106  *References* property ``dict["RefShapeType"]`` is the empty string ``""``.
107 
108  :note:
109  Undefined behaviour if one of the members has a *References* property
110  which is empty.
111 
112  :note:
113  Undefined behaviour if the type of the references of one object are not
114  all the same.
115 
116  :note:
117  Inheritance of Fem types is not checked. If *obj* uses Fems typesystem the
118  type is just checked for equality. If the type doesn't match
119  ``obj.isDerivedFrom`` is called as usual. See
120  https://forum.freecad.org/viewtopic.php?f=10&t=32625
121  """
122  # if no member is found, an empty list is returned
123  objs = get_member(analysis, t)
124  members = []
125  for m in objs:
126  obj_dict = {}
127  obj_dict["Object"] = m
128  obj_dict["RefShapeType"] = femutils.get_refshape_type(m)
129  members.append(obj_dict)
130  return members

133 def get_mesh_to_solve(analysis):
134  """Find one and only mesh object of *analysis*.
135 
136  :returns:
137  A tuple ``(object, message)``. If and only if the analysis contains
138  exactly one mesh object the first value of the tuple is the mesh document
139  object. Otherwise the first value is ``None`` and the second value is a
140  error message indicating what went wrong.
141  """
142  mesh_to_solve = None
143  for m in analysis.Group:
144  if m.isDerivedFrom("Fem::FemMeshObject") and not m.Suppressed:
145  if not mesh_to_solve:
146  mesh_to_solve = m
147  else:
148  return (None, "FEM: multiple mesh in analysis not yet supported!")
149  if mesh_to_solve is not None:
150  return (mesh_to_solve, "")
151  else:
152  return (None, "FEM: no mesh object found in analysis.")
