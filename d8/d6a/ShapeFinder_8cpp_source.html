107 TopoDS_Shape ShapeFinder::getLocatedShape(const App::DocumentObject& rootObject,
108  const std::string& leafSub)
109 {
110  auto resolved = resolveSelection(rootObject, leafSub);
111  auto target = &resolved.getTarget();
112  auto shortSub = resolved.getShortSub();
113  if (!target) {
114  return {};
115  }
116 
117  TopoDS_Shape shape = Part::Feature::getShape(target);
118  if (isShapeReallyNull(shape)) {
119  return {};
120  }
121 
122  auto cleanSub = removeTnpInfo(leafSub);
123  auto transform = getGlobalTransform(rootObject, cleanSub);
124 
125  shape = transformShape(shape, transform.first, transform.second);
126  Part::TopoShape tShape {shape};
127  if (!shortSub.empty()) {
128  return tShape.getSubTopoShape(shortSub.c_str()).getShape();
129  }
130 
131  return tShape.getShape();
132 }

147 void ShapeFinder::crawlPlacementChain(std::vector<Base::Placement>& plmStack,
148  std::vector<Base::Matrix4D>& scaleStack,
149  const App::DocumentObject& rootObject,
150  const std::string& leafSub)
151 {
152  auto currentSub = leafSub;
153  std::string previousSub {};
154  while (!currentSub.empty() && currentSub != previousSub) {
155  auto resolved = resolveSelection(rootObject, currentSub);
156  auto target = &resolved.getTarget();
157  if (!target) {
158  return;
159  }
160  auto currentPlacement = getPlacement(target);
161  auto currentScale = getScale(target);
162  if (!currentPlacement.isIdentity() || !currentScale.isUnity()) {
163  plmStack.push_back(currentPlacement);
164  scaleStack.push_back(currentScale);
165  }
166  previousSub = currentSub;
167  currentSub = pruneLastTerm(currentSub);
168  }
169 }

174 TopoDS_Shape ShapeFinder::transformShape(TopoDS_Shape& inShape,
175  const Base::Placement& placement,
176  const Base::Matrix4D& scaler)
177 {
178  if (isShapeReallyNull(inShape)) {
179  return {};
180  }
181  // we modify the parameter shape here. we don't claim to be const, but may be better to copy
182  // the shape?
183  Part::TopoShape tshape {inShape};
184  if (tshape.isInfinite()) {
185  inShape = stripInfiniteShapes(inShape);
186  }
187 
188  // copying the shape prevents "non-orthogonal GTrsf" errors in some versions
189  // of OCC. Something to do with triangulation of shape??
190  // it may be that incremental mesh would work here too.
191  BRepBuilderAPI_Copy copier(inShape);
192  tshape = Part::TopoShape(copier.Shape());
193  if (tshape.isNull()) {
194  return {};
195  }
196 
197  tshape.transformShape(scaler, true, true);
198  tshape.setPlacement(placement);
199 
200  return tshape.getShape();
201 }

262 TopoDS_Shape ShapeFinder::stripInfiniteShapes(const TopoDS_Shape& inShape)
263 {
264  BRep_Builder builder;
265  TopoDS_Compound comp;
266  builder.MakeCompound(comp);
267 
268  TopoDS_Iterator it(inShape);
269  for (; it.More(); it.Next()) {
270  TopoDS_Shape shape = it.Value();
271  if (shape.ShapeType() < TopAbs_SOLID) {
272  // look inside composite shapes
273  shape = stripInfiniteShapes(shape);
274  }
275  else if (Part::TopoShape(shape).isInfinite()) {
276  continue;
277  }
278  // simple shape & finite
279  builder.Add(comp, shape);
280  }
281 
282  return {std::move(comp)};
283 }

299 std::pair<Base::Placement, Base::Matrix4D>
300 ShapeFinder::getGlobalTransform(const App::DocumentObject& rootObject, const std::string& leafSub)
301 {
302  // we prune the last term if it is a vertex, edge or face
303  std::string newSub = removeGeometryTerm(leafSub);
304 
305  std::vector<Base::Placement> plmStack;
306  std::vector<Base::Matrix4D> scaleStack;
307  // get transforms below rootObject
308  // Note: root object is provided by the caller and may or may not be a top level object
309  crawlPlacementChain(plmStack, scaleStack, rootObject, newSub);
310 
311  auto pathTransform = sumTransforms(plmStack, scaleStack);
312 
313  // apply the placements in reverse order - top to bottom
314  // should this be rootObject's local transform?
315  auto rootTransform = getGlobalTransform(&rootObject);
316 
317  auto netPlm = rootTransform.first * pathTransform.first;
318  auto netScale = rootTransform.second * pathTransform.second;
319 
320  return {netPlm, netScale};
321 }

326 std::pair<Base::Placement, Base::Matrix4D>
327 ShapeFinder::getGlobalTransform(const App::DocumentObject* cursorObject)
328 {
329  if (!cursorObject) {
330  return {};
331  }
332 
333  Base::Placement netPlm;
334  Base::Matrix4D netScale = getScale(cursorObject);
335 
336  Base::Placement geoPlm;
337  auto geoCursor = dynamic_cast<const App::GeoFeature*>(cursorObject);
338  if (!isLinkLike(cursorObject) && geoCursor) {
339  netPlm = geoCursor->globalPlacement();
340  return {netPlm, netScale};
341  }
342 
343  netPlm = getPlacement(cursorObject);
344 
345  return {netPlm, netScale};
346 }

351 std::pair<Base::Placement, Base::Matrix4D>
352 ShapeFinder::sumTransforms(const std::vector<Base::Placement>& plmStack,
353  const std::vector<Base::Matrix4D>& scaleStack)
354 {
355  Base::Placement netPlm;
356  Base::Matrix4D netScale;
357 
358  auto itRevPlm = plmStack.rbegin();
359  for (; itRevPlm != plmStack.rend(); itRevPlm++) {
360  netPlm *= *itRevPlm;
361  }
362  auto itRevScale = scaleStack.rbegin();
363  for (; itRevScale != scaleStack.rend(); itRevScale++) {
364  netScale *= *itRevScale;
365  }
366 
367  return {netPlm, netScale};
368 }
