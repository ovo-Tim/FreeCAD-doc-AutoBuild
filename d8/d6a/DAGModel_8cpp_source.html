269 void Model::slotDeleteObject(const ViewProviderDocumentObject &VPDObjectIn)
270 {
271  Vertex vertex = findRecord(&VPDObjectIn, *graphLink).vertex;
272 
273  //remove items from scene.
274  removeVertexItemsFromScene(vertex);
275 
276  //remove connector items
277  auto outRange = boost::out_edges(vertex, *theGraph);
278  for (auto outEdgeIt = outRange.first; outEdgeIt != outRange.second; ++outEdgeIt)
279  this->removeItem((*theGraph)[*outEdgeIt].connector.get());
280  auto inRange = boost::in_edges(vertex, *theGraph);
281  for (auto inEdgeIt = inRange.first; inEdgeIt != inRange.second; ++inEdgeIt)
282  this->removeItem((*theGraph)[*inEdgeIt].connector.get());
283 
284  if (vertex == lastAddedVertex)
285  lastAddedVertex = Graph::null_vertex();
286 
287  (*theGraph)[vertex].connChangeIcon.disconnect();
288 
289  //remove the actual vertex.
290  boost::clear_vertex(vertex, *theGraph);
291  boost::remove_vertex(vertex, *theGraph);
292 
293  eraseRecord(&VPDObjectIn, *graphLink);
294  graphDirty = true;
295 }

343 void Model::selectionChanged(const SelectionChanges& msg)
344 {
345  //TODO: note that treeview uses set selection which sends a message with just a document name
346  //and no object name. Have to explore further.
347 
348  auto getAllEdges = [this](const Vertex &vertexIn)
349  {
350  //is there really no function to get both in and out edges?
351  std::vector<Edge> out;
352 
353  OutEdgeIterator outIt, outItEnd;
354  for (boost::tie(outIt, outItEnd) = boost::out_edges(vertexIn, *theGraph); outIt != outItEnd; ++outIt)
355  out.push_back(*outIt);
356 
357  InEdgeIterator inIt, inItEnd;
358  for (boost::tie(inIt, inItEnd) = boost::in_edges(vertexIn, *theGraph); inIt != inItEnd; ++inIt)
359  out.push_back(*inIt);
360 
361  return out;
362  };
363 
364  auto highlightConnectorOn = [this, getAllEdges](const Vertex &vertexIn)
365  {
366  QColor color = (*theGraph)[vertexIn].text->defaultTextColor();
367  QPen pen(color);
368  pen.setWidth(3.0);
369  auto edges = getAllEdges(vertexIn);
370  for (auto edge : edges)
371  {
372  (*theGraph)[edge].connector->setPen(pen);
373  (*theGraph)[edge].connector->setZValue(1.0);
374  }
375  };
376 
377  auto highlightConnectorOff = [this, getAllEdges](const Vertex &vertexIn)
378  {
379  auto edges = getAllEdges(vertexIn);
380  for (auto edge : edges)
381  {
382  (*theGraph)[edge].connector->setPen(QPen());
383  (*theGraph)[edge].connector->setZValue(0.0);
384  }
385  };
386 
387  //lambda for clearing selections.
388  auto clearSelection = [this, highlightConnectorOff]()
389  {
390  BGL_FORALL_VERTICES(currentVertex, *theGraph, Graph)
391  {
392  RectItem *rect = (*theGraph)[currentVertex].rectangle.get();
393  assert(rect);
394  rect->selectionOff();
395  highlightConnectorOff(currentVertex);
396  }
397  };
398 
399  //lambda for getting rectangle.
400  auto getRectangle = [this](const char *in)
401  {
402  assert(in);
403  std::string name(in);
404  assert(!name.empty());
405  const GraphLinkRecord &record = findRecord(name, *graphLink);
406  RectItem *rect = (*theGraph)[record.vertex].rectangle.get();
407  assert(rect);
408  return rect;
409  };
410 
411  if (msg.Type == SelectionChanges::AddSelection)
412  {
413  if (msg.pObjectName)
414  {
415  RectItem *rect = getRectangle(msg.pObjectName);
416  rect->selectionOn();
417  highlightConnectorOn(findRecord(std::string(msg.pObjectName), *graphLink).vertex);
418  }
419  }
420  else if(msg.Type == SelectionChanges::RmvSelection)
421  {
422  if (msg.pObjectName)
423  {
424  RectItem *rect = getRectangle(msg.pObjectName);
425  rect->selectionOff();
426  highlightConnectorOff(findRecord(std::string(msg.pObjectName), *graphLink).vertex);
427  }
428  }
429  else if(msg.Type == SelectionChanges::SetSelection)
430  {
431  clearSelection();
432 
433  auto selections = Gui::Selection().getSelection(msg.pDocName);
434  for (const auto &selection : selections)
435  {
436  assert(selection.FeatName);
437  RectItem *rect = getRectangle(selection.FeatName);
438  rect->selectionOn();
439  highlightConnectorOn(findRecord(selection.FeatName, *graphLink).vertex);
440  }
441  }
442  else if(msg.Type == SelectionChanges::ClrSelection)
443  {
444  clearSelection();
445  }
446 
447  this->invalidate();
448 }

460 void Model::updateSlot()
461 {
462  //empty outList means it is a root.
463  //empty inList means it is a leaf.
464 
465  //NOTE: some of the following loops can/should be combined
466  //for speed. Not doing yet, as I want a simple algorithm until
467  //a more complete picture is formed.
468 
469  Base::TimeElapsed startTime;
470 
471  //here we will cycle through the graph updating edges.
472  //we have to do this first and in isolation because everything is dependent on an up to date graph.
473  BGL_FORALL_VERTICES(currentVertex, *theGraph, Graph)
474  {
475  const App::DocumentObject *currentDObject = findRecord(currentVertex, *graphLink).DObject;
476  std::vector<App::DocumentObject *> otherDObjects = currentDObject->getOutList();
477  for (auto &currentOtherDObject : otherDObjects)
478  {
479  if (!hasRecord(currentOtherDObject, *graphLink))
480  continue;
481  Vertex otherVertex = findRecord(currentOtherDObject, *graphLink).vertex;
482  bool result;
483  Edge edge;
484  boost::tie(edge, result) = boost::add_edge(currentVertex, otherVertex, *theGraph);
485  if (result)
486  {
487  (*theGraph)[edge].connector = std::make_shared<QGraphicsPathItem>();
488  (*theGraph)[edge].connector->setZValue(0.0);
489  }
490  }
491  }
492 
493  //apply filters.
494  BGL_FORALL_VERTICES(currentVertex, *theGraph, Graph)
495  {
496  (*theGraph)[currentVertex].dagVisible = true; //default to shown.
497  for (const auto &currentFilter : filters)
498  {
499  if (!currentFilter->enabled || currentFilter->type != FilterBase::Type::Exclusion)
500  continue;
501  if (currentFilter->goFilter(currentVertex, *theGraph, *graphLink))
502  (*theGraph)[currentVertex].dagVisible = false;
503  }
504  }
505  //inclusion takes precedence. Separate loop because filters might probe
506  //children and parents. So we want to ensure all exclusions are done
507  //before inclusions start.
508  BGL_FORALL_VERTICES(currentVertex, *theGraph, Graph)
509  {
510  for (const auto &currentFilter : filters)
511  {
512  if (!currentFilter->enabled || currentFilter->type != FilterBase::Type::Inclusion)
513  continue;
514  if (currentFilter->goFilter(currentVertex, *theGraph, *graphLink))
515  (*theGraph)[currentVertex].dagVisible = true;
516  }
517  }
518 
519  //sync scene items to graph vertex dagVisible.
520  BGL_FORALL_VERTICES(currentVertex, *theGraph, Graph)
521  {
522  if ((*theGraph)[currentVertex].dagVisible && (!(*theGraph)[currentVertex].rectangle->scene()))
523  addVertexItemsToScene(currentVertex);
524  if ((!(*theGraph)[currentVertex].dagVisible) && (*theGraph)[currentVertex].rectangle->scene())
525  removeVertexItemsFromScene(currentVertex);
526  }
527 
528  //sync scene items for graph edge.
529  BGL_FORALL_EDGES(currentEdge, *theGraph, Graph)
530  {
531  Vertex source = boost::source(currentEdge, *theGraph);
532  Vertex target = boost::target(currentEdge, *theGraph);
533 
534  bool edgeVisible = (*theGraph)[source].dagVisible && (*theGraph)[target].dagVisible;
535  if (edgeVisible && (!(*theGraph)[currentEdge].connector->scene()))
536  this->addItem((*theGraph)[currentEdge].connector.get());
537  if ((!edgeVisible) && (*theGraph)[currentEdge].connector->scene())
538  this->removeItem((*theGraph)[currentEdge].connector.get());
539  }
540 
541  indexVerticesEdges();
542  Path sorted;
543  try
544  {
545  boost::topological_sort(*theGraph, std::back_inserter(sorted));
546  }
547  catch(const boost::not_a_dag &)
548  {
549  Base::Console().Error("not a dag exception in DAGView::Model::updateSlot()\n");
550  //do not continuously report an error for cyclic graphs
551  graphDirty = false;
552  return;
553  }
554  //index the vertices in sort order.
555  int tempIndex = 0;
556  for (const auto &currentVertex : sorted)
557  {
558  (*theGraph)[currentVertex].topoSortIndex = tempIndex;
559  tempIndex++;
560  }
561 
562  //draw graph(nodes and connectors).
563  int currentRow = 0;
564  int currentColumn = -1; //we know first column is going to be root so will be kicked up to 0.
565  int maxColumn = currentColumn; //used for determining offset of icons and text.
566  float maxTextLength = 0;
567  for (const auto &currentVertex : sorted)
568  {
569  if (!(*theGraph)[currentVertex].dagVisible)
570  continue;
571 
572  if (boost::out_degree(currentVertex, *theGraph) == 0)
573  currentColumn = 0;
574  else
575  {
576  //loop parents and find an acceptable column.
577  int farthestParentIndex = sorted.size();
578  ColumnMask columnMask;
579  Path parentVertices;
580  OutEdgeIterator it, itEnd;
581  boost::tie(it, itEnd) = boost::out_edges(currentVertex, *theGraph);
582  for (;it != itEnd; ++it)
583  {
584 // std::cout << std::endl << "name: " << findRecord(currentVertex, *graphLink).DObject->Label.getValue() << std::endl;
585 
586  Vertex target = boost::target(*it, *theGraph);
587  parentVertices.push_back(target);
588  int currentParentIndex = (*theGraph)[target].topoSortIndex;
589  if (currentParentIndex < farthestParentIndex)
590  {
591  Path::const_iterator start = sorted.begin() + currentParentIndex + 1; // 1 after
592  Path::const_iterator end = sorted.begin() + (*theGraph)[currentVertex].topoSortIndex; // 1 before
593  Path::const_iterator it;
594  for (it = start; it != end; ++it)
595  {
596 // std::cout << " parent: " << findRecord(*it, *graphLink).DObject->Label.getValue() << std::endl;
597 
598  columnMask |= (*theGraph)[*it].column;
599  }
600  farthestParentIndex = currentParentIndex;
601  }
602  }
603 
604  //have to create a smaller subset to get through std::cout.
605 // std::bitset<8> testSet;
606 // for (unsigned int index = 0; index < testSet.size(); ++index)
607 // testSet[index]= columnMask[index];
608 // std::cout << "mask for " << findRecord(currentVertex, *graphLink).DObject->Label.getValue() << " " <<
609 // testSet.to_string() << std::endl;
610 
611  //now we should have a mask representing the columns that are being used.
612  //this is from the lowest parent, in the topo sort, to last entry.
613 
614  //try to use the same column as one of the parents.(*theGraph)[*it].column
615  int destinationColumn = 0; //default to first column
616  for (const auto &currentParent : parentVertices)
617  {
618  if (((*theGraph)[currentParent].column & columnMask).none())
619  {
620  //go with first visible parent for now.
621  if (!(*theGraph)[currentParent].dagVisible)
622  continue;
623  destinationColumn = static_cast<int>(columnFromMask((*theGraph)[currentParent].column));
624  break;
625  }
626  }
627  //if destination not valid look for the first open column.
628  if (columnMask.test(destinationColumn))
629  {
630  for (std::size_t index = 0; index < columnMask.size(); ++index)
631  {
632  if (! columnMask.test(index))
633  {
634  destinationColumn = index;
635  break;
636  }
637  }
638  }
639 
640  currentColumn = destinationColumn;
641  }
642 
643  assert(currentColumn < static_cast<int>(ColumnMask().size())); //temp limitation.
644 
645  maxColumn = std::max(currentColumn, maxColumn);
646  QBrush currentBrush(forgroundBrushes.at(currentColumn % forgroundBrushes.size()));
647 
648  auto rectangle = (*theGraph)[currentVertex].rectangle.get();
649  rectangle->setRect(-rowPadding, 0.0, rowPadding, rowHeight); //calculate actual length later.
