65 bool ViewProviderHole::setEdit(int ModNum)
66 {
67  if (ModNum == ViewProvider::Default ) {
68  // When double-clicking on the item for this hole the
69  // object unsets and sets its edit mode without closing
70  // the task panel
71  Gui::TaskView::TaskDialog *dlg = Gui::Control().activeDialog();
72  TaskDlgHoleParameters *holeDlg = qobject_cast<TaskDlgHoleParameters *>(dlg);
73  if (holeDlg && holeDlg->getViewObject() != this)
74  holeDlg = nullptr; // another hole left open its task panel
75  if (dlg && !holeDlg) {
76  QMessageBox msgBox;
77  msgBox.setText(QObject::tr("A dialog is already open in the task panel"));
78  msgBox.setInformativeText(QObject::tr("Do you want to close this dialog?"));
79  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
80  msgBox.setDefaultButton(QMessageBox::Yes);
81  int ret = msgBox.exec();
82  if (ret == QMessageBox::Yes)
83  Gui::Control().closeDialog();
84  else
85  return false;
86  }
87 
88  // clear the selection (convenience)
89  Gui::Selection().clearSelection();
90 
91  // always change to PartDesign WB, remember where we come from
92  oldWb = Gui::Command::assureWorkbench("PartDesignWorkbench");
93 
94  // start the edit dialog
95  if (holeDlg)
96  Gui::Control().showDialog(holeDlg);
97  else
98  Gui::Control().showDialog(new TaskDlgHoleParameters(this));
99 
100  return true;
101  }
102  else {
103  return PartGui::ViewProviderPart::setEdit(ModNum); // clazy:exclude=skipped-base-method
104  }
105 }
