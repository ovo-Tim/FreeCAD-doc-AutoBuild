57 const std::list<gp_Trsf>
58 Scaled::getTransformations(const std::vector<App::DocumentObject*> originals)
59 {
60  double const factor = Factor.getValue();
61  if (factor < Precision::Confusion()) {
62  throw Base::ValueError("Scaling factor too small");
63  }
64  int const occurrences = Occurrences.getValue();
65  if (occurrences < 2) {
66  throw Base::ValueError("At least two occurrences required");
67  }
68 
69  double f = (factor - 1.0) / double(occurrences - 1);
70 
71  gp_Pnt cog;
72  if (!originals.empty()) {
73  // Find centre of gravity of first original
74  // FIXME: This method will NOT give the expected result for more than one original!
75  if (auto feature = Base::freecad_dynamic_cast<PartDesign::FeatureAddSub>(originals.front())) {
76  TopoDS_Shape original = feature->AddSubShape.getShape().getShape();
77 
78  GProp_GProps props;
79  BRepGProp::VolumeProperties(original, props);
80  cog = props.CentreOfMass();
81  }
82  }
83 
84  // Note: The original feature is NOT included in the list of transformations! Therefore
85  // we start with occurrence number 1, not number 0
86  std::list<gp_Trsf> transformations;
87  gp_Trsf trans;
88  transformations.push_back(trans); // identity transformation
89 
90  for (int i = 1; i < occurrences; i++) {
91  trans.SetScale(cog, 1.0 + double(i) * f);
92  transformations.push_back(trans);
93  }
94 
95  return transformations;
96 }
