93 App::DocumentObjectExecReturn *DrawViewArch::execute()
94 {
95  if (!keepUpdated()) {
96  return App::DocumentObject::StdReturn;
97  }
98 
99  App::DocumentObject* sourceObj = Source.getValue();
100  if (sourceObj) {
101  //if (sourceObj is not ArchSection) return
102  App::Property* proxy = sourceObj->getPropertyByName("Proxy");
103  if (!proxy) {
104  Base::Console().Error("DVA::execute - %s is not an ArchSection\n", sourceObj->Label.getValue());
105  //this is definitely not an ArchSection
106  return DrawView::execute();
107  }
108 
109  //std::string svgFrag;
110  std::string svgHead = getSVGHead();
111  std::string svgTail = getSVGTail();
112  std::string FeatName = getNameInDocument();
113  std::string SourceName = sourceObj->getNameInDocument();
114  // ArchSectionPlane.getSVG(section, allOn=False, renderMode="Wireframe", showHidden=False, showFill=False, scale=1, linewidth=1, fontsize=1):
115 
116  std::stringstream paramStr;
117  paramStr << ", allOn=" << (AllOn.getValue() ? "True" : "False")
118  << ", renderMode=" << RenderMode.getValue()
119  << ", showHidden=" << (ShowHidden.getValue() ? "True" : "False")
120  << ", showFill=" << (ShowFill.getValue() ? "True" : "False")
121  << ", scale=" << getScale()
122  << ", linewidth=" << LineWidth.getValue()
123  << ", fontsize=" << FontSize.getValue()
124  << ", techdraw=True"
125  << ", rotation=" << Rotation.getValue()
126  << ", fillSpaces=" << (FillSpaces.getValue() ? "True" : "False")
127  << ", cutlinewidth=" << CutLineWidth.getValue()
128  << ", linespacing=" << LineSpacing.getValue()
129  << ", joinArch=" << (JoinArch.getValue() ? "True" : "False");
130 
131  Base::Interpreter().runString("import ArchSectionPlane");
132  Base::Interpreter().runStringArg("svgBody = ArchSectionPlane.getSVG(App.activeDocument().%s %s)",
133  SourceName.c_str(), paramStr.str().c_str());
134  Base::Interpreter().runStringArg("App.activeDocument().%s.Symbol = '%s' + svgBody + '%s'",
135  FeatName.c_str(), svgHead.c_str(), svgTail.c_str());
136  }
137  overrideKeepUpdated(false);
138  return DrawView::execute();
139 }
