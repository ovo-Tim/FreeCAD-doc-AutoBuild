42 def write_fenics_mesh_xml(fem_mesh_obj, outputfile):
43  """
44  For the export, we only have to use the highest dimensional entities and their
45  vertices to be exported.
46  For second order elements, we have to delete the mid element nodes.
47  """
48 
49  # TODO: check for second order elements
50  # (reduce element order would be ok - all elements have at least the first
51  # necessary nodes)
52 
53  FreeCAD_to_Fenics_dict = {
54  "Triangle": "triangle",
55  "Tetra": "tetrahedron",
56  "Hexa": "hexahedron",
57  "Edge": "interval",
58  "Node": "point",
59  "Quadrangle": "quadrilateral",
60  "Polygon": "unknown",
61  "Polyhedron": "unknown",
62  "Prism": "unknown",
63  "Pyramid": "unknown",
64  }
65 
66  XML_Number_of_Nodes_dict = {
67  "point": 1,
68  "interval": 2,
69  "triangle": 3,
70  "quadrilateral": 4,
71  "tetrahedron": 4,
72  "hexahedron": 8,
73  }
74 
75  Console.PrintMessage(f"Converting {fem_mesh_obj.Label} to fenics XML File\n")
76  Console.PrintMessage(f"Dimension of mesh: {get_FemMeshObjectDimension(fem_mesh_obj)}\n")
77 
78  elements_in_mesh = get_FemMeshObjectElementTypes(fem_mesh_obj)
79  Console.PrintMessage(f"Elements appearing in mesh: {str(elements_in_mesh)}\n")
80  celltype_in_mesh = get_MaxDimElementFromList(elements_in_mesh)
81  (num_cells, cellname_fc, dim_cell) = celltype_in_mesh
82  cellname_fenics = FreeCAD_to_Fenics_dict[cellname_fc]
83  num_verts_cell = XML_Number_of_Nodes_dict[cellname_fenics]
84  Console.PrintMessage(
85  f"Celltype in mesh -> {str(celltype_in_mesh)} "
86  + f"and its Fenics name: {cellname_fenics}\n"
87  )
88 
89  root = ET.Element("dolfin", dolfin="http://fenicsproject.org")
90  meshchild = ET.SubElement(root, "mesh", celltype=cellname_fenics, dim=str(dim_cell))
91  vertices = ET.SubElement(meshchild, "vertices", size=str(fem_mesh_obj.FemMesh.NodeCount))
92 
93  for nodeind, fc_vec in list(fem_mesh_obj.FemMesh.Nodes.items()):
94  ET.SubElement(
95  vertices,
96  "vertex",
97  index=str(nodeind - 1),
98  # FC starts from 1, fenics starts from 0 to size-1
99  x=str(fc_vec[0]),
100  y=str(fc_vec[1]),
101  z=str(fc_vec[2]),
102  )
103 
104  cells = ET.SubElement(meshchild, "cells", size=str(num_cells))
105  if dim_cell == 3:
106  fc_cells = fem_mesh_obj.FemMesh.Volumes
107  elif dim_cell == 2:
108  fc_cells = fem_mesh_obj.FemMesh.Faces
109  elif dim_cell == 1:
110  fc_cells = fem_mesh_obj.FemMesh.Edges
111  else:
112  fc_cells = ()
113 
114  for fen_ind, fc_volume_ind in enumerate(fc_cells):
115  # FC starts after all other entities, fenics start from 0 to size-1
116  nodeindices = fem_mesh_obj.FemMesh.getElementNodes(fc_volume_ind)
117 
118  cell_args = {}
119  for vi, ni in enumerate(nodeindices):
120  if vi < num_verts_cell: # XML only supports first order meshs
121  cell_args["v" + str(vi)] = str(ni - 1)
122  # generate as many v entries in dict as nodes are listed in cell
123  # works only for first order elements
124 
125  ET.SubElement(cells, cellname_fenics, index=str(fen_ind), **cell_args)
126 
127  # ET.SubElement(meshchild, "data")
128  # removed to eliminate warning from meshio
129 
130  fp = open(outputfile, "wb")
131  fp.write(ET.tostring(root))
132  # xml core functionality does not support pretty printing
133  # so the output file looks quite ugly
134  fp.close()
