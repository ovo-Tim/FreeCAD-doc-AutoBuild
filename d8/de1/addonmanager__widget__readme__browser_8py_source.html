45 class WidgetReadmeBrowser(QtWidgets.QTextBrowser):
46  """A QTextBrowser widget that emits signals for each requested image resource, allowing an external controller
47  to load and re-deliver those images. Once all resources have been re-delivered, the original data is redisplayed
48  with the images in-line. Call setUrl prior to calling setMarkdown or setHtml to ensure URLs are resolved
49  correctly."""

65  def setMarkdown(self, md: str):

67  have native markdown support. Lacking that, plaintext is displayed."""

85  def _clean_markdown(self, md: str):
86  # Remove some HTML tags ( for now just img and br, which are the most common offenders that break rendering )
87  br_re = re.compile(r"<br\s*/?>")
88  img_re = re.compile(r"<img\s.*?src=(?:'|\")([^'\">]+)(?:'|\").*?\/?>")
89 
90  cleaned = br_re.sub(r"\n", md)
91  cleaned = img_re.sub(r"[html tag removed]", cleaned)
92 
93  return cleaned

95  def set_resource(self, resource_url: str, image: Optional[QtGui.QImage]):
96  """Once a resource has been fetched (or the fetch has failed), this method should be used to inform the widget
97  that the resource has been loaded. Note that the incoming image is scaled to 97% of the widget width if it is
98  larger than that."""
99  self.image_map[resource_url] = self._ensure_appropriate_width(image)

101  def loadResource(self, resource_type: int, name: QtCore.QUrl) -> object:
102  """Callback for resource loading. Called automatically by underlying Qt
103  code when external resources are needed for rendering. In particular,
104  here it is used to download and cache (in RAM) the images needed for the
105  README and Wiki pages."""
106  if resource_type == QtGui.QTextDocument.ImageResource and not self.stop:
107  full_url = self._create_full_url(name.toString())
108  if full_url not in self.image_map:
109  self.load_resource.emit(full_url)
110  self.image_map[full_url] = None
111  return self.image_map[full_url]
112  elif resource_type == QtGui.QTextDocument.MarkdownResource:
113  self.follow_link.emit(name.toString())
114  return self.toMarkdown()
115  elif resource_type == QtGui.QTextDocument.HtmlResource:
116  self.follow_link.emit(name.toString())
117  return self.toHtml()
118  return super().loadResource(resource_type, name)

120  def _ensure_appropriate_width(self, image: QtGui.QImage) -> QtGui.QImage:
121  ninety_seven_percent = self.width() * 0.97
122  if image.width() < ninety_seven_percent:
123  return image
124  return image.scaledToWidth(ninety_seven_percent)

126  def _create_full_url(self, url: str) -> str:
127  if url.startswith("http"):
128  return url

130  return url

132  return lhs + slash + url
