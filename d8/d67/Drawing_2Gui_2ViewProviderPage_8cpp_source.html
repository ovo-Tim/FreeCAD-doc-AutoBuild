104 void ViewProviderDrawingPage::updateData(const App::Property* prop)
105 {
106  Gui::ViewProviderDocumentObjectGroup::updateData(prop);
107  if (prop->is<App::PropertyFileIncluded>()) {
108  if (std::string(getPageObject()->PageResult.getValue()) != "") {
109  if (view) {
110  view->load(QString::fromUtf8(getPageObject()->PageResult.getValue()));
111  if (view->isHidden()) {
112  QTimer::singleShot(300, view, SLOT(viewAll()));
113  }
114  else {
115  view->viewAll();
116  }
117  }
118  }
119  }
120  else if (pcObject && prop == &pcObject->Label) {
121  if (view) {
122  const char* objname = pcObject->Label.getValue();
123  view->setObjectName(QString::fromUtf8(objname));
124  Gui::Document* doc = Gui::Application::Instance->getDocument(pcObject->getDocument());
125  view->onRelabel(doc);
126  }
127  }
128 }

158 DrawingView* ViewProviderDrawingPage::showDrawingView()
159 {

162  view = new DrawingView(doc, Gui::getMainWindow());
163  view->setWindowIcon(Gui::BitmapFactory().pixmap("actions/drawing-landscape"));
164 
165  const char* objname = pcObject->Label.getValue();

167  view->onRelabel(doc);
