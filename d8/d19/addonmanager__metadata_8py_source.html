75 class Version:
76  """Provide a more useful representation of Version information"""
77 
78  def __init__(self, from_string: str = None, from_list=None):
79  """If from_string is a string, it is parsed to get the version. If from_list
80  exists (and no string was provided), it is treated as a version list of
81  [major:int, minor:int, patch:int, pre:str]"""
82  self.version_as_list = [0, 0, 0, ""]
83  if from_string is not None:
84  self._init_from_string(from_string)
85  elif from_list is not None:
86  self._init_from_list(from_list)
87 
88  def _init_from_string(self, from_string: str):
89  """Find the first digit in the given string, and send that substring off for
90  parsing."""
91  counter = 0
92  for char in from_string:
93  if char.isdigit():
94  break
95  counter += 1
96  self._parse_string_to_tuple(from_string[counter:])
97 
98  def _init_from_list(self, from_list):
99  for index, element in enumerate(from_list):
100  if index < 3:
101  self.version_as_list[index] = int(element)
102  elif index == 3:
103  self.version_as_list[index] = str(element)
104  else:
105  break
106 
107  def _parse_string_to_tuple(self, from_string: str):
108  """We hand-parse only simple version strings, of the form 1.2.3suffix -- only
109  the first digit is required."""
110  splitter = from_string.split(".", 2)
111  counter = 0
112  for component in splitter:
113  try:
114  self.version_as_list[counter] = int(component)
115  counter += 1
116  except ValueError:
117  if counter == 0:
118  raise ValueError(f"Invalid version string {from_string}")
119  number, text = self._parse_final_entry(component)
120  self.version_as_list[counter] = number
121  self.version_as_list[3] = text
122 
123  @staticmethod
124  def _parse_final_entry(final_string: str) -> Tuple[int, str]:
125  """The last value is permitted to contain both a number and a word, and needs
126  to be split"""
127  digits = ""
128  for c in final_string:
129  if c.isdigit():
130  digits += c
131  else:
132  break
133  return int(digits), final_string[len(digits) :]
134 
135  def __repr__(self) -> str:
136  v = self.version_as_list
137  return f"{v[0]}.{v[1]}.{v[2]} {v[3]}"
138 
139  def __eq__(self, other) -> bool:
140  return self.version_as_list == other.version_as_list
141 
142  def __ne__(self, other) -> bool:
143  return not (self == other)
144 
145  def __lt__(self, other) -> bool:
146  for a, b in zip(self.version_as_list, other.version_as_list):
147  if a != b:
148  return a < b
149  return False
150 
151  def __gt__(self, other) -> bool:
152  if self.version_as_list == other.version_as_list:
153  return False
154  return not (self < other)
155 
156  def __ge__(self, other) -> bool:
157  return self > other or self == other
158 
159  def __le__(self, other) -> bool:
160  return self < other or self == other

221 def get_first_supported_freecad_version(metadata: Metadata) -> Optional[Version]:
222  """Look through all content items of this metadata element and determine what the
223  first version of freecad that ANY of the items support is. For example, if it
224  contains several workbenches, some of which require v0.20, and some 0.21, then
225  0.20 is returned. Returns None if frecadmin is unset by any part of this object."""
226 
227  current_earliest = metadata.freecadmin if metadata.freecadmin is not None else None
228  for content_class in metadata.content.values():
229  for content_item in content_class:
230  content_first = get_first_supported_freecad_version(content_item)
231  if content_first is not None:
232  if current_earliest is None:
233  current_earliest = content_first
234  else:
235  current_earliest = min(current_earliest, content_first)
236 
237  return current_earliest

283  @staticmethod
284  def _parse_child_element(namespace: str, child: ET.Element, metadata: Metadata):
285  """Figure out what sort of metadata child represents, and add it to the
286  metadata object."""
287 
288  tag = child.tag[len(namespace) :]
289  if tag in ["name", "date", "description", "type", "icon", "classname", "subdirectory"]:
290  # Text-only elements
291  metadata.__dict__[tag] = child.text
292  elif tag in ["version", "freecadmin", "freecadmax", "pythonmin"]:
293  try:
294  metadata.__dict__[tag] = Version(from_string=child.text)
295  except ValueError:
296  print(
297  f"Invalid version specified for tag {tag} in Addon {metadata.name}: {child.text}"
298  )
299  metadata.__dict__[tag] = Version(from_list=[0, 0, 0])
300  elif tag in ["tag", "file"]:
301  # Lists of strings
302  if child.text:
303  metadata.__dict__[tag].append(child.text)
304  elif tag in ["maintainer", "author"]:
305  # Lists of contacts
306  metadata.__dict__[tag].append(MetadataReader._parse_contact(child))
307  elif tag == "license":
308  # List of licenses
309  metadata.license.append(MetadataReader._parse_license(child))
310  elif tag == "url":
311  # List of urls
312  metadata.url.append(MetadataReader._parse_url(child))
313  elif tag in ["depend", "conflict", "replace"]:
314  # Lists of dependencies
315  metadata.__dict__[tag].append(MetadataReader._parse_dependency(child))
316  elif tag == "content":
317  MetadataReader._parse_content(namespace, metadata, child)

366  @staticmethod
367  def _parse_content(namespace: str, metadata: Metadata, root: ET.Element):
368  """Given a content node, loop over its children, and if they are a recognized
369  element type, recurse into each one to parse it."""
370  known_content_types = ["workbench", "macro", "preferencepack"]
371  for child in root:
