149  def get_avail_version(self):
150  """Retrieves an available ifcopenshell version"""
151 
152  result = self.run_pip(["index", "versions", "ifcopenshell"])

154  if result.stdout and "versions" in result.stdout:
155  result = result.stdout.split()
156  result = result[result.index("versions:")+1:]

162  def compare_versions(self, v1, v2):
163  """Compare two version strings in the form '0.7.0' or v0.7.0"""
164 
165  # code from https://www.geeksforgeeks.org/compare-two-version-numbers
166 
167  arr1 = v1.replace("v","").split(".")
168  arr2 = v2.replace("v","").split(".")
169  n = len(arr1)
170  m = len(arr2)
171  arr1 = [int(i) for i in arr1]
172  arr2 = [int(i) for i in arr2]
173  if n > m:
174  for i in range(m, n):
175  arr2.append(0)
176  elif m > n:
177  for i in range(n, m):
178  arr1.append(0)
179  for i in range(len(arr1)):
180  if arr1[i] > arr2[i]:
181  return 1
182  elif arr2[i] > arr1[i]:
183  return -1
184  return 0
