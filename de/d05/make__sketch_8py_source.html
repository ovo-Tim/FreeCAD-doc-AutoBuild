41 def make_sketch(objects_list, autoconstraints=False, addTo=None,
42  delete=False, name="Sketch", radiusPrecision=-1, tol=1e-3):
43  """make_sketch(objects_list, [autoconstraints], [addTo], [delete],
44  [name], [radiusPrecision], [tol])
45 
46  Makes a Sketch objects_list with the given Draft objects.
47 
48  Parameters
49  ----------
50  objects_list: can be single or list of objects of Draft type objects,
51  Part::Feature, Part.Shape, or mix of them.
52 
53  autoconstraints(False): if True, coincident, horizontal and vertical
54  constraints will be added automatically.
55 
56  addTo(None): if set to an existing sketch, geometry will be added to it
57  instead of creating a new one.
58 
59  delete(False): if True, the original object will be deleted.
60  If set to a string "all" the object and all its linked object will be
61  deleted.
62 
63  name("Sketch"): the name for the new sketch object.
64 
65  radiusPrecision(-1): If <0, disable radius constraint. If =0, add individual
66  radius constraint. If >0, the radius will be rounded according to this
67  precision, and "Equal" constraint will be added to curve with equal
68  radius within precision.
69 
70  tol(1e-3): Tolerance used to check if the shapes are planar and coplanar.
71  Consider change to tol=-1 for a more accurate analysis.
72  """
73 
74  if not App.ActiveDocument:
75  App.Console.PrintError("No active document. Aborting\n")
76  return
77 
78  import Part
79 
80  if App.GuiUp:
81  v_dir = gui_utils.get_3d_view().getViewDirection()
82  else:

85  # lists to accumulate shapes with defined normal and undefined normal
86  shape_norm_yes = list()
87  shape_norm_no = list()
88

93  if isinstance(obj,Part.Shape):
94  shape = obj
95  elif not hasattr(obj, "Shape"):
96  App.Console.PrintError(translate("draft",
97  "No shape found") + "\n")
98  return None
99  else:
100  shape = obj.Shape
101 
102  if not DraftGeomUtils.is_planar(shape, tol):
103  App.Console.PrintError(translate("draft",
104  "All Shapes must be planar") + "\n")
105  return None
106 
107  if DraftGeomUtils.get_normal(shape, tol):
108  shape_norm_yes.append(shape)
109  else:
110  shape_norm_no.append(shape)
111 
112 
113  shapes_list = shape_norm_yes + shape_norm_no
114 
115  # test if all shapes are coplanar
116  if len(shape_norm_yes) >= 1:
117  for shape in shapes_list[1:]:
118  if not DraftGeomUtils.are_coplanar(shapes_list[0], shape, tol):
119  App.Console.PrintError(translate("draft",
120  "All Shapes must be coplanar") + "\n")
121  return None
122  # define sketch normal
123  normal = DraftGeomUtils.get_normal(shapes_list[0], tol)
124 
125  else:
126  # suppose all geometries are straight lines or points
127  points = [vertex.Point for shape in shapes_list for vertex in shape.Vertexes]
128  if len(points) >= 2:
129  poly = Part.makePolygon(points)
130  if not DraftGeomUtils.is_planar(poly, tol):
131  App.Console.PrintError(translate("draft",
132  "All Shapes must be coplanar") + "\n")
133  return None
134  normal = DraftGeomUtils.get_normal(poly, tol)
135  if not normal:
136  # all points aligned
137  poly_dir = poly.Edges[0].Curve.Direction
138  normal = (v_dir - v_dir.dot(poly_dir)*poly_dir).normalize()
139  normal = normal.negative()
140  else:
141  # only one point
142  normal = v_dir.negative()
143 
144 
145  if addTo:
146  nobj = addTo
147  else:
148  nobj = App.ActiveDocument.addObject("Sketcher::SketchObject", name)
149 
150  # Collect constraints and add in one go to improve performance
151  constraints = []
152  radiuses = {}
153 
154  def addRadiusConstraint(edge):
155  try:
156  if radiusPrecision<0:
157  return
158  if radiusPrecision==0:
159  constraints.append(Constraint("Radius",
160  nobj.GeometryCount-1, edge.Curve.Radius))
161  return
162  r = round(edge.Curve.Radius,radiusPrecision)
163  constraints.append(Constraint("Equal",
164  radiuses[r], nobj.GeometryCount-1))
165  except KeyError:
166  radiuses[r] = nobj.GeometryCount-1
167  constraints.append(Constraint("Radius", nobj.GeometryCount-1, r))
168  except AttributeError:
169  pass
170 
171  def convertBezier(edge):
172  if DraftGeomUtils.geomType(edge) == "BezierCurve":
173  return(edge.Curve.toBSpline(edge.FirstParameter,edge.LastParameter).toShape())
174  else:
175  return edge
176 
177 
178  axis = App.Vector(0, 0, 1).cross(normal)
179  if axis.Length > 1e-6:
180  axis.normalize()
181  elif normal.z >= 0:
182  axis = App.Vector(0, 0, 1)
183  else:
184  axis = App.Vector(0, 0, -1)
185  angle = math.degrees(DraftVecUtils.angle(normal, App.Vector(0, 0, 1)))
186  rotation = App.Rotation(axis, angle)
187 
188  point = shapes_list[0].Vertexes[0].Point
189  base = App.Vector(normal)
190  base.Length = point.dot(base.normalize()) # See https://forum.freecad.org/viewtopic.php?f=22&t=69304#p601149
191 
192  nobj.Placement = App.Placement(base, rotation)
193 
194  for obj in objects_list:
195  ok = False
196  tp = utils.get_type(obj)
197 
198  if tp == "Point":
199  # obj.Shape.copy() does not work properly for a Draft_Point.
200  # The coords of the point are multiplied by 2.
201  # We therefore create a Part Vertex instead.
202  shape = Part.Vertex(obj.Shape.Point)
203  if angle:
204  shape.rotate(App.Vector(0, 0, 0), axis, -angle)
205  point = Part.Point(shape.Point)
206  nobj.addGeometry(point)
207  ok = True
208 
209  elif tp == "Shape" or hasattr(obj, "Shape"):
210  shape = obj if tp == "Shape" else obj.Shape
211  for e in shape.Edges:
212  newedge = convertBezier(e)
213  nobj.addGeometry(DraftGeomUtils.orientEdge(
214  newedge, normal, make_arc=True))
215  addRadiusConstraint(newedge)
216  ok = True
217 
218  if ok and delete:
219  def delObj(obj):
220  if obj.InList:
221  App.Console.PrintWarning(translate("draft",
222  "Cannot delete object {} with dependency".format(obj.Label)) + "\n")
223  else:
224  obj.Document.removeObject(obj.Name)
225  try:
226  if delete == "all":
227  objs = [obj]
228  while objs:
229  obj = objs[0]
230  objs = objs[1:] + obj.OutList
231  delObj(obj)
232  else:
233  delObj(obj)
234  except Exception as ex:
235  App.Console.PrintWarning(translate("draft",
236  "Failed to delete object {}: {}".format(obj.Label, ex)) + "\n")
237 
238  nobj.addConstraint(constraints)
239  if autoconstraints:
240  nobj.detectMissingPointOnPointConstraints(utils.tolerance())
241  nobj.makeMissingPointOnPointCoincident(True)
242  nobj.detectMissingVerticalHorizontalConstraints(utils.tolerance())
243  nobj.makeMissingVerticalHorizontal(True)
244 
245  return nobj
