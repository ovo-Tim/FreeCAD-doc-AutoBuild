43 def BoxCircle(SketchFeature):
44  # Square with a circle centered at the upper right corner
45  top_edge = int(SketchFeature.GeometryCount)
46  right_edge = top_edge + 1
47  CreateRectangleSketch(SketchFeature, [0, 0], [2, 2])
48  top_midpoint = App.Vector(1, 2, 0)
49  right_midpoint = App.Vector(2, 1, 0)
50  circle = int(SketchFeature.GeometryCount)
51  SketchFeature.addGeometry(Part.Circle(App.Vector(0, 0, 0), App.Vector(0, 0, 1), 1), False)
52  SketchFeature.addConstraint(Sketcher.Constraint("Radius", circle, 1))
53  SketchFeature.addConstraint(Sketcher.Constraint("Coincident", top_edge, 2, circle, 3))
54  # Since the circle center is coincident with the corner, there are three coincident points
55  # and thus the simpler fillet() call would get confused. Instead we'll need to point at the two
56  # lines and their midpoints
57  SketchFeature.fillet(top_edge, right_edge, top_midpoint, right_midpoint, 0.25, True, True)

218  def testDistance(self):
219  SketchFeature = self.Doc.addObject("Sketcher::SketchObject", "Distance")
220  # We'll end up implicitly testing line length constraints as well since that's
221  # what CreateRectangleSketch uses to enforce side length. If the side length doesn't
222  # switch to a point-to-point distance constraint with the original corner as expected,
223  # the point won't end up at its expected destination.
224  CreateRectangleSketch(SketchFeature, [0, 0], [2, 2])
225 
226  point = int(SketchFeature.GeometryCount)

229  SketchFeature.addConstraint(Sketcher.Constraint("DistanceY", 0, 2, point, 1, 0))
230 
231  self.Doc.recompute()
232  self.assertTrue(SketchFeature.FullyConstrained)
233  SketchFeature.fillet(0, 2, 0.25, True, True)
234  SketchFeature.addConstraint(Sketcher.Constraint("Radius", 5, 0.25))
235  self.Doc.recompute()
236  # If any constraints disappeared then we won't be fully constrained
237  self.assertTrue(SketchFeature.FullyConstrained)
