98  "https://raw.githubusercontent.com/FreeCAD/FreeCAD-addons/master/addonflags.json", 5000

116  self._process_deprecated(j["deprecated"])

126  def _process_deprecated(self, deprecated_addons):
127  """Parse the section on deprecated addons"""
128 
129  fc_major = int(FreeCAD.Version()[0])
130  fc_minor = int(FreeCAD.Version()[1])
131  for item in deprecated_addons:
132  if "as_of" in item and "name" in item:
133  try:
134  version_components = item["as_of"].split(".")
135  major = int(version_components[0])
136  if len(version_components) > 1:
137  minor = int(version_components[1])
138  else:
139  minor = 0
140  if major < fc_major or (major == fc_major and minor <= fc_minor):
141  if "kind" not in item or item["kind"] == "mod":
142  self.obsolete.append(item["name"])
143  elif item["kind"] == "macro":
144  self.macros_reject_list.append(item["name"])
145  else:
146  FreeCAD.Console.PrintMessage(
147  f'Unrecognized Addon kind {item["kind"]} in deprecation list.'
148  )
149  except ValueError:
150  FreeCAD.Console.PrintMessage(
151  f"Failed to parse version from {item['name']}, version {item['as_of']}"
152  )

154  def _get_custom_addons(self):
155 
156  # querying custom addons first
157  addon_list = (
158  FreeCAD.ParamGet("User parameter:BaseApp/Preferences/Addons")
159  .GetString("CustomRepositories", "")
160  .split("\n")
161  )
162  custom_addons = []
163  for addon in addon_list:
164  if " " in addon:
165  addon_and_branch = addon.split(" ")
166  custom_addons.append({"url": addon_and_branch[0], "branch": addon_and_branch[1]})
167  else:
168  custom_addons.append({"url": addon, "branch": "master"})
169  for addon in custom_addons:
170  if self.current_thread.isInterruptionRequested():
171  return
172  if addon and addon["url"]:
173  if addon["url"][-1] == "/":
174  addon["url"] = addon["url"][0:-1] # Strip trailing slash
175  addon["url"] = addon["url"].split(".git")[0] # Remove .git
176  name = addon["url"].split("/")[-1]
177  if name in self.package_names:
178  # We already have something with this name, skip this one
179  FreeCAD.Console.PrintWarning(
180  translate("AddonsInstaller", "WARNING: Duplicate addon {} ignored").format(
181  name
182  )
183  )
184  continue
185  FreeCAD.Console.PrintLog(
186  f"Adding custom location {addon['url']} with branch {addon['branch']}\n"
187  )
188  self.package_names.append(name)
189  addondir = os.path.join(self.moddir, name)
190  if os.path.exists(addondir) and os.listdir(addondir):
191  state = Addon.Status.UNCHECKED
192  else:
193  state = Addon.Status.NOT_INSTALLED
194  repo = Addon(name, addon["url"], state, addon["branch"])
195  md_file = os.path.join(addondir, "package.xml")
196  if os.path.isfile(md_file):
197  try:
198  repo.installed_metadata = MetadataReader.from_file(md_file)
199  repo.installed_version = repo.installed_metadata.version
200  repo.updated_timestamp = os.path.getmtime(md_file)
201  repo.verify_url_and_branch(addon["url"], addon["branch"])
202  except xml.etree.ElementTree.ParseError:
203  fci.Console.PrintWarning(
204  "An invalid or corrupted package.xml file was installed for"
205  )
206  fci.Console.PrintWarning(
207  f" custom addon {self.name}... ignoring the bad data.\n"
208  )
209 
210  self.addon_repo.emit(repo)

212  def _get_official_addons(self):
213  # querying official addons
214  p = NetworkManager.AM_NETWORK_MANAGER.blocking_get(
215  "https://raw.githubusercontent.com/FreeCAD/FreeCAD-addons/master/.gitmodules", 5000
216  )
217  if not p:
218  return
219  p = p.data().decode("utf8")
220  p = re.findall(
221  (
222  r'(?m)\[submodule\s*"(?P<name>.*)"\]\s*'
223  r"path\s*=\s*(?P<path>.+)\s*"
224  r"url\s*=\s*(?P<url>https?://.*)\s*"
225  r"(branch\s*=\s*(?P<branch>[^\s]*)\s*)?"
226  ),
227  p,
228  )
229  for name, _, url, _, branch in p:
230  if self.current_thread.isInterruptionRequested():
231  return
232  if name in self.package_names:
233  # We already have something with this name, skip this one
234  continue
235  self.package_names.append(name)
236  if branch is None or len(branch) == 0:
237  branch = "master"
238  url = url.split(".git")[0]
239  addondir = os.path.join(self.moddir, name)
240  if os.path.exists(addondir) and os.listdir(addondir):
241  # make sure the folder exists and it contains files!
242  state = Addon.Status.UNCHECKED
243  else:
244  state = Addon.Status.NOT_INSTALLED
245  repo = Addon(name, url, state, branch)
246  md_file = os.path.join(addondir, "package.xml")
247  if os.path.isfile(md_file):
248  try:
249  repo.installed_metadata = MetadataReader.from_file(md_file)
250  repo.installed_version = repo.installed_metadata.version
251  repo.updated_timestamp = os.path.getmtime(md_file)
252  repo.verify_url_and_branch(url, branch)
253  except xml.etree.ElementTree.ParseError:
254  fci.Console.PrintWarning(
255  "An invalid or corrupted package.xml file was installed for"
256  )
257  fci.Console.PrintWarning(f" addon {self.name}... ignoring the bad data.\n")
258 
259  if name in self.py2only:
260  repo.python2 = True
261  if name in self.mod_reject_list:
262  repo.rejected = True
263  if name in self.obsolete:
264  repo.obsolete = True
265  self.addon_repo.emit(repo)

267  def _retrieve_macros_from_git(self):
268  """Retrieve macros from FreeCAD-macros.git
269 
270  Emits a signal for each macro in
271  https://github.com/FreeCAD/FreeCAD-macros.git
272  """
273 
274  macro_cache_location = utils.get_cache_file_name("Macros")
275 
276  if not self.git_manager:
277  message = translate(
278  "AddonsInstaller",
279  "Git is disabled, skipping Git macros",
280  )
281  FreeCAD.Console.PrintWarning(message + "\n")
282  return
283 
284  update_succeeded = self._update_local_git_repo()
285  if not update_succeeded:
286  return
287 
288  n_files = 0
289  for _, _, filenames in os.walk(macro_cache_location):
290  n_files += len(filenames)
291  counter = 0
292  for dirpath, _, filenames in os.walk(macro_cache_location):
293  counter += 1
294  if self.current_thread.isInterruptionRequested():
295  return
296  if ".git" in dirpath:
297  continue

299  if self.current_thread.isInterruptionRequested():

303  if macro.name in self.package_names:

307  continue # We already have a macro with this name
308  self.package_names.append(macro.name)
309  macro.on_git = True
310  macro.src_filename = os.path.join(dirpath, filename)
311  macro.fill_details_from_file(macro.src_filename)
312  repo = Addon.from_macro(macro)
313  FreeCAD.Console.PrintLog(f"Found macro {repo.name}\n")
314  repo.url = "https://github.com/FreeCAD/FreeCAD-macros.git"
315  utils.update_macro_installation_details(repo)

318  def _update_local_git_repo(self) -> bool:
319  macro_cache_location = utils.get_cache_file_name("Macros")
320  try:
321  if os.path.exists(macro_cache_location):
322  if not os.path.exists(os.path.join(macro_cache_location, ".git")):
323  FreeCAD.Console.PrintWarning(
324  translate(
325  "AddonsInstaller",
326  "Attempting to change non-Git Macro setup to use Git\n",
327  )
328  )
329  self.git_manager.repair(
330  "https://github.com/FreeCAD/FreeCAD-macros.git",
331  macro_cache_location,
332  )
333  self.git_manager.update(macro_cache_location)
334  else:
335  self.git_manager.clone(
336  "https://github.com/FreeCAD/FreeCAD-macros.git",
337  macro_cache_location,
338  )
339  except GitFailed as e:
340  FreeCAD.Console.PrintMessage(
341  translate(
342  "AddonsInstaller",
343  "An error occurred updating macros from GitHub, trying clean checkout...",
344  )
345  + f":\n{e}\n"
346  )
347  FreeCAD.Console.PrintMessage(f"{macro_cache_location}\n")
348  FreeCAD.Console.PrintMessage(
349  translate("AddonsInstaller", "Attempting to do a clean checkout...") + "\n"
350  )
351  try:
352  os.chdir(
353  os.path.join(macro_cache_location, "..")
354  ) # Make sure we are not IN this directory
355  shutil.rmtree(macro_cache_location, onerror=self._remove_readonly)
356  self.git_manager.clone(
357  "https://github.com/FreeCAD/FreeCAD-macros.git",
358  macro_cache_location,
359  )
360  FreeCAD.Console.PrintMessage(
361  translate("AddonsInstaller", "Clean checkout succeeded") + "\n"
362  )
363  except GitFailed as e2:
364  # The Qt Python translation extractor doesn't support splitting this string (yet)
365  # pylint: disable=line-too-long
366  FreeCAD.Console.PrintWarning(
367  translate(
368  "AddonsInstaller",
369  "Failed to update macros from GitHub -- try clearing the Addon Manager's cache.",
370  )
371  + f":\n{str(e2)}\n"
372  )
373  return False
374  return True

409  (macname not in self.macros_reject_list)

411  and (macname not in macro_names)

416  FreeCAD.Console.PrintLog(

565 class UpdateChecker:
566  """A utility class used by the CheckWorkbenchesForUpdatesWorker class. Each function is
567  designed for a specific Addon type, and modifies the passed-in Addon with the determined
568  update status."""
569 
570  def __init__(self):
571  self.basedir = FreeCAD.getUserAppDataDir()
572  self.moddir = os.path.join(self.basedir, "Mod")
573  self.git_manager = initialize_git()
574 
575  def override_mod_directory(self, moddir):
576  """Primarily for use when testing, sets an alternate directory to use for mods"""
577  self.moddir = moddir
578 
579  def check_workbench(self, wb):
580  """Given a workbench Addon wb, check it for updates using git. If git is not
581  available, does nothing."""
582  if not self.git_manager:
583  wb.set_status(Addon.Status.CANNOT_CHECK)
584  return
585  clonedir = os.path.join(self.moddir, wb.name)
586  if os.path.exists(clonedir):
587  # mark as already installed AND already checked for updates
588  if not os.path.exists(os.path.join(clonedir, ".git")):
589  with wb.git_lock:
590  self.git_manager.repair(wb.url, clonedir)
591  with wb.git_lock:
592  try:
593  status = self.git_manager.status(clonedir)
594  if "(no branch)" in status:
595  # By definition, in a detached-head state we cannot
596  # update, so don't even bother checking.
597  wb.set_status(Addon.Status.NO_UPDATE_AVAILABLE)
598  wb.branch = self.git_manager.current_branch(clonedir)
599  return
600  except GitFailed as e:
601  FreeCAD.Console.PrintWarning(
602  "AddonManager: "
603  + translate(
604  "AddonsInstaller",
605  "Unable to fetch Git updates for workbench {}",
606  ).format(wb.name)
607  + "\n"
608  )
609  FreeCAD.Console.PrintWarning(str(e) + "\n")
610  wb.set_status(Addon.Status.CANNOT_CHECK)
611  else:
612  try:
613  if self.git_manager.update_available(clonedir):
614  wb.set_status(Addon.Status.UPDATE_AVAILABLE)
615  else:
616  wb.set_status(Addon.Status.NO_UPDATE_AVAILABLE)
617  except GitFailed:
618  FreeCAD.Console.PrintWarning(
619  translate("AddonsInstaller", "Git status failed for {}").format(wb.name)
620  + "\n"
621  )
622  wb.set_status(Addon.Status.CANNOT_CHECK)

802  def _process_queue(self, num_macros) -> bool:
803  """Spools up six network connections and downloads the macro code. Returns True if
804  it was interrupted by user request, or False if it ran to completion."""
805 
806  # Emulate QNetworkAccessManager and spool up six connections:
807  for _ in range(6):
808  self.update_and_advance(None)
809 
810  current_thread = QtCore.QThread.currentThread()
811  while True:
812  if current_thread.isInterruptionRequested():
813  for worker in self.workers:
814  worker.blockSignals(True)
815  worker.requestInterruption()
816  if not worker.wait(100):
817  FreeCAD.Console.PrintWarning(
818  translate(
819  "AddonsInstaller",
820  "Addon Manager: a worker process failed to halt ({name})",
821  ).format(name=worker.macro.name)
822  + "\n"
823  )
824  return True
825  # Ensure our signals propagate out by running an internal thread-local event loop
826  QtCore.QCoreApplication.processEvents()
827  with self.lock:
828  if self.counter >= num_macros:
829  break
830  time.sleep(0.1)
831  return False

833  def update_and_advance(self, repo: Optional[Addon]) -> None:
834  """Emit the updated signal and launch the next item from the queue."""
835  if repo is not None:
836  if repo.macro.name not in self.failed:
837  self.update_macro.emit(repo)
838  self.repo_queue.task_done()
839  with self.lock:
840  self.counter += 1
841 
842  if QtCore.QThread.currentThread().isInterruptionRequested():
843  return
844 
845  if repo is not None:
846  message = translate("AddonsInstaller", "Caching {} macro").format(repo.display_name)
847  else:
848  message = translate("AddonsInstaller", "Caching macros")
849  self.progress_made.emit(message, len(self.repos) - self.repo_queue.qsize(), len(self.repos))
850 
851  try:
852  next_repo = self.repo_queue.get_nowait()
853  worker = GetMacroDetailsWorker(next_repo)
854  worker.finished.connect(lambda: self.update_and_advance(next_repo))
855  with self.lock:
856  self.workers.append(worker)
857  self.terminators.append(
858  QtCore.QTimer.singleShot(10000, lambda: self.terminate(worker))
859  )
860  worker.start()
861  except queue.Empty:
862  pass

865  """Shut down all running workers and exit the thread"""
866  if not worker.isFinished():
867  macro_name = worker.macro.name

869  translate(
870  "AddonsInstaller",
871  "Timeout while fetching metadata for macro {}",
872  ).format(macro_name)
873  + "\n"
874  )

876  worker.requestInterruption()
877  worker.wait(100)
878  if worker.isRunning():
879  FreeCAD.Console.PrintError(
880  translate(
881  "AddonsInstaller",
882  "Failed to kill process for macro {}!\n",
883  ).format(macro_name)
884  )
885  with self.lock:
886  self.failed.append(macro_name)
