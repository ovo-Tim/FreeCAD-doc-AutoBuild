169 void SoWidgetShape::computeBBox(SoAction *action, SbBox3f &box, SbVec3f &center)
170 {
171  // ignore if node is empty
172  if (this->image.isNull())
173  return;
174 
175  SbVec3f v0, v1, v2, v3;
176  // this will cause a cache dependency on the view volume,
177  // model matrix and viewport.
178  this->getQuad(action->getState(), v0, v1, v2, v3);
179 
180  box.makeEmpty();
181  box.extendBy(v0);
182  box.extendBy(v1);
183  box.extendBy(v2);
184  box.extendBy(v3);
185  center = box.getCenter();

188 // Calculates the quad in 3D.
189 void
190 SoWidgetShape::getQuad(SoState * state, SbVec3f & v0, SbVec3f & v1,
191  SbVec3f & v2, SbVec3f & v3)
192 {
193  SbVec3f nilpoint(0.0f, 0.0f, 0.0f);
194  const SbMatrix & mat = SoModelMatrixElement::get(state);
195  mat.multVecMatrix(nilpoint, nilpoint);
196 
197  const SbViewVolume &vv = SoViewVolumeElement::get(state);
198 
199  SbVec3f screenpoint;
200  vv.projectToScreen(nilpoint, screenpoint);
201 
202  const SbViewportRegion & vp = SoViewportRegionElement::get(state);
203  SbVec2s vpsize = vp.getViewportSizePixels();
204 
205  // find normalized width and height of image
206  float nw = (float)this->image.width();
207  nw /= (float)vpsize[0];
208  float nh = (float)this->image.height();
209  nh /= (float)vpsize[1];
210 
211  // need only half the width
212  nw *= 0.5f;
213  nh *= 0.5f;
214 
215  SbVec2f n0, n1, n2, n3;
216 
217  n0 = SbVec2f(screenpoint[0]-nw, screenpoint[1]-nh);
218  n1 = SbVec2f(screenpoint[0]+nw, screenpoint[1]-nh);
219  n2 = SbVec2f(screenpoint[0]+nw, screenpoint[1]+nh);
220  n3 = SbVec2f(screenpoint[0]-nw, screenpoint[1]+nh);
221 
222  // get distance from nilpoint to camera plane
223  float dist = -vv.getPlane(0.0f).getDistance(nilpoint);
224 
225  // find the four image points in the plane
226  v0 = vv.getPlanePoint(dist, n0);
227  v1 = vv.getPlanePoint(dist, n1);
228  v2 = vv.getPlanePoint(dist, n2);
229  v3 = vv.getPlanePoint(dist, n3);
230 
231  // transform back to object space
232  SbMatrix inv = mat.inverse();
233  inv.multVecMatrix(v0, v0);
234  inv.multVecMatrix(v1, v1);
235  inv.multVecMatrix(v2, v2);
236  inv.multVecMatrix(v3, v3);
237 }

239 void SoWidgetShape::generatePrimitives(SoAction *action)
240 {
241  if (this->image.isNull())
242  return;
243 
244  SoState *state = action->getState();
245  state->push();
246 
247  SbVec2s size;
248  SbVec3f v0, v1, v2, v3;
249  this->getQuad(action->getState(), v0, v1, v2, v3);
250 
251  SbVec3f n = (v1-v0).cross(v2-v0);
252  n.normalize();
253 
254  this->beginShape(action, SoShape::QUADS);
255  SoPrimitiveVertex vertex;
256  vertex.setNormal(n);
257 
258  vertex.setTextureCoords(SbVec2f(0,0));
259  vertex.setPoint(v0);
260  this->shapeVertex(&vertex);
261 
262  vertex.setTextureCoords(SbVec2f(1,0));
263  vertex.setPoint(v1);
264  this->shapeVertex(&vertex);
265 
266  vertex.setTextureCoords(SbVec2f(1,1));
267  vertex.setPoint(v2);
268  this->shapeVertex(&vertex);
269 
270  vertex.setTextureCoords(SbVec2f(0,1));
271  vertex.setPoint(v3);
272  this->shapeVertex(&vertex);
273 
274  this->endShape();
275 
276  state->pop();
277 }
