63 def setup(doc=None, solvertype="ccxtools"):
64 
65  # init FreeCAD document
66  if doc is None:
67  doc = init_doc()
68 
69  # explanation object
70  # just keep the following line and change text string in get_explanation method
71  manager.add_explanation_obj(doc, get_explanation(manager.get_header(get_information())))
72 
73  # setup box static, add a fixed, force and a pressure constraint
74  doc = setup_boxanalysisbase(doc, solvertype)
75  geom_obj = doc.Box
76  analysis = doc.Analysis
77 
78  # solver
79  if solvertype == "ccxtools":
80  solver_obj = ObjectsFem.makeSolverCalculiXCcxTools(doc, "CalculiXCcxTools")
81  solver_obj.WorkingDir = ""
82  elif solvertype == "elmer":
83  solver_obj = ObjectsFem.makeSolverElmer(doc, "SolverElmer")
84  ObjectsFem.makeEquationElasticity(doc, solver_obj)
85  else:
86  FreeCAD.Console.PrintWarning(
87  "Unknown or unsupported solver type: {}. "
88  "No solver object was created.\n".format(solvertype)
89  )
90  if solvertype == "ccxtools":
91  solver_obj.SplitInputWriter = False
92  solver_obj.AnalysisType = "static"
93  solver_obj.GeometricalNonlinearity = "linear"
94  solver_obj.ThermoMechSteadyState = False
95  solver_obj.MatrixSolverType = "default"
96  solver_obj.IterationsControlParameterTimeUse = False
97  analysis.addObject(solver_obj)
98 
99  # constraint fixed
100  con_fixed = ObjectsFem.makeConstraintFixed(doc, "FemConstraintFixed")
101  con_fixed.References = [(geom_obj, "Face1")]
102  analysis.addObject(con_fixed)
103 
104  # constraint force
105  con_force = ObjectsFem.makeConstraintForce(doc, "FemConstraintForce")
106  con_force.References = [(geom_obj, "Face6")]
107  con_force.Force = "40000.0 N"
108  con_force.Direction = (geom_obj, ["Edge5"])
109  con_force.Reversed = True
110  analysis.addObject(con_force)
111 
112  # constraint pressure
113  con_pressure = ObjectsFem.makeConstraintPressure(doc, name="FemConstraintPressure")
114  con_pressure.References = [(geom_obj, "Face2")]
115  con_pressure.Pressure = "1000.0 MPa"
116  con_pressure.Reversed = False
117  analysis.addObject(con_pressure)
118 
119  doc.recompute()
120  return doc
