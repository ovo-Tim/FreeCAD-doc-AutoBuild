348 Base::Vector3d BaseGeom::getMidPoint()
349 {
350  // Midpoint calculation - additional details here: https://forum.freecad.org/viewtopic.php?f=35&t=59582
351 
352  BRepAdaptor_Curve curve(occEdge);
353 
354  // As default, set the midpoint curve parameter value by simply averaging start point and end point values
355  double midParam = (curve.FirstParameter() + curve.LastParameter())/2.0;
356 
357  // GCPnts_AbscissaPoint allows us to compute the parameter value depending on the distance along a curve.
358  // In this case we want the curve parameter value for the half of the whole curve length,
359  // thus GCPnts_AbscissaPoint::Length(curve)/2 is the distance to go from the start point.
360  GCPnts_AbscissaPoint abscissa(Precision::Confusion(), curve, GCPnts_AbscissaPoint::Length(curve)/2.0,
361  curve.FirstParameter());
362  if (abscissa.IsDone()) {
363  // The computation was successful - otherwise keep the average, it is better than nothing
364  midParam = abscissa.Parameter();
365  }
366 
367  // Now compute coordinates of the point on curve for curve parameter value equal to midParam
368  BRepLProp_CLProps props(curve, midParam, 0, Precision::Confusion());
369  const gp_Pnt &point = props.Value();
370 
371  return Base::Vector3d(point.X(), point.Y(), point.Z());
372 }

374 std::vector<Base::Vector3d> BaseGeom::getQuads()
375 {
376  std::vector<Base::Vector3d> result;
377  BRepAdaptor_Curve adapt(occEdge);
378  double u = adapt.FirstParameter();
379  double v = adapt.LastParameter();
380  double range = v - u;
381  double q1 = u + (range / 4.0);
382  double q2 = u + (range / 2.0);
383  double q3 = u + (3.0 * range / 4.0);
384  BRepLProp_CLProps prop(adapt, q1, 0, Precision::Confusion());
385  const gp_Pnt& p1 = prop.Value();
386  result.emplace_back(p1.X(), p1.Y(), 0.0);
387  prop.SetParameter(q2);
388  const gp_Pnt& p2 = prop.Value();
389  result.emplace_back(p2.X(), p2.Y(), 0.0);
390  prop.SetParameter(q3);
391  const gp_Pnt& p3 = prop.Value();
392  result.emplace_back(p3.X(), p3.Y(), 0.0);
393  return result;
394 }

481 // this should not return nullptr as things will break later on.
482 // regular geometry is stored scaled, but cosmetic geometry is stored in 1:1 scale, so the crazy edge
483 // check is not appropriate.
484 BaseGeomPtr BaseGeom::baseFactory(TopoDS_Edge edge, bool isCosmetic)
485 {
486  if (edge.IsNull()) {
487  Base::Console().Message("BG::baseFactory - input edge is NULL \n");
488  }
489  //weed out rubbish edges before making geometry
490  if (!isCosmetic && !validateEdge(edge)) {
491  return nullptr;
492  }
493 
494  BaseGeomPtr result = std::make_shared<Generic> (edge);
495 
496  BRepAdaptor_Curve adapt(edge);
497  switch(adapt.GetType()) {
498  case GeomAbs_Circle: {
499  double f = adapt.FirstParameter();
500  double l = adapt.LastParameter();
501  gp_Pnt s = adapt.Value(f);
502  gp_Pnt e = adapt.Value(l);
503 
504  //don't understand this test.
505  //if first to last is > 1 radian? are circles parameterize by rotation angle?
506  //if start and end points are close?
507  if (fabs(l-f) > 1.0 && s.SquareDistance(e) < 0.001) {
508  result = std::make_shared<Circle>(edge);
509  } else {
510  result = std::make_shared<AOC>(edge);
511  }
512  } break;
513  case GeomAbs_Ellipse: {
514  double f = adapt.FirstParameter();
515  double l = adapt.LastParameter();
516  gp_Pnt s = adapt.Value(f);
517  gp_Pnt e = adapt.Value(l);
518  if (fabs(l-f) > 1.0 && s.SquareDistance(e) < 0.001) {
519  result = std::make_shared<Ellipse>(edge);
520  } else {
521  result = std::make_shared<AOE>(edge);
522  }
523  } break;
524  case GeomAbs_BezierCurve: {
525  Handle(Geom_BezierCurve) bez = adapt.Bezier();
526  //if (bez->Degree() < 4) {
527  result = std::make_shared<BezierSegment>(edge);
528  if (edge.Orientation() == TopAbs_REVERSED) {
529  result->reversed = true;
530  }
531 
532  // OCC is quite happy with Degree > 3 but QtGui handles only 2, 3
533  } break;
534  case GeomAbs_BSplineCurve: {
535  TopoDS_Edge circEdge;
536 
537  bool isArc = false;
538  try {
539  BSplinePtr bspline = std::make_shared<BSpline>(edge);
540  if (bspline->isLine()) {
541  result = std::make_shared<Generic>(edge);
542  } else if (bspline->isCircle()) {
543  circEdge = bspline->asCircle(isArc);
544  if (circEdge.IsNull()) {
545  result = bspline;
546  } else {
547  if (isArc) {
548  result = std::make_shared<AOC>(circEdge);
549  } else {
550  result = std::make_shared<Circle>(circEdge);
551  }
552  }
553  } else {
554 // Base::Console().Message("Geom::baseFactory - circEdge is Null\n");
555  result = bspline;
556  }
557  break;
558  }
559  catch (const Standard_Failure& e) {
560  Base::Console().Log("Geom::baseFactory - OCC error - %s - while making spline\n",
561  e.GetMessageString());
562  break;
563  }
564  catch (...) {
565  Base::Console().Log("Geom::baseFactory - unknown error occurred while making spline\n");
566  break;
567  } break;
568  } // end bspline case
569  default: {
570  result = std::make_unique<Generic>(edge);
571  } break;
572  }
573 
574  return result;
575 }

582 TopoDS_Edge BaseGeom::completeEdge(const TopoDS_Edge &edge) {
583  // Extend given edge so we can get intersections even outside its boundaries
584  try {
585  BRepAdaptor_Curve curve(edge);
586  switch (curve.GetType()) {
587  case GeomAbs_Line:
588  // Edge longer than 10m is considered "crazy", thus limit intersection(s) to this perimeter
589  return BRepBuilderAPI_MakeEdge(curve.Line(), -10000.0, +10000.0);
590  case GeomAbs_Circle:
591  // If an arc of circle was provided, return full circle
592  return BRepBuilderAPI_MakeEdge(curve.Circle());
593  case GeomAbs_Ellipse:
594  // If an arc of ellipse was provided, return full ellipse
595  return BRepBuilderAPI_MakeEdge(curve.Ellipse());
596  default:
597  // Currently we are not extrapolating B-splines, though it is technically possible
598  return BRepBuilderAPI_MakeEdge(curve.Curve().Curve());
599  }
600  }
601  catch (Standard_Failure &e) {
602  Base::Console().Error("BaseGeom::completeEdge OCC error: %s\n", e.GetMessageString());
603  }
604 
605  return TopoDS_Edge();
606 }

608 std::vector<Base::Vector3d> BaseGeom::intersection(TechDraw::BaseGeomPtr geom2)
609 {
610  // find intersection vertex(es) between two edges
611  // call: interPoints = line1.intersection(line2);
612  std::vector<Base::Vector3d> interPoints;
613 
614  TopoDS_Edge edge1 = completeEdge(this->getOCCEdge());
615  if (edge1.IsNull()) {
616  return interPoints;
617  }
618 
619  TopoDS_Edge edge2 = completeEdge(geom2->getOCCEdge());
620  if (edge2.IsNull()) {
621  return interPoints;
622  }
623 
624  FCBRepAlgoAPI_Section sectionOp(edge1, edge2);
625  sectionOp.SetFuzzyValue(FUZZYADJUST*EWTOLERANCE);
626  sectionOp.SetNonDestructive(true);
627 
628  sectionOp.Build();
629  if (!sectionOp.HasErrors()) {
630  TopoDS_Shape sectionShape = sectionOp.Shape();
631  if (!sectionShape.IsNull()) {

1567 bool GeometryUtils::getCircleParms(TopoDS_Edge occEdge, double& radius, Base::Vector3d& center, bool& isArc)

1570  double curveLimit = EWTOLERANCE;
1571  BRepAdaptor_Curve c(occEdge);
1572  Handle(Geom_BSplineCurve) spline = c.BSpline();
1573  double f, l;
1574  f = c.FirstParameter();
1575  l = c.LastParameter();
1576  double parmRange = fabs(l - f);
1577  double parmStep = parmRange/testCount;
1578  std::vector<double> curvatures;
1579  std::vector<gp_Pnt> centers;
1580  gp_Pnt curveCenter;
1581  double sumCurvature = 0;
1582  Base::Vector3d sumCenter, valueAt;
1583  try {
1584  GeomLProp_CLProps prop(spline, f, 3, Precision::Confusion());
1585 
1586  // check only the interior points of the edge
1587  for (int i = 1; i < (testCount - 1); i++) {
1588  prop.SetParameter(parmStep * i);
1589  curvatures.push_back(prop.Curvature());
1590  sumCurvature += prop.Curvature();
1591  prop.CentreOfCurvature(curveCenter);
1592  centers.push_back(curveCenter);
1593  sumCenter += DrawUtil::toVector3d(curveCenter);
1594  }
1595 
1596  }
1597  catch (Standard_Failure&) {
1598  return false;
1599  }
1600  Base::Vector3d avgCenter = sumCenter/ centers.size();
1601 
1602  double avgCurve = sumCurvature/ centers.size();
1603  double errorCurve = 0;
1604  // sum the errors in curvature
1605  for (auto& cv: curvatures) {
1606  errorCurve += avgCurve - cv;
1607  }
1608 
1609  double errorCenter{0};
1610  for (auto& observe : centers) {
1611  auto error = (DU::toVector3d(observe)- avgCenter).Length();
1612  errorCenter += error;
1613  }
1614 
1615  // calculate average error in curvature. we are only interested in the magnitude of the error
1616  errorCurve = fabs(errorCurve / curvatures.size());
1617  // calculate the average error in center of curvature
1618  errorCenter = errorCenter / curvatures.size();
1619  auto edgeLong = edgeLength(occEdge);
1620  double centerLimit = edgeLong * 0.01;
1621 
1622  isArc = !c.IsClosed();
1623  bool isCircle(false);
1624  if ( errorCurve <= curveLimit &&
1625  errorCenter <= centerLimit) {
1626  isCircle = true;
1627  radius = 1.0/avgCurve;
1628  center = avgCenter;
1629  }
1630
