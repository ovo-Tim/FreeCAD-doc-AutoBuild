368 void MDIViewPage::print(QPrinter* printer)
369 {
370  // As size of the render area paperRect() should be used. When performing a real
371  // print pageRect() may also work but the output is cropped at the bottom part.
372  // So, independent whether pageRect() or paperRect() is used there is no scaling effect.
373  // However, when using a different paper size as set in the drawing template (e.g.
374  // DIN A5 instead of DIN A4) then the output is scaled.
375  //
376  // When creating a PDF file there seems to be no difference between pageRect() and
377  // paperRect().
378  //
379  // When showing the preview of a print paperRect() must be used because with pageRect()
380  // a certain scaling effect can be observed and the content becomes smaller.
381 
382  QPaintEngine::Type paintType = printer->paintEngine()->type();
383  auto pageAttr = PagePrinter::getPaperAttributes(getViewProviderPage());
384  if (printer->outputFormat() == QPrinter::NativeFormat) {
385  QPageSize::PageSizeId psPrtSetting = printer->pageLayout().pageSize().id();
386 
387  // for the preview a 'Picture' paint engine is used which we don't
388  // care if it uses wrong printer settings
389  bool doPrint = paintType != QPaintEngine::Picture;
390 
391  if (doPrint && printer->pageLayout().orientation() != pageAttr.orientation()) {
392  int ret = QMessageBox::warning(
393  this, tr("Different orientation"),
394  tr("The printer uses a different orientation than the drawing.\n"
395  "Do you want to continue?"),
396  QMessageBox::Yes | QMessageBox::No);
397  if (ret != QMessageBox::Yes) {
398  return;
399  }
400  }
401  if (doPrint && psPrtSetting != pageAttr.pageSize()) {
402  int ret = QMessageBox::warning(
403  this, tr("Different paper size"),
404  tr("The printer uses a different paper size than the drawing.\n"
405  "Do you want to continue?"),
406  QMessageBox::Yes | QMessageBox::No);
407  if (ret != QMessageBox::Yes) {
408  return;
409  }
410  }
411  }
412 
413  PagePrinter::print(getViewProviderPage(), printer);
414 }

648 //really "onTreeSelectionChanged"
649 void MDIViewPage::onSelectionChanged(const Gui::SelectionChanges& msg)
650 {
651  blockSceneSelection(true);
652 
653  if (msg.Type == Gui::SelectionChanges::ClrSelection || msg.Type == Gui::SelectionChanges::SetSelection) {
654  clearSceneSelection();
655 
656  if (msg.Type == Gui::SelectionChanges::SetSelection) {
657  std::vector<Gui::SelectionObject> selObjs = Gui::Selection().getSelectionEx(msg.pDocName);
658  for (auto &so : selObjs) {
659  App::DocumentObject *docObj = so.getObject();
660  if (docObj->isDerivedFrom(TechDraw::DrawView::getClassTypeId())) {
661  selectQGIView(docObj, true, so.getSubNames());
662  }
663  }
664  }
665  }
666  else if (msg.Type == Gui::SelectionChanges::AddSelection || msg.Type == Gui::SelectionChanges::RmvSelection) {
667  App::DocumentObject *docObj = msg.Object.getSubObject();
668  if (docObj->isDerivedFrom(TechDraw::DrawView::getClassTypeId())) {
669  bool isSelected = msg.Type != Gui::SelectionChanges::RmvSelection;
670  selectQGIView(docObj, isSelected, std::vector(1, std::string(msg.pSubName ? msg.pSubName : "")));
671  }
672  }
673 
674  blockSceneSelection(false);
675 }

972 bool MDIViewPage::compareSelections(std::vector<Gui::SelectionObject> treeSel,
973  QList<QGraphicsItem*> sceneSel)
974 {
975  if (treeSel.empty() && sceneSel.empty()) {
976  return true;
977  }
978  else if (treeSel.empty() && !sceneSel.empty()) {
979  return false;
980  }
981  else if (!treeSel.empty() && sceneSel.empty()) {
982  return false;
983  }
984 
985  int treeCount = 0;
986  int subCount = 0;
987  int sceneCount = 0;
988  int ppCount = 0;
989 
990  std::vector<std::string> treeNames;
991  std::vector<std::string> sceneNames;
992 
993  for (auto tn : treeSel) {
994  if (tn.getObject()->isDerivedFrom(TechDraw::DrawView::getClassTypeId())) {
995  std::string s = tn.getObject()->getNameInDocument();
996  treeNames.push_back(s);
997  subCount += tn.getSubNames().size();
998  }
999  }
1000  std::sort(treeNames.begin(), treeNames.end());
1001  treeCount = treeNames.size();
1002 
1003  for (auto sn : sceneSel) {
1004  QGIView* itemView = dynamic_cast<QGIView*>(sn);
1005  if (!itemView) {
1006  QGIDatumLabel* dl = dynamic_cast<QGIDatumLabel*>(sn);
1007  QGIPrimPath* pp = dynamic_cast<QGIPrimPath*>(sn);//count Vertex/Edge/Face
1008  if (pp) {
1009  ppCount++;
1010  }
1011  else if (dl) {
1012  //get dim associated with this label
1013  QGraphicsItem* qgi = dl->parentItem();
1014  if (qgi) {
1015  QGIViewDimension* vd = dynamic_cast<QGIViewDimension*>(qgi);
1016  if (vd) {
1017  std::string s = vd->getViewNameAsString();
1018  sceneNames.push_back(s);
1019  }
1020  }
1021  }
1022  }
1023  else {
1024  std::string s = itemView->getViewNameAsString();
1025  sceneNames.push_back(s);
1026  }
1027  }
1028  std::sort(sceneNames.begin(), sceneNames.end());
1029  sceneCount = sceneNames.size();
1030 
1031  //different # of DrawView* vs QGIV*
1032  if (sceneCount != treeCount) {
1033  return false;
1034  }
1035 
1036  // even of counts match, have to check that names in scene == names in tree
1037  auto treePtr = treeNames.begin();
1038  for (auto& s : sceneNames) {
1039  if (s == (*treePtr)) {
1040  treePtr++;
1041  continue;
1042  }
1043  else {
1044  return false;
1045  }
1046  }
1047 
1048  //Objects all match, check subs
1049  if (subCount != ppCount) {
1050  return false;
1051  }
1052 
1053  return true;
1054 }
