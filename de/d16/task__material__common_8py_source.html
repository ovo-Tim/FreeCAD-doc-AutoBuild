173  def get_material_uuid(self, material):
174  if self.uuid:
175  try:
176  self.material_manager.getMaterial(self.uuid)
177  return self.uuid
178  except:
179  return ""
180 
181  if not self.material:
182  return ""
183 
184  for a_mat in self.material_manager.Materials:
185  # check if every item of the current material fits to a known material card
186  # if all items were found we know it is the right card
187  # we can hereby not simply perform
188  # set(self.materials[a_mat].items()) ^ set(material.items())
189  # because entries are often identical, just appear in the set in a different order
190  unmatched_item = False
191  a_mat_prop = self.material_manager.getMaterial(a_mat).Properties.items()
192  for item in material.items():
193  if item not in a_mat_prop:
194  unmatched_item = True
195  # often the difference is just a decimal e.g. "120" to "120.0"
196  # therefore first check if the item name exists
197  for a_mat_item in a_mat_prop:
198  if item[0] == a_mat_item[0]:
199  # now check if we have a number value in a unit
200  if item[1].split() and not self.isfloat(item[1].split()[0]):
201  break
202  if item[1].split() and float(item[1].split()[0]) == float(
203  a_mat_item[1].split()[0]
204  ):
205  unmatched_item = False
206  elif not item[1].split():
207  # handle the case where item[1] is an empty string
208  if not self.isfloat(item[1]):
209  break
210  if float(item[1]) == float(a_mat_item[1]):
211  unmatched_item = False
212  break
213  if not unmatched_item:
214  return a_mat
215  return ""
