33 def ParseUserData(element):
34  res = {}
35  for i in element.findall("{http://www.plmxml.org/Schemas/PLMXMLSchema}UserData"):
36  for value in i.findall("{http://www.plmxml.org/Schemas/PLMXMLSchema}UserValue"):
37  res[value.attrib["title"]] = value.attrib["value"]
38  return res

157 
158  tree = ET.parse(fileName)
159  root = tree.getroot()
160  ProductDef = root.find("{http://www.plmxml.org/Schemas/PLMXMLSchema}ProductDef")

162  res = ParseUserData(ProductDef.find("{http://www.plmxml.org/Schemas/PLMXMLSchema}UserData"))
163 
164  InstanceGraph = ProductDef.find("{http://www.plmxml.org/Schemas/PLMXMLSchema}InstanceGraph")
165 
166  # get all the special elements we can read
167  Instances = InstanceGraph.findall("{http://www.plmxml.org/Schemas/PLMXMLSchema}Instance")
168  Parts = InstanceGraph.findall("{http://www.plmxml.org/Schemas/PLMXMLSchema}Part")
169  ProductInstances = InstanceGraph.findall(

173  "{http://www.plmxml.org/Schemas/PLMXMLSchema}ProductRevisionView"
174  )
175 
176  instanceTypesSet = set()

178  instanceTypesSet.add(child.tag)
179  print("All types below the InstanceGraph:")

181  print(i)
