110 void TaskMirroredParameters::updateUI()
111 {
112  if (blockUpdate) {
113  return;
114  }
115  blockUpdate = true;
116 
117  auto pcMirrored = getObject<PartDesign::Mirrored>();
118 
119  if (planeLinks.setCurrentLink(pcMirrored->MirrorPlane) == -1) {
120  // failed to set current, because the link isn't in the list yet
121  planeLinks.addLink(
122  pcMirrored->MirrorPlane,
123  getRefStr(pcMirrored->MirrorPlane.getValue(), pcMirrored->MirrorPlane.getSubValues()));
124  planeLinks.setCurrentLink(pcMirrored->MirrorPlane);
125  }
126 
127  blockUpdate = false;
128 }

130 void TaskMirroredParameters::onSelectionChanged(const Gui::SelectionChanges& msg)
131 {
132  if (selectionMode != SelectionMode::None && msg.Type == Gui::SelectionChanges::AddSelection) {
133 
134  if (originalSelected(msg)) {
135  exitSelectionMode();
136  }
137  else {
138  auto pcMirrored = getObject<PartDesign::Mirrored>();
139 
140  std::vector<std::string> mirrorPlanes;
141  App::DocumentObject* selObj = nullptr;
142  getReferencedSelection(pcMirrored, msg, selObj, mirrorPlanes);
143  if (!selObj) {
144  return;
145  }
146 
147  if (selectionMode == SelectionMode::Reference || selObj->isDerivedFrom<App::Plane>()) {
148  setupTransaction();
149  pcMirrored->MirrorPlane.setValue(selObj, mirrorPlanes);
150  recomputeFeature();
151  updateUI();
152  }
153  exitSelectionMode();
154  }
155  }
156 }

158 void TaskMirroredParameters::onPlaneChanged(int /*num*/)
159 {
160  if (blockUpdate) {
161  return;
162  }
163  setupTransaction();
164  auto pcMirrored = getObject<PartDesign::Mirrored>();
165  try {
166  if (!planeLinks.getCurrentLink().getValue()) {
167  // enter reference selection mode
168  hideObject();
169  showBase();
170  selectionMode = SelectionMode::Reference;
171  Gui::Selection().clearSelection();
172  addReferenceSelectionGate(AllowSelection::FACE | AllowSelection::PLANAR);
173  }
174  else {
175  exitSelectionMode();
176  pcMirrored->MirrorPlane.Paste(planeLinks.getCurrentLink());
177  }
178  }
179  catch (Base::Exception& e) {
180  QMessageBox::warning(nullptr, tr("Error"), QApplication::translate("Exception", e.what()));
181  }
182 
183  recomputeFeature();
184 }
