150  def save_meta(self, styles):
151  """Save a dict to the document Meta attribute and update objects."""
152  # save meta
153  changedstyles = []
154  meta = self.doc.Meta
155  for key, value in styles.items():
156  try:
157  strvalue = json.dumps(value)
158  except Exception:
159  print("debug: unable to serialize this:", value)
160  if ("Draft_Style_" + key in meta
161  and meta["Draft_Style_" + key] != strvalue):
162  changedstyles.append(key)
163  meta["Draft_Style_" + key] = strvalue
164 
165  # remove deleted styles
166  todelete = []
167  for key, value in meta.items():
168  if key.startswith("Draft_Style_"):
169  if key[12:] not in styles:
170  todelete.append(key)
171  for key in todelete:
172  del meta[key]
173 
174  self.doc.Meta = meta

176  # propagate changes to all annotations
177  for obj in self.get_annotations():
178  vobj = obj.ViewObject
179  try:
180  current = vobj.AnnotationStyle
181  except AssertionError:
182  # empty annotation styles list
183  pass
184  else:
185  if vobj.AnnotationStyle in self.renamed:
186  # the style has been renamed
187  # temporarily add the new style and switch to it
188  vobj.AnnotationStyle = [vobj.AnnotationStyle, self.renamed[vobj.AnnotationStyle]]

334  def repair_style(self, style):
335  """Repair a V0.19 or V0.20 style.
336 
337  Some properties were missing or misspelled.
338  Some float values were wrongly stored as strings.
339  """
340  default = utils.get_default_annotation_style()
341  new = {}
342  for key, val in default.items():
343  if style.get(key) is None:
344  new[key] = val[1]
345  elif type(style[key]) == type(val[1]):
346  new[key] = style[key]
347  elif isinstance(style[key], str):
348  new[key] = float(style[key].replace(",", "."))
349  else:
350  new[key] = val[1]
351  return new

353  def fill_editor(self, style=None):
354  """Fill the editor fields with the contents of a style."""
355  default = utils.get_default_annotation_style()
356  if style is None or style == "":
357  style = {}
358  for key, val in default.items():
359  style[key] = val[1]
360  elif isinstance(style, dict):
361  pass
362  elif isinstance(style, str) and style in self.styles:
363  style = self.styles[style]

367  for key, value in style.items():
368  control = getattr(self.form, key)
369  if default[key][0] == "str":
370  control.setText(value)

381  else:
