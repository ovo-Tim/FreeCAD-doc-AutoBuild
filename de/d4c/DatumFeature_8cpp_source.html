59  PyObject **pyObj, Base::Matrix4D *pmat, bool transform, int depth) const
60 {
61  // For the sake of simplicity, we don't bother to check for subname, just
62  // return the shape as it is, because a datum object only holds shape with
63  // one single geometry element.
64  (void)subname;
65  (void)depth;
66 
67  if(pmat && transform)
68  *pmat *= Placement.getValue().toMatrix();
69 
70  if(!pyObj)
71  return const_cast<Datum*>(this);
72 
73  Base::PyGILStateLocker lock;
74  PY_TRY {
75  TopoShape ts(getShape().Located(TopLoc_Location()));
76  if(pmat && !ts.isNull())
77  ts.transformShape(*pmat,false,true);
78  *pyObj = Py::new_reference_to(shape2pyshape(ts.getShape()));
79  return const_cast<Datum*>(this);
80  } PY_CATCH_OCC
