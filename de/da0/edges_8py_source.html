55 def orientEdge(edge, normal=None, make_arc=False):
56  """Re-orient the edge such that it is in the XY plane.
57 
58  Re-orients `edge` such that it is in the XY plane.
59  If `normal` is passed, this is used as the basis for the rotation,
60  otherwise the placement of `edge` is used.
61  """
62  # This 'normalizes' the placement to the xy plane
63  edge = edge.copy()
64  xyDir = App.Vector(0, 0, 1)
65  base = App.Vector(0, 0, 0)
66 
67  if normal:
68  angle = DraftVecUtils.angle(normal, xyDir) * App.Units.Radian
69  axis = normal.cross(xyDir)
70  else:
71  axis = edge.Placement.Rotation.Axis
72  angle = -1 * edge.Placement.Rotation.Angle * App.Units.Radian
73  if axis == App.Vector(0.0, 0.0, 0.0):
74  axis = App.Vector(0.0, 0.0, 1.0)
75  if angle:
76  edge.rotate(base, axis, angle)
77  if isinstance(edge.Curve, Part.Line):
78  return Part.LineSegment(edge.Curve,
79  edge.FirstParameter,
80  edge.LastParameter)
81  elif make_arc and isinstance(edge.Curve, Part.Circle) and not edge.Closed:
82  return Part.ArcOfCircle(edge.Curve,
83  edge.FirstParameter,
84  edge.LastParameter,
85  edge.Curve.Axis.z > 0)
86  elif make_arc and isinstance(edge.Curve, Part.Ellipse) and not edge.Closed:
87  return Part.ArcOfEllipse(edge.Curve,
88  edge.FirstParameter,
89  edge.LastParameter,
90  edge.Curve.Axis.z > 0)
91  return edge.Curve

114 def is_line(bspline):
115  """Return True if the given BSpline curve is a straight line."""
116 
117 # previous implementation may fail for a multipole straight spline due
118 # a second order error in tolerance, which introduce a difference of 1e-14
119 # in the values of the tangents. Also, may fail on a periodic spline.
120 # step = bspline.LastParameter/10
121 # b = bspline.tangent(0)
122 #
123 # for i in range(10):
124 # if bspline.tangent(i * step) != b:
125 # return False
126 
127  start_point = bspline.StartPoint
128  end_point = bspline.EndPoint
129  dist_start_end = end_point.distanceToPoint(start_point)
130  if abs(bspline.length() - dist_start_end) < 1e-7:
131  return True

174 def getTangent(edge, from_point=None):
175  """Return the tangent to an edge, including BSpline and circular arcs.
176 
177  If from_point is given, it is used to calculate the tangent,
178  only useful for a circular arc.
179  """
180  if geomType(edge) == "Line":
181  return vec(edge)
182 
183  elif (geomType(edge) == "BSplineCurve"
184  or geomType(edge) == "BezierCurve"):
185  if not from_point:
186  return None
187  cp = edge.Curve.parameter(from_point)
188  return edge.Curve.tangent(cp)[0]
189 
190  elif geomType(edge) == "Circle":
191  if not from_point:
192  v1 = edge.Vertexes[0].Point.sub(edge.Curve.Center)
193  else:
194  v1 = from_point.sub(edge.Curve.Center)
195  return v1.cross(edge.Curve.Axis)
196 
197  return None
