78 void ViewProviderFemConstraintFluidBoundary::updateData(const App::Property* prop)
79 {
80  // Gets called whenever a property of the attached object changes
81  Fem::ConstraintFluidBoundary* pcConstraint = this->getObject<Fem::ConstraintFluidBoundary>();
82  float scaledwidth =
83  WIDTH * pcConstraint->Scale.getValue(); // OvG: Calculate scaled values once only
84  float scaledheight = HEIGHT * pcConstraint->Scale.getValue();
85 
86  float scaledheadradius =
87  ARROWHEADRADIUS * pcConstraint->Scale.getValue(); // OvG: Calculate scaled values once only
88  float scaledlength = ARROWLENGTH * pcConstraint->Scale.getValue();
89 
90  std::string boundaryType = pcConstraint->BoundaryType.getValueAsString();
91  if (prop == &pcConstraint->BoundaryType) {
92  if (boundaryType == "wall") {
93  ShapeAppearance.setDiffuseColor(0.0, 1.0, 1.0);
94  }
95  else if (boundaryType == "interface") {
96  ShapeAppearance.setDiffuseColor(0.0, 1.0, 0.0);
97  }
98  else if (boundaryType == "freestream") {
99  ShapeAppearance.setDiffuseColor(1.0, 1.0, 0.0);
100  }
101  else if (boundaryType == "inlet") {
102  ShapeAppearance.setDiffuseColor(1.0, 0.0, 0.0);
103  }
104  else { //(boundaryType == "outlet")
105  ShapeAppearance.setDiffuseColor(0.0, 0.0, 1.0);
106  }
107  }
108 
109  if (boundaryType == "inlet" || boundaryType == "outlet") {
110 #ifdef USE_MULTIPLE_COPY
111  // OvG: need access to cp for scaling
112  SoMultipleCopy* cp = new SoMultipleCopy();
113  if (pShapeSep->getNumChildren() == 0) {
114  // Set up the nodes
115  cp->matrix.setNum(0);
116  cp->addChild(
117  (SoNode*)GuiTools::createArrow(scaledlength, scaledheadradius)); // OvG: Scaling
118  pShapeSep->addChild(cp);
119  }
120 #endif
121 
122  if (prop == &pcConstraint->Points) {
123  const std::vector<Base::Vector3d>& points = pcConstraint->Points.getValues();
124 
125 #ifdef USE_MULTIPLE_COPY
126  cp = static_cast<SoMultipleCopy*>(pShapeSep->getChild(0));
127  cp->matrix.setNum(points.size());
128  SbMatrix* matrices = cp->matrix.startEditing();
129  int idx = 0;
130 #else
131  // Redraw all arrows
132  Gui::coinRemoveAllChildren(pShapeSep);
133 #endif
134  // This should always point outside of the solid
135  Base::Vector3d normal = pcConstraint->NormalDirection.getValue();
136 
137  // Get default direction (on first call to method)
138  Base::Vector3d forceDirection = pcConstraint->DirectionVector.getValue();
139  if (forceDirection.Length() < Precision::Confusion()) {
140  forceDirection = normal;
141  }
142 
143  SbVec3f dir(forceDirection.x, forceDirection.y, forceDirection.z);
144  SbRotation rot(SbVec3f(0, 1, 0), dir);
145 
146  for (const auto& point : points) {
147  SbVec3f base(point.x, point.y, point.z);
148  if (forceDirection.GetAngle(normal)
149  < M_PI_2) { // Move arrow so it doesn't disappear inside the solid
150  base = base + dir * scaledlength; // OvG: Scaling
151  }
152 #ifdef USE_MULTIPLE_COPY
153  SbMatrix m;
154  m.setTransform(base, rot, SbVec3f(1, 1, 1));
155  matrices[idx] = m;
156  idx++;
157 #else
158  SoSeparator* sep = new SoSeparator();
159  GuiTools::createPlacement(sep, base, rot);
160  GuiTools::createArrow(sep, scaledlength, scaledheadradius); // OvG: Scaling
161  pShapeSep->addChild(sep);
162 #endif
163  }
164 #ifdef USE_MULTIPLE_COPY
165  cp->matrix.finishEditing();
166 #endif
167  }
168  else if (prop == &pcConstraint->DirectionVector) {
169  // Note: "Reversed" also triggers "DirectionVector"
170  // Re-orient all arrows
171  Base::Vector3d normal = pcConstraint->NormalDirection.getValue();
172  Base::Vector3d forceDirection = pcConstraint->DirectionVector.getValue();
173  if (forceDirection.Length() < Precision::Confusion()) {
174  forceDirection = normal;
175  if (boundaryType == "inlet") {
176  forceDirection = -normal;
177  }
178  }
179 
180  SbVec3f dir(forceDirection.x, forceDirection.y, forceDirection.z);
181  SbRotation rot(SbVec3f(0, 1, 0), dir);
182 
183  const std::vector<Base::Vector3d>& points = pcConstraint->Points.getValues();
184 
185 #ifdef USE_MULTIPLE_COPY
186  SoMultipleCopy* cp = static_cast<SoMultipleCopy*>(pShapeSep->getChild(0));
187  cp->matrix.setNum(points.size());
188  SbMatrix* matrices = cp->matrix.startEditing();
189 #endif
190  int idx = 0;
191 
192  for (const auto& point : points) {
193  SbVec3f base(point.x, point.y, point.z);
194  if (forceDirection.GetAngle(normal) < M_PI_2) {
195  base = base + dir * scaledlength; // OvG: Scaling
196  }
197 #ifdef USE_MULTIPLE_COPY
198  SbMatrix m;
199  m.setTransform(base, rot, SbVec3f(1, 1, 1));
200  matrices[idx] = m;
201 #else
202  SoSeparator* sep = static_cast<SoSeparator*>(pShapeSep->getChild(idx));
203  GuiTools::updatePlacement(sep, 0, base, rot);
204  GuiTools::updateArrow(sep, 2, scaledlength, scaledheadradius); // OvG: Scaling
205 #endif
206  idx++;
207  }
208 #ifdef USE_MULTIPLE_COPY
209  cp->matrix.finishEditing();
210 #endif
211  }
212  }
213  else { // not inlet or outlet boundary type
214 
215 #ifdef USE_MULTIPLE_COPY
216  // OvG: always need access to cp for scaling
217  SoMultipleCopy* cp = new SoMultipleCopy();
218  if (pShapeSep->getNumChildren() == 0) {
219  // Set up the nodes
220  cp->matrix.setNum(0);
221  cp->addChild(
222  (SoNode*)GuiTools::createFixed(scaledheight, scaledwidth)); // OvG: Scaling
223  pShapeSep->addChild(cp);
224  }
225 #endif
226 
227  if (prop == &pcConstraint->Points) {
228  const std::vector<Base::Vector3d>& points = pcConstraint->Points.getValues();
229  const std::vector<Base::Vector3d>& normals = pcConstraint->Normals.getValues();
230  if (points.size() != normals.size()) {
231  return;
232  }
233  std::vector<Base::Vector3d>::const_iterator n = normals.begin();
234 
235 #ifdef USE_MULTIPLE_COPY
236  cp = static_cast<SoMultipleCopy*>(pShapeSep->getChild(0));
237  cp->matrix.setNum(points.size());
238  SbMatrix* matrices = cp->matrix.startEditing();
239  int idx = 0;
240 #else
241  // Note: Points and Normals are always updated together
242  Gui::coinRemoveAllChildren(pShapeSep);
243 #endif
244 
245  for (const auto& point : points) {
246  SbVec3f base(point.x, point.y, point.z);
247  SbVec3f dir(n->x, n->y, n->z);
248  SbRotation rot(SbVec3f(0, -1, 0), dir);
249 #ifdef USE_MULTIPLE_COPY
250  SbMatrix m;
251  m.setTransform(base, rot, SbVec3f(1, 1, 1));
252  matrices[idx] = m;
253  idx++;
254 #else
255  SoSeparator* sep = new SoSeparator();
256  GuiTools::createPlacement(sep, base, rot);
257  GuiTools::createFixed(sep, scaledheight, scaledwidth); // OvG: Scaling
258  pShapeSep->addChild(sep);
259 #endif
260  n++;
261  }
262 #ifdef USE_MULTIPLE_COPY
263  cp->matrix.finishEditing();
264 #endif
265  }
266  }
