63 void VRMLObject::onChanged(const App::Property* prop)
64 {
65  if (restoreData) {
66  return;
67  }
68 
69  if (prop == &VrmlFile) {
70  std::string orig = VrmlFile.getOriginalFileName();
71  if (!orig.empty()) {
72  // store the path name of the VRML file
73  Base::FileInfo fi(orig);
74  this->vrmlPath = fi.dirPath();
75  }
76  }
77  else if (prop == &Urls) {
78  // save the relative paths to the resource files in the project file
79  Resources.setSize(Urls.getSize());
80  const std::vector<std::string>& urls = Urls.getValues();
81  int index = 0;
82  for (const auto& it : urls) {
83  std::string output = getRelativePath(this->vrmlPath, it);
84  Resources.set1Value(index, output);
85  ++index;
86  }
87  }
88  GeoFeature::onChanged(prop);
89 }

100 std::string VRMLObject::getRelativePath(const std::string& prefix,
101  const std::string& resource) const
102 {
103  std::string str;
104  std::string intname = this->getNameInDocument();
105  if (!prefix.empty()) {
106  if (resource.substr(0, prefix.size()) == prefix) {
107  std::string suffix = resource.substr(prefix.size());
108  str = intname + suffix;
109  }
110  }
111 
112  if (str.empty()) {
113  Base::FileInfo fi(resource);
114  str = intname + "/" + fi.fileName();
115  }
116 
117  return str;
118 }

120 std::string VRMLObject::fixRelativePath(const std::string& name, const std::string& resource)
121 {
122  // the part before the first '/' must match with object's internal name
123  std::string::size_type pos = resource.find('/');
124  if (pos != std::string::npos) {
125  std::string prefix = resource.substr(0, pos);
126  std::string suffix = resource.substr(pos);
127  if (prefix != name) {
128  return name + suffix;
129  }
130  }
131  return resource;
132 }

134 void VRMLObject::makeDirectories(const std::string& path, const std::string& subdir)
135 {
136  std::string::size_type pos = subdir.find('/');
137  while (pos != std::string::npos) {
138  std::string sub = subdir.substr(0, pos);
139  std::string dir = path;
140  dir += '/';
141  dir += sub;
142  Base::FileInfo fi(dir);
143  if (!fi.createDirectory()) {
144  break;
145  }
146  pos = subdir.find('/', pos + 1);
147  }
148 }

177 void VRMLObject::SaveDocFile(Base::Writer& writer) const
178 {
179  // store the inline files of the VRML file
180  if (this->indexSave < Urls.getSize()) {
181  std::string url = Urls[this->indexSave];
182 
183  Base::FileInfo fi(url);
184  // it can happen that the transient directory has changed after
185  // saving the 'URLs' in RestoreDocFile() and then we have to
186  // try again with the new transient directory.
187  if (!fi.exists()) {
188  std::string path = getDocument()->TransientDir.getValue();
189  url = Resources[this->indexSave];
190  url = path + "/" + url;
191  fi.setFile(url);
192  }
193 
194  this->indexSave++;
195  Base::ifstream file(fi, std::ios::in | std::ios::binary);
196  if (file) {
197  writer.Stream() << file.rdbuf();
198  }
199  }
200 }

202 bool VRMLObject::restoreTextureFinished(Base::Reader& reader)
203 {
204  Base::StateLocker locker(restoreData, true);
205  if (this->indexRestore < Resources.getSize()) {
206  std::string path = getDocument()->TransientDir.getValue();
207  std::string url = Resources[this->indexRestore];
208  std::string intname = this->getNameInDocument();
209  url = fixRelativePath(intname, url);
210  Resources.set1Value(this->indexRestore, url);
211  makeDirectories(path, url);
212 
213  url = path + "/" + url;
214  Base::FileInfo fi(url);
215  Urls.set1Value(this->indexRestore, url);
216  this->indexRestore++;
217 
218  Base::ofstream file(fi, std::ios::out | std::ios::binary);
219  if (file) {
220  reader >> file.rdbuf();
221  file.close();
222  }
223 
224  return (this->indexRestore == Urls.getSize());
225  }
226 
227  return false;
228 }
