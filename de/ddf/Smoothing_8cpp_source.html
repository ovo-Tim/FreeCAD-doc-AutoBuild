56 void PlaneFitSmoothing::Smooth(unsigned int iterations)
57 {
58  MeshCore::MeshPoint center;
59  MeshCore::MeshPointArray PointArray = kernel.GetPoints();
60 
61  MeshCore::MeshPointIterator v_it(kernel);
62  MeshCore::MeshRefPointToPoints vv_it(kernel);
63  MeshCore::MeshPointArray::_TConstIterator v_beg = kernel.GetPoints().begin();
64 
65  for (unsigned int i = 0; i < iterations; i++) {
66  Base::Vector3f N, L;
67  for (v_it.Begin(); v_it.More(); v_it.Next()) {
68  MeshCore::PlaneFit pf;
69  pf.AddPoint(*v_it);
70  center = *v_it;
71  const std::set<PointIndex>& cv = vv_it[v_it.Position()];
72  if (cv.size() < 3) {
73  continue;
74  }
75 
76  std::set<PointIndex>::const_iterator cv_it;
77  for (cv_it = cv.begin(); cv_it != cv.end(); ++cv_it) {
78  pf.AddPoint(v_beg[*cv_it]);
79  center += v_beg[*cv_it];
80  }
81 
82  float scale = 1.0F / (static_cast<float>(cv.size()) + 1.0F);
83  center.Scale(scale, scale, scale);
84 
85  // get the mean plane of the current vertex with the surrounding vertices
86  pf.Fit();
87  N = pf.GetNormal();
88  N.Normalize();
89 
90  // look in which direction we should move the vertex
91  L.Set(v_it->x - center.x, v_it->y - center.y, v_it->z - center.z);
92  if (N * L < 0.0F) {
93  N.Scale(-1.0, -1.0, -1.0);
94  }
95 
96  // maximum value to move is distance to mean plane
97  float d = std::min<float>(std::fabs(this->maximum), fabs(N * L));
98  N.Scale(d, d, d);
99 
100  PointArray[v_it.Position()].Set(v_it->x - N.x, v_it->y - N.y, v_it->z - N.z);
101  }
102 
103  // assign values without affecting iterators
104  PointIndex count = kernel.CountPoints();
105  for (PointIndex idx = 0; idx < count; idx++) {
106  kernel.SetPoint(idx, PointArray[idx]);
107  }
108  }
109 }

172 void LaplaceSmoothing::Umbrella(const MeshRefPointToPoints& vv_it,
173  const MeshRefPointToFacets& vf_it,
174  double stepsize)
175 {
176  const MeshCore::MeshPointArray& points = kernel.GetPoints();
177  MeshCore::MeshPointArray::_TConstIterator v_it, v_beg = points.begin(), v_end = points.end();
178 
179  PointIndex pos = 0;
180  for (v_it = points.begin(); v_it != v_end; ++v_it, ++pos) {
181  const std::set<PointIndex>& cv = vv_it[pos];
182  if (cv.size() < 3) {
183  continue;
184  }
185  if (cv.size() != vf_it[pos].size()) {
186  // do nothing for border points
187  continue;
188  }
189 
190  size_t n_count = cv.size();
191  double w {};
192  w = 1.0 / double(n_count);
193 
194  double delx = 0.0, dely = 0.0, delz = 0.0;
195  std::set<PointIndex>::const_iterator cv_it;
196  for (cv_it = cv.begin(); cv_it != cv.end(); ++cv_it) {
197  delx += w * static_cast<double>((v_beg[*cv_it]).x - v_it->x);
198  dely += w * static_cast<double>((v_beg[*cv_it]).y - v_it->y);
199  delz += w * static_cast<double>((v_beg[*cv_it]).z - v_it->z);
200  }
201 
202  float x = static_cast<float>(static_cast<double>(v_it->x) + stepsize * delx);
203  float y = static_cast<float>(static_cast<double>(v_it->y) + stepsize * dely);
204  float z = static_cast<float>(static_cast<double>(v_it->z) + stepsize * delz);
205  kernel.SetPoint(pos, x, y, z);
206  }
207 }

300 using AngleNormal = std::pair<double, Base::Vector3d>;
301 inline Base::Vector3d find_median(std::vector<AngleNormal>& container)
302 {
303  auto compare_angle_normal = [](const AngleNormal& an1, const AngleNormal& an2) {
304  return an1.first < an2.first;
305  };
306  size_t n = container.size() / 2;
307  std::nth_element(container.begin(),
308  container.begin() + n,
309  container.end(),
310  compare_angle_normal);
311 
312  if ((container.size() % 2) == 1) {
313  return container[n].second;
314  }
315 
316  // even sized vector -> average the two middle values
317  auto max_it = std::max_element(container.begin(), container.begin() + n, compare_angle_normal);
318  Base::Vector3d vec = (max_it->second + container[n].second) / 2.0;
319  vec.Normalize();
320  return vec;
321 }

351 void MedianFilterSmoothing::UpdatePoints(const MeshRefFacetToFacets& ff_it,
352  const MeshRefPointToFacets& vf_it,
353  const std::vector<PointIndex>& point_indices)
354 {
355  const MeshCore::MeshPointArray& points = kernel.GetPoints();
356  const MeshCore::MeshFacetArray& facets = kernel.GetFacets();
357 
358  // Initialize the array with the real normals
359  std::vector<Base::Vector3d> faceNormals;
360  faceNormals.reserve(facets.size());
361  MeshCore::MeshFacetIterator iter(kernel);
362  for (iter.Init(); iter.More(); iter.Next()) {
363  faceNormals.emplace_back(Base::toVector<double>(iter->GetNormal()));
364  }
365 
366  // Step 1: determine face normals
367  for (FacetIndex pos = 0; pos < facets.size(); pos++) {
368  iter.Set(pos);
369  Base::Vector3d refNormal = Base::toVector<double>(iter->GetNormal());
370  const std::set<FacetIndex>& cv = ff_it[pos];
371  const MeshCore::MeshFacet& facet = facets[pos];
372 
373  std::vector<AngleNormal> anglesWithFaces;
374  for (auto fi : cv) {
375  iter.Set(fi);
376  Base::Vector3d faceNormal = Base::toVector<double>(iter->GetNormal());
377  double angle = refNormal.GetAngle(faceNormal);
378 
379  int absWeight = std::abs(weights);
380  if (absWeight > 1 && facet.IsNeighbour(fi)) {
381  if (weights < 0) {
382  angle = -angle;
383  }
384  for (int i = 0; i < absWeight; i++) {
385  anglesWithFaces.emplace_back(angle, faceNormal);
386  }
387  }
388  else {
389  anglesWithFaces.emplace_back(angle, faceNormal);
390  }
391  }
392 
393  faceNormals[pos] = find_median(anglesWithFaces);
394  }
395 
396  // Step 2: move vertices
397  for (auto pos : point_indices) {
398  Base::Vector3d P = Base::toVector<double>(points[pos]);
399  const std::set<FacetIndex>& cv = vf_it[pos];
400 
401  double totalArea = 0.0;
402  Base::Vector3d totalvT;
403  for (auto it : cv) {
404  iter.Set(it);
405 
406  double faceArea = iter->Area();
407  totalArea += faceArea;
408 
409  Base::Vector3d C = Base::toVector<double>(iter->GetGravityPoint());
410 
411  Base::Vector3d PC = C - P;
412  Base::Vector3d mT = faceNormals[it];
413  Base::Vector3d vT = (PC * mT) * mT;
414  totalvT += vT * faceArea;
415  }
416 
417  P = P + totalvT / totalArea;
418  kernel.SetPoint(pos, Base::toVector<float>(P));
419  }
420 }
