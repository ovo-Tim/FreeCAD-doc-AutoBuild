338 void TaskLeaderLine::createLeaderFeature(std::vector<Base::Vector3d> sceneDeltas)
339 {
340  const std::string objectName{"LeaderLine"};
341  std::string m_leaderName = m_basePage->getDocument()->getUniqueObjectName(objectName.c_str());
342  m_leaderType = "TechDraw::DrawLeaderLine";
343 
344  std::string PageName = m_basePage->getNameInDocument();
345 
346  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create Leader"));
347  Command::doCommand(Command::Doc, "App.activeDocument().addObject('%s', '%s')",
348  m_leaderType.c_str(), m_leaderName.c_str());
349  Command::doCommand(Command::Doc, "App.activeDocument().%s.translateLabel('DrawLeaderLine', 'LeaderLine', '%s')",
350  m_leaderName.c_str(), m_leaderName.c_str());
351 
352  Command::doCommand(Command::Doc, "App.activeDocument().%s.addView(App.activeDocument().%s)",
353  PageName.c_str(), m_leaderName.c_str());
354 
355  double baseRotation{0};
356  if (m_baseFeat) {
357  Command::doCommand(Command::Doc, "App.activeDocument().%s.LeaderParent = App.activeDocument().%s",
358  m_leaderName.c_str(), m_baseFeat->getNameInDocument());
359  baseRotation = m_baseFeat->Rotation.getValue();
360  }
361 
362  App::DocumentObject* obj = m_basePage->getDocument()->getObject(m_leaderName.c_str());
363  if (!obj) {
364  throw Base::RuntimeError("TaskLeaderLine - new markup object not found");
365  }
366 
367  if (obj->isDerivedFrom(TechDraw::DrawLeaderLine::getClassTypeId())) {
368  m_lineFeat = static_cast<TechDraw::DrawLeaderLine*>(obj);
369  auto forMath{m_attachPoint};
370  if (baseRotation != 0) {
371  forMath = DU::invertY(forMath);
372  forMath.RotateZ(-Base::toRadians(baseRotation));
373  forMath = DU::invertY(forMath);
374  }
375  m_attachPoint = forMath;
376  m_lineFeat->setPosition(Rez::appX(m_attachPoint.x), Rez::appX(- m_attachPoint.y), true);
377  if (!sceneDeltas.empty()) {
378  std::vector<Base::Vector3d> pageDeltas;
379  // convert deltas to mm. leader points are stored inverted, so we do not convert to conventional Y axis
380  for (auto& delta : sceneDeltas) {
381  Base::Vector3d deltaInPageCoords = Rez::appX(delta);
382  pageDeltas.push_back(deltaInPageCoords);
383  }
384 
385  // already unrotated, now convert to unscaled, but inverted
386  bool doScale{true};
387  bool doRotate{false};
388  auto temp = m_lineFeat->makeCanonicalPointsInverted(pageDeltas, doScale, doRotate);
389  m_lineFeat->WayPoints.setValues(temp);
390  }
391  commonFeatureUpdate();
392  }
393 
394  if (m_lineFeat) {
395  Gui::ViewProvider* vp = QGIView::getViewProvider(m_lineFeat);
396  auto leadVP = dynamic_cast<ViewProviderLeader*>(vp);
397  if (leadVP) {
398  App::Color ac;
399  ac.setValue<QColor>(ui->cpLineColor->color());
400  leadVP->Color.setValue(ac);
401  leadVP->LineWidth.setValue(ui->dsbWeight->rawValue());
402  leadVP->LineStyle.setValue(ui->cboxStyle->currentIndex());
403  }
404  }
405 
406  Gui::Command::updateActive();
407  Gui::Command::commitCommand();
408 
409  //trigger claimChildren in tree
410  if (m_baseFeat) {
411  m_baseFeat->touch();
412  }
413 
414  m_basePage->touch();
415 
416  if (m_lineFeat) {
417  m_lineFeat->requestPaint();
418  }
419 }

487 void TaskLeaderLine::onTrackerClicked(bool clicked)
488 {
489  Q_UNUSED(clicked);
490  if (!m_vpp->getMDIViewPage()) {
491  Base::Console().Message("TLL::onTrackerClicked - no Mdi, no Tracker!\n");
492  return;
493  }
494 
495  if ( m_pbTrackerState == TrackerAction::SAVE &&
496  getCreateMode() ){
497  if (m_tracker) {
498  m_tracker->terminateDrawing();
499  }
500  m_pbTrackerState = TrackerAction::PICK;
501  ui->pbTracker->setText(tr("Pick Points"));
502  ui->pbCancelEdit->setEnabled(false);
503  enableTaskButtons(true);
504 
505  setEditCursor(Qt::ArrowCursor);
506  return;
507  }
508 
509  if ( m_pbTrackerState == TrackerAction::SAVE &&
510  !getCreateMode() ) { //edit mode
511  if (m_qgLeader) {
512  m_qgLeader->closeEdit();
513  }
514  m_pbTrackerState = TrackerAction::PICK;
515  ui->pbTracker->setText(tr("Edit Points"));
516  ui->pbCancelEdit->setEnabled(false);
517  enableTaskButtons(true);
518 
519  setEditCursor(Qt::ArrowCursor);
520  return;
521  }
522 
523  //TrackerAction::PICK or TrackerAction::EDIT
524  if (getCreateMode()) {
525  m_inProgressLock = true;
526  m_saveContextPolicy = m_vpp->getMDIViewPage()->contextMenuPolicy();
527  m_vpp->getMDIViewPage()->setContextMenuPolicy(Qt::PreventContextMenu);
528  m_trackerMode = QGTracker::TrackerMode::Line;
529  setEditCursor(Qt::CrossCursor);
530  startTracker();
531 
532  QString msg = tr("Pick a starting point for leader line");
533  getMainWindow()->statusBar()->show();
534  Gui::getMainWindow()->showMessage(msg, MessageDisplayTime);
535  ui->pbTracker->setText(tr("Save Points"));
536  ui->pbTracker->setEnabled(true);
537  ui->pbCancelEdit->setEnabled(true);
538  m_pbTrackerState = TrackerAction::SAVE;
539  enableTaskButtons(false);
540  } else { //edit mode
541  // pageDeltas are in mm with conventional Y axis
542  auto pageDeltas = m_lineFeat->getScaledAndRotatedPoints();
543 
544  m_sceneDeltas.clear();
545  m_sceneDeltas.reserve(pageDeltas.size());
546  // now convert to sceneUnits and Qt Y axis
547  for (auto& entry : pageDeltas) {
548  m_sceneDeltas.push_back(DGU::toSceneCoords(entry, false));
549  }
550 
551  if (!m_sceneDeltas.empty()) { //regular edit session
552  m_inProgressLock = true;
553  m_saveContextPolicy = m_vpp->getMDIViewPage()->contextMenuPolicy();
554  m_vpp->getMDIViewPage()->setContextMenuPolicy(Qt::PreventContextMenu);
555  QGIView* qgiv = m_vpp->getQGSPage()->findQViewForDocObj(m_lineFeat);
556  auto qgLead = dynamic_cast<QGILeaderLine*>(qgiv);
557 
558  if (!qgLead) {
559  //tarfu
560  Base::Console().Error("TaskLeaderLine - can't find leader graphic\n");
561  //now what? throw will generate "unknown unhandled exception"
562  } else {
563  m_qgLeader = qgLead;
564  connect(qgLead, &QGILeaderLine::editComplete,
565  this, &TaskLeaderLine::onPointEditComplete);
566  qgLead->startPathEdit();
567  QString msg = tr("Click and drag markers to adjust leader line");
568  getMainWindow()->statusBar()->show();
569  Gui::getMainWindow()->showMessage(msg, MessageDisplayTime);
570  ui->pbTracker->setText(tr("Save changes"));
571  ui->pbTracker->setEnabled(true);
572  ui->pbCancelEdit->setEnabled(true);
573  m_pbTrackerState = TrackerAction::SAVE;
574  enableTaskButtons(false);
575  }
576  } else { // need to recreate leaderline
577  m_inProgressLock = true;
578  m_saveContextPolicy = m_vpp->getMDIViewPage()->contextMenuPolicy();
579  m_vpp->getMDIViewPage()->setContextMenuPolicy(Qt::PreventContextMenu);
580  m_trackerMode = QGTracker::TrackerMode::Line;
581  setEditCursor(Qt::CrossCursor);
582  startTracker();
583 
584  QString msg = tr("Pick a starting point for leader line");
585  getMainWindow()->statusBar()->show();
586  Gui::getMainWindow()->showMessage(msg, MessageDisplayTime);
587  ui->pbTracker->setText(tr("Save changes"));
588  ui->pbTracker->setEnabled(true);
589  ui->pbCancelEdit->setEnabled(true);
590  m_pbTrackerState = TrackerAction::SAVE;
591  enableTaskButtons(false);
592  }
593  }

596 void TaskLeaderLine::startTracker()
597 {
598 // Base::Console().Message("TTL::startTracker()\n");
599  if (!m_vpp->getQGSPage()) {
600  return;
601  }
602 
603  if (m_trackerMode == QGTracker::TrackerMode::None) {
604  return;
605  }
606 
607  if (!m_tracker) {
608  m_tracker = new QGTracker(m_vpp->getQGSPage(), m_trackerMode);
609  QObject::connect(
610  m_tracker, &QGTracker::drawingFinished,
611  this , &TaskLeaderLine::onTrackerFinished
612  );
613  } else {
614  //this is too harsh. but need to avoid restarting process

722 // from scene QPointF to zero origin (delta from p0) Vector3d points
723 std::vector<Base::Vector3d> TaskLeaderLine::scenePointsToDeltas(std::vector<QPointF> scenePoints)
724 {
725  if (scenePoints.empty()) {
726  return {};
727  }
728 
729  std::vector<Base::Vector3d> result;
730  auto frontPoint = DU::toVector3d(m_qgParent->mapFromScene(scenePoints.front()));
731  result.reserve(scenePoints.size());
732  for (auto& point: scenePoints) {
733  auto viewPoint = m_qgParent->mapFromScene(point);
734  auto vPoint = DU::toVector3d(viewPoint);
735  auto delta = vPoint - frontPoint;
736  auto rotationDeg = m_baseFeat->Rotation.getValue();
737  auto deltaUnrotated{delta};
738  if (rotationDeg != 0) {
739  deltaUnrotated = DU::invertY(deltaUnrotated);
740  deltaUnrotated.RotateZ(-Base::toRadians(rotationDeg));
741  deltaUnrotated = DU::invertY(deltaUnrotated);
742  }
743 
744  result.push_back(deltaUnrotated);
745  }
746  return result;
747 }
