78 bool VectorTableModel::setData(const QModelIndex &index, const QVariant &value, int role)
79 {
80  int r = index.row();
81  int c = index.column();
82  if (role == Qt::EditRole && r < vectors.size()) {
83  if (value.canConvert<Base::Vector3d>()) {
84  vectors[r] = value.value<Base::Vector3d>();
85  Q_EMIT dataChanged(index, index.sibling(index.row(), 2));
86  return true;
87  }
88  else if (c < 3) {
89  double d = value.toDouble();
90  if (c == 0)
91  vectors[r].x = d;
92  else if (c == 1)
93  vectors[r].y = d;
94  else if (c == 2)
95  vectors[r].z = d;
96  Q_EMIT dataChanged(index, index);
97  return true;
98  }
99  }
100  return QAbstractTableModel::setData(index, value, role);
101 }

155 void Gui::VectorTableModel::pasteFromClipboard()
156 {
157  QClipboard *clipboard = QApplication::clipboard();
158  QStringList lines = clipboard->text().split(QLatin1Char('\n'));
159  bool okAll = !lines.empty();
160  QList<Base::Vector3d> newVectors;
161  QLatin1Char tab('\t');
162  QLatin1Char semicolon(';');
163  QLatin1Char comma(',');
164 
165  for (const QString& line : lines) {
166  if (line.isEmpty()) {
167  continue;
168  }
169  QChar delimiter = line.count(tab) == 2 ? tab
170  : line.count(semicolon) == 2 ? semicolon
171  : line.count(comma) == 2 ? comma
172  : QChar(QChar::Null);
173 
174  if (delimiter.isNull()) {
175  okAll = false;
176  break;
177  }
178 
179  QStringList components = line.split(delimiter);
180 
181  if (components.size() == 3) {
182  bool okX, okY, okZ;
183  double x = components.at(0).toDouble(&okX);
184  double y = components.at(1).toDouble(&okY);
185  double z = components.at(2).toDouble(&okZ);
186 
187  if (!okX || !okY || !okZ) {
188  okAll = false;
189  break;
190  }
191  newVectors.append(Base::Vector3d(x, y, z));
192  }
193  else {
194  okAll = false;
195  break;
196  }
197  }
198 
199  if (okAll) {
200  setValues(newVectors);
201  }
202  else {
203  QString msg(tr("Unsupported format. Must be 3 values per row separated by tabs, semicolons, or commas:") + QLatin1String("\n"));
204  msg += clipboard->text();
205  Base::Console().Error(msg.toStdString().c_str());
206  }
207 }
