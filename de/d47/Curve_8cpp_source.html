75 bool CCurve::CheckForArc(const CVertex& prev_vt,
76  std::list<const CVertex*>& might_be_an_arc,
77  CArc& arc_returned)
78 {
79  // this examines the vertices in might_be_an_arc
80  // if they do fit an arc, set arc to be the arc that they fit and return true
81  // returns true, if arc added
82  if (might_be_an_arc.size() < 2) {
83  return false;
84  }
85 
86  // find middle point
87  std::size_t num = might_be_an_arc.size();
88  std::size_t i = 0;
89  const CVertex* mid_vt = NULL;
90  std::size_t mid_i = (num - 1) / 2;
91  for (std::list<const CVertex*>::iterator It = might_be_an_arc.begin();
92  It != might_be_an_arc.end();
93  It++, i++) {
94  if (i == mid_i) {
95  mid_vt = *It;
96  break;
97  }
98  }
99 
100  if (mid_vt == NULL) {
101  return false;
102  }
103 
104  // create a circle to test
105  Point p0(prev_vt.m_p);
106  Point p1(mid_vt->m_p);
107  Point p2(might_be_an_arc.back()->m_p);
108  Circle c(p0, p1, p2);
109 
110  const CVertex* current_vt = &prev_vt;
111  // It seems that ClipperLib's offset ArcTolerance (same as m_accuracy here)
112  // is not exactly what's documented at https://goo.gl/4odfQh. Test shows the
113  // maximum arc distance deviate at about 2.2*ArcTolerance units. The maximum
114  // deviance seems to always occur at the end of arc.
115  double accuracy = CArea::m_accuracy * 2.3 / CArea::m_units;
116  for (std::list<const CVertex*>::iterator It = might_be_an_arc.begin();
117  It != might_be_an_arc.end();
118  It++) {
119  const CVertex* vt = *It;
120 
121  if (!c.LineIsOn(current_vt->m_p, vt->m_p, accuracy)) {
122  return false;
123  }
124  current_vt = vt;
125  }
126 
127  CArc arc;
128  arc.m_c = c.m_c;
129  arc.m_s = prev_vt.m_p;
130  arc.m_e = might_be_an_arc.back()->m_p;
131  arc.SetDirWithPoint(might_be_an_arc.front()->m_p);
132  arc.m_user_data = might_be_an_arc.back()->m_user_data;
133 
134  double angs = atan2(arc.m_s.y - arc.m_c.y, arc.m_s.x - arc.m_c.x);
135  double ange = atan2(arc.m_e.y - arc.m_c.y, arc.m_e.x - arc.m_c.x);
136  if (arc.m_dir) {
137  // make sure ange > angs
138  if (ange < angs) {
139  ange += 6.2831853071795864;
140  }
141  }
142  else {
143  // make sure angs > ange
144  if (angs < ange) {
145  angs += 6.2831853071795864;
146  }
147  }
148 
149  if (arc.IncludedAngle() >= 3.15) { // We don't want full arcs, so limit to about 180 degrees
150  return false;
151  }
152 
153  for (std::list<const CVertex*>::iterator It = might_be_an_arc.begin();
154  It != might_be_an_arc.end();
155  It++) {
156  const CVertex* vt = *It;
157  double angp = atan2(vt->m_p.y - arc.m_c.y, vt->m_p.x - arc.m_c.x);
158  if (arc.m_dir) {
159  // make sure angp > angs
160  if (angp < angs) {
161  angp += 6.2831853071795864;
162  }
163  if (angp > ange) {
164  return false;
165  }
166  }
167  else {
168  // make sure angp > ange
169  if (angp < ange) {
170  angp += 6.2831853071795864;
171  }
172  if (angp > angs) {
173  return false;
174  }
175  }
176  }
177 
178  arc_returned = arc;
179  return true;
180 }

231 void CCurve::FitArcs(bool retry)
232 {
233  std::list<CVertex> new_vertices;
234 
235  std::list<const CVertex*> might_be_an_arc;
236  CArc arc;
237  bool arc_found = false;
238  bool arc_added = false;
239  int i = 0;
240  for (std::list<CVertex>::iterator It = m_vertices.begin(); It != m_vertices.end(); It++, i++) {
241  CVertex& vt = *It;
242  if (vt.m_type || i == 0) {
243  if (i != 0) {
244  AddArcOrLines(false, new_vertices, might_be_an_arc, arc, arc_found, arc_added);
245  }
246  new_vertices.push_back(vt);
247  }
248  else {
249  might_be_an_arc.push_back(&vt);
250 
251  if (might_be_an_arc.size() == 1) {}
252  else {
253  AddArcOrLines(true, new_vertices, might_be_an_arc, arc, arc_found, arc_added);
254  }
255  }
256  }
257 
258  if (might_be_an_arc.size() > 0) {
259  // check if the last edge can form an arc with the starting edge
260  if (!retry && m_vertices.size() > 2 && m_vertices.begin()->m_type == 0 && IsClosed()) {
261  std::list<const CVertex*> tmp;
262  auto it = m_vertices.begin();
263  tmp.push_back(&(*it++));
264 
265  // this condition check is to skip the situation when both the
266  // starting and ending has already been fitted with some arc
267  if (!arc_found || it->m_type == 0) {
268  tmp.push_back(&(*it));
269  CArc tmpArc;
270  auto itEnd = m_vertices.end();
271  --itEnd;
272  --itEnd;
273  if (CheckForArc(*itEnd, tmp, tmpArc)) {
274  if (arc_found) {
275  // this means the last edge has already been fitted with
276  // some arc, so we move the first edge to the end
277 
278  // Must pop first, because this is a closed curve,
279  // meaning the last point must be equal to the first
280  // point.
281  m_vertices.pop_front();
282  m_vertices.push_back(m_vertices.front());
283  }
284  else {
285  m_vertices.push_front(CVertex(new_vertices.back().m_p));
286  m_vertices.pop_back();
287  }
288  FitArcs(true);
289  return;
290  }
291  }
292  }
293  AddArcOrLines(false, new_vertices, might_be_an_arc, arc, arc_found, arc_added);
294  }
295 
296  if (arc_added) {
297  for (auto* v : might_be_an_arc) {
298  new_vertices.push_back(*v);
299  }
300  m_vertices.swap(new_vertices);
301  }
302 }

304 void CCurve::UnFitArcs()
305 {
306  std::list<Point> new_pts;
307 
308  const CVertex* prev_vertex = NULL;
309  for (std::list<CVertex>::const_iterator It2 = m_vertices.begin(); It2 != m_vertices.end();
310  It2++) {
311  const CVertex& vertex = *It2;
312  if (vertex.m_type == 0 || prev_vertex == NULL) {
313  new_pts.push_back(vertex.m_p * CArea::m_units);
314  }
315  else {
316  if (vertex.m_p != prev_vertex->m_p) {
317  double phi, dphi, dx, dy;
318  int Segments;
319  int i;
320  double ang1, ang2, phit;
321 
322  dx = (prev_vertex->m_p.x - vertex.m_c.x) * CArea::m_units;
323  dy = (prev_vertex->m_p.y - vertex.m_c.y) * CArea::m_units;
324 
325  ang1 = atan2(dy, dx);
326  if (ang1 < 0) {
327  ang1 += 2.0 * PI;
328  }
329  dx = (vertex.m_p.x - vertex.m_c.x) * CArea::m_units;
330  dy = (vertex.m_p.y - vertex.m_c.y) * CArea::m_units;
331  ang2 = atan2(dy, dx);
332  if (ang2 < 0) {
333  ang2 += 2.0 * PI;
334  }
335 
336  if (vertex.m_type == -1) { // clockwise
337  if (ang2 > ang1) {
338  phit = 2.0 * PI - ang2 + ang1;
339  }
340  else {
341  phit = ang1 - ang2;
342  }
343  }
344  else { // counter_clockwise
345  if (ang1 > ang2) {
346  phit = -(2.0 * PI - ang1 + ang2);
347  }
348  else {
349  phit = -(ang2 - ang1);
350  }
351  }
352 
353  // what is the delta phi to get an accuracy of aber
354  double radius = sqrt(dx * dx + dy * dy);
355  dphi = 2 * acos((radius - CArea::m_accuracy) / radius);
356 
357  // set the number of segments
358  if (phit > 0) {
359  Segments = (int)ceil(phit / dphi);
360  }
361  else {
362  Segments = (int)ceil(-phit / dphi);
363  }
364 
365  if (Segments < 1) {
366  Segments = 1;
367  }
368  if (Segments > 100) {
369  Segments = 100;
370  }
371 
372  dphi = phit / (Segments);
373 
374  double px = prev_vertex->m_p.x * CArea::m_units;
375  double py = prev_vertex->m_p.y * CArea::m_units;
376 
377  for (i = 1; i <= Segments; i++) {
378  dx = px - vertex.m_c.x * CArea::m_units;
379  dy = py - vertex.m_c.y * CArea::m_units;
380  phi = atan2(dy, dx);
381 
382  double nx = vertex.m_c.x * CArea::m_units + radius * cos(phi - dphi);
383  double ny = vertex.m_c.y * CArea::m_units + radius * sin(phi - dphi);
384 
385  new_pts.emplace_back(nx, ny);
386 
387  px = nx;
388  py = ny;
389  }
390  }
391  }
392  prev_vertex = &vertex;
393  }
394 
395  m_vertices.clear();
396 
397  for (std::list<Point>::iterator It = new_pts.begin(); It != new_pts.end(); It++) {
398  Point& pt = *It;
399  CVertex vertex(0, pt / CArea::m_units, Point(0.0, 0.0));
400  m_vertices.push_back(vertex);
401  }
402 }

608 void CCurve::ExtractSeparateCurves(const std::list<Point>& ordered_points,
609  std::list<CCurve>& separate_curves) const
610 {
611  // returns separate curves for this curve split at points
612  // the points must be in order along this curve, already, and lie on this curve
613  const Point* prev_p = NULL;
614 
615  if (ordered_points.size() == 0) {
616  separate_curves.push_back(*this);
617  return;
618  }
619 
620  CCurve current_curve;
621 
622  std::list<Point>::const_iterator PIt = ordered_points.begin();
623  Point point = *PIt;
624 
625  for (std::list<CVertex>::const_iterator VIt = m_vertices.begin(); VIt != m_vertices.end();
626  VIt++) {
627  const CVertex& vertex = *VIt;
628  if (prev_p) // not the first vertex
629  {
630  Span span(*prev_p, vertex);
631  while ((PIt != ordered_points.end()) && span.On(point)) {
632  CVertex v(vertex);
633  v.m_p = point;
634  current_curve.m_vertices.push_back(v);
635  if (current_curve.m_vertices.size() > 1) { // don't add single point curves
636  separate_curves.push_back(current_curve); // add the curve
637  }
638  current_curve = CCurve(); // make a new curve
639  current_curve.m_vertices.push_back(v); // add it's first point
640  PIt++;
641  if (PIt != ordered_points.end()) {
642  point = *PIt; // increment the point
643  }
644  }
645 
646  // add the end of span
647  if (current_curve.m_vertices.back().m_p != vertex.m_p) {
648  current_curve.m_vertices.push_back(vertex);
649  }
650  }
651  if ((current_curve.m_vertices.size() == 0)
652  || (current_curve.m_vertices.back().m_p != vertex.m_p)) {
653  // very first vertex, start the current curve
654  current_curve.m_vertices.push_back(vertex);
655  }
656  prev_p = &(vertex.m_p);
657  }
658 
659  // add whatever is left
660  if (current_curve.m_vertices.size() > 1) { // don't add single point curves
661  separate_curves.push_back(current_curve); // add the curve
662  }
663 }

782 bool CCurve::Offset(double leftwards_value)
783 {
784  // use the kurve code donated by Geoff Hawkesford, to offset the curve as an open curve
785  // returns true for success, false for failure
786  bool success = true;
787 
788  CCurve save_curve = *this;
789 
790  try {
791  geoff_geometry::Kurve k = MakeKurve(*this);
792  geoff_geometry::Kurve kOffset;
793  int ret = 0;
794  k.OffsetMethod1(kOffset, fabs(leftwards_value), (leftwards_value > 0) ? 1 : -1, 1, ret);
795  success = (ret == 0);
796  if (success) {
797  *this = MakeCCurve(kOffset);
798  }
799  }
800  catch (...) {
801  success = false;
802  }
803 
804  if (!success) {
805  if (this->IsClosed()) {
806  double inwards_offset = leftwards_value;
807  bool cw = false;
808  if (this->IsClockwise()) {
809  inwards_offset = -inwards_offset;
810  cw = true;
811  }
812  CArea a;
813  a.append(*this);
814  a.Offset(inwards_offset);
815  if (a.m_curves.size() == 1) {
816  Span* start_span = NULL;
817  if (this->m_vertices.size() > 1) {
818  std::list<CVertex>::iterator It = m_vertices.begin();
819  CVertex& v0 = *It;
820  It++;
821  CVertex& v1 = *It;
822  start_span = new Span(v0.m_p, v1, true);
823  }
824  *this = a.m_curves.front();
825  if (this->IsClockwise() != cw) {
826  this->Reverse();
827  }
828  if (start_span) {
829  Point forward = start_span->GetVector(0.0);
830  Point left(-forward.y, forward.x);
831  Point offset_start = start_span->m_p + left * leftwards_value;
832  this->ChangeStart(this->NearestPoint(offset_start));
833  delete start_span;
834  }
835  success = true;
836  }

855 void CCurve::OffsetForward(double forwards_value, bool refit_arcs)
856 {
857  // for drag-knife compensation
858 
859  // replace arcs with lines
860  UnFitArcs();
861 
862  std::list<Span> spans;
863  GetSpans(spans);
864 
865  m_vertices.clear();
866 
867  // shift all the spans
868  for (std::list<Span>::iterator It = spans.begin(); It != spans.end(); It++) {
869  Span& span = *It;
870  Point v = span.GetVector(0.0);
871  v.normalize();
872  Point shift = v * forwards_value;
873  span.m_p = span.m_p + shift;
874  span.m_v.m_p = span.m_v.m_p + shift;
875  }
876 
877  // loop through the shifted spans
878  for (std::list<Span>::iterator It = spans.begin(); It != spans.end();) {
879  Span& span = *It;
880  Point v = span.GetVector(0.0);
881  v.normalize();
882 
883  // add the span
884  if (It == spans.begin()) {
885  m_vertices.push_back(span.m_p);
886  }
887  m_vertices.push_back(span.m_v.m_p);
888 
889  It++;
890  if (It != spans.end()) {
891  Span& next_span = *It;
892  Point nv = next_span.GetVector(0.0);
893  nv.normalize();
894  double sin_angle = v ^ nv;
895  bool sharp_corner = (fabs(sin_angle) > 0.5); // angle > 30 degrees
896 
897  if (sharp_corner) {
898  // add an arc to the start of the next span
899  int arc_type = ((sin_angle > 0) ? 1 : (-1));
900  Point centre = span.m_v.m_p - v * forwards_value;
901  m_vertices.emplace_back(arc_type, next_span.m_p, centre);
902  }
903  }
904  }
905 
906  if (refit_arcs) {
907  FitArcs(); // find the arcs again
908  }
909  else {
910  UnFitArcs(); // convert those little arcs added to lines
911  }
912 }

1227 double IncludedAngle(const Point& v0, const Point& v1, int dir)
1228 {
1229  // returns the absolute included angle between 2 vectors in the direction of dir ( 1=acw -1=cw)
1230  double inc_ang = v0 * v1;
1231  if (inc_ang > 1. - 1.0e-10) {
1232  return 0;
1233  }
1234  if (inc_ang < -1. + 1.0e-10) {
1235  inc_ang = PI;
1236  }
1237  else { // dot product, v1 . v2 = cos ang
1238  if (inc_ang > 1.0) {
1239  inc_ang = 1.0;
1240  }
1241  inc_ang = acos(inc_ang); // 0 to pi radians
1242 
1243  if (dir * (v0 ^ v1) < 0) {
1244  inc_ang = 2 * PI - inc_ang; // cp
1245  }
1246  }
1247  return dir * inc_ang;
1248 }
