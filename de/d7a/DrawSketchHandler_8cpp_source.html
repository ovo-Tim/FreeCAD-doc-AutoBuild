193 std::vector<Base::Vector2d> CurveConverter::toVector2D(const Part::Geometry* geometry)
194 {
195  std::vector<Base::Vector2d> vector2d;
196 
197  const auto type = geometry->getTypeId();
198 
199  auto emplaceasvector2d = [&vector2d](const Base::Vector3d& point) {
200  vector2d.emplace_back(point.x, point.y);
201  };
202 
203  auto isconic = type.isDerivedFrom(Part::GeomConic::getClassTypeId());
204  auto isbounded = type.isDerivedFrom(Part::GeomBoundedCurve::getClassTypeId());
205 
206  if (type == Part::GeomLineSegment::getClassTypeId()) { // add a line
207  auto geo = static_cast<const Part::GeomLineSegment*>(geometry);
208 
209  emplaceasvector2d(geo->getStartPoint());
210  emplaceasvector2d(geo->getEndPoint());
211  }
212  else if (isconic || isbounded) {
213 
214  auto geo = static_cast<const Part::GeomConic*>(geometry);
215 
216  double segment =
217  (geo->getLastParameter() - geo->getFirstParameter()) / curvedEdgeCountSegments;
218 
219  for (int i = 0; i < curvedEdgeCountSegments; i++) {
220  emplaceasvector2d(geo->value(geo->getFirstParameter() + i * segment));
221  }
222 
223  // either close the curve for untrimmed conic or set the last point for bounded curves
224  emplaceasvector2d(isconic ? geo->value(0) : geo->value(geo->getLastParameter()));
225  }
226 
227  return vector2d;
228 }

381 std::vector<QPixmap>
382 DrawSketchHandler::suggestedConstraintsPixmaps(std::vector<AutoConstraint>& suggestedConstraints)
383 {
384  std::vector<QPixmap> pixmaps;
385  // Iterate through AutoConstraints types and get their pixmaps
386  for (auto& autoCstr : suggestedConstraints) {
387  QString iconType;
388  switch (autoCstr.Type) {
389  case Horizontal:
390  iconType = QString::fromLatin1("Constraint_Horizontal");
391  break;
392  case Vertical:
393  iconType = QString::fromLatin1("Constraint_Vertical");
394  break;
395  case Coincident:
396  iconType = QString::fromLatin1("Constraint_PointOnPoint");
397  break;
398  case PointOnObject:
399  iconType = QString::fromLatin1("Constraint_PointOnObject");
400  break;
401  case Symmetric:
402  iconType = QString::fromLatin1("Constraint_Symmetric");
403  break;
404  case Tangent:
405  iconType = QString::fromLatin1("Constraint_Tangent");
406  break;
407  default:
408  break;
409  }
410  if (!iconType.isEmpty()) {
411  qreal pixelRatio = 1;
412  Gui::View3DInventorViewer* viewer = getViewer();
413  if (viewer) {
414  pixelRatio = viewer->devicePixelRatio();
415  }
416  int iconWidth = 16 * pixelRatio;
417  QPixmap icon = Gui::BitmapFactory().pixmapFromSvg(iconType.toStdString().c_str(),
418  QSize(iconWidth, iconWidth));
419  pixmaps.push_back(icon);
420  }
421  }
422  return pixmaps;
423 }

425 int DrawSketchHandler::seekAutoConstraint(std::vector<AutoConstraint>& suggestedConstraints,
426  const Base::Vector2d& Pos,
427  const Base::Vector2d& Dir,
428  AutoConstraint::TargetType type)
429 {
430  suggestedConstraints.clear();
431  SketchObject* obj = sketchgui->getSketchObject();
432 
433  if (!sketchgui->Autoconstraints.getValue()) {
434  return 0; // If Autoconstraints property is not set quit
435  }
436  // direction of hit shape (if it is a line, the direction of the line)
437  Base::Vector3d hitShapeDir = Base::Vector3d(0, 0, 0);
438  bool preselectIsLine = false;
439 
440  // Get Preselection
441  int preSelPnt = getPreselectPoint();
442  int preSelCrv = getPreselectCurve();
443  int preSelCrs = getPreselectCross();
444 
445  int GeoId = GeoEnum::GeoUndef;
446  PointPos PosId = PointPos::none;
447 
448  if (preSelPnt != -1) {
449  obj->getGeoVertexIndex(preSelPnt, GeoId, PosId);
450  }
451  else if (preSelCrv != -1) {
452  const Part::Geometry* geom = obj->getGeometry(preSelCrv);
453 
454  // ensure geom exists in case object was called before preselection is updated
455  if (geom) {
456  GeoId = preSelCrv;
457  if (geom->is<Part::GeomLineSegment>()) {
458  auto* line = static_cast<const Part::GeomLineSegment*>(geom);
459  hitShapeDir = line->getEndPoint() - line->getStartPoint();
460  preselectIsLine = true;
461  }
462  }
463  }
464  else if (preSelCrs == 0) {
465  // root point
466  GeoId = Sketcher::GeoEnum::RtPnt;
467  PosId = PointPos::start;
468  }
469  else if (preSelCrs == 1) {
470  // x axis
471  GeoId = Sketcher::GeoEnum::HAxis;
472  hitShapeDir = Base::Vector3d(1, 0, 0);
473  preselectIsLine = true;
474  }
475  else if (preSelCrs == 2) {
476  // y axis
477  GeoId = Sketcher::GeoEnum::VAxis;
478  hitShapeDir = Base::Vector3d(0, 1, 0);
479  preselectIsLine = true;
480  }
481 
482  if (GeoId != GeoEnum::GeoUndef) {
483  // Currently only considers objects in current Sketcher
484  AutoConstraint constr;
485  constr.Type = Sketcher::None;
486  constr.GeoId = GeoId;
487  constr.PosId = PosId;
488  if (type == AutoConstraint::VERTEX || type == AutoConstraint::VERTEX_NO_TANGENCY) {
489  if (PosId == PointPos::none) {
490  bool lineCenter = false;
491  auto* geo = obj->getGeometry(GeoId);
492  if (geo->is<Part::GeomLineSegment>()) {
493  auto* line = static_cast<const Part::GeomLineSegment*>(geo);
494 
495  Base::Vector2d startPoint = toVector2d(line->getStartPoint());
496  Base::Vector2d endPoint = toVector2d(line->getEndPoint());
497  Base::Vector2d midPoint = (startPoint + endPoint) / 2;
498 
499  // Check if we are at middle of the line
500  if ((Pos - midPoint).Length() < (endPoint - startPoint).Length() * 0.05) {
501  lineCenter = true;
502  }
503  }
504 
505  constr.Type = lineCenter ? Sketcher::Symmetric : Sketcher::PointOnObject;
506  }
507  else {
508  constr.Type = Sketcher::Coincident;
509  }
510  }
511  else if (type == AutoConstraint::CURVE && PosId != PointPos::none) {
512  constr.Type = Sketcher::PointOnObject;
513  }
514  else if (type == AutoConstraint::CURVE && PosId == PointPos::none) {
515  constr.Type = Sketcher::Tangent;
516  }
517 
518  if (constr.Type == Sketcher::Tangent && preselectIsLine) {
519  if (Dir.Length() < 1e-8 || hitShapeDir.Length() < 1e-8) {
520  // Direction not set so return;
521  return suggestedConstraints.size();
522  }
523  // We are hitting a line and have hitting vector information
524  Base::Vector3d dir3d = Base::Vector3d(Dir.x, Dir.y, 0);
525  double cosangle = dir3d.Normalize() * hitShapeDir.Normalize();
526 
527  // the angle between the line and the hitting direction are over around 6 degrees (it is
528  // substantially parallel) or if it is an sketch axis (that can not move to accommodate
529  // to the shape), then only if it is around 6 degrees with the normal (around 84
530  // degrees)
531  if (fabs(cosangle) < 0.995f
532  || ((GeoId == Sketcher::GeoEnum::HAxis || GeoId == Sketcher::GeoEnum::VAxis)
533  && fabs(cosangle) < 0.1)) {
534  suggestedConstraints.push_back(constr);
535  }
536 
537 
538  return suggestedConstraints.size();
539  }
540 
541  if (constr.Type != Sketcher::None) {
542  suggestedConstraints.push_back(constr);
543  }
544  }
545 
546  if (Dir.Length() < 1e-8 || type == AutoConstraint::CURVE) {
547  // Direction not set so return;
548  return suggestedConstraints.size();
549  }
550 
551  // Suggest vertical and horizontal constraints
552 
553  // Number of Degree of deviation from horizontal or vertical lines
554  const double angleDev = 2;
555  const double angleDevRad = angleDev * M_PI / 180.;
556 
557  AutoConstraint constr;
558  constr.Type = Sketcher::None;
559  constr.GeoId = GeoEnum::GeoUndef;
560  constr.PosId = PointPos::none;
561  double angle = std::abs(atan2(Dir.y, Dir.x));
562  if (angle < angleDevRad || (M_PI - angle) < angleDevRad) {
563  // Suggest horizontal constraint
564  constr.Type = Sketcher::Horizontal;
565  }
566  else if (std::abs(angle - M_PI_2) < angleDevRad) {
567  // Suggest vertical constraint
568  constr.Type = Sketcher::Vertical;
569  }
570 
571  if (constr.Type != Sketcher::None) {
572  suggestedConstraints.push_back(constr);
573  }
574 
575  // Do not seek for tangent if we are actually building a primitive
576  if (type == AutoConstraint::VERTEX_NO_TANGENCY) {
577  return suggestedConstraints.size();
578  }
579 
580  // Find if there are tangent constraints (currently arcs and circles)
581 
582  int tangId = GeoEnum::GeoUndef;
583 
584  // Do not consider if distance is more than that.
585  // Decrease this value when a candidate is found.
586  double tangDeviation = 0.1 * sketchgui->getScaleFactor();
587 
588  // Get geometry list
589  const std::vector<Part::Geometry*> geomlist = obj->getCompleteGeometry();
590 
591  Base::Vector3d tmpPos(Pos.x, Pos.y, 0.f); // Current cursor point
592  Base::Vector3d tmpDir(Dir.x, Dir.y, 0.f); // Direction of line
593  Base::Vector3d tmpStart(Pos.x - Dir.x, Pos.y - Dir.y, 0.f); // Start point
594 
595  int i = -1;
596  for (auto* geo : geomlist) {
597  i++;
598 
599  if (geo->is<Part::GeomCircle>()) {
600  auto* circle = static_cast<const Part::GeomCircle*>(geo);
601 
602  Base::Vector3d center = circle->getCenter();
603 
604  double radius = circle->getRadius();
605 
606  // ignore if no touch (use dot product)
607  if (tmpDir * (center - tmpPos) > 0 || tmpDir * (center - tmpStart) < 0) {
608  continue;
609  }
610 
611  Base::Vector3d projPnt(0.f, 0.f, 0.f);
612  projPnt = projPnt.ProjectToLine(center - tmpPos, tmpDir);
613  double projDist = std::abs(projPnt.Length() - radius);
614 
615  // Find if nearest
616  if (projDist < tangDeviation) {
617  tangId = i;
618  tangDeviation = projDist;
619  }
620  }
621  else if (geo->is<Part::GeomEllipse>()) {
622  auto* ellipse = static_cast<const Part::GeomEllipse*>(geo);
623 
624  Base::Vector3d center = ellipse->getCenter();
625 
626  double a = ellipse->getMajorRadius();
627  double b = ellipse->getMinorRadius();
628  Base::Vector3d majdir = ellipse->getMajorAxisDir();
629 
630  double cf = sqrt(a * a - b * b);
631 
632  Base::Vector3d focus1P = center + cf * majdir;
633  Base::Vector3d focus2P = center - cf * majdir;
634 
635  Base::Vector3d norm = Base::Vector3d(Dir.y, -Dir.x).Normalize();
636 
637  double distancetoline = norm * (tmpPos - focus1P); // distance focus1 to line
638 
639  // mirror of focus1 with respect to the line
640  Base::Vector3d focus1PMirrored = focus1P + 2 * distancetoline * norm;
641 
642  double error = fabs((focus1PMirrored - focus2P).Length() - 2 * a);
643 
644  if (error < tangDeviation) {
645  tangId = i;
646  tangDeviation = error;
647  }
648  }
649  else if (geo->is<Part::GeomArcOfCircle>()) {
650  auto* arc = static_cast<const Part::GeomArcOfCircle*>(geo);
651 
652  Base::Vector3d center = arc->getCenter();
653  double radius = arc->getRadius();
654 
655  // ignore if no touch (use dot product)
656  if (tmpDir * (center - tmpPos) > 0 || tmpDir * (center - tmpStart) < 0) {
657  continue;
658  }
659 
660  Base::Vector3d projPnt(0.f, 0.f, 0.f);
661  projPnt = projPnt.ProjectToLine(center - tmpPos, tmpDir);
662  double projDist = std::abs(projPnt.Length() - radius);
663 
664  if (projDist < tangDeviation) {
665  double startAngle, endAngle;
666  arc->getRange(startAngle, endAngle, /*emulateCCW=*/true);
667 
668  double angle = atan2(projPnt.y, projPnt.x);
669  while (angle < startAngle) {
670  angle += 2 * D_PI; // Bring it to range of arc
671  }
672 
673  // if the point is on correct side of arc
674  if (angle <= endAngle) { // Now need to check only one side
675  tangId = i;
676  tangDeviation = projDist;
677  }
678  }
679  }
680  else if (geo->is<Part::GeomArcOfEllipse>()) {
681  auto* aoe = static_cast<const Part::GeomArcOfEllipse*>(geo);
682 
683  Base::Vector3d center = aoe->getCenter();
684 
685  double a = aoe->getMajorRadius();
686  double b = aoe->getMinorRadius();
687  Base::Vector3d majdir = aoe->getMajorAxisDir();
688 
689  double cf = sqrt(a * a - b * b);
690 
691  Base::Vector3d focus1P = center + cf * majdir;
692  Base::Vector3d focus2P = center - cf * majdir;
693 
694  Base::Vector3d norm = Base::Vector3d(Dir.y, -Dir.x).Normalize();
695 
696  double distancetoline = norm * (tmpPos - focus1P); // distance focus1 to line
697 
698  Base::Vector3d focus1PMirrored =
699  focus1P + 2 * distancetoline * norm; // mirror of focus1 with respect to the line
700 
701  double error = fabs((focus1PMirrored - focus2P).Length() - 2 * a);
702 
703  if (error < tangDeviation) {
704  tangId = i;
705  tangDeviation = error;
706  }
707 
708  if (error < tangDeviation) {
709  double startAngle, endAngle;
710  aoe->getRange(startAngle, endAngle, /*emulateCCW=*/true);
711 
712  double angle = Base::fmod(
713  atan2(
714  -aoe->getMajorRadius()
715  * ((tmpPos.x - center.x) * majdir.y - (tmpPos.y - center.y) * majdir.x),
716  aoe->getMinorRadius()
717  * ((tmpPos.x - center.x) * majdir.x + (tmpPos.y - center.y) * majdir.y))
718  - startAngle,
719  2.f * M_PI);
720 
721  while (angle < startAngle) {
722  angle += 2 * D_PI; // Bring it to range of arc
723  }
724 
725  // if the point is on correct side of arc
726  if (angle <= endAngle) { // Now need to check only one side
727  tangId = i;
728  tangDeviation = error;
729  }
730  }
731  }
732  }
733 
734  if (tangId != GeoEnum::GeoUndef) {
735  if (tangId > getHighestCurveIndex()) { // external Geometry
736  tangId = getHighestCurveIndex() - tangId;
737  }
738  // Suggest vertical constraint
739  constr.Type = Tangent;
740  constr.GeoId = tangId;
741  constr.PosId = PointPos::none;
742  suggestedConstraints.push_back(constr);
743  }
744 
745  return suggestedConstraints.size();
746 }

748 void DrawSketchHandler::createAutoConstraints(const std::vector<AutoConstraint>& autoConstrs,
749  int geoId1,
750  Sketcher::PointPos posId1,
751  bool createowncommand /*= true*/)
752 {
753  if (!sketchgui->Autoconstraints.getValue()) {
754  return; // If Autoconstraints property is not set quit
755  }
756 
757  if (!autoConstrs.empty()) {
758 
759  if (createowncommand) {
760  // Open the Command
761  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Add auto constraints"));
762  }
763 
764  // Iterate through constraints
765  for (auto& cstr : autoConstrs) {
766  int geoId2 = cstr.GeoId;
767 
768  switch (cstr.Type) {
769  case Sketcher::Coincident: {
770  if (posId1 == Sketcher::PointPos::none) {
771  continue;
772  }
773  // If the auto constraint has a point create a coincident otherwise it is an
774  // edge on a point
775  Gui::cmdAppObjectArgs(
776  sketchgui->getObject(),
777  "addConstraint(Sketcher.Constraint('Coincident',%d,%d,%d,%d)) ",
778  geoId1,
779  static_cast<int>(posId1),
780  cstr.GeoId,
781  static_cast<int>(cstr.PosId));
782  } break;
783  case Sketcher::PointOnObject: {
784  Sketcher::PointPos posId2 = cstr.PosId;
785  if (posId1 == Sketcher::PointPos::none) {
786  // Auto constraining an edge so swap parameters
787  std::swap(geoId1, geoId2);
788  std::swap(posId1, posId2);
789  }
790 
791  Gui::cmdAppObjectArgs(
792  sketchgui->getObject(),
793  "addConstraint(Sketcher.Constraint('PointOnObject',%d,%d,%d)) ",
794  geoId1,
795  static_cast<int>(posId1),
796  geoId2);
797  } break;
798  case Sketcher::Symmetric: {
799  Gui::cmdAppObjectArgs(
800  sketchgui->getObject(),
801  "addConstraint(Sketcher.Constraint('Symmetric',%d,1,%d,2,%d,%d)) ",
802  geoId2,
803  geoId2,
804  geoId1,
805  static_cast<int>(posId1));
806  } break;
807  // In special case of Horizontal/Vertical constraint, geoId2 is normally unused
808  // and should be 'Constraint::GeoUndef' However it can be used as a way to
809  // require the function to apply these constraints on another geometry In this
810  // case the caller as to set geoId2, then it will be used as target instead of
811  // geoId2
812  case Sketcher::Horizontal: {
813  Gui::cmdAppObjectArgs(sketchgui->getObject(),
814  "addConstraint(Sketcher.Constraint('Horizontal',%d)) ",
815  geoId2 != GeoEnum::GeoUndef ? geoId2 : geoId1);
816  } break;
817  case Sketcher::Vertical: {
818  Gui::cmdAppObjectArgs(sketchgui->getObject(),
819  "addConstraint(Sketcher.Constraint('Vertical',%d)) ",
820  geoId2 != GeoEnum::GeoUndef ? geoId2 : geoId1);
821  } break;
822  case Sketcher::Tangent: {
823  Sketcher::SketchObject* Obj = sketchgui->getSketchObject();
824 
825  const Part::Geometry* geom1 = Obj->getGeometry(geoId1);
826  const Part::Geometry* geom2 = Obj->getGeometry(cstr.GeoId);
827 
828  // ellipse tangency support using construction elements (lines)
829  if (geom1 && geom2
830  && (geom1->is<Part::GeomEllipse>() || geom2->is<Part::GeomEllipse>())) {
831 
832  if (!geom1->is<Part::GeomEllipse>()) {
833  std::swap(geoId1, geoId2);
834  }
835 
836  // geoId1 is the ellipse
837  geom1 = Obj->getGeometry(geoId1);
838  geom2 = Obj->getGeometry(geoId2);
839 
840  if (geom2->is<Part::GeomEllipse>() || geom2->is<Part::GeomArcOfEllipse>()
841  || geom2->is<Part::GeomCircle>()
842  || geom2->is<Part::GeomArcOfCircle>()) {
843  // in all these cases an intermediate element is needed
844  makeTangentToEllipseviaNewPoint(
845  Obj,
846  static_cast<const Part::GeomEllipse*>(geom1),
847  geom2,
848  geoId1,
849  geoId2);
850  return;
851  }
852  }
853 
854  // arc of ellipse tangency support using external elements
855  if (geom1 && geom2
856  && (geom1->is<Part::GeomArcOfEllipse>()
857  || geom2->is<Part::GeomArcOfEllipse>())) {
858 
859  if (!geom1->is<Part::GeomArcOfEllipse>()) {
860  std::swap(geoId1, geoId2);
861  }
862 
863  // geoId1 is the arc of ellipse
864  geom1 = Obj->getGeometry(geoId1);
865  geom2 = Obj->getGeometry(geoId2);
866 
867  if (geom2->is<Part::GeomArcOfEllipse>() || geom2->is<Part::GeomCircle>()
868  || geom2->is<Part::GeomArcOfCircle>()) {
869  // in all these cases an intermediate element is needed
870  makeTangentToArcOfEllipseviaNewPoint(
871  Obj,
872  static_cast<const Part::GeomArcOfEllipse*>(geom1),
873  geom2,
874  geoId1,
875  geoId2);
876  return;
877  }
878  }
879 
880  Gui::cmdAppObjectArgs(sketchgui->getObject(),
881  "addConstraint(Sketcher.Constraint('Tangent',%d, %d)) ",
882  geoId1,
883  cstr.GeoId);
884  } break;
885  default:
886  break;
887  }
888 
889  if (createowncommand) {
890  Gui::Command::commitCommand();
891  }
892  // Gui::Command::updateActive(); // There is already an recompute in each command
893  // creation, this is redundant.
894  }
895  }
896 }
