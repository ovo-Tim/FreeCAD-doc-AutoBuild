265 bool FeaturePythonImp::getSubObject(DocumentObject*& ret,
266  const char* subname,
267  PyObject** pyObj,
268  Base::Matrix4D* _mat,
269  bool transform,
270  int depth) const
271 {
272  FC_PY_CALL_CHECK(getSubObject);
273  Base::PyGILStateLocker lock;
274  try {
275  Py::Tuple args(6);
276  args.setItem(0, Py::Object(object->getPyObject(), true));
277  if (!subname) {
278  subname = "";
279  }
280  args.setItem(1, Py::String(subname));
281  args.setItem(2, Py::Int(pyObj ? 2 : 1));
282  Base::MatrixPy* pyMat = new Base::MatrixPy(new Base::Matrix4D);
283  if (_mat) {
284  *pyMat->getMatrixPtr() = *_mat;
285  }
286  args.setItem(3, Py::asObject(pyMat));
287  args.setItem(4, Py::Boolean(transform));
288  args.setItem(5, Py::Int(depth));
289 
290  Py::Object res(Base::pyCall(py_getSubObject.ptr(), args.ptr()));
291  if (res.isNone()) {
292  ret = nullptr;
293  return true;
294  }
295  if (!res.isTrue()) {
296  return false;
297  }
298  if (!res.isSequence()) {
299  throw Py::TypeError("getSubObject expects return type of tuple");
300  }
301  Py::Sequence seq(res);
302  if (seq.length() < 2
303  || (!seq.getItem(0).isNone()
304  && !PyObject_TypeCheck(seq.getItem(0).ptr(), &DocumentObjectPy::Type))
305  || !PyObject_TypeCheck(seq.getItem(1).ptr(), &Base::MatrixPy::Type)) {
306  throw Py::TypeError("getSubObject expects return type of (obj,matrix,pyobj)");
307  }
308  if (_mat) {
309  *_mat = *static_cast<Base::MatrixPy*>(seq.getItem(1).ptr())->getMatrixPtr();
310  }
311  if (pyObj) {
312  if (seq.length() > 2) {
313  *pyObj = Py::new_reference_to(seq.getItem(2));
314  }
315  else {
316  *pyObj = Py::new_reference_to(Py::None());
317  }
318  }
319  if (seq.getItem(0).isNone()) {
320  ret = nullptr;
321  }
322  else {
323  ret = static_cast<DocumentObjectPy*>(seq.getItem(0).ptr())->getDocumentObjectPtr();
324  }
325  return true;
326  }
327  catch (Py::Exception&) {
328  if (PyErr_ExceptionMatches(PyExc_NotImplementedError)) {
329  PyErr_Clear();
330  return false;
331  }
332  Base::PyException e; // extract the Python error text
333  e.ReportException();
334  ret = nullptr;
335  return true;
336  }
