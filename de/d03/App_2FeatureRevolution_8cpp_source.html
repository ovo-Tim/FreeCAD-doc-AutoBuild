81 bool Revolution::fetchAxisLink(const App::PropertyLinkSub &axisLink,
82  Base::Vector3d& center,
83  Base::Vector3d& dir,
84  double& angle)
85 {
86  if (!axisLink.getValue())
87  return false;
88 
89  auto linked = axisLink.getValue();
90 
91  TopoDS_Shape axEdge;
92  if (!axisLink.getSubValues().empty() && axisLink.getSubValues()[0].length() > 0){
93  axEdge = Feature::getTopoShape(linked, axisLink.getSubValues()[0].c_str(), true /*need element*/).getShape();
94  } else {
95  axEdge = Feature::getShape(linked);
96  }
97 
98  if (axEdge.IsNull())
99  throw Base::ValueError("AxisLink shape is null");
100  if (axEdge.ShapeType() != TopAbs_EDGE)
101  throw Base::TypeError("AxisLink shape is not an edge");
102 
103  BRepAdaptor_Curve crv(TopoDS::Edge(axEdge));
104  gp_Pnt base;
105  gp_Dir occdir;
106  bool reversed = axEdge.Orientation() == TopAbs_REVERSED;
107  if (crv.GetType() == GeomAbs_Line){
108  base = crv.Value(reversed ? crv.FirstParameter() : crv.LastParameter());
109  occdir = crv.Line().Direction();
110  } else if (crv.GetType() == GeomAbs_Circle) {
111  base = crv.Circle().Axis().Location();
112  occdir = crv.Circle().Axis().Direction();
113  angle = crv.LastParameter() - crv.FirstParameter();
114  } else {
115  throw Base::TypeError("AxisLink edge is neither line nor arc of circle.");
116  }
117  if (reversed)
118  occdir.Reverse();
119  center.Set(base.X(), base.Y(),base.Z());
120  dir.Set(occdir.X(), occdir.Y(), occdir.Z());
121  return true;
122 }

124 App::DocumentObjectExecReturn *Revolution::execute()
125 {
126  App::DocumentObject* link = Source.getValue();
127  if (!link)
128  return new App::DocumentObjectExecReturn("No object linked");
129 
130  try {
131  //read out axis link
132  double angle_edge = 0;
133  Base::Vector3d b = Base.getValue();
134  Base::Vector3d v = Axis.getValue();
135  bool linkFetched = this->fetchAxisLink(this->AxisLink, b, v, angle_edge);
136  if (linkFetched){
137  this->Base.setValue(b);
138  this->Axis.setValue(v);
139  }
140 
141  gp_Pnt pnt(b.x,b.y,b.z);
142  gp_Dir dir(v.x,v.y,v.z);
143  gp_Ax1 revAx(pnt, dir);
144 
145  //read out revolution angle
146  double angle = Angle.getValue()/180.0f*M_PI;
147  if (fabs(angle) < Precision::Angular())
148  angle = angle_edge;
149 
150  //apply "midplane" symmetry
151  TopoShape sourceShape = Feature::getTopoShape(link);
152  if (Symmetric.getValue()) {
153  //rotate source shape backwards by half angle, to make resulting revolution symmetric to the profile
154  gp_Trsf mov;
155  mov.SetRotation(revAx, angle * (-0.5));
156  TopLoc_Location loc(mov);
157  sourceShape.setShape(sourceShape.getShape().Moved(loc));
158  }
159  TopoShape revolve(0);
160  revolve.makeElementRevolve(sourceShape,
161  revAx,
162  angle,
163  Solid.getValue() ? FaceMakerClass.getValue() : 0);
164  if (revolve.isNull()) {
165  return new App::DocumentObjectExecReturn("Resulting shape is null");
166  }
167  this->Shape.setValue(revolve);
168  return Part::Feature::execute();
169  }
170  catch (Standard_Failure& e) {
171  return new App::DocumentObjectExecReturn(e.GetMessageString());
172  }
173 }
