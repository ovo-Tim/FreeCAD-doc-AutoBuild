107 void ViewProviderDatumPlane::setExtents (Base::BoundBox3d bbox) {
108  PartDesign::Plane* pcDatum = this->getObject<PartDesign::Plane>();
109  if (pcDatum->ResizeMode.getValue() != 0) {
110  setExtents(pcDatum->Length.getValue(), pcDatum->Width.getValue());
111  return;
112  }
113 
114  Base::Placement plm = pcDatum->Placement.getValue ().inverse ();
115 
116  // Transform the box to the line's coordinates, the result line will be larger than the bbox
117  bbox = bbox.Transformed ( plm.toMatrix() );
118  // Add origin of the plane to the box if it's not
119  bbox.Add ( Base::Vector3d (0, 0, 0) );
120 
121  double margin = sqrt(bbox.LengthX ()*bbox.LengthY ()) * marginFactor ();
122 
123  pcDatum->Length.setValue(bbox.LengthX() + 2*margin);
124  pcDatum->Width.setValue(bbox.LengthY() + 2*margin);
125 
126  // Change the coordinates of the line
127  pCoords->point.setNum (4);
128  pCoords->point.set1Value(0, bbox.MaxX + margin, bbox.MaxY + margin, 0 );
129  pCoords->point.set1Value(1, bbox.MinX - margin, bbox.MaxY + margin, 0 );
130  pCoords->point.set1Value(2, bbox.MinX - margin, bbox.MinY - margin, 0 );
131  pCoords->point.set1Value(3, bbox.MaxX + margin, bbox.MinY - margin, 0 );
132 }
