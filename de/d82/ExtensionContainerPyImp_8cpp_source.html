44 int ExtensionContainerPy::initialization()
45 {
46 
47  if (!this->ob_type->tp_dict) {
48  if (PyType_Ready(this->ob_type) < 0) {
49  return 0;
50  }
51  }
52 
53  ExtensionContainer::ExtensionIterator it = this->getExtensionContainerPtr()->extensionBegin();
54  for (; it != this->getExtensionContainerPtr()->extensionEnd(); ++it) {
55 
56  // The PyTypeObject is shared by all instances of this type and therefore
57  // we have to add new methods only once.
58  PyObject* obj = (*it).second->getExtensionPyObject();
59  PyMethodDef* meth = obj->ob_type->tp_methods;
60  PyTypeObject* type = this->ob_type;
61  PyObject* dict = type->tp_dict;
62 
63  // make sure to do the initialization only once
64  if (meth->ml_name) {
65  PyObject* item = PyDict_GetItemString(dict, meth->ml_name);
66  if (!item) {
67  // Note: this adds the methods to the type object to make sure
68  // it appears in the call tips. The function will not be bound
69  // to an instance
70  Py_INCREF(dict);
71  while (meth->ml_name) {
72  PyObject* func;
73  func = PyCFunction_New(meth, 0);
74  if (!func) {
75  break;
76  }
77  if (PyDict_SetItemString(dict, meth->ml_name, func) < 0) {
78  break;
79  }
80  Py_DECREF(func);
81  ++meth;
82  }
83 
84  Py_DECREF(dict);
85  }
86  }
87 
88  Py_DECREF(obj);
89  }
90  return 1;
91 }

118 PyObject* ExtensionContainerPy::getCustomAttributes(const char* attr) const
119 {
120  if (Base::streq(attr, "__dict__")) {
121  PyObject* dict = PyDict_New();
122  PyObject* props = PropertyContainerPy::getCustomAttributes("__dict__");
123  if (props && PyDict_Check(props)) {
124  PyDict_Merge(dict, props, 0);
125  Py_DECREF(props);
126  }
127 
128  ExtensionContainer::ExtensionIterator it =
129  this->getExtensionContainerPtr()->extensionBegin();
130  for (; it != this->getExtensionContainerPtr()->extensionEnd(); ++it) {
131  // The PyTypeObject is shared by all instances of this type and therefore
132  // we have to add new methods only once.
133  PyObject* obj = (*it).second->getExtensionPyObject();
134  PyTypeObject* tp = Py_TYPE(obj);
135  if (tp && tp->tp_dict) {
136  Py_XINCREF(tp->tp_dict);
137  PyDict_Merge(dict, tp->tp_dict, 0);
138  Py_XDECREF(tp->tp_dict);
139  }
140  Py_DECREF(obj);
141  }
142 
143  return dict;
144  }
145  // Search for the method called 'attr' in the extensions. If the search with
146  // Py_FindMethod is successful then a PyCFunction_New instance is returned
147  // with the PyObject pointer of the extension to make sure the method will
148  // be called for the correct instance.
149  PyObject* func = nullptr;
150  ExtensionContainer::ExtensionIterator it = this->getExtensionContainerPtr()->extensionBegin();
151  for (; it != this->getExtensionContainerPtr()->extensionEnd(); ++it) {
152  // The PyTypeObject is shared by all instances of this type and therefore
153  // we have to add new methods only once.
154  PyObject* obj = (*it).second->getExtensionPyObject();
155  PyObject* nameobj = PyUnicode_FromString(attr);
156  func = PyObject_GenericGetAttr(obj, nameobj);
157  Py_DECREF(nameobj);
158  Py_DECREF(obj);
159  if (func && PyCFunction_Check(func)) {
160  PyCFunctionObject* cfunc = reinterpret_cast<PyCFunctionObject*>(func);
161 
162  // OK, that's what we wanted
163  if (cfunc->m_self == obj) {
164  break;
165  }
166  // otherwise cleanup the result again
167  Py_DECREF(func);
168  func = nullptr;
169  }
170  PyErr_Clear(); // clear the error set inside Py_FindMethod
171  }
172 
173  return func;
174 }

181 PyObject* ExtensionContainerPy::hasExtension(PyObject* args)
182 {
183 
184  char* type;
185  PyObject* deriv = Py_True;
186  if (!PyArg_ParseTuple(args, "s|O!", &type, &PyBool_Type, &deriv)) {
187  return nullptr;
188  }
189 
190  // get the extension type asked for
191  bool derived = Base::asBoolean(deriv);
192  Base::Type extension = Base::Type::fromName(type);
193  if (extension.isBad() || !extension.isDerivedFrom(App::Extension::getExtensionClassTypeId())) {
194  std::stringstream str;
195  str << "No extension found of type '" << type << "'" << std::ends;
196  throw Py::TypeError(str.str());
197  }
198 
199  bool val = false;
200  if (getExtensionContainerPtr()->hasExtension(extension, derived)) {
201  val = true;
202  }
203 
204  return PyBool_FromLong(val ? 1 : 0);
205 }
