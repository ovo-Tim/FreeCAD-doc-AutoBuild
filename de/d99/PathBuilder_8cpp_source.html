265 void PathBuilder::pathArc(QPainterPath& path, double rx, double ry, double x_axis_rotation,
266  bool large_arc_flag, bool sweep_flag, double x, double y, double curx,
267  double cury) const
268 {
269  double sin_th, cos_th;
270  double a00, a01, a10, a11;
271  double x0, y0, x1, y1, xc, yc;
272  double d, sfactor, sfactor_sq;
273  double th0, th1, th_arc;
274  int i, n_segs;
275  double dx, dy, dx1, dy1, Pr1, Pr2, Px, Py, check;
276 
277  rx = qAbs(rx);
278  ry = qAbs(ry);
279 
280  sin_th = qSin(x_axis_rotation);
281  cos_th = qCos(x_axis_rotation);
282 
283  dx = (curx - x) / 2.0;
284  dy = (cury - y) / 2.0;
285  dx1 = cos_th * dx + sin_th * dy;
286  dy1 = -sin_th * dx + cos_th * dy;
287  Pr1 = rx * rx;
288  Pr2 = ry * ry;
289  Px = dx1 * dx1;
290  Py = dy1 * dy1;
291  /* Spec : check if radii are large enough */
292  check = Px / Pr1 + Py / Pr2;
293  if (check > 1) {
294  rx = rx * qSqrt(check);
295  ry = ry * qSqrt(check);
296  }
297 
298  a00 = cos_th / rx;
299  a01 = sin_th / rx;
300  a10 = -sin_th / ry;
301  a11 = cos_th / ry;
302  x0 = a00 * curx + a01 * cury;
303  y0 = a10 * curx + a11 * cury;
304  x1 = a00 * x + a01 * y;
305  y1 = a10 * x + a11 * y;
306  /* (x0, y0) is current point in transformed coordinate space.
307  (x1, y1) is new point in transformed coordinate space.
308 
309  The arc fits a unit-radius circle in this space.
310  */
311  d = (x1 - x0) * (x1 - x0) + (y1 - y0) * (y1 - y0);
312  sfactor_sq = 1.0 / d - 0.25;
313  if (sfactor_sq < 0)
314  sfactor_sq = 0;
315 
316  sfactor = qSqrt(sfactor_sq);
317 
318  if (sweep_flag == large_arc_flag)
319  sfactor = -sfactor;
320 
321  xc = 0.5 * (x0 + x1) - sfactor * (y1 - y0);
322  yc = 0.5 * (y0 + y1) + sfactor * (x1 - x0);
323  /* (xc, yc) is center of the circle. */
324 
325  th0 = qAtan2(y0 - yc, x0 - xc);
326  th1 = qAtan2(y1 - yc, x1 - xc);
327 
328  th_arc = th1 - th0;
329  if (th_arc < 0 && sweep_flag)
330  th_arc += 2 * M_PI;
331  else if (th_arc > 0 && !sweep_flag)
332  th_arc -= 2 * M_PI;
333 
334  n_segs = qCeil(qAbs(th_arc / (M_PI * 0.5 + 0.001)));
335 
336  path.moveTo(curx, cury);
337 
338  for (i = 0; i < n_segs; i++) {
339  pathArcSegment(path, xc, yc, th0 + i * th_arc / n_segs, th0 + (i + 1) * th_arc / n_segs, rx,
340  ry, x_axis_rotation);
341  }
342 }

344 void PathBuilder::pathArcSegment(QPainterPath& path, double xc, double yc, double th0, double th1,
345  double rx, double ry, double xAxisRotation) const
346 {
347  double sinTh, cosTh;
348  double a00, a01, a10, a11;
349  double x1, y1, x2, y2, x3, y3;
350  double t;
351  double thHalf;
352 
353  sinTh = qSin(xAxisRotation);
354  cosTh = qCos(xAxisRotation);
355 
356  a00 = cosTh * rx;
357  a01 = -sinTh * ry;
358  a10 = sinTh * rx;
359  a11 = cosTh * ry;
360 
361  thHalf = 0.5 * (th1 - th0);
362  t = (8.0 / 3.0) * qSin(thHalf * 0.5) * qSin(thHalf * 0.5) / qSin(thHalf);
363  x1 = xc + qCos(th0) - t * qSin(th0);
364  y1 = yc + qSin(th0) + t * qCos(th0);
365  x3 = xc + qCos(th1);
366  y3 = yc + qSin(th1);
367  x2 = x3 + t * qSin(th1);
368  y2 = y3 - t * qCos(th1);
369 
370  path.cubicTo(a00 * x1 + a01 * y1, a10 * x1 + a11 * y1, a00 * x2 + a01 * y2, a10 * x2 + a11 * y2,
371  a00 * x3 + a01 * y3, a10 * x3 + a11 * y3);
372 }
