41 class XDMFReader:
42  """
43  Reads XDMF file and provides unified interface for returning
44  cell functions or facet functions.
45  """
46 
47  def __init__(self, xdmffilename):
48  """
49  Sets filename and sets mesh instance to None.
50  """
51  self.xdmffilename = xdmffilename
52  self.mesh = None
53 
54  def resetMesh(self):
55  """
56  Resets mesh instance to None.
57  """
58  self.mesh = None

60  def readMesh(self):
61  """
62  If mesh instance is None, read mesh instance from file denoted
63  by filename property.
64  """
65  # TODO: implement mesh read in for open file
66  if self.mesh is None:
67  xdmffile = fenics.XDMFFile(self.xdmffilename)
68  self.mesh = fenics.Mesh()
69  xdmffile.read(self.mesh)
70  xdmffile.close()

72  def readCellExpression(
73  self, group_value_dict, value_type="scalar", overlap=lambda x: x[0], *args, **kwargs
74  ):
75  """
76  Reads cell expression and returns it.
77  """
78  value_type_dictionary = {
79  "scalar": ScalarCellExpressionFromXDMF,
80  "vector2d": Vector2DCellExpressionFromXDMF,
81  "vector3d": Vector3DCellExpressionFromXDMF,
82  }
83 
84  self.readMesh()
85  xdmffile = fenics.XDMFFile(self.xdmffilename)
86  cf = value_type_dictionary[value_type.lower()](
87  group_value_dict, overlap=overlap, *args, **kwargs
88  )
89  cf.init()
90  for key, value in cf.group_value_dict.items():
91  cf.markers[key] = fenics.MeshFunction("size_t", self.mesh, self.mesh.topology().dim())
92  xdmffile.read(cf.markers[key], key)
93  cf.dx[key] = fenics.Measure("dx", domain=self.mesh, subdomain_data=cf.markers[key])
94  xdmffile.close()
95  return cf

97  def readFacetFunction(self, group_value_dict, *args, **kwargs):
98  """
99  Reads facet function and returns it.
100  """
101  self.readMesh()
102  xdmffile = fenics.XDMFFile(self.xdmffilename)
103  ff = FacetFunctionFromXDMF(group_value_dict, *args, **kwargs)
104  ff.init()
105  for key, value in ff.group_value_dict.items():
106  ff.markers[key] = fenics.MeshFunction(
107  "size_t", self.mesh, self.mesh.topology().dim() - 1
108  )
109  xdmffile.read(ff.markers[key], key)
110  ff.marked[key] = value.get("marked", 1)
111  ff.ds[key] = fenics.Measure("ds", domain=self.mesh, subdomain_data=ff.markers[key])
112  ff.bcs[key] = value
113  xdmffile.close()
114  return ff

143  def eval_cell_backend(self, values, x, cell):
144 
145  values_list = [
146  func(x)
147  for (key, func) in self.group_value_dict.items()
148  if self.check_marked(self.markers[key][cell.index])
149  ]
150  return_value = self.overlap(values_list)
151 
152  if values_list:
153  self.assign_values(values, return_value)
154  else:
155  self.assign_values(values, self.default(x))
