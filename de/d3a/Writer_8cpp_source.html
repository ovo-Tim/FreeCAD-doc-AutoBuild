45 // boost iostream filter to escape ']]>' in text file saved into CDATA section.
46 // It does not check if the character is valid utf8 or not.
47 struct cdata_filter
48 {
49 
50  using char_type = char;
51  using category = boost::iostreams::output_filter_tag;
52 
53  template<typename Device>
54  inline bool put(Device& dev, char ch)
55  {
56  switch (state) {
57  case 0:
58  case 1:
59  if (ch == ']') {
60  ++state;
61  }
62  else {
63  state = 0;
64  }
65  break;
66  case 2:
67  if (ch == '>') {
68  static const char escape[] = "]]><![CDATA[";
69  boost::iostreams::write(dev, escape, sizeof(escape) - 1);
70  }
71  state = 0;
72  break;
73  }
74  return boost::iostreams::put(dev, ch);
75  }
76 
77  int state = 0;
78 };

91 std::ostream& Writer::beginCharStream(CharStreamFormat format)
92 {
93  if (CharStream) {
94  throw Base::RuntimeError("Writer::beginCharStream(): invalid state");
95  }
96  charStreamFormat = format;
97  if (format == CharStreamFormat::Base64Encoded) {
98  CharStream = create_base64_encoder(Stream(), Base::base64DefaultBufferSize);
99  }
100  else {
101  Stream() << "<![CDATA[";
102  CharStream = std::make_unique<boost::iostreams::filtering_ostream>();
103  auto* filteredStream = dynamic_cast<boost::iostreams::filtering_ostream*>(CharStream.get());
104  filteredStream->push(cdata_filter());
105  filteredStream->push(Stream());
106  *filteredStream << std::setprecision(std::numeric_limits<double>::digits10 + 1);
107  }
108  return *CharStream;
109 }

153 {

261 std::string Writer::getUniqueFileName(const char* Name)
262 {
263  // name in use?
264  std::string CleanName = (Name ? Name : "");
265  std::vector<std::string>::const_iterator pos;
266  pos = find(FileNames.begin(), FileNames.end(), CleanName);
267 
268  if (pos == FileNames.end()) {
269  // if not, name is OK
270  return CleanName;
271  }
272 
273  std::vector<std::string> names;
274  names.reserve(FileNames.size());
275  FileInfo fi(CleanName);
276  CleanName = fi.fileNamePure();
277  std::string ext = fi.extension();
278  for (pos = FileNames.begin(); pos != FileNames.end(); ++pos) {
279  fi.setFile(*pos);
280  std::string FileName = fi.fileNamePure();
281  if (fi.extension() == ext) {
282  names.push_back(FileName);
283  }
284  }
285 
286  std::stringstream str;
287  str << Base::Tools::getUniqueName(CleanName, names);
288  if (!ext.empty()) {
289  str << "." << ext;
290  }
291 
292  return str.str();
293 }

405  // processing the files new ones can be added

407  this->FileStream.close();
