127 template<typename SelectModeT>
128 class StateMachine
129 {
130 public:
131  StateMachine()
132  : Mode(static_cast<SelectModeT>(0))
133  {}
134  virtual ~StateMachine()
135  {}
136 
137 protected:
138  void setState(SelectModeT mode)
139  {
140  Mode = mode;
141  onModeChanged();
142  }
143 
144  void ensureState(SelectModeT mode)
145  {
146  if (Mode != mode) {
147  Mode = mode;
148  onModeChanged();
149  }
150  }
151 
159  void ensureStateIfEarlier(SelectModeT mode)
160  {
161  if (Mode != mode) {
162  if (mode < Mode) {
163  Mode = mode;
164  onModeChanged();
165  }
166  }
167  }
168 
169  SelectModeT state() const
170  {
171  return Mode;
172  }
173 
174  bool isState(SelectModeT state) const
175  {
176  return Mode == state;
177  }
178 
179  bool isFirstState() const
180  {
181  return Mode == (static_cast<SelectModeT>(0));
182  }
183 
184  bool isLastState() const
185  {
186  return Mode == SelectModeT::End;
187  }
188 
189  constexpr SelectModeT getFirstState() const
190  {
191  return static_cast<SelectModeT>(0);
192  }
193 
194  SelectModeT getNextMode() const
195  {
196  auto modeint = static_cast<int>(state());
197 
198 
199  if (modeint < maxMode) {
200  auto newmode = static_cast<SelectModeT>(modeint + 1);
201  return newmode;
202  }
203  else {
204  return SelectModeT::End;
205  }
206  }
207 
208  void moveToNextMode()
209  {
210  setState(getNextMode());
211  }
212 
213  void reset()
214  {
215  if (Mode != static_cast<SelectModeT>(0)) {
216  setState(static_cast<SelectModeT>(0));
217  }
218  }
219 
220  virtual bool onModeChanged()
221  {
222  return true;
223  };
224 
225 private:
226  SelectModeT Mode;
227  static const constexpr int maxMode = static_cast<int>(SelectModeT::End);
228 };

659  void generateAutoConstraintsOnElement(const std::vector<AutoConstraint>& autoConstrs,
660  int geoId1,
661  Sketcher::PointPos posId1)
662  {
663  if (!sketchgui->Autoconstraints.getValue()) {
664  return;
665  }
666 
667  if (autoConstrs.size() > 0) {
668  for (auto& ac : autoConstrs) {
669  int geoId2 = ac.GeoId;
670 
671  switch (ac.Type) {
672  case Sketcher::Coincident: {
673  if (posId1 == Sketcher::PointPos::none) {
674  continue;
675  }
676 
677  // find if there is already a matching tangency
678  auto itOfTangentConstraint = AutoConstraints.end();
679  if ((posId1 == Sketcher::PointPos::start
680  || posId1 == Sketcher::PointPos::end)
681  && (ac.PosId == Sketcher::PointPos::start
682  || ac.PosId == Sketcher::PointPos::end)) {
683  itOfTangentConstraint =
684  std::find_if(AutoConstraints.begin(),
685  AutoConstraints.end(),
686  [&](const auto& ace) {
687  return ace->Type == Sketcher::Tangent
688  && ace->First == geoId1
689  && ace->Second == ac.GeoId;
690  });
691  }
692 
693 
694  if (itOfTangentConstraint != AutoConstraints.end()) {
695  // modify tangency to endpoint-to-endpoint
696  (*itOfTangentConstraint)->FirstPos = posId1;
697  (*itOfTangentConstraint)->SecondPos = ac.PosId;
698  }
699  else {
700  auto c = std::make_unique<Sketcher::Constraint>();
701  c->Type = Sketcher::Coincident;
702  c->First = geoId1;
703  c->FirstPos = posId1;
704  c->Second = ac.GeoId;
705  c->SecondPos = ac.PosId;
706  AutoConstraints.push_back(std::move(c));
707  }
708 
709  } break;
710  case Sketcher::PointOnObject: {
711  Sketcher::PointPos posId2 = ac.PosId;
712  if (posId1 == Sketcher::PointPos::none) {
713  // Auto constraining an edge so swap parameters
714  std::swap(geoId1, geoId2);
715  std::swap(posId1, posId2);
716  }
717 
718  auto itOfTangentConstraint = AutoConstraints.end();
719  if (posId1 == Sketcher::PointPos::start
720  || posId1 == Sketcher::PointPos::end) {
721  itOfTangentConstraint =
722  std::find_if(AutoConstraints.begin(),
723  AutoConstraints.end(),
724  [&](const auto& ace) {
725  return ace->Type == Sketcher::Tangent
726  && ace->First == geoId1
727  && ace->Second == ac.GeoId;
728  });
729  }
730 
731  // if tangency, convert to point-to-edge tangency
732  if (itOfTangentConstraint != AutoConstraints.end()) {
733  if ((*itOfTangentConstraint)->First != geoId1) {
734  std::swap((*itOfTangentConstraint)->Second,
735  (*itOfTangentConstraint)->First);
736  }
737 
738  (*itOfTangentConstraint)->FirstPos = posId1;
739  }
740  else {
741  auto c = std::make_unique<Sketcher::Constraint>();
742  c->Type = Sketcher::PointOnObject;
743  c->First = geoId1;
744  c->FirstPos = posId1;
745  c->Second = geoId2;
746  AutoConstraints.push_back(std::move(c));
747  }
748  } break;
749  case Sketcher::Symmetric: {
750  auto c = std::make_unique<Sketcher::Constraint>();
751  c->Type = Sketcher::Symmetric;
752  c->First = geoId2;
753  c->FirstPos = Sketcher::PointPos::start;
754  c->Second = geoId2;
755  c->SecondPos = Sketcher::PointPos::end;
756  c->Third = geoId1;
757  c->ThirdPos = posId1;
758  AutoConstraints.push_back(std::move(c));
759  } break;
760  // In special case of Horizontal/Vertical constraint, geoId2 is normally
761  // unused and should be 'Constraint::GeoUndef' However it can be used as a
762  // way to require the function to apply these constraints on another
763  // geometry In this case the caller as to set geoId2, then it will be used
764  // as target instead of geoId2
765  case Sketcher::Horizontal: {
766  auto c = std::make_unique<Sketcher::Constraint>();
767  c->Type = Sketcher::Horizontal;
768  c->First = (geoId2 != Sketcher::GeoEnum::GeoUndef ? geoId2 : geoId1);
769  AutoConstraints.push_back(std::move(c));
770  } break;
771  case Sketcher::Vertical: {
772  auto c = std::make_unique<Sketcher::Constraint>();
773  c->Type = Sketcher::Vertical;
774  c->First = (geoId2 != Sketcher::GeoEnum::GeoUndef ? geoId2 : geoId1);
775  AutoConstraints.push_back(std::move(c));
776  } break;
777  case Sketcher::Tangent: {
778  Sketcher::SketchObject* Obj =
779  sketchgui->getObject<Sketcher::SketchObject>();
780 
781  const Part::Geometry* geom1 = Obj->getGeometry(geoId1);
782  const Part::Geometry* geom2 = Obj->getGeometry(ac.GeoId);
783 
784  // ellipse tangency support using construction elements (lines)
785  if (geom1 && geom2
786  && (geom1->is<Part::GeomEllipse>() || geom2->is<Part::GeomEllipse>())) {
787  if (!geom1->is<Part::GeomEllipse>()) {
788  std::swap(geoId1, geoId2);
789  }
790 
791  // geoId1 is the ellipse
792  geom1 = Obj->getGeometry(geoId1);
793  geom2 = Obj->getGeometry(geoId2);
794 
795  if (geom2->is<Part::GeomEllipse>()
796  || geom2->is<Part::GeomArcOfEllipse>()
797  || geom2->is<Part::GeomCircle>()
798  || geom2->is<Part::GeomArcOfCircle>()) {
799  // in all these cases an intermediate element is needed
800  // makeTangentToEllipseviaNewPoint(
801  // Obj,
802  // static_cast<const Part::GeomEllipse *>(geom1),
803  // geom2, geoId1, geoId2);
804  // NOTE: Temporarily deactivated
805  return;
806  }
807  }
808 
809  // arc of ellipse tangency support using external elements
810  if (geom1 && geom2
811  && (geom1->is<Part::GeomArcOfEllipse>()
812  || geom2->is<Part::GeomArcOfEllipse>())) {
813 
814  if (!geom1->is<Part::GeomArcOfEllipse>()) {
815  std::swap(geoId1, geoId2);
816  }
817 
818  // geoId1 is the arc of ellipse
819  geom1 = Obj->getGeometry(geoId1);
820  geom2 = Obj->getGeometry(geoId2);
821 
822  if (geom2->is<Part::GeomArcOfEllipse>() || geom2->is<Part::GeomCircle>()
823  || geom2->is<Part::GeomArcOfCircle>()) {
824  // in all these cases an intermediate element is needed
825  // makeTangentToArcOfEllipseviaNewPoint(
826  // Obj,
827  // static_cast<const Part::GeomArcOfEllipse*>(geom1), geom2,
828  // geoId1, geoId2);
829  // NOTE: Temporarily deactivated
830  return;
831  }
832  }
833 
834  auto resultcoincident =
835  std::find_if(AutoConstraints.begin(),
836  AutoConstraints.end(),
837  [&](const auto& ace) {
838  return ace->Type == Sketcher::Coincident
839  && ace->First == geoId1 && ace->Second == ac.GeoId;
840  });
841 
842  auto resultpointonobject = std::find_if(
843  AutoConstraints.begin(),
844  AutoConstraints.end(),
845  [&](const auto& ace) {
846  return ace->Type == Sketcher::PointOnObject
847  && ((ace->First == geoId1 && ace->Second == ac.GeoId)
848  || (ace->First == ac.GeoId && ace->Second == geoId1));
849  });
850 
851  if (resultcoincident != AutoConstraints.end()
852  && ((*resultcoincident)->FirstPos == Sketcher::PointPos::start
853  || (*resultcoincident)->FirstPos == Sketcher::PointPos::end)
854  && ((*resultcoincident)->SecondPos == Sketcher::PointPos::start
855  || (*resultcoincident)->SecondPos == Sketcher::PointPos::end)) {
856  // endpoint-to-endpoint tangency
857  (*resultcoincident)->Type = Sketcher::Tangent;
858  }
859  else if (resultpointonobject != AutoConstraints.end()
860  && ((*resultpointonobject)->FirstPos == Sketcher::PointPos::start
861  || (*resultpointonobject)->FirstPos
862  == Sketcher::PointPos::end)) {
863  // endpoint-to-edge tangency
864  (*resultpointonobject)->Type = Sketcher::Tangent;
865  }
866  else if (resultcoincident != AutoConstraints.end()
867  && (*resultcoincident)->FirstPos == Sketcher::PointPos::mid
868  && (*resultcoincident)->SecondPos == Sketcher::PointPos::mid
869  && geom1 && geom2
870  && (geom1->is<Part::GeomCircle>()
871  || geom1->is<Part::GeomArcOfCircle>())
872  && (geom2->is<Part::GeomCircle>()
873  || geom2->is<Part::GeomArcOfCircle>())) {
874  // equality
875  auto c = std::make_unique<Sketcher::Constraint>();
876  c->Type = Sketcher::Equal;
877  c->First = geoId1;
878  c->Second = ac.GeoId;
879  AutoConstraints.push_back(std::move(c));
880  }
881  else { // regular edge to edge tangency
882  auto c = std::make_unique<Sketcher::Constraint>();
883  c->Type = Sketcher::Tangent;
884  c->First = geoId1;
885  c->Second = ac.GeoId;
886  AutoConstraints.push_back(std::move(c));
887  }
888  } break;
889  default:
890  break;
891  }
892  }
893  }

944  void removeRedundantAutoConstraints()
945  {
946 
947  if (AutoConstraints.empty()) {
948  return;
949  }
950 
951  auto sketchobject = getSketchObject();
952 
953  auto autoConstraints = toPointerVector(AutoConstraints);
954 
955  // Allows a diagnose with the new autoconstraints as if they were part of the sketchobject,
956  // but WITHOUT adding them to the sketchobject..
957  sketchobject->diagnoseAdditionalConstraints(autoConstraints);
958 
959  if (sketchobject->getLastHasRedundancies()) {
960  Base::Console().Warning(
961  QT_TRANSLATE_NOOP("Notifications",
962  "Autoconstraints cause redundancy. Removing them") "\n");
963 
964  auto lastsketchconstraintindex = sketchobject->Constraints.getSize() - 1;
965 
966  auto redundants = sketchobject->getLastRedundant(); // redundants is always sorted
967 
968  for (int index = redundants.size() - 1; index >= 0; index--) {
969  int redundantconstraintindex = redundants[index] - 1;
970  if (redundantconstraintindex > lastsketchconstraintindex) {

976  // 1) Geometry (and constraints of the geometry in case of a multicurve shape)

978  // constraints are added

983  "or additional constraints were added. Please report!") "\n");
984  }
985  }
986

988  // redundants anymore
989  }

992  // just a safeguard.
