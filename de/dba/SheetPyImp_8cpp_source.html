96 PyObject* SheetPy::get(PyObject* args)
97 {
98  const char* address;
99  const char* address2 = nullptr;
100 
101  if (!PyArg_ParseTuple(args, "s|s:get", &address, &address2)) {
102  return nullptr;
103  }
104 
105  PY_TRY
106  {
107  if (address2) {
108  auto a1 = getSheetPtr()->getAddressFromAlias(address);
109  if (a1.empty()) {
110  a1 = address;
111  }
112  auto a2 = getSheetPtr()->getAddressFromAlias(address2);
113  if (a2.empty()) {
114  a2 = address2;
115  }
116  Range range(a1.c_str(), a2.c_str());
117  Py::Tuple tuple(range.size());
118  int i = 0;
119  do {
120  App::Property* prop = getSheetPtr()->getPropertyByName(range.address().c_str());
121  if (!prop) {
122  PyErr_Format(PyExc_ValueError,
123  "Invalid address '%s' in range %s:%s",
124  range.address().c_str(),
125  address,
126  address2);
127  return nullptr;
128  }
129  tuple.setItem(i++, Py::Object(prop->getPyObject(), true));
130  } while (range.next());
131  return Py::new_reference_to(tuple);
132  }
133  }
134  PY_CATCH;
135 
136  App::Property* prop = this->getSheetPtr()->getPropertyByName(address);
137 
138  if (!prop) {
139  PyErr_Format(PyExc_ValueError, "Invalid cell address or property: %s", address);
140  return nullptr;
141  }
142  return prop->getPyObject();
143 }

350 PyObject* SheetPy::setStyle(PyObject* args)
351 {
352  const char* cell;
353  PyObject* value;
354  std::set<std::string> style;
355  const char* options = "replace";
356 
357  if (!PyArg_ParseTuple(args, "sO|s:setStyle", &cell, &value, &options)) {
358  return nullptr;
359  }
360 
361  if (PySet_Check(value)) {
362  PyObject* copy = PySet_New(value);
363 
364  while (PySet_Size(copy) > 0) {
365  PyObject* item = PySet_Pop(copy);
366 
367  // check on the key:
368  if (PyUnicode_Check(item)) {
369  style.insert(PyUnicode_AsUTF8(item));
370  }
371  else {
372  std::string error = std::string("type of the set need to be a string, not ")
373  + item->ob_type->tp_name;
374  PyErr_SetString(PyExc_TypeError, error.c_str());
375  Py_DECREF(copy);
376  return nullptr;
377  }
378  }
379  Py_DECREF(copy);
380  }
381  else if (PyUnicode_Check(value)) {
382  using namespace boost;
383 
384  escaped_list_separator<char> e('\0', '|', '\0');
385  std::string line = PyUnicode_AsUTF8(value);
386  tokenizer<escaped_list_separator<char>> tok(line, e);
387 
388  for (tokenizer<escaped_list_separator<char>>::iterator i = tok.begin(); i != tok.end();
389  ++i) {
390  style.insert(*i);
391  }
392  }
393  else {
394  std::string error =
395  std::string("style must be either set or string, not ") + value->ob_type->tp_name;
396  PyErr_SetString(PyExc_TypeError, error.c_str());
397  return nullptr;
398  }
399 
400  if (strcmp(options, "replace") == 0) {
401  Range rangeIter(cell);
402  do {
403  getSheetPtr()->setStyle(*rangeIter, style);
404  } while (rangeIter.next());
405  }
406  else if (strcmp(options, "add") == 0) {
407  Range rangeIter(cell);
408 
409  do {
410  std::set<std::string> oldStyle;
411  const Cell* cell = getSheetPtr()->getCell(*rangeIter);
412 
413  // Get old styles first
414  if (cell) {
415  cell->getStyle(oldStyle);
416  }
417 
418  for (const auto& it : oldStyle) {
419  style.insert(it);
420  }
421 
422  // Set new style
423  getSheetPtr()->setStyle(*rangeIter, style);
424  } while (rangeIter.next());
425  }
426  else if (strcmp(options, "remove") == 0) {
427  Range rangeIter(cell);
428 
429  do {
430  std::set<std::string> oldStyle;
431  const Cell* cell = getSheetPtr()->getCell(*rangeIter);
432 
433  // Get old styles first
434  if (cell) {
435  cell->getStyle(oldStyle);
436  }
437 
438  for (const auto& it : style) {
439  oldStyle.erase(it);
440  }
441 
442  // Set new style
443  getSheetPtr()->setStyle(*rangeIter, oldStyle);
444  } while (rangeIter.next());
445  }
446  else if (strcmp(options, "invert") == 0) {
447  Range rangeIter(cell);
448 
449  do {
450  std::set<std::string> oldStyle;
451  std::set<std::string> newStyle;
452  const Cell* cell = getSheetPtr()->getCell(*rangeIter);
453 
454  // Get old styles first
455  if (cell) {
456  cell->getStyle(oldStyle);
457  newStyle = oldStyle;
458  }
459 
460  for (const auto& i : style) {
461  if (oldStyle.find(i) == oldStyle.end()) {
462  // Not found in oldstyle; add it to newStyle
463  newStyle.insert(i);
464  }
465  else {
466  // Found in oldStyle, remove it from newStyle
467  newStyle.erase(i);
468  }
469  }
470 
471  // Set new style
472  getSheetPtr()->setStyle(*rangeIter, newStyle);
473  } while (rangeIter.next());
474  }
475  else {
476  PyErr_SetString(
477  PyExc_ValueError,
478  "Optional parameter must be either 'replace', 'add', 'remove', or 'invert'");
479  return nullptr;
480  }
481 
482  Py_Return;
483 }

645 PyObject* SheetPy::setAlignment(PyObject* args)
646 {
647  const char* cell;
648  PyObject* value;
649  int alignment = 0;
650  const char* options = "replace";
651 
652  if (!PyArg_ParseTuple(args, "sO|s:setAlignment", &cell, &value, &options)) {
653  return nullptr;
654  }
655 
656  if (PySet_Check(value)) {
657  // Argument is a set of strings
658  PyObject* copy = PySet_New(value);
659  int n = PySet_Size(copy);
660 
661  while (n-- > 0) {
662  PyObject* item = PySet_Pop(copy);
663 
664  if (PyUnicode_Check(item)) {
665  alignment = Cell::decodeAlignment(PyUnicode_AsUTF8(item), alignment);
666  }
667  else {
668  std::string error = std::string("type of the key need to be a string, not")
669  + item->ob_type->tp_name;
670  PyErr_SetString(PyExc_TypeError, error.c_str());
671  Py_DECREF(copy);
672  return nullptr;
673  }
674  }
675 
676  Py_DECREF(copy);
677  }
678  else if (PyUnicode_Check(value)) {
679  // Argument is a string, combination of alignments, separated by the pipe character
680  using namespace boost;
681 
682  escaped_list_separator<char> e('\0', '|', '\0');
683  std::string line = PyUnicode_AsUTF8(value);
684  tokenizer<escaped_list_separator<char>> tok(line, e);
685 
686  for (tokenizer<escaped_list_separator<char>>::iterator i = tok.begin(); i != tok.end();
687  ++i) {
688  if (!i->empty()) {
689  alignment = Cell::decodeAlignment(*i, alignment);
690  }
691  }
692  }
693  else {
694  std::string error =
695  std::string("style must be either set or string, not ") + value->ob_type->tp_name;
696  PyErr_SetString(PyExc_TypeError, error.c_str());
697  return nullptr;
698  }
699 
700  // Set alignment depending on 'options' variable
701  if (strcmp(options, "replace") == 0) {
702  Range rangeIter(cell);
703 
704  do {
705  getSheetPtr()->setAlignment(*rangeIter, alignment);
706  } while (rangeIter.next());
707  }
708  else if (strcmp(options, "keep") == 0) {
709  Range rangeIter(cell);
710 
711  do {
712  int oldAlignment = 0;
713  const Cell* cell = getSheetPtr()->getCell(*rangeIter);
714 
715  if (cell) {
716  cell->getAlignment(oldAlignment);
717  }
718 
719  if (alignment & Cell::ALIGNMENT_VERTICAL) {
720  oldAlignment &= ~Cell::ALIGNMENT_VERTICAL;
721  }
722  if (alignment & Cell::ALIGNMENT_HORIZONTAL) {
723  oldAlignment &= ~Cell::ALIGNMENT_HORIZONTAL;
724  }
725 
726  getSheetPtr()->setAlignment(*rangeIter, alignment | oldAlignment);
727  } while (rangeIter.next());
728  }
729  else {
730  PyErr_SetString(PyExc_ValueError, "Optional parameter must be either 'replace' or 'keep'");
731  return nullptr;
732  }
733  Py_Return;
734 }
