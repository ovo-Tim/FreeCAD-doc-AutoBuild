62 QVariant PropertyMeshKernelItem::value(const App::Property*) const
63 {
64  int ctP = 0;
65  int ctE = 0;
66  int ctF = 0;
67 
68  const std::vector<App::Property*>& props = getPropertyData();
69  for (auto prop : props) {
70  Mesh::PropertyMeshKernel* pPropMesh = static_cast<Mesh::PropertyMeshKernel*>(prop);
71  const MeshKernel& rMesh = pPropMesh->getValue().getKernel();
72  ctP += (int)rMesh.CountPoints();
73  ctE += (int)rMesh.CountEdges();
74  ctF += (int)rMesh.CountFacets();
75  }
76 
77  QString str = QObject::tr("[Points: %1, Edges: %2, Faces: %3]").arg(ctP).arg(ctE).arg(ctF);
78  return {str};
79 }
