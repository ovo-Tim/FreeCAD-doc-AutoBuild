91 void QGIViewClip::drawClip()
92 {
93  auto viewClip( dynamic_cast<TechDraw::DrawViewClip *>(getViewObject()) );
94 
95  if (!viewClip)
96  return;
97 
98  prepareGeometryChange();
99  double h = viewClip->Height.getValue();
100  double w = viewClip->Width.getValue();
101  QRectF r = QRectF(-Rez::guiX(w)/2.0, -Rez::guiX(h)/2.0, Rez::guiX(w), Rez::guiX(h));
102  m_frame->setRect(r); // (-50, -50) -> (50, 50)
103  m_frame->setPos(0., 0.);
104  if (viewClip->ShowFrame.getValue()) {
105  m_frame->show();
106  } else {
107  m_frame->hide();
108  }
109 
110  //probably a slicker way to do this?
111  QPointF midFrame = m_frame->boundingRect().center();
112  QPointF midMapped = mapFromItem(m_frame, midFrame);
113  QPointF clipOrigin = mapToItem(m_cliparea, midMapped);
114 
115  m_cliparea->setRect(r.adjusted(-1, -1, 1,1));
116 
117  std::vector<std::string> childNames = viewClip->getChildViewNames();
118  //for all child Views in Clip, add the graphics representation of the View to the Clip group
119  for(std::vector<std::string>::iterator it = childNames.begin(); it != childNames.end(); it++) {
120  QGIView* qgiv = getQGIVByName((*it));
121  if (qgiv) {
122  //TODO: why is qgiv never already in a group?
123  if (qgiv->group() != m_cliparea) {
124  qgiv->hide();
125  scene()->removeItem(qgiv);
126  m_cliparea->addToGroup(qgiv);
127  qgiv->isInnerView(true);
128  double x = Rez::guiX(qgiv->getViewObject()->X.getValue());
129  double y = Rez::guiX(qgiv->getViewObject()->Y.getValue());
130  qgiv->setPosition(clipOrigin.x() + x, clipOrigin.y() + y);
131 // if (viewClip->ShowLabels.getValue()) {
132 // qgiv->toggleBorder(true);
133 // } else {
134 // qgiv->toggleBorder(false);
135 // }
136  qgiv->show();
137  }
138  } else {
139  Base::Console().Warning("Logic error? - drawClip() - qgiv for %s not found\n", (*it).c_str()); //gview for feature !exist
140  }
141  }
142 
143  //for all graphic views in qgigroup, remove from qgigroup the ones that aren't in ViewClip
144  QList<QGraphicsItem*> qgItems = m_cliparea->childItems();
145  QList<QGraphicsItem*>::iterator it = qgItems.begin();
146  for (; it != qgItems.end(); it++) {
147  QGIView* qv = dynamic_cast<QGIView*>((*it));
148  if (qv) {
149  std::string qvName = std::string(qv->getViewName());
150  if (std::find(childNames.begin(), childNames.end(), qvName) == childNames.end()) {
151  m_cliparea->removeFromGroup(qv);
152  removeFromGroup(qv);
153  qv->isInnerView(false);
154 // qv->toggleBorder(true);
155  }
156  }
157  }
158 }
