139 void Metadata::loadFromInputSource(const InputSource& source)
140 {
141  // Any exception thrown by the XML code propagates out and prevents object creation
142  XMLPlatformUtils::Initialize();
143 
144  _parser = std::make_shared<XercesDOMParser>();
145  _parser->setValidationScheme(XercesDOMParser::Val_Never);
146  _parser->setDoNamespaces(true);
147 
148  auto errHandler = std::make_unique<MetadataInternal::XMLErrorHandler>();
149  _parser->setErrorHandler(errHandler.get());
150 
151  _parser->parse(source);
152 
153  auto doc = _parser->getDocument();
154  _dom = doc->getDocumentElement();
155 
156  auto rootTagName = StrXUTF8(_dom->getTagName()).str;
157  if (rootTagName != "package") {
158  throw Base::XMLBaseException(
159  "Malformed package.xml document: Root <package> group not found");
160  }
161  auto formatVersion = XMLString::parseInt(_dom->getAttribute(XUTF8Str("format").unicodeForm()));
162  switch (formatVersion) {
163  case 1:
164  parseVersion1(_dom);
165  break;
166  default:
167  throw Base::XMLBaseException(
168  "package.xml format version is not supported by this version of FreeCAD");
169  }
170 }

608 void Metadata::write(const fs::path& file) const
609 {
610  DOMImplementation* impl =
611  DOMImplementationRegistry::getDOMImplementation(XUTF8Str("Core LS").unicodeForm());
612 
613  DOMDocument* doc = impl->createDocument(nullptr, XUTF8Str("package").unicodeForm(), nullptr);
614  DOMElement* root = doc->getDocumentElement();
615  root->setAttribute(XUTF8Str("format").unicodeForm(), XUTF8Str("1").unicodeForm());
616  root->setAttribute(XUTF8Str("xmlns").unicodeForm(),
617  XUTF8Str("https://wiki.freecad.org/Package_Metadata").unicodeForm());
618 
619  appendToElement(root);
620 
621  DOMLSSerializer* theSerializer = ((DOMImplementationLS*)impl)->createLSSerializer();
622  DOMConfiguration* config = theSerializer->getDomConfig();
623  if (config->canSetParameter(XMLUni::fgDOMWRTFormatPrettyPrint, true)) {
624  config->setParameter(XMLUni::fgDOMWRTFormatPrettyPrint, true);
625  }
626 
627  // set feature if the serializer supports the feature/mode
628  if (config->canSetParameter(XMLUni::fgDOMWRTSplitCdataSections, true)) {
629  config->setParameter(XMLUni::fgDOMWRTSplitCdataSections, true);
630  }
631 
632  if (config->canSetParameter(XMLUni::fgDOMWRTDiscardDefaultContent, true)) {
633  config->setParameter(XMLUni::fgDOMWRTDiscardDefaultContent, true);
634  }
635 
636  try {
637  XMLFormatTarget* myFormTarget = new LocalFileFormatTarget(file.string().c_str());
638  DOMLSOutput* theOutput = ((DOMImplementationLS*)impl)->createLSOutput();
639 
640  theOutput->setByteStream(myFormTarget);
641  theSerializer->write(doc, theOutput);
642 
643  theOutput->release();
644  theSerializer->release();
645  delete myFormTarget;
646  }
647  catch (const XMLException& toCatch) {
648  char* message = XMLString::transcode(toCatch.getMessage());
649  std::string what = message;
650  XMLString::release(&message);
651  throw Base::XMLBaseException(what);
652  }
653  catch (const DOMException& toCatch) {
654  char* message = XMLString::transcode(toCatch.getMessage());
655  std::string what = message;
656  XMLString::release(&message);
657  throw Base::XMLBaseException(what);
658  }
659 
660  doc->release();
661 }

663 bool Metadata::satisfies(const Meta::Dependency& dep)
664 {
665  if (dep.package != _name) {
666  return false;
667  }
668  // The "condition" attribute allows an expression to enable or disable this dependency check: it
669  // must contain a valid FreeCAD Expression. If it evaluates to false, this dependency is
670  // bypassed (e.g. this function returns false).
671  if (!dep.condition.empty()) {
672  auto injectedString = dep.condition;
673  std::map<std::string, std::string> replacements;
674  std::map<std::string, std::string>& config = App::Application::Config();
675  replacements.insert(std::make_pair("$BuildVersionMajor", config["BuildVersionMajor"]));
676  replacements.insert(std::make_pair("$BuildVersionMinor", config["BuildVersionMinor"]));
677  replacements.insert(std::make_pair("$BuildVersionMinor", config["BuildVersionPoint"]));
678  replacements.insert(std::make_pair("$BuildRevision", config["BuildRevision"]));
679  for (const auto& replacement : replacements) {
680  auto pos = injectedString.find(replacement.first);
681  while (pos != std::string::npos) {
682  injectedString.replace(pos, replacement.first.length(), replacement.second);
683  pos = injectedString.find(replacement.first);
684  }
685  }
686  auto parsedExpression = App::Expression::parse(nullptr, dep.condition);
687  auto result = parsedExpression->eval();
688  if (!boost::any_cast<bool>(result->getValueAsAny())) {
689  return false;
690  }
691  }
692 
693  if (!dep.version_eq.empty()) {
694  return _version == Meta::Version(dep.version_eq);
695  }
696  // Any of the others might be specified in pairs, so only return the "false" case
697 
698  if (!dep.version_lt.empty()) {
699  if (!(_version < Meta::Version(dep.version_lt))) {
700  return false;
701  }
702  }
703  if (!dep.version_lte.empty()) {
704  if (!(_version <= Meta::Version(dep.version_lt))) {
705  return false;
706  }
707  }
708  if (!dep.version_gt.empty()) {
709  if (!(_version > Meta::Version(dep.version_lt))) {
710  return false;
711  }
712  }
713  if (!dep.version_gte.empty()) {
714  if (!(_version >= Meta::Version(dep.version_lt))) {
715  return false;
716  }
717  }
718  return true;
719 }

721 bool Metadata::supportsCurrentFreeCAD() const
722 {
723  static auto fcVersion = Meta::Version();
724  if (fcVersion == Meta::Version()) {
725  std::map<std::string, std::string>& config = App::Application::Config();
726  std::stringstream ss;
727  ss << config["BuildVersionMajor"] << "." << config["BuildVersionMinor"] << "."
728  << config["BuildVersionPoint"] << "."
729  << (config["BuildRevision"].empty() ? "0" : config["BuildRevision"]);
730  fcVersion = Meta::Version(ss.str());
731  }
732 
733  if (_freecadmin != Meta::Version() && _freecadmin > fcVersion) {
734  return false;
735  }
736  if (_freecadmax != Meta::Version() && _freecadmax < fcVersion) {
737  return false;
738  }
739  return true;
740 }

742 void Metadata::appendToElement(DOMElement* root) const
743 {
744  appendSimpleXMLNode(root, "name", _name);
745  appendSimpleXMLNode(root, "type", _type);
746  appendSimpleXMLNode(root, "description", _description);
747  if (_version != Meta::Version()) {
748  // Only append version if it's not 0.0.0
749  appendSimpleXMLNode(root, "version", _version.str());
750  }
751 
752  if (!_date.empty()) {
753  appendSimpleXMLNode(root, "date", _date);
754  }
755 
756  for (const auto& maintainer : _maintainer) {
