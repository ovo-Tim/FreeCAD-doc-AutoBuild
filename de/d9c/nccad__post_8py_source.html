30 TOOLTIP = """
31 This is a postprocessor file for the Path workbench. It is used to take
32 a pseudo-G-code fragment output by a Path object and output real G-code
33 suitable for the Max Computer GmbH nccad9 Computer Numeric Control.
34 
35 Supported features:
36 
37 - 3-axis milling
38 - manual tool change with tool number as comment
39 - spindle speed as comment
40 
41 !!! gCode files must use the suffix .knc !!!
42 
43 import nccad_post
44 nccad_post.export([object], "/path/to/file.knc", "")
45 """

47 MACHINE_NAME = """Max Computer GmbH nccad9 MCS/KOSY"""

49 # gCode for changing tools
50 # M01 <String> ; Displays <String> and waits for user interaction
51 TOOL_CHANGE = """G77 ; Move to release position
52 M10 O6.0 ; Stop spindle
53 M01 Insert tool TOOL
54 G76 ; Move to reference point to ensure correct coordinates after tool change
55 M10 O6.1 ; Start spindle"""
56 
57 
58 # gCode finishing the program
59 POSTAMBLE = """G77 ; Move to release position
60 M10 O6.0 ; Stop spindle"""
61 
62 
63 # gCode header with information about CAD-software, post-processor
64 # and date/time
65 if FreeCAD.ActiveDocument:
66  cam_file = FreeCAD.ActiveDocument.FileName
67 else:
68  cam_file = "<None>"

71 ;Post Processor: {}
72 ;CAM file: {}
73 ;Output Time: {}
74 """.format(
75  __name__, cam_file, str(datetime.datetime.now())

79 def export(objectslist, filename, argstring):
80  """Export the list of objects into a filename.
81 
82  Parameters
83  ----------
84  objectslists: list
85  List of objects.
86 
87  filename: str
88  Name of the output file ending in `'.knc'`.
89  """
90  gcode = HEADER
91 
92  for obj in objectslist:
93  for command in PathUtils.getPathWithPlacement(obj).Commands:
94  # Manipulate tool change commands
95  if "M6" == command.Name:
96  gcode += TOOL_CHANGE.replace("TOOL", str(int(command.Parameters["T"])))
97  elif "M3" == command.Name:
98  # Convert spindle speed (rpm) command to comment
99  gcode += (
100  "M01 Set spindle speed to "
101  + str(int(command.Parameters["S"]))
102  + " rounds per minute"
103  )
104  else:
105  # Add other commands
106  gcode += command.Name
107 
108  # Loop through command parameters
109  for parameter, value in command.Parameters.items():
110  # Multiply F parameter value by 10,
111  # FreeCAD = mm/s, nccad = 1/10 mm/s
112  if "F" == parameter:
113  value *= 10
114  # Add command parameters and values and round float
115  # as nccad9 does not support exponents
116  gcode += " " + parameter + str(round(value, 5))
117 
118  gcode += "\n"
119 
120  gcode += POSTAMBLE + "\n"
121 
122  # Open editor window
123  if FreeCAD.GuiUp:
124  dia = PostUtils.GCodeEditorDialog()
125  dia.editor.setText(gcode)
126  result = dia.exec_()
127  if result:
128  gcode = dia.editor.toPlainText()
129 
130  # Save to file
131  if filename != "-":
132  gfile = open(filename, "w")
133  gfile.write(gcode)
134  gfile.close()
135 
136  return filename
