113 void ViewProviderLoft::highlightSection(bool on)
114 {
115  PartDesign::Loft* pcLoft = getObject<PartDesign::Loft>();
116  auto sections = pcLoft->Sections.getSubListValues();
117  for (auto& it : sections) {
118  // only take the entire shape when we have a sketch selected, but
119  // not a point of the sketch
120  auto subName = it.second.empty() ? "" : it.second.front();
121  if (it.first->isDerivedFrom(Part::Part2DObject::getClassTypeId()) && subName.compare(0, 6, "Vertex") != 0) {
122  it.second.clear();
123  }
124  highlightReferences(dynamic_cast<Part::Feature*>(it.first), it.second, on);
125  }
126 }

146 void ViewProviderLoft::highlightReferences(Part::Feature* base, const std::vector<std::string>& elements, bool on)
147 {
148  if (!base)
149  return;
150 
151  PartGui::ViewProviderPart* svp = dynamic_cast<PartGui::ViewProviderPart*>(
152  Gui::Application::Instance->getViewProvider(base));
153  if (!svp)
154  return;
155 
156  std::vector<App::Color>& edgeColors = originalLineColors[base->getID()];
157 
158  if (on) {
159  edgeColors = svp->LineColorArray.getValues();
160  std::vector<App::Color> colors = edgeColors;
161 
162  PartGui::ReferenceHighlighter highlighter(base->Shape.getValue(), svp->LineColor.getValue());
163  highlighter.getEdgeColors(elements, colors);
164  svp->LineColorArray.setValues(colors);
165  }
166  else {
167  svp->LineColorArray.setValues({svp->LineColor.getValue()});
168  edgeColors.clear();
169  }
170 }
