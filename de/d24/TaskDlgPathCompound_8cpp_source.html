48 TaskWidgetPathCompound::TaskWidgetPathCompound(ViewProviderPathCompound* CompoundView,
49  QWidget* parent)

58  this->groupLayout()->addWidget(proxy);

60  Path::FeatureCompound* pcCompound = CompoundView->getObject<Path::FeatureCompound>();
61  const std::vector<App::DocumentObject*>& Paths = pcCompound->Group.getValues();
