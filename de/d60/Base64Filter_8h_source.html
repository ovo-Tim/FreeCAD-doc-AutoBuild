100  template<typename Device>
101  std::streamsize write(Device& dev, const char_type* str, std::streamsize n)
102  {
103  std::streamsize res = n;
104 
105  if (pending_size > 0) {
106  while (n && pending_size < 3) {
107  pending[pending_size] = *str++;
108  ++pending_size;
109  --n;
110  }
111  if (pending_size != 3) {
112  return res;
113  }
114 
115  base64_encode(buffer, pending.data(), 3);
116  }
117  pending_size = n % 3;
118  n = n / 3 * 3;
119  base64_encode(buffer, str, n);
120  str += n;
121  for (unsigned i = 0; i < pending_size; ++i) {
122  pending[i] = str[i];
123  }
124 
125  const char* buf = buffer.c_str();
126  const char* end = buf + buffer.size();
127  if (line_size && buffer.size() >= line_size - pos) {
128  bio::write(dev, buf, line_size - pos);
129  bio::put(dev, '\n');
130  buf += line_size - pos;
131  pos = 0;
132  for (; end - buf >= (int)line_size; buf += line_size) {
133  bio::write(dev, buf, line_size);
134  bio::put(dev, '\n');
135  }
136  }
137  pos += end - buf;
138  bio::write(dev, buf, end - buf);
139  buffer.clear();
140  return n;
141  }

144  std::size_t pos = 0;
145  std::size_t pending_size = 0;
146  std::array<unsigned char, 3> pending {};

179  template<typename Device>
180  std::streamsize read(Device& dev, char_type* str, std::streamsize n)
181  {
182  static auto table = base64_decode_table();
183 
184  if (!n) {
185  return 0;
186  }
187 
188  std::streamsize count = 0;
189 
190  for (;;) {
191  while (pending_out < out_count) {
192  *str++ = char_array_3[pending_out++];
193  ++count;
194  if (--n == 0) {
195  return count;
196  }
197  }
198 
199  if (eof) {
200  return count ? count : -1;
201  }
202 
203  for (;;) {
204  int newChar = bio::get(dev);
205  if (newChar < 0) {
206  eof = true;
207  if (pending_in <= 1) {
208  if (pending_in == 1 && errHandling == Base64ErrorHandling::throws) {
209  throw BOOST_IOSTREAMS_FAILURE("Unexpected ending of base64 string");
210  }
211  return count ? count : -1;
212  }
213  out_count = pending_in - 1;
214  pending_in = 4;
215  }
216  else {
217  signed char decodedChar = table[newChar];
218  if (decodedChar < 0) {
219  if (decodedChar == -2 || errHandling == Base64ErrorHandling::silent) {
220  continue;
221  }
222  throw BOOST_IOSTREAMS_FAILURE("Invalid character in base64 string");
223  }
224  char_array_4[pending_in++] = (char)decodedChar;
225  }
226  if (pending_in == 4) {
227  pending_out = pending_in = 0;
228  char_array_3[0] =
229  static_cast<char>((char_array_4[0] << 2) + ((char_array_4[1] & 0x30) >> 4));
230  char_array_3[1] = static_cast<char>(((char_array_4[1] & 0xf) << 4)
231  + ((char_array_4[2] & 0x3c) >> 2));
232  char_array_3[2] =
233  static_cast<char>(((char_array_4[2] & 0x3) << 6) + char_array_4[3]);
234  break;
235  }
236  }
237  }
238  }
239 
240  std::size_t line_size;
241  std::uint8_t pending_in = 0;
242  std::array<char, 4> char_array_4 {};
243  std::uint8_t pending_out = 3;
244  std::uint8_t out_count = 3;
245  std::array<char, 3> char_array_3 {};
246  Base64ErrorHandling errHandling;
247  bool eof = false;
