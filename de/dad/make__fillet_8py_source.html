51 def _preprocess(objs, radius, chamfer):
52  """Check the inputs and return the edges for the fillet and the objects to be deleted."""
53  edges = []
54  del_objs = []
55  if objs[0].isDerivedFrom("Gui::SelectionObject"):
56  for sel in objs:
57  for sub in sel.SubElementNames if sel.SubElementNames else [""]:
58  shape = sel.Object.getSubObject(sub)
59  if shape.ShapeType == "Edge":
60  edges.append(shape)
61  if sel.Object not in del_objs:
62  del_objs.append(sel.Object)
63  else:
64  for obj in objs:
65  if hasattr(obj, "Shape"):
66  shape = obj.Shape
67  del_objs.append(obj)
68  else:
69  shape = obj
70  if hasattr(shape, "ShapeType") and shape.ShapeType in ("Wire", "Edge"):
71  edges.append(shape.Edges[0])
72 
73  if len(edges) != 2:
74  _err(translate("draft", "Two edges are needed."))
75  return None, None
76 
77  edges = DraftGeomUtils.fillet(edges, radius, chamfer)
78  if len(edges) < 3:
79  _err(translate("draft", "Edges are not connected or radius is too large."))
80  return None, None
81 
82  return edges, del_objs

85 def make_fillet(objs, radius=100, chamfer=False, delete=False):
86  """Create a fillet between two edges.
87 
88  Parameters
89  ----------
90  objs: list
91  A list of two objects or shapes of type wire (1st edge is used) or edge,
92  or a 2 edge selection set:`FreeCADGui.Selection.getSelectionEx("", 0)`.
93 
94  radius: float, optional
95  It defaults to 100. The curvature of the fillet.
96 
97  chamfer: bool, optional
98  It defaults to `False`. If it is `True` it no longer produces
99  a rounded fillet but a chamfer (straight edge)
100  with the value of the `radius`.
101 
102  delete: bool, optional
103  It defaults to `False`. If `True` the source objects are deleted.
104  Ignored for shapes.
105 
106  Returns
107  -------
108  Part::Part2DObjectPython
109  The object of Proxy type `'Fillet'`.
110  It returns `None` if it fails producing the object.
111  """
112 
113  edges, del_objs = _preprocess(objs, radius, chamfer)
114  if edges is None:
115  return
116 
117  try:
118  wire = Part.Wire(edges)
119  except Part.OCCError:
120  return None
121 
122  doc = App.activeDocument()
123  obj = doc.addObject("Part::Part2DObjectPython", "Fillet")
124  fillet.Fillet(obj)
125  obj.Shape = wire
126  obj.Length = wire.Length
127  obj.Start = wire.Vertexes[0].Point
128  obj.End = wire.Vertexes[-1].Point
129  obj.FilletRadius = radius
130 
131  if delete:
132  for del_obj in del_objs:
133  doc.removeObject(del_obj.Name)
134 
135  if App.GuiUp:
136  view_fillet.ViewProviderFillet(obj.ViewObject)
137  gui_utils.format_object(obj)
138  gui_utils.select(obj)
139 
140  return obj
