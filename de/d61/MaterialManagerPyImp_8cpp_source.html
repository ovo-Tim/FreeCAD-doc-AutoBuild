226 PyObject* MaterialManagerPy::save(PyObject* args, PyObject* kwds)
227 {

257  return nullptr;

264 
265  std::shared_ptr<MaterialLibrary> library;

267  library = getMaterialManagerPtr()->getLibrary(QString::fromUtf8(libraryName));

270  PyErr_SetString(PyExc_LookupError, "Unknown library");
271  return nullptr;
272  }
273 
274 
275  getMaterialManagerPtr()->saveMaterial(library,
276  sharedMaterial,
277  QString::fromUtf8(path),
278  PyObject_IsTrue(overwrite),
279  PyObject_IsTrue(saveAsCopy),
280  PyObject_IsTrue(saveInherited));
