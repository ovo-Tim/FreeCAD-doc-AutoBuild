65 float cStock::FindRectTop(int& xp, int& yp, int& x_size, int& y_size, bool scanHoriz)
66 {
67  float z = m_stock[xp][yp];
68  bool xr_ok = true;
69  bool xl_ok = scanHoriz;
70  bool yu_ok = true;
71  bool yd_ok = !scanHoriz;
72  x_size = 1;
73  y_size = 1;
74  while (xr_ok || xl_ok || yu_ok || yd_ok) {
75  // sweep right x direction
76  if (xr_ok) {
77  int tx = xp + x_size;
78  if (tx >= m_x) {
79  xr_ok = false;
80  }
81  else {
82  for (int y = yp; y < yp + y_size; y++) {
83  if ((m_attr[tx][y] & SIM_TESSEL_TOP) != 0 || fabs(z - m_stock[tx][y]) > m_res) {
84  xr_ok = false;
85  break;
86  }
87  }
88  if (xr_ok) {
89  x_size++;
90  }
91  }
92  }
93 
94  // sweep left x direction
95  if (xl_ok) {
96  int tx = xp - 1;
97  if (tx < 0) {
98  xl_ok = false;
99  }
100  else {
101  for (int y = yp; y < yp + y_size; y++) {
102  if ((m_attr[tx][y] & SIM_TESSEL_TOP) != 0 || fabs(z - m_stock[tx][y]) > m_res) {
103  xl_ok = false;
104  break;
105  }
106  }
107  if (xl_ok) {
108  x_size++;
109  xp--;
110  }
111  }
112  }
113 
114  // sweep up y direction
115  if (yu_ok) {
116  int ty = yp + y_size;
117  if (ty >= m_y) {
118  yu_ok = false;
119  }
120  else {
121  for (int x = xp; x < xp + x_size; x++) {

156 int cStock::TesselTop(int xp, int yp)
157 {
158  int x_size, y_size;
159  float z = FindRectTop(xp, yp, x_size, y_size, true);
160  bool farRect = false;
161  while (y_size / x_size > 5) {
162  farRect = true;
163  yp += x_size * 5;
164  z = FindRectTop(xp, yp, x_size, y_size, true);
165  }
166 
167  while (x_size / y_size > 5) {
168  farRect = true;
169  xp += y_size * 5;
170  z = FindRectTop(xp, yp, x_size, y_size, false);
171  }
172 
173  // mark all points inside
174  for (int y = yp; y < yp + y_size; y++) {
175  for (int x = xp; x < xp + x_size; x++) {
176  m_attr[x][y] |= SIM_TESSEL_TOP;
177  }
178  }
179 
180  if (z > m_pz + m_res) {
181  // generate 4 3d points
182  Point3D pbl(xp, yp, z);
183  Point3D pbr(xp + x_size, yp, z);
184  Point3D ptl(xp, yp + y_size, z);
185  Point3D ptr(xp + x_size, yp + y_size, z);
186  if (fabs(m_pz + m_lz - z) < SIM_EPSILON) {
187  AddQuad(pbl, pbr, ptr, ptl, facetsOuter);
188  }
189  else {
190  AddQuad(pbl, pbr, ptr, ptl, facetsInner);
191  }
192  }
193 
194  if (farRect) {
195  return -1;
196  }
197  return std::max(0, x_size - 1);
198  // return 0;
199 }

344  facets = &facetsOuter;

350  float newz1 = m_pz;

448 void cStock::Tessellate(Mesh::MeshObject& meshOuter, Mesh::MeshObject& meshInner)
449 {
450  // reset attribs
451  for (int y = 0; y < m_y; y++) {
452  for (int x = 0; x < m_x; x++) {
453  m_attr[x][y] = 0;
454  }
455  }
456 
457  facetsOuter.clear();
458  facetsInner.clear();
459 
460  for (int y = 0; y < m_y; y++) {
461  for (int x = 0; x < m_x; x++) {

466  }
467  }

469  for (int x = 0; x < m_x; x++) {

471  m_attr[x][y] |= SIM_TESSEL_BOT;
472  }
473  if ((m_attr[x][y] & SIM_TESSEL_BOT) == 0) {
474  x += TesselBot(x, y);
475  }
476  }
477  }
478  for (int y = 0; y <= m_y; y++) {
479  TesselSidesX(y);
480  }
481  for (int x = 0; x <= m_x; x++) {
482  TesselSidesY(x);
483  }
484  meshOuter.addFacets(facetsOuter);
485  meshInner.addFacets(facetsInner);
486  facetsOuter.clear();
487  facetsInner.clear();
488 }

491 void cStock::CreatePocket(float cxf, float cyf, float radf, float height)
492 {
493  int cx = (int)((cxf - m_px) / m_res);
494  int cy = (int)((cyf - m_py) / m_res);
495  int rad = (int)(radf / m_res);

497  int ys = std::max(0, cy - rad);

500  int xe = std::min(m_x, cx + rad);
501  for (int y = ys; y < ye; y++) {
502  for (int x = xs; x < xe; x++) {
503  if (((x - cx) * (x - cx) + (y - cy) * (y - cy)) < drad) {
504  if (m_stock[x][y] > height) {
505  m_stock[x][y] = height;

512 void cStock::ApplyLinearTool(Point3D& p1, Point3D& p2, cSimTool& tool)
513 {
514  // translate coordinates
515  Point3D pi1 = ToInner(p1);
516  Point3D pi2 = ToInner(p2);
517  float rad = tool.radius;
518  rad /= m_res;
519  float cupAngle = 180;
520 
521  // strait motion
522  float perpDirX = 1;
523  float perpDirY = 0;
524  cLineSegment path(pi1, pi2);
525  if (path.lenXY > SIM_EPSILON) // only if moving along xy
526  {
527  perpDirX = -path.pDirXY.y;
528  perpDirY = path.pDirXY.x;
529  Point3D start(perpDirX * rad + pi1.x, perpDirY * rad + pi1.y, pi1.z);
530  Point3D mainWay = path.pDir * SIM_WALK_RES;
531  Point3D sideWay(-perpDirX * SIM_WALK_RES, -perpDirY * SIM_WALK_RES, 0);
532  int lenSteps = (int)(path.len / SIM_WALK_RES) + 1;
533  int radSteps = (int)(rad * 2 / SIM_WALK_RES) + 1;
534  float zstep = (pi2.z - pi1.z) / radSteps;
535  float tstep = 2.0 / radSteps;
536  float t = -1;
537  for (int j = 0; j < radSteps; j++) {
538  float z = pi1.z + tool.GetToolProfileAt(t);
539  Point3D p = start;
540  for (int i = 0; i < lenSteps; i++) {
541  int x = (int)p.x;
542  int y = (int)p.y;
543  if (x >= 0 && y >= 0 && x < m_x && y < m_y) {
544  if (m_stock[x][y] > z) {
545  m_stock[x][y] = z;
546  }
547  }
548  p.Add(mainWay);
549  z += zstep;
550  }
551  t += tstep;
552  start.Add(sideWay);
553  }
554  }
555  else {
556  cupAngle = 360;
557  }
558 
559  // end cup
560  for (float r = 0.5f; r <= rad; r += (float)SIM_WALK_RES) {
561  Point3D cupCirc(perpDirX * r, perpDirY * r, pi2.z);
562  float rotang = 180 * SIM_WALK_RES / (3.1415926535 * r);
563  cupCirc.SetRotationAngle(-rotang);
564  float z = pi2.z + tool.GetToolProfileAt(r / rad);
565  for (float a = 0; a < cupAngle; a += rotang) {
566  int x = (int)(pi2.x + cupCirc.x);
567  int y = (int)(pi2.y + cupCirc.y);
568  if (x >= 0 && y >= 0 && x < m_x && y < m_y) {
569  if (m_stock[x][y] > z) {
570  m_stock[x][y] = z;
571  }
572  }
573  cupCirc.Rotate();
574  }
575  }
576 }

578 void cStock::ApplyCircularTool(Point3D& p1, Point3D& p2, Point3D& cent, cSimTool& tool, bool isCCW)
579 {
580  // translate coordinates
581  Point3D pi1 = ToInner(p1);
582  Point3D pi2 = ToInner(p2);
583  Point3D centi(cent.x / m_res, cent.y / m_res, cent.z);
584  float rad = tool.radius;
585  rad /= m_res;
586  float cpx = centi.x;
587  float cpy = centi.y;
588 
589  Point3D xynorm = unit(Point3D(-cpx, -cpy, 0));
590  float crad = sqrt(cpx * cpx + cpy * cpy);
591  // bool shortRad = (crad - rad) < 0.0001;
592  // if (shortRad)
593  // rad = crad;
594  float crad1 = std::max((float)0.5, crad - rad);
595  float crad2 = crad + rad;
596 
597  float sang = atan2(-cpy, -cpx); // start angle
598 
599  cpx += pi1.x;
600  cpy += pi1.y;
601  double eang = atan2(pi2.y - cpy, pi2.x - cpx); // end angle
602 
603  double ang = eang - sang;
604  if (!isCCW && ang > 0) {
605  ang -= 2 * 3.1415926;
606  }
607  if (isCCW && ang < 0) {
608  ang += 2 * 3.1415926;
609  }
610  ang = fabs(ang);
611 
612  // apply path
613  Point3D cupCirc;
614  float tstep = (float)SIM_WALK_RES / rad;
615  float t = -1;
616  for (float r = crad1; r <= crad2; r += (float)SIM_WALK_RES) {
617  cupCirc.x = xynorm.x * r;
618  cupCirc.y = xynorm.y * r;
619  float rotang = (float)SIM_WALK_RES / r;
620  int ndivs = (int)(ang / rotang) + 1;
621  if (!isCCW) {
622  rotang = -rotang;
623  }
624  cupCirc.SetRotationAngleRad(rotang);
625  float z = pi1.z + tool.GetToolProfileAt(t);
626  float zstep = (pi2.z - pi1.z) / ndivs;
627  for (int i = 0; i < ndivs; i++) {
628  int x = (int)(cpx + cupCirc.x);
629  int y = (int)(cpy + cupCirc.y);
630  if (x >= 0 && y >= 0 && x < m_x && y < m_y) {
631  if (m_stock[x][y] > z) {
632  m_stock[x][y] = z;
633  }
634  }
