63 def readPresets():
64 
65  Presets=[]
66  bid = 1 #Unique index
67  for profilefile in profilefiles:
68  if os.path.exists(profilefile):
69  try:
70  with open(profilefile, "r") as csvfile:
71  beamreader = csv.reader(csvfile)
72  for row in beamreader:
73  if (not row) or row[0].startswith("#"):
74  continue
75  try:
76  r=[bid, row[0], row[1], row[2]]
77  for i in range(3,len(row)):
78  r=r+[float(row[i])]
79  if not r in Presets:
80  Presets.append(r)
81  bid=bid+1
82  except ValueError:
83  print("Skipping bad line: "+str(row))
84  except IOError:
85  print("Could not open ",profilefile)
86  return Presets
