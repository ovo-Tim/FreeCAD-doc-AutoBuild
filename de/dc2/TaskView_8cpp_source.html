144 QSize TaskBox::minimumSizeHint() const
145 {
146  // ActionGroup returns a size of 200x100 which leads to problems
147  // when there are several task groups in a panel and the first
148  // one is collapsed. In this case the task panel doesn't expand to
149  // the actually required size and all the remaining groups are
150  // squeezed into the available space and thus the widgets in there
151  // often can't be used any more.
152  // To fix this problem minimumSizeHint() is implemented to again
153  // respect the layout's minimum size.
154  QSize s1 = QSint::ActionGroup::minimumSizeHint();
155  QSize s2 = QWidget::minimumSizeHint();
156  return {qMax(s1.width(), s2.width()), qMax(s1.height(), s2.height())};
157 }

166 void TaskBox::hideGroupBox()
167 {
168  if (!wasShown) {
169  // get approximate height
170  int h=0;
171  int ct = groupLayout()->count();
172  for (int i=0; i<ct; i++) {
173  QLayoutItem* item = groupLayout()->itemAt(i);
174  if (item && item->widget()) {
175  QWidget* w = item->widget();
176  h += w->height();
177  }
178  }
179 
180  m_tempHeight = m_fullHeight = h;
181  // For the very first time the group gets shown
182  // we cannot do the animation because the layouting
183  // is not yet fully done
184  m_foldDelta = 0;
185  }
186  else {
187  m_tempHeight = m_fullHeight = myGroup->height();
188  m_foldDelta = m_fullHeight / myScheme->groupFoldSteps;
189  }
190 
191  m_foldStep = 0.0;
192  m_foldDirection = -1;
193 
194  // make sure to have the correct icon
195  bool block = myHeader->blockSignals(true);
196  myHeader->fold();
197  myHeader->blockSignals(block);
198 
199  myDummy->setFixedHeight(0);
200  myDummy->hide();
201  myGroup->hide();
202 
203  m_foldPixmap = QPixmap();
204  setFixedHeight(myHeader->height());
205  setSizePolicy(QSizePolicy::Preferred, QSizePolicy::Preferred);
206 }

336 bool TaskView::event(QEvent* event)
337 {
338  // Workaround for a limitation in Qt (#0003794)
339  // Line edits and spin boxes don't handle the key combination
340  // Shift+Keypad button (if NumLock is activated)
341  if (event->type() == QEvent::ShortcutOverride) {
342  QWidget* focusWidget = qApp->focusWidget();
343  bool isLineEdit = qobject_cast<QLineEdit*>(focusWidget);
344  bool isSpinBox = qobject_cast<QAbstractSpinBox*>(focusWidget);
345 
346  if (isLineEdit || isSpinBox) {
347  auto kevent = static_cast<QKeyEvent*>(event);
348  Qt::KeyboardModifiers ShiftKeypadModifier = Qt::ShiftModifier | Qt::KeypadModifier;
349  if (kevent->modifiers() == Qt::NoModifier ||
350  kevent->modifiers() == Qt::ShiftModifier ||
351  kevent->modifiers() == Qt::KeypadModifier ||
352  kevent->modifiers() == ShiftKeypadModifier) {
353  switch (kevent->key()) {
354  case Qt::Key_Delete:
355  case Qt::Key_Home:
356  case Qt::Key_End:
357  case Qt::Key_Backspace:
358  case Qt::Key_Left:
359  case Qt::Key_Right:
360  kevent->accept();
361  default:
362  break;
363  }
364  }
365  }
366  }
367  return QScrollArea::event(event);
368 }

370 void TaskView::keyPressEvent(QKeyEvent* ke)
371 {
372  if (ActiveCtrl && ActiveDialog) {
373  if (ke->key() == Qt::Key_Return || ke->key() == Qt::Key_Enter) {
374  // get all buttons of the complete task dialog
375  QList<QPushButton*> list = this->findChildren<QPushButton*>();
376  for (auto pb : list) {
377  if (pb->isDefault() && pb->isVisible()) {
378  if (pb->isEnabled()) {
379 #if defined(FC_OS_MACOSX)
380  // #0001354: Crash on using Enter-Key for confirmation of chamfer or fillet entries
381  QPoint pos = QCursor::pos();
382  QCursor::setPos(pb->parentWidget()->mapToGlobal(pb->pos()));
383 #endif
384  pb->click();
385 #if defined(FC_OS_MACOSX)
386  QCursor::setPos(pos);
387 #endif
388  }
389  return;
390  }
391  }
392  }
393  else if (ke->key() == Qt::Key_Escape && ActiveDialog->isEscapeButtonEnabled()) {
394  // get only the buttons of the button box
395  QDialogButtonBox* box = ActiveCtrl->standardButtons();
396  QList<QAbstractButton*> list = box->buttons();
397  for (auto pb : list) {
398  if (box->buttonRole(pb) == QDialogButtonBox::RejectRole) {
399  if (pb->isEnabled()) {
400 #if defined(FC_OS_MACOSX)
401  // #0001354: Crash on using Enter-Key for confirmation of chamfer or fillet entries
402  QPoint pos = QCursor::pos();
403  QCursor::setPos(pb->parentWidget()->mapToGlobal(pb->pos()));
404 #endif
405  pb->click();
406 #if defined(FC_OS_MACOSX)
407  QCursor::setPos(pos);
408 #endif
409  }
410  return;
411  }
412  }
413 
414  // In case a task panel has no Close or Cancel button
415  // then invoke resetEdit() directly
416  // See also ViewProvider::eventCallback
417  auto func = new Gui::TimerFunction();
418  func->setAutoDelete(true);
419  Gui::Document* doc = Gui::Application::Instance->getDocument(ActiveDialog->getDocumentName().c_str());
420  if (doc) {
421  func->setFunction([doc](){
422  doc->resetEdit();
423  });
424  func->singleShot(0);
425  }
426  }
427  }
428  else {
429  QScrollArea::keyPressEvent(ke);
430  }
431 }

560 void TaskView::showDialog(TaskDialog *dlg)
561 {
562  // if trying to open the same dialog twice nothing needs to be done
563  if (ActiveDialog == dlg)
564  return;
565  assert(!ActiveDialog);
566  assert(!ActiveCtrl);
567 
568  // remove the TaskWatcher as long as the Dialog is up
569  removeTaskWatcher();
570 
571  // first create the control element, set it up and wire it:
572  ActiveCtrl = new TaskEditControl(this);
573  ActiveCtrl->buttonBox->setStandardButtons(dlg->getStandardButtons());
574  TaskDialogAttorney::setButtonBox(dlg, ActiveCtrl->buttonBox);
575 
576  // clang-format off
577  // make connection to the needed signals
578  connect(ActiveCtrl->buttonBox, &QDialogButtonBox::accepted,
579  this, &TaskView::accept);
580  connect(ActiveCtrl->buttonBox, &QDialogButtonBox::rejected,
581  this, &TaskView::reject);
582  connect(ActiveCtrl->buttonBox, &QDialogButtonBox::helpRequested,
583  this, &TaskView::helpRequested);
584  connect(ActiveCtrl->buttonBox, &QDialogButtonBox::clicked,
585  this, &TaskView::clicked);
586  // clang-format on
587 
588  const std::vector<QWidget*>& cont = dlg->getDialogContent();
589 
590  // give to task dialog to customize the button box
591  dlg->modifyStandardButtons(ActiveCtrl->buttonBox);
592 
593  if (dlg->buttonPosition() == TaskDialog::North) {
594  taskPanel->addWidget(ActiveCtrl);
595  for (const auto & it : cont){
596  taskPanel->addWidget(it);
597  }
598  }
599  else {
600  for (const auto & it : cont){
601  taskPanel->addWidget(it);
602  }
603  taskPanel->addWidget(ActiveCtrl);
604  }
605 
606  taskPanel->setScheme(QSint::FreeCADPanelScheme::defaultScheme());
607 
608  if (!dlg->needsFullSpace())
609  taskPanel->addStretch();
610 
611  // set as active Dialog
612  ActiveDialog = dlg;
613 
614  ActiveDialog->open();
615 
616  saveCurrentWidth();
617  getMainWindow()->updateActions();
618 
619  triggerMinimumSizeHint();
620 
621  Q_EMIT taskUpdate();

665 void TaskView::updateWatcher()
666 {
667  if (ActiveWatcher.empty()) {
668  auto panel = Gui::Control().taskPanel();
669  if (panel && panel->ActiveWatcher.size())
670  takeTaskWatcher(panel);
671  }
672 
673  // In case a child of the TaskView has the focus and get hidden we have
674  // to make sure to set the focus on a widget that won't be hidden or
675  // deleted because otherwise Qt may forward the focus via focusNextPrevChild()
676  // to the mdi area which may switch to another mdi view which is not an
677  // acceptable behaviour.
678  QWidget *fw = QApplication::focusWidget();
679  if (!fw)
680  this->setFocus();
681  QPointer<QWidget> fwp = fw;
682  while (fw && !fw->isWindow()) {
683  if (fw == this) {
684  this->setFocus();
685  break;
686  }
687  fw = fw->parentWidget();
688  }
689 
690  // add all widgets for all watcher to the task view
691  for (const auto & it : ActiveWatcher) {
692  bool match = it->shouldShow();
693  std::vector<QWidget*> &cont = it->getWatcherContent();
694  for (auto & it2 : cont) {
695  if (match)
696  it2->show();
697  else
698  it2->hide();
699  }
700  }
701 
702  // In case the previous widget that had the focus is still visible
703  // give it the focus back.
704  if (fwp && fwp->isVisible())
705  fwp->setFocus();
706 
707  triggerMinimumSizeHint();
708 
709  Q_EMIT taskUpdate();
710 }

740 void TaskView::addTaskWatcher()
741 {
742  // add all widgets for all watcher to the task view
743  for (TaskWatcher* tw : ActiveWatcher) {
744  std::vector<QWidget*> &cont = tw->getWatcherContent();
745  for (QWidget* w : cont) {
746  taskPanel->addWidget(w);
747  }
748  }
749 
750  if (!ActiveWatcher.empty())
751  taskPanel->addStretch();
752  updateWatcher();
753 
754 #if QT_VERSION >= QT_VERSION_CHECK(5, 12, 0)
755  // Workaround to avoid a crash in Qt. See also
756  // https://forum.freecad.org/viewtopic.php?f=8&t=39187
757  //
758  // Notify the button box about a style change so that it can
759  // safely delete the style animation of its push buttons.
760  auto box = taskPanel->findChild<QDialogButtonBox*>();

798 void TaskView::removeTaskWatcher()
799 {
800  // In case a child of the TaskView has the focus and get hidden we have
801  // to make sure that set the focus on a widget that won't be hidden or
802  // deleted because otherwise Qt may forward the focus via focusNextPrevChild()
803  // to the mdi area which may switch to another mdi view which is not an
804  // acceptable behaviour.
805  QWidget *fw = QApplication::focusWidget();
806  if (!fw)
807  this->setFocus();
808  while (fw && !fw->isWindow()) {
809  if (fw == this) {
810  this->setFocus();
811  break;
812  }
813  fw = fw->parentWidget();
814  }
815 
816  // remove all widgets
817  for (TaskWatcher* tw : ActiveWatcher) {
818  std::vector<QWidget*> &cont = tw->getWatcherContent();
819  for (QWidget* w : cont) {
820  w->hide();
821  taskPanel->removeWidget(w);
822  }
823  }
824 
825  taskPanel->removeStretch();
826 }
