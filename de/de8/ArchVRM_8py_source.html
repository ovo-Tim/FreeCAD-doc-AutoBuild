187  def projectFace(self,face):
188  "projects a single face on the WP"
189  #print("VRM: projectFace start: ",len(face[0].Vertexes)," verts, ",len(face[0].Edges)," edges")
190  wires = []
191  if not face[0].Wires:
192  if DEBUG: print("Error: Unable to project face on the WP")
193  return None
194  norm = face[0].normalAt(0,0)
195  for w in face[0].Wires:
196  verts = []
197  edges = Part.__sortEdges__(w.Edges)
198  #print(len(edges)," edges after sorting")
199  for e in edges:
200  v = e.Vertexes[0].Point
201  #print(v)
202  v = self.wp.get_local_coords(v)
203  verts.append(v)
204  verts.append(verts[0])
205  if len(verts) > 2:
206  #print("new wire with ",len(verts))
207  wires.append(Part.makePolygon(verts))
208  try:
209  sh = ArchCommands.makeFace(wires)
210  except Exception:
211  if DEBUG: print("Error: Unable to project face on the WP")
212  return None
213  else:
214  # restoring flipped normals
215  vnorm = self.wp.get_local_coords(norm)
216  if vnorm.getAngle(sh.normalAt(0,0)) > 1:
217  sh.reverse()
218  #print("VRM: projectFace end: ",len(sh.Vertexes)," verts")
219  return [sh]+face[1:]

289  def isInside(self,vert,face):
290  "Returns True if the vert is inside the face in Z projection"
291 
292  if not face:
293  return False
294 
295  # http://paulbourke.net/geometry/insidepoly/
296  count = 0
297  p = self.wp.get_local_coords(vert.Point)
298  for e in face[0].Edges:
299  p1 = e.Vertexes[0].Point
300  p2 = e.Vertexes[-1].Point
301  if p.y > min(p1.y,p2.y):
302  if p.y <= max(p1.y,p2.y):
303  if p.x <= max(p1.x,p2.x):
304  if p1.y != p2.y:
305  xinters = (p.y-p1.y)*(p2.x-p1.x)/(p2.y-p1.y)+p1.x
306  if (p1.x == p2.x) or (p.x <= xinters):
307  count += 1
308  if count % 2 == 0:
309  return False
310  else:
311  return True

313  def zOverlaps(self,face1,face2):
314  "Checks if face1 overlaps face2 in Z direction"
315  face1 = self.flattenFace(face1)
316  face2 = self.flattenFace(face2)
317 
318  if (not face1) or (not face2):
319  return False
320 
321  # first we check if one of the verts is inside the other face
322  for v in face1[0].Vertexes:
323  if self.isInside(v,face2):
324  return True
325 
326  # even so, faces can still overlap if their edges cross each other
327  for e1 in face1[0].Edges:
328  for e2 in face2[0].Edges:
329  if DraftGeomUtils.findIntersection(e1,e2):
330  return True
331  return False

333  def compare(self,face1,face2):
334  "zsorts two faces. Returns 1 if face1 is closer, 2 if face2 is closer, 0 otherwise"
335 
336  #print(face1,face2)
337 
338  if not face1:
339  if DEBUG: print("Warning, undefined face!")
340  return 31
341  elif not face2:
342  if DEBUG: print("Warning, undefined face!" )
343  return 32
344 
345  # theory from
346  # http://www.siggraph.org/education/materials/HyperGraph/scanline/visibility/painter.htm
347  # and practical application http://vrm.ao2.it/ (blender vector renderer)
348 
349  b1 = face1[0].BoundBox
350  b2 = face2[0].BoundBox
351 
352  # test 1: if faces don't overlap, no comparison possible
353  if DEBUG: print("doing test 1")
354  if b1.XMax < b2.XMin:
355  return 0
356  if b1.XMin > b2.XMax:
357  return 0
358  if b1.YMax < b2.YMin:
359  return 0
360  if b1.YMin > b2.YMax:
361  return 0
362  if DEBUG: print("failed, faces bboxes are not distinct")
363 
364  # test 2: if Z bounds don't overlap, it's easy to know the closest
365  if DEBUG: print("doing test 2")
366  if b1.ZMax < b2.ZMin:
367  return 2
368  if b2.ZMax < b1.ZMin:
369  return 1
370  if DEBUG: print("failed, faces Z are not distinct")
371 
372  # test 3: all verts of face1 are in front or behind the plane of face2
373  if DEBUG: print("doing test 3")
374  norm = face2[0].normalAt(0,0)
375  behind = 0
376  front = 0
377  for v in face1[0].Vertexes:
378  dv = v.Point.sub(face2[0].Vertexes[0].Point)
379  dv = DraftVecUtils.project(dv,norm)
380  if DraftVecUtils.isNull(dv):
381  behind += 1
382  front += 1
383  else:
384  if dv.getAngle(norm) > 1:
385  behind += 1
386  else:
387  front += 1
388  if DEBUG: print("front: ",front," behind: ",behind)
389  if behind == len(face1[0].Vertexes):
390  return 2
391  elif front == len(face1[0].Vertexes):
392  return 1
393  if DEBUG: print("failed, cannot say if face 1 is in front or behind")
394 
395  # test 4: all verts of face2 are in front or behind the plane of face1
396  if DEBUG: print("doing test 4")
397  norm = face1[0].normalAt(0,0)
398  behind = 0
399  front = 0
400  for v in face2[0].Vertexes:
401  dv = v.Point.sub(face1[0].Vertexes[0].Point)
402  dv = DraftVecUtils.project(dv,norm)
403  if DraftVecUtils.isNull(dv):
404  behind += 1
405  front += 1
406  else:
407  if dv.getAngle(norm) > 1:
408  behind += 1
409  else:
410  front += 1
411  if DEBUG: print("front: ",front," behind: ",behind)
412  if behind == len(face2[0].Vertexes):
413  return 1
414  elif front == len(face2[0].Vertexes):
415  return 2
416  if DEBUG: print("failed, cannot say if face 2 is in front or behind")
417 
418  # test 5: see if faces projections don't overlap, vertexwise
419  if DEBUG: print("doing test 5")
420  if not self.zOverlaps(face1,face2):
421  return 0
422  elif not self.zOverlaps(face2,face1):
423  return 0
424  if DEBUG: print("failed, faces are overlapping")
425 
426  if DEBUG: print("Houston, all tests passed, and still no results")
427  return 0

459  def findPosition(self,f1,faces):
460  "Finds the position of a face in a list of faces"
461  l = None
462  h = None
463  for f2 in faces:
464  if DEBUG: print("comparing face",str(self.faces.index(f1))," with face",str(self.faces.index(f2)))
465  r = self.compare(f1,f2)
466  if r == 1:
467  l = faces.index(f2)
468  elif r == 2:
469  if h is None:
470  h = faces.index(f2)
471  else:
472  if faces.index(f2) < h:
473  h = faces.index(f2)
474  if l is not None:
475  return l + 1
476  elif h is not None:
477  return h
478  else:
479  return None

481  def sort(self):
482  "projects a shape on the WP"
483  if DEBUG: print("\n\n======> Starting sort\n\n")
484  if len(self.faces) <= 1:
485  return
486  if not self.trimmed:
487  self.removeHidden()
488  if DEBUG: print("Done hidden face removal")
489  if len(self.faces) == 1:
490  return
491  if not self.oriented:
492  self.reorient()
493  if DEBUG: print("Done reorientation")
494  faces = self.faces[:]
495  if DEBUG: print("sorting ",len(self.faces)," faces")
496  sfaces = []
497  loopcount = 0
498  notfoundstack = 0
499  while faces:
500  if DEBUG: print("loop ", loopcount)
501  f1 = faces[0]
502  if sfaces and (notfoundstack < len(faces)):
503  if DEBUG: print("using ordered stack, notfound = ",notfoundstack)
504  p = self.findPosition(f1,sfaces)
505  if p is None:
506  # no position found, we move the face to the end of the pile
507  faces.remove(f1)
508  faces.append(f1)
509  notfoundstack += 1
510  else:
511  # position found, we insert it
512  faces.remove(f1)
513  sfaces.insert(p,f1)
514  notfoundstack = 0
515  else:
516  # either there is no stack, or no more face can be compared
517  # find a root, 2 faces that can be compared
518  if DEBUG: print("using unordered stack, notfound = ",notfoundstack)
519  for f2 in faces[1:]:
520  if DEBUG: print("comparing face",str(self.faces.index(f1))," with face",str(self.faces.index(f2)))
521  r = self.compare(f1,f2)
522  print("comparison result:",r)
523  if r == 1:
524  faces.remove(f2)
525  sfaces.append(f2)
526  faces.remove(f1)
527  sfaces.append(f1)
528  notfoundstack = 0
529  break
530  elif r == 2:
531  faces.remove(f1)
532  sfaces.append(f1)
533  faces.remove(f2)
534  sfaces.append(f2)
535  notfoundstack = 0
536  break
537  elif r == 31:
538  if f1 in faces:
539  faces.remove(f1)
540  elif r == 32:
541  if f2 in faces:
542  faces.remove(f2)
543  else:
544  # nothing found, move the face to the end of the pile
545  if f1 in faces:
546  faces.remove(f1)
547  faces.append(f1)
548  loopcount += 1
549  if loopcount == MAXLOOP * len(self.faces):
550  if DEBUG: print("Too many loops, aborting.")
551  break
552 
553  if DEBUG: print("done Z sorting. ", len(sfaces), " faces retained, ", len(self.faces)-len(sfaces), " faces lost.")
554  self.faces = sfaces
555  self.sorted = True
556  if DEBUG: print("\n\n======> Finished sort\n\n")
