76 std::vector<App::DocumentObject*> ViewProviderWeld::claimChildren() const
77 {
78  // Collect any child Document Objects and put them in the right place in the Feature tree
79  // valid children of a DrawWeldSymbol are:
80  // - DrawTiles
81  std::vector<App::DocumentObject*> temp;
82  const std::vector<App::DocumentObject *> &tiles = getFeature()->getInList();
83  try {
84  for(std::vector<App::DocumentObject *>::const_iterator it = tiles.begin(); it != tiles.end(); ++it) {
85  if ((*it)->isDerivedFrom<TechDraw::DrawTile>()) {
86  temp.push_back((*it));
87  }
88  }
89  return temp;
90  } catch (...) {
91  return std::vector<App::DocumentObject*>();
92  }
93 }

132 bool ViewProviderWeld::onDelete(const std::vector<std::string> &)
133 {
134  // a weld cannot be deleted if it has a tile
135 
136  // get childs
137  auto childs = claimChildren();
138 
139  if (!childs.empty()) {
140  QString bodyMessage;
141  QTextStream bodyMessageStream(&bodyMessage);
142  bodyMessageStream << qApp->translate("Std_Delete",
143  "You cannot delete this weld symbol because\nit has a tile weld that would become broken.");
144  QMessageBox::warning(Gui::getMainWindow(),
145  qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
146  QMessageBox::Ok);
147  return false;
148  }
149 
150  return true;
151 }
