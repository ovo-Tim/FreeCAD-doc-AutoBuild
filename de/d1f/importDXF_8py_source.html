342 def calcBulge(v1, bulge, v2):
343  """Calculate intermediary vertex for a curved segment.
344 
345  Considering an arc of a circle, it can be defined by two vertices `v1`
346  and `v2`, and a `bulge` value that indicates how curved the arc is.
347  A `bulge` of 0 is a straight line, while a `bulge` of 1 is the maximum
348  curvature, or a semicircle.
349 
350  A vertex that is in the curve, equidistant to the two vertices,
351  can be found by finding the sagitta of the arc, that is,
352  the perpendicular to the chord that goes from `v1` to `v2`.
353 
354  It uses the algorithm from http://www.afralisp.net/lisp/Bulges1.htm
355 
356  Parameters
357  ----------
358  v1 : Base::Vector3
359  The first point.
360  bulge : float
361  The bulge is the tangent of 1/4 of the included angle for the arc
362  between `v1` and `v2`. A negative `bulge` indicates that the arc
363  goes clockwise from `v1` to `v2`. A `bulge` of 0 indicates
364  a straight segment, and a `bulge` of 1 is a semicircle.
365  v2 : Base::Vector3
366  The second point.
367 
368  Returns
369  -------
370  Base::Vector3
371  The new point between `v1` and `v2`.
372  """
373  chord = v2.sub(v1)
374  sagitta = (bulge * chord.Length)/2
375  perp = chord.cross(Vector(0, 0, 1))
376  startpoint = v1.add(chord.multiply(0.5))
377  if not DraftVecUtils.isNull(perp):
378  perp.normalize()
379  endpoint = perp.multiply(sagitta)
380  return startpoint.add(endpoint)

718 def placementFromDXFOCS(ent):
719  """Return the placement of an object from AutoCAD's OCS.
720 
721  In AutoCAD DXF's the points of each entity are expressed in terms
722  of the entity's object coordinate system (OCS).
723  Then to determine the entity's position in 3D space,
724  what is needed is a 3D vector defining the Z axis of the OCS,
725  and the elevation value over it.
726 
727  It uses `WorkingPlane.align_to_point_and_axis()` to align the working plane
728  to the origin and to `ent.extrusion` (the plane's `axis`).
729  Then it gets the global coordinates of the entity
730  by using `WorkingPlane.get_global_coords()`
731  and either `ent.elevation` (Z coordinate) or `ent.loc` a `(x,y,z)` tuple.
732 
733  Parameters
734  ----------
735  ent : A DXF entity
736  It could be of several types, like `lwpolyline`, `polyline`,
737  and others, and with `ent.extrusion`, `ent.elevation`
738  or `ent.loc` attributes.
739 
740  Returns
741  -------
742  Base::Placement
743  A placement, comprised of a `Base` (`Base::Vector3`),
744  and a `Rotation` (`Base::Rotation`).
745 
746  See also
747  --------
748  WorkingPlane.align_to_point_and_axis, WorkingPlane.get_global_coords
749  """
750  draftWPlane = WorkingPlane.PlaneBase()
751  draftWPlane.align_to_point_and_axis(Vector(0.0, 0.0, 0.0),
752  vec(ent.extrusion), 0.0)
753  # Object Coordinate Systems (OCS)
754  # http://docs.autodesk.com/ACD/2011/ENU/filesDXF/WS1a9193826455f5ff18cb41610ec0a2e719-7941.htm
755  # Arbitrary Axis Algorithm
756  # http://docs.autodesk.com/ACD/2011/ENU/filesDXF/WS1a9193826455f5ff18cb41610ec0a2e719-793d.htm#WSc30cd3d5faa8f6d81cb25f1ffb755717d-7ff5
757  # Riferimenti dell'algoritmo dell'asse arbitrario in italiano
758  # http://docs.autodesk.com/ACD/2011/ITA/filesDXF/WS1a9193826455f5ff18cb41610ec0a2e719-7941.htm
759  # http://docs.autodesk.com/ACD/2011/ITA/filesDXF/WS1a9193826455f5ff18cb41610ec0a2e719-793d.htm#WSc30cd3d5faa8f6d81cb25f1ffb755717d-7ff5
760  if (draftWPlane.axis == FreeCAD.Vector(1.0, 0.0, 0.0)):
761  draftWPlane.u = FreeCAD.Vector(0.0, 1.0, 0.0)
762  draftWPlane.v = FreeCAD.Vector(0.0, 0.0, 1.0)
763  elif (draftWPlane.axis == FreeCAD.Vector(-1.0, 0.0, 0.0)):
764  draftWPlane.u = FreeCAD.Vector(0.0, -1.0, 0.0)
765  draftWPlane.v = FreeCAD.Vector(0.0, 0.0, 1.0)
766  else:
767  if ((abs(ent.extrusion[0]) < (1.0 / 64.0)) and (abs(ent.extrusion[1]) < (1.0 / 64.0))):
768  draftWPlane.u = FreeCAD.Vector(0.0, 1.0, 0.0).cross(draftWPlane.axis)
769  else:
770  draftWPlane.u = FreeCAD.Vector(0.0, 0.0, 1.0).cross(draftWPlane.axis)
771  draftWPlane.u.normalize()
772  draftWPlane.v = draftWPlane.axis.cross(draftWPlane.u)
773  draftWPlane.v.normalize()
774  draftWPlane.position = Vector(0.0, 0.0, 0.0)
775 
776  pl = draftWPlane.get_placement()
777  if ((ent.type == "lwpolyline") or (ent.type == "polyline")):
778  pl.Base = draftWPlane.get_global_coords(vec([0.0, 0.0, ent.elevation]))
779  else:
780  pl.Base = draftWPlane.get_global_coords(vec(ent.loc))
781  return pl

910  if not DraftVecUtils.equals(v1, v2):
911  if DraftVecUtils.isColinear([v1, cv, v2]):
912  try:
913  edges.append(Part.LineSegment(v1, v2).toShape())
914  except Part.OCCError:

916  else:
917  try:

919  except Part.OCCError:
