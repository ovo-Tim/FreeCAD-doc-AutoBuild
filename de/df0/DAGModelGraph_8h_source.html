174  class RakeLeaves
175  {
176  using GraphInVertex = boost::graph_traits<Graph>::vertex_descriptor;
177  using GraphInVertices = std::vector<GraphInVertex>;
178  public:
179  explicit RakeLeaves(const GraphIn &graphIn) : graph(graphIn) {}
180  GraphInVertices operator()() const
181  {
182  GraphInVertices out;
183  BGL_FORALL_VERTICES_T(currentVertex, graph, GraphIn)
184  {
185  if (boost::out_degree(currentVertex, graph) == 0)
186  out.push_back(currentVertex);
187  }
188  return out;
189  }
190  private:
191  const GraphIn &graph;
192  };
