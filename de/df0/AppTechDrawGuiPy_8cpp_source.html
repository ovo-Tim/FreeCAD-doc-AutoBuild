118  Py::Object exporter(const Py::Tuple& args)
119  {
120  PyObject* object;
121  char* Name;
122  if (!PyArg_ParseTuple(args.ptr(), "Oet", &object, "utf-8", &Name))
123  throw Py::Exception();
124 
125  std::string EncodedName = std::string(Name);
126  PyMem_Free(Name);
127 
128  TechDraw::DrawPage* page = nullptr;
129  Py::Sequence list(object);
130  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
131  PyObject* item = (*it).ptr();
132  if (PyObject_TypeCheck(item, &(App::DocumentObjectPy::Type))) {
133  App::DocumentObject* obj =
134  static_cast<App::DocumentObjectPy*>(item)->getDocumentObjectPtr();
135  if (obj->isDerivedFrom<TechDraw::DrawPage>()) {
136  page = static_cast<TechDraw::DrawPage*>(obj);
137  Gui::Document* activeGui =
138  Gui::Application::Instance->getDocument(page->getDocument());
139  Gui::ViewProvider* vp = activeGui->getViewProvider(obj);
140  ViewProviderPage* vpPage = dynamic_cast<ViewProviderPage*>(vp);
141  if (!vpPage) {
142  throw Py::TypeError("TechDraw can not find Page");
143  }
144 
145  Base::FileInfo fi_out(EncodedName.c_str());
146 
147  if (fi_out.hasExtension("svg")) {
148  PagePrinter::saveSVG(vpPage, EncodedName);
149  }
150  else if (fi_out.hasExtension("dxf")) {
151  PagePrinter::saveDXF(vpPage, EncodedName);
152  }
153  else if (fi_out.hasExtension("pdf")) {
154  PagePrinter::savePDF(vpPage, EncodedName);
155  }
156  else {
157  throw Py::TypeError("TechDraw can not export this file format");
158  }
159  }
160  else {
161  throw Py::TypeError("No Technical Drawing Page found in selection.");
162  }
163  }
164  }
165 
166  return Py::None();
167  }
