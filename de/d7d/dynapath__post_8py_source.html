288 def parse(pathobj):
289  global PRECISION
290  global UNIT_FORMAT
291  global UNIT_SPEED_FORMAT
292 
293  out = ""
294  lastcommand = None
295 
296  precision_string = "." + str(PRECISION) + "f"
297 
298  # params = ['X','Y','Z','A','B','I','J','K','F','S'] #This list control the order of parameters
299  params = ["X", "Y", "Z", "A", "B", "I", "J", "F", "S", "T", "Q", "R", "L"]
300 
301  if hasattr(pathobj, "Group"): # We have a compound or project.
302  if OUTPUT_COMMENTS:
303  out += linenumber() + "(compound: " + pathobj.Label + ")\n"
304  for p in pathobj.Group:
305  out += parse(p)
306  return out
307  else: # parsing simple path
308 
309  if not hasattr(pathobj, "Path"): # groups might contain non-path things like stock.
310  return out
311 
312  if OUTPUT_COMMENTS:
313  out += linenumber() + "(Path: " + pathobj.Label + ")\n"
314 
315  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
316  outstring = []
317  command = c.Name
318  outstring.append(command)
319  # if modal: only print the command if it is not the same as the last one
320  if MODAL is True:
321  if command == lastcommand:
322  outstring.pop(0)
323 
324  # Now add the remaining parameters in order
325  for param in params:
326  if param in c.Parameters:
327  if param == "F":
328  speed = Units.Quantity(c.Parameters["F"], FreeCAD.Units.Velocity)
329  if speed.getValueAs(UNIT_SPEED_FORMAT) > 0.0:
330  outstring.append(
331  param
332  + format(
333  float(speed.getValueAs(UNIT_SPEED_FORMAT)),
334  precision_string,
335  )
336  )
337  elif param == "S":
338  outstring.append(param + format(c.Parameters[param], precision_string))
339  elif param == "T":
340  outstring.append(param + format(c.Parameters["T"], precision_string))
341  else:
342  pos = Units.Quantity(c.Parameters[param], FreeCAD.Units.Length)
343  outstring.append(
344  param + format(float(pos.getValueAs(UNIT_FORMAT)), precision_string)
345  )
346 
347  # store the latest command
348  lastcommand = command
349 
350  # Check for Tool Change:
351  if command == "M6":
352  if OUTPUT_COMMENTS:
353  out += linenumber() + "(begin toolchange)\n"
354  for line in TOOL_CHANGE.splitlines(True):
355  out += linenumber() + line
356 
357  if command == "message":
358  if OUTPUT_COMMENTS is False:
359  out = []
360  else:
361  outstring.pop(0) # remove the command
362 
363  # prepend a line number and append a newline
364  if len(outstring) >= 1:
365  if OUTPUT_LINE_NUMBERS:
366  outstring.insert(0, (linenumber()))
367 
368  # append the line to the final output
369  for w in outstring:
370  out += w + COMMAND_SPACE
371  out = out.strip() + "\n"
372 
373  return out
