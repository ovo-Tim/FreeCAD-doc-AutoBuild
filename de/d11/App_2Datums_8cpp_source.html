155 App::DatumElement* LocalCoordinateSystem::getDatumElement(const char* role) const
156 {
157  const auto& features = OriginFeatures.getValues();
158  auto featIt = std::find_if(features.begin(), features.end(), [role](App::DocumentObject* obj) {
159  return obj->isDerivedFrom(App::DatumElement::getClassTypeId())
160  && strcmp(static_cast<App::DatumElement*>(obj)->Role.getValue(), role) == 0;
161  });
162  if (featIt != features.end()) {
163  return static_cast<App::DatumElement*>(*featIt);
164  }
165  std::stringstream err;
166  err << "LocalCoordinateSystem \"" << getFullName() << "\" doesn't contain feature with role \""
167  << role << '"';
168  throw Base::RuntimeError(err.str().c_str());
169 }

221 App::DocumentObjectExecReturn* LocalCoordinateSystem::execute()
222 {
223  try { // try to find all base axis and planes in the origin
224  for (const char* role : AxisRoles) {
225  App::Line* axis = getAxis(role);

227  (void)axis;

229  for (const char* role : PlaneRoles) {

259 DatumElement* LocalCoordinateSystem::createDatum(const SetupData& data)
260 {
261  App::Document* doc = getDocument();
262  std::string objName = doc->getUniqueObjectName(data.role);
263  App::DocumentObject* featureObj = doc->addObject(data.type.getName(), objName.c_str());
264 
265  assert(featureObj && featureObj->isDerivedFrom(App::DatumElement::getClassTypeId()));
266 
267  QByteArray byteArray = data.label.toUtf8();
268  featureObj->Label.setValue(byteArray.constData());
269 
270  auto* feature = static_cast<App::DatumElement*>(featureObj);
271  feature->Placement.setValue(Base::Placement(Base::Vector3d(), data.rot));
272  feature->Role.setValue(data.role);
273 
274  feature->Placement.setStatus(Property::Hidden, true);
275  return feature;
276 }

300 void LocalCoordinateSystem::unsetupObject()
301 {
302  const auto& objsLnk = OriginFeatures.getValues();
303  // Copy to set to assert we won't call method more then one time for each object
304  std::set<App::DocumentObject*> objs(objsLnk.begin(), objsLnk.end());
305  // Remove all controlled objects
306  for (auto obj : objs) {
307  // Check that previous deletes didn't indirectly remove one of our objects
308  const auto& objsLnk = OriginFeatures.getValues();
309  if (std::find(objsLnk.begin(), objsLnk.end(), obj) != objsLnk.end()) {
310  if (!obj->isRemoving()) {
311  obj->getDocument()->removeObject(obj->getNameInDocument());
312  }
313  }
314  }

325 void LocalCoordinateSystem::migrateOriginPoint()
326 {
327  auto features = OriginFeatures.getValues();
328 
329  auto isOrigin = [](App::DocumentObject* obj) {
330  return obj->isDerivedFrom<App::DatumElement>() &&
331  strcmp(static_cast<App::DatumElement*>(obj)->Role.getValue(), PointRoles[0]) == 0;
332  };
333  if (std::none_of(features.begin(), features.end(), isOrigin)) {
334  auto data = getData(PointRoles[0]);
335  auto* origin = createDatum(data);
336  origin->purgeTouched();
337  features.push_back(origin);
338  OriginFeatures.setValues(features);
339  }
340 }

355 bool LocalCoordinateSystem::LCSExtension::extensionGetSubObject(DocumentObject*& ret,
356  const char* subname,
357  PyObject** pyobj,
358  Base::Matrix4D* mat,
359  bool,
360  int depth) const
361 {
362  if (!subname || subname[0] == '\0') {
363  return false;
364  }
365 
366  // mapping of object name to role name
367  std::string name(subname);
368  for (int i = 0; i < 3; i++) {
369  if (name.rfind(LocalCoordinateSystem::AxisRoles[i], 0) == 0) {
370  name = LocalCoordinateSystem::AxisRoles[i];
371  break;
372  }
373  if (name.rfind(LocalCoordinateSystem::PlaneRoles[i], 0) == 0) {
374  name = LocalCoordinateSystem::PlaneRoles[i];
375  break;
376  }
377  }
378 
379  if (name.rfind(LocalCoordinateSystem::PointRoles[0], 0) == 0) {
380  name = LocalCoordinateSystem::PointRoles[0];
381  }
382 
383  try {
384  ret = obj->getDatumElement(name.c_str());
385  if (!ret) {
386  return false;
387  }
388  const char* dot = strchr(subname, '.');
389  if (dot) {
390  subname = dot + 1;
391  }
392  else {
393  subname = "";
394  }
395  ret = ret->getSubObject(subname, pyobj, mat, true, depth + 1);
396  return true;
397  }
398  catch (const Base::Exception& e) {
399  e.ReportException();
400  return false;
401  }
402 }
