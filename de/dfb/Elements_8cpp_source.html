241 bool MeshGeomEdge::IntersectWithLine(const Base::Vector3f& rclPt,
242  const Base::Vector3f& rclDir,
243  Base::Vector3f& rclRes) const
244 {
245  const float eps = 1e-06F;
246  Base::Vector3f n = _aclPoints[1] - _aclPoints[0];
247 
248  // check angle between edge and the line direction, FLOAT_MAX is
249  // returned for degenerated edges
250  float fAngle = rclDir.GetAngle(n);
251  if (fAngle == 0) {
252  // parallel lines
253  float distance = _aclPoints[0].DistanceToLine(rclPt, rclDir);
254  if (distance < eps) {
255  // lines are equal
256  rclRes = _aclPoints[0];
257  return true;
258  }
259 
260  return false; // no intersection possible
261  }
262 
263  // that's the normal of a helper plane and its base at _aclPoints
264  Base::Vector3f normal = n.Cross(rclDir);
265 
266  // if the distance of rclPt to the plane is higher than eps then the
267  // two lines are warped and there is no intersection possible
268  if (fabs(rclPt.DistanceToPlane(_aclPoints[0], normal)) > eps) {
269  return false;
270  }
271 
272  // get a second helper plane and get the intersection with the line
273  Base::Vector3f normal2 = normal.Cross(n);
274 
275  float s = ((_aclPoints[0] - rclPt) * normal2) / (rclDir * normal2);
276  rclRes = rclPt + s * rclDir;
277 
278  float dist1 = Base::Distance(_aclPoints[0], _aclPoints[1]);
279  float dist2 = Base::Distance(_aclPoints[0], rclRes);
280  float dist3 = Base::Distance(_aclPoints[1], rclRes);
281 
282  return dist2 + dist3 <= dist1 + eps;
283 }

304 bool MeshGeomEdge::IntersectWithEdge(const MeshGeomEdge& edge, Base::Vector3f& res) const
305 {
306  const float eps = 1e-06F;
307  Base::Vector3f p(_aclPoints[0]);
308  Base::Vector3f r(_aclPoints[1] - _aclPoints[0]);
309  Base::Vector3f q(edge._aclPoints[0]);
310  Base::Vector3f s(edge._aclPoints[1] - edge._aclPoints[0]);
311  Base::Vector3f n = r.Cross(s);
312  Base::Vector3f d = q - p;
313 
314  // lines are collinear or parallel
315  if (n.IsNull()) {
316  if (d.Cross(r).IsNull()) {
317  // Collinear
318  if (IsProjectionPointOf(edge._aclPoints[0])) {
319  res = edge._aclPoints[0];
320  return true;
321  }
322  if (IsProjectionPointOf(edge._aclPoints[1])) {
323  res = edge._aclPoints[1];
324  return true;
325  }
326 
327  return false;
328  }
329 
330  // Parallel
331  return false;
332  }
333  else {
334  // Get the distance of q to the plane defined by p and n
335  float distance = q.DistanceToPlane(p, n);
336 
337  // lines are warped
338  if (std::fabs(distance) > eps) {
339  return false;
340  }
341 
342  float t = d.Cross(s).Dot(n) / n.Sqr();
343  float u = d.Cross(r).Dot(n) / n.Sqr();
344 
345  auto is_in_range = [](float v) {
346  return v >= 0.0F && v <= 1.0F;
347  };
348 
349  if (is_in_range(t) && is_in_range(u)) {
350  res = p + t * r; // equal to q + u * s
351  return true;
352  }
353 
354  return false;
355  }
356 }

388  const Base::Vector3f& lineDir,

431 {

634 bool MeshGeomFacet::IsDegenerated(float epsilon) const
635 {
636  // The triangle has the points A,B,C where we can define the vector u and v
637  // u = b-a and v = c-a. Then we define the line g: r = a+t*u and the plane
638  // E: (x-c)*u=0. The intersection point of g and E is S.
639  // The vector to S can be computed with a+(uv)/(uu)*u. The difference of
640  // C and S then is v-(u*v)/(u*u)*u. The square distance leads to the formula
641  // (v-(u*v)/(u*u)*u)^2 < eps which means that C and S is considered equal if
642  // the square distance is less than an epsilon and thus the triangle is de-
643  // generated.
644  // After a few calculation step we get the formula:
645  // (u*u)*(v*v)-(u*v)*(u*v) < eps*(u*u)
646  // So, if we do the same except that we define a line h which goes through
647  // A and C and a plane going through B we get a similar formula:
648  // (u*u)*(v*v)-(u*v)*(u*v) < eps*(v*v).
649  // As end formula we can write then:
650  // (u*u)*(v*v)-(u*v)*(u*v) < max(eps*(u*u),eps*(v*v)).
651  //
652  // BTW (u*u)*(v*v)-(u*v)*(u*v) is the same as (uxv)*(uxv).
653  Base::Vector3d p1 = Base::convertTo<Base::Vector3d>(this->_aclPoints[0]);
654  Base::Vector3d p2 = Base::convertTo<Base::Vector3d>(this->_aclPoints[1]);
655  Base::Vector3d p3 = Base::convertTo<Base::Vector3d>(this->_aclPoints[2]);
656 
657  Base::Vector3d u = p2 - p1;
658  Base::Vector3d v = p3 - p1;
659 
660  double eps = static_cast<double>(epsilon);
661  double uu = u * u;
662  if (uu <= eps) {
663  return true;
664  }
665  double vv = v * v;
666  if (vv <= eps) {
667  return true;
668  }
669  double uv = u * v;
670  double crosssqr = uu * vv - uv * uv;
671  if (crosssqr <= eps * std::max<double>(uu, vv)) {
672  return true;
673  }
674  return false;
675 }

880 bool MeshGeomFacet::Foraminate(const Base::Vector3f& P,
881  const Base::Vector3f& dir,
882  Base::Vector3f& I,
883  float fMaxAngle) const
884 {
885  const float eps = 1e-06F;
886  Base::Vector3f n = this->GetNormal();
887 
888  // check angle between facet normal and the line direction, FLOAT_MAX is
889  // returned for degenerated facets
890  float fAngle = dir.GetAngle(n);
891  if (fAngle > fMaxAngle) {
892  return false;
893  }
894 
895  float nn = n * n;
896  float nd = n * dir;
897  float dd = dir * dir;
898 
899  // the line mustn't be parallel to the triangle
900  if ((nd * nd) <= (eps * dd * nn)) {
901  return false;
902  }
903 
904  Base::Vector3f u = this->_aclPoints[1] - this->_aclPoints[0];
905  Base::Vector3f v = this->_aclPoints[2] - this->_aclPoints[0];
906 
907  Base::Vector3f w0 = P - this->_aclPoints[0];
908  float r = -(n * w0) / nd;
909  Base::Vector3f w = w0 + r * dir;
910 
911  float uu = u * u;
912  float uv = u * v;
913  float vv = v * v;
914  float wu = w * u;
915  float wv = w * v;
916  float det = float(std::fabs((uu * vv) - (uv * uv)));
917 
918  float s = (vv * wu) - (uv * wv);
919  float t = (uu * wv) - (uv * wu);
920 
921  // is the intersection point inside the triangle?
922  if ((s >= 0.0F) && (t >= 0.0F) && ((s + t) <= det)) {
923  I = w + this->_aclPoints[0];
924  return true;
925  }
926 
927  return false;
928 }

1009

1011  float fLenAB = clVecAB.Length();
1012  float fLenAC = clVecAC.Length();
1013  float fLenBC = clVecBC.Length();
1014 
1015  if (fLenAC > fLenAB) {
1016  std::swap(B, C);
1017  std::swap(fLenAB, fLenAC);
1018  }
1019  if (fLenBC > fLenAB) {
1020  std::swap(A, C);
1021  std::swap(fLenBC, fLenAB);
1022  }
1023 
1024  clVecAB = (B - A);
1025  clVecAC = (C - A);
1026  clVecBC = (C - B);

1030  clVecHNorm.Normalize();

1033  float cy = float(std::sin(clVecAB.GetAngle(clVecAC)) * fLenAC);
1034  float cx = float(std::sqrt(std::fabs(fLenAC * fLenAC - cy * cy)));

1039  {
1040  for (float py = (fStep / 2.0f); py < cy; py += fStep) // NOLINT
1041  {
1042  float u = (bx * cy + cx * py - px * cy - bx * py) / fDetABC;
1043  float v = (px * cy - cx * py) / fDetABC;
1044  float w = (bx * py) / fDetABC;

1050  }
1051  else {
1052  break;
1053  }
1054  }
1055  }
1056 
1057  // if couldn't subsample the facet take gravity center
1058  if (clPoints.empty()) {
1059  clPoints.push_back(this->GetGravityPoint());
1060  }
1061 
1062  rclPoints.insert(rclPoints.end(), clPoints.begin(), clPoints.end());
