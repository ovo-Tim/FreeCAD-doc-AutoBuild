397  if not hasattr(pathobj, "Path"): # groups might contain non-path things like stock.
398  return out

401  out += linenumber() + "(Path: " + pathobj.Label + ")\n"

404  outstring = []

406

424  outstring.append(param + str(int(c.Parameters[param])))

427  elif param in ["A", "B", "C"]:
428  outstring.append(param + format(c.Parameters[param], precision_string))

479  if command in SUPPRESS_COMMANDS:

481  outstring.append(")")
482

496 def drill_translate(outstring, cmd, params):
497  global DRILL_RETRACT_MODE
498  global MOTION_MODE
499  global CURRENT_X
500  global CURRENT_Y
501  global CURRENT_Z
502  global UNITS
503  global UNIT_FORMAT
504  global UNIT_SPEED_FORMAT
505 
506  strFormat = "." + str(PRECISION) + "f"
507 
508  trBuff = ""
509 
510  if OUTPUT_COMMENTS: # Comment the original command
511  outstring[0] = "(" + outstring[0]
512  outstring[-1] = outstring[-1] + ")"
513  trBuff += linenumber() + format_outstring(outstring) + "\n"
514 
515  # cycle conversion
516  # currently only cycles in XY are provided (G17)
517  # other plains ZX (G18) and YZ (G19) are not dealt with : Z drilling only.
518  drill_X = Units.Quantity(params["X"], FreeCAD.Units.Length)
519  drill_Y = Units.Quantity(params["Y"], FreeCAD.Units.Length)
520  drill_Z = Units.Quantity(params["Z"], FreeCAD.Units.Length)
521  RETRACT_Z = Units.Quantity(params["R"], FreeCAD.Units.Length)
522  # R less than Z is error
523  if RETRACT_Z < drill_Z:
524  trBuff += linenumber() + "(drill cycle error: R less than Z )\n"
525  return trBuff
526 
527  if MOTION_MODE == "G91": # G91 relative movements
528  drill_X += CURRENT_X
529  drill_Y += CURRENT_Y
530  drill_Z += CURRENT_Z
531  RETRACT_Z += CURRENT_Z
532 
533  if DRILL_RETRACT_MODE == "G98" and CURRENT_Z >= RETRACT_Z:
534  RETRACT_Z = CURRENT_Z
535 
536  # get the other parameters
537  drill_feedrate = Units.Quantity(params["F"], FreeCAD.Units.Velocity)
538  if cmd == "G83":
539  drill_Step = Units.Quantity(params["Q"], FreeCAD.Units.Length)
540  a_bit = (
541  drill_Step * 0.05
542  ) # NIST 3.5.16.4 G83 Cycle: "current hole bottom, backed off a bit."
543  elif cmd == "G82":
544  drill_DwellTime = params["P"]
545 
546  # wrap this block to ensure machine MOTION_MODE is restored in case of error
547  try:
548  strG0_RETRACT_Z = (
549  "G0 Z" + format(float(RETRACT_Z.getValueAs(UNIT_FORMAT)), strFormat) + "\n"
550  )
551  strF_Feedrate = (
552  " F" + format(float(drill_feedrate.getValueAs(UNIT_SPEED_FORMAT)), ".2f") + "\n"
553  )
554  print(strF_Feedrate)
555 
556  # preliminary movement(s)
557  if CURRENT_Z < RETRACT_Z:
558  trBuff += linenumber() + strG0_RETRACT_Z
559  trBuff += (
560  linenumber()
561  + "G0 X"
562  + format(float(drill_X.getValueAs(UNIT_FORMAT)), strFormat)
563  + " Y"
564  + format(float(drill_Y.getValueAs(UNIT_FORMAT)), strFormat)
565  + "\n"
566  )
567  if CURRENT_Z > RETRACT_Z:
568  # NIST GCODE 3.5.16.1 Preliminary and In-Between Motion says G0 to RETRACT_Z. Here use G1 since retract height may be below surface !
569  trBuff += (
570  linenumber()
571  + "G1 Z"
572  + format(float(RETRACT_Z.getValueAs(UNIT_FORMAT)), strFormat)
573  + strF_Feedrate
574  )
575  last_Stop_Z = RETRACT_Z
576 
577  # drill moves
578  if cmd in ("G81", "G82"):
579  trBuff += (
580  linenumber()
581  + "G1 Z"
582  + format(float(drill_Z.getValueAs(UNIT_FORMAT)), strFormat)
583  + strF_Feedrate
584  )
585  # pause where applicable
586  if cmd == "G82":
587  trBuff += linenumber() + "G4 P" + str(drill_DwellTime) + "\n"
588  trBuff += linenumber() + strG0_RETRACT_Z
589  else: # 'G83'
590  if params["Q"] != 0:
591  while 1:
592  if last_Stop_Z != RETRACT_Z:
593  clearance_depth = (
594  last_Stop_Z + a_bit
595  ) # rapid move to just short of last drilling depth
596  trBuff += (
597  linenumber()
598  + "G0 Z"
599  + format(
600  float(clearance_depth.getValueAs(UNIT_FORMAT)),
601  strFormat,
602  )
603  + "\n"
604  )
605  next_Stop_Z = last_Stop_Z - drill_Step
606  if next_Stop_Z > drill_Z:
607  trBuff += (
608  linenumber()
609  + "G1 Z"
610  + format(float(next_Stop_Z.getValueAs(UNIT_FORMAT)), strFormat)
611  + strF_Feedrate
612  )
613  trBuff += linenumber() + strG0_RETRACT_Z
614  last_Stop_Z = next_Stop_Z
615  else:
616  trBuff += (
617  linenumber()
618  + "G1 Z"
619  + format(float(drill_Z.getValueAs(UNIT_FORMAT)), strFormat)
620  + strF_Feedrate
621  )
622  trBuff += linenumber() + strG0_RETRACT_Z
623  break
624 
625  except Exception as e:
626  pass
627 
628  return trBuff
