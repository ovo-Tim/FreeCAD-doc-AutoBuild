77  def set_z88_elparam(self):
78  # TODO: param should be moved to the solver object like the known analysis
79  z8804 = {"INTORD": "0", "INTOS": "0", "IHFLAG": "0", "ISFLAG": "1"} # seg2 --> stab4
80  z8824 = {"INTORD": "7", "INTOS": "7", "IHFLAG": "1", "ISFLAG": "1"} # tria6 --> schale24
81  z8823 = {"INTORD": "3", "INTOS": "0", "IHFLAG": "1", "ISFLAG": "0"} # quad8 --> schale23
82  z8817 = {"INTORD": "4", "INTOS": "0", "IHFLAG": "0", "ISFLAG": "0"} # tetra4 --> volume17
83  z8816 = {"INTORD": "4", "INTOS": "0", "IHFLAG": "0", "ISFLAG": "0"} # tetra10 --> volume16
84  z8801 = {"INTORD": "2", "INTOS": "2", "IHFLAG": "0", "ISFLAG": "1"} # hexa8 --> volume1
85  z8810 = {"INTORD": "3", "INTOS": "0", "IHFLAG": "0", "ISFLAG": "0"} # hexa20 --> volume10
86  param = {4: z8804, 24: z8824, 23: z8823, 17: z8817, 16: z8816, 1: z8801, 10: z8810}
87  # TODO: test elements 17, 16, 10, INTORD etc
88  self.z88_element_type = importZ88Mesh.get_z88_element_type(
89  self.femmesh, self.femelement_table
90  )
91  if self.z88_element_type in param:
92  self.z88_elparam = param[self.z88_element_type]
93  else:
94  FreeCAD.Console.PrintError(
95  "Element type not supported by Z88. Can not write Z88 solver input.\n"
96  )
97  return False
98  FreeCAD.Console.PrintMessage(self.z88_elparam)
99  FreeCAD.Console.PrintMessage("\n")

117  def write_z88_constraints(self):
118  constraints_data = [] # will be a list of tuple for better sorting
119 
120  # fixed constraints
121  # write nodes to constraints_data (different from writing to file in ccxInpWriter
122  for femobj in self.member.cons_fixed:
123  for n in femobj["Nodes"]:
124  constraints_data.append((n, f"{n} 1 2 0\n"))
125  constraints_data.append((n, f"{n} 2 2 0\n"))
126  constraints_data.append((n, f"{n} 3 2 0\n"))
127 
128  # forces constraints
129  # write node loads to constraints_data
130  # a bit different from writing to file for ccxInpWriter
131  for femobj in self.member.cons_force:
132  # femobj --> dict, FreeCAD document object is femobj["Object"]
133  direction_vec = femobj["Object"].DirectionVector
134  for ref_shape in femobj["NodeLoadTable"]:
135  for n in sorted(ref_shape[1]):
136  # the loads in ref_shape[1][n] are without unit
137  node_load = ref_shape[1][n]
138  if direction_vec.x != 0.0:
139  v1 = direction_vec.x * node_load
140  constraints_data.append((n, f"{n} 1 1 {v1}\n"))
141  if direction_vec.y != 0.0:
142  v2 = direction_vec.y * node_load
143  constraints_data.append((n, f"{n} 2 1 {v2}\n"))
144  if direction_vec.z != 0.0:
145  v3 = direction_vec.z * node_load
146  constraints_data.append((n, f"{n} 3 1 {v3}\n"))
147 
148  # write constraints_data to file
149  constraints_file_path = self.file_name + "i2.txt"
150  f = open(constraints_file_path, "w")
151  f.write(str(len(constraints_data)) + "\n")
152  for c in sorted(constraints_data):
153  f.write(c[1])
154  f.close()

185  def write_z88_elements_properties(self):
186  element_properties_file_path = self.file_name + "elp.txt"
187  elements_data = []
188  if meshtools.is_edge_femmesh(self.femmesh):
189  beam_obj = self.member.geos_beamsection[0]["Object"]
190  area = 0
191  if beam_obj.SectionType == "Rectangular":
192  width = beam_obj.RectWidth.getValueAs("mm").Value
193  height = beam_obj.RectHeight.getValueAs("mm").Value
194  area = width * height
195  elif beam_obj.SectionType == "Circular":
196  diameter = beam_obj.CircDiameter.getValueAs("mm").Value
197  from math import pi
198 
199  area = 0.25 * pi * diameter * diameter
200  else:
201  FreeCAD.Console.PrintError(
202  "Cross section type {} not supported, "
203  "cross section area will be 0 in solver input.\n".format(beam_obj.SectionType)
204  )
205  # TODO make the check in prechecks and delete it here
206  # no extensive errorhandling in writer
207  # this way the solver will fail and an exception is raised somehow
208  elements_data.append(f"1 {self.element_count} {area} 0 0 0 0 0 0 ")
209  FreeCAD.Console.PrintWarning("Be aware, only trusses are supported for edge meshes!\n")
210  elif meshtools.is_face_femmesh(self.femmesh):
211  thick_obj = self.member.geos_shellthickness[0]["Object"]
212  thickness = thick_obj.Thickness.getValueAs("mm").Value
213  elements_data.append(f"1 {self.element_count} {thickness} 0 0 0 0 0 0 ")
214  elif meshtools.is_solid_femmesh(self.femmesh):
215  elements_data.append(f"1 {self.element_count} 0 0 0 0 0 0 0")
216  else:
217  FreeCAD.Console.PrintError("Error!\n")
218  f = open(element_properties_file_path, "w")
219  f.write(str(len(elements_data)) + "\n")
220  for e in elements_data:
221  f.write(e)
222  f.write("\n")
223  f.close()
