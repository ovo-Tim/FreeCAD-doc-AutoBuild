69 def parse(inputstring):
70  "parse(inputstring): returns a parsed output string"
71  print("preprocessing...")
72  Path.Log.track(inputstring)
73  # split the input by line
74  lines = inputstring.split("\n")
75  output = []
76  lastcommand = None
77 
78  for lin in lines:
79  # remove any leftover trailing and preceding spaces
80  lin = lin.strip()
81  if not lin:
82  # discard empty lines
83  continue
84  if lin[0].upper() in ["N"]:
85  # remove line numbers
86  lin = lin.split(" ", 1)
87  if len(lin) >= 1:
88  lin = lin[1].strip()
89  else:
90  continue
91 
92  if lin[0] in ["(", "%", "#", ";"]:
93  # discard comment and other non strictly gcode lines
94  continue
95  if lin[0].upper() in ["G", "M"]:
96  # found a G or M command: we store it
97  output.append(Path.Command(str(lin)))
98  last = lin[0].upper()
99  for c in lin[1:]:
100  if not c.isdigit():
101  break
102  else:
103  last += c
104  lastcommand = last
105  elif lastcommand:
106  # no G or M command: we repeat the last one
107  output.append(Path.Command(str(lastcommand + " " + lin)))
108 
109  print("done preprocessing.")
110  return output
