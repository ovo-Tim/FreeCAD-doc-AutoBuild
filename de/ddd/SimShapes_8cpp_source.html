36 void Shape::GenerateSinTable(int nSlices)
37 {
38  if (nSlices == lastNumSlices) {
39  return;
40  }
41 
42  float slice = (float)(2 * PI / nSlices);
43  int nvals = nSlices + 1;
44  sinTable.resize(nvals);
45  cosTable.resize(nvals);
46  for (int i = 0; i < nvals; i++) {
47  sinTable[i] = sinf(slice * i);
48  cosTable[i] = cosf(slice * i);
49  }
50  lastNumSlices = nvals;
51 }

54 void Shape::RotateProfile(float* profPoints,
55  int nPoints,
56  float distance,
57  float /* deltaHeight */,
58  int nSlices,
59  bool isHalfTurn)
60 {
61  int vidx = 0;
62  int iidx = 0;
63  int numVerts, numIndices;
64  int vstart;
65 
66  numVerts = nPoints * 2 * (nSlices + 1);
67  numIndices = (nPoints - 1) * nSlices * 6;
68 
69  std::vector<Vertex> vbuffer(numVerts);
70  std::vector<GLushort> ibuffer(numIndices);
71  int nsinvals = nSlices;
72  if (isHalfTurn) {
73  nsinvals *= 2;
74  }
75  GenerateSinTable(nsinvals);
76 
77  for (int i = 0; i < nPoints; i++) {
78  int i2 = i * 2;
79 
80  float prevy = i > 0 ? profPoints[i2 - 2] : 0;
81  float prevz = i > 0 ? profPoints[i2 - 1] : profPoints[i2 + 1];
82  float prevrad = fabsf(prevy);
83  float rad = fabsf(profPoints[i2]);
84  float z2 = profPoints[i2 + 1];
85  float diffy = profPoints[i2] - prevy;
86  float diffz = z2 - prevz;
87  float len = sqrtf(diffy * diffy + diffz * diffz);
88  float nz = diffy / len;
89  vstart = i * 2 * (nSlices + 1);
90 
91  for (int j = 0; j <= nSlices; j++) {
92  // generate vertices
93  float sx = sinTable[j];
94  float sy = cosTable[j];
95  float x1 = prevrad * sx + distance;
96  float y1 = prevrad * sy;
97  float x2 = rad * sx + distance;
98  float y2 = rad * sy;
99 
100  // generate normals
101  float ny = -diffz / len;
102  float nx = ny * sx;
103  ny *= sy;
104 
105  vbuffer[vidx++] = {x1, y1, prevz, nx, ny, nz};
106  vbuffer[vidx++] = {x2, y2, z2, nx, ny, nz};
107 
108  if (j != nSlices) {
109  // generate indices { 0, 3, 1, 0, 2, 3 }
110  int pos = vstart + 2 * j;
111  if (i < (nPoints - 1)) {
112  SET_TRIPLE(ibuffer, iidx, pos, pos + 3, pos + 1);
113  }
114  if (i > 0) {
115  SET_TRIPLE(ibuffer, iidx, pos, pos + 2, pos + 3);
116  }
117  }
118  }
119  }
120 
121  SetModelData(vbuffer, ibuffer);
122 }

150 void Shape::ExtrudeProfileRadial(float* profPoints,
151  int nPoints,
152  float radius,
153  float angleRad,
154  float deltaHeight,
155  bool capStart,
156  bool capEnd)
157 {
158  int vidx = 0, vc1idx, vc2idx;
159  int iidx = 0, ic1idx, ic2idx;
160  int numVerts, numIndices;
161 
162  CalculateExtrudeBufferSizes(nPoints,
163  capStart,
164  capEnd,
165  &numVerts,
166  &numIndices,
167  &vc1idx,
168  &vc2idx,
169  &ic1idx,
170  &ic2idx);
171  int vc1start = vc1idx;
172  int vc2start = vc2idx;
173 
174  std::vector<Vertex> vbuffer(numVerts);
175  std::vector<GLushort> ibuffer(numIndices);
176 
177  bool is_clockwise = angleRad > 0;
178  angleRad = (float)fabs(angleRad);
179  float dir = is_clockwise ? 1.0f : -1.0f;
180  int offs1 = is_clockwise ? -1 : 0;
181  int offs2 = is_clockwise ? 0 : -1;
182 
183  float cosAng = cosf(angleRad);
184  float sinAng = sinf(angleRad);
185  for (int i = 0; i < nPoints; i++) {
186  int p1 = i * 2;
187  float y1 = profPoints[p1] + radius;
188  float z1 = profPoints[p1 + 1];
189  int p2 = (p1 + 2) % (nPoints * 2);
190  float y2 = profPoints[p2] + radius;
191  float z2 = profPoints[p2 + 1];
192 
193  // normals
194  float ydiff = y2 - y1;
195  float zdiff = z2 - z1;
196  float len = sqrtf(ydiff * ydiff + zdiff * zdiff);
197  float ny = -zdiff / len;
198  float nz = ydiff / len;
199  float nx = -sinAng * ny;
200  ny *= cosAng;
201 
202  // start verts
203  vbuffer[vidx++] = {0, y1, z1, nx, ny, nz};
204  vbuffer[vidx++] = {0, y2, z2, nx, ny, nz};
205 
206  if (capStart) {
207  vbuffer[vc1idx++] = {0, y1, z1, -1 * dir, 0, 0};
208  if (i > 1) {
209  SET_TRIPLE(ibuffer, ic1idx, vc1start, vc1start + i + offs1, vc1start + i + offs2);
210  }
211  }
212 
213  float x1 = y1 * sinAng * dir;
214  float x2 = y2 * sinAng * dir;
215  y1 *= cosAng;
216  y2 *= cosAng;
217  z1 += deltaHeight;
218  z2 += deltaHeight;
219  vbuffer[vidx++] = {x1, y1, z1, nx, ny, nz};
220  vbuffer[vidx++] = {x2, y2, z2, nx, ny, nz};
221 
222  // face have 2 triangles { 0, 2, 3, 0, 3, 1 };
223  GLushort vistart = i * 4;
224  if (is_clockwise) {
225  SET_TRIPLE(ibuffer, iidx, vistart, vistart + 2, vistart + 3);
226  SET_TRIPLE(ibuffer, iidx, vistart, vistart + 3, vistart + 1);
227  }
228  else {
229  SET_TRIPLE(ibuffer, iidx, vistart, vistart + 3, vistart + 2);
230  SET_TRIPLE(ibuffer, iidx, vistart, vistart + 1, vistart + 3);
231  }
232 
233  if (capEnd) {
234  vbuffer[vc2idx++] = {x1, y1, z1, cosAng * dir, -sinAng, 0};
235  if (i > 1) {
236  SET_TRIPLE(ibuffer, ic2idx, vc2start, vc2start + i + offs2, vc2start + i + offs1);
237  }
238  }
239  }
240 
241  SetModelData(vbuffer, ibuffer);
242 }
