101 std::shared_ptr<Material2DArray> MaterialYamlEntry::read2DArray(const YAML::Node& node, int columns)
102 {
103  auto array2d = std::make_shared<Material2DArray>();
104  array2d->setColumns(columns);
105 
106  if (node.size() == 1 || node.size() == 2) {
107  // There used to be a default value. Ignore it.
108  auto yamlArray = node[0];
109  if (node.size() == 2) {
110  yamlArray = node[1];
111  }
112 
113  for (std::size_t i = 0; i < yamlArray.size(); i++) {
114  auto yamlRow = yamlArray[i];
115 
116  auto row = std::make_shared<QList<QVariant>>();
117  for (std::size_t j = 0; j < yamlRow.size(); j++) {
118  Base::Quantity qq = Base::Quantity::parse(yamlRow[j].as<std::string>());
119  qq.setFormat(MaterialValue::getQuantityFormat());
120  row->push_back(QVariant::fromValue(qq));
121  }
122  array2d->addRow(row);
123  }
124  }
125 
126  return array2d;
127 }

129 std::shared_ptr<Material3DArray> MaterialYamlEntry::read3DArray(const YAML::Node& node, int columns)
130 {
131  auto array3d = std::make_shared<Material3DArray>();
132  array3d->setColumns(columns - 1); // First column is third dimension
133 
134  if (node.size() == 1 || node.size() == 2) {
135  // There used to be a default value. Ignore it.
136  auto yamlArray = node[0];
137  if (node.size() == 2) {
138  yamlArray = node[1];
139  }
140 
141  for (std::size_t depth = 0; depth < yamlArray.size(); depth++) {
142  auto yamlDepth = yamlArray[depth];
143  for (auto it = yamlDepth.begin(); it != yamlDepth.end(); it++) {
144  auto depthValue = Base::Quantity::parse(it->first.as<std::string>());
145  depthValue.setFormat(MaterialValue::getQuantityFormat());
146  array3d->addDepth(depth, depthValue);
147

150  auto yamlRow = yamlTable[i];

167 void MaterialYamlEntry::addToTree(
168  std::shared_ptr<std::map<QString, std::shared_ptr<Material>>> materialMap)
169 {
170  std::set<QString> exclude;
171  exclude.insert(QString::fromStdString("General"));
172  exclude.insert(QString::fromStdString("Inherits"));
173 
174  auto yamlModel = getModel();
175  auto library = getLibrary();
176  auto name = getName();
177  auto directory = getDirectory();
178  QString uuid = getUUID();
179 
180  QString author = yamlValue(yamlModel["General"], "Author", "");
181  QString license = yamlValue(yamlModel["General"], "License", "");
182  QString description = yamlValue(yamlModel["General"], "Description", "");
183 
184  std::shared_ptr<Material> finalModel =
185  std::make_shared<Material>(library, directory, uuid, name);
186  finalModel->setAuthor(author);
187  finalModel->setLicense(license);
188  finalModel->setDescription(description);
189 
190  // Add inheritance list
191  if (yamlModel["Inherits"]) {
192  auto inherits = yamlModel["Inherits"];
193  for (auto it = inherits.begin(); it != inherits.end(); it++) {
194  auto nodeName = it->second["UUID"].as<std::string>();
195 
196  finalModel->setParentUUID(
197  QString::fromStdString(nodeName)); // Should only be one. Need to check
198  }
199  }
200 
201  // Add material models
202  if (yamlModel["Models"]) {
203  auto models = yamlModel["Models"];
204  for (auto it = models.begin(); it != models.end(); it++) {
205  auto modelName = (it->first).as<std::string>();
206 
207  // Add the model uuid
208  auto modelNode = models[modelName];
209  auto modelUUID = modelNode["UUID"].as<std::string>();
210  finalModel->addPhysical(QString::fromStdString(modelUUID));
211 
212  // Add the property values
213  auto properties = yamlModel["Models"][modelName];
214  for (auto itp = properties.begin(); itp != properties.end(); itp++) {
215  auto propertyName = (itp->first).as<std::string>();
216  if (finalModel->hasPhysicalProperty(QString::fromStdString(propertyName))) {
217  auto prop =
218  finalModel->getPhysicalProperty(QString::fromStdString(propertyName));
219  auto type = prop->getType();
220 
221  try {
222  if (type == MaterialValue::List || type == MaterialValue::FileList) {
223  auto list = readList(itp->second);
224  finalModel->setPhysicalValue(QString::fromStdString(propertyName),
225  list);
226  }
227  else if (type == MaterialValue::ImageList) {
228  auto list = readImageList(itp->second);
229  finalModel->setPhysicalValue(QString::fromStdString(propertyName),
230  list);
231  }
232  else if (type == MaterialValue::Array2D) {
233  auto array2d = read2DArray(itp->second, prop->columns());
234  finalModel->setPhysicalValue(QString::fromStdString(propertyName),
235  array2d);
236  }
237  else if (type == MaterialValue::Array3D) {
238  auto array3d = read3DArray(itp->second, prop->columns());
239  finalModel->setPhysicalValue(QString::fromStdString(propertyName),
240  array3d);
241  }
242  else {
243  QString propertyValue =
244  QString::fromStdString((itp->second).as<std::string>());
245  if (type == MaterialValue::Image) {
246  propertyValue = propertyValue.remove(
247  QRegularExpression(QString::fromStdString("[\r\n]")));
248  }
249  finalModel->setPhysicalValue(QString::fromStdString(propertyName),
250  propertyValue);
251  }
252  }
253  catch (const YAML::BadConversion& e) {
254  Base::Console().Log("Exception %s <%s:%s> - ignored\n",
255  e.what(),
256  name.toStdString().c_str(),
257  propertyName.c_str());
258  }
259  }
260  else if (propertyName != "UUID") {
261  Base::Console().Log("\tProperty '%s' is not described by any model. Ignored\n",
262  propertyName.c_str());
263  }
264  }
265  }
266  }
267 
268  // Add appearance models
269  if (yamlModel["AppearanceModels"]) {
270  auto models = yamlModel["AppearanceModels"];
271  for (auto it = models.begin(); it != models.end(); it++) {
272  auto modelName = (it->first).as<std::string>();
273 
274  // Add the model uuid
275  auto modelNode = models[modelName];
276  auto modelUUID = modelNode["UUID"].as<std::string>();
277  finalModel->addAppearance(QString::fromStdString(modelUUID));
278 
279  // Add the property values
280  auto properties = yamlModel["AppearanceModels"][modelName];
281  for (auto itp = properties.begin(); itp != properties.end(); itp++) {
282  auto propertyName = (itp->first).as<std::string>();
283  if (finalModel->hasAppearanceProperty(QString::fromStdString(propertyName))) {
284  auto prop =
285  finalModel->getAppearanceProperty(QString::fromStdString(propertyName));
286  auto type = prop->getType();
287 
288  try {
289  if (type == MaterialValue::List || type == MaterialValue::FileList) {
290  auto list = readList(itp->second);
291  finalModel->setAppearanceValue(QString::fromStdString(propertyName),
292  list);
293  }
294  else if (type == MaterialValue::ImageList) {
295  auto list = readImageList(itp->second);
296  finalModel->setAppearanceValue(QString::fromStdString(propertyName),
297  list);
298  }
299  else if (type == MaterialValue::Array2D) {
300  auto array2d = read2DArray(itp->second, prop->columns());
301  finalModel->setAppearanceValue(QString::fromStdString(propertyName),
302  array2d);
303  }
304  else if (type == MaterialValue::Array3D) {
305  auto array3d = read3DArray(itp->second, prop->columns());
306  finalModel->setAppearanceValue(QString::fromStdString(propertyName),
307  array3d);
308  }
309  else {
310  QString propertyValue =
311  QString::fromStdString((itp->second).as<std::string>());
312  if (type == MaterialValue::Image) {
313  propertyValue = propertyValue.remove(
314  QRegularExpression(QString::fromStdString("[\r\n]")));
315  }
316  finalModel->setAppearanceValue(QString::fromStdString(propertyName),
317  propertyValue);
318  }
319  }
320  catch (const YAML::BadConversion& e) {
321  Base::Console().Log("Exception %s <%s:%s> - ignored\n",
322  e.what(),
323  name.toStdString().c_str(),
324  propertyName.c_str());
325  }
326  }
327  else if (propertyName != "UUID") {
328  Base::Console().Log("\tProperty '%s' is not described by any model. Ignored\n",
329  propertyName.c_str());
330  }
331  }
332  }
333  }
334 
335  QString path = QDir(directory).absolutePath();
336  (*materialMap)[uuid] = library->addMaterial(finalModel, path);
337 }

389 std::shared_ptr<MaterialEntry>
390 MaterialLoader::getMaterialFromPath(const std::shared_ptr<MaterialLibrary>& library,
391  const QString& path) const
392 {
393  std::shared_ptr<MaterialEntry> model = nullptr;
394 
395  // Used for debugging
396  std::string pathName = path.toStdString();
397 
398  if (MaterialConfigLoader::isConfigStyle(path)) {
399  auto material = MaterialConfigLoader::getMaterialFromPath(library, path);
400  if (material) {
401  (*_materialMap)[material->getUUID()] = library->addMaterial(material, path);
402  }
403 
404  // Return the nullptr as there are no intermediate steps to take, such
405  // as checking inheritance
406  return model;
407  }
408 
409  Base::FileInfo info(pathName);
410  Base::ifstream fin(info);
411  if (!fin) {
412  Base::Console().Error("YAML file open error: '%s'\n", pathName.c_str());
413  return model;
414  }
415 
416  YAML::Node yamlroot;
417  try {
418  yamlroot = YAML::Load(fin);
419 
420  model = getMaterialFromYAML(library, yamlroot, path);
421  }
422  catch (YAML::Exception const& e) {
423  Base::Console().Error("YAML parsing error: '%s'\n", pathName.c_str());
424  Base::Console().Error("\t'%s'\n", e.what());
425  showYaml(yamlroot);
426  }
427 
428 
429  return model;
430 }

442 void MaterialLoader::dereference(
443  const std::shared_ptr<std::map<QString, std::shared_ptr<Material>>>& materialMap,
444  const std::shared_ptr<Material>& material)
445 {
446  // Avoid recursion
447  if (material->getDereferenced()) {
448  return;
449  }
450 
451  auto parentUUID = material->getParentUUID();
452  if (parentUUID.size() > 0) {
453  std::shared_ptr<Material> parent;
454  try {
455  parent = materialMap->at(parentUUID);
456  }
457  catch (std::out_of_range&) {
458  Base::Console().Log(
459  "Unable to apply inheritance for material '%s', parent '%s' not found.\n",
460  material->getName().toStdString().c_str(),
461  parentUUID.toStdString().c_str());
462  return;
463  }
464 
465  // Ensure the parent has been dereferenced
466  dereference(materialMap, parent);
467 
468  // Add physical models
469  auto modelVector = parent->getPhysicalModels();
470  for (auto& model : *modelVector) {
471  if (!material->hasPhysicalModel(model)) {
472  material->addPhysical(model);
473  }
474  }
475 
476  // Add appearance models
477  modelVector = parent->getAppearanceModels();
478  for (auto& model : *modelVector) {
479  if (!material->hasAppearanceModel(model)) {
480  material->addAppearance(model);
481  }
482  }
483 
484  // Add values
485  auto properties = parent->getPhysicalProperties();
486  for (auto& itp : properties) {
487  auto name = itp.first;
488  auto property = itp.second;
489 
490  if (material->getPhysicalProperty(name)->isNull()) {
491  material->getPhysicalProperty(name)->setValue(property->getValue());
492  }
493  }
494 
495  properties = parent->getAppearanceProperties();
496  for (auto& itp : properties) {
497  auto name = itp.first;
498  auto property = itp.second;
499 
500  if (material->getAppearanceProperty(name)->isNull()) {
501  material->getAppearanceProperty(name)->setValue(property->getValue());
502  }
503  }
504  }
505 
506  material->markDereferenced();
507 }

514 void MaterialLoader::loadLibrary(const std::shared_ptr<MaterialLibrary>& library)
515 {
516  if (_materialEntryMap == nullptr) {
517  _materialEntryMap = std::make_unique<std::map<QString, std::shared_ptr<MaterialEntry>>>();
518  }
519 
520  QDirIterator it(library->getDirectory(), QDirIterator::Subdirectories);
521  while (it.hasNext()) {
522  auto pathname = it.next();
523  QFileInfo file(pathname);
524  if (file.isFile()) {
525  if (file.suffix().toStdString() == "FCMat") {
526  try {
527  auto model = getMaterialFromPath(library, file.canonicalFilePath());
528  if (model) {
529  (*_materialEntryMap)[model->getUUID()] = model;
530  }
531  }
532  catch (const MaterialReadError&) {
533  // Ignore the file. Error messages should have already been logged
534  }
535  }
536  }
537  }
538 
539  for (auto& it : *_materialEntryMap) {
540  it.second->addToTree(_materialMap);
541  }
542 }
