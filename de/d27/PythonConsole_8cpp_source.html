259 bool InteractiveInterpreter::runSource(const char* source) const
260 {
261  Base::PyGILStateLocker lock;
262  PyObject* code;
263  try {
264  code = compile(source);
265  } catch (const Base::Exception&) {
266  // A system, overflow or value error was raised.
267  // We clear the traceback info as this might be a longly
268  // message we don't need.
269  PyObject *errobj, *errdata, *errtraceback;
270  PyErr_Fetch(&errobj, &errdata, &errtraceback);
271  PyErr_Restore(errobj, errdata, nullptr);
272  // print error message
273  if (PyErr_Occurred()) PyErr_Print();
274  return false;
275  }
276 
277  // the command is incomplete
278  if (PyObject_TypeCheck(Py_None, code->ob_type)) {
279  Py_DECREF(code);
280  return true;
281  }
282 
283  // run the code and return false
284  runCode((PyCodeObject*)code);
285  return false;

298 /* Execute a code object.
299  *
300  * When an exception occurs, a traceback is displayed.
301  * All exceptions are caught except SystemExit, which is reraised.
302  */
303 void InteractiveInterpreter::runCode(PyCodeObject* code) const
304 {
305  if (isOccupied()) {
306  return;
307  }
308 
309  d->trace.fetchFromSettings();
310  PythonTracingLocker tracelock(d->trace);
311 
312  Base::PyGILStateLocker lock;
313  PyObject *module, *dict, *presult; /* "exec code in d, d" */
314  module = PyImport_AddModule("__main__"); /* get module, init python */
315  if (!module) {
316  throw Base::PyException(); /* not incref'd */
317  }
318  dict = PyModule_GetDict(module); /* get dict namespace */
319  if (!dict) {
320  throw Base::PyException(); /* not incref'd */
321  }
322 
323  // It seems that the return value is always 'None' or Null
324  presult = PyEval_EvalCode((PyObject*)code, dict, dict); /* run compiled bytecode */
325  Py_XDECREF(code); /* decref the code object */
326  if (!presult) {
327  if (PyErr_ExceptionMatches(PyExc_SystemExit)) {
328  // throw SystemExit exception
329  throw Base::SystemExitException();
330  }
331  if (PyErr_Occurred()) { /* get latest python exception information */
332  PyObject *errobj, *errdata, *errtraceback;
333  PyErr_Fetch(&errobj, &errdata, &errtraceback);
334  // the error message can be empty so errdata will be null
335  if (errdata && PyDict_Check(errdata)) {
336  PyObject* value = PyDict_GetItemString(errdata, "swhat");
337  if (value) {
338  Base::RuntimeError e;
339  e.setPyObject(errdata);
340  Py_DECREF(errdata);
341 
342  std::stringstream str;
343  str << e.what();
344  if (!e.getFunction().empty()) {
345  str << " In " << e.getFunction();
346  }
347  if (!e.getFile().empty() && e.getLine() > 0) {
348  std::string file = e.getFile();
349  std::size_t pos = file.find("src");
350  if (pos!=std::string::npos) {
351  file = file.substr(pos);
352  }
353  str << " in " << file << ":" << e.getLine();
354  }
355 
356  std::string err = str.str();
357  errdata = PyUnicode_FromString(err.c_str());
358  }
359  }
360  PyErr_Restore(errobj, errdata, errtraceback);
361  PyErr_Print(); /* and print the error to the error output */
362  }
363  } else {
364  Py_DECREF(presult);
365  }
366 }

543 void PythonConsole::keyPressEvent(QKeyEvent * e)
544 {
545  bool restartHistory = true;
546  QTextCursor cursor = this->textCursor();
547  QTextCursor inputLineBegin = this->inputBegin();
548 
549  if (e->key() == Qt::Key_C && e->modifiers() == Qt::ControlModifier) {
550  if (d->interpreter->interrupt()) {
551  return;
552  }
553  }
554 
555  if (!cursorBeyond( cursor, inputLineBegin ))
556  {
565  switch (e->key())
566  {
567  case Qt::Key_Return:
568  case Qt::Key_Enter:
569  case Qt::Key_Escape:
570  case Qt::Key_Backspace:
571  this->moveCursor( QTextCursor::End );
572  break;
573 
574  default:
575  if (e->text().isEmpty() ||
576  e->matches(QKeySequence::Copy) ||
577  e->matches(QKeySequence::SelectAll)) {
578  PythonTextEditor::keyPressEvent(e);
579  }
580  else if (!e->text().isEmpty() &&
581  (e->modifiers() == Qt::NoModifier ||
582  e->modifiers() == Qt::ShiftModifier)) {
583  this->moveCursor(QTextCursor::End);
584  PythonTextEditor::keyPressEvent(e);
585  }
586  break;
587  }
588  }
589  else
590  {
598  QTextBlock inputBlock = inputLineBegin.block(); //< get the last paragraph's text
599  QString inputLine = inputBlock.text();
600  QString inputStrg = stripPromptFrom( inputLine );
601  if (this->_sourceDrain && !this->_sourceDrain->isEmpty()) {
602  inputStrg = inputLine.mid(this->_sourceDrain->length());
603  }
604 
605  switch (e->key())
606  {
607  case Qt::Key_Escape:
608  {
609  // disable current input string - i.e. put it to history but don't execute it.
610  if (!inputStrg.isEmpty())
611  {
612  d->history.append( QLatin1String("# ") + inputStrg ); //< put commented string to history ...
613  inputLineBegin.insertText( QString::fromLatin1("# ") ); //< and comment it on console
614  setTextCursor( inputLineBegin );
615  printPrompt(d->interpreter->hasPendingInput() //< print adequate prompt
616  ? PythonConsole::Incomplete
617  : PythonConsole::Complete);
618  }
619  } break;
620 
621  case Qt::Key_Return:
622  case Qt::Key_Enter:
623  {
624  d->history.append( inputStrg ); //< put statement to history
625  runSource( inputStrg ); //< commit input string
626  } break;
627 
628  case Qt::Key_Period:
629  {
630  // In Qt 4.8 there is a strange behaviour because when pressing ":"
631  // then key is also set to 'Period' instead of 'Colon'. So we have
632  // to make sure we only handle the period.
633  if (e->text() == QLatin1String(".")) {
634  // analyse context and show available call tips
635  int contextLength = cursor.position() - inputLineBegin.position();
636  PythonTextEditor::keyPressEvent(e);
637  d->callTipsList->showTips( inputStrg.left( contextLength ) );
638  }
639  else {
640  PythonTextEditor::keyPressEvent(e);
641  }
642  } break;
643 
644  case Qt::Key_Home:
645  {
646  QTextCursor::MoveMode mode = (e->modifiers() & Qt::ShiftModifier)? QTextCursor::KeepAnchor
647  /* else */ : QTextCursor::MoveAnchor;
648  cursor.setPosition( inputLineBegin.position(), mode );
649  setTextCursor( cursor );
650  ensureCursorVisible();
651  } break;
652 
653  case Qt::Key_Up:
654  {
655  // if possible, move back in history
656  if (d->history.prev( inputStrg ))
657  { overrideCursor( d->history.value() ); }
658  restartHistory = false;
659  } break;
660 
661  case Qt::Key_Down:
662  {
663  // if possible, move forward in history
664  if (d->history.next())
665  { overrideCursor( d->history.value() ); }
666  restartHistory = false;
667  } break;
668 
669  case Qt::Key_Left:
670  {
671  if (cursor > inputLineBegin)
672  { PythonTextEditor::keyPressEvent(e); }
673  restartHistory = false;
674  } break;
675 
676  case Qt::Key_Right:
677  {
678  PythonTextEditor::keyPressEvent(e);
679  restartHistory = false;
680  } break;
681 
682  case Qt::Key_Backspace:
683  {
684  if (cursorBeyond( cursor, inputLineBegin, +1 ))
685  { PythonTextEditor::keyPressEvent(e); }
686  } break;
687 
688  default:
689  {
690  PythonTextEditor::keyPressEvent(e);
691  } break;
692  }
693  // This can't be done in CallTipsList::eventFilter() because we must first perform
694  // the event and afterwards update the list widget
695  if (d->callTipsList->isVisible())
696  { d->callTipsList->validateCursor(); }
697 
698  // disable history restart if input line changed
699  restartHistory &= (inputLine != inputBlock.text());
700  }
701  // any cursor move resets the history to its latest item.
702  if (restartHistory) {
703  d->history.restart();
704  }
705 }

754  }

758

762  cursor.movePosition(QTextCursor::End);

766  // In this case, however, we should not add a new text block. We force
767  // the current block to be normal text (user state = 0) to be highlighted

769  if (block.length() > 1)

778  break;
779  case PythonConsole::Complete:
780  cursor.insertText(QString::fromLatin1(">>> "));
781  break;
782  default:
783  break;
784  }

786  }

1208 void PythonConsole::runSourceFromMimeData(const QString& source)
1209 {
1210  // When inserting a big text block we must break it down into several command
1211  // blocks instead of processing the text block as a whole or each single line.
1212  // If we processed the complete block as a whole only the first valid Python
1213  // command would be executed and the rest would be ignored. However, if we
1214  // processed each line separately the interpreter might be confused that a block
1215  // is complete but it might be not. This is for instance, if a class or method
1216  // definition contains several empty lines which leads to error messages (almost
1217  // indentation errors) later on.
1218  QString text = source;
1219  if (text.isNull()) {
1220  return;
1221  }
1222 
1223 #if defined (Q_OS_LINUX)
1224  // Need to convert CRLF to LF
1225  text.replace(QLatin1String("\r\n"), QLatin1String("\n"));
1226 #elif defined(Q_OS_WIN32)
1227  // Need to convert CRLF to LF
1228  text.replace(QLatin1String("\r\n"), QLatin1String("\n"));
1229 #elif defined(Q_OS_MACOS)
1230  //need to convert CR to LF
1231  text.replace(QLatin1Char('\r'), QLatin1Char('\n'));
1232 #endif
1233 
1234  // separate the lines and get the last one
1235  QStringList lines = text.split(QLatin1Char('\n'));
1236  QString last = lines.back();
1237  lines.pop_back();
1238 
1239  QTextCursor cursor = textCursor();
1240  QStringList buffer = d->interpreter->getBuffer();
1241  d->interpreter->clearBuffer();
1242 
1243  int countNewlines = lines.count(), i = 0;
1244  for (QStringList::Iterator it = lines.begin(); it != lines.end(); ++it, ++i) {
1245  QString line = *it;
1246 
1247  // insert the text to the current cursor position
1248  cursor.insertText(*it);
1249 
1250  // for the very first line get the complete block
1251  // because it may differ from the inserted text
1252  if (i == 0) {
1253  // get the text from the current cursor position to the end, remove it
1254  // and add it to the last line
1255  cursor.movePosition(QTextCursor::End, QTextCursor::KeepAnchor);
1256  QString select = cursor.selectedText();
1257  cursor.removeSelectedText();
1258  last = last + select;
1259  line = stripPromptFrom( cursor.block().text() );
1260  }
1261 
1262  // put statement to the history
1263  d->history.append(line);
1264 
1265  buffer.append(line);
1266  int ret = d->interpreter->compileCommand(buffer.join(QLatin1String("\n")).toUtf8());
1267  if (ret == 1) { // incomplete
1268  printPrompt(PythonConsole::Incomplete);
1269  }
1270  else if (ret == 0) { // complete
1271  // check if the following lines belong to the previous block
1272  int k=i+1;
1273  QString nextline;
1274  while ((nextline.isEmpty() || isComment(nextline)) && k < countNewlines) {
1275  nextline = lines[k];
1276  k++;
1277  }
1278 
1279  int ret = d->interpreter->compileCommand(nextline.toUtf8());
1280 
1281  // If the line is valid, i.e. complete or incomplete the previous block
1282  // is finished
1283  if (ret == -1) {
1284  // the command is not finished yet
1285  printPrompt(PythonConsole::Incomplete);
1286  }
1287  else {
1288  runSource(buffer.join(QLatin1String("\n")));
1289  buffer.clear();
1290  }
1291  }
1292  else { // invalid
1293  runSource(buffer.join(QLatin1String("\n")));
1294  ensureCursorVisible();
1295  return; // exit the method on error
1296  }
1297  }
1298 
1299  // set the incomplete block to the interpreter and insert the last line
1300  d->interpreter->setBuffer(buffer);
1301  cursor.insertText(last);
1302  ensureCursorVisible();
1303 }
