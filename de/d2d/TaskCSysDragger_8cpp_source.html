59 void alignGridLayoutColumns(const std::list<QGridLayout*>& layouts, unsigned column = 0)
60 {
61  std::vector<int> widths;
62 
63  auto getActualWidth = [&](const QGridLayout* layout) -> int {
64  if (auto const item = layout->itemAtPosition(0, column)) {
65  return item->geometry().width();
66  }
67 
68  return 0;
69  };
70 
71  for (const auto layout : layouts) {
72  widths.push_back(getActualWidth(layout));
73  }
74 
75  const auto maxWidth = *std::max_element(widths.begin(), widths.end());
76  for (const auto layout : layouts) {
77  layout->setColumnMinimumWidth(column, maxWidth);
78  }
79 }

134 void TaskTransform::dragMotionCallback(void* data, [[maybe_unused]] SoDragger* dragger)
135 {
136  auto task = static_cast<TaskTransform*>(data);
137 
138  const auto currentRotation = task->vp->getOriginalDraggerPlacement().getRotation();
139  const auto updatedRotation = task->vp->getDraggerPlacement().getRotation();
140 
141  const auto rotationAxisHasChanged = [task](auto first, auto second) {
142  double alpha, beta, gamma;
143 
144  (first.inverse() * second).getEulerAngles(task->eulerSequence(), alpha, beta, gamma);
145 
146  auto angles = {alpha, beta, gamma};
147  const int changed = std::count_if(angles.begin(), angles.end(), [](double angle) {
148  return std::fabs(angle) > tolerance;
149  });
150 
151  // if representation of both differs by more than one axis the axis of rotation must be
152  // different
153  return changed > 1;
154  };
155 
156  if (!updatedRotation.isSame(currentRotation, tolerance)) {
157  task->resetReferencePlacement();
158 
159  if (rotationAxisHasChanged(task->referenceRotation, updatedRotation)) {
160  task->referenceRotation = currentRotation;
161  }
162  }
163 
164  task->updatePositionAndRotationUi();
165 }

300 void TaskTransform::updatePositionAndRotationUi() const
301 {
302  const auto referencePlacement = currentCoordinateSystem().origin;
303 
304  const auto xyzPlacement = vp->getDraggerPlacement();
305  const auto uvwPlacement = referencePlacement.inverse() * xyzPlacement;
306 
307  auto fixNegativeZero = [](const double value) {
308  return std::fabs(value) < Base::Precision::Confusion() ? 0.0 : value;
309  };
310 
311  auto setPositionValues = [&](const Base::Vector3d& vec, auto* x, auto* y, auto* z) {
312  [[maybe_unused]]
313  auto blockers = {QSignalBlocker(x), QSignalBlocker(y), QSignalBlocker(z)};
314 
315  x->setValue(fixNegativeZero(vec.x));
316  y->setValue(fixNegativeZero(vec.y));
317  z->setValue(fixNegativeZero(vec.z));
318  };
319 
320  auto setRotationValues = [&](const Base::Rotation& rot, auto* x, auto* y, auto* z) {
321  [[maybe_unused]]
322  auto blockers = {QSignalBlocker(x), QSignalBlocker(y), QSignalBlocker(z)};
323 
324  double alpha, beta, gamma;
325  rot.getEulerAngles(eulerSequence(), alpha, beta, gamma);
326 
327  x->setValue(fixNegativeZero(alpha));
328  y->setValue(fixNegativeZero(beta));
329  z->setValue(fixNegativeZero(gamma));
330  };
331 
332  setPositionValues(uvwPlacement.getPosition(),
333  ui->xPositionSpinBox,
334  ui->yPositionSpinBox,
335  ui->zPositionSpinBox);
336 
337  setRotationValues(positionMode == PositionMode::Local
338  ? referenceRotation.inverse() * xyzPlacement.getRotation()
339  : uvwPlacement.getRotation(),
340  ui->xRotationSpinBox,
341  ui->yRotationSpinBox,
342  ui->zRotationSpinBox);
343 }

448 void TaskTransform::onSelectionChanged(const SelectionChanges& msg)
449 {
450  const auto isSupportedMessage =
451  msg.Type == SelectionChanges::AddSelection || msg.Type == SelectionChanges::SetPreselect;
452 
453  if (!isSupportedMessage) {
454  return;
455  }
456 
457  if (!subObjectPlacementProvider) {
458  return;
459  }
460 
461  if (!msg.pOriginalMsg) {
462  // this should not happen! Original should contain unresolved message.
463  return;
464  }
465 
466  auto doc = Application::Instance->getDocument(msg.pDocName);
467  auto obj = doc->getDocument()->getObject(msg.pObjectName);
468 
469  auto orgDoc = Application::Instance->getDocument(msg.pOriginalMsg->pDocName);
470  auto orgObj = orgDoc->getDocument()->getObject(msg.pOriginalMsg->pObjectName);
471 
472  auto globalPlacement = App::GeoFeature::getGlobalPlacement(obj, orgObj, msg.pOriginalMsg->pSubName);
473  auto localPlacement = App::GeoFeature::getPlacementFromProp(obj, "Placement");
474  auto rootPlacement = App::GeoFeature::getGlobalPlacement(vp->getObject());
475  auto attachedPlacement = subObjectPlacementProvider->calculate(msg.Object, localPlacement);
476 
477  auto selectedObjectPlacement = rootPlacement.inverse() * globalPlacement * attachedPlacement;
478 
479  auto label = QStringLiteral("%1#%2.%3")
480  .arg(QLatin1String(msg.pOriginalMsg->pObjectName),
481  QLatin1String(msg.pObjectName),
482  QLatin1String(msg.pSubName));
483 
484  switch (selectionMode) {
485  case SelectionMode::SelectTransformOrigin: {
486  if (msg.Type == SelectionChanges::AddSelection) {
487  ui->referenceLineEdit->setText(label);
488  customTransformOrigin = selectedObjectPlacement;
489  updateTransformOrigin();
490  setSelectionMode(SelectionMode::None);
491  } else {
492  vp->setTransformOrigin(selectedObjectPlacement);
493  }
494 
495  break;
496  }
497 
498  case SelectionMode::SelectAlignTarget: {
499  vp->setDraggerPlacement(vp->getObjectPlacement() * selectedObjectPlacement);
500 
501  if (msg.Type == SelectionChanges::AddSelection) {
502  moveObjectToDragger();
503 
504  setSelectionMode(SelectionMode::None);
505  }
506 
507  break;
508  }
509 
510  default:
511  // no-op
512  break;
513  }
514 }

568 void TaskTransform::updateTransformOrigin()
569 {
570  auto getTransformOrigin = [this](const PlacementMode& mode) -> Base::Placement {
571  switch (mode) {
572  case PlacementMode::ObjectOrigin:
573  return {};
574  case PlacementMode::Centroid:
575  if (const auto com = centerOfMassProvider->ofDocumentObject(vp->getObject())) {
576  return {*com, {}};
577  }
578  return {};
579  case PlacementMode::Custom:
580  return customTransformOrigin.value_or(Base::Placement {});
581  default:
582  return {};
583  }
584  };
585 
586  ui->referencePickerWidget->setVisible(placementMode == PlacementMode::Custom);
587 
588  if (placementMode == PlacementMode::Custom && !customTransformOrigin.has_value()) {
589  setSelectionMode(SelectionMode::SelectTransformOrigin);
590  return;
591  }
592 
593  auto transformOrigin = getTransformOrigin(placementMode);
594  if (isDraggerAlignedToCoordinateSystem()) {
595  transformOrigin.setRotation(
596  (vp->getObjectPlacement().inverse() * globalCoordinateSystem().origin).getRotation());
597  }
598 
599  vp->setTransformOrigin(transformOrigin);
600 
601  resetReferencePlacement();
602  resetReferenceRotation();
603 
604  updatePositionAndRotationUi();
605  updateDraggerLabels();
606 }

674 void TaskTransform::onRotationChange(QuantitySpinBox* changed)
675 {
676  if (positionMode == PositionMode::Local) {
677  for (auto rotationSpinBox : {ui->xRotationSpinBox,
678  ui->yRotationSpinBox,
679  ui->zRotationSpinBox}) {
680  QSignalBlocker blocker(rotationSpinBox);
681 
682  // if any other spinbox contains non-zero value we need to reset rotation reference first
683  if (std::fabs(rotationSpinBox->rawValue()) > tolerance && rotationSpinBox != changed) {
684  resetReferenceRotation();
685  rotationSpinBox->setValue(0.0);
686  }
687  }
688  }
689 
690  const auto uvwRotation = Base::Rotation::fromEulerAngles(eulerSequence(),
691  ui->xRotationSpinBox->rawValue(),
692  ui->yRotationSpinBox->rawValue(),
693  ui->zRotationSpinBox->rawValue());
694 
695  auto referenceRotation = positionMode == PositionMode::Local
696  ? this->referenceRotation
697  : currentCoordinateSystem().origin.getRotation();
698 
699  const auto xyzRotation = referenceRotation * uvwRotation;
700 
701  const auto placement = vp->getDraggerPlacement();
702 
703  vp->setDraggerPlacement({placement.getPosition(), xyzRotation});
704 
705  vp->updateTransformFromDragger();
706  vp->updatePlacementFromDragger();
707 
708  resetReferencePlacement();
709 }
