1164  switch (mode) {
1165  case 1:
1166  sf = 1.0;
1167  break;
1168  case 2:
1169  sf = 0.625;
1170  break;
1171  case 3:
1172  sf = 1.6;
1173  break;
1174  default:
1175  throw Base::ValueError("Unsupported argument");
1176  }
1177  for (auto it : ThreadRunout) {
1178  double p = it[0];
1179  if (pitch <= p) {
1180  return sf * it[1];

1182  }

1241 {
1242  // Diameter parameter depends on Threaded, ThreadType, ThreadSize, and ThreadFit
1243 
1244  int threadType = ThreadType.getValue();
1245  int threadSize = ThreadSize.getValue();
1246  if (threadType < 0) {
1247  // When restoring the feature it might be in an inconsistent state.
1248  // So, just silently ignore it instead of throwing an exception.
1249  if (isRestoring())
1250  return std::nullopt;
1251  throw Base::IndexError("Thread type out of range");
1252  }
1253  if (threadSize < 0) {
1254  // When restoring the feature it might be in an inconsistent state.
1255  // So, just silently ignore it instead of throwing an exception.
1256  if (isRestoring())
1257  return std::nullopt;
1258  throw Base::IndexError("Thread size out of range");
1259  }
1260  double diameter = threadDescription[threadType][threadSize].diameter;
1261  double pitch = threadDescription[threadType][threadSize].pitch;
1262  double clearance = 0.0;
1263 
1264  if (threadType == 0)
1265  return std::nullopt;
1266 
1267  if (Threaded.getValue()) {
1268 
1269  if (ModelThread.getValue()) {
1270  if (UseCustomThreadClearance.getValue())
1271  clearance = CustomThreadClearance.getValue();
1272  else
1273  clearance = getThreadClassClearance();
1274  }
1275 
1276  // use normed diameters if possible
1277  std::string threadTypeStr = ThreadType.getValueAsString();
1278  if (threadDescription[threadType][threadSize].TapDrill > 0) {
1279  diameter = threadDescription[threadType][threadSize].TapDrill + clearance;
1280  } // if nothing is available, we must calculate
1281  else if (
1282  threadTypeStr == "BSP"
1283  || threadTypeStr == "BSW"
1284  || threadTypeStr == "BSF"
1285  ) {
1286  double thread = 2 * (0.640327 * pitch);
1287  // truncation is allowed by ISO-228 and BS 84
1288  diameter = diameter - thread * 0.75 + clearance;
1289  }
1290  else if (threadTypeStr == "NPT") {
1291  double thread = 2 * (0.8 * pitch);
1292  diameter = diameter - thread * 0.75 + clearance;
1293  }
1294  else {
1295  // this fits exactly the definition for ISO metric fine
1296  diameter = diameter - pitch + clearance;
1297  }
1298  }

1811  double vy1 = pa1.Y() - pa2.Y();
1812  double vx2 = pb1.X() - pb2.X();
1813  double vy2 = pb1.Y() - pb2.Y();
1814  double x1 = pa1.X();
1815  double y1 = pa1.Y();
1816  double x2 = pb1.X();
1817  double y2 = pb1.Y();
1818 
1819  /* Solve the system
1820  x1 + t1 * vx1 = x2 + t2 * vx2
1821  y1 + t1 * vy1 = y2 + t2 * vy2
1822 
1823  =>
1824 
1825  [ vx1 -vx2 ] [ t1 ] = [ x2 - x1 ]
1826  [ vy1 -vy2 ] [ t2 ] = [ y2 - y1 ]
1827 
1828  =>
1829 
1830  [ t1 ] = f * [ -vy2 vx2 ] [ x2 - x1 ]
1831  [ t2 ] = [ -vy1 vx1 ] [ y2 - y1 ]
1832 
1833  */
1834 
1835  assert(((vx1 * -vy2) - (-vx2 * vy1)) != 0);
1836 
1837  double f = 1 / ((vx1 * -vy2) - (-vx2 * vy1));
1838 
1839  double t1 = -vy2 * f * (x2 - x1) + vx2 * f * (y2 - y1);
1840

1843

1845  assert((y1 + t1 * vy1) - (y2 + t2 * vy2) < 1e-6);
1846 #endif
1847
