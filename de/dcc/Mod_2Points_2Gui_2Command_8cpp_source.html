76 void CmdPointsImport::activated(int iMsg)
77 {
78  Q_UNUSED(iMsg);
79 
80  QString fn = Gui::FileDialog::getOpenFileName(
81  Gui::getMainWindow(),
82  QString(),
83  QString(),
84  QString::fromLatin1("%1 (*.asc *.pcd *.ply);;%2 (*.*)")
85  .arg(QObject::tr("Point formats"), QObject::tr("All Files")));
86  if (fn.isEmpty()) {
87  return;
88  }
89 
90  if (!fn.isEmpty()) {
91  fn = Base::Tools::escapeEncodeFilename(fn);
92  Gui::Document* doc = getActiveGuiDocument();
93  openCommand(QT_TRANSLATE_NOOP("Command", "Import points"));
94  addModule(Command::App, "Points");
95  doCommand(Command::Doc,
96  "Points.insert(\"%s\", \"%s\")",
97  fn.toUtf8().data(),
98  doc->getDocument()->getName());
99  commitCommand();
100 
101  updateActive();
102 
108  auto* pcFtr = dynamic_cast<Points::Feature*>(doc->getDocument()->getActiveObject());
109  if (pcFtr) {
110  auto points = pcFtr->Points.getValue();
111  auto bbox = points.getBoundBox();
112  auto center = bbox.GetCenter();
113 
114  if (!bbox.IsInBox(Base::Vector3d(0, 0, 0))) {
115  QMessageBox msgBox;
116  msgBox.setIcon(QMessageBox::Question);
117  msgBox.setWindowTitle(QObject::tr("Points not at Origin"));
118  msgBox.setText(QObject::tr(
119  "The Bounding Box of the imported points does not contain the origin. "
120  "Do you want to translate it to the origin?"));
121  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
122  msgBox.setDefaultButton(QMessageBox::Yes);
123  auto ret = msgBox.exec();
124 
125  if (ret == QMessageBox::Yes) {
126  Points::PointKernel translatedPoints;
127  for (const auto& point : points) {
128  translatedPoints.push_back(point - center);
129  }
130  pcFtr->Points.setValue(translatedPoints);
131  }
132  }
133  }
134  }
135 }

247 void CmdPointsConvert::activated(int iMsg)
248 {
249  Q_UNUSED(iMsg);
250  double STD_OCC_TOLERANCE = 1e-6;
251 
252  int decimals = Base::UnitsApi::getDecimals();
253  double tolerance_from_decimals = pow(10., -decimals);
254 
255  double minimal_tolerance =
256  tolerance_from_decimals < STD_OCC_TOLERANCE ? STD_OCC_TOLERANCE : tolerance_from_decimals;
257 
258  bool ok;
259  double tol = QInputDialog::getDouble(Gui::getMainWindow(),
260  QObject::tr("Distance"),
261  QObject::tr("Enter maximum distance:"),
262  0.1,
263  minimal_tolerance,
264  10.0,
265  decimals,
266  &ok,
267  Qt::MSWindowsFixedSizeDialogHint);
268  if (!ok) {
269  return;
270  }
271 
272  Gui::WaitCursor wc;
273  openCommand(QT_TRANSLATE_NOOP("Command", "Convert to points"));
274  std::vector<App::GeoFeature*> geoObject = getSelection().getObjectsOfType<App::GeoFeature>();
275 
276  auto run_python = [](const std::vector<App::GeoFeature*>& geoObject, double tol) -> bool {
277  Py::List list;
278  for (auto it : geoObject) {
279  const App::PropertyComplexGeoData* prop = it->getPropertyOfGeometry();
280  if (prop) {
281  list.append(Py::asObject(it->getPyObject()));
282  }
283  }
284 
285  if (list.size() > 0) {
286  PyObject* module = PyImport_ImportModule("pointscommands.commands");
287  if (!module) {
288  throw Py::Exception();
289  }
290 
291  Py::Module commands(module, true);
292  commands.callMemberFunction("make_points_from_geometry",
293  Py::TupleN(list, Py::Float(tol)));
294  return true;
295  }
296 
297  return false;
298  };
299 
300  Base::PyGILStateLocker lock;
301  try {
302  if (run_python(geoObject, tol)) {
303  commitCommand();
304  }
305  else {
306  abortCommand();
307  }
308  }
309  catch (const Py::Exception&) {
310  abortCommand();
311  Base::PyException e;
312  e.ReportException();
313  }
314 }

446 void CmdPointsStructure::activated(int iMsg)
447 {
448  Q_UNUSED(iMsg);
449 
450  App::Document* doc = App::GetApplication().getActiveDocument();
451  doc->openTransaction("Structure point cloud");
452 
453  std::vector<App::DocumentObject*> docObj =
454  Gui::Selection().getObjectsOfType(Points::Feature::getClassTypeId());
455  for (auto it : docObj) {
456  std::string name = it->Label.getValue();
457  name += " (Structured)";
458  Points::Structured* output =
459  static_cast<Points::Structured*>(doc->addObject("Points::Structured", name.c_str()));
460  output->Label.setValue(name);
461 
462  // Already sorted, so just make a copy
463  if (it->isDerivedFrom<Points::Structured>()) {
464  Points::Structured* input = static_cast<Points::Structured*>(it);
465 
466  Points::PointKernel* kernel = output->Points.startEditing();
467  const Points::PointKernel& k = input->Points.getValue();
468 
469  kernel->resize(k.size());
470  for (std::size_t i = 0; i < k.size(); ++i) {
471  kernel->setPoint(i, k.getPoint(i));
472  }
473  output->Points.finishEditing();
474  output->Width.setValue(input->Width.getValue());
475  output->Height.setValue(input->Height.getValue());
476  }
477  // Sort the points
478  else {
479  Points::Feature* input = static_cast<Points::Feature*>(it);
480 
481  Points::PointKernel* kernel = output->Points.startEditing();
482  const Points::PointKernel& k = input->Points.getValue();
483 
484  Base::BoundBox3d bbox = input->Points.getBoundingBox();
485  double width = bbox.LengthX();
486  double height = bbox.LengthY();
487 
488  // Count the number of different x or y values to get the size
489  std::set<double> countX, countY;
490  for (std::size_t i = 0; i < k.size(); ++i) {
491  Base::Vector3d pnt = k.getPoint(i);
492  countX.insert(pnt.x);
493  countY.insert(pnt.y);
494  }
495 
496  long width_l = long(countX.size());
497  long height_l = long(countY.size());
498 
499  double dx = width / (width_l - 1);
500  double dy = height / (height_l - 1);
501 
502  // Pre-fill the vector with <nan, nan, nan> points and afterwards replace them
503  // with valid point coordinates
504  double nan = std::numeric_limits<double>::quiet_NaN();
505  std::vector<Base::Vector3d> sortedPoints(width_l * height_l,
506  Base::Vector3d(nan, nan, nan));
507 
508  for (std::size_t i = 0; i < k.size(); ++i) {
509  Base::Vector3d pnt = k.getPoint(i);
510  double xi = (pnt.x - bbox.MinX) / dx;
511  double yi = (pnt.y - bbox.MinY) / dy;
512 
513  double xx = std::fabs(xi - std::round(xi));
514  double yy = std::fabs(yi - std::round(yi));
515  if (xx < 0.01 && yy < 0.01) {
516  xi = std::round(xi);
517  yi = std::round(yi);
518  long index = long(yi * width_l + xi);
519  sortedPoints[index] = pnt;
520  }
521  }
522 
523  kernel->resize(sortedPoints.size());
524  for (std::size_t index = 0; index < sortedPoints.size(); index++) {
525  kernel->setPoint(index, sortedPoints[index]);
526  }
527 
528  output->Points.finishEditing();
529  output->Width.setValue(width_l);
530  output->Height.setValue(height_l);
531  }
532  }
533 
534  doc->commitTransaction();
535  updateActive();
536 }
