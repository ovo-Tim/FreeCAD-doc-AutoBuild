61 void ShapeFromMesh::perform()

66  Gui::WaitCursor wc;

71 
72  Gui::doCommandT(Gui::Command::Doc, "import Part");
73  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Convert mesh"));
74 
75  for (auto it : meshes) {

77  std::string mesh = it->getNameInDocument();
78  std::string name = doc->getUniqueObjectName(mesh.c_str());
79 
80  Gui::cmdAppDocumentArgs(doc, "addObject('%s', '%s')", "Part::Feature", name);
81  std::string partObj = App::DocumentObjectT(doc, name).getObjectPython();
82  std::string meshObj = App::DocumentObjectT(doc, mesh).getObjectPython();
83 
84  Gui::doCommandT(Gui::Command::Doc, "__shape__ = Part.Shape()");
85  Gui::doCommandT(Gui::Command::Doc, "__shape__.makeShapeFromMesh(%s.Mesh.Topology, %f, %s)", meshObj, tolerance, (sewShape ? "True" : "False"));
86  Gui::doCommandT(Gui::Command::Doc, partObj + ".Shape = __shape__");
