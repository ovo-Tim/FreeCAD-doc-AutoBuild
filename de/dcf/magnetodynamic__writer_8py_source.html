44  def getMagnetodynamicSolver(self, equation):
45  # output the equation parameters
46  s = self.write.createNonlinearSolver(equation)
47  if not equation.IsHarmonic:
48  s["Equation"] = "MgDyn"
49  s["Procedure"] = sifio.FileAttr("MagnetoDynamics/WhitneyAVSolver")
50  s["Variable"] = "av"
51  else:
52  s["Equation"] = "MgDynHarmonic"
53  s["Procedure"] = sifio.FileAttr("MagnetoDynamics/WhitneyAVHarmonicSolver")
54  s["Variable"] = "av[av re:1 av im:1]"
55  # round to get rid of numerical artifacts
56  frequency = float(Units.Quantity(equation.AngularFrequency).Value)
57  s["Angular Frequency"] = round(frequency, 6)
58  s["Exec Solver"] = "Always"
59  s["Optimize Bandwidth"] = True
60  s["Stabilize"] = equation.Stabilize
61  if equation.LinearSystemRefactorize is True:
62  s["Linear System Refactorize"] = True
63  if equation.UsePiolaTransform is True:
64  s["Use Piola Transform"] = True
65  if equation.QuadraticApproximation is True:
66  s["Quadratic Approximation"] = True
67  if equation.StaticConductivity is True:
68  s["Static Conductivity"] = True
69  if equation.FixInputCurrentDensity is True:
70  s["Fix Input Current Density"] = True
71  if equation.AutomatedSourceProjectionBCs is True:
72  s["Automated Source Projection BCs"] = True
73  if equation.UseLagrangeGauge is True:
74  s["Use Lagrange Gauge"] = True
75  if equation.LagrangeGaugePenalizationCoefficient != 0.0:
76  s["Lagrange Gauge Penalization Coefficient"] = (
77  equation.LagrangeGaugePenalizationCoefficient
78  )
79  if equation.UseTreeGauge is True:
80  s["Use Tree Gauge"] = True
81  return s

131  def handleMagnetodynamicMaterial(self, bodies):
132  # check that all bodies have a set material
133  for name in bodies:
134  if self.write.getBodyMaterial(name) is None:
135  raise general_writer.WriteError(
136  f"The body {name} is not referenced in any material.\n\n"
137  )
138  for obj in self.write.getMember("App::MaterialObject"):
139  m = obj.Material
140  refs = obj.References[0][1] if obj.References else self.write.getAllBodies()
141  for name in (n for n in refs if n in bodies):
142  if "ElectricalConductivity" not in m:
143  Console.PrintMessage(f"m: {m}\n")
144  raise general_writer.WriteError(
145  "The electrical conductivity must be specified for all materials.\n\n"
146  )
147  if "RelativePermeability" not in m:
148  Console.PrintMessage(f"m: {m}\n")
149  raise general_writer.WriteError(
150  "The relative permeability must be specified for all materials.\n\n"
151  )
152  self.write.material(name, "Name", m["Name"])
153  conductivity = self.write.convert(m["ElectricalConductivity"], "T^3*I^2/(L^3*M)")
154  conductivity = round(conductivity, 10) # to get rid of numerical artifacts
155  self.write.material(name, "Electric Conductivity", conductivity)
156  self.write.material(name, "Relative Permeability", float(m["RelativePermeability"]))
157  # permittivity might be necessary for the post processor
158  if "RelativePermittivity" in m:
159  self.write.material(
160  name, "Relative Permittivity", float(m["RelativePermittivity"])
161  )

223  def handleMagnetodynamicBodyForces(self, bodies, equation):
224  # the current density can either be a body force or a boundary constraint
225  # therefore only output here if a solid is referenced
226  currentDensities = self.write.getMember("Fem::ConstraintCurrentDensity")
227  for obj in currentDensities:
228  if obj.References:
229  firstName = obj.References[0][1][0]
230  firstName = firstName.rstrip("0123456789")
231  if firstName == "Solid":
232  for name in obj.References[0][1]:
233  self._outputMagnetodynamicBodyForce(obj, name, equation)
234  self.write.handled(obj)
235  else:
236  # if there is only one current density without a reference,
237  # add it to all bodies
238  if len(currentDensities) == 1:
239  for name in bodies:
240  self._outputMagnetodynamicBodyForce(obj, name, equation)
241  else:
242  raise general_writer.WriteError(
