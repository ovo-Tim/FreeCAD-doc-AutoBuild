150 class SegmentationManual::Private
151 {
152 public:
153  static void findGeometry(
154  int minFaces,
155  double tolerance,
156  std::function<MeshCore::AbstractSurfaceFit*(const std::vector<Base::Vector3f>&,
157  const std::vector<Base::Vector3f>&)> fitFunc)
158  {
159  Gui::Document* gdoc = Gui::Application::Instance->activeDocument();
160  if (!gdoc) {
161  return;
162  }
163 
164  App::Document* adoc = gdoc->getDocument();
165  std::vector<Mesh::Feature*> meshes = adoc->getObjectsOfType<Mesh::Feature>();
166  for (auto it : meshes) {
167  MeshGui::ViewProviderMesh* vpm =
168  static_cast<MeshGui::ViewProviderMesh*>(gdoc->getViewProvider(it));
169  const Mesh::MeshObject& mesh = it->Mesh.getValue();
170 
171  if (mesh.hasSelectedFacets()) {
172  const MeshCore::MeshKernel& kernel = mesh.getKernel();
173 
174  std::vector<MeshCore::FacetIndex> facets;
175  std::vector<MeshCore::PointIndex> vertexes;
176  mesh.getFacetsFromSelection(facets);
177  vertexes = mesh.getPointsFromFacets(facets);
178  MeshCore::MeshPointArray coords = kernel.GetPoints(vertexes);
179 
180  std::vector<Base::Vector3f> points, normals;
181  normals = kernel.GetFacetNormals(facets);
182  points.insert(points.end(), coords.begin(), coords.end());
183  coords.clear();
184 
185  MeshCore::AbstractSurfaceFit* surfFit = fitFunc(points, normals);
186  if (surfFit) {
187  MeshCore::MeshSegmentAlgorithm finder(kernel);
188 
189  std::vector<MeshCore::MeshSurfaceSegmentPtr> segm;
190  segm.emplace_back(
191  std::make_shared<MeshCore::MeshDistanceGenericSurfaceFitSegment>(
192  surfFit,
193  kernel,
194  minFaces,
195  tolerance));
196  finder.FindSegments(segm);
197 
198  for (const auto& segmIt : segm) {
199  const std::vector<MeshCore::MeshSegment>& data = segmIt->GetSegments();
200  for (const auto& dataIt : data) {
201  vpm->addSelection(dataIt);
202  }
203  }
204  }
205  }
206  }
207  }
208 };

210 void SegmentationManual::onPlaneDetectClicked()
211 {
212  auto func = [=](const std::vector<Base::Vector3f>& points,
213  const std::vector<Base::Vector3f>& normal) -> MeshCore::AbstractSurfaceFit* {
214  Q_UNUSED(normal)
215 
216  MeshCore::PlaneFit fit;
217  fit.AddPoints(points);
218  if (fit.Fit() < FLOAT_MAX) {
219  Base::Vector3f base = fit.GetBase();
220  Base::Vector3f axis = fit.GetNormal();
221  return new MeshCore::PlaneSurfaceFit(base, axis);
222  }
223 
224  return nullptr;
225  };
226  Private::findGeometry(ui->numPln->value(), ui->tolPln->value(), func);
227 }

229 void SegmentationManual::onCylinderDetectClicked()
230 {
231  auto func = [=](const std::vector<Base::Vector3f>& points,
232  const std::vector<Base::Vector3f>& normal) -> MeshCore::AbstractSurfaceFit* {
233  Q_UNUSED(normal)
234 
235  MeshCore::CylinderFit fit;
236  fit.AddPoints(points);
237  if (!normal.empty()) {
238  Base::Vector3f base = fit.GetGravity();
239  Base::Vector3f axis = fit.GetInitialAxisFromNormals(normal);
240  fit.SetInitialValues(base, axis);
241  }
242  if (fit.Fit() < FLOAT_MAX) {
243  Base::Vector3f base = fit.GetBase();
244  Base::Vector3f axis = fit.GetAxis();
245  float radius = fit.GetRadius();
246  return new MeshCore::CylinderSurfaceFit(base, axis, radius);
247  }
248 
249  return nullptr;
250  };
251  Private::findGeometry(ui->numCyl->value(), ui->tolCyl->value(), func);
252 }

273 void SegmentationManual::createSegment()
274 {
275  Gui::Document* gdoc = Gui::Application::Instance->activeDocument();
276  if (!gdoc) {
277  return;
278  }
279  // delete all selected faces
280  App::Document* adoc = gdoc->getDocument();
281  gdoc->openCommand(QT_TRANSLATE_NOOP("Command", "Segmentation"));
282 
283  std::vector<Mesh::Feature*> meshes = adoc->getObjectsOfType<Mesh::Feature>();
284  bool selected = false;
285  for (auto it : meshes) {
286  const Mesh::MeshObject& mesh = it->Mesh.getValue();
287  const MeshCore::MeshKernel& kernel = mesh.getKernel();
288 
289  MeshCore::MeshAlgorithm algo(kernel);
290  unsigned long ct = algo.CountFacetFlag(MeshCore::MeshFacet::SELECTED);
291  if (ct > 0) {
292  selected = true;
293 
294  std::vector<MeshCore::FacetIndex> facets;
295  algo.GetFacetsFlag(facets, MeshCore::MeshFacet::SELECTED);
296 
297  std::unique_ptr<Mesh::MeshObject> segment(mesh.meshFromSegment(facets));
298  Mesh::Feature* feaSegm =
299  static_cast<Mesh::Feature*>(adoc->addObject("Mesh::Feature", "Segment"));
300  Mesh::MeshObject* feaMesh = feaSegm->Mesh.startEditing();
301  feaMesh->swap(*segment);
302  feaMesh->clearFacetSelection();
303  feaSegm->Mesh.finishEditing();
304 
305  if (ui->checkBoxHideSegm->isChecked()) {
306  feaSegm->Visibility.setValue(false);
307  }
308 
309  if (ui->checkBoxCutSegm->isChecked()) {
310  Mesh::MeshObject* editmesh = it->Mesh.startEditing();
311  editmesh->deleteFacets(facets);
312  it->Mesh.finishEditing();
313  }
314  }
315  }
316 
317  if (!selected) {
318  gdoc->abortCommand();
319  }
320  else {
321  gdoc->commitCommand();
322  }
323  meshSel.clearSelection();
324 }
