171  def _getObjects(self, obj, names):
172  objects = []
173  if not hasattr(obj, "Shape"):
174  FreeCAD.Console.PrintMessage("Selected object has no Shape.\n")
175  return objects
176  shape = obj.Shape
177  for n in names:
178  if n.startswith("Face"):
179  objects.append(shape.Faces[int(n[4:]) - 1])
180  elif n.startswith("Edge"):
181  objects.append(shape.Edges[int(n[4:]) - 1])
182  elif n.startswith("Vertex"):
183  objects.append(shape.Vertexes[int(n[6:]) - 1])
184  elif n.startswith("Solid"):
185  objects.append(shape.Solids[int(n[5:]) - 1])
186  return objects

188  def _getSolidOfSub(self, obj, sub):
189  foundSolids = set()
190  if sub.ShapeType == "Solid":
191  for solidId, solid in enumerate(obj.Shape.Solids):
192  if sub.isSame(solid):
193  foundSolids.add("Solid" + str(solidId + 1))
194  elif sub.ShapeType == "Face":
195  for solidId, solid in enumerate(obj.Shape.Solids):
196  if self._findSub(sub, solid.Faces):
197  foundSolids.add("Solid" + str(solidId + 1))
198  elif sub.ShapeType == "Edge":
199  for solidId, solid in enumerate(obj.Shape.Solids):
200  if self._findSub(sub, solid.Edges):
201  foundSolids.add("Solid" + str(solidId + 1))
202  elif sub.ShapeType == "Vertex":
203  for solidId, solid in enumerate(obj.Shape.Solids):
204  if self._findSub(sub, solid.Vertexes):
205  foundSolids.add("Solid" + str(solidId + 1))
206  if len(foundSolids) == 1:
207  it = iter(foundSolids)
208  return next(it)
209  return None

456  def selectionParser(self, selection):

464  elt = sobj.Shape.getElement(selection[1])
465  ele_ShapeType = elt.ShapeType
466  if self.selection_mode_solid and "Solid" in self.sel_elem_types:
467  # in solid selection mode use edges and faces for selection of a solid
468  # adapt selection variable to hold the Solid
469  solid_to_add = None
