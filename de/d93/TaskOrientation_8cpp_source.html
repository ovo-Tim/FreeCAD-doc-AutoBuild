98 void TaskOrientation::restore(const Base::Placement& plm)
99 {
100  auto isReversed = [](Camera::Orientation type) {
101  return (type == Camera::Bottom) || (type == Camera::Rear) || (type == Camera::Left);
102  };
103  std::map<Camera::Orientation, Base::Rotation> rotations {
104  {Camera::Top, Camera::convert(Camera::Top)},
105  {Camera::Bottom, Camera::convert(Camera::Bottom)},
106  {Camera::Front, Camera::convert(Camera::Front)},
107  {Camera::Rear, Camera::convert(Camera::Rear)},
108  {Camera::Right, Camera::convert(Camera::Right)},
109  {Camera::Left, Camera::convert(Camera::Left)}};
110 
111  Base::Rotation rot = plm.getRotation();
112  Base::Vector3d pos = plm.getPosition();
113 
114  double prec = 1.0e-5;
115  for (const auto& it : rotations) {
116  if (rot.isSame(it.second, prec)) {
117  if (it.first == Camera::Top || it.first == Camera::Bottom) {
118  ui->XY_radioButton->setChecked(true);
119  ui->Offset_doubleSpinBox->setValue(pos.z);
120  }
121  else if (it.first == Camera::Front || it.first == Camera::Rear) {
122  ui->XZ_radioButton->setChecked(true);
123  ui->Offset_doubleSpinBox->setValue(pos.y);
124  }
125  else if (it.first == Camera::Right || it.first == Camera::Left) {
126  ui->YZ_radioButton->setChecked(true);
127  ui->Offset_doubleSpinBox->setValue(pos.x);
128  }
129 
130  if (isReversed(it.first)) {
131  ui->Reverse_checkBox->setChecked(true);
132  }
133  }
134  }
135 
136  onPreview();
137 }

139 void TaskOrientation::updatePlacement()
140 {
141  Base::Placement Pos;
142  double offset = ui->Offset_doubleSpinBox->value().getValue();
143  bool reverse = ui->Reverse_checkBox->isChecked();
144  if (ui->XY_radioButton->isChecked()) {
145  if (!reverse) {
146  Pos = Base::Placement(Base::Vector3d(0, 0, offset), Camera::convert(Camera::Top));
147  }
148  else {
149  Pos = Base::Placement(Base::Vector3d(0, 0, offset), Camera::convert(Camera::Bottom));
150  }
151  }
152  else if (ui->XZ_radioButton->isChecked()) {
153  if (!reverse) {
154  Pos = Base::Placement(Base::Vector3d(0, offset, 0), Camera::convert(Camera::Front));
155  }
156  else {
157  Pos = Base::Placement(Base::Vector3d(0, offset, 0), Camera::convert(Camera::Rear));
158  }
159  }
160  else if (ui->YZ_radioButton->isChecked()) {
161  if (!reverse) {
162  Pos = Base::Placement(Base::Vector3d(offset, 0, 0), Camera::convert(Camera::Right));
163  }
164  else {
165  Pos = Base::Placement(Base::Vector3d(offset, 0, 0), Camera::convert(Camera::Left));
166  }
167  }
168 
169  if (!feature.expired()) {
170  feature->Placement.setValue(Pos);
171  }
172 }
