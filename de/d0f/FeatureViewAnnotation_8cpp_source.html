58 App::DocumentObjectExecReturn* FeatureViewAnnotation::execute(void)
59 {
60  stringstream result, hr, hg, hb;
61  const App::Color& c = TextColor.getValue();
62  hr << hex << setfill('0') << setw(2) << (int)(255.0 * c.r);
63  hg << hex << setfill('0') << setw(2) << (int)(255.0 * c.g);
64  hb << hex << setfill('0') << setw(2) << (int)(255.0 * c.b);
65 
66  result << "<g transform=\"translate(" << X.getValue() << "," << Y.getValue() << ")"
67  << " rotate(" << Rotation.getValue() << ")\">" << endl
68  << "<text id=\"" << Label.getValue() << "\"" << endl
69  << " font-family=\"" << Font.getValue() << "\"" << endl
70  << " font-size=\"" << Scale.getValue() << "\"" << endl
71  << " fill=\"#" << hr.str() << hg.str() << hb.str() << "\">" << endl;
72 
73  for (vector<string>::const_iterator it = Text.getValues().begin(); it != Text.getValues().end();
74  ++it) {
75  result << "<tspan x=\"0\" dy=\"1em\">" << it->c_str() << "</tspan>" << endl;
76  }
77 
78  result << "</text>" << endl << "</g>" << endl;
79 
80  // Apply the resulting fragment
81  ViewResult.setValue(result.str().c_str());
82 
83  return App::DocumentObject::StdReturn;
84 }
