178 void SceneModel::setNode(QModelIndex index, SoNode* node)
179 {
180  this->setData(index.siblingAtColumn(static_cast<int>(Column::INVENTOR_TREE)),
181  QVariant(QString::fromLatin1(QByteArray(node->getTypeId().getName()))));
182 
183  QHash<SoNode*, QString>::iterator it = nodeNames.find(node);
184  const QString name {
185  (it != nodeNames.end()) ? it.value()
186  : QString::fromLatin1(QByteArray(node->getName()))
187  };
188  this->setData(index.siblingAtColumn(static_cast<int>(Column::NAME)), QVariant(name));
189 
190  this->setData(index.siblingAtColumn(static_cast<int>(Column::MEMORY_ADDRESS)),
191  QVariant(QString::fromStdString(fmt::format("{}", (void*)node))));
192 
193  QString data;
194  QTextStream stream(&data);
195  if(static_cast<bool>(node->isOfType(SoSwitch::getClassTypeId()))) {
196  auto pcSwitch = static_cast<SoSwitch*>(node);
197  auto value = pcSwitch->whichChild.getValue();
198  stream << fmt::format("Which: {} ({})", formatSoSwitchValue(value), value).c_str();
199  } else if (static_cast<bool>(node->isOfType(SoSeparator::getClassTypeId()))) {
200  auto pcSeparator = static_cast<SoSeparator*>(node);
201  auto value = pcSeparator->renderCaching.getValue();
202  stream << fmt::format("RenderCaching: {} ({})", formatSoSeparatorCacheEnabled(value), value).c_str();
203  } else if (static_cast<bool>(node->isOfType(SoDrawStyle::getClassTypeId()))) {
204  auto pcDrawStyle = static_cast<SoDrawStyle*>(node);
205  auto value = pcDrawStyle->style.getValue();
206  stream << fmt::format("Style: {} ({})", formatSoDrawStyleElement(value), value).c_str();
207  } else if (static_cast<bool>(node->isOfType(SoPickStyle::getClassTypeId()))) {
208  auto pcPickStyle = static_cast<SoPickStyle*>(node);
209  auto value = pcPickStyle->style.getValue();
210  stream << fmt::format("Style: {} ({})", formatSoPickStyleElement(value), value).c_str();
211  } else if (static_cast<bool>(node->isOfType(SoCoordinate3::getClassTypeId()))) {
212  auto pcCoords = static_cast<SoCoordinate3*>(node);
213  auto values = pcCoords->point.getValues(0);
214  if (values) {
215  float x { 0 };
216  float y { 0 };
217  float z { 0 };
218  values->getValue(x, y, z);
219  stream << fmt::format("XYZ: {}, {}, {}", x, y, z).c_str();
220  }
221  }
222 
223  this->setData(index.siblingAtColumn((int)Column::DATA), QVariant(data));
224 
225  if (static_cast<bool>(node->getTypeId().isDerivedFrom(SoGroup::getClassTypeId()))) {
226  auto group = static_cast<SoGroup*>(node);
227  this->insertColumns(0, static_cast<int>(Column::COUNT), index);
228  this->insertRows(0, group->getNumChildren(), index);
229  for (int i=0; i < group->getNumChildren(); i++) {
230  SoNode* child = group->getChild(i);
231  setNode(this->index(i, 0, index), child);
232  }
233  }
234 }

294 void DlgInspector::setNodeNames(Gui::Document* doc)
295 {
296  std::vector<Gui::ViewProvider*> vps = doc->getViewProvidersOfType
297  (Gui::ViewProviderDocumentObject::getClassTypeId());
298  QHash<SoNode*, QString> nodeNames;
299  for (const auto & it : vps) {
300  auto vp = static_cast<Gui::ViewProviderDocumentObject*>(it);
301  App::DocumentObject* obj = vp->getObject();
302  if (obj) {
303  QString label = QString::fromUtf8(obj->Label.getValue());
304  nodeNames[vp->getRoot()] = label;
305  }
306 
307  std::vector<std::string> modes = vp->getDisplayMaskModes();
308  for (const auto & mode : modes) {
309  SoNode* node = vp->getDisplayMaskMode(mode.c_str());
310  if (node) {
311  nodeNames[node] = QString::fromStdString(mode);
312  }
313  }
314  }
315 
316  auto model = static_cast<SceneModel*>(ui->treeView->model());
317  model->setNodeNames(nodeNames);
318 }
