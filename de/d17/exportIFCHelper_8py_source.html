40 def writeUnits(ifcfile,unit="metre"):
41  """adds additional units settings to the given ifc file if needed"""
42  # so far, only metre or foot possible (which is all revit knows anyway)
43 
44  if unit == "foot":
45  d1 = ifcfile.createIfcDimensionalExponents(1,0,0,0,0,0,0)
46  d2 = ifcfile.createIfcMeasureWithUnit(ifcfile.createIfcRatioMeasure(0.3048),ifcfile[13])
47  d3 = ifcfile.createIfcConversionBasedUnit(d1,'LENGTHUNIT','FOOT',d2)
48  d4 = ifcfile.createIfcDimensionalExponents(2,0,0,0,0,0,0)
49  d5 = ifcfile.createIfcMeasureWithUnit(ifcfile.createIfcRatioMeasure(0.09290304000000001),ifcfile[14])
50  d6 = ifcfile.createIfcConversionBasedUnit(d4,'AREAUNIT','SQUARE FOOT',d5)
51  d7 = ifcfile.createIfcDimensionalExponents(3,0,0,0,0,0,0)
52  d8 = ifcfile.createIfcMeasureWithUnit(ifcfile.createIfcRatioMeasure(0.028316846592),ifcfile[15])
53  d9 = ifcfile.createIfcConversionBasedUnit(d7,'VOLUMEUNIT','CUBIC FOOT',d8)
54  ifcfile.createIfcUnitAssignment((d3,d6,d9,ifcfile[18]))
55  else: # default = metre, no need to add anything
56  ifcfile.createIfcUnitAssignment((ifcfile[13],ifcfile[14],ifcfile[15],ifcfile[18]))
57  return ifcfile

60 def writeQuantities(ifcfile, obj, product, history, scale):
61  "append quantities to the given object"
62 
63  if hasattr(obj,"IfcData"):
64  quantities = []
65  if ("ExportHeight" in obj.IfcData) and obj.IfcData["ExportHeight"] and hasattr(obj,"Height"):
66  quantities.append(ifcfile.createIfcQuantityLength('Height',None,None,obj.Height.Value*scale))
67  if ("ExportWidth" in obj.IfcData) and obj.IfcData["ExportWidth"] and hasattr(obj,"Width"):
68  quantities.append(ifcfile.createIfcQuantityLength('Width',None,None,obj.Width.Value*scale))
69  if ("ExportLength" in obj.IfcData) and obj.IfcData["ExportLength"] and hasattr(obj,"Length"):
70  quantities.append(ifcfile.createIfcQuantityLength('Length',None,None,obj.Length.Value*scale))
71  if ("ExportHorizontalArea" in obj.IfcData) and obj.IfcData["ExportHorizontalArea"] and hasattr(obj,"HorizontalArea"):
72  quantities.append(ifcfile.createIfcQuantityArea('HorizontalArea',None,None,obj.HorizontalArea.Value*(scale**2)))
73  if ("ExportVerticalArea" in obj.IfcData) and obj.IfcData["ExportVerticalArea"] and hasattr(obj,"VerticalArea"):
74  quantities.append(ifcfile.createIfcQuantityArea('VerticalArea',None,None,obj.VerticalArea.Value*(scale**2)))
75  if ("ExportVolume" in obj.IfcData) and obj.IfcData["ExportVolume"] and obj.isDerivedFrom("Part::Feature"):
76  quantities.append(ifcfile.createIfcQuantityVolume('Volume',None,None,obj.Shape.Volume*(scale**3)))
77  if quantities:
78  eltq = ifcfile.createIfcElementQuantity(
79  ifcopenshell.guid.new(),
80  history,
81  "ElementQuantities",
82  None,
83  "FreeCAD",quantities
84  )
85  ifcfile.createIfcRelDefinesByProperties(
86  ifcopenshell.guid.new(),
87  history,
88  None,
89  None,
90  [product],eltq
91  )
