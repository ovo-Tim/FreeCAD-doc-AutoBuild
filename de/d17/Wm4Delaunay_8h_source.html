48  // The interpretations of the return values of these functions depends on
49  // the dimension. Generally, SQ = GetSimplexQuantity() is the number of
50  // simplices in the mesh. The array returned by I = GetIndices() contains
51  // SQ tuples, each tuple having d+1 elements and representing a simplex.
52  // An index I[*] is relative to the vertex array V. The array returned by
53  // A = GetAdjacencies() contains SQ tuples, each tuple having d+1 elements
54  // and representing those simplices adjacent to the d+1 faces of a
55  // simplex. An index A[*] is relative to the index array I.
56  int GetSimplexQuantity () const;
57  const int* GetIndices () const;
58  const int* GetAdjacencies () const;
59 
60  // Dimension d = 0.
61  // SQ = 1
62  // I = null (use index zero for vertices)
63  // A = null (use index zero for vertices)
64 
65  // Dimension d = 1.
66  // SQ = VQ-1
67  // I = Array of 2-tuples of indices into V that represent the
68  // segments (2*SQ total elements).
69  // A = Array of 2-tuples of indices into I that represent the
70  // adjacent segments (2*SQ total elements).
71  // The i-th segment has vertices
72  // vertex[0] = V[I[2*i+0]]
73  // vertex[1] = V[I[2*i+1]].
74  // The segments adjacent to these vertices have indices
75  // adjacent[0] = A[2*i+0] is the segment sharing vertex[0]
76  // adjacent[1] = A[2*i+1] is the segment sharing vertex[1]
77  // If there is no adjacent segment, the A[*] value is set to -1. The
78  // segment adjacent to vertex[j] has vertices
79  // adjvertex[0] = V[I[2*adjacent[j]+0]]
80  // adjvertex[1] = V[I[2*adjacent[j]+1]]
81 
82  // Dimension d = 2.
83  // SQ = number of triangles
84  // I = Array of 3-tuples of indices into V that represent the
85  // triangles (3*SQ total elements).
86  // A = Array of 3-tuples of indices into I that represent the
87  // adjacent triangles (3*SQ total elements).
88  // The i-th triangle has vertices
89  // vertex[0] = V[I[3*i+0]]
90  // vertex[1] = V[I[3*i+1]]
91  // vertex[2] = V[I[3*i+2]]
92  // and edge index pairs
93  // edge[0] = <I[3*i+0],I[3*i+1]>
94  // edge[1] = <I[3*i+1],I[3*i+2]>
95  // edge[2] = <I[3*i+2],I[3*i+0]>
96  // The triangles adjacent to these edges have indices
97  // adjacent[0] = A[3*i+0] is the triangle sharing edge[0]
98  // adjacent[1] = A[3*i+1] is the triangle sharing edge[1]
99  // adjacent[2] = A[3*i+2] is the triangle sharing edge[2]
100  // If there is no adjacent triangle, the A[*] value is set to -1. The
101  // triangle adjacent to edge[j] has vertices
102  // adjvertex[0] = V[I[3*adjacent[j]+0]]
103  // adjvertex[1] = V[I[3*adjacent[j]+1]]
104  // adjvertex[2] = V[I[3*adjacent[j]+2]]
105 
106  // Dimension d = 3.
107  // SQ = number of tetrahedra
108  // I = Array of 4-tuples of indices into V that represent the
109  // tetrahedra (4*SQ total elements).
110  // A = Array of 4-tuples of indices into I that represent the
111  // adjacent tetrahedra (4*SQ total elements).
112  // The i-th tetrahedron has vertices
113  // vertex[0] = V[I[4*i+0]]
114  // vertex[1] = V[I[4*i+1]]
115  // vertex[2] = V[I[4*i+2]]
116  // vertex[3] = V[I[4*i+3]]
117  // and face index triples listed below. The face vertex ordering when
118  // viewed from outside the tetrahedron is counterclockwise.
119  // face[0] = <I[4*i+1],I[4*i+2],I[4*i+3]>
120  // face[1] = <I[4*i+0],I[4*i+3],I[4*i+2]>
121  // face[2] = <I[4*i+0],I[4*i+1],I[4*i+3]>
122  // face[3] = <I[4*i+0],I[4*i+2],I[4*i+1]>
123  // The tetrahedra adjacent to these faces have indices
124  // adjacent[0] = A[4*i+0] is the tetrahedron opposite vertex[0], so it
125  // is the tetrahedron sharing face[0].
126  // adjacent[1] = A[4*i+1] is the tetrahedron opposite vertex[1], so it
127  // is the tetrahedron sharing face[1].
128  // adjacent[2] = A[4*i+2] is the tetrahedron opposite vertex[2], so it
129  // is the tetrahedron sharing face[2].
130  // adjacent[3] = A[4*i+3] is the tetrahedron opposite vertex[3], so it
131  // is the tetrahedron sharing face[3].
132  // If there is no adjacent tetrahedron, the A[*] value is set to -1. The
133  // tetrahedron adjacent to face[j] has vertices
134  // adjvertex[0] = V[I[4*adjacent[j]+0]]
135  // adjvertex[1] = V[I[4*adjacent[j]+1]]
136  // adjvertex[2] = V[I[4*adjacent[j]+2]]
137  // adjvertex[3] = V[I[4*adjacent[j]+3]]
138
