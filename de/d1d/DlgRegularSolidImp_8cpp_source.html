117 void DlgRegularSolidImp::onCreateSolidButtonClicked()
118 {
119  try {
120  Gui::WaitCursor wc;
121  std::string cmd, name;
122  App::Document* doc = App::GetApplication().getActiveDocument();
123  if (!doc) {
124  QMessageBox::warning(this, tr("Create %1").arg(ui->comboBox1->currentText()), tr("No active document"));
125  return;
126  }
127  switch (ui->comboBox1->currentIndex()) {
128  case 0:
129  name = doc->getUniqueObjectName("Cube");
130  cmd = fmt::format(
131  "App.ActiveDocument.addObject(\"Mesh::Cube\",\"{0}\")\n"
132  "App.ActiveDocument.{0}.Length={1}\n"
133  "App.ActiveDocument.{0}.Width={2}\n"
134  "App.ActiveDocument.{0}.Height={3}\n", name,
135  Base::UnitsApi::toNumber(ui->boxLength->value()),
136  Base::UnitsApi::toNumber(ui->boxWidth->value()),
137  Base::UnitsApi::toNumber(ui->boxHeight->value()));
138  break;
139  case 1:
140  name = doc->getUniqueObjectName("Cylinder");
141  cmd = fmt::format(
142  "App.ActiveDocument.addObject(\"Mesh::Cylinder\",\"{0}\")\n"
143  "App.ActiveDocument.{0}.Radius={1}\n"
144  "App.ActiveDocument.{0}.Length={2}\n"
145  "App.ActiveDocument.{0}.EdgeLength={3}\n"
146  "App.ActiveDocument.{0}.Closed={4}\n"
147  "App.ActiveDocument.{0}.Sampling={5}\n", name,
148  Base::UnitsApi::toNumber(ui->cylinderRadius->value()),
149  Base::UnitsApi::toNumber(ui->cylinderLength->value()),
150  Base::UnitsApi::toNumber(ui->cylinderEdgeLength->value()),
151  ui->cylinderClosed->isChecked() ? "True" : "False",
152  ui->cylinderCount->value());
153  break;
154  case 2:
155  name = doc->getUniqueObjectName("Cone");
156  cmd = fmt::format(
157  "App.ActiveDocument.addObject(\"Mesh::Cone\",\"{0}\")\n"
158  "App.ActiveDocument.{0}.Radius1={1}\n"
159  "App.ActiveDocument.{0}.Radius2={2}\n"
160  "App.ActiveDocument.{0}.Length={3}\n"
161  "App.ActiveDocument.{0}.EdgeLength={4}\n"
162  "App.ActiveDocument.{0}.Closed={5}\n"
163  "App.ActiveDocument.{0}.Sampling={6}\n", name,
164  Base::UnitsApi::toNumber(ui->coneRadius1->value()),
165  Base::UnitsApi::toNumber(ui->coneRadius2->value()),
166  Base::UnitsApi::toNumber(ui->coneLength->value()),
167  Base::UnitsApi::toNumber(ui->coneEdgeLength->value()),
168  ui->coneClosed->isChecked() ? "True" : "False",
169  ui->coneCount->value());
170  break;
171  case 3:
172  name = doc->getUniqueObjectName("Sphere");
173  cmd = fmt::format(
174  "App.ActiveDocument.addObject(\"Mesh::Sphere\",\"{0}\")\n"
175  "App.ActiveDocument.{0}.Radius={1}\n"
176  "App.ActiveDocument.{0}.Sampling={2}\n", name,
177  Base::UnitsApi::toNumber(ui->sphereRadius->value()),
178  ui->sphereCount->value());
179  break;
180  case 4:
181  name = doc->getUniqueObjectName("Ellipsoid");
182  cmd = fmt::format(
183  "App.ActiveDocument.addObject(\"Mesh::Ellipsoid\",\"{0}\")\n"
184  "App.ActiveDocument.{0}.Radius1={1}\n"
185  "App.ActiveDocument.{0}.Radius2={2}\n"
186  "App.ActiveDocument.{0}.Sampling={3}\n", name,
187  Base::UnitsApi::toNumber(ui->ellipsoidRadius1->value()),
188  Base::UnitsApi::toNumber(ui->ellipsoidRadius2->value()),
189  ui->ellipsoidCount->value());
190  break;
191  case 5:
192  name = doc->getUniqueObjectName("Torus");
193  cmd = fmt::format(
194  "App.ActiveDocument.addObject(\"Mesh::Torus\",\"{0}\")\n"
195  "App.ActiveDocument.{0}.Radius1={1}\n"
196  "App.ActiveDocument.{0}.Radius2={2}\n"
197  "App.ActiveDocument.{0}.Sampling={3}\n", name,
198  Base::UnitsApi::toNumber(ui->toroidRadius1->value()),
199  Base::UnitsApi::toNumber(ui->toroidRadius2->value()),
200  ui->toroidCount->value());
201  break;
202  }
203 
204  // Execute the Python block
205  QString solid = tr("Create %1").arg(ui->comboBox1->currentText());
206  Gui::Application::Instance->activeDocument()->openCommand(solid.toUtf8());
207  Gui::Command::doCommand(Gui::Command::Doc, cmd.c_str());
208  Gui::Application::Instance->activeDocument()->commitCommand();
209  Gui::Command::doCommand(Gui::Command::Doc, "App.activeDocument().recompute()");
210  Gui::Command::doCommand(Gui::Command::Gui, "Gui.SendMsgToActiveView(\"ViewFit\")");
211  }
212  catch (const Base::PyException& e) {
213  QMessageBox::warning(this, tr("Create %1").arg(ui->comboBox1->currentText()),
214  QString::fromLatin1(e.what()));
215  }
216 }
