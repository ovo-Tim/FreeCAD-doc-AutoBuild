150  def createObject(self):
151  """Create object in the current document."""
152  String = self.form.leString.text().replace('\\', '\\\\').replace('"', '\\"')
153  String = '"' + String + '"'
154  FFile = '"' + str(self.fileSpec) + '"'
155 
156  Size = str(App.Units.Quantity(self.form.sbHeight.text()).Value)
157  Tracking = str(0.0)
158  x = App.Units.Quantity(self.form.sbX.text()).Value
159  y = App.Units.Quantity(self.form.sbY.text()).Value
160  z = App.Units.Quantity(self.form.sbZ.text()).Value
161  ssBase = App.Vector(x, y, z)
162  try:
163  qr, sup, points, fil = self.sourceCmd.getStrings()
164  Gui.addModule("Draft")
165  self.sourceCmd.commit(translate('draft', 'Create ShapeString'),
166  ['ss = Draft.make_shapestring(String=' + String + ', FontFile=' + FFile + ', Size=' + Size + ', Tracking=' + Tracking + ')',
167  'plm = FreeCAD.Placement()',
168  'plm.Base = ' + toString(ssBase),
169  'plm.Rotation.Q = ' + qr,
170  'ss.Placement = plm',
171  'ss.AttachmentSupport = ' + sup,
172  'Draft.autogroup(ss)',
173  'FreeCAD.ActiveDocument.recompute()'])
174  except Exception:
