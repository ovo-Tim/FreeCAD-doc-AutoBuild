123  if (e->type() == QEvent::ParentChange || e->type() == QEvent::ParentAboutToChange)

187 void DlgCustomActionsImp::onActionListWidgetItemActivated(QTreeWidgetItem* item)
188 {
189  if (!item) {
190  return; // no valid item
191  }
192 
193  // search for the command in the manager and if necessary in the temporary created ones
194  QByteArray actionName = item->data(1, Qt::UserRole).toByteArray();
195  CommandManager& rclMan = Application::Instance->commandManager();
196  Command* pCmd = rclMan.getCommandByName(actionName.constData());
197  auto pScript = dynamic_cast<MacroCommand*>(pCmd);
198 
199  // if valid command
200  if (pScript) {
201  bool bFound = false;
202  QString scriptName = QString::fromUtf8(pScript->getScriptName());
203  for (int i = 0; i < ui->actionMacros->count(); i++) {
204  if (ui->actionMacros->itemText(i).startsWith(scriptName, Qt::CaseSensitive)) {
205  bFound = true;
206  ui->actionMacros->setCurrentIndex(i);
207  break;
208  }
209  }
210 
211  if (!bFound) {
212  QMessageBox::critical(this,
213  tr("Macro not found"),
214  tr("Sorry, couldn't find macro file '%1'.").arg(scriptName));
215  }
216 
217  // fill up labels with the command's data
218  ui->actionWhatsThis->setText(QString::fromUtf8(pScript->getWhatsThis()));
219  ui->actionMenu->setText(QString::fromUtf8(pScript->getMenuText()));
220  ui->actionToolTip->setText(QString::fromUtf8(pScript->getToolTipText()));
221  ui->actionStatus->setText(QString::fromUtf8(pScript->getStatusTip()));
222  ui->actionAccel->setText(
223  ShortcutManager::instance()->getShortcut(actionName.constData(), pScript->getAccel()));
224  ui->pixmapLabel->clear();
225  m_sPixmap.clear();
226  const char* name = pScript->getPixmap();
227  if (name && std::strlen(name) > 2) {
228  QPixmap p = Gui::BitmapFactory().pixmap(pScript->getPixmap());
229  ui->pixmapLabel->setPixmap(p);
230  m_sPixmap = QString::fromUtf8(name); // can also be a path
231  }
232  }

452 void IconDialog::onAddIconPath()
453 {
454  // Add the user defined paths
455  Base::Reference<ParameterGrp> group =
456  App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/Bitmaps");
457  std::vector<std::string> paths = group->GetASCIIs("CustomPath");
458  QStringList pathList;
459  for (const auto& path : paths) {
460  pathList << QString::fromUtf8(path.c_str());
461  }
462 
463  IconFolders dlg(pathList, this);
464  dlg.setWindowTitle(tr("Icon folders"));
465  if (dlg.exec()) {
466  QStringList paths = dlg.getPaths();
467 
468  // Write to user config
469  group->Clear();
470  int index = 0;
471  for (QStringList::iterator it = paths.begin(); it != paths.end(); ++it, ++index) {
472  std::stringstream str;
473  str << "CustomPath" << index;
474  group->SetASCII(str.str().c_str(), (const char*)it->toUtf8());
475  }
476 
477  QStringList search = BitmapFactory().getPaths();
478  for (auto& it : search) {
479  it = QDir::toNativeSeparators(it);
480  }
481  for (const auto& path : paths) {
482  if (search.indexOf(path) < 0) {
483  QStringList filters;
484  QList<QByteArray> formats = QImageReader::supportedImageFormats();
485  for (const auto& format : formats) {
486  filters << QString::fromLatin1("*.%1").arg(
487  QString::fromLatin1(format).toLower());
488  }
489  QDir d(path);
490  d.setNameFilters(filters);
491  QFileInfoList fi = d.entryInfoList();
492  for (const auto& jt : fi) {
493  QString file = jt.absoluteFilePath();
494  auto item = new QListWidgetItem(ui->listWidget);
495  item->setIcon(QIcon(file));
496  item->setText(jt.baseName());
497  item->setToolTip(file);
498  }
499 
500  BitmapFactory().addPath(path);
501  }
502  }
503  }
504 }
