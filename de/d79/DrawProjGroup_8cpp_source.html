310 {
311  // Base::Console().Message("DPG::getRect - views: %d\n", Views.getValues().size());

327 //find area consumed by Views only - scaled or unscaled
328 void DrawProjGroup::getViewArea(std::array<DrawProjGroupItem*, MAXPROJECTIONCOUNT>& viewPtrs,
329  double& width, double& height, bool scaled) const
330 {
331  // Get the child view bounding boxes
332  std::array<Base::BoundBox3d, MAXPROJECTIONCOUNT> bboxes;
333  makeViewBbs(viewPtrs, bboxes, scaled);
334 
335  //TODO: note that TLF/TRF/BLF, BRF extend a bit farther than a strict row/col arrangement would suggest.
336  //get widest view in each row/column
337  double col0w =
338  std::max(std::max(bboxes[0].LengthX(), bboxes[3].LengthX()), bboxes[7].LengthX()),
339  col1w =
340  std::max(std::max(bboxes[1].LengthX(), bboxes[4].LengthX()), bboxes[8].LengthX()),
341  col2w =
342  std::max(std::max(bboxes[2].LengthX(), bboxes[5].LengthX()), bboxes[9].LengthX()),
343  col3w = bboxes[6].LengthX(),
344  row0h =
345  std::max(std::max(bboxes[0].LengthY(), bboxes[1].LengthY()), bboxes[2].LengthY()),
346  row1h = std::max(std::max(bboxes[3].LengthY(), bboxes[4].LengthY()),
347  std::max(bboxes[5].LengthY(), bboxes[6].LengthY())),
348  row2h =
349  std::max(std::max(bboxes[7].LengthY(), bboxes[8].LengthY()), bboxes[9].LengthY());
350 
351  width = col0w + col1w + col2w + col3w;
352  height = row0h + row1h + row2h;

601  // Base::Console().Message("DPG::getXYPosition(%s)\n", Label.getValue());
602  // Third Angle: FTL T FTRight 0 1 2

607  // Right F L Rear 3 4 5 6
608  // FTRight T FTL 7 8 9
609

611  if (strcmp(viewTypeCStr, "Front") == 0) {// Front!

613  }
614  const int idxCount = MAXPROJECTIONCOUNT;

616  arrangeViewPointers(viewPtrs);
617  int viewIndex = getViewIndex(viewTypeCStr);
618 
619  //TODO: bounding boxes do not take view orientation into account
620  // i.e. X&Y widths might be swapped on page
621 
622  if (viewPtrs[viewIndex]->LockPosition.getValue() || !AutoDistribute.getValue()) {
