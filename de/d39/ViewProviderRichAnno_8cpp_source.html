144 void ViewProviderRichAnno::handleChangedPropertyType(Base::XMLReader &reader, const char *TypeName, App::Property *prop)
145 // transforms properties that had been changed
146 {
147  // property LineWidth had App::PropertyFloat and was changed to App::PropertyLength
148  if (prop == &LineWidth && strcmp(TypeName, "App::PropertyFloat") == 0) {
149  App::PropertyFloat LineWidthProperty;
150  // restore the PropertyFloat to be able to set its value
151  LineWidthProperty.Restore(reader);
152  LineWidth.setValue(LineWidthProperty.getValue());
153  }
154 
155  // property LineStyle had App::PropertyInteger and was changed to App::PropertyIntegerConstraint
156  else if (prop == &LineStyle && strcmp(TypeName, "App::PropertyInteger") == 0) {
157  App::PropertyInteger LineStyleProperty;
158  // restore the PropertyInteger to be able to set its value
159  LineStyleProperty.Restore(reader);
160  LineStyle.setValue(LineStyleProperty.getValue());
161  }
162 
163  // property LineStyle had App::PropertyIntegerConstraint and was changed to App::PropertyEnumeration
164  else if (prop == &LineStyle && strcmp(TypeName, "App::PropertyIntegerConstraint") == 0) {
165  App::PropertyIntegerConstraint LineStyleProperty;
166  // restore the PropertyIntegerConstraint to be able to set its value
167  LineStyleProperty.Restore(reader);
168  LineStyle.setValue(LineStyleProperty.getValue());
169  }
170 
171  else {
172  ViewProviderDrawingView::handleChangedPropertyType(reader, TypeName, prop);
173  }
174 }

187 std::vector<App::DocumentObject*> ViewProviderRichAnno::claimChildren() const
188 {
189  // What can reasonably have a RichAnno as a parent? A leader? a bit unconventional, but not forbidden.
190  // Another RichAnno? Maybe? Balloons? Dimensions? This is a bit of a corner case. Typically, a
191  // RichAnno would belong to something rather than owning something.
192 
193  std::vector<App::DocumentObject*> temp;
194  const std::vector<App::DocumentObject*>& candidates = getViewObject()->getInList();
195  for (auto& obj : candidates) {
196  if (obj->isDerivedFrom<TechDraw::DrawViewBalloon>() ||
197  obj->isDerivedFrom<TechDraw::DrawLeaderLine>() ||
198  obj->isDerivedFrom<TechDraw::DrawRichAnno>() ||
199  obj->isDerivedFrom<TechDraw::DrawViewDimension>() ) {
200  temp.push_back(obj);
201  }
202  }
203  return temp;
204 }
