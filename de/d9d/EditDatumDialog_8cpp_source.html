73 int EditDatumDialog::exec(bool atCursor)
74 {
75  // Return if constraint doesn't have editable value
76  if (Constr->isDimensional()) {
77 
78  if (sketch->hasConflicts()) {
79  Gui::TranslatedUserWarning(sketch,
80  QObject::tr("Dimensional constraint"),
81  QObject::tr("Not allowed to edit the datum because the "
82  "sketch contains conflicting constraints"));
83  return QDialog::Rejected;
84  }
85 
86  Base::Quantity init_val;
87 
88  QDialog dlg(Gui::getMainWindow());
89  if (!ui_ins_datum) {
90  ui_ins_datum.reset(new Ui_InsertDatum);
91  ui_ins_datum->setupUi(&dlg);
92  }
93  double datum = Constr->getValue();
94 
95  ui_ins_datum->labelEdit->setEntryName(QByteArray("DatumValue"));
96  if (Constr->Type == Sketcher::Angle) {
97  datum = Base::toDegrees<double>(datum);
98  dlg.setWindowTitle(tr("Insert angle"));
99  init_val.setUnit(Base::Unit::Angle);
100  ui_ins_datum->label->setText(tr("Angle:"));
101  ui_ins_datum->labelEdit->setParamGrpPath(
102  QByteArray("User parameter:BaseApp/History/SketcherAngle"));
103  }
104  else if (Constr->Type == Sketcher::Radius) {
105  dlg.setWindowTitle(tr("Insert radius"));
106  init_val.setUnit(Base::Unit::Length);
107  ui_ins_datum->label->setText(tr("Radius:"));
108  ui_ins_datum->labelEdit->setParamGrpPath(
109  QByteArray("User parameter:BaseApp/History/SketcherLength"));
110  }
111  else if (Constr->Type == Sketcher::Diameter) {
112  dlg.setWindowTitle(tr("Insert diameter"));
113  init_val.setUnit(Base::Unit::Length);
114  ui_ins_datum->label->setText(tr("Diameter:"));
115  ui_ins_datum->labelEdit->setParamGrpPath(
116  QByteArray("User parameter:BaseApp/History/SketcherLength"));
117  }
118  else if (Constr->Type == Sketcher::Weight) {
119  dlg.setWindowTitle(tr("Insert weight"));
120  ui_ins_datum->label->setText(tr("Weight:"));
121  ui_ins_datum->labelEdit->setParamGrpPath(
122  QByteArray("User parameter:BaseApp/History/SketcherWeight"));
123  }
124  else if (Constr->Type == Sketcher::SnellsLaw) {
125  dlg.setWindowTitle(tr("Refractive index ratio", "Constraint_SnellsLaw"));
126  ui_ins_datum->label->setText(tr("Ratio n2/n1:", "Constraint_SnellsLaw"));
127  ui_ins_datum->labelEdit->setParamGrpPath(
128  QByteArray("User parameter:BaseApp/History/SketcherRefrIndexRatio"));
129  ui_ins_datum->labelEdit->setSingleStep(0.05);
130  }
131  else {
132  dlg.setWindowTitle(tr("Insert length"));
133  init_val.setUnit(Base::Unit::Length);
134  ui_ins_datum->label->setText(tr("Length:"));
135  ui_ins_datum->labelEdit->setParamGrpPath(
136  QByteArray("User parameter:BaseApp/History/SketcherLength"));
137  }
138 
139  init_val.setValue(datum);
140 
141  ui_ins_datum->labelEdit->setValue(init_val);
142  ui_ins_datum->labelEdit->pushToHistory();
143  ui_ins_datum->labelEdit->selectNumber();
144  ui_ins_datum->labelEdit->bind(sketch->Constraints.createPath(ConstrNbr));
145  ui_ins_datum->name->setText(QString::fromStdString(Constr->Name));
146 
147  ui_ins_datum->cbDriving->setChecked(!Constr->isDriving);
148 
149  connect(ui_ins_datum->cbDriving,
150  &QCheckBox::toggled,
151  this,
152  &EditDatumDialog::drivingToggled);
153  connect(ui_ins_datum->labelEdit,
154  qOverload<const Base::Quantity&>(&Gui::QuantitySpinBox::valueChanged),
155  this,
156  &EditDatumDialog::datumChanged);
157  connect(ui_ins_datum->labelEdit,
158  &Gui::QuantitySpinBox::showFormulaDialog,
159  this,
160  &EditDatumDialog::formEditorOpened);
161  connect(&dlg, &QDialog::accepted, this, &EditDatumDialog::accepted);
162  connect(&dlg, &QDialog::rejected, this, &EditDatumDialog::rejected);
163 
164  if (atCursor) {
165  dlg.show(); // Need to show the dialog so geometry is computed
166  QRect pg = dlg.parentWidget()->geometry();
167  int Xmin = pg.x() + 10;
168  int Ymin = pg.y() + 10;
169  int Xmax = pg.x() + pg.width() - dlg.geometry().width() - 10;
170  int Ymax = pg.y() + pg.height() - dlg.geometry().height() - 10;
171  int x = Xmax < Xmin ? (Xmin + Xmax) / 2
172  : std::min(std::max(QCursor::pos().x(), Xmin), Xmax);
173  int y = Ymax < Ymin ? (Ymin + Ymax) / 2
174  : std::min(std::max(QCursor::pos().y(), Ymin), Ymax);
175  dlg.setGeometry(x, y, dlg.geometry().width(), dlg.geometry().height());
176  }
177 
178  return dlg.exec();
179  }
180 
181  return QDialog::Rejected;
182 }

184 void EditDatumDialog::accepted()
185 {
186  Base::Quantity newQuant = ui_ins_datum->labelEdit->value();
187  if (newQuant.isQuantity() || (Constr->Type == Sketcher::SnellsLaw && newQuant.isDimensionless())
188  || (Constr->Type == Sketcher::Weight && newQuant.isDimensionless())) {
189 
190  // save the value for the history
191  ui_ins_datum->labelEdit->pushToHistory();
192 
193  double newDatum = newQuant.getValue();
194 
195  try {
196 
197  /*if (ui_ins_datum->cbDriving->isChecked() == Constr->isDriving) {
198  Gui::cmdAppObjectArgs(sketch, "toggleDriving(%i)", ConstrNbr);
199  }*/
200 
201  if (!ui_ins_datum->cbDriving->isChecked()) {
202  if (ui_ins_datum->labelEdit->hasExpression()) {
203  ui_ins_datum->labelEdit->apply();
204  }
205  else {
206  auto unitString = newQuant.getUnit().getString();
207  unitString = Base::Tools::escapeQuotesFromString(unitString);
208  Gui::cmdAppObjectArgs(sketch,
209  "setDatum(%i,App.Units.Quantity('%f %s'))",
210  ConstrNbr,
211  newDatum,
212  unitString);
213  }
214  }
215 
216  QString constraintName = ui_ins_datum->name->text().trimmed();
217  if (constraintName.toStdString() != sketch->Constraints[ConstrNbr]->Name) {
218  std::string escapedstr =
219  Base::Tools::escapedUnicodeFromUtf8(constraintName.toUtf8().constData());
220  escapedstr = Base::Tools::escapeQuotesFromString(escapedstr);
221  Gui::cmdAppObjectArgs(sketch,
222  "renameConstraint(%d, u'%s')",
223  ConstrNbr,
224  escapedstr.c_str());
225  }
226 
227  Gui::Command::commitCommand();
228 
229  // THIS IS A WORK-AROUND NOT TO DELAY 0.19 RELEASE
230  //
231  // depsAreTouched is not returning true in this case:
232  // https://forum.freecad.org/viewtopic.php?f=3&t=55633&p=481061#p478477
233  //
234  // It appears related to a drastic change in how dependencies are calculated, see:
235  // https://forum.freecad.org/viewtopic.php?f=3&t=55633&p=481061#p481061
236  //
237  // This is NOT the solution, as there is no point in systematically executing the
238  // ExpressionEngine on every dimensional constraint change. Just a quick fix to avoid
239  // clearly unwanted behaviour in absence of time to actually fix the root cause.
240 
241  // if (sketch->noRecomputes && sketch->ExpressionEngine.depsAreTouched()) {
242  sketch->ExpressionEngine.execute();
243  sketch->solve();
244  //}
245 
246  tryAutoRecompute(sketch);
247  success = true;
248  }
249  catch (const Base::Exception& e) {
250  Gui::NotifyUserError(sketch,
251  QT_TRANSLATE_NOOP("Notifications", "Value Error"),
252  e.what());
253 
254  Gui::Command::abortCommand();
255 
256  if (sketch->noRecomputes) { // if setdatum failed, it is highly likely that solver
257  // information is invalid.
258  sketch->solve();
259  }
260  }
261  }
262 }
