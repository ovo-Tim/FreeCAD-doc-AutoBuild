51 App::DocumentObjectExecReturn* Cut::execute()
52 {
53  // Perform error checking
54 
55  try {
56  std::vector<App::DocumentObject*> shapes = ShapeList.getValues();
57  if (shapes.size() != 2) {
58  return new App::DocumentObjectExecReturn(
59  "Two shapes must be entered at a time for a cut operation");
60  }
61 
62  Part::TopoShape ts1;
63  Part::TopoShape ts2;
64 
65  // Get first toposhape
66  if (shapes[0]->isDerivedFrom<Part::Feature>()) {
67  ts1 = static_cast<Part::Feature*>(shapes[0])->Shape.getShape(); // Part::TopoShape 1
68  }
69  else {
70  return new App::DocumentObjectExecReturn("Shape1 not from Part::Feature");
71  }
72 
73  // Get second toposhape
74  if (shapes[1]->isDerivedFrom<Part::Feature>()) {
75  ts2 = static_cast<Part::Feature*>(shapes[1])->Shape.getShape();
76  }
77  else {
78  return new App::DocumentObjectExecReturn("Shape2 not from Part::Feature");
79  }
80 
81  // Cut Shape1 by Shape2
82  TopoDS_Shape aCutShape;
83  aCutShape = ts1.cut(ts2.getShape());
84 
85  // Check if resulting shell is null
86  if (aCutShape.IsNull()) {
87  return new App::DocumentObjectExecReturn("Resulting shape is null");
88  }
89 
90  this->Shape.setValue(aCutShape);
91  return nullptr;
92  }
93  catch (Standard_Failure& e) {
94 
95  return new App::DocumentObjectExecReturn(e.GetMessageString());
96  }
97 }
