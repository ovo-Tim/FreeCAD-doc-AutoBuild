85  Gui::View3DInventor* view = static_cast<Gui::View3DInventor*>(doc->getActiveView());
86  if (view) {

95  InspectionGui::ViewProviderInspection::inspectCallback);

99 bool CmdInspectElement::isActive()
100 {
101  App::Document* doc = App::GetApplication().getActiveDocument();
102  if (!doc || doc->countObjectsOfType<Inspection::Feature>() == 0) {
103  return false;
104  }
105 
106  Gui::MDIView* view = Gui::getMainWindow()->activeWindow();
107  if (view && view->isDerivedFrom(Gui::View3DInventor::getClassTypeId())) {
108  Gui::View3DInventorViewer* viewer = static_cast<Gui::View3DInventor*>(view)->getViewer();
109  return !viewer->isEditing();
110  }
111 
112  return false;
113 }
