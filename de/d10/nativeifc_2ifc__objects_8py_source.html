52  def onChanged(self, obj, prop):
53  # link class property to its hidder IfcClass counterpart
54  if prop == "IfcClass" and hasattr(obj, "Class") and obj.Class != obj.IfcClass:
55  obj.Class = obj.IfcClass

59  self.rebuild_classlist(obj, setprops=True)
60  elif prop == "Schema":

62  elif prop == "Type":
63  self.Classification(obj)
64  elif prop == "Classification":
65  self.edit_classification(obj)
66  elif prop == "Group":
67  self.edit_group(obj)
68  elif hasattr(obj, prop) and obj.getGroupOfProperty(prop) == "IFC":
69  if prop not in ["StepId"]:
70  self.edit_attribute(obj, prop)
71  elif prop == "Label":
72  self.edit_attribute(obj, "Name", obj.Label)
73  elif prop == "Text":
74  self.edit_annotation(obj, "Text", "\n".join(obj.Text))
75  elif prop in ["Start", "End"]:
76  self.edit_annotation(obj, prop)
77  elif prop in ["DisplayLength","DisplayHeight","Depth"]:
78  self.edit_annotation(obj, prop)
79  elif prop == "Placement":
80  if getattr(self, "virgin_placement", False):
81  self.virgin_placement = False
82  elif obj.Placement != getattr(self, "old_placement", None):
83  # print("placement changed for",obj.Label,"to",obj.Placement)
84  self.edit_placement(obj)
85  elif prop == "Modified":
86  if obj.ViewObject:
87  obj.ViewObject.signalChangeIcon()
88  elif hasattr(obj, prop) and obj.getGroupOfProperty(prop) == "Geometry":
89  self.edit_geometry(obj, prop)
90  elif hasattr(obj, prop) and obj.getGroupOfProperty(prop) == "Quantities":
91  self.edit_quantity(obj, prop)
92  elif hasattr(obj, prop) and obj.getGroupOfProperty(prop) not in NON_PSETS:
93  # Treat all property groups outside the default ones as Psets
94  # print("DEBUG: editinog pset prop",prop)
95  self.edit_pset(obj, prop)

182  def edit_annotation(self, obj, attribute, value=None):
183  """Edits an attribute of an underlying IFC annotation"""
184 
185  from nativeifc import ifc_tools # lazy import
186  from nativeifc import ifc_export
187 
188  if not value:
189  if hasattr(obj, attribute):
190  value = obj.getPropertyByName(attribute)
191  ifcfile = ifc_tools.get_ifcfile(obj)
192  elt = ifc_tools.get_ifc_element(obj, ifcfile)
193  if elt:
194  if attribute == "Text":
195  text = ifc_export.get_text(elt)
196  if text:
197  ifc_tools.set_attribute(ifcfile, text, "Literal", value)
198  elif attribute in ["Start", "End"]:

200  if dim:
201  rep = dim[0]

205  continue

208  points = sub.Points
209  value = list(points.CoordList)
210  is2d = "2D" in points.is_a()
211  if attribute == "Start":
212  value[0] = ifc_export.get_scaled_point(obj.Start, ifcfile, is2d)
213  else:
214  value[-1] = ifc_export.get_scaled_point(obj.End, ifcfile, is2d)
215  ifc_tools.set_attribute(ifcfile, points, "CoordList", value)
216  else:
217  print("DEBUG: unknown dimension curve type:",sub)
218  elif attribute in ["DisplayLength","DisplayHeight","Depth"]:
219  l = w = h = 1000.0
220  if obj.ViewObject:
221  if obj.ViewObject.DisplayLength.Value:
222  l = ifc_export.get_scaled_value(obj.ViewObject.DisplayLength.Value, ifcfile)
223  if obj.ViewObject.DisplayHeight.Value:
224  w = ifc_export.get_scaled_value(obj.ViewObject.DisplayHeight.Value, ifcfile)
225  if obj.Depth.Value:
226  h = ifc_export.get_scaled_value(obj.Depth.Value, ifcfile)
227  if elt.Representation.Representations:
228  for rep in elt.Representation.Representations:
229  for item in rep.Items:
230  if item.is_a("IfcCsgSolid"):
231  if item.TreeRootExpression.is_a("IfcBlock"):
232  block = item.TreeRootExpression
233  loc = block.Position.Location
234  ifc_tools.set_attribute(ifcfile, block, "XLength", l)
235  ifc_tools.set_attribute(ifcfile, block, "YLength", w)
236  ifc_tools.set_attribute(ifcfile, block, "ZLength", h)
237  ifc_tools.set_attribute(ifcfile, loc, "Coordinates", (-l/2, -h/2, -h))
