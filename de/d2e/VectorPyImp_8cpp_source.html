84  catch (const Py::Exception&) {
85  return -1;
86  }
87  }

89  PyErr_SetString(PyExc_TypeError, "Either three floats, tuple or Vector expected");

227 // http://renesd.blogspot.de/2009/07/python3-c-api-simple-slicing-sqslice.html
228 PyObject* VectorPy::mapping_subscript(PyObject* self, PyObject* item)
229 {
230  if (PyIndex_Check(item)) {
231  Py_ssize_t i = PyNumber_AsSsize_t(item, PyExc_IndexError);
232  if (i == -1 && PyErr_Occurred()) {
233  return nullptr;
234  }
235  if (i < 0) {
236  i += sequence_length(self);
237  }
238  return sequence_item(self, i);
239  }
240  if (PySlice_Check(item)) {
241  Py_ssize_t start = 0, stop = 0, step = 0, slicelength = 0, cur = 0, i = 0;
242  PyObject* slice = item;
243 
244  if (PySlice_GetIndicesEx(slice, sequence_length(self), &start, &stop, &step, &slicelength)
245  < 0) {
246  return nullptr;
247  }
248 
249  if (slicelength <= 0) {
250  return PyTuple_New(0);
251  }
252  if (start == 0 && step == 1 && slicelength == sequence_length(self)
253  && PyObject_TypeCheck(self, &(VectorPy::Type))) {
254  Base::Vector3d v = static_cast<VectorPy*>(self)->value();
255  Py::Tuple xyz(3);
256  xyz.setItem(0, Py::Float(v.x));
257  xyz.setItem(1, Py::Float(v.y));
258  xyz.setItem(2, Py::Float(v.z));
259  return Py::new_reference_to(xyz);
260  }
261  if (PyObject_TypeCheck(self, &(VectorPy::Type))) {
262  Base::Vector3d v = static_cast<VectorPy*>(self)->value();
263  Py::Tuple xyz(static_cast<size_t>(slicelength));
264 
265  for (cur = start, i = 0; i < slicelength; cur += step, i++) {
266  unsigned short pos = cur % 3;
267  xyz.setItem(static_cast<Py::sequence_index_type>(i), Py::Float(v[pos]));
268  }
269 
270  return Py::new_reference_to(xyz);
271  }
272  }
273 
274  PyErr_Format(PyExc_TypeError,
275  "Vector indices must be integers or slices, not %.200s",
276  Py_TYPE(item)->tp_name);
277  return nullptr;
278 }

323 PyObject* VectorPy::richCompare(PyObject* v, PyObject* w, int op)
324 {
325  if (PyObject_TypeCheck(v, &(VectorPy::Type)) && PyObject_TypeCheck(w, &(VectorPy::Type))) {
326  Vector3d v1 = static_cast<VectorPy*>(v)->value();
327  Vector3d v2 = static_cast<VectorPy*>(w)->value();
328 
329  PyObject* res = nullptr;
330  if (op != Py_EQ && op != Py_NE) {
331  PyErr_SetString(PyExc_TypeError, "no ordering relation is defined for Vector");
332  return nullptr;
333  }
334  if (op == Py_EQ) {
335  res = (v1 == v2) ? Py_True : Py_False; // NOLINT
336  Py_INCREF(res);
337  return res;
338  }
339  res = (v1 != v2) ? Py_True : Py_False; // NOLINT
340  Py_INCREF(res);
341  return res;
342  }
343  // This always returns False
344  Py_INCREF(Py_NotImplemented);
345  return Py_NotImplemented;
346 }
