89 void CmdMeshPartTrimByPlane::activated(int)
90 {
91  Base::Type partType = Base::Type::fromName("Part::Plane");
92  std::vector<App::DocumentObject*> plane = getSelection().getObjectsOfType(partType);
93  if (plane.empty()) {
94  QMessageBox::warning(Gui::getMainWindow(),
95  qApp->translate("MeshPart_TrimByPlane", "Select plane"),
96  qApp->translate("MeshPart_TrimByPlane",
97  "Please select a plane at which you trim the mesh."));
98  return;
99  }
100 
101  QMessageBox msgBox(Gui::getMainWindow());
102  msgBox.setIcon(QMessageBox::Question);
103  msgBox.setWindowTitle(qApp->translate("MeshPart_TrimByPlane", "Trim by plane"));
104  msgBox.setText(qApp->translate("MeshPart_TrimByPlane", "Select the side you want to keep."));
105  QPushButton* inner =
106  msgBox.addButton(qApp->translate("MeshPart_TrimByPlane", "Below"), QMessageBox::ActionRole);
107  QPushButton* outer =
108  msgBox.addButton(qApp->translate("MeshPart_TrimByPlane", "Above"), QMessageBox::ActionRole);
109  QPushButton* split =
110  msgBox.addButton(qApp->translate("MeshPart_TrimByPlane", "Split"), QMessageBox::ActionRole);
111  msgBox.addButton(QMessageBox::Cancel);
112  msgBox.setDefaultButton(inner);
113  msgBox.exec();
114  QAbstractButton* click = msgBox.clickedButton();
115 
116  Gui::SelectionRole role;
117  if (inner == click) {
118  role = Gui::SelectionRole::Inner;
119  }
120  else if (outer == click) {
121  role = Gui::SelectionRole::Outer;
122  }
123  else if (split == click) {
124  role = Gui::SelectionRole::Split;
125  }
126  else {
127  // abort
128  return;
129  }
130 
131  Base::Placement plnPlacement =
132  static_cast<App::GeoFeature*>(plane.front())->Placement.getValue();
133 
134  openCommand(QT_TRANSLATE_NOOP("Command", "Trim with plane"));
135  std::vector<App::DocumentObject*> docObj =
136  Gui::Selection().getObjectsOfType(Mesh::Feature::getClassTypeId());
137  for (auto it : docObj) {
138  Base::Vector3d normal(0, 0, 1);
139  plnPlacement.getRotation().multVec(normal, normal);
140  Base::Vector3d base = plnPlacement.getPosition();
141 
142  Mesh::MeshObject* mesh = static_cast<Mesh::Feature*>(it)->Mesh.startEditing();
143 
144  Base::Vector3f plnBase = Base::convertTo<Base::Vector3f>(base);
145  Base::Vector3f plnNormal = Base::convertTo<Base::Vector3f>(normal);
146 
147  if (role == Gui::SelectionRole::Inner) {
148  mesh->trimByPlane(plnBase, plnNormal);
149  static_cast<Mesh::Feature*>(it)->Mesh.finishEditing();
150  }
151  else if (role == Gui::SelectionRole::Outer) {
152  mesh->trimByPlane(plnBase, -plnNormal);
153  static_cast<Mesh::Feature*>(it)->Mesh.finishEditing();
154  }
155  else if (role == Gui::SelectionRole::Split) {
156  Mesh::MeshObject copy(*mesh);
157  mesh->trimByPlane(plnBase, plnNormal);
158  static_cast<Mesh::Feature*>(it)->Mesh.finishEditing();
159 
160  copy.trimByPlane(plnBase, -plnNormal);
161  App::Document* doc = it->getDocument();
162  Mesh::Feature* fea = static_cast<Mesh::Feature*>(doc->addObject("Mesh::Feature"));
163  fea->Label.setValue(it->Label.getValue());
164  Mesh::MeshObject* feamesh = fea->Mesh.startEditing();
165  feamesh->swap(copy);
166  fea->Mesh.finishEditing();
167  }
168  it->purgeTouched();
169  }
170  commitCommand();
171 }

195 void CmdMeshPartSection::activated(int)
196 {
197  Base::Type partType = Base::Type::fromName("Part::Plane");
198  std::vector<App::DocumentObject*> plane = getSelection().getObjectsOfType(partType);
199  if (plane.empty()) {
200  QMessageBox::warning(
201  Gui::getMainWindow(),
202  qApp->translate("MeshPart_Section", "Select plane"),
203  qApp->translate("MeshPart_Section",
204  "Please select a plane at which you section the mesh."));
205  return;
206  }
207 
208  Base::Placement plm = static_cast<App::GeoFeature*>(plane.front())->Placement.getValue();
209  Base::Vector3d normal(0, 0, 1);
210  plm.getRotation().multVec(normal, normal);
211  Base::Vector3d base = plm.getPosition();
212 
213  openCommand(QT_TRANSLATE_NOOP("Command", "Section with plane"));
214  std::vector<App::DocumentObject*> docObj =
215  Gui::Selection().getObjectsOfType(Mesh::Feature::getClassTypeId());
216  Mesh::MeshObject::TPlane tplane;
217  tplane.first = Base::convertTo<Base::Vector3f>(base);
218  tplane.second = Base::convertTo<Base::Vector3f>(normal);
219  std::vector<Mesh::MeshObject::TPlane> sections;
220  sections.push_back(tplane);
221 
222  Py::Module partModule(PyImport_ImportModule("Part"), true);
223  Py::Callable makeWire(partModule.getAttr("makePolygon"));
224  Py::Module appModule(PyImport_ImportModule("FreeCAD"), true);
225  Py::Callable addObject(appModule.getAttr("ActiveDocument").getAttr("addObject"));
226  for (auto it : docObj) {
227  const Mesh::MeshObject* mesh = static_cast<Mesh::Feature*>(it)->Mesh.getValuePtr();
228  std::vector<Mesh::MeshObject::TPolylines> polylines;
229  const float minSectionLength = 1e-7F;
230  mesh->crossSections(sections, polylines, minSectionLength);
231 
232  for (const auto& it2 : polylines) {
233  for (const auto& it3 : it2) {
234  Py::Tuple arg(1);
235  Py::List list;
236  for (auto it4 : it3) {
237  Py::Tuple pnt(3);
238  pnt.setItem(0, Py::Float(it4.x));
239  pnt.setItem(1, Py::Float(it4.y));
240  pnt.setItem(2, Py::Float(it4.z));
241  list.append(pnt);
242  }
243  arg.setItem(0, list);
244  Py::Object wire = makeWire.apply(arg);
245 
246  Py::Tuple arg2(2);
247  arg2.setItem(0, Py::String("Part::Feature"));
248  arg2.setItem(1, Py::String("Section"));
249  Py::Object obj = addObject.apply(arg2);
250  obj.setAttr("Shape", wire);
251  }
252  }
253  }
254  updateActive();
255  commitCommand();
256 }
