61 App::DocumentObjectExecReturn* SetOperations::execute()
62 {
63  Mesh::Feature* mesh1 = dynamic_cast<Mesh::Feature*>(Source1.getValue());
64  Mesh::Feature* mesh2 = dynamic_cast<Mesh::Feature*>(Source2.getValue());
65 
66  if (mesh1 && mesh2) {
67  const MeshObject& meshKernel1 = mesh1->Mesh.getValue();
68  const MeshObject& meshKernel2 = mesh2->Mesh.getValue();
69 
70  std::unique_ptr<MeshObject> pcKernel(new MeshObject()); // Result Meshkernel
71 
72  MeshCore::SetOperations::OperationType type {};
73  string ot(OperationType.getValue());
74  if (ot == "union") {
75  type = MeshCore::SetOperations::Union;
76  }
77  else if (ot == "intersection") {
78  type = MeshCore::SetOperations::Intersect;
79  }
80  else if (ot == "difference") {
81  type = MeshCore::SetOperations::Difference;
82  }
83  else if (ot == "inner") {
84  type = MeshCore::SetOperations::Inner;
85  }
86  else if (ot == "outer") {
87  type = MeshCore::SetOperations::Outer;
88  }
89  else {
90  throw Base::ValueError("Operation type must either be 'union' or 'intersection'"
91  " or 'difference' or 'inner' or 'outer'");
92  }
93 
94  MeshCore::SetOperations setOp(meshKernel1.getKernel(),
95  meshKernel2.getKernel(),
96  pcKernel->getKernel(),
97  type,
98  1.0e-5F);
99  setOp.Do();
100  Mesh.setValuePtr(pcKernel.release());
101  }
102  else {
103  // Error mesh property
104  if (!mesh1) {
105  throw Base::ValueError("First input mesh not set");
106  }
107  if (!mesh2) {
108  throw Base::ValueError("Second input mesh not set");
109  }
110  }
111 
112  return App::DocumentObject::StdReturn;
113 }
