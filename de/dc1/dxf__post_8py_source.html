89 def parse(pathobj):
90  """accepts a Path object. Returns a list of wires"""
91 
92  feedcommands = Path.Geom.CmdMove
93  rapidcommands = Path.Geom.CmdMoveRapid
94 
95  edges = []
96  objlist = []
97 
98  # Gotta start somewhere. Assume 0,0,0
99  curPoint = FreeCAD.Vector(0, 0, 0)
100  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
101  Path.Log.debug("{} -> {}".format(curPoint, c))
102  if "Z" in c.Parameters:
103  newparams = c.Parameters
104  newparams.pop("Z", None)
105  flatcommand = Path.Command(c.Name, newparams)
106  c.Parameters = newparams
107  else:
108  flatcommand = c
109 
110  # ignore gcode that isn't moving
111  if flatcommand.Name not in feedcommands + rapidcommands:
112  Path.Log.debug("non move")
113  continue
114 
115  # ignore pure vertical feed and rapid
116  if (
117  flatcommand.Parameters.get("X", curPoint.x) == curPoint.x
118  and flatcommand.Parameters.get("Y", curPoint.y) == curPoint.y
119  ):
120  Path.Log.debug("vertical")
121  continue
122 
123  # feeding move. Build an edge
124  if flatcommand.Name in feedcommands:
125  edges.append(Path.Geom.edgeForCmd(flatcommand, curPoint))
126  Path.Log.debug("feeding move")
127 
128  # update the curpoint
129  curPoint.x = flatcommand.Parameters.get("X", curPoint.x)
130  curPoint.y = flatcommand.Parameters.get("Y", curPoint.y)
131 
132  if len(edges) > 0:
133  candidates = Part.sortEdges(edges)
134  for c in candidates:
135  obj = FreeCAD.ActiveDocument.addObject("Part::Feature", "Wire")
136  obj.Shape = Part.Wire(c)
137  objlist.append(obj)
138 
139  return objlist
