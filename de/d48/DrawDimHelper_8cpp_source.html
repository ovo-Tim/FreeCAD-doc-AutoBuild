82 {

100  DrawPage* page = dvp->findParentPage();
101  std::string pageName = page->getNameInDocument();

136  dimExt->recomputeFeature();

138  return dimExt;

204 std::pair<Base::Vector3d, Base::Vector3d>
205 DrawDimHelper::minMax(DrawViewPart* dvp, std::vector<std::string> edgeNames, int direction)
206 {
207  // Base::Console().Message("DDH::minMax() - edgeName: %d\n", edgeNames.size());
208  std::pair<Base::Vector3d, Base::Vector3d> result;
209  Base::Vector3d refMin;
210  Base::Vector3d refMax;
211 
212  gp_Ax3 projAx3;// OXYZ
213  gp_Pln projPlane(projAx3);
214 
215  BaseGeomPtrVector edgeGeomList;
216  if (!edgeNames.empty() && !edgeNames.front().empty()) {
217  //we have edge names and the first one isn't null
218  for (auto& n : edgeNames) {
219  std::string geomType = DrawUtil::getGeomTypeFromName(n);
220  if (geomType == "Edge") {
221  int i = DrawUtil::getIndexFromName(n);
222  BaseGeomPtr bg = dvp->getGeomByIndex(i);
223  if (bg) {
224  edgeGeomList.push_back(bg);
225  }
226  }
227  }
228  }
229  else {
230  for (auto& edge : dvp->getEdgeGeometry()) {
231  if (!edge->getCosmetic()) {
232  // skip cosmetic edges
233  edgeGeomList.push_back(edge);
234  }
235  }
236  }
237 
238  if (edgeGeomList.empty()) {
239  return result;
240  }
241 
242  Bnd_Box edgeBbx;
243  edgeBbx.SetGap(1.0);//make the box a bit bigger
244 
245  std::vector<TopoDS_Edge> inEdges;
246  for (auto& bg : edgeGeomList) {
247  inEdges.push_back(bg->getOCCEdge());
248  BRepBndLib::Add(bg->getOCCEdge(), edgeBbx);
249  }
250 
251  double minX, minY, minZ, maxX, maxY, maxZ;
252  edgeBbx.Get(minX, minY, minZ, maxX, maxY, maxZ);
253  double xMid = (maxX + minX) / 2.0;
254  double yMid = (maxY + minY) / 2.0;
255 
256  gp_Pnt rightMid(maxX, yMid, 0.0);
257  gp_Pnt leftMid(minX, yMid, 0.0);
258  gp_Pnt topMid(xMid, maxY, 0.0);
259  gp_Pnt bottomMid(xMid, minY, 0.0);
260 
261  gp_Dir xDir(1.0, 0.0, 0.0);
262  gp_Dir yDir(0.0, 1.0, 0.0);
263 
264  if (direction == HORIZONTAL) {
265  Handle(Geom_Line) lineLeft = new Geom_Line(leftMid, yDir);
266  BRepBuilderAPI_MakeEdge mkEdgeLeft(lineLeft);
267  TopoDS_Edge edgeLeft = mkEdgeLeft.Edge();
268  gp_Pnt leftPoint = findClosestPoint(inEdges, edgeLeft);
269  Handle(Geom_Line) lineRight = new Geom_Line(rightMid, yDir);
270  BRepBuilderAPI_MakeEdge mkEdgeRight(lineRight);
271  TopoDS_Edge edgeRight = mkEdgeRight.Edge();
272  gp_Pnt rightPoint = findClosestPoint(inEdges, edgeRight);
273 
274  refMin = Base::Vector3d(leftPoint.X(), leftPoint.Y(), 0.0);
275  refMax = Base::Vector3d(rightPoint.X(), rightPoint.Y(), 0.0);
276  }
277  else if (direction == VERTICAL) {
278  Handle(Geom_Line) lineBottom = new Geom_Line(bottomMid, xDir);
279  BRepBuilderAPI_MakeEdge mkEdgeBottom(lineBottom);
280  TopoDS_Edge edgeBottom = mkEdgeBottom.Edge();
281  gp_Pnt bottomPoint = findClosestPoint(inEdges, edgeBottom);
282  Handle(Geom_Line) lineTop = new Geom_Line(topMid, xDir);
283  BRepBuilderAPI_MakeEdge mkEdgeTop(lineTop);
284  TopoDS_Edge edgeTop = mkEdgeTop.Edge();
285  gp_Pnt topPoint = findClosestPoint(inEdges, edgeTop);
286  refMin = Base::Vector3d(bottomPoint.X(), bottomPoint.Y(), 0.0);
287  refMax = Base::Vector3d(topPoint.X(), topPoint.Y(), 0.0);
288  }
289 
290  result.first = refMin;
291  result.second = refMax;
292  return result;
293 }

295 //given list of curves, find the closest point from any curve to a boundary
296 //computation intensive for a cosmetic result.
297 gp_Pnt DrawDimHelper::findClosestPoint(std::vector<TopoDS_Edge> inEdges, TopoDS_Edge& boundary)
298 {
299  // Base::Console().Message("DDH::findClosestPoint() - edges: %d\n", inEdges.size());
300  //
301  //find an extent point that is actually on one of the curves
302  double minDistance(std::numeric_limits<float>::max());
303  gp_Pnt nearPoint;
304  for (auto& edge : inEdges) {
305  BRepExtrema_DistShapeShape extss(edge, boundary);
306  if (!extss.IsDone()) {
307  Base::Console().Warning(
308  "DDH::findClosestPoint - BRepExtrema_DistShapeShape failed - 1\n");
309  continue;
310  }
311  if (extss.NbSolution() == 0) {
312  Base::Console().Warning(
313  "DDH::findClosestPoint - BRepExtrema_DistShapeShape failed - 2\n");
314  continue;
315  }
316  if (extss.Value() < minDistance) {
317  minDistance = extss.Value();
318  nearPoint = extss.PointOnShape1(1);
319  }
320  }
321  return nearPoint;
322 }

327  // Base::Console().Message("DDH::minMax3d() - references: %d\n", references.size());

332  gp_Ax3 projAx3;//OXYZ
333  gp_Pln projPlane(projAx3);
334 
335  BRep_Builder builder;
336  TopoDS_Compound comp;
337  builder.MakeCompound(comp);
338  for (auto& ref : references) {
339  auto tempGeom = ref.getGeometry();
340  if (tempGeom.IsNull()) {
341  continue;
342  }
343  builder.Add(comp, tempGeom);
344  }
345  Base::Vector3d centroid = dvp->getOriginalCentroid();
346  TopoDS_Shape centeredShape =//this result is a throw away. We will work with comp.

348

350  TechDraw::GeometryObjectPtr go(
351  std::make_shared<TechDraw::GeometryObject>(std::string(), nullptr));
352  go->setIsoCount(0);

380  gp_Pnt bottomMid(xMid, minY, 0.0);
381 
382  gp_Dir xDir(1.0, 0.0, 0.0);
383  gp_Dir yDir(0.0, 1.0, 0.0);
384 
385  if (direction == HORIZONTAL) {
386  Handle(Geom_Line) lineLeft = new Geom_Line(leftMid, yDir);
387  BRepBuilderAPI_MakeEdge mkEdgeLeft(lineLeft);
388  TopoDS_Edge edgeLeft = mkEdgeLeft.Edge();
389  gp_Pnt leftPoint = findClosestPoint(inEdges, edgeLeft);
390  Handle(Geom_Line) lineRight = new Geom_Line(rightMid, yDir);
391  BRepBuilderAPI_MakeEdge mkEdgeRight(lineRight);
392  TopoDS_Edge edgeRight = mkEdgeRight.Edge();
393  gp_Pnt rightPoint = findClosestPoint(inEdges, edgeRight);
394 
395  refMin = Base::Vector3d(leftPoint.X(), leftPoint.Y(), 0.0);
396  refMax = Base::Vector3d(rightPoint.X(), rightPoint.Y(), 0.0);
397  }
398  else if (direction == VERTICAL) {

400  BRepBuilderAPI_MakeEdge mkEdgeBottom(lineBottom);
401  TopoDS_Edge edgeBottom = mkEdgeBottom.Edge();
402  gp_Pnt bottomPoint = findClosestPoint(inEdges, edgeBottom);
403  Handle(Geom_Line) lineTop = new Geom_Line(topMid, xDir);

406  gp_Pnt topPoint = findClosestPoint(inEdges, edgeTop);
407  refMin = Base::Vector3d(bottomPoint.X(), bottomPoint.Y(), 0.0);
408  refMax = Base::Vector3d(topPoint.X(), topPoint.Y(), 0.0);

413  return result;
