291  {
292  _out << object << '\n';
293  return *this;
294  }

296  TextOutputStream& operator<<(const char* object)
297  {
298  std::string_view str(object);
299 
300  // Count the lines so that we can deal with potential EOL conversions by external software
301  uint32_t lineCount = 0;
302  for (const auto character : str) {
303  if (character == '\n') {
304  ++lineCount;
305  }
306  }
307  // Stores the line count followed by a colon as the delimiter. We don't use
308  // whitespace because the input stream may also start with whitespace.
309  _out << lineCount << ':';
310 
311  // Store the text, and normalize the end of line to a single '\n'.
312  bool foundSlashR = false;
313  for (const auto character : str) {
314  if (character == '\r') {
315  foundSlashR = true;
316  continue;
317  }
318  if (foundSlashR) {
319  if (character != '\n') {
320  // We allow '\r' if it is not at the end of a line
321  _out.put('\r');
322  }
323  foundSlashR = false; // Reset for next time
324  }
325  _out.put(character);
326  }
327 
328  // Add an extra newline as the delimiter for the following data.
329  _out.put('\n');
330  return *this;
331  }
332 
333  TextOutputStream& operator<<(const std::string& object)
334  {
335  return (*this) << object.c_str();
336  }
337 
338  TextOutputStream& operator<<(char object)
339  {
340  _out.put(object);
341  return *this;
342  }
343 
344  explicit operator bool() const
345  {
346  // test if _Ipfx succeeded
347  return !_out.eof();
348  }
349 
350 private:
351  std::ostream& _out;
352 };
