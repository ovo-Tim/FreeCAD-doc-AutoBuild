64 class GraphvizWorker : public QThread {
65  Q_OBJECT
66 public:
67  explicit GraphvizWorker(QObject * parent = nullptr)
68  : QThread(parent)
69  {
70  }
71 
72  ~GraphvizWorker() override
73  {
74  dotProc.moveToThread(this);
75  unflattenProc.moveToThread(this);
76  }
77 
78  void setData(const QByteArray & data)
79  {
80  str = data;
81  }
82 
83  void startThread() {
84  // This doesn't actually run a thread but calls the function
85  // directly in the main thread.
86  // This is needed because embedding a QProcess into a QThread
87  // causes some problems with Qt5.
88  run();
89  // Can't use the finished() signal of QThread
90  Q_EMIT emitFinished();
91  }
92 
93  void run() override {
94  QByteArray preprocessed = str;
95 
96  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/DependencyGraph");
97  if(hGrp->GetBool("Unflatten", true)) {
98  // Write data to unflatten process
99  unflattenProc.write(str);
100  unflattenProc.closeWriteChannel();
101  //no error handling: unflatten is optional
102  unflattenProc.waitForFinished();
103  preprocessed = unflattenProc.readAll();
104  } else {
105  unflattenProc.closeWriteChannel();
106  unflattenProc.waitForFinished();
107  }
108 
109  dotProc.write(preprocessed);
110  dotProc.closeWriteChannel();
111  if (!dotProc.waitForFinished()) {
112  Q_EMIT error();
113  quit();
114  }
115 
116  // Emit result; it will get queued for processing in the main thread
117  Q_EMIT svgFileRead(dotProc.readAll());
118  }
119 
120  QProcess * dotProcess() {
121  return &dotProc;
122  }
123 
124  QProcess * unflattenProcess() {
125  return &unflattenProc;
126  }
127 
128 Q_SIGNALS:
129  void svgFileRead(const QByteArray & data);
130  void error();
131  void emitFinished();
132 
133 private:
134  QProcess dotProc, unflattenProc;
135  QByteArray str, flatStr;
136 };

178 void GraphvizGraphicsView::mouseMoveEvent(QMouseEvent *e)
179 {
180  if (!e)
181  return;
182 
183  if (isPanning) {
184  auto *horizontalScrollbar = horizontalScrollBar();
185  auto *verticalScrollbar = verticalScrollBar();
186  if (!horizontalScrollbar || !verticalScrollbar)
187  return;
188 
189  auto direction = e->pos() - panStart;
190  horizontalScrollbar->setValue(horizontalScrollbar->value() - direction.x());
191  verticalScrollbar->setValue(verticalScrollbar->value() - direction.y());
192 
193  panStart = e->pos();
194  e->accept();
195  }
196 
197  QGraphicsView::mouseMoveEvent(e);
198 
199  return;
200 }

271 void GraphvizView::updateSvgItem(const App::Document &doc)
272 {
273  nPending++;
274 
275  // Skip if thread is working now
276  if (nPending > 1)
277  return;
278 
279  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/Paths");
280  QProcess * dotProc = thread->dotProcess();
281  QProcess * flatProc = thread->unflattenProcess();
282  QStringList args, flatArgs;
283  // TODO: Make -Granksep flag value variable depending on number of edges,
284  // the downside is that the value affects all subgraphs
285  args << QLatin1String("-Granksep=2") << QLatin1String("-Goutputorder=edgesfirst")
286  << QLatin1String("-Gsplines=ortho") << QLatin1String("-Tsvg");
287  flatArgs << QLatin1String("-c2 -l2");
288  auto dot = QString::fromLatin1("dot");
289  auto unflatten = QString::fromLatin1("unflatten");
290  auto path = QString::fromUtf8(hGrp->GetASCII("Graphviz").c_str());
291  bool pathChanged = false;
292  QDir dir;
293  if (!path.isEmpty()) {
294  dir = QDir(path);
295  dot = dir.filePath(QString::fromLatin1("dot"));
296  unflatten = dir.filePath(QString::fromLatin1("unflatten"));
297  }
298  dotProc->setEnvironment(QProcess::systemEnvironment());
299  flatProc->setEnvironment(QProcess::systemEnvironment());
300  do {
301  flatProc->start(unflatten, flatArgs);
302  bool value = flatProc->waitForStarted();
303  Q_UNUSED(value); // quieten code analyzer
304  dotProc->start(dot, args);
305  if (!dotProc->waitForStarted()) {
306  int ret = QMessageBox::warning(Gui::getMainWindow(),
307  tr("Graphviz not found"),
308  QString::fromLatin1("<html><head/><body>%1 "
309  "<a href=\"https://www.freecad.org/wiki/Std_DependencyGraph\">%2"
310  "</a><p>%3</p></body></html>")
311  .arg(tr("Graphviz couldn't be found on your system."),
312  tr("Read more about it here."),
313  tr("Do you want to specify its installation path if it's already installed?")),
314  QMessageBox::Yes, QMessageBox::No);
315  if (ret == QMessageBox::No) {
316  disconnectSignals();
317  return;
318  }
319  path = QFileDialog::getExistingDirectory(Gui::getMainWindow(),
320  tr("Graphviz installation path"));
321  if (path.isEmpty()) {
322  disconnectSignals();
323  return;
324  }
325  else {
326  dir = QDir(path);
327  dot = dir.filePath(QString::fromLatin1("dot"));
328  unflatten = dir.filePath(QString::fromLatin1("unflatten"));
329  pathChanged = true;
330  }
331  }
332  else {
333  if (pathChanged)
334  hGrp->SetASCII("Graphviz", (const char*)path.toUtf8());
335  break;
336  }
337  }
338  while(true);
339 
340  // Create graph in dot format
341  std::stringstream stream;
342  doc.exportGraphviz(stream);
343  graphCode = stream.str();
344 
345  // Update worker thread, and start it
346  thread->setData(QByteArray(graphCode.c_str(), graphCode.size()));
347  thread->startThread();
348 }

389 QByteArray GraphvizView::exportGraph(const QString& format)
390 {
391  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/Paths");
392  QProcess dotProc, flatProc;
393  QStringList args, flatArgs;
394  args << QString::fromLatin1("-T%1").arg(format);
395  flatArgs << QLatin1String("-c2 -l2");
396 
397 #ifdef FC_OS_LINUX
398  QString path = QString::fromUtf8(hGrp->GetASCII("Graphviz", "/usr/bin").c_str());
399 #else
400  QString path = QString::fromUtf8(hGrp->GetASCII("Graphviz").c_str());
401 #endif
402 
403 #ifdef FC_OS_WIN32
404  QString exe = QString::fromLatin1("\"%1/dot\"").arg(path);
405  QString unflatten = QString::fromLatin1("\"%1/unflatten\"").arg(path);
406 #else
407  QString exe = QString::fromLatin1("%1/dot").arg(path);
408  QString unflatten = QString::fromLatin1("%1/unflatten").arg(path);
409 #endif
410 
411  dotProc.setEnvironment(QProcess::systemEnvironment());
412  dotProc.start(exe, args);
413  if (!dotProc.waitForStarted()) {
414  return {};
415  }
416 
417  ParameterGrp::handle depGrp = App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/DependencyGraph");
418  if(depGrp->GetBool("Unflatten", true)) {
419  flatProc.setEnvironment(QProcess::systemEnvironment());
420  flatProc.start(unflatten, flatArgs);
421  if (!flatProc.waitForStarted()) {
422  return {};
423  }
424  flatProc.write(graphCode.c_str(), graphCode.size());
425  flatProc.closeWriteChannel();
426  if (!flatProc.waitForFinished())
427  return {};
428 
429  dotProc.write(flatProc.readAll());
430  }
431  else
432  dotProc.write(graphCode.c_str(), graphCode.size());
433 
434  dotProc.closeWriteChannel();
435  if (!dotProc.waitForFinished())
436  return {};
437 
438  return dotProc.readAll();
439 }

441 bool GraphvizView::onMsg(const char* pMsg, const char**)
442 {
443  if (strcmp("Save",pMsg) == 0 || strcmp("SaveAs",pMsg) == 0) {
444  QList< QPair<QString, QString> > formatMap;
445  formatMap << qMakePair(QString::fromLatin1("%1 (*.gv)").arg(tr("Graphviz format")), QString::fromLatin1("gv"));
446  formatMap << qMakePair(QString::fromLatin1("%1 (*.png)").arg(tr("PNG format")), QString::fromLatin1("png"));
447  formatMap << qMakePair(QString::fromLatin1("%1 (*.bmp)").arg(tr("Bitmap format")), QString::fromLatin1("bmp"));
448  formatMap << qMakePair(QString::fromLatin1("%1 (*.gif)").arg(tr("GIF format")), QString::fromLatin1("gif"));
449  formatMap << qMakePair(QString::fromLatin1("%1 (*.jpg)").arg(tr("JPG format")), QString::fromLatin1("jpg"));
450  formatMap << qMakePair(QString::fromLatin1("%1 (*.svg)").arg(tr("SVG format")), QString::fromLatin1("svg"));
451  formatMap << qMakePair(QString::fromLatin1("%1 (*.pdf)").arg(tr("PDF format")), QString::fromLatin1("pdf"));
452 
453  QStringList filter;
454  for (const auto & it : std::as_const(formatMap)) {
455  filter << it.first;
456  }
457 
458  QString selectedFilter;
459  QString fn = Gui::FileDialog::getSaveFileName(this, tr("Export graph"), QString(), filter.join(QLatin1String(";;")), &selectedFilter);
460  if (!fn.isEmpty()) {
461  QString format;
462  for (const auto & it : std::as_const(formatMap)) {
463  if (selectedFilter == it.first) {
464  format = it.second;
465  break;
466  }
467  }
468  QByteArray buffer;
469  if (format == QLatin1String("gv")) {
470  std::stringstream str;
471  doc.exportGraphviz(str);
472  buffer = QByteArray::fromStdString(str.str());
473  }
474  else {
475  buffer = exportGraph(format);
476  }
477  if (buffer.isEmpty()) {
478  return true;
479  }
480  QFile file(fn);
481  if (file.open(QFile::WriteOnly)) {
482  file.write(buffer);
483  file.close();
484  }
485  }
486  return true;
487  }
488  else if (strcmp("Print",pMsg) == 0) {
489  print();
490  return true;
491  }
492  else if (strcmp("PrintPreview",pMsg) == 0) {
493  printPreview();
494  return true;
495  }
496  else if (strcmp("PrintPdf",pMsg) == 0) {
497  printPdf();
498  return true;
499  }
500 
501  return false;
502 }
