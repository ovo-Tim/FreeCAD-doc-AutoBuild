28 //----------------------------------------------------------------------------
29 template <class Real>
30 QuadricSurface<Real>::QuadricSurface (const Real afCoeff[10])
31 {
32  for (int i = 0; i < 10; i++)
33  {
34  m_afCoeff[i] = afCoeff[i];
35  }
36 
37  // compute A, B, C
38  m_fC = m_afCoeff[0];
39  m_kB[0] = m_afCoeff[1];
40  m_kB[1] = m_afCoeff[2];
41  m_kB[2] = m_afCoeff[3];
42  m_kA[0][0] = m_afCoeff[4];
43  m_kA[0][1] = ((Real)0.5)*m_afCoeff[5];
44  m_kA[0][2] = ((Real)0.5)*m_afCoeff[6];
45  m_kA[1][0] = m_kA[0][1];
46  m_kA[1][1] = m_afCoeff[7];
47  m_kA[1][2] = ((Real)0.5)*m_afCoeff[8];
48  m_kA[2][0] = m_kA[0][2];
49  m_kA[2][1] = m_kA[1][2];
50  m_kA[2][2] = m_afCoeff[9];
51 }

155 //----------------------------------------------------------------------------
156 template <class Real>
157 int QuadricSurface<Real>::GetType () const
158 {
159  // Convert the coefficients to their rational representations and
160  // compute various derived quantities.
161  RReps kReps(m_afCoeff);
162 
163  // use Sturm sequences to determine the signs of the roots
164  int iPositiveRoots, iNegativeRoots, iZeroRoots;
165  GetRootSigns(kReps,iPositiveRoots,iNegativeRoots,iZeroRoots);
166 
167  // classify the solution set to the equation
168  int eType = QT_NONE;
169  switch (iZeroRoots)
170  {
171  case 0:
172  eType = ClassifyZeroRoots0(kReps,iPositiveRoots);
173  break;
174  case 1:
175  eType = ClassifyZeroRoots1(kReps,iPositiveRoots);
176  break;
177  case 2:
178  eType = ClassifyZeroRoots2(kReps,iPositiveRoots);
179  break;
180  case 3:
181  eType = ClassifyZeroRoots3(kReps);
182  break;
183  }
184  return eType;
185 }
186 //----------------------------------------------------------------------------
187 template <class Real>
188 void QuadricSurface<Real>::GetRootSigns (RReps& rkReps,
189  int& riPositiveRoots, int& riNegativeRoots, int& riZeroRoots)
190 {
191  // use Sturm sequences to determine the signs of the roots
192  int iSignChangeMI, iSignChange0, iSignChangePI, iDistinctNonzeroRoots;
193  Rational akValue[4];
194  if (rkReps.c0 != 0)
195  {
196  rkReps.c3 = Rational(2,9)*rkReps.c2*rkReps.c2 -
197  Rational(2,3)*rkReps.c1;
198  rkReps.c4 = rkReps.c0 - Rational(1,9)*rkReps.c1*rkReps.c2;
199 
200  if (rkReps.c3 != 0)
201  {
202  rkReps.c5 = -(rkReps.c1 + ((Rational(2)*rkReps.c2*rkReps.c3 +
203  Rational(3)*rkReps.c4)*rkReps.c4)/(rkReps.c3*rkReps.c3));
204 
205  akValue[0] = 1;
206  akValue[1] = -rkReps.c3;
207  akValue[2] = rkReps.c5;
208  iSignChangeMI = 1 + GetSignChanges(3,akValue);
209 
210  akValue[0] = -rkReps.c0;
211  akValue[1] = rkReps.c1;
212  akValue[2] = rkReps.c4;
213  akValue[3] = rkReps.c5;
214  iSignChange0 = GetSignChanges(4,akValue);
215 
216  akValue[0] = 1;
217  akValue[1] = rkReps.c3;
218  akValue[2] = rkReps.c5;
219  iSignChangePI = GetSignChanges(3,akValue);
220  }
221  else
222  {
223  akValue[0] = -rkReps.c0;
224  akValue[1] = rkReps.c1;
225  akValue[2] = rkReps.c4;
226  iSignChange0 = GetSignChanges(3,akValue);
227 
228  akValue[0] = 1;
229  akValue[1] = rkReps.c4;
230  iSignChangePI = GetSignChanges(2,akValue);
231  iSignChangeMI = 1 + iSignChangePI;
232  }
233 
234  riPositiveRoots = iSignChange0 - iSignChangePI;
235  assert(riPositiveRoots >= 0);
236  riNegativeRoots = iSignChangeMI - iSignChange0;
237  assert(riNegativeRoots >= 0);
238  riZeroRoots = 0;
239 
240  iDistinctNonzeroRoots = riPositiveRoots + riNegativeRoots;
241  if (iDistinctNonzeroRoots == 2)
242  {
243  if (riPositiveRoots == 2)
244  {
245  riPositiveRoots = 3;
246  }
247  else if (riNegativeRoots == 2)
248  {
249  riNegativeRoots = 3;
250  }
251  else
252  {
253  // One root is positive and one is negative. One root has
254  // multiplicity 2, the other of multiplicity 1. Distinguish
255  // between the two cases by computing the sign of the
256  // polynomial at the inflection point L = c2/3.
257  Rational kX = Rational(1,3)*rkReps.c2;
258  Rational kPoly = kX*(kX*(kX-rkReps.c2)+rkReps.c1)-rkReps.c0;
259  if (kPoly > 0)
260  {
261  riPositiveRoots = 2;
262  }
263  else
264  {
265  riNegativeRoots = 2;
266  }
267  }
268  }
269  else if (iDistinctNonzeroRoots == 1)
270  {
271  // root of multiplicity 3
272  if (riPositiveRoots == 1)
273  {
274  riPositiveRoots = 3;
275  }
276  else
277  {
278  riNegativeRoots = 3;
279  }
280  }
281 
282  return;
283  }
284 
285  if (rkReps.c1 != 0)
286  {
287  rkReps.c3 = Rational(1,4)*rkReps.c2*rkReps.c2 - rkReps.c1;
288 
289  akValue[0] = -1;
290  akValue[1] = rkReps.c3;
291  iSignChangeMI = 1 + GetSignChanges(2,akValue);
292 
293  akValue[0] = rkReps.c1;
294  akValue[1] = -rkReps.c2;
295  akValue[2] = rkReps.c3;
296  iSignChange0 = GetSignChanges(3,akValue);
297 
298  akValue[0] = 1;
299  akValue[1] = rkReps.c3;
300  iSignChangePI = GetSignChanges(2,akValue);
301 
302  riPositiveRoots = iSignChange0 - iSignChangePI;
303  assert( riPositiveRoots >= 0 );
304  riNegativeRoots = iSignChangeMI - iSignChange0;
305  assert( riNegativeRoots >= 0 );
306  riZeroRoots = 1;
307 
308  iDistinctNonzeroRoots = riPositiveRoots + riNegativeRoots;
309  if (iDistinctNonzeroRoots == 1)
310  {
311  riPositiveRoots = 2;
312  }
313 
314  return;
315  }
316 
317  if (rkReps.c2 != 0)
318  {
319  riZeroRoots = 2;
320  if (rkReps.c2 > 0)
321  {
322  riPositiveRoots = 1;
323  riNegativeRoots = 0;
324  }
325  else
326  {
327  riPositiveRoots = 0;
328  riNegativeRoots = 1;
329  }
330  return;
331  }
332 
333  riPositiveRoots = 0;
334  riNegativeRoots = 0;
335  riZeroRoots = 3;
336 }
337 //----------------------------------------------------------------------------
338 template <class Real>
339 int QuadricSurface<Real>::GetSignChanges (int iQuantity,
340  const Rational* akValue)
341 {
342  int iSignChanges = 0;
343  Rational kZero(0);
344 
345  Rational kPrev = akValue[0];
346  for (int i = 1; i < iQuantity; i++)
347  {
348  Rational kNext = akValue[i];
349  if (kNext != kZero)
350  {
351  if (kPrev*kNext < kZero)
352  {
353  iSignChanges++;
354  }
355 
356  kPrev = kNext;
357  }
358  }
359 
360  return iSignChanges;
361 }
362 //----------------------------------------------------------------------------
363 template <class Real>
364 int QuadricSurface<Real>::ClassifyZeroRoots0 (const RReps& rkReps,
365  int iPositiveRoots)
366 {
367  // inverse matrix is
368  // +- -+
369  // | Sub00 -Sub01 Sub02 |
370  // | -Sub01 Sub11 -Sub12 | * (1/det)
371  // | Sub02 -Sub12 Sub22 |
372  // +- -+
373  Rational kFourDet = Rational(4)*rkReps.c0;
374 
375  Rational kQForm = rkReps.b0*(rkReps.Sub00*rkReps.b0 -
376  rkReps.Sub01*rkReps.b1 + rkReps.Sub02*rkReps.b2) -
377  rkReps.b1*(rkReps.Sub01*rkReps.b0 - rkReps.Sub11*rkReps.b1 +
378  rkReps.Sub12*rkReps.b2) + rkReps.b2*(rkReps.Sub02*rkReps.b0 -
379  rkReps.Sub12*rkReps.b1 + rkReps.Sub22*rkReps.b2);
380 
381  Rational kR = Rational(1,4)*kQForm/kFourDet - rkReps.c;
382  if (kR > 0)
383  {
384  if (iPositiveRoots == 3)
385  {
386  return QT_ELLIPSOID;
387  }
388  else if (iPositiveRoots == 2)
389  {
390  return QT_HYPERBOLOID_ONE_SHEET;
391  }
392  else if (iPositiveRoots == 1)
393  {
394  return QT_HYPERBOLOID_TWO_SHEETS;
395  }
396  else
397  {
398  return QT_NONE;
399  }
400  }
401  else if (kR < 0)
402  {
403  if (iPositiveRoots == 3)
404  {
405  return QT_NONE;
406  }
407  else if (iPositiveRoots == 2)
408  {
409  return QT_HYPERBOLOID_TWO_SHEETS;
410  }
411  else if (iPositiveRoots == 1)
412  {
413  return QT_HYPERBOLOID_ONE_SHEET;
414  }
415  else
416  {
417  return QT_ELLIPSOID;
418  }
419  }
420 
421  // else kR == 0
422  if (iPositiveRoots == 3 || iPositiveRoots == 0)
423  {
424  return QT_POINT;
425  }
426 
427  return QT_ELLIPTIC_CONE;
428 }
429 //----------------------------------------------------------------------------
430 template <class Real>
431 int QuadricSurface<Real>::ClassifyZeroRoots1 (const RReps& rkReps,
432  int iPositiveRoots)
433 {
434  // Generate an orthonormal set {p0,p1,p2}, where p0 is an eigenvector
435  // of A corresponding to eigenvalue zero.
436  QSVector kP0, kP1, kP2;
437 
438  if (rkReps.Sub00 != 0 || rkReps.Sub01 != 0 || rkReps.Sub02 != 0)
439  {
440  // rows 1 and 2 are linearly independent
441  kP0 = QSVector(rkReps.Sub00,-rkReps.Sub01,rkReps.Sub02);
442  kP1 = QSVector(rkReps.a01,rkReps.a11,rkReps.a12);
443  kP2 = kP0.Cross(kP1);
444  return ClassifyZeroRoots1(rkReps,iPositiveRoots,kP0,kP1,kP2);
445  }
446 
447  if (rkReps.Sub01 != 0 || rkReps.Sub11 != 0 || rkReps.Sub12 != 0)
448  {
449  // rows 2 and 0 are linearly independent
450  kP0 = QSVector(-rkReps.Sub01,rkReps.Sub11,-rkReps.Sub12);
451  kP1 = QSVector(rkReps.a02,rkReps.a12,rkReps.a22);
452  kP2 = kP0.Cross(kP1);
453  return ClassifyZeroRoots1(rkReps,iPositiveRoots,kP0,kP1,kP2);
454  }
455 
456  // rows 0 and 1 are linearly independent
457  kP0 = QSVector(rkReps.Sub02,-rkReps.Sub12,rkReps.Sub22);
458  kP1 = QSVector(rkReps.a00,rkReps.a01,rkReps.a02);
459  kP2 = kP0.Cross(kP1);
460  return ClassifyZeroRoots1(rkReps,iPositiveRoots,kP0,kP1,kP2);
461 }
462 //----------------------------------------------------------------------------
463 template <class Real>
464 int QuadricSurface<Real>::ClassifyZeroRoots1 (const RReps& rkReps,
465  int iPositiveRoots, const QSVector& rkP0, const QSVector& rkP1,
466  const QSVector& rkP2)
467 {
468  Rational kE0 = rkP0.X()*rkReps.b0 + rkP0.Y()*rkReps.b1 +
469  rkP0.Z()*rkReps.b2;
470 
471  if (kE0 != 0)
472  {
473  if (iPositiveRoots == 1)
474  {
475  return QT_HYPERBOLIC_PARABOLOID;
476  }
477  else
478  {
479  return QT_ELLIPTIC_PARABOLOID;
480  }
481  }
482 
483  // matrix F
484  Rational kF11 = rkP1.X()*(rkReps.a00*rkP1.X() + rkReps.a01*rkP1.Y() +
485  rkReps.a02*rkP1.Z()) + rkP1.Y()*(rkReps.a01*rkP1.X() +
486  rkReps.a11*rkP1.Y() + rkReps.a12*rkP1.Z()) + rkP1.Z()*(
487  rkReps.a02*rkP1.X() + rkReps.a12*rkP1.Y() + rkReps.a22*rkP1.Z());
488 
489  Rational kF12 = rkP2.X()*(rkReps.a00*rkP1.X() + rkReps.a01*rkP1.Y() +
490  rkReps.a02*rkP1.Z()) + rkP2.Y()*(rkReps.a01*rkP1.X() +
491  rkReps.a11*rkP1.Y() + rkReps.a12*rkP1.Z()) + rkP2.Z()*(
492  rkReps.a02*rkP1.X() + rkReps.a12*rkP1.Y() + rkReps.a22*rkP1.Z());
493 
494  Rational kF22 = rkP2.X()*(rkReps.a00*rkP2.X() + rkReps.a01*rkP2.Y() +
495  rkReps.a02*rkP2.Z()) + rkP2.Y()*(rkReps.a01*rkP2.X() +
496  rkReps.a11*rkP2.Y() + rkReps.a12*rkP2.Z()) + rkP2.Z()*(
497  rkReps.a02*rkP2.X() + rkReps.a12*rkP2.Y() + rkReps.a22*rkP2.Z());
498 
499  // vector g
500  Rational kG1 = rkP1.X()*rkReps.b0 + rkP1.Y()*rkReps.b1 +
501  rkP1.Z()*rkReps.b2;
502  Rational kG2 = rkP2.X()*rkReps.b0 + rkP2.Y()*rkReps.b1 +
503  rkP2.Z()*rkReps.b2;
504 
505  // compute g^T*F^{-1}*g/4 - c
506  Rational kFourDet = Rational(4)*(kF11*kF22 - kF12*kF12);
507  Rational kR = (kG1*(kF22*kG1-kF12*kG2)+kG2*(kF11*kG2-kF12*kG1))/kFourDet
508  - rkReps.c;
509 
510  if (kR > 0)
511  {
512  if (iPositiveRoots == 2)
513  {
514  return QT_ELLIPTIC_CYLINDER;
515  }
516  else if (iPositiveRoots == 1)
517  {
518  return QT_HYPERBOLIC_CYLINDER;
519  }
520  else
521  {
522  return QT_NONE;
523  }
524  }
525  else if (kR < 0)
526  {
527  if (iPositiveRoots == 2)
528  {
529  return QT_NONE;
530  }
531  else if (iPositiveRoots == 1)
532  {
533  return QT_HYPERBOLIC_CYLINDER;
534  }
535  else
536  {
537  return QT_ELLIPTIC_CYLINDER;
538  }
539  }
540 
541  // else kR == 0
542  return (iPositiveRoots == 1 ? QT_TWO_PLANES : QT_LINE);
543 }
544 //----------------------------------------------------------------------------
545 template <class Real>
546 int QuadricSurface<Real>::ClassifyZeroRoots2 (const RReps& rkReps,
547  int iPositiveRoots)
548 {
549  // Generate an orthonormal set {p0,p1,p2}, where p0 and p1 are
550  // eigenvectors of A corresponding to eigenvalue zero. The vector p2 is
551  // an eigenvector of A corresponding to eigenvalue c2.
552  QSVector kP0, kP1, kP2;
553 
554  if (rkReps.a00 != 0 || rkReps.a01 != 0 || rkReps.a02 != 0)
555  {
556  // row 0 is not zero
557  kP2 = QSVector(rkReps.a00,rkReps.a01,rkReps.a02);
558  }
559  else if (rkReps.a01 != 0 || rkReps.a11 != 0 || rkReps.a12 != 0)
560  {
561  // row 1 is not zero
562  kP2 = QSVector(rkReps.a01,rkReps.a11,rkReps.a12);
563  }
564  else
565  {
566  // row 2 is not zero
567  kP2 = QSVector(rkReps.a02,rkReps.a12,rkReps.a22);
568  }
569 
570  if (kP2.X() != 0)
571  {
572  kP1.X() = kP2.Y();
573  kP1.Y() = -kP2.X();
574  kP1.Z() = 0;
575  }
576  else
577  {
578  kP1.X() = 0;
579  kP1.Y() = kP2.Z();
580  kP1.Z() = -kP2.Y();
581  }
582  kP0 = kP1.Cross(kP2);
583 
584  return ClassifyZeroRoots2(rkReps,iPositiveRoots,kP0,kP1,kP2);
585 }
586 //----------------------------------------------------------------------------
587 template <class Real>
588 int QuadricSurface<Real>::ClassifyZeroRoots2 (const RReps& rkReps,
589  int iPositiveRoots, const QSVector& rkP0, const QSVector& rkP1,
590  const QSVector& rkP2)
591 {
592  Rational kE0 = rkP0.X()*rkReps.b0 + rkP0.Y()*rkReps.b1 +
593  rkP0.Z()*rkReps.b1;
594 
595  if (kE0 != 0)
596  {
597  return QT_PARABOLIC_CYLINDER;
598  }
599 
600  Rational kE1 = rkP1.X()*rkReps.b0 + rkP1.Y()*rkReps.b1 +
601  rkP1.Z()*rkReps.b1;
602 
603  if (kE1 != 0)
604  {
605  return QT_PARABOLIC_CYLINDER;
606  }
607 
608  Rational kF2 = rkReps.c2*(rkP2.Dot(rkP2));
609  Rational kE2 = rkP2.X()*rkReps.b0 + rkP2.Y()*rkReps.b1 +
610  rkP2.Z()*rkReps.b1;
611 
612  Rational kR = kE2*kE2/(Rational(4)*kF2) - rkReps.c;
613  if (kR > 0)
