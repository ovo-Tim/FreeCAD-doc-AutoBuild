314  """Execute when the command is called."""
315  import Part
316 
317  # If there is a selection, and this selection contains various
318  # two-point lines, their shapes are extracted, and we attempt
319  # to join them into a single Wire (polyline),
320  # then the old lines are removed.
321  if len(Gui.Selection.getSelection()) > 1:
322  edges = []
323  for o in Gui.Selection.getSelection():
324  if utils.get_type(o) != "Wire":
325  edges = []
326  break
327  edges.extend(o.Shape.Edges)
328  if edges:
329  try:
330  w = Part.Wire(Part.__sortEdges__(edges))
331  except Exception:

333  "Unable to create a Wire "
334  "from selected objects"))
335  else:
336  # Points of the new fused Wire in string form
337  # 'FreeCAD.Vector(x,y,z), FreeCAD.Vector(x1,y1,z1), ...'
338  pts = ", ".join([str(v.Point) for v in w.Vertexes])
339  pts = pts.replace("Vector ", "FreeCAD.Vector")
340 
341  # List of commands to remove the old objects
342  rems = list()
343  for o in Gui.Selection.getSelection():
344  rems.append('FreeCAD.ActiveDocument.'
345  'removeObject("' + o.Name + '")')
346

349  # to be committed through the `draftutils.todo.ToDo` class

351  _cmd += '[' + pts + '], closed=' + str(w.isClosed())
352  _cmd += ')'
353  _cmd_list = [_cmd]

356  _cmd_list.append('FreeCAD.ActiveDocument.recompute()')
