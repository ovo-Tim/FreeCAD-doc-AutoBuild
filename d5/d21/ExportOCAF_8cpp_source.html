89 std::vector<App::DocumentObject*> ExportOCAF::filterPart(App::Part* part) const
90 {
91  // Ignore shape of a Part that is referenced by a FeatureBase (#0003807)
92  //
93  std::vector<App::DocumentObject*> entries = part->Group.getValues();
94 
95  // get FeatureBases of the out-lists of the features of the Part
96  Base::Type featureBase = Base::Type::fromName("PartDesign::FeatureBase");
97  std::vector<App::DocumentObject*> filterType;
98  for (auto it : entries) {
99  std::vector<App::DocumentObject*> outList = it->getOutList();
100  for (auto jt : outList) {
101  if (jt->getTypeId() == featureBase) {
102  filterType.push_back(jt);
103  }
104  }
105  }
106 
107  // now check if for a feature of the Part it must be filtered
108  if (!filterType.empty()) {
109  std::vector<App::DocumentObject*> keepObjects;
110  for (auto it : entries) {
111  std::vector<App::DocumentObject*> inList = it->getInList();
112  bool accept = true;
113  for (auto jt : inList) {
114  auto kt = std::find(filterType.begin(), filterType.end(), jt);
115  if (kt != filterType.end()) {
116  accept = false;
117  break;
118  }
119  }
120 
121  if (accept) {
122  keepObjects.push_back(it);
123  }
124  }
125 
126  entries.swap(keepObjects);
127  }
128 
129  return entries;
130 }

156 int ExportOCAF::exportObject(App::DocumentObject* obj,
157  std::vector<TDF_Label>& hierarchical_label,
158  std::vector<TopLoc_Location>& hierarchical_loc,
159  std::vector<App::DocumentObject*>& hierarchical_part)
160 {
161  std::vector<int> local_label;
162  int root_id;
163  int return_label = -1;
164 
165  if (obj->isDerivedFrom<App::Part>()) {
166  App::Part* part = static_cast<App::Part*>(obj);
167  // I shall recursively select the elements and call back
168  std::vector<App::DocumentObject*> entries = part->Group.getValues();
169  std::vector<App::DocumentObject*>::iterator it;
170 
171  if (filterBaseFeature) {
172  entries = filterPart(part);
173  }
174 
175  for (it = entries.begin(); it != entries.end(); ++it) {
176  int new_label = 0;
177  new_label =
178  exportObject((*it), hierarchical_label, hierarchical_loc, hierarchical_part);
179  local_label.push_back(new_label);
180  }
181 
182  createNode(part, root_id, hierarchical_label, hierarchical_loc, hierarchical_part);
183  std::vector<int>::iterator label_it;
184  for (label_it = local_label.begin(); label_it != local_label.end(); ++label_it) {
185  pushNode(root_id, (*label_it), hierarchical_label, hierarchical_loc);
186  }
187 
188  return_label = root_id;
189  }
190 
191  if (obj->isDerivedFrom<Part::Feature>()) {
192  Part::Feature* part = static_cast<Part::Feature*>(obj);
193  std::vector<App::Color> colors;
194  findColors(part, colors);
195 
196  return_label =
197  saveShape(part, colors, hierarchical_label, hierarchical_loc, hierarchical_part);
198  }
199 
200  return return_label;
201 }

234 int ExportOCAF::saveShape(Part::Feature* part,
235  const std::vector<App::Color>& colors,
236  std::vector<TDF_Label>& hierarchical_label,
237  std::vector<TopLoc_Location>& hierarchical_loc,
238  std::vector<App::DocumentObject*>& hierarchical_part)

282  // Add color information
283  Quantity_ColorRGBA col;
284 
285  std::set<int> face_index;
286  TopTools_IndexedMapOfShape faces;
287  TopExp_Explorer xp(baseShape, TopAbs_FACE);

308  aShapeTool->SetShape(faceLabel, xp.Current());
309  }
310  else {
311  aShapeTool->FindShape(xp.Current(), faceLabel);
312  }
313 
314  if (!faceLabel.IsNull()) {
315  const App::Color& color = colors[index - 1];
316  col = convertColor(color);
317  aColorTool->SetColor(faceLabel, col, XCAFDoc_ColorSurf);
318  }
319  }

321  }
322  }
323  else if (!colors.empty()) {
324  App::Color color = colors.front();
325  col = convertColor(color);
326  aColorTool->SetColor(shapeLabel, col, XCAFDoc_ColorGen);
327  }
328 
329  hierarchical_label.push_back(shapeLabel);
330  hierarchical_loc.push_back(MyLoc);
331  hierarchical_part.push_back(part);
332 
333  return (hierarchical_label.size());

343 {

444 void ExportOCAF::pushNode(int root_id,
445  int node_id,
446  std::vector<TDF_Label>& hierarchical_label,
447  std::vector<TopLoc_Location>& hierarchical_loc)
448 {
449  auto isValidIndex = [&](std::size_t root, std::size_t node) {
450  // NOLINTBEGIN
451  if (root >= hierarchical_label.size()) {
452  return false;
453  }
454  if (node >= hierarchical_label.size() || node >= hierarchical_loc.size()) {
455  return false;
456  }
457 
458  return true;
459  // NOLINTEND
460  };
461  if (isValidIndex(root_id - 1, node_id - 1)) {
462  TDF_Label root;
463  TDF_Label node;
464  TopLoc_Location locn;
465  root = hierarchical_label.at(root_id - 1);
466  node = hierarchical_label.at(node_id - 1);
467  locn = hierarchical_loc.at(node_id - 1);
468 
469  XCAFDoc_DocumentTool::ShapeTool(root)->AddComponent(root, node, locn);
470  }
471 }
