105 bool MeasureDistance::isValidSelection(const App::MeasureSelection& selection)
106 {
107 
108  if (selection.size() != 2) {
109  return false;
110  }
111 
112  for (auto element : selection) {
113  auto type = App::MeasureManager::getMeasureElementType(element);
114 
115  if (type == App::MeasureElementType::INVALID) {
116  return false;
117  }
118 
119  if (type != App::MeasureElementType::POINT && type != App::MeasureElementType::LINE
120  && type != App::MeasureElementType::LINESEGMENT
121  && type != App::MeasureElementType::CIRCLE && type != App::MeasureElementType::ARC
122  && type != App::MeasureElementType::CURVE && type != App::MeasureElementType::PLANE
123  && type != App::MeasureElementType::CYLINDER) {
124  return false;
125  }
126  }
127  return true;
128 }

163 bool MeasureDistance::getShape(App::PropertyLinkSub* prop, TopoDS_Shape& rShape)
164 {
165 
166  App::DocumentObject* ob = prop->getValue();
167  std::vector<std::string> subs = prop->getSubValues();
168 
169  if (!ob || !ob->isValid() || subs.empty()) {

172 
173  std::string subName = subs.at(0);
174  App::SubObjectT subject {ob, subName.c_str()};
175 
176  auto info = getMeasureInfo(subject);

181  auto distanceInfo = std::dynamic_pointer_cast<Part::MeasureDistanceInfo>(info);
182 
183  rShape = distanceInfo->getShape();
184  return true;

187 bool MeasureDistance::distanceCircleCircle(const TopoDS_Shape& shape1, const TopoDS_Shape& shape2)
188 {
189  auto circle1 = asCircle(shape1);
190  auto circle2 = asCircle(shape2);
191  if (!circle1.IsNull() && !circle2.IsNull()) {
192  const gp_Pnt p1 = circle1->Location();
193  const gp_Pnt p2 = circle2->Location();
194  setValues(p1, p2);
195  return true;
196  }
197 
198  return false;
199 }

201 void MeasureDistance::distanceGeneric(const TopoDS_Shape& shape1, const TopoDS_Shape& shape2)
202 {
203  // Calculate the extrema
204  BRepExtrema_DistShapeShape measure(shape1, shape2);
205  if (!measure.IsDone() || measure.NbSolution() < 1) {
206  throw Base::RuntimeError("Could not get extrema");
207  }
208 
209  const gp_Pnt p1 = measure.PointOnShape1(1);
210  const gp_Pnt p2 = measure.PointOnShape2(1);
211  setValues(p1, p2);
212 }

214 void MeasureDistance::setValues(const gp_Pnt& p1, const gp_Pnt& p2)
215 {
216  Position1.setValue(p1.X(), p1.Y(), p1.Z());
217  Position2.setValue(p2.X(), p2.Y(), p2.Z());
218 
219  const gp_Pnt delta = p2.XYZ() - p1.XYZ();
220  Distance.setValue(p1.Distance(p2));
221  DistanceX.setValue(std::fabs(delta.X()));
222  DistanceY.setValue(std::fabs(delta.Y()));
223  DistanceZ.setValue(std::fabs(delta.Z()));
224 }

226 App::DocumentObjectExecReturn* MeasureDistance::execute()
227 {
228 
229  App::DocumentObject* ob1 = Element1.getValue();
230  std::vector<std::string> subs1 = Element1.getSubValues();
231 
232  App::DocumentObject* ob2 = Element2.getValue();
233  std::vector<std::string> subs2 = Element2.getSubValues();
234 
235  if (!ob1 || !ob1->isValid() || !ob2 || !ob2->isValid()) {
236  return new App::DocumentObjectExecReturn("Submitted object(s) is not valid");
237  }
238 
239  if (subs1.empty() || subs2.empty()) {
240  return new App::DocumentObjectExecReturn("No geometry element picked");
241  }
242 
243  // Get both shapes from geometry handler
244  TopoDS_Shape shape1;
245  if (!this->getShape(&Element1, shape1)) {
246  return new App::DocumentObjectExecReturn("Could not get shape");
247  }
248 
249  TopoDS_Shape shape2;
250  if (!this->getShape(&Element2, shape2)) {
251  return new App::DocumentObjectExecReturn("Could not get shape");
252  }
253 
254  if (distanceCircleCircle(shape1, shape2)) {
255  return DocumentObject::StdReturn;
256  }
257 
258  distanceGeneric(shape1, shape2);
259  return DocumentObject::StdReturn;
260 }
