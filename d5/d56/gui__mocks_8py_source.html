39 class DialogInteractor(QtCore.QObject):
40  """Takes the title of the dialog and a callable. The callable is passed the widget

62  def run(self):
63  widget = QtWidgets.QApplication.activeModalWidget()
64  if widget and self._dialog_matches(widget):
65  # Found the dialog we are looking for: now try to run the interaction
66  if self.interaction is not None and callable(self.interaction):
67  self.interaction(widget)
68  self.dialog_found = True
69  self.timer.stop()

74  print("Stopped timer after 100 iterations")

78  # Is this the widget we are looking for? Only applies on Linux and Windows: macOS
79  # doesn't set the title of a modal dialog:
80  os = QtCore.QSysInfo.productType() # Qt5 gives "osx", Qt6 gives "macos"
