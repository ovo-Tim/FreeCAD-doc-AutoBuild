44 class ExtendSelection: public Gui::SelectionFilterGate

54 
55  bool allow(App::Document* /*pDoc*/, App::DocumentObject* pObj, const char* sSubName) override
56  {
57  if (pObj != this->object) {
58  return false;
59  }
60  if (!sSubName || sSubName[0] == '\0') {
61  return false;
62  }
63  if (disabled) {
64  return true;
65  }
66  std::string element(sSubName);
67  if (element.substr(0, 4) == "Edge") {
68  int GeoId = std::atoi(element.substr(4, 4000).c_str()) - 1;
69  Sketcher::SketchObject* Sketch = static_cast<Sketcher::SketchObject*>(object);
70  const Part::Geometry* geom = Sketch->getGeometry(GeoId);
71  if (geom->is<Part::GeomLineSegment>() || geom->is<Part::GeomArcOfCircle>()) {
72  return true;
73  }
74  }
75  return false;
76  }
77 
78  void setDisabled(bool isDisabled)
79  {
80  disabled = isDisabled;
81  }

89 {
90 public:

104  enum SelectMode
105  {
106  STATUS_SEEK_First,
107  STATUS_SEEK_Second,
108  };

110  void mouseMove(Base::Vector2d onSketchPos) override
111  {
112  Q_UNUSED(onSketchPos);
113  if (Mode == STATUS_SEEK_Second) {
114  const Part::Geometry* geom = sketchgui->getSketchObject()->getGeometry(BaseGeoId);
115  if (geom->is<Part::GeomLineSegment>()) {
116  const Part::GeomLineSegment* lineSeg =
117  static_cast<const Part::GeomLineSegment*>(geom);
118  // project point to the existing curve
119  Base::Vector3d start3d = lineSeg->getStartPoint();
120  Base::Vector3d end3d = lineSeg->getEndPoint();
121 
122  Base::Vector2d startPoint = Base::Vector2d(start3d.x, start3d.y);
123  Base::Vector2d endPoint = Base::Vector2d(end3d.x, end3d.y);
124  Base::Vector2d recenteredLine = endPoint - startPoint;
125  Base::Vector2d recenteredPoint = onSketchPos - startPoint;
126  Base::Vector2d projection;
127  projection.ProjectToLine(recenteredPoint, recenteredLine);
128  if (recenteredPoint.Length() < recenteredPoint.Distance(recenteredLine)) {
129  EditCurve[0] = startPoint + projection;
130  EditCurve[1] = endPoint;
131  }
132  else {
133  EditCurve[0] = startPoint;
134  EditCurve[1] = startPoint + projection;
135  }
143  bool inCurve = (projection.Length() < recenteredLine.Length()
144  && projection.GetAngle(recenteredLine)
145  < 0.1); // Two possible values here, M_PI and 0, but 0.1 is to
146  // avoid floating point problems.
147  if (inCurve) {
148  Increment = SavedExtendFromStart
149  ? -1 * projection.Length()
150  : projection.Length() - recenteredLine.Length();
151  ExtendFromStart = SavedExtendFromStart;
152  }
153  else {
154  ExtendFromStart =
155  onSketchPos.Distance(startPoint) < onSketchPos.Distance(endPoint);
156  Increment = ExtendFromStart ? projection.Length()
157  : projection.Length() - recenteredLine.Length();
158  }
159  drawEdit(EditCurve);
160  }
161  else if (geom->is<Part::GeomArcOfCircle>()) {
162  const Part::GeomArcOfCircle* arc = static_cast<const Part::GeomArcOfCircle*>(geom);
163  Base::Vector3d center = arc->getCenter();
164  double radius = arc->getRadius();
165 
166  double start, end;
167  arc->getRange(start, end, true);
168  double arcAngle = end - start;
169 
170  Base::Vector2d angle =
171  Base::Vector2d(onSketchPos.x - center.x, onSketchPos.y - center.y);
172  Base::Vector2d startAngle = Base::Vector2d(cos(start), sin(start));
173  Base::Vector2d endAngle = Base::Vector2d(cos(end), sin(end));
174 
175  Base::Vector2d arcHalf =
176  Base::Vector2d(cos(start + arcAngle / 2.0), sin(start + arcAngle / 2.0));
177  double angleToEndAngle = angle.GetAngle(endAngle);
178  double angleToStartAngle = angle.GetAngle(startAngle);
179 
180 
181  double modStartAngle = start;
182  double modArcAngle = end - start;
183  bool outOfArc = arcHalf.GetAngle(angle) * 2.0 > arcAngle;
184  if (ExtendFromStart) {
185  bool isCCWFromStart = crossProduct(angle, startAngle) < 0;
186  if (outOfArc) {
187  if (isCCWFromStart) {
188  modStartAngle -= 2 * M_PI - angleToStartAngle;
189  modArcAngle += 2 * M_PI - angleToStartAngle;
190  }
191  else {
192  modStartAngle -= angleToStartAngle;
193  modArcAngle += angleToStartAngle;
194  }
195  }
196  else {
197  if (isCCWFromStart) {
198  modStartAngle += angleToStartAngle;
199  modArcAngle -= angleToStartAngle;
200  }
201  else {
202  modStartAngle += 2 * M_PI - angleToStartAngle;
203  modArcAngle -= 2 * M_PI - angleToStartAngle;
204  }
205  }
206  }
207  else {
208  bool isCWFromEnd = crossProduct(angle, endAngle) >= 0;
209  if (outOfArc) {
210  if (isCWFromEnd) {
211  modArcAngle += 2 * M_PI - angleToEndAngle;
212  }
213  else {
214  modArcAngle += angleToEndAngle;
215  }
216  }
217  else {
218  if (isCWFromEnd) {
219  modArcAngle -= angleToEndAngle;
220  }
221  else {
222  modArcAngle -= 2 * M_PI - angleToEndAngle;
223  }
224  }
225  }
226  Increment = modArcAngle - (end - start);
227  for (int i = 0; i < 31; i++) {
228  double angle = modStartAngle + i * modArcAngle / 30.0;
229  EditCurve[i] = Base::Vector2d(center.x + radius * cos(angle),
230  center.y + radius * sin(angle));
231  }
232  drawEdit(EditCurve);
233  }
234  int curveId = getPreselectCurve();
235  if (BaseGeoId != curveId) {
236  seekAndRenderAutoConstraint(SugConstr, onSketchPos, Base::Vector2d(0.f, 0.f));
237  }
238  }
239  }

247  bool releaseButton(Base::Vector2d onSketchPos) override
248  {
249  Q_UNUSED(onSketchPos);
250  if (Mode == STATUS_SEEK_First) {
251  BaseGeoId = getPreselectCurve();
252  if (BaseGeoId > -1) {
253  const Part::Geometry* geom = sketchgui->getSketchObject()->getGeometry(BaseGeoId);
254  if (geom->is<Part::GeomLineSegment>()) {
255  const Part::GeomLineSegment* seg =
256  static_cast<const Part::GeomLineSegment*>(geom);
257  Base::Vector3d start3d = seg->getStartPoint();
258  Base::Vector3d end3d = seg->getEndPoint();
259  Base::Vector2d start = Base::Vector2d(start3d.x, start3d.y);
260  Base::Vector2d end = Base::Vector2d(end3d.x, end3d.y);
261  SavedExtendFromStart =
262  (onSketchPos.Distance(start) < onSketchPos.Distance(end));
263  ExtendFromStart = SavedExtendFromStart;
264  Mode = STATUS_SEEK_Second;
265  }
266  else if (geom->is<Part::GeomArcOfCircle>()) {
267  const Part::GeomArcOfCircle* arc =
268  static_cast<const Part::GeomArcOfCircle*>(geom);
269  double start, end;
270  arc->getRange(start, end, true);
271 
272  Base::Vector3d center = arc->getCenter();
273  Base::Vector2d angle =
274  Base::Vector2d(onSketchPos.x - center.x, onSketchPos.y - center.y);
275  double angleToStart = angle.GetAngle(Base::Vector2d(cos(start), sin(start)));
276  double angleToEnd = angle.GetAngle(Base::Vector2d(cos(end), sin(end)));
277  ExtendFromStart =
278  (angleToStart
279  < angleToEnd); // move start point if closer to angle than end point
280  EditCurve.resize(31);
281  Mode = STATUS_SEEK_Second;
282  }
283  filterGate->setDisabled(true);
284  }
285  }
286  else if (Mode == STATUS_SEEK_Second) {
287  try {
288  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Extend edge"));
289  Gui::cmdAppObjectArgs(sketchgui->getObject(),
290  "extend(%d, %f, %d)\n", // GeoId, increment, PointPos
291  BaseGeoId,
292  Increment,
293  ExtendFromStart ? static_cast<int>(Sketcher::PointPos::start)
294  : static_cast<int>(Sketcher::PointPos::end));
295  Gui::Command::commitCommand();
296 
297  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
298  "User parameter:BaseApp/Preferences/Mod/Sketcher");
299  bool autoRecompute = hGrp->GetBool("AutoRecompute", false);
300  if (autoRecompute) {
301  Gui::Command::updateActive();
302  }
303 
304  // constrain chosen point
305  if (!SugConstr.empty()) {
306  createAutoConstraints(SugConstr,
307  BaseGeoId,
308  (ExtendFromStart) ? Sketcher::PointPos::start
309  : Sketcher::PointPos::end);
310  SugConstr.clear();
311  }
312  bool continuousMode = hGrp->GetBool("ContinuousCreationMode", true);
313 
314  if (continuousMode) {
315  // This code enables the continuous creation mode.
316  Mode = STATUS_SEEK_First;
317  filterGate->setDisabled(false);
318  EditCurve.clear();
319  drawEdit(EditCurve);
320  EditCurve.resize(2);
321  applyCursor();
322  /* this is ok not to call to purgeHandler
323  * in continuous creation mode because the
324  * handler is destroyed by the quit() method on pressing the
325  * right button of the mouse */
326  }
327  else {
328  sketchgui->purgeHandler(); // no code after this line, Handler get deleted in
329  // ViewProvider
330  }
331  }
332  catch (const Base::Exception&) {
333  Gui::NotifyError(sketchgui,
334  QT_TRANSLATE_NOOP("Notifications", "Error"),
335  QT_TRANSLATE_NOOP("Notifications", "Failed to extend edge"));
336  Gui::Command::abortCommand();
337  }
338  }
339  else { // exit extension tool if user clicked on empty space
340  BaseGeoId = -1;
341  sketchgui
342  ->purgeHandler(); // no code after this line, Handler get deleted in ViewProvider
343  }
344  return true;
345  }
