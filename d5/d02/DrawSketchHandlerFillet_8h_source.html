55  bool allow(App::Document* /*pDoc*/, App::DocumentObject* pObj, const char* sSubName) override
56  {
57  if (pObj != this->object) {
58  return false;
59  }
60  if (!sSubName || sSubName[0] == '\0') {
61  return false;
62  }
63  std::string element(sSubName);
64  if (element.substr(0, 4) == "Edge") {
65  int GeoId = std::atoi(element.substr(4, 4000).c_str()) - 1;
66  Sketcher::SketchObject* Sketch = static_cast<Sketcher::SketchObject*>(object);
67  const Part::Geometry* geom = Sketch->getGeometry(GeoId);
68  if (geom->isDerivedFrom<Part::GeomBoundedCurve>()) {
69  return true;
70  }
71  }
72  if (element.substr(0, 6) == "Vertex") {
73  int VtId = std::atoi(element.substr(6, 4000).c_str()) - 1;
74  Sketcher::SketchObject* Sketch = static_cast<Sketcher::SketchObject*>(object);
75  std::vector<int> GeoIdList;
76  std::vector<Sketcher::PointPos> PosIdList;
77  Sketch->getDirectlyCoincidentPoints(VtId, GeoIdList, PosIdList);
78  if (GeoIdList.size() == 2 && GeoIdList[0] >= 0 && GeoIdList[1] >= 0) {
79  const Part::Geometry* geom1 = Sketch->getGeometry(GeoIdList[0]);
80  const Part::Geometry* geom2 = Sketch->getGeometry(GeoIdList[1]);
81  if (geom1->is<Part::GeomLineSegment>() && geom2->is<Part::GeomLineSegment>()) {
82  return true;
83  }
84  }
85  }
86  return false;
87  }
88 };

156  void executeCommands() override
157  {
158  SketchObject* obj = sketchgui->getSketchObject();
159 
160  bool construction = false;
161  bool isChamfer = constructionMethod() == ConstructionMethod::Chamfer;
162 
163  if (vtId != -1) {
164  int GeoId;
165  PointPos PosId = PointPos::none;
166  obj->getGeoVertexIndex(vtId, GeoId, PosId);
167  const Part::Geometry* geom = obj->getGeometry(GeoId);
168  if (isLineSegment(*geom) && (PosId == PointPos::start || PosId == PointPos::end)) {
169 
170  // guess fillet radius
171  double radius = -1;
172  std::vector<int> GeoIdList;
173  std::vector<Sketcher::PointPos> PosIdList;
174  obj->getDirectlyCoincidentPoints(GeoId, PosId, GeoIdList, PosIdList);
175  if (GeoIdList.size() == 2 && GeoIdList[0] >= 0 && GeoIdList[1] >= 0) {
176  const Part::Geometry* geo1 = obj->getGeometry(GeoIdList[0]);
177  const Part::Geometry* geo2 = obj->getGeometry(GeoIdList[1]);
178 
179  construction = GeometryFacade::getConstruction(geo1)
180  && GeometryFacade::getConstruction(geo2);
181 
182  if (isLineSegment(*geo1) && isLineSegment(*geo2)) {
183  auto* line1 = static_cast<const Part::GeomLineSegment*>(geo1);
184  auto* line2 = static_cast<const Part::GeomLineSegment*>(geo2);
185  Base::Vector3d dir1 = line1->getEndPoint() - line1->getStartPoint();
186  Base::Vector3d dir2 = line2->getEndPoint() - line2->getStartPoint();
187  if (PosIdList[0] == PointPos::end) {
188  dir1 *= -1;
189  }
190  if (PosIdList[1] == PointPos::end) {
191  dir2 *= -1;
192  }
193  double l1 = dir1.Length();
194  double l2 = dir2.Length();
195  double angle = dir1.GetAngle(dir2);
196  radius = (l1 < l2 ? l1 : l2) * 0.2 * sin(angle / 2);
197  }
198  }
199  if (radius < 0) {
200  return;
201  }
202 
203  int filletGeoId = getHighestCurveIndex() + (isChamfer ? 2 : 1);
204  // create fillet at point
205  try {
206  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create fillet"));
207  Gui::cmdAppObjectArgs(obj,
208  "fillet(%d,%d,%f,%s,%s,%s)",
209  GeoId,
210  static_cast<int>(PosId),
211  radius,
212  "True",
213  preserveCorner ? "True" : "False",
214  isChamfer ? "True" : "False");
215 
216  if (construction) {
217  Gui::cmdAppObjectArgs(obj, "toggleConstruction(%d) ", filletGeoId);
218  }
219 
220  Gui::Command::commitCommand();
221  }
222  catch (const Base::Exception& e) {
223  Gui::NotifyUserError(
224  obj,
225  QT_TRANSLATE_NOOP("Notifications", "Failed to create fillet"),
226  e.what());
227  Gui::Command::abortCommand();
228  }
229 
230  tryAutoRecomputeIfNotSolve(obj);
231  }
232  }
233 
234  else {
235  Base::Vector3d refPnt1(firstPos.x, firstPos.y, 0.f);
236  Base::Vector3d refPnt2(secondPos.x, secondPos.y, 0.f);
237 
238  const Part::Geometry* geo1 = obj->getGeometry(geoId1);
239  const Part::Geometry* geo2 = obj->getGeometry(geoId2);
240 
241  construction =
242  GeometryFacade::getConstruction(geo1) && GeometryFacade::getConstruction(geo2);
243 
244  double radius = 0;
245 
246  if (isLineSegment(*geo1) && isLineSegment(*geo2)) {
247  // guess fillet radius
248  auto* line1 = static_cast<const Part::GeomLineSegment*>(geo1);
249  auto* line2 = static_cast<const Part::GeomLineSegment*>(geo2);
250 
251  radius = Part::suggestFilletRadius(line1, line2, refPnt1, refPnt2);
252  if (radius < 0) {
253  return;
254  }
255  }
256 
257  int filletGeoId = getHighestCurveIndex() + (isChamfer ? 2 : 1);
258 
259  // create fillet between lines
260  try {
261  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create fillet"));
262  Gui::cmdAppObjectArgs(
263  obj,
264  "fillet(%d,%d,App.Vector(%f,%f,0),App.Vector(%f,%f,0),%f,%s,%s,%s)",
265  geoId1,
266  geoId2,
267  firstPos.x,
268  firstPos.y,
269  secondPos.x,
270  secondPos.y,
271  radius,
272  "True",
273  preserveCorner ? "True" : "False",
274  isChamfer ? "True" : "False");
275  Gui::Command::commitCommand();
276  }
277  catch (const Base::CADKernelError& e) {
278  if (e.getTranslatable()) {
279  Gui::TranslatedUserError(sketchgui,
280  QObject::tr("CAD Kernel Error"),
281  QObject::tr(e.getMessage().c_str()));
282  }
283  Gui::Selection().clearSelection();
284  Gui::Command::abortCommand();
285  }
286  catch (const Base::ValueError& e) {
287  Gui::TranslatedUserError(sketchgui,
288  QObject::tr("Value Error"),
289  QObject::tr(e.getMessage().c_str()));
290  Gui::Selection().clearSelection();
291  Gui::Command::abortCommand();
292  }
293 
294  tryAutoRecompute(obj);
295 
296  if (construction) {
297  Gui::cmdAppObjectArgs(obj, "toggleConstruction(%d) ", filletGeoId);
298  }
299 
300 
301  Gui::Selection().clearSelection();
302  }
303  }
