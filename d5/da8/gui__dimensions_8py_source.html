532  def set_constraint_node(self):

534  by projecting on the working plane.
535  """

537  self.proj_point1 = self.wp.project_point(self.node[0])
538  self.proj_point2 = self.wp.project_point(self.node[1])
539  proj_u= self.wp.u.dot(self.proj_point2 - self.proj_point1)
540  proj_v= self.wp.v.dot(self.proj_point2 - self.proj_point1)
541  active_view = Gui.ActiveDocument.ActiveView
542  cursor = active_view.getCursorPos()
543  cursor_point = active_view.getPoint(cursor)
544  self.point = self.wp.project_point(cursor_point)
545  if not self.force:
546  ref_point = self.point - (self.proj_point2 + self.proj_point1)*1/2
547  ref_angle = abs(ref_point.getAngle(self.wp.u))
548  if (ref_angle > math.pi/4) and (ref_angle <= 0.75*math.pi):
549  self.force = 2
