123 AppearancePreview::AppearancePreview(QWidget* parent)
124  : Gui::View3DInventorViewer(parent)
125 {
126  setRedirectToSceneGraph(true);
127  setViewing(true);
128  setPopupMenuEnabled(false);
129 
130  applySettings();
131  // setBackground();
132  setEnabledNaviCube(false);
133 
134  _group = dynamic_cast<SoSeparator*>(getSceneGraph());
135  _group->ref();
136 
137  _switch = new SoSwitch();
138  _switch->ref();
139  _material = new SoMaterial();
140  _material->ref();
141  _texture = new SoTexture2();
142  _texture->ref();
143  _environment = new SoTextureCoordinateEnvironment();
144  _environment->ref();
145 
146  _switch->addChild(_material);
147  _switch->addChild(_texture);
148  _switch->whichChild.setValue(0);
149 
150  _group->addChild(_switch);
151  _group->addChild(new SoSphere());
152 
153  setCameraType(SoOrthographicCamera::getClassTypeId());
154  setViewDirection(SbVec3f(1, 1, -5));
155  viewAll();
156 }

158 AppearancePreview::~AppearancePreview()
159 {
160  if (_switch) {
161  if (_switch->findChild(_material) > -1) {
162  _switch->removeChild(_material);
163  }
164  if (_switch->findChild(_texture) > -1) {
165  _switch->removeChild(_texture);
166  }
167  }
168  if (_group) {
169  if (_group->findChild(_switch) > -1) {
170  _group->removeChild(_switch);
171  }
172  }
173 
174  _group->unref();
175  _group = nullptr;
176  _switch->unref();
177  _switch = nullptr;
178  _material->unref();
179  _material = nullptr;
180  _texture->unref();
181  _texture = nullptr;
182  _environment->unref();
183  _environment = nullptr;
184 }
