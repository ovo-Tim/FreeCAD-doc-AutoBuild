64  GENERAL NOTE ON COIN3D CUSTOM DRAGGERS
65  * You basically have two choices for creating custom dragger geometry.
66  * 1) create an .iv file and set environment variable to the file path. This
67  * comes with install headaches.
68  * 2) create an .iv file and run through a mock compiler that generates a header
69  * file to include in the project. I would have gone this way but after installing
70  * inventor-demo(ubuntu), the mock compiler tool was there only in source and make
71  * didn't do anything. Didn't want to put any time into something I didn't like anyway.
72  *
73  * static SbList <SoNode*> * defaultdraggerparts = NULL; is a global definition
74  * in SoInteractionKit that contains the geometry. There doesn't appear to be anyway
75  * to add to this other than readDefaultParts, that takes a file. So maybe a temp file?
76  *
77  * naming appears to be central to the core. It looks like as long as an object
78  * is alive SoNode::getByName() will find it. So maybe just create my own little
79  * container of objects to keep the default geometry alive....This appears to be
80  * working and I like this solution.
81  *
82  * SoInteractionKit warns about these
83  * names all being the same scope and do NOT have to be unique. Need to make names
84  * descriptive to avoid collisions.
85 
86  * this is point of the SoGroup accessed from SoFCDB::getStorage().

304 void TDragger::dragStart()
305 {
306  SoSwitch* sw;
307  sw = SO_GET_ANY_PART(this, "activeSwitch", SoSwitch);
308  SoInteractionKit::setSwitchValue(sw, SO_SWITCH_ALL);
309 
310  // do an initial projection to eliminate discrepancies
311  // in arrow head pick. we define the arrow in the y+ direction
312  // and we know local space will be relative to this. so y vector
313  // line projection will work.
314  projector.setViewVolume(this->getViewVolume());
315  projector.setWorkingSpace(this->getLocalToWorldMatrix());
316  projector.setLine(SbLine(SbVec3f(0.0, 0.0, 0.0), SbVec3f(0.0, 1.0, 0.0)));
317  SbVec3f hitPoint = projector.project(getNormalizedLocaterPosition());
318 
319  projector.setLine(SbLine(SbVec3f(0.0, 0.0, 0.0), hitPoint));
320 
321  SbMatrix localToWorld = getLocalToWorldMatrix();
322  localToWorld.multVecMatrix(hitPoint, hitPoint);
323  setStartingPoint((hitPoint));
324 
325  translationIncrementCount.setValue(0);
326 }

328 void TDragger::drag()
329 {
330  projector.setViewVolume(this->getViewVolume());
331  projector.setWorkingSpace(this->getLocalToWorldMatrix());
332 
333  SbVec3f hitPoint = projector.project(getNormalizedLocaterPosition());
334  SbVec3f startingPoint = getLocalStartingPoint();
335  SbVec3f localMovement = hitPoint - startingPoint;
336 
337  // scale the increment to match local space.
338  float scaledIncrement =
339  static_cast<float>(translationIncrement.getValue()) / autoScaleResult.getValue();
340 
341  localMovement = roundTranslation(localMovement, scaledIncrement);
342  // when the movement vector is null either the appendTranslation or
343  // the setMotionMatrix doesn't work. either way it stops translating
344  // back to its initial starting point.
345  if (localMovement.equals(SbVec3f(0.0, 0.0, 0.0), 0.00001f)) {
346  setMotionMatrix(getStartMotionMatrix());
347  // don't know why I need the following but if I don't have it
348  // it won't return to original position.
349  this->valueChanged();
350  }
351  else {
352  setMotionMatrix(appendTranslation(getStartMotionMatrix(), localMovement));
353  }
354 
355  Base::Quantity quantity(static_cast<double>(translationIncrementCount.getValue())
356  * translationIncrement.getValue(),
357  Base::Unit::Length);
358 
359  QString message =
360  QString::fromLatin1("%1 %2").arg(QObject::tr("Translation:"), QString::fromStdString(quantity.getUserString()));
361  getMainWindow()->showMessage(message, 3000);
362 }

396 SbVec3f TDragger::roundTranslation(const SbVec3f& vecIn, float incrementIn)
397 {
398  // everything is transformed into local space. That means we only have
399  // worry about the y-value.
400 
401  int yCount = 0;
402  float yValue = vecIn[1];
403 
404  if (fabs(yValue) > (incrementIn / 2.0)) {
405  yCount = static_cast<int>(yValue / incrementIn);
406  float remainder = fmod(yValue, incrementIn);
407  if (remainder >= (incrementIn / 2.0)) {
408  yCount++;
409  }
410  }
411 
412  translationIncrementCount.setValue(yCount);
413 
414  SbVec3f out;
415  out[0] = 0;
416  out[1] = static_cast<float>(yCount) * incrementIn;
417  out[2] = 0.0;
418 
419  return out;
420 }

684 SbVec3f TPlanarDragger::roundTranslation(const SbVec3f& vecIn, float incrementIn)
685 {
686  int xCount = 0;
687  float xValue = vecIn[0];
688 
689  if (fabs(xValue) > (incrementIn / 2.0)) {
690  xCount = static_cast<int>(xValue / incrementIn);
691  float remainder = fmod(xValue, incrementIn);
692  if (remainder >= (incrementIn / 2.0)) {
693  xCount++;
694  }
695  }
696 
697  translationIncrementXCount.setValue(xCount);
698 
699  int yCount = 0;
700  float yValue = vecIn[1];
701 
702  if (fabs(yValue) > (incrementIn / 2.0)) {
703  yCount = static_cast<int>(yValue / incrementIn);
704  float remainder = fmod(yValue, incrementIn);
705  if (remainder >= (incrementIn / 2.0)) {
706  yCount++;
707  }
708  }
709 
710  translationIncrementYCount.setValue(yCount);
711 
712  SbVec3f out;
713  out[0] = static_cast<float>(xCount) * incrementIn;
714  out[1] = static_cast<float>(yCount) * incrementIn;
715  out[2] = 0.0;
716 
717  return out;
718 }
