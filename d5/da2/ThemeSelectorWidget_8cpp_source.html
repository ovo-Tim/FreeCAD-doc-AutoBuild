47 static bool isSystemInDarkMode()
48 {
49  // Auto-detect system setting and default to light mode
50 #if QT_VERSION >= QT_VERSION_CHECK(6, 5, 0)
51  // https://www.qt.io/blog/dark-mode-on-windows-11-with-qt-6.5
52  const auto scheme = QGuiApplication::styleHints()->colorScheme();
53  return scheme == Qt::ColorScheme::Dark;
54 #elif QT_VERSION >= QT_VERSION_CHECK(6, 4, 0)
55  // https://www.qt.io/blog/dark-mode-on-windows-11-with-qt-6.5
56  const QPalette defaultPalette;
57  const auto text = defaultPalette.color(QPalette::WindowText);
58  const auto window = defaultPalette.color(QPalette::Window);
59  return text.lightness() > window.lightness();
60 #else
61 #ifdef FC_OS_MACOSX
62  auto key = CFSTR("AppleInterfaceStyle");
63  if (auto value = CFPreferencesCopyAppValue(key, kCFPreferencesAnyApplication)) {
64  // If the value is "Dark", Dark Mode is enabled
65  if (CFGetTypeID(value) == CFStringGetTypeID()) {
66  if (CFStringCompare((CFStringRef)value, CFSTR("Dark"), kCFCompareCaseInsensitive)
67  == kCFCompareEqualTo) {
68  CFRelease(value);
69  return true;
70  }
71  }
72  CFRelease(value);
73  }
74 #endif // FC_OS_MACOSX
75 #endif // QT_VERSION >= 6.4+
76  return false;
77 }

80 static bool shouldHideClassicTheme()
81 {
82  // Classic on macOS and windows 11 with qt6(.4+?) doesn't work when system
83  // is in dark mode and to make matter worse, on macOS there's a setting that
84  // changes mode depending on time of day.
85 #if QT_VERSION >= QT_VERSION_CHECK(6, 4, 0) || defined(FC_OS_MACOSX) || defined(FC_OS_WIN32)
86  return true;
87 #else
88  return false;
89 #endif
90 }

108 void ThemeSelectorWidget::setupButtons(QBoxLayout* layout)
109 {
110  if (!layout) {
111  return;
112  }
113  std::map<Theme, QString> themeMap {{Theme::Classic, tr("FreeCAD Classic")},
114  {Theme::Dark, tr("FreeCAD Dark")},
115  {Theme::Light, tr("FreeCAD Light")}};
116  std::map<Theme, QIcon> iconMap {
117  {Theme::Classic, QIcon(QLatin1String(":/thumbnails/Theme_thumbnail_classic.png"))},
118  {Theme::Light, QIcon(QLatin1String(":/thumbnails/Theme_thumbnail_light.png"))},
119  {Theme::Dark, QIcon(QLatin1String(":/thumbnails/Theme_thumbnail_dark.png"))}};
120  auto hGrp = App::GetApplication().GetParameterGroupByPath(
121  "User parameter:BaseApp/Preferences/MainWindow");
122  auto styleSheetName = QString::fromStdString(hGrp->GetASCII("StyleSheet"));
123  for (const auto& theme : themeMap) {
124  auto button = gsl::owner<QToolButton*>(new QToolButton());
125 
126  if (theme.first == Theme::Classic && shouldHideClassicTheme()) {
127  button->setVisible(false);
128  }
129 
130  button->setCheckable(true);
131  button->setAutoExclusive(true);
132  button->setToolButtonStyle(Qt::ToolButtonStyle::ToolButtonTextUnderIcon);
133  button->setText(theme.second);
134  button->setIcon(iconMap[theme.first]);
135  button->setIconSize(iconMap[theme.first].actualSize(QSize(256, 256)));
136  if (theme.first == Theme::Classic && styleSheetName.isEmpty()) {
137  button->setChecked(true);
138  }
139  else if (theme.first == Theme::Light
140  && styleSheetName.contains(QLatin1String("FreeCAD Light"),
141  Qt::CaseSensitivity::CaseInsensitive)) {
142  button->setChecked(true);
143  }
144  else if (theme.first == Theme::Dark
145  && styleSheetName.contains(QLatin1String("FreeCAD Dark"),
146  Qt::CaseSensitivity::CaseInsensitive)) {
147  button->setChecked(true);
148  }
149  connect(button, &QToolButton::clicked, this, [this, theme] {
150  themeChanged(theme.first);
151  });
152  layout->addWidget(button);
153  _buttons[static_cast<int>(theme.first)] = button;
154  }
155 }

171 void ThemeSelectorWidget::onLinkActivated(const QString& link)
172 {
173  auto const addonManagerLink = QStringLiteral("freecad:Std_AddonMgr");
174 
175  if (link != addonManagerLink) {
176  return;
177  }
178 
179  // Set the user preferences to include only preference packs.
180  // This is a quick and dirty way to open Addon Manager with only themes.
181  auto pref =
182  App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/Addons");
183  pref->SetInt("PackageTypeSelection", 3); // 3 stands for Preference Packs
184  pref->SetInt("StatusSelection", 0); // 0 stands for any installation status
185 
186  Gui::Application::Instance->commandManager().runCommandByName("Std_AddonMgr");
187 }

189 void ThemeSelectorWidget::preselectThemeFromSystemSettings()
190 {
191  auto nullStyle("<N/A>");
192  auto hGrp = App::GetApplication().GetParameterGroupByPath(
193  "User parameter:BaseApp/Preferences/MainWindow");
194  auto styleSheetName = QString::fromStdString(hGrp->GetASCII("StyleSheet", nullStyle));
195  if (styleSheetName == QString::fromStdString(nullStyle)) {
196  auto theme = isSystemInDarkMode() ? Theme::Dark : Theme::Light;
197  themeChanged(theme);
198  }

201 void ThemeSelectorWidget::themeChanged(Theme newTheme)
202 {
203  // Run the appropriate preference pack:
204  auto prefPackManager = Gui::Application::Instance->prefPackManager();
205  switch (newTheme) {
206  case Theme::Classic:
207  prefPackManager->apply("FreeCAD Classic");
208  break;
209  case Theme::Dark:
210  prefPackManager->apply("FreeCAD Dark");
211  break;
212  case Theme::Light:
213  prefPackManager->apply("FreeCAD Light");
214  break;
215  }
216  ParameterGrp::handle hGrp =
217  App::GetApplication().GetParameterGroupByPath("User parameter:BaseApp/Preferences/Themes");
218  const unsigned long nonExistentColor = -1434171135;
219  const unsigned long defaultAccentColor = 1434171135;
220  unsigned long longAccentColor1 = hGrp->GetUnsigned("ThemeAccentColor1", nonExistentColor);
221  if (longAccentColor1 == nonExistentColor) {
222  hGrp->SetUnsigned("ThemeAccentColor1", defaultAccentColor);
223  hGrp->SetUnsigned("ThemeAccentColor2", defaultAccentColor);
224  hGrp->SetUnsigned("ThemeAccentColor3", defaultAccentColor);
225  }
226 }
