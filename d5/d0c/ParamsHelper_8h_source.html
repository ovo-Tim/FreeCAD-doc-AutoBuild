439 /* Convert ParamHelper defined enum type to user defined ones
440  *
441  * This assumes the user defined enum type is given in \ref ParamSeq "seq_type"
442  * of the parameter definition, and it has the same postfix as the ones
443  * specified in \ref ParamSeq "seq" member of the parameter definition. See
444  * \ref ParamEnumHelper "here" for implementations
445  *
446  * \ingroup ParamEnumHelper
447  *
448  * \arg \c _src: Macro to generate source variable. The signature must be
449  * <tt>_src(_param)<\tt>, where \c _param is the tuple defining the parameter.
450  * You pass any of the \ref ParamAccessor "parameter accessors" to directly
451  * access the field. Or, supply your own macro to append any prefix as you
452  * like.
453  * \arg \c _dst: Same as above.
454  * \arg \c _default: A macro to call for invalid value. Signature should be
455  * <tt>_default(_param)<\tt>, where \c _param is the parameter definition. You
456  * can use #PARAM_ENUM_EXCEPT to throw Base::ValueError exception in FreeCAD
457  * \arg \c _seq: Parameter sequence
458  *
459  * For example, with the following parameter definition
460  * \code{.unparsed}
461  * #define MY_PARAM_TEST \
462  * ((enum,test1,Test1,0,"it's a test",(Foo)(Bar),(MyEnum1,myEnum1)) \
463  * ((enum,test2,Test2,0,"it's a test",(Foo)(Bar),(MyEnum2,myEnum2)))
464  *
465  * #define MY_DST(_param) BOOST_PP_CAT(my,PARAM_FNAME(_param))
466  * \code{.unparsed}
467  *
468  * calling
469  * \code{.unparsed}
470  * PARAM_ENUM_CONVERT(PARAM_FNAME,MY_DST,My,PARAM_ENUM_EXCEP,MY_PARAM_TEST)
471  * \code{.unparsed}
472  *
473  * expands to
474  * \code{.unparsed}
475  * MyEnum1 myTest1;
476  * switch(Test1) {
477  * case Test1Foo:
478  * myTest1 = myEnum1Foo;
479  * break;
480  * case Test1Bar:
481  * myTest1 = myEnum1Bar;
482  * break;
483  * default:
484  * throw Base::ValueError("invalid value for enum Test1");
485  * }
486  * MyEnum2 myTest2;
487  * switch(Test2) {
488  * case Test1Foo:
489  * myTest2 = myEnum2Foo;
490  * break;
491  * case Test2Bar:
492  * myTest2 = myEnum2Bar;
493  * break;
494  * default:
495  * throw Base::ValueError("invalid value for enum Test2");
496  * }
497  * \endcode
498  *
499  * The above code assumes you've already defined \a Test1 and \a Test2 some
500  * where as the source variable.
501  */
502 #define PARAM_ENUM_CONVERT(_src, _dst, _default, _seq) \
503  BOOST_PP_SEQ_FOR_EACH(PARAM_ENUM_CONVERT_, (_src, _dst, _default), _seq)
