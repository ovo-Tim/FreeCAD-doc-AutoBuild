269  def _splitInCompound(self, compound, existing_pieces):
270  """Splits aggregates inside compound. Returns None if nothing is split, otherwise
271  returns compound.
272  existing_pieces is a dict. Key is deep hash. Value is tuple (int, shape). It is
273  used to search for if this split piece was already generated, and reuse the old
274  one."""
275 
276  changed = False
277  new_children = []
278  for piece in compound.childShapes():
279  if piece.ShapeType == "Compound":
280  subspl = self._splitInCompound(piece, existing_pieces)
281  if subspl is None:
282  new_children.append(piece)
283  else:
284  new_children.append(subspl)
285  changed = True
286  else:
287  new_pieces = self.makeSplitPieces(piece)
288  changed = changed or len(new_pieces)>1
289  for new_piece in new_pieces:
290  hash = HashableShape_Deep(new_piece)
291  dummy,ex_piece = existing_pieces.get(hash, (None, None))
292  if ex_piece is not None:
293  new_children.append(ex_piece)
294  changed = True
295  else:
296  new_children.append(new_piece)
297  existing_pieces[hash] = (-1, new_piece)

304  def makeSplitPieces(self, shape):
305  """makeSplitPieces(self, shape): splits a shell, wire or compsolid into pieces where
306  it intersects with other shapes.
307 
308  Returns list of split pieces. If no splits were done, returns list containing the
309  original shape."""
310 
311  if shape.ShapeType == "Wire":
312  bit_extractor = lambda sh: sh.Edges
313  joint_extractor = lambda sh: sh.Vertexes
314  elif shape.ShapeType == "Shell":
315  bit_extractor = lambda sh: sh.Faces
316  joint_extractor = lambda sh: sh.Edges
317  elif shape.ShapeType == "CompSolid":
318  bit_extractor = lambda sh: sh.Solids
319  joint_extractor = lambda sh: sh.Faces
320  else:
321  #can't split the shape
322  return [shape]
323 
324  # for each joint, test if all bits it's connected to are from same number of sources.
325  # If not, this is a joint for splitting
326  # FIXME: this is slow, and maybe can be optimized
327  splits = []
328  for joint in joint_extractor(shape):
329  joint_overlap_count = len(self._element_to_source[HashableShape(joint)])
330  if joint_overlap_count > 1:
331  # find elements in pieces that are connected to joint
332  for bit in bit_extractor(self.gfa_return[0]):
333  for joint_bit in joint_extractor(bit):
334  if joint_bit.isSame(joint):
335  #bit is connected to joint!
336  bit_overlap_count = len(self._element_to_source[HashableShape(bit)])
337  assert(bit_overlap_count <= joint_overlap_count)
338  if bit_overlap_count < joint_overlap_count:
339  if len(splits) == 0 or splits[-1] is not joint:
340  splits.append(joint)
341  if len(splits)==0:
342  #shape was not split - no split points found
343  return [shape]
344 
345  from . import ShapeMerge
346 
347  new_pieces = ShapeMerge.mergeShapes(bit_extractor(shape), split_connections= splits, bool_compsolid= True).childShapes()
348  if len(new_pieces) == 1:
349  #shape was not split (split points found, but the shape remained in one piece).
350  return [shape]
351  return new_pieces
