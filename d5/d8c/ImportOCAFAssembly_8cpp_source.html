103 void ImportOCAFAssembly::loadShapes(const TDF_Label& label,
104  const TopLoc_Location& loc,
105  const std::string& defaultname,
106  const std::string& assembly,
107  bool isRef,
108  int dep)
109 {
110  int hash = 0;
111  TopoDS_Shape aShape;
112  if (aShapeTool->GetShape(label, aShape)) {
113  hash = Part::ShapeMapHasher {}(aShape);
114  }
115 
116  Handle(TDataStd_Name) name;
117  std::string part_name = defaultname;
118  if (label.FindAttribute(TDataStd_Name::GetID(), name)) {
119  TCollection_ExtendedString extstr = name->Get();
120  char* str = new char[extstr.LengthOfCString() + 1];
121  extstr.ToUTF8CString(str);
122  part_name = str;
123  delete[] str;
124  if (part_name.empty()) {
125  part_name = defaultname;
126  }
127  else {
128  bool ws = true;
129  for (std::string::iterator it = part_name.begin(); it != part_name.end(); ++it) {
130  if (*it != ' ') {
131  ws = false;
132  break;
133  }
134  }
135  if (ws) {
136  part_name = defaultname;
137  }
138  }
139  }
140 
141  TopLoc_Location part_loc = loc;
142  Handle(XCAFDoc_Location) hLoc;
143  if (label.FindAttribute(XCAFDoc_Location::GetID(), hLoc)) {
144  if (isRef) {
145  part_loc = part_loc * hLoc->Get();
146  }
147  else {
148  part_loc = hLoc->Get();
149  }
150  }
151 
152 #ifdef FC_DEBUG
153  const char* s;
154  if (!hLoc.IsNull()) {
155  s = hLoc->Get().IsIdentity() ? "0" : "1";
156  }
157  else {
158  s = "0";
159  }
160 
161  std::stringstream str;
162 
163  Base::Console().Log("H:%-9d \tN:%-30s \tTop:%d, Asm:%d, Shape:%d, Compound:%d, Simple:%d, "
164  "Free:%d, Ref:%d, Component:%d, SubShape:%d\tTrf:%s-- Dep:%d \n",
165  hash,
166  part_name.c_str(),
167  aShapeTool->IsTopLevel(label),
168  aShapeTool->IsAssembly(label),
169  aShapeTool->IsShape(label),
170  aShapeTool->IsCompound(label),
171  aShapeTool->IsSimpleShape(label),
172  aShapeTool->IsFree(label),
173  aShapeTool->IsReference(label),
174  aShapeTool->IsComponent(label),
175  aShapeTool->IsSubShape(label),
176  s,
177  dep);
178 
179  label.Dump(str);
180  Base::Console().Message(str.str().c_str());
181 #endif
182 
183  std::string asm_name = assembly;
184  if (aShapeTool->IsAssembly(label)) {
185  asm_name = part_name;
186  }
187 
188  TDF_Label ref;
189  if (aShapeTool->IsReference(label) && aShapeTool->GetReferredShape(label, ref)) {
190  loadShapes(ref, part_loc, part_name, asm_name, true, dep + 1);
191  }
192 
193  if (isRef || myRefShapes.find(hash) == myRefShapes.end()) {
194  TopoDS_Shape aShape;
195  if (isRef && aShapeTool->GetShape(label, aShape)) {
196  myRefShapes.insert(Part::ShapeMapHasher {}(aShape));
197  }
198 
199  if (aShapeTool->IsSimpleShape(label) && (isRef || aShapeTool->IsFree(label))) {
200  if (!asm_name.empty()) {
201  part_name = asm_name;
202  }
203  if (isRef) {
204  createShape(label, loc, part_name);
205  }
206  else {
207  createShape(label, part_loc, part_name);
208  }
209  }
210  else {
211  for (TDF_ChildIterator it(label); it.More(); it.Next()) {
212  loadShapes(it.Value(), part_loc, part_name, asm_name, isRef, dep + 1);
213  }
214  }
215  }
216 }

218 void ImportOCAFAssembly::createShape(const TDF_Label& label,
219  const TopLoc_Location& loc,
220  const std::string& name)
221 {
222  Base::Console().Log("-create Shape\n");
223  const TopoDS_Shape& aShape = aShapeTool->GetShape(label);
224  if (!aShape.IsNull() && aShape.ShapeType() == TopAbs_COMPOUND) {
225  TopExp_Explorer xp;
226  int ctSolids = 0, ctShells = 0;
227  for (xp.Init(aShape, TopAbs_SOLID); xp.More(); xp.Next(), ctSolids++) {
228  createShape(xp.Current(), loc, name);
229  }
230  for (xp.Init(aShape, TopAbs_SHELL, TopAbs_SOLID); xp.More(); xp.Next(), ctShells++) {
231  createShape(xp.Current(), loc, name);
232  }
233  if (ctSolids > 0 || ctShells > 0) {
234  return;
235  }
236  }
237  createShape(aShape, loc, name);
238 }

240 void ImportOCAFAssembly::createShape(const TopoDS_Shape& aShape,
241  const TopLoc_Location& loc,
242  const std::string& name)
243 {
244  Part::Feature* part = static_cast<Part::Feature*>(doc->addObject("Part::Feature"));
245  if (!loc.IsIdentity()) {
246  part->Shape.setValue(aShape.Moved(loc));
247  }
248  else {
249  part->Shape.setValue(aShape);
250  }
251  part->Label.setValue(name);
252 
253  Quantity_Color aColor;
254  App::Color color(0.8f, 0.8f, 0.8f);
255  if (aColorTool->GetColor(aShape, XCAFDoc_ColorGen, aColor)
256  || aColorTool->GetColor(aShape, XCAFDoc_ColorSurf, aColor)
257  || aColorTool->GetColor(aShape, XCAFDoc_ColorCurv, aColor)) {
258  color.r = (float)aColor.Red();
259  color.g = (float)aColor.Green();
260  color.b = (float)aColor.Blue();
261  std::vector<App::Color> colors;
262  colors.push_back(color);
263  applyColors(part, colors);
264  }
265 
266  TopTools_IndexedMapOfShape faces;
267  TopExp_Explorer xp(aShape, TopAbs_FACE);
268  while (xp.More()) {
269  faces.Add(xp.Current());
270  xp.Next();
271  }
272  bool found_face_color = false;
273  std::vector<App::Color> faceColors;
274  faceColors.resize(faces.Extent(), color);
275  xp.Init(aShape, TopAbs_FACE);
276  while (xp.More()) {
277  if (aColorTool->GetColor(xp.Current(), XCAFDoc_ColorGen, aColor)
278  || aColorTool->GetColor(xp.Current(), XCAFDoc_ColorSurf, aColor)
279  || aColorTool->GetColor(xp.Current(), XCAFDoc_ColorCurv, aColor)) {
280  int index = faces.FindIndex(xp.Current());
281  color.r = (float)aColor.Red();
282  color.g = (float)aColor.Green();
283  color.b = (float)aColor.Blue();
284  faceColors[index - 1] = color;
285  found_face_color = true;
286  }
287  xp.Next();
288  }
289 
290  if (found_face_color) {
291  applyColors(part, faceColors);
292  }
293 }
