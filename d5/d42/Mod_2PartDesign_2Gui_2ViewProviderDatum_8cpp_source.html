101 void ViewProviderDatum::attach(App::DocumentObject *obj)
102 {
103  if (auto geo = dynamic_cast<App::GeoFeature*>(obj)) {
104  geo->setMaterialAppearance(ShapeAppearance[0]);
105  }
106 
107  ViewProviderGeometryObject::attach ( obj );
108 
109  // TODO remove this field (2015-09-08, Fat-Zer)
110  App::DocumentObject* o = getObject();
111  if (o->is<PartDesign::Plane>()) {
112  datumType = QString::fromLatin1("Plane");
113  datumText = QObject::tr("Plane");
114  datumMenuText = tr("Datum Plane parameters");
115  }
116  else if (o->is<PartDesign::Line>()) {
117  datumType = QString::fromLatin1("Line");
118  datumText = QObject::tr("Line");
119  datumMenuText = tr("Datum Line parameters");
120  }
121  else if (o->is<PartDesign::Point>()) {
122  datumType = QString::fromLatin1("Point");
123  datumText = QObject::tr("Point");
124  datumMenuText = tr("Datum Point parameters");
125  }
126  else if (o->is<PartDesign::CoordinateSystem>()) {
127  datumType = QString::fromLatin1("CoordinateSystem");
128  datumText = QObject::tr("Coordinate System");
129  datumMenuText = tr("Local Coordinate System parameters");
130  }
131 
132  SoShapeHints* hints = new SoShapeHints();
133  hints->shapeType.setValue(SoShapeHints::UNKNOWN_SHAPE_TYPE);
134  hints->vertexOrdering.setValue(SoShapeHints::COUNTERCLOCKWISE);
135  SoDrawStyle* fstyle = new SoDrawStyle();
136  fstyle->style = SoDrawStyle::FILLED;
137  fstyle->lineWidth = 3;
138  fstyle->pointSize = 5;
139  pPickStyle->style = SoPickStyle::SHAPE;
140  SoMaterialBinding* matBinding = new SoMaterialBinding;
141  matBinding->value = SoMaterialBinding::OVERALL;
142 
143  SoSeparator* sep = new SoSeparator();
144  sep->addChild(hints);
145  sep->addChild(fstyle);
146  sep->addChild(pPickStyle);
147  sep->addChild(matBinding);
148  sep->addChild(pcShapeMaterial);
149  sep->addChild(pShapeSep);
150 
151  addDisplayMaskMode(sep, "Base");
152 }

176 std::string ViewProviderDatum::getElement(const SoDetail* detail) const
177 {
178  if (detail) {
179  int element = 1;
180 
181  if (detail->getTypeId() == SoLineDetail::getClassTypeId()) {
182  const SoLineDetail* line_detail = static_cast<const SoLineDetail*>(detail);
183  element = line_detail->getLineIndex();
184  } else if (detail->getTypeId() == SoFaceDetail::getClassTypeId()) {
185  const SoFaceDetail* face_detail = static_cast<const SoFaceDetail*>(detail);
186  element = face_detail->getFaceIndex();
187  } else if (detail->getTypeId() == SoPointDetail::getClassTypeId()) {
188  const SoPointDetail* point_detail = static_cast<const SoPointDetail*>(detail);
189  element = point_detail->getCoordinateIndex();
190  }
191 
192  if (element == 0)
193  return datumType.toStdString();
194  }
195 
196  return {};
197 }

234 bool ViewProviderDatum::setEdit(int ModNum)
235 {
236  if (!ViewProvider::setEdit(ModNum))
237  return false;
238  // TODO Share this code with Features view providers somehow (2015-09-08, Fat-Zer)
239  if (ModNum == ViewProvider::Default ) {
240  // When double-clicking on the item for this datum feature the
241  // object unsets and sets its edit mode without closing
242  // the task panel
243  Gui::TaskView::TaskDialog *dlg = Gui::Control().activeDialog();
244  TaskDlgDatumParameters *datumDlg = qobject_cast<TaskDlgDatumParameters *>(dlg);
245  if (datumDlg && datumDlg->getViewProvider() != this)
246  datumDlg = nullptr; // another datum feature left open its task panel
247  if (dlg && !datumDlg) {
248  QMessageBox msgBox;
249  msgBox.setText(QObject::tr("A dialog is already open in the task panel"));
250  msgBox.setInformativeText(QObject::tr("Do you want to close this dialog?"));
251  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
252  msgBox.setDefaultButton(QMessageBox::Yes);
253  int ret = msgBox.exec();
254  if (ret == QMessageBox::Yes)
255  Gui::Control().closeDialog();
256  else
257  return false;
258  }
259 
260  // clear the selection (convenience)
261  Gui::Selection().clearSelection();
262 
263  oldWb = Gui::Command::assureWorkbench("PartDesignWorkbench");
264 
265  // start the edit dialog
266  if (datumDlg)
267  Gui::Control().showDialog(datumDlg);
268  else
269  Gui::Control().showDialog(new TaskDlgDatumParameters(this));
270 
271  return true;
272  }
273  else {
274  return ViewProvider::setEdit(ModNum);
275  }
276 }

331 SbBox3f ViewProviderDatum::getRelevantBoundBox () const {
332  std::vector<App::DocumentObject *> objs;
333 
334  // Probe body first
335  PartDesign::Body* body = PartDesign::Body::findBodyOf ( this->getObject() );
336  if (body) {
337  objs = body->getFullModel ();
338  } else {
339  // Probe if we belongs to some group
340  App::DocumentObject* group = App::DocumentObjectGroup::getGroupOfObject ( this->getObject () );
341 
342  if(group) {
343  auto* ext = group->getExtensionByType<App::GroupExtension>();
344  if(ext)
345  objs = ext->getObjects ();
346  } else {
347  // Fallback to whole document
348  objs = this->getObject ()->getDocument ()->getObjects ();
349  }
350  }
351 
352  Gui::View3DInventor* view = dynamic_cast<Gui::View3DInventor*>(this->getActiveView());
353  if(view){
354  Gui::View3DInventorViewer* viewer = view->getViewer();
355  SoGetBoundingBoxAction bboxAction(viewer->getSoRenderManager()->getViewportRegion());
356  SbBox3f bbox = getRelevantBoundBox (bboxAction, objs);
357 
358  if ( bbox.getVolume () < Precision::Confusion() ) {
359  bbox.extendBy ( defaultBoundBox () );
360  }
361  return bbox;
362  } else {
363  return defaultBoundBox();
364  }
365 }

367 SbBox3f ViewProviderDatum::getRelevantBoundBox (
368  SoGetBoundingBoxAction &bboxAction, const std::vector <App::DocumentObject *> &objs )
369 {
370  SbBox3f bbox = defaultBoundBox();
371 
372  // Adds the bbox of given feature to the output
373  for (auto obj :objs) {
374  ViewProvider *vp = Gui::Application::Instance->getViewProvider(obj);
375  if (!vp) { continue; }
376  if (!vp->isVisible ()) { continue; }
377 
378  if (obj->isDerivedFrom (Part::Datum::getClassTypeId() ) ) {
379  // Treat datums only as their basepoint
380  // I hope it's ok to take FreeCAD's point here
381  Base::Vector3d basePoint = static_cast<Part::Datum *> ( obj )->getBasePoint ();
382  bbox.extendBy (SbVec3f(basePoint.x, basePoint.y, basePoint.z ));
383  } else {
384  bboxAction.apply ( vp->getRoot () );
385  SbBox3f obj_bbox = bboxAction.getBoundingBox ();
386 
387  if ( obj_bbox.getVolume () < Precision::Infinite () ) {
388  bbox.extendBy ( obj_bbox );
389  }
390  }
391  }
392 
393  // TODO: shrink bbox when all other elements are too small
394  return bbox;
395 }
