1357 void MarkerBitmaps::createBitmap(const std::string& name, int px, int width, int height, const char* marker)
1358 {
1359  int byteidx = 0;
1360  const int byteWidth = (width + 7) / 2;
1361  int size = byteWidth * height;
1362  std::vector<unsigned char> bitmapbytes(size);
1363 
1364  for (int h = 0; h < height; h++) {
1365  unsigned char bits = 0;
1366  for (int w = 0; w < width; w++) {
1367  if (marker[(h * width) + w] != ' ') {
1368  bits |= (0x80 >> (w % 8));
1369  }
1370  if ((((w + 1) % 8) == 0) || (w == width - 1)) {
1371  bitmapbytes[byteidx++] = bits;
1372  bits = 0;
1373  }
1374  }
1375  }
1376 
1377  int MY_BITMAP_IDX = SoMarkerSet::getNumDefinedMarkers(); // add at end
1378  SoMarkerSet::addMarker(MY_BITMAP_IDX, SbVec2s(width, height),
1379  &(bitmapbytes[0]), FALSE, TRUE);
1380 
1381  markerIndex[std::make_pair(name, px)] = MY_BITMAP_IDX;
1382 }
