85  def recursive_widget_saver(self, widget):
86  """Writes out the data for this widget and all of its children, recursively."""
87  if isinstance(widget, QWidget):
88  # See if it's one of ours:
89  pref_path = widget.property("prefPath")
90  pref_entry = widget.property("prefEntry")
91  if pref_path and pref_entry:
92  pref_path = pref_path.data()
93  pref_entry = pref_entry.data()
94  pref_access_string = f"User parameter:BaseApp/Preferences/{str(pref_path,'utf-8')}"
95  pref = FreeCAD.ParamGet(pref_access_string)
96  if isinstance(widget, QCheckBox):
97  checked = widget.isChecked()
98  pref.SetBool(str(pref_entry, "utf-8"), checked)
99  elif isinstance(widget, QRadioButton):
100  checked = widget.isChecked()
101  pref.SetBool(str(pref_entry, "utf-8"), checked)
102  elif isinstance(widget, QComboBox):
103  new_index = widget.currentIndex()
104  pref.SetInt(str(pref_entry, "utf-8"), new_index)
105  elif isinstance(widget, QTextEdit):
106  text = widget.toPlainText()
107  pref.SetString(str(pref_entry, "utf-8"), text)
108  elif isinstance(widget, QLineEdit):
109  text = widget.text()
110  pref.SetString(str(pref_entry, "utf-8"), text)
111  elif widget.metaObject().className() == "Gui::PrefFileChooser":
112  filename = str(widget.property("fileName"))
113  filename = pref.SetString(str(pref_entry, "utf-8"), filename)
114 
115  # Recurse over children
116  if isinstance(widget, QtCore.QObject):
117  for child in widget.children():
118  self.recursive_widget_saver(child)

190 class CustomRepoDataModel(QtCore.QAbstractTableModel):
191  """The model for the custom repositories: wraps the underlying preference data and uses that
192  as its main data store."""
193 
194  def __init__(self):
195  super().__init__()
196  pref_access_string = "User parameter:BaseApp/Preferences/Addons"
197  self.pref = FreeCAD.ParamGet(pref_access_string)
198  self.load_model()
199 
200  def load_model(self):
201  """Load the data from the preferences entry"""
202  pref_entry: str = self.pref.GetString("CustomRepositories", "")
203 
204  # The entry is saved as a space- and newline-delimited text block: break it into its
205  # constituent parts
206  lines = pref_entry.split("\n")
207  self.model = []
208  for line in lines:
209  if not line:
210  continue
211  split_data = line.split()
212  if len(split_data) > 1:
213  branch = split_data[1]
214  else:
215  branch = "master"
216  url = split_data[0]
217  self.model.append([url, branch])
218 
219  def save_model(self):
220  """Save the data into a preferences entry"""
221  entry = ""
222  for row in self.model:
223  entry += f"{row[0]} {row[1]}\n"
224  self.pref.SetString("CustomRepositories", entry)

238  def data(self, index, role=QtCore.Qt.DisplayRole):
239  """The data at an index."""
240  if role != QtCore.Qt.DisplayRole:
241  return None
242  row = index.row()
243  column = index.column()
244  if row > len(self.model):
245  return None
246  if column > 1:
247  return None
248  return self.model[row][column]

251  """Get the row and column header data."""

271  """Remove rows"""
