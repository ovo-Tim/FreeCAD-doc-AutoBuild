84  def show_repo(self, repo: Addon) -> None:
85  """The main entry point for this class, shows the package details and related buttons
86  for the provided repo."""
87  self.addon = repo
88  self.readme_controller.set_addon(repo)
89  self.original_disabled_state = self.addon.is_disabled()
90  if repo is not None:
91  self.ui.button_bar.show()
92  else:
93  self.ui.button_bar.hide()
94 
95  if self.worker is not None:
96  if not self.worker.isFinished():
97  self.worker.requestInterruption()
98  self.worker.wait()
99 
100  installed = self.addon.status() != Addon.Status.NOT_INSTALLED
101  self.ui.set_installed(installed)
102  if repo.metadata is not None:
103  self.ui.set_url(get_repo_url_from_metadata(repo.metadata))
104  else:
105  self.ui.set_url(None) # to reset it and hide it
106  update_info = UpdateInformation()
107  if installed:
108  update_info.unchecked = self.addon.status() == Addon.Status.UNCHECKED
109  update_info.update_available = self.addon.status() == Addon.Status.UPDATE_AVAILABLE
110  update_info.check_in_progress = False # TODO: Implement the "check in progress" status
111  if repo.metadata:
112  update_info.branch = get_branch_from_metadata(repo.metadata)
113  update_info.version = repo.metadata.version
114  elif repo.macro:
115  update_info.version = repo.macro.version
116  self.ui.set_update_available(update_info)
117  self.ui.set_location(os.path.join(self.addon.mod_directory, self.addon.name))
118  self.ui.set_disabled(self.addon.is_disabled())
119  self.ui.allow_running(repo.repo_type == Addon.Kind.MACRO)
120  self.ui.allow_disabling(repo.repo_type != Addon.Kind.MACRO)
121 
122  if repo.repo_type == Addon.Kind.MACRO:
123  self.update_macro_info(repo)
124 
125  if repo.status() == Addon.Status.UNCHECKED:
126  self.ui.button_bar.check_for_update.show()
127  self.ui.button_bar.check_for_update.setText(
128  translate("AddonsInstaller", "Check for " "update")
129  )
130  self.ui.button_bar.check_for_update.setEnabled(True)
131  if not self.update_check_thread:
132  self.update_check_thread = QtCore.QThread()
133  self.check_for_update_worker = CheckSingleUpdateWorker(repo)
134  self.check_for_update_worker.moveToThread(self.update_check_thread)
135  self.update_check_thread.finished.connect(self.check_for_update_worker.deleteLater)
136  self.ui.button_bar.check_for_update.clicked.connect(
137  self.check_for_update_worker.do_work
138  )
139  self.check_for_update_worker.update_status.connect(self.display_repo_status)
140  self.update_check_thread.start()
141  else:
142  self.ui.button_bar.check_for_update.hide()
143 
144  flags = WarningFlags()
145  flags.required_freecad_version = self.requires_newer_freecad()
146  flags.obsolete = repo.obsolete
147  flags.python2 = repo.python2
148  self.ui.set_warning_flags(flags)
149  self.set_change_branch_button_state()

151  def requires_newer_freecad(self) -> Optional[Version]:
152  """If the current package is not installed, returns the first supported version of
153  FreeCAD, if one is set, or None if no information is available (or if the package is
154  already installed)."""
155 
156  # If it's not installed, check to see if it's for a newer version of FreeCAD
157  if self.addon.status() == Addon.Status.NOT_INSTALLED and self.addon.metadata:
158  # Only hide if ALL content items require a newer version, otherwise
159  # it's possible that this package actually provides versions of itself
160  # for newer and older versions
161 
162  first_supported_version = get_first_supported_freecad_version(self.addon.metadata)
163  if first_supported_version is not None:
164  fc_version = Version(from_list=fci.Version())
165  if first_supported_version > fc_version:
166  return first_supported_version
167  return None

169  def set_change_branch_button_state(self):
170  """The change branch button is only available for installed Addons that have a .git directory
171  and in runs where the git is available."""
172 
173  self.ui.button_bar.change_branch.hide()
174 
175  pref = fci.ParamGet("User parameter:BaseApp/Preferences/Addons")
176  show_switcher = pref.GetBool("ShowBranchSwitcher", False)
177  if not show_switcher:
178  return
179 
180  # Is this repo installed? If not, return.
181  if self.addon.status() == Addon.Status.NOT_INSTALLED:
182  return
183 
184  # Is it a Macro? If so, return:
185  if self.addon.repo_type == Addon.Kind.MACRO:
186  return
187 
188  # Can we actually switch branches? If not, return.
189  if not self.git_manager:
190  return
191 
192  # Is there a .git subdirectory? If not, return.
193  basedir = fci.getUserAppDataDir()
194  path_to_git = os.path.join(basedir, "Mod", self.addon.name, ".git")
195  if not os.path.isdir(path_to_git):
196  return
197 
198  # If all four above checks passed, then it's possible for us to switch
199  # branches, if there are any besides the one we are on: show the button
200  self.ui.button_bar.change_branch.show()

244  def branch_changed(self, old_branch: str, name: str) -> None:
245  """Displays a dialog confirming the branch changed, and tries to access the
246  metadata file from that branch."""
247  QtWidgets.QMessageBox.information(
248  self.ui,
249  translate("AddonsInstaller", "Success"),
250  translate(
251  "AddonsInstaller",
252  "Branch change succeeded.\n"
253  "Moved\n"
254  "from: {}\n"
255  "to: {}\n"
256  "Please restart to use the new version.",
257  ).format(old_branch, name),
258  )
259  # See if this branch has a package.xml file:
260  basedir = fci.getUserAppDataDir()
261  path_to_metadata = os.path.join(basedir, "Mod", self.addon.name, "package.xml")
262  if os.path.isfile(path_to_metadata):
263  self.addon.load_metadata_file(path_to_metadata)
264  self.addon.installed_version = self.addon.metadata.version
265  else:
266  self.addon.repo_type = Addon.Kind.WORKBENCH
267  self.addon.metadata = None
268  self.addon.installed_version = None
269  self.addon.updated_timestamp = QtCore.QDateTime.currentDateTime().toSecsSinceEpoch()
270  self.addon.branch = name
271  self.addon.set_status(Addon.Status.PENDING_RESTART)
272  self.ui.set_new_branch(name)
273  self.update_status.emit(self.addon)
