56 Py::Object wrapFromWidgetFactory(const Py::Tuple& args, const std::function<QWidget*(const QString&, QWidget *, const QString&)> & callableFunc)
57 {
58  Gui::PythonWrapper wrap;
59 
60  // 1st argument
61  Py::String str(args[0]);
62  std::string className;
63  className = str.as_std_string("utf-8");
64  // 2nd argument
65  QWidget* parent = nullptr;
66  if (wrap.loadCoreModule() && args.size() > 1) {
67  QObject* object = wrap.toQObject(args[1]);
68  if (object)
69  parent = qobject_cast<QWidget*>(object);
70  }
71 
72  // 3rd argument
73  std::string objectName;
74  if (args.size() > 2) {
75  Py::String str(args[2]);
76  objectName = str.as_std_string("utf-8");
77  }
78 
79  QWidget* widget = callableFunc(QString::fromLatin1(className.c_str()), parent,
80  QString::fromLatin1(objectName.c_str()));
81  if (!widget) {
82  return Py::None();
83  // std::string err = "No such widget class '";
84  // err += className;
85  // err += "'";
86  // throw Py::RuntimeError(err);
87  }
88  wrap.loadGuiModule();
89  wrap.loadWidgetsModule();
90 
91  const char* typeName = wrap.getWrapperName(widget);
92  return wrap.fromQWidget(widget, typeName);
93 }

110 Py::Object PySideUicModule::loadUiType(const Py::Tuple& args)
111 {
112  Base::PyGILStateLocker lock;
113  PyObject* main = PyImport_AddModule("__main__");
114  PyObject* dict = PyModule_GetDict(main);
115  Py::Dict d(PyDict_Copy(dict), true);
116  Py::String uiFile(args.getItem(0));
117  std::string file = uiFile.as_string();
118  std::replace(file.begin(), file.end(), '\\', '/');
119 
120  QString cmd;
121  QTextStream str(&cmd);
122  // https://github.com/albop/dolo/blob/master/bin/load_ui.py
123  str << "import pyside2uic\n"
124  << "from PySide import QtCore, QtGui, QtWidgets\n"
125  << "import xml.etree.ElementTree as xml\n"
126  << "try:\n"
127  << " from cStringIO import StringIO\n"
128  << "except Exception:\n"
129  << " from io import StringIO\n"
130  << "\n"
131  << "uiFile = \"" << file.c_str() << "\"\n"
132  << "parsed = xml.parse(uiFile)\n"
133  << "widget_class = parsed.find('widget').get('class')\n"
134  << "form_class = parsed.find('class').text\n"
135  << "with open(uiFile, 'r') as f:\n"
136  << " o = StringIO()\n"
137  << " frame = {}\n"
138  << " pyside2uic.compileUi(f, o, indent=0)\n"
139  << " pyc = compile(o.getvalue(), '<string>', 'exec')\n"
140  << " exec(pyc, frame)\n"
141  << " #Fetch the base_class and form class based on their type in the xml from designer\n"
142  << " form_class = frame['Ui_%s'%form_class]\n"
143  << " base_class = eval('QtWidgets.%s'%widget_class)\n";
144 
145  PyObject* result = PyRun_String((const char*)cmd.toLatin1(), Py_file_input, d.ptr(), d.ptr());
146  if (result) {
147  Py_DECREF(result);
148  if (d.hasKey("form_class") && d.hasKey("base_class")) {
149  Py::Tuple t(2);
150  t.setItem(0, d.getItem("form_class"));
151  t.setItem(1, d.getItem("base_class"));
152  return t;
153  }
154  }
155  else {
156  throw Py::Exception();
157  }
158 
159  return Py::None();
160 }

162 Py::Object PySideUicModule::loadUi(const Py::Tuple& args)
163 {
164  Base::PyGILStateLocker lock;
165  PyObject* main = PyImport_AddModule("__main__");
166  PyObject* dict = PyModule_GetDict(main);
167  Py::Dict d(PyDict_Copy(dict), true);
168  d.setItem("uiFile_", args[0]);
169  if (args.size() > 1)
170  d.setItem("base_", args[1]);
171  else
172  d.setItem("base_", Py::None());
173 
174  QString cmd;
175  QTextStream str(&cmd);
176 
177  str << "from PySide import QtCore, QtGui, QtWidgets\n"
178  << "import FreeCADGui"
179  << "\n"
180  << "loader = FreeCADGui.UiLoader()\n"
181  << "widget = loader.load(globals()[\"uiFile_\"])\n"
182  << "\n";
183 
184 
185  PyObject* result = PyRun_String((const char*)cmd.toLatin1(), Py_file_input, d.ptr(), d.ptr());
186  if (result) {
187  Py_DECREF(result);
188  if (d.hasKey("widget")) {
189  return d.getItem("widget");
190  }
191  }
192  else {
193  throw Py::Exception();
194  }
195 
196  return Py::None();
197 }

570 Py::Object UiLoaderPy::load(const Py::Tuple& args)
571 {
572  Gui::PythonWrapper wrap;
573  if (wrap.loadCoreModule()) {
574  std::string fn;
575  QFile file;
576  QIODevice* device = nullptr;
577  QWidget* parent = nullptr;
578  if (wrap.toCString(args[0], fn)) {
579  file.setFileName(QString::fromUtf8(fn.c_str()));
580  if (!file.open(QFile::ReadOnly))
581  throw Py::RuntimeError("Cannot open file");
582  device = &file;
583  }
584  else if (args[0].isString()) {
585  fn = static_cast<std::string>(Py::String(args[0]));
586  file.setFileName(QString::fromUtf8(fn.c_str()));
587  if (!file.open(QFile::ReadOnly))
588  throw Py::RuntimeError("Cannot open file");
589  device = &file;
590  }
591  else {
592  QObject* obj = wrap.toQObject(args[0]);
593  device = qobject_cast<QIODevice*>(obj);
594  }
595 
596  if (args.size() > 1) {
597  QObject* obj = wrap.toQObject(args[1]);
598  parent = qobject_cast<QWidget*>(obj);
599  }
600 
601  if (device) {
602  QWidget* widget = loader->load(device, parent);
603  if (widget) {
604  wrap.loadGuiModule();
605  wrap.loadWidgetsModule();
606 
607  const char* typeName = wrap.getWrapperName(widget);
608  Py::Object pyWdg = wrap.fromQWidget(widget, typeName);
609  wrap.createChildrenNameAttributes(*pyWdg, widget);
610  wrap.setParent(*pyWdg, parent);
611  return pyWdg;
612  }
613  }
614  else {
615  throw Py::TypeError("string or QIODevice expected");
616  }
617  }
618  return Py::None();
619 }
