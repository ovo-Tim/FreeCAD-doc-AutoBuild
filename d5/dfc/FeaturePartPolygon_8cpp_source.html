51 App::DocumentObjectExecReturn *Part::Polygon::execute()
52 {
53  BRepBuilderAPI_MakePolygon poly;
54  const std::vector<Base::Vector3d> nodes = Nodes.getValues();
55 
56  for (const auto & node : nodes) {
57  gp_Pnt pnt(node.x, node.y, node.z);
58  poly.Add(pnt);
59  }
60 
61  if (Close.getValue())
62  poly.Close();
63 
64  if (!poly.IsDone())
65  throw Base::CADKernelError("Cannot create polygon because less than two vertices are given");
66  TopoDS_Wire wire = poly.Wire();
67  this->Shape.setValue(wire);

69  return App::DocumentObject::StdReturn;
