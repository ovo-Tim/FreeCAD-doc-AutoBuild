68  void update()
69  {
70  auto &manager = Application::Instance->commandManager();
71  if (revision == _CommandRevision && _CommandRevision == manager.getRevision())
72  return;
73  beginResetModel();
74  revision = manager.getRevision();
75  if (revision != _CommandRevision) {
76  _CommandRevision = revision;
77  _CommandRevision = manager.getRevision();
78  _Commands.clear();
79  for (auto &v : manager.getCommands()) {
80  _Commands.emplace_back();
81  auto &info = _Commands.back();
82  info.cmd = v.second;
83  }
84  }
85  endResetModel();
86  }

93  QVariant data(const QModelIndex & index, int role) const override
94  {
95  if (index.row() < 0 || index.row() >= (int)_Commands.size())
96  return {};
97 
98  auto &info = _Commands[index.row()];
99 
100  switch(role) {
101  case Qt::DisplayRole:
102  case Qt::EditRole: {
103  QString title = QStringLiteral("%1 (%2)").arg(
104  Action::commandMenuText(info.cmd),
105  QString::fromUtf8(info.cmd->getName()));
106  QString shortcut = info.cmd->getShortcut();
107  if (!shortcut.isEmpty())
108  title += QStringLiteral(" (%1)").arg(shortcut);
109  return title;
110  }
111  case Qt::ToolTipRole:
112  return Action::commandToolTip(info.cmd);
113 
114  case Qt::DecorationRole:
115  if (!info.iconChecked) {
116  info.iconChecked = true;
117  if(info.cmd->getPixmap())
118  info.icon = BitmapFactory().iconFromTheme(info.cmd->getPixmap());
119  }
120  return info.icon;
121 
122  case CommandNameRole:
123  return QByteArray(info.cmd->getName());
124 
125  default:
126  break;
127  }
128  return {};
129  }

166 bool CommandCompleter::eventFilter(QObject *o, QEvent *ev)
167 {
168  if (ev->type() == QEvent::KeyPress
169  && (o == this->widget() || o == this->popup()))
170  {
171  QKeyEvent * ke = static_cast<QKeyEvent*>(ev);
172  switch(ke->key()) {
173  case Qt::Key_Escape: {
174  auto edit = qobject_cast<QLineEdit*>(this->widget());
175  if (edit && edit->text().size()) {
176  edit->setText(QString());
177  popup()->hide();
178  return true;
179  } else if (popup()->isVisible()) {
180  popup()->hide();
181  return true;
182  }
183  break;
184  }
185  case Qt::Key_Tab: {
186  if (this->popup()->isVisible()) {
187  QKeyEvent kevent(ke->type(), Qt::Key_Down, Qt::NoModifier);
188  qApp->sendEvent(this->popup(), &kevent);
189  return true;
190  }
191  break;
192  }
193  case Qt::Key_Backtab: {
194  if (this->popup()->isVisible()) {
195  QKeyEvent kevent(ke->type(), Qt::Key_Up, Qt::NoModifier);
196  qApp->sendEvent(this->popup(), &kevent);
197  return true;
198  }
199  break;
200  }
201  case Qt::Key_Enter:
202  case Qt::Key_Return:
203  if (o == this->widget()) {
204  auto index = currentIndex();
205  if (index.isValid())
206  onCommandActivated(index);
207  else
208  complete();
209  ev->setAccepted(true);
210  return true;
211  }
212  default:
213  break;
214  }
215  }
216  return QCompleter::eventFilter(o, ev);
217 }

225 void CommandCompleter::onTextChanged(const QString &txt)
226 {
227  // Do not activate completer if less than 3 characters for better
228  // performance.
229  if (txt.size() < 3 || !widget())
230  return;
231 
232  static_cast<CommandModel*>(this->model())->update();
233 
234  this->setCompletionPrefix(txt);
235  QRect rect = widget()->rect();
236  if (rect.width() < 300)
237  rect.setWidth(300);
238  this->complete(rect);
239 }
