151 bool Color::fromHexString(const std::string& hex)
152 {
153  if (hex.size() < 7 || hex[0] != '#') {
154  return false;
155  }
156 
157  // #RRGGBB
158  if (hex.size() == 7) {
159  std::stringstream ss(hex);
160  unsigned int rgb;
161  char ch {};
162 
163  ss >> ch >> std::hex >> rgb;
164  int rc = (rgb >> 16) & 0xff;
165  int gc = (rgb >> 8) & 0xff;
166  int bc = rgb & 0xff;
167 
168  r = rc / 255.0F;
169  g = gc / 255.0F;
170  b = bc / 255.0F;
171 
172  return true;
173  }
174  // #RRGGBBAA
175  if (hex.size() == 9) {
176  std::stringstream ss(hex);
177  unsigned int rgba;
178  char ch {};
179 
180  ss >> ch >> std::hex >> rgba;
181  int rc = (rgba >> 24) & 0xff;
182  int gc = (rgba >> 16) & 0xff;
183  int bc = (rgba >> 8) & 0xff;
184  int ac = rgba & 0xff;
185 
186  r = rc / 255.0F;
187  g = gc / 255.0F;
188  b = bc / 255.0F;
189  a = ac / 255.0F;
190 
191  return true;
192  }
193 
194  return false;
