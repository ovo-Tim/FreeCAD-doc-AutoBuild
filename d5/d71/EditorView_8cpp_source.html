200 void EditorView::checkTimestamp()
201 {
202  QFileInfo fi(d->fileName);
203  qint64 timeStamp = fi.lastModified().toSecsSinceEpoch();
204  if (timeStamp != d->timeStamp) {
205  switch (QMessageBox::question(this,
206  tr("Modified file"),
207  tr("%1.\n\nThis has been modified outside of the source "
208  "editor. Do you want to reload it?")
209  .arg(d->fileName),
210  QMessageBox::Yes | QMessageBox::No,
211  QMessageBox::Yes)) {
212  case QMessageBox::Yes:
213  // updates time stamp and timer
214  open(d->fileName);
215  return;
216  case QMessageBox::No:
217  d->timeStamp = timeStamp;
218  break;
219  default:
220  break;
221  }
222  }
223 
224  d->activityTimer->setSingleShot(true);
225  d->activityTimer->start(3000);
226 }

231 bool EditorView::onMsg(const char* pMsg, const char** /*ppReturn*/)
232 {
233  // don't allow any actions if the editor is being closed
234  if (d->aboutToClose) {
235  return false;
236  }
237 
238  if (strcmp(pMsg, "Save") == 0) {
239  saveFile();
240  return true;
241  }
242  else if (strcmp(pMsg, "SaveAs") == 0) {
243  saveAs();
244  return true;
245  }
246  else if (strcmp(pMsg, "Cut") == 0) {
247  cut();
248  return true;
249  }
250  else if (strcmp(pMsg, "Copy") == 0) {
251  copy();
252  return true;
253  }
254  else if (strcmp(pMsg, "Paste") == 0) {
255  paste();
256  return true;
257  }
258  else if (strcmp(pMsg, "Undo") == 0) {
259  undo();
260  return true;
261  }
262  else if (strcmp(pMsg, "Redo") == 0) {
263  redo();
264  return true;
265  }
266  else if (strcmp(pMsg, "ViewFit") == 0) {
267  // just ignore this
268  return true;
269  }
270 
271  return false;
272 }

278 bool EditorView::onHasMsg(const char* pMsg) const
279 {
280  // don't allow any actions if the editor is being closed
281  if (d->aboutToClose) {
282  return false;
283  }
284  if (strcmp(pMsg, "Run") == 0) {
285  return true;
286  }
287  if (strcmp(pMsg, "DebugStart") == 0) {
288  return true;
289  }
290  if (strcmp(pMsg, "DebugStop") == 0) {
291  return true;
292  }
293  if (strcmp(pMsg, "SaveAs") == 0) {
294  return true;
295  }
296  if (strcmp(pMsg, "Print") == 0) {
297  return true;
298  }
299  if (strcmp(pMsg, "PrintPreview") == 0) {
300  return true;
301  }
302  if (strcmp(pMsg, "PrintPdf") == 0) {
303  return true;
304  }
305  if (strcmp(pMsg, "Save") == 0) {
306  return d->textEdit->document()->isModified();
307  }
308  else if (strcmp(pMsg, "Cut") == 0) {
309  bool canWrite = !d->textEdit->isReadOnly();
310  return (canWrite && (d->textEdit->textCursor().hasSelection()));
311  }
312  else if (strcmp(pMsg, "Copy") == 0) {
313  return (d->textEdit->textCursor().hasSelection());
314  }
315  else if (strcmp(pMsg, "Paste") == 0) {
316  QClipboard* cb = QApplication::clipboard();
317  QString text;
318 
319  // Copy text from the clipboard (paste)
320  text = cb->text();
321 
322  bool canWrite = !d->textEdit->isReadOnly();
323  return (!text.isEmpty() && canWrite);
324  }
325  else if (strcmp(pMsg, "Undo") == 0) {
326  return d->textEdit->document()->isUndoAvailable();
327  }
328  else if (strcmp(pMsg, "Redo") == 0) {
329  return d->textEdit->document()->isRedoAvailable();
330  }
331 
332  return false;
333 }

384 bool EditorView::open(const QString& fileName)
385 {
386  if (!QFile::exists(fileName)) {
387  return false;
388  }
389  QFile file(fileName);
390  if (!file.open(QFile::ReadOnly)) {
391  return false;
392  }
393 
394  d->lock = true;
395  d->textEdit->setPlainText(QString::fromUtf8(file.readAll()));
396  d->lock = false;
397  d->undos.clear();
398  d->redos.clear();
399  file.close();
400 
401  QFileInfo fi(fileName);
402  d->timeStamp = fi.lastModified().toSecsSinceEpoch();
403  d->activityTimer->setSingleShot(true);
404  d->activityTimer->start(3000);
405 
406  setCurrentFileName(fileName);
407  return true;
408 }

558 bool EditorView::saveFile()
559 {
560  if (d->fileName.isEmpty()) {
561  return saveAs();
562  }
563 
564  QFile file(d->fileName);
565  if (!file.open(QFile::WriteOnly)) {
566  return false;
567  }
568  QTextStream ts(&file);
569 #if QT_VERSION < 0x060000
570  ts.setCodec("UTF-8");
571 #endif
572  ts << d->textEdit->document()->toPlainText();
573  file.close();
574  d->textEdit->document()->setModified(false);
575 
576  QFileInfo fi(d->fileName);
577  d->timeStamp = fi.lastModified().toSecsSinceEpoch();
578  return true;
579 }

696 void PythonEditorView::executeScript()
697 {
698  // always save the macro when it is modified
699  if (EditorView::onHasMsg("Save")) {
700  EditorView::onMsg("Save", nullptr);
701  }
702  try {
703  getMainWindow()->setCursor(Qt::WaitCursor);
704  PythonTracingLocker tracelock(watcher->getTrace());
705  Application::Instance->macroManager()->run(Gui::MacroManager::File, fileName().toUtf8());
706  getMainWindow()->unsetCursor();
707  }
708  catch (const Base::SystemExitException&) {
709  // handle SystemExit exceptions
710  Base::PyGILStateLocker locker;
711  Base::PyException e;
712  e.ReportException();
713  getMainWindow()->unsetCursor();
714  }
715 }
