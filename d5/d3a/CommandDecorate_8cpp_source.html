84 void CmdTechDrawHatch::activated(int iMsg)
85 {
86  Q_UNUSED(iMsg);
87  if (!_checkSelectionHatch(this)) {
88  return;
89  }
90 
91  std::vector<Gui::SelectionObject> selection = getSelection().getSelectionEx();
92  auto partFeat( dynamic_cast<TechDraw::DrawViewPart *>(selection[0].getObject()) );
93  if (!partFeat) {
94  return;
95  }
96  const std::vector<std::string> &subNames = selection[0].getSubNames();
97  TechDraw::DrawPage* page = partFeat->findParentPage();
98  std::string PageName = page->getNameInDocument();
99  std::vector<int> selFaces;
100  for (auto& s: subNames) {
101  int f = TechDraw::DrawUtil::getIndexFromName(s);
102  selFaces.push_back(f);
103  }
104 
105  bool removeOld = false;
106  std::vector<TechDraw::DrawHatch*> hatchObjs = partFeat->getHatches();
107  for (auto& s: subNames) { //all the faces selected in DVP
108  int face = TechDraw::DrawUtil::getIndexFromName(s);
109  if (TechDraw::DrawHatch::faceIsHatched(face, hatchObjs)) {
110  QMessageBox::StandardButton rc =
111  QMessageBox::question(Gui::getMainWindow(), QObject::tr("Replace Hatch?"),
112  QObject::tr("Some Faces in selection are already hatched. Replace?"));
113  if (rc == QMessageBox::StandardButton::NoButton) {
114  return;
115  }
116 
117  removeOld = true;
118  break;
119  }
120  }
121 
122  if (removeOld) {
123  openCommand(QT_TRANSLATE_NOOP("Command", "Remove old Hatch"));
124  std::vector<std::pair< int, TechDraw::DrawHatch*> > toRemove;
125  for (auto& h: hatchObjs) { //all the hatch objects for selected DVP
126  std::vector<std::string> hatchSubs = h->Source.getSubValues();
127  for (auto& hs: hatchSubs) { //all the Faces in this hatch object
128  int hatchFace = TechDraw::DrawUtil::getIndexFromName(hs);
129  std::vector<int>::iterator it = std::find(selFaces.begin(), selFaces.end(), hatchFace);
130  if (it != selFaces.end()) {
131  std::pair< int, TechDraw::DrawHatch*> removeItem;
132  removeItem.first = hatchFace;
133  removeItem.second = h;
134  toRemove.push_back(removeItem);
135  }
136  }
137  }
138  for (auto& r: toRemove) {
139  r.second->removeSub(r.first);
140  if (r.second->empty()) {
141  doCommand(Doc, "App.activeDocument().removeObject('%s')", r.second->getNameInDocument());
142  }
143  }
144  commitCommand();
145  }
146 
147  // dialog to fill in hatch values
148  Gui::Control().showDialog(new TaskDlgHatch(partFeat, subNames));
149 
150  // Touch the parent feature so the hatching in tree view appears as a child
151  partFeat->touch();
152  getDocument()->recompute();
153 }

384 bool _checkSelectionHatch(Gui::Command* cmd) {
385  std::vector<Gui::SelectionObject> selection = cmd->getSelection().getSelectionEx();
386  if (selection.empty()) {
387  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Incorrect selection"),
388  QObject::tr("Select a Face first"));
389  return false;
390  }
391 
392  TechDraw::DrawViewPart * objFeat = dynamic_cast<TechDraw::DrawViewPart *>(selection[0].getObject());
393  if(!objFeat) {
394  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Incorrect selection"),
395  QObject::tr("No TechDraw object in selection"));
396  return false;
397  }
398 
399  std::vector<App::DocumentObject*> pages = cmd->getDocument()->getObjectsOfType(TechDraw::DrawPage::getClassTypeId());
400  if (pages.empty()){
401  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Incorrect selection"),
402  QObject::tr("Create a page to insert."));
403  return false;
404  }
405 
406  const std::vector<std::string> &SubNames = selection[0].getSubNames();
407  if (SubNames.empty()) {
408  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Incorrect Selection"),
409  QObject::tr("No Faces to hatch in this selection"));
410  return false;
411  }
412  std::string gType = TechDraw::DrawUtil::getGeomTypeFromName(SubNames.at(0));
413  if (!(gType == "Face")) {
414  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Incorrect Selection"),
415  QObject::tr("No Faces to hatch in this selection"));
416  return false;
417  }
418 
419  return true;
420 }
