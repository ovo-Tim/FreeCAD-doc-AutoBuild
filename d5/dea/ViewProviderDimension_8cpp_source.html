153 void ViewProviderDimension::updateData(const App::Property* prop)
154 {
155  if (prop == &(getViewObject()->Type)) {
156  setPixmapForType();
157  }
158 
159  //Dimension handles X, Y updates differently that other QGIView
160  //call QGIViewDimension::updateView
161  if (prop == &(getViewObject()->X) ||
162  prop == &(getViewObject()->Y) ||
163  prop == &(getViewObject()->FormatSpec) ||
164  prop == &(getViewObject()->Arbitrary) ||
165  prop == &(getViewObject()->FormatSpecOverTolerance) ||
166  prop == &(getViewObject()->FormatSpecUnderTolerance) ||
167  prop == &(getViewObject()->ArbitraryTolerances) ||
168  prop == &(getViewObject()->MeasureType) ||
169  prop == &(getViewObject()->TheoreticalExact) ||
170  prop == &(getViewObject()->EqualTolerance) ||
171  prop == &(getViewObject()->OverTolerance) ||
172  prop == &(getViewObject()->UnderTolerance) ||
173  prop == &(getViewObject()->Inverted) ) {
174 
175  QGIView* qgiv = getQView();
176  if (qgiv) {
177  qgiv->updateView(true);
178  }
179  return;
180  }
181 
182  //Skip QGIView X, Y processing - do not call ViewProviderDrawingView
183  Gui::ViewProviderDocumentObject::updateData(prop);
184 }

205 void ViewProviderDimension::onChanged(const App::Property* prop)
206 {
207  if ((prop == &Font) ||
208  (prop == &Fontsize) ||
209  (prop == &Arrowsize) ||
210  (prop == &LineWidth) ||
211  (prop == &StandardAndStyle) ||
212  (prop == &RenderingExtent) ||
213  (prop == &FlipArrowheads) ||
214  (prop == &GapFactorASME) ||
215  (prop == &GapFactorISO) ||
216  prop == &LineSpacingFactorISO) {
217  auto* qgiv = getQView();
218  if (qgiv) {
219  qgiv->updateView(true);
220  }
221  }
222  if (prop == &Color) {
223  auto* qgiv = getQView();
224  if (qgiv) {
225  auto* qgivd = dynamic_cast<QGIViewDimension*>(qgiv);
226  if (qgivd) {
227  qgivd->setNormalColorAll();
228  }
229  }
230  }
231 
232  ViewProviderDrawingView::onChanged(prop);
233 }

312 std::vector<App::DocumentObject*> ViewProviderDimension::claimChildren() const
313 {
314  // What can reasonably have a Dimension as a parent? A leader? a bit unconventional, but not forbidden.
315  // A RichAnno? Maybe? Balloons? This is a bit of a corner case. Typically, a
316  // Dimension would belong to something rather than owning something.
317  // Pages will appear in the inList, but should not be treated as a child of the dimension!
318 
319  std::vector<App::DocumentObject*> temp;
320  const std::vector<App::DocumentObject*>& candidates = getViewObject()->getInList();
321  for (auto& obj : candidates) {
322  if (obj->isDerivedFrom<TechDraw::DrawViewBalloon>() ||
323  obj->isDerivedFrom<TechDraw::DrawLeaderLine>() ||
324  obj->isDerivedFrom<TechDraw::DrawRichAnno>() ) {
325  temp.push_back(obj);
326  }
327  }
328  return temp;
329 }
