178  def execute(self,obj):
179 
180  pl = obj.Placement
181  rowsizes,columnsizes,spangroups = self.getSizes(obj)
182  #print rowsizes,columnsizes,spangroups
183  # create one face for each cell
184  faces = []
185  facenumber = 0
186  rowoffset = 0
187  for row in rowsizes:
188  columnoffset = 0
189  for column in columnsizes:
190  v1 = FreeCAD.Vector(columnoffset,rowoffset,0)
191  v2 = v1.add(FreeCAD.Vector(column,0,0))
192  v3 = v2.add(FreeCAD.Vector(0,-row,0))
193  v4 = v3.add(FreeCAD.Vector(-column,0,0))
194  f = Part.Face(Part.makePolygon([v1,v2,v3,v4,v1]))
195  if not facenumber in obj.HiddenFaces:
196  spanning = False
197  for i in range(len(spangroups)):
198  if facenumber in spangroups[i]:
199  g = spangroups[i]
200  g[g.index(facenumber)] = f
201  spangroups[i] = g
202  spanning = True
203  break
204  if not spanning:
205  faces.append(f)
206  facenumber += 1
207  columnoffset += column
208  rowoffset -= row
