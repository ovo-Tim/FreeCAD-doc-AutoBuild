212  TopoDS_Shape makeOffsetShape(bool allowOpenResult = false)
213  {
214  // in OCC the JointTypes are : Arc(0), Tangent(1), Intersection(2)
215  short joinType =
216  constructionMethod() == DrawSketchHandlerOffset::ConstructionMethod::Arc ? 0 : 2;
217 
218  // Offset will fail for single lines if we don't set a plane in ctor.
219  // But if we set a plane, then the direction of offset is forced...
220  // so we set a plane if and only if there are not a single sourceWires with more than single
221  // line.
222  BRepOffsetAPI_MakeOffset mkOffset;
223 
224  if (onlySingleLines) {
225  TopoDS_Face workingPlane = BRepBuilderAPI_MakeFace(gp_Pln(gp::Origin(), gp::DZ()));
226  mkOffset = BRepOffsetAPI_MakeOffset(workingPlane);
227  }
228  mkOffset.Init(GeomAbs_JoinType(joinType), allowOpenResult);
229 
230  for (TopoDS_Wire& wire : sourceWires) {
231  mkOffset.AddWire(wire);
232  }
233  try {
234 #if defined(__GNUC__) && defined(FC_OS_LINUX)
235  Base::SignalException se;
236 #endif
237  mkOffset.Perform(offsetLength);
238  }
239  catch (Standard_Failure&) {
240  throw;
241  }
242  catch (...) {
243  throw Base::CADKernelError(
244  "BRepOffsetAPI_MakeOffset has crashed! (Unknown exception caught)");
245  }
246 
247  TopoDS_Shape offsetShape = mkOffset.Shape();
248 
249  if (offsetShape.IsNull()) {
250  throw Base::CADKernelError("makeOffset2D: result of offsetting is null!");
251  }
252 
253  // Copying shape to fix strange orientation behavior, OCC7.0.0. See bug #2699
254  // http://www.freecadweb.org/tracker/view.php?id=2699
255  offsetShape = BRepBuilderAPI_Copy(offsetShape).Shape();
256  return offsetShape;
257  }

259  Part::Geometry* curveToLine(BRepAdaptor_Curve curve)
260  {
261  double first = curve.FirstParameter();
262  if (fabs(first) > 1E99) {
263  first = -10000;
264  }
265 
266  double last = curve.LastParameter();
267  if (fabs(last) > 1E99) {
268  last = +10000;
269  }
270 
271  gp_Pnt P1 = curve.Value(first);
272  gp_Pnt P2 = curve.Value(last);
273 
274  Base::Vector3d p1(P1.X(), P1.Y(), P1.Z());
275  Base::Vector3d p2(P2.X(), P2.Y(), P2.Z());
276  auto* line = new Part::GeomLineSegment();
277  line->setPoints(p1, p2);
278  GeometryFacade::setConstruction(line, false);
279  return line;
280  }
281  Part::Geometry* curveToCircleOrArc(BRepAdaptor_Curve curve)
282  {
283  gp_Circ circle = curve.Circle();
284  gp_Pnt cnt = circle.Location();
285  gp_Pnt beg = curve.Value(curve.FirstParameter());
286  gp_Pnt end = curve.Value(curve.LastParameter());
287 
288  if (beg.SquareDistance(end) < Precision::Confusion()) {
289  auto* gCircle = new Part::GeomCircle();
290  gCircle->setRadius(circle.Radius());
291  gCircle->setCenter(Base::Vector3d(cnt.X(), cnt.Y(), cnt.Z()));
292 
293  GeometryFacade::setConstruction(gCircle, false);
294  return gCircle;
295  }
296  else {
297  auto* gArc = new Part::GeomArcOfCircle();
298  Handle(Geom_Curve) hCircle = new Geom_Circle(circle);
299  Handle(Geom_TrimmedCurve) tCurve =
300  new Geom_TrimmedCurve(hCircle, curve.FirstParameter(), curve.LastParameter());
301  gArc->setHandle(tCurve);
302  GeometryFacade::setConstruction(gArc, false);
303  return gArc;
304  }
305  }
306  Part::Geometry* curveToEllipseOrArc(BRepAdaptor_Curve curve)
307  {
308  gp_Elips ellipse = curve.Ellipse();
309  // gp_Pnt cnt = ellipse.Location();
310  gp_Pnt beg = curve.Value(curve.FirstParameter());
311  gp_Pnt end = curve.Value(curve.LastParameter());
312 
313  if (beg.SquareDistance(end) < Precision::Confusion()) {
314  auto* gEllipse = new Part::GeomEllipse();

316  gEllipse->setHandle(hEllipse);

318  return gEllipse;

321  Handle(Geom_Curve) hEllipse = new Geom_Ellipse(ellipse);
322  Handle(Geom_TrimmedCurve) tCurve =

331  void getOffsetGeos(std::vector<Part::Geometry*>& geometriesToAdd,
332  std::vector<int>& listOfOffsetGeoIds)
333  {
334  TopoDS_Shape offsetShape = makeOffsetShape();
335 
336  TopExp_Explorer expl(offsetShape, TopAbs_EDGE);
337  int geoIdToAdd = firstCurveCreated;
338  for (; expl.More(); expl.Next(), geoIdToAdd++) {
339 
340  const TopoDS_Edge& edge = TopoDS::Edge(expl.Current());
341  BRepAdaptor_Curve curve(edge);
342  if (curve.GetType() == GeomAbs_Line) {
343  geometriesToAdd.push_back(curveToLine(curve));
344  listOfOffsetGeoIds.push_back(geoIdToAdd);
345  }
346  else if (curve.GetType() == GeomAbs_Circle) {
347  geometriesToAdd.push_back(curveToCircleOrArc(curve));
348  listOfOffsetGeoIds.push_back(geoIdToAdd);
349  }
350  else if (curve.GetType() == GeomAbs_Ellipse) {
351  geometriesToAdd.push_back(curveToEllipseOrArc(curve));
352  listOfOffsetGeoIds.push_back(geoIdToAdd);
353  }
354  // TODO Bspline support
355  }

392  void jointOffsetCurves(std::vector<int>& listOfOffsetGeoIds)
393  {
394  std::stringstream stream;
395  stream << "conList = []\n";
396  for (size_t i = 0; i < listOfOffsetGeoIds.size() - 1; i++) {
397  for (size_t j = i + 1; j < listOfOffsetGeoIds.size(); j++) {
398  // There's a bug with created arcs. They end points seems to swap at some point...
399  // Here we make coincidences based on distance. So they must change after.
400  Base::Vector3d firstStartPoint, firstEndPoint, secondStartPoint, secondEndPoint;
401  if (!getFirstSecondPoints(listOfOffsetGeoIds[i], firstStartPoint, firstEndPoint)
402  || !getFirstSecondPoints(listOfOffsetGeoIds[j],
403  secondStartPoint,
404  secondEndPoint)) {
405  continue;
406  }
407 
408  bool create = false;
409  PointPos posi, posj;
410 
411  if ((firstStartPoint - secondStartPoint).Length() < Precision::Confusion()) {
412  create = true;
413  posi = PointPos::start;
414  posj = PointPos::start;
415  }
416  else if ((firstStartPoint - secondEndPoint).Length() < Precision::Confusion()) {
417  create = true;
418  posi = PointPos::start;
419  posj = PointPos::end;
420  }
421  else if ((firstEndPoint - secondStartPoint).Length() < Precision::Confusion()) {
422  create = true;
423  posi = PointPos::end;
424  posj = PointPos::start;
425  }
426  else if ((firstEndPoint - secondEndPoint).Length() < Precision::Confusion()) {
427  create = true;
428  posi = PointPos::end;
429  posj = PointPos::end;
430  }
431 
432  if (create) {
433  bool tangent =
434  needTangent(listOfOffsetGeoIds[i], listOfOffsetGeoIds[j], posi, posj);
435  stream << "conList.append(Sketcher.Constraint('"
436  << (tangent ? "Tangent" : "Coincident");
437  stream << "'," << listOfOffsetGeoIds[i] << "," << static_cast<int>(posi) << ", "
438  << listOfOffsetGeoIds[j] << "," << static_cast<int>(posj) << "))\n";
439  }
440  }
441  }
442 
443  stream << Gui::Command::getObjectCmd(sketchgui->getObject()) << ".addConstraint(conList)\n";
444  stream << "del conList\n";
445  Gui::Command::doCommand(Gui::Command::Doc, stream.str().c_str());
446  }

448  bool needTangent(int geoId1, int geoId2, PointPos pos1, PointPos pos2)
449  {
450  // Todo: add cases for arcOfellipse parabolas hyperbolas bspline
451 
452  SketchObject* Obj = sketchgui->getSketchObject();
453  const Part::Geometry* geo1 = Obj->getGeometry(geoId1);
454  const Part::Geometry* geo2 = Obj->getGeometry(geoId2);
455 
456  if (!isArcOfCircle(*geo1) && !isArcOfCircle(*geo2)) {
457  return false;
458  }
459 
460  Base::Vector3d perpendicular1, perpendicular2, p1, p2;
461  if (isArcOfCircle(*geo1)) {
462  auto* arcOfCircle = static_cast<const Part::GeomArcOfCircle*>(geo1);
463  p1 = pos1 == PointPos::start ? arcOfCircle->getStartPoint(true)
464  : arcOfCircle->getEndPoint(true);
465 
466  perpendicular1.x = -(arcOfCircle->getCenter() - p1).y;
467  perpendicular1.y = (arcOfCircle->getCenter() - p1).x;
468  }
469  else if (isLineSegment(*geo1)) {
470  auto* line = static_cast<const Part::GeomLineSegment*>(geo1);
471  perpendicular1 = line->getStartPoint() - line->getEndPoint();
472  }
473  else {
474  return false;
475  }
476 
477  if (isArcOfCircle(*geo2)) {
478  auto* arcOfCircle = static_cast<const Part::GeomArcOfCircle*>(geo2);
479  p2 = pos2 == PointPos::start ? arcOfCircle->getStartPoint(true)
480  : arcOfCircle->getEndPoint(true);
481 
482  perpendicular2.x = -(arcOfCircle->getCenter() - p2).y;
483  perpendicular2.y = (arcOfCircle->getCenter() - p2).x;
484  }
485  else if (isLineSegment(*geo2)) {
486  auto* line = static_cast<const Part::GeomLineSegment*>(geo2);
487  perpendicular2 = line->getStartPoint() - line->getEndPoint();
488  }
489  else {
490  return false;
491  }
492 
493  // if lines are parallel
494  if ((perpendicular1 % perpendicular2).Length() < Precision::Confusion()) {
495  return true;
496  }
497 
498  return false;
499  }

518  void makeOffsetConstraint(std::vector<int>& listOfOffsetGeoIds)
519  {
520  SketchObject* Obj = sketchgui->getSketchObject();
521 
522  std::stringstream stream;
523  stream << "conList = []\n";
524  // We separate the constraints of new lines in case the construction lines are not needed.
525  std::stringstream newLinesStream;
526  newLinesStream << "conList2 = []\n";
527 
528  vCCO = generatevCC(listOfOffsetGeoIds);
529 
530  int geoIdCandidate1 {};
531  int geoIdCandidate2 {};
532 
533  int newCurveCounter = 0;
534  int prevCurveCounter = 0;
535  std::vector<Part::Geometry*> geometriesToAdd;
536  for (auto& curve : vCCO) {
537  // Check if curve is closed. Note as we use pipe it should always be closed but in case
538  // we enable 'Skin' in the future.
539  bool closed = isCurveClosed(curve);
540  bool atLeastOneLine = false;
541  bool rerunFirstAfterThis = false;
542  bool rerunningFirst = false;
543  bool inTangentGroup = false;
544 
545  for (size_t j = 0; j < curve.size(); j++) {
546 
547  // Tangent constraint is constraining the offset already. So if there are tangents
548  // we should not create the construction lines. Hence the code below.
549  bool createLine = true;
550  bool forceCreate = false;
551  if (!inTangentGroup && (!closed || j != 0 || rerunningFirst)) {
552  createLine = true;
553  atLeastOneLine = true;
554  }
555  else {
556  // include case of j == 0 and closed curve, because if required the line
557  // will be made after last.
558  createLine = false;
559  }
560 
561  if (j + 1 < curve.size()) {
562  inTangentGroup = areTangentCoincident(curve[j], curve[j + 1]);
563  }
564  else if (j == curve.size() - 1 && closed) {
565  // Case of last geoId for closed curves.
566  inTangentGroup = areTangentCoincident(curve[j], curve[0]);
567  if (inTangentGroup) {
568  if (!atLeastOneLine) { // We need at least one line
569  createLine = true;
570  forceCreate = true;
571  }
572  }
573  else {
574  // We rerun the for at j=0 after this run to create line for j = 0.
575  rerunFirstAfterThis = true;
576  }
577  }
578 
579  const Part::Geometry* geo = Obj->getGeometry(curve[j]);
580  for (auto geoId : listOfGeoIds) {
581  // Check if geoId is the offsetted curve giving curve[j].
582  const Part::Geometry* geo2 = Obj->getGeometry(geoId);
583 
584  if (isCircle(*geo) && isCircle(*geo2)) {
585  auto* circle = static_cast<const Part::GeomCircle*>(geo);
586  auto* circle2 = static_cast<const Part::GeomCircle*>(geo2);
587  Base::Vector3d p1 = circle->getCenter();
588  Base::Vector3d p2 = circle2->getCenter();
589  if ((p1 - p2).Length() < Precision::Confusion()) {
590  // coincidence of center
591  stream << "conList.append(Sketcher.Constraint('Coincident'," << curve[j]
592  << ",3, " << geoId << ",3))\n";
593 
594  // Create line between both circles.
595  auto* line = new Part::GeomLineSegment();
596  p1.x = p1.x + circle->getRadius();
597  p2.x = p2.x + circle2->getRadius();
598  line->setPoints(p1, p2);
599  GeometryFacade::setConstruction(line, true);
600  geometriesToAdd.push_back(line);
601  newCurveCounter++;
602  newLinesStream << "conList2.append(Sketcher.Constraint('Perpendicular',"
603  << getHighestCurveIndex() + newCurveCounter << ", "
604  << curve[j] << "))\n";
605  newLinesStream << "conList2.append(Sketcher.Constraint('PointOnObject',"
606  << getHighestCurveIndex() + newCurveCounter << ",1, "
607  << curve[j] << "))\n";
608  newLinesStream << "conList2.append(Sketcher.Constraint('PointOnObject',"
609  << getHighestCurveIndex() + newCurveCounter << ",2, "
610  << geoId << "))\n";
611 
612  geoIdCandidate1 = curve[j];
613  geoIdCandidate2 = geoId;
614 
615  break;
616  }
617  }
618  else if (isEllipse(*geo) && isEllipse(*geo2)) {
619  // same as circle but 2 lines
620  }
621  else if (isLineSegment(*geo) && isLineSegment(*geo2)) {
622  auto* lineSeg1 = static_cast<const Part::GeomLineSegment*>(geo);
623  auto* lineSeg2 = static_cast<const Part::GeomLineSegment*>(geo2);
624  Base::Vector3d p1[2], p2[2];
625  p1[0] = lineSeg1->getStartPoint();
626  p1[1] = lineSeg1->getEndPoint();
627  p2[0] = lineSeg2->getStartPoint();
628  p2[1] = lineSeg2->getEndPoint();
629  // if lines are parallel
630  if (((p1[1] - p1[0]) % (p2[1] - p2[0])).Length()
631  < Precision::Intersection()) {
632  // If the lines are space by offsetLength distance
633  Base::Vector3d projectedP;
634  projectedP.ProjectToLine(p1[0] - p2[0], p2[1] - p2[0]);
635 
636  if ((projectedP).Length() - fabs(offsetLength)
637  < Precision::Confusion()) {
638  if (!forceCreate && !rerunningFirst) {
639  stream << "conList.append(Sketcher.Constraint('Parallel',"
640  << curve[j] << ", " << geoId << "))\n";
641  }
642 
643  // We don't need a construction line if the line has a tangent at
644  // one end. Unless it's the first line that we're making.
645  if (createLine) {
646  auto* line = new Part::GeomLineSegment();
647  line->setPoints(p1[0], p1[0] + projectedP);
648  GeometryFacade::setConstruction(line, true);
649  geometriesToAdd.push_back(line);
650  newCurveCounter++;

838  std::vector<std::vector<int>> generatevCC(std::vector<int>& listOfGeo)
839  {
840  // This function separates all the selected geometries into separate continuous curves.
841  SketchObject* Obj = sketchgui->getSketchObject();
842  std::vector<std::vector<int>> vcc;
843 
844  for (auto geoId : listOfGeo) {
845  std::vector<int> vecOfGeoIds;
846  const Part::Geometry* geo = Obj->getGeometry(geoId);
847  if (isCircle(*geo) || isEllipse(*geo)) {
848  vecOfGeoIds.push_back(geoId);
849  vcc.push_back(vecOfGeoIds);
850  continue;
851  }
852 
853  bool inserted = false;
854  int insertedIn = -1;
855  for (size_t j = 0; j < vcc.size(); j++) {
856  for (size_t k = 0; k < vcc[j].size(); k++) {
857  if (!areCoincident(geoId, vcc[j][k])) {
858  continue;
859  }
860 
861  if (inserted && insertedIn != int(j)) {
862  // if it's already inserted in another continuous curve then we need
863  // to merge both curves together. There're 2 cases, it could have
864  // been inserted at the end or at the beginning.
865  if (vcc[insertedIn][0] == geoId) {
866  // Two cases. Either the coincident is at the beginning or at
867  // the end.
868  if (k == 0) {
869  std::reverse(vcc[j].begin(), vcc[j].end());
870  }
871  vcc[j].insert(vcc[j].end(),
872  vcc[insertedIn].begin(),
873  vcc[insertedIn].end());
874  vcc.erase(vcc.begin() + insertedIn);
875  }
876  else {
877  if (k != 0) { // ie k is vcc[j].size()-1
878  std::reverse(vcc[j].begin(), vcc[j].end());
879  }
880  vcc[insertedIn].insert(vcc[insertedIn].end(),
881  vcc[j].begin(),
882  vcc[j].end());
883  vcc.erase(vcc.begin() + j);
884  }
885  j--;
886  }
887  else {
888  // we need to get the curves in the correct order.
889  if (k == vcc[j].size() - 1) {
890  vcc[j].push_back(geoId);
891  }
892  else {
893  // in this case k should actually be 0.
894  vcc[j].insert(vcc[j].begin() + k, geoId);
895  }
896  insertedIn = j;
897  inserted = true;
898  }
899  // printCCeVec();
900  break;
901  }
902  }
903  if (!inserted) {
904  vecOfGeoIds.push_back(geoId);
905  vcc.push_back(vecOfGeoIds);
906  }
907  }
908  return vcc;

995  CoincidencePointPos checkForCoincidence(int geoId1, int geoId2, bool tangentOnly = false)
996  {
997  // This function looks up for 2 coincidence between 2 edges (arc + line can have 2)
998  SketchObject* Obj = sketchgui->getSketchObject();
999  const std::vector<Constraint*>& vals = Obj->Constraints.getValues();
1000  CoincidencePointPos positions;
1001  positions.firstPos1 = PointPos::none;
1002  positions.secondPos1 = PointPos::none;
1003  positions.firstPos2 = PointPos::none;
1004  positions.secondPos2 = PointPos::none;
1005  bool firstCoincidenceFound = false;
1006  for (auto* cstr : vals) {
1007  if (((tangentOnly || cstr->Type != Coincident) && cstr->Type != Tangent)
1008  || cstr->FirstPos == PointPos::mid || cstr->FirstPos == PointPos::none
1009  || cstr->SecondPos == PointPos::mid || cstr->SecondPos == PointPos::none) {
1010  continue;
1011  }
1012 
1013  if ((cstr->First == geoId1 && cstr->Second == geoId2)
1014  || (cstr->First == geoId2 && cstr->Second == geoId1)) {
1015  if (!firstCoincidenceFound) {
1016  positions.firstPos1 = cstr->First == geoId1 ? cstr->FirstPos : cstr->SecondPos;
1017  positions.secondPos1 = cstr->First == geoId2 ? cstr->FirstPos : cstr->SecondPos;
1018  firstCoincidenceFound = true;
1019  }
1020  else {
1021  positions.firstPos2 = cstr->First == geoId1 ? cstr->FirstPos : cstr->SecondPos;
1022  positions.secondPos2 = cstr->First == geoId2 ? cstr->FirstPos : cstr->SecondPos;
1023  break;
1024  }
1025  }
1026  }
1027  return positions;
