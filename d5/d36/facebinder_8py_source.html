136  def _build_shape(self, obj, faces, sew=False):
137  """returns the built shape and the area of the offset faces"""
138  import Part
139  offs_val = getattr(obj, "Offset", 0)
140  extr_val = getattr(obj, "Extrusion", 0)
141 
142  shp = Part.Compound(faces)
143  # Sew before offsetting to ensure corners stay connected:
144  if sew:
145  shp.sewShape()
146  if shp.ShapeType != "Compound":
147  shp = Part.Compound([shp])
148 
149  if offs_val:
150  offsets = []
151  for sub in shp.SubShapes:
152  offsets.append(sub.makeOffsetShape(offs_val, 1e-7, join=2))
153  shp = Part.Compound(offsets)
154 
155  area = shp.Area # take area after offsetting original faces, but before extruding
156 
157  if extr_val:
158  extrudes = []
159  for sub in shp.SubShapes:
160  ext = sub.makeOffsetShape(extr_val, 1e-7, inter=True, join=2, fill=True)
161  extrudes.append(self._convert_to_planar(obj, ext))
162  shp = Part.Compound(extrudes)
163 
164  subs = shp.SubShapes
165  shp = subs.pop()
166  if subs:
167  shp = shp.fuse(subs)
168 
169  if len(shp.Faces) > 1:
170  if getattr(obj, "RemoveSplitter", True):
171  shp = shp.removeSplitter()
172 
173  return shp, area

175  def _convert_to_planar(self, obj, shp):
176  """convert flat B-spline faces to planar faces if possible"""
177  import Part
178  faces = []
179  for face in shp.Faces:
180  if face.Surface.TypeId == "Part::GeomPlane":
181  faces.append(face)
182  elif not geometry.is_planar(face):
183  faces.append(face)
184  else:
185  edges = []
186  for edge in face.Edges:
187  if edge.Curve.TypeId == "Part::GeomLine" or geometry.is_straight_line(edge):
188  verts = edge.Vertexes
189  edges.append(Part.makeLine(verts[0].Point, verts[1].Point))
190  else:
191  edges.append(edge)
192  wires = [Part.Wire(x) for x in Part.sortEdges(edges)]
193  face = Part.makeFace(wires, "Part::FaceMakerCheese")
194  face.fix(1e-7, 0, 1)
195  faces.append(face)
196  solid = Part.makeSolid(Part.makeShell(faces))
197  if solid.isValid():
198  return solid
199  _msg(obj.Label + ": " + translate("draft",
200  "Converting flat B-spline faces of Facebinder to planar faces failed"
201  ))
202  return shp
