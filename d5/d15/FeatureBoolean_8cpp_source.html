72 App::DocumentObjectExecReturn *Boolean::execute()
73 {
74  // Get the operation type
75  std::string type = Type.getValueAsString();
76 
77  // Check the parameters
78  const Part::Feature* baseFeature = this->getBaseObject(/* silent = */ true);
79 
80  if (!baseFeature && type == "Cut") {
81  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Cannot do boolean cut without BaseFeature"));
82  }
83 
84  std::vector<App::DocumentObject*> tools = Group.getValues();
85  if (tools.empty())
86  return App::DocumentObject::StdReturn;
87 
88  // Get the base shape to operate on
89  Part::TopoShape baseTopShape;
90  if(baseFeature)
91  baseTopShape = baseFeature->Shape.getShape();
92  else {
93  auto feature = tools.back();
94  if(!feature->isDerivedFrom(Part::Feature::getClassTypeId()))
95  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Cannot do boolean with anything but Part::Feature and its derivatives"));
96 
97  baseTopShape = static_cast<Part::Feature*>(feature)->Shape.getShape();
98  tools.pop_back();
99  }
100 
101  if (baseTopShape.getShape().IsNull())
102  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Cannot do boolean operation with invalid base shape"));
103 
104  //get the body this boolean feature belongs to
105  Part::BodyBase* baseBody = Part::BodyBase::findBodyOf(this);
106 
107  if(!baseBody)
108  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Cannot do boolean on feature which is not in a body"));
109 
110  std::vector<TopoShape> shapes;
111  shapes.push_back(baseTopShape);
112  for(auto it=tools.begin(); it<tools.end(); ++it) {
113  auto shape = getTopoShape(*it);
114  if (shape.isNull())
115  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception","Tool shape is null"));
116  shapes.push_back(shape);
117  }
118  TopoShape result(baseTopShape);
119 
120  Base::Placement bodyPlacement = baseBody->globalPlacement().inverse();
121  for (auto tool : tools)
122  {
123  if(!tool->isDerivedFrom(Part::Feature::getClassTypeId()))
124  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Cannot do boolean with anything but Part::Feature and its derivatives"));
125 
126  Part::TopoShape toolShape = static_cast<Part::Feature*>(tool)->Shape.getShape();
127  if ( UsePlacement.getValue() )
128  toolShape.setPlacement(bodyPlacement * toolShape.getPlacement());
129  TopoDS_Shape shape = toolShape.getShape();
130  TopoDS_Shape boolOp;
131 
132  // Must not pass null shapes to the boolean operations
133  if (result.isNull())
134  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Base shape is null"));
135 
136  if (shape.IsNull())
137  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Tool shape is null"));
138 
139  const char *op = nullptr;
140  if (type == "Fuse")
141  op = Part::OpCodes::Fuse;
142  else if(type == "Cut")
143  op = Part::OpCodes::Cut;
144  else if(type == "Common")
145  op = Part::OpCodes::Common;
146  // LinkStage3 defines these other types of Boolean operations. Removed for now pending
147  // decision to bring them in or not.
148  // else if(type == "Compound")
149  // op = Part::OpCodes::Compound;
150  // else if(type == "Section")
151  // op = Part::OpCodes::Section;
152  else
153  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Unsupported boolean operation"));
154 
155  try {
156  result.makeElementBoolean(op, shapes);
157  } catch (Standard_Failure &e) {
158  FC_ERR("Boolean operation failed: " << e.GetMessageString());
159  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Boolean operation failed"));
160  }
161  }
162 
163  result = refineShapeIfActive(result);
164 
165  if (!isSingleSolidRuleSatisfied(result.getShape())) {
166  return new App::DocumentObjectExecReturn(QT_TRANSLATE_NOOP("Exception", "Result has multiple solids: that is not currently supported."));
167  }
168 
169  this->Shape.setValue(getSolid(result));
170  return App::DocumentObject::StdReturn;
171 }
