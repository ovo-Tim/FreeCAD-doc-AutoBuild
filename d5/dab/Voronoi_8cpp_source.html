210 void Voronoi::colorExterior(const Voronoi::diagram_type::edge_type* edge, std::size_t colorValue)
211 {
212  if (edge->color()) {
213  // end recursion
214  return;
215  }
216  edge->color(colorValue);
217  edge->twin()->color(colorValue);
218  auto v = edge->vertex1();
219  if (!v || !edge->is_primary()) {
220  return;
221  }
222  v->color(colorValue);
223  auto e = v->incident_edge();
224  do {
225  colorExterior(e, colorValue);
226  e = e->rot_next();
227  } while (e != v->incident_edge());
228 }

253 double Voronoi::diagram_type::angleOfSegment(int i, Voronoi::diagram_type::angle_map_t* angle) const
254 {
255  Voronoi::diagram_type::angle_map_t::const_iterator a =
256  angle ? angle->find(i) : Voronoi::diagram_type::angle_map_t::const_iterator();
257  if (!angle || a == angle->end()) {
258  Voronoi::point_type p0 = low(segments[i]);
259  Voronoi::point_type p1 = high(segments[i]);
260  double ang = 0;
261  if (p0.x() == p1.x()) {
262  if (p0.y() < p1.y()) {
263  ang = M_PI_2;
264  }
265  else {
266  ang = -M_PI_2;
267  }
268  }
269  else {
270  ang = atan((p0.y() - p1.y()) / (p0.x() - p1.x()));
271  }
272  if (angle) {
273  angle->insert(angle_map_t::value_type(i, ang));
274  }
275  return ang;
276  }
277  return a->second;
278 }

293 void Voronoi::colorColinear(Voronoi::color_type color, double degree)
294 {
295  double rad = degree * M_PI / 180;
296 
297  Voronoi::diagram_type::angle_map_t angle;
298  int psize = vd->points.size();
299 
300  for (diagram_type::const_edge_iterator it = vd->edges().begin(); it != vd->edges().end();
301  ++it) {
302  int i0 = it->cell()->source_index() - psize;
303  int i1 = it->twin()->cell()->source_index() - psize;
304  if (it->color() == 0 && it->cell()->contains_segment()
305  && it->twin()->cell()->contains_segment() && vd->segmentsAreConnected(i0, i1)) {
306  double a0 = vd->angleOfSegment(i0, &angle);
307  double a1 = vd->angleOfSegment(i1, &angle);
308  double a = a0 - a1;
309  if (a > M_PI_2) {
310  a -= M_PI;
311  }
312  else if (a < -M_PI_2) {
313  a += M_PI;
314  }
315  if (fabs(a) < rad) {
316  it->color(color);
317  it->twin()->color(color);
318  }
319  }
320  }
321 }
