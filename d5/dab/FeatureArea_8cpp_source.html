75 App::DocumentObjectExecReturn* FeatureArea::execute()
76 {
77  myInited = true;
78 
79  std::vector<App::DocumentObject*> links = Sources.getValues();
80  if (links.empty()) {
81  return new App::DocumentObjectExecReturn("No shapes linked");
82  }
83 
84  for (std::vector<App::DocumentObject*>::iterator it = links.begin(); it != links.end(); ++it) {
85  if (!(*it && (*it)->isDerivedFrom(Part::Feature::getClassTypeId()))) {
86  return new App::DocumentObjectExecReturn(
87  "Linked object is not a Part object (has no Shape).");
88  }
89  TopoDS_Shape shape = static_cast<Part::Feature*>(*it)->Shape.getShape().getShape();
90  if (shape.IsNull()) {
91  return new App::DocumentObjectExecReturn("Linked shape object is empty");
92  }
93  }
94 
95  FC_TIME_INIT(t);
96 
97  AreaParams params;
98 
99 #define AREA_PROP_GET(_param) params.PARAM_FNAME(_param) = PARAM_FNAME(_param).getValue();
100  PARAM_FOREACH(AREA_PROP_GET, AREA_PARAMS_CONF)
101 
102  myArea.clean(true);
103  myArea.setParams(params);
104 
105  TopoDS_Shape workPlane = WorkPlane.getShape().getShape();
106  myArea.setPlane(workPlane);
107 
108  for (std::vector<App::DocumentObject*>::iterator it = links.begin(); it != links.end(); ++it) {
109  myArea.add(static_cast<Part::Feature*>(*it)->Shape.getShape().getShape(),
110  PARAM_PROP_ARGS(AREA_PARAMS_OPCODE));
111  }
112 
113  myShapes.clear();
114  if (myArea.getSectionCount() == 0) {
115  myShapes.push_back(myArea.getShape(-1));
116  }
117  else {
118  myShapes.reserve(myArea.getSectionCount());
119  for (int i = 0; i < (int)myArea.getSectionCount(); ++i) {
120  myShapes.push_back(myArea.getShape(i));
121  }
122  }
123 
124  bool hasShape = false;
125  if (myShapes.empty()) {
126  Shape.setValue(TopoDS_Shape());
127  }
128  else {
129  // compound is built even if there is only one shape to save the
130  // trouble of messing around with placement
131  BRep_Builder builder;
132  TopoDS_Compound compound;
133  builder.MakeCompound(compound);
134  for (auto& shape : myShapes) {
135  if (shape.IsNull()) {
136  continue;
137  }
138  hasShape = true;
139  builder.Add(compound, shape);
140  }
141  Shape.setValue(compound);
142  }
143 
144  FC_TIME_LOG(t, "feature execute");
145 
146  if (!hasShape) {
147  return new App::DocumentObjectExecReturn("no output shape");
148  }
149 
150  return DocumentObject::StdReturn;
151 }

198 std::list<TopoDS_Shape> FeatureAreaView::getShapes()
199 {
200  std::list<TopoDS_Shape> shapes;
201  App::DocumentObject* pObj = Source.getValue();
202  if (!pObj) {
203  return shapes;
204  }
205  if (!pObj->isDerivedFrom(FeatureArea::getClassTypeId())) {
206  return shapes;
207  }
208 
209  auto all_shapes = static_cast<FeatureArea*>(pObj)->getShapes();
210 
211  if (all_shapes.empty()) {
212  return shapes;
213  }
214 
215  int index = SectionIndex.getValue(), count = SectionCount.getValue();
216  if (index < 0) {
217  index += ((int)all_shapes.size());
218  if (index < 0) {
219  return shapes;
220  }
221  if (count <= 0 || index + 1 - count < 0) {
222  count = index + 1;
223  index = 0;
224  }
225  else {
226  index -= count - 1;
227  }
228  }
229  else if (index >= (int)all_shapes.size()) {
230  return shapes;
231  }
232 
233  if (count <= 0) {
234  count = all_shapes.size();
235  }
236  count += index;
237  if (count > (int)all_shapes.size()) {
238  count = all_shapes.size();
239  }
240  for (int i = index; i < count; ++i) {
241  shapes.push_back(all_shapes[i]);
242  }
243  return shapes;
244 }

246 App::DocumentObjectExecReturn* FeatureAreaView::execute()
247 {
248  App::DocumentObject* pObj = Source.getValue();
249  if (!pObj) {
250  return new App::DocumentObjectExecReturn("No shape linked");
251  }
252 
253  if (!pObj->isDerivedFrom(FeatureArea::getClassTypeId())) {
254  return new App::DocumentObjectExecReturn("Linked object is not a FeatureArea");
255  }
256 
257  bool hasShape = false;
258  std::list<TopoDS_Shape> shapes = getShapes();
259  if (shapes.empty()) {
260  Shape.setValue(TopoDS_Shape());
261  }
262  else {
263  BRep_Builder builder;
264  TopoDS_Compound compound;
265  builder.MakeCompound(compound);
266  for (auto& shape : shapes) {
267  if (shape.IsNull()) {
268  continue;
269  }
270  hasShape = true;
271  builder.Add(compound, shape);
272  }
273  Shape.setValue(compound);
274  }
275 
276  if (!hasShape) {
277  return new App::DocumentObjectExecReturn("no output shape");
278  }
279 
280  return DocumentObject::StdReturn;
281 }
