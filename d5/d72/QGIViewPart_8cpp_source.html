260 void QGIViewPart::drawAllFaces(void)
261 {
262  // dvp already validated
263  auto dvp(static_cast<TechDraw::DrawViewPart*>(getViewObject()));
264 
265  QColor faceColor;
266  auto vpp = dynamic_cast<ViewProviderViewPart *>(getViewProvider(getViewObject()));
267  if (vpp) {
268  faceColor = vpp->FaceColor.getValue().asValue<QColor>();
269  faceColor.setAlpha((100 - vpp->FaceTransparency.getValue())*255/100);
270  }
271 
272  std::vector<TechDraw::DrawHatch*> regularHatches = dvp->getHatches();
273  std::vector<TechDraw::DrawGeomHatch*> geomHatches = dvp->getGeomHatches();
274  const std::vector<TechDraw::FacePtr>& faceGeoms = dvp->getFaceGeometry();
275  int iFace(0);
276  for (auto& face : faceGeoms) {
277  QGIFace* newFace = drawFace(face, iFace);
278  if (faceColor.isValid()) {
279  newFace->setFillColor(faceColor);
280  newFace->setFillMode(faceColor.alpha() ? QGIFace::PlainFill : QGIFace::NoFill);
281  }
282 
283  TechDraw::DrawHatch* fHatch = faceIsHatched(iFace, regularHatches);
284  TechDraw::DrawGeomHatch* fGeom = faceIsGeomHatched(iFace, geomHatches);
285  if (fGeom) {
286  // geometric hatch (from PAT hatch specification)
287  newFace->isHatched(true);
288  newFace->setFillMode(QGIFace::GeomHatchFill);
289  std::vector<LineSet> lineSets = fGeom->getTrimmedLines(iFace);
290  if (!lineSets.empty()) {
291  // this face has geometric hatch lines
292  newFace->clearLineSets();
293  for (auto& ls : lineSets) {
294  newFace->addLineSet(ls);
295  }
296  }
297  double hatchScale = fGeom->ScalePattern.getValue();
298  if (hatchScale > 0.0) {
299  newFace->setHatchScale(fGeom->ScalePattern.getValue());
300  }
301  newFace->setHatchRotation(fGeom->PatternRotation.getValue());
302  newFace->setHatchOffset(fGeom->PatternOffset.getValue());
303  newFace->setHatchFile(fGeom->PatIncluded.getValue());
304  Gui::ViewProvider* gvp = QGIView::getViewProvider(fGeom);
305  ViewProviderGeomHatch* geomVp = dynamic_cast<ViewProviderGeomHatch*>(gvp);
306  if (geomVp) {
307  newFace->setHatchColor(geomVp->ColorPattern.getValue());
308  newFace->setLineWeight(geomVp->WeightPattern.getValue());
309  }
310  } else if (fHatch) {
311  // svg or bitmap hatch
312  newFace->isHatched(true);
313  if (!fHatch->SvgIncluded.isEmpty()) {
314  newFace->setHatchFile(fHatch->SvgIncluded.getValue());
315  }
316  if (fHatch->isSvgHatch()) {
317  // svg tile hatch
318  newFace->setFillMode(QGIFace::SvgFill);
319  } else {
320  //bitmap hatch
321  newFace->setFillMode(QGIFace::BitmapFill);
322  }
323 
324  // get the properties from the hatch viewprovider
325  Gui::ViewProvider* gvp = QGIView::getViewProvider(fHatch);
326  ViewProviderHatch* hatchVp = dynamic_cast<ViewProviderHatch*>(gvp);
327  if (hatchVp) {
328  if (hatchVp->HatchScale.getValue() > 0.0) {
329  newFace->setHatchScale(hatchVp->HatchScale.getValue());
330  }
331  newFace->setHatchColor(hatchVp->HatchColor.getValue());
332  newFace->setHatchRotation(hatchVp->HatchRotation.getValue());
333  newFace->setHatchOffset(hatchVp->HatchOffset.getValue());
334  }
335  }
336 
337  newFace->setDrawEdges(prefFaceEdges());
338  newFace->setZValue(ZVALUE::FACE);
339  newFace->setPrettyNormal();
340  newFace->draw();
341  iFace++;
342  }

345 void QGIViewPart::drawAllEdges()
346 {
347  // dvp and vp already validated
348  auto dvp(static_cast<TechDraw::DrawViewPart*>(getViewObject()));
349  auto vp = static_cast<ViewProviderViewPart*>(getViewProvider(getViewObject()));
350 
351  const TechDraw::BaseGeomPtrVector& geoms = dvp->getEdgeGeometry();
352  TechDraw::BaseGeomPtrVector::const_iterator itGeom = geoms.begin();
353  QGIEdge* item;
354  for (int iEdge = 0; itGeom != geoms.end(); itGeom++, iEdge++) {
355  bool showItem = true;
356  if (!showThisEdge(*itGeom)) {
357  continue;
358  }
359 
360  item = new QGIEdge(iEdge);
361  addToGroup(item); //item is created at scene(0, 0), not group(0, 0)
362  item->setPath(drawPainterPath(*itGeom));
363  item->setSource((*itGeom)->source());
364 
365  item->setNormalColor(PreferencesGui::getAccessibleQColor(PreferencesGui::normalQColor()));
366  if ((*itGeom)->getCosmetic()) {
367  // cosmetic edge - format appropriately
368  int source = (*itGeom)->source();
369  if (source == COSMETICEDGE) {
370  std::string cTag = (*itGeom)->getCosmeticTag();
371  showItem = formatGeomFromCosmetic(cTag, item);
372  }
373  else if (source == CENTERLINE) {
374  std::string cTag = (*itGeom)->getCosmeticTag();
375  showItem = formatGeomFromCenterLine(cTag, item);
376  }
377  else {
378  Base::Console().Message("QGIVP::drawVP - cosmetic edge: %d is confused - source: %d\n",
379  iEdge, source);
380  }
381  } else {
382  // geometry edge - apply format if applicable
383  TechDraw::GeomFormat* gf = dvp->getGeomFormatBySelection(iEdge);
384  if (gf) {
385  App::Color color = Preferences::getAccessibleColor(gf->m_format.getColor());
386  item->setNormalColor(color.asValue<QColor>());
387  int lineNumber = gf->m_format.getLineNumber();
388  int qtStyle = gf->m_format.getStyle();
389  item->setLinePen(m_dashedLineGenerator->getBestPen(lineNumber, (Qt::PenStyle)qtStyle,
390  gf->m_format.getWidth()));
391  // but we need to actually draw the lines in QGScene coords (0.1 mm).
392  item->setWidth(Rez::guiX(gf->m_format.getWidth()));
393  showItem = gf->m_format.getVisible();
394  } else {
395  if (!(*itGeom)->getHlrVisible()) {
396  // hidden line without a format
397  item->setLinePen(m_dashedLineGenerator->getLinePen(Preferences::HiddenLineStyle(),
398  vp->LineWidth.getValue()));
399  item->setHiddenEdge(true);
400  item->setWidth(Rez::guiX(vp->HiddenWidth.getValue())); //thin
401  item->setZValue(ZVALUE::HIDEDGE);
402  } else {
403  // unformatted visible line, draw as continuous line
404  item->setLinePen(m_dashedLineGenerator->getLinePen(1, vp->LineWidth.getValue()));
405  item->setWidth(Rez::guiX(vp->LineWidth.getValue()));
406  }
407  }
408  }
409 
410  if ((*itGeom)->getClassOfEdge() == ecUVISO) {
411  // we don't have a style option for iso-parametric lines so draw continuous
412  item->setLinePen(m_dashedLineGenerator->getLinePen(1, vp->IsoWidth.getValue()));
413  item->setWidth(Rez::guiX(vp->IsoWidth.getValue())); //graphic
414  }
415 
416  item->setPos(0.0, 0.0);//now at group(0, 0)
417  item->setZValue(ZVALUE::EDGE);
418  item->setPrettyNormal();
419 
420  if (!vp->ShowAllEdges.getValue() && !showItem) {
421  //view level "show" status && individual edge "show" status
422  item->hide();
423  }
424 
425  //debug a path
426  // QPainterPath edgePath=drawPainterPath(*itGeom);
427  // std::stringstream edgeId;
428  // edgeId << "QGIVP.edgePath" << i;
429  // dumpPath(edgeId.str().c_str(), edgePath);
430  }
431 }

579 QGIFace* QGIViewPart::drawFace(TechDraw::FacePtr f, int idx)
580 {
581  // Base::Console().Message("QGIVP::drawFace - %d\n", idx);
582  std::vector<TechDraw::Wire*> fWires = f->wires;
583  QPainterPath facePath;
584  for (std::vector<TechDraw::Wire*>::iterator wire = fWires.begin(); wire != fWires.end();
585  ++wire) {
586  TechDraw::BaseGeomPtrVector geoms = (*wire)->geoms;
587  if (geoms.empty())
588  continue;
589 
590  TechDraw::BaseGeomPtr firstGeom = geoms.front();
591  QPainterPath wirePath;
592  //QPointF startPoint(firstGeom->getStartPoint().x, firstGeom->getStartPoint().y);
593  //wirePath.moveTo(startPoint);
594  QPainterPath firstSeg = drawPainterPath(firstGeom);
595  wirePath.connectPath(firstSeg);
596  for (TechDraw::BaseGeomPtrVector::iterator edge = ((*wire)->geoms.begin()) + 1;
597  edge != (*wire)->geoms.end(); ++edge) {
598  QPainterPath edgePath = drawPainterPath(*edge);
599  //handle section faces differently
600  if (idx == -1) {
601  QPointF wEnd = wirePath.currentPosition();
602  auto element = edgePath.elementAt(0);
603  QPointF eStart(element.x, element.y);
604  QPointF eEnd = edgePath.currentPosition();
605  QPointF sVec = wEnd - eStart;
606  QPointF eVec = wEnd - eEnd;
607  double sDist2 = sVec.x() * sVec.x() + sVec.y() * sVec.y();
