649  void setModeOnViewParameters()
650  {
651  // before each mode change we reset the dynamic override
652  ovpVisibilityManager.resetDynamicOverride();
653 
654  bool firstOfMode = true;
655  parameterWithFocus = -1;
656 
657  for (size_t i = 0; i < onViewParameters.size(); i++) {
658 
659  if (!isOnViewParameterOfCurrentMode(i)) {
660  onViewParameters[i]->stopEdit();
661  if (!onViewParameters[i]->isSet || handler->state() == SelectMode::End) {
662  onViewParameters[i]->deactivate();
663  }
664  }
665  else {
666 
667  if (firstOfMode) {
668  parameterWithFocus = static_cast<int>(i);
669  firstOfMode = false;
670  }
671 
672  bool visible = isOnViewParameterVisible(i);
673 
674  if (visible) {
675  onViewParameters[i]->activate();
676 
677  // points/value will be overridden by the mouseMove triggered by the mode
678  // change.
679  onViewParameters[i]->setPoints(Base::Vector3d(), Base::Vector3d());
680  onViewParameters[i]->startEdit(0.0, keymanager.get());
681  }
682  }
683  }

703  void passFocusToNextOnViewParameter()
704  {
705  unsigned int index = parameterWithFocus + 1;
706 
707  if (index >= onViewParameters.size()) {
708  index = 0;
709  }
710 
711  auto trySetFocus = [this](unsigned int& idx) -> bool {
712  while (idx < onViewParameters.size()) {
713  if (isOnViewParameterOfCurrentMode(idx) && isOnViewParameterVisible(idx)) {
714  setFocusToOnViewParameter(idx);
715  return true;
716  }
717  idx++;
718  }
719  return false;
720  };
721 
722  if (!trySetFocus(index)) {
723  // We have not found a parameter in this mode after current.
724  // So we go back to start and retry.
725  index = 0;
726  trySetFocus(index);
727  }
728 
729  // At that point if no onViewParameter is found, there is none.
730  }
