153 class RollbackImporter:
154  """This tricky little class is used to make sure that modules under test
155  will be reloaded the next time they are imported.
156  """
157 
158  def __init__(self):
159  self.previousModules = sys.modules.copy()
160 
161  def rollbackImports(self):
162  for modname in sys.modules.keys():
163  if modname not in self.previousModules:
164  # Force reload when modname next imported
165  del sys.modules[modname]
