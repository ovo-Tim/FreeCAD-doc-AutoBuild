32 class DraftObject(object):
33  """The base class for Draft objects.
34 
35  Parameters
36  ----------
37  obj : a base C++ object
38  The base object instantiated during creation,
39  which commonly may be of types `Part::Part2DObjectPython`,
40  `Part::FeaturePython`, or `App::FeaturePython`.
41 
42  >>> obj = App.ActiveDocument.addObject('Part::Part2DObjectPython')
43  >>> DraftObject(obj)
44 
45  This class instance is stored in the `Proxy` attribute
46  of the base object.
47  ::
48  obj.Proxy = self
49 
50  tp : str, optional
51  It defaults to `'Unknown'`.
52  It indicates the type of this scripted object,
53  which will be assigned to the Proxy's `Type` attribute.
54 
55  This is useful to distinguish different types of scripted objects
56  that may be derived from the same C++ class.
57 
58  Attributes
59  ----------
60  Type : str
61  It indicates the type of scripted object.
62  Normally `Type = tp`.
63 
64  All objects have a `TypeId` attribute, but this attribute
65  refers to the C++ class only. Using the `Type` attribute
66  allows distinguishing among various types of objects
67  derived from the same C++ class.
68 
69  >>> print(A.TypeId, "->", A.Proxy.Type)
70  Part::Part2DObjectPython -> Wire
71  >>> print(B.TypeId, "->", B.Proxy.Type)
72  Part::Part2DObjectPython -> Circle
73 
74  This class attribute is accessible through the `Proxy` object:
75  `obj.Proxy.Type`.
76  """

162  def props_changed_store(self, prop):
163  """Store the name of the property that will be changed in the
164  self.props_changed list.
165 
166  The function should be called at the start of onChanged or onBeforeChange.
167 
168  The list is used to detect Placement-only changes. In such cases the
169  Shape of an object does not need to be updated. Not updating the Shape
170  avoids Uno/Redo issues for the Windows version and is also more efficient.
171  """
172  if not hasattr(self, "props_changed"):
173  self.props_changed = [prop]
174  else:
175  self.props_changed.append(prop)

185  def props_changed_placement_only(self, obj=None):
186  """Return `True` if the self.props_changed list, after removing
187  `_LinkTouched`, `Shape`, `Density`, `Volume` and `Mass` items,
188  only contains `Placement` items.
189 
190  Parameters
191  ----------
192  obj : the scripted object. Need only be supplied if the Shape of obj
193  is, or can be, derived from other objects.
194 
195  """
196  if not hasattr(self, "props_changed"):
197  return False
198 
199  # For some objects a dummy Placement property change (new and old
200  # Placement are the same) is used in cases where a full recompute is
201  # required. This function should then return `False`. The common
202  # denominator seems to be a non-empty OutList.
203  # https://github.com/FreeCAD/FreeCAD/issues/8771
204  # https://forum.freecad.org/viewtopic.php?t=82436
205  if obj is not None and obj.OutList:
206  return False
207 
208  props = set(self.props_changed)
209  for prop in ("_LinkTouched", "Shape", "Density", "Volume", "Mass"):
210  if prop in props:
211  props.remove(prop)
212  return props == {"Placement"}
