45 template <class Real>
46 bool IntrSegment3Plane3<Real>::Test ()
47 {
48  Vector3<Real> kP0 = m_rkSegment.GetNegEnd();
49  Real fSDistance0 = m_rkPlane.DistanceTo(kP0);
50  if (Math<Real>::FAbs(fSDistance0) <= Math<Real>::ZERO_TOLERANCE)
51  {
52  fSDistance0 = (Real)0.0;
53  }
54 
55  Vector3<Real> kP1 = m_rkSegment.GetPosEnd();
56  Real fSDistance1 = m_rkPlane.DistanceTo(kP1);
57  if (Math<Real>::FAbs(fSDistance1) <= Math<Real>::ZERO_TOLERANCE)
58  {
59  fSDistance1 = (Real)0.0;
60  }
61 
62  Real fProd = fSDistance0*fSDistance1;
63  if (fProd < (Real)0.0)
64  {
65  // The segment passes through the plane.
66  m_iIntersectionType = IT_POINT;
67  return true;
68  }
69 
70  if (fProd > (Real)0.0)
71  {
72  // The segment is on one side of the plane.
73  m_iIntersectionType = IT_EMPTY;
74  return false;
75  }
76 
77  if (fSDistance0 != (Real)0.0 || fSDistance1 != (Real)0.0)
78  {
79  // A segment end point touches the plane.
80  m_iIntersectionType = IT_POINT;
81  return true;
82  }
83 
84  // The segment is coincident with the plane.
85  m_iIntersectionType = IT_SEGMENT;
86  return true;
87 }

89 template <class Real>
90 bool IntrSegment3Plane3<Real>::Find ()
91 {
92  Line3<Real> kLine(m_rkSegment.Origin,m_rkSegment.Direction);
93  IntrLine3Plane3<Real> kIntr(kLine,m_rkPlane);
94  if (kIntr.Find())
95  {
96  // The line intersects the plane, but possibly at a point that is
97  // not on the segment.
98  m_iIntersectionType = kIntr.GetIntersectionType();
99  m_fSegmentT = kIntr.GetLineT();
100  return Math<Real>::FAbs(m_fSegmentT) <= m_rkSegment.Extent;
101  }
102 
103  m_iIntersectionType = IT_EMPTY;
104  return false;
105 }
