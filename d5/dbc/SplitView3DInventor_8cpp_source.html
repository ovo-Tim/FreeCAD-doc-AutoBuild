576 SplitView3DInventor::SplitView3DInventor(int views, Gui::Document* pcDocument, QWidget* parent, Qt::WindowFlags wflags)
577  : AbstractSplitView(pcDocument,parent, wflags)
578 {
579  //anti-aliasing settings
580  bool smoothing = false;
581  bool glformat = false;
582  int samples = View3DInventorViewer::getNumSamples();
583  QtGLFormat f;
584 
585  if (samples > 1) {
586  glformat = true;
587  f.setSamples(samples);
588  }
589  else if (samples > 0) {
590  smoothing = true;
591  }
592 
593  // minimal 2 views
594  while (views < 2)
595  views ++;
596 
597  QSplitter* mainSplitter = nullptr;
598 
599  // if views < 3 show them as a row
600  if (views <= 3) {
601  mainSplitter = new QSplitter(Qt::Horizontal, this);
602  for (int i=0; i < views; i++) {
603  if (glformat)
604  _viewer.push_back(new View3DInventorViewer(f, mainSplitter));
605  else
606  _viewer.push_back(new View3DInventorViewer(mainSplitter));
607  }
608  }
609  else {
610  mainSplitter = new QSplitter(Qt::Vertical, this);
611  auto topSplitter = new QSplitter(Qt::Horizontal, mainSplitter);
612  auto botSplitter = new QSplitter(Qt::Horizontal, mainSplitter);
613 
614  if (glformat) {
615  _viewer.push_back(new View3DInventorViewer(f, topSplitter));
616  _viewer.push_back(new View3DInventorViewer(f, topSplitter));
617  }
618  else {
619  _viewer.push_back(new View3DInventorViewer(topSplitter));
620  _viewer.push_back(new View3DInventorViewer(topSplitter));
621  }
622 
623  for (int i=2;i<views;i++) {
624  if (glformat)
625  _viewer.push_back(new View3DInventorViewer(f, botSplitter));
626  else
627  _viewer.push_back(new View3DInventorViewer(botSplitter));
628  }
629 
630  topSplitter->setOpaqueResize(true);
631  botSplitter->setOpaqueResize(true);
632  }
633 
634  if (smoothing) {
635  for (std::vector<int>::size_type i = 0; i != _viewer.size(); i++)
636  _viewer[i]->getSoRenderManager()->getGLRenderAction()->setSmoothing(true);
637  }
638 
639  mainSplitter->show();
640  setCentralWidget(mainSplitter);
641 
642  setDocumentOfViewers(pcDocument);
643 
644  // apply the user settings
645  setupSettings();
646 }
