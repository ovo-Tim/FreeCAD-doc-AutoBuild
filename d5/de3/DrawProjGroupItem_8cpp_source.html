109 App::DocumentObjectExecReturn *DrawProjGroupItem::execute()
110 {
111 // Base::Console().Message("DPGI::execute() - %s / %s\n", getNameInDocument(), Label.getValue());
112  if (!keepUpdated()) {
113  return DrawView::execute();
114  }
115 
116  if (waitingForHlr()) {
117  return DrawView::execute();
118  }
119 
120  bool haveX = checkXDirection();
121  if (!haveX) {
122  //block touch/onChanged stuff
123  Base::Vector3d newX = getXDirection();
124  XDirection.setValue(newX);
125  XDirection.purgeTouched(); //don't trigger updates!
126  //unblock
127  }
128 
129  if (DrawUtil::checkParallel(Direction.getValue(), getXDirection())) {
130  return new App::DocumentObjectExecReturn("DPGI: Direction and XDirection are parallel");
131  }
132 
133  return DrawViewPart::execute();
134 }

151 void DrawProjGroupItem::autoPosition()
152 {
153  DrawProjGroup* pGroup = getPGroup();
154  if (!pGroup) {
155  return;
156  }
157 // Base::Console().Message("DPGI::autoPosition(%s)\n", Label.getValue());
158  if (LockPosition.getValue()) {
159  return;
160  }
161  Base::Vector3d newPos;
162  if (pGroup && pGroup->AutoDistribute.getValue()) {
163  newPos = pGroup->getXYPosition(Type.getValueAsString());
164  X.setValue(newPos.x);
165  Y.setValue(newPos.y);
166  requestPaint();
167  purgeTouched(); //prevents "still touched after recompute" message
168  pGroup->purgeTouched(); //changing dpgi x, y marks parent dpg as touched
169  }
170 }

192 Base::Vector3d DrawProjGroupItem::getXDirection() const
193 {
194 // Base::Console().Message("DPGI::getXDirection() - %s\n", Label.getValue());
195  Base::Vector3d result(1.0, 0.0, 0.0); //default X
196  App::Property* prop = getPropertyByName("XDirection");
197  if (prop) {
198  Base::Vector3d propVal = XDirection.getValue();
199  if (DrawUtil::fpCompare(propVal.Length(), 0.0)) { //have XDirection property, but not set
200  prop = getPropertyByName("RotationVector");
201  if (prop) {
202  result = RotationVector.getValue(); //use RotationVector if we have it
203  }
204  else {
205  result = DrawViewPart::getXDirection(); //over complex.
206  }
207  }
208  else {
209  result = DrawViewPart::getXDirection();
210  }
211  }
212  else { //not sure this branch can actually happen
213  Base::Console().Message("DPGI::getXDirection - unexpected branch taken!\n");
214  prop = getPropertyByName("RotationVector");
215  if (prop) {
216  result = RotationVector.getValue();
217 
218  }
219  else {
220  Base::Console().Message("DPGI::getXDirection - missing RotationVector and XDirection\n");
221  }
222  }
223  return result;
224 }

226 Base::Vector3d DrawProjGroupItem::getLegacyX(const Base::Vector3d& pt,
227  const Base::Vector3d& axis,

255 //get the angle between the current RotationVector vector and the original X dir angle
256 double DrawProjGroupItem::getRotateAngle()
257 {
258  gp_Ax2 viewAxis;
259  Base::Vector3d x = getXDirection(); //current rotation
260  Base::Vector3d nx = x;
261  nx.Normalize();
262  Base::Vector3d na = Direction.getValue();
263  na.Normalize();
264  Base::Vector3d org(0.0, 0.0, 0.0);
265 
266  viewAxis = getProjectionCS(org);
267  gp_Dir gxDir = viewAxis.XDirection();
268  Base::Vector3d origX(gxDir.X(), gxDir.Y(), gxDir.Z());
269  origX.Normalize();
270  double angle = origX.GetAngle(nx);
271 
272  Base::Vector3d rotAxis = origX.Cross(nx);
273  if (rotAxis == Direction.getValue()) {
274  angle *= -1.0;
275  }
276  return angle;
277 }
