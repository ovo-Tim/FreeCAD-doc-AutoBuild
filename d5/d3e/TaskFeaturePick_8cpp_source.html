59 const QString TaskFeaturePick::getFeatureStatusString(const featureStatus st)
60 {
61  switch (st) {
62  case validFeature:
63  return tr("Valid");
64  case invalidShape:
65  return tr("Invalid shape");
66  case noWire:
67  return tr("No wire in sketch");
68  case isUsed:
69  return tr("Sketch already used by other feature");
70  case otherBody:
71  return tr("Belongs to another body");
72  case otherPart:
73  return tr("Belongs to another part");
74  case notInBody:
75  return tr("Doesn't belong to any body");
76  case basePlane:
77  return tr("Base plane");
78  case afterTip:
79  return tr("Feature is located after the tip feature");
80  }
81 
82  return QString();

89  : TaskBox(Gui::BitmapFactory().pixmap("edit-select-all"), tr("Select attachment"), true, parent)

92 {
93 
94  proxy = new QWidget(this);

97  // clang-format off
98  connect(ui->checkUsed, &QCheckBox::toggled, this, &TaskFeaturePick::onUpdate);
99  connect(ui->checkOtherBody, &QCheckBox::toggled, this, &TaskFeaturePick::onUpdate);
100  connect(ui->checkOtherPart, &QCheckBox::toggled, this, &TaskFeaturePick::onUpdate);
101  connect(ui->radioIndependent, &QRadioButton::toggled, this, &TaskFeaturePick::onUpdate);

104  connect(ui->listWidget, &QListWidget::itemSelectionChanged, this, &TaskFeaturePick::onItemSelectionChanged);
105  connect(ui->listWidget, &QListWidget::itemDoubleClicked, this, &TaskFeaturePick::onDoubleClick);
106  // clang-format on
107 
108 
109  if (!singleFeatureSelect) {
110  ui->listWidget->setSelectionMode(QAbstractItemView::ExtendedSelection);
111  }
112 
113  enum
114  {
115  axisBit = 0,
116  planeBit = 1
117  };
118 
119  // NOTE: generally there shouldn't be more then one origin
120  std::map<App::Origin*, std::bitset<2>> originVisStatus;
121 
122  auto statusIt = status.cbegin();
123  auto objIt = objects.begin();
124  assert(status.size() == objects.size());
125 
126  bool attached = false;
127  for (; statusIt != status.end(); ++statusIt, ++objIt) {
128  QListWidgetItem* item = new QListWidgetItem(
129  QString::fromLatin1("%1 (%2)").arg(QString::fromUtf8((*objIt)->Label.getValue()),
130  getFeatureStatusString(*statusIt)));
131  item->setData(Qt::UserRole, QString::fromLatin1((*objIt)->getNameInDocument()));
132  ui->listWidget->addItem(item);
133 
134  App::Document* pDoc = (*objIt)->getDocument();
135  documentName = pDoc->getName();
136  if (!attached) {
137  attached = true;
138  attachDocument(Gui::Application::Instance->getDocument(pDoc));
139  }
140 
141  // check if we need to set any origin in temporary visibility mode
142  auto* datum = dynamic_cast<App::DatumElement*>(*objIt);
143  if (*statusIt != invalidShape && datum) {
144  App::Origin* origin = dynamic_cast<App::Origin*>(datum->getLCS());
145  if (origin) {
146  if ((*objIt)->isDerivedFrom(App::Plane::getClassTypeId())) {
147  originVisStatus[origin].set(planeBit, true);
148  }
149  else if ((*objIt)->isDerivedFrom(App::Line::getClassTypeId())) {
150  originVisStatus[origin].set(axisBit, true);
151  }
152  }
153  }
154  }
155 
156  // Setup the origin's temporary visibility
157  for (const auto& originPair : originVisStatus) {
158  const auto& origin = originPair.first;
159 
160  Gui::ViewProviderCoordinateSystem* vpo = static_cast<Gui::ViewProviderCoordinateSystem*>(
161  Gui::Application::Instance->getViewProvider(origin));
162  if (vpo) {
163  vpo->setTemporaryVisibility(originVisStatus[origin][axisBit],
164  originVisStatus[origin][planeBit]);
165  vpo->setTemporaryScale(4.0); // NOLINT
166  vpo->setPlaneLabelVisibility(true);
167  origins.push_back(vpo);
168  }
169  }
170 
171  // TODO may be update origin API to show only some objects (2015-08-31, Fat-Zer)
172 
173  groupLayout()->addWidget(proxy);
174  statuses = status;
175  updateList();
176 }

268 std::vector<App::DocumentObject*> TaskFeaturePick::buildFeatures()
269 {
270  int index = 0;
271  std::vector<App::DocumentObject*> result;
272  try {
273  auto activeBody = PartDesignGui::getBody(false);
274  if (!activeBody) {
275  return result;
276  }
277 
278  auto activePart = PartDesignGui::getPartFor(activeBody, false);
279 
280  for (auto status : statuses) {
281  QListWidgetItem* item = ui->listWidget->item(index);
282 
283  if (item->isSelected() && !item->isHidden()) {
284  QString t = item->data(Qt::UserRole).toString();
285  auto obj = App::GetApplication()
286  .getDocument(documentName.c_str())
287  ->getObject(t.toLatin1().data());
288 
289  // build the dependent copy or reference if wanted by the user
290  if (status == otherBody || status == otherPart || status == notInBody) {
291  if (!ui->radioXRef->isChecked()) {
292  auto copy = makeCopy(obj, "", ui->radioIndependent->isChecked());
293 
294  if (status == otherBody) {
295  activeBody->addObject(copy);
296  }
297  else if (status == otherPart) {
298  auto oBody = PartDesignGui::getBodyFor(obj, false);
299  if (!oBody) {
300  activePart->addObject(copy);
301  }
302  else {
303  activeBody->addObject(copy);
304  }
305  }
306  else if (status == notInBody) {
307  activeBody->addObject(copy);
308  // doesn't supposed to get here anything but sketch but to be on the
309  // safe side better to check
310  if (copy->isDerivedFrom<Sketcher::SketchObject>()) {
311  Sketcher::SketchObject* sketch =
312  static_cast<Sketcher::SketchObject*>(copy);
313  PartDesignGui::fixSketchSupport(sketch);
314  }
315  }
316  result.push_back(copy);
317  }
318  else {
319  result.push_back(obj);
320  }
321  }
322  else {
323  result.push_back(obj);
324  }
325  }
326 
327  index++;
328  }
329  }
330  catch (const Base::Exception& e) {
331  e.ReportException();
332  }
333  catch (Py::Exception& e) {
334  // reported by code analyzers
335  e.clear();
336  Base::Console().Warning("Unexpected PyCXX exception\n");
337  }
338  catch (const boost::exception&) {
339  // reported by code analyzers
340  Base::Console().Warning("Unexpected boost exception\n");
341  }
342 
343  return result;
344 }

346 App::DocumentObject*
347 TaskFeaturePick::makeCopy(App::DocumentObject* obj, std::string sub, bool independent)
348 {
349 
350  App::DocumentObject* copy = nullptr;
351  // Check for null to avoid segfault
352  if (!obj) {
353  return copy;
354  }
355  if (independent
356  && (obj->isDerivedFrom(Sketcher::SketchObject::getClassTypeId())
357  || obj->isDerivedFrom(PartDesign::FeaturePrimitive::getClassTypeId()))) {
358 
359  // we do know that the created instance is a document object, as obj is one. But we do not
360  // know which exact type
361  auto name = std::string("Copy") + std::string(obj->getNameInDocument());
362  copy = App::GetApplication().getActiveDocument()->addObject(obj->getTypeId().getName(),
363  name.c_str());
364 
365  // copy over all properties
366  std::vector<App::Property*> props;
367  std::vector<App::Property*> cprops;
368  obj->getPropertyList(props);
369  copy->getPropertyList(cprops);
370 
371  auto it = cprops.begin();
372  for (App::Property* prop : props) {
373 
374  // independent copies don't have links and are not attached
375  if (independent
376  && (prop->isDerivedFrom<App::PropertyLink>()
377  || prop->isDerivedFrom<App::PropertyLinkList>()
378  || prop->isDerivedFrom<App::PropertyLinkSub>()
379  || prop->isDerivedFrom<App::PropertyLinkSubList>()
380  || (prop->getGroup() && strcmp(prop->getGroup(), "Attachment") == 0))) {
381 
382  ++it;
383  continue;
384  }
385 
386  App::Property* cprop = *it++;
387 
388  if (prop->getName() && strcmp(prop->getName(), "Label") == 0) {
389  static_cast<App::PropertyString*>(cprop)->setValue(name.c_str());
390  continue;
391  }
392 
393  cprop->Paste(*prop);
394 
395  // we are a independent copy, therefore no external geometry was copied. WE therefore
396  // can delete all constraints
397  if (obj->isDerivedFrom(Sketcher::SketchObject::getClassTypeId())) {
398  static_cast<Sketcher::SketchObject*>(copy)->delConstraintsToExternal();
399  }
400  }
401  }
402  else {
403 
404  std::string name;
405  if (!independent) {
406  name = std::string("Reference");
407  }
408  else {
409  name = std::string("Copy");
410  }
411  name += std::string(obj->getNameInDocument());
412 
413  std::string entity;
414  if (!sub.empty()) {
415  entity = sub;
416  }
417 
418  Part::PropertyPartShape* shapeProp = nullptr;
419 
420  // TODO Replace it with commands (2015-09-11, Fat-Zer)
421  if (obj->isDerivedFrom(Part::Datum::getClassTypeId())) {
422  copy = App::GetApplication().getActiveDocument()->addObject(obj->getTypeId().getName(),
423  name.c_str());
424 
425  assert(copy->isDerivedFrom(Part::Datum::getClassTypeId()));
426 
427  // we need to reference the individual datums and make again datums. This is important
428  // as datum adjust their size dependent on the part size, hence simply copying the shape
429  // is not enough
430  long int mode = mmDeactivated;
431  Part::Datum* datumCopy = static_cast<Part::Datum*>(copy);
432 
433  if (obj->is<PartDesign::Point>()) {
434  mode = mm0Vertex;
435  }
436  else if (obj->is<PartDesign::Line>()) {
437  mode = mm1TwoPoints;
438  }
439  else if (obj->is<PartDesign::Plane>()) {
440  mode = mmFlatFace;
441  }
442  else {
443  return copy;
444  }
445 
446  // TODO Recheck this. This looks strange in case of independent copy (2015-10-31,
447  // Fat-Zer)
448  if (!independent) {
449  datumCopy->AttachmentSupport.setValue(obj, entity.c_str());
450  datumCopy->MapMode.setValue(mode);
451  }
452  else if (!entity.empty()) {
453  datumCopy->Shape.setValue(
454  static_cast<Part::Datum*>(obj)->Shape.getShape().getSubShape(entity.c_str()));
455  }
456  else {
457  datumCopy->Shape.setValue(static_cast<Part::Datum*>(obj)->Shape.getValue());
458  }
459  }
460  else if (obj->is<PartDesign::ShapeBinder>()
461  || obj->isDerivedFrom(Part::Feature::getClassTypeId())) {
462 
463  copy = App::GetApplication().getActiveDocument()->addObject("PartDesign::ShapeBinder",
464  name.c_str());
465 
466  if (!independent) {
467  static_cast<PartDesign::ShapeBinder*>(copy)->Support.setValue(obj, entity.c_str());
468  }
469  else {
470  shapeProp = &static_cast<PartDesign::ShapeBinder*>(copy)->Shape;
471  }
472  }
473  else if (obj->isDerivedFrom(App::Plane::getClassTypeId())
474  || obj->isDerivedFrom(App::Line::getClassTypeId())) {
475 
476  copy = App::GetApplication().getActiveDocument()->addObject("PartDesign::ShapeBinder",
477  name.c_str());
478 
479  if (!independent) {
480  static_cast<PartDesign::ShapeBinder*>(copy)->Support.setValue(obj, entity.c_str());
481  }
482  else {
483  App::GeoFeature* geo = static_cast<App::GeoFeature*>(obj);
484  std::vector<std::string> subvalues;
485  subvalues.push_back(entity);
486  Part::TopoShape shape =
487  PartDesign::ShapeBinder::buildShapeFromReferences(geo, subvalues);
488  static_cast<PartDesign::ShapeBinder*>(copy)->Shape.setValue(shape);
489  }
490  }
491 
492  if (independent && shapeProp) {
493  if (entity.empty()) {
494  shapeProp->setValue(static_cast<Part::Feature*>(obj)->Shape.getValue());
495  }
496  else {
497  shapeProp->setValue(
498  static_cast<Part::Feature*>(obj)->Shape.getShape().getSubShape(entity.c_str()));
499  }
500  }
501  }
502 
503  return copy;
504 }
