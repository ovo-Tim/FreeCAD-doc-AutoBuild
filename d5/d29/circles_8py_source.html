50  closest = None

58 def getCircleFromSpline(edge):
59  """Return a circle-based edge from a bspline-based edge.
60 
61  Return None if the edge is not a BSplineCurve.
62  """
63  if geomType(edge) != "BSplineCurve" or len(edge.Vertexes) != 1:
64  return None
65 
66  # get 2 points
67  p1 = edge.Curve.value(0)
68  p2 = edge.Curve.value(math.pi/2)
69  # get 2 tangents
70  t1 = edge.Curve.tangent(0)[0]
71  t2 = edge.Curve.tangent(math.pi/2)[0]
72  # get normal
73  n = p1.cross(p2)
74  if DraftVecUtils.isNull(n):
75  return None
76 
77  # get rays
78  r1 = DraftVecUtils.rotate(t1, math.pi/2, n)
79  r2 = DraftVecUtils.rotate(t2, math.pi/2, n)
80  # get center (intersection of rays)
81  i = findIntersection(p1, p1.add(r1), p2, p2.add(r2), True, True)
82  if not i:
83  return None
84 
85  c = i[0]
86  r = (p1.sub(c)).Length
87  circle = Part.makeCircle(r, c, n)
88  # print(circle.Curve)
89  return circle

92 def circlefrom1Line2Points(edge, p1, p2):
93  """Return a list of circles created from an edge and two points.
94 
95  It calculates up to 2 possible centers.
96  """
97  p1_p2 = edg(p1, p2)
98  s = findIntersection(edge, p1_p2, True, True)
99  if not s:
100  return None
101 
102  s = s[0]
103  v1 = p1.sub(s)
104  v2 = p2.sub(s)
105  projectedDist = math.sqrt(abs(v1.dot(v2)))
106  edgeDir = vec(edge)
107  edgeDir.normalize()
108 
109  projectedCen1 = App.Vector.add(s, App.Vector(edgeDir).multiply(projectedDist))
110  projectedCen2 = App.Vector.add(s, App.Vector(edgeDir).multiply(-projectedDist))
111  perpEdgeDir = edgeDir.cross(App.Vector(0, 0, 1))
112  perpCen1 = App.Vector.add(projectedCen1, perpEdgeDir)
113  perpCen2 = App.Vector.add(projectedCen2, perpEdgeDir)
114 
115  mid = findMidpoint(p1_p2)
116  x = DraftVecUtils.crossproduct(vec(p1_p2))
117  x.normalize()
118  perp_mid = App.Vector.add(mid, x)
119  cen1 = findIntersection(edg(projectedCen1, perpCen1),
120  edg(mid, perp_mid), True, True)
121  cen2 = findIntersection(edg(projectedCen2, perpCen2),
122  edg(mid, perp_mid), True, True)
123 
124  circles = []
125  if cen1:
126  radius = DraftVecUtils.dist(projectedCen1, cen1[0])
127  circles.append(Part.Circle(cen1[0], NORM, radius))
128 
129  if cen2:
130  radius = DraftVecUtils.dist(projectedCen2, cen2[0])
131  circles.append(Part.Circle(cen2[0], NORM, radius))
132 
133  if circles:
134  return circles
135  else:
136  return None

149 def circleFrom2LinesRadius(edge1, edge2, radius):
150  """Return a list of circles from two edges and one radius.
151 
152  It calculates 4 centers.
153  """
154  intsec = findIntersection(edge1, edge2, True, True)
155  if not intsec:
156  return None
157 
158  intsec = intsec[0]
159  bis12 = angleBisection(edge1, edge2)
160  bis21 = Part.LineSegment(bis12.Vertexes[0].Point,
161  DraftVecUtils.rotate(vec(bis12), math.pi/2.0))
162  ang12 = abs(DraftVecUtils.angle(vec(edge1), vec(edge2)))
163  ang21 = math.pi - ang12
164  dist12 = radius / math.sin(ang12 * 0.5)
165  dist21 = radius / math.sin(ang21 * 0.5)
166 
167  circles = []
168  cen = App.Vector.add(intsec, vec(bis12).multiply(dist12))
169  circles.append(Part.Circle(cen, NORM, radius))
170 
171  cen = App.Vector.add(intsec, vec(bis12).multiply(-dist12))
172  circles.append(Part.Circle(cen, NORM, radius))
173 
174  cen = App.Vector.add(intsec, vec(bis21).multiply(dist21))
175  circles.append(Part.Circle(cen, NORM, radius))
176 
177  cen = App.Vector.add(intsec, vec(bis21).multiply(-dist21))
178  circles.append(Part.Circle(cen, NORM, radius))
179 
180  return circles

245 def circleFromPointLineRadius(point, edge, radius):
246  """Return a list of circles from one point, one edge, and one radius.
247 
248  It calculates up to 2 possible centers.
249  """
250  dist = findDistance(point, edge, False)
251  center1 = None
252  center2 = None
253 
254  if dist.Length == 0:
255  segment = vec(edge)
256  perpVec = DraftVecUtils.crossproduct(segment)
257  perpVec.normalize()
258  normPoint_c1 = App.Vector(perpVec).multiply(radius)
259  normPoint_c2 = App.Vector(perpVec).multiply(-radius)
260  center1 = point.add(normPoint_c1)
261  center2 = point.add(normPoint_c2)
262  elif dist.Length > 2 * radius:
263  return None
264  elif dist.Length == 2 * radius:
265  normPoint = point.add(findDistance(point, edge, False))
266  dummy = (normPoint.sub(point)).multiply(0.5)
267  cen = point.add(dummy)
268  circ = Part.Circle(cen, NORM, radius)
269  if circ:
270  return [circ]
271  else:
272  return None
273  else:
274  normPoint = point.add(findDistance(point, edge, False))
275  normDist = DraftVecUtils.dist(normPoint, point)
276  dist = math.sqrt(radius**2 - (radius - normDist)**2)
277  centerNormVec = DraftVecUtils.scaleTo(point.sub(normPoint), radius)
278  edgeDir = edge.Vertexes[0].Point.sub(normPoint)
279  edgeDir.normalize()
280  center1 = centerNormVec.add(normPoint.add(App.Vector(edgeDir).multiply(dist)))
281  center2 = centerNormVec.add(normPoint.add(App.Vector(edgeDir).multiply(-dist)))
282 
283  circles = []
284  if center1:
285  circ = Part.Circle(center1, NORM, radius)
286  if circ:
287  circles.append(circ)
288 
289  if center2:
290  circ = Part.Circle(center2, NORM, radius)
291  if circ:
292  circles.append(circ)
293 
294  if circles:
295  return circles
296  else:
297  return None

343 
344  Return None if the objects are not circles, or if they are concentric.
345

347  http://mathworld.wolfram.com/HomotheticCenter.html
348  """

350  print("debug: findHomotheticCenterOfCircles bad parameters!")
351  return None

354  circle2.Curve.Center):
355  return None
356 
357  cen1_cen2 = Part.LineSegment(circle1.Curve.Center,
358  circle2.Curve.Center).toShape()
359  cenDir = vec(cen1_cen2)
360  cenDir.normalize()
361 
362  # Get the perpendicular vector.
363  perpCenDir = cenDir.cross(App.Vector(0, 0, 1))
364  perpCenDir.normalize()
365 
366  # Get point on first circle
367  p1 = App.Vector.add(circle1.Curve.Center,
368  App.Vector(perpCenDir).multiply(circle1.Curve.Radius))
369 
370  centers = []
371  # Calculate inner homothetic center
372  # Get point on second circle
373  p2_inner = App.Vector.add(circle1.Curve.Center,
374  App.Vector(perpCenDir).multiply(-circle1.Curve.Radius))
375  hCenterInner = DraftVecUtils.intersect(circle1.Curve.Center,
376  circle2.Curve.Center,
377  p1, p2_inner,
378  True, True)
379  if hCenterInner:
380  centers.append(hCenterInner)

383  # have different radii

385  # Get point on second circle
386  p2_outer = App.Vector.add(circle1.Curve.Center,

401 def findRadicalAxis(circle1, circle2):
402  """Calculate the radical axis of two circles.
403 
404  On the radical axis (also called power line) of two circles any
405  tangents drawn from a point on the axis to both circles have
406  the same length.
407 
408  http://en.wikipedia.org/wiki/Radical_axis
409  http://mathworld.wolfram.com/RadicalLine.html
410 
411  See Also
412  --------
413  findRadicalCenter
414  """
415  if (geomType(circle1) == "Circle") and (geomType(circle2) == "Circle"):
416  print("debug: findRadicalAxis bad parameters! Must be circles.")
417  return None
418 
419  if DraftVecUtils.equals(circle1.Curve.Center, circle2.Curve.Center):
420  return None
421 
422  r1 = circle1.Curve.Radius
423  r2 = circle1.Curve.Radius
424  cen1 = circle1.Curve.Center
425  # dist .. the distance from cen1 to cen2.
426  dist = DraftVecUtils.dist(cen1, circle2.Curve.Center)
427  cenDir = cen1.sub(circle2.Curve.Center)
428  cenDir.normalize()
429 
430  # Get the perpendicular vector.
431  perpCenDir = cenDir.cross(App.Vector(0, 0, 1))
432  perpCenDir.normalize()
433 
434  # J ... The radical center.
435  # K ... The point where the cadical axis crosses the line of cen1->cen2.
436  # k1 ... Distance from cen1 to K.
437  # k2 ... Distance from cen2 to K.
438  # dist = k1 + k2
439 
440  k1 = (dist + (r1**2 - r2**2) / dist) / 2.0
441  # k2 = dist - k1
442 
443  K = App.Vector.add(cen1, cenDir.multiply(k1))
444 
445  # K_ .. A point somewhere between K and J; actually with a distance
446  # of 1 unit from K.
447  K_ = App.Vector.add(K, perpCenDir)
448 
449  # Original code didn't specify the value of origin nor dir,
450  # so this is a guess:
451  # radicalAxis = Part.LineSegment(K, Vector.add(origin, dir))
452 
453  origin = App.Vector(0, 0, 0)
454  radicalAxis = Part.LineSegment(K, App.Vector.add(origin, perpCenDir))
455 
456  if radicalAxis:
457  return radicalAxis
458  else:
459  return None

462 def findRadicalCenter(circle1, circle2, circle3):
463  """Calculate the radical center of three circles.
464 
465  It is also called the power center.
466  It is the intersection point of the three radical axes of the pairs
467  of circles.
468 
469  http://en.wikipedia.org/wiki/Power_center_(geometry)
470  http://mathworld.wolfram.com/RadicalCenter.html
471 
472  See Also
473  --------
474  findRadicalAxis
475  """
476  if (geomType(circle1) != "Circle"
477  or geomType(circle2) != "Circle"
478  or geomType(circle3) != "Circle"):
479  print("debug: findRadicalCenter bad parameters! Must be circles.")
480  return None
481 
482  radicalAxis12 = findRadicalAxis(circle1, circle2)
483  radicalAxis23 = findRadicalAxis(circle2, circle3)
484 
485  if not radicalAxis12 or not radicalAxis23:
486  # No radical center could be calculated.
487  return None
488 
489  intersect = findIntersection(radicalAxis12, radicalAxis23, True, True)
490 
491  if intersect:
492  return intersect
493  else:
494  # No radical center could be calculated.
495  return None
