33 def create_material(element, parent, recursive=False):
34  """Creates a material object in the given project or parent material"""
35 
36  if not element:
37  return
38  if isinstance(element, (tuple, list)):
39  for e in element:
40  create_material(e, parent, recursive)
41  return
42  if hasattr(parent, "Document"):
43  doc = parent.Document
44  else:
45  doc = parent
46  exobj = ifc_tools.get_object(element, doc)
47  if exobj:
48  return exobj
49  obj = ifc_tools.add_object(doc, otype="material")
50  ifcfile = ifc_tools.get_ifcfile(parent)
51  ifc_tools.add_properties(obj, ifcfile, element)
52  if parent.isDerivedFrom("App::MaterialObject"):
53  parent.Proxy.addObject(parent, obj)
54  else:
55  ifc_tools.get_group(parent, "IfcMaterialsGroup").addObject(obj)
56  if recursive:
57  submat = get_material(obj)
58  if isinstance(submat, list):
59  for s in submat:
60  create_material(s, obj, recursive)
61  else:
62  create_material(submat, obj, recursive)

91 def get_material(obj):
92  """Returns a material attached to this object"""
93 
94  element = ifc_tools.get_ifc_element(obj)
95  if not element:
96  return None
97  if element.is_a("IfcMaterialConstituentSet"):
98  return element.MaterialConstituents
99  elif element.is_a() in [
100  "IfcMaterialLayer",
101  "IfcMaterialConstituent",
102  "IfcMaterialProfile",
103  ]:
104  return element.Material
105  elif element.is_a("IfcMaterialLayerSet"):
106  return element.MaterialLayers
107  elif element.is_a("IfcMaterialProfileSet"):
108  return element.MaterialProfiles
109  else:
110  material = ifcopenshell.util.element.get_material(
111  element, should_skip_usage=True
112  )
