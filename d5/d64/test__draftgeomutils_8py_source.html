37  def check_wire(self, wire):
38  offset_values = (2000.0, 0.0, -1000, -2000, -3000, -5500)
39  for offset_start in offset_values:
40  for offset_end in offset_values:
41  if offset_start + offset_end > -wire.Length:
42  # print ("'start={0}, end={1}' failed".format(offset_start, offset_end))
43  try:
44  extended = DraftGeomUtils.get_extended_wire(wire, offset_start, offset_end)
45  # Test that the extended wire's length is correctly changed
46  self.assertAlmostEqual(extended.Length, wire.Length + offset_start + offset_end,
47  DraftGeomUtils.precision(), "'start={0}, end={1}' failed".format(offset_start, offset_end))
48  if offset_start == 0.0:
49  # If offset_start is 0.0, check that the wire's start point is unchanged
50  self.assertAlmostEqual(extended.OrderedVertexes[0].Point.distanceToPoint(wire.OrderedVertexes[0].Point), 0.0,
51  DraftGeomUtils.precision(), "'start={0}, end={1}' failed".format(offset_start, offset_end))
52  if offset_end == 0.0:
53  # If offset_end is 0.0, check that the wire's end point is unchanged
54  self.assertAlmostEqual(extended.OrderedVertexes[-1].Point.distanceToPoint(wire.OrderedVertexes[-1].Point), 0.0,
55  DraftGeomUtils.precision(), "'start={0}, end={1}' failed".format(offset_start, offset_end))
56  except Exception as exc:
57  print ("get_extended_wire failed for 'start={0}, end={1}'".format(offset_start, offset_end))
58  raise exc
