165  function baseDecode(input) {
166  const baseCt = base.length;
167  const output = [];
168  const len = parseInt(input[0]); // num chars of each element
169  for (let i = 1; i < input.length; i += len) {
170  const str = input.substring(i, i + len).trim();
171  let val = 0;
172  for (let s = 0; s < str.length; s++) {
173  const ind = base.indexOf(str[s]);
174  val += ind * Math.pow(baseCt, s);
175  }
176  output.push(val);
177  }
178  return output;
179  }

181  function floatDecode(input) {
182  const baseCt = base.length;
183  const baseFloatCt = baseFloat.length;
184  let numString = '';
185  for (let i = 0; i < input.length; i += 4) {
186  const b90chunk = input.substring(i, i + 4).trim();
187  let quotient = 0;
188  for (let s = 0; s < b90chunk.length; s++) {
189  const ind = base.indexOf(b90chunk[s]);
190  quotient += ind * Math.pow(baseCt, s);
191  }
192  let buffer = '';
193  for (let s = 0; s < 7; s++) {
194  buffer = baseFloat[quotient % baseFloatCt] + buffer;
195  quotient = parseInt(quotient / baseFloatCt);
196  }
197  numString += buffer;
198  }
199  let trailingCommas = 0;
200  for (let s = 1; s < 7; s++) {
201  if (numString[numString.length - s] == baseFloat[0]) {
202  trailingCommas++;
203  }
204  }
205  numString = numString.substring(0, numString.length - trailingCommas);
206  return numString;
207  }

891  """Compress floats to base 90
892

894  """

899  baseCt = len(base)

903  for s in range(len(str7)):
904  quotient += baseFloat.find(str7[s]) * pow(baseFloatCt, (6 - s))
