76 void CmdPrimtiveCompAdditive::activated(int iMsg)
77 {
78  App::Document *doc = getDocument();
79 
80  // We need either an active Body, or for there to be no Body objects
81  // (in which case, just make one) to make a new additive shape.
82 
83  PartDesign::Body *pcActiveBody = PartDesignGui::getBody( /* messageIfNot = */ false );
84 
85  auto shouldMakeBody( false );
86  if (!pcActiveBody) {
87  if ( doc->getObjectsOfType(PartDesign::Body::getClassTypeId()).empty() ) {
88  shouldMakeBody = true;
89  } else {
90  PartDesignGui::DlgActiveBody dia(Gui::getMainWindow(), doc);
91  if (dia.exec() == QDialog::DialogCode::Accepted)
92  pcActiveBody = dia.getActiveBody();
93  if (!pcActiveBody)
94  return;
95  }
96  }
97 
98  Gui::ActionGroup* pcAction = qobject_cast<Gui::ActionGroup*>(_pcAction);
99  pcAction->setIcon(pcAction->actions().at(iMsg)->icon());
100 
101  auto shapeType( primitiveIntToName(iMsg) );
102 
103  Gui::Command::openCommand( (std::string("Make additive ") + shapeType).c_str() );
104  if (shouldMakeBody) {
105  pcActiveBody = PartDesignGui::makeBody(doc);
106  }
107 
108  if (!pcActiveBody) {
109  return;
110  }
111 
112  auto FeatName( getUniqueObjectName(shapeType, pcActiveBody) );
113 
114  FCMD_OBJ_DOC_CMD(pcActiveBody,"addObject('PartDesign::Additive"<<shapeType<<"','"<<FeatName<<"')");
115 
116  auto* prm = static_cast<PartDesign::FeaturePrimitive*>(
117  pcActiveBody->getDocument()->getObject(FeatName.c_str()));
118 
119  if(!prm)
120  return;
121  FCMD_OBJ_CMD(pcActiveBody,"addObject("<<getObjectCmd(prm)<<")");
122  Gui::Command::updateActive();
123 
124  auto base = prm->BaseFeature.getValue();
125  FCMD_OBJ_HIDE(base);
126 
127  if(!base)
128  base = pcActiveBody;
129  copyVisual(prm, "ShapeAppearance", base);
130  copyVisual(prm, "LineColor", base);
131  copyVisual(prm, "PointColor", base);
132  copyVisual(prm, "Transparency", base);
133  copyVisual(prm, "DisplayMode", base);
134 
135  PartDesignGui::setEdit(prm,pcActiveBody);
136 }

249 void CmdPrimtiveCompSubtractive::activated(int iMsg)
250 {
251  PartDesign::Body *pcActiveBody = PartDesignGui::getBody(true);
252 
253  if (!pcActiveBody)
254  return;
255 
256  Gui::ActionGroup* pcAction = qobject_cast<Gui::ActionGroup*>(_pcAction);
257  pcAction->setIcon(pcAction->actions().at(iMsg)->icon());
258 
259  //check if we already have a feature as subtractive ones work only if we have
260  //something to subtract from.
261  App::DocumentObject *prevSolid = pcActiveBody->Tip.getValue();
262  if(!prevSolid) {
263  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("No previous feature found"),
264  QObject::tr("It is not possible to create a subtractive feature without a base feature available"));
265  return;
266  }
267 
268  auto shapeType( primitiveIntToName(iMsg) );
269  auto FeatName( getUniqueObjectName(shapeType, pcActiveBody) );
270 
271  Gui::Command::openCommand( (std::string("Make subtractive ") + shapeType).c_str() );
272  FCMD_OBJ_CMD(pcActiveBody,"newObject('PartDesign::Subtractive"<<shapeType<<"','"<<FeatName<<"')");
273  Gui::Command::updateActive();
274 
275  auto Feat = pcActiveBody->getDocument()->getObject(FeatName.c_str());
276  copyVisual(Feat, "ShapeAppearance", prevSolid);
277  copyVisual(Feat, "LineColor", prevSolid);
278  copyVisual(Feat, "PointColor", prevSolid);
279  copyVisual(Feat, "Transparency", prevSolid);
280  copyVisual(Feat, "DisplayMode", prevSolid);
281 
282  if ( isActiveObjectValid() ) {
283  // TODO (2015-08-05, Fat-Zer)
284  FCMD_OBJ_HIDE(prevSolid);
285  }
286 
287  PartDesignGui::setEdit(Feat,pcActiveBody);
288 }
