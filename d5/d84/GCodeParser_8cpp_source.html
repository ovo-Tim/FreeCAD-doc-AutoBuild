100 const char* GCodeParser::ParseFloat(const char* ptr, float* retFloat)
101 {
102  float decPos = 10;
103  float sign = 1;
104  bool decimalPointFound = false;
105  float res = 0;
106  while (*ptr != 0) {
107  char letter = toupper(*ptr);
108  ptr++;
109 
110  if (letter == ' ') {
111  continue;
112  }
113 
114  if (letter == '-') {
115  sign = -1;
116  }
117  else if (letter == '.') {
118  decimalPointFound = true;
119  }
120  else if (letter >= '0' && letter <= '9') {
121  float digitVal = (float)(letter - '0');
122  if (decimalPointFound) {
123  res = res + digitVal / decPos;
124  decPos *= 10;
125  }
126  else {
127  res = res * 10 + digitVal;
128  }
129  }
130  else {
131  ptr--;
132  break;
133  }
134  }
135  *retFloat = res * sign;
136  return ptr;
137 }

140 {
141  GCToken token;

145  while (*ptr != 0 && !exitLoop) {
