215 def parse(pathobj):
216  out = ""
217  lastcommand = None
218 
219  params = [
220  "X",
221  "Y",
222  "Z",
223  "A",
224  "B",
225  "I",
226  "J",

236  if OUTPUT_COMMENTS:
237  out += linenumber() + "(compound: " + pathobj.Label + ")\n"
238  for p in pathobj.Group:
239  out += parse(p)
240  return out
241  else: # parsing simple path
242 
243  if not hasattr(pathobj, "Path"): # groups might contain non-path things like stock.
244  return out
245 
246  if OUTPUT_COMMENTS:
247  out += linenumber() + "(Path: " + pathobj.Label + ")\n"
248 
249  for c in PathUtils.getPathWithPlacement(pathobj).Commands:
250  outstring = []
251  command = c.Name
252 
253  # fablin does not support parenthesis syntax, so removing that (pocket) in the agnostic gcode
254  if command[0] == "(":
255  if not OUTPUT_COMMENTS:
256  pass
257  else:
258  outstring.append(command)
259 
260  # if modal: only print the command if it is not the same as the last one
261  if MODAL is True:
262  if command == lastcommand:
263  outstring.pop(0)
264 
265  # Now add the remaining parameters in order
266  for param in params:
267  if param in c.Parameters:
268  if param == "F":
269  if command not in RAPID_MOVES:
270  outstring.append(param + format(c.Parameters["F"], ".2f"))
271  elif param == "T":
272  outstring.append(param + str(c.Parameters["T"]))
273  else:
274  outstring.append(param + format(c.Parameters[param], ".4f"))
275 
276  if command in RAPID_MOVES and command != lastcommand:
277  outstring.append("F" + format(RAPID_FEEDRATE))
278 
279  # store the latest command
280  lastcommand = command
281 
282  # Check for Tool Change:
283  if command == "M6":
284  if OUTPUT_COMMENTS:
285  out += linenumber() + "(begin toolchange)\n"
286  if not OUTPUT_TOOL_CHANGE:
287  outstring.insert(0, ";")
288  else:
289  for line in TOOL_CHANGE.splitlines(True):
290  out += linenumber() + line
291 
292  if command == "message":
293  if OUTPUT_COMMENTS is False:
294  out = []
295  else:
296  outstring.pop(0) # remove the command
297 
298  if command in SUPPRESS_COMMANDS:
299  outstring = []
300 
301  # prepend a line number and append a newline
302  if len(outstring) >= 1:
303  if OUTPUT_LINE_NUMBERS:
304  outstring.insert(0, (linenumber()))
305 
306  # append the line to the final output
307  for w in outstring:
308  out += w + COMMAND_SPACE
309  out = out.strip() + "\n"
310 
311  return out
