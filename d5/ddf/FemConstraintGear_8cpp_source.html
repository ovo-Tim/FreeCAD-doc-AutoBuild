61 void ConstraintGear::onChanged(const App::Property* prop)
62 {
63  ConstraintBearing::onChanged(prop);
64 
65  if (prop == &Direction) {
66  Base::Vector3d direction = getDirection(Direction);
67  if (direction.Length() < Precision::Confusion()) {
68  return;
69  }
70  naturalDirectionVector = direction;
71  if (Reversed.getValue()) {
72  direction = -direction;
73  }
74  DirectionVector.setValue(direction);
75  DirectionVector.touch();
76  }
77  else if (prop == &Reversed) {
78  if (Reversed.getValue() && (DirectionVector.getValue() == naturalDirectionVector)) {
79  DirectionVector.setValue(-naturalDirectionVector);
80  DirectionVector.touch();
81  }
82  else if (!Reversed.getValue() && (DirectionVector.getValue() != naturalDirectionVector)) {
83  DirectionVector.setValue(naturalDirectionVector);
84  DirectionVector.touch();
85  }
86  }
87  // The computation for the force angle is simpler in the ViewProvider directly
