214 QIcon DrawGuiUtil::iconForLine(size_t lineNumber,
215  TechDraw::LineGenerator* generator)
216 {
217  // Base::Console().Message("DGU::iconForLine(lineNumber: %d)\n", lineNumber);
218  constexpr int iconSize {64};
219  constexpr int borderSize {4};
220  constexpr double iconLineWeight {1.0};
221  size_t lineCount {4};
222  double maxLineLength = iconSize - borderSize * 2.0;
223  QBitmap bitmap {iconSize, iconSize};
224  bitmap.clear();
225 
226  QPainter painter(&bitmap);
227  QPen linePen = generator->getLinePen(lineNumber, iconLineWeight);
228  linePen.setDashOffset(0.0);
229  linePen.setCapStyle(Qt::FlatCap);
230  linePen.setColor(Qt::color1);
231 
232  QSize lineIconSize(iconSize, iconSize);
233 
234  auto curStyleSheet =
235  App::GetApplication()
236  .GetParameterGroupByPath("User parameter:BaseApp/Preferences/MainWindow")
237  ->GetASCII("StyleSheet", "None");
238  QColor textColor{Qt::black};
239  if (isStyleSheetDark(curStyleSheet)) {
240  textColor = Preferences::lightTextColor().asValue<QColor>();
241  }
242 
243  // handle simple case of continuous line
244  if (linePen.style() == Qt::SolidLine) {
245  linePen.setWidthF(iconLineWeight * lineCount);
246  painter.setPen(linePen);
247  painter.drawLine(borderSize, iconSize / 2, iconSize - borderSize, iconSize / 2);
248  if (isStyleSheetDark(curStyleSheet)) {
249  QIcon lineItemIcon(bitmap);
250  return QIcon(maskBlackPixels(lineItemIcon, lineIconSize, textColor));
251  }
252  else {
253  return QIcon(bitmap);
254  }
255  }
256 
257  // dashed line
258  linePen.setWidthF(iconLineWeight);
259  painter.setPen(linePen);
260  double yHeight = (iconSize / 2) - (lineCount * iconLineWeight);
261  size_t iLine = 0;
262  // draw multiple lines to stretch the line vertically without horizontal
263  // distortion
264  for (; iLine < lineCount; iLine++) {
265  painter.drawLine(borderSize, yHeight, maxLineLength, yHeight);
266  yHeight += iconLineWeight;
267  }
268  if (isStyleSheetDark(curStyleSheet)) {
269  QIcon lineItemIcon(bitmap);
270  return QIcon(maskBlackPixels(lineItemIcon, lineIconSize, textColor));
271  }
272  else {
273  return QIcon(bitmap);
274  }
275 }

282 // find a page in Selection, Document or CurrentWindow.
283 TechDraw::DrawPage* DrawGuiUtil::findPage(Gui::Command* cmd, bool findAny)
284 {
285  // Base::Console().Message("DGU::findPage()\n");
286  std::vector<std::string> names;
287  std::vector<std::string> labels;
288  auto docs = App::GetApplication().getDocuments();
289 
290  if (findAny) {
291  // find a page in any open document
292  std::vector<App::DocumentObject*> foundPageObjects;
293  // no page found in the usual places, but we have been asked to search all
294  // open documents for a page.
295  auto docsAll = App::GetApplication().getDocuments();
296  for (auto& doc : docsAll) {
297  auto docPages = doc->getObjectsOfType(TechDraw::DrawPage::getClassTypeId());
298  if (docPages.empty()) {
299  // this open document has no TD pages
300  continue;
301  }
302  foundPageObjects.insert(foundPageObjects.end(), docPages.begin(), docPages.end());
303  }
304  if (foundPageObjects.empty()) {
305  QMessageBox::warning(Gui::getMainWindow(),
306  QObject::tr("No page found"),
307  QObject::tr("No Drawing Pages available."));
308  return nullptr;
309  }
310  else if (foundPageObjects.size() > 1) {
311  // multiple pages available, ask for help
312  for (auto obj : foundPageObjects) {
313  std::string name = obj->getNameInDocument();
314  names.push_back(name);
315  std::string label = obj->Label.getValue();
316  labels.push_back(label);
317  }
318  DlgPageChooser dlg(labels, names, Gui::getMainWindow());
319  if (dlg.exec() == QDialog::Accepted) {
320  std::string selName = dlg.getSelection();
321  App::Document* doc = cmd->getDocument();
322  return static_cast<TechDraw::DrawPage*>(doc->getObject(selName.c_str()));
323  }
324  }
325  else {
326  // only 1 page found
327  return static_cast<TechDraw::DrawPage*>(foundPageObjects.front());
328  }
329  }
330 
331  // check Selection for a page
332  std::vector<App::DocumentObject*> selPages =
333  cmd->getSelection().getObjectsOfType(TechDraw::DrawPage::getClassTypeId());
334  if (selPages.empty()) {
335  // no page in selection, try this document
336  auto docPages = cmd->getDocument()->getObjectsOfType(TechDraw::DrawPage::getClassTypeId());
337  if (docPages.empty()) {
338  // we are only to look in this document, and there is no page in this document
339  QMessageBox::warning(Gui::getMainWindow(),
340  QObject::tr("No page found"),
341  QObject::tr("No Drawing Pages in document."));
342  return nullptr;
343  }
344  else if (docPages.size() > 1) {
345  // multiple pages in document, use active page if there is one
346  Gui::MainWindow* w = Gui::getMainWindow();
347  Gui::MDIView* mv = w->activeWindow();
348  MDIViewPage* mvp = dynamic_cast<MDIViewPage*>(mv);
349  if (mvp) {
350  QGSPage* qp = mvp->getViewProviderPage()->getQGSPage();
351  return qp->getDrawPage();
352  }
353  else {
354  // none of pages in document is active, ask for help
355  for (auto obj : docPages) {
356  std::string name = obj->getNameInDocument();
357  names.push_back(name);
358  std::string label = obj->Label.getValue();
359  labels.push_back(label);
360  }
361  DlgPageChooser dlg(labels, names, Gui::getMainWindow());
362  if (dlg.exec() == QDialog::Accepted) {
363  std::string selName = dlg.getSelection();
364  App::Document* doc = cmd->getDocument();
365  return static_cast<TechDraw::DrawPage*>(doc->getObject(selName.c_str()));
366  }
367  return nullptr;
368  }
369  }
370  else {
371  // only 1 page in document - use it
372  return static_cast<TechDraw::DrawPage*>(docPages.front());
373  }
374  }
375  else if (selPages.size() > 1) {
376  // multiple pages in selection
377  for (auto obj : selPages) {
378  std::string name = obj->getNameInDocument();
379  names.push_back(name);
380  std::string label = obj->Label.getValue();
381  labels.push_back(label);
382  }
383  DlgPageChooser dlg(labels, names, Gui::getMainWindow());
384  if (dlg.exec() == QDialog::Accepted) {
385  std::string selName = dlg.getSelection();
386  App::Document* doc = cmd->getDocument();
387  return static_cast<TechDraw::DrawPage*>(doc->getObject(selName.c_str()));
388  }
389  }
390  else {
391  // exactly 1 page in selection, use it
392  return static_cast<TechDraw::DrawPage*>(selPages.front());
393  }
394 
395  // we can not actually reach this point.
396  return nullptr;
397 }

399 bool DrawGuiUtil::isDraftObject(App::DocumentObject* obj)
400 {
401  bool result = false;
402  App::PropertyPythonObject* proxy =
403  dynamic_cast<App::PropertyPythonObject*>(obj->getPropertyByName("Proxy"));
404 
405  if (proxy) {
406  // if no proxy, can not be Draft obj
407  // if has proxy, might be Draft obj
408  std::stringstream ss;
409  Py::Object proxyObj = proxy->getValue();
410  Base::PyGILStateLocker lock;
411  try {
412  if (proxyObj.hasAttr("__module__")) {
413  Py::String mod(proxyObj.getAttr("__module__"));
414  ss << (std::string)mod;
415  if (ss.str().find("Draft") != std::string::npos) {
416  result = true;
417  }
418  else if (ss.str().find("draft") != std::string::npos) {
419  result = true;
420  }
421  }
422  }
423  catch (Py::Exception&) {
424  Base::PyException e; // extract the Python error text
425  e.ReportException();
426  result = false;
427  }
428  }
429  return result;
430 }

567 std::pair<Base::Vector3d, Base::Vector3d> DrawGuiUtil::get3DDirAndRot()
568 {
569  std::pair<Base::Vector3d, Base::Vector3d> result;
570  Base::Vector3d viewDir(0.0, -1.0, 0.0); // default to front
571  Base::Vector3d viewUp(0.0, 0.0, 1.0); // default to top
572  Base::Vector3d viewRight(1.0, 0.0, 0.0); // default to right
573  std::list<Gui::MDIView*> mdis = Gui::Application::Instance->activeDocument()->getMDIViews();
574  Gui::View3DInventor* view;
575  Gui::View3DInventorViewer* viewer = nullptr;
576  for (auto& m : mdis) { // find the 3D viewer
577  view = dynamic_cast<Gui::View3DInventor*>(m);
578  if (view) {
579  viewer = view->getViewer();
580  break;
581  }
582  }
583  if (!viewer) {
584  return std::make_pair(viewDir, viewRight);
585  }
586 
587  // Coin is giving us a values like 0.000000134439 instead of 0.000000000000.
588  // This small difference caused circles to be projected as ellipses among other
589  // problems.
590  // Since SbVec3f is single precision floating point, it is only good to 6-9
591  // significant decimal digits, and the rest of TechDraw works with doubles
592  // that are good to 15-18 significant decimal digits.
593  // But. When a float is promoted to double the value is supposed to be unchanged!
594  // So where do the garbage digits come from???
595  // In any case, if we restrict directions to 6 digits, we avoid the problem.
596  int digits(6);
597  SbVec3f dvec = viewer->getViewDirection();
598  double dvecX = roundToDigits(dvec[0], digits);
599  double dvecY = roundToDigits(dvec[1], digits);
600  double dvecZ = roundToDigits(dvec[2], digits);
601  viewDir = Base::Vector3d(dvecX, dvecY, dvecZ);
602  viewDir = viewDir * (-1.0); // Inventor dir is opposite TD projection dir
603 
604  SbVec3f upvec = viewer->getUpDirection();
605  double upvecX = roundToDigits(upvec[0], digits);
606  double upvecY = roundToDigits(upvec[1], digits);
607  double upvecZ = roundToDigits(upvec[2], digits);
608  viewUp = Base::Vector3d(upvecX, upvecY, upvecZ);
609 
610  Base::Vector3d right = viewUp.Cross(viewDir);
611 
612  result = std::make_pair(viewDir, right);
613  return result;
614 }

616 std::pair<Base::Vector3d, Base::Vector3d> DrawGuiUtil::getProjDirFromFace(App::DocumentObject* obj,
617  std::string faceName)
618 {
619  std::pair<Base::Vector3d, Base::Vector3d> d3Dirs = get3DDirAndRot();
620  std::pair<Base::Vector3d, Base::Vector3d> dirs;
621  dirs.first = Base::Vector3d(0.0, 0.0, 1.0); // set a default
622  dirs.second = Base::Vector3d(1.0, 0.0, 0.0);
623  Base::Vector3d projDir, rotVec;
624  projDir = d3Dirs.first;
625  rotVec = d3Dirs.second;
626 
627  auto ts = Part::Feature::getShape(obj, faceName.c_str(), true);
628  if (ts.IsNull() || ts.ShapeType() != TopAbs_FACE) {
629  Base::Console().Warning("getProjDirFromFace(%s) is not a Face\n", faceName.c_str());
630  return dirs;
631  }
632 
633  const TopoDS_Face& face = TopoDS::Face(ts);
634  TopAbs_Orientation orient = face.Orientation();
635  BRepAdaptor_Surface adapt(face);
636 
637  double u1 = adapt.FirstUParameter();
