40 def move(objectslist, vector, copy=False):
41  """move(objects,vector,[copy])
42 
43  Move the objects contained in objects (that can be an object or a
44  list of objects) in the direction and distance indicated by the given
45  vector.
46 
47  Parameters
48  ----------
49  objectslist : list
50 
51  vector : Base.Vector
52  Delta Vector to move the clone from the original position.
53 
54  copy : bool
55  If copy is True, the actual objects are not moved, but copies
56  are created instead.
57 
58  Return
59  ----------
60  The objects (or their copies) are returned.
61  """
62  utils.type_check([(vector, App.Vector), (copy,bool)], "move")
63  if not isinstance(objectslist, list):
64  objectslist = [objectslist]
65 
66  objectslist.extend(groups.get_movable_children(objectslist))
67  newobjlist = []
68  newgroups = {}
69  objectslist = utils.filter_objects_for_modifiers(objectslist, copy)
70 
71  if copy:
72  doc = App.ActiveDocument
73  for obj in objectslist:
74  if obj.isDerivedFrom("App::DocumentObjectGroup") \
75  and obj.Name not in newgroups:
76  newgroups[obj.Name] = doc.addObject(obj.TypeId,
77  utils.get_real_name(obj.Name))
78 
79  for obj in objectslist:
80  newobj = None
81 
82  # real_vector have been introduced to take into account
83  # the possibility that object is inside an App::Part
84  # TODO: Make Move work also with App::Link
85  if hasattr(obj, "getGlobalPlacement"):
86  v_minus_global = obj.getGlobalPlacement().inverse().Rotation.multVec(vector)
87  real_vector = obj.Placement.Rotation.multVec(v_minus_global)
88  else:
89  real_vector = vector
90 
91  if obj.isDerivedFrom("App::DocumentObjectGroup"):
92  if copy:
93  newobj = newgroups[obj.Name]
94  else:
95  newobj = obj
96 
97  elif hasattr(obj, "Shape"):
98  if copy:
99  newobj = make_copy.make_copy(obj)
100  else:
101  newobj = obj
102  pla = newobj.Placement
103  pla.move(real_vector)
104 
105  elif obj.isDerivedFrom("App::Annotation"):
106  if copy:
107  newobj = make_copy.make_copy(obj)
108  else:
109  newobj = obj
110  newobj.Position = obj.Position.add(real_vector)
111 
112  elif utils.get_type(obj) in ["Text", "DraftText"]:
113  if copy:
114  newobj = make_copy.make_copy(obj)
115  else:
116  newobj = obj
117  newobj.Placement.Base = obj.Placement.Base.add(real_vector)
118 
119  elif utils.get_type(obj) in ["Dimension", "LinearDimension"]:
120  if copy:
121  newobj = make_copy.make_copy(obj)
122  else:
123  newobj = obj
124  newobj.Start = obj.Start.add(real_vector)
125  newobj.End = obj.End.add(real_vector)
126  newobj.Dimline = obj.Dimline.add(real_vector)
127 
128  elif utils.get_type(obj) == "AngularDimension":
129  if copy:
130  newobj = make_copy.make_copy(obj)
131  else:
132  newobj = obj
133  newobj.Center = obj.Center.add(real_vector)
134  newobj.Dimline = obj.Dimline.add(real_vector)
135 
136  elif hasattr(obj, "Placement"):
137  if copy:
138  newobj = make_copy.make_copy(obj)
139  else:
140  newobj = obj
141  pla = newobj.Placement
142  pla.move(real_vector)
143 
144  if newobj is not None:
145  newobjlist.append(newobj)
146  if copy:
147  for parent in obj.InList:
148  if parent.isDerivedFrom("App::DocumentObjectGroup") \
149  and (parent in objectslist):
150  newgroups[parent.Name].addObject(newobj)
151  if utils.get_type(parent) == "Layer":
152  parent.Proxy.addObject(parent ,newobj)
153 
154  if copy and params.get_param("selectBaseObjects"):
155  gui_utils.select(objectslist)
156  else:
157  gui_utils.select(newobjlist)
158 
159  if len(newobjlist) == 1:
160  return newobjlist[0]
161  return newobjlist

202 def copy_moved_edge(object, edge_index, vector):
203  """
204  Needed for SubObjects modifiers.

206  """
