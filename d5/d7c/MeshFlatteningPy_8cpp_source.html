55 const TopoDS_Face& getTopoDSFace(py::object* face)
56 {
57  if (PyObject_TypeCheck(face->ptr(), &(Part::TopoShapeFacePy::Type)))
58  {
59  const Part::TopoShapeFacePy* f = static_cast<Part::TopoShapeFacePy*>(face->ptr());
60  const TopoDS_Face& myFace = TopoDS::Face(f->getTopoShapePtr()->getShape());
61  return myFace;
62  }
63  else
64  throw std::invalid_argument("must be a face");
65 }

67 const TopoDS_Edge& getTopoDSEdge(py::object* edge)
68 {
69  if (PyObject_TypeCheck(edge->ptr(), &(Part::TopoShapeEdgePy::Type)))
70  {
71  const Part::TopoShapeEdgePy* e = static_cast<Part::TopoShapeEdgePy*>(edge->ptr());
72  const TopoDS_Edge& myEdge = TopoDS::Edge(e->getTopoShapePtr()->getShape());
73  return myEdge;
74  }
75  else
76  throw std::invalid_argument("must be an edge");
