119  """Execute when clicking the OK button or pressing Enter key.
120 
121  It changes the slope of the line that lies on the XY plane.
122 
123  TODO: make it work also with lines lying on the YZ and XZ planes.
124  """
125  if hasattr(self, "spinbox"):
126  pc = self.spinbox.value()
127  self.doc.openTransaction("Change slope")
128  for obj in Gui.Selection.getSelection():
129  if utils.get_type(obj) == "Wire":
130  if len(obj.Points) > 1:
131  lp = None
132  np = []
133  for p in obj.Points:
134  if not lp:
135  lp = p
136  else:
137  v = p.sub(lp)
138  z = pc * App.Vector(v.x, v.y, 0).Length
139  lp = App.Vector(p.x, p.y, lp.z + z)
140  np.append(lp)
141  obj.Points = np
