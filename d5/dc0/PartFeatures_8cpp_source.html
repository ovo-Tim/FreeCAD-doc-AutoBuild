89 App::DocumentObjectExecReturn* RuledSurface::getShape(const App::PropertyLinkSub& link,
90  TopoDS_Shape& shape) const
91 {
92  App::DocumentObject* obj = link.getValue();
93  const Part::TopoShape part = Part::Feature::getTopoShape(obj);
94  if (part.isNull()) {
95  return new App::DocumentObjectExecReturn("No shape linked.");
96  }
97 
98  // if no explicit sub-shape is selected use the whole part
99  const std::vector<std::string>& element = link.getSubValues();
100  if (element.empty()) {
101  shape = part.getShape();
102  return nullptr;
103  }
104  else if (element.size() != 1) {
105  return new App::DocumentObjectExecReturn("Not exactly one sub-shape linked.");
106  }
107 
108  if (!part.getShape().IsNull()) {
109  if (!element[0].empty()) {
110  // shape = Part::Feature::getTopoShape(obj, element[0].c_str(), true /*need
111  // element*/).getShape();
112  shape = part.getSubShape(element[0].c_str());
113  }
114  else {
115  // the sub-element is an empty string, so use the whole part
116  shape = part.getShape();
117  }
118  }
119 
120  return nullptr;
121 }

123 App::DocumentObjectExecReturn* RuledSurface::execute()
124 {
125  try {
126  std::vector<TopoShape> shapes;
127  std::array<App::PropertyLinkSub*, 2> links = {&Curve1, &Curve2};
128  for (auto link : links) {
129  App::DocumentObject* obj = link->getValue();
130  const Part::TopoShape part = Part::Feature::getTopoShape(obj);
131  if (part.isNull()) {
132  return new App::DocumentObjectExecReturn("No shape linked.");
133  }
134  // if no explicit sub-shape is selected use the whole part
135  const auto& subs = link->getSubValues();
136  if (subs.empty()) {
137  shapes.push_back(part);
138  }
139  else if (subs.size() != 1) {
140  return new App::DocumentObjectExecReturn("Not exactly one sub-shape linked.");
141  }
142  else {
143  shapes.push_back(getTopoShape(link->getValue(), subs.front().c_str(), true));
144  }
145  if (shapes.back().isNull()) {
146  return new App::DocumentObjectExecReturn("Invalid link.");
147  }
148  }
149  TopoShape res(0);
150  res.makeElementRuledSurface(shapes, Orientation.getValue());
151  this->Shape.setValue(res);
152  return Part::Feature::execute();
153 
154  }
155  catch (Standard_Failure& e) {
156 
157  return new App::DocumentObjectExecReturn(e.GetMessageString());
158  }
159  catch (...) {
160  return new App::DocumentObjectExecReturn("General error in RuledSurface::execute()");
161  }
162 }

210 App::DocumentObjectExecReturn* Loft::execute()
211 {
212  if (Sections.getSize() == 0) {
213  return new App::DocumentObjectExecReturn("No sections linked.");
214  }
215 
216  try {
217  std::vector<TopoShape> shapes;
218  for (auto& obj : Sections.getValues()) {
219  shapes.emplace_back(getTopoShape(obj));
220  if (shapes.back().isNull()) {
221  return new App::DocumentObjectExecReturn("Invalid section link");
222  }
223  }
224  IsSolid isSolid = Solid.getValue() ? IsSolid::solid : IsSolid::notSolid;
225  IsRuled isRuled = Ruled.getValue() ? IsRuled::ruled : IsRuled::notRuled;
226  IsClosed isClosed = Closed.getValue() ? IsClosed::closed : IsClosed::notClosed;
227  int degMax = MaxDegree.getValue();
228  TopoShape result(0);
229  result.makeElementLoft(shapes, isSolid, isRuled, isClosed, degMax);
230  if (Linearize.getValue()) {
231  result.linearize( LinearizeFace::linearizeFaces, LinearizeEdge::noEdges);
232  }
233  this->Shape.setValue(result);
234  return Part::Feature::execute();
235  }
236  catch (Standard_Failure& e) {
237 
238  return new App::DocumentObjectExecReturn(e.GetMessageString());
239  }

295 App::DocumentObjectExecReturn* Sweep::execute()
296 {
297  if (Sections.getSize() == 0) {
298  return new App::DocumentObjectExecReturn("No sections linked.");
299  }
300  if (!Spine.getValue()) {
301  return new App::DocumentObjectExecReturn("No spine");
302  }
303  TopoShape spine = getTopoShape(Spine.getValue());
304  const auto& subs = Spine.getSubValues();
305  if (spine.isNull()) {
306  return new App::DocumentObjectExecReturn("Invalid spine");
307  }
308  if (subs.size()) {
309  std::vector<TopoShape> spineShapes;
310  for (auto sub : subs) {
311  auto shape = spine.getSubTopoShape(sub.c_str());
312  if (shape.isNull()) {
313  return new App::DocumentObjectExecReturn("Invalid spine");
314  }
315  spineShapes.push_back(shape);
316  }
317  spine = TopoShape().makeElementCompound(spineShapes, 0, TopoShape::SingleShapeCompoundCreationPolicy::returnShape);
318  }
319  std::vector<TopoShape> shapes;
320  shapes.push_back(spine);
321  for (auto& obj : Sections.getValues()) {
322  shapes.emplace_back(getTopoShape(obj));
323  if (shapes.back().isNull()) {
324  return new App::DocumentObjectExecReturn("Invalid section link");
325  }
326  }
327  MakeSolid isSolid = Solid.getValue() ? MakeSolid::makeSolid : MakeSolid::noSolid;
328  Standard_Boolean isFrenet = Frenet.getValue() ? Standard_True : Standard_False;
329  auto transMode = static_cast<TransitionMode>(Transition.getValue());
330  try {
331  TopoShape result(0);
332  result.makeElementPipeShell(shapes, isSolid, isFrenet, transMode, Part::OpCodes::Sweep);
333  if (Linearize.getValue()) {
334  result.linearize(LinearizeFace::linearizeFaces, LinearizeEdge::noEdges);
335  }
336  this->Shape.setValue(result);
337  return App::DocumentObject::StdReturn;
338  }
339  catch (Standard_Failure& e) {
340 
341  return new App::DocumentObjectExecReturn(e.GetMessageString());
342  }
343  catch (...) {
344  return new App::DocumentObjectExecReturn("A fatal error occurred when making the sweep");
345  }
346 }

415 App::DocumentObjectExecReturn* Thickness::execute()
416 {
417  std::vector<TopoShape> shapes;
418  auto base = getTopoShape(Faces.getValue());
419  if (base.isNull()) {
420  return new App::DocumentObjectExecReturn("Invalid source shape");
421  }
422  if (base.countSubShapes(TopAbs_SOLID) != 1) {
423  return new App::DocumentObjectExecReturn("Source shape is not single solid.");
424  }
425  for (auto& sub : Faces.getSubValues(true)) {
426  shapes.push_back(base.getSubTopoShape(sub.c_str()));
427  if (shapes.back().getShape().ShapeType() != TopAbs_FACE) {
428  return new App::DocumentObjectExecReturn("Invalid face selection");
429  }
430  }
431  double thickness = Value.getValue();
432  double tol = Precision::Confusion();
433  bool inter = Intersection.getValue();
434  bool self = SelfIntersection.getValue();
435  short mode = (short)Mode.getValue();
436  short join = (short)Join.getValue();
437 
438  this->Shape.setValue(TopoShape(0,getDocument()->getStringHasher())
439  .makeElementThickSolid(base,
440  shapes,
441  thickness,
442  tol,
443  inter,
444  self,
445  mode,
446  static_cast<JoinType>(join)));
447  return Part::Feature::execute();
448 }
