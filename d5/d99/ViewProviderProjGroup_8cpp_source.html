93 bool ViewProviderProjGroup::onDelete(const std::vector<std::string> & parms)
94 {
95  Q_UNUSED(parms)
96  // warn the user if the ProjGroup is not empty
97 
98  QString bodyMessage;
99  QTextStream bodyMessageStream(&bodyMessage);
100  TechDraw::DrawProjGroupItem* Item = nullptr;
101  std::vector<std::string> ViewList;
102 
103  // get the items in the group
104  auto objs = claimChildren();
105 
106  // iterate over all item to check which ones have a section or detail view
107  for (auto ObjectIterator : objs) {
108  // get item
109  Item = static_cast<TechDraw::DrawProjGroupItem*>(ObjectIterator);
110  // get its section views
111  auto viewSection = Item->getSectionRefs();
112  // add names to a list
113  if (!viewSection.empty()) {
114  for (auto SecIterator : viewSection) {
115  ViewList.emplace_back(SecIterator->Label.getValue());
116  }
117  }
118  // get its detail views
119  auto viewDetail = Item->getDetailRefs();
120  if (!viewDetail.empty()) {
121  for (auto DetIterator : viewDetail) {
122  ViewList.emplace_back(DetIterator->Label.getValue());
123  }
124  }
125  // get its leader lines
126  auto viewLead = Item->getLeaders();
127  if (!viewLead.empty()) {
128  for (auto LeadIterator : viewLead) {
129  ViewList.emplace_back(LeadIterator->Label.getValue());
130  }
131  }
132  }
133 
134  // if there are section or detail views we cannot delete because this would break them
135  if (!ViewList.empty()) {
136  bodyMessageStream << qApp->translate("Std_Delete",
137  "The group cannot be deleted because its items have the following\nsection or detail views, or leader lines that would get broken:");
138  bodyMessageStream << '\n';
139  for (const auto& ListIterator : ViewList) {
140  bodyMessageStream << '\n' << QString::fromUtf8(ListIterator.c_str());
141  }
142  QMessageBox::warning(Gui::getMainWindow(),
143  qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
144  QMessageBox::Ok);
145  return false;
146  }
147 
148  if (!objs.empty())
149  {
150  // generate dialog
151  bodyMessageStream << qApp->translate("Std_Delete",
152  "The projection group is not empty, therefore\nthe following referencing objects might be lost:");
153  bodyMessageStream << '\n';
154  for (auto ObjIterator : objs) {
155  bodyMessageStream << '\n' << QString::fromUtf8(ObjIterator->Label.getValue());
156  }
157  bodyMessageStream << "\n\n" << QObject::tr("Are you sure you want to continue?");
158  // show and evaluate dialog
159  int DialogResult = QMessageBox::warning(Gui::getMainWindow(),
160  qApp->translate("Std_Delete", "Object dependencies"), bodyMessage,
161  QMessageBox::Yes, QMessageBox::No);
162  return (DialogResult == QMessageBox::Yes);
163  }
164  return true;
165 }

176 std::vector<App::DocumentObject*> ViewProviderProjGroup::claimChildren() const
177 {
178  // Collect any child Document Objects and put them in the right place in the Feature tree
179  // valid children of an ProjGroup are:
180  // - Balloons
181  // - Leaders
182  // - RichAnno
183  std::vector<App::DocumentObject*> temp;
184  const std::vector<App::DocumentObject*>& candidates = getViewObject()->getInList();
185  // DPGI's do not point at the DPG, the DPG/DVC maintains links to the items
186 
187  // why does this need a try/catch??
188  try {
189  for (auto& obj : candidates) {
190  if (obj->isDerivedFrom<TechDraw::DrawViewBalloon>() ||
191  obj->isDerivedFrom<TechDraw::DrawLeaderLine>() ||
192  obj->isDerivedFrom<TechDraw::DrawRichAnno>()) {
193  temp.push_back(obj);
194  }
195  }
196 // return temp;
197  }
198  catch (...) {
199  return {};
200  }
201 
202  // plus the individual ProjGroupItems
203  for (auto& view : getViewObject()->Views.getValues()) {
204  temp.push_back(view);
205  }
206 
207  return temp;
208 }
