86 TopoShape FeatureAddSub::refineShapeIfActive(const TopoShape& oldShape, const RefineErrorPolicy onError) const
87 {
88  if (this->Refine.getValue()) {
89  TopoShape shape(oldShape);
90  // this->fixShape(shape); // Todo: Not clear that this is required
91  try{

94  catch (Standard_Failure& err) {
95  if(onError == RefineErrorPolicy::Warn){
96  Base::Console().Warning((std::string("Refine failed: ") + err.GetMessageString()).c_str());
