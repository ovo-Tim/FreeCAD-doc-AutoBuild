59 void StdCmdDlgMacroRecord::activated(int iMsg)
60 {
61  Q_UNUSED(iMsg);
62  if (!getGuiApplication()->macroManager()->isOpen()){
63  Gui::Dialog::DlgMacroRecordImp cDlg(getMainWindow());
64  if (cDlg.exec() && getAction()) {
65  getAction()->setIcon(Gui::BitmapFactory().iconFromTheme("media-playback-stop"));
66  getAction()->setText(QCoreApplication::translate("StdCmdDlgMacroRecord", "S&top macro recording"));
67  getAction()->setToolTip(QCoreApplication::translate("StdCmdDlgMacroRecord", "Stop the macro recording session"));
68  }
69  }
70  else {
71  getGuiApplication()->macroManager()->commit();
72  if (getAction()) {
73  getAction()->setIcon(Gui::BitmapFactory().iconFromTheme("media-record"));
74  getAction()->setText(QString::fromLatin1(sMenuText));
75  getAction()->setToolTip(QString::fromLatin1(sToolTipText));
76  }
77  }
78 }
