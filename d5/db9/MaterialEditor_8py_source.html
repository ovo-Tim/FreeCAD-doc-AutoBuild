571  def openfile(self):
572  if self.verifyMaterial():
573  """
574  Save any unchanged data
575  """
576  self.edited = False

579 
580  "Opens a FCMat file"
581  if self.category == "Solid":
582  directory = self.directory + os.sep + "StandardMaterial"
583  else:
584  directory = self.directory + os.sep + "FluidMaterial"
585  if self.card_path is None or len(self.card_path) == 0:
586  self.card_path = directory
587  filetuple = QtGui.QFileDialog.getOpenFileName(
588  QtGui.QApplication.activeWindow(),
589  "Open FreeCAD Material file",
590  self.card_path,
591  "*.FCMat"
592  )
593  if os.path.isfile(filetuple[0]):
594  card_path_new = filetuple[0]
595  else:
596  FreeCAD.Console.PrintError("Error: Invalid path to the material file\n")
597  return
598  # we cannot simply execute findData(self.card_path) because e.g. on Windows
599  # the return path has "/" as folder separator, but the paths in the ComboMaterial
600  # have also some "\" in them. For example a path can look like this:
601  # D:/FreeCAD-build/data/Mod\Material\FluidMaterial\Air.FCMat
602  # To keep it simple, we take a path from the ComboMaterial and change only the
603  # material card filename
604  #
605  # Using the initialIndex variable won't work before a card os selected for the
606  # first time, so use index 1. Index 0 is a blank entry
607  if self.widget.ComboMaterial.count() > 1:
608  path = self.widget.ComboMaterial.itemData(1)
609  # at first check if we have a uniform usage
610  # (if a character is not present, rsplit delivers the initial string)
611  testBackslash = path.rsplit('\\', 1)[0]
612  testSlash = path.rsplit('/', 1)[0]
613  if testBackslash == path:
614  path = testBackslash.rsplit('/', 1)[0] + '/'
615  elif testSlash == path:
616  path = testSlash.rsplit('\\', 1)[0] + '\\'
617  # since we don't know if we have to deal with slash or backslash, take the
618  # longest result as path
619  else:
620  pathBackslash = path.rsplit('\\', 1)[0]
621  pathSlash = path.rsplit('/', 1)[0]
622  if len(pathBackslash) > len(pathSlash):
623  path = pathBackslash + '\\'
624  else:
625  path = pathSlash + '/'
626  # we know that card_path_new has uniformly either / or \ but not yet what
627  testBackslash = card_path_new.rsplit('\\', 1)[0]
628  if testBackslash == card_path_new:
629  self.card_path = path + card_path_new.rsplit('/', 1)[1]
630  else:
631  self.card_path = path + card_path_new.rsplit('\\', 1)[1]
632  index = self.widget.ComboMaterial.findData(self.card_path)
633 
634  # check if card_path is in known path, means it is in combo box already
635  # if not print message, and give some feedbach that the card parameter are loaded
636  if index == -1:
637  FreeCAD.Console.PrintMessage(
638  "Card path: {} not found in known cards.\n"
639  "The material parameter only are loaded.\n"
640  .format(self.card_path)
641  )
