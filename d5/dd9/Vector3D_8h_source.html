262 template<class float_type>
263 inline float_type Distance(const Vector3<float_type>& v1, const Vector3<float_type>& v2)
264 {
265  float_type x = v1.x - v2.x;
266  float_type y = v1.y - v2.y;
267  float_type z = v1.z - v2.z;
268  return static_cast<float_type>(sqrt((x * x) + (y * y) + (z * z)));
269 }
270 
272 template<class float_type>
273 inline float_type DistanceP2(const Vector3<float_type>& v1, const Vector3<float_type>& v2)
274 {
275  float_type x = v1.x - v2.x;
276  float_type y = v1.y - v2.y;
277  float_type z = v1.z - v2.z;
278  return x * x + y * y + z * z;
279 }
