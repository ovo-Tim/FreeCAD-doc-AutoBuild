564 class BarThread : public QThread
565 {
566 public:
567  explicit BarThread(unsigned long s) : steps(s)
568  {
569  }
570  ~BarThread() override = default;
571  void run() override
572  {
573  QMutex mutex;
574  QMutexLocker ml(&mutex);
575 
576  try
577  {
578  Base::SequencerLauncher seq("Starting progress bar in thread", steps);
579 
580  for (unsigned long i=0; i<this->steps;i++)
581  {
582  seq.next(true);
583  QWaitCondition().wait(&mutex, 5);
584  }
585  }
586  catch (...)
587  {
588  }
589 
590  this->deleteLater();
591  Base::Console().Message("Thread with %d steps finished\n",this->steps);
592  }
593 
594 private:
595  unsigned long steps;
596 };
