229  outerConeBottomRadius = 40 / 2
230  outerConeTopRadius = 15 / 2
231  innerConeBottomRadius = 25 / 2
232  innerConeTopRadius = 8 / 2
233  coneHeight = 20.0
234  padHeight = 10.0
235  # Frustum volumes
236  outerConeVolume = 1.0 / 3.0 * math.pi * coneHeight * (
237  outerConeBottomRadius ** 2 + outerConeTopRadius ** 2 +
238  outerConeBottomRadius * outerConeTopRadius )
239  innerConeVolume = 1.0 / 3.0 * math.pi * coneHeight * (
240  innerConeBottomRadius ** 2 + innerConeTopRadius ** 2 +
241  innerConeBottomRadius * innerConeTopRadius )
242  frustumVolume = outerConeVolume - innerConeVolume
243  topArea = math.pi * (outerConeTopRadius ** 2) - math.pi * (innerConeTopRadius ** 2)
244  bottomArea = math.pi * (outerConeBottomRadius ** 2) - math.pi * (innerConeBottomRadius ** 2)
245  padVolume = topArea * padHeight
246  self.assertEqual(len(cone.Shape.Faces), 4) # Bottom, Inner, Top, Outer
247  self.assertAlmostEqual(cone.Shape.Faces[2].Area, topArea)
248  self.assertAlmostEqual(cone.Shape.Faces[0].Area, bottomArea)
249  self.assertAlmostEqual(cone.Shape.Volume, frustumVolume)
250  self.assertAlmostEqual(pad.Shape.Volume, frustumVolume + padVolume) # contains volume of previous in Body
251  self.assertAlmostEqual(body.Shape.Volume, frustumVolume + padVolume) # Overall body volume matches
