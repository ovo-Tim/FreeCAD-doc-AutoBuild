107 void ViewProviderSections::highlightReferences(ShapeType type, const References& refs, bool on)
108 {
109  for (const auto& it : refs) {
110  Part::Feature* base = dynamic_cast<Part::Feature*>(it.first);
111  if (base) {
112  PartGui::ViewProviderPartExt* svp = dynamic_cast<PartGui::ViewProviderPartExt*>(
113  Gui::Application::Instance->getViewProvider(base));
114  if (svp) {
115  switch (type) {
116  case ViewProviderSections::Vertex:
117  if (on) {
118  std::vector<App::Color> colors;
119  TopTools_IndexedMapOfShape vMap;
120  TopExp::MapShapes(base->Shape.getValue(), TopAbs_VERTEX, vMap);
121  colors.resize(vMap.Extent(), svp->PointColor.getValue());
122 
123  for (const auto& jt : it.second) {
124  // check again that the index is in range because it's possible that
125  // the sub-names are invalid
126  std::size_t idx =
127  static_cast<std::size_t>(std::stoi(jt.substr(6)) - 1);
128  if (idx < colors.size()) {
129  colors[idx] = App::Color(1.0, 0.0, 1.0); // magenta
130  }
131  }
132 
133  svp->setHighlightedPoints(colors);
134  }
135  else {
136  svp->unsetHighlightedPoints();
137  }
138  break;
139  case ViewProviderSections::Edge:
140  if (on) {
141  std::vector<App::Color> colors;
142  TopTools_IndexedMapOfShape eMap;
143  TopExp::MapShapes(base->Shape.getValue(), TopAbs_EDGE, eMap);
144  colors.resize(eMap.Extent(), svp->LineColor.getValue());
145 
146  for (const auto& jt : it.second) {
147  std::size_t idx =
148  static_cast<std::size_t>(std::stoi(jt.substr(4)) - 1);
149  // check again that the index is in range because it's possible that
150  // the sub-names are invalid
151  if (idx < colors.size()) {
152  colors[idx] = App::Color(1.0, 0.0, 1.0); // magenta
153  }
154  }
155 
156  svp->setHighlightedEdges(colors);
157  }
158  else {
159  svp->unsetHighlightedEdges();
160  }
161  break;
162  case ViewProviderSections::Face:
163  if (on) {
164  std::vector<App::Material> materials;
165  TopTools_IndexedMapOfShape fMap;
166  TopExp::MapShapes(base->Shape.getValue(), TopAbs_FACE, fMap);
167  materials.resize(fMap.Extent(), svp->ShapeAppearance[0]);
168 
169  for (const auto& jt : it.second) {
170  std::size_t idx =
171  static_cast<std::size_t>(std::stoi(jt.substr(4)) - 1);
172  // check again that the index is in range because it's possible that
173  // the sub-names are invalid
174  if (idx < materials.size()) {
175  // magenta
176  materials[idx].diffuseColor = App::Color(1.0, 0.0, 1.0);
177  }
178  }
179 
180  svp->setHighlightedFaces(materials);
181  }
182  else {
183  svp->unsetHighlightedFaces();
184  }
185  break;
186  }
187  }
188  }
189  }
190 }

194 class SectionsPanel::ShapeSelection: public Gui::SelectionFilterGate
195 {
196 public:
197  ShapeSelection(SectionsPanel::SelectionMode& mode, Surface::Sections* editedObject)
198  : Gui::SelectionFilterGate(nullPointer())
199  , mode(mode)
200  , editedObject(editedObject)
201  {}
202  ~ShapeSelection() override
203  {
204  mode = SectionsPanel::None;
205  }
209  bool allow(App::Document*, App::DocumentObject* pObj, const char* sSubName) override
210  {
211  // don't allow references to itself
212  if (pObj == editedObject) {
213  return false;
214  }
215  if (!pObj->isDerivedFrom(Part::Feature::getClassTypeId())) {
216  return false;
217  }
218 
219  if (!sSubName || sSubName[0] == '\0') {
220  return false;
221  }
222 
223  switch (mode) {
224  case SectionsPanel::AppendEdge:
225  return allowEdge(true, pObj, sSubName);
226  case SectionsPanel::RemoveEdge:
227  return allowEdge(false, pObj, sSubName);
228  default:
229  return false;
230  }
231  }
232 
233 private:
234  bool allowEdge(bool appendEdges, App::DocumentObject* pObj, const char* sSubName)
235  {
236  std::string element(sSubName);
237  if (element.substr(0, 4) != "Edge") {
238  return false;
239  }
240 
241  auto links = editedObject->NSections.getSubListValues();
242  for (const auto& it : links) {
243  if (it.first == pObj) {
244  for (const auto& jt : it.second) {
245  if (jt == sSubName) {
246  return !appendEdges;
247  }
248  }
249  }
250  }
251 
252  return appendEdges;
253  }
254 
255 private:
256  SectionsPanel::SelectionMode& mode;
257  Surface::Sections* editedObject;
258 };

312 void SectionsPanel::setEditedObject(Surface::Sections* fea)
313 {
314  editedObject = fea;
315 
316  // get the section edges
317  auto objects = editedObject->NSections.getValues();
318  auto edges = editedObject->NSections.getSubValues();
319  auto count = objects.size();
320 
321  App::Document* doc = editedObject->getDocument();
322  for (std::size_t i = 0; i < count; i++) {
323  App::DocumentObject* obj = objects[i];
324  std::string edge = edges[i];
325 
326  QListWidgetItem* item = new QListWidgetItem(ui->listSections);
327  ui->listSections->addItem(item);
328 
329  QString text = QString::fromLatin1("%1.%2").arg(QString::fromUtf8(obj->Label.getValue()),
330  QString::fromStdString(edge));
331  item->setText(text);
332 
333  // The user data field of a list widget item
334  // is a list of five elementa:
335  // 1. document name
336  // 2. object name
337  // 3. sub-element name of the edge
338  QList<QVariant> data;
339  data << QByteArray(doc->getName());
340  data << QByteArray(obj->getNameInDocument());
341  data << QByteArray(edge.c_str());
342  item->setData(Qt::UserRole, data);
343  }
344 
345  // attach this document observer
346  attachDocument(Gui::Application::Instance->getDocument(doc));
347 }

465 void SectionsPanel::onSelectionChanged(const Gui::SelectionChanges& msg)
466 {
467  if (selectionMode == None) {
468  return;
469  }
470 
471  if (msg.Type == Gui::SelectionChanges::AddSelection) {
472  checkOpenCommand();
473  if (selectionMode == AppendEdge) {
474  QListWidgetItem* item = new QListWidgetItem(ui->listSections);
475  ui->listSections->addItem(item);
476 
477  Gui::SelectionObject sel(msg);
478  QString text = QString::fromLatin1("%1.%2").arg(
479  QString::fromUtf8(sel.getObject()->Label.getValue()),
480  QString::fromLatin1(msg.pSubName));
481  item->setText(text);
482 
483  QList<QVariant> data;
484  data << QByteArray(msg.pDocName);
485  data << QByteArray(msg.pObjectName);
486  data << QByteArray(msg.pSubName);
487  item->setData(Qt::UserRole, data);
488 
489  appendCurve(sel.getObject(), msg.pSubName);
490  }
491  else if (selectionMode == RemoveEdge) {
492  Gui::SelectionObject sel(msg);
493  QList<QVariant> data;
494  data << QByteArray(msg.pDocName);
495  data << QByteArray(msg.pObjectName);
496  data << QByteArray(msg.pSubName);
497 
498  // only the three first elements must match
499  for (int i = 0; i < ui->listSections->count(); i++) {
500  QListWidgetItem* item = ui->listSections->item(i);
501  QList<QVariant> userdata = item->data(Qt::UserRole).toList();
502  if (userdata.mid(0, 3) == data) {
503  ui->listSections->takeItem(i);
504  delete item;
505  break;
506  }
507  }
508 
509  removeCurve(sel.getObject(), msg.pSubName);
510  }
511 
512  editedObject->recomputeFeature();
513  QTimer::singleShot(50, this, &SectionsPanel::clearSelection);
514  }
515 }

536 void SectionsPanel::onIndexesMoved()
537 {
538  QAbstractItemModel* model = qobject_cast<QAbstractItemModel*>(sender());
539  if (!model) {
540  return;
541  }
542 
543  std::vector<App::DocumentObject*> objects;

545 
546  int rows = model->rowCount();
547  for (int i = 0; i < rows; i++) {

552  App::Document* doc = App::GetApplication().getDocument(data[0].toByteArray());

577 void SectionsPanel::removeCurve(App::DocumentObject* obj, const std::string& subname)
578 {
579  this->vp->highlightReferences(ViewProviderSections::Edge,
580  editedObject->NSections.getSubListValues(),
581  false);
582

584  auto element = editedObject->NSections.getSubValues();
585 
586  auto it = objects.begin();
587  auto jt = element.begin();
588  for (; it != objects.end() && jt != element.end(); ++it, ++jt) {
589  if (*it == obj && *jt == subname) {
