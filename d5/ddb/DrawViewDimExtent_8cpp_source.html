112 pointPair DrawViewDimExtent::getPointsExtent(ReferenceVector references)
113 {
114 // Base::Console().Message("DVD::getPointsExtent() - %s\n", getNameInDocument());
115  App::DocumentObject* refObject = references.front().getObject();
116  int direction = DirExtent.getValue();
117  if (refObject->isDerivedFrom(TechDraw::DrawViewPart::getClassTypeId())) {
118  auto dvp = static_cast<TechDraw::DrawViewPart*>(refObject);
119 
120  std::vector<std::string> edgeNames; //empty list means we are using all the edges
121  if (!references.at(0).getSubName().empty()) {
122  //this is the usual case of selected edges in a dvp
123  for (auto& ref : references) {
124  if (ref.getSubName().empty()) {
125  continue;
126  }
127  std::string geomType = DrawUtil::getGeomTypeFromName(ref.getSubName());
128  if (geomType == "Edge") {
129  edgeNames.push_back(ref.getSubName());
130  }
131  }
132  }
133  std::pair<Base::Vector3d, Base::Vector3d> endPoints =
134  DrawDimHelper::minMax(dvp,
135  edgeNames,
136  direction);
137  return pointPair(endPoints.first, endPoints.second);
138  }
139 
140  //this is a 3d reference
141  std::pair<Base::Vector3d, Base::Vector3d> endPoints =
142  DrawDimHelper::minMax3d(getViewPart(),
143  references,
144  direction);
145  return pointPair(endPoints.first, endPoints.second);
146 }
