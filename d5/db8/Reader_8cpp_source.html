164  std::string what = message;

169

181 void Base::XMLReader::readElement(const char* ElementName)
182 {
183  bool ok {};
184 
185  endCharStream();
186  int currentLevel = Level;
187  std::string currentName = LocalName;
188  do {
189  ok = read();
190  if (!ok) {
191  break;
192  }
193  if (ReadType == EndElement && currentName == LocalName && currentLevel >= Level) {
194  // we have reached the end of the element when calling this method
195  // thus we must stop reading on.
196  break;
197  }
198  if (ReadType == EndDocument) {
199  // the end of the document has been reached but we still try to continue on reading
200  throw Base::XMLParseException("End of document reached");
201  }
202  } while ((ReadType != StartElement && ReadType != StartEndElement)
203  || (ElementName && LocalName != ElementName));
204 }

252 {
253  endCharStream();
254 
255  // if we are already at the end of the current element
256  if ((ReadType == EndElement || ReadType == StartEndElement) && ElementName
257  && LocalName == ElementName && (level < 0 || level == Level)) {
258  return;
259  }
260  if (ReadType == EndDocument) {
261  // the end of the document has been reached but we still try to continue on reading

263  }
264 
265  bool ok {};

267  ok = read();
268  if (!ok) {
269  break;
270  }

272  break;
273  }
274  } while (ReadType != EndElement
275  || (ElementName && (LocalName != ElementName || (level >= 0 && level != Level))));

291 std::streamsize Base::XMLReader::read(char_type* s, std::streamsize n)
292 {
293 
294  char_type* buf = s;
295  if (CharacterOffset < 0) {
296  return -1;
297  }
298 
299  for (;;) {
300  std::streamsize copy_size =
301  static_cast<std::streamsize>(Characters.size()) - CharacterOffset;
302  if (n < copy_size) {
303  copy_size = n;
304  }
305  std::memcpy(s, Characters.c_str() + CharacterOffset, copy_size);
306  n -= copy_size;
307  s += copy_size;
308  CharacterOffset += copy_size;
309 
310  if (!n) {
311  break;
312  }
313 
314  if (ReadType == Chars) {
315  read();
316  }
317  else {
318  CharacterOffset = -1;
319  break;
320  }
321  }
322 
323  return s - buf;
324 }

340 std::istream& Base::XMLReader::beginCharStream(CharStreamFormat format)
341 {
342  if (CharStream) {
343  throw Base::XMLParseException("recursive character stream");
344  }
345 
346  // TODO: An XML element can actually contain a mix of child elements and
347  // characters. So we should not actually demand 'StartElement' here. But
348  // with the current implementation of character stream, we cannot track
349  // child elements and character content at the same time.
350  if (ReadType == StartElement) {
351  CharacterOffset = 0;
352  read();
353  }
354  else if (ReadType == StartEndElement) {
355  // If we are currently at a self-closing element, just leave the offset
356  // as negative and do not read any characters. This will result in an
357  // empty input stream for the caller.
358  CharacterOffset = -1;
359  }
360  else {
361  throw Base::XMLParseException("invalid state while reading character stream");
362  }
363 
364  CharStream = std::make_unique<boost::iostreams::filtering_istream>();
365  auto* filteringStream = dynamic_cast<boost::iostreams::filtering_istream*>(CharStream.get());
366  if (format == CharStreamFormat::Base64Encoded) {
367  filteringStream->push(
368  base64_decoder(Base::base64DefaultBufferSize, Base64ErrorHandling::silent));
369  }
370  filteringStream->push(boost::ref(*this));
371  return *CharStream;
372 }

394 void Base::XMLReader::readFiles(zipios::ZipInputStream& zipstream) const
395 {
396  // It's possible that not all objects inside the document could be created, e.g. if a module
397  // is missing that would know these object types. So, there may be data files inside the zip
398  // file that cannot be read. We simply ignore these files.
399  // On the other hand, however, it could happen that a file should be read that is not part of
400  // the zip file. This happens e.g. if a document is written without GUI up but is read with GUI
401  // up. In this case the associated GUI document asks for its file which is not part of the ZIP
402  // file, then.
403  // In either case it's guaranteed that the order of the files is kept.
404  zipios::ConstEntryPointer entry;
405  try {
406  entry = zipstream.getNextEntry();
407  }
408  catch (const std::exception&) {
409  // There is no further file at all. This can happen if the
410  // project file was created without GUI
411  return;
412  }
413  std::vector<FileEntry>::const_iterator it = FileList.begin();
414  Base::SequencerLauncher seq("Importing project files...", FileList.size());
415  while (entry->isValid() && it != FileList.end()) {
416  std::vector<FileEntry>::const_iterator jt = it;
417  // Check if the current entry is registered, otherwise check the next registered files as
418  // soon as both file names match
419  while (jt != FileList.end() && entry->getName() != jt->FileName) {
420  ++jt;
421  }
422  // If this condition is true both file names match and we can read-in the data, otherwise
423  // no file name for the current entry in the zip was registered.
424  if (jt != FileList.end()) {
425  try {
426  Base::Reader reader(zipstream, jt->FileName, FileVersion);
427  jt->Object->RestoreDocFile(reader);
428  if (reader.getLocalReader()) {
429  reader.getLocalReader()->readFiles(zipstream);
430  }
431  }
432  catch (...) {
433  // For any exception we just continue with the next file.
434  // It doesn't matter if the last reader has read more or
435  // less data than the file size would allow.
436  // All what we need to do is to notify the user about the
437  // failure.
438  Base::Console().Error("Reading failed from embedded file: %s\n",
439  entry->toString().c_str());
440  FailedFiles.push_back(jt->FileName);
441  }
442  // Go to the next registered file name
443  it = jt + 1;
444  }
445 
446  seq.next();
447 
448  // In either case we must go to the next entry
449  try {
450  entry = zipstream.getNextEntry();
451  }
452  catch (const std::exception&) {
453  // there is no further entry
454  break;
455  }
456  }
457 }
