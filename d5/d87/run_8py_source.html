72 def run_fem_solver(solver, working_dir=None):
73  """Execute *solver* of the solver framework.
74 
75  Uses :meth:`getMachine <femsolver.solverbase.Proxy.getMachine>` to obtain a
76  :class:`Machine` instance of the solver. It than executes the Machine with
77  using the ``RESULTS`` target (see :class:`Machine` for infos about
78  different targets). This method is blocking, it waits for the solver to
79  finished before returning. Be aware of :class:`Machine` caching when using
80  the function.
81 
82  :param solver:
83  A document object which must be a framework compliant solver. This means
84  that it should be derived from the document object provided by
85  :mod:`femsolver.solverbase` and implement all required methods
86  correctly. Of particular importance is :meth:`getMachine
87  <femsolver.solverbase.Proxy.getMachine>` as it is used by this method
88  the get the :class:`Machine` used to execute the solver.
89 
90  :param working_dir:
91  If specified it overwrites the automatic and user configurable working
92  directory management of the Solver framework. Should always be a
93  absolute path because the location of the binary is not consistent
94  among platforms. If ``None`` the automatic working directory management
95  is used.
96 
97  :note:
98  There is some legacy code to execute the old Calculix solver
99  (pre-framework) which behaives differently because it does not
100  use a :class:`Machine`.
101  """
102 
103  if solver.Proxy.Type == "Fem::SolverCcxTools":
104  from femtools.ccxtools import CcxTools as ccx
105 
106  App.Console.PrintMessage("Run of CalxuliX ccx tools solver started.\n")
107  fea = ccx(solver)
108  fea.reset_mesh_purge_results_checked()
109  if working_dir is None:
110  fea.run() # standard, no working dir is given in solver
111  else:
112  # not the standard way
113  fea.update_objects()
114  fea.setup_working_dir(working_dir)
115  fea.setup_ccx()
116  message = fea.check_prerequisites()
117  if not message:
118  fea.write_inp_file()
119  fea.ccx_run()
120  fea.load_results()
121  else:
122  App.Console.PrintError(f"Houston, we have a problem...!\n{message}\n")
123  App.Console.PrintMessage("Run of CalxuliX ccx tools solver finished.\n")
124  else:
125  # App.Console.PrintMessage("Frame work solver!\n")
126  try:
127  if working_dir is not None:
128  machine = getMachine(solver, working_dir)
129  else:
130  machine = getMachine(solver)
131  except MustSaveError:
132  error_message = (
133  "Please save the file before executing the solver. "
134  "This must be done because the location of the working "
135  'directory is set to "Beside *.FCStd File".'
136  )
137  App.Console.PrintError(error_message + "\n")
138  if App.GuiUp:
139  QtGui.QMessageBox.critical(
140  FreeCADGui.getMainWindow(), "Can't start Solver", error_message
141  )
142  return
143  except DirectoryDoesNotExistError:
144  error_message = "Selected working directory doesn't exist."
145  App.Console.PrintError(error_message + "\n")
146  if App.GuiUp:
147  QtGui.QMessageBox.critical(
148  FreeCADGui.getMainWindow(), "Can't start Solver", error_message
149  )
150  return
151  if not machine.running:
152  machine.reset()
153  machine.target = RESULTS
154  machine.start()
155  machine.join() # wait for the machine to finish.
156  if machine.failed is True:
157  App.Console.PrintError("Machine failed to run.\n")
158  from .report import displayLog
159 
160  displayLog(machine.report)
161  if App.GuiUp:
162  error_message = (
163  "Failed to run. Please try again after all "
164  "of the following errors are resolved."
165  )
166  from .report import display
167 
168  display(machine.report, "Run Report", error_message)

171 def getMachine(solver, path=None):
172  """Get or create :class:`Machine` using caching mechanism.
173 
174  :param solver:
175  A document object which must be a framework compliant solver. This means
176  that it should be derived from the document object provided by
177  :mod:`femsolver.solverbase` and implement all required methods
178  correctly. Of particular importance is :meth:`getMachine
179  <femsolver.solverbase.Proxy.getMachine>` as it is used by this method
180  to create a new :class:`Machine` on cache miss.
181 
182  :param path:
183  A valid filesystem path which shall be associetad with the machine.
184  """
185  # print(path)
186  _DocObserver.attach()
187  m = _machines.get(solver)
188  if m is None or not _isPathValid(m, path):
189  m = _createMachine(solver, path, testmode=False)
190  # print(m.__dir__()) # document these attributes somewhere
191  # print(m.directory)
192  return m

195 def _isPathValid(m, path):
196  t = _dirTypes.get(m.directory) # setting default None
197  setting = settings.get_dir_setting()
198  if path is not None:
199  return t is None and m.directory == path
200  if setting == settings.DirSetting.BESIDE:
201  if t == settings.DirSetting.BESIDE:
202  base = os.path.split(m.directory.rstrip("/"))[0]
203  return base == _getBesideBase(m.solver)
204  return False
205  if setting == settings.DirSetting.TEMPORARY:
206  return t == settings.DirSetting.TEMPORARY
207  if setting == settings.DirSetting.CUSTOM:
208  if t == settings.DirSetting.CUSTOM:
209  firstBase = os.path.split(m.directory.rstrip("/"))[0]
210  customBase = os.path.split(firstBase)[0]
211  return customBase == _getCustomBase(m.solver)
212  return False

215 def _createMachine(solver, path, testmode):
216  global _dirTypes
217  setting = settings.get_dir_setting()
218  if path is not None:
219  _dirTypes[path] = None
220  elif setting == settings.DirSetting.BESIDE:
221  path = _getBesideDir(solver)
222  _dirTypes[path] = settings.DirSetting.BESIDE
223  elif setting == settings.DirSetting.TEMPORARY:
224  path = _getTempDir(solver)
225  _dirTypes[path] = settings.DirSetting.TEMPORARY
226  elif setting == settings.DirSetting.CUSTOM:
227  path = _getCustomDir(solver)
228  _dirTypes[path] = settings.DirSetting.CUSTOM
229  m = solver.Proxy.createMachine(solver, path, testmode)
230  oldMachine = _machines.get(solver)
231  if oldMachine is not None and _dirTypes.get(oldMachine.directory) is not None:
232  del _dirTypes[oldMachine.directory]
233  _machines[solver] = m
234  return m

338  def run(self):
339  self._confTasks()
340  self._isReset = False
341  self._pendingState = self.state
342  while not self.aborted and not self.failed and self._pendingState <= self.target:
343  task = self._getTask(self._pendingState)
344  self._runTask(task)
345  self.report.extend(task.report)
346  if task.failed:
347  self.fail()
348  elif task.aborted:
349  self.abort()
350  else:
351  self._pendingState += 1
352  self._applyPending()
353 
354  def reset(self, newState=CHECK):
355  state = self.state if self._pendingState is None else self._pendingState
356  if newState < state:
357  self._isReset = True
358  self._state = newState
359  signal.notify(self.signalState)
360 
361  def _confTasks(self):
362  tasks = [self.check, self.prepare, self.solve, self.results]
363  for t in tasks:
364  t.solver = self.solver
365  t.directory = self.directory
366  t.testmode = self.testmode
367 
368  def _applyPending(self):
369  if not self._isReset:
370  self._state = self._pendingState
371  signal.notify(self.signalState)
372  self._isReset = False
373  self._pendingState = None

375  def _runTask(self, task):
376 
377  def statusProxy(line):
378  self.pushStatus(line)
379 
380  def killer():
381  task.abort()
382 
383  self.signalAbort.add(killer)
384  task.signalStatus.add(statusProxy)
385  task.start()
386  task.join()
387  self.signalAbort.remove(killer)
