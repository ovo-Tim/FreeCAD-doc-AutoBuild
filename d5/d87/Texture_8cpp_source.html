38 bool Texture::LoadImage(unsigned int* image, int _width, int _height)
39 {
40  DestroyTexture();
41  width = _width;
42  height = _height;
43  glGenTextures(1, &mTextureId);
44  glBindTexture(GL_TEXTURE_2D, mTextureId);
45  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
46  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
47  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
48  glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
49  glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA8, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, image);
50  glBindTexture(GL_TEXTURE_2D, 0);
51  return true;
52 }
