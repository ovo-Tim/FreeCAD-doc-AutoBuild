595 def HEIDEN_Line(line_Params, line_comp, line_feed, line_rapid, line_M_funct, Cmd_Number):
596  global FEED_MAX_SPEED
597  global COMPENSATION_DIFF_STATUS
598  global G_FUNCTION_STORE
599  global MACHINE_WORK_AXIS
600  global MACHINE_LAST_POSITION
601  global MACHINE_STORED_PARAMS
602  global MACHINE_SKIP_PARAMS
603  global MACHINE_USE_FMAX
604  H_Line_New = { # axis initial values to overwrite
605  "X": 99999,
606  "Y": 99999,
607  "Z": 99999,
608  "A": 99999,
609  "B": 99999,
610  "C": 99999,
611  }
612  H_Line = "L"
613  H_Line_Params = [["X", "Y", "Z"], ["R0", line_feed, line_M_funct]]
614 
615  # check and hide duplicated axis movements, not last, update with new ones
616  for i in H_Line_New:
617  if G_FUNCTION_STORE["G91"]: # incremental
618  H_Line_New[i] = 0
619  if i in line_Params:
620  if line_Params[i] != 0 or line_M_funct != "":
621  H_Line += " I" + HEIDEN_Format(i, line_Params[i]) # print incremental
622  # update to absolute position
623  H_Line_New[i] = MACHINE_LAST_POSITION[i] + H_Line_New[i]
624  else: # absolute
625  H_Line_New[i] = MACHINE_LAST_POSITION[i]
626  if i in line_Params:
627  if line_Params[i] != H_Line_New[i] or line_M_funct != "":
628  H_Line += " " + HEIDEN_Format(i, line_Params[i])
629  H_Line_New[i] = line_Params[i]
630 
631  if H_Line == "L": # No movements no line
632  return None
633 
634  if COMPENSATION_DIFF_STATUS[0]: # Diff from compensated ad not compensated path
635  if COMPENSATION_DIFF_STATUS[1]: # skip if already compensated, not active by now
636  Cmd_Number -= 1 # align
637  # initialize like true, set false if not same point compensated and not compensated
638  i = True
639  for j in H_Line_Params[0]:
640  if j in STORED_COMPENSATED_OBJ[Cmd_Number].Parameters and j in line_Params:
641  if STORED_COMPENSATED_OBJ[Cmd_Number].Parameters[j] != line_Params[j]:
642  i = False
643  if i == False:
644  H_Line_Params[1][0] = "R" + line_comp
645  # we can skip this control if already in compensation
646  # COMPENSATION_DIFF_STATUS[1] = False
647  else:
648  H_Line_Params[1][0] = "R" + line_comp # not used by now
649 
650  # check if we need to skip already active parameters
651  # R parameter
652  if MACHINE_SKIP_PARAMS == False or H_Line_Params[1][0] != MACHINE_STORED_PARAMS[0]:
653  MACHINE_STORED_PARAMS[0] = H_Line_Params[1][0]
654  H_Line += " " + H_Line_Params[1][0]
655 
656  # F parameter (check rapid o feed)
657  if line_rapid:
