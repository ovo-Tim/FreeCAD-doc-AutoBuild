167  Py::Object write(const Py::Tuple& args)
168  {
169  char* Name;
170  PyObject* pObj;
171  if (!PyArg_ParseTuple(args.ptr(), "Oet", &pObj, "utf-8", &Name)) {
172  throw Py::Exception();
173  }
174  std::string EncodedName = std::string(Name);
175  PyMem_Free(Name);
176  Base::FileInfo file(EncodedName.c_str());
177 
178  if (PyObject_TypeCheck(pObj, &(App::DocumentObjectPy::Type))) {
179  App::DocumentObject* obj =
180  static_cast<App::DocumentObjectPy*>(pObj)->getDocumentObjectPtr();
181  if (obj->getTypeId().isDerivedFrom(Base::Type::fromName("Path::Feature"))) {
182  const Path::Toolpath& path = static_cast<Path::Feature*>(obj)->Path.getValue();
183  std::string gcode = path.toGCode();
184  Base::ofstream ofile(file);
185  ofile << gcode;
186  ofile.close();
187  }
188  else {
189  throw Py::RuntimeError("The given file is not a path");
190  }
191  }
192 
193  return Py::None();

197  Py::Object read(const Py::Tuple& args)
198  {
199  char* Name;
200  const char* DocName = nullptr;
201  if (!PyArg_ParseTuple(args.ptr(), "et|s", "utf-8", &Name, &DocName)) {
202  throw Py::Exception();
203  }
204  std::string EncodedName = std::string(Name);
205  PyMem_Free(Name);
206 
207  Base::FileInfo file(EncodedName.c_str());
208  if (!file.exists()) {
209  throw Py::RuntimeError("File doesn't exist");
210  }
211 
212  App::Document* pcDoc;
213  if (DocName) {
214  pcDoc = App::GetApplication().getDocument(DocName);
215  }
216  else {
217  pcDoc = App::GetApplication().getActiveDocument();
218  }
219  if (!pcDoc) {
220  pcDoc = App::GetApplication().newDocument(DocName);
221  }
222 
223  try {
224  // read the gcode file
225  Base::ifstream filestr(file);
226  std::stringstream buffer;
227  buffer << filestr.rdbuf();
228  std::string gcode = buffer.str();
229  Path::Toolpath path;
230  path.setFromGCode(gcode);
231  Path::Feature* object = static_cast<Path::Feature*>(
232  pcDoc->addObject("Path::Feature", file.fileNamePure().c_str()));
233  object->Path.setValue(path);
234  pcDoc->recompute();
235  }
236  catch (const Base::Exception& e) {
237  throw Py::RuntimeError(e.what());
238  }
239 
240  return Py::None();
241  }

276  Py::Object fromShape(const Py::Tuple& args)
277  {
278  PyObject* pcObj;
279  if (!PyArg_ParseTuple(args.ptr(), "O", &pcObj)) {
280  throw Py::Exception();
281  }
282  TopoDS_Shape shape;
283  try {
284  if (PyObject_TypeCheck(pcObj, &(Part::TopoShapePy::Type))) {
285  shape = static_cast<Part::TopoShapePy*>(pcObj)->getTopoShapePtr()->getShape();
286  }
287  else {
288  throw Py::TypeError("the given object is not a shape");
289  }
290  if (!shape.IsNull()) {
291  if (shape.ShapeType() == TopAbs_WIRE) {
292  Path::Toolpath result;
293  bool first = true;
294  Base::Placement last;
295 
296  TopExp_Explorer ExpEdges(shape, TopAbs_EDGE);
297  while (ExpEdges.More()) {
298  const TopoDS_Edge& edge = TopoDS::Edge(ExpEdges.Current());
299  TopExp_Explorer ExpVerts(edge, TopAbs_VERTEX);
300  bool vfirst = true;
301  while (ExpVerts.More()) {
302  const TopoDS_Vertex& vert = TopoDS::Vertex(ExpVerts.Current());
303  gp_Pnt pnt = BRep_Tool::Pnt(vert);
304  Base::Placement tpl;
305  tpl.setPosition(Base::Vector3d(pnt.X(), pnt.Y(), pnt.Z()));
306  if (first) {
307  // add first point as a G0 move
308  Path::Command cmd;
309  std::ostringstream ctxt;
310  ctxt << "G0 X" << tpl.getPosition().x << " Y" << tpl.getPosition().y
311  << " Z" << tpl.getPosition().z;
312  cmd.setFromGCode(ctxt.str());
313  result.addCommand(cmd);
314  first = false;
315  vfirst = false;
316  }
317  else {
318  if (vfirst) {
319  vfirst = false;
320  }
321  else {
322  Path::Command cmd;
323  cmd.setFromPlacement(tpl);
324 
325  // write arc data if needed
326  BRepAdaptor_Curve adapt(edge);
327  if (adapt.GetType() == GeomAbs_Circle) {
328  gp_Circ circ = adapt.Circle();
329  gp_Pnt c = circ.Location();
330  bool clockwise = false;
331  gp_Dir n = circ.Axis().Direction();
332  if (n.Z() < 0) {
333  clockwise = true;
334  }
335  Base::Vector3d center = Base::Vector3d(c.X(), c.Y(), c.Z());
336  // center coords must be relative to last point
337  center -= last.getPosition();
338  cmd.setCenter(center, clockwise);
339  }
340  result.addCommand(cmd);
341  }
342  }
343  ExpVerts.Next();
344  last = tpl;
345  }
346  ExpEdges.Next();
347  }
348  return Py::asObject(new Path::PathPy(new Path::Toolpath(result)));
349  }
350  else {
351  throw Py::TypeError("the given shape must be a wire");
352  }
353  }
354  else {
355  throw Py::TypeError("the given shape is empty");
356  }
357  }
358  catch (const Base::Exception& e) {
359  throw Py::RuntimeError(e.what());
360  }
361  }

363  Py::Object fromShapes(const Py::Tuple& args, const Py::Dict& kwds)
364  {
365  PARAM_PY_DECLARE_INIT(PARAM_FARG, AREA_PARAMS_PATH)
366  PyObject* pShapes = nullptr;
367  PyObject* start = nullptr;
368  PyObject* return_end = Py_False;
369  static const std::array<const char*, 22> kwd_list {
370  "shapes",
371  "start",
372  "return_end",
373  PARAM_FIELD_STRINGS(ARG, AREA_PARAMS_PATH),
374  nullptr};
375  if (!Base::Wrapped_ParseTupleAndKeywords(args.ptr(),
376  kwds.ptr(),
377  "O|O!O!" PARAM_PY_KWDS(AREA_PARAMS_PATH),
378  kwd_list,
379  &pShapes,
380  &(Base::VectorPy::Type),
381  &start,
382  &PyBool_Type,
383  &return_end,
384  PARAM_REF(PARAM_FARG, AREA_PARAMS_PATH))) {
385  throw Py::Exception();
386  }
387 
388  std::list<TopoDS_Shape> shapes;
389  if (PyObject_TypeCheck(pShapes, &(Part::TopoShapePy::Type))) {
390  shapes.push_back(
391  static_cast<Part::TopoShapePy*>(pShapes)->getTopoShapePtr()->getShape());
392  }
393  else if (PyObject_TypeCheck(pShapes, &(PyList_Type))
394  || PyObject_TypeCheck(pShapes, &(PyTuple_Type))) {
395  Py::Sequence shapeSeq(pShapes);
396  for (Py::Sequence::iterator it = shapeSeq.begin(); it != shapeSeq.end(); ++it) {
397  PyObject* item = (*it).ptr();
398  if (!PyObject_TypeCheck(item, &(Part::TopoShapePy::Type))) {
399  PyErr_SetString(PyExc_TypeError, "non-shape object in sequence");
400  throw Py::Exception();
401  }
402  shapes.push_back(
403  static_cast<Part::TopoShapePy*>(item)->getTopoShapePtr()->getShape());
404  }
405  }
406 
407  gp_Pnt pstart;
408  if (start) {
409  Base::Vector3d vec = static_cast<Base::VectorPy*>(start)->value();
410  pstart.SetCoord(vec.x, vec.y, vec.z);
411  }
412 
413  try {
414  gp_Pnt pend;
415  std::unique_ptr<Path::Toolpath> path(new Path::Toolpath);
416  Path::Area::toPath(*path,
417  shapes,
418  start ? &pstart : nullptr,
419  &pend,
420  PARAM_PY_FIELDS(PARAM_FARG, AREA_PARAMS_PATH));
421  if (!Base::asBoolean(return_end)) {
422  return Py::asObject(new Path::PathPy(path.release()));
423  }
424  Py::Tuple tuple(2);
425  tuple.setItem(0, Py::asObject(new Path::PathPy(path.release())));
426  tuple.setItem(
427  1,
428  Py::asObject(new Base::VectorPy(Base::Vector3d(pend.X(), pend.Y(), pend.Z()))));
429  return tuple;
430  }
431  PATH_CATCH
432  }
