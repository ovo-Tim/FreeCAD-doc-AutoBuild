303 QVariant QGIViewBalloon::itemChange(GraphicsItemChange change, const QVariant& value)
304 {
305  if (change == ItemSelectedHasChanged && scene()) {
306  if (isSelected()) {
307  balloonLabel->setSelected(true);
308  }
309  else {
310  balloonLabel->setSelected(false);
311  }
312  draw();
313  return value;
314  }
315 
316  if(change == ItemPositionChange && scene()) {
317  // QGIVBalloon doesn't really change position the way other views do.
318  // If we call QGIView::itemChange it will set the position to (0,0) instead of
319  // using the label's position, and the Balloon will be in the wrong place.
320  // QGIVDimension behaves the same way.
321  return QGraphicsItem::itemChange(change, value);
322  }
323 
324  return QGIView::itemChange(change, value);
325 }

413 //update the bubble contents
414 void QGIViewBalloon::updateBalloon(bool obtuse)
415 {
416  (void)obtuse;
417  const auto balloon(dynamic_cast<TechDraw::DrawViewBalloon*>(getViewObject()));
418  if (!balloon) {
419  return;
420  }
421  auto vp = static_cast<ViewProviderBalloon*>(getViewProvider(getViewObject()));
422  if (!vp) {
423  return;
424  }
425  const TechDraw::DrawView* refObj = balloon->getParentView();
426  if (!refObj) {
427  return;
428  }
429 
430  QFont font;
431  font.setFamily(QString::fromUtf8(vp->Font.getValue()));
432  font.setPixelSize(exactFontSize(vp->Font.getValue(), vp->Fontsize.getValue()));
433  balloonLabel->setFont(font);
434 
435  QString labelText = QString::fromUtf8(balloon->Text.getStrValue().data());
436  balloonLabel->setVerticalSep(false);
437  balloonLabel->setSeps(std::vector<int>());
438 
439  if (strcmp(balloon->BubbleShape.getValueAsString(), "Rectangle") == 0) {
440  std::vector<int> newSeps;
441  while (labelText.contains(QString::fromUtf8("|"))) {
442  int pos = labelText.indexOf(QString::fromUtf8("|"));
443  labelText.replace(pos, 1, QString::fromUtf8(" "));
444  QFontMetrics fm(balloonLabel->getFont());
445  newSeps.push_back(Gui::QtTools::horizontalAdvance(fm, labelText.left(pos + 2)));
446  balloonLabel->setVerticalSep(true);
447  }
448  balloonLabel->setSeps(newSeps);
449  }
450 
451  balloonLabel->setDimString(labelText, Rez::guiX(balloon->TextWrapLen.getValue()));
452  float x = Rez::guiX(balloon->X.getValue() * refObj->getScale());
453  float y = Rez::guiX(balloon->Y.getValue() * refObj->getScale());
454  balloonLabel->setPosFromCenter(x, -y);
455 }

457 void QGIViewBalloon::balloonLabelDragged(bool originDrag)
458 {
459  auto dvb(dynamic_cast<TechDraw::DrawViewBalloon*>(getViewObject()));
460  if (!dvb) {
461  return;
462  }
463 
464  if (!m_dragInProgress) {//first drag movement
465  m_dragInProgress = true;
466  if (originDrag) {//moving whole thing, remember Origin offset from Bubble
467  m_saveOriginOffset = dvb->getOriginOffset();
468  m_saveOrigin = DU::toVector3d(arrow->pos());
469  m_savePosition = DU::toVector3d(balloonLabel->pos());
470  }
471  }
472 
473  // store if origin is also moving to be able to later calc new origin and update feature
474  if (originDrag) {
475  m_originDragged = true;
476  }
477 
478  DrawView* balloonParent = getSourceView();
479  if (balloonParent) {
480  // redraw the balloon at the new position
481  // note that we don't store the new position to the X/Y properties
482  // since the dragging is not yet finished
483  // TODO: we don't need to redraw if the graphic is not changing (other than location).
484  // all of the balloon components (box, text, line, arrow, etc) would have to be placed
485  // into a QGraphicsItemGroup (or made children of a single QGI) in order to move them in
486  // sync
487  drawBalloon(true);
488  }
489 }

491 void QGIViewBalloon::balloonLabelDragFinished()
492 {
493  // stores the final drag position for undo
494 
495  auto dvb(dynamic_cast<TechDraw::DrawViewBalloon*>(getViewObject()));
496  if (!dvb) {
497  return;
498  }
499 
500  double scale = 1.0;
501  DrawView* balloonParent = getSourceView();
502  if (!balloonParent) {
503  return;
504  }
505  scale = balloonParent->getScale();
506 
507  //set feature position (x, y) from graphic position
508  double x = Rez::appX(balloonLabel->getCenterX() / scale);
509  double y = Rez::appX(balloonLabel->getCenterY() / scale);
510  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Drag Balloon"));
511  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.%s.X = %f",
512  dvb->getNameInDocument(), x);
513  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.%s.Y = %f",
514  dvb->getNameInDocument(), -y);
515 
516  // for the case that origin was also dragged, calc new origin and update feature
517  if (m_originDragged) {
518  auto originGui = arrowPosInDrag(); // Qt rotated
519  auto originApp = originGui / scale; // Qt rotated unscaled
520  originApp = Rez::appX(DU::invertY(originApp)); // App rotated
521 
522  auto originAppUnrotated = originApp;
523  auto rotationDeg = balloonParent->Rotation.getValue();
524  if (rotationDeg != 0) {
525  originAppUnrotated.RotateZ(Base::toRadians(-rotationDeg));
526  }
527 
528  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.%s.OriginX = %f",
529  dvb->getNameInDocument(), originAppUnrotated.x);
530  Gui::Command::doCommand(Gui::Command::Doc, "App.ActiveDocument.%s.OriginY = %f",
531  dvb->getNameInDocument(), originAppUnrotated.y);
532  }
533 
534  Gui::Command::commitCommand();
535 
536  m_dragInProgress = false;
537  m_originDragged = false;
538  drawBalloon(false);
539 }

541 //from QGVP::mouseReleaseEvent - pos = eventPos in scene coords?
542 void QGIViewBalloon::placeBalloon(QPointF pos)
543 {
544  // Base::Console().Message("QGIVB::placeBalloon(%s)\n",
545  // DrawUtil::formatVector(pos).c_str());
546  auto balloon(dynamic_cast<TechDraw::DrawViewBalloon*>(getViewObject()));
547  if (!balloon) {
548  return;
549  }
550 
551  DrawView* balloonParent = dynamic_cast<DrawView*>(balloon->SourceView.getValue());
552  if (!balloonParent) {
553  return;
554  }
555 
556  auto featPage = balloonParent->findParentPage();
557  if (!featPage) {
558  return;
559  }
560 
561  auto vp = static_cast<ViewProviderBalloon*>(getViewProvider(getViewObject()));
562  if (!vp) {
563  return;
564  }
565 
566  QGIView* qgivParent = nullptr;
567  QPointF viewPos;
568  Gui::ViewProvider* objVp = QGIView::getViewProvider(balloonParent);
569  auto partVP = dynamic_cast<ViewProviderViewPart*>(objVp);
570  if (partVP) {
571  qgivParent = partVP->getQView();
572  if (qgivParent) {
573  //tip position is mouse release pos in parentView coords ==> OriginX, OriginY
574  //bubble pos is some arbitrary shift from tip position ==> X, Y
575  viewPos = qgivParent->mapFromScene(pos);
576  balloon->OriginX.setValue(Rez::appX(viewPos.x()) / balloonParent->getScale());
577  balloon->OriginY.setValue(-Rez::appX(viewPos.y()) / balloonParent->getScale());
578  balloon->X.setValue(Rez::appX((viewPos.x() + 200.0) / balloonParent->getScale()));
579  balloon->Y.setValue(-Rez::appX((viewPos.y() - 200.0) / balloonParent->getScale()));
580  }
581  }
582 
583  int idx = featPage->getNextBalloonIndex();
584  QString labelText = QString::number(idx);
585  balloon->Text.setValue(std::to_string(idx).c_str());
586 
587  QFont font = balloonLabel->getFont();
588  font.setPixelSize(calculateFontPixelSize(vp->Fontsize.getValue()));
589  font.setFamily(QString::fromUtf8(vp->Font.getValue()));
590  font.setPixelSize(exactFontSize(vp->Font.getValue(), vp->Fontsize.getValue()));
591  balloonLabel->setFont(font);
592 
593  prepareGeometryChange();
594 
595  // Default label position
596  balloonLabel->setPosFromCenter(viewPos.x() + 200, viewPos.y() - 200);
597  balloonLabel->setDimString(labelText, Rez::guiX(balloon->TextWrapLen.getValue()));
598 
599  draw();
600 }

608 void QGIViewBalloon::drawBalloon(bool originDrag)
609 {
610  if ((!originDrag) && m_dragInProgress) {
611  // TODO there are 2 drag status variables. m_draggingInProgress appears to be the one to use?
612  // dragged shows false while drag is still in progress.
613  return;
614  }
615 
616  prepareGeometryChange();
617 
618  TechDraw::DrawViewBalloon* balloon = dynamic_cast<TechDraw::DrawViewBalloon*>(getViewObject());
619  if ((!balloon) ||
620  (!balloon->isDerivedFrom(TechDraw::DrawViewBalloon::getClassTypeId()))) {
621  //nothing to draw, don't try
622  return;
623  }
624  if (balloon->isRestoring()) {
625  // don't try to draw yet
626  return;
627  }
628 
629  TechDraw::DrawView* refObj = balloon->getParentView();
630  auto vp = static_cast<ViewProviderBalloon*>(getViewProvider(getViewObject()));
631  if (!refObj || !vp) {
632  // can't draw this. probably restoring.
633  return;
634  }
635 
636  m_lineWidth = Rez::guiX(vp->LineWidth.getValue());
637 
638  double textWidth = balloonLabel->getDimText()->boundingRect().width();
639  double textHeight = balloonLabel->getDimText()->boundingRect().height();
640 
641  float arrowTipX;
642  Base::Vector3d arrowTipPosInParent;
643  bool isDragging = originDrag || m_dragInProgress;
644  Base::Vector3d labelPos;
645  getBalloonPoints(balloon, refObj, isDragging, labelPos, arrowTipPosInParent);
646  arrowTipX = arrowTipPosInParent.x;
647  Base::Vector3d lblCenter(labelPos.x, -labelPos.y, 0.0);
648 
649  if (balloon->isLocked()) {
650  balloonLabel->setFlag(QGraphicsItem::ItemIsMovable, false);
651  }
652  else {
653  balloonLabel->setFlag(QGraphicsItem::ItemIsMovable, true);
654  }
655 
656  Base::Vector3d dLineStart;
657  Base::Vector3d kinkPoint;
658  double kinkLength = Rez::guiX(balloon->KinkLength.getValue());
659 
660  const char* balloonType = balloon->BubbleShape.getValueAsString();
661 
662  float scale = balloon->ShapeScale.getValue();
663  double offsetLR = 0;
664  double offsetUD = 0;
665  QPainterPath balloonPath;
666 
667  if (strcmp(balloonType, "Circular") == 0) {
668  double balloonRadius = sqrt(pow((textHeight / 2.0), 2) + pow((textWidth / 2.0), 2));
669  balloonRadius = balloonRadius * scale;

823  arAngle = 270.0;
824  }
825  else {
826  arAngle = 0;
827  }
828  double radAngle = arAngle * M_PI / 180.0;
829  double sinAngle = sin(radAngle);
830  double cosAngle = cos(radAngle);
831  xAdj = Rez::guiX(arrowAdj * cosAngle);
832  yAdj = Rez::guiX(arrowAdj * sinAngle);
833  }
834  arrow->setRotation(arAngle);
835  arrow->show();
836  }
837  dLinePath.lineTo(arrowTipPosInParent.x - xAdj, arrowTipPosInParent.y - yAdj);
838  balloonLines->setPath(dLinePath);
839 
840  // This overwrites the previously created QPainterPath with empty one, in case it should be hidden. Should be refactored.
841  if (!vp->LineVisible.getValue()) {
842  arrow->hide();
843  balloonLines->setPath(QPainterPath());
844  }

846  // redraw the Balloon and the parent View
847  if (hasHover && !isSelected()) {
848  setPrettyPre();
849  }
850  else if (isSelected()) {
851  setPrettySel();
852  }
853  else {
