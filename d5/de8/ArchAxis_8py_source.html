95  pl = obj.Placement

100  if hasattr(obj, "Distances"):

102  if hasattr(obj, "Angles"):
103  angles = obj.Angles
104  if distances and obj.Length.Value:
105  if angles and len(distances) == len(angles):
106  for i in range(len(distances)):
107  if hasattr(obj.Length,"Value"):
108  l = obj.Length.Value
109  else:
110  l = obj.Length
111  dist += distances[i]
112  ang = math.radians(angles[i])

540  def getNumber(self,vobj,num):
541 
542  chars = "abcdefghijklmnopqrstuvwxyz"
543  roman=(('M',1000),('CM',900),('D',500),('CD',400),
544  ('C',100),('XC',90),('L',50),('XL',40),
545  ('X',10),('IX',9),('V',5),('IV',4),('I',1))
546  if hasattr(vobj.Object,"CustomNumber") and vobj.Object.CustomNumber:
547  return vobj.Object.CustomNumber
548  elif hasattr(vobj,"NumberingStyle"):
549  if vobj.NumberingStyle == "1,2,3":
550  return str(num+1)
551  elif vobj.NumberingStyle == "01,02,03":
552  return str(num+1).zfill(2)
553  elif vobj.NumberingStyle == "001,002,003":
554  return str(num+1).zfill(3)
555  elif vobj.NumberingStyle == "A,B,C":
556  result = ""
557  base = num//26
558  if base:
559  result += chars[base].upper()
560  remainder = num % 26
561  result += chars[remainder].upper()
562  return result
563  elif vobj.NumberingStyle == "a,b,c":
564  result = ""
565  base = num//26
566  if base:
567  result += chars[base]
568  remainder = num % 26
569  result += chars[remainder]
570  return result
571  elif vobj.NumberingStyle == "I,II,III":
572  result = ""
573  n = num
574  n += 1
575  for numeral, integer in roman:
576  while n >= integer:
577  result += numeral
578  n -= integer
579  return result
580  elif vobj.NumberingStyle == "L0,L1,L2":
581  return "L"+str(num)
582  else:
583  return str(num+1)
