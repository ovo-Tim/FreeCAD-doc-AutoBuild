102  def mirror(self, p1, p2, copy=False):
103  """Mirror the real shapes."""
104  sel = '['
105  for o in self.sel:
106  if len(sel) > 1:
107  sel += ', '
108  sel += 'FreeCAD.ActiveDocument.' + o.Name
109  sel += ']'
110  Gui.addModule("Draft")

122  def action(self, arg):
123  """Handle the 3D scene events.
124 
125  This is installed as an EventCallback in the Inventor view.
126 
127  Parameters
128  ----------
129  arg: dict
130  Dictionary with strings that indicates the type of event received
131  from the 3D view.
132  """
133  if arg["Type"] == "SoKeyboardEvent":
134  if arg["Key"] == "ESCAPE":
135  self.finish()
136  elif arg["Type"] == "SoLocation2Event": # mouse movement detection
137  self.point, ctrlPoint, info = gui_tool_utils.getPoint(self, arg)
138  if len(self.node) > 0:
139  last = self.node[-1]
140  if self.ghost:
141  tol = 1e-7
142  if self.point.sub(last).Length > tol:
143  # The normal of the mirror plane must be same as in mirror.py.
144  nor = self.point.sub(last).cross(self.wp.axis)
145  if nor.Length > tol:
146  nor.normalize()
147  mtx = DraftGeomUtils.mirror_matrix(App.Matrix(), last, nor)
148  self.ghost.setMatrix(mtx) # Ignores the position of the matrix.
149  self.ghost.move(App.Vector(mtx.col(3)[:3]))
150  if self.extendedCopy:
151  if not gui_tool_utils.hasMod(arg, gui_tool_utils.get_mod_alt_key()):
152  self.finish()
153  gui_tool_utils.redraw3DView()
154  elif arg["Type"] == "SoMouseButtonEvent":
155  if (arg["State"] == "DOWN") and (arg["Button"] == "BUTTON1"):
156  if self.point:
157  self.ui.redraw()
158  if (self.node == []):
159  self.node.append(self.point)
160  self.ui.isRelative.show()
161  if self.ghost:
162  self.ghost.on()
163  _toolmsg(translate("draft",
164  "Pick end point of mirror line"))
165  if self.planetrack:
166  self.planetrack.set(self.point)
167  else:
168  last = self.node[0]
169  if (self.ui.isCopy.isChecked()
170  or gui_tool_utils.hasMod(arg, gui_tool_utils.get_mod_alt_key())):
171  self.mirror(last, self.point, True)
172  else:
173  self.mirror(last, self.point)
174  if gui_tool_utils.hasMod(arg, gui_tool_utils.get_mod_alt_key()):
175  self.extendedCopy = True
176  else:
177  self.finish()
