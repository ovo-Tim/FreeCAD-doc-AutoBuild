287 void TaskFemConstraintTransform::addToSelection()
288 {
289  int rows = ui->lw_Rect->model()->rowCount();
290  std::vector<Gui::SelectionObject> selection =
291  Gui::Selection().getSelectionEx(); // gets vector of selected objects of active document
292  if (selection.empty()) {
293  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
294  return;
295  }
296 
297  if (rows == 1) {
298  QMessageBox::warning(this,
299  tr("Selection error"),
300  tr("Only one face for rectangular local coordinate system!"));
301  Gui::Selection().clearSelection();
302  return;
303  }
304 
305  if ((rows == 0) && (selection.size() >= 2)) {
306  QMessageBox::warning(this,
307  tr("Selection error"),
308  tr("Only one face for rectangular local coordinate system!"));
309  Gui::Selection().clearSelection();
310  return;
311  }
312 
313  Fem::ConstraintTransform* pcConstraint = ConstraintView->getObject<Fem::ConstraintTransform>();
314  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
315  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
316 
317  std::vector<App::DocumentObject*> ObjDispl = pcConstraint->RefDispl.getValues();
318  std::vector<std::string> SubElemDispl = pcConstraint->RefDispl.getSubValues();
319  for (auto& it : selection) { // for every selected object
320  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
321  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
322  return;
323  }
324  const std::vector<std::string>& subNames = it.getSubNames();
325  App::DocumentObject* obj = it.getObject();
326  if (subNames.size() != 1) {
327  QMessageBox::warning(this,
328  tr("Selection error"),
329  tr("Only one face for local coordinate system!"));
330  Gui::Selection().clearSelection();
331  return;
332  }
333  for (const auto& subName : subNames) { // for every selected sub element
334  bool addMe = true;
335  if (subName.substr(0, 4) != "Face") {
336  QMessageBox::warning(this, tr("Selection error"), tr("Only faces can be picked"));
337  return;
338  }
339  if (subName.substr(0, 4) == "Face") {
340  if (ui->rb_cylin->isChecked()) {
341  Part::Feature* feat = static_cast<Part::Feature*>(obj);
342  TopoDS_Shape ref = feat->Shape.getShape().getSubShape(subName.c_str());
343  BRepAdaptor_Surface surface(TopoDS::Face(ref));
344  if (surface.GetType() != GeomAbs_Cylinder) {
345  QMessageBox::warning(this,
346  tr("Selection error"),
347  tr("Only cylindrical faces can be picked"));
348  return;
349  }
350  }
351  }
352  for (std::vector<std::string>::iterator itr =
353  std::find(SubElements.begin(), SubElements.end(), subName);
354  itr != SubElements.end();
355  itr = std::find(++itr,
356  SubElements.end(),
357  subName)) { // for every sub element in selection that
358  // matches one in old list
359  if (obj
360  == Objects[std::distance(
361  SubElements.begin(),
362  itr)]) { // if selected sub element's object equals the one in old list
363  // then it was added before so don't add
364  addMe = false;
365  }
366  }
367  if (addMe) {
368  disconnect(ui->lw_Rect,
369  &QListWidget::currentItemChanged,
370  this,
371  &TaskFemConstraintTransform::setSelection);
372  for (std::size_t i = 0; i < ObjDispl.size(); i++) {
373  if ((makeRefText(ObjDispl[i], SubElemDispl[i]))
374  == (makeRefText(obj, subName))) {
375  Objects.push_back(obj);
376  SubElements.push_back(subName);
377  ui->lw_Rect->addItem(makeRefText(obj, subName));
378  connect(ui->lw_Rect,
379  &QListWidget::currentItemChanged,
380  this,
381  &TaskFemConstraintTransform::setSelection);
382  }
383  }
384  if (Objects.empty()) {
385  QMessageBox::warning(
386  this,
387  tr("Selection error"),
388  tr("Only transformable faces can be selected! Apply displacement boundary "
389  "condition to surface first then apply local coordinate system to "
390  "surface"));
391  Gui::Selection().clearSelection();
392  return;
393  }
394  }
395  }
396  }
397  // Update UI
398  pcConstraint->References.setValues(Objects, SubElements);
399  updateUI();
400  if (ui->rb_rect->isChecked()) {
401  Base::Vector3d normal = pcConstraint->NormalDirection.getValue();
402  Base::Rotation rot(Base::Vector3d(0, 0, 1), normal);
403  Base::Vector3d axis;
404  double angle;
405  rot.getValue(axis, angle);
406  ui->spb_rot_axis_x->setValue(axis.x);
407  ui->spb_rot_axis_y->setValue(axis.y);
408  ui->spb_rot_axis_z->setValue(axis.z);
409  Base::Quantity rotAngle(angle, "rad");
410  ui->qsb_rot_angle->setValue(rotAngle.getValueAs(Base::Quantity::Degree));
411  }
412 }

414 void TaskFemConstraintTransform::removeFromSelection()
415 {
416  std::vector<Gui::SelectionObject> selection =
417  Gui::Selection().getSelectionEx(); // gets vector of selected objects of active document
418  if (selection.empty()) {
419  QMessageBox::warning(this, tr("Selection error"), tr("Nothing selected!"));
420  return;
421  }
422  Fem::ConstraintTransform* pcConstraint = ConstraintView->getObject<Fem::ConstraintTransform>();
423  std::vector<App::DocumentObject*> Objects = pcConstraint->References.getValues();
424  std::vector<std::string> SubElements = pcConstraint->References.getSubValues();
425  std::vector<size_t> itemsToDel;
426  for (const auto& it : selection) { // for every selected object
427  if (!it.isObjectTypeOf(Part::Feature::getClassTypeId())) {
428  QMessageBox::warning(this, tr("Selection error"), tr("Selected object is not a part!"));
429  return;
430  }
431  const std::vector<std::string>& subNames = it.getSubNames();
432  const App::DocumentObject* obj = it.getObject();
433 
434  for (const auto& subName : subNames) { // for every selected sub element
435  for (std::vector<std::string>::iterator itr =
436  std::find(SubElements.begin(), SubElements.end(), subName);
437  itr != SubElements.end();
438  itr = std::find(++itr,
439  SubElements.end(),
440  subName)) { // for every sub element in selection that
441  // matches one in old list
442  if (obj
443  == Objects[std::distance(
444  SubElements.begin(),
445  itr)]) { // if selected sub element's object equals the one in old list
446  // then it was added before so mark for deletion
447  itemsToDel.push_back(std::distance(SubElements.begin(), itr));
448  }
449  }
450  }
451  }
452  std::sort(itemsToDel.begin(), itemsToDel.end());
453  while (!itemsToDel.empty()) {
454  Objects.erase(Objects.begin() + itemsToDel.back());
455  SubElements.erase(SubElements.begin() + itemsToDel.back());
456  itemsToDel.pop_back();
457  }
458  // Update UI
459  {
460  QSignalBlocker block(ui->lw_Rect);
461  ui->lw_Rect->clear();
462  }
463  pcConstraint->References.setValues(Objects, SubElements);
464  updateUI();
465  ui->spb_rot_axis_x->setValue(0);
466  ui->spb_rot_axis_y->setValue(0);
467  ui->spb_rot_axis_z->setValue(1);
468  ui->qsb_rot_angle->setValue(0);
469 }
