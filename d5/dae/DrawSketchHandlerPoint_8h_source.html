232 template<>
233 void DSHPointController::addConstraints()
234 {
235  int firstCurve = handler->getHighestCurveIndex();
236 
237  auto x0 = onViewParameters[OnViewParameter::First]->getValue();
238  auto y0 = onViewParameters[OnViewParameter::Second]->getValue();
239 
240  auto x0set = onViewParameters[OnViewParameter::First]->isSet;
241  auto y0set = onViewParameters[OnViewParameter::Second]->isSet;
242 
243  using namespace Sketcher;
244 
245  if (x0set && y0set && x0 == 0. && y0 == 0.) {
246  ConstraintToAttachment(GeoElementId(firstCurve, PointPos::start),
247  GeoElementId::RtPnt,
248  x0,
249  handler->sketchgui->getObject());
250  }
251  else {
252  if (x0set) {
253  ConstraintToAttachment(GeoElementId(firstCurve, PointPos::start),
254  GeoElementId::VAxis,
255  x0,
256  handler->sketchgui->getObject());
257  }
258 
259  if (y0set) {
260  ConstraintToAttachment(GeoElementId(firstCurve, PointPos::start),
261  GeoElementId::HAxis,
262  y0,
263  handler->sketchgui->getObject());
264  }
265  }
