37 def CreateRectangleSketch(SketchFeature, corner, lengths):
38  hmin, hmax = corner[0], corner[0] + lengths[0]
39  vmin, vmax = corner[1], corner[1] + lengths[1]
40 
41  # add the geometry and grab the count offset
42  i = int(SketchFeature.GeometryCount)
43  SketchFeature.addGeometry(
44  Part.LineSegment(FreeCAD.Vector(hmin, vmax), FreeCAD.Vector(hmax, vmax, 0))
45  )
46  SketchFeature.addGeometry(
47  Part.LineSegment(FreeCAD.Vector(hmax, vmax, 0), FreeCAD.Vector(hmax, vmin, 0))
48  )
49  SketchFeature.addGeometry(
50  Part.LineSegment(FreeCAD.Vector(hmax, vmin, 0), FreeCAD.Vector(hmin, vmin, 0))
51  )
52  SketchFeature.addGeometry(
53  Part.LineSegment(FreeCAD.Vector(hmin, vmin, 0), FreeCAD.Vector(hmin, vmax, 0))
54  )
55 
56  # add the rectangular constraints
57  SketchFeature.addConstraint(Sketcher.Constraint("Coincident", i + 0, 2, i + 1, 1))
58  SketchFeature.addConstraint(Sketcher.Constraint("Coincident", i + 1, 2, i + 2, 1))
59  SketchFeature.addConstraint(Sketcher.Constraint("Coincident", i + 2, 2, i + 3, 1))
60  SketchFeature.addConstraint(Sketcher.Constraint("Coincident", i + 3, 2, i + 0, 1))
61  SketchFeature.addConstraint(Sketcher.Constraint("Horizontal", i + 0))
62  SketchFeature.addConstraint(Sketcher.Constraint("Horizontal", i + 2))
63  SketchFeature.addConstraint(Sketcher.Constraint("Vertical", i + 1))
64  SketchFeature.addConstraint(Sketcher.Constraint("Vertical", i + 3))
65 
66  # Fix the bottom left corner of the rectangle
67  SketchFeature.addConstraint(Sketcher.Constraint("DistanceX", i + 2, 2, corner[0]))
68  SketchFeature.addConstraint(Sketcher.Constraint("DistanceY", i + 2, 2, corner[1]))
69 
70  # add dimensions
71  if lengths[0] == lengths[1]:
72  SketchFeature.addConstraint(Sketcher.Constraint("Equal", i + 2, i + 3))
73  SketchFeature.addConstraint(Sketcher.Constraint("Distance", i + 0, hmax - hmin))
74  else:
75  SketchFeature.addConstraint(Sketcher.Constraint("Distance", i + 1, vmax - vmin))
76  SketchFeature.addConstraint(Sketcher.Constraint("Distance", i + 0, hmax - hmin))

454  def testCircleToLineDistance_Reference_Secant(self):
455  sketch = self.Doc.addObject("Sketcher::SketchObject", "Sketch")
456  radius = 20
457  c_idx = sketch.addGeometry(Part.Circle(vec(0, 0), xy_normal, radius))
458  l_idx = sketch.addGeometry(
459  Part.LineSegment(vec(-radius, radius / 2), vec(radius, radius / 2))
460  )
461  # The block constraints are required to ensure the geometry does not move.
462  # Without this, the solver may find another valid solution than what we assert.
463  sketch.addConstraint(
464  [Sketcher.Constraint("Block", c_idx), Sketcher.Constraint("Block", l_idx)]
465  )
466  # use a negative distance to tell "line is within the circle"
467  expected_distance = -radius / 2 # note that we don't set this in the constraint below!
468  # TODO: addConstraint(constraint) triggers a solve (for godd reasons) however, this way
469  # one cannot add non-driving constraints. In contrast, addConstraint(list(constraint))
470  # does not solve automatically, thus we use this "overload".
471  # Much nicer would be an addConstraint(constraint, isReference=False), like addGeometry
472  dist_idx = sketch.addConstraint([Sketcher.Constraint("Distance", c_idx, l_idx, 0)])[0]
473  sketch.setDriving(dist_idx, False)
474  self.assertSuccessfulSolve(sketch)
475  actual_distance = sketch.Constraints[dist_idx].Value
476  self.assertAlmostEqual(
477  expected_distance,
