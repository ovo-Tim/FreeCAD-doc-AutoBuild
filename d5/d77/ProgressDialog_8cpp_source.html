84 void SequencerDialog::startStep()
85 {
86  QThread *currentThread = QThread::currentThread();
87  QThread *thr = d->dlg->thread(); // this is the main thread
88  if (thr != currentThread) {
89  d->guiThread = false;
90  QMetaObject::invokeMethod(d->dlg, "setRangeEx", Qt::QueuedConnection,
91  Q_ARG(int, 0), Q_ARG(int, (int)nTotalSteps));
92  d->dlg->setModal(false);
93  if (nTotalSteps == 0) {
94  d->progressTime.start();
95  }
96 
97  d->measureTime.start();
98  QMetaObject::invokeMethod(d->dlg, "setValueEx", Qt::QueuedConnection,
99  Q_ARG(int,0));
100  QMetaObject::invokeMethod(d->dlg, "aboutToShow", Qt::QueuedConnection);
101  }
102  else {
103  d->guiThread = true;
104  d->dlg->setRangeEx(0, (int)nTotalSteps);
105  d->dlg->setModal(true);
106  if (nTotalSteps == 0) {
107  d->progressTime.start();
108  }
109 
110  d->measureTime.start();
111  d->dlg->setValueEx(0);
112  d->dlg->aboutToShow();
113  }

116 void SequencerDialog::nextStep(bool canAbort)
117 {
118  d->canabort = canAbort;
119  QThread *currentThread = QThread::currentThread();
120  QThread *thr = d->dlg->thread(); // this is the main thread
121  if (thr != currentThread) {
122  if (wasCanceled() && canAbort) {
123  abort();
124  }
125  else {
126  setValue((int)nProgress + 1);
127  }
128  }
129  else {
130  if (wasCanceled() && canAbort) {
131  // restore cursor
132  pause();
133  bool ok = d->dlg->canAbort();
134  // continue and show up wait cursor if needed
135  resume();
136 
137  // force to abort the operation
138  if ( ok ) {
139  abort();
140  } else {
141  rejectCancel();
142  setValue((int)nProgress+1);
143  }
144  }
145  else {
146  setValue((int)nProgress+1);
147  }
148  }
149 }

165 void SequencerDialog::setValue(int step)
166 {
167  QThread *currentThread = QThread::currentThread();
168  QThread *thr = d->dlg->thread(); // this is the main thread
169  // if number of total steps is unknown then increment only by one
170  if (nTotalSteps == 0) {
171  int elapsed = d->progressTime.elapsed();
172  // allow an update every 500 milliseconds only
173  if (elapsed > 500) {
174  d->progressTime.restart();
175  if (thr != currentThread) {
176  QMetaObject::invokeMethod(d->dlg, "setValueEx", Qt::/*Blocking*/QueuedConnection,
177  Q_ARG(int,d->dlg->value()+1));
178  }
179  else {
180  d->dlg->setValueEx(d->dlg->value()+1);
181  qApp->processEvents();
182  }
183  }
184  }
185  else {
186  if (thr != currentThread) {
187  QMetaObject::invokeMethod(d->dlg, "setValueEx", Qt::/*Blocking*/QueuedConnection,
188  Q_ARG(int,step));
189  if (d->dlg->isVisible())
190  showRemainingTime();
191  }
192  else {
193  d->dlg->setValueEx(step);
194  if (d->dlg->isVisible())
195  showRemainingTime();
196  qApp->processEvents();
197  }
198  }

201 void SequencerDialog::showRemainingTime()
202 {
203  QThread *currentThread = QThread::currentThread();
204  QThread *thr = d->dlg->thread(); // this is the main thread
205 
206  int elapsed = d->measureTime.elapsed();
207  int progress = d->dlg->value();
208  int totalSteps = d->dlg->maximum() - d->dlg->minimum();
209 
210  QString txt = d->text;
211  // More than 5 percent complete or more than 5 secs have elapsed.
212  if (progress * 20 > totalSteps || elapsed > 5000) {
213  int rest = (int) ( (double) totalSteps/progress * elapsed ) - elapsed;
214 
215  // more than 1 secs have elapsed and at least 100 ms are remaining
216  if (elapsed > 1000 && rest > 100) {
217  QTime time( 0,0, 0);
218  time = time.addSecs( rest/1000 );
219  QString remain = Gui::ProgressDialog::tr("Remaining: %1").arg(time.toString());
220  QString status = QString::fromLatin1("%1\t[%2]").arg(txt, remain);
221 
222  if (thr != currentThread) {
223  QMetaObject::invokeMethod(d->dlg, "setLabelText",
224  Qt::/*Blocking*/QueuedConnection,
225  Q_ARG(QString,status));
226  }
227  else {
228  d->dlg->setLabelText(status);
229  }
230  }
231  }

234 void SequencerDialog::resetData()
235 {
236  QThread *currentThread = QThread::currentThread();
237  QThread *thr = d->dlg->thread(); // this is the main thread
238  if (thr != currentThread) {
239  QMetaObject::invokeMethod(d->dlg, "resetEx", Qt::QueuedConnection);
240  QMetaObject::invokeMethod(d->dlg, "hide", Qt::QueuedConnection);
241  QMetaObject::invokeMethod(d->dlg, "setLabelText",
242  Qt::/*Blocking*/QueuedConnection,
243  Q_ARG(QString,QString()));
244  }
245  else {
246  d->dlg->resetEx();
247  // Note: Under Qt 4.1.4 this forces to run QWindowsStyle::eventFilter() twice
248  // handling the same event thus a warning is printed. Possibly, this is a bug
249  // in Qt. The message is QEventDispatcherUNIX::unregisterTimer: invalid argument.
250  d->dlg->hide();
251  d->dlg->setLabelText(QString());
252  }
