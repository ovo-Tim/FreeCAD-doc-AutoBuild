50 class ExternalSelection: public Gui::SelectionFilterGate
51 {
52  App::DocumentObject* object;
53 
54 public:

56  : Gui::SelectionFilterGate(nullPointer())
57  , object(obj)

60  bool allow(App::Document* pDoc, App::DocumentObject* pObj, const char* sSubName) override
61  {
62  Sketcher::SketchObject* sketch = static_cast<Sketcher::SketchObject*>(object);
63 
64  this->notAllowedReason = "";
65  Sketcher::SketchObject::eReasonList msg;
66  if (!sketch->isExternalAllowed(pDoc, pObj, &msg)) {
67  switch (msg) {
68  case Sketcher::SketchObject::rlCircularReference:
69  this->notAllowedReason =
70  QT_TR_NOOP("Linking this will cause circular dependency.");
71  break;
72  case Sketcher::SketchObject::rlOtherDoc:
73  this->notAllowedReason = QT_TR_NOOP("This object is in another document.");
74  break;
75  case Sketcher::SketchObject::rlOtherBody:
76  this->notAllowedReason =
77  QT_TR_NOOP("This object belongs to another body, can't link.");
78  break;
79  case Sketcher::SketchObject::rlOtherPart:
80  this->notAllowedReason =
81  QT_TR_NOOP("This object belongs to another part, can't link.");
82  break;
83  default:
84  break;
85  }
86  return false;
87  }
88 
89  // Note: its better to search the support of the sketch in case the sketch support is a base
90  // plane
91  // Part::BodyBase* body = Part::BodyBase::findBodyOf(sketch);
92  // if ( body && body->hasFeature ( pObj ) && body->isAfter ( pObj, sketch ) ) {
93  // Don't allow selection after the sketch in the same body
94  // NOTE: allowness of features in other bodies is handled by
95  // SketchObject::isExternalAllowed()
96  // TODO may be this should be in SketchObject::isExternalAllowed() (2015-08-07, Fat-Zer)
97  // return false;
98  //}
99 
100  if (!sSubName || sSubName[0] == '\0') {
101  return false;
102  }
103  std::string element(sSubName);
104  if ((element.size() > 4 && element.substr(0, 4) == "Edge")
105  || (element.size() > 6 && element.substr(0, 6) == "Vertex")
106  || (element.size() > 4 && element.substr(0, 4) == "Face")) {
107  return true;
108  }
109  if (pObj->isDerivedFrom<App::Plane>() || pObj->isDerivedFrom<Part::Datum>()) {
110  return true;
111  }
112  return false;
113  }

143  {
144  Q_UNUSED(onSketchPos);
145  /* this is ok not to call to purgeHandler
146  * in continuous creation mode because the
147  * handler is destroyed by the quit() method on pressing the
148  * right button of the mouse */
149  return true;

152  bool onSelectionChanged(const Gui::SelectionChanges& msg) override
153  {
154  if (msg.Type == Gui::SelectionChanges::AddSelection) {
155  App::DocumentObject* obj =
156  sketchgui->getObject()->getDocument()->getObject(msg.pObjectName);
157  if (!obj) {
158  throw Base::ValueError("Sketcher: External geometry: Invalid object in selection");
159  }
160  std::string subName(msg.pSubName);
161  if (obj->isDerivedFrom<App::Plane>() || obj->isDerivedFrom<Part::Datum>()
162  || (subName.size() > 4 && subName.substr(0, 4) == "Edge")
163  || (subName.size() > 6 && subName.substr(0, 6) == "Vertex")
164  || (subName.size() > 4 && subName.substr(0, 4) == "Face")) {
165  try {
166  Gui::Command::openCommand(
167  QT_TRANSLATE_NOOP("Command", "Add external geometry"));
168  Gui::cmdAppObjectArgs(sketchgui->getObject(),
169  "addExternal(\"%s\",\"%s\", %s, %s)",
170  msg.pObjectName,
171  msg.pSubName,
172  alwaysReference || isConstructionMode() ? "False"
173  : "True",
174  intersection ? "True" : "False");
175 
176  Gui::Command::commitCommand();
177 
178  // adding external geometry does not require a solve() per se (the DoF is the
179  // same), however a solve is required to update the amount of solver geometry,
180  // because we only redraw a changed Sketch if the solver geometry amount is the
181  // same as the SkethObject geometry amount (as this avoids other issues). This
182  // solver is a very low cost one anyway (there is actually nothing to solve).
183  tryAutoRecomputeIfNotSolve(sketchgui->getObject<Sketcher::SketchObject>());
184 
185  Gui::Selection().clearSelection();
186  /* this is ok not to call to purgeHandler
187  * in continuous creation mode because the
188  * handler is destroyed by the quit() method on pressing the
189  * right button of the mouse */
190  }
191  catch (const Base::Exception&) {
192  Gui::NotifyError(
193  sketchgui,
194  QT_TRANSLATE_NOOP("Notifications", "Error"),
195  QT_TRANSLATE_NOOP("Notifications", "Failed to add external geometry"));
196  Gui::Selection().clearSelection();
197  Gui::Command::abortCommand();
198  }
199  return true;
200  }
201  }
202  return false;
203  }

217  }
