57 bool MeshTopoAlgorithm::InsertVertex(FacetIndex ulFacetPos, const Base::Vector3f& rclPoint)
58 {
59  MeshFacet& rclF = _rclMesh._aclFacetArray[ulFacetPos];
60  MeshFacet clNewFacet1, clNewFacet2;
61 
62  // insert new point
63  PointIndex ulPtCnt = _rclMesh._aclPointArray.size();
64  PointIndex ulPtInd = this->GetOrAddIndex(rclPoint);
65  FacetIndex ulSize = _rclMesh._aclFacetArray.size();
66 
67  if (ulPtInd < ulPtCnt) {
68  return false; // the given point is already part of the mesh => creating new facets would
69  // be an illegal operation
70  }
71 
72  // adjust the facets
73  //
74  // first new facet
75  clNewFacet1._aulPoints[0] = rclF._aulPoints[1];
76  clNewFacet1._aulPoints[1] = rclF._aulPoints[2];
77  clNewFacet1._aulPoints[2] = ulPtInd;
78  clNewFacet1._aulNeighbours[0] = rclF._aulNeighbours[1];
79  clNewFacet1._aulNeighbours[1] = ulSize + 1;
80  clNewFacet1._aulNeighbours[2] = ulFacetPos;
81  // second new facet
82  clNewFacet2._aulPoints[0] = rclF._aulPoints[2];
83  clNewFacet2._aulPoints[1] = rclF._aulPoints[0];
84  clNewFacet2._aulPoints[2] = ulPtInd;
85  clNewFacet2._aulNeighbours[0] = rclF._aulNeighbours[2];
86  clNewFacet2._aulNeighbours[1] = ulFacetPos;
87  clNewFacet2._aulNeighbours[2] = ulSize;
88  // adjust the neighbour facet
89  if (rclF._aulNeighbours[1] != FACET_INDEX_MAX) {
90  _rclMesh._aclFacetArray[rclF._aulNeighbours[1]].ReplaceNeighbour(ulFacetPos, ulSize);
91  }
92  if (rclF._aulNeighbours[2] != FACET_INDEX_MAX) {
93  _rclMesh._aclFacetArray[rclF._aulNeighbours[2]].ReplaceNeighbour(ulFacetPos, ulSize + 1);
94  }
95  // original facet
96  rclF._aulPoints[2] = ulPtInd;
97  rclF._aulNeighbours[1] = ulSize;
98  rclF._aulNeighbours[2] = ulSize + 1;
99 
100  // insert new facets
101  _rclMesh._aclFacetArray.push_back(clNewFacet1);
102  _rclMesh._aclFacetArray.push_back(clNewFacet2);
103 
104  return true;

107 bool MeshTopoAlgorithm::SnapVertex(FacetIndex ulFacetPos, const Base::Vector3f& rP)
108 {
109  MeshFacet& rFace = _rclMesh._aclFacetArray[ulFacetPos];
110  if (!rFace.HasOpenEdge()) {
111  return false;
112  }
113  Base::Vector3f cNo1 = _rclMesh.GetNormal(rFace);
114  for (unsigned short i = 0; i < 3; i++) {
115  if (rFace._aulNeighbours[i] == FACET_INDEX_MAX) {
116  const Base::Vector3f& rPt1 = _rclMesh._aclPointArray[rFace._aulPoints[i]];
117  const Base::Vector3f& rPt2 = _rclMesh._aclPointArray[rFace._aulPoints[(i + 1) % 3]];
118  Base::Vector3f cNo2 = (rPt2 - rPt1) % cNo1;
119  Base::Vector3f cNo3 = (rP - rPt1) % (rPt2 - rPt1);
120  float fD2 = Base::DistanceP2(rPt1, rPt2);
121  float fTV = (rP - rPt1) * (rPt2 - rPt1);
122 
123  // Point is on the edge
124  if (cNo3.Length() < FLOAT_EPS) {
125  return SplitOpenEdge(ulFacetPos, i, rP);
126  }
127  if ((rP - rPt1) * cNo2 > 0.0F && fD2 >= fTV && fTV >= 0.0F) {
128  MeshFacet cTria;
129  cTria._aulPoints[0] = this->GetOrAddIndex(rP);
130  cTria._aulPoints[1] = rFace._aulPoints[(i + 1) % 3];
131  cTria._aulPoints[2] = rFace._aulPoints[i];
132  cTria._aulNeighbours[1] = ulFacetPos;
133  rFace._aulNeighbours[i] = _rclMesh.CountFacets();
134  _rclMesh._aclFacetArray.push_back(cTria);
135  return true;
136  }
137  }
138  }
139 
140  return false;

143 void MeshTopoAlgorithm::OptimizeTopology(float fMaxAngle)
144 {
145  // For each internal edge get the adjacent facets. When doing an edge swap we must update
146  // this structure.
147  std::map<std::pair<PointIndex, PointIndex>, std::vector<FacetIndex>> aEdge2Face;
148  for (auto pI = _rclMesh._aclFacetArray.begin(); pI != _rclMesh._aclFacetArray.end(); ++pI) {
149  for (int i = 0; i < 3; i++) {
150  // ignore open edges
151  if (pI->_aulNeighbours[i] != FACET_INDEX_MAX) {
152  PointIndex ulPt0 =
153  std::min<PointIndex>(pI->_aulPoints[i], pI->_aulPoints[(i + 1) % 3]);
154  PointIndex ulPt1 =
155  std::max<PointIndex>(pI->_aulPoints[i], pI->_aulPoints[(i + 1) % 3]);
156  aEdge2Face[std::pair<PointIndex, PointIndex>(ulPt0, ulPt1)].push_back(
157  pI - _rclMesh._aclFacetArray.begin());
158  }
159  }
160  }
161 
162  // fill up this list with all internal edges and perform swap edges until this list is empty
163  std::list<std::pair<PointIndex, PointIndex>> aEdgeList;
164  std::map<std::pair<PointIndex, PointIndex>, std::vector<FacetIndex>>::iterator pE;
165  for (pE = aEdge2Face.begin(); pE != aEdge2Face.end(); ++pE) {
166  if (pE->second.size() == 2) { // make sure that we really have an internal edge
167  aEdgeList.push_back(pE->first);
168  }
169  }
170 
171  // to be sure to avoid an endless loop
172  unsigned long uMaxIter = 5 * aEdge2Face.size();
173 
174  // Perform a swap edge where needed
175  while (!aEdgeList.empty() && uMaxIter > 0) {
176  // get the first edge and remove it from the list
177  std::pair<PointIndex, PointIndex> aEdge = aEdgeList.front();
178  aEdgeList.pop_front();
179  uMaxIter--;
180 
181  // get the adjacent facets to this edge
182  pE = aEdge2Face.find(aEdge);
183 
184  // this edge has been removed some iterations before
185  if (pE == aEdge2Face.end()) {
186  continue;
187  }
188 
189  // Is swap edge allowed and sensible?
190  if (!ShouldSwapEdge(pE->second[0], pE->second[1], fMaxAngle)) {
191  continue;
192  }
193 
194  // ok, here we should perform a swap edge to minimize the maximum angle
195  if (/*fMax12 > fMax34*/ true) {
196  // swap the edge
197  SwapEdge(pE->second[0], pE->second[1]);
198 
199  MeshFacet& rF1 = _rclMesh._aclFacetArray[pE->second[0]];
200  MeshFacet& rF2 = _rclMesh._aclFacetArray[pE->second[1]];
201  unsigned short side1 = rF1.Side(aEdge.first, aEdge.second);
202  unsigned short side2 = rF2.Side(aEdge.first, aEdge.second);
203 
204  // adjust the edge list
205  for (int i = 0; i < 3; i++) {
206  std::map<std::pair<PointIndex, PointIndex>, std::vector<FacetIndex>>::iterator it;
207  // first facet
208  PointIndex ulPt0 =
209  std::min<PointIndex>(rF1._aulPoints[i], rF1._aulPoints[(i + 1) % 3]);
210  PointIndex ulPt1 =
211  std::max<PointIndex>(rF1._aulPoints[i], rF1._aulPoints[(i + 1) % 3]);
212  it = aEdge2Face.find(std::make_pair(ulPt0, ulPt1));
213  if (it != aEdge2Face.end()) {
214  if (it->second[0] == pE->second[1]) {
215  it->second[0] = pE->second[0];
216  }
217  else if (it->second[1] == pE->second[1]) {
218  it->second[1] = pE->second[0];
219  }
220  aEdgeList.push_back(it->first);
221  }
222 
223  // second facet
224  ulPt0 = std::min<PointIndex>(rF2._aulPoints[i], rF2._aulPoints[(i + 1) % 3]);
225  ulPt1 = std::max<PointIndex>(rF2._aulPoints[i], rF2._aulPoints[(i + 1) % 3]);
226  it = aEdge2Face.find(std::make_pair(ulPt0, ulPt1));
227  if (it != aEdge2Face.end()) {
228  if (it->second[0] == pE->second[0]) {
229  it->second[0] = pE->second[1];
230  }
231  else if (it->second[1] == pE->second[0]) {
232  it->second[1] = pE->second[1];
233  }
234  aEdgeList.push_back(it->first);
235  }
236  }
237 
238  // Now we must remove the edge and replace it through the new edge
239  PointIndex ulPt0 = std::min<PointIndex>(rF1._aulPoints[(side1 + 1) % 3],
240  rF2._aulPoints[(side2 + 1) % 3]);
241  PointIndex ulPt1 = std::max<PointIndex>(rF1._aulPoints[(side1 + 1) % 3],
242  rF2._aulPoints[(side2 + 1) % 3]);
243  std::pair<PointIndex, PointIndex> aNewEdge = std::make_pair(ulPt0, ulPt1);
244  aEdge2Face[aNewEdge] = pE->second;
245  aEdge2Face.erase(pE);
246  }
247  }
248 }

250 // Cosine of the maximum angle in triangle (v1,v2,v3)
251 static float
252 cos_maxangle(const Base::Vector3f& v1, const Base::Vector3f& v2, const Base::Vector3f& v3)
253 {
254  float a = Base::Distance(v2, v3);
255  float b = Base::Distance(v3, v1);
256  float c = Base::Distance(v1, v2);
257  float A = a * (b * b + c * c - a * a);
258  float B = b * (c * c + a * a - b * b);
259  float C = c * (a * a + b * b - c * c);
260  return 0.5F * std::min<float>(std::min<float>(A, B), C)
261  / (a * b * c); // min cosine == max angle
262 }

264 static float swap_benefit(const Base::Vector3f& v1,
265  const Base::Vector3f& v2,
266  const Base::Vector3f& v3,
267  const Base::Vector3f& v4)
268 {
269  Base::Vector3f n124 = (v4 - v2) % (v1 - v2);
270  Base::Vector3f n234 = (v3 - v2) % (v4 - v2);
271  if ((n124 * n234) <= 0.0F) {
272  return 0.0F; // avoid normal flip
273  }
274 
275  return std::max<float>(-cos_maxangle(v1, v2, v3), -cos_maxangle(v1, v3, v4))
276  - std::max<float>(-cos_maxangle(v1, v2, v4), -cos_maxangle(v2, v3, v4));
277 }

279 float MeshTopoAlgorithm::SwapEdgeBenefit(FacetIndex f, int e) const
280 {
281  const MeshFacetArray& faces = _rclMesh.GetFacets();
282  const MeshPointArray& vertices = _rclMesh.GetPoints();
283 
284  FacetIndex n = faces[f]._aulNeighbours[e];
285  if (n == FACET_INDEX_MAX) {
286  return 0.0F; // border edge
287  }
288 
289  PointIndex v1 = faces[f]._aulPoints[e];
290  PointIndex v2 = faces[f]._aulPoints[(e + 1) % 3];
291  PointIndex v3 = faces[f]._aulPoints[(e + 2) % 3];
292  unsigned short s = faces[n].Side(faces[f]);
293  if (s == USHRT_MAX) {
294  std::cerr << "MeshTopoAlgorithm::SwapEdgeBenefit: error in neighbourhood "
295  << "of faces " << f << " and " << n << std::endl;
296  return 0.0F; // topological error
297  }
298  PointIndex v4 = faces[n]._aulPoints[(s + 2) % 3];
299  if (v3 == v4) {
300  std::cerr << "MeshTopoAlgorithm::SwapEdgeBenefit: duplicate faces " << f << " and " << n
301  << std::endl;
302  return 0.0F; // duplicate faces
303  }
304  return swap_benefit(vertices[v2], vertices[v3], vertices[v1], vertices[v4]);
305 }

310 void MeshTopoAlgorithm::OptimizeTopology()
311 {
312  // Find all edges that can be swapped and insert them into a
313  // priority queue
314  const MeshFacetArray& faces = _rclMesh.GetFacets();
315  FacetIndex nf = _rclMesh.CountFacets();
316  std::priority_queue<FaceEdgePriority> todo;
317  for (FacetIndex i = 0; i < nf; i++) {
318  for (int j = 0; j < 3; j++) {
319  float b = SwapEdgeBenefit(i, j);
320  if (b > 0.0F) {
321  todo.push(std::make_pair(b, std::make_pair(i, j)));
322  }
323  }
324  }
325 
326  // Edges are sorted in decreasing order with respect to their benefit
327  while (!todo.empty()) {
328  FacetIndex f = todo.top().second.first;
329  int e = todo.top().second.second;
330  todo.pop();
331  // Check again if the swap should still be done
332  if (SwapEdgeBenefit(f, e) <= 0.0F) {
333  continue;
334  }
335  // OK, swap the edge
336  FacetIndex f2 = faces[f]._aulNeighbours[e];
337  SwapEdge(f, f2);
338  // Insert new edges into queue, if necessary
339  for (int j = 0; j < 3; j++) {
340  float b = SwapEdgeBenefit(f, j);
341  if (b > 0.0F) {
342  todo.push(std::make_pair(b, std::make_pair(f, j)));
343  }
344  }
345  for (int j = 0; j < 3; j++) {
346  float b = SwapEdgeBenefit(f2, j);
347  if (b > 0.0F) {
348  todo.push(std::make_pair(b, std::make_pair(f2, j)));
349  }
350  }
351  }
352 }

354 void MeshTopoAlgorithm::DelaunayFlip(float fMaxAngle)
355 {
356  // For each internal edge get the adjacent facets.
357  std::set<std::pair<FacetIndex, FacetIndex>> aEdge2Face;
358  FacetIndex index = 0;
359  for (auto pI = _rclMesh._aclFacetArray.begin(); pI != _rclMesh._aclFacetArray.end();
360  ++pI, index++) {
361  for (FacetIndex nbIndex : pI->_aulNeighbours) {
362  // ignore open edges
363  if (nbIndex != FACET_INDEX_MAX) {
364  FacetIndex ulFt0 = std::min<FacetIndex>(index, nbIndex);
365  FacetIndex ulFt1 = std::max<FacetIndex>(index, nbIndex);
366  aEdge2Face.insert(std::pair<FacetIndex, FacetIndex>(ulFt0, ulFt1));
367  }
368  }
369  }
370 
371  Base::Vector3f center;
372  while (!aEdge2Face.empty()) {
373  std::set<std::pair<FacetIndex, FacetIndex>>::iterator it = aEdge2Face.begin();
374  std::pair<FacetIndex, FacetIndex> edge = *it;
375  aEdge2Face.erase(it);
376  if (ShouldSwapEdge(edge.first, edge.second, fMaxAngle)) {
377  float radius = _rclMesh.GetFacet(edge.first).CenterOfCircumCircle(center);
378  radius *= radius;
379  const MeshFacet& face_1 = _rclMesh._aclFacetArray[edge.first];
380  const MeshFacet& face_2 = _rclMesh._aclFacetArray[edge.second];
381  unsigned short side = face_2.Side(edge.first);
382  MeshPoint vertex = _rclMesh.GetPoint(face_2._aulPoints[(side + 1) % 3]);
383  if (Base::DistanceP2(center, vertex) < radius) {
384  SwapEdge(edge.first, edge.second);
385  for (int i = 0; i < 3; i++) {
386  if (face_1._aulNeighbours[i] != FACET_INDEX_MAX
387  && face_1._aulNeighbours[i] != edge.second) {
388  FacetIndex ulFt0 =
389  std::min<FacetIndex>(edge.first, face_1._aulNeighbours[i]);
390  FacetIndex ulFt1 =
391  std::max<FacetIndex>(edge.first, face_1._aulNeighbours[i]);
392  aEdge2Face.insert(std::pair<FacetIndex, FacetIndex>(ulFt0, ulFt1));
393  }
394  if (face_2._aulNeighbours[i] != FACET_INDEX_MAX
395  && face_2._aulNeighbours[i] != edge.first) {
396  FacetIndex ulFt0 =
397  std::min<FacetIndex>(edge.second, face_2._aulNeighbours[i]);
398  FacetIndex ulFt1 =
399  std::max<FacetIndex>(edge.second, face_2._aulNeighbours[i]);
400  aEdge2Face.insert(std::pair<FacetIndex, FacetIndex>(ulFt0, ulFt1));
401  }
402  }
403  }
404  }
405  }

408 int MeshTopoAlgorithm::DelaunayFlip()
409 {
410  int cnt_swap = 0;
411  _rclMesh._aclFacetArray.ResetFlag(MeshFacet::TMP0);
412  size_t cnt_facets = _rclMesh._aclFacetArray.size();
413  for (size_t i = 0; i < cnt_facets; i++) {
414  const MeshFacet& f_face = _rclMesh._aclFacetArray[i];
415  if (f_face.IsFlag(MeshFacet::TMP0)) {
416  continue;
417  }
418  for (int j = 0; j < 3; j++) {
419  FacetIndex n = f_face._aulNeighbours[j];
420  if (n != FACET_INDEX_MAX) {
421  const MeshFacet& n_face = _rclMesh._aclFacetArray[n];
422  if (n_face.IsFlag(MeshFacet::TMP0)) {
423  continue;
424  }
425  unsigned short k = n_face.Side(f_face);
426  MeshGeomFacet f1 = _rclMesh.GetFacet(f_face);
427  MeshGeomFacet f2 = _rclMesh.GetFacet(n_face);
428  Base::Vector3f c1, c2, p1, p2;
429  p1 = f1._aclPoints[(j + 2) % 3];
430  p2 = f2._aclPoints[(k + 2) % 3];
431  float r1 = f1.CenterOfCircumCircle(c1);
432  r1 = r1 * r1;
433  float r2 = f2.CenterOfCircumCircle(c2);
434  r2 = r2 * r2;
435  float d1 = Base::DistanceP2(c1, p2);
436  float d2 = Base::DistanceP2(c2, p1);
437  if (d1 < r1 || d2 < r2) {
438  SwapEdge(i, n);
439  cnt_swap++;
440  f_face.SetFlag(MeshFacet::TMP0);
441  n_face.SetFlag(MeshFacet::TMP0);
442  }
443  }
444  }
445  }
446 
447  return cnt_swap;

450 void MeshTopoAlgorithm::AdjustEdgesToCurvatureDirection()
451 {
452  std::vector<Wm4::Vector3<float>> aPnts;
453  MeshPointIterator cPIt(_rclMesh);
454  aPnts.reserve(_rclMesh.CountPoints());
455  for (cPIt.Init(); cPIt.More(); cPIt.Next()) {
456  aPnts.emplace_back(cPIt->x, cPIt->y, cPIt->z);
457  }
458 
459  // get all point connections
460  std::vector<int> aIdx;
461  const MeshFacetArray& raFts = _rclMesh.GetFacets();
462  aIdx.reserve(3 * raFts.size());
463 
464  // Build map of edges to the referencing facets
465  FacetIndex k = 0;
466  std::map<std::pair<PointIndex, PointIndex>, std::list<FacetIndex>> aclEdgeMap;
467  for (auto jt = raFts.begin(); jt != raFts.end(); ++jt, k++) {
468  for (int i = 0; i < 3; i++) {
469  PointIndex ulT0 = jt->_aulPoints[i];
470  PointIndex ulT1 = jt->_aulPoints[(i + 1) % 3];
471  PointIndex ulP0 = std::min<PointIndex>(ulT0, ulT1);
472  PointIndex ulP1 = std::max<PointIndex>(ulT0, ulT1);
473  aclEdgeMap[std::make_pair(ulP0, ulP1)].push_front(k);
474  aIdx.push_back(static_cast<int>(jt->_aulPoints[i]));
475  }
476  }
477 
478  // compute vertex based curvatures
479  Wm4::MeshCurvature<float> meshCurv(static_cast<int>(_rclMesh.CountPoints()),
480  aPnts.data(),
481  static_cast<int>(_rclMesh.CountFacets()),
482  aIdx.data());
483 
484  // get curvature information now
485  const Wm4::Vector3<float>* aMaxCurvDir = meshCurv.GetMaxDirections();
486  const Wm4::Vector3<float>* aMinCurvDir = meshCurv.GetMinDirections();
487  const float* aMaxCurv = meshCurv.GetMaxCurvatures();
488  const float* aMinCurv = meshCurv.GetMinCurvatures();
489 
490  raFts.ResetFlag(MeshFacet::VISIT);
491  const MeshPointArray& raPts = _rclMesh.GetPoints();
492  for (auto& kt : aclEdgeMap) {
493  if (kt.second.size() == 2) {
494  PointIndex uPt1 = kt.first.first;
495  PointIndex uPt2 = kt.first.second;
496  FacetIndex uFt1 = kt.second.front();
497  FacetIndex uFt2 = kt.second.back();
498 
499  const MeshFacet& rFace1 = raFts[uFt1];
500  const MeshFacet& rFace2 = raFts[uFt2];
501  if (rFace1.IsFlag(MeshFacet::VISIT) || rFace2.IsFlag(MeshFacet::VISIT)) {
502  continue;
503  }
504 
505  PointIndex uPt3 {}, uPt4 {};
506  unsigned short side = rFace1.Side(uPt1, uPt2);
507  uPt3 = rFace1._aulPoints[(side + 2) % 3];
508  side = rFace2.Side(uPt1, uPt2);
509  uPt4 = rFace2._aulPoints[(side + 2) % 3];
510 
511  Wm4::Vector3<float> dir;
512  float fActCurvature {};
513  if (std::fabs(aMinCurv[uPt1]) > std::fabs(aMaxCurv[uPt1])) {
514  fActCurvature = aMinCurv[uPt1];
515  dir = aMaxCurvDir[uPt1];
516  }
517  else {
518  fActCurvature = aMaxCurv[uPt1];
519  dir = aMinCurvDir[uPt1];
520  }
521 
522  Base::Vector3f cMinDir(dir.X(), dir.Y(), dir.Z());
523  Base::Vector3f cEdgeDir1 = raPts[uPt1] - raPts[uPt2];
524  Base::Vector3f cEdgeDir2 = raPts[uPt3] - raPts[uPt4];
525  cMinDir.Normalize();
526  cEdgeDir1.Normalize();
527  cEdgeDir2.Normalize();
528 
529  // get the plane and calculate the distance to the fourth point
530  MeshGeomFacet cPlane(raPts[uPt1], raPts[uPt2], raPts[uPt3]);
531  // positive or negative distance
532  float fDist = raPts[uPt4].DistanceToPlane(cPlane._aclPoints[0], cPlane.GetNormal());
533 
534  float fLength12 = Base::Distance(raPts[uPt1], raPts[uPt2]);
535  float fLength34 = Base::Distance(raPts[uPt3], raPts[uPt4]);
536  if (fabs(cEdgeDir1 * cMinDir) < fabs(cEdgeDir2 * cMinDir)) {
537  if (IsSwapEdgeLegal(uFt1, uFt2) && fLength34 < 1.05F * fLength12
538  && fActCurvature * fDist > 0.0F) {
539  SwapEdge(uFt1, uFt2);
540  rFace1.SetFlag(MeshFacet::VISIT);
541  rFace2.SetFlag(MeshFacet::VISIT);
542  }
543  }
544  }

548 bool MeshTopoAlgorithm::InsertVertexAndSwapEdge(FacetIndex ulFacetPos,

595 bool MeshTopoAlgorithm::IsSwapEdgeLegal(FacetIndex ulFacetPos, FacetIndex ulNeighbour) const
596 {
597  MeshFacet& rclF = _rclMesh._aclFacetArray[ulFacetPos];
598  MeshFacet& rclN = _rclMesh._aclFacetArray[ulNeighbour];
599 
600  unsigned short uFSide = rclF.Side(rclN);
601  unsigned short uNSide = rclN.Side(rclF);
602 
603  if (uFSide == USHRT_MAX || uNSide == USHRT_MAX) {
604  return false; // not neighbours
605  }
606 
607  Base::Vector3f cP1 = _rclMesh._aclPointArray[rclF._aulPoints[uFSide]];
608  Base::Vector3f cP2 = _rclMesh._aclPointArray[rclF._aulPoints[(uFSide + 1) % 3]];
609  Base::Vector3f cP3 = _rclMesh._aclPointArray[rclF._aulPoints[(uFSide + 2) % 3]];
610  Base::Vector3f cP4 = _rclMesh._aclPointArray[rclN._aulPoints[(uNSide + 2) % 3]];
611 
612  // do not allow one to create degenerated triangles
613  MeshGeomFacet cT3(cP4, cP3, cP1);
614  if (cT3.IsDegenerated(MeshDefinitions::_fMinPointDistanceP2)) {
615  return false;
616  }
617  MeshGeomFacet cT4(cP3, cP4, cP2);
618  if (cT4.IsDegenerated(MeshDefinitions::_fMinPointDistanceP2)) {
619  return false;
620  }
621 
622  // We must make sure that the two adjacent triangles builds a convex polygon, otherwise
623  // the swap edge operation is illegal
624  Base::Vector3f cU = cP2 - cP1;
625  Base::Vector3f cV = cP4 - cP3;
626  // build a helper plane through cP1 that must separate cP3 and cP4
627  Base::Vector3f cN1 = (cU % cV) % cU;
628  if (((cP3 - cP1) * cN1) * ((cP4 - cP1) * cN1) >= 0.0F) {
629  return false; // not convex
630  }
631  // build a helper plane through cP3 that must separate cP1 and cP2
632  Base::Vector3f cN2 = (cU % cV) % cV;
633  if (((cP1 - cP3) * cN2) * ((cP2 - cP3) * cN2) >= 0.0F) {
634  return false; // not convex
635  }
636 
637  return true;
638 }

640 bool MeshTopoAlgorithm::ShouldSwapEdge(FacetIndex ulFacetPos,
641  FacetIndex ulNeighbour,
642  float fMaxAngle) const
643 {
644  if (!IsSwapEdgeLegal(ulFacetPos, ulNeighbour)) {
645  return false;
646  }
647 
648  MeshFacet& rclF = _rclMesh._aclFacetArray[ulFacetPos];
649  MeshFacet& rclN = _rclMesh._aclFacetArray[ulNeighbour];
650 
651  unsigned short uFSide = rclF.Side(rclN);
652  unsigned short uNSide = rclN.Side(rclF);
653 
654  Base::Vector3f cP1 = _rclMesh._aclPointArray[rclF._aulPoints[uFSide]];
655  Base::Vector3f cP2 = _rclMesh._aclPointArray[rclF._aulPoints[(uFSide + 1) % 3]];
656  Base::Vector3f cP3 = _rclMesh._aclPointArray[rclF._aulPoints[(uFSide + 2) % 3]];
657  Base::Vector3f cP4 = _rclMesh._aclPointArray[rclN._aulPoints[(uNSide + 2) % 3]];
658 
659  MeshGeomFacet cT1(cP1, cP2, cP3);
660  float fMax1 = cT1.MaximumAngle();
661  MeshGeomFacet cT2(cP2, cP1, cP4);
662  float fMax2 = cT2.MaximumAngle();
663  MeshGeomFacet cT3(cP4, cP3, cP1);
664  float fMax3 = cT3.MaximumAngle();
665  MeshGeomFacet cT4(cP3, cP4, cP2);
666  float fMax4 = cT4.MaximumAngle();
667 
668  // get the angle between the triangles
669  Base::Vector3f cN1 = cT1.GetNormal();

671  if (cN1.GetAngle(cN2) > fMaxAngle) {
672  return false;
673  }

676  float fMax34 = std::max<float>(fMax3, fMax4);
677 
678  return fMax12 > fMax34;

681 void MeshTopoAlgorithm::SwapEdge(FacetIndex ulFacetPos, FacetIndex ulNeighbour)
682 {
683  MeshFacet& rclF = _rclMesh._aclFacetArray[ulFacetPos];
684  MeshFacet& rclN = _rclMesh._aclFacetArray[ulNeighbour];
685 
686  unsigned short uFSide = rclF.Side(rclN);
687  unsigned short uNSide = rclN.Side(rclF);
688 
689  if (uFSide == USHRT_MAX || uNSide == USHRT_MAX) {
690  return; // not neighbours
691  }
692 
693  // adjust the neighbourhood
694  if (rclF._aulNeighbours[(uFSide + 1) % 3] != FACET_INDEX_MAX) {
695  _rclMesh._aclFacetArray[rclF._aulNeighbours[(uFSide + 1) % 3]].ReplaceNeighbour(
696  ulFacetPos,
697  ulNeighbour);

700  _rclMesh._aclFacetArray[rclN._aulNeighbours[(uNSide + 1) % 3]].ReplaceNeighbour(ulNeighbour,
701  ulFacetPos);
702  }
703 
704  // swap the point and neighbour indices
705  rclF._aulPoints[(uFSide + 1) % 3] = rclN._aulPoints[(uNSide + 2) % 3];
706  rclN._aulPoints[(uNSide + 1) % 3] = rclF._aulPoints[(uFSide + 2) % 3];
707  rclF._aulNeighbours[uFSide] = rclN._aulNeighbours[(uNSide + 1) % 3];
708  rclN._aulNeighbours[uNSide] = rclF._aulNeighbours[(uFSide + 1) % 3];
709  rclF._aulNeighbours[(uFSide + 1) % 3] = ulNeighbour;
710  rclN._aulNeighbours[(uNSide + 1) % 3] = ulFacetPos;

714  FacetIndex ulNeighbour,

719 
720  unsigned short uFSide = rclF.Side(rclN);
721  unsigned short uNSide = rclN.Side(rclF);
722 
723  if (uFSide == USHRT_MAX || uNSide == USHRT_MAX) {
724  return false; // not neighbours
725  }
726

728  PointIndex uPtInd = this->GetOrAddIndex(rP);
729  FacetIndex ulSize = _rclMesh._aclFacetArray.size();

732  // be an illegal operation
733  if (uPtInd < uPtCnt) {
734  return false;
735  }
736

738  if (rclF._aulNeighbours[(uFSide + 1) % 3] != FACET_INDEX_MAX) {

751  cNew1._aulNeighbours[0] = ulSize + 1;
752  cNew1._aulNeighbours[1] = rclF._aulNeighbours[(uFSide + 1) % 3];
753  cNew1._aulNeighbours[2] = ulFacetPos;
754 
755  cNew2._aulPoints[0] = rclN._aulPoints[uNSide];
756  cNew2._aulPoints[1] = uPtInd;

758  cNew2._aulNeighbours[0] = ulSize;
759  cNew2._aulNeighbours[1] = ulNeighbour;
760  cNew2._aulNeighbours[2] = rclN._aulNeighbours[(uNSide + 2) % 3];
761 
762  // adjust the facets
763  rclF._aulPoints[(uFSide + 1) % 3] = uPtInd;
764  rclF._aulNeighbours[(uFSide + 1) % 3] = ulSize;
765  rclN._aulPoints[uNSide] = uPtInd;
766  rclN._aulNeighbours[(uNSide + 2) % 3] = ulSize + 1;
767 
768  // insert new facets
769  _rclMesh._aclFacetArray.push_back(cNew1);
770  _rclMesh._aclFacetArray.push_back(cNew2);
771 
772  return true;

905 bool MeshTopoAlgorithm::CollapseVertex(const VertexCollapse& vc)
906 {
907  if (vc._circumFacets.size() != vc._circumPoints.size()) {
908  return false;
909  }
910 
911  if (vc._circumFacets.size() != 3) {
912  return false;
913  }
914 
915  if (!_rclMesh._aclPointArray[vc._point].IsValid()) {
916  return false; // the point is marked invalid from a previous run
917  }
918 
919  MeshFacet& rFace1 = _rclMesh._aclFacetArray[vc._circumFacets[0]];
920  MeshFacet& rFace2 = _rclMesh._aclFacetArray[vc._circumFacets[1]];
921  MeshFacet& rFace3 = _rclMesh._aclFacetArray[vc._circumFacets[2]];
922 
923  // get the point that is not shared by rFace1
924  PointIndex ptIndex = POINT_INDEX_MAX;
925  std::vector<PointIndex>::const_iterator it;
926  for (it = vc._circumPoints.begin(); it != vc._circumPoints.end(); ++it) {
927  if (!rFace1.HasPoint(*it)) {
928  ptIndex = *it;
929  break;
930  }
931  }
932 
933  if (ptIndex == POINT_INDEX_MAX) {
934  return false;
935  }
936 
937  FacetIndex neighbour1 = FACET_INDEX_MAX;
938  FacetIndex neighbour2 = FACET_INDEX_MAX;
939 
940  const std::vector<FacetIndex>& faces = vc._circumFacets;
941  // get neighbours that are not part of the faces to be removed
942  for (int i = 0; i < 3; i++) {
943  if (std::find(faces.begin(), faces.end(), rFace2._aulNeighbours[i]) == faces.end()) {
944  neighbour1 = rFace2._aulNeighbours[i];
945  }
946  if (std::find(faces.begin(), faces.end(), rFace3._aulNeighbours[i]) == faces.end()) {
947  neighbour2 = rFace3._aulNeighbours[i];
948  }
949  }
950 
951  // adjust point and neighbour indices
952  rFace1.Transpose(vc._point, ptIndex);
953  rFace1.ReplaceNeighbour(vc._circumFacets[1], neighbour1);
954  rFace1.ReplaceNeighbour(vc._circumFacets[2], neighbour2);
955 
956  if (neighbour1 != FACET_INDEX_MAX) {
957  MeshFacet& rFace4 = _rclMesh._aclFacetArray[neighbour1];
958  rFace4.ReplaceNeighbour(vc._circumFacets[1], vc._circumFacets[0]);
959  }
960  if (neighbour2 != FACET_INDEX_MAX) {
961  MeshFacet& rFace5 = _rclMesh._aclFacetArray[neighbour2];
962  rFace5.ReplaceNeighbour(vc._circumFacets[2], vc._circumFacets[0]);
963  }
964 
965  // the two facets and the point can be marked for removal
966  rFace2.SetInvalid();
967  rFace3.SetInvalid();
968  _rclMesh._aclPointArray[vc._point].SetInvalid();
969 
970  _needsCleanup = true;
971 
972  return true;
973 }

975 bool MeshTopoAlgorithm::CollapseEdge(FacetIndex ulFacetPos, FacetIndex ulNeighbour)
976 {

1005  ulFacetPos,

1010  ulFacetPos,

1034 
1035  _needsCleanup = true;
1036 
1037  return true;
