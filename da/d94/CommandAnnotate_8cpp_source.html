330 void execMidpoints(Gui::Command* cmd)
331 {
332 // Base::Console().Message("execMidpoints()\n");
333  TechDraw::DrawViewPart * dvp = nullptr;
334  std::vector<std::string> selectedEdges = CommandHelpers::getSelectedSubElements(cmd, dvp, "Edge");
335 
336  if (!dvp || selectedEdges.empty())
337  return;
338 
339  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Add Midpoint Vertices"));
340 
341  const TechDraw::BaseGeomPtrVector edges = dvp->getEdgeGeometry();
342  for (auto& s: selectedEdges) {
343  int GeoId(TechDraw::DrawUtil::getIndexFromName(s));
344  TechDraw::BaseGeomPtr geom = edges.at(GeoId);
345  Base::Vector3d mid = geom->getMidPoint();
346  // invert the point so the math works correctly
347  mid = DrawUtil::invertY(mid);
348  mid = CosmeticVertex::makeCanonicalPoint(dvp, mid);
349  dvp->addCosmeticVertex(mid);
350  }
351 
352  Gui::Command::commitCommand();
353 
354  dvp->recomputeFeature();
355 }
