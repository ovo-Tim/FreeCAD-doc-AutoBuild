39 def calc_length_adaptive(kink, angle, nominal_length, custom_length):
40  Path.Log.track(kink, angle, nominal_length, custom_length)
41 
42  if Path.Geom.isRoughly(abs(kink.deflection()), 0):
43  return 0
44 
45  # If the kink poses a 180deg turn the adaptive length is undefined. Mathematically
46  # it's infinite but that is not practical.
47  # We define the adaptive length to be the nominal length for this case.
48  if Path.Geom.isRoughly(abs(kink.deflection()), PI):
49  return nominal_length
50 
51  # The distance of the (estimated) corner from the kink position depends only on the
52  # deflection of the kink.
53  # Some sample values to build up intuition:
54  # deflection : dog bone : norm distance : calc
55  # ----------------:-------------:---------------:--------------
56  # 0 : -PI/2 : 1
57  # PI/6 : -5*PI/12 : 1.03528 : 1/cos( (pi/6) / 2)
58  # PI/4 : -3*PI/8 : 1.08239 : 1/cos( (pi/4) / 2)
59  # PI/3 : -PI/3 : 1.1547 : 1/cos( (pi/3) / 2)
60  # PI/2 : -PI/4 : 1.41421 : 1/cos( (pi/2) / 2)
61  # 2*PI/3 : -PI/6 : 2 : 1/cos((2*pi/3) / 2)
62  # 3*PI/4 : -PI/8 : 2.61313 : 1/cos((3*pi/4) / 2)
63  # 5*PI/6 : -PI/12 : 3.8637 : 1/cos((5*pi/6) / 2)
64  # PI : 0 : nan <-- see above
65  # The last column can be geometrically derived or found by experimentation.
66  dist = nominal_length / math.cos(kink.deflection() / 2)
67 
68  # The depth of the bone depends on the direction of the bone in relation to the
69  # direction of the corner. If the direction is identical then the depth is the same
70  # as the distance of the corner minus the nominal_length (which corresponds to the
71  # radius of the tool).
72  # If the corner's direction is PI/4 off the bone angle the intersecion of the tool
73  # with the corner is the projection of the corner onto the bone.
74  # If the corner's direction is perpendicular to the bone's angle there is, strictly
75  # speaking no intersection and the bone is ineffective. However, giving it our
76  # best shot we should probably move the entire depth.
77 
78  da = Path.Geom.normalizeAngle(kink.normAngle() - angle)
79  depth = dist * math.cos(da)
80  if depth < 0:
81  Path.Log.debug(
82  f"depth={depth:4f}: kink={kink}, angle={180*angle/PI}, dist={dist:.4f}, da={180*da/PI} -> depth=0.0"
83  )
84  depth = 0
85  else:
86  height = dist * abs(math.sin(da))
87  if height < nominal_length:
88  depth = depth - math.sqrt(nominal_length * nominal_length - height * height)
89  Path.Log.debug(
90  f"{kink}: angle={180*angle/PI}, dist={dist:.4f}, da={180*da/PI}, depth={depth:.4f}"
91  )
92 
93  return depth
