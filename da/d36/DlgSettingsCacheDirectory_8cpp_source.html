205 bool ApplicationCache::periodicCheckOfSize() const
206 {
207  QString vendor = QString::fromLatin1(App::Application::Config()["ExeVendor"].c_str());
208  QString application = QString::fromStdString(App::Application::getExecutableName());
209 
210  QSettings settings(vendor, application);
211  QString key = QString::fromLatin1("LastCacheCheck");
212  QDate date = settings.value(key).toDate();
213  QDate now = QDate::currentDate();
214 
215  // get the days since the last check
216  int days = date.daysTo(now);
217  if (date.isNull()) {
218  days = 1000;
219  }
220 
221  if (days >= numDays) {
222  settings.setValue(key, now);
223  return true;
224  }
225 
226  return false;
227 }

234 bool ApplicationCache::performAction(qint64 total)
235 {
236  bool performed = false;
237  if (total > limit) {
238  QString path = QString::fromStdString(App::Application::getUserCachePath());
239  QMessageBox msgBox(Gui::getMainWindow());
240  msgBox.setIcon(QMessageBox::Warning);
241  msgBox.setWindowTitle(tr("Cache directory"));
242 
243  QString hint = tr("The cache directory %1 exceeds the size of %2.").arg(path, ApplicationCache::toString(limit));
244  QString ask = tr("Do you want to clear it now?");
245  QString warn = tr("Warning: Please make sure that this is the only running %1 instance "
246  "and that no documents are opened as this may result into data loss!").arg(QCoreApplication::applicationName());
247 
248  msgBox.setText(QString::fromLatin1("%1 %2\n\n\n%3").arg(hint, ask, warn));
249  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No | QMessageBox::Open);
250  msgBox.setDefaultButton(QMessageBox::No);
251 
252  while (true) {
253  int ret = msgBox.exec();
254  if (ret == QMessageBox::Open) {
255  QUrl url = QUrl::fromLocalFile(path);
256  QDesktopServices::openUrl(url);
257  }
258  else {
259  if (ret == QMessageBox::Yes) {
260  clearDirectory(path);
261  performed = true;
262  }
263  break;
264  }
265  }
266  }
267 
268  return performed;
269 }

285 void ApplicationCache::clearDirectory(const QString& path)
286 {
287  // Add the transient directories and the lock files to the ignore list
288  QDir tmp = QString::fromUtf8(App::Application::getUserCachePath().c_str());
289  tmp.setNameFilters(QStringList() << QString::fromLatin1("*.lock"));
290  tmp.setFilter(QDir::Files);
291 
292  QList<QFileInfo> dirs;
293  std::vector<App::Document*> docs = App::GetApplication().getDocuments();
294  for (auto it : docs) {
295  QDir dir(QString::fromStdString(it->TransientDir.getStrValue()));
296  QFileInfo fi(dir.absolutePath());
297  dirs.append(fi);
298  }
299 
300  DocumentRecoveryCleaner cleaner;
301  cleaner.setIgnoreFiles(tmp.entryList());
302  cleaner.setIgnoreDirectories(dirs);
303  cleaner.clearDirectory(QFileInfo(path));
304 }

309 qint64 ApplicationCache::dirSize(QString dirPath) const
310 {
311  qint64 total = 0;
312  QDir dir(dirPath);
313 
314  QDir::Filters fileFilters = QDir::Files;
315  const auto& files = dir.entryList(fileFilters);
316  for (const QString& filePath : files) {
317  QFileInfo fi(dir, filePath);
318  total += fi.size();
319  }
320 
321  // traverse sub-directories recursively
322  QDir::Filters dirFilters = QDir::Dirs | QDir::NoDotAndDotDot;
323  const auto& dirs = dir.entryList(dirFilters);
324  for (const QString& subDirPath : dirs)
325  total += dirSize(dirPath + QDir::separator() + subDirPath);
326  return total;
327 }

348 QString ApplicationCache::toString(qint64 size)
349 {
350  QStringList units = {QString::fromLatin1("Bytes"),
351  QString::fromLatin1("KB"),
352  QString::fromLatin1("MB"),
353  QString::fromLatin1("GB")};
354  int i;
355  double outputSize = size;
356  for (i=0; i<units.size()-1; i++) {
357  if (outputSize < 1024)
358  break;
359  outputSize /= 1024;
360  }
361 
362  return QString::fromLatin1("%1 %2").arg(QLocale().toString(outputSize, 'f', 2), units[i]);
363 }
