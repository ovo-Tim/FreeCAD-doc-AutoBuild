70 void ConstraintPulley::onChanged(const App::Property* prop)
71 {
72  ConstraintGear::onChanged(prop);
73 
74  if ((prop == &Diameter) || (prop == &OtherDiameter) || (prop == &CenterDistance)) {
75  if (CenterDistance.getValue() > Precision::Confusion()) {
76  BeltAngle.setValue(asin((Diameter.getValue() - OtherDiameter.getValue()) / 2
77  / CenterDistance.getValue()));
78  BeltAngle.touch();
79  }
80  }
81  else if ((prop == &Force) || (prop == &TensionForce) || (prop == &IsDriven)) {
82  double radius = Diameter.getValue() / 2.0;
83  if (radius < Precision::Confusion()) {
84  return;
85  }
86  double force = Force.getValue() / (radius / 1000);
87  if (fabs(force) < Precision::Confusion()) {
88  return;
89  }
90  bool neg = (force < 0.0);
91  if (neg) {
92  force *= -1.0;
93  }
94 
95  if (IsDriven.getValue() == neg) {
96  BeltForce1.setValue(force + TensionForce.getValue());
97  BeltForce2.setValue(TensionForce.getValue());
98  }
99  else {
100  BeltForce2.setValue(force + TensionForce.getValue());
101  BeltForce1.setValue(TensionForce.getValue());
102  }
103  BeltForce1.touch();
104  }
105 }
