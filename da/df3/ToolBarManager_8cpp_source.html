299 void ToolBarGrip::mouseMoveEvent(QMouseEvent *me)
300 {
301  auto toolbar = qobject_cast<ToolBar*>(parentWidget());
302  if (!toolbar) {
303  return;
304  }
305 
306  auto area = ToolBarManager::getInstance()->toolBarAreaWidget(toolbar);
307  if (!area) {
308  return;
309  }
310 
311 #if QT_VERSION < QT_VERSION_CHECK(6,0,0)
312  QPoint pos = me->globalPos();
313 #else
314  QPoint pos = me->globalPosition().toPoint();
315 #endif
316  QRect rect(toolbar->mapToGlobal(QPoint(0,0)), toolbar->size());
317 
318  // if mouse did not leave the area of toolbar do not continue with undocking it
319  if (rect.contains(pos)) {
320  return;
321  }
322 
323  toolbar->undock();
324 
325  // After removing from area, this grip will be deleted. In order to
326  // continue toolbar dragging (because the mouse button is still pressed),
327  // we fake mouse events and send to toolbar. For some reason,
328  // send/postEvent() does not work, only timer works.
329  QPointer tb(toolbar);
330  QTimer::singleShot(0, [tb] {
331  auto modifiers = QApplication::queryKeyboardModifiers();
332  auto buttons = QApplication::mouseButtons();
333  if (buttons != Qt::LeftButton
334  || QWidget::mouseGrabber()
335  || modifiers != Qt::NoModifier
336  || !tb) {
337  return;
338  }
339 
340  QPoint pos(10, 10);
341  QPoint globalPos(tb->mapToGlobal(pos));
342  QMouseEvent mouseEvent(
343  QEvent::MouseButtonPress,
344  pos, globalPos, Qt::LeftButton, buttons, modifiers);
345  QApplication::sendEvent(tb, &mouseEvent);
346 
347  // Mouse follow the mouse press event with mouse move with some offset
348  // in order to activate toolbar dragging.
349  QPoint offset(30, 30);
350  QMouseEvent mouseMoveEvent(
351  QEvent::MouseMove,
352  pos+offset, globalPos+offset,
353  Qt::LeftButton, buttons, modifiers);
354  QApplication::sendEvent(tb, &mouseMoveEvent);
355  });
356 }

536 ToolBarArea ToolBarManager::toolBarArea(QWidget *widget) const
537 {
538  if (auto toolBar = qobject_cast<QToolBar*>(widget)) {
539  if (toolBar->isFloating()) {
540  return ToolBarArea::NoToolBarArea;
541  }
542 
543  auto qtToolBarArea = getMainWindow()->toolBarArea(toolBar);
544  switch (qtToolBarArea) {
545  case Qt::LeftToolBarArea:
546  return ToolBarArea::LeftToolBarArea;
547  case Qt::RightToolBarArea:
548  return ToolBarArea::RightToolBarArea;
549  case Qt::TopToolBarArea:
550  return ToolBarArea::TopToolBarArea;
551  case Qt::BottomToolBarArea:
552  return ToolBarArea::BottomToolBarArea;
553  default:
554  // no-op
555  break;
556  }
557  }
558 
559  if (auto areaWidget = toolBarAreaWidget(widget)) {
560  return areaWidget->area();
561  }
562 
563  return ToolBarArea::NoToolBarArea;
564 }

603 int ToolBarManager::toolBarIconSize(QWidget *widget) const
604 {
605  int s = _toolBarIconSize;
606  if (widget) {
607  if (widget->parentWidget() == statusBarAreaWidget) {
608  if (_statusBarIconSize > 0) {
609  s = _statusBarIconSize;
610  }
611  else {
612  s *= 0.6;
613  }

615  else if (widget->parentWidget() == menuBarLeftAreaWidget

618  s = _menuBarIconSize;

622  }
623  }
624  }

653 void ToolBarManager::setup(ToolBarItem* toolBarItems)
654 {
655  if (!toolBarItems) {
656  return; // empty menu bar
657  }
658 
659  QPointer<QWidget> actionWidget = createActionWidget();
660 
661  saveState();
662  this->toolbarNames.clear();
663 
664  int max_width = getMainWindow()->width();
665  int top_width = 0;
666 
667  bool nameAsToolTip = App::GetApplication().GetUserParameter().GetGroup("BaseApp")
668  ->GetGroup("Preferences")
669  ->GetGroup("MainWindow")
670  ->GetBool("ToolBarNameAsToolTip", true);
671 
672  QList<ToolBarItem*> items = toolBarItems->getItems();
673  QList<ToolBar*> toolbars = toolBars();
674 
675  for (ToolBarItem* it : items) {
676  // search for the toolbar
677  QString name = QString::fromUtf8(it->command().c_str());
678  this->toolbarNames << name;
679  ToolBar* toolbar = findToolBar(toolbars, name);
680  std::string toolbarName = it->command();
681  bool toolbar_added = false;
682 
683  if (!toolbar) {
684  toolbar = new ToolBar(getMainWindow());
685  toolbar->setWindowTitle(QApplication::translate("Workbench", toolbarName.c_str()));
686  toolbar->setObjectName(name);
687 
688  getMainWindow()->addToolBar(toolbar);
689 
690  if (nameAsToolTip) {
691  auto tooltip = QChar::fromLatin1('[')
692  + QApplication::translate("Workbench", toolbarName.c_str())
693  + QChar::fromLatin1(']');
694  toolbar->setToolTip(tooltip);
695  }
696  toolbar_added = true;
697  }
698  else {
699  int index = toolbars.indexOf(toolbar);
700  toolbars.removeAt(index);
701  }
702 
703  bool visible = false;
704 
705  // If visibility policy is custom, the toolbar is initialised as not visible, and the
706  // toggleViewAction to control its visibility is not visible either.
707  //
708  // Both are managed under the responsibility of the client code
709  if (it->visibilityPolicy != ToolBarItem::DefaultVisibility::Unavailable) {
710  bool defaultvisibility = it->visibilityPolicy == ToolBarItem::DefaultVisibility::Visible;
711 
712  visible = hPref->GetBool(toolbarName.c_str(), defaultvisibility);
713 
714  // Enable automatic handling of visibility via, for example, (contextual) menu
715  toolbar->toggleViewAction()->setVisible(true);
716  }
717  else {
718  // ToolBarItem::DefaultVisibility::Unavailable
719  // Prevent that the action to show/hide a toolbar appears on the (contextual) menus.
720  // This is also managed by the client code for a toolbar with custom policy
721  toolbar->toggleViewAction()->setVisible(false);
722  }
723 
724  // Initialise toolbar item visibility
725  toolbar->setVisible(visible);
726 
727  // Store item visibility policy within the action
728  QAction* toggle = toolbar->toggleViewAction();
729  toggle->setProperty("DefaultVisibility", static_cast<int>(it->visibilityPolicy));
730 
731  // setup the toolbar
732  setup(it, toolbar);
733  auto actions = toolbar->actions();
734  for (auto action : actions) {
735  actionWidget->addAction(action);
736  }
737 
738  // try to add some breaks to avoid to have all toolbars in one line
739  if (toolbar_added) {
740  if (top_width > 0 && getMainWindow()->toolBarBreak(toolbar)) {
741  top_width = 0;
742  }
743 
744  // the width() of a toolbar doesn't return useful results so we estimate
745  // its size by the number of buttons and the icon size
746  QList<QToolButton*> btns = toolbar->findChildren<QToolButton*>();
747  top_width += (btns.size() * toolbar->iconSize().width());
748  if (top_width > max_width) {
749  top_width = 0;
750  getMainWindow()->insertToolBarBreak(toolbar);
751  }
752  }
753  }
754  // hide all unneeded toolbars
755  for (QToolBar* it : std::as_const(toolbars)) {
756  // make sure that the main window has the focus when hiding the toolbar with
757  // the combo box inside
758  QWidget *fw = QApplication::focusWidget();
759  while (fw && !fw->isWindow()) {
760  if (fw == it) {
761  getMainWindow()->setFocus();
762  break;
763  }
764  fw = fw->parentWidget();
765  }
766  // ignore toolbars which do not belong to the previously active workbench
767  //QByteArray toolbarName = it->objectName().toUtf8();
768  if (!it->toggleViewAction()->isVisible()) {
769  continue;
770  }
771  //hPref->SetBool(toolbarName.constData(), it->isVisible());
772  it->hide();
773  it->toggleViewAction()->setVisible(false);
774  }
775 
776  setMovable(!areToolBarsLocked());
777 }

779 void ToolBarManager::setup(ToolBarItem* item, QToolBar* toolbar) const
780 {
781  CommandManager& mgr = Application::Instance->commandManager();
782  QList<ToolBarItem*> items = item->getItems();
783  QList<QAction*> actions = toolbar->actions();
784  for (ToolBarItem* it : items) {
785  // search for the action item

823 void ToolBarManager::saveState() const
824 {

827  // In this case the current state is not saved.

849  continue;

900 bool ToolBarManager::addToolBarToArea(QObject *source, QMouseEvent *ev)
901 {

903  if (!statusBar || !statusBar->isVisible()) {
904  statusBar = nullptr;
905  }
906 
907  auto menuBar = getMainWindow()->menuBar();
908  if (!menuBar || !menuBar->isVisible()) {
909  if (!statusBar) {
910  return false;

912  menuBar = nullptr;
913  }
914 
915  auto tb = qobject_cast<QToolBar*>(source);
916  if (!tb || !tb->isFloating()) {
917  return false;
918  }
919 
920  static QPointer<OverlayDragFrame> tbPlaceholder;
921  static QPointer<ToolBarAreaWidget> lastArea;
922  static int tbIndex = -1;
923  if (ev->type() == QEvent::MouseMove) {
924  if (tb->orientation() != Qt::Horizontal
925  || ev->buttons() != Qt::LeftButton) {
926  if (tbIndex >= 0) {
927  if (lastArea) {
928  lastArea->removeWidget(tbPlaceholder);
929  lastArea = nullptr;
930  }
931  tbPlaceholder->hide();
932  tbIndex = -1;
933  }
934  return false;
935  }
936  }
937 
938  if (ev->type() == QEvent::MouseButtonRelease && ev->button() != Qt::LeftButton) {
939  return false;
940  }
941 
942  QPoint pos = QCursor::pos();
943  ToolBarAreaWidget *area = nullptr;
944  if (statusBar) {
945  QRect rect(statusBar->mapToGlobal(QPoint(0,0)), statusBar->size());
946  if (rect.contains(pos)) {
947  area = statusBarAreaWidget;
948  }
949  }
950  if (!area) {
951  if (!menuBar) {
952  return false;
953  }
954  QRect rect(menuBar->mapToGlobal(QPoint(0,0)), menuBar->size());
955  if (rect.contains(pos)) {
956  if (pos.x() - rect.left() < menuBar->width()/2) {
957  area = menuBarLeftAreaWidget;
958  }
959  else {
960  area = menuBarRightAreaWidget;
961  }
962  }
963  else {
964  if (tbPlaceholder) {
965  if (lastArea) {
966  lastArea->removeWidget(tbPlaceholder);
967  lastArea = nullptr;
968  }
969  tbPlaceholder->hide();
970  tbIndex = -1;
971  }
972  return false;
973  }
974  }
975 
976  int idx = 0;
977  for (int c = area->count(); idx < c ;++idx) {
978  auto widget = area->widgetAt(idx);
979  if (!widget || widget->isHidden()) {
980  continue;
981  }
982  int p = widget->mapToGlobal(widget->rect().center()).x();
983  if (pos.x() < p) {
984  break;
985  }
986  }
987  if (tbIndex >= 0 && tbIndex == idx-1) {
988  idx = tbIndex;
989  }
990  if (ev->type() == QEvent::MouseMove) {
991  if (!tbPlaceholder) {
992  tbPlaceholder = new OverlayDragFrame(getMainWindow());
993  tbPlaceholder->hide();
994  tbIndex = -1;
995  }
996  if (tbIndex != idx) {
997  tbIndex = idx;
998  tbPlaceholder->setSizePolicy(tb->sizePolicy());
999  tbPlaceholder->setMinimumWidth(tb->minimumWidth());
1000  tbPlaceholder->resize(tb->size());
1001  area->insertWidget(idx, tbPlaceholder);
1002  lastArea = area;
1003  tbPlaceholder->adjustSize();
1004  tbPlaceholder->show();
1005  }
1006  } else {
1007  tbIndex = idx;
1008  QTimer::singleShot(10, tb, [tb]() {
1009  if (!lastArea) {
1010  return;
1011  }
1012 
1013  {
1014  tbPlaceholder->hide();
1015  QSignalBlocker block(tb);
1016  lastArea->removeWidget(tbPlaceholder);
1017  getMainWindow()->removeToolBar(tb);
1018  tb->setOrientation(Qt::Horizontal);
1019  lastArea->insertWidget(tbIndex, tb);
1020  tb->setVisible(true);
1021  lastArea = nullptr;
1022  }
1023 
1024  Q_EMIT tb->topLevelChanged(false);
1025  tbIndex = -1;
1026  });
1027  }
1028  return false;
1029 }

1138 {

1170  }
