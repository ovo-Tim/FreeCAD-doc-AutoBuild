54 int MappedName::findTagInElementName(long* tagOut,
55  int* lenOut,
56  std::string* postfixOut,
57  char* typeOut,
58  bool negative,
59  bool recursive) const
60 {
61  bool hex = true;
62  int pos = this->rfind(POSTFIX_TAG);
63 
64  // Example name, POSTFIX_TAG == ;:H
65  // #94;:G0;XTR;:H19:8,F;:H1a,F;BND:-1:0;:H1b:10,F
66  // ^
67  // |
68  // pos
69 
70  if (pos < 0) {
71  pos = this->rfind(POSTFIX_DECIMAL_TAG);
72  if (pos < 0) {
73  return -1;
74  }
75  hex = false;
76  }
77  int offset = pos + (int)POSTFIX_TAG_SIZE;
78  long _tag = 0;
79  int _len = 0;
80  char sep = 0;
81  char sep2 = 0;
82  char tp = 0;
83  char eof = 0;
84 
85  int size {0};
86  const char* nameAsChars = this->toConstString(offset, size);
87 
88  // check if the number followed by the tagPosfix is negative
89  bool isNegative = (nameAsChars[0] == '-');
90  if (isNegative) {
91  ++nameAsChars;
92  --size;
93  }
94  boost::iostreams::stream<boost::iostreams::array_source> iss(nameAsChars, size);
95  if (!hex) {
96  // no hex is an older version of the encoding scheme
97  iss >> _tag >> sep;
98  }
99  else {
100  // The purpose of tagOut postfixOut is to encode one model operation. The
101  // 'tagOut' field is used to record the own object ID of that model shape,
102  // and the 'lenOut' field indicates the length of the operation codes
103  // before the tagOut postfixOut. These fields are in hex. The trailing 'F' is
104  // the shape typeOut of this element, 'F' for face, 'E' edge, and 'V' vertex.
105  //
106  // #94;:G0;XTR;:H19:8,F;:H1a,F;BND:-1:0;:H1b:10,F
107  // | | ^^ ^^
108  // | | | |
109  // ---lenOut = 0x10--- tagOut lenOut
110 
111  iss >> std::hex;
112  // _tag field can be skipped, if it is 0
113  if (nameAsChars[0] == ',' || nameAsChars[0] == ':') {
114  iss >> sep;
115  }
116  else {
117  iss >> _tag >> sep;
118  }
119  }
120 
121  if (isNegative) {
122  _tag = -_tag;
123  }
124 
125  if (sep == ':') {
126  // ':' is followed by _len field.
127  //
128  // For decTagPostfix() (i.e. older encoding scheme), this is the length
129  // of the string before the entire postfixOut (A postfixOut may contain
130  // multiple segments usually separated by ELEMENT_MAP_PREFIX.
131  //
132  // For newer POSTFIX_TAG, this counts the number of characters that
133  // proceeds this tagOut postfixOut segment that forms the op code (see
134  // example above).
135  //
136  // The reason of this change is so that the postfixOut can stay the same
137  // regardless of the prefix, which can increase memory efficiency.
138  //
139  iss >> _len >> sep2 >> tp >> eof;
140 
141  // The next separator to look for is either ':' for older tagOut postfixOut, or ','
142  if (!hex && sep2 == ':') {
143  sep2 = ',';
144  }
145  }
146  else if (hex && sep == ',') {
147  // ',' is followed by a single character that indicates the element typeOut.
148  iss >> tp >> eof;
149  sep = ':';
150  sep2 = ',';
151  }
152 
153  if (_len < 0 || sep != ':' || sep2 != ',' || tp == 0 || eof != 0) {
154  return -1;
155  }
156 
157  if (hex) {
158  if (pos - _len < 0) {
159  return -1;
160  }
161  if ((_len != 0) && recursive && (tagOut || lenOut)) {
162  // in case of recursive tagOut postfixOut (used by hierarchy element
163  // map), look for any embedded tagOut postfixOut
164  int next = MappedName::fromRawData(*this, pos - _len, _len).rfind(POSTFIX_TAG);
165  if (next >= 0) {
166  next += pos - _len;
167  // #94;:G0;XTR;:H19:8,F;:H1a,F;BND:-1:0;:H1b:10,F
168  // ^ ^
169  // | |
170  // next pos
171  //
172  // There maybe other operation codes after this embedded tagOut
173  // postfixOut, search for the separator.
174  //
175  int end {0};
176  if (pos == next) {
177  end = -1;
178  }
179  else {
180  end = MappedName::fromRawData(*this, next + 1, pos - next - 1)
181  .find(ELEMENT_MAP_PREFIX);
182  }
183  if (end >= 0) {
184  end += next + 1;
185  // #94;:G0;XTR;:H19:8,F;:H1a,F;BND:-1:0;:H1b:10,F
186  // ^
187  // |
188  // end
189  _len = pos - end;
190  // #94;:G0;XTR;:H19:8,F;:H1a,F;BND:-1:0;:H1b:10,F
191  // | |
192  // -- lenOut --
193  }
194  else {
195  _len = 0;
196  }
197  }
198  }
199 
200  // Now convert the 'lenOut' field back to the length of the remaining name
201  //
202  // #94;:G0;XTR;:H19:8,F;:H1a,F;BND:-1:0;:H1b:10,F
203  // | |
204  // ----------- lenOut -----------
205  _len = pos - _len;
206  }
207  if (typeOut) {
208  *typeOut = tp;
209  }
210  if (tagOut) {
211  if (_tag == 0 && recursive) {
212  return MappedName(*this, 0, _len)
213  .findTagInElementName(tagOut, lenOut, postfixOut, typeOut, negative);
214  }
215  if (_tag > 0 || negative) {
216  *tagOut = _tag;
217  }
218  else {
219  *tagOut = -_tag;
220  }
221  }
222  if (lenOut) {
223  *lenOut = _len;
224  }
225  if (postfixOut) {
226  *postfixOut = this->toString(pos);
227  }
228  return pos;
229 }
