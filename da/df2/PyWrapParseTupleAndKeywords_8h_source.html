40 bool Wrapped_ParseTupleAndKeywords(PyObject* args,
41  PyObject* kw,
42  const char* format,
43  const std::array<const char*, arraySize> keywords,
44  ...)
45 {
46  static_assert(arraySize > 0, "keywords array must have at least a single nullptr in it");
47  if (keywords.back()) {
48  PyErr_SetString(PyExc_ValueError, "Last element of keywords array is not null");
49  return false;
50  }

52  // NOTE: This code is from getargs.c in the Python source code (modified to use the public

54 
55  if ((args == nullptr || !PyTuple_Check(args)) || (kw != nullptr && !PyDict_Check(kw))
56  || format == nullptr) {
57  PyErr_BadInternalCall();
58  return false;
59  }
60

63  // NOLINTNEXTLINE(cppcoreguidelines-pro-type-const-cast)
64  int retval =
65  PyArg_VaParseTupleAndKeywords(args, kw, format, const_cast<char**>(keywords.data()), va);
66  va_end(va);

68 }
