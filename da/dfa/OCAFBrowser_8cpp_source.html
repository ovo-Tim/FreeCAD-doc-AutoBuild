108 void OCAFBrowser::load(const TDF_Label& label, QTreeWidgetItem* item, const QString& str)
109 {
110  label.Dump(std::cout);
111 
112  Handle(TDataStd_Name) name;
113  if (label.FindAttribute(TDataStd_Name::GetID(), name)) {
114  QString text =
115  QString::fromLatin1("%1 %2").arg(str, QString::fromUtf8(toString(name->Get()).c_str()));
116  item->setText(0, text);
117  }
118 
119 
120  TDF_IDList localList;
121  TDF_AttributeIterator itr(label);
122  for (; itr.More(); itr.Next()) {
123  localList.Append(itr.Value()->ID());
124  }
125 
126  // Hint: In OCC 7.7 use XCAFDoc::AttributeInfo
127  for (TDF_ListIteratorOfIDList it(localList); it.More(); it.Next()) {
128  Handle(TDF_Attribute) attr;
129  if (label.FindAttribute(it.Value(), attr)) {
130  QTreeWidgetItem* child = new QTreeWidgetItem();
131  item->addChild(child);
132  if (it.Value() == TDataStd_Name::GetID()) {
133  QString text;
134  QTextStream str(&text);
135  str << attr->DynamicType()->Name();
136  str << " = " << toString(Handle(TDataStd_Name)::DownCast(attr)->Get()).c_str();
137  child->setText(0, text);
138  }
139  else if (it.Value() == TNaming_UsedShapes::GetID()) {
140  QString text;
141  QTextStream str(&text);
142  str << attr->DynamicType()->Name();
143  str << " = " << Handle(TNaming_UsedShapes)::DownCast(attr)->Map().Size();
144  child->setText(0, text);
145  }
146  else if (it.Value() == TDF_TagSource::GetID()) {
147  QString text;
148  QTextStream str(&text);
149  str << attr->DynamicType()->Name();
150  str << " = " << Handle(TDF_TagSource)::DownCast(attr)->Get();
151  child->setText(0, text);
152  }
153  else if (it.Value() == TDataStd_Integer::GetID()) {
154  QString text;
155  QTextStream str(&text);
156  str << attr->DynamicType()->Name();
157  str << " = " << Handle(TDataStd_Integer)::DownCast(attr)->Get();
158  child->setText(0, text);
159  }
160  else if (it.Value() == TDataStd_Real::GetID()) {
161  QString text;
162  QTextStream str(&text);
163  str << attr->DynamicType()->Name();
164  str << " = " << Handle(TDataStd_Real)::DownCast(attr)->Get();
165  child->setText(0, text);
166  }
167  else if (it.Value() == XCAFDoc_Color::GetID()) {
168  Quantity_ColorRGBA rgba = Handle(XCAFDoc_Color)::DownCast(attr)->GetColorRGBA();
169  Quantity_Color rgb = rgba.GetRGB();
170  QString tuple = QString::fromLatin1("(%1, %2, %3, %4)")
171  .arg(rgb.Red())
172  .arg(rgb.Green())
173  .arg(rgb.Blue())
174  .arg(rgba.Alpha());
175  QString text;
176  QTextStream str(&text);
177  str << attr->DynamicType()->Name();
178  str << " = " << Quantity_Color::StringName(rgb.Name()) << " " << tuple;
179  child->setText(0, text);
180  }
181  else if (it.Value() == XCAFDoc_ShapeMapTool::GetID()) {
182  QString text;
183  QTextStream str(&text);
184  str << attr->DynamicType()->Name();
185  str << " = " << Handle(XCAFDoc_ShapeMapTool)::DownCast(attr)->GetMap().Size();
186  child->setText(0, text);
187  }
188  else if (it.Value() == TNaming_NamedShape::GetID()) {
189  child->setText(0, toText(Handle(TNaming_NamedShape)::DownCast(attr)));
190  }
191  else if (!Handle(TDataStd_TreeNode)::DownCast(attr).IsNull()) {
192  Handle(TDataStd_TreeNode) aTreeNode = Handle(TDataStd_TreeNode)::DownCast(attr);
193  child->setText(0, toText(aTreeNode));
194  }
195  else {
196  child->setText(0, QLatin1String(attr->DynamicType()->Name()));
197  }
198  }
199  }
200 
201 
202  int index = 1;
203  for (TDF_ChildIterator it(label); it.More(); it.Next(), index++) {
204  QString text = QString::fromLatin1("%1:%2").arg(str).arg(index);
205  QTreeWidgetItem* child = new QTreeWidgetItem();
206  child->setText(0, text);
207  child->setIcon(0, myGroupIcon);
208  item->addChild(child);
209  load(it.Value(), child, text);
210  }
211 }

254 QString OCAFBrowser::toText(const Handle(TDataStd_TreeNode) & treeNode)
255 {
256  Standard_CString type = "";
257  if (treeNode->ID() == XCAFDoc::ShapeRefGUID()) {
258  type = "Shape Instance Link";
259  }
260  else if (treeNode->ID() == XCAFDoc::ColorRefGUID(XCAFDoc_ColorGen)) {
261  type = "Generic Color Link";
262  }
263  else if (treeNode->ID() == XCAFDoc::ColorRefGUID(XCAFDoc_ColorSurf)) {
264  type = "Surface Color Link";
265  }
266  else if (treeNode->ID() == XCAFDoc::ColorRefGUID(XCAFDoc_ColorCurv)) {
267  type = "Curve Color Link";
268  }
269  else if (treeNode->ID() == XCAFDoc::DimTolRefGUID()) {
270  type = "DGT Link";
271  }
272  else if (treeNode->ID() == XCAFDoc::DatumRefGUID()) {
273  type = "Datum Link";
274  }
275  else if (treeNode->ID() == XCAFDoc::MaterialRefGUID()) {
276  type = "Material Link";
277  }
278 
279  QString prefix = QString::fromLatin1(type);
280 
281  QString display;
282  if (treeNode->HasFather()) {
283  TDF_Label aLabel = treeNode->Father()->Label();
284  TCollection_AsciiString entry;
285  TDF_Tool::Entry(aLabel, entry);
286  QString suffix = QString::fromLatin1(entry.ToCString());
287  display = QString::fromLatin1("%1 ==> %2").arg(prefix, suffix);
288  }
289  else {
290  Handle(TDataStd_TreeNode) aFirstChild = treeNode->First();
291  QStringList aRefs;
292  while (!aFirstChild.IsNull()) {
293  TDF_Label aLabel = aFirstChild->Label();
294  TCollection_AsciiString entry;
295  TDF_Tool::Entry(aLabel, entry);
296  QString suffix = QString::fromLatin1(entry.ToCString());
297  aRefs.append(suffix);
298  aFirstChild = aFirstChild->Next();
299  }
300  display = QString::fromLatin1("%1 <== %2").arg(prefix, aRefs.join(QChar::fromLatin1(',')));
301  }
302 
303  QString text;
304  QTextStream str(&text);
305  str << treeNode->DynamicType()->Name();
306  if (!display.isEmpty()) {
307  str << " = " << display;
308  }
309 
310  return text;
311 }
