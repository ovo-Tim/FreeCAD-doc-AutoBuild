150 TopoDS_Shape DrawComplexSection::makeCuttingTool(double dMax)
151 {
152  // Base::Console().Message("DCS::makeCuttingTool()\n");
153  TopoDS_Wire profileWire = makeProfileWire();
154  if (profileWire.IsNull()) {
155  throw Base::RuntimeError("Can not make wire from cutting tool (1)");
156  }
157 
158  if (debugSection()) {
159  //the nose to tail version of the profile
160  BRepTools::Write(profileWire, "DCSProfileWire.brep");//debug
161  }
162 
163  gp_Ax2 sectionCS = getSectionCS();
164  gp_Dir gClosestBasis;//direction perpendicular to profile & section normal
165  bool isPositionOK = validateProfilePosition(profileWire, sectionCS, gClosestBasis);
166  if (!isPositionOK) {
167  //profile is not in a good position. Result might not be right.
168  Base::Console().Warning("DCS::makeCuttingTool - %s - profile is outside shape box\n",
169  getNameInDocument());
170  }
171 
172  //move the profile wire to one side of the shape
173  gp_Trsf mov;
174  mov.SetTranslation(gp_Vec(gClosestBasis) * (-dMax));
175  TopLoc_Location loc(mov);
176  profileWire.Move(loc);
177 
178  gp_Vec extrudeDir(0.0, 0.0, 1.0);//arbitrary default
179  if (BRep_Tool::IsClosed(profileWire)) {
180  // Wire is closed, so make a face from it and extrude "vertically"
181  BRepBuilderAPI_MakeFace mkFace(profileWire);
182  TopoDS_Face toolFace = mkFace.Face();
183  if (toolFace.IsNull()) {
184  return TopoDS_Shape();
185  }
186  gp_Dir gpNormal = getFaceNormal(toolFace);
187  extrudeDir = 2.0 * dMax * gpNormal;
188  return BRepPrimAPI_MakePrism(toolFace, extrudeDir).Shape();
189  }
190 
191  // if the wire is open (the normal case of a more or less linear profile),
192  // we need to make a "face" from the wire by extruding it
193  // in the direction of gClosestBasis , then extrude the face in the direction of the section normal
194 
195  if (ProjectionStrategy.getValue() == 0) {
196  // Offset. Warn if profile is not quite aligned with section normal. if
197  // the profile and normal are misaligned, the check below for empty "solids"
198  // will not be correct.
199  double angleThresholdDeg = 5.0;
200  // bool isOK =
201  validateOffsetProfile(profileWire, SectionNormal.getValue(), angleThresholdDeg);
202  }
203 
204  m_toolFaceShape = extrudeWireToFace(profileWire, gClosestBasis, 2.0 * dMax);
205  if (debugSection()) {
206  BRepTools::Write(m_toolFaceShape, "DCSToolFaceShape.brep");//debug
207  }
208  extrudeDir = dMax * sectionCS.Direction();
209  TopoDS_Shape roughTool = BRepPrimAPI_MakePrism(m_toolFaceShape, extrudeDir).Shape();
210  if (roughTool.ShapeType() == TopAbs_COMPSOLID ||
211  roughTool.ShapeType() == TopAbs_COMPOUND) {
212  //Composite Solids do not cut well if they contain "solids" with no volume. This
213  //happens if the profile has segments parallel to the extrude direction.
214  //We need to disassemble it and only keep the real solids.
215  BRep_Builder builder;
216  TopoDS_Compound comp;
217  builder.MakeCompound(comp);
218  TopExp_Explorer expSolids(roughTool, TopAbs_SOLID);
219  for (; expSolids.More(); expSolids.Next()) {
220  TopoDS_Solid solid = TopoDS::Solid(expSolids.Current());
221  GProp_GProps gprops;
222  BRepGProp::VolumeProperties(solid, gprops);
223  double volume = gprops.Mass();
224  if (volume > EWTOLERANCE) {
225  builder.Add(comp, solid);
226  }
227  }
228  return comp;
229  }
230 
231  return BRepPrimAPI_MakePrism(m_toolFaceShape, extrudeDir).Shape();
232 }

276 void DrawComplexSection::makeSectionCut(const TopoDS_Shape& baseShape)
277 {
278  // Base::Console().Message("DCS::makeSectionCut() - %s - baseShape.IsNull: %d\n",
279  // getNameInDocument(), baseShape.IsNull());
280  if (ProjectionStrategy.getValue() == 0) {
281  //Offset. Use regular section behaviour
282  return DrawViewSection::makeSectionCut(baseShape);
283  }
284 
285  try {
286  connectAlignWatcher =
287  QObject::connect(&m_alignWatcher, &QFutureWatcherBase::finished, &m_alignWatcher,
288  [this] { this->onSectionCutFinished(); });
289 
290  // We create a lambda closure to hold a copy of baseShape.
291  // This is important because this variable might be local to the calling
292  // function and might get destructed before the parallel processing finishes.
293  auto lambda = [this, baseShape]{this->makeAlignedPieces(baseShape);};
294  m_alignFuture = QtConcurrent::run(std::move(lambda));
295  m_alignWatcher.setFuture(m_alignFuture);
296  waitingForAlign(true);
297  }
298  catch (...) {
299  Base::Console().Message("DCS::makeSectionCut - failed to make alignedPieces");
300  return;
301  }
302 
303  return DrawViewSection::makeSectionCut(baseShape);
304 }

322 //for Aligned strategy, cut the rawShape by each segment of the tool
323 //TODO: this process should replace the "makeSectionCut" from DVS
324 void DrawComplexSection::makeAlignedPieces(const TopoDS_Shape& rawShape)
325 {
326  // Base::Console().Message("DCS::makeAlignedPieces() - rawShape.isNull: %d\n", rawShape.IsNull());
327 
328  if (!canBuild(getSectionCS(), CuttingToolWireObject.getValue())) {
329  throw Base::RuntimeError("Profile is parallel to Section Normal");
330  }
331 
332  std::vector<TopoDS_Shape> pieces;
333  std::vector<double> pieceXSizeAll;//size in sectionCS.XDirection (width)
334  std::vector<double> pieceYSizeAll;//size in sectionCS.YDirection (height)
335  std::vector<double> pieceZSizeAll; //size in sectionCS.Direction (depth)
336  std::vector<double> pieceVerticalAll; // displacement of piece in vertical direction
337 
338 
339  //make a CS from the section CS ZX plane to allow piece positioning (left-right)
340  //with the section view vertical centerline and (in-out, forward-backward) with
341  //the effective section plane for cut surface identification.
342  gp_Ax3 stdCS; //OXYZ
343  gp_Vec gProjectionUnit = gp_Vec(getSectionCS().Direction());
344 
345  //get a vector that describes the profile's 3d gross orientation
346  TopoDS_Wire profileWire = makeProfileWire();
347  if (profileWire.IsNull()) {
348  throw Base::RuntimeError("Can not make wire from cutting tool (2)");
349  }
350  gp_Vec gProfileVec = makeProfileVector(profileWire);
351 
352  // we will rotate the pieces around a line perpendicular to both section
353  // normal and the profile vector
354  gp_Vec rotateAxis = (getSectionCS().Direction()).Crossed(gProfileVec);
355 
356  //now we want to know what the profileVector looks like on the page (only X,Y coords)
357  //so we know if we are going to stack views vertically or horizontally and if the segments
358  //will occur (left to right or right to left) or (top to bottom or bottom to top)
359  gProfileVec = projectVector(gProfileVec).Normalized();
360 
361  bool isProfileVertical = true;
362  if (fabs(gProfileVec.Dot(gp::OY().Direction().XYZ())) != 1.0) {
363  //profile is not parallel with stdY (paper space Up).
364  //this test is not good enough for "vertical-ish" diagonal profiles
365  isProfileVertical = false;
366  }
367 
368  double horizReverser = 1.0;//profile vector points to right, so we move to right
369  if (gProfileVec.Dot(gp_Vec(gp::OX().Direction().XYZ())) < 0.0) {
370  //profileVec does not point towards stdX (right in paper space)
371  horizReverser = -1.0;
372  }
373  double verticalReverser = 1.0;//profile vector points to top, so we move to top
374  if (gProfileVec.Dot(gp_Vec(gp::OY().Direction().XYZ())) < 0.0) {
375  //profileVec does not point towards stdY (up in paper space)
376  verticalReverser = -1.0;
377  }
378 
379  TopExp_Explorer expFaces(m_toolFaceShape, TopAbs_FACE);
380  for (int iPiece = 0; expFaces.More(); expFaces.Next(), iPiece++) {
381  TopoDS_Face face = TopoDS::Face(expFaces.Current());
382  gp_Vec segmentNormal = gp_Vec(getFaceNormal(face));
383  if (!showSegment(gp_Dir(segmentNormal))) {
384  //skip this segment of the profile
385  continue;
386  }
387  //we only want to reverse the segment normal if it is not perpendicular to section normal
388  if (segmentNormal.Dot(gProjectionUnit) != 0.0
389  && segmentNormal.Angle(gProjectionUnit) <= M_PI_2) {
390  segmentNormal.Reverse();
391  }
392 
393  gp_Vec extrudeDir = segmentNormal * m_shapeSize;
394  BRepPrimAPI_MakePrism mkPrism(face, extrudeDir);
395  TopoDS_Shape segmentTool = mkPrism.Shape();
396  TopoDS_Shape intersect = shapeShapeIntersect(segmentTool, rawShape);
397  if (intersect.IsNull()) {
398  continue;
399  }
400  if (debugSection()) {
401  stringstream ss;
402  ss << "DCSAintersect" << iPiece << ".brep";
403  BRepTools::Write(intersect, ss.str().c_str());//debug
404  ss.clear();
405  ss.str(std::string());
406  }
407 
408  // move intersection shape to the origin so we can rotate it without worrying about
409  // center of rotation.
410  gp_Trsf xPieceCenter;
411  gp_Vec pieceCentroid = gp_Vec(ShapeUtils::findCentroid(intersect).XYZ());
412 
413  // save the amount we moved this piece in the vertical direction so we can
414  // put it back in the right place later
415  gp_Vec maskedVertical = DU::maskDirection(pieceCentroid, rotateAxis);
416  maskedVertical = pieceCentroid - maskedVertical;
417  double verticalDisplacement = maskedVertical.X() + maskedVertical.Y() + maskedVertical.Z();
418  pieceVerticalAll.push_back(verticalDisplacement);
419 
420  xPieceCenter.SetTranslation(pieceCentroid * -1.0);
421  BRepBuilderAPI_Transform mkTransXLate(intersect, xPieceCenter, true);
422  TopoDS_Shape pieceCentered = mkTransXLate.Shape();
423  if (debugSection()) {
424  stringstream ss;
425  ss << "DCSBpieceCentered" << iPiece << ".brep";
426  BRepTools::Write(pieceCentered, ss.str().c_str());//debug
427  ss.clear();
428  ss.str(std::string());
429  }
430 
431  //rotate the intersection so interesting face is aligned with what will
432  // become the paper plane.
433  double faceAngle =
434  gp_Vec(getSectionCS().Direction().Reversed()).AngleWithRef(segmentNormal, rotateAxis);
435  gp_Ax1 faceAxis(gp_Pnt(0.0, 0.0, 0.0), rotateAxis);
436  gp_Ax3 pieceCS;// OXYZ
437  pieceCS.Rotate(faceAxis, faceAngle);
438  gp_Trsf xPieceRotate;
439  xPieceRotate.SetTransformation(stdCS, pieceCS);
440  BRepBuilderAPI_Transform mkTransRotate(pieceCentered, xPieceRotate, true);
441  TopoDS_Shape pieceRotated = mkTransRotate.Shape();
442 
443  if (debugSection()) {
444  stringstream ss;
445  ss << "DCSCpieceRotated" << iPiece << ".brep";
446  BRepTools::Write(pieceRotated, ss.str().c_str());//debug
447  ss.clear();
448  ss.str(std::string());
449  }
450 
451  //align a copy of the piece with OXYZ so we can use bounding box to get
452  //width, depth, height of the piece. We copy the piece so the transformation
453  //does not affect the original.
454  BRepBuilderAPI_Copy BuilderPieceCopy(pieceRotated);
455  TopoDS_Shape copyPieceRotatedShape = BuilderPieceCopy.Shape();
456  gp_Trsf xPieceAlign;
457  xPieceAlign.SetTransformation(stdCS, getProjectionCS());
458  BRepBuilderAPI_Transform mkTransAlign(copyPieceRotatedShape, xPieceAlign);
459  TopoDS_Shape pieceAligned = mkTransAlign.Shape();
460  // we may have shifted our piece off center, so we better recenter here
461  gp_Trsf xPieceRecenter;
462  gp_Vec rotatedCentroid = gp_Vec(ShapeUtils::findCentroid(pieceAligned).XYZ());
463  xPieceRecenter.SetTranslation(rotatedCentroid * -1.0);
464  BRepBuilderAPI_Transform mkTransRecenter(pieceAligned, xPieceRecenter, true);
465  pieceAligned = mkTransRecenter.Shape();
466 
467  if (debugSection()) {
468  stringstream ss;
469  ss << "DCSDpieceAligned" << iPiece << ".brep";
470  BRepTools::Write(pieceAligned, ss.str().c_str());//debug
471  ss.clear();
472  ss.str(std::string());
473  }
474  Bnd_Box shapeBox;
475  shapeBox.SetGap(0.0);
476  BRepBndLib::AddOptimal(pieceAligned, shapeBox);
477  double xMin = 0, xMax = 0, yMin = 0, yMax = 0, zMin = 0, zMax = 0;
478  shapeBox.Get(xMin, yMin, zMin, xMax, yMax, zMax);
479  double pieceXSize(xMax - xMin);
480  double pieceYSize(yMax - yMin);
481  double pieceZSize(zMax - zMin);
482 
483  pieceXSizeAll.push_back(pieceXSize); // size in ProjectionCS.
484  pieceYSizeAll.push_back(pieceYSize);
485  pieceZSizeAll.push_back(pieceZSize);
486 
487  //now we need to move the piece so that the interesting face is coincident
488  //with the paper plane (stdXY). This will be a move along stdZ by -zMax.
489  gp_Vec toPaperPlane = gp::OZ().Direction().XYZ() * zMax * -1.0;
490  gp_Trsf xPieceToPlane;
491  xPieceToPlane.SetTranslation(toPaperPlane);
492  BRepBuilderAPI_Transform mkTransDisplace(pieceAligned, xPieceToPlane, true);
493  TopoDS_Shape pieceToPlane = mkTransDisplace.Shape();
494 
495  if (debugSection()) {
496  stringstream ss;
497  ss << "DCSEpieceToPlane" << iPiece << ".brep";
498  BRepTools::Write(pieceToPlane, ss.str().c_str());//debug
499  }
500  pieces.push_back(pieceToPlane);
501  // piece is on the paper plane, with piece centroid at the origin
502  }
503 
504  if (pieces.empty()) {
505  m_alignResult = TopoDS_Compound();
506  return;
507  }
508 
509  //space the pieces "horizontally" or "vertically" in OXYZ
510  double movementReverser = isProfileVertical ? verticalReverser : horizReverser;
511  gp_Vec movementAxis = gp_Vec(gp::OX().Direction());
512  gp_Vec alignmentAxis = gp_Vec(gp::OY().Direction().Reversed());
513  if (isProfileVertical) {
514  movementAxis = gp_Vec(gp::OY().Direction());
515  alignmentAxis = gp_Vec(gp::OX().Direction());
516  }
517  gp_Vec gMovementVector = movementAxis * movementReverser;
518 
519  int stopAt = pieces.size();
520  double distanceToMove = 0.0;
521  for (int iPiece = 0; iPiece < stopAt; iPiece++) {
522  // this movement needs to be smarter about what direction is L/R and which
523  // is up/down?
524  double pieceSizeMoveDist = pieceXSizeAll.at(iPiece);
525  if (isProfileVertical) {
526  pieceSizeMoveDist = pieceYSizeAll.at(iPiece); // for spacing
527  }
528  double myDistanceToMove = distanceToMove + pieceSizeMoveDist / 2.0;
529  gp_Vec alignmentVector = alignmentAxis * pieceVerticalAll.at(iPiece) * -1.0;
530  gp_Vec netDisplacementVector = gMovementVector * myDistanceToMove + alignmentVector;
531  gp_Trsf xPieceDistribute;
532  xPieceDistribute.SetTranslation(netDisplacementVector);
533  BRepBuilderAPI_Transform mkTransDistribute(pieces.at(iPiece), xPieceDistribute, true);
534  pieces.at(iPiece) = mkTransDistribute.Shape();
535  distanceToMove += pieceSizeMoveDist;
536 
537  if (debugSection()) {
538  stringstream ss;
539  ss << "DCSFpieceSpaced" << iPiece << ".brep";
540  BRepTools::Write(pieces.at(iPiece), ss.str().c_str());//debug
541  }
542  }
543 
544  //make a compound of the aligned pieces
545  BRep_Builder builder;
546  TopoDS_Compound comp;
547  builder.MakeCompound(comp);
548  for (auto& piece : pieces) {
549  builder.Add(comp, piece);
550  }
551 
552  //center the compound along SectionCS XDirection
553  Base::Vector3d centerVector = DU::toVector3d(gMovementVector) * distanceToMove / -2.0;
554  TopoDS_Shape centeredCompound = ShapeUtils::moveShape(comp, centerVector);
555 
556  if (debugSection()) {
557  BRepTools::Write(centeredCompound, "DCSmap40CenteredCompound.brep");//debug
558  }
559 
560  // re-align our shape with the projection CS
561  gp_Trsf xPieceAlign;
562  xPieceAlign.SetTransformation(getProjectionCS(), stdCS);
563  BRepBuilderAPI_Transform mkTransAlign(centeredCompound, xPieceAlign);
564  TopoDS_Shape alignedCompound = mkTransAlign.Shape();

879 //find the points where the section line changes direction, and the direction
880 //of the profile before and after the point
881 ChangePointVector DrawComplexSection::getChangePointsFromSectionLine()
882 {
883  // Base::Console().Message("DCS::getChangePointsFromSectionLine()\n");
884  ChangePointVector result;
885  std::vector<gp_Pnt> allPoints;
886  DrawViewPart* baseDvp = dynamic_cast<DrawViewPart*>(BaseView.getValue());
887  if (baseDvp) {
888  TopoDS_Wire lineWire = makeSectionLineWire();
889  TopoDS_Shape projectedWire =
890  GeometryObject::projectSimpleShape(lineWire, baseDvp->getProjectionCS());
891  if (projectedWire.IsNull()) {
892  return result;
893  }
894  //get UNIQUE points along the projected profile
895  TopExp_Explorer expVertex(projectedWire, TopAbs_VERTEX);
896  gp_Pnt previousPoint(Precision::Infinite(), Precision::Infinite(), Precision::Infinite());
897  for (; expVertex.More(); expVertex.Next()) {
898  TopoDS_Vertex vert = TopoDS::Vertex(expVertex.Current());
899  gp_Pnt gPoint = BRep_Tool::Pnt(vert);
900  if (gPoint.IsEqual(previousPoint, 2.0 * EWTOLERANCE)) {
901  continue;
902  }
903  allPoints.push_back(gPoint);
904  previousPoint = gPoint;
905  }
906 
907  //make the intermediate marks
908  for (size_t iPoint = 1; iPoint < allPoints.size() - 1; iPoint++) {
909  gp_Pnt location = allPoints.at(iPoint);
910  gp_Dir preDir = gp_Dir(allPoints.at(iPoint - 1).XYZ() - allPoints.at(iPoint).XYZ());
911  gp_Dir postDir = gp_Dir(allPoints.at(iPoint + 1).XYZ() - allPoints.at(iPoint).XYZ());
912  ChangePoint point(location, preDir, postDir);
913  result.push_back(point);
914  }
915 
916  //make start and end marks
917  gp_Pnt location0 = allPoints.at(0);
918  gp_Pnt location1 = allPoints.at(1);
919  gp_Dir postDir = gp_Dir(location1.XYZ() - location0.XYZ());
920  gp_Dir preDir = postDir.Reversed();
921  ChangePoint startPoint(location0, preDir, postDir);
922  result.push_back(startPoint);
923  location0 = allPoints.at(allPoints.size() - 1);
924  location1 = allPoints.at(allPoints.size() - 2);
925  preDir = gp_Dir(location0.XYZ() - location1.XYZ());
926  postDir = preDir.Reversed();
927  ChangePoint endPoint(location0, preDir, postDir);
928  result.push_back(endPoint);
929  }
930  return result;
931 }
