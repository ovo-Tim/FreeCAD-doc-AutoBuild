82 void ConstraintForce::onChanged(const App::Property* prop)
83 {
84  // Note: If we call this at the end, then the arrows are not oriented correctly initially
85  // because the NormalDirection has not been calculated yet
86  Constraint::onChanged(prop);
87 
88  if (prop == &Direction) {
89  Base::Vector3d direction = getDirection(Direction);
90  if (direction.Length() < Precision::Confusion()) {
91  return;
92  }
93  naturalDirectionVector = direction;
94  if (Reversed.getValue()) {
95  direction = -direction;
96  }
97  DirectionVector.setValue(direction);
98  }
99  else if (prop == &Reversed) {
100  // if the direction is invalid try to compute it again
101  if (naturalDirectionVector.Length() < Precision::Confusion()) {
102  naturalDirectionVector = getDirection(Direction);
103  }
104  if (naturalDirectionVector.Length() >= Precision::Confusion()) {
105  if (Reversed.getValue() && (DirectionVector.getValue() == naturalDirectionVector)) {
106  DirectionVector.setValue(-naturalDirectionVector);
107  }
108  else if (!Reversed.getValue()
109  && (DirectionVector.getValue() != naturalDirectionVector)) {
110  DirectionVector.setValue(naturalDirectionVector);
111  }
112  }
113  }
114  else if (prop == &NormalDirection) {
115  // Set a default direction if no direction reference has been given
116  if (!Direction.getValue()) {
117  Base::Vector3d direction = NormalDirection.getValue();
118  if (Reversed.getValue()) {
119  direction = -direction;
120  }
121  DirectionVector.setValue(direction);
122  naturalDirectionVector = direction;
123  }
124  }
125 }
