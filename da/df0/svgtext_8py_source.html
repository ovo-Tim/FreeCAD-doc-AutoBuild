39 def _get_text_techdraw(text, tcolor, fontsize, anchor,
40  align, fontname, angle, base,
41  linespacing):
42  """Return the SVG representation of text for TechDraw display.
43 
44  `text` is a list of textual elements; they are iterated, styled,
45  and added around a `<text>` tag.
46  ::
47  <text ...> text[0] </text>
48  <text ...> text[1] </text>
49  """
50  svg = ""
51  for i in range(len(text)):
52  delta = App.Vector(0, -i * linespacing, 0)
53  point = base + DraftVecUtils.rotate(delta, angle)
54 
55  _t = text[i].replace("&", "&amp;")
56  _t = _t.replace("<", "&lt;")
57  t = _t.replace(">", "&gt;")
58 
59  svg += '<text '
60  svg += 'stroke-width="0" stroke="{}" '.format(tcolor)
61  svg += 'fill="{}" font-size="{}" '.format(tcolor, fontsize)
62  svg += 'style="text-anchor:{};text-align:{};'.format(anchor,
63  align.lower())
64  svg += 'font-family:{}" '.format(fontname)
65  svg += 'transform="'
66  svg += 'rotate({},{},{}) '.format(math.degrees(angle),
67  point.x,
68  point.y)
69  svg += 'translate({},{}) '.format(point.x,
70  point.y)
71  svg += 'scale(1,-1)"'
72  # svg += 'freecad:skip="1"'
73  svg += '>\n'
74  svg += t
75  svg += '</text>\n'
76  return svg

79 def _get_text_header(tcolor, fontsize, anchor, align,
80  fontname, angle, base, flip):
81  """Return the initial <text> tag with style options.
82 
83  The text must be added after this tag, and then must be closed.
84  ::
85  <text ...>
86  ...
87  </text>
88  """
89  svg = '<text '
90  svg += 'stroke-width="0" stroke="{}" '.format(tcolor)
91  svg += 'fill="{}" font-size="{}" '.format(tcolor, fontsize)
92  svg += 'style="text-anchor:{};text-align:{};'.format(anchor,
93  align.lower())
94  svg += 'font-family:{}" '.format(fontname)
95  svg += 'transform="'
96  svg += 'rotate({},{},{}) '.format(math.degrees(angle),
97  base.x,
98  base.y)
99  if flip:
100  svg += 'translate({},{}) '.format(base.x, base.y)
101  else:
102  svg += 'translate({},{}) '.format(base.x, -base.y)
103  # svg += 'scale({},-{}) '.format(tmod/2000, tmod/2000)
104 
105  if flip:
106  svg += 'scale(1,-1) '
107  else:
108  svg += 'scale(1,1) '
109 
110  svg += '" '
111  svg += 'freecad:skip="1"'
112  svg += '>\n'
113  return svg

116 def get_text(plane, techdraw,
117  tcolor, fontsize, fontname,
118  angle, base, text,
119  linespacing=0.5, align="center", flip=True):
120  """Get the SVG representation of a textual element."""
121  if isinstance(angle, App.Rotation):
122  if not plane:
123  angle = angle.Angle
124  else:
125  if plane.axis.getAngle(angle.Axis) < 0.001:
126  angle = angle.Angle
127  elif abs(plane.axis.getAngle(angle.Axis) - math.pi) < 0.001:
128  if abs(angle.Angle) > 0.1:
129  angle = -angle.Angle
130  else:
131  angle = angle.Angle
132  elif abs(plane.axis.getAngle(angle.Axis) - math.pi/2) < 0.001:
133  # text is perpendicular to view, so it shouldn't appear
134  return ""
135  else:
136  # Compute an X vector
137  vec = App.Vector(1,0,0)
138  vec = angle.multVec(App.Vector(1,0,0))
139  angle = DraftVecUtils.angle(vec,plane.u)
140 
141  # text should be a list of strings separated by a newline
142  if not isinstance(text, list):
143  text = text.split("\n")
144 
145  if align.lower() == "center":
146  anchor = "middle"
147  elif align.lower() == "left":
148  anchor = "start"
149  else:
150  anchor = "end"
151 
152  if techdraw:
153  # For TechDraw display each item in the text list is placed
154  # in an individual tag.
155  # <text ...> text[0] </text>
156  # <text ...> text[1] </text>
157  svg = _get_text_techdraw(text, tcolor, fontsize, anchor,
158  align, fontname, angle, base,
159  linespacing)
160  else:
161  # If the SVG is not for TechDraw, and there is a single item
162  # in the text list, place it in a single tag.
163  # <text ...> text </text>
164  #
165  # For multiple elements, place each element inside a <tspan> tag.
166  # <text ...>
167  # <tspan>text[0]</tspan>
168  # <tspan>text[1]</tspan>
169  # </text>
170  svg = _get_text_header(tcolor, fontsize, anchor, align,
171  fontname, angle, base, flip)
172 
173  if len(text) == 1:
174  _t = text[0].replace("&", "&amp;").replace("<", "&lt;")
175  svg += _t.replace(">", "&gt;")
176  else:
177  for i in range(len(text)):
178  if i == 0:
179  svg += '<tspan>'
180  else:
181  svg += '<tspan x="0" dy="{}">'.format(linespacing)
182  _t = text[i].replace("&", "&amp;").replace("<", "&lt;")
183  svg += _t.replace(">", "&gt;")
184  svg += '</tspan>\n'
185  svg += '</text>\n'
186  return svg
