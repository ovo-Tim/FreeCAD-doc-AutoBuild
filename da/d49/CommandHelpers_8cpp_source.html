93 std::vector<std::string> CommandHelpers::getSelectedSubElements(Gui::Command* cmd,
94  TechDraw::DrawViewPart* &dvp,
95  std::string subType)
96 {
97  std::vector<std::string> selectedSubs;
98  std::vector<std::string> subNames;
99  dvp = nullptr;
100  std::vector<Gui::SelectionObject> selection = cmd->getSelection().getSelectionEx();
101  std::vector<Gui::SelectionObject>::iterator itSel = selection.begin();
102  for (; itSel != selection.end(); itSel++) {
103  if ((*itSel).getObject()->isDerivedFrom(TechDraw::DrawViewPart::getClassTypeId())) {
104  dvp = static_cast<TechDraw::DrawViewPart*> ((*itSel).getObject());
105  subNames = (*itSel).getSubNames();
106  break;
107  }
108  }
109  if (!dvp) {
110  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Wrong Selection"),
111  QObject::tr("No Part View in Selection"));
112  return selectedSubs;
113  }
114 
115  for (auto& s: subNames) {
116  if (TechDraw::DrawUtil::getGeomTypeFromName(s) == subType) {
117  selectedSubs.push_back(s);
118  }
119  }
120 
121  if (selectedSubs.empty()) {
122  QMessageBox::warning(Gui::getMainWindow(),
123  QObject::tr("Wrong Selection"),
124  QObject::tr("No %1 in Selection")
125  .arg(QString::fromStdString(subType)));
126  return selectedSubs;
127  }
128 
129  return selectedSubs;
130 }

134 void CommandHelpers::getSelectedShapes(Gui::Command* cmd,
135  std::vector<App::DocumentObject*>& shapes,
136  std::vector<App::DocumentObject*>& xShapes,
137  App::DocumentObject* faceObj,
138  std::string& faceName)
139 {
140  Gui::ResolveMode resolve = Gui::ResolveMode::OldStyleElement;//mystery
141  bool single = false; //mystery
142  auto selection = cmd->getSelection().getSelectionEx(nullptr, App::DocumentObject::getClassTypeId(),
143  resolve, single);
144  for (auto& sel : selection) {
145  bool is_linked = false;
146  auto obj = sel.getObject();
147  if (obj->isDerivedFrom(TechDraw::DrawPage::getClassTypeId())) {
148  continue;
149  }
150  if (obj->isDerivedFrom(App::LinkElement::getClassTypeId())
151  || obj->isDerivedFrom(App::LinkGroup::getClassTypeId())
152  || obj->isDerivedFrom(App::Link::getClassTypeId())) {
153  is_linked = true;
154  }
155  // If parent of the obj is a link to another document, we possibly need to treat non-link obj as linked, too
156  // 1st, is obj in another document?
157  if (obj->getDocument() != cmd->getDocument()) {
158  std::set<App::DocumentObject*> parents = obj->getInListEx(true);
159  for (auto& parent : parents) {
160  // Only consider parents in the current document, i.e. possible links in this View's document
161  if (parent->getDocument() != cmd->getDocument()) {
162  continue;
163  }
164  // 2nd, do we really have a link to obj?
165  if (parent->isDerivedFrom(App::LinkElement::getClassTypeId())
166  || parent->isDerivedFrom(App::LinkGroup::getClassTypeId())
167  || parent->isDerivedFrom(App::Link::getClassTypeId())) {
168  // We have a link chain from this document to obj, and obj is in another document -> it is an XLink target
169  is_linked = true;
170  }
171  }
172  }
173  if (is_linked) {
174  xShapes.push_back(obj);
175  continue;
176  }
177  //not a Link and not null. assume to be drawable. Undrawables will be
178  // skipped later.
179  shapes.push_back(obj);
180  if (faceObj) {
181  continue;
182  }
183  //don't know if this works for an XLink
184  for (auto& sub : sel.getSubNames()) {
185  if (TechDraw::DrawUtil::getGeomTypeFromName(sub) == "Face") {
186  faceName = sub;
187  //
188  faceObj = obj;
189  break;
190  }
191  }
192  }
193 }
