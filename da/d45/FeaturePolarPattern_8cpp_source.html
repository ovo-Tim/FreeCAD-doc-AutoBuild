77  Reversed.isTouched() ||

87 const std::list<gp_Trsf> PolarPattern::getTransformations(const std::vector<App::DocumentObject*>)
88 {
89  int occurrences = Occurrences.getValue();
90  if (occurrences < 1)
91  throw Base::ValueError("At least one occurrence required");
92 
93  if (occurrences == 1)
94  return {gp_Trsf()};
95 
96  bool reversed = Reversed.getValue();
97 
98  App::DocumentObject* refObject = Axis.getValue();
99  if (!refObject)
100  throw Base::ValueError("No axis reference specified");
101  std::vector<std::string> subStrings = Axis.getSubValues();
102  if (subStrings.empty())
103  throw Base::ValueError("No axis reference specified");
104 
105  gp_Pnt axbase;
106  gp_Dir axdir;
107  if (refObject->isDerivedFrom<Part::Part2DObject>()) {
108  Part::Part2DObject* refSketch = static_cast<Part::Part2DObject*>(refObject);
109  Base::Axis axis;
110  if (subStrings[0] == "H_Axis")
111  axis = refSketch->getAxis(Part::Part2DObject::H_Axis);
112  else if (subStrings[0] == "V_Axis")
113  axis = refSketch->getAxis(Part::Part2DObject::V_Axis);
114  else if (subStrings[0] == "N_Axis")
115  axis = refSketch->getAxis(Part::Part2DObject::N_Axis);
116  else if (subStrings[0].compare(0, 4, "Axis") == 0) {
117  int AxId = std::atoi(subStrings[0].substr(4,4000).c_str());
118  if (AxId >= 0 && AxId < refSketch->getAxisCount())
119  axis = refSketch->getAxis(AxId);
120  }
121  axis *= refSketch->Placement.getValue();
122  axbase = gp_Pnt(axis.getBase().x, axis.getBase().y, axis.getBase().z);
123  axdir = gp_Dir(axis.getDirection().x, axis.getDirection().y, axis.getDirection().z);
124  } else if (refObject->isDerivedFrom<PartDesign::Line>()) {
125  PartDesign::Line* line = static_cast<PartDesign::Line*>(refObject);
126  Base::Vector3d base = line->getBasePoint();
127  axbase = gp_Pnt(base.x, base.y, base.z);
128  Base::Vector3d dir = line->getDirection();
129  axdir = gp_Dir(dir.x, dir.y, dir.z);
130  } else if (refObject->isDerivedFrom<App::Line>()) {
131  App::Line* line = static_cast<App::Line*>(refObject);
132  Base::Vector3d d = line->getDirection();
133  axdir = gp_Dir(d.x, d.y, d.z);
134  } else if (refObject->isDerivedFrom<Part::Feature>()) {
135  if (subStrings[0].empty())
136  throw Base::ValueError("No axis reference specified");
137  Part::Feature* refFeature = static_cast<Part::Feature*>(refObject);
138  Part::TopoShape refShape = refFeature->Shape.getShape();
139  TopoDS_Shape ref = refShape.getSubShape(subStrings[0].c_str());
140 
141  if (ref.ShapeType() == TopAbs_EDGE) {
142  TopoDS_Edge refEdge = TopoDS::Edge(ref);
143  if (refEdge.IsNull())
144  throw Base::ValueError("Failed to extract axis edge");
145  BRepAdaptor_Curve adapt(refEdge);
146  if (adapt.GetType() == GeomAbs_Line) {
147  axbase = adapt.Line().Location();
148  axdir = adapt.Line().Direction();
149  } else if (adapt.GetType() == GeomAbs_Circle) {
150  axbase = adapt.Circle().Location();
151  axdir = adapt.Circle().Axis().Direction();
152  } else {
153  throw Base::TypeError("Rotation edge must be a straight line, circle or arc of circle");
154  }
155  } else {
156  throw Base::TypeError("Axis reference must be an edge");
157  }
158  } else {
159  throw Base::TypeError("Axis reference must be edge of a feature or datum line");
160  }
161  TopLoc_Location invObjLoc = this->getLocation().Inverted();
162  axbase.Transform(invObjLoc.Transformation());
163  axdir.Transform(invObjLoc.Transformation());
164 
165  gp_Ax2 axis(axbase, axdir);
166 
167  if (reversed)
168  axis.SetDirection(axis.Direction().Reversed());
169 
170  double angle;
171 
172  switch (static_cast<PolarPatternMode>(Mode.getValue())) {
173  case PolarPatternMode::angle:
174  angle = Angle.getValue();
175 
176  if (std::fabs(angle - 360.0) < Precision::Confusion())
177  angle /= occurrences; // Because e.g. two occurrences in 360 degrees need to be 180 degrees apart
178  else
179  angle /= occurrences - 1;
180 
181  break;
182 
183  case PolarPatternMode::offset:
184  angle = Offset.getValue();
185  break;
186 
187  default:
188  throw Base::ValueError("Invalid mode");
189  }
190 
191  double offset = Base::toRadians<double>(angle);
192 
193  if (offset < Precision::Angular())
194  throw Base::ValueError("Pattern angle too small");
195 
196  std::list<gp_Trsf> transformations;
197  gp_Trsf trans;
198  transformations.push_back(trans);
199 
200  // Note: The original feature is already included in the list of transformations!
201  // Therefore we start with occurrence number 1
202  for (int i = 1; i < occurrences; i++) {
203  trans.SetRotation(axis.Axis(), i * offset);
204  transformations.push_back(trans);
205  }
206 
207  return transformations;
208 }
