92 def loopdetect(obj, edge1, edge2):
93  """
94  Returns a loop wire that includes the two edges.
95  Useful for detecting boundaries of negative space features ie 'holes'
96  If a unique loop is not found, returns None
97  edge1 = edge
98  edge2 = edge
99  """
100 
101  Path.Log.track()
102  candidates = []
103  for wire in obj.Shape.Wires:
104  for e in wire.Edges:
105  if e.hashCode() == edge1.hashCode():
106  candidates.append((wire.hashCode(), wire))
107  if e.hashCode() == edge2.hashCode():
108  candidates.append((wire.hashCode(), wire))
109  loop = set([x for x in candidates if candidates.count(x) > 1]) # return the duplicate item
110  if len(loop) != 1:
111  return None
112  loopwire = next(x for x in loop)[1]
113  return loopwire

130 def horizontalFaceLoop(obj, face, faceList=None):
131  """horizontalFaceLoop(obj, face, faceList=None) ... returns a list of face names which form the walls of a vertical hole face is a part of.
132  All face names listed in faceList must be part of the hole for the solution to be returned."""
133 
134  wires = [horizontalEdgeLoop(obj, e) for e in face.Edges]
135  # Not sure if sorting by Area is a premature optimization - but it seems
136  # the loop we're looking for is typically the biggest of the them all.
137  wires = sorted([w for w in wires if w], key=lambda w: Part.Face(w).Area)
138 
139  for wire in wires:
140  hashes = [e.hashCode() for e in wire.Edges]
141 
142  # find all faces that share a an edge with the wire and are vertical
143  faces = [
144  "Face%d" % (i + 1)
145  for i, f in enumerate(obj.Shape.Faces)
146  if any(e.hashCode() in hashes for e in f.Edges) and Path.Geom.isVertical(f)
147  ]
148 
149  if faceList and not all(f in faces for f in faceList):
150  continue
151 
152  # verify they form a valid hole by getting the outline and comparing
153  # the resulting XY footprint with that of the faces
154  comp = Part.makeCompound([obj.Shape.getElement(f) for f in faces])
155  outline = TechDraw.findShapeOutline(comp, 1, Vector(0, 0, 1))
156 
157  # findShapeOutline always returns closed wires, by removing the
158  # trace-backs single edge spikes don't contribute to the bound box
159  uniqueEdges = []
160  for edge in outline.Edges:
161  if any(Path.Geom.edgesMatch(edge, e) for e in uniqueEdges):
162  continue
163  uniqueEdges.append(edge)
164  w = Part.Wire(uniqueEdges)
165 
166  # if the faces really form the walls of a hole then the resulting
167  # wire is still closed and it still has the same footprint
168  bb1 = comp.BoundBox
169  bb2 = w.BoundBox
170  if (
171  w.isClosed()
172  and Path.Geom.isRoughly(bb1.XMin, bb2.XMin)
173  and Path.Geom.isRoughly(bb1.XMax, bb2.XMax)
174  and Path.Geom.isRoughly(bb1.YMin, bb2.YMin)
175  and Path.Geom.isRoughly(bb1.YMax, bb2.YMax)
176  ):
177  return faces
178  return None

445 def sort_locations(locations, keys, attractors=None):
446  """sort holes by the nearest neighbor method
447  keys: two-element list of keys for X and Y coordinates. for example ['x','y']
448  originally written by m0n5t3r for PathHelix
449  """
450  from queue import PriorityQueue
451  from collections import defaultdict
452 
453  if attractors is None:
454  attractors = []
455 
456  attractors = attractors or [keys[0]]
457 
458  def sqdist(a, b):
459  """square Euclidean distance"""
460  d = 0
461  for k in keys:
462  d += (a[k] - b[k]) ** 2
463 
464  return d
465 
466  def weight(location):
467  w = 0
468 
469  for k in attractors:
470  w += abs(location[k])
471 
472  return w
473 
474  def find_closest(location_list, location, dist):
475  q = PriorityQueue()
476 
477  for i, j in enumerate(location_list):
478  # prevent dictionary comparison by inserting the index
479  q.put((dist(j, location) + weight(j), i, j))
480 
481  prio, i, result = q.get()
482 
483  return result
484 
485  out = []
486  zero = defaultdict(lambda: 0)
487 
488  out.append(find_closest(locations, zero, sqdist))
489  locations.remove(out[-1])
490 
491  while locations:
492  closest = find_closest(locations, out[-1], sqdist)
493  out.append(closest)
494  locations.remove(closest)
495 
496  return out

688  if self.__z_finish_step < total_depth:
689  depths.append(self.__z_finish_step + self.__final_depth)
690  else:

692 
693  if equalstep:

695  else:
696  depths += self.__fixed_steps(self.__start_depth, depths[-1], self.__step_down)[1:]

705  def __filter_roughly_equal_depths(self, depths):
706  """Depths arrive sorted from largest to smallest, positive to negative.
707  Return unique list of depths, using Path.Geom.isRoughly() method to determine
708  if the two values are equal. Only one of two consecutive equals are removed.
709 
710  The assumption is that there are not enough consecutively roughly-equal depths
711  to be removed, so as to eliminate an effective step-down depth with the removal
712  of repetitive roughly-equal values."""
713 
714  depthcopy = sorted(depths) # make a copy and sort low to high
715  keep = [depthcopy[0]]
716  for depth in depthcopy[1:]:
717  if not Path.Geom.isRoughly(depth, keep[-1]):
718  keep.append(depth)
719  keep.reverse() # reverse results back high to low
720  return keep

722  def __equal_steps(self, start, stop, max_size):
723  """returns a list of depths beginning with the bottom (included), ending
724  with the top (not included).
725  all steps are of equal size, which is as big as possible but not bigger
726  than max_size."""
727 
728  steps_needed = math.ceil((start - stop) / max_size)
729  depths = list(linspace(stop, start, steps_needed, endpoint=False))
730 
731  return depths

733  def __fixed_steps(self, start, stop, size):
734  """returns a list of depths beginning with the bottom (included), ending
735  with the top (not included).
736  all steps are of size 'size' except the one at the bottom which can be
737  smaller."""
738 
739  fullsteps = int((start - stop) / size)
740  last_step = start - (fullsteps * size)
741  depths = list(linspace(last_step, start, fullsteps, endpoint=False))
742 
743  if last_step == stop:
744  return depths
745  else:
746  return [stop] + depths

749 def simplify3dLine(line, tolerance=1e-4):
750  """Simplify a line defined by a list of App.Vectors, while keeping the
751  maximum deviation from the original line within the defined tolerance.
752  Implementation of
753  https://en.wikipedia.org/wiki/Ramer%E2%80%93Douglas%E2%80%93Peucker_algorithm"""
754  stack = [(0, len(line) - 1)]
755  results = []
756 
757  def processRange(start, end):
758  """Internal worker. Process a range of Vector indices within the
759  line."""
760  if end - start < 2:
761  results.extend(line[start:end])
762  return
763  # Find point with maximum distance
764  maxIndex, maxDistance = 0, 0.0
765  startPoint, endPoint = (line[start], line[end])
766  for i in range(start + 1, end):
767  v = line[i]
768  distance = v.distanceToLineSegment(startPoint, endPoint).Length
769  if distance > maxDistance:
770  maxDistance = distance
771  maxIndex = i
772  if maxDistance > tolerance:
773  # Push second branch first, to be executed last
774  stack.append((maxIndex, end))
775  stack.append((start, maxIndex))
776  else:
777  results.append(line[start])
778 
779  while len(stack):
780  processRange(*stack.pop())
781  # Each segment only appended its start point to the final result, so fill in
782  # the last point.
783  results.append(line[-1])
784  return results

787 def RtoIJ(startpoint, command):
788  """
789  This function takes a startpoint and an arc command in radius mode and
790  returns an arc command in IJ mode. Useful for preprocessor scripts
791  """
792  if "R" not in command.Parameters:
793  raise ValueError("No R parameter in command")
794  if command.Name not in ["G2", "G02", "G03", "G3"]:
795  raise ValueError("Not an arc command")
796 
797  endpoint = command.Placement.Base
798  radius = command.Parameters["R"]
799 
800  # calculate the IJ
801  # we take a vector between the start and endpoints
802  chord = endpoint.sub(startpoint)
803 
804  # Take its perpendicular (we assume the arc is in the XY plane)
805  perp = chord.cross(Vector(0, 0, 1))
806 
807  # use pythagoras to get the perp length
808  plength = math.sqrt(radius**2 - (chord.Length / 2) ** 2)
809  perp.normalize()
810  perp.scale(plength, plength, plength)
811 
812  # Calculate the relative center
813  relativecenter = chord.scale(0.5, 0.5, 0.5).add(perp)
814 
815  # build new command
816  params = {c: command.Parameters[c] for c in "XYZF" if c in command.Parameters}
817  params["I"] = relativecenter.x
818  params["J"] = relativecenter.y
819 
820  newcommand = Path.Command(command.Name)
821  newcommand.Parameters = params
822 
823  return newcommand
