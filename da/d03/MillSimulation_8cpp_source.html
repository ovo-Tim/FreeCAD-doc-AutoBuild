87 void MillSimulation::InitSimulation(float quality)
88 {
89  ClearMillPathSegments();
90  millPathLine.Clear();
91  simDisplay.applySSAO = guiDisplay.IsChecked(eGuiItemAmbientOclusion);
92 
93  mDestMotion = mZeroPos;
94  // gDestPos = curMillOperation->startPos;
95  mCurStep = 0;
96  mPathStep = -1;
97  mNTotalSteps = 0;
98  mSimPlaying = false;
99  mSimSpeed = 1;
100  MillPathSegment::SetQuality(quality, simDisplay.maxFar);
101  int nOperations = (int)mCodeParser.Operations.size();
102  int segId = 0;
103  for (int i = 0; i < nOperations; i++) {
104  mCurMotion = mDestMotion;
105  mDestMotion = mCodeParser.Operations[i];
106  EndMill* tool = GetTool(mDestMotion.tool);
107  if (tool != nullptr) {
108  MillSim::MillPathSegment* segment =
109  new MillSim::MillPathSegment(tool, &mCurMotion, &mDestMotion);
110  segment->indexInArray = i;
111  segment->segmentIndex = segId++;
112  mNTotalSteps += segment->numSimSteps;
113  MillPathSegments.push_back(segment);
114  segment->AppendPathPoints(millPathLine.MillPathPointsBuffer);
115  }
116  }
117  mNPathSteps = (int)MillPathSegments.size();
118  millPathLine.GenerateModel();
119  InitDisplay(quality);
120 }

247 void MillSimulation::CalcSegmentPositions()
248 {
249  mSubStep = mCurStep;
250  for (mPathStep = 0; mPathStep < mNPathSteps; mPathStep++) {
251  MillSim::MillPathSegment* p = MillPathSegments[mPathStep];
252  if (mSubStep < p->numSimSteps) {
253  break;
254  }
255  mSubStep -= p->numSimSteps;
256  }
257  if (mPathStep >= mNPathSteps) {
258  mPathStep = mNPathSteps - 1;
259  mSubStep = MillPathSegments[mPathStep]->numSimSteps;
260  }
261  else {
262  mSubStep++;
263  }
264 }

266 void MillSimulation::RenderSimulation()
267 {
268  if ((mViewItems & VIEWITEM_SIMULATION) == 0) {
269  return;
270  }
271 
272  simDisplay.StartDepthPass();
273 
274  GlsimStart();
275  mStockObject.render();
276 
277  GlsimToolStep2();
278 
279  for (int i = 0; i <= mPathStep; i++) {
280  renderSegmentForward(i);
281  }
282 
283  for (int i = mPathStep; i >= 0; i--) {
284  renderSegmentForward(i);
285  }
286 
287  for (int i = 0; i < mPathStep; i++) {
288  renderSegmentReversed(i);
289  }
290 
291  for (int i = mPathStep; i >= 0; i--) {
292  renderSegmentReversed(i);
293  }
294 
295  GlsimClipBack();
296  mStockObject.render();
297 
298  // start coloring
299  simDisplay.StartGeometryPass(stockColor, false);
300  GlsimRenderStock();
301  mStockObject.render();
302 
303  // render cuts (back faces of tools)
304  simDisplay.StartGeometryPass(cutColor, true);
305  GlsimRenderTools();
306  for (int i = 0; i <= mPathStep; i++) {
307  MillSim::MillPathSegment* p = MillPathSegments.at(i);
308  int step = (i == mPathStep) ? mSubStep : p->numSimSteps;
309  int start = p->isMultyPart ? 1 : step;
310  for (int j = start; j <= step; j++) {
311  MillPathSegments.at(i)->render(j);
312  }
313  }
314 
315  GlsimEnd();
316 }

399 void MillSimulation::ProcessSim(unsigned int time_ms)
400 {
401 
402  static int ancient = 0;
403  static unsigned int last = 0;
404  static unsigned int msec = 0xFFFFFFFF;
405  static int fps = 0;
406  static int renderTime = 0;
407 
408  last = msec == 0xFFFFFFFF ? time_ms : msec;
409  msec = time_ms;
410  if (guiDisplay.IsChecked(eGuiItemRotate)) {
411  simDisplay.RotateEye((msec - last) / 4600.0f);
412  }
413 
414  if (last / 1000 != msec / 1000) {
415  float calcFps = 1000.0f * fps / (msec - ancient);
416  mFpsStream.str("");
417  mFpsStream << "fps: " << calcFps << " rendertime:" << renderTime
418  << " zpos:" << mDestMotion.z << std::ends;
419  ancient = msec;
420  fps = 0;
421  }
422 
423  if (mSimPlaying || mSingleStep) {
424  SimNext();
425  mSingleStep = false;
426  }
427 
428  Render();
429 
430  ++fps;
431 }
