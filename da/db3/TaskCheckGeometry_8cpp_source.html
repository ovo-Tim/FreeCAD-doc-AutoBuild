203 void ResultEntry::buildEntryName()
204 {
205  ResultEntry* parentEntry = this;
206  while (parentEntry->parent) {
207  ResultEntry* temp = parentEntry->parent;
208  if (!temp->parent)
209  break;
210  parentEntry = parentEntry->parent;
211  }
212 
213  QString stringOut;
214  QTextStream stream(&stringOut);
215  TopTools_IndexedMapOfShape shapeMap;
216  int index(-1);
217 
218  switch (this->shape.ShapeType())
219  {
220  case TopAbs_COMPOUND:
221  TopExp::MapShapes(parentEntry->shape, TopAbs_COMPOUND, shapeMap);
222  stream << "Compound";
223  break;
224  case TopAbs_COMPSOLID:
225  TopExp::MapShapes(parentEntry->shape, TopAbs_COMPSOLID, shapeMap);
226  stream << "CompSolid";
227  break;
228  case TopAbs_SOLID:
229  TopExp::MapShapes(parentEntry->shape, TopAbs_SOLID, shapeMap);
230  stream << "Solid";
231  break;
232  case TopAbs_SHELL:
233  TopExp::MapShapes(parentEntry->shape, TopAbs_SHELL, shapeMap);
234  stream << "Shell";
235  break;
236  case TopAbs_WIRE:
237  TopExp::MapShapes(parentEntry->shape, TopAbs_WIRE, shapeMap);
238  stream << "Wire";
239  break;
240  case TopAbs_FACE:
241  TopExp::MapShapes(parentEntry->shape, TopAbs_FACE, shapeMap);
242  stream << "Face";
243  break;
244  case TopAbs_EDGE:
245  TopExp::MapShapes(parentEntry->shape, TopAbs_EDGE, shapeMap);
246  stream << "Edge";

413 void TaskCheckGeometryResults::goCheck()
414 {
415  Gui::WaitCursor wc;
416  auto selection = Gui::Selection().getSelection();
417 
418  int selectedCount(0), checkedCount(0), invalidShapes(0);
419  ResultEntry *theRoot = new ResultEntry();
420  reportViewStrings.clear();
421  reportViewStrings << QLatin1String("\n");
422 
423  std::string scopeName {tr("Boolean operation check...").toStdString()};
424 #if OCC_VERSION_HEX < 0x070500
425  Handle(Message_ProgressIndicator) theProgress = new BOPProgressIndicator(tr("Check geometry"),
426  Gui::getMainWindow());
427  theProgress->NewScope(scopeName.c_str());
428  theProgress->Show();
429 #else
430  Handle(Message_ProgressIndicator) theProgress = new BOPProgressIndicator(tr("Check geometry"),
431  Gui::getMainWindow());
432  Message_ProgressRange theRange(theProgress->Start());
433  Message_ProgressScope theScope(theRange,
434  TCollection_AsciiString(scopeName.c_str()),
435  selection.size());
436  theScope.Show();
437 #endif // 0x070500
438 
439  for(const auto &sel : selection) {
440  selectedCount++;
441  int localInvalidShapeCount(0);
442  QString baseName;
443  QTextStream baseStream(&baseName);
444  baseStream << sel.DocName;
445  baseStream << "." << sel.FeatName;
446  std::string label = sel.pObject->Label.getValue();
447  if (sel.FeatName != label) {
448  baseStream << " (" << label.c_str() << ")";
449  }
450 
451  TopoDS_Shape shape = Part::Feature::getShape(sel.pObject,sel.SubName,true);
452  if (shape.IsNull()) {
453  ResultEntry *entry = new ResultEntry();
454  entry->parent = theRoot;
455  entry->name = baseName;
456  entry->type = tr("Null shape");
457  entry->error = tr("Skipped");
458  reportViewStrings.append(entry->name + QLatin1String(" | ")
459  + entry->type + QLatin1String(" | ") + entry->error);
460  theRoot->children.push_back(entry);
461  continue;
462  }
463  if (shape.Infinite()) {
464  ResultEntry *entry = new ResultEntry();
465  entry->parent = theRoot;
466  entry->name = baseName;
467  entry->type = tr("Infinite shape");
468  entry->error = tr("Skipped");
469  reportViewStrings.append(entry->name + QLatin1String(" | ")
470  + entry->type + QLatin1String(" | ") + entry->error);
471  theRoot->children.push_back(entry);
472  continue;
473  }
474  currentSeparator = Gui::Application::Instance->getViewProvider(sel.pObject)->getRoot();
475  if (!currentSeparator) {
476  continue;
477  }
478 
479  checkedCount++;
480  checkedMap.Clear();
481 
482  buildShapeContent(sel.pObject, baseName, shape);
483 
484  BRepCheck_Analyzer shapeCheck(shape);
485  if (!shapeCheck.IsValid())
486  {
487  invalidShapes++;
488  localInvalidShapeCount++;
489  ResultEntry *entry = new ResultEntry();
490  entry->parent = theRoot;
491  entry->shape = shape;
492  entry->name = baseName;
493  entry->type = shapeEnumToString(shape.ShapeType());
494  entry->error = tr("Invalid");
495  reportViewStrings.append(entry->name + QLatin1String(" | ")
496  + entry->type + QLatin1String(" | ") + entry->error);
497  entry->viewProviderRoot = currentSeparator;
498  entry->viewProviderRoot->ref();
499  goSetupResultBoundingBox(entry);
500  theRoot->children.push_back(entry);
501  recursiveCheck(shapeCheck, shape, entry);
502  continue; //don't run BOPAlgo_ArgumentAnalyzer if BRepCheck_Analyzer finds something.
503  }
504  else
505  {
506  //BOPAlgo_ArgumentAnalyzer can be really slow!
507  //so only run it when the shape seems valid to BRepCheck_Analyzer And
508  //when the option is set.
509 
510  ParameterGrp::handle group = App::GetApplication().GetUserParameter().
511  GetGroup("BaseApp")->GetGroup("Preferences")->GetGroup("Mod")->GetGroup("Part")->GetGroup("CheckGeometry");
512  bool runSignal = group->GetBool("RunBOPCheck", false);
513  group->SetBool("RunBOPCheck", runSignal);
514  if (runSignal) {
515  std::string label = tr("Checking").toStdString() + " ";
516  label += sel.pObject->Label.getStrValue();
517  label += "...";
518 #if OCC_VERSION_HEX < 0x070500
519  theProgress->NewScope(label.c_str());
520  localInvalidShapeCount += goBOPSingleCheck(shape, theRoot, baseName, theProgress);
521  invalidShapes += localInvalidShapeCount;
522  theProgress->EndScope();
523  if (theProgress->UserBreak())
524  break;
525 #else
526  Message_ProgressScope theInnerScope(theScope.Next(), TCollection_AsciiString(label.c_str()), 1);
527  theInnerScope.Show();
528  localInvalidShapeCount += goBOPSingleCheck(shape, theRoot, baseName, theInnerScope);
529  invalidShapes += localInvalidShapeCount;
530  theInnerScope.Close();
531  if (theScope.UserBreak())
532  break;
533 #endif
534  }
535  }
536  // create an entry for shapes without errors
537  if (localInvalidShapeCount == 0) {
538  ResultEntry *entry = new ResultEntry();
539  entry->parent = theRoot;
540  entry->name = baseName;
541  entry->type = shapeEnumToString(shape.ShapeType());
542  entry->error = tr("No errors");
543  reportViewStrings.append(entry->name + QLatin1String(" | ")
544  + entry->type + QLatin1String(" | ") + entry->error);
545  entry->viewProviderRoot = currentSeparator;
546  entry->viewProviderRoot->ref();
547  theRoot->children.push_back(entry);
548  }
549  }
550  model->setResults(theRoot);
551  treeView->expandAll();
552  treeView->header()->resizeSections(QHeaderView::ResizeToContents);
553  QString aMessage {tr("%1 processed out of %2 selected").arg(checkedCount).arg(selectedCount)};
554  aMessage += QLatin1String("<br/>") + tr("%n invalid shapes.", "", invalidShapes);
555  aMessage += QLatin1String(" <a href=\"#report\">") + tr("To Report view.") + QLatin1String("</a>");
556  message->setText(aMessage);
557 }

566 void TaskCheckGeometryResults::recursiveCheck(const BRepCheck_Analyzer &shapeCheck, const TopoDS_Shape &shape,
567  ResultEntry *parent)
568 {
569  ResultEntry *branchNode = parent;
570  BRepCheck_ListIteratorOfListOfStatus listIt;
571  if (!shapeCheck.Result(shape).IsNull() && !checkedMap.Contains(shape))
572  {
573  listIt.Initialize(shapeCheck.Result(shape)->Status());
574  if (listIt.Value() != BRepCheck_NoError)
575  {
576  ResultEntry *entry = new ResultEntry();
577  entry->parent = parent;
578  entry->shape = shape;
579  entry->buildEntryName();
580  entry->type = shapeEnumToString(shape.ShapeType());
581  entry->error = checkStatusToString(listIt.Value());
582  reportViewStrings.append(entry->name + QLatin1String(" | ")
583  + entry->type + QLatin1String(" | ") + entry->error);
584  entry->viewProviderRoot = currentSeparator;
585  entry->viewProviderRoot->ref();
586  dispatchError(entry, listIt.Value());
587  parent->children.push_back(entry);
588  branchNode = entry;
589  }
590  }
591  checkedMap.Add(shape);
592 
593  if (shape.ShapeType() == TopAbs_SOLID)
594  checkSub(shapeCheck, shape, TopAbs_SHELL, branchNode);
595  if (shape.ShapeType() == TopAbs_EDGE)
596  checkSub(shapeCheck, shape, TopAbs_VERTEX, branchNode);
597  if (shape.ShapeType() == TopAbs_FACE)
598  {
599  checkSub(shapeCheck, shape, TopAbs_WIRE, branchNode);
600  checkSub(shapeCheck, shape, TopAbs_EDGE, branchNode);
601  checkSub(shapeCheck, shape, TopAbs_VERTEX, branchNode);
602  }
603 
604  for (TopoDS_Iterator it(shape); it.More(); it.Next())
605  recursiveCheck(shapeCheck, it.Value(), branchNode);
606 }

608 void TaskCheckGeometryResults::checkSub(const BRepCheck_Analyzer &shapeCheck, const TopoDS_Shape &shape,
609  const TopAbs_ShapeEnum subType, ResultEntry *parent)
610 {
611  BRepCheck_ListIteratorOfListOfStatus itl;
612  TopExp_Explorer exp;
613  for (exp.Init(shape,subType); exp.More(); exp.Next())
614  {
615  const Handle(BRepCheck_Result)& res = shapeCheck.Result(exp.Current());
616  const TopoDS_Shape& sub = exp.Current();
