341  "recursively get the shapes of objects inside this BuildingPart"
342 
343  shapes = []

346  for child in Draft.get_group_contents(obj, walls=True):

351  matname = "Undefined"
352  if hasattr(child,"Material") and child.Material:
353  matname = child.Material.Name
354  if matname in materialstable:
355  materialstable[matname] = materialstable[matname]+","+str(solidindex)
356  else:
357  materialstable[matname] = str(solidindex)
358  solidindex += 1
359  return shapes,materialstable
