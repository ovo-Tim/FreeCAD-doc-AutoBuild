274 bool AttachExtension::changeAttacherType(const char* typeName, bool base)
275 {
276  auto& prop = base ? _baseProps : _props;
277 
278  // check if we need to actually change anything
279  if (prop.attacher) {
280  if (strcmp(prop.attacher->getTypeId().getName(), typeName) == 0) {
281  return false;
282  }
283  }
284  else if (strlen(typeName) == 0) {
285  return false;
286  }
287  if (strlen(typeName) == 0) {
288  setAttacher(nullptr, base);
289  return true;
290  }
291  Base::Type t = Base::Type::fromName(typeName);
292  if (t.isDerivedFrom(AttachEngine::getClassTypeId())) {
293  AttachEngine* pNewAttacher =
294  static_cast<Attacher::AttachEngine*>(Base::Type::createInstanceByName(typeName));
295  this->setAttacher(pNewAttacher, base);
296  return true;
297  }
298 
299  std::stringstream errMsg;
300  errMsg << "Object if this type is not derived from AttachEngine: " << typeName;
301  throw AttachEngineException(errMsg.str());
302 }

304 bool AttachExtension::positionBySupport()
305 {
306  _active = 0;
307  if (!_props.attacher) {
308  throw Base::RuntimeError(
309  "AttachExtension: can't positionBySupport, because no AttachEngine is set.");
310  }
311  updateAttacherVals();
312  Base::Placement plaOriginal = getPlacement().getValue();
313  try {
314  if (_props.attacher->mapMode == mmDeactivated) {
315  return false;
316  }
317  bool subChanged = false;
318 
319  getPlacement().setValue(Base::Placement());
320 
321  Base::Placement basePlacement;
322  if (_baseProps.attacher && _baseProps.attacher->mapMode != mmDeactivated) {
323  basePlacement =
324  _baseProps.attacher->calculateAttachedPlacement(Base::Placement(), &subChanged);
325  if (subChanged) {
326  _baseProps.attachment->setValues(_baseProps.attachment->getValues(),
327  _baseProps.attacher->getSubValues());
328  }
329  }
330 
331  subChanged = false;
332  _props.attacher->setOffset(AttachmentOffset.getValue() * basePlacement.inverse());
333  auto placement = _props.attacher->calculateAttachedPlacement(plaOriginal, &subChanged);
334  if (subChanged) {
335  Base::ObjectStatusLocker<App::Property::Status, App::Property> guard(
336  App::Property::User3,
337  &AttachmentSupport);
338  AttachmentSupport.setValues(AttachmentSupport.getValues(),
339  _props.attacher->getSubValues());
340  }
341  getPlacement().setValue(placement);
342  _active = 1;
343  return true;
344  }
345  catch (ExceptionCancel&) {
346  // disabled, don't do anything
347  getPlacement().setValue(plaOriginal);
348  return false;
349  }
350  catch (Base::Exception&) {
351  getPlacement().setValue(plaOriginal);
352  throw;
353  }
354  catch (Standard_Failure&) {
355  getPlacement().setValue(plaOriginal);
356  throw;
357  }
358 }

401 void AttachExtension::extensionOnChanged(const App::Property* prop)
402 {
403  if (!getExtendedObject()->isRestoring()) {
404  // If we change anything that affects our position, update it immediately so you can see it
405  // interactively.
406  if ((prop == &AttachmentSupport
407  || prop == &MapMode
408  || prop == &MapPathParameter
409  || prop == &MapReversed
410  || prop == &AttachmentOffset)){
411  bool bAttached = false;
412  try{
413  bAttached = positionBySupport();
414  }
415  catch (Base::Exception &e) {
416  getExtendedObject()->setStatus(App::Error, true);
417  Base::Console().Error("PositionBySupport: %s\n",e.what());
418  //set error message - how?
419  }
420  catch (Standard_Failure &e){
421  getExtendedObject()->setStatus(App::Error, true);
422  Base::Console().Error("PositionBySupport: %s\n",e.GetMessageString());
423  }
424 
425  updateSinglePropertyStatus(bAttached);
426  }
427  if (prop == &AttacherEngine) {
428  AttacherType.setValue(enumToClass(AttacherEngine.getValueAsString()));
429  }
430  else if (_props.matchProperty(prop)) {
431  _active = -1;
432  updateAttacherVals(/*base*/ false);
433  updatePropertyStatus(isAttacherActive());
434  }
435  else if (_baseProps.matchProperty(prop)) {
436  _active = -1;
437  updateAttacherVals(/*base*/ true);
438  updatePropertyStatus(isAttacherActive(), /*base*/ true);
439  }
440  }
441  if (prop == &(this->AttacherType)) {
442  this->changeAttacherType(this->AttacherType.getValue());
443  }
444  else if (prop == _baseProps.attacherType) {
445  this->changeAttacherType(_baseProps.attacherType->getValue());
446  }
447 
448  App::DocumentObjectExtension::extensionOnChanged(prop);
449 }

451 bool AttachExtension::extensionHandleChangedPropertyName(Base::XMLReader& reader,
452  const char* TypeName,
453  const char* PropName)
454 {
455  Base::Type type = Base::Type::fromName(TypeName);
456  // superPlacement -> AttachmentOffset
457  if (strcmp(PropName, "superPlacement") == 0 && AttachmentOffset.getClassTypeId() == type) {
458  AttachmentOffset.Restore(reader);
459  return true;
460  }
461  // Support -> AttachmentSupport
462  if (strcmp(PropName, "Support") == 0) {
463  // At one point, the type of Support changed from PropertyLinkSub to its present type
464  // of PropertyLinkSubList. Later, the property name changed to AttachmentSupport
465  App::PropertyLinkSub tmp;
466  if (strcmp(tmp.getTypeId().getName(), TypeName) == 0) {
467  tmp.setContainer(this->getExtendedContainer());
468  tmp.Restore(reader);
469  AttachmentSupport.setValue(tmp.getValue(), tmp.getSubValues());
470  this->MapMode.setValue(Attacher::mmFlatFace);
471  return true;
472  }
473  if (AttachmentSupport.getClassTypeId() == type) {
474  AttachmentSupport.Restore(reader);
475  return true;
476  }
477  }
478  return App::DocumentObjectExtension::extensionHandleChangedPropertyName(reader, TypeName, PropName);
479 }

506 void AttachExtension::updateSinglePropertyStatus(bool bAttached, bool base)
507 {
508  auto& props = base ? this->_baseProps : this->_props;
509  if (!props.mapMode) {
510  return;
511  }
512 
513  // Hide properties when not applicable to reduce user confusion
514  eMapMode mmode = eMapMode(props.mapMode->getValue());
515  bool modeIsPointOnCurve =
516  (mmode == mmNormalToPath || mmode == mmFrenetNB || mmode == mmFrenetTN
517  || mmode == mmFrenetTB || mmode == mmRevolutionSection || mmode == mmConcentric);
518 
519  // MapPathParameter is only used if there is a reference to one edge and not edge + vertex
520  bool hasOneRef = props.attacher && props.attacher->subnames.size() == 1;
521 
522  props.mapPathParameter->setStatus(App::Property::Status::Hidden, !bAttached || !(modeIsPointOnCurve && hasOneRef));
523  props.mapReversed->setStatus(App::Property::Status::Hidden, !bAttached);
524 
525  if (base) {
526  props.attachment->setStatus(App::Property::Status::Hidden, !bAttached);
527  }
528  else {
529  this->AttachmentOffset.setStatus(App::Property::Status::Hidden, !bAttached);
530  if (getExtendedContainer()) {
531  // for mmTranslate, orientation should remain editable even when attached.
532  getPlacement().setReadOnly(bAttached && mmode != mmTranslate);
533  }
534  }
