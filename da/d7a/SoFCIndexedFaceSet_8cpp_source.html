118 void MeshRenderer::Private::generateGLArrays(SoGLRenderAction* action,
119  SoMaterialBindingElement::Binding matbind,
120  std::vector<float>& vertex,
121  std::vector<int32_t>& index)
122 {
123  if (vertex.empty() || index.empty()) {
124  return;
125  }
126 
127  // lazy initialization
128  vertices.setCurrentContext(action->getCacheContext());
129  indices.setCurrentContext(action->getCacheContext());
130 
131  initialized = true;
132  vertices.create();
133  indices.create();
134 
135  vertices.bind();
136  vertices.allocate(vertex.data(), vertex.size() * sizeof(float));
137  vertices.release();
138 
139  indices.bind();
140  indices.allocate(index.data(), index.size() * sizeof(int32_t));
141  indices.release();
142  this->matbinding = matbind;
143 }

145 void MeshRenderer::Private::renderGLArray(SoGLRenderAction* action, GLenum mode)
146 {
147  if (!initialized) {
148  SoDebugError::postWarning("MeshRenderer", "not initialized");
149  return;
150  }
151 
152  vertices.setCurrentContext(action->getCacheContext());
153  indices.setCurrentContext(action->getCacheContext());
154 
155  glEnableClientState(GL_VERTEX_ARRAY);
156  glEnableClientState(GL_NORMAL_ARRAY);
157  glEnableClientState(GL_COLOR_ARRAY);
158 
159  vertices.bind();
160  indices.bind();
161 
162  if (matbinding != SoMaterialBindingElement::OVERALL) {
163  glInterleavedArrays(GL_C4F_N3F_V3F, 0, nullptr);
164  }
165  else {
166  glInterleavedArrays(GL_N3F_V3F, 0, nullptr);
167  }
168 
169  glDrawElements(mode, indices.size() / sizeof(uint32_t), GL_UNSIGNED_INT, nullptr);
170 
171  vertices.release();
172  indices.release();
173 
174  glDisableClientState(GL_COLOR_ARRAY);
175  glDisableClientState(GL_NORMAL_ARRAY);
176  glDisableClientState(GL_VERTEX_ARRAY);
177 }

389 bool MeshRenderer::matchMaterial(SoState* state) const
390 {
391  // FIXME: There is sometimes a minor problem that in wireframe
392  // mode the colors do not match. The steps to reproduce
393  // * set mesh to shaded mode
394  // * open function to remove components and select an area
395  // * set to wireframe mode
396  // => the material of the shaded mode instead of that of the
397  // wireframe mode
398  SoMaterialBindingElement::Binding matbind = SoMaterialBindingElement::get(state);
399  if (p->matbinding != matbind) {
400  return false;
401  }
402  // the buffer doesn't contain color information
403  if (matbind == SoMaterialBindingElement::OVERALL) {
404  return true;
405  }
406  const SbColor* pcolors = nullptr;
407  SoGLLazyElement* gl = SoGLLazyElement::getInstance(state);
408  if (gl) {
409  pcolors = gl->getDiffusePointer();
410  }
411  return p->pcolors == pcolors;
412 }

474 void SoFCIndexedFaceSet::GLRender(SoGLRenderAction* action)
475 {
476  if (this->coordIndex.getNum() < 3) {
477  return;
478  }
479 
480  if (!this->shouldGLRender(action)) {
481  // Transparency is handled inside 'shouldGLRender' but the base class
482  // somehow misses to reset the blending mode. This causes SoGLLazyElement
483  // not to switch on and off GL_BLEND mode and thus transparency doesn't
484  // work as expected. Calling SoMaterialBundle::sendFirst seems to fix the
485  // problem.
486  SoMaterialBundle mb(action);
487  mb.sendFirst();
488  return;
489  }
490 
491 #if defined(RENDER_GL_VAO)
492  SoState* state = action->getState();
493 
494  // get the VBO status of the viewer
495  SbBool useVBO = true;
496  Gui::SoGLVBOActivatedElement::get(state, useVBO);
497 
498  // Check for a matching OpenGL context
499  if (!render.canRenderGLArray(action)) {
500  useVBO = false;
501  }
502 
503  // use VBO for fast rendering if possible
504  if (useVBO) {
505  if (updateGLArray.getValue()) {
506  updateGLArray.setValue(false);
507  render.update();
508  generateGLArrays(action);
509  }
510  else if (render.needUpdate(action)) {
511  generateGLArrays(action);
512  }
513 
514  if (render.matchMaterial(state)) {
515  SoMaterialBundle mb(action);
516  mb.sendFirst();
517  render.renderFacesGLArray(action);
518  }
519  else {
520  drawFaces(action);
521  }
522  }
523  else {
524  drawFaces(action);
525  }
526 #else
527  drawFaces(action);
528 #endif
529 }

531 void SoFCIndexedFaceSet::drawFaces(SoGLRenderAction* action)
532 {
533  SoState* state = action->getState();
534  SbBool mode = Gui::SoFCInteractiveElement::get(state);
535 
536  unsigned int num = this->coordIndex.getNum() / 4;
537  if (!mode || num <= this->renderTriangleLimit) {
538 #ifdef RENDER_GLARRAYS
539  SoMaterialBindingElement::Binding matbind = SoMaterialBindingElement::get(state);
540 
541  SbBool matchCtx = render.canRenderGLArray(action);
542  if (matbind == SoMaterialBindingElement::OVERALL && matchCtx) {
543  SoMaterialBundle mb(action);
544  mb.sendFirst();
545  if (updateGLArray.getValue()) {
546  updateGLArray.setValue(false);
547  generateGLArrays(action);
548  }
549  render.renderFacesGLArray(action);
550  }
551  else {
552  inherited::GLRender(action);
553  }
554 #else
555  inherited::GLRender(action);
556 #endif
557  }
558  else {
559 #if 0 && defined(RENDER_GLARRAYS)
560  SoMaterialBundle mb(action);
561  mb.sendFirst();
562  render.renderCoordsGLArray(action);
563 #else
564  SoMaterialBindingElement::Binding matbind = SoMaterialBindingElement::get(state);
565  int32_t binding = (int32_t)(matbind);
566 
567  const SoCoordinateElement* coords = nullptr;
568  const SbVec3f* normals = nullptr;
569  const int32_t* cindices = nullptr;
570  int numindices = 0;
571  const int32_t* nindices = nullptr;
572  const int32_t* tindices = nullptr;
573  const int32_t* mindices = nullptr;
574  SbBool normalCacheUsed {};
575 
576  SoMaterialBundle mb(action);
577 
578  SoTextureCoordinateBundle tb(action, true, false);
579  SbBool sendNormals = !mb.isColorOnly() || tb.isFunction();
580 
581  this->getVertexData(state,
582  coords,
583  normals,
584  cindices,
585  nindices,
586  tindices,
587  mindices,
588  numindices,
589  sendNormals,
590  normalCacheUsed);
591 
592  mb.sendFirst(); // make sure we have the correct material
593 
594  drawCoords(static_cast<const SoGLCoordinateElement*>(coords),
595  cindices,
596  numindices,
597  normals,
598  nindices,
599  &mb,
600  mindices,
601  binding,
602  &tb,
603  tindices);
604 
605  // getVertexData() internally calls readLockNormalCache() that read locks
606  // the normal cache. When the cache is not needed any more we must call
607  // readUnlockNormalCache()
608  if (normalCacheUsed) {
609  this->readUnlockNormalCache();
610  }
611 
612  // Disable caching for this node
613  SoGLCacheContextElement::shouldAutoCache(state, SoGLCacheContextElement::DONT_AUTO_CACHE);
614 #endif
615  }

618 void SoFCIndexedFaceSet::drawCoords(const SoGLCoordinateElement* const vertexlist,
619  const int32_t* vertexindices,
620  int numindices,
621  const SbVec3f* normals,
622  const int32_t* normalindices,
623  SoMaterialBundle* materials,
624  const int32_t* /*matindices*/,
625  const int32_t binding,
626  const SoTextureCoordinateBundle* const /*texcoords*/,
627  const int32_t* /*texindices*/)
628 {
629  const SbVec3f* coords3d = nullptr;
630  coords3d = vertexlist->getArrayPtr3();
631 
632  int mod = numindices / (4 * this->renderTriangleLimit) + 1;
633  float size = std::min<float>((float)mod, 3.0F);
634  glPointSize(size);
635 
636  SbBool per_face = false;
637  SbBool per_vert = false;
638  switch (binding) {
639  case SoMaterialBindingElement::PER_FACE:
640  per_face = true;
641  break;
642  case SoMaterialBindingElement::PER_VERTEX:
643  per_vert = true;
644  break;
645  default:
646  break;
647  }
648 
649  int ct = 0;
650  const int32_t* viptr = vertexindices;
651  int32_t v1 {}, v2 {}, v3 {};
652  SbVec3f dummynormal(0, 0, 1);
653  const SbVec3f* currnormal = &dummynormal;
654  if (normals) {
655  currnormal = normals;
656  }
657 
658  glBegin(GL_POINTS);
659  for (int index = 0; index < numindices; ct++) {
660  if (ct % mod == 0) {
661  if (per_face) {
662  materials->send(ct, true);
663  }
664  v1 = *viptr++;
665  index++;
666  if (per_vert) {
667  materials->send(v1, true);
668  }
669  if (normals) {
670  currnormal = &normals[*normalindices++];
671  }
672  glNormal3fv((const GLfloat*)currnormal);
673  glVertex3fv((const GLfloat*)(coords3d + v1));
674 
675  v2 = *viptr++;
676  index++;
677  if (per_vert) {
678  materials->send(v2, true);
679  }
680  if (normals) {
681  currnormal = &normals[*normalindices++];
682  }
683  glNormal3fv((const GLfloat*)currnormal);
684  glVertex3fv((const GLfloat*)(coords3d + v2));
685 
686  v3 = *viptr++;
687  index++;
688  if (per_vert) {
689  materials->send(v3, true);
690  }
691  if (normals) {
692  currnormal = &normals[*normalindices++];
693  }
694  glNormal3fv((const GLfloat*)currnormal);
695  glVertex3fv((const GLfloat*)(coords3d + v3));
696  }
697  else {
698  viptr++;
699  index++;
700  normalindices++;
701  viptr++;
702  index++;
703  normalindices++;
704  viptr++;
705  index++;
706  normalindices++;
707  }
708 
709  viptr++;
710  index++;
711  normalindices++;
712  }
713  glEnd();
714 }

722 {
723  const SoCoordinateElement* coords = nullptr;
724  const SbVec3f* normals = nullptr;
725  const int32_t* cindices = nullptr;
726  const SbColor* pcolors = nullptr;
727  const float* transp = nullptr;
728  int numindices = 0, numcolors = 0, numtransp = 0;
729  const int32_t* nindices = nullptr;
730  const int32_t* tindices = nullptr;
731  const int32_t* mindices = nullptr;
732  SbBool normalCacheUsed {};
733 
734  SbBool sendNormals = true;
735 
736  SoState* state = action->getState();
737  this->getVertexData(state,
738  coords,
739  normals,
740  cindices,
741  nindices,
742  tindices,
743  mindices,

746  normalCacheUsed);
747 
748  const SbVec3f* points = coords->getArrayPtr3();
