1946 void CDxfRead::ResolveEntityAttributes()
1947 {
1948  m_entityAttributes.ResolveBylayerAttributes(*this);
1949  // TODO: Look at the space and layer (hidden/frozen?) and options and return false if the entity
1950  // is not needed.
1951  // TODO: INSERT must not call this because an INSERT on a hidden layer should always be
1952  // honoured.
1953 
1954  // Calculate the net entity transformation.
1955 
1956  // This is to handle the Object Coordinate System used in many DXF records. Note that versions
1957  // before R13 used the term ECS (Entity Coordinate System) instead. Here's who uses OCS: Lines
1958  // and Points use WCS except they can be extruded (have nonzero Thickness (39)) which occurs in
1959  // the OCS Z direction all 3D objects use WCS entirely Dimensions use a mix of OCS and WCS,
1960  // Circle, Arc, Dolid, Trace, Text, Attib, Attdef, Shape, Insert, (lw)Polyline/Vertex, hatch,
1961  // image all use the OCS
1962  //
1963  // The transformed Z axis is in EntityNormalVector, but we rescale it in case the DXF contains
1964  // an unnormalized value
1965  if (EntityNormalVector.IsNull()) {
1966  ImportError("Entity has zero-length extrusion direction\n");
1967  }
1968  EntityNormalVector.Normalize();
1969  // Apply the Arbitrary Axis Algorithm to determine the X and Y directions
1970  // The purpose of this algorithm is to calculate a conventional 3d orientation based only on a Z
1971  // direction, while avoiding taking the cross product of two vectors that are nearly parallel,
1972  // which would be subject to a lot of numerical inaccuracy. In this case, "close to" the Z axis
1973  // means the X and Y components of EntityNormalVector are less than 1/64, a value chosen because
1974  // it is exactly representable in all binary floating-point systems.
1975  Base::Vector3d xDirection;
1976  if (EntityNormalVector.x < ArbitraryAxisAlgorithmThreshold
1977  && EntityNormalVector.y < ArbitraryAxisAlgorithmThreshold) {
1978  // The Z axis is close to the UCS Z axis, the X direction is UCSY × OCSZ
1979  static const Base::Vector3d UCSYAxis(0, 1, 0);
1980  xDirection = UCSYAxis % EntityNormalVector;
1981  }
1982  else {
1983  // otherwise, the X direction is UCSZ × OCSZ
1984  static const Base::Vector3d UCSZAxis(0, 0, 1);
1985  xDirection = UCSZAxis % EntityNormalVector;
1986  }
1987  OCSOrientationTransform.setCol(0, xDirection);
1988  // In all cases the Y direction is the Zdirection × XDirection which gives a right-hand
1989  // orthonormal coordinate system
1990  OCSOrientationTransform.setCol(1, EntityNormalVector % xDirection);
1991  // and EntityNormalVector is of course the direction of the Z axis in the UCS.
1992  OCSOrientationTransform.setCol(2, EntityNormalVector);
1993 }

2554 {

3006 inline static double level(int distance, double blackLevel)
3007 {
3008  // Distance is the number of 24ths around the color wheel between the desired hue and
3009  // the primary hue in question. Coming in it is a ordinate difference and so can be negative
3010  // so the first thing we do is take its absolute value.
3011  if (distance < 0) {
3012  distance = -distance;
3013  }
3014  // If the distance is greater than 12, it is measuring the long way around the color wheel so we
3015  // reduce it to measuring along the short way instead
3016  if (distance > 12) {
3017  distance = 24 - distance;
3018  }
3019  if (distance <= 4) {
3020  // A distance 4 or less givs full intensity of the primary color
3021  return 1.0;
3022  }
3023  if (distance < 8) {
3024  // Between 4 and 8 gives a blend of the full primary and the black level
3025  return ((8 - distance) + blackLevel * (distance - 4)) / 4;
3026  }
3027  // 8 and beyond yield the black level
3028  return blackLevel;
3029 }

3036 App::Color CDxfRead::ObjectColor(ColorIndex_t index)
3037 {

3042 
3043  // The first 7 colors (1-7) have ad hoc names red, yellow, green, cyan, blue, magenta, and

3045  // to contrast with the background color.

3047  // color circle, with even numbers fading to black on the tens digit,
