194  def align_to_edge_or_wire(self, shape, offset=0):
195  """Align the WP to an edge or wire with an optional offset.
196 
197  The shape must define a plane.
198 
199  If the shape is an edge with a `Center` then that defines the WP
200  `position`. The vector between the center and its start point then
201  defines the WP `u` axis.
202 
203  In other cases the start point of the first edge defines the WP
204  `position` and the 1st derivative at that point the WP `u` axis.
205 
206  Parameters
207  ----------
208  shape: Part.Edge or Part.Wire
209  Edge or wire.
210  offset: float, optional
211  Defaults to zero.
212  Offset along the WP `axis`.
213 
214  Returns
215  -------
216  `True`/`False`
217  `True` if successful.
218  """
219  tol = 1e-7
220  plane = shape.findPlane()
221  if plane is None:
222  return False
223  self.axis = plane.Axis
224  if shape.ShapeType == "Edge" and hasattr(shape.Curve, "Center"):
225  pos = shape.Curve.Center
226  vec = shape.Vertexes[0].Point - pos
227  if vec.Length > tol:
228  self.u = vec
229  self.u.normalize()
230  self.v = self.axis.cross(self.u)
231  else:
232  self.u, self.v, _ = self._axes_from_rotation(plane.Rotation)
233  elif shape.Edges[0].Length > tol:
234  pos = shape.Vertexes[0].Point
235  self.u = shape.Edges[0].derivative1At(0)
236  self.u.normalize()
237  self.v = self.axis.cross(self.u)
238  else:
239  pos = shape.Vertexes[0].Point
240  self.u, self.v, _ = self._axes_from_rotation(plane.Rotation)
241  self.position = pos + (self.axis * offset)
242  return True

293  def align_to_point_and_axis(self, point, axis, offset=0, upvec=Vector(1, 0, 0)):
294  """Align the WP to a point and an axis with an optional offset and an
295  optional up-vector.
296 
297  If the axis and up-vector are parallel the FreeCAD.Rotation algorithm
298  will replace the up-vector: Vector(0, 0, 1) is tried first, then
299  Vector(0, 1, 0), and finally Vector(1, 0, 0).
300 
301  Parameters
302  ----------
303  point: Base.Vector
304  New WP `position`.
305  axis: Base.Vector
306  New WP `axis`.
307  offset: float, optional
308  Defaults to zero.
309  Offset along the WP `axis`.
310  upvec: Base.Vector, optional
311  Defaults to Vector(1, 0, 0).
312  Up-vector.
313 
314  Returns
315  -------
316  `True`
317  """
318  tol = 1e-7
319  if axis.Length < tol:
320  return False
321  if upvec.Length < tol:
322  return False
323  axis = Vector(axis).normalize()
324  upvec = Vector(upvec).normalize()
325  if axis.isEqual(upvec, tol) or axis.isEqual(upvec.negative(), tol):

327  rot = FreeCAD.Rotation(Vector(), upvec, axis, "ZYX")
328  self.u, self.v, _ = self._axes_from_rotation(rot)
329  self.axis = axis
330  self.position = point + (self.axis * offset)

333  def align_to_point_and_axis_svg(self, point, axis, offset=0):
334  """Align the WP to a point and an axis with an optional offset.
335 
336  It aligns `u` and `v` based on the magnitude of the components
337  of `axis`.
338 
339  Parameters
340  ----------
341  point: Base.Vector
342  The new `position` of the plane, adjusted by
343  the `offset`.
344  axis: Base.Vector
345  A vector whose unit vector will be used as the new `axis`
346  of the plane.
347  The magnitudes of the `x`, `y`, `z` components of the axis
348  determine the orientation of `u` and `v` of the plane.
349  offset: float, optional
350  Defaults to zero. A value which will be used to offset
351  the plane in the direction of its `axis`.
352 
353  Returns
354  -------
355  `True`
356 
357  Cases
358  -----
359  The `u` and `v` are always calculated the same
360 
361  * `u` is the cross product of the positive or negative of `axis`
362  with a `reference vector`.
363  ::
364  u = [+1|-1] axis.cross(ref_vec)
365  * `v` is `u` rotated 90 degrees around `axis`.
366 
367  Whether the `axis` is positive or negative, and which reference
368  vector is used, depends on the absolute values of the `x`, `y`, `z`
369  components of the `axis` unit vector.
370 
371  #. If `x > y`, and `y > z`
372  The reference vector is +Z
373  ::
374  u = -1 axis.cross(+Z)
375  #. If `y > z`, and `z >= x`
376  The reference vector is +X.
377  ::
378  u = -1 axis.cross(+X)
379  #. If `y >= x`, and `x > z`
380  The reference vector is +Z.
381  ::
382  u = +1 axis.cross(+Z)
383  #. If `x > z`, and `z >= y`
384  The reference vector is +Y.
385  ::
386  u = +1 axis.cross(+Y)
387  #. If `z >= y`, and `y > x`
388  The reference vector is +X.
389  ::
390  u = +1 axis.cross(+X)
391  #. otherwise
392  The reference vector is +Y.
393  ::
394  u = -1 axis.cross(+Y)
395  """
396  self.axis = Vector(axis).normalize()
397  ref_vec = Vector(0.0, 1.0, 0.0)
398 
399  if ((abs(axis.x) > abs(axis.y)) and (abs(axis.y) > abs(axis.z))):
400  ref_vec = Vector(0.0, 0., 1.0)
401  self.u = axis.negative().cross(ref_vec)
402  self.u.normalize()
403  self.v = DraftVecUtils.rotate(self.u, math.pi/2, self.axis)
404  # projcase = "Case new"
405 
406  elif ((abs(axis.y) > abs(axis.z)) and (abs(axis.z) >= abs(axis.x))):
407  ref_vec = Vector(1.0, 0.0, 0.0)
408  self.u = axis.negative().cross(ref_vec)
409  self.u.normalize()
410  self.v = DraftVecUtils.rotate(self.u, math.pi/2, self.axis)
411  # projcase = "Y>Z, View Y"
412 
413  elif ((abs(axis.y) >= abs(axis.x)) and (abs(axis.x) > abs(axis.z))):
414  ref_vec = Vector(0.0, 0., 1.0)
415  self.u = axis.cross(ref_vec)
416  self.u.normalize()
417  self.v = DraftVecUtils.rotate(self.u, math.pi/2, self.axis)
418  # projcase = "ehem. XY, Case XY"
419 
420  elif ((abs(axis.x) > abs(axis.z)) and (abs(axis.z) >= abs(axis.y))):
421  self.u = axis.cross(ref_vec)
422  self.u.normalize()
423  self.v = DraftVecUtils.rotate(self.u, math.pi/2, self.axis)
424  # projcase = "X>Z, View X"
425 
426  elif ((abs(axis.z) >= abs(axis.y)) and (abs(axis.y) > abs(axis.x))):
427  ref_vec = Vector(1.0, 0., 0.0)
428  self.u = axis.cross(ref_vec)
429  self.u.normalize()
430  self.v = DraftVecUtils.rotate(self.u, math.pi/2, self.axis)
431  # projcase = "Y>X, Case YZ"
432 
433  else:
434  self.u = axis.negative().cross(ref_vec)
435  self.u.normalize()
436  self.v = DraftVecUtils.rotate(self.u, math.pi/2, self.axis)
437  # projcase = "else"
438

440  # spat_res = spat_vec.dot(axis)
441  # Console.PrintMessage(projcase + " spat Prod = " + str(spat_res) + "\n")

443  offsetVector = Vector(axis)
444  offsetVector.multiply(offset)
445  self.position = point.add(offsetVector)
