35 EndMill::EndMill(const std::vector<float>& toolProfile, int toolid, float diameter)
36  : EndMill(toolid, diameter)
37 {
38  profilePoints.clear();
39 
40  int srcBuffSize = toolProfile.size();
41  nPoints = srcBuffSize / 2;
42  if (nPoints < 2) {
43  return;
44  }
45 
46  // make sure last point is at 0,0 else, add it
47  bool missingCenterPoint = fabs(toolProfile[nPoints * 2 - 2]) > 0.0001f;
48  if (missingCenterPoint) {
49  nPoints++;
50  }
51 
52  int buffSize = PROFILE_BUFFER_SIZE(nPoints);
53  profilePoints.resize(buffSize);
54 
55  // copy profile points
56  for (int i = 0; i < srcBuffSize; i += 2) {
57  // add some width to reduce simulation artifacts
58  profilePoints[i] = toolProfile[i] + diameter * 0.01f;
59  profilePoints[i + 1] = toolProfile[i + 1] - diameter * 0.01f;
60  }
61  if (missingCenterPoint) {
62  profilePoints[srcBuffSize] = 0.0F;
63  profilePoints[srcBuffSize + 1] = profilePoints[srcBuffSize - 1];
64  }
65 
66  MirrorPointBuffer();
67 }
