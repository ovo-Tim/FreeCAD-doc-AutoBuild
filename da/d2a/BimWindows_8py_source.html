98  for window in windows:

102  elif self.form.groupMode.currentIndex() == 2:
103  if window.CloneOf:
104  group = window.CloneOf.Label
105  else:
106  group = window.Name
107  elif self.form.groupMode.currentIndex() == 3:
108  group = window.Tag
109  else:
110  if window.Material:
111  group = window.Material.Label
112  if not group:
113  group = "None"
114  if group in groups:
115  groups[group].append(window)
116  else:
117  groups[group] = [window]
118  for group in groups.keys():
119  s1 = group
120  top = QtGui.QTreeWidgetItem([s1, ""])
121  top.setExpanded(True)
122  self.form.windows.addTopLevelItem(top)
123  for window in groups[group]:
124  s1 = window.Label
125  s2 = window.Tag
126  it = QtGui.QTreeWidgetItem([s1, s2])
127  it.setIcon(0, QtGui.QIcon(":/icons/Arch_Window_Tree.svg"))
128  it.setToolTip(0, window.Name)
