58 MillPathSegment::MillPathSegment(EndMill* _endmill, MillMotion* from, MillMotion* to)
59 {
60  mat4x4_identity(mShearMat);
61  MotionPosToVec(mStartPos, from);
62  MotionPosToVec(mDiff, to);
63  vec3_sub(mDiff, mDiff, mStartPos);
64  mXYDistance = sqrtf(mDiff[PX] * mDiff[PX] + mDiff[PY] * mDiff[PY]);
65  mZDistance = fabsf(mDiff[PY]);
66  mXYZDistance = sqrtf(mXYDistance * mXYDistance + mDiff[PZ] * mDiff[PZ]);
67  mXYAngle = atan2f(mDiff[PY], mDiff[PX]);
68  endmill = _endmill;
69  mStartAngRad = mStepAngRad = 0;
70  if (IsArcMotion(to)) {
71  mMotionType = MTCurved;
72  mRadius = sqrtf(to->j * to->j + to->i * to->i);
73  mSmallRad = mRadius <= endmill->radius;
74 
75  if (mSmallRad) {
76  mStepAngRad = mSmallRadStep;
77  }
78  else {
79  mStepAngRad = asinf(mResolution / mRadius);
80  if (mStepAngRad > MAX_SEG_DEG) {
81  mStepAngRad = MAX_SEG_DEG;
82  }
83  else if (mStepAngRad < NIN_SEG_DEG) {
84  mStepAngRad = NIN_SEG_DEG;
85  }
86  }
87 
88  MotionPosToVec(mCenter, from);
89  mCenter[PX] += to->i;
90  mCenter[PY] += to->j;
91  mArcDir = to->cmd == eRotateCCW ? -1.f : 1.f;
92  mStartAngRad = atan2f(mCenter[PX] - from->x, from->y - mCenter[PY]);
93  float endAng = atan2f(mCenter[PX] - to->x, to->y - mCenter[PY]);
94  mSweepAng = (mStartAngRad - endAng) * mArcDir;
95  if (mSweepAng < EPSILON) {
96  mSweepAng += PI * 2;
97  }
98  numSimSteps = (int)(mSweepAng / mStepAngRad) + 1;
99  mStepAngRad = mArcDir * mSweepAng / numSimSteps;
100  if (mSmallRad) {
101  // when the radius is too small, we just use the tool itself to carve the stock
102  mShape = endmill->toolShape;
103  }
104  else {
105  endmill->GenerateArcSegmentDL(mRadius,
106  mStepAngRad * SWEEP_ARC_PAD,
107  mDiff[PZ] / numSimSteps,
108  &mShape);
109  numSimSteps++;
110  }
111 
112  isMultyPart = true;
113  }
114  else {
115  numSimSteps = (int)(mXYZDistance / mResolution);
116  if (numSimSteps == 0) {
117  numSimSteps = 1;
118  }
119  isMultyPart = false;
120  mStepDistance = mXYDistance / numSimSteps;
121  mStepLength[PX] = mDiff[PX];
122  mStepLength[PY] = mDiff[PY];
123  mStepLength[PZ] = mDiff[PZ];
124  vec3_scale(mStepLength, mStepLength, 1.f / (float)numSimSteps);
125 
126  if (IsVerticalMotion(from, to)) {
127  mMotionType = MTVertical;
128  }
129  else {
130  mMotionType = MTHorizontal;
131  mShearMat[0][2] = mDiff[PZ] / mXYDistance;
132  }
133  }
134 }

142 void MillPathSegment::AppendPathPoints(std::vector<MillPathPosition>& pointsBuffer)
143 {
144  MillPathPosition mpPos;
145  if (mMotionType == MTCurved) {
146  float ang = mStartAngRad;
147  float z = mStartPos[PZ];
148  float zStep = mDiff[PZ] / numSimSteps;
149  for (int i = 1; i < numSimSteps; i++) {
150  ang -= mStepAngRad;
151  z += zStep;
152  mpPos.X = mCenter[PX] - sinf(ang) * mRadius;
153  mpPos.Y = mCenter[PY] + cosf(ang) * mRadius;
154  mpPos.Z = z;
155  mpPos.SegmentId = segmentIndex;
156  pointsBuffer.push_back(mpPos);
157  }
158  }
159  else {
160  mpPos.X = mStartPos[PX] + mDiff[PX];
161  mpPos.Y = mStartPos[PY] + mDiff[PY];
162  mpPos.Z = mStartPos[PZ] + mDiff[PZ];
163  mpPos.SegmentId = segmentIndex;
164  pointsBuffer.push_back(mpPos);
165  }
166 }

168 void MillPathSegment::render(int step)
169 {
170  mStepNumber = step;
171  mat4x4 mat, mat2, rmat;
172  mat4x4_identity(mat);
173  mat4x4_identity(rmat);
174  if (mMotionType == MTCurved) {
175  mat4x4_translate_in_place(mat,
176  mCenter[PX],
177  mCenter[PY],
178  mCenter[PZ] + mDiff[PZ] * (step - 1) / numSimSteps);
179  mat4x4_rotate_Z(mat, mat, mStartAngRad - (step - 1) * mStepAngRad);
180  mat4x4_rotate_Z(rmat, rmat, mStartAngRad - (step - 1) * mStepAngRad);
181 
182  if (mSmallRad || step == numSimSteps) {
183  mat4x4_translate_in_place(mat, 0, mRadius, 0);
184  endmill->toolShape.Render(mat, rmat);
185  }
186  else {
187  mShape.Render(mat, rmat);
188  }
189  }
190  else {
191  if (mMotionType == MTVertical) {
192  if (mStepLength[PZ] > 0) {
193  mat4x4_translate_in_place_v(mat, mStartPos);
194  }
195  else {
196  mat4x4_translate_in_place(mat,
197  mStartPos[PX],
198  mStartPos[PY],
199  mStartPos[PZ] + mStepNumber * mStepLength[PZ]);
200  }
201  endmill->toolShape.Render(mat, rmat);
202  }
203  else {
204  float renderDist = step * mStepDistance;
205  mat4x4_translate_in_place_v(mat, mStartPos);
206  mat4x4_rotate_Z(mat, mat, mXYAngle);
207  mat4x4_rotate_Z(rmat, rmat, mXYAngle);
208  mat4x4_dup(mat2, mat);
209  if (mDiff[PZ] != 0.0) {
210  mat4x4_mul(mat2, mat2, mShearMat);
211  }
212  mat4x4_scale_aniso(mat2, mat2, renderDist, 1, 1);
213  endmill->pathShape.Render(mat2, rmat);
214  mat4x4_translate_in_place(mat, renderDist, 0, mDiff[PZ]);
215  endmill->halfToolShape.Render(mat, rmat);
216  }
217  }
218 }

220 void MillPathSegment::GetHeadPosition(vec3 headPos)
221 {
222  if (mMotionType == MTCurved) {
223  float angRad = mStartAngRad - mStepNumber * mStepAngRad;
224  vec3_set(mHeadPos, -mRadius * sinf(angRad), mRadius * cosf(angRad), 0);
225  vec3_add(mHeadPos, mHeadPos, mCenter);
226  }
227  else {
228  vec3_dup(mHeadPos, mStepLength);
229  vec3_scale(mHeadPos, mHeadPos, (float)mStepNumber);
230  vec3_add(mHeadPos, mHeadPos, mStartPos);
231  }
232  vec3_dup(headPos, mHeadPos);
233 }
234 float MillPathSegment::SetQuality(float quality, float maxStockDimension)
235 {
236  mResolution = maxStockDimension * 0.05 / quality;
237  if (mResolution > 4) {
238  mResolution = 4;
239  }
240  if (mResolution < 0.5) {
241  mResolution = 0.5;
242  }
243  mSmallRadStep = PI / 8;
244  if (quality < 4) {
245  mSmallRadStep = PI / 2;
246  }
247  else if (quality < 8) {
248  mSmallRadStep = PI / 4;
249  }
250  return mResolution;
