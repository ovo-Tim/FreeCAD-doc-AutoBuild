553 // The purpose of this function is to synchronize the mutated copy to the
554 // original linked CopyOnChange object. It will make a new copy if any of the
555 // non-CopyOnChange property of the original object has changed.
556 void LinkBaseExtension::syncCopyOnChange()
557 {
558  if (!getLinkCopyOnChangeValue()) {
559  return;
560  }
561  auto linkProp = getLinkedObjectProperty();
562  auto srcProp = getLinkCopyOnChangeSourceProperty();
563  auto srcTouched = getLinkCopyOnChangeTouchedProperty();
564  if (!linkProp || !srcProp || !srcTouched || !srcProp->getValue() || !linkProp->getValue()
565  || srcProp->getValue() == linkProp->getValue()) {
566  return;
567  }
568 
569  auto parent = getContainer();
570 
571  auto linked = linkProp->getValue();
572 
573  std::vector<App::DocumentObjectT> oldObjs;
574  std::vector<App::DocumentObject*> objs;
575 
576  // CopyOnChangeGroup is a hidden dynamic property for holding a LinkGroup
577  // for holding the mutated copy of the original linked object and all its
578  // dependencies.
579  LinkGroup* copyOnChangeGroup = nullptr;
580  if (auto prop = getLinkCopyOnChangeGroupProperty()) {
581  copyOnChangeGroup = Base::freecad_dynamic_cast<LinkGroup>(prop->getValue());
582  if (!copyOnChangeGroup) {
583  // Create the LinkGroup if not exist
584  auto group = new LinkGroup;
585  group->LinkMode.setValue(LinkModeAutoDelete);
586  parent->getDocument()->addObject(group, "CopyOnChangeGroup");
587  prop->setValue(group);
588  }
589  else {
590  // If it exists, then obtain all copied objects. Note that we stores
591  // the dynamic property _SourceUUID in oldObjs if possible, in order
592  // to match the possible new copy later.
593  objs = copyOnChangeGroup->ElementList.getValues();
594  for (auto obj : objs) {
595  if (!obj->isAttachedToDocument()) {
596  continue;
597  }
598  auto prop =
599  Base::freecad_dynamic_cast<PropertyUUID>(obj->getPropertyByName("_SourceUUID"));
600  if (prop && prop->getContainer() == obj) {
601  oldObjs.emplace_back(prop);
602  }
603  else {
604  oldObjs.emplace_back(obj);
605  }
606  }
607  std::sort(objs.begin(), objs.end());
608  }
609  }
610 
611  // Obtain the original linked object and its dependency in depending order.
612  // The last being the original linked object.
613  auto srcObjs = getOnChangeCopyObjects();
614  // Refresh signal connection to monitor changes
615  monitorOnChangeCopyObjects(srcObjs);
616 
617  // Copy the objects. Document::export/importObjects() (called by
618  // copyObject()) will generate a _ObjectUUID for each source object and
619  // match it with a _SourceUUID in the copy.
620  auto copiedObjs = parent->getDocument()->copyObject(srcObjs);
621  if (copiedObjs.empty()) {
622  return;
623  }
624 
625  // copyObject() will return copy in order of the same order of the input,
626  // so the last object will be the copy of the original linked object
627  auto newLinked = copiedObjs.back();
628 
629  // We are copying from the original linked object and we've already mutated
630  // it, so we need to copy all CopyOnChange properties from the mutated
631  // object to the new copy.
632  std::vector<App::Property*> propList;
633  linked->getPropertyList(propList);
634  for (auto prop : propList) {
635  if (!prop->testStatus(Property::CopyOnChange) || prop->getContainer() != linked) {
636  continue;
637  }
638  auto p = newLinked->getPropertyByName(prop->getName());
639  if (p && p->getTypeId() == prop->getTypeId()) {
640  std::unique_ptr<Property> pCopy(prop->Copy());
641  p->Paste(*pCopy);
642  }
643  }
644 
645  if (copyOnChangeGroup) {
646  // The order of the copied objects is in dependency order (because of
647  // getOnChangeCopyObjects()). We reverse it here so that we can later
648  // on delete it in reverse order to avoid error (because some parent
649  // objects may want to delete their own children).
650  std::reverse(copiedObjs.begin(), copiedObjs.end());
651  copyOnChangeGroup->ElementList.setValues(copiedObjs);
652  }
653 
654  // Create a map to find the corresponding replacement of the new copies to
655  // the mutated object. The reason for doing so is that we are copying from
656  // the original linked object and its dependency, not the mutated objects
657  // which are old copies. There could be arbitrary changes in the originals
658  // which may add or remove or change depending orders, while the
659  // replacement happens between the new and old copies.
660 
661  std::map<Base::Uuid, App::DocumentObjectT> newObjs;
662  for (auto obj : copiedObjs) {
663  auto prop = Base::freecad_dynamic_cast<PropertyUUID>(obj->getPropertyByName("_SourceUUID"));
664  if (prop) {
665  newObjs.insert(std::make_pair(prop->getValue(), obj));
666  }
667  }
668 
669  std::vector<std::pair<App::DocumentObject*, App::DocumentObject*>> replacements;
670  for (const auto& objT : oldObjs) {
671  auto prop = Base::freecad_dynamic_cast<PropertyUUID>(objT.getProperty());
672  if (!prop) {
673  continue;
674  }
675  auto it = newObjs.find(prop->getValue());
676  if (it == newObjs.end()) {
677  continue;
678  }
679  auto oldObj = objT.getObject();
680  auto newObj = it->second.getObject();
681  if (oldObj && newObj) {
682  replacements.emplace_back(oldObj, newObj);
683  }
684  }
685 
686  std::vector<std::pair<App::DocumentObjectT, std::unique_ptr<App::Property>>> propChanges;
687  if (!replacements.empty()) {
688  std::sort(copiedObjs.begin(), copiedObjs.end());
689 
690  // Global search for links affected by the replacement. We accumulate
691  // the changes in propChanges without applying, in order to avoid any
692  // side effect of changing while searching.
693  for (auto doc : App::GetApplication().getDocuments()) {
694  for (auto o : doc->getObjects()) {
695  if (o == parent || std::binary_search(objs.begin(), objs.end(), o)
696  || std::binary_search(copiedObjs.begin(), copiedObjs.end(), o)) {
697  continue;
698  }
699  propList.clear();
700  o->getPropertyList(propList);
701  for (auto prop : propList) {
702  if (prop->getContainer() != o) {
703  continue;
704  }

956 {

958  if (pauseCopyOnChange || !linked) {
959  return nullptr;
960  }

963  for (auto obj : srcobjs) {
964  if (obj->testStatus(App::PartialObject)) {
965  FC_THROWM(Base::RuntimeError,
966  "Cannot copy partial loaded object: " << obj->getFullName());

972  }

975

982  getLinkCopyOnChangeProperty()->setValue(CopyOnChangeOwned);

1233  if (!dot) {

1235  }
1236

1238  // If the name start with digits, treat as index reference
1239  idx = getArrayIndex(subname, nullptr);
1240  if (idx < 0) {

1244  if (idx >= _getElementCountValue()) {

1249  return -1;
1250  }
1251  }
1252  else if (!_getShowElementValue() && _getElementCountValue()) {
1253  // If elements are collapsed, we check first for LinkElement naming
1254  // pattern, which is the owner object name + "_i" + index
1255  const char* name = subname[0] == '$' ? subname + 1 : subname;
1256  auto owner = getContainer();
1257  if (owner && owner->isAttachedToDocument()) {
1258  std::string ownerName(owner->getNameInDocument());
1259  ownerName += '_';
1260  if (boost::algorithm::starts_with(name, ownerName.c_str())) {
1261  for (const char* txt = dot - 1; txt >= name + ownerName.size(); --txt) {
1262  if (*txt == 'i') {
1263  idx = getArrayIndex(txt + 1, nullptr);
1264  if (idx < 0 || idx >= _getElementCountValue()) {
1265  idx = -1;
1266  }
1267  break;
1268  }
1269  if (!isdigit(*txt)) {
1270  break;
1271  }
1272  }
1273  }
1274  }

1277  // redirect that reference to the first array element

1284  if (boost::equals(sub, linked->Label.getValue())) {
1285  idx = 0;
1286  }
1287  }
1288  else {
1289  CharRange sub(subname, dot);
1290  if (boost::equals(sub, linked->getNameInDocument())) {
1291  idx = 0;
1292  }
1293  }

1296  auto sobj = linked->getSubObject(std::string(subname, dot - subname + 1).c_str());
1297  if (!sobj) {

1300  if (psubname) {
1301  *psubname = subname;
1302  }
1303  return 0;
1304  }
1305  }
1306  }
1307  else if (subname[0] != '$') {
1308  // Try search by element objects' name
1309  std::string name(subname, dot);

1316  if (nidx >= 0) {

1319  }
1320  }
1321  }
1322  else if (getElementListProperty()) {
1323  getElementListProperty()->find(name.c_str(), &idx);
1324  }
1325  if (idx < 0) {
1326  return -1;
1327  }
1328  }
1329  else {
1330  // Try search by label if the reference name start with '$'
1331  ++subname;
1332  std::string name(subname, dot - subname);

1336  cacheChildLabel(1);

1340  return -1;

1343  }
1344  else {
1345  idx = 0;
1346  for (auto element : elements) {
1347  if (element->Label.getStrValue() == name) {

1351  }
1352  }

1364  }
1365  }
1366  }
