130 bool PropertySheet::isValidCellAddressName(const std::string& candidate)
131 {
132  /* Check if it matches a cell reference */
133  static const boost::regex e("\\${0,1}([A-Z]{1,2})\\${0,1}([0-9]{1,5})");
134  boost::cmatch cm;
135 
136  if (boost::regex_match(candidate.c_str(), cm, e)) {
137  const boost::sub_match<const char*> colstr = cm[1];
138  const boost::sub_match<const char*> rowstr = cm[2];
139 
140  if (App::validRow(rowstr.str()) >= 0 && App::validColumn(colstr.str())) {
141  return true;
142  }
143  }
144  return false;

147 bool PropertySheet::isValidAlias(const std::string& candidate)
148 {
149  /* Ensure it only contains allowed characters */
150  static const boost::regex gen("^[A-Za-z][_A-Za-z0-9]*$");
151  boost::cmatch cm;
152  if (!boost::regex_match(candidate.c_str(), cm, gen)) {
153  return false;
154  }
155 
156  /* Check if it is used before */
157  if (getValueFromAlias(candidate)) {
158  return false;
159  }
160 
161  /* Check if it would be a valid cell address name, e.g. "A2" or "C3" */
162  if (isValidCellAddressName(candidate)) {
163  return false;
164  }
165 
166  /* Check to make sure it doesn't clash with a reserved name */
167  if (ExpressionParser::isTokenAUnit(candidate)
168  || ExpressionParser::isTokenAConstant(candidate)) {
169  return false;
170  }
171 
172  return true;
173 }

178 // A utility function that gets the range (the minimum and maximum row and column) out of a
179 // vector of cell addresses. Note that it's possible that neither cell in the tuple itself
180 // has data in it, but operating on all cells in the inclusive range specified by the tuple
181 // is guaranteed to include all cells in the passed-in vector, and no cells exist to the
182 // left, right, above, or below the block specified.
183 std::tuple<CellAddress, CellAddress> extractRange(const std::vector<CellAddress>& cells)
184 {
185  CellAddress firstRowAndColumn;
186  CellAddress lastRowAndColumn;
187  for (const auto& cell : cells) {
188  int row = cell.row();
189  int column = cell.col();
190  if (row < firstRowAndColumn.row() || !firstRowAndColumn.isValid()) {
191  firstRowAndColumn.setRow(row);
192  }
193  if (column < firstRowAndColumn.col() || !firstRowAndColumn.isValid()) {
194  firstRowAndColumn.setCol(column);
195  }
196  if (row > lastRowAndColumn.row() || !lastRowAndColumn.isValid()) {
197  lastRowAndColumn.setRow(row);
198  }
199  if (column > lastRowAndColumn.col() || !lastRowAndColumn.isValid()) {
200  lastRowAndColumn.setCol(column);
201  }
202  }
203  return std::make_tuple(firstRowAndColumn, lastRowAndColumn);
204 }

315 void PropertySheet::Paste(const Property& from)
316 {
317  const PropertySheet& froms = dynamic_cast<const PropertySheet&>(from);
318 
319  AtomicPropertyChange signaller(*this);
320 
321  std::map<CellAddress, Cell*>::iterator icurr = data.begin();
322 
323  /* Mark all first */
324  while (icurr != data.end()) {
325  icurr->second->mark();
326  ++icurr;
327  }
328 
329  std::map<CellAddress, Cell*>::const_iterator ifrom = froms.data.begin();
330  std::vector<CellAddress> spanChanges;
331  int rows, cols;
332  while (ifrom != froms.data.end()) {
333  auto& cell = data[ifrom->first];
334 
335  if (cell) {
336  int r, c;
337  cell->getSpans(rows, cols);
338  ifrom->second->getSpans(r, c);
339  if (rows != r || cols != c) {
340  spanChanges.push_back(ifrom->first);
341  }
342  *cell = *(ifrom->second); // Exists; assign cell directly
343  }
344  else {
345  cell = new Cell(this,
346  *(ifrom->second)); // Doesn't exist, copy using Cell's copy constructor
347  if (cell->getSpans(rows, cols)) {
348  spanChanges.push_back(ifrom->first);
349  }
350  }
351  recomputeDependencies(ifrom->first);
352 
353  /* Set dirty */
354  setDirty(ifrom->first);
355  ++ifrom;
356  }
357 
358  /* Remove all that are still marked */
359  icurr = data.begin();
360  while (icurr != data.end()) {
361  Cell* cell = icurr->second;
362 
363  if (cell->isMarked()) {
364  if (cell->getSpans(rows, cols)) {
365  spanChanges.push_back(icurr->first);
366  }
367 
368  std::map<CellAddress, Cell*>::iterator next = icurr;
369 
370  ++next;
371  clear(icurr->first);
372  icurr = next;
373  }
374  else {
375  ++icurr;
376  }
377  }
378 
379  if (!spanChanges.empty()) {
380  mergedCells = froms.mergedCells;
381  if (auto sheet = Base::freecad_dynamic_cast<Sheet>(getContainer())) {
382  for (const auto& addr : spanChanges) {
383  sheet->cellSpanChanged(addr);
384  }
385  }
386  }
387  signaller.tryInvoke();
388 }

490 void PropertySheet::pasteCells(XMLReader& reader, Range dstRange)
491 {
492  reader.readElement("Cells");
493  int rangeCount = reader.getAttributeAsInteger("count");
494  if (rangeCount <= 0) {
495  return;
496  }
497 
498  int dstRows = dstRange.rowCount();
499  int dstCols = dstRange.colCount();
500  CellAddress dstFrom = dstRange.from();
501 
502  int roffset = 0, coffset = 0;
503 
504  AtomicPropertyChange signaller(*this);
505  for (int ri = 0; ri < rangeCount; ++ri) {
506  reader.readElement("Range");
507  CellAddress from(reader.getAttribute("from"));
508  CellAddress to(reader.getAttribute("to"));
509  int cellCount = reader.getAttributeAsInteger("count");
510 
511  Range range(from, to);
512 
513  CellAddress addr(dstFrom);
514  if (ri == 0) {
515  roffset = addr.row() - from.row();
516  coffset = addr.col() - from.col();
517  }
518 
519  int rcount, ccount;
520  if (rangeCount > 1) {
521  rcount = 1;
522  ccount = 1;
523  }
524  else {
525  rcount = dstRows / range.rowCount();
526  if (rcount == 0) {
527  rcount = 1;
528  }
529  ccount = dstCols / range.colCount();
530  if (ccount == 0) {
531  ccount = 1;
532  }
533  }
534  for (int ci = 0; ci < cellCount; ++ci) {
535  reader.readElement("Cell");
536  CellAddress src(reader.getAttribute("address"));
537 
538  if (ci) {
539  range.next();
540  }
541 
542  while (src != *range) {
543  for (int r = 0; r < rcount; ++r) {
544  for (int c = 0; c < ccount; ++c) {
545  CellAddress dst(range.row() + roffset + r * range.rowCount(),
546  range.column() + coffset + c * range.colCount());
547  if (!dst.isValid()) {
548  continue;
549  }
550  owner->clear(dst);
551  }
552  }
553  range.next();
554  }
555 
556  CellAddress newCellAddr;
557  for (int r = 0; r < rcount; ++r) {
558  for (int c = 0; c < ccount; ++c) {
559  CellAddress dst(src.row() + roffset + r * range.rowCount(),
560  src.col() + coffset + c * range.colCount());
561  if (!dst.isValid()) {
562  continue;
563  }
564 
565  auto cell = owner->getNewCell(dst);
566  splitCell(dst);
567 
568  int roffset_cur, coffset_cur;
569  if (!newCellAddr.isValid()) {
570  roffset_cur = roffset;
571  coffset_cur = coffset;
572  newCellAddr = dst;
573  cell->restore(reader, true);
574  }
575  else {
576  roffset_cur = r * range.rowCount();
577  coffset_cur = c * range.colCount();
578  auto newCell = owner->getCell(newCellAddr);
579  const Expression* expr;
580  if (!newCell || !(expr = newCell->getExpression(true))) {
581  FC_THROWM(Base::RuntimeError,
582  "Failed to copy cell " << getFullName() << '.'
583  << dst.toString() << " from "
584  << newCellAddr.toString());
585  }
586  cell->setExpression(ExpressionPtr(expr->copy()));
587  }
588 
589  int rows, cols;
590  if (cell->getSpans(rows, cols)) {
591  mergeCells(dst, CellAddress(dst.row() + rows - 1, dst.col() + cols - 1));
592  }
593  else {
594  splitCell(dst);
595  }
596 
597  if (roffset_cur || coffset_cur) {
598  OffsetCellsExpressionVisitor<PropertySheet> visitor(*this,
599  roffset_cur,
600  coffset_cur);
601  cell->visit(visitor);
602  if (visitor.changed()) {
603  recomputeDependencies(dst);
604  }
605  }
606  dirty.insert(dst);
607  }
608  }
609  }
610  if (cellCount == 0 || range.next()) {
611  do {
612  for (int r = 0; r < rcount; ++r) {
613  for (int c = 0; c < ccount; ++c) {
614  CellAddress dst(range.row() + roffset + r * range.rowCount(),
615  range.column() + coffset + c * range.colCount());
616  if (!dst.isValid()) {
617  continue;
618  }
619  owner->clear(dst);
620  }
621  }
622  } while (range.next());
623  }
624  owner->rangeUpdated(Range(from, to));
625  }
626  signaller.tryInvoke();
627 }

745 void PropertySheet::setAlias(CellAddress address, const std::string& alias)
746 {

748  throw Base::ValueError("Invalid alias");
749  }
750 
751  const Cell* aliasedCell = getValueFromAlias(alias);
752  Cell* cell = nonNullCellAt(address);
753  assert(cell);
754 
755  if (aliasedCell == cell) {
756  return;
757  }
758 
759  if (aliasedCell) {
760  throw Base::ValueError("Alias already defined.");
761  }
762

764

766  * disappears */
767  std::string fullName = owner->getFullName() + "." + address.toString();
768

1473 {

1475  if (itD != _Deps.end() && itD->second) {
1476  // Check for hidden reference. Because a hidden reference is not
1477  // protected by cyclic dependency checking, we need to take special
1478  // care to prevent it from misbehave.
1479  Sheet* sheet = Base::freecad_dynamic_cast<Sheet>(getContainer());
1480  if (!sheet || sheet->testStatus(App::ObjectStatus::Recompute2) || !owner

1484  }
1485

1487  // references, i.e indirect references. The dependencies of these
1488  // references are too complex to track exactly, so we only track the
1489  // top parent object instead, and mark the involved expression
1490  // whenever the top parent changes.
1491  std::string fullName = owner->getFullName() + ".";
1492  auto it = propertyNameToCellMap.find(fullName);
1493  if (it != propertyNameToCellMap.end()) {
1494  for (auto& cell : it->second) {

1500  // Now, we check for direct property references
