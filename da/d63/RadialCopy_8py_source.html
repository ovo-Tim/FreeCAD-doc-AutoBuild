14 def makeCopy(shape, radius, angle):
15  mat = Base.Matrix()
16  mat.rotateZ(math.radians(angle))
17  step = int(360.0 / angle)
18  shape = shape.copy()
19  shape.translate((radius, 0, 0))
20  comp = shape.copy()
21  for i in range(step):
22  shape.transformShape(mat)
23  comp = comp.fuse(shape)
24  return comp
