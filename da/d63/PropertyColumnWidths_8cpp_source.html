126 void PropertyColumnWidths::Restore(Base::XMLReader& reader)
127 {
128  int Cnt;
129 
130  // Column info
131  reader.readElement("ColumnInfo");
132  Cnt = reader.hasAttribute("Count") ? reader.getAttributeAsInteger("Count") : 0;
133  for (int i = 0; i < Cnt; i++) {
134  reader.readElement("Column");
135  const char* name = reader.hasAttribute("name") ? reader.getAttribute("name") : nullptr;
136  const char* width = reader.hasAttribute("width") ? reader.getAttribute("width") : nullptr;
137 
138  try {
139  if (name && width) {
140  int col = App::decodeColumn(name);
141  int colWidth = atoi(width);
142 
143  setValue(col, colWidth);
144  }
145  }
146  catch (...) {
147  // Something is wrong, skip this column
148  }
149  }
150  reader.readEndElement("ColumnInfo");
151 }
