85 CosmeticEdge::CosmeticEdge(const TechDraw::BaseGeomPtr g)
86 {
87 // Base::Console().Message("CE::CE(bg)\n");
88  m_geometry = g;
89  //we assume input edge is already in Yinverted coordinates
90  permaStart = m_geometry->getStartPoint();
91  permaEnd = m_geometry->getEndPoint();
92  if ((g->getGeomType() == TechDraw::GeomType::CIRCLE) ||
93  (g->getGeomType() == TechDraw::GeomType::ARCOFCIRCLE)) {
94  TechDraw::CirclePtr circ = std::static_pointer_cast<TechDraw::Circle>(g);
95  permaStart = circ->center;

97  permaRadius = circ->radius;

99  TechDraw::AOCPtr aoc = std::static_pointer_cast<TechDraw::AOC>(circ);
100  aoc->clockwiseAngle(g->clockwiseAngle());
101  aoc->startPnt = g->getStartPoint();
102  aoc->startAngle = g->getStartAngle();
103  aoc->endPnt = g->getEndPoint();
104  aoc->endAngle = g->getEndAngle();
105  // aoc->largeArc = g->largeArc;
106  }
107  }

148 TechDraw::BaseGeomPtr CosmeticEdge::scaledAndRotatedGeometry(const double scale, const double rotDegrees)
149 {
150  TopoDS_Edge e = m_geometry->getOCCEdge();
151  bool saveCW = m_geometry->clockwiseAngle();
152  // Mirror shape in Y and scale
153  TopoDS_Shape s = ShapeUtils::mirrorShape(e, gp_Pnt(0.0, 0.0, 0.0), scale);
154  // rotate using OXYZ as the coordinate system
155  s = ShapeUtils::rotateShape(s, gp_Ax2(), rotDegrees);
156  s = ShapeUtils::mirrorShape(s);
157  TopoDS_Edge newEdge = TopoDS::Edge(s);
158  TechDraw::BaseGeomPtr newGeom = TechDraw::BaseGeom::baseFactory(newEdge);
159  newGeom->setClassOfEdge(ecHARD);
160  newGeom->setHlrVisible( true);
161  newGeom->setCosmetic(true);
162  newGeom->source(COSMETICEDGE);
163  newGeom->setCosmeticTag(getTagAsString());
164  newGeom->clockwiseAngle(saveCW);
165  return newGeom;
166 }

243 void CosmeticEdge::Restore(Base::XMLReader &reader)
244 {
245  if (!CosmeticVertex::restoreCosmetic()) {
246  return;
247  }
248 // Base::Console().Message("CE::Restore - reading elements\n");
249  reader.readElement("Style");
250  m_format.setStyle(reader.getAttributeAsInteger("value"));
251  reader.readElement("Weight");
252  m_format.setWidth(reader.getAttributeAsFloat("value"));
253  reader.readElement("Color");
254  std::string tempHex = reader.getAttribute("value");
255  App::Color tempColor;
256  tempColor.fromHexString(tempHex);
257  m_format.setColor(tempColor);
258  reader.readElement("Visible");
259  m_format.setVisible(reader.getAttributeAsInteger("value") != 0);
260 
261  reader.readElement("GeometryType");
262  TechDraw::GeomType gType = static_cast<TechDraw::GeomType>(reader.getAttributeAsInteger("value"));
263 
264  if (gType == TechDraw::GeomType::GENERIC) {
265  TechDraw::GenericPtr gen = std::make_shared<TechDraw::Generic> ();
266  gen->Restore(reader);
267  gen->setOCCEdge(GeometryUtils::edgeFromGeneric(gen));
268  m_geometry = gen;
269  permaStart = gen->getStartPoint();
270  permaEnd = gen->getEndPoint();
271  } else if (gType == TechDraw::GeomType::CIRCLE) {
272  TechDraw::CirclePtr circ = std::make_shared<TechDraw::Circle> ();
273  circ->Restore(reader);
274  circ->setOCCEdge(GeometryUtils::edgeFromCircle(circ));
275  m_geometry = circ;
276  permaRadius = circ->radius;
277  permaStart = circ->center;
278  permaEnd = circ->center;
279  } else if (gType == TechDraw::GeomType::ARCOFCIRCLE) {
280  TechDraw::AOCPtr aoc = std::make_shared<TechDraw::AOC> ();
281  aoc->Restore(reader);
282  aoc->setOCCEdge(GeometryUtils::edgeFromCircleArc(aoc));
283  m_geometry = aoc->inverted();
284  permaStart = aoc->center;
285  permaEnd = aoc->center;
286  permaRadius = aoc->radius;
287  } else {
288  Base::Console().Warning("CE::Restore - unimplemented geomType: %d\n", static_cast<int>(gType));
289  }
290  // older documents may not have the LineNumber element, so we need to check the
291  // next entry. if it is a start element, then we check if it is a start element
292  // for LineNumber.
293  if (reader.readNextElement()) {
294  if(strcmp(reader.localName(),"LineNumber") == 0 ) {
295  // this CosmeticEdge has an LineNumber attribute
296  m_format.setLineNumber(reader.getAttributeAsInteger("value"));
297  } else {
298  // LineNumber not found.
299  // TODO: line number should be set to DashedLineGenerator.fromQtStyle(m_format.m_style)
300  m_format.setLineNumber(LineFormat::InvalidLine);
301  }
302  }
303 }

315 void CosmeticEdge::createNewTag()
316 {
317  // Initialize a random number generator, to avoid Valgrind false positives.
318  // The random number generator is not threadsafe so we guard it. See
319  // https://www.boost.org/doc/libs/1_62_0/libs/uuid/uuid.html#Design%20notes
320  static boost::mt19937 ran;
321  static bool seeded = false;
322  static boost::mutex random_number_mutex;
323 
324  boost::lock_guard<boost::mutex> guard(random_number_mutex);
325 
326  if (!seeded) {
327  ran.seed(static_cast<unsigned int>(std::time(nullptr)));
328  seeded = true;
329  }
330  static boost::uuids::basic_random_generator<boost::mt19937> gen(&ran);
331 
332  tag = gen();
333 }
