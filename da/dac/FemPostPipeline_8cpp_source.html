104 void FemPostPipeline::read(Base::FileInfo File)
105 {
106 
107  // checking on the file
108  if (!File.isReadable()) {
109  throw Base::FileException("File to load not existing or not readable", File);
110  }
111 
112  if (File.hasExtension("vtu")) {
113  readXMLFile<vtkXMLUnstructuredGridReader>(File.filePath());
114  }
115  else if (File.hasExtension("pvtu")) {
116  readXMLFile<vtkXMLPUnstructuredGridReader>(File.filePath());
117  }
118  else if (File.hasExtension("vtp")) {
119  readXMLFile<vtkXMLPolyDataReader>(File.filePath());
120  }
121  else if (File.hasExtension("vts")) {
122  readXMLFile<vtkXMLStructuredGridReader>(File.filePath());
123  }
124  else if (File.hasExtension("vtr")) {
125  readXMLFile<vtkXMLRectilinearGridReader>(File.filePath());
126  }
127  else if (File.hasExtension("vti")) {
128  readXMLFile<vtkXMLImageDataReader>(File.filePath());
129  }
130  else if (File.hasExtension("vtk")) {
131  readXMLFile<vtkDataSetReader>(File.filePath());
132  }
133  else {
134  throw Base::FileException("Unknown extension");

143 void FemPostPipeline::onChanged(const Property* prop)
144 {
145  if (prop == &Filter || prop == &Mode) {
146 
147  // if we are in custom mode the user is free to set the input
148  // thus nothing needs to be done here
149  if (Mode.getValue() == 2) { // custom
150  return;
151  }
152 
153  // we check if all connections are right and add new ones if needed
154  std::vector<App::DocumentObject*> objs = Filter.getValues();
155 
156  if (objs.empty()) {
157  return;
158  }
159 
160  std::vector<App::DocumentObject*>::iterator it = objs.begin();
161  FemPostFilter* filter = static_cast<FemPostFilter*>(*it);
162 
163  // the first filter must always grab the data
164  if (filter->Input.getValue()) {
165  filter->Input.setValue(nullptr);
166  }
167 
168  // all the others need to be connected to the previous filter or grab the data,
169  // dependent on mode
170  ++it;
171  for (; it != objs.end(); ++it) {
172  auto* nextFilter = static_cast<FemPostFilter*>(*it);
173 
174  if (Mode.getValue() == 0) { // serial mode
175  if (nextFilter->Input.getValue() != filter) {
176  nextFilter->Input.setValue(filter);
177  }
178  }
179  else { // Parallel mode
180  if (nextFilter->Input.getValue()) {
181  nextFilter->Input.setValue(nullptr);
182  }
183  }
184 
185  filter = nextFilter;
186  };
187  }
188 
189  App::GeoFeature::onChanged(prop);
190 }

222 void FemPostPipeline::load(FemResultObject* res)
223 {
224  if (!res->Mesh.getValue()) {
225  Base::Console().Log("Result mesh object is empty.\n");
226  return;
227  }
228  if (!res->Mesh.getValue()->isDerivedFrom(Fem::FemMeshObject::getClassTypeId())) {
229  Base::Console().Log("Result mesh object is not derived from Fem::FemMeshObject.\n");
230  return;
231  }
232 
233  // first copy the mesh over
234  // ***************************
235  const FemMesh& mesh = static_cast<FemMeshObject*>(res->Mesh.getValue())->FemMesh.getValue();
236  vtkSmartPointer<vtkUnstructuredGrid> grid = vtkSmartPointer<vtkUnstructuredGrid>::New();
237  FemVTKTools::exportVTKMesh(&mesh, grid);
238 
239  // Now copy the point data over
240  // ***************************
241  FemVTKTools::exportFreeCADResult(res, grid);
242 
243  Data.setValue(grid);
244 }
