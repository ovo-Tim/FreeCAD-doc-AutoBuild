24 """
25 Provide tools to access the FreeCAD documentation.
26 
27 The main usage is using the "show" function. It can retrieve an URL,
28 a local file (markdown or html), or find a page automatically from
29 the settings set under Preferences->General->Help.
30 
31 It doesn't matter what you give, the system will recognize if the contents are
32 HTML or Markdown and render it appropriately.
33 
34 Basic usage:
35 
36  import Help
37  Help.show("Draft Line")
38  Help.show("Draft_Line") # works with spaces or underscores
39  Help.show("https://wiki.freecadweb.org/Draft_Line")
40  Help.show("https://gitlab.com/freecad/FreeCAD-documentation/-/raw/main/wiki/Draft_Line.md")
41  Help.show("/home/myUser/.FreeCAD/Documentation/Draft_Line.md")
42  Help.show("http://myserver.com/myfolder/Draft_Line.html")
43 
44 Preferences keys (in "User parameter:BaseApp/Preferences/Mod/Help"):
45 
46  optionBrowser/optionTab/optionDialog (bool): Where to open the help dialog
47  optionOnline/optionOffline (bool): where to fetch the documentation from
48  URL (string): online location
49  Location (string): offline location
50  Suffix (string): a suffix to add to the URL, ex: /fr
51  StyleSheet (string): optional CSS stylesheet to style the output
52 
53 Defaults are to open the wiki in the desktop browser
54 """

91 def show(page, view=None, conv=None):
92  """
93  show(page,view=None, conv=None):
94  Opens a help viewer and shows the given help page.
95  The given help page can be a URL pointing to a markdown or HTML file,
96  a name page / command name, or a file path pointing to a markdown
97  or HTML file. If view is given (an instance of openBrowserHTML.HelpPage or
98  any other object with a 'setHtml()' method), the page will be
99  rendered there, otherwise a new tab/widget will be created according to
100  preferences settings. If conv is given (markdown, pandoc, github, builtin or
101  none), the corresponding markdown conversion method is used. Otherwise, the
102  module will use the best available.
103  In non-GUI mode, this function simply outputs the markdown or HTML text.
104  """
105 
106  page = underscore_page(page)
107  location, _pagename = get_location(page)
108  FreeCAD.Console.PrintLog("Help: opening " + location + "\n")
109  if not location:
110  FreeCAD.Console.PrintError(LOCTXT + "\n")
111  return
112  md = get_contents(location)
113  html = convert(md, conv)
114  baseurl = get_uri(location)
115  if _pagename != "":
116  pagename = _pagename
117  else:
118  pagename = os.path.basename(page.replace("_", " ").replace(".md", ""))
119  title = translate("Help", "Help") + ": " + pagename
120  if FreeCAD.GuiUp:
121  if PREFS.GetBool("optionTab", False) and get_qtwebwidgets():
122  # MDI tab
123  show_tab(html, baseurl, title, view)
124  elif PREFS.GetBool("optionDialog", False) and get_qtwebwidgets():
125  # floating dock window
126  show_dialog(html, baseurl, title, view)
127  else:
128  # desktop web browser - default
129  show_browser(location)
130  else:
131  # console mode, we just print the output
132  print(md)

159 def location_url(url_localized: str, url_english: str) -> tuple:
160  """
161  Returns localized documentation url and page name, if they exist,
162  otherwise defaults to english version.
163  Page name is gotten from:
164  a) Name/* metadata tag on raw markdown files from github,
165  Wiki translators should make sure to add it.
166  b) <title> HTML tag
167  """
168  try:
169  req = urllib.request.Request(url_localized)
170  with urllib.request.urlopen(req) as response:
171  html = response.read().decode("utf-8")
172  if re.search(MD_RAW_URL, url_localized):
173  pagename_match = re.search(r"Name/.*?:\s*(.+)", html)
174  else:
175  # Pages from FreeCAD Wiki fall here
176  pagename_match = re.search(r"<title>(.*?) - .*?</title>", html)
177  if pagename_match is not None:
178  return (url_localized, pagename_match.group(1))
179  else:
180  return (url_localized, "")
181  except urllib.error.HTTPError as e:
182  return (url_english, "")

185 def get_location(page) -> tuple:
186  """retrieves the location (online or offline) of a given page. Returns the location and the page name"""
187 
188  location = ""
189  if page.startswith("http"):
190  # NOTE: This gets activated when you open a link on the built-in browser, since
191  # we don't know the URL, using location_url() fallback to the HTML <title> tag
192  return location_url(page, page)
193  if page.startswith("file://"):
194  return (page[7:], "")
195  # offline location
196  if os.path.exists(page):
197  return (page, "")
198  page = page.replace(".md", "")
199  page = page.replace(" ", "_")
200  page = page.replace("wiki/", "")
201  page = page.split("#")[0]
202  suffix = PREFS.GetString("Suffix", "")
203  pagename = ""
204  if suffix:
205  if not suffix.startswith("/"):
206  suffix = "/" + suffix
207  if PREFS.GetBool("optionWiki", True): # default
208  location, pagename = location_url(WIKI_URL + "/" + page + suffix, WIKI_URL + "/" + page)
209  elif PREFS.GetBool("optionMarkdown", False):
210  if PREFS.GetBool("optionBrowser", False):
211  location = MD_RENDERED_URL
212  else:
213  location = MD_RAW_URL
214  if suffix:
215  location, pagename = location_url(
216  location + "/" + MD_TRANSLATIONS_FOLDER + suffix + "/" + page + ".md",
217  location + "/" + page + ".md",
218  )
219  else:
220  location += "/" + page + ".md"
221  elif PREFS.GetBool("optionGithub", False):
222  location = MD_RENDERED_URL
223  if suffix:
224  location, pagename = location_url(
225  location + "/" + MD_TRANSLATIONS_FOLDER + suffix + "/" + page + ".md",
226  location + "/" + page + ".md",
227  )
228  else:
229  location += "/" + page + ".md"
230  elif PREFS.GetBool("optionCustom", False):
231  location = PREFS.GetString("Location", "")
232  if not location:
233  location = os.path.join(
234  FreeCAD.getUserAppDataDir(),
235  "Mod",
236  "offline-documentation",
237  "FreeCAD-documentation-main",
238  "wiki",
239  )
240  location = os.path.join(location, page + ".md")
241  return (location, pagename)

358  def convert_raw(m):
359  # simple and dirty regex-based markdown to html
360 
361  import re
362 
363  f = re.DOTALL | re.MULTILINE
364  m = re.sub(r"^##### (.*?)\n", r"<h5>\1</h5>\n", m, flags=f) # ##### titles
365  m = re.sub(r"^#### (.*?)\n", r"<h4>\1</h4>\n", m, flags=f) # #### titles
366  m = re.sub(r"^### (.*?)\n", r"<h3>\1</h3>\n", m, flags=f) # ### titles
367  m = re.sub(r"^## (.*?)\n", r"<h2>\1</h2>\n", m, flags=f) # ## titles
368  m = re.sub(r"^# (.*?)\n", r"<h1>\1</h1>\n", m, flags=f) # # titles
369  m = re.sub(r"!\[(.*?)\]\((.*?)\)", r'<img alt="\1" src="\2">', m, flags=f) # images
370  m = re.sub(r"\[(.*?)\]\((.*?)\)", r'<a href="\2">\1</a>', m, flags=f) # links
371  m = re.sub(r"\*\*(.*?)\*\*", r"<b>\1</b>", m) # bold
372  m = re.sub(r"\*(.*?)\*", r"<i>\1</i>", m) # italic
373  m = re.sub(r"\n\n", r"<br/>", m, flags=f) # double new lines
374  m += "\n<br/><hr/><small>" + CONVERTTXT + "</small>"
375  return m
