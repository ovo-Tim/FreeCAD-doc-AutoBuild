270 PyObject* Application::sLoadFile(PyObject* /*self*/, PyObject* args)
271 {
272  const char* path;
273  const char* doc = "";
274  const char* mod = "";
275  if (!PyArg_ParseTuple(args, "s|ss", &path, &doc, &mod)) {
276  return nullptr;
277  }
278  try {
279  Base::FileInfo fi(path);
280  if (!fi.isFile() || !fi.exists()) {
281  PyErr_Format(PyExc_IOError, "File %s doesn't exist.", path);
282  return nullptr;
283  }
284 
285  std::string module = mod;
286  if (module.empty()) {
287  std::string ext = fi.extension();
288  std::vector<std::string> modules = GetApplication().getImportModules(ext.c_str());
289  if (modules.empty()) {
290  PyErr_Format(PyExc_IOError, "Filetype %s is not supported.", ext.c_str());
291  return nullptr;
292  }
293  else {
294  module = modules.front();
295  }
296  }
297 
298  // path could contain characters that need escaping, such as quote signs
299  // therefore use its representation in the Python code string
300  PyObject* pathObj = PyUnicode_FromString(path);
301  PyObject* pathReprObj = PyObject_Repr(pathObj);
302  const char* pathRepr = PyUnicode_AsUTF8(pathReprObj);
303 
304  std::stringstream str;
305  str << "import " << module << std::endl;
306  if (fi.hasExtension("FCStd")) {
307  str << module << ".openDocument(" << pathRepr << ")" << std::endl;
308  }
309  else {
310  str << module << ".insert(" << pathRepr << ",'" << doc << "')" << std::endl;
311  }
312 
313  Py_DECREF(pathObj);
314  Py_DECREF(pathReprObj);
315 
316  Base::Interpreter().runString(str.str().c_str());
317  Py_Return;
318  }
319  catch (const Base::Exception& e) {
320  e.setPyException();
321  return nullptr;
322  }
323  catch (const std::exception& e) {
324  // might be subclass from zipios
325  PyErr_Format(PyExc_IOError, "Invalid project file %s: %s", path, e.what());
326  return nullptr;
327  }
328 }
