69 def get_scales(unit_system = 0):
70  """
71  returns the list of preset scales according to unit system.
72 
73  Parameters:
74  unit_system = 0 : default from user preferences
75  1 : metrics
76  2 : imperial architectural
77  3 : imperial engineering
78  """

80  if unit_system == 0:
81  scale_units_system = params.get_param("UserSchema", path="Units")
82  if scale_units_system in [2, 3, 5]:
83  return draft_scales_arch_imperial
84  elif scale_units_system in [7]:
85  return draft_scales_eng_imperial
86  else:
87  return draft_scales_metrics

96 def scale_to_label(scale):
97  """
98  transform a float number into a 1:X or X:1 scale and return it as label
99  """
100  if scale <= 0:
101  return "1:1"
102  f = round(scale, 2)
103  if f == 1:
104  return "1:1"
105  if f > 1:
106  f = f.as_integer_ratio()
107  if f[1] == 1:
108  return str(f[0]) + ":1"
109  return str(scale)
110  f = round(1/scale, 2)
111  f = f.as_integer_ratio()
112  if f[1] == 1:
113  return "1:" + str(f[0])
114  return str(scale)

116 def label_to_scale(label):
117  """
118  transform a scale string into scale factor as float
119  """
120  try :
121  scale = float(label)
122  return scale
123  except Exception:
124  if ":" in label:
125  f = label.split(":")
126  elif "=" in label:
127  f = label.split("=")
128  else:
129  return None
130  if len(f) == 2:
131  try:
132  num = App.Units.Quantity(f[0]).Value
133  den = App.Units.Quantity(f[1]).Value
134  scale = num/den
135  return scale
136  except Exception:
137  err = translate("draft",
138  "Unable to convert input into a scale factor")
139  App.Console.PrintWarning(err)
140  return None

176 def init_draft_statusbar_scale():
177  """
178  this function initializes draft statusbar scale widget
179  """
180  mw = Gui.getMainWindow()
181  sb = mw.statusBar()
182 
183  scale_widget = QtWidgets.QToolBar()
184  # prevent the widget from showing up in the toolbar area context menu:
185  scale_widget.toggleViewAction().setVisible(False)
186  scale_widget.setObjectName("draft_scale_widget")
187  # WindowTitle is just in case, should not be visible in the GUI.
188  scale_widget.setWindowTitle(translate("draft", "Draft scale widget"))
189 
190  # get scales list according to system units
191  draft_scales = get_scales()
192 
193  # get draft annotation scale
194  scale_multiplier = params.get_param("DefaultAnnoScaleMultiplier")
195  annotation_scale = 1 / scale_multiplier if scale_multiplier > 0 else 1
196 
197  # initializes scale widget
198  scale_widget.draft_scales = draft_scales
199  scaleLabel = QtWidgets.QPushButton("Scale")
200  scaleLabel.setObjectName("ScaleLabel")
201  scaleLabel.setFlat(True)
202  menu = QtWidgets.QMenu(scaleLabel)
203  gUnits = QtGui.QActionGroup(menu)
204  for u in draft_scales:
205  a = QtGui.QAction(gUnits)
206  a.setText(u)
207  menu.addAction(a)
208  scaleLabel.setMenu(menu)
209  gUnits.triggered.connect(_set_scale)
210  scale_label = scale_to_label(annotation_scale)
211  scaleLabel.setText(scale_label)
212  scaleLabel.setToolTip(translate("draft",
213  "Set the scale used by draft annotation tools"))
214  scale_widget.addWidget(scaleLabel)
215  scale_widget.scaleLabel = scaleLabel
216 
217  # add scale widget to the statusbar
218  sb.insertPermanentWidget(3, scale_widget)
219  scale_widget.show()
