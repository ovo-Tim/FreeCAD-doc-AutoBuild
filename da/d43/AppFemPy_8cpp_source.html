191  Py::Object exporter(const Py::Tuple& args)
192  {
193  PyObject* object;
194  char* Name;
195  if (!PyArg_ParseTuple(args.ptr(), "Oet", &object, "utf-8", &Name)) {
196  throw Py::Exception();
197  }
198 
199  Base::FileInfo file(Name);
200  PyMem_Free(Name);
201 
202  ParameterGrp::handle hGrp = App::GetApplication().GetParameterGroupByPath(
203  "User parameter:BaseApp/Preferences/Mod/Fem");
204 
205  Py::Sequence list(object);
206  Base::Type meshId = Base::Type::fromName("Fem::FemMeshObject");
207  for (Py::Sequence::iterator it = list.begin(); it != list.end(); ++it) {
208  PyObject* item = (*it).ptr();
209  if (PyObject_TypeCheck(item, &(App::DocumentObjectPy::Type))) {
210  App::DocumentObject* obj =
211  static_cast<App::DocumentObjectPy*>(item)->getDocumentObjectPtr();
212  if (obj->getTypeId().isDerivedFrom(meshId)) {
213  auto femMesh = static_cast<FemMeshObject*>(obj)->FemMesh.getValue();
214  if (file.hasExtension({"vtk", "vtu"})) {
215  // get VTK prefs
216  ParameterGrp::handle g = hGrp->GetGroup("InOutVtk");
217  std::string level = g->GetASCII("MeshExportLevel", "Highest");
218  femMesh.writeVTK(file.filePath().c_str(),
219  level == "Highest" ? true : false);
220  }
221  else if (file.hasExtension("inp")) {
222  // get Abaqus inp prefs
223  ParameterGrp::handle g = hGrp->GetGroup("Abaqus");
224  int elemParam = g->GetInt("AbaqusElementChoice", 1);
225  bool groupParam = g->GetBool("AbaqusWriteGroups", false);
226  // write ABAQUS Output
227  femMesh.writeABAQUS(file.filePath(), elemParam, groupParam);
228  }
229  else {
230  femMesh.write(file.filePath().c_str());
231  }
232  return Py::None();
233  }
234  }
235  }
236 
237  throw Py::RuntimeError("No FEM mesh for export selected");
238  }
