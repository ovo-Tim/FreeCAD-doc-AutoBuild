47 def exportMeshToTetGenPoly(meshToExport, filePath, beVerbose=1):
48  """Export mesh to TetGen *.poly file format"""
49  # ********** Part 1 - write node list to output file
50  if beVerbose == 1:
51  Console.PrintMessage("\nExport of mesh to TetGen file ...")
52  (allVertices, allFacets) = meshToExport.Topology
53  f = open(filePath, "w")
54  f.write("# This file was generated from FreeCAD geometry\n")
55  f.write("# Part 1 - node list\n")
56  f.write(
57  "TotalNumOfPoints: {}, NumOfDimensions; {}, "
58  "NumOfProperties: {}, BoundaryMarkerExists: {}\n".format(len(allVertices), 3, 0, 0)
59  )
60  for PointIndex in range(len(allVertices)):
61  f.write(
62  "%(PointIndex)5i %(x) e %(y) e %(z) e\n"
63  % {
64  "PointIndex": PointIndex,
65  "x": allVertices[PointIndex].x,
66  "y": allVertices[PointIndex].y,
67  "z": allVertices[PointIndex].z,
68  }
69  )
70 
71  # Find out BoundaryMarker for each facet. If edge connects only two facets,
72  # then this facets should have the same BoundaryMarker
73  BoundaryMarkerExists = 1
74  PointList = [allFacets[0][1], allFacets[0][0]]
75  PointList.sort()
76  EdgeFacets = {(PointList[0], PointList[1]): {0}}
77  Edge = []
78 
79  # Find all facets for each edge
80  for FacetIndex in range(len(allFacets)):
81  Facet = allFacets[FacetIndex]
82  for i in range(0, -len(Facet), -1):
83  tmpEdge = [Facet[i], Facet[i + 1]]
84  tmpEdge.sort()
85  Edge.append(tmpEdge)
86  for i in range(len(Edge)):
87  EdgeIndex = (Edge[i][0], Edge[i][1])
88  if EdgeIndex in EdgeFacets:
89  EdgeFacets[EdgeIndex].add(FacetIndex)
90  else:
91  EdgeFacets[EdgeIndex] = {FacetIndex}
92  Edge = []
93 
94  # Find BoundaryMarker for each facet
95  BoundaryMarker = []
96  for index in range(len(allFacets)):
97  BoundaryMarker.append(0)
98  MinMarker = -1
99  InitialFacet = 0
100  BoundaryMarker[InitialFacet] = MinMarker
101  EdgeKeys = EdgeFacets.keys()
102  # disconnectedEdges = len(EdgeKeys)
103  if beVerbose == 1:
104  Console.PrintMessage(
105  "\nBoundaryMarker:" + repr(BoundaryMarker) + " " + repr(len(EdgeFacets))
106  )
107  searchForPair = 1
108 
109  # Main loop: first search for all complementary facets
110  # then fill one branch and repeat while edges are available
111  while len(EdgeFacets) > 0:
112  removeEdge = 0
113  for EdgeIndex in EdgeKeys:
114  if len(EdgeFacets[EdgeIndex]) == 1:
115  removeEdge = 1
116  break
117  if len(EdgeFacets[EdgeIndex]) == 2:
118  FacetPair = []
119  for facet in EdgeFacets[EdgeIndex]:
120  FacetPair.append(facet)
121  if (BoundaryMarker[FacetPair[0]] == 0) and (BoundaryMarker[FacetPair[1]] == 0):
122  continue
123  if (BoundaryMarker[FacetPair[0]] != 0) and (BoundaryMarker[FacetPair[1]] != 0):
124  removeEdge = 1
125  break
126  if BoundaryMarker[FacetPair[0]] != 0:
127  BoundaryMarker[FacetPair[1]] = BoundaryMarker[FacetPair[0]]
128  else:
129  BoundaryMarker[FacetPair[0]] = BoundaryMarker[FacetPair[1]]
130  removeEdge = 1
131  break
132  if searchForPair == 1:
133  continue
134  FacetTree = []
135  # AllMarkers = 1
136  MarkerSum = 0
137  for facet in EdgeFacets[EdgeIndex]:
138  FacetTree.append(facet)
139  MarkerSum += BoundaryMarker[facet]
140  if MarkerSum == 0:
141  continue
142  for facet in EdgeFacets[EdgeIndex]:
143  if BoundaryMarker[facet] == 0:
144  MinMarker -= 1
145  BoundaryMarker[facet] = MinMarker
146  searchForPair = 1
147  removeEdge = 1
148  break
149  if removeEdge == 1:
150  del EdgeFacets[EdgeIndex]
151  EdgeKeys = EdgeFacets.keys()
152  continue
153  searchForPair = 0
154  # End of main loop
155  if beVerbose == 1:
156  Console.PrintMessage(
157  "\nNew BoundaryMarker:" + repr(BoundaryMarker) + " " + repr(len(EdgeFacets))
158  )
159 
160  # ********** Part 2 - write all facets to *.poly file
161  f.write("# Part 2 - facet list\n")
162  f.write(
163  "%(TotalNumOfFacets)i %(BoundaryMarkerExists)i\n"
164  % {"TotalNumOfFacets": len(allFacets), "BoundaryMarkerExists": BoundaryMarkerExists}
165  )
166  for FacetIndex in range(len(allFacets)):
167  f.write("# FacetIndex = %(Index)i\n" % {"Index": FacetIndex})
168  f.write("%(NumOfPolygons)3i " % {"NumOfPolygons": 1})
169  if BoundaryMarkerExists == 1:
170  f.write("0 %(BoundaryMarker)i" % {"BoundaryMarker": BoundaryMarker[FacetIndex]})
171  f.write("\n%(NumOfConers)3i " % {"NumOfConers": len(allFacets[FacetIndex])})
172  for PointIndex in range(len(allFacets[FacetIndex])):
173  # f.write(repr(allFacets[FacetIndex][PointIndex]))
174  f.write("%(PointIndex)i " % {"PointIndex": allFacets[FacetIndex][PointIndex]})
175  f.write("\n")
176  # ********** Part 3 and Part 4 are zero
177  f.write("# Part 3 - the hole list.\n# There is no hole in bar.\n0\n")
178  f.write("# Part 4 - the region list.\n# There is no region defined.\n0\n")
179  f.write("# This file was generated from FreeCAD geometry\n")
180  f.close()
