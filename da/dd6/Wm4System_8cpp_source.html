46 //----------------------------------------------------------------------------
47 void System::SwapBytes (int iSize, void* pvValue)
48 {
49  // size must be even
50  assert(iSize >= 2 && (iSize & 1) == 0);
51 
52  char* acBytes = (char*) pvValue;
53  for (int i0 = 0, i1 = iSize-1; i0 < iSize/2; i0++, i1--)
54  {
55  char cSave = acBytes[i0];
56  acBytes[i0] = acBytes[i1];
57  acBytes[i1] = cSave;
58  }
59 }
60 //----------------------------------------------------------------------------
61 void System::SwapBytes (int iSize, int iQuantity, void* pvValue)
62 {
63  // size must be even
64  assert(iSize >= 2 && (iSize & 1) == 0);
65 
66  char* acBytes = (char*) pvValue;
67  for (int i = 0; i < iQuantity; i++, acBytes += iSize)
68  {
69  for (int i0 = 0, i1 = iSize-1; i0 < iSize/2; i0++, i1--)
70  {
71  char cSave = acBytes[i0];
72  acBytes[i0] = acBytes[i1];
73  acBytes[i1] = cSave;
74  }
75  }
76 }

104 double System::GetTime ()
105 {
106 #if !defined(_WIN32)
107  if (!gs_bInitializedTime)
108  {
109  gs_bInitializedTime = true;
110  gettimeofday(&gs_kInitial, nullptr);
111  }
112 
113  struct timeval kCurrent;
114  gettimeofday(&kCurrent,nullptr);
115 
116  struct timeval kDelta;
117  timersub(&kCurrent,&gs_kInitial,&kDelta);
118 
119  return 0.001*(double)(1000*kDelta.tv_sec + kDelta.tv_usec/1000);
120 #else
121  struct timeb kTB;
122 
123  if (!gs_bInitializedTime)
124  {
125  gs_bInitializedTime = true;
126  ftime(&kTB);
127  gs_lInitialSec = (long)kTB.time;
128  gs_lInitialUSec = 1000*kTB.millitm;
129  }
130 
131  ftime(&kTB);
132  long lCurrentSec = (long)kTB.time;
133  long lCurrentUSec = 1000*kTB.millitm;
134  long lDeltaSec = lCurrentSec - gs_lInitialSec;
135  long lDeltaUSec = lCurrentUSec -gs_lInitialUSec;
136  if (lDeltaUSec < 0)
137  {
138  lDeltaUSec += 1000000;
139  lDeltaSec--;
140  }
141 
142  return 0.001*(double)(1000*lDeltaSec + lDeltaUSec/1000);
143 #endif
144 }

146 bool System::Load (const char* acFilename, char*& racBuffer, int& riSize)
147 {
148  struct stat kStat;
149  if (stat(acFilename,&kStat) != 0)
150  {
151  // file does not exist
152  racBuffer = nullptr;
153  riSize = 0;
154  return false;
155  }
156 
157  FILE* pkFile = System::Fopen(acFilename,"rb");
158  assert(pkFile);
159  if (!pkFile)
160  {
161  racBuffer = nullptr;
162  riSize = 0;
163  return false;
164  }
165 
166  riSize = kStat.st_size;
167  racBuffer = WM4_NEW char[riSize];
168  int iRead = (int)fread(racBuffer,sizeof(char),riSize,pkFile);
169  if (System::Fclose(pkFile) != 0 || iRead != riSize)
170  {
171  assert(false);
172  WM4_DELETE[] racBuffer;
173  racBuffer = nullptr;
174  riSize = 0;
175  return false;
176  }
177 
178  return true;
179 }

568 //----------------------------------------------------------------------------
569 const char* System::GetPath (const char* acDirectory, const char* acFilename)
570 {
571  // #0000656: WildMagic4 doesn't build on 64-bit Mac OS
572 #if defined(__APPLE__) && !defined(__aarch64__) && !defined(__x86_64__)
573  // An application-relative path is needed for the applications to be able
574  // to find the input data sets. Unfortunately, there is no exact way to
575  // predict which directory the application is run from, since this depends
576  // on whether it was launched by the Finder, Xcode, gdb, the console, etc.
577  // To work around this, the following code switches to the application
578  // directory. Beware of your applications themselves changing directory,
579  // which could cause the input data sets not to be found.
580 
581  // WARNING. The engine supports XCode 2.x, but not XCode 1.x. The
582  // directory structure assumed here is based on the default Xcode 2.x
583  // preferences:
584  // "Put build products in project directory"
585  // "Put intermediate build files with build products"
586  // If you change the Xcode preferences, you will need to modify this code
587  // to correctly locate the application directory.
588 
589  OSStatus eError;
590  FSRef kExeRef; // ${ApplicationPath}/build/WildMagic/executableFile
591  FSRef kWMRef; // ${ApplicationPath}/build/WildMagic
592  FSRef kBuildRef; // ${ApplicationPath}/build
593  FSRef kAppRef; // ${ApplicationPath}
594  ProcessSerialNumber kPSN;
595 
596  MacGetCurrentProcess(&kPSN);
597  eError = GetProcessBundleLocation(&kPSN,&kExeRef);
598  eError = FSGetCatalogInfo(&kExeRef,0,0,0,0,&kWMRef);
599  eError = FSGetCatalogInfo(&kWMRef,0,0,0,0,&kBuildRef);
600  eError = FSGetCatalogInfo(&kBuildRef,0,0,0,0,&kAppRef);
601  eError = FSRefMakePath(&kAppRef,(UInt8*)ms_acPath,SYSTEM_MAX_PATH);
602  eError = chdir(ms_acPath);
603 
604 #if 0
605  // If you really must use Xcode version 1.x, then the path searching
606  // code must be the following.
607  OSStatus eError;
608  FSRef kExeRef; // ${ApplicationPath}/build/executableFile
609  FSRef kBuildRef; // ${ApplicationPath}/build
610  FSRef kAppRef; // ${ApplicationPath}
611  ProcessSerialNumber kPSN;
612 
613  MacGetCurrentProcess(&kPSN);
614  eError = GetProcessBundleLocation(&kPSN,&kExeRef);
615  eError = FSGetCatalogInfo(&kExeRef,0,0,0,0,&kBuildRef);
616  eError = FSGetCatalogInfo(&kBuildRef,0,0,0,0,&kAppRef);
617  eError = FSRefMakePath(&kAppRef,(UInt8*)ms_acPath,SYSTEM_MAX_PATH);
618  eError = chdir(ms_acPath);
619 #endif
620 #endif
621 
622  size_t uiDLength = strlen(acDirectory);
623  size_t uiFLength = strlen(acFilename);
624  if (uiDLength + uiFLength + 1 <= SYSTEM_MAX_PATH)
625  {
626  System::Strcpy(ms_acPath,SYSTEM_MAX_PATH,acDirectory);
627  System::Strcat(ms_acPath,SYSTEM_MAX_PATH,acFilename);
628  return ms_acPath;
629  }
630  return nullptr;
631 }
