53 VEManifoldMesh::EPtr VEManifoldMesh::InsertEdge (int iV0, int iV1)
54 {
55  std::pair<int,int> kEKey(iV0,iV1);
56  EMapIterator pkEIter = m_kEMap.find(kEKey);
57  if (pkEIter != m_kEMap.end())
58  {
59  // edge already exists
60  return nullptr;
61  }
62 
63  // add new edge
64  EPtr pkEdge = m_oECreator(iV0,iV1);
65  m_kEMap[kEKey] = pkEdge;
66 
67  // add vertices to mesh
68  for (int i = 0; i < 2; i++)
69  {
70  int iV = pkEdge->V[i];
71  VPtr pkVertex;
72  VMapIterator pkVIter = m_kVMap.find(iV);
73  if (pkVIter == m_kVMap.end())
74  {
75  // first time vertex encountered
76  pkVertex = m_oVCreator(iV);
77  m_kVMap[iV] = pkVertex;
78 
79  // update vertex
80  pkVertex->E[0] = pkEdge;
81  }
82  else
83  {
84  // second time vertex encountered
85  pkVertex = pkVIter->second;
86  assert(pkVertex);
87 
88  // update vertex
89  if (pkVertex->E[1])
90  {
91  assert(false); // mesh must be manifold
92  return nullptr;
93  }
94  pkVertex->E[1] = pkEdge;
95 
96  // update adjacent edge
97  EPtr pkAdjacent = pkVertex->E[0];
98  assert(pkAdjacent);
99  for (int j = 0; j < 2; j++)
100  {
101  if (pkAdjacent->V[j] == iV)
102  {
103  pkAdjacent->E[j] = pkEdge;
104  break;
105  }
106  }
107 
108  // update edge
109  pkEdge->E[i] = pkAdjacent;
110  }
111  }
112 
113  return pkEdge;
114 }

116 bool VEManifoldMesh::RemoveEdge (int iV0, int iV1)
117 {
118  std::pair<int,int> kEKey(iV0,iV1);
119  EMapIterator pkEIter = m_kEMap.find(kEKey);
120  if (pkEIter == m_kEMap.end())
121  {
122  // edge does not exist
123  return false;
124  }
125 
126  EPtr pkEdge = pkEIter->second;
127  for (int i = 0; i < 2; i++)
128  {
129  // inform vertices you are going away
130  VMapIterator pkVIter = m_kVMap.find(pkEdge->V[i]);
131  assert(pkVIter != m_kVMap.end());
132  Vertex* pkVertex = pkVIter->second;
133  assert(pkVertex);
134  if (pkVertex->E[0] == pkEdge)
135  {
136  // one-edge vertices always have pointer in slot zero
137  pkVertex->E[0] = pkVertex->E[1];
138  pkVertex->E[1] = nullptr;
139  }
140  else if (pkVertex->E[1] == pkEdge)
141  {
142  pkVertex->E[1] = nullptr;
143  }
144  else
145  {
146  assert(false);
147  return false;
148  }
149 
150  // remove vertex if you had the last reference to it
151  if (!pkVertex->E[0] && !pkVertex->E[1])
152  {
153  m_kVMap.erase(pkVertex->V);
154  WM4_DELETE pkVertex;
155  }
156 
157  // inform adjacent edges you are going away
158  EPtr pkAdjacent = pkEdge->E[i];
159  if (pkAdjacent)
160  {
161  for (int j = 0; j < 2; j++)
162  {
163  if (pkAdjacent->E[j] == pkEdge)
164  {
165  pkAdjacent->E[j] = nullptr;
166  break;
167  }
168  }
169  }
170  }
171 
172  m_kEMap.erase(kEKey);
173  WM4_DELETE pkEdge;
174  return true;
175 }
