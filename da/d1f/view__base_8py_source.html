51 class ViewProviderDraft(object):
52  """The base class for Draft view providers.
53 
54  Parameters
55  ----------
56  vobj : a base C++ view provider
57  The view provider of the scripted object (`obj.ViewObject`),
58  which commonly may be of types `PartGui::ViewProvider2DObjectPython`,
59  `PartGui::ViewProviderPython`, or `Gui::ViewProviderFeaturePython`.
60 
61  A basic view provider is instantiated during the creation
62  of the base C++ object, for example,
63  `Part::Part2DObjectPython`, `Part::FeaturePython`,
64  or `App::FeaturePython`.
65 
66  >>> obj = App.ActiveDocument.addObject('Part::Part2DObjectPython')
67  >>> vobj = obj.ViewObject
68  >>> ViewProviderDraft(vobj)
69 
70  This view provider class instance is stored in the `Proxy` attribute
71  of the base view provider.
72  ::
73  vobj.Proxy = self
74 
75  Attributes
76  ----------
77  Object : the base C++ object
78  The scripted document object that is associated
79  with this view provider, which commonly may be of types
80  `Part::Part2DObjectPython`, `Part::FeaturePython`,
81  or `App::FeaturePython`.
82 
83  texture : coin.SoTexture2
84  A texture that could be added to this object.
85 
86  texcoords : coin.SoTextureCoordinatePlane
87  The coordinates defining a plane to use for aligning the texture.
88 
89  These class attributes are accessible through the `Proxy` object:
90  `vobj.Proxy.Object`, `vobj.Proxy.texture`, etc.

257  def onChanged(self, vobj, prop):
258  """Run when a view property is changed.
259 
260  Override this method to handle the behavior
261  of the view provider depending on changes that occur to its properties
262  such as line color, line width, point color, point size,
263  draw style, shape color, transparency, and others.
264 
265  This method updates the texture and pattern if
266  the properties `TextureImage`, `Pattern`, `ShapeAppearance`,
267  and `PatternSize` change.
268 
269  Parameters
270  ----------
271  vobj : the view provider of the scripted object.
272  This is `obj.ViewObject`.
273 
274  prop : str
275  Name of the property that was modified.
276  """
277  # treatment of patterns and image textures
278  if prop in ("TextureImage", "Pattern", "ShapeAppearance"):
279  if hasattr(self.Object, "Shape"):
280  if self.Object.Shape.Faces:
281  path = None
282  if hasattr(vobj, "TextureImage"):
283  if vobj.TextureImage:
284  path = vobj.TextureImage
285  if not path:
286  if hasattr(vobj, "Pattern"):
287  if str(vobj.Pattern) in utils.svg_patterns():
288  path = utils.svg_patterns()[vobj.Pattern][1]
289  else:
290  path = "None"
291  if path and vobj.RootNode:
292  switch = gui_utils.find_coin_node(vobj.RootNode, coin.SoSwitch)
293  if switch is not None:
294  if switch.getChildren().getLength() > 0:
295  innodes = switch.getChild(0).getChildren().getLength()
296  if innodes > 2:
297  r = switch.getChild(0).getChild(innodes-1)
298  i = QtCore.QFileInfo(path)
299  if self.texture:
300  r.removeChild(self.texture)
301  self.texture = None
302  if self.texcoords:
303  r.removeChild(self.texcoords)
304  self.texcoords = None
305  if i.exists():
306  size = None
307  if ".SVG" in path.upper():
308  size = params.get_param("HatchPatternResolution")
309  if not size:
310  size = 128
311  im = gui_utils.load_texture(path, size)
312  if im:
313  self.texture = coin.SoTexture2()
314  self.texture.image = im
315  r.insertChild(self.texture, 1)
316  if size:
317  s = 1
318  if hasattr(vobj, "PatternSize"):
319  if vobj.PatternSize:
320  s = vobj.PatternSize
321  self.texcoords = coin.SoTextureCoordinatePlane()
322  self.texcoords.directionS.setValue(s, 0, 0)
323  self.texcoords.directionT.setValue(0, s, 0)
324  r.insertChild(self.texcoords, 2)
325  elif prop == "PatternSize":
326  if hasattr(self, "texcoords"):
327  if self.texcoords:
328  s = 1
329  if vobj.PatternSize:
330  s = vobj.PatternSize
331  vS = App.Vector(self.texcoords.directionS.getValue().getValue())
332  vT = App.Vector(self.texcoords.directionT.getValue().getValue())
333  vS.Length = s
334  vT.Length = s
335  self.texcoords.directionS.setValue(vS.x, vS.y, vS.z)
336  self.texcoords.directionT.setValue(vT.x, vT.y, vT.z)
337  return
