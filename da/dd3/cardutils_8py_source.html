359 def get_material_template(withSpaces=False):
360  # material properties
361  # see the following resources in the FreeCAD wiki for more
362  # information about the material specific properties:
363  # https://www.freecad.org/wiki/Material_data_model
364  # https://www.freecad.org/wiki/Material
365 
366  print("Call to get_material_template() successful")
367 
368  import yaml
369  template_data = yaml.safe_load(
370  open(join(FreeCAD.ConfigGet('AppHomePath'), 'Mod/Material/Templatematerial.yml'))
371  )
372  if withSpaces:
373  # on attributes, add a space before a capital letter
374  # will be used for better display in the ui
375  import re
376  new_template = []
377  for group in template_data:
378  new_group = {}
379  gg = list(group)[0] # group dict has only one key
380  # iterating over a dict and changing it is not allowed
381  # thus it is iterated over a list of the keys
382  new_group[gg] = {}
383  for proper in list(group[gg]):
384  new_proper = re.sub(r"(\w)([A-Z]+)", r"\1 \2", proper)
385  # strip underscores of vectorial properties
386  new_proper = new_proper.replace("_", " ")
387  new_group[gg][new_proper] = group[gg][proper]
388  new_template.append(new_group)
389  template_data = new_template
390  return template_data
