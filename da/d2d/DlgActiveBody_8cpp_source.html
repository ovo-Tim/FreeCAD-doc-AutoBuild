44  : QDialog(parent)

52  this, &DlgActiveBody::accept);
53 
54  if (!infoText.isEmpty()) {
55  ui->label->setText(infoText + QString::fromUtf8("\n\n") + tr("Please select"));
56  }
57 
58  auto bodies = _doc->getObjectsOfType(PartDesign::Body::getClassTypeId());
59 
60  PartDesign::Body* bodyOfActiveObject = nullptr;
61  for (const auto &obj : Gui::Selection().getSelection()) {
62  bodyOfActiveObject = PartDesign::Body::findBodyOf(obj.pObject);
63  break; // Just get the body for first selected object
64  }
65 
66  for (const auto &body : bodies) {

68  item->setData(Qt::UserRole, QVariant::fromValue(body));
69  ui->bodySelect->addItem(item);
70 
71  if (body == bodyOfActiveObject) {
72  item->setSelected(true);
73  }
74 
75  // TODO: Any other logic (hover, select effects on view etc.)
76  }
77 
78  if (!bodyOfActiveObject) {
79  // by default select the first item so that the user
80  // can continue by clicking Ok without further action
81  QListWidgetItem* first = ui->bodySelect->item(0);
82  if (first)
83  first->setSelected(true);

89 void DlgActiveBody::accept()
90 {
91  auto selectedItems = ui->bodySelect->selectedItems();
92  if (selectedItems.empty())
93  return;
94 
95  App::DocumentObject* selectedBody =
96  selectedItems[0]->data(Qt::UserRole).value<App::DocumentObject*>();
97  if (selectedBody) {
98  activeBody = makeBodyActive(selectedBody, _doc);
99  }
100  else {
101  // A transaction must be created as otherwise the undo/redo is broken
102  App::GetApplication().setActiveTransaction(QT_TRANSLATE_NOOP("Command", "Add a Body"), true);
103  activeBody = makeBody(_doc);
104  App::GetApplication().closeActiveTransaction();
105  }
106 
107  QDialog::accept();
108 }
