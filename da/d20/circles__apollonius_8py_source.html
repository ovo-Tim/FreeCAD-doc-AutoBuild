24 """Provides various functions for Apollonius and Soddy circle operations.

62 def outerSoddyCircle(circle1, circle2, circle3):
63  """Compute the outer soddy circle for three tightly packed circles.
64 
65  Original Java code Copyright (rc) 2008 Werner Randelshofer
66  Converted to python by Martin Buerbaum 2009
67  http://www.randelshofer.ch/treeviz/
68  Either Creative Commons Attribution 3.0, the MIT license,
69  or the GNU Lesser General License LGPL.
70  """
71  if (geomType(circle1) != "Circle"
72  or geomType(circle2) != "Circle"
73  or geomType(circle3) != "Circle"):
74  print("debug: outerSoddyCircle bad parameters! Must be circles.")
75  return None
76 
77  A = circle1.Curve.Center
78  B = circle2.Curve.Center
79  C = circle3.Curve.Center
80 
81  ra = circle1.Curve.Radius
82  rb = circle2.Curve.Radius
83  rc = circle3.Curve.Radius
84 
85  # Solution using Descartes' theorem, as described here:
86  # http://en.wikipedia.org/wiki/Descartes%27_theorem
87  k1 = 1 / ra
88  k2 = 1 / rb
89  k3 = 1 / rc
90  k4 = abs(k1 + k2 + k3 - 2 * math.sqrt(k1 * k2 + k2 * k3 + k3 * k1))
91 
92  q1 = (k1 + 0j) * (A.x + A.y * 1j)
93  q2 = (k2 + 0j) * (B.x + B.y * 1j)
94  q3 = (k3 + 0j) * (C.x + C.y * 1j)
95 
96  temp = ((q1 * q2) + (q2 * q3) + (q3 * q1))
97  q4 = q1 + q2 + q3 - ((2 + 0j) * cmath.sqrt(temp))
98 
99  z = q4 / (k4 + 0j)
100 
101  # If the formula is not solvable, we return no circle.
102  if not z or not (1 / k4):
103  return None
104 
105  X = -z.real
106  Y = -z.imag
107  print("Outer Soddy circle: " + str(X) + " " + str(Y) + "\n") # Debug
108 
109  # The Radius of the outer soddy circle can also be calculated
110  # with the following formula:
111  # radiusOuter = abs(r1*r2*r3 / (r1*r2 + r1*r3 + r2*r3 - 2 * math.sqrt(r1*r2*r3 * (r1+r2+r3))))
112  circ = Part.Circle(App.Vector(X, Y, A.z), NORM, 1 / k4)
113 
114  return circ
