95 void CmdPartDesignBody::activated(int iMsg)
96 {
97  Q_UNUSED(iMsg);
98 
99  App::Part *actPart = PartDesignGui::getActivePart ();
100  App::Part* partOfBaseFeature = nullptr;
101 
102  std::vector<App::DocumentObject*> features =
103  getSelection().getObjectsOfType(Part::Feature::getClassTypeId());
104  App::DocumentObject* baseFeature = nullptr;
105  bool addtogroup = false;
106 
107 
108  if (!features.empty()) {
109  if (features.size() == 1) {
110  baseFeature = features[0];
111  if ( baseFeature->isDerivedFrom ( PartDesign::Feature::getClassTypeId() ) &&
112  PartDesign::Body::findBodyOf ( baseFeature ) ) {
113  // Prevent creating bodies based on features already belonging to other bodies
114  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Bad base feature"),
115  QObject::tr("Body can't be based on a PartDesign feature."));
116  baseFeature = nullptr;
117  }
118  else if (PartDesign::Body::findBodyOf ( baseFeature )){
119  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Bad base feature"),
120  QObject::tr("%1 already belongs to a body, can't use it as base feature for another body.")
121  .arg(QString::fromUtf8(baseFeature->Label.getValue())));
122  baseFeature = nullptr;
123  }
124  else if ( baseFeature->isDerivedFrom ( Part::BodyBase::getClassTypeId() ) ) {
125  // Prevent creating bodies based on bodies (but don't pop-up a dialog)
126  baseFeature = nullptr;
127  }
128  else {
129  partOfBaseFeature = App::Part::getPartOfObject(baseFeature);
130  if (partOfBaseFeature && partOfBaseFeature != actPart){
131  //prevent cross-part mess
132  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Bad base feature"),
133  QObject::tr("Base feature (%1) belongs to other part.")
134  .arg(QString::fromUtf8(baseFeature->Label.getValue())));
135  baseFeature = nullptr;
136  }
137  else if (baseFeature->isDerivedFrom(Sketcher::SketchObject::getClassTypeId())) {
138  // Add sketcher to the body's group property
139  addtogroup = true;
140  }
141  // if a standard Part feature (not a PartDesign feature) is selected then check
142  // the number of solids/shells
143  else if (!baseFeature->isDerivedFrom(PartDesign::Feature::getClassTypeId())) {
144  const TopoDS_Shape& shape = static_cast<Part::Feature*>(baseFeature)->Shape.getValue();
145  if (!shape.IsNull()) {
146  int numSolids = 0;
147  int numShells = 0;
148  for (TopExp_Explorer xp(shape, TopAbs_SOLID); xp.More(); xp.Next()) {
149  numSolids++;
150  }
151  for (TopExp_Explorer xp(shape, TopAbs_SHELL, TopAbs_SOLID); xp.More(); xp.Next()) {
152  numShells++;
153  }
154 
155  QString warning;
156  if (numSolids > 1 && numShells == 0) {
157  warning = QObject::tr("The selected shape consists of multiple solids.\n"
158  "This may lead to unexpected results.");
159  }
160  else if (numShells > 1 && numSolids == 0) {
161  warning = QObject::tr("The selected shape consists of multiple shells.\n"
162  "This may lead to unexpected results.");
163  }
164  else if (numShells == 1 && numSolids == 0) {
165  warning = QObject::tr("The selected shape consists of only a shell.\n"
166  "This may lead to unexpected results.");
167  }
168  else if (numSolids + numShells > 1) {
169  warning = QObject::tr("The selected shape consists of multiple solids or shells.\n"
170  "This may lead to unexpected results.");
171  }
172 
173  if (!warning.isEmpty()) {
174  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Base feature"), warning);
175  }
176  }
177  }
178  }
179 
180  } else {
181  QMessageBox::warning(Gui::getMainWindow(), QObject::tr("Bad base feature"),
182  QObject::tr("Body may be based on no more than one feature."));
183  return;
184  }
185  }
186 
187 
188  openCommand(QT_TRANSLATE_NOOP("Command", "Add a Body"));
189 
190  std::string bodyName = getUniqueObjectName("Body");
191  const char* bodyString = bodyName.c_str();
192 
193  // add the Body feature itself, and make it active
194  doCommand(Doc,"App.activeDocument().addObject('PartDesign::Body','%s')", bodyString);
195  // set Label for i18n/L10N
196  std::string labelString = QObject::tr("Body").toUtf8().toStdString();
197  labelString = Base::Tools::escapeEncodeString(labelString);
198  doCommand(Doc,"App.ActiveDocument.getObject('%s').Label = '%s'", bodyString, labelString.c_str());
199  if (baseFeature) {
200  if (partOfBaseFeature){
201  //withdraw base feature from Part, otherwise visibility madness results
202  doCommand(Doc,"App.activeDocument().%s.removeObject(App.activeDocument().%s)",
203  partOfBaseFeature->getNameInDocument(), baseFeature->getNameInDocument());
204  }
205  if (addtogroup) {
206  doCommand(Doc,"App.activeDocument().%s.Group = [App.activeDocument().%s]",
207  bodyString, baseFeature->getNameInDocument());
208  }
209  else {
210  doCommand(Doc,"App.activeDocument().%s.BaseFeature = App.activeDocument().%s",
211  bodyString, baseFeature->getNameInDocument());
212  }
213  }
214  addModule(Gui,"PartDesignGui"); // import the Gui module only once a session
215 
216  if (actPart) {
217  doCommand(Doc,"App.activeDocument().%s.addObject(App.ActiveDocument.%s)",
218  actPart->getNameInDocument(), bodyString);
219  }
220 
221  doCommand(Gui::Command::Gui, "Gui.activateView('Gui::View3DInventor', True)\n"
222  "Gui.activeView().setActiveObject('%s', App.activeDocument().%s)",
223  PDBODYKEY, bodyString);
224 
225  // Make the "Create sketch" prompt appear in the task panel
226  doCommand(Gui,"Gui.Selection.clearSelection()");
227  doCommand(Gui,"Gui.Selection.addSelection(App.ActiveDocument.%s)", bodyString);
228 
229  // check if a proxy object has been created for the base feature inside the body
230  if (baseFeature) {
231  PartDesign::Body* body = dynamic_cast<PartDesign::Body*>
232  (baseFeature->getDocument()->getObject(bodyString));
233  if (body) {
234  std::vector<App::DocumentObject*> links = body->Group.getValues();
235  for (auto it : links) {
236  if (it->isDerivedFrom<PartDesign::FeatureBase>()) {
237  PartDesign::FeatureBase* base = static_cast<PartDesign::FeatureBase*>(it);
238  if (base && base->BaseFeature.getValue() == baseFeature) {
239  Gui::Application::Instance->hideViewProvider(baseFeature);
240  break;
241  }
242  }
243  }
244 
245  // for sketches open the feature dialog to rebase it to a new plane
246  // as requested in issue #0002862
247  if (addtogroup) {
248  std::vector<App::DocumentObject*> planes;
249  std::vector<PartDesignGui::TaskFeaturePick::featureStatus> status;
250  unsigned validPlaneCount = 0;
251  for (auto plane: body->getOrigin ()->planes()) {
252  planes.push_back (plane);
253  status.push_back(PartDesignGui::TaskFeaturePick::basePlane);
254  validPlaneCount++;
255  }
256 
257  if (validPlaneCount > 1) {
258  // Determines if user made a valid selection in dialog
259  auto accepter = [](const std::vector<App::DocumentObject*>& features) -> bool {
260  return !features.empty();
261  };
262 
263  // Called by dialog when user hits "OK" and accepter returns true
264  auto worker = [baseFeature](const std::vector<App::DocumentObject*>& features) {
265  // may happen when the user switched to an empty document while the
266  // dialog is open
267  if (features.empty())
268  return;
269  App::Plane* plane = static_cast<App::Plane*>(features.front());
270  std::string supportString = Gui::Command::getObjectCmd(plane,"(",", [''])");
271 
272  FCMD_OBJ_CMD(baseFeature,"AttachmentSupport = " << supportString);
273  FCMD_OBJ_CMD(baseFeature,"MapMode = '" << Attacher::AttachEngine::getModeName(Attacher::mmFlatFace) << "'");
274  Gui::Command::updateActive();
275  };
276 
277  // Called by dialog for "Cancel", or "OK" if accepter returns false
278  std::string docname = getDocument()->getName();
279  auto quitter = [docname]() {
280  Gui::Document* document = Gui::Application::Instance->getDocument(docname.c_str());
281  if (document)
282  document->abortCommand();
283  };
284 
285  // Show dialog and let user pick plane
286  Gui::TaskView::TaskDialog *dlg = Gui::Control().activeDialog();
287  if (!dlg) {
288  Gui::Selection().clearSelection();
289  Gui::Control().showDialog(new PartDesignGui::TaskDlgFeaturePick(planes, status, accepter, worker, true, quitter));
290  }
291  }
292  }
293  }
294  }
295 
296  updateActive();
297 }

322 void CmdPartDesignMigrate::activated(int iMsg)
323 {
324  Q_UNUSED(iMsg);
325  App::Document *doc = getDocument();
326 
327  std::set<PartDesign::Feature*> migrateFeatures;
328 
329 
330  // Retrieve all PartDesign Features objects and filter out features already belonging to some body
331  for ( const auto & feat: doc->getObjects( ) ) {
332  if( feat->isDerivedFrom( PartDesign::Feature::getClassTypeId() ) &&
333  !PartDesign::Body::findBodyOf( feat ) && PartDesign::Body::isSolidFeature ( feat ) ) {
334  migrateFeatures.insert ( static_cast <PartDesign::Feature *>( feat ) );
335  }
336  }
337 
338  if ( migrateFeatures.empty() ) {
339  if ( !PartDesignGui::isModernWorkflow ( doc ) ) {
340  // If there is nothing to migrate and workflow is still old just set it to modern
341  PartDesignGui::WorkflowManager::instance()->forceWorkflow (
342  doc, PartDesignGui::Workflow::Modern );
343  } else {
344  // Huh? nothing to migrate?
345  QMessageBox::warning ( nullptr, QObject::tr ( "Nothing to migrate" ),
346  QObject::tr ( "No PartDesign features found that don't belong to a body."
347  " Nothing to migrate." ) );
348  }
349  return;
350  }
351 
352  // Note: this action is undoable, should it be?
353  PartDesignGui::WorkflowManager::instance()->forceWorkflow ( doc, PartDesignGui::Workflow::Modern );
354 
355  // Put features into chains. Each chain should become a separate body.
356  std::list< std::list<PartDesign::Feature *> > featureChains;
357  std::list<PartDesign::Feature *> chain; //< the current chain we are working on
358 
359  for (auto featIt = migrateFeatures.begin(); !migrateFeatures.empty(); ) {
360  Part::Feature *base = (*featIt)->getBaseObject( /*silent =*/ true );
361 
362  chain.push_front ( *featIt );
363 
364  if ( !base || !base->isDerivedFrom (PartDesign::Feature::getClassTypeId () ) ||
365  PartDesignGui::isAnyNonPartDesignLinksTo ( static_cast <PartDesign::Feature *>(base),
366  /*respectGroups=*/ true ) ) {
367  // a feature based on nothing as well as on non-partdesign solid starts a new chain
368  auto newChainIt = featureChains.emplace (featureChains.end());
369  newChainIt->splice (newChainIt->end(), chain);
370  } else {
371  // we are basing on some partdesign feature which supposed to belong to some body
372  PartDesign::Feature *baseFeat = static_cast <PartDesign::Feature *>( base );
373 
374  auto baseFeatSetIt = migrateFeatures.find(baseFeat);
375 
376  if ( baseFeatSetIt != migrateFeatures.end() ) {
377  // base feature is pending for migration, switch to it and continue over
378  migrateFeatures.erase(featIt);
379  featIt = baseFeatSetIt;
380  continue;
381  } else {
382  // The base feature seems already assigned to some chain
383  // Find which
384  std::list<PartDesign::Feature *>::iterator baseFeatIt;
385  auto chainIt = std::find_if( featureChains.begin(), featureChains.end(),
386  [baseFeat, &baseFeatIt] ( std::list<PartDesign::Feature *>&chain ) mutable -> bool {
387  baseFeatIt = std::find( chain.begin(), chain.end(), baseFeat );
388  return baseFeatIt != chain.end();
389  } );
390 
391  if ( chainIt != featureChains.end() ) {
392  assert (baseFeatIt != chainIt->end());
393  if ( std::next ( baseFeatIt ) == chainIt->end() ) {
394  // just append our chain to already found
395  chainIt->splice ( chainIt->end(), chain );
396  // TODO: If we will hit a third part everything will be messed up again.
397  // Probably it will require a yet another smart-ass find_if. (2015-08-10, Fat-Zer)
398  } else {
399  // We have a fork of a partDesign feature here
400  // add a chain for current body
401  auto newChainIt = featureChains.emplace (featureChains.end());
402  newChainIt->splice (newChainIt->end(), chain);
403  // add a chain for forked one
404  newChainIt = featureChains.emplace (featureChains.end());
405  newChainIt->splice (newChainIt->end(), *chainIt,
406  std::next ( baseFeatIt ), chainIt->end());
407  }
408  } else {
409  // The feature is not present in list pending for migration,
410  // This generally shouldn't happen but may be if we run into some broken file
411  // Try to find out the body we should insert into
412  // TODO: Some error/warning is needed here (2015-08-10, Fat-Zer)
413  auto newChainIt = featureChains.emplace (featureChains.end());
414  newChainIt->splice (newChainIt->end(), chain);
415  }
416  }
417  }
418  migrateFeatures.erase ( featIt );
419  featIt = migrateFeatures.begin ();
420  // TODO: Align visibility (2015-08-17, Fat-Zer)
421  } /* for */
422 
423  // TODO: make it work without parts (2015-09-04, Fat-Zer)
424  // add a part if there is no active yet
425  App::Part *actPart = PartDesignGui::assertActivePart ();
426 
427  if (!actPart) {
428  return;
429  }
430 
431  // do the actual migration
432  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Migrate legacy Part Design features to Bodies"));
433 
434  for ( auto chainIt = featureChains.begin(); !featureChains.empty();
435  featureChains.erase (chainIt), chainIt = featureChains.begin () ) {
436 #ifndef FC_DEBUG
437  if ( chainIt->empty () ) { // prevent crash in release in case of errors
438  continue;
439  }
440 #else
441  assert ( !chainIt->empty () );
442 #endif
443  Part::Feature *base = chainIt->front()->getBaseObject ( /*silent =*/ true );
444 
445  // Find a suitable chain to work with
446  for( ; chainIt != featureChains.end(); chainIt ++) {
447  base = chainIt->front()->getBaseObject ( /*silent =*/ true );
448  if (!base || !base->isDerivedFrom ( PartDesign::Feature::getClassTypeId () ) ) {
449  break; // no base is ok
450  } else {
451  // The base feature is a PartDesign, it's a fork, try to reassign it to a body...
452  base = PartDesign::Body::findBodyOf ( base );
453  if ( base ) {
454  break;
455  }
456  }
457  }
458 
459  if ( chainIt == featureChains.end() ) {
460  // Shouldn't happen, may be only in case of some circular dependency?
461  // TODO Some error message (2015-08-11, Fat-Zer)
462  chainIt = featureChains.begin();
463  base = chainIt->front()->getBaseObject ( /*silent =*/ true );
464  }
465 
466  // Construct a Pretty Body name based on the Tip
467  std::string bodyName = getUniqueObjectName (
468  std::string ( chainIt->back()->getNameInDocument() ).append ( "Body" ).c_str () ) ;
469 
470  // Create a body for the chain
471  doCommand ( Doc,"App.activeDocument().addObject('PartDesign::Body','%s')", bodyName.c_str () );
472  doCommand ( Doc,"App.activeDocument().%s.addObject(App.ActiveDocument.%s)",
473  actPart->getNameInDocument (), bodyName.c_str () );
474  if (base) {
475  doCommand ( Doc,"App.activeDocument().%s.BaseFeature = App.activeDocument().%s",
476  bodyName.c_str (), base->getNameInDocument () );
477  }
478 
479  // Fill the body with features
480  for ( auto feature: *chainIt ) {
481  if ( feature->isDerivedFrom ( PartDesign::ProfileBased::getClassTypeId() ) ) {
482  // add the sketch and also reroute it if needed
483  PartDesign::ProfileBased *sketchBased = static_cast<PartDesign::ProfileBased *> ( feature );
484  Part::Part2DObject *sketch = sketchBased->getVerifiedSketch( /*silent =*/ true);
485  if ( sketch ) {
486  doCommand ( Doc,"App.activeDocument().%s.addObject(App.activeDocument().%s)",
487  bodyName.c_str (), sketch->getNameInDocument() );
488 
489  if ( sketch->isDerivedFrom ( Sketcher::SketchObject::getClassTypeId() ) ) {
490  try {
491  PartDesignGui::fixSketchSupport ( static_cast<Sketcher::SketchObject *> ( sketch ) );
492  } catch (Base::Exception &) {
493  QMessageBox::critical(Gui::getMainWindow(),
494  QObject::tr("Sketch plane cannot be migrated"),
495  QObject::tr("Please edit '%1' and redefine it to use a Base or "
496  "Datum plane as the sketch plane.").
497  arg(QString::fromUtf8(sketch->Label.getValue()) ) );
498  }
499  } else {
500  // TODO: Message that sketchbased is based not on a sketch (2015-08-11, Fat-Zer)
501  }
502  }
503  }
504  doCommand ( Doc,"App.activeDocument().%s.addObject(App.activeDocument().%s)",
505  bodyName.c_str (), feature->getNameInDocument() );
506 
507  PartDesignGui::relinkToBody ( feature );
508  }
509 
510  }
511 
512  updateActive();
513 }

537 void CmdPartDesignMoveTip::activated(int iMsg)
538 {
539  Q_UNUSED(iMsg);
540  std::vector<App::DocumentObject*> features = getSelection().getObjectsOfType(
541  Part::Feature::getClassTypeId() );
542  App::DocumentObject* selFeature;
543  PartDesign::Body* body= nullptr;
544 
545  if ( features.size() == 1 ) {
546  selFeature = features.front();
547  if ( selFeature->getTypeId().isDerivedFrom ( PartDesign::Body::getClassTypeId() ) ) {
548  body = static_cast<PartDesign::Body *> ( selFeature );
549  } else {
550  body = PartDesignGui::getBodyFor ( selFeature, /* messageIfNot =*/ false );
551  }
552  } else {
553  selFeature = nullptr;
554  }
555 
556  if (!selFeature) {
557  QMessageBox::warning (nullptr, QObject::tr( "Selection error" ),
558  QObject::tr( "Select exactly one PartDesign feature or a body." ) );
559  return;
560  } else if (!body) {
561  QMessageBox::warning (nullptr, QObject::tr( "Selection error" ),
562  QObject::tr( "Couldn't determine a body for the selected feature '%s'.", selFeature->Label.getValue() ) );
563  return;
564  } else if ( !selFeature->isDerivedFrom(PartDesign::Feature::getClassTypeId () ) &&
565  selFeature != body && body->BaseFeature.getValue() != selFeature ) {
566  QMessageBox::warning (nullptr, QObject::tr( "Selection error" ),
567  QObject::tr( "Only a solid feature can be the tip of a body." ) );
568  return;
569  }
570 
571  App::DocumentObject* oldTip = body->Tip.getValue();
572  if (oldTip == selFeature) { // it's not generally an error, so print only a console message
573  Base::Console().Message ("%s is already the tip of the body\n", selFeature->getNameInDocument () );
574  return;
575  }
576 
577  openCommand(QT_TRANSLATE_NOOP("Command", "Move tip to selected feature"));
578 
579  if (selFeature == body) {
580  FCMD_OBJ_CMD(body,"Tip = None");
581  } else {
582  FCMD_OBJ_CMD(body,"Tip = " << getObjectCmd(selFeature));
583 
584  // Adjust visibility to show only the Tip feature
585  FCMD_OBJ_SHOW(selFeature);
586  }
587 
588  // TODO: Hide all datum features after the Tip feature? But the user might have already hidden some and wants to see
589  // others, so we would have to remember their state somehow
590  updateActive();
591 }

843 void CmdPartDesignMoveFeatureInTree::activated(int iMsg)
844 {
845  Q_UNUSED(iMsg);
846  std::vector<App::DocumentObject*> features = getSelection().getObjectsOfType(Part::Feature::getClassTypeId());
847  if (features.empty())
848  return;
849 
850  PartDesign::Body *body = PartDesignGui::getBodyFor ( features.front(), false );
851  App::DocumentObject * bodyBase = nullptr;
852  // sanity check
853  bool allFeaturesFromSameBody = true;
854 
855  if ( body ) {
856  bodyBase = body->BaseFeature.getValue();
857  for ( auto feat: features ) {
858  if ( !body->hasObject ( feat ) ) {
859  allFeaturesFromSameBody = false;
860  break;
861  }
862  if ( bodyBase == feat) {
863  QMessageBox::warning (nullptr, QObject::tr( "Selection error" ),
864  QObject::tr( "Impossible to move the base feature of a body." ) );
865  return;
866  }
867  }
868  }
869  if (!body || ! allFeaturesFromSameBody) {
870  QMessageBox::warning (nullptr, QObject::tr( "Selection error" ),
871  QObject::tr( "Select one or more features from the same body." ) );
872  return;
873  }
874 
875  // Create a list of all features in this body
876  const std::vector<App::DocumentObject*> & model = body->Group.getValues();
877 
878  // Ask user to select the target feature
879  bool ok;
880  QStringList items;
881  if ( bodyBase ) {
882  items.push_back( QString::fromUtf8 ( bodyBase->Label.getValue () ) );
883  } else {
884  items.push_back( QObject::tr( "Beginning of the body" ) );
885  }
886  for ( auto feat: model ) {
887  items.push_back( QString::fromUtf8 ( feat->Label.getValue() ) );
888  }
889 
890  QString text = QInputDialog::getItem(Gui::getMainWindow(),
891  qApp->translate("PartDesign_MoveFeatureInTree", "Select feature"),
892  qApp->translate("PartDesign_MoveFeatureInTree", "Select a feature from the list"),
893  items, 0, false, &ok, Qt::MSWindowsFixedSizeDialogHint);
894  if (!ok)
895  return;
896  int index = items.indexOf(text);
897  // first object is the beginning of the body
898  App::DocumentObject* target = index != 0 ? model[index-1] : nullptr;
899 
900  openCommand(QT_TRANSLATE_NOOP("Command", "Move an object inside tree"));
901 
902  App::DocumentObject* lastObject = target;
903  for ( auto feat: features ) {
904  if ( feat == target ) continue;
905 
906  // Remove and re-insert the feature to/from the Body, preserving their order.
907  // TODO: if tip was moved the new position of tip is quite undetermined (2015-08-07, Fat-Zer)
908  // TODO: warn the user if we are moving an object to some place before the object's link (2015-08-07, Fat-Zer)
909  FCMD_OBJ_CMD(body,"removeObject(" << getObjectCmd(feat) << ")");
910  FCMD_OBJ_CMD(body,"insertObject(" << getObjectCmd(feat) << ","<< getObjectCmd(lastObject) << ", True)");
911 
912  lastObject = feat;
913  }
914 
915  // Dependency order check.
916  // We must make sure the resulting objects of PartDesign::Feature do not
917  // depend on later objects
918  std::vector<App::DocumentObject*> bodyFeatures;
919  std::map<App::DocumentObject*,size_t> orders;
920  for(auto obj : body->Group.getValues()) {
921  if(obj->isDerivedFrom(PartDesign::Feature::getClassTypeId())) {
922  orders.emplace(obj,bodyFeatures.size());
923  bodyFeatures.push_back(obj);
924  }
925  }
926  bool failed = false;
927  std::ostringstream ss;
928  for(size_t i=0;i<bodyFeatures.size();++i) {
929  auto feat = bodyFeatures[i];
930  for(auto obj : feat->getOutList()) {
931  if(obj->isDerivedFrom(PartDesign::Feature::getClassTypeId()))
932  continue;
933  for(auto dep : App::Document::getDependencyList({obj})) {
934  auto it = orders.find(dep);
935  if(it != orders.end() && it->second > i) {
936  ss << feat->Label.getValue() << ", " <<
937  obj->Label.getValue() << " -> " <<
938  it->first->Label.getValue();
939  if(!failed)
940  failed = true;
941  else
942  ss << std::endl;
943  }
944  }
945  }
946  }
947  if(failed) {
948  QMessageBox::critical (nullptr, QObject::tr( "Dependency violation" ),
949  QObject::tr( "Early feature must not depend on later feature.\n\n")
950  + QString::fromUtf8(ss.str().c_str()));
951  abortCommand();
952  return;
953  }
954 
955  // If the selected objects have been moved after the current tip then ask the
956  // user if they want the last object to be the new tip.
957  // Only do this for features that can hold a tip (not for e.g. datums)
958  if ( lastObject != target && body->Tip.getValue() == target
959  && lastObject->isDerivedFrom(PartDesign::Feature::getClassTypeId()) ) {
960  QMessageBox msgBox(Gui::getMainWindow());
961  msgBox.setIcon(QMessageBox::Question);
962  msgBox.setWindowTitle(qApp->translate("PartDesign_MoveFeatureInTree", "Move tip"));
963  msgBox.setText(qApp->translate("PartDesign_MoveFeatureInTree", "The moved feature appears after the currently set tip."));
964  msgBox.setInformativeText(qApp->translate("PartDesign_MoveFeatureInTree", "Do you want the last feature to be the new tip?"));
965  msgBox.setStandardButtons(QMessageBox::Yes | QMessageBox::No);
966  msgBox.setDefaultButton(QMessageBox::No);
967  int ret = msgBox.exec();
968  if (ret == QMessageBox::Yes)
969  FCMD_OBJ_CMD(body, "Tip = " << getObjectCmd(lastObject));
970  }
971 
972  updateActive();
973 }
