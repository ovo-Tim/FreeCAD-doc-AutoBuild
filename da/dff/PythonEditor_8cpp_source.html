181 void PythonEditor::keyPressEvent(QKeyEvent* e)
182 {
188  if (e->key() == Qt::Key_Enter || e->key() == Qt::Key_Return) {
189  bool shiftPressed = e->modifiers() & Qt::ShiftModifier;
190  ParameterGrp::handle hPrefGrp = getWindowParameter();
191  int indent = hPrefGrp->GetInt( "IndentSize", 4 );
192  bool space = hPrefGrp->GetBool( "Spaces", true );
193  QString ch = space ? QString::fromLatin1(" ")
194  : QString::fromLatin1("\t");
195 
196  QTextCursor cursor = textCursor();
197  QString currentLineText = cursor.block().text();
198  bool endsWithColon = currentLineText.endsWith(QLatin1Char(':'));
199  int currentIndentation = 0;
200  //count spaces/tabs at start of current line
201  for (auto c : currentLineText) {
202  if (c == ch) {
203  currentIndentation++;
204  } else {
205  break;
206  }
207  }
208  cursor.insertBlock(); //new line
209  cursor.movePosition(QTextCursor::StartOfBlock); //carriage return
210  //Shift+Enter means dedent, but ensure we are not at column 0
211  if (shiftPressed && currentIndentation >= indent){
212  currentIndentation -= indent;
213  }
214  //insert appropriate number of spaces/tabs to match current indentation
215  cursor.insertText(QString(currentIndentation, ch[0]));
216  //if the line ended in a colon, then we need to add another tab or multiple spaces
217  if (endsWithColon) {
218  if (space){
219  cursor.insertText(QString(indent, ch[0])); //4 more spaces by default
220  } else {
221  cursor.insertText(ch); //1 more tab
222  }
223  }
224  setTextCursor(cursor);
225  return; //skip default handler
226  }
227  PythonTextEditor::keyPressEvent(e); //wasn't enter key, so let base class handle it
228 }

230 void PythonEditor::onComment()
231 {
232  QTextCursor cursor = textCursor();
233  int selStart = cursor.selectionStart();
234  int selEnd = cursor.selectionEnd();
235  QTextBlock block;
236  cursor.beginEditBlock();
237  for (block = document()->begin(); block.isValid(); block = block.next()) {
238  int pos = block.position();
239  int off = block.length()-1;
240  // at least one char of the block is part of the selection
241  if ( pos >= selStart || pos+off >= selStart) {
242  if ( pos+1 > selEnd )
243  break; // end of selection reached
244  cursor.setPosition(block.position());
245  cursor.insertText(QLatin1String("#"));
246  selEnd++;
247  }
248  }
249 
250  cursor.endEditBlock();
251 }

253 void PythonEditor::onUncomment()
254 {
255  QTextCursor cursor = textCursor();
256  int selStart = cursor.selectionStart();
257  int selEnd = cursor.selectionEnd();
258  QTextBlock block;
259  cursor.beginEditBlock();
260  for (block = document()->begin(); block.isValid(); block = block.next()) {
261  int pos = block.position();
262  int off = block.length()-1;
263  // at least one char of the block is part of the selection
264  if ( pos >= selStart || pos+off >= selStart) {
265  if ( pos+1 > selEnd )
266  break; // end of selection reached
267  if (block.text().startsWith(QLatin1String("#"))) {
268  cursor.setPosition(block.position());
269  cursor.deleteChar();
270  selEnd--;
271  }
272  }
273  }
274 
275  cursor.endEditBlock();

328 void PythonSyntaxHighlighter::highlightBlock (const QString & text)
329 {
330  int i = 0;
331  QChar prev, ch;
332 
333  const int Standard = 0; // Standard text
334  const int Digit = 1; // Digits
335  const int Comment = 2; // Comment begins with #
336  const int Literal1 = 3; // String literal beginning with "
337  const int Literal2 = 4; // Other string literal beginning with '
338  const int Blockcomment1 = 5; // Block comments beginning and ending with """
339  const int Blockcomment2 = 6; // Other block comments beginning and ending with '''
340  const int ClassName = 7; // Text after the keyword class
341  const int DefineName = 8; // Text after the keyword def
342 
343  int endStateOfLastPara = previousBlockState();
344  if (endStateOfLastPara < 0 || endStateOfLastPara > maximumUserState())
345  endStateOfLastPara = Standard;
346 
347  while ( i < text.length() )
348  {
349  ch = text.at( i );
350 
351  switch ( endStateOfLastPara )
352  {
353  case Standard:
354  {
355  switch ( ch.unicode() )
356  {
357  case '#':
358  {
359  // begin a comment
360  setFormat( i, 1, this->colorByType(SyntaxHighlighter::Comment));
361  endStateOfLastPara=Comment;
362  } break;
363  case '"':
364  {
365  // Begin either string literal or block comment
366  if ((i>=2) && text.at(i-1) == QLatin1Char('"') &&
367  text.at(i-2) == QLatin1Char('"'))
368  {
369  setFormat( i-2, 3, this->colorByType(SyntaxHighlighter::BlockComment));
370  endStateOfLastPara=Blockcomment1;
371  }
372  else
373  {
374  setFormat( i, 1, this->colorByType(SyntaxHighlighter::String));
375  endStateOfLastPara=Literal1;
376  }
377  } break;
378  case '\'':
379  {
380  // Begin either string literal or block comment
381  if ((i>=2) && text.at(i-1) == QLatin1Char('\'') &&
382  text.at(i-2) == QLatin1Char('\''))
383  {
384  setFormat( i-2, 3, this->colorByType(SyntaxHighlighter::BlockComment));
385  endStateOfLastPara=Blockcomment2;
386  }
387  else
388  {
389  setFormat( i, 1, this->colorByType(SyntaxHighlighter::String));
390  endStateOfLastPara=Literal2;
391  }
392  } break;
393  case ' ':
394  case '\t':
395  {
396  // ignore whitespaces
397  } break;
398  case '(': case ')': case '[': case ']':
399  case '+': case '-': case '*': case '/':
400  case ':': case '%': case '^': case '~':
401  case '!': case '=': case '<': case '>': // possibly two characters
402  {
403  setFormat(i, 1, this->colorByType(SyntaxHighlighter::Operator));
404  endStateOfLastPara=Standard;
405  } break;
406  default:
407  {
408  // Check for normal text
409  if ( ch.isLetter() || ch == QLatin1Char('_') )
410  {
411  QString buffer;
412  int j=i;
413  while ( ch.isLetterOrNumber() || ch == QLatin1Char('_') ) {
414  buffer += ch;
415  ++j;
416  if (j >= text.length())
417  break; // end of text
418  ch = text.at(j);
419  }
420 
421  if ( d->keywords.contains( buffer ) != 0 ) {
422  if ( buffer == QLatin1String("def"))
423  endStateOfLastPara = DefineName;
424  else if ( buffer == QLatin1String("class"))
425  endStateOfLastPara = ClassName;
426 
427  QTextCharFormat keywordFormat;
428  keywordFormat.setForeground(this->colorByType(SyntaxHighlighter::Keyword));
429  keywordFormat.setFontWeight(QFont::Bold);
430  setFormat( i, buffer.length(), keywordFormat);
431  }
432  else {
433  setFormat( i, buffer.length(),this->colorByType(SyntaxHighlighter::Text));
434  }
435 
436  // increment i
437  if ( !buffer.isEmpty() )
438  i = j-1;
439  }
440  // this is the beginning of a number
441  else if ( ch.isDigit() )
442  {
443  setFormat(i, 1, this->colorByType(SyntaxHighlighter::Number));
444  endStateOfLastPara=Digit;
445  }
446  // probably an operator
447  else if ( ch.isSymbol() || ch.isPunct() )
448  {
449  setFormat( i, 1, this->colorByType(SyntaxHighlighter::Operator));
450  }
451  }
452  }
453  } break;
454  case Comment:
455  {
456  setFormat( i, 1, this->colorByType(SyntaxHighlighter::Comment));
457  } break;
458  case Literal1:
459  {
460  setFormat( i, 1, this->colorByType(SyntaxHighlighter::String));
461  if ( ch == QLatin1Char('"') )
462  endStateOfLastPara = Standard;
463  } break;
464  case Literal2:
465  {
466  setFormat( i, 1, this->colorByType(SyntaxHighlighter::String));
467  if ( ch == QLatin1Char('\'') )
468  endStateOfLastPara = Standard;
469  } break;
470  case Blockcomment1:
471  {
472  setFormat( i, 1, this->colorByType(SyntaxHighlighter::BlockComment));
473  if ( i>=2 && ch == QLatin1Char('"') &&
474  text.at(i-1) == QLatin1Char('"') &&
475  text.at(i-2) == QLatin1Char('"'))
476  endStateOfLastPara = Standard;
477  } break;
478  case Blockcomment2:
479  {
480  setFormat( i, 1, this->colorByType(SyntaxHighlighter::BlockComment));
481  if ( i>=2 && ch == QLatin1Char('\'') &&
482  text.at(i-1) == QLatin1Char('\'') &&
483  text.at(i-2) == QLatin1Char('\''))
484  endStateOfLastPara = Standard;
485  } break;
486  case DefineName:
487  {
488  if ( ch.isLetterOrNumber() || ch == QLatin1Char(' ') || ch == QLatin1Char('_') )
489  {
490  setFormat( i, 1, this->colorByType(SyntaxHighlighter::Defname));
491  }
492  else
493  {
494  if ( ch.isSymbol() || ch.isPunct() )
495  setFormat(i, 1, this->colorByType(SyntaxHighlighter::Operator));
496  endStateOfLastPara = Standard;
497  }
498  } break;
499  case ClassName:
500  {
501  if ( ch.isLetterOrNumber() || ch == QLatin1Char(' ') || ch == QLatin1Char('_') )
502  {
503  setFormat( i, 1, this->colorByType(SyntaxHighlighter::Classname));
504  }
505  else
506  {
507  if (ch.isSymbol() || ch.isPunct() )
508  setFormat( i, 1, this->colorByType(SyntaxHighlighter::Operator));
509  endStateOfLastPara = Standard;
510  }
511  } break;
512  case Digit:
513  {
514  if (ch.isDigit() || ch == QLatin1Char('.'))
515  {
516  setFormat( i, 1, this->colorByType(SyntaxHighlighter::Number));
517  }
518  else
519  {
520  if ( ch.isSymbol() || ch.isPunct() )
521  setFormat( i, 1, this->colorByType(SyntaxHighlighter::Operator));
522  endStateOfLastPara = Standard;
523  }
524  }break;
525  }
526 
527  prev = ch;
528  i++;
529  }
530 
531  // only block comments can have several lines
532  if ( endStateOfLastPara != Blockcomment1 && endStateOfLastPara != Blockcomment2 )
533  {
534  endStateOfLastPara = Standard ;
535  }
536 
537  setCurrentBlockState(endStateOfLastPara);
538 }
