62 void TaskGeomHatch::initUi()
63 {
64  ui->fcFile->setFileName(QString::fromUtf8(m_file.data(), m_file.size()));
65  std::vector<std::string> names = PATLineSpec::getPatternList(m_file);
66  QStringList qsNames = listToQ(names);
67 
68  ui->cbName->addItems(qsNames);
69  int nameIndex = ui->cbName->findText(QString::fromUtf8(m_name.data(), m_name.size()));
70  if (nameIndex > -1) {
71  ui->cbName->setCurrentIndex(nameIndex);
72  } else {
73  Base::Console().Warning("Warning - Pattern name *%s* not found in current PAT File\n", m_name.c_str());
74  }
75  connect(ui->cbName, qOverload<int>(&QComboBox::currentIndexChanged), this, &TaskGeomHatch::onNameChanged);
76 
77  ui->sbScale->setValue(m_scale);
78  ui->sbScale->setSingleStep(0.1);
79  connect(ui->sbScale, qOverload<double>(&QuantitySpinBox::valueChanged), this, &TaskGeomHatch::onScaleChanged);
80  ui->sbWeight->setValue(m_weight);
81  ui->sbWeight->setSingleStep(0.1);
82  connect(ui->sbWeight, qOverload<double>(&QuantitySpinBox::valueChanged), this, &TaskGeomHatch::onLineWeightChanged);
83  ui->ccColor->setColor(m_color.asValue<QColor>());
84  connect(ui->ccColor, &ColorButton::changed, this, &TaskGeomHatch::onColorChanged);
85 
86  ui->dsbRotation->setValue(m_rotation);
87  connect(ui->dsbRotation, qOverload<double>(&QDoubleSpinBox::valueChanged), this, &TaskGeomHatch::onRotationChanged);
88  ui->dsbOffsetX->setValue(m_offset.x);
89  connect(ui->dsbOffsetX, qOverload<double>(&QDoubleSpinBox::valueChanged), this, &TaskGeomHatch::onOffsetChanged);
90  ui->dsbOffsetY->setValue(m_offset.y);
91  connect(ui->dsbOffsetY, qOverload<double>(&QDoubleSpinBox::valueChanged), this, &TaskGeomHatch::onOffsetChanged);
92 }

164 bool TaskGeomHatch::reject()
165 {
166  if (getCreateMode()) {
167  std::string HatchName = m_hatch->getNameInDocument();
168  Gui::Command::doCommand(Gui::Command::Gui, "App.activeDocument().removeObject('%s')", HatchName.c_str());
169  Gui::Command::doCommand(Gui::Command::Gui, "Gui.ActiveDocument.resetEdit()");
170  m_source->touch();
171  m_source->getDocument()->recompute();
172  } else {
173  m_hatch->FilePattern.setValue(m_origFile);
174  m_hatch->NamePattern.setValue(m_origName);
175  m_hatch->ScalePattern.setValue(m_origScale);
176  m_hatch->PatternRotation.setValue(m_origRotation);
177  m_hatch->PatternOffset.setValue(m_origOffset);
178  m_Vp->ColorPattern.setValue(m_origColor);
179  m_Vp->WeightPattern.setValue(m_origWeight);
180  }
181  return false;
182 }
