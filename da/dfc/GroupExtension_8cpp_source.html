82 std::vector<DocumentObject*> GroupExtension::addObjects(std::vector<DocumentObject*> objs)
83 {
84 
85  std::vector<DocumentObject*> added;
86  std::vector<DocumentObject*> grp = Group.getValues();
87  for (auto obj : objs) {
88 
89  if (!allowObject(obj)) {
90  continue;
91  }
92 
93  if (hasObject(obj)) {
94  continue;
95  }
96 
97  // only one group per object. Note that it is allowed to be in a group and geofeaturegroup.
98  // However, getGroupOfObject() returns only normal groups, no GeoFeatureGroups. Hence this
99  // works.
100  auto* group = App::GroupExtension::getGroupOfObject(obj);
101  if (group && group != getExtendedObject()) {
102  group->getExtensionByType<App::GroupExtension>()->removeObject(obj);
103  }
104 
105  // if we are in a geofeaturegroup we need to ensure the object is too
106  auto geogrp = GeoFeatureGroupExtension::getGroupOfObject(getExtendedObject());
107  auto objgrp = GeoFeatureGroupExtension::getGroupOfObject(obj);
108  if (geogrp != objgrp) {
109  // what to do depends on if we are in geofeature group or not
110  if (geogrp) {
111  geogrp->getExtensionByType<GeoFeatureGroupExtension>()->addObject(obj);
112  }
113  else {
114  objgrp->getExtensionByType<GeoFeatureGroupExtension>()->removeObject(obj);
115  }
116  }
117 
118  grp.push_back(obj);
119  added.push_back(obj);
120  }
121 
122  Group.setValues(grp);
123 
124  return added;
125 }

140 std::vector<DocumentObject*> GroupExtension::removeObjects(std::vector<DocumentObject*> objs)
141 {
142 
143  const std::vector<DocumentObject*>& grp = Group.getValues();
144  std::vector<DocumentObject*> newGrp = grp;
145  std::vector<DocumentObject*> removed;
146 
147  std::vector<DocumentObject*>::iterator end = newGrp.end();
148  for (auto obj : objs) {
149  auto res = std::remove(newGrp.begin(), end, obj);
150  if (res != end) {
151  end = res;
152  removed.push_back(obj);
153  }
154  }
155 
156  newGrp.erase(end, newGrp.end());
157  if (grp.size() != newGrp.size()) {
158  Group.setValues(newGrp);
159  }
160 
161  return removed;
162 }

164 void GroupExtension::removeObjectsFromDocument()
165 {

175 void GroupExtension::removeObjectFromDocument(DocumentObject* obj)
176 {
177  // check that object is not invalid
178  if (!obj || !obj->isAttachedToDocument()) {
179  return;
180  }
181 
182  // remove all children
183  if (obj->hasExtension(GroupExtension::getExtensionClassTypeId())) {
184  GroupExtension* grp = static_cast<GroupExtension*>(
185  obj->getExtension(GroupExtension::getExtensionClassTypeId()));
186 
187  // recursive call to remove all subgroups
188  grp->removeObjectsFromDocument();
189  }

203 bool GroupExtension::hasObject(const DocumentObject* obj, bool recursive) const
204 {
205  if (obj == getExtendedObject()) {
206  return false;
207  }
208 
209  try {
210  const std::vector<DocumentObject*>& grp = Group.getValues();
211  for (auto child : grp) {
212 
213  if (!child) {
214  continue;
215  }
216 
217  if (child == obj) {
218  return true;
219  }
220  else if (child == getExtendedObject()) {
221  throw Base::RuntimeError(
222  "Cyclic dependencies detected: Search cannot be performed");
223  }
224  else if (recursive && child->hasExtension(GroupExtension::getExtensionClassTypeId())) {
225  App::GroupExtension* subGroup = static_cast<App::GroupExtension*>(
226  child->getExtension(GroupExtension::getExtensionClassTypeId()));
227  std::vector<const GroupExtension*> history;
228  history.push_back(this);
229 
230  if (subGroup->recursiveHasObject(obj, subGroup, history)) {
231  return true;
232  }
233  }
234  }
235 
236  return false;
237  }
238  catch (const Base::RuntimeError& e) {
239  e.ReportException();
240  return false;
241  }
242 }

244 bool GroupExtension::recursiveHasObject(const DocumentObject* obj,
245  const GroupExtension* group,
246  std::vector<const GroupExtension*> history) const
247 {
248 
249  // the purpose is to prevent infinite recursion when groups form a cyclic graph. To do this
250  // we store every group we processed on the current leave of the tree, and if we reach an
251  // already processed group we know that it not really is a tree but a cycle.
252  history.push_back(this);
253 
254  // we use hasObject with out recursion to allow override in derived classes
255  if (group->hasObject(obj, false)) {
256  return true;
257  }
258 
259  // we checked for the searched object already with hasObject and did not find it, now we need to
260  // do the same for all subgroups
261  for (auto child : group->Group.getValues()) {
262 
263  if (!child) {
264  continue;
265  }
266 
267  if (child->hasExtension(GroupExtension::getExtensionClassTypeId())) {
268 
269  auto ext = child->getExtensionByType<GroupExtension>();
270 
271  if (std::find(history.begin(), history.end(), ext) != history.end()) {
272  throw Base::RuntimeError(
273  "Cyclic dependencies detected: Search cannot be performed");
274  }
275 
276  if (recursiveHasObject(obj, ext, history)) {
277  return true;
278  }
279  }
280  }
281  return false;
282 }

320 DocumentObject* GroupExtension::getGroupOfObject(const DocumentObject* obj)
321 {
322  // note that we return here only Groups, but nothing derived from it, e.g. no GeoFeatureGroups.
323  // That is important as there are clear differences between groups/geofeature groups (e.g. an
324  // object can be in only one group, and only one geofeaturegroup, however, it can be in both at
325  // the same time)
326  for (auto o : obj->getInList()) {
327  if (o->hasExtension(App::GroupExtension::getExtensionClassTypeId(), false)) {
328  return o;
329  }
330  if (o->hasExtension(App::GroupExtensionPython::getExtensionClassTypeId(), false)) {
331  return o;
332  }
333  }
334 
335  return nullptr;

349 void GroupExtension::extensionOnChanged(const Property* p)
350 {
351 
352  // objects are only allowed in a single group. Note that this check must only be done for normal
353  // groups, not any derived classes
354  if ((this->getExtensionTypeId() == GroupExtension::getExtensionClassTypeId()) && p == &Group
355  && !Group.testStatus(Property::User3)) {
356  if (!getExtendedObject()->isRestoring()
357  && !getExtendedObject()->getDocument()->isPerformingTransaction()) {
358 
359  bool error = false;
360  auto corrected = Group.getValues();
361  for (auto obj : Group.getValues()) {
362 
363  // we have already set the obj into the group, so in a case of multiple groups
364  // getGroupOfObject would return anyone of it and hence it is possible that we miss
365  // an error. We need a custom check
366  auto list = obj->getInList();
367  for (auto in : list) {
368  if (in->hasExtension(App::GroupExtension::getExtensionClassTypeId(), false)
369  && in != getExtendedObject()) {
370  error = true;
371  corrected.erase(std::remove(corrected.begin(), corrected.end(), obj),
372  corrected.end());
373  }
374  }
375  }
376 
377  // if an error was found we need to correct the values and inform the user
378  if (error) {
379  Base::ObjectStatusLocker<Property::Status, Property> guard(Property::User3, &Group);
380  Group.setValues(corrected);
381  throw Base::RuntimeError("Object can only be in a single Group");
382  }
383  }
384  }
385 
386  if (p == &Group) {
387  _Conns.clear();
388  for (auto obj : Group.getValue()) {
389  if (obj && obj->isAttachedToDocument()) {
390  // NOLINTBEGIN
391  _Conns[obj] = obj->signalChanged.connect(
392  std::bind(&GroupExtension::slotChildChanged, this, sp::_1, sp::_2));
393  // NOLINTEND
394  }
395  }
396  }
397 
398  App::Extension::extensionOnChanged(p);
399 }

408 bool GroupExtension::extensionGetSubObject(DocumentObject*& ret,
409  const char* subname,
410  PyObject** pyObj,
411  Base::Matrix4D* mat,
412  bool /*transform*/,
413  int depth) const
414 {
415  const char* dot;
416  if (!subname || *subname == 0) {
417  auto obj = Base::freecad_dynamic_cast<const DocumentObject>(getExtendedContainer());
418  ret = const_cast<DocumentObject*>(obj);
419  return true;
420  }
421  dot = strchr(subname, '.');
422  if (!dot) {
423  return false;
424  }
425  if (subname[0] != '$') {
426  ret = Group.findUsingMap(std::string(subname, dot));
427  }
428  else {
429  std::string name = std::string(subname + 1, dot);
430  for (auto child : Group.getValues()) {
431  if (name == child->Label.getStrValue()) {
432  ret = child;
433  break;
434  }
435  }
436  }
437  if (!ret) {
438  return false;
439  }
440  return ret->getSubObject(dot + 1, pyObj, mat, true, depth + 1);
