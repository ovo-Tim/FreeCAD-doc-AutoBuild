61  Recompute2 = 9, // set when the object is being recomputed in the second pass
62  PartialObject = 10,
63  PendingRecompute = 11, // set by Document, indicating the object is in recomputation queue
64  ObjImporting = 13, // Mark the object as importing
65  NoTouch = 14, // no touch on any property change
66  GeoExcluded = 15, // mark as a member but not claimed by GeoFeatureGroup
67  Expand = 16, // indicate the object's tree item expansion status

70  RecomputeExtension = 19, // mark the object to recompute its extensions
71  TouchOnColorChange = 20, // inform view provider touch object on color change

73 };
