44 def make_text(string, placement=None, screen=False, height=None, line_spacing=1):
45  """Create a Text object containing the given list of strings.
46 
47  The current color and text height and font specified in preferences
48  are used.
49 
50  Parameters
51  ----------
52  string: str, or list of str
53  String to display on screen.
54  If it is a list, each element in the list represents a new text line.
55 
56  placement: Base::Placement, Base::Vector3, or Base::Rotation, optional
57  Defaults to `None`.
58  If it is provided, it is the placement of the new text.
59  The input could be a full placement, just a vector indicating
60  the translation, or just a rotation.
61 
62  screen: bool or None, optional
63  Defaults to `False`.
64  If it is `True`, the DisplayMode is set to "Screen".
65  If it is `False`, it is set to "World".
66  If it is `None`, the DisplayMode depends on the current preferences.
67 
68  height: float or None, optional
69  Defaults to `None`.
70  A height value for the text, in mm.
71  If it is `None` or zero, the FontSize depends on the current preferences.
72 
73  line_spacing: float or None, optional
74  Defaults to 1.
75  The line spacing factor.
76  If it is `None` or zero, the LineSpacing depends on the current preferences.
77 
78  Returns
79  -------
80  App::FeaturePython
81  A scripted object of type `'Text'`.
82  This object does not have a `Shape` attribute, as the text is created
83  on screen by Coin (pivy).
84 
85  None
86  If there is a problem it will return `None`.
87  """
88  _name = "make_text"
89 
90  found, doc = utils.find_doc(App.activeDocument())
91  if not found:
92  _err(translate("draft","No active document. Aborting."))
93  return None
94 
95  try:
96  utils.type_check([(string, (str, list))], name=_name)
97  except TypeError:
98  _err(translate("draft","Wrong input: must be a list of strings or a single string."))
99  return None
100 
101  if (type(string) is list
102  and not all(isinstance(element, str) for element in string)):
103  _err(translate("draft","Wrong input: must be a list of strings or a single string."))
104  return None
105 
106  if not placement:
107  placement = App.Placement()
108  try:
109  utils.type_check([(placement, (App.Placement,
110  App.Vector,
111  App.Rotation))], name=_name)
112  except TypeError:
113  _err(translate("draft","Wrong input: must be a placement, a vector, or a rotation."))
114  return None
115 
116  # Convert the vector or rotation to a full placement
117  if isinstance(placement, App.Vector):
118  placement = App.Placement(placement, App.Rotation())
119  elif isinstance(placement, App.Rotation):
120  placement = App.Placement(App.Vector(), placement)
121 
122  new_obj = doc.addObject("App::FeaturePython", "Text")
123  Text(new_obj)
124  new_obj.Text = string
125  new_obj.Placement = placement
126 
127  if App.GuiUp:
128  ViewProviderText(new_obj.ViewObject)
129 
130  if screen is None:
131  # Keep value as set by viewprovider
132  pass
133  elif screen:
134  new_obj.ViewObject.DisplayMode = "Screen"
135  else:
136  new_obj.ViewObject.DisplayMode = "World"
137 
138  if height: # Keep value as set by viewprovider if zero or None
139  new_obj.ViewObject.FontSize = height
140 
141  if line_spacing: # Keep value as set by viewprovider if zero or None
142  new_obj.ViewObject.LineSpacing = line_spacing
143 
144  gui_utils.format_object(new_obj)
145  gui_utils.select(new_obj)
146 
147  return new_obj

157 def convert_draft_texts(textslist=None):
158  """Convert the given Annotation to a Draft text.
159 
160  In the past, the `Draft Text` object didn't exist; text objects
161  were of type `App::Annotation`. This function was introduced
162  to convert those older objects to a `Draft Text` scripted object.
163 
164  This function was already present at splitting time during v0.19.
165 
166  Parameters
167  ----------
168  textslist: list of objects, optional
169  It defaults to `None`.
170  A list containing `App::Annotation` objects or a single of these
171  objects.
172  If it is `None` it will convert all objects in the current document.
173  """
174  _name = "convert_draft_texts"
175 
176  found, doc = utils.find_doc(App.activeDocument())
177  if not found:
178  _err(translate("draft","No active document. Aborting."))
179  return None
180 
181  if not textslist:
182  textslist = list()
183  for obj in doc.Objects:
184  if obj.TypeId == "App::Annotation":
185  textslist.append(obj)
186 
187  if not isinstance(textslist, list):
188  textslist = [textslist]
189 
190  to_delete = []
191 
192  for obj in textslist:
193  label = obj.Label
194  obj.Label = label + ".old"
195 
196  # Create a new Draft Text object
197  new_obj = make_text(obj.LabelText, placement=obj.Position)
198  new_obj.Label = label
199  to_delete.append(obj)
200 
201  # Move the new object to the group which contained the old object
202  for in_obj in obj.InList:
203  if in_obj.isDerivedFrom("App::DocumentObjectGroup"):
204  if obj in in_obj.Group:
205  group = in_obj.Group
206  group.append(new_obj)
207  in_obj.Group = group
208 
209  for obj in to_delete:
210  doc.removeObject(obj.Name)
