137  # MaterialManager is unaware of the material until it is saved
138  #
139  # When initially creating the material, setting overwrite=True preserves the UUID. This should not
140  # be used when saving after properties have been edited as this could adversely affect other
141  # documents or parts using the same material. Setting overwrite=False, or omitting it, will change
142  # the UUID. It will also fail if the material file already exists.
143  #
144  # Similarly, saveAsCopy=True preserves the UUID and should be used carefully. It will save an
145  # identical copy of the original but in a different location.
146  #
147  # The third optional parameter is saveInherited. When set to true it will mark models and properties
148  # as inherited without duplicating them. When false, they will be copied as uninherited. Avoid
149  # self-inheritance as this creates an invalid model. It will have a different UUID than the original.
150  #
151  self.MaterialManager.save("User", material, "Example/Frakenstein.FCMat", overwrite=True)
152 
153  # Now the UUID is valid
154  self.assertEqual(len(material.UUID), 36)
155  self.assertEqual(material.UUID, uuid)
156  self.assertIn(uuid, self.MaterialManager.Materials)
157  self.assertIsNotNone(self.MaterialManager.getMaterialByPath("Example/Frakenstein.FCMat", "User"))
158  self.assertIsNotNone(self.MaterialManager.getMaterial(uuid))
