69 std::vector<int> getListOfSelectedGeoIds(bool forceInternalSelection)
70 {
71  std::vector<int> listOfGeoIds = {};
72 
73  // get the selection
74  std::vector<Gui::SelectionObject> selection;
75  selection = Gui::Selection().getSelectionEx(0, Sketcher::SketchObject::getClassTypeId());
76 
77  // only one sketch with its subelements are allowed to be selected
78  if (selection.size() != 1) {
79  QMessageBox::warning(Gui::getMainWindow(),
80  QObject::tr("Wrong selection"),
81  QObject::tr("Select elements from a single sketch."));
82  return {};
83  }
84 
85  // get the needed lists and objects
86  auto* Obj = static_cast<Sketcher::SketchObject*>(selection[0].getObject());
87  const std::vector<std::string>& subNames = selection[0].getSubNames();
88  if (!subNames.empty()) {
89 
90  for (auto& name : subNames) {
91  if (name.size() > 4 && name.substr(0, 4) == "Edge") {
92  int geoId = std::atoi(name.substr(4, 4000).c_str()) - 1;
93  listOfGeoIds.push_back(geoId);
94  }
95  else if (name.size() > 12 && name.substr(0, 12) == "ExternalEdge") {
96  int geoId = -std::atoi(name.substr(12, 4000).c_str()) - 2;
97  listOfGeoIds.push_back(geoId);
98  }
99  else if (name.size() > 6 && name.substr(0, 6) == "Vertex") {
100  // only if it is a GeomPoint
101  int VtId = std::atoi(name.substr(6, 4000).c_str()) - 1;
102  int geoId;
103  Sketcher::PointPos PosId;
104  Obj->getGeoVertexIndex(VtId, geoId, PosId);
105  if (isPoint(*Obj->getGeometry(geoId))) {
106  if (geoId >= 0) {
107  listOfGeoIds.push_back(geoId);
108  }
109  }
110  }
111  }
112  }
113 
114  if (forceInternalSelection) {
115  size_t loopSize = listOfGeoIds.size();
116  for (size_t i = 0; i < loopSize; i++) {
117  const Part::Geometry* geo = Obj->getGeometry(listOfGeoIds[i]);
118  if (isEllipse(*geo) || isArcOfEllipse(*geo) || isArcOfHyperbola(*geo) || isArcOfParabola(*geo) || isBSplineCurve(*geo)) {
119  const std::vector<Sketcher::Constraint*>& constraints = Obj->Constraints.getValues();
120  for (auto constr : constraints) {
121  if (constr->Type == InternalAlignment && constr->Second == listOfGeoIds[i]) {
122  if (std::find(listOfGeoIds.begin(), listOfGeoIds.end(), constr->First) == listOfGeoIds.end()) {
123  // If the value is not found, add it to the vector
124  listOfGeoIds.push_back(constr->First);
125  }
126  }
127  }
128  }
129  }
130  }
131 
132  if (listOfGeoIds.empty()) {
133  Gui::NotifyUserError(Obj,
134  QT_TRANSLATE_NOOP("Notifications", "Invalid selection"),
135  QT_TRANSLATE_NOOP("Notifications", "Selection has no valid geometries."));
136  }
137 
138  return listOfGeoIds;
139 }

153 bool copySelectionToClipboard(Sketcher::SketchObject* obj) {
154  std::vector<int> listOfGeoId = getListOfSelectedGeoIds(true);
155  if (listOfGeoId.empty()) { return false; }
156  sort(listOfGeoId.begin(), listOfGeoId.end());
157 
158  //Export selected geometries as a formatted string.
159  std::vector<Part::Geometry*> shapeGeometry;
160  for (auto geoId : listOfGeoId) {
161  Part::Geometry* geoNew = obj->getGeometry(geoId)->copy();
162  shapeGeometry.push_back(geoNew);
163  }
164  std::string geosAsStr = Sketcher::PythonConverter::convert(
165  "objectStr",
166  shapeGeometry,
167  Sketcher::PythonConverter::Mode::OmitInternalGeometry);
168 
169  // Export constraints of selected geos.
170  std::vector<Sketcher::Constraint*> shapeConstraints;
171  for (auto constr : obj->Constraints.getValues()) {
172 
173  auto isSelectedGeoOrAxis = [](const std::vector<int>& vec, int value) {
174  return (std::find(vec.begin(), vec.end(), value) != vec.end())
175  || value == GeoEnum::GeoUndef || value == GeoEnum::RtPnt
176  || value == GeoEnum::VAxis || value == GeoEnum::HAxis;
177  };
178 
179  if (!isSelectedGeoOrAxis(listOfGeoId, constr->First)
180  || !isSelectedGeoOrAxis(listOfGeoId, constr->Second)
181  || !isSelectedGeoOrAxis(listOfGeoId, constr->Third)) {
182  continue;
183  }
184 
185  Constraint* temp = constr->copy();
186  for (size_t j = 0; j < listOfGeoId.size(); j++) {
187  if (temp->First == listOfGeoId[j]) {
188  temp->First = j;
189  }
190  if (temp->Second == listOfGeoId[j]) {
191  temp->Second = j;
192  }
193  if (temp->Third == listOfGeoId[j]) {
194  temp->Third = j;
195  }
196  }
197  shapeConstraints.push_back(temp);
198  }
199  std::string cstrAsStr = Sketcher::PythonConverter::convert("objectStr", shapeConstraints, Sketcher::PythonConverter::GeoIdMode::AddLastGeoIdToGeoIds);
200 
201  std::string exportedData = "# Copied from sketcher. From:\n#objectStr = " + Gui::Command::getObjectCmd(obj) + "\n"
202  + geosAsStr + "\n" + cstrAsStr;
203 
204  if (!exportedData.empty()) {
205  QClipboard* clipboard = QGuiApplication::clipboard();
206  clipboard->setText(QString::fromStdString(exportedData));
207  return true;
208  }
209  return false;
210 }

355

358  SketcherGui::ReleaseHandler(doc);

371  Sketcher::SketchObject* Obj = static_cast<Sketcher::SketchObject*>(selection[0].getObject());
372  const std::vector<Sketcher::Constraint*>& vals = Obj->Constraints.getValues();
373 
374  std::string doc_name = Obj->getDocument()->getName();
375  std::string obj_name = Obj->getNameInDocument();
376 
377  getSelection().clearSelection();
378 
379  std::vector<std::string> constraintSubNames;
380  // go through the selected subelements
381  for (std::vector<std::string>::const_iterator it = SubNames.begin(); it != SubNames.end();
382  ++it) {
383  // only handle edges
384  if (it->size() > 4 && it->substr(0, 4) == "Edge") {
385  int GeoId = std::atoi(it->substr(4, 4000).c_str()) - 1;
386 
387  // push all the constraints
388  int i = 0;
389  for (std::vector<Sketcher::Constraint*>::const_iterator it = vals.begin();
390  it != vals.end();
391  ++it, ++i) {
392  if ((*it)->First == GeoId || (*it)->Second == GeoId || (*it)->Third == GeoId) {
393  constraintSubNames.push_back(
394  Sketcher::PropertyConstraintList::getConstraintName(i));
395  }
396  }
397  }
398  }
399 
400  if (!constraintSubNames.empty())
401  Gui::Selection().addSelections(doc_name.c_str(), obj_name.c_str(), constraintSubNames);
402 }
