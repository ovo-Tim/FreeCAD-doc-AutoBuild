125  double min = static_cast<double>(ui->sliderReduction->minimum());

147 bool TaskDecimating::accept()
148 {
149  std::vector<Mesh::Feature*> meshes = Gui::Selection().getObjectsOfType<Mesh::Feature>();
150  if (meshes.empty()) {
151  return true;
152  }
153  Gui::Selection().clearSelection();
154 
155  Gui::WaitCursor wc;
156  Gui::Command::openCommand(QT_TRANSLATE_NOOP("Command", "Mesh Decimating"));
157 
158  float tolerance = float(widget->tolerance());
159  float reduction = float(widget->reduction());
160  bool absolute = widget->isAbsoluteNumber();
161  int targetSize = 0;
162  if (absolute) {
163  targetSize = widget->targetNumberOfTriangles();
164  }
165  for (auto mesh : meshes) {
166  if (absolute) {
167  Gui::cmdAppObjectArgs(mesh, "decimate(%i)", targetSize);
168  }
169  else {
170  Gui::cmdAppObjectArgs(mesh, "decimate(%f, %f)", tolerance, reduction);
171  }
172  }
173 
174  Gui::Command::commitCommand();
175  return true;
176 }
