32 Circle::Circle(const Point& p0, const Point& p1, const Point& p2)
33 {
34  // from TangentCircles in http://code.google.com/p/heekscad/source/browse/trunk/src/Geom.cpp
35 
36  // set default values, in case this fails
37  m_radius = 0.0;
38  m_c = Point(0, 0);
39 
40  double x1 = p0.x;
41  double y1 = p0.y;
42  double x2 = p1.x;
43  double y2 = p1.y;
44  double x3 = p2.x;
45  double y3 = p2.y;
46 
47  double a = 2 * (x1 - x2);
48  double b = 2 * (y1 - y2);
49  double d = (x1 * x1 + y1 * y1) - (x2 * x2 + y2 * y2);
50 
51  double A = 2 * (x1 - x3);
52  double B = 2 * (y1 - y3);
53  double D = (x1 * x1 + y1 * y1) - (x3 * x3 + y3 * y3);
54 
55  double aBmbA = (a * B - b * A); // aB - bA
56 
57  // x = k + Kr where
58  double k = (B * d - b * D) / aBmbA;
59 
60  // y = l + Lr where
61  double l = (-A * d + a * D) / aBmbA;
62 
63  double qa = -1;
64  double qb = 0.0;
65  double qc = k * k + x1 * x1 - 2 * k * x1 + l * l + y1 * y1 - 2 * l * y1;
66 
67  // solve the quadratic equation, r = (-b +- sqrt(b*b - 4*a*c))/(2 * a)
68  for (int qs = 0; qs < 2; qs++) {
69  double bb = qb * qb;
70  double ac4 = 4 * qa * qc;
71  if (ac4 <= bb) {
72  double r = (-qb + ((qs == 0) ? 1 : -1) * sqrt(bb - ac4)) / (2 * qa);
73  double x = k;
74  double y = l;
75 
76  // set the circle
77  if (r >= 0.0) {
78  m_c = Point(x, y);
79  m_radius = r;
80  }
81  }
82  }
83 }

86 {
87  double rp = p.dist(m_c);
88  bool on = fabs(m_radius - rp) < accuracy;
