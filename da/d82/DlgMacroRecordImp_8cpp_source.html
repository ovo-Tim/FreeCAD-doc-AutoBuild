100 void DlgMacroRecordImp::onButtonStartClicked()
101 {
102  // test if the path already set
103  if (ui->lineEditPath->text().isEmpty()) {
104  QMessageBox::information(getMainWindow(),
105  tr("Macro recorder"),
106  tr("Specify first a place to save."));
107  return;
108  }
109 
110  QDir dir(macroPath);
111  if (!dir.exists()) {
112  QMessageBox::information(
113  getMainWindow(),
114  tr("Macro recorder"),
115  tr("The macro directory doesn't exist. Please, choose another one."));
116  return;
117  }
118 
119  // search in the macro path first for an already existing macro
120  QString fn = this->macroPath + ui->lineEditPath->text();
121  if (!fn.endsWith(QLatin1String(".FCMacro"))) {
122  fn += QLatin1String(".FCMacro");
123  }
124 
125  QFileInfo fi(fn);
126  if (fi.isFile() && fi.exists()) {
127  if (QMessageBox::question(
128  this,
129  tr("Existing macro"),
130  tr("The macro '%1' already exists. Do you want to overwrite?").arg(fn),
131  QMessageBox::Yes | QMessageBox::No,
132  QMessageBox::No)
133  == QMessageBox::No) {
134  return;
135  }
136  }
137 
138  QFile file(fn);
139  if (!file.open(QFile::WriteOnly)) {
140  QMessageBox::information(
141  getMainWindow(),
142  tr("Macro recorder"),
143  tr("You have no write permission for the directory. Please, choose another one."));
144  return;
145  }
146  file.close();
147 
148  // open the macro recording
149  this->macroManager->open(MacroManager::File, fn.toUtf8().constData());
150 
151  ui->buttonStart->setEnabled(false);
152  ui->buttonStop->setEnabled(true);
153  ui->buttonClose->setEnabled(false);
154  QDialog::accept();
155 }
