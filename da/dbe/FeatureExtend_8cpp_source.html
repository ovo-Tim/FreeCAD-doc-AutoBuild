95 App::DocumentObjectExecReturn* Extend::execute()
96 {
97  App::DocumentObject* part = Face.getValue();
98  if (!part || !part->isDerivedFrom<Part::Feature>()) {
99  return new App::DocumentObjectExecReturn("No shape linked.");
100  }
101  const auto& faces = Face.getSubValues();
102  if (faces.size() != 1) {
103  return new App::DocumentObjectExecReturn("Not exactly one sub-shape linked.");
104  }
105 
106  TopoDS_Shape shape =
107  static_cast<Part::Feature*>(part)->Shape.getShape().getSubShape(faces[0].c_str());
108  if (shape.IsNull() || shape.ShapeType() != TopAbs_FACE) {
109  return new App::DocumentObjectExecReturn("Sub-shape is not a face.");
110  }
111 
112  const TopoDS_Face& face = TopoDS::Face(shape);
113  BRepAdaptor_Surface adapt(face);
114  double u1 = adapt.FirstUParameter();
115  double u2 = adapt.LastUParameter();
116  double v1 = adapt.FirstVParameter();
117  double v2 = adapt.LastVParameter();
118 
119  double ur = u2 - u1;
120  double vr = v2 - v1;
121  double eu1 = u1 - ur * ExtendUNeg.getValue();
122  double eu2 = u2 + ur * ExtendUPos.getValue();
123  double ev1 = v1 - vr * ExtendVNeg.getValue();
124  double ev2 = v2 + vr * ExtendVPos.getValue();
125  double eur = eu2 - eu1;
126  double evr = ev2 - ev1;
127 
128  long numU = SampleU.getValue();
129  long numV = SampleV.getValue();
130  TColgp_Array2OfPnt approxPoints(1, numU, 1, numV);
131  for (long u = 0; u < numU; u++) {
132  double uu = eu1 + u * eur / (numU - 1);
133  for (long v = 0; v < numV; v++) {
134  double vv = ev1 + v * evr / (numV - 1);
135  BRepLProp_SLProps prop(adapt, uu, vv, 0, Precision::Confusion());
136  const gp_Pnt& pnt = prop.Value();
137  approxPoints(u + 1, v + 1) = pnt;
138  }
139  }
140 
141  Approx_ParametrizationType ParType = Approx_ChordLength;
142  Standard_Integer DegMin = 3;
143  Standard_Integer DegMax = 5;
144  GeomAbs_Shape Continuity = GeomAbs_C2;
145  Standard_Real Tol3d = Tolerance.getValue();
146 
147  GeomAPI_PointsToBSplineSurface approx;
148  approx.Init(approxPoints, ParType, DegMin, DegMax, Continuity, Tol3d);
149 
150  Handle(Geom_BSplineSurface) surface(approx.Surface());
151  BRepBuilderAPI_MakeFace mkFace(surface, Precision::Confusion());
152 
153  Shape.setValue(mkFace.Face());
154 
155  return StdReturn;
156 }

158 void Extend::onChanged(const App::Property* prop)
159 {
160  // using a mutex and lock to protect a recursive calling when setting the new values
161  if (lockOnChangeMutex) {
162  return;
163  }
164  Base::StateLocker lock(lockOnChangeMutex);
165 
166  if (ExtendUSymetric.getValue()) {
167  if (prop == &ExtendUNeg || prop == &ExtendUPos) {
168  auto changedValue = dynamic_cast<const App::PropertyFloat*>(prop);
169  if (changedValue) {
170  ExtendUNeg.setValue(changedValue->getValue());
171  ExtendUPos.setValue(changedValue->getValue());
172  }
173  }
174  }
175 
176  if (ExtendVSymetric.getValue()) {
177  if (prop == &ExtendVNeg || prop == &ExtendVPos) {
178  auto changedValue = dynamic_cast<const App::PropertyFloat*>(prop);
179  if (changedValue) {
180  ExtendVNeg.setValue(changedValue->getValue());
181  ExtendVPos.setValue(changedValue->getValue());
182  }
183  }
184  }
185  Part::Spline::onChanged(prop);
