106 int ImpExpDxfReadGui::GetDrawStyle() const
107 {
108  // The DXF line type can be quite complex, including both patterns, in-line text (for instance,
109  // so a line can mark itself as "Gas Main"), and shapes from a .SHX file (so a line can have,
110  // say, diamond shapes along it). The coin package we use to render objects into the screen has
111  // no facility for doing the in-line text or shapes, and has limited control over line/space
112  // patterns. The line pattern is divided into 16 pieces which can be either drawn or not. There
113  // is also support for line pattern scaling (to match the LTSCALE DXF property). Unfortunately
114  // neither of these is exposed at the FreeCAD level; line types are dumbed down to four choices:
115  // 0 "Solid" 0xffff i.e. solid
116  // 1 "Dashed" 0xf00f i.e. coarse dashes
117  // 2 "Dotted" 0x0f0f i.e. fine (half-size) dashes
118  // 3 "DashDot" 0xff88 i.e. long dash/dot (like a center line)
119  // Right now we infer these from the DXF LINETYPE name. In the long run we should look at the
120  // actual pattern, but the CDxfRead class does nothing with the LTYPE table. In the longer run,
121  // FreeCAD should expose the line pattern (the 16-bit number) for more versatility.
122 
123  // Make an array of patterns whose index is the "draw style" enum value for the pattern.
124  static std::array<std::regex, 4> matchers {
125  // Starts with "cont" (i.e. continuous) : return solid
126  // I can't find anything that actually says what happens if you have a top-level (i.e. not
127  // in a BLOCK) object whose line type is BYBLOCK. We treat this as Continuous.
128  std::basic_regex("^cont|^byblock$", std::regex::icase),
129  // starts with hidden, border, dash : return dashed
130  std::basic_regex("^hidden|^border|^dash", std::regex::icase),
131  // starts with dot : return dotted
132  std::basic_regex("^dot", std::regex::icase),
133  // dash & dot or center/centre/divide/phantom : return DashDot
134  std::basic_regex("dot.*dash|dash.*dot|^cent(er|re)|^divide|^phantom", std::regex::icase)};
135  // Some line type names can match several patterns, in particular dotdash can also match ^dot
136  // and dashdot can match ^dash Rather than code the latter patterns to forbid subsequent "dash"
137  // or "dot" so they are mutually exclusive, we just match the more specific pattern, which is
138  // the last one, first.
139  for (int i = matchers.size(); --i >= 0;) {
140  if (regex_search(m_entityAttributes.m_LineType, matchers.at(i))) {
141  return i;
142  }
143  }
144  // If we don't understand it, return solid
145  return 0;
146 }
