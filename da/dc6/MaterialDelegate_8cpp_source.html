175 bool MaterialDelegate::editorEvent(QEvent* event,
176  QAbstractItemModel* model,
177  const QStyleOptionViewItem& option,
178  const QModelIndex& index)
179 {
180  if (index.column() == 1) {
181  if (event->type() == QEvent::MouseButtonDblClick) {
182  auto treeModel = dynamic_cast<const QStandardItemModel*>(index.model());
183 
184  // Check we're not the material model root. This is also used to access the entry
185  // columns
186  auto item = treeModel->itemFromIndex(index);
187  auto group = item->parent();
188  if (!group) {
189  return QStyledItemDelegate::editorEvent(event, model, option, index);
190  }
191 
192  int row = index.row();
193 
194  // QString propertyName = group->child(row, 0)->text();
195  QString propertyName = group->child(row, 0)->data().toString();
196 
197  auto type = getType(index);
198  if (type == Materials::MaterialValue::Color) {
199  showColorModal(propertyName, item);
200  // Mark as handled
201  return true;
202  }
203  if (type == Materials::MaterialValue::MultiLineString) {
204  showMultiLineStringModal(propertyName, item);
205  // Mark as handled
206  return true;
207  }
208  if (type == Materials::MaterialValue::List || type == Materials::MaterialValue::FileList
209  || type == Materials::MaterialValue::ImageList) {
210  showListModal(propertyName, item);
211  // Mark as handled
212  return true;
213  }
214  if (type == Materials::MaterialValue::Array2D) {
215  showArray2DModal(propertyName, item);
216  // Mark as handled
217  return true;
218  }
219  if (type == Materials::MaterialValue::Array3D) {
220  showArray3DModal(propertyName, item);
221  // Mark as handled
222  return true;
223  }
224  if (type == Materials::MaterialValue::Image || type == Materials::MaterialValue::SVG) {
225  showImageModal(propertyName, item);
226  // Mark as handled
227  return true;
228  }
229  }
230  }
231  return QStyledItemDelegate::editorEvent(event, model, option, index);
232 }

234 void MaterialDelegate::showColorModal(const QString& propertyName, QStandardItem* item)
235 {
236  QColor currentColor; // = d->col;
237  currentColor.setRgba(parseColor(item->text()));
238  auto dlg = new QColorDialog(currentColor);
239 
240  dlg->setAttribute(Qt::WA_DeleteOnClose);
241  if (Gui::DialogOptions::dontUseNativeColorDialog()) {
242  dlg->setOptions(QColorDialog::DontUseNativeDialog);
243  }
244  dlg->setOption(QColorDialog::ColorDialogOption::ShowAlphaChannel, false);
245 
246  dlg->setCurrentColor(currentColor);
247  dlg->adjustSize();
248 
249  connect(dlg, &QColorDialog::finished, this, [&](int result) {
250  if (result == QDialog::Accepted) {
251  QColor color = dlg->selectedColor();
252  if (color.isValid()) {
253  QString colorText = QString(QString::fromStdString("(%1,%2,%3,%4)"))
254  .arg(color.red() / 255.0)
255  .arg(color.green() / 255.0)
256  .arg(color.blue() / 255.0)
257  .arg(color.alpha() / 255.0);
258  item->setText(colorText);
259  Q_EMIT const_cast<MaterialDelegate*>(this)->propertyChange(propertyName,
260  item->text());
261  }
262  }
263  });
264 
265  dlg->exec();
266 }

417 QWidget* MaterialDelegate::createWidget(QWidget* parent,
418  const QVariant& item,
419  const QModelIndex& index) const
420 {
421  QWidget* widget = nullptr;
422 
423  auto type = getType(index);
424  if (type == Materials::MaterialValue::Integer) {
425  auto spinner = new Gui::IntSpinBox(parent);
426  spinner->setMinimum(0);
427  spinner->setMaximum(INT_MAX);
428  spinner->setValue(item.toInt());
429  widget = spinner;
430  }
431  else if (type == Materials::MaterialValue::Float) {
432  auto spinner = new Gui::DoubleSpinBox(parent);
433 
434  // the magnetic permeability is the parameter for which many decimals matter
435  // the most however, even for this, 6 digits are sufficient
436  spinner->setDecimals(6);
437 
438  // for almost all Float parameters of materials a step of 1 would be too large
439  spinner->setSingleStep(0.1);
440 
441  spinner->setMinimum(std::numeric_limits<double>::min());
442  spinner->setMaximum(std::numeric_limits<double>::max());
443  spinner->setValue(item.toDouble());
444  widget = spinner;
445  }
446  else if (type == Materials::MaterialValue::Boolean) {
447  auto combo = new Gui::PrefComboBox(parent);
448  combo->insertItem(0, QString::fromStdString(""));
449  combo->insertItem(1, tr("False"));
450  combo->insertItem(2, tr("True"));
451  combo->setCurrentText(item.toString());
452  widget = combo;
453  }
454  else if (type == Materials::MaterialValue::Quantity) {
455  // auto input = new Gui::InputField(parent);
456  auto input = new Gui::QuantitySpinBox(parent);
457  input->setMinimum(std::numeric_limits<double>::min());
458  input->setMaximum(std::numeric_limits<double>::max());
459  input->setUnitText(getUnits(index));
460  // input->setPrecision(6);
461  input->setValue(item.value<Base::Quantity>());
462 
463  widget = input;
464  }
465  else if (type == Materials::MaterialValue::File) {
466  auto chooser = new Gui::FileChooser(parent);
467  if (!item.toString().isEmpty()) {
468  chooser->setFileName(item.toString());
469  }
470 
471  widget = chooser;
472  }
473  else {
474  // Default editor
475  auto lineEdit = new Gui::PrefLineEdit(parent);
476  lineEdit->setText(item.toString());
477  widget = lineEdit;
478  }
479 
480  return widget;
481 }
