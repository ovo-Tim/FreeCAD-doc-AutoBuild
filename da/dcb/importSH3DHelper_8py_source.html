219  def _import_home(self, home):
220  doc = App.ActiveDocument
221  self.total_object_count = self._get_object_count(home)
222  _msg(f"Importing home '{home.get('name')}' ...")
223  # Create the groups to organize the different resources together
224  self._create_groups()
225 
226  # Get all the FreeCAD object in the active doc, in order to allow
227  # for merge of existing object
228  if self.preferences["MERGE"]:
229  for object in doc.Objects:
230  if hasattr(object, 'id'):
231  self.fc_objects[object.id] = object
232 
233  # Let's create the project and site for this import
234  self._setup_project(home)
235 
236  # Import the <level> element if any. If none are defined
237  # create a default one.
238  if home.find(path='level') != None:
239  self._import_elements(home, 'level')
240  else:
241  # Has the default floor already been created from a
242  # previous import?
243  if self.preferences["DEBUG"]: _log("No level defined. Using default level ...")
244  self.default_floor = self.fc_objects.get('Level') if 'Level' in self.fc_objects else self._create_default_floor()
245  self.add_floor(self.default_floor)
246 
247  # Importing <room> elements ...
248  self._import_elements(home, 'room')
249 
250  # Importing <wall> elements ...
251  self._import_elements(home, 'wall')
252 
253  self._refresh()
254  if App.GuiUp and self.preferences["FIT_VIEW"]:
255  Gui.SendMsgToActiveView("ViewFit")
256 
257  # Importing <doorOrWindow> elements ...
258  if self.preferences["IMPORT_DOORS_AND_WINDOWS"]:
259  self._import_elements(home, 'doorOrWindow')
260  self._refresh()
261 
262  # Importing <pieceOfFurniture> && <furnitureGroup> elements ...
263  if self.preferences["IMPORT_FURNITURES"]:
264  self._import_elements(home, 'pieceOfFurniture')
265  for furniture_group in home.findall('furnitureGroup'):
266  self._import_elements(furniture_group, 'pieceOfFurniture', False)
267  self._refresh()
268 
269  # Importing <light> elements ...
270  if self.preferences["IMPORT_LIGHTS"]:
271  self._import_elements(home, 'light')
272  self._refresh()
273 
274  # Importing <observerCamera> elements ...
275  if self.preferences["IMPORT_CAMERAS"]:
276  self._import_elements(home, 'observerCamera')
277  self._import_elements(home, 'camera')
278  self._refresh()
279 
280  if self.preferences["CREATE_RENDER_PROJECT"] and self.site:
281  Project.create(doc, renderer="Povray", template="povray_standard.pov")
282  Gui.Selection.clearSelection()
283  Gui.Selection.addSelection(self.site)
284  Gui.runCommand('Render_View', 0)
285  self._refresh()

287  _msg(f"Successfully imported home '{home.get('name')}' ...")

520  def _import_elements(self, parent, tag, update_progress=True):
521  """Generic function to import a specific element.
522 
523  This function will lookup the handler registered for the elements
524  `tag` and then call it on each item. It also provides some update
525  on the whole process.
526 
527  Args:
528  parent (Element): the parent of the elements to be imported.
529  Usually the <home> element.
530  tag (str): the tag of the elements to be imported.
531  update_progress (bool, optional): whether to update the
532  progress. Set to false when importing a group of elements.
533  Defaults to True.
534  """
535  tags = list(self.handlers.keys())
536  elements = parent.findall(tag)
537  if update_progress and self.progress_bar:
538  self.progress_bar.stop()
539  self.progress_bar.start(f"Step {tags.index(tag)+1}/{len(tags)}: importing {len(elements)} '{tag}' elements. Please wait ...", len(elements))
540  _msg(f"Importing {len(elements)} '{tag}' elements ...")
541  def _process(tuple):
542  (i, elm) = tuple
543  _msg(f"Importing {tag}#{i} ({self.current_object_count + 1}/{self.total_object_count}) ...")
544  try:
545  self.handlers[tag].process(parent, i, elm)
546  except Exception as e:
547  _err(f"Failed to import <{tag}>#{i} ({elm.get('id', elm.get('name'))}):")
548  _err(str(e))
549  if update_progress and self.progress_bar:
550  self.progress_bar.next()
551  self.current_object_count = self.current_object_count + 1
552  list(map(_process, enumerate(elements)))

778  def _create_wall(self, floor, prev, next, elm):
779  """Create an Arch::Structure from an SH3D Element.
780 
781  The constructed wall will either be a straight wall or a curved
782  wall depending on the `elm` attributes.
783 
784  Args:
785  floor (Arch::Structure): The floor the wall belongs to
786  prev (Element): the xml element for the previous sibling wall
787  next (Element): the xml element for the next sibling wall
788  elm (Element): the xml element for the wall to be imported
789 
790  Returns:
791  Arch::Wall: the newly created wall
792  """
793  wall_details = self._get_wall_details(floor, elm)
794  assert wall_details is not None, f"Fail to get details of wall {elm.get('id')}. Bailing out! {elm} / {wall_details}"
795 
796  # Both the wall at start or the wall at end can be None.
797  prev_wall_details = self._get_wall_details(floor, prev)
798  next_wall_details = self._get_wall_details(floor, next)
799 
800  # Is the wall curved (i.e. arc_extent != 0) ?
801  if wall_details[5] != 0:
802  section_start, section_end, spine = self._create_curved_segment(
803  wall_details,
804  prev_wall_details,
805  next_wall_details)
806  else:
807  section_start, section_end, spine = self._create_straight_segment(
808  wall_details,
809  prev_wall_details,
810  next_wall_details)
811 
812  sweep = App.ActiveDocument.addObject('Part::Sweep')
813  sweep.Sections = [section_start, section_end]
814  sweep.Spine = spine
815  sweep.Solid = True
816  sweep.Frenet = False
817  section_start.Visibility = False
818  section_end.Visibility = False
819  spine.Visibility = False
820  App.ActiveDocument.recompute([sweep])
821  # Sometimes the Part::Sweep creates a "twisted" sweep which
822  # result in a broken wall. The solution is to use a compound
823  # object based on ruled surface instead.
824  if FIX_INVALID_SWEEP and (sweep.Shape.isNull() or not sweep.Shape.isValid()):
825  _log(f"Part::Sweep for wall#{elm.get('id')} is invalid. Using ruled surface instead ...")
826  ruled_surface = App.ActiveDocument.addObject('Part::RuledSurface')
827  ruled_surface.Curve1 = section_start
828  ruled_surface.Curve2 = section_end
829  App.ActiveDocument.recompute([ruled_surface])
830  _log(f"Creating compound object ...")
831  compound = App.ActiveDocument.addObject('Part::Compound')
832  compound.Links = [ruled_surface, section_start, section_end]
833  App.ActiveDocument.recompute([compound])
834  _log(f"Creating solid ...")
835  solid = App.ActiveDocument.addObject("Part::Feature")
836  solid.Shape = Part.Solid(Part.Shell(compound.Shape.Faces))
837  doc = App.ActiveDocument
838  doc.removeObject(compound.Label)
839  doc.recompute()
840  doc.removeObject(ruled_surface.Label)
841  doc.recompute()
842  doc.removeObject(sweep.Label)
843  doc.recompute()
844  doc.removeObject(spine.Label)
845  doc.recompute()
846  doc.removeObject(section_start.Label)
847  doc.removeObject(section_end.Label)
848  wall = Arch.makeWall(solid)
849  else:
850  wall = Arch.makeWall(sweep)
851  # For some reason the Length of the spine is not propagated to the
852  # wall itself...
853  wall.Length = spine.Length
854  return wall

946  def _get_section(self, wall_details, at_start, sibling_details):
947  """Returns a rectangular section at the specified coordinate.
948 
949  Returns a Rectangle that is then used as a section in the Part::Sweep
950  used to construct a wall. Depending whether the wall should be joined
951  with its siblings, the rectangle is either created and rotated around
952  the endpoint of the line that will be used as the spline of the sweep
953  or it is calculated as the intersection profile of the 2 walls.
954 
955  Args:
956  wall_details (tuple): The details of the wall
957  at_start (bool): indicate whether the section is for the start
958  point or the end point of the wall.
959  sibling_details (tuple): The details of the sibling wall
960 
961  Returns:
962  Rectangle: the section properly positioned
963  """
964  if self.importer.preferences["JOIN_ARCH_WALL"] and sibling_details:
965  # In case the walls are to be joined we determine the intersection
966  # of both wall which depends on their respective thickness.
967  # Calculate the left and right side of each wall
968  (start, end, thickness, height_start, height_end, _) = wall_details
969  (s_start, s_end, s_thickness, _, _, _) = sibling_details
970 
971  lside, rside = self._get_sides(start, end, thickness)
972  s_lside, s_rside = self._get_sides(s_start, s_end, s_thickness)
973  i_start, i_end = self._get_intersection_edge(lside, rside, s_lside, s_rside)
974 
975  height = height_start if at_start else height_end
976  i_start_z = i_start + App.Vector(0, 0, height)
977  i_end_z = i_end + App.Vector(0, 0, height)
978 
979  if self.importer.preferences["DEBUG"]:
980  _log(f"Joining wall {self._pv(end-start)}@{self._pv(start)} and wall {self._pv(s_end-s_start)}@{self._pv(s_start)}")
981  _log(f" wall: {self._pe(lside)},{self._pe(rside)}")
982  _log(f" sibling: {self._pe(s_lside)},{self._pe(s_rside)}")
983  _log(f"intersec: {self._pv(i_start)},{self._pv(i_end)}")
984  section = Draft.makeRectangle([i_start, i_end, i_end_z, i_start_z])
985  if self.importer.preferences["DEBUG"]:
986  _log(f"section: {section}")
987  else:
988  (start, end, thickness, height_start, height_end, _) = wall_details
989  height = height_start if at_start else height_end
990  center = start if at_start else end
991  a1, a2, _ = self._get_normal_angles(wall_details)
992  z_rotation = a1 if at_start else a2
993  section = Draft.makeRectangle(thickness, height)
994  Draft.move([section], App.Vector(-thickness/2, 0, 0))
995  Draft.rotate([section], 90, ORIGIN, X_NORM)

1007  for i in range(0, len(point_colors)):
1008  point_colors[i] = hex2rgb(DEBUG_POINT_COLORS[i%len(DEBUG_POINT_COLORS)])
1009  view.PointColorArray = point_colors
1010  view.PointSize = 5
1011 
1012  return section

1034  def _get_normal_angles(self, wall_details):
1035  """Return the angles of the normal at the endpoints of the wall.
1036 
1037  This method returns the normal angle of the sections that constitute
1038  the wall sweep. These angles can then be used to create the
1039  corresponding sections. Depending on whether the wall section is
1040  straight or curved, the section will be calculated slightly
1041  differently.
1042 
1043  Args:
1044  wall_details (tuple): The details of the wall
1045 
1046  Returns:
1047  float: the angle of the normal at the starting point
1048  float: the angle of the normal at the ending point
1049  bool: the angle of the normal at the ending point
1050  Vector: the center of the circle for a curved wall section
1051  float: the radius of said circle
1052  """
1053  (start, end, thickness, height_start, height_end, arc_extent) = wall_details
1054 
1055  angle_start = angle_end = 0
1056  invert_angle = False
1057  center = radius = None
1058  if arc_extent == 0:
1059  angle_start = angle_end = 90-math.degrees(DraftVecUtils.angle(end-start, X_NORM))
1060  else:
1061  # Calculate the circle that pases through the center of both rectangle
1062  # and has the correct angle between p1 and p2
1063  chord = DraftVecUtils.dist(start, end)
1064  radius = abs(chord / (2*math.sin(arc_extent/2)))
1065 
1066  circles = DraftGeomUtils.circleFrom2PointsRadius(start, end, radius)
1067  # We take the center that preserve the arc_extent orientation (in FC
1068  # coordinate). The orientation is calculated from start to end
1069  center = circles[0].Center
1070  if numpy.sign(arc_extent) != numpy.sign(DraftVecUtils.angle(start-center, end-center, Z_NORM)):
1071  invert_angle = True
1072  center = circles[1].Center
1073 
1074  # radius1 and radius2 are the vector from center to start and end respectively
1075  radius1 = start - center
1076  radius2 = end - center
1077 
1078  angle_start = math.degrees(DraftVecUtils.angle(X_NORM, radius1, Z_NORM))
1079  angle_end = math.degrees(DraftVecUtils.angle(X_NORM, radius2, Z_NORM))
1080 
1081  return angle_start, angle_end, (invert_angle, center, radius)
