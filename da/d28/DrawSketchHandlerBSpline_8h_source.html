454  bool canGoToNextMode() override
455  {
456  Sketcher::PointPos pointPos = constructionMethod() == ConstructionMethod::ControlPoints
457  ? Sketcher::PointPos::mid
458  : Sketcher::PointPos::start;
459  if (state() == SelectMode::SeekFirst) {
460  // insert point for pole/knot, defer internal alignment constraining.
461  if (!addPos()) {
462  return false;
463  }
464 
465  // add auto constraints on pole/knot
466  auto& ac0 = sugConstraints[0];
467  generateAutoConstraintsOnElement(ac0, geoIds.back(), pointPos);
468 
469  sketchgui->getSketchObject()->solve();
470  }
471  else if (state() == SelectMode::SeekSecond) {
472  // We stay in SeekSecond unless the user closed the bspline.
473  bool isClosed = false;
474 
475  // check if coincident with first pole/knot
476  for (auto& ac : sugConstraints.back()) {
477  if (ac.Type == Sketcher::Coincident) {
478  if (ac.GeoId == geoIds[0]) {
479  isClosed = true;
480  }
481  else {
482  // The coincidence with first point may be indirect
483  const auto coincidents =
484  sketchgui->getSketchObject()->getAllCoincidentPoints(ac.GeoId,
485  ac.PosId);
486  if (coincidents.find(geoIds[0]) != coincidents.end()) {
487  isClosed = true;
488  }
489  }
490  }
491  }
492 
493  if (isClosed) {
494  if (periodic) { // if periodic we do not need the last pole/knot
495  return true;
496  }
497  }
498  else {
499  setAngleSnapping(true, getLastPoint());
500  resetSeekSecond = true;
501  }
502 
503  // insert circle point for pole/knot, defer internal alignment constraining.
504  if (!addPos()) {
505  return false;
506  }
507 
508  // add auto constraints on pole/knot
509  auto& ac1 = sugConstraints[1];
510  generateAutoConstraintsOnElement(ac1, geoIds.back(), pointPos);
511  sugConstraintsBackup.push_back(std::move(ac1));
512  ac1.clear();
513 
514  return isClosed;
515  }
516  return true;
517  }

569  void undoLastPoint()
570  {
571  // can only delete last pole/knot if it exists
572  if (state() != SelectMode::SeekSecond) {
573  return;
574  }
575 
576  // if only first pole/knot exists it's equivalent to canceling current spline
577  if (geoIds.size() == 1) {
578  // this also exits b-spline creation if continuous mode is off
579  quit();
580  return;
581  }
582 
583  // reverse the steps of press/release button
584  try {
585  // already ensured that CurrentConstraint == EditCurve.size() > 1
586  const int delGeoId = geoIds.back();
587  const auto& constraints = sketchgui->getSketchObject()->Constraints.getValues();
588  for (int i = constraints.size() - 1; i >= 0; --i) {
589  if (delGeoId == constraints[i]->First || delGeoId == constraints[i]->Second
590  || delGeoId == constraints[i]->Third) {
591  Gui::cmdAppObjectArgs(sketchgui->getObject(), "delConstraint(%d)", i);
592  }
593  }
594 
595  // Remove pole/knot
596  Gui::cmdAppObjectArgs(sketchgui->getObject(), "delGeometry(%d)", delGeoId);
597 
598  sketchgui->getSketchObject()->solve();
599 
600  geoIds.pop_back();
601  points.pop_back();
602  multiplicities.pop_back();
603  distances.pop_back();
604 
605  updateDataAndDrawToPosition(prevCursorPosition);
606  }
607  catch (const Base::Exception&) {
608  Gui::NotifyError(sketchgui,
609  QT_TRANSLATE_NOOP("Notifications", "Error"),
610  QT_TRANSLATE_NOOP("Notifications", "Error deleting last pole/knot"));

614
