59 void FixedTimeAnimation::initialize()
60 {
61 #if (COIN_MAJOR_VERSION * 100 + COIN_MINOR_VERSION * 10 + COIN_MICRO_VERSION < 403)
62  navigation->findBoundingSphere();
63 #endif
64 
65  prevAngle = 0;
66  prevTranslation = SbVec3f(0, 0, 0);
67 
68  // Find an axis and angle to rotate from the camera orientation to the target orientation using post-multiplication
69  SbVec3f rotationAxisPost;
70  float angle;
71  SbRotation(navigation->getCamera()->orientation.getValue().inverse() * targetOrientation).getValue(rotationAxisPost, angle);
72  if (angle > M_PI) {
73  angle -= float(2 * M_PI);
74  }
75 
76  // Convert post-multiplication axis to a pre-multiplication axis
77  navigation->getCamera()->orientation.getValue().inverse().multVec(rotationAxisPost, rotationAxis);
78 
79  angularVelocity = angle / duration();
80  linearVelocity = targetTranslation / duration();
81 }

86 void FixedTimeAnimation::update(const QVariant& value)
87 {
88  SoCamera* camera = navigation->getCamera();
89  if (!camera) {
90  return;
91  }
92 
93  float angle = value.toFloat() * angularVelocity;
94  SbVec3f translation = value.toFloat() * linearVelocity;
95 
96  SbRotation rotation(rotationAxis, angle - prevAngle);
97 
98  camera->position = camera->position.getValue() - prevTranslation;
99  navigation->reorientCamera(camera, rotation, rotationCenter);
100  camera->position = camera->position.getValue() + translation;
101 
102  prevAngle = angle;
103  prevTranslation = translation;
104 }

109 void FixedTimeAnimation::onStop(bool finished)
110 {
111  if (finished) {
112  SoCamera* camera = navigation->getCamera();
113  if (!camera) {
114  return;
115  }
116 
117  // Set exact target orientation
118  camera->orientation = targetOrientation;
119  camera->position = camera->position.getValue() + targetTranslation - prevTranslation;
120  }
121 }
