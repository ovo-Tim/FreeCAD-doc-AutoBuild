175 void TaskHatch::apply(bool forceUpdate)

193 void TaskHatch::createHatch()
194 {
195 // Base::Console().Message("TH::createHatch()\n");
196  App::Document* doc = m_dvp->getDocument();
197 
198  // TODO: the structured label for Hatch (and GeomHatch) should be retired.
199  const std::string objectName("Hatch");
200  std::string FeatName = doc->getUniqueObjectName(objectName.c_str());
201 
202  Command::openCommand(QT_TRANSLATE_NOOP("Command", "Create Hatch"));
203 
204  Command::doCommand(Command::Doc, "App.activeDocument().addObject('TechDraw::DrawHatch', '%s')", FeatName.c_str());
205  Command::doCommand(Command::Doc, "App.activeDocument().%s.translateLabel('DrawHatch', 'Hatch', '%s')",
206  FeatName.c_str(), FeatName.c_str());
207 
208  m_hatch = static_cast<TechDraw::DrawHatch *>(doc->getObject(FeatName.c_str()));
209  m_hatch->Source.setValue(m_dvp, m_subs);
210 
211  auto filespec = ui->fcFile->fileName().toStdString();
212  filespec = DU::cleanFilespecBackslash(filespec);
213  Command::doCommand(Command::Doc, "App.activeDocument().%s.HatchPattern = '%s'",
214  FeatName.c_str(),
215  filespec.c_str());
216 
217  //view provider properties
218  Gui::ViewProvider* vp = Gui::Application::Instance->getDocument(doc)->getViewProvider(m_hatch);
219  m_vp = dynamic_cast<TechDrawGui::ViewProviderHatch*>(vp);
220  if (m_vp) {
221  App::Color ac;
222  ac.setValue<QColor>(ui->ccColor->color());
223  m_vp->HatchColor.setValue(ac);
224  m_vp->HatchScale.setValue(ui->sbScale->value().getValue());
225  m_vp->HatchRotation.setValue(ui->dsbRotation->value());
226  Base::Vector3d offset(ui->dsbOffsetX->value(), ui->dsbOffsetY->value(), 0.0);
227  m_vp->HatchOffset.setValue(offset);
228  } else {
229  Base::Console().Error("TaskHatch - Hatch has no ViewProvider\n");
230  }
231  Command::commitCommand();
232 }
