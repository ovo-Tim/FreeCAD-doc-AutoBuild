84 bool ViewProviderTextDocument::doubleClicked()
85 {
86  if (!activateView()) {
87  editorWidget = new QPlainTextEdit {};
88  editorWidget->setReadOnly(ReadOnly.getValue());
89  FontName.touch();
90  SyntaxHighlighter.touch();
91 
92  getMainWindow()->addWindow(

95  editorWidget, getMainWindow()});

100 void ViewProviderTextDocument::onChanged(const App::Property* prop)
101 {
102  if (editorWidget) {
103  if (prop == &ReadOnly) {
104  editorWidget->setReadOnly(ReadOnly.getValue());
105  }
106  else if (prop == &FontSize || prop == &FontName) {
107  QFont font(QString::fromLatin1(this->FontName.getValue()), (int)this->FontSize.getValue());
108  editorWidget->setFont(font);
109  }
110  else if (prop == &SyntaxHighlighter) {
111  long value = SyntaxHighlighter.getValue();
112  if (value == 1) {
113  auto pythonSyntax = new PythonSyntaxHighlighter(editorWidget);
114  pythonSyntax->setDocument(editorWidget->document());
115  }
116  else {
117  auto shl = editorWidget->findChild<QSyntaxHighlighter*>();
118  if (shl)
119  shl->deleteLater();
120  }
121  }
122  }
123  ViewProviderDocumentObject::onChanged(prop);
124 }
