57 class NameValidator(QValidator):
58  """Simple validator to exclude characters that are not valid in filenames."""
59 
60  invalid = '/\\?%*:|"<>'
61 
62  def validate(self, value: str, _: int):
63  """Check the value against the validator"""
64  for char in value:
65  if char in NameValidator.invalid:
66  return QValidator.Invalid
67  return QValidator.Acceptable
68 
69  def fixup(self, value: str) -> str:
70  """Remove invalid characters from value"""
71  result = ""
72  for char in value:
73  if char not in NameValidator.invalid:

95 class SemVerValidator(RegexValidatorWrapper):
96  """Implements the officially-recommended regex validator for Semantic version numbers."""
97 
98  # https://semver.org/#is-there-a-suggested-regular-expression-regex-to-check-a-semver-string
99  semver_re = RegexWrapper(
100  r"^(?P<major>0|[1-9]\d*)\.(?P<minor>0|[1-9]\d*)\.(?P<patch>0|[1-9]\d*)"
101  + r"(?:-(?P<prerelease>(?:0|[1-9]\d*|\d*[a-zA-Z-][0-9a-zA-Z-]*)"
102  + r"(?:\.(?:0|[1-9]\d*|\d*[a-zA-Z-][0-9a-zA-Z-]*))*))?"
103  + r"(?:\+(?P<buildmetadata>[0-9a-zA-Z-]+(?:\.[0-9a-zA-Z-]+)*))?$"
104  )
105 
106  def __init__(self):
107  super().__init__()
108  if hasattr(self, "setRegularExpression"):
109  self.setRegularExpression(SemVerValidator.semver_re)
110  else:
111  self.setRegExp(SemVerValidator.semver_re)

143 class VersionValidator(QValidator):
144  """Implements the officially-recommended regex validator for Semantic version numbers, and a
145  decent approximation of the same thing for CalVer-style version numbers."""
146 
147  def __init__(self):
148  super().__init__()
149  self.semver = SemVerValidator()
150  self.calver = CalVerValidator()
151 
152  def validate(self, value: str, position: int):
153  """Called for validation, returns a tuple of the validation state, the value, and the
154  position."""
155  semver_result = self.semver.validate(value, position)
156  calver_result = self.calver.validate(value, position)
157 
158  if semver_result[0] == QValidator.Acceptable:
159  return semver_result
160  if calver_result[0] == QValidator.Acceptable:
161  return calver_result
162  if semver_result[0] == QValidator.Intermediate:
163  return semver_result
164  if calver_result[0] == QValidator.Intermediate:
165  return calver_result
166  return QValidator.Invalid, value, position
