75 App::DocumentObjectExecReturn* Feature::recompute()
76 {
77  setMaterialToBodyMaterial();
78 
79  SuppressedShape.setValue(TopoShape());
80 
81  if (!Suppressed.getValue()) {
82  return Part::Feature::recompute();
83  }
84 
85  bool failed = false;
86  try {
87  std::unique_ptr<App::DocumentObjectExecReturn> ret(Part::Feature::recompute());
88  if (ret) {
89  throw Base::RuntimeError(ret->Why);
90  }
91  }
92  catch (Base::AbortException&) {
93  throw;
94  }
95  catch (Base::Exception& e) {
96  failed = true;
97  e.ReportException();
98  FC_ERR("Failed to recompute suppressed feature " << getFullName());
99  }
100 
101  if (!failed) {
102  updateSuppressedShape();
103  }
104  else {
105  Shape.setValue(getBaseTopoShape(true));
106  }
107  return App::DocumentObject::StdReturn;
108 }

122 void Feature::updateSuppressedShape()
123 {
124  auto baseShape = getBaseTopoShape(true);
125  TopoShape res(getID());
126  TopoShape shape = Shape.getShape();
127  shape.setPlacement(Base::Placement());
128  std::vector<TopoShape> generated;
129  if(!shape.isNull()) {
130  unsigned count = shape.countSubShapes(TopAbs_FACE);
131  for(unsigned i=1; i<=count; ++i) {
132  Data::MappedName mapped = shape.getMappedName(
133  Data::IndexedName::fromConst("Face", i));
134  if(mapped && shape.isElementGenerated(mapped))
135  generated.push_back(shape.getSubTopoShape(TopAbs_FACE, i));
136  }
137  }
138  if(!generated.empty()) {
139  res.makeElementCompound(generated);
140  res.setPlacement(Placement.getValue());
141  }
142  Shape.setValue(baseShape);
143  SuppressedShape.setValue(res);
144 }

153 TopoShape Feature::getSolid(const TopoShape& shape) const
154 {
155  if (shape.isNull()) {
156  throw Part::NullShapeException("Null shape");
157  }
158 
159  // If single solid rule is not enforced we simply return the shape as is
160  if (singleSolidRuleMode() != Feature::SingleSolidRuleMode::Enforced) {
161  return shape;
162  }
163 
164  int count = shape.countSubShapes(TopAbs_SOLID);
165  if (count) {
166  auto res = shape.getSubTopoShape(TopAbs_SOLID, 1);
167  res.fixSolidOrientation();
168  return res;
169  }
170 
171  return shape;
172 }

174 void Feature::onChanged(const App::Property *prop)
175 {
176  if (!this->isRestoring()
177  && this->getDocument()
178  && !this->getDocument()->isPerformingTransaction()) {
179  if (prop == &Visibility || prop == &BaseFeature) {
180  auto body = Body::findBodyOf(this);
181  if (body) {
182  if (prop == &BaseFeature && BaseFeature.getValue()) {
183  int idx = -1;
184  body->Group.find(this->getNameInDocument(), &idx);
185  int baseidx = -1;
186  body->Group.find(BaseFeature.getValue()->getNameInDocument(), &idx);
187  if (idx >= 0 && baseidx >= 0 && baseidx+1 != idx)
188  body->insertObject(BaseFeature.getValue(), this);
189  }
190  }
191  } else if (prop == &ShapeMaterial) {
192  auto body = Body::findBodyOf(this);
193  if (body) {
194  if (body->ShapeMaterial.getValue().getUUID()
195  != ShapeMaterial.getValue().getUUID()) {
196  body->ShapeMaterial.setValue(ShapeMaterial.getValue());
197  }
198  }
199  }
200  }
201  Part::Feature::onChanged(prop);
202 }

299  throw Base::ValueError("Base feature's shape is not a solid");

304 Part::TopoShape Feature::getBaseTopoShape(bool silent) const
305 {
306  Part::TopoShape result;
307 
308  const Part::Feature* BaseObject = getBaseObject(silent);
309  if (!BaseObject) {
310  return result;
311  }
312 
313  if (BaseObject != BaseFeature.getValue()) {
314  auto body = getFeatureBody();
315  if (!body) {
316  if (silent) {
317  return result;
318  }
319  throw Base::RuntimeError("Missing container body");
320  }
321  if (BaseObject->isDerivedFrom(PartDesign::ShapeBinder::getClassTypeId())
322  || BaseObject->isDerivedFrom(PartDesign::SubShapeBinder::getClassTypeId())) {
323  if (silent) {
324  return result;
325  }
326  throw Base::ValueError("Base shape of shape binder cannot be used");
327  }
328  }
329 
330  result = BaseObject->Shape.getShape();
331  if (!silent) {
332  if (result.isNull()) {
333  throw Base::ValueError("Base feature's TopoShape is invalid");
334  }
335  if (!result.hasSubShape(TopAbs_SOLID)) {
336  throw Base::ValueError("Base feature's shape is not a solid");
337  }
338  }
339  return result;
340 }

358 {

408 App::DocumentObject *Feature::getSubObject(const char *subname,
409  PyObject **pyObj, Base::Matrix4D *pmat, bool transform, int depth) const
410 {
411  if (subname && subname != Data::findElementName(subname)) {
412  const char * dot = strchr(subname,'.');
413  if (dot) {
414  auto body = PartDesign::Body::findBodyOf(this);
415  if (body) {
416  auto feat = body->Group.findUsingMap(std::string(subname, dot));
417  if (feat) {
418  Base::Matrix4D _mat;
419  if (!transform) {
420  // Normally the parent object is supposed to transform
421  // the sub-object using its own placement. So, if no
422  // transform is requested, (i.e. no parent
423  // transformation), we just need to NOT apply the
424  // transformation.
425  //
426  // But PartDesign features (including sketch) are
427  // supposed to be contained inside a body. It makes
428  // little sense to transform its sub-object. So if 'no
429  // transform' is requested, we need to actively apply
430  // an inverse transform.
431  _mat = Placement.getValue().inverse().toMatrix();
432  if (pmat)
433  *pmat *= _mat;
434  else
435  pmat = &_mat;
436  }
437  return feat->getSubObject(dot+1, pyObj, pmat, true, depth+1);
438  }
439  }
440  }
441  }
442  return Part::Feature::getSubObject(subname, pyObj, pmat, transform, depth);
443 }
