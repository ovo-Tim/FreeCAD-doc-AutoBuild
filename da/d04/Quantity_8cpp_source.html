84 Quantity::Quantity(double value, const std::string& unit)
85  : myValue {0.0}
86 {
87  if (unit.empty()) {
88  this->myValue = value;
89  this->myUnit = Unit();
90  return;
91  }
92 
93  try {
94  auto tmpQty = parse(unit);
95  this->myUnit = tmpQty.getUnit();
96  this->myValue = value * tmpQty.getValue();
97  }
98  catch (const Base::ParserError&) {
99  this->myValue = 0.0;
100  this->myUnit = Unit();
101  }
102 }

457 /* helper function for tuning number strings with groups in a locale agnostic way... */
458 // NOLINTBEGIN
459 double num_change(char* yytext, char dez_delim, char grp_delim)
460 {
461  double ret_val {};
462  const int num = 40;
463  std::array<char, num> temp {};
464  int iter = 0;
465  for (char* ch = yytext; *ch != '\0'; ch++) {
466  // skip group delimiter
467  if (*ch == grp_delim) {
468  continue;
469  }
470  // check for a dez delimiter other then dot
471  if (*ch == dez_delim && dez_delim != '.') {
472  temp[iter++] = '.';
473  }
474  else {
475  temp[iter++] = *ch;
476  }
477  // check buffer overflow
478  if (iter >= num) {
479  return 0.0;
480  }
481  }
482 
483  temp[iter] = '\0';
484 
485  ret_val = atof(temp.data());
486  return ret_val;
487 }
