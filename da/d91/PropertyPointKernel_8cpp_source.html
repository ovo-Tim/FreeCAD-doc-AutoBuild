167 void PropertyPointKernel::removeIndices(const std::vector<unsigned long>& uIndices)
168 {
169  // We need a sorted array
170  std::vector<unsigned long> uSortedInds = uIndices;
171  std::sort(uSortedInds.begin(), uSortedInds.end());
172 
173  assert(uSortedInds.size() <= _cPoints->size());
174  if (uSortedInds.size() > _cPoints->size()) {
175  return;
176  }
177 
178  PointKernel kernel;
179  kernel.setTransform(_cPoints->getTransform());
180  kernel.reserve(_cPoints->size() - uSortedInds.size());
181 
182  std::vector<unsigned long>::iterator pos = uSortedInds.begin();
183  unsigned long index = 0;
184  for (PointKernel::const_iterator it = _cPoints->begin(); it != _cPoints->end(); ++it, ++index) {
185  if (pos == uSortedInds.end()) {
186  kernel.push_back(*it);
187  }
188  else if (index != *pos) {
189  kernel.push_back(*it);
190  }
191  else {
192  ++pos;
193  }
194  }
195 
196  setValue(kernel);
197 }
