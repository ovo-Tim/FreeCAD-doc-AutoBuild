32 // minimizes ( 0.5 * x^T * H * x + g^T * x ) under the condition ( A*x + c = 0 )
33 // it returns the solution in x, the row-space of A in Y, and the null space of A in Z
34 int qp_eq(MatrixXd& H, VectorXd& g, MatrixXd& A, VectorXd& c, VectorXd& x, MatrixXd& Y, MatrixXd& Z)
35 {
36  FullPivHouseholderQR<MatrixXd> qrAT(A.transpose());
37  MatrixXd Q = qrAT.matrixQ();
38 
39  size_t params_num = qrAT.rows();
40  size_t constr_num = qrAT.cols();
41  size_t rank = qrAT.rank();
42 
43  if (rank != constr_num || constr_num > params_num) {
44  return -1;
45  }
46 
47  // A^T = Q*R*P^T = Q1*R1*P^T
48  // Q = [Q1,Q2], R=[R1;0]
49  // Y = Q1 * inv(R^T) * P^T
50  // Z = Q2
51  Y = qrAT.matrixQR()
52  .topRows(constr_num)
53  .triangularView<Upper>()
54  .transpose()
55  .solve<OnTheRight>(Q.leftCols(rank))
56  * qrAT.colsPermutation().transpose();
57  if (params_num == rank) {
58  x = -Y * c;
59  }
60  else {
61  Z = Q.rightCols(params_num - rank);
62 
63  MatrixXd ZTHZ = Z.transpose() * H * Z;
64  VectorXd rhs = Z.transpose() * (H * Y * c - g);
65 
66  VectorXd y = ZTHZ.colPivHouseholderQr().solve(rhs);
67 
68  x = -Y * c + Z * y;
69  }
70 
71  return 0;
72 }
