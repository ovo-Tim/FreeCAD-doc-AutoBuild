104 App::DocumentObjectExecReturn* Boolean::execute()
105 {
106  try {
107 #if defined(__GNUC__) && defined(FC_OS_LINUX)
108  Base::SignalException se;
109 #endif
110  auto base = Base.getValue();
111  auto tool = Tool.getValue();
112 
113  if (!base || !tool) {
114  return new App::DocumentObjectExecReturn("Linked object is not a Part object");
115  }
116  std::vector<TopoShape> shapes;
117  shapes.reserve(2);
118  // Now, let's get the TopoDS_Shape
119  shapes.push_back(Feature::getTopoShape(Base.getValue()));
120  auto BaseShape = shapes[0].getShape();
121  if (BaseShape.IsNull()) {
122  throw NullShapeException("Base shape is null");
123  }
124  shapes.push_back(Feature::getTopoShape(Tool.getValue()));
125  auto ToolShape = shapes[1].getShape();
126  if (ToolShape.IsNull()) {
127  throw NullShapeException("Tool shape is null");
128  }
129 
130  std::unique_ptr<BRepAlgoAPI_BooleanOperation> mkBool(makeOperation(BaseShape, ToolShape));
131  if (!mkBool->IsDone()) {
132  std::stringstream error;
133  error << "Boolean operation failed";
134  if (BaseShape.ShapeType() != TopAbs_SOLID) {
135  error << std::endl << base->Label.getValue() << " is not a solid";
136  }
137  if (ToolShape.ShapeType() != TopAbs_SOLID) {
138  error << std::endl << tool->Label.getValue() << " is not a solid";
139  }
140  return new App::DocumentObjectExecReturn(error.str());
141  }
142  TopoDS_Shape resShape = mkBool->Shape();
143  if (resShape.IsNull()) {
144  return new App::DocumentObjectExecReturn("Resulting shape is null");
145  }
146 
147  throwIfInvalidIfCheckModel(resShape);
148 
149  TopoShape res(0);
150  res.makeElementShape(*mkBool, shapes, opCode());
151  if (this->Refine.getValue()) {
152  res = res.makeElementRefine();
153  }
154  this->Shape.setValue(res);
155  copyMaterial(base);
156  return Part::Feature::execute();
157  }
158  catch (const Base::Exception& e) {
159  return new App::DocumentObjectExecReturn(e.what());
160  }
161  catch (...) {
162  return new App::DocumentObjectExecReturn(
163  "A fatal error occurred when running boolean operation");
164  }
165 }
