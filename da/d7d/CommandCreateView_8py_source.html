321  def applyStep(self, move, com=App.Vector(), size=100):
322  if not UtilsAssembly.isRefValid(move.References, 1):
323  return
324 
325  positions = []
326  if move.MoveType == "Radial":
327  distance = move.MovementTransform.Base.Length
328  factor = 4 * distance / size
329 
330  subs = move.References[1]
331  for sub in subs:
332  ref = [move.References[0], [sub]]
333  obj = UtilsAssembly.getObject(ref)
334  if not obj:
335  continue
336 
337  if move.ViewObject:
338  startPos = UtilsAssembly.getCenterOfBoundingBox([obj], [ref])
339 
340  if move.MoveType == "Radial":
341  objCom, objSize = UtilsAssembly.getComAndSize(obj)
342  init_vec = objCom - com
343  obj.Placement.Base = obj.Placement.Base + init_vec * factor
344  else:
345  obj.Placement = move.MovementTransform * obj.Placement
346 
347  if move.ViewObject:
348  endPos = UtilsAssembly.getCenterOfBoundingBox([obj], [ref])
349  positions.append([startPos, endPos])
350 
351  if move.ViewObject:
352  move.ViewObject.Proxy.redrawLines(move, positions)
353 
354  return positions
