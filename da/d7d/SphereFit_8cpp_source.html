106 float SphereFit::GetStdDeviation() const
107 {
108  // Mean: M=(1/N)*SUM Xi
109  // Variance: VAR=(N/N-1)*[(1/N)*SUM(Xi^2)-M^2]
110  // Standard deviation: SD=SQRT(VAR)
111  if (!_bIsFitted) {
112  return FLOAT_MAX;
113  }
114 
115  double sumXi = 0.0, sumXi2 = 0.0, dist = 0.0;
116  for (auto it : _vPoints) {
117  dist = GetDistanceToSphere(it);
118  sumXi += dist;
119  sumXi2 += (dist * dist);
120  }
121 
122  double N = static_cast<double>(CountPoints());
123  double mean = sumXi / N;
124  return sqrt((N / (N - 1.0)) * (sumXi2 / N - mean * mean));
125 }

127 void SphereFit::ProjectToSphere()
128 {
129  for (auto& cPnt : _vPoints) {
130  // Compute unit vector from sphere centre to point.
131  // Because this vector is orthogonal to the sphere's surface at the
132  // intersection point we can easily compute the projection point on the
133  // closest surface point using the radius of the sphere
134  Base::Vector3d diff((double)cPnt.x - _vCenter.x,
135  (double)cPnt.y - _vCenter.y,
136  (double)cPnt.z - _vCenter.z);
137  double length = diff.Length();
138  if (length == 0.0) {
139  // Point is exactly at the sphere center, so it can be projected in any direction onto
140  // the sphere! So here just project in +Z direction
141  cPnt.z += (float)_dRadius;
142  }
143  else {
144  diff /= length; // normalizing the vector
145  Base::Vector3d proj = _vCenter + diff * _dRadius;
146  cPnt.x = (float)proj.x;
147  cPnt.y = (float)proj.y;
148  cPnt.z = (float)proj.z;
149  }
150  }
151 }

153 // Compute approximations for the parameters using all points:
154 // Set centre to centre of gravity of points and radius to the average
155 // distance from the centre of gravity to the points.
156 void SphereFit::ComputeApproximations()
157 {
158  _bIsFitted = false;
159  _fLastResult = FLOAT_MAX;
160  _numIter = 0;
161  _vCenter.Set(0.0, 0.0, 0.0);
162  _dRadius = 0.0;
163  if (!_vPoints.empty()) {
164  std::list<Base::Vector3f>::const_iterator cIt;
165  for (cIt = _vPoints.begin(); cIt != _vPoints.end(); ++cIt) {
166  _vCenter.x += cIt->x;
167  _vCenter.y += cIt->y;
168  _vCenter.z += cIt->z;
169  }
170  _vCenter /= (double)_vPoints.size();
171 
172  for (cIt = _vPoints.begin(); cIt != _vPoints.end(); ++cIt) {
173  Base::Vector3d diff((double)cIt->x - _vCenter.x,
174  (double)cIt->y - _vCenter.y,
175  (double)cIt->z - _vCenter.z);
176  _dRadius += diff.Length();
177  }
178  _dRadius /= (double)_vPoints.size();
179  }
180 }

182 float SphereFit::Fit()
183 {
184  _bIsFitted = false;
185  _fLastResult = FLOAT_MAX;
186  _numIter = 0;
187 
188  // A minimum of 4 surface points is needed to define a sphere
189  if (CountPoints() < 4) {
190  return FLOAT_MAX;
191  }
192 
193  // If approximations have not been set/computed then compute some now
194  if (_dRadius == 0.0) {
195  ComputeApproximations();
196  }
197 
198  // Initialise some matrices and vectors
199  std::vector<Base::Vector3d> residuals(CountPoints(), Base::Vector3d(0.0, 0.0, 0.0));
200  Matrix4x4 atpa;
201  Eigen::VectorXd atpl(4);
202 
203  // Iteration loop...
204  double sigma0 {};
205  bool cont = true;
206  while (cont && (_numIter < _maxIter)) {
207  ++_numIter;
208 
209  // Set up the quasi parametric normal equations
210  setupNormalEquationMatrices(residuals, atpa, atpl);
211 
212  // Solve the equations for the unknown corrections
213  Eigen::LLT<Matrix4x4> llt(atpa);
214  if (llt.info() != Eigen::Success) {
215  return FLOAT_MAX;
216  }
217  Eigen::VectorXd x = llt.solve(atpl);
218 
219  // Check parameter convergence (order of parameters: X,Y,Z,R)
220  cont = false;
221  if ((fabs(x(0)) > _posConvLimit) || (fabs(x(1)) > _posConvLimit)
222  || (fabs(x(2)) > _posConvLimit) || (fabs(x(3)) > _posConvLimit)) {
223  cont = true;
224  }
225 
226  // Before updating the unknowns, compute the residuals and sigma0 and check the residual
227  // convergence
228  bool vConverged {};
229  if (!computeResiduals(x, residuals, sigma0, _vConvLimit, vConverged)) {
230  return FLOAT_MAX;
231  }
232  if (!vConverged) {
233  cont = true;
234  }
235 
236  // Update the parameters (order of parameters: X,Y,Z,R)
237  _vCenter.x += x(0);
238  _vCenter.y += x(1);
239  _vCenter.z += x(2);
240  _dRadius += x(3);
241  }
242 
243  // Check for convergence
244  if (cont) {
245  return FLOAT_MAX;
246  }
247 
248  _bIsFitted = true;
249  _fLastResult = sigma0;
250 
251  return _fLastResult;
252 }

255 // atpa ... 4x4 normal matrix
256 // atpl ... 4x1 matrix (right-hand side of equation)
257 void SphereFit::setupNormalEquationMatrices(const std::vector<Base::Vector3d>& residuals,
258  Matrix4x4& atpa,
259  Eigen::VectorXd& atpl) const
260 {
261  // Zero matrices
262  atpa.setZero();
263  atpl.setZero();
264 
265  // For each point, setup the observation equation coefficients and add their
266  // contribution into the normal equation matrices
267  double a[4] {}, b[3] {};
268  double f0 {}, qw {};
269  std::vector<Base::Vector3d>::const_iterator vIt = residuals.begin();
270  for (auto cIt = _vPoints.begin(); cIt != _vPoints.end(); ++cIt, ++vIt) {
271  // if (using this point) { // currently all given points are used (could modify this if
272  // eliminating outliers, etc....
273  setupObservation(*cIt, *vIt, a, f0, qw, b);
274  addObservationU(a, f0, qw, atpa, atpl);
275  // }
276  }
277  setLowerPart(atpa);
278 }

280 // Sets up contributions of given observation to the quasi parametric
281 // normal equation matrices. Assumes uncorrelated coordinates.
282 // point ... point
283 // residual ... residual for this point computed from previous iteration (zero for first iteration)
284 // a[4] ... parameter partials (order of parameters: X,Y,Z,R)
285 // f0 ... reference to f0 term
286 // qw ... reference to quasi weight (here we are assuming equal unit weights for each observed
287 // point coordinate) b[3] ... observation partials
288 void SphereFit::setupObservation(const Base::Vector3f& point,
289  const Base::Vector3d& residual,
290  double a[4],
291  double& f0,
292  double& qw,
293  double b[3]) const
294 {
295  // This adjustment requires an update of the observation approximations
296  // because the residuals do not have a linear relationship.
297  // New estimates for the observations:
298  double xEstimate = (double)point.x + residual.x;
299  double yEstimate = (double)point.y + residual.y;
300  double zEstimate = (double)point.z + residual.z;
301 
302  // partials of the observations
303  double dx = xEstimate - _vCenter.x;
304  double dy = yEstimate - _vCenter.y;
305  double dz = zEstimate - _vCenter.z;
306  b[0] = 2.0 * dx;
307  b[1] = 2.0 * dy;
308  b[2] = 2.0 * dz;
309 
310  // partials of the parameters
311  a[0] = -b[0];
312  a[1] = -b[1];
313  a[2] = -b[2];
314  a[3] = -2.0 * _dRadius;
315 
316  // free term
317  f0 = _dRadius * _dRadius - dx * dx - dy * dy - dz * dz + b[0] * residual.x + b[1] * residual.y
318  + b[2] * residual.z;
319 
320  // quasi weight (using equal weights for sphere point coordinate observations)
321  // w[0] = 1.0;
322  // w[1] = 1.0;
323  // w[2] = 1.0;
324  // qw = 1.0 / (b[0] * b[0] / w[0] + b[1] * b[1] / w[1] + b[2] * b[2] / w[2]);
325  qw = 1.0 / (b[0] * b[0] + b[1] * b[1] + b[2] * b[2]);
326 }
327 
328 // Computes contribution of the given observation equation on the normal equation matrices
329 // Call this for each observation (point)
330 // Here we only add the contribution to the upper part of the normal matrix
331 // and then after all observations have been added we need to set the lower part
332 // (which is symmetrical to the upper part)
333 // a[4] ... parameter partials
334 // li ... free term (f0)
335 // pi ... weight of observation (= quasi weight qw for this solution)
336 // atpa ... 4x4 normal equation matrix
337 // atpl ... 4x1 matrix/vector (right-hand side of equations)
338 void SphereFit::addObservationU(double a[4],
339  double li,
340  double pi,
341  Matrix4x4& atpa,
342  Eigen::VectorXd& atpl) const
343 {
344  for (int i = 0; i < 4; ++i) {
345  double aipi = a[i] * pi;
346  for (int j = i; j < 4; ++j) {
347  atpa(i, j) += aipi * a[j];
348  // atpa(j, i) = atpa(i, j); // it's a symmetrical matrix, we'll set this later after all
349  // observations processed
350  }
351  atpl(i) += aipi * li;
352  }
353 }

367 bool SphereFit::computeResiduals(const Eigen::VectorXd& x,
368  std::vector<Base::Vector3d>& residuals,
369  double& sigma0,
370  double vConvLimit,
371  bool& vConverged) const
372 {
373  vConverged = true;
374  int nPtsUsed = 0;
375  sigma0 = 0.0;
376  double a[4] {}, b[3] {};
377  double f0 {}, qw {};
378  // double maxdVx = 0.0;
379  // double maxdVy = 0.0;
380  // double maxdVz = 0.0;
381  // double rmsVv = 0.0;
382  std::vector<Base::Vector3d>::iterator vIt = residuals.begin();
383  for (auto cIt = _vPoints.begin(); cIt != _vPoints.end(); ++cIt, ++vIt) {
384  // if (using this point) { // currently all given points are used (could modify this if
385  // eliminating outliers, etc....
386  ++nPtsUsed;
387  Base::Vector3d& v = *vIt;
388  setupObservation(*cIt, v, a, f0, qw, b);
389  double qv = -f0;
390  for (int i = 0; i < 4; ++i) {
391  qv += a[i] * x(i);
392  }
393 
394  // We are using equal weights for sphere point coordinate observations (see
395  // setupObservation) i.e. w[0] = w[1] = w[2] = 1.0;
396  // double vx = -qw * qv * b[0] / w[0];
397  // double vy = -qw * qv * b[1] / w[1];
398  // double vz = -qw * qv * b[2] / w[2];
399  double vx = -qw * qv * b[0];
400  double vy = -qw * qv * b[1];
401  double vz = -qw * qv * b[2];
402  double dVx = fabs(vx - v.x);
403  double dVy = fabs(vy - v.y);
404  double dVz = fabs(vz - v.z);
405  v.x = vx;
406  v.y = vy;
407  v.z = vz;
408 
409  // double vv = v.x * v.x + v.y * v.y + v.z * v.z;
410  // rmsVv += vv * vv;
411 
412  // sigma0 += v.x * w[0] * v.x + v.y * w[1] * v.y + v.z * w[2] * v.z;
413  sigma0 += v.x * v.x + v.y * v.y + v.z * v.z;
414 
415  if ((dVx > vConvLimit) || (dVy > vConvLimit) || (dVz > vConvLimit)) {
416  vConverged = false;
417  }
418 
419  // if (dVx > maxdVx)
420  // maxdVx = dVx;
421  // if (dVy > maxdVy)
422  // maxdVy = dVy;
423  // if (dVz > maxdVz)
424  // maxdVz = dVz;
425  }
426 
427  // Compute degrees of freedom and sigma0
428  if (nPtsUsed < 4) // A minimum of 4 surface points is needed to define a sphere
429  {
430  sigma0 = 0.0;
431  return false;
432  }
433  int df = nPtsUsed - 4;
434  if (df == 0) {
435  sigma0 = 0.0;
436  }
437  else {
438  sigma0 = sqrt(sigma0 / (double)df);
439  }
