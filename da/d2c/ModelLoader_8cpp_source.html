179 void ModelLoader::dereference(std::shared_ptr<ModelEntry> model,
180  std::map<std::pair<QString, QString>, QString>* inheritances)
181 {
182  // Avoid recursion
183  if (model->getDereferenced()) {
184  return;
185  }
186 
187  auto yamlModel = model->getModel();
188  auto base = model->getBase().toStdString();
189  if (yamlModel[base]["Inherits"]) {
190  auto inherits = yamlModel[base]["Inherits"];
191  for (auto it = inherits.begin(); it != inherits.end(); it++) {
192  QString nodeName = QString::fromStdString((*it)["UUID"].as<std::string>());
193 
194  // This requires that all models have already been loaded undereferenced
195  try {
196  std::shared_ptr<ModelEntry> child = (*_modelEntryMap)[nodeName];
197  dereference(model->getUUID(), model, child, inheritances);
198  }
199  catch (const std::out_of_range&) {
200  Base::Console().Log("Unable to find '%s' in model map\n",
201  nodeName.toStdString().c_str());
202  }
203  }
204  }
205 
206  model->markDereferenced();
207 }

219 void ModelLoader::addToTree(std::shared_ptr<ModelEntry> model,
220  std::map<std::pair<QString, QString>, QString>* inheritances)
221 {
222  std::set<QString> exclude;
223  exclude.insert(QString::fromStdString("Name"));
224  exclude.insert(QString::fromStdString("UUID"));
225  exclude.insert(QString::fromStdString("URL"));
226  exclude.insert(QString::fromStdString("Description"));
227  exclude.insert(QString::fromStdString("DOI"));
228  exclude.insert(QString::fromStdString("Inherits"));
229 
230  auto yamlModel = model->getModel();
231  auto library = model->getLibrary();
232  auto base = model->getBase().toStdString();
233  auto name = model->getName();
234  auto directory = model->getDirectory();
235  auto uuid = model->getUUID();
236 
237  QString description = yamlValue(yamlModel[base], "Description", "");
238  QString url = yamlValue(yamlModel[base], "URL", "");
239  QString doi = yamlValue(yamlModel[base], "DOI", "");
240 
241  Model::ModelType type =
242  (base == "Model") ? Model::ModelType_Physical : Model::ModelType_Appearance;
243 
244  Model finalModel(library, type, name, directory, uuid, description, url, doi);
245 
246  // Add inheritance list
247  if (yamlModel[base]["Inherits"]) {
248  auto inherits = yamlModel[base]["Inherits"];
249  for (auto it = inherits.begin(); it != inherits.end(); it++) {
250  QString nodeName = QString::fromStdString((*it)["UUID"].as<std::string>());
251 
252  finalModel.addInheritance(nodeName);
253  }
254  }
255 
256  // Add property list
257  auto yamlProperties = yamlModel[base];
258  for (auto it = yamlProperties.begin(); it != yamlProperties.end(); it++) {
259  std::string propName = it->first.as<std::string>();
260  if (exclude.count(QString::fromStdString(propName)) == 0) {
261  // showYaml(it->second);
262  auto yamlProp = yamlProperties[propName];
263  auto propDisplayName = yamlValue(yamlProp, "DisplayName", "");
264  auto propType = yamlValue(yamlProp, "Type", "");
265  auto propUnits = yamlValue(yamlProp, "Units", "");
266  auto propURL = yamlValue(yamlProp, "URL", "");
267  auto propDescription = yamlValue(yamlProp, "Description", "");
268  // auto inherits = yamlValue(yamlProp, "Inherits", "");
269 
270  ModelProperty property(QString::fromStdString(propName),
271  propDisplayName,
272  propType,
273  propUnits,
274  propURL,
275  propDescription);
276 
277  if (propType == QString::fromStdString("2DArray")
278  || propType == QString::fromStdString("3DArray")) {
279  // Base::Console().Log("Reading columns\n");
280  // Read the columns
281  auto cols = yamlProp["Columns"];
282  for (const auto& col : cols) {
283  std::string colName = col.first.as<std::string>();
284  // Base::Console().Log("\tColumns '%s'\n", colName.c_str());
285 
286  auto colProp = cols[colName];
287  auto colPropDisplayName = yamlValue(colProp, "DisplayName", "");
288  auto colPropType = yamlValue(colProp, "Type", "");
289  auto colPropUnits = yamlValue(colProp, "Units", "");
290  auto colPropURL = yamlValue(colProp, "URL", "");
291  auto colPropDescription = yamlValue(colProp, "Description", "");
292  ModelProperty colProperty(QString::fromStdString(colName),
293  colPropDisplayName,
294  colPropType,
295  colPropUnits,
296  colPropURL,
297  colPropDescription);
298 
299  property.addColumn(colProperty);
300  }
301  }
302 
303  auto key = std::pair<QString, QString>(uuid, QString::fromStdString(propName));
304  if (inheritances->count(key) > 0) {
305  property.setInheritance((*inheritances)[key]);
306  }
307 
308  finalModel.addProperty(property);
309  }
310  }
311 
312  (*_modelMap)[uuid] = library->addModel(finalModel, directory);
313 }

315 void ModelLoader::loadLibrary(std::shared_ptr<ModelLibrary> library)
316 {
317  if (_modelEntryMap == nullptr) {
318  _modelEntryMap = std::make_unique<std::map<QString, std::shared_ptr<ModelEntry>>>();
319  }
320 
321  QDirIterator it(library->getDirectory(), QDirIterator::Subdirectories);
322  while (it.hasNext()) {
323  auto pathname = it.next();
324  QFileInfo file(pathname);
325  if (file.isFile()) {
326  if (file.suffix().toStdString() == "yml") {
327  try {
328  auto model = getModelFromPath(library, file.canonicalFilePath());
329  (*_modelEntryMap)[model->getUUID()] = model;
330  // showYaml(model->getModel());
331  }
332  catch (InvalidModel const&) {
333  Base::Console().Log("Invalid model '%s'\n", pathname.toStdString().c_str());
334  }
335  }
336  }
337  }
338 
339  std::map<std::pair<QString, QString>, QString> inheritances;
340  for (auto it = _modelEntryMap->begin(); it != _modelEntryMap->end(); it++) {
341  dereference(it->second, &inheritances);
342  }
343 
344  for (auto it = _modelEntryMap->begin(); it != _modelEntryMap->end(); it++) {
345  addToTree(it->second, &inheritances);
346  }
347 }

366  bool useMatFromCustomDir = param->GetBool("UseMaterialsFromCustomDir", true);
367

380  "User parameter:BaseApp/Preferences/Mod/Material/Resources/Modules");

390  auto libData = std::make_shared<ModelLibrary>(moduleName, modelDir, modelIcon);
391  _libraryList->push_back(libData);

395  }

404  QString::fromStdString("User"),

419  resourceDir,

423  }
