54 ETManifoldMesh::TPtr ETManifoldMesh::InsertTriangle (int iV0, int iV1,
55  int iV2)
56 {
57  TriangleKey kTKey(iV0,iV1,iV2);
58  TMapIterator pkTIter = m_kTMap.find(kTKey);
59  if (pkTIter != m_kTMap.end())
60  {
61  // triangle already exists
62  return nullptr;
63  }
64 
65  // add new triangle
66  TPtr pkTriangle = m_oTCreator(iV0,iV1,iV2);
67  m_kTMap[kTKey] = pkTriangle;
68 
69  // add edges to mesh
70  for (int i0 = 2, i1 = 0; i1 < 3; i0 = i1++)
71  {
72  EdgeKey kEKey(pkTriangle->V[i0],pkTriangle->V[i1]);
73  EPtr pkEdge;
74  EMapIterator pkEIter = m_kEMap.find(kEKey);
75  if (pkEIter == m_kEMap.end())
76  {
77  // first time edge encountered
78  pkEdge = m_oECreator(pkTriangle->V[i0],pkTriangle->V[i1]);
79  m_kEMap[kEKey] = pkEdge;
80 
81  // update edge and triangle
82  pkEdge->T[0] = pkTriangle;
83  pkTriangle->E[i0] = pkEdge;
84  }
85  else
86  {
87  // second time edge encountered
88  pkEdge = pkEIter->second;
89  assert(pkEdge);
90 
91  // update edge
92  if (pkEdge->T[1])
93  {
94  assert(false); // mesh must be manifold
95  return nullptr;
96  }
97  pkEdge->T[1] = pkTriangle;
98 
99  // update adjacent triangles
100  TPtr pkAdjacent = pkEdge->T[0];
101  assert(pkAdjacent);
102  for (int i = 0; i < 3; i++)
103  {
104  if (pkAdjacent->E[i] == pkEdge)
105  {
106  pkAdjacent->T[i] = pkTriangle;
107  break;
108  }
109  }
110 
111  // update triangle
112  pkTriangle->E[i0] = pkEdge;
113  pkTriangle->T[i0] = pkAdjacent;
114  }
115  }
116 
117  return pkTriangle;
118 }

120 bool ETManifoldMesh::RemoveTriangle (int iV0, int iV1, int iV2)
121 {
122  TriangleKey kTKey(iV0,iV1,iV2);
123  TMapIterator pkTIter = m_kTMap.find(kTKey);
124  if (pkTIter == m_kTMap.end())
125  {
126  // triangle does not exist
127  return false;
128  }
129 
130  TPtr pkTriangle = pkTIter->second;
131  for (int i = 0; i < 3; i++)
132  {
133  // inform edges you are going away
134  Edge* pkEdge = pkTriangle->E[i];
135  assert(pkEdge);
136  if (pkEdge->T[0] == pkTriangle)
137  {
138  // one-triangle edges always have pointer in slot zero
139  pkEdge->T[0] = pkEdge->T[1];
140  pkEdge->T[1] = nullptr;
141  }
142  else if (pkEdge->T[1] == pkTriangle)
143  {
144  pkEdge->T[1] = nullptr;
145  }
146  else
147  {
148  assert(false);
149  return false;
150  }
151 
152  // remove edge if you had the last reference to it
153  if (!pkEdge->T[0] && !pkEdge->T[1])
154  {
155  EdgeKey kEKey(pkEdge->V[0],pkEdge->V[1]);
156  m_kEMap.erase(kEKey);
157  WM4_DELETE pkEdge;
158  }
159 
160  // inform adjacent triangles you are going away
161  TPtr pkAdjacent = pkTriangle->T[i];
162  if (pkAdjacent)
163  {
164  for (int j = 0; j < 3; j++)
165  {
166  if (pkAdjacent->T[j] == pkTriangle)
167  {
168  pkAdjacent->T[j] = nullptr;
169  break;
170  }
171  }
172  }
173  }
174 
175  m_kTMap.erase(kTKey);
176  WM4_DELETE pkTriangle;
177  return true;
178 }
