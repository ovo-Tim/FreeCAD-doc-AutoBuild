339 void ViewProviderMeasureBase::updateData(const App::Property* prop)
340 {
341  bool doUpdate = false;
342 
343  auto obj = getMeasureObject();
344  if (!obj) {
345  return;
346  }
347 
348  if (strcmp(prop->getName(), "Label") == 0) {
349  doUpdate = true;
350  }
351 
352  // Check if one of the input properties has been changed
353  auto inputProps = obj->getInputProps();
354  if (std::find(inputProps.begin(), inputProps.end(), std::string(prop->getName()))
355  != inputProps.end()) {
356  doUpdate = true;
357 
358  // Add connections to be notified when the measured objects are changed
359  connectToSubject(obj->getSubject());
360  }
361 
362  // Check if the result prop has been changed
363  auto resultProp = obj->getResultProp();
364  if (resultProp && prop == resultProp) {
365  doUpdate = true;
366  }
367 
368  if (doUpdate) {
369  redrawAnnotation();
370 
371  // Update label
372  std::string userLabel(obj->Label.getValue());
373  std::string name = userLabel.substr(0, userLabel.find(":"));
374  obj->Label.setValue((name + ": ") + obj->getResultString().toStdString());
375  }
376 
377  ViewProviderDocumentObject::updateData(prop);
378 }

438 Base::Vector3d ViewProviderMeasureBase::getTextDirection(Base::Vector3d elementDirection,
439  double tolerance)
440 {
441  // TODO: this can fail if the active view is not a 3d view (spreadsheet, techdraw page) and
442  // something causes a measure to try to update we need to search through the mdi views for a 3d
443  // view and take the direction from it (or decide that if the active view is not 3d, assume we
444  // are looking from the front).
445  Base::Vector3d viewDirection;
446  Base::Vector3d upDirection;
447 
448  Gui::View3DInventor* view = nullptr;
449  try {
450  view = dynamic_cast<Gui::View3DInventor*>(this->getActiveView());
451  }
452  catch (const Base::RuntimeError&) {
453  Base::Console().Log(
454  "ViewProviderMeasureBase::getTextDirection: Could not get active view\n");
455  }
456 
457  if (view) {
458  Gui::View3DInventorViewer* viewer = view->getViewer();
459  viewDirection = toVector3d(viewer->getViewDirection()).Normalize();
460  upDirection = toVector3d(viewer->getUpDirection()).Normalize();
461  // Measure doesn't work with this kind of active view. Might be dependency graph, might be
462  // TechDraw, or ????
463  // throw Base::RuntimeError("Measure doesn't work with this kind of active view.");
464  }
465  else {
466  viewDirection = Base::Vector3d(0.0, 1.0, 0.0);
467  upDirection = Base::Vector3d(0.0, 0.0, 1.0);
468  }
469 
470  Base::Vector3d textDirection = elementDirection.Cross(viewDirection);
471  if (textDirection.Length() < tolerance) {
472  // either elementDirection and viewDirection are parallel or one of them is null.
473  textDirection = elementDirection.Cross(upDirection);
474  }
475 
476  return textDirection.Normalize();
477 }

482 bool ViewProviderMeasureBase::isSubjectVisible()
483 {
484  Gui::Document* guiDoc = nullptr;
485  try {
486  guiDoc = this->getDocument();
487  }
488  catch (const Base::RuntimeError&) {
489  Base::Console().Log("ViewProviderMeasureBase::isSubjectVisible: Could not get document\n");
490  return false;
491  }
492 
493  // we need these things to proceed
494  if (!getMeasureObject() || !guiDoc) {
495  return false;
496  }
497 
498  // Show the measurement if it doesn't track any subjects
499  if (getMeasureObject()->getSubject().empty()) {
500  return true;
501  }
502 
503  for (auto& obj : getMeasureObject()->getSubject()) {
504  Gui::ViewProvider* vp = guiDoc->getViewProvider(obj);
505  if (!vp || !vp->isVisible()) {
506  return false;
507  }
508  }
509 
510  // all of the subject objects are visible
511  return true;
512 }

539 float ViewProviderMeasureBase::getViewScale()
540 {
541  float scale = 1.0;
542 
543  Gui::View3DInventor* view = dynamic_cast<Gui::View3DInventor*>(this->getActiveView());
544  if (!view) {
545  Base::Console().Log("ViewProviderMeasureBase::getViewScale: Could not get active view\n");
546  return scale;
547  }
548  Gui::View3DInventorViewer* viewer = view->getViewer();
549

552  return false;
553  }
554

557  scale = volume.getWorldToScreenScale(center, 1.0);
558  return scale;
559 }

590  pCoords->point.connectFrom(engineCat->output);

593  pLines->ref();
594  pLines->coordIndex.setNum(lineCount);
595  pLines->coordIndex.setValues(0, lineCount, lines);
596

598  lineSep->addChild(pCoords);
599  lineSep->addChild(pLines);

606 
607  // Connect dragger local orientation to view orientation

631 void ViewProviderMeasure::positionAnno(const Measure::MeasureBase* measureObject)
632 {
633  (void)measureObject;
634 
635  // Initialize the text position
636  Base::Vector3d textPos = getTextPosition();
637  auto srcVec = SbVec3f(textPos.x, textPos.y, textPos.z);
638 
639  // Translate the position by the local dragger matrix (pDraggerOrientation)
640  Gui::View3DInventor* view = nullptr;
641  try {
642  view = dynamic_cast<Gui::View3DInventor*>(this->getActiveView());
643  }
644  catch (const Base::RuntimeError&) {
645  Base::Console().Log("ViewProviderMeasure::positionAnno: Could not get active view\n");
646  }
647 
648  if (!view) {
649  return;
650  }
651 
652  Gui::View3DInventorViewer* viewer = view->getViewer();
653  auto gma = SoGetMatrixAction(viewer->getSoRenderManager()->getViewportRegion());
654  gma.apply(pDraggerOrientation);
655  auto mat = gma.getMatrix();
656  SbVec3f destVec(0, 0, 0);
657  mat.multVecMatrix(srcVec, destVec);
658 
659  setLabelTranslation(destVec);
660  updateView();
661 }

694 Base::Vector3d ViewProviderMeasure::getTextPosition()
695 {
696  // Return the initial position relative to the base position
697  auto basePoint = getBasePosition();
698 
699  Gui::View3DInventor* view = dynamic_cast<Gui::View3DInventor*>(this->getActiveView());
700  if (!view) {
701  Base::Console().Log(
702  "ViewProviderMeasureBase::getTextPosition: Could not get active view\n");
703  return Base::Vector3d();
704  }
705 
706  Gui::View3DInventorViewer* viewer = view->getViewer();
707 
708  // Convert to screenspace, offset and convert back to world space
709  SbVec2s screenPos = viewer->getPointOnViewport(SbVec3f(basePoint.x, basePoint.y, basePoint.z));
710  SbVec3f vec = viewer->getPointOnFocalPlane(screenPos + SbVec2s(30.0, 30.0));
711  Base::Vector3d textPos(vec[0], vec[1], vec[2]);
712 
713  return textPos - basePoint;
714 }
