107 inline bool MeshSearchNeighbourFacetsVisitor::Visit(const MeshFacet& rclFacet,
108  const MeshFacet& rclFrom,
109  FacetIndex ulFInd,
110  unsigned long ulLevel)
111 {
112  (void)rclFrom;
113  if (ulLevel > _ulCurrentLevel) {
114  if (!_bFacetsFoundInCurrentLevel) {
115  return false;
116  }
117  _ulCurrentLevel = ulLevel;
118  _bFacetsFoundInCurrentLevel = false;
119  }
120 
121  for (PointIndex ptIndex : rclFacet._aulPoints) {
122  if (Base::Distance(_clCenter, _rclMeshBase.GetPoint(ptIndex)) < _fRadius) {
123  _vecFacets.push_back(ulFInd);
124  _bFacetsFoundInCurrentLevel = true;
125  return true;
126  }
127  }
128 
129  return true;
130 }
