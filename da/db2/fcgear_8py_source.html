80 def midpoints(p1, p2, r):
81  """A very ugly function that returns the midpoint of a p1 and p2
82  on the circle which radius is r and which pass through p1 and
83  p2
84 
85  Return the 2 possible solutions
86  """
87  vx, vy = p2[0]-p1[0], p2[1]-p1[1]
88  b = (vx**2 + vy**2)**.5
89  v = (vx/b, vy/b)
90  cosA = b**2 / (2*b*r)
91  A = acos(cosA)
92 
93  vx, vy = rotate(v, A)
94  c1 = (p1[0]+r*vx, p1[1]+r*vy)
95  m1x, m1y = ((p1[0]+p2[0])/2 - c1[0], (p1[1]+p2[1])/2 - c1[1])
96  dm1 = (m1x**2+m1y**2)**.5
97  m1x, m1y = (c1[0] + r*m1x/dm1, c1[1] + r*m1y/dm1)
98  m1 = (m1x, m1y)
99 
100  vx, vy = rotate(v, -A)
101  c2 = (p1[0]+r*vx, p1[1]+r*vy)
102  m2x, m2y = ((p1[0]+p2[0])/2 - c2[0], (p1[1]+p2[1])/2 - c2[1])
103  dm2 = (m2x**2+m2y**2)**.5
104  m2x, m2y = (c2[0] + r*m2x/dm2, c2[1] + r*m2y/dm2)
105  m2 = (m2x, m2y)
106 
107  return m1, m2
