394 void MRichTextEdit::textStyle(int index) {
395  //TODO: would prefer select font vs paragraph style.
396  QTextCursor cursor = f_textedit->textCursor();
397  cursor.beginEditBlock();
398 
399  // standard
400  if (!cursor.hasSelection()) {
401  cursor.select(QTextCursor::BlockUnderCursor);
402  }
403  QTextCharFormat fmt;
404  cursor.setCharFormat(fmt);
405  f_textedit->setCurrentCharFormat(fmt);
406 
407  if (index == ParagraphHeading1
408  || index == ParagraphHeading2
409  || index == ParagraphHeading3
410  || index == ParagraphHeading4 ) {
411  if (index == ParagraphHeading1) {
412  fmt.setFontPointSize(m_fontsize_h1);
413  }
414  if (index == ParagraphHeading2) {
415  fmt.setFontPointSize(m_fontsize_h2);
416  }
417  if (index == ParagraphHeading3) {
418  fmt.setFontPointSize(m_fontsize_h3);
419  }
420  if (index == ParagraphHeading4) {
421  fmt.setFontPointSize(m_fontsize_h4);
422  }
423  if (index == ParagraphHeading2 || index == ParagraphHeading4) {
424  fmt.setFontItalic(true);
425  }
426 
427  fmt.setFontWeight(QFont::Bold);
428  }
429  if (index == ParagraphMonospace) {
430  fmt = cursor.charFormat();
431 #if QT_VERSION < QT_VERSION_CHECK(5,13,0)
432  fmt.setFontFamily(QString::fromUtf8("Monospace"));
433 #else
434  fmt.setFontFamilies(QStringList() << QString::fromUtf8("Monospace"));
435 #endif
436  fmt.setFontStyleHint(QFont::Monospace);
437  fmt.setFontFixedPitch(true);
438  }
439  cursor.setCharFormat(fmt);
440  f_textedit->setCurrentCharFormat(fmt);
441 
442  cursor.endEditBlock();
443 }

532 void MRichTextEdit::slotCursorPositionChanged() {
533 // qDebug() << "MRTE::slotCursorPositionChanged()";
534  //why do we change text style when selecting text?
535  QTextCursor cursor = f_textedit->textCursor();
536  if (cursor.hasSelection()) { //let selection logic handle this
537  return;
538  }
539  //set font size widget to match curr char size
540  QTextCharFormat fmt = cursor.charFormat();
541  double currSize = fmt.fontPointSize();
542  int fSize = f_fontsize->findText(QString::number(currSize));
543  f_fontsize ->setCurrentIndex(fSize);
544 
545  QTextList *l = f_textedit->textCursor().currentList();
546 
547  if (m_lastBlockList &&
548  (l == m_lastBlockList ||
549  (l && m_lastBlockList && l->format().style() == m_lastBlockList->format().style()) ) ) {
550  return;
551  }
552  m_lastBlockList = l;
553  if (l) {
554  QTextListFormat lfmt = l->format();
555  if (lfmt.style() == QTextListFormat::ListDisc) {
556  f_list_bullet->setChecked(true);
557  f_list_ordered->setChecked(false);
558  } else if (lfmt.style() == QTextListFormat::ListDecimal) {
559  f_list_bullet->setChecked(false);
560  f_list_ordered->setChecked(true);
561  } else {
562  f_list_bullet->setChecked(false);
563  f_list_ordered->setChecked(false);
564  }
565  } else {
566  f_list_bullet->setChecked(false);
567  f_list_ordered->setChecked(false);

741 {
742 // qDebug() << "MRTE::hasMultipleSizes()";
743  QTextCursor cursor = f_textedit->textCursor();
744  if (cursor.hasSelection()) {
745  int begin = cursor.selectionStart();
746  int end = cursor.selectionEnd();
747  int currPos;
748  std::vector<QString> foundSizes;
749  std::map<QString, int> countMap;
750  for (currPos = begin; currPos < end; currPos++) {
751  cursor.setPosition(currPos);
752  QTextCharFormat fmt = cursor.charFormat();
753  double currSize = fmt.fontPointSize();
754  QString asQS = QString::number(currSize, 'f', 2);
755  foundSizes.push_back(asQS);
756  auto ret = countMap.insert(std::pair<QString, int>(asQS, 1));
757  if (!ret.second) { //already have this size
758  ret.first->second++; //bump count
759  }
760  }
761  if (countMap.size() > 1) {
762  return true;
763  }
764  }
