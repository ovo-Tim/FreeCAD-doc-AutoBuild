174 void GmshWidget::accept()
175 {
176  if (d->gmsh.state() == QProcess::Running) {
177  Base::Console().Warning("Cannot start gmsh because it's already running\n");
178  return;
179  }
180 
181  // clang-format off
182  QString inpFile;
183  QString outFile;
184  if (writeProject(inpFile, outFile)) {
185  // ./gmsh - -bin -2 /tmp/mesh.geo -o /tmp/best.stl
186  QString proc = d->ui.fileChooser->fileName();
187  QStringList args;
188  args << QLatin1String("-")
189  << QLatin1String("-bin")
190  << QLatin1String("-2")
191  << inpFile
192  << QLatin1String("-o")
193  << outFile;
194  d->gmsh.start(proc, args);
195 
196  d->time.start();
197  d->ui.labelTime->setText(tr("Time:"));
198  }
199  // clang-format on
200 }

202 void GmshWidget::readyReadStandardError()
203 {
204  QByteArray msg = d->gmsh.readAllStandardError();
205  if (msg.startsWith("\0[1m\0[31m")) {
206  msg = msg.mid(9);
207  }
208  if (msg.endsWith("\0[0m")) {
209  msg.chop(5);
210  }
211 
212  QString text = QString::fromUtf8(msg.data());
213  d->appendText(text, true);
214 }

311 bool RemeshGmsh::writeProject(QString& inpFile, QString& outFile)
312 {
313  // clang-format off
314  if (!d->mesh.expired()) {
315  Base::FileInfo stl(d->stlFile);
316  MeshCore::MeshOutput output(d->copy);
317  Base::ofstream stlOut(stl, std::ios::out | std::ios::binary);
318  output.SaveBinarySTL(stlOut);
319  stlOut.close();
320 
321  // Parameters
322  int algorithm = meshingAlgorithm();
323  double maxSize = getMaxSize();
324  if (maxSize == 0.0) {
325  maxSize = 1.0e22;
326  }
327  double minSize = getMinSize();
328  double angle = getAngle();
329  int maxAngle = 120;
330  int minAngle = 20;
331 
332  // Gmsh geo file
333  Base::FileInfo geo(d->geoFile);
334  Base::ofstream geoOut(geo, std::ios::out);
335  // Examples on how to use Gmsh: https://sfepy.org/doc-devel/preprocessing.html
336  // https://gmsh.info//doc/texinfo/gmsh.html
337  // https://docs.salome-platform.org/latest/gui/GMSHPLUGIN/gmsh_2d_3d_hypo_page.html
338  geoOut << "// geo file for meshing with Gmsh meshing software created by FreeCAD\n"
339  << "If(GMSH_MAJOR_VERSION < 4)\n"
340  << " Error(\"Too old Gmsh version %g.%g. At least 4.x is required\", GMSH_MAJOR_VERSION, GMSH_MINOR_VERSION);\n"
341  << " Exit;\n"
342  << "EndIf\n"
343  << "Merge \"" << stl.filePath() << "\";\n\n"
344  << "// 2D mesh algorithm (1=MeshAdapt, 2=Automatic, 5=Delaunay, 6=Frontal, 7=BAMG, 8=Frontal Quad, 9=Packing of Parallelograms, 11=Quasi-structured Quad)\n"
345  << "Mesh.Algorithm = " << algorithm << ";\n\n"
346  << "// 3D mesh algorithm (1=Delaunay, 2=New Delaunay, 4=Frontal, 7=MMG3D, 9=R-tree, 10=HTX)\n"
347  << "// Mesh.Algorithm3D = 1;\n\n"
348  << "Mesh.CharacteristicLengthMax = " << maxSize << ";\n"
349  << "Mesh.CharacteristicLengthMin = " << minSize << ";\n\n"
350  << "// We first classify (\"color\") the surfaces by splitting the original surface\n"
351  << "// along sharp geometrical features. This will create new discrete surfaces,\n"
352  << "// curves and points.\n"
353  << "angle = DefineNumber[" << angle << ", Min " << minAngle << ", Max " << maxAngle << ", Step 1,\n"
354  << " Name \"Parameters/Angle for surface detection\" ];\n\n"
355  << "forceParametrizablePatches = DefineNumber[0, Choices{0,1},\n"
356  << " Name \"Parameters/Create surfaces guaranteed to be parametrizable\"];\n\n"
357  << "includeBoundary = 1;\n"
358  << "ClassifySurfaces{angle * Pi/180, includeBoundary, forceParametrizablePatches};\n"
359  << "// Create a geometry for all the discrete curves and surfaces in the mesh, by\n"
360  << "// computing a parametrization for each one\n"
361  << "CreateGeometry;\n\n"
362  << "// Create a volume as usual\n"
363  << "Surface Loop(1) = Surface{:};\n"
364  << "Volume(1) = {1};\n";
365  geoOut.close();
366 
367  inpFile = QString::fromUtf8(d->geoFile.c_str());
368  outFile = QString::fromUtf8(d->stlFile.c_str());
369 
370  return true;
371  }
372 
373  return false;
374  // clang-format on
375 }

377 bool RemeshGmsh::loadOutput()
378 {
379  if (d->mesh.expired()) {
380  return false;
381  }
382 
383  // Now read-in modified mesh
384  Base::FileInfo stl(d->stlFile);
385  Base::FileInfo geo(d->geoFile);
386 
387  Mesh::MeshObject kernel;
388  MeshCore::MeshInput input(kernel.getKernel());
389  Base::ifstream stlIn(stl, std::ios::in | std::ios::binary);
390  input.LoadBinarySTL(stlIn);
391  stlIn.close();
392  kernel.harmonizeNormals();
393 
394  Mesh::Feature* fea = d->mesh.get<Mesh::Feature>();
395  App::Document* doc = fea->getDocument();
396  doc->openTransaction("Remesh");
397  fea->Mesh.setValue(kernel.getKernel());
398  doc->commitTransaction();
399  stl.deleteFile();
400  geo.deleteFile();
401 
402  return true;
403 }
