25 """
26 This is an example preprocessor file for the Path workbench. Its aim is to
27 open a gcode file, parse its contents, and create the appropriate objects
28 in FreeCAD.
29 
30 This preprocessor will split gcode on tool changes and create one or more
31 PathCustom objects in the job. Tool Change commands themselves are not
32 preserved. It is up to the user to create and assign appropriate tool
33 controllers.
34 
35 Only gcodes that are supported by Path are imported. Thus things like G43
36 are suppressed.
37 
38 Importing gcode is inherently dangerous because context cannot be safely
39 assumed. The user should carefully examine the resulting gcode!
40 
41 Read the Path Workbench documentation to know how to create Path objects
42 from GCode.
43 """

118 def parse(inputstring):
119  "parse(inputstring): returns a parsed output string"
120 
121  supported = [
122  "G0",
123  "G00",
124  "G1",
125  "G01",
126  "G2",
127  "G02",
128  "G3",
129  "G03",
130  "G81",
131  "G82",
132  "G83",
133  "G90",
134  "G91",
135  ]
136 
137  axis = ["X", "Y", "Z", "A", "B", "C", "U", "V", "W"]
138 
139  FreeCAD.Console.PrintMessage("preprocessing...\n")
140  Path.Log.track(inputstring)
141  # split the input by line
142  lines = inputstring.splitlines()
143  output = []
144  lastcommand = None
145 
146  for lin in lines:
147  # remove any leftover trailing and preceding spaces
148  lin = lin.strip()
149 
150  # discard empty lines
151  if not lin:
152  continue
153 
154  # remove line numbers
155  if lin[0].upper() in ["N"]:
156  lin = lin.split(" ", 1)
157  if len(lin) >= 1:
158  lin = lin[1].strip()
159  else:
160  continue
161 
162  # Anything else not a G/M code or an axis move is ignored.
163  if lin[0] not in ["G", "M", "X", "Y", "Z", "A", "B", "C", "U", "V", "W"]:
164  continue
165 
166  # if the remaining line is supported, store it
167  currcommand = lin.split()[0]
168 
169  if currcommand in supported:
170  output.append(lin)
171  lastcommand = currcommand
172 
173  # modal commands have no G or M but have axis moves. append those too.
174  elif currcommand[0] in axis and lastcommand:
175  output.append(lastcommand + " " + lin)
176 
177  FreeCAD.Console.PrintMessage("done preprocessing.\n")
178  return output

181 def _identifygcodeByToolNumberList(filename):
182  """called when freecad imports a file"""
183  Path.Log.track(filename)
184  gcodeByToolNumberList = []
185 
186  gfile = pyopen(filename)
187  gcode = gfile.read()
188  gfile.close()
189 
190  # Regular expression to match tool changes in the format 'M6 Tn'
191  p = re.compile(r"[mM]+?\s?0?6\s?T\d*\s")
192 
193  # split the gcode on tool changes
194  paths = re.split(r"([mM]+?\s?0?6\s?T\d*\s)", gcode)
195 
196  # iterate the gcode sections and add customs for each
197  toolnumber = 0
198 
199  for path in paths:
200 
201  # if the section is a tool change, extract the tool number
202  m = p.match(path)
203  if m:
204  toolnumber = int(m.group().split("T")[-1])
205  continue
206 
207  # Parse the gcode and throw away any empty lists
208  gcode = parse(path)
209  if len(gcode) > 0:
210  gcodeByToolNumberList.append((gcode, toolnumber))
211 
212  return gcodeByToolNumberList

215 def insert(filename, docname=None):
216  """called when freecad imports a file"""
217  Path.Log.track(filename)
218 
219  try:
220  if not _isImportEnvironmentReady():
221  return
222  except PathNoActiveDocumentException:
223  Path.Log.error(translate("CAM_Gcode_pre", "No active document"))
224  return
225  except PathNoJobException:
226  Path.Log.error(translate("CAM_Gcode_pre", "No job object"))
227  return
228 
229  # Create a Custom operation for each gcode-toolNumber pair
230  for gcode, toolNumber in _identifygcodeByToolNumberList(filename):
231  obj = PathCustom.Create("Custom")
232 
233  # Set the gcode and try to match a tool controller
234  obj.Gcode = gcode
235  obj.ToolController = matchToolController(obj, toolNumber)
236  if docname:
237  obj.Label = obj.Label + "_" + docname
238 
239  if FreeCAD.GuiUp:
240  # Add a view provider to a Custom operation object
241  obj.ViewObject.Proxy = PathCustomGui.PathOpGui.ViewProvider(
242  obj.ViewObject, PathCustomGui.Command.res

244  obj.ViewObject.Proxy.setDeleteObjectsOnReject(False)

246  FreeCAD.ActiveDocument.recompute()
